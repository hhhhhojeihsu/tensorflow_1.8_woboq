128/* _PyLong_FromByteArray: View the n unsigned bytes as a binary integer in
129 base 256, and return a Python int with the same numeric value.
130 If n is 0, the integer is 0. Else:
131 If little_endian is 1/true, bytes[n-1] is the MSB and bytes[0] the LSB;
132 else (little_endian is 0/false) bytes[0] is the MSB and bytes[n-1] the
133 LSB.
134 If is_signed is 0/false, view the bytes as a non-negative integer.
135 If is_signed is 1/true, view the bytes as a 2's-complement integer,
136 non-negative if bit 0x80 of the MSB is clear, negative if set.
137 Error returns:
138 + Return NULL with the appropriate exception set if there's not
139 enough memory to create the Python int.
140*/
141PyAPI_FUNC(PyObject *) _PyLong_FromByteArray(
142 const unsigned char* bytes, size_t n,
143 int little_endian, int is_signed);
144
145/* _PyLong_AsByteArray: Convert the least-significant 8*n bits of long
146 v to a base-256 integer, stored in array bytes. Normally return 0,
147 return -1 on error.
148 If little_endian is 1/true, store the MSB at bytes[n-1] and the LSB at
149 bytes[0]; else (little_endian is 0/false) store the MSB at bytes[0] and
150 the LSB at bytes[n-1].
151 If is_signed is 0/false, it's an error if v < 0; else (v >= 0) n bytes
152 are filled and there's nothing special about bit 0x80 of the MSB.
153 If is_signed is 1/true, bytes is filled with the 2's-complement
154 representation of v's value. Bit 0x80 of the MSB is the sign bit.
155 Error returns (-1):
156 + is_signed is 0 and v < 0. TypeError is set in this case, and bytes
157 isn't altered.
158 + n isn't big enough to hold the full mathematical value of v. For
159 example, if is_signed is 0 and there are more digits in the v than
160 fit in n; or if is_signed is 1, v < 0, and n is just 1 bit shy of
161 being large enough to hold a sign bit. OverflowError is set in this
162 case, but bytes holds the least-signficant n bytes of the true value.
163*/
164PyAPI_FUNC(int) _PyLong_AsByteArray(PyLongObject* v,
165 unsigned char* bytes, size_t n,
166 int little_endian, int is_signed);
