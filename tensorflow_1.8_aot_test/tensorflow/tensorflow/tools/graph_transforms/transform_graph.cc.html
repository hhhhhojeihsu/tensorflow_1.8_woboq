33Status ParseTransformParameters(const string& transforms_string,
34 TransformParameters* params_list) {
35 params_list->clear();
36 enum {
37 TRANSFORM_NAME,
38 TRANSFORM_PARAM_NAME,
39 TRANSFORM_PARAM_VALUE,
40 } state = TRANSFORM_NAME;
41 StringPiece remaining(transforms_string);
42 StringPiece match;
43 StringPiece transform_name;
44 StringPiece parameter_name;
45 StringPiece parameter_value;
46 TransformFuncParameters func_parameters;
47 while (!remaining.empty()) {
48 if (state == TRANSFORM_NAME) {
49 // Reset the list of parameters.
50 func_parameters.clear();
51 // Eat up any leading spaces.
52 Scanner(remaining).AnySpace().GetResult(&remaining, &match);
53 if (remaining.empty()) {
54 // Nothing remains after consuming trailing spaces.
55 // Consumed all transform parameter string without errors.
56 return Status::OK();
57 }
58 // See if we have a valid transform name.
59 const bool found_transform_name =
60 Scanner(remaining)
61 .Many(Scanner::LETTER_DIGIT_UNDERSCORE)
62 .GetResult(&remaining, &transform_name);
63 if (!found_transform_name) {
64 return errors::InvalidArgument("Looking for transform name, but found ",
65 remaining.ToString().c_str());
66 }
67 if (Scanner(remaining).OneLiteral("(").GetResult(&remaining, &match)) {
68 state = TRANSFORM_PARAM_NAME;
69 } else {
70 // Add a transform with no parameters.
71 params_list->push_back({transform_name.ToString(), func_parameters});
72 transform_name = "";
73 state = TRANSFORM_NAME;
74 }
75 } else if (state == TRANSFORM_PARAM_NAME) {
76 if (Scanner(remaining).OneLiteral(")").GetResult(&remaining, &match)) {
77 params_list->push_back({transform_name.ToString(), func_parameters});
78 transform_name = "";
79 state = TRANSFORM_NAME;
80 } else {
81 // Eat up any leading spaces or commas.
82 Scanner(remaining).ZeroOrOneLiteral(",").GetResult(&remaining, &match);
83 Scanner(remaining).AnySpace().GetResult(&remaining, &match);
84 // See if we have a valid parameter name.
85 const bool found_parameter_name =
86 Scanner(remaining)
87 .Many(Scanner::LETTER_DIGIT_UNDERSCORE)
88 .GetResult(&remaining, &parameter_name);
89 if (!found_parameter_name) {
90 return errors::InvalidArgument(
91 "Looking for parameter name, but found ",
92 remaining.ToString().c_str());
93 }
94 if (Scanner(remaining).OneLiteral("=").GetResult(&remaining, &match)) {
95 state = TRANSFORM_PARAM_VALUE;
96 } else {
97 return errors::InvalidArgument("Looking for =, but found ",
98 remaining.ToString().c_str());
99 }
100 }
101 } else if (state == TRANSFORM_PARAM_VALUE) {
102 bool found_parameter_value;
103 // Deal with quoted values.
104 if (Scanner(remaining).OneLiteral("\"").GetResult(&remaining, &match)) {
105 found_parameter_value =
106 Scanner(remaining).ScanEscapedUntil('"').GetResult(
107 &remaining, &parameter_value);
108 if (found_parameter_value) {
109 Scanner(remaining).OneLiteral("\"").GetResult(&remaining, &match);
110 }
111 } else {
112 // See if we have a valid parameter name.
113 found_parameter_value =
114 Scanner(remaining)
115 .Many(Scanner::LETTER_DIGIT_DASH_DOT_SLASH_UNDERSCORE)
116 .GetResult(&remaining, &parameter_value);
117 }
118 if (!found_parameter_value) {
119 return errors::InvalidArgument("Looking for parameter name, but found ",
120 remaining.ToString().c_str());
121 }
122 func_parameters[parameter_name.ToString()].push_back(
123 parameter_value.ToString());
124 // Eat up any trailing quotes.
125 Scanner(remaining).ZeroOrOneLiteral("\"").GetResult(&remaining, &match);
126 Scanner(remaining).ZeroOrOneLiteral("'").GetResult(&remaining, &match);
127 state = TRANSFORM_PARAM_NAME;
128 }
129 }
130 return Status::OK();
131}

133int ParseFlagsAndTransformGraph(int argc, char* argv[], bool init_main) {
134 string in_graph = "";
135 string out_graph = "";
136 string inputs_string = "";
137 string outputs_string = "";
138 string transforms_string = "";
139 bool output_as_text = false;
140 std::vector<Flag> flag_list = {
141 Flag("in_graph", &in_graph, "input graph file name"),
142 Flag("out_graph", &out_graph, "output graph file name"),
143 Flag("inputs", &inputs_string, "inputs"),
144 Flag("outputs", &outputs_string, "outputs"),
145 Flag("transforms", &transforms_string, "list of transforms"),
146 Flag("output_as_text", &output_as_text,
147 "whether to write the graph in text protobuf format"),
148 };
149 string usage = Flags::Usage(argv[0], flag_list);
150 usage += "\nTransforms are:\n";
151 TransformRegistry* transform_registry = GetTransformRegistry();
152 for (const auto& pair : *transform_registry) {
153 usage += pair.first + "\n";
154 }
155
156 const bool parse_result = Flags::Parse(&argc, argv, flag_list);
157 // We need to call this to set up global state for TensorFlow.
158 if (init_main) {
159 port::InitMain(argv[0], &argc, &argv);
160 }
161 if (!parse_result) {
162 LOG(ERROR) << usage;
163 return -1;
164 }
165 if (argc > 1) {
166 LOG(ERROR) << "Unknown argument " << argv[1] << ".\n" << usage;
167 return -1;
168 }
169 if (in_graph.empty()) {
170 LOG(ERROR) << "in_graph graph can't be empty.\n" << usage;
171 return -1;
172 }
173 if (out_graph.empty()) {
174 LOG(ERROR) << "out_graph graph can't be empty.\n" << usage;
175 return -1;
176 }
177 if (transforms_string.empty()) {
178 LOG(ERROR) << "You must specify at least one transform.\n" << usage;
179 return -1;
180 }
181
182 std::vector<string> inputs = str_util::Split(inputs_string, ',');
183 std::vector<string> outputs = str_util::Split(outputs_string, ',');
184 TransformParameters transform_params;
185 Status parse_status =
186 ParseTransformParameters(transforms_string, &transform_params);
187 if (!parse_status.ok()) {
188 LOG(ERROR) << "Failed to parse --transform argument, error was "
189 << parse_status.error_message();
190 return -1;
191 }
192 if (transform_params.empty()) {
193 LOG(ERROR) << "You must specify at least one transform.\n" << usage;
194 return -1;
195 }
196
197 GraphDef graph_def;
198 Status load_status = LoadTextOrBinaryGraphFile(in_graph, &graph_def);
199 if (!load_status.ok()) {
200 LOG(ERROR) << "Loading graph '" << in_graph << "' failed with "
201 << load_status.error_message();
202 LOG(ERROR) << usage;
203 return -1;
204 }
205
206 Status transform_result =
207 TransformGraph(inputs, outputs, transform_params, &graph_def);
208
209 if (!transform_result.ok()) {
210 LOG(ERROR) << transform_result.error_message();
211 LOG(ERROR) << usage;
212 return -1;
213 }
214
215 Status save_status;
216 if (output_as_text) {
217 save_status = WriteTextProto(Env::Default(), out_graph, graph_def);
218 } else {
219 save_status = WriteBinaryProto(Env::Default(), out_graph, graph_def);
220 }
221 if (!save_status.ok()) {
222 LOG(ERROR) << "Saving graph '" << out_graph << "' failed with "
223 << save_status.error_message();
224 return -1;
225 }
226
227 return 0;
228}

230Status ShouldIgnoreErrors(const TransformFuncParameters& transform_params,
231 bool* ignore_errors) {
232 *ignore_errors = false;
233 if (transform_params.count("ignore_errors") &&
234 (!transform_params.at("ignore_errors").empty())) {
235 const string& ignore_errors_string =
236 str_util::Lowercase(transform_params.at("ignore_errors").at(0));
237 if (ignore_errors_string == "true") {
238 *ignore_errors = true;
239 } else if (ignore_errors_string == "false") {
240 *ignore_errors = false;
241 } else {
242 return errors::InvalidArgument(
243 "ignore_errors should be true or false, found ",
244 ignore_errors_string);
245 }
246 }
247 return Status::OK();
248}

250Status TransformGraph(const std::vector<string>& inputs,
251 const std::vector<string>& outputs,
252 const TransformParameters& transform_params,
253 GraphDef* graph_def) {
254 TransformRegistry* transform_registry = GetTransformRegistry();
255 for (const auto& transform_info : transform_params) {
256 const string& transform_name = transform_info.first;
257 if (transform_name.empty()) {
258 continue;
259 }
260 if (!transform_registry->count(transform_name)) {
261 return errors::InvalidArgument("Transform '", transform_name,
262 "' not recognized.");
263 }
264 LOG(INFO) << "Applying " << transform_name;
265 const TransformFunc& transform_func =
266 transform_registry->at(transform_name);
267 TransformFuncContext context;
268 context.input_names = inputs;
269 context.output_names = outputs;
270 context.params = transform_info.second;
271 bool ignore_errors;
272 TF_RETURN_IF_ERROR(
273 ShouldIgnoreErrors(transform_info.second, &ignore_errors));
274 GraphDef transformed_graph_def;
275 Status transform_result =
276 transform_func(*graph_def, context, &transformed_graph_def);
277 if (!transform_result.ok()) {
278 if (ignore_errors) {
279 LOG(ERROR) << transform_name << ": Ignoring error "
280 << transform_result.error_message();
281 transformed_graph_def = *graph_def;
282 } else {
283 return transform_result;
284 }
285 }
286 // Copy over the library from the original input graph.
287 *transformed_graph_def.mutable_library() = graph_def->library();
288 TF_RETURN_IF_ERROR(IsGraphValid(transformed_graph_def));
289
290 *graph_def = transformed_graph_def;
291 }
292 return Status::OK();
293}
