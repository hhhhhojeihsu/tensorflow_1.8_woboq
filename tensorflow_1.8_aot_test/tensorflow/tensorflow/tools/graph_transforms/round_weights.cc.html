32// Rounds any large float constants to the specified number of levels.
33Status RoundWeights(const GraphDef& input_graph_def,
34 const TransformFuncContext& context,
35 GraphDef* output_graph_def) {
36 int32 num_steps;
37 TF_RETURN_IF_ERROR(
38 context.GetOneInt32Parameter("num_steps", 256, &num_steps));
39 TF_RETURN_IF_ERROR(ReplaceMatchingOpTypes(
40 input_graph_def, {"Const"},
41 [num_steps](const NodeMatch& match, const std::set<string>& input_nodes,
42 const std::set<string>& output_nodes,
43 std::vector<NodeDef>* new_nodes) {
44 const NodeDef& old_const_node = match.node;
45 if (!old_const_node.attr().count("dtype")) {
46 return errors::InvalidArgument("No 'dtype' attribute for Const node ",
47 old_const_node.name());
48 }
49 if (!old_const_node.attr().count("value")) {
50 return errors::InvalidArgument("No 'value' attribute for Const node ",
51 old_const_node.name());
52 }
53 const DataType old_dtype = old_const_node.attr().at("dtype").type();
54 Tensor old_tensor;
55 if (!old_tensor.FromProto(old_const_node.attr().at("value").tensor())) {
56 return errors::InvalidArgument("Decoding Tensor failed for node",
57 old_const_node.name());
58 }
59 const size_t num_elements = old_tensor.NumElements();
60 // If this isn't a float constant, or it's too small, then reuse the
61 // same node with no changes. The size is important because small
62 // constants tend to be used for more accuracy-sensitive calculations,
63 // and the benefit of shrinking them is very marginal.
64 if ((old_dtype != DT_FLOAT) || (num_elements < 16)) {
65 new_nodes->push_back(old_const_node);
66 return Status::OK();
67 }
68 const float* old_values = old_tensor.flat<float>().data();
69 float min = std::numeric_limits<float>::max();
70 float max = std::numeric_limits<float>::min();
71 for (int i = 0; i < num_elements; ++i) {
72 const float value = old_values[i];
73 min = std::min(min, value);
74 max = std::max(max, value);
75 }
76 // min_value == max_value is a tricky case. It can occur for general
77 // tensors, and of course for scalars. The quantized ops cannot deal
78 // with this case, so we set max_value to something else.
79 // It's a tricky question what is the numerically best solution to
80 // deal with this degeneracy.
81 // TODO(petewarden): Better use a tolerance than a hard comparison?
82 if (min == max) {
83 if (std::abs(min) < 0.000001f) {
84 max = min + 1.0f;
85 } else if (min > 0) {
86 max = 2.0f * min;
87 } else {
88 min = 2.0f * max;
89 }
90 }
91 Tensor rounded_tensor(DT_FLOAT, old_tensor.shape());
92 float* rounded_values = rounded_tensor.flat<float>().data();
93 const float bucket_width = (max - min) / num_steps;
94 for (int i = 0; i < num_elements; ++i) {
95 const int32 bucket = std::floor((old_values[i] - min) / bucket_width);
96 rounded_values[i] = min + (bucket_width * (bucket + 0.5f));
97 }
98
99 NodeDef rounded_const_node;
100 rounded_const_node.set_op("Const");
101 rounded_const_node.set_name(old_const_node.name());
102 SetNodeAttr("dtype", DT_FLOAT, &rounded_const_node);
103 SetNodeTensorAttr<float>("value", rounded_tensor, &rounded_const_node);
104 new_nodes->push_back(rounded_const_node);
105
106 return Status::OK();
107 },
108 {}, output_graph_def));
109
110 return Status::OK();
111}
112
113REGISTER_GRAPH_TRANSFORM("round_weights", RoundWeights);
