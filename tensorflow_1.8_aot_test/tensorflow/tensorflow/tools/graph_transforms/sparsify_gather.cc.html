40// Sparsify Tensor of shape [N, 1]. Return the indices and values vectors for
41// non-zero tensor content.
42Status SparsifyWeights(const Tensor& tensor, Tensor* indices_tensor,
43 Tensor* values_tensor) {
44 if (tensor.dims() != 2 || tensor.dim_size(1) != 1) {
45 return tensorflow::errors::FailedPrecondition(
46 "Transform only applicable to subgraph with 'Const' with "
47 "tensor of shape [N, 1]. But instead get shape ",
48 tensor.shape().DebugString(), ".");
49 }
50
51 auto flat = tensor.flat<float>();
52 std::vector<int64> indices;
53 std::vector<float> values;
54
55 for (int64 i = 0; i < flat.size(); i++) {
56 float val = flat(i);
57 if (std::abs(val) >= 1.0e-5) {
58 indices.push_back(i);
59 values.push_back(val);
60 }
61 }
62
63 // During model initialization, InitializeTableOp makes use of
64 // KeyValueTensorIterator, which does not accept empty keys or values.
65 // Consequently, adding a dummy pair of indices and values as a walkaround.
66 if (indices.empty() || values.empty()) {
67 indices.push_back(0);
68 values.push_back(0);
69 }
70 *indices_tensor = Tensor(DataTypeToEnum<int64>::value,
71 {static_cast<int64>(indices.size())});
72 std::copy_n(indices.begin(), indices.size(),
73 indices_tensor->flat<int64>().data());
74
75 *values_tensor =
76 Tensor(DataTypeToEnum<float>::value, {static_cast<int64>(values.size())});
77 std::copy_n(values.begin(), values.size(),
78 values_tensor->flat<float>().data());
79
80 return Status::OK();
81}

90string GetMonolithicTensorKey(const string& tensor_slice_name) {
91 std::vector<string> names = Split(tensor_slice_name, "/");
92 if (str_util::StartsWith(names[names.size() - 1], "part_")) {
93 CHECK_GE(names.size(), 2);
94 names.pop_back();
95 }
96 return Join(names, "/");
97}

99Status ObtainTensorSlice(const GraphDef& input_graph_def,
100 const string& target_name,
101 string* shape_slice_string) {
102 string restore_node_name;
103 for (const auto& node : input_graph_def.node()) {
104 std::vector<string> node_name_parts = Split(node.name(), "/");
105 if (node_name_parts.size() == 2 &&
106 str_util::StartsWith(node_name_parts[0], "save") &&
107 str_util::StartsWith(node_name_parts[1], "Assign") &&
108 node.input(0) == target_name) {
109 restore_node_name = node.input(1);
110 break;
111 }
112 }
113
114 std::vector<string> restore_node_parts = Split(restore_node_name, ":");
115 CHECK_LE(restore_node_parts.size(), 2);
116 string tensor_names_node;
117 string shape_and_slices_node;
118 for (const auto& node : input_graph_def.node()) {
119 if ((node.name() == restore_node_parts[0]) && (node.op() == "RestoreV2")) {
120 tensor_names_node = node.input(1);
121 shape_and_slices_node = node.input(2);
122 break;
123 }
124 }
125
126 int offset = -1;
127 for (const auto& node : input_graph_def.node()) {
128 if (node.name() == tensor_names_node) {
129 Tensor tensor_names_tensor;
130 TF_RETURN_IF_ERROR(GetNodeAttr(node, "value", &tensor_names_tensor));
131 const auto& tensor_names_value = tensor_names_tensor.flat<string>();
132 for (int i = 0; i < tensor_names_value.size(); i++) {
133 if (tensor_names_value(i) == GetMonolithicTensorKey(target_name)) {
134 offset = i;
135 break;
136 }
137 }
138 }
139 }
140 if (offset == -1) {
141 return errors::Internal("Unable to find RestoreV2 entry for variable: ",
142 target_name);
143 }
144 for (const auto& node : input_graph_def.node()) {
145 if (node.name() == shape_and_slices_node) {
146 Tensor shape_and_slices_tensor;
147 TF_RETURN_IF_ERROR(GetNodeAttr(node, "value", &shape_and_slices_tensor));
148 const auto& shape_and_slices_value =
149 shape_and_slices_tensor.flat<string>();
150 *shape_slice_string = shape_and_slices_value(offset);
151 return Status::OK();
152 }
153 }
154 return errors::Internal("Unable to find slice for variable: ", target_name);
155}

157Status ReadTensorFromCheckpoint(
158 const string& tensor_name, const std::unique_ptr<BundleReader>& ckpt_reader,
159 const string& shape_and_slice, Tensor* tensor) {
160 if (ckpt_reader) {
161 TensorShape parsed_full_shape;
162 TensorSlice parsed_slice;
163 TensorShape parsed_slice_shape;
164
165 bool get_slice = false;
166 if (!shape_and_slice.empty()) {
167 TF_RETURN_IF_ERROR(
168 checkpoint::ParseShapeAndSlice(shape_and_slice, &parsed_full_shape,
169 &parsed_slice, &parsed_slice_shape));
170 get_slice = (parsed_full_shape != parsed_slice_shape);
171 }
172 if (get_slice) {
173 TF_RETURN_IF_ERROR(ckpt_reader->LookupSlice(
174 GetMonolithicTensorKey(tensor_name), parsed_slice, tensor));
175 } else {
176 TF_RETURN_IF_ERROR(
177 ckpt_reader->Lookup(GetMonolithicTensorKey(tensor_name), tensor));
178 }
179 return Status::OK();
180 }
181 return errors::Internal("Checkpoint reader was not initialized. ");
182}

184Status InitializeCheckpointReader(const TransformFuncContext& context,
185 std::unique_ptr<BundleReader>* ckpt_reader) {
186 if (context.params.count("input_checkpoint")) {
187 const string input_checkpoint = context.params.at("input_checkpoint")[0];
188 ckpt_reader->reset(new BundleReader(Env::Default(), input_checkpoint));
189 TF_RETURN_IF_ERROR((*ckpt_reader)->status());
190 }
191 return Status::OK();
192}

194Status ObtainVariableInfo(
195 const GraphDef& input_graph_def,
196 std::unique_ptr<std::unordered_map<string, string> >* shapes_and_slices) {
197 shapes_and_slices->reset(new std::unordered_map<string, string>());
198 for (const auto& node : input_graph_def.node()) {
199 if ((node.op() == "Variable") || (node.op() == "VariableV2")) {
200 string s;
201 TF_RETURN_IF_ERROR(ObtainTensorSlice(input_graph_def, node.name(), &s));
202 (**shapes_and_slices)[node.name()] = s;
203 }
204 }
205 return Status::OK();
206}

208Status RemoveInputAtIndex(NodeDef* n, int index) {
209 for (int i = index; i < n->input_size() - 1; i++) {
210 n->mutable_input()->SwapElements(i, i + 1);
211 }
212 n->mutable_input()->RemoveLast();
213 return Status::OK();
214}
215
216Status RemoveNodeAtIndex(GraphDef* g, int index) {
217 for (int i = index; i < g->node_size() - 1; i++) {
218 g->mutable_node()->SwapElements(i, i + 1);
219 }
220 g->mutable_node()->RemoveLast();
221 return Status::OK();
222}

224Status SparsifyGatherInternal(
225 const GraphDef& input_graph_def,
226 const std::unique_ptr<std::unordered_map<string, string> >&
227 shapes_and_slices,
228 const TransformFuncContext& context, const OpTypePattern& pattern,
229 const std::unique_ptr<BundleReader>& ckpt_reader,
230 GraphDef* output_graph_def) {
231 string group_init_node = "group_deps";
232 if (context.params.count("group_init_node")) {
233 group_init_node = context.params.at("group_init_node")[0];
234 }
235 GraphDef current_graph_def = input_graph_def;
236 bool any_match_found = false;
237
238 // Populate references.
239 std::unordered_map<string, int> refs;
240 for (const auto& node : current_graph_def.node()) {
241 for (const auto& input : node.input()) {
242 auto parsed_input = StringReplace(input, "^", "", true);
243 refs[parsed_input] += 1;
244 }
245 }
246
247 // The subgraphs may have overlapping components, therefore GraphMatcher
248 // doesn't return all subgraphs in one round -- this has to be multi-round
249 // update.
250 do {
251 any_match_found = false;
252 GraphDef replaced_graph_def = current_graph_def;
253 std::vector<string> init_table_node_names;
254 std::vector<string> removed_node_names;
255
256 TF_RETURN_IF_ERROR(ReplaceMatchingOpTypes(
257 current_graph_def, pattern,
258 [&ckpt_reader, &any_match_found, &init_table_node_names,
259 &shapes_and_slices, &removed_node_names,
260 &refs](const NodeMatch& match, const std::set<string>& input_nodes,
261 const std::set<string>& output_nodes,
262 std::vector<NodeDef>* new_nodes) {
263 any_match_found = true;
264
265 // The captured subgraph should be of the following pattern:
266 // Const --> Identity --> Gather --> ...
267 // ^
268 // |
269 // (ids)
270 //
271 // After transform, it becomes:
272 // --> NoOp(group_deps)
273 // |
274 // Const --> InitializeTable --> HashTable
275 // ^ |
276 // | |
277 // Const ------------- |
278 // v
279 // (ids) ---> LookupTableFind <--- Const(default)
280 // |
281 // v
282 // ...
283
284 // clang-format off
285 // For each subgraph, do the following
286 // 1. Sparsify the `Const`, creating two `Const`, for hashtable
287 // key/val.
288 // 2. Create a `InitializeTable` op connecting to the above 2 `Const`.
289 // 3. Create a `HashTable` op connecting to `InitializeTable` op.
290 // 4. Replace the `Gather` with a `LookupTableFind` op.
291 // 5. Connect the `LookupTableFind` with
292 // a. `HashTable`
293 // b. `Gather`'s ids input
294 // c. a `default_val` arg, valued at 0
295 // clang-format on
296 const NodeDef& gather_node = match.node;
297
298 // GatherV2 adds an "axis" parameter. sparsify_gather only supports
299 // axis 0 gathers.
300 if (gather_node.op() == "GatherV2") {
301 // Per the OpTypePattern, the 3rd input to Gather must be a Const.
302 const NodeDef& axis_node = match.inputs[2].node;
303
304 Tensor axis_t;
305 TF_RETURN_IF_ERROR(GetNodeAttr(axis_node, "value", &axis_t));
306 int64 axis = 0;
307 if (axis_t.dtype() == DT_INT32) {
308 axis = axis_t.scalar<int32>()();
309 } else if (axis_t.dtype() == DT_INT64) {
310 axis = axis_t.scalar<int64>()();
311 } else {
312 return tensorflow::errors::FailedPrecondition(
313 "Gather axis was not int32 or int64.");
314 }
315
316 if (axis != 0) {
317 return tensorflow::errors::FailedPrecondition(
318 "Transform only applicable to subgraph with GatherV2 over "
319 "axis 0. Found axis ",
320 axis, ".");
321 }
322 }
323
324 const NodeDef& weights_node = match.inputs[0].inputs[0].node;
325
326 DataType data_type;
327 TF_RETURN_IF_ERROR(GetNodeAttr(weights_node, "dtype", &data_type));
328 if (data_type != DT_FLOAT) {
329 return tensorflow::errors::FailedPrecondition(
330 "Transform only applicable to subgraph with 'Const',"
331 "'Variable', or 'VariableV2' of dtype "
332 "'DT_FLOAT'. Found '" +
333 weights_node.op() + "' with name '",
334 weights_node.name(), "' and dtype '", data_type, "'.");
335 }
336
337 Tensor weight;
338 if (weights_node.op() == "Const") {
339 weight = GetNodeTensorAttr(weights_node, "value");
340 } else {
341 TF_RETURN_IF_ERROR(ReadTensorFromCheckpoint(
342 weights_node.name(), ckpt_reader,
343 (*shapes_and_slices)[weights_node.name()], &weight));
344 }
345 // Add both both weight and identity node names.
346 removed_node_names.push_back(weights_node.name());
347 removed_node_names.push_back(match.inputs[0].node.name());
348 for (auto input_node : match.inputs[0].node.input()) {
349 auto parsed_input = StringReplace(input_node, "^", "", true);
350 refs[parsed_input]--;
351 }
352 Tensor indices_tensor;
353 Tensor values_tensor;
354 TF_RETURN_IF_ERROR(
355 SparsifyWeights(weight, &indices_tensor, &values_tensor));
356
357 // indices and values of sparsified `Const`
358 DataType key_dtype = DT_INT64;
359 NodeDef indices_node;
360 CreateConstNode(indices_tensor,
361 StrCat(weights_node.name(), "/indices"),
362 &indices_node);
363 SetNodeAttr("dtype", key_dtype, &indices_node);
364
365 NodeDef values_node;
366 CreateConstNode(values_tensor, StrCat(weights_node.name(), "/values"),
367 &values_node);
368 SetNodeAttr("dtype", data_type, &values_node);
369
370 // HashTable node
371 NodeDef hashtable_node;
372 hashtable_node.set_op("HashTable");
373 hashtable_node.set_name(StrCat(weights_node.name(), "/HashTable"));
374 SetNodeAttr("key_dtype", key_dtype, &hashtable_node);
375 SetNodeAttr("value_dtype", data_type, &hashtable_node);
376
377 // InitializeTable node
378 NodeDef init_table_node;
379 init_table_node.set_op("InitializeTable");
380 init_table_node.set_name(
381 StrCat(weights_node.name(), "/InitializeTable"));
382 SetNodeAttr("Tkey", key_dtype, &init_table_node);
383 SetNodeAttr("Tval", data_type, &init_table_node);
384 init_table_node_names.push_back(init_table_node.name());
385
386 // LookupTableFind node
387 NodeDef lookup_node;
388 lookup_node.set_op("LookupTableFind");
389 lookup_node.set_name(StrCat(gather_node.name(), "/LookupTableFind"));
390 SetNodeAttr("Tin", key_dtype, &lookup_node);
391 SetNodeAttr("Tout", data_type, &lookup_node);
392
393 // Default return value of hashtable lookup
394 Tensor zero_tensor(data_type, TensorShape({}));
395 zero_tensor.flat<float>()(0) = 0.0;
396 NodeDef default_value_node;
397 CreateConstNode(zero_tensor, StrCat(gather_node.name(), "/Const"),
398 &default_value_node);
399 SetNodeAttr("dtype", data_type, &default_value_node);
400
401 // ExpandDims argument
402 Tensor dim_idx(DT_INT32, TensorShape({}));
403 dim_idx.flat<int32>()(0) = -1;
404 NodeDef dim_idx_node;
405 dim_idx_node.set_op("Const");
406 dim_idx_node.set_name(
407 StrCat(gather_node.name(), "/ExpandDims/Const"));
408 SetNodeAttr("value", dim_idx, &dim_idx_node);
409 SetNodeAttr("dtype", DT_INT32, &dim_idx_node);
410
411 // ExpandDims node
412 NodeDef expand_dims_node;
413 expand_dims_node.set_op("ExpandDims");
414 // Reuse gather_node's name so not to change dependent's inputs
415 expand_dims_node.set_name(gather_node.name());
416 SetNodeAttr("T", data_type, &expand_dims_node);
417
418 // Connect nodes
419 AddNodeInput(hashtable_node.name(), &init_table_node);
420 refs[hashtable_node.name()]++;
421 AddNodeInput(indices_node.name(), &init_table_node);
422 refs[indices_node.name()]++;
423 AddNodeInput(values_node.name(), &init_table_node);
424 refs[values_node.name()]++;
425
426 AddNodeInput(hashtable_node.name(), &lookup_node);
427 refs[hashtable_node.name()]++;
428 AddNodeInput(gather_node.input(1), &lookup_node);
429 refs[gather_node.input(1)]++;
430 AddNodeInput(default_value_node.name(), &lookup_node);
431 refs[default_value_node.name()]++;
432
433 AddNodeInput(lookup_node.name(), &expand_dims_node);
434 refs[lookup_node.name()]++;
435 AddNodeInput(dim_idx_node.name(), &expand_dims_node);
436 refs[dim_idx_node.name()]++;
437
438 // Copy 'ids' input of original 'Gather'
439 new_nodes->push_back(match.inputs[1].node);
440 new_nodes->push_back(indices_node);
441 new_nodes->push_back(values_node);
442 new_nodes->push_back(hashtable_node);
443 new_nodes->push_back(init_table_node);
444 new_nodes->push_back(lookup_node);
445 new_nodes->push_back(default_value_node);
446 new_nodes->push_back(dim_idx_node);
447 new_nodes->push_back(expand_dims_node);
448
449 return Status::OK();
450 },
451 {true}, &replaced_graph_def));
452
453 NodeDef* init_op = nullptr;
454 for (int i = 0; i < replaced_graph_def.node_size(); i++) {
455 if (replaced_graph_def.node(i).name() == group_init_node &&
456 replaced_graph_def.node(i).op() == "NoOp") {
457 init_op = replaced_graph_def.mutable_node(i);
458 break;
459 }
460 }
461 if (!init_op) {
462 // Init node
463 init_op = replaced_graph_def.mutable_node()->Add();
464 init_op->set_op("NoOp");
465 init_op->set_name(group_init_node);
466 }
467 for (const string& name : init_table_node_names) {
468 // Add control dependence from init_table_node to group_deps_node
469 AddNodeInput(StrCat("^", name), init_op);
470 refs[name]++;
471 }
472
473 // Erase inputs and outputs as they are not considered for deletion.
474 for (const auto& output : context.output_names) {
475 refs.erase(output);
476 }
477
478 for (const auto& input : context.input_names) {
479 refs.erase(input);
480 }
481
482 // Add nodes with a reference count of 0 for deletion.
483 for (auto entry : refs) {
484 if (entry.second == 0) {
485 removed_node_names.push_back(entry.first);
486 }
487 }
488
489 while (!removed_node_names.empty()) {
490 auto name = removed_node_names.back();
491 removed_node_names.pop_back();
492
493 int i = 0;
494 while (i < replaced_graph_def.node_size()) {
495 // Revisit this to see if we can safely remove RestoreV2 nodes.
496 if ((replaced_graph_def.node(i).name() == name) &&
497 (replaced_graph_def.node(i).op() != "RestoreV2")) {
498 for (const auto& input : replaced_graph_def.node(i).input()) {
499 auto parsed_input = StringReplace(input, "^", "", true);
500 refs[parsed_input] -= 1;
501 if (refs[parsed_input] == 0) {
502 removed_node_names.push_back(parsed_input);
503 }
504 }
505 TF_RETURN_IF_ERROR(RemoveNodeAtIndex(&replaced_graph_def, i));
506 continue;
507 }
508 int j = 0;
509 bool deleted_inputs = false;
510 while (j < replaced_graph_def.node(i).input_size()) {
511 if (replaced_graph_def.node(i).input(j) == name ||
512 replaced_graph_def.node(i).input(j) == ("^" + name)) {
513 TF_RETURN_IF_ERROR(
514 RemoveInputAtIndex(replaced_graph_def.mutable_node(i), j));
515 deleted_inputs = true;
516 continue;
517 }
518 j++;
519 }
520 if (deleted_inputs) {
521 if (replaced_graph_def.node(i).op() == "ConcatV2") {
522 if (replaced_graph_def.node(i).input_size() > 2) {
523 SetNodeAttr("N", replaced_graph_def.node(i).input_size() - 1,
524 replaced_graph_def.mutable_node(i));
525 } else if (replaced_graph_def.node(i).input_size() == 2) {
526 if (refs[replaced_graph_def.node(i).input(1)] != 1) {
527 return errors::Internal(
528 "Expect axis tensor of ConcatV2 node to only be referenced "
529 "once.");
530 }
531 refs[replaced_graph_def.node(i).input(1)] -= 1;
532 removed_node_names.push_back(replaced_graph_def.node(i).input(1));
533 replaced_graph_def.mutable_node(i)->mutable_input()->RemoveLast();
534 replaced_graph_def.mutable_node(i)->mutable_attr()->erase("N");
535 replaced_graph_def.mutable_node(i)->set_op("Identity");
536 } else {
537 return errors::Internal(
538 "ConcatV2 should have at least two elements");
539 }
540 }
541 if ((replaced_graph_def.node(i).op() == "Assign" ||
542 replaced_graph_def.node(i).op() == "Reshape" ||
543 replaced_graph_def.node(i).op() == "Equal" ||
544 replaced_graph_def.node(i).op() == "Mean" ||
545 replaced_graph_def.node(i).op() == "ScalarSummary") &&
546 replaced_graph_def.node(i).input_size() == 1) {
547 removed_node_names.push_back(replaced_graph_def.node(i).name());
548 }
549 if (!replaced_graph_def.node(i).input_size()) {
550 removed_node_names.push_back(replaced_graph_def.node(i).name());
551 }
552 }
553 i++;
554 }
555 }
556 current_graph_def = replaced_graph_def;
557 } while (any_match_found);
558 *output_graph_def = current_graph_def;
559 return Status::OK();
560}

562Status SparsifyGather(const GraphDef& input_graph_def,
563 const TransformFuncContext& context,
564 GraphDef* output_graph_def) {
565 // clang-format off
566 const OpTypePattern gather_pattern =
567 {"Gather",
568 {
569 {"Identity",
570 {
571 {"Const|Variable|VariableV2"}
572 }
573 },
574 {"*"},
575 }
576 };
577 const OpTypePattern gather_v2_pattern =
578 {"GatherV2",
579 {
580 {"Identity",
581 {
582 {"Const|Variable|VariableV2"}
583 }
584 },
585 {"*"},
586 // GatherV2's axis must be constant.
587 {"Const"},
588 }
589 };
590 // clang-format on
591
592 GraphDef cleaned_input_graph_def;
593 RemoveAttributes(input_graph_def, {"_output_shapes"},
594 &cleaned_input_graph_def);
595
596 GraphDef temp_output;
597
598 std::unique_ptr<BundleReader> ckpt_reader;
599 TF_RETURN_IF_ERROR(InitializeCheckpointReader(context, &ckpt_reader));
600
601 std::unique_ptr<std::unordered_map<string, string> > shapes_and_slices;
602 TF_RETURN_IF_ERROR(
603 ObtainVariableInfo(cleaned_input_graph_def, &shapes_and_slices));
604
605 TF_RETURN_IF_ERROR(SparsifyGatherInternal(
606 cleaned_input_graph_def, shapes_and_slices, context, gather_pattern,
607 ckpt_reader, &temp_output));
608
609 TF_RETURN_IF_ERROR(SparsifyGatherInternal(temp_output, shapes_and_slices,
610 context, gather_v2_pattern,
611 ckpt_reader, output_graph_def));
612
613 return Status::OK();
614}
