30// Converts Conv2D or MatMul ops followed by column-wise Muls into equivalent
31// ops with the Mul baked into the convolution weights, to save computation
32// during inference.
33Status FoldBatchNorms(const GraphDef& input_graph_def,
34 const TransformFuncContext& context,
35 GraphDef* output_graph_def) {
36 GraphDef replaced_graph_def;
37 TF_RETURN_IF_ERROR(ReplaceMatchingOpTypes(
38 input_graph_def, // clang-format off
39 {"Mul", // mul_node
40 {
41 {"Conv2D|MatMul", // conv_node
42 {
43 {"*"}, // input_node
44 {"Const"}, // weights_node
45 }
46 },
47 {"Const"}, // mul_values_node
48 }
49 }, // clang-format on
50 [](const NodeMatch& match, const std::set<string>& input_nodes,
51 const std::set<string>& output_nodes,
52 std::vector<NodeDef>* new_nodes) {
53 // Find all the nodes we expect in the subgraph.
54 const NodeDef& mul_node = match.node;
55 const NodeDef& conv_node = match.inputs[0].node;
56 const NodeDef& input_node = match.inputs[0].inputs[0].node;
57 const NodeDef& weights_node = match.inputs[0].inputs[1].node;
58 const NodeDef& mul_values_node = match.inputs[1].node;
59
60 // Check that nodes that we use are not used somewhere else.
61 for (const auto& node : {conv_node, weights_node, mul_values_node}) {
62 if (output_nodes.count(node.name())) {
63 // Return original nodes.
64 new_nodes->insert(new_nodes->end(),
65 {mul_node, conv_node, input_node, weights_node,
66 mul_values_node});
67 return Status::OK();
68 }
69 }
70
71 Tensor weights = GetNodeTensorAttr(weights_node, "value");
72 Tensor mul_values = GetNodeTensorAttr(mul_values_node, "value");
73
74 // Make sure all the inputs really are vectors, with as many entries as
75 // there are columns in the weights.
76 const int weights_cols_index = conv_node.op() == "Conv2D" ? 3 : 1;
77 const int64 weights_cols = weights.shape().dim_size(weights_cols_index);
78 if ((mul_values.shape().dims() != 1) ||
79 (mul_values.shape().dim_size(0) != weights_cols)) {
80 return errors::InvalidArgument(
81 "Mul constant input to batch norm has bad shape: ",
82 mul_values.shape().DebugString());
83 }
84
85 // Multiply the original weights by the scale vector.
86 auto weights_matrix = weights.flat_inner_dims<float>();
87 Tensor scaled_weights(DT_FLOAT, weights.shape());
88 auto scaled_weights_matrix = scaled_weights.flat_inner_dims<float>();
89 for (int64 row = 0; row < weights_matrix.dimension(0); ++row) {
90 for (int64 col = 0; col < weights_cols; ++col) {
91 scaled_weights_matrix(row, col) =
92 weights_matrix(row, col) * mul_values.flat<float>()(col);
93 }
94 }
95
96 // Construct the new nodes.
97 NodeDef scaled_weights_node;
98 scaled_weights_node.set_op("Const");
99 scaled_weights_node.set_name(weights_node.name());
100 SetNodeAttr("dtype", DT_FLOAT, &scaled_weights_node);
101 SetNodeTensorAttr<float>("value", scaled_weights, &scaled_weights_node);
102 new_nodes->push_back(scaled_weights_node);
103
104 new_nodes->push_back(input_node);
105
106 NodeDef new_conv_node;
107 new_conv_node = conv_node;
108 new_conv_node.set_name(mul_node.name());
109 new_nodes->push_back(new_conv_node);
110
111 return Status::OK();
112 },
113 {}, &replaced_graph_def));
114 *output_graph_def = replaced_graph_def;
115 return Status::OK();
116}
117
118REGISTER_GRAPH_TRANSFORM("fold_batch_norms", FoldBatchNorms);
