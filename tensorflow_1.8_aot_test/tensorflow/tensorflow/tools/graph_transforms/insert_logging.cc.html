31// Clears the device field of all ops in the graph.
32Status InsertLogging(const GraphDef& input_graph_def,
33 const TransformFuncContext& context,
34 GraphDef* output_graph_def) {
35 std::unordered_set<string> ops;
36 bool has_ops;
37 if (context.params.count("op")) {
38 has_ops = true;
39 for (const string& op : context.params.at("op")) {
40 ops.insert(op);
41 }
42 } else {
43 has_ops = false;
44 }
45
46 std::unordered_set<string> prefixes;
47 bool has_prefixes;
48 if (context.params.count("prefix")) {
49 has_prefixes = true;
50 for (const string& prefix : context.params.at("prefix")) {
51 prefixes.insert(prefix);
52 }
53 } else {
54 has_prefixes = false;
55 }
56
57 string message;
58 TF_RETURN_IF_ERROR(context.GetOneStringParameter("message", "", &message));
59
60 bool show_name;
61 TF_RETURN_IF_ERROR(
62 context.GetOneBoolParameter("show_name", false, &show_name));
63
64 bool show_op;
65 TF_RETURN_IF_ERROR(context.GetOneBoolParameter("show_op", false, &show_op));
66
67 int32 first_n;
68 TF_RETURN_IF_ERROR(context.GetOneInt32Parameter("first_n", -1, &first_n));
69
70 int32 summarize;
71 TF_RETURN_IF_ERROR(
72 context.GetOneInt32Parameter("summarize", 1024, &summarize));
73
74 std::unordered_map<string, std::set<int>> node_outputs;
75 for (const NodeDef& node : input_graph_def.node()) {
76 for (const string& input : node.input()) {
77 const string canonical_input = CanonicalInputName(input);
78 string prefix;
79 string name;
80 string suffix;
81 NodeNamePartsFromInput(canonical_input, &prefix, &name, &suffix);
82 const string output_index_string = suffix.substr(1, suffix.size() - 1);
83 int32 output_index;
84 if (!strings::safe_strto32(output_index_string, &output_index)) {
85 return errors::InvalidArgument("Couldn't understand output number in ",
86 input);
87 }
88 node_outputs[name].insert(output_index);
89 }
90 }
91
92 std::map<string, string> inputs_to_rename;
93 std::unordered_set<string> ignore_when_renaming;
94 GraphDef logged_graph_def;
95 for (const NodeDef& node : input_graph_def.node()) {
96 NodeDef* new_node = logged_graph_def.mutable_node()->Add();
97 *new_node = node;
98 if (node_outputs[node.name()].empty()) {
99 // There were no outputs found to this node, so skip it.
100 continue;
101 }
102 const bool op_matches = (ops.count(node.op()) > 0);
103 bool prefix_matches = false;
104 for (const string& prefix : prefixes) {
105 if (str_util::StartsWith(node.name(), prefix)) {
106 prefix_matches = true;
107 }
108 }
109 // If we're not looking for ops, or we found the right op, and if we're not
110 // looking for prefixes or we found the right prefix, then add logging here.
111 if ((!has_ops || op_matches) && (!has_prefixes || prefix_matches)) {
112 const string name_suffix = "__print__";
113 DataTypeVector input_types;
114 DataTypeVector output_types;
115 TF_RETURN_IF_ERROR(GetInOutTypes(node, &input_types, &output_types));
116 NodeDef* print_node = logged_graph_def.mutable_node()->Add();
117 print_node->set_op("Print");
118 print_node->set_name(strings::StrCat(node.name(), name_suffix));
119 string node_message;
120 if (show_op) {
121 node_message += ";" + node.op() + ";";
122 }
123 if (show_name) {
124 node_message += ";" + print_node->name() + ";";
125 }
126 node_message += message;
127 SetNodeAttr("message", node_message, print_node);
128 SetNodeAttr("first_n", first_n, print_node);
129 SetNodeAttr("summarize", summarize, print_node);
130 print_node->add_input(node.name() + ":0");
131 SetNodeAttr("T", output_types[0], print_node);
132 for (int output_index : node_outputs[node.name()]) {
133 print_node->add_input(strings::StrCat(node.name(), ":", output_index));
134 }
135 SetNodeAttr("U", output_types, print_node);
136 ignore_when_renaming.insert(print_node->name());
137 // Rewrite the graph so all references to the first input of the original
138 // op now pull from the print op instead, so it's executed.
139 inputs_to_rename[node.name() + ":0"] =
140 strings::StrCat(node.name(), name_suffix, ":0");
141 }
142 }
143
144 output_graph_def->Clear();
145 return RenameNodeInputs(logged_graph_def, inputs_to_rename,
146 ignore_when_renaming, output_graph_def);
147}

149REGISTER_GRAPH_TRANSFORM("insert_logging", InsertLogging);
