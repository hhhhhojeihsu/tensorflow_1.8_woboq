32// Holds the information we need to translate from a float version of this op
33// into the quantized equivalent.
34struct QuantizedOpInfo {
35 // The name of the float op.
36 string float_name;
37 // Which attributes to copy directly over.
38 std::vector<string> attrs_to_copy;
39 // Extra data type attributes we need to set.
40 std::vector<std::pair<string, DataType>> dtypes_to_set;
41 // What depth of inputs the op can read in.
42 DataType input_bit_depth;
43 // The depth of the op's quantized outputs.
44 DataType output_bit_depth;
45 // Which inputs (e.g. shapes) aren't involved in the quantization process.
46 std::set<int32> unquantized_inputs;
47 // How the outputs are arranged, either
48 // [input0, input1, min0, max0, min1, max1] for contiguous, or
49 // [input0, input1, min0, min1, max0, max1] for separate.
50 // The separate order is needed because it's the only way to specify unknown
51 // numbers of inputs for ops like Concat.
52 enum { CONTIGUOUS_MIN_MAX, SEPARATE_MIN_MAX } min_max_order;
53};

148// Replaces invalid characters in input names to get a unique node name.
149string UniqueNodeNameFromInput(const string& input_name) {
150 string prefix;
151 string node_name;
152 string suffix;
153 NodeNamePartsFromInput(input_name, &prefix, &node_name, &suffix);
154 string result;
155 if (prefix == "^") {
156 result += "__hat__";
157 }
158 result += node_name;
159 if (!suffix.empty()) {
160 result += "__port__" + suffix.substr(1, suffix.size() - 1);
161 }
162 return result;
163}

165// Pulls two float values from the named parameters, with a lot of checking.
166Status ExtractRangeFromParams(const TransformFuncContext& context,
167 const string& min_name, const string& max_name,
168 float* min_value, float* max_value,
169 bool* has_range) {
170 // See if we've been given quantized inputs with a known range.
171 const bool has_min = (context.params.count(min_name) != 0);
172 const bool has_max = (context.params.count(max_name) != 0);
173 *has_range = (has_min || has_max);
174 if (!*has_range) {
175 return Status::OK();
176 }
177 if (!has_min || !has_max) {
178 return errors::InvalidArgument("You must pass both ", min_name, " and ",
179 max_name, " into quantize_nodes");
180 }
181 TF_RETURN_IF_ERROR(context.GetOneFloatParameter(min_name, 0.0f, min_value));
182 TF_RETURN_IF_ERROR(context.GetOneFloatParameter(max_name, 0.0f, max_value));
183 return Status::OK();
184}

188// Analyzes all the nodes in the graph to figure out which ones are duplicates
189// apart from their names. This commonly includes identical Const nodes, but can
190// also be simple operations that are repeated on multiple outputs of a
191// particular node. The complexity is managed using a hash function that avoids
192// the need for any O(n^2) algorithms when identifying duplicates.
193Status MergeDuplicateNodes(const GraphDef& input_graph_def,
194 const TransformFuncContext& context,
195 GraphDef* output_graph_def) {
196 // Make sure we can look up inputs and outputs quickly.
197 std::set<string> input_names(context.input_names.begin(),
198 context.input_names.end());
199 std::set<string> output_names(context.output_names.begin(),
200 context.output_names.end());
201 GraphDef current_graph_def = input_graph_def;
202 // Keep running the merging until no more duplicates are found.
203 bool any_duplicates_found;
204 do {
205 any_duplicates_found = false;
206 // First arrange all of the nodes by a hash of their contents.
207 std::map<uint64, std::vector<const NodeDef*>> hashed_nodes;
208 for (const NodeDef& node : current_graph_def.node()) {
209 NodeDef nameless_node = node;
210 // The name matters if it's being used as an input or output node,
211 // otherwise ignore it when looking for duplicates.
212 if (!input_names.count(node.name()) && !output_names.count(node.name())) {
213 nameless_node.set_name("");
214 }
215 const uint64 hash = HashNodeDef(nameless_node);
216 hashed_nodes[hash].push_back(&node);
217 }
218 // If we have multiple nodes with the same hash, then we know they're
219 // duplicates and can be removed, unless they're stateful.
220 std::map<string, string> inputs_to_rename;
221 GraphDef merged_graph_def;
222 for (const std::pair<uint64, std::vector<const NodeDef*>> hashed_node_info :
223 hashed_nodes) {
224 const std::vector<const NodeDef*>& hash_node_list =
225 hashed_node_info.second;
226 for (int i = 0; i < hash_node_list.size(); ++i) {
227 const NodeDef* current_node = hash_node_list[i];
228 const OpDef* op_def = nullptr;
229 TF_RETURN_IF_ERROR(
230 OpRegistry::Global()->LookUpOpDef(current_node->op(), &op_def));
231 const bool is_duplicate = ((!op_def->is_stateful()) && (i > 0));
232 if (is_duplicate) {
233 const string original_name = hash_node_list[0]->name();
234 inputs_to_rename[current_node->name() + ":*"] = original_name;
235 any_duplicates_found = true;
236 } else {
237 NodeDef* new_node = merged_graph_def.mutable_node()->Add();
238 *new_node = *current_node;
239 }
240 }
241 }
242 // Update the graph so that any nodes that referred to removed inputs now
243 // pull from the remaining duplicate.
244 TF_RETURN_IF_ERROR(RenameNodeInputs(merged_graph_def, inputs_to_rename,
245 std::unordered_set<string>(),
246 &current_graph_def));
247 } while (any_duplicates_found);
248
249 *output_graph_def = current_graph_def;
250
251 return Status::OK();
252}

254// Looks for the patterns that indicate there are two eight-bit ops feeding into
255// each other, separated by a conversion up to float and back again. These occur
256// during the initial conversion of ops to their quantized forms. Because we're
257// only looking at an individual op in that phase and don't know if its inputs
258// and outputs are eight-bit-capable, we start by converting the actual op into
259// quantized form, but add float conversions before and after. This pass gets
260// rid of those conversions if it turns out we do have adjacent ops capable of
261// eight-bit processing.
262Status RemoveRedundantQuantizations(const GraphDef& input_graph_def,
263 const TransformFuncContext& context,
264 GraphDef* output_graph_def) {
265 std::set<string> graph_outputs;
266 for (const string& output_name : context.output_names) {
267 graph_outputs.insert(NodeNameFromInput(output_name));
268 }
269 std::map<string, string> inputs_to_rename;
270 GraphDef replaced_graph_def;
271 TF_RETURN_IF_ERROR(ReplaceMatchingOpTypes(
272 input_graph_def, // clang-format off
273 {"QuantizeV2",
274 {
275 {"Dequantize"},
276 {"Min"},
277 {"Max"},
278 }
279 }, // clang-format on
280 [&inputs_to_rename, &graph_outputs](const NodeMatch& match,
281 const std::set<string>& input_nodes,
282 const std::set<string>& output_nodes,
283 std::vector<NodeDef>* new_nodes) {
284 const NodeDef& quantize_node = match.node;
285 const NodeDef& dequantize_node = match.inputs[0].node;
286 inputs_to_rename[quantize_node.name() + ":0"] =
287 dequantize_node.input(0);
288 inputs_to_rename[quantize_node.name() + ":1"] =
289 dequantize_node.input(1);
290 inputs_to_rename[quantize_node.name() + ":2"] =
291 dequantize_node.input(2);
292
293 // Are other sub-graphs using the float intermediate result? If so,
294 // preserve it, but the input renaming still rewires the eight-bit ops
295 // so they don't go through float.
296 if (output_nodes.count(dequantize_node.name()) ||
297 graph_outputs.count(dequantize_node.name())) {
298 CopyOriginalMatch(match, new_nodes);
299 }
300
301 return Status::OK();
302 },
303 {true}, &replaced_graph_def));
304
305 return RenameNodeInputs(replaced_graph_def, inputs_to_rename,
306 std::unordered_set<string>(), output_graph_def);
307}

309// If the user has passed in the input_min and input_max args, then we need to
310// convert any input placeholders from float to eight bit, so quantized inputs
311// can be fed directly into the graph.
312Status QuantizePlaceholders(const GraphDef& input_graph_def,
313 const TransformFuncContext& context,
314 GraphDef* output_graph_def) {
315 float input_min;
316 float input_max;
317 bool has_input_range;
318 TF_RETURN_IF_ERROR(ExtractRangeFromParams(context, "input_min", "input_max",
319 &input_min, &input_max,
320 &has_input_range));
321 if (!has_input_range) {
322 *output_graph_def = input_graph_def;
323 return Status::OK();
324 }
325 std::map<string, string> inputs_to_rename_first_pass;
326 std::map<string, string> inputs_to_rename_second_pass;
327 GraphDef placeholder_graph_def;
328 placeholder_graph_def.Clear();
329 for (const NodeDef& node : input_graph_def.node()) {
330 if (node.op() != "Placeholder") {
331 *(placeholder_graph_def.mutable_node()->Add()) = node;
332 } else {
333 string namespace_prefix = node.name() + "_eightbit";
334
335 NodeDef quantized_placeholder;
336 quantized_placeholder = node;
337 SetNodeAttr("dtype", DT_QUINT8, &quantized_placeholder);
338 *(placeholder_graph_def.mutable_node()->Add()) = quantized_placeholder;
339
340 NodeDef min_node;
341 min_node.set_op("Const");
342 min_node.set_name(namespace_prefix + "/min");
343 SetNodeAttr("dtype", DT_FLOAT, &min_node);
344 Tensor min_tensor(DT_FLOAT, {});
345 min_tensor.flat<float>()(0) = input_min;
346 SetNodeTensorAttr<float>("value", min_tensor, &min_node);
347 *(placeholder_graph_def.mutable_node()->Add()) = min_node;
348
349 NodeDef max_node;
350 max_node.set_op("Const");
351 max_node.set_name(namespace_prefix + "/max");
352 SetNodeAttr("dtype", DT_FLOAT, &max_node);
353 Tensor max_tensor(DT_FLOAT, {});
354 max_tensor.flat<float>()(0) = input_max;
355 SetNodeTensorAttr<float>("value", max_tensor, &max_node);
356 *(placeholder_graph_def.mutable_node()->Add()) = max_node;
357
358 const string rename_suffix = "__RENAMED_PLACEHOLDER__";
359 NodeDef dequantize_node;
360 dequantize_node.set_op("Dequantize");
361 dequantize_node.set_name(namespace_prefix + "/dequantize");
362 SetNodeAttr("T", DT_QUINT8, &dequantize_node);
363 SetNodeAttr("mode", "MIN_FIRST", &dequantize_node);
364 AddNodeInput(node.name() + rename_suffix, &dequantize_node);
365 AddNodeInput(min_node.name(), &dequantize_node);
366 AddNodeInput(max_node.name(), &dequantize_node);
367 *(placeholder_graph_def.mutable_node()->Add()) = dequantize_node;
368
369 // First make sure that any internal references to the old placeholder
370 // now point to the dequantize result.
371 inputs_to_rename_first_pass[node.name()] = dequantize_node.name();
372 // Then fix up the dequantize op so that it really points to the
373 // placeholder.
374 inputs_to_rename_second_pass[node.name() + rename_suffix] = node.name();
375 }
376 }
377
378 GraphDef first_pass_graph_def;
379 TF_RETURN_IF_ERROR(
380 RenameNodeInputs(placeholder_graph_def, inputs_to_rename_first_pass,
381 std::unordered_set<string>(), &first_pass_graph_def));
382 TF_RETURN_IF_ERROR(
383 RenameNodeInputs(first_pass_graph_def, inputs_to_rename_second_pass,
384 std::unordered_set<string>(), output_graph_def));
385
386 return Status::OK();
387}

389// During training, FakeQuantWithMinMaxVars ops capture a good min/max range for
390// an activation layer. To use these during inference, this pass converts those
391// ops into Requantizes with the trained min/maxes as constant inputs.
392Status ConvertFakeQuantsToRequantize(const GraphDef& input_graph_def,
393 const TransformFuncContext& context,
394 GraphDef* output_graph_def) {
395 TF_RETURN_IF_ERROR(ReplaceMatchingOpTypes(
396 input_graph_def, // clang-format off
397 {"FakeQuantWithMinMaxVars",
398 {
399 {"*"},
400 {"Const"},
401 {"Const"},
402 }
403 }, // clang-format on
404 [](const NodeMatch& match, const std::set<string>& input_nodes,
405 const std::set<string>& output_nodes,
406 std::vector<NodeDef>* new_nodes) {
407 const NodeDef& fake_quant_node = match.node;
408 const NodeDef& original_op_node = match.inputs[0].node;
409 const NodeDef& fake_quant_min_node = match.inputs[1].node;
410 const NodeDef& fake_quant_max_node = match.inputs[2].node;
411
412 string namespace_prefix = fake_quant_node.name() + "_eightbit";
413
414 new_nodes->push_back(original_op_node);
415 new_nodes->push_back(fake_quant_min_node);
416 new_nodes->push_back(fake_quant_max_node);
417
418 NodeDef quantize_node;
419 quantize_node.set_op("QuantizeV2");
420 quantize_node.set_name(namespace_prefix + "/quantize");
421 SetNodeAttr("T", DT_QINT32, &quantize_node);
422 SetNodeAttr("mode", "MIN_FIRST", &quantize_node);
423 AddNodeInput(fake_quant_node.input(0), &quantize_node);
424 AddNodeInput(fake_quant_min_node.name(), &quantize_node);
425 AddNodeInput(fake_quant_max_node.name(), &quantize_node);
426 new_nodes->push_back(quantize_node);
427
428 NodeDef requantize_node;
429 requantize_node.set_op("Requantize");
430 requantize_node.set_name(namespace_prefix + "/requantize");
431 SetNodeAttr("Tinput", DT_QINT32, &requantize_node);
432 SetNodeAttr("out_type", DT_QUINT8, &requantize_node);
433 AddNodeInput(quantize_node.name() + ":0", &requantize_node);
434 AddNodeInput(quantize_node.name() + ":1", &requantize_node);
435 AddNodeInput(quantize_node.name() + ":2", &requantize_node);
436 AddNodeInput(fake_quant_min_node.name(), &requantize_node);
437 AddNodeInput(fake_quant_max_node.name(), &requantize_node);
438 new_nodes->push_back(requantize_node);
439
440 // Convert the 8-bit result back into float for the final output.
441 NodeDef dequantize_node;
442 dequantize_node.set_op("Dequantize");
443 dequantize_node.set_name(fake_quant_node.name());
444 SetNodeAttr("T", DT_QUINT8, &dequantize_node);
445 SetNodeAttr("mode", "MIN_FIRST", &dequantize_node);
446 AddNodeInput(requantize_node.name() + ":0", &dequantize_node);
447 AddNodeInput(requantize_node.name() + ":1", &dequantize_node);
448 AddNodeInput(requantize_node.name() + ":2", &dequantize_node);
449 new_nodes->push_back(dequantize_node);
450
451 return Status::OK();
452 },
453 {}, output_graph_def));
454
455 return Status::OK();
456}

458// We always generate Requantize ops driven by dynamic RequantizationRange
459// calculations when we produce quantized ops like Conv2D or BiasAdd with
460// 32-bit results. If there were FakeQuant ops already for those activation
461// layers, then there will be a later Requantize op with constant min/max
462// inputs, which is preferable for fast inference. This pass looks for those
463// later Requantize ops, and replaces the dynamic version with them.
464Status MergeAdjacentRequantizes(const GraphDef& input_graph_def,
465 const TransformFuncContext& context,
466 GraphDef* output_graph_def) {
467 TF_RETURN_IF_ERROR(ReplaceMatchingOpTypes(
468 input_graph_def, // clang-format off
469 {"Requantize",
470 {
471 {"QuantizeV2",
472 {
473 {"Dequantize",
474 {
475 {"Requantize",
476 {
477 {"*"},
478 {"*"},
479 {"*"},
480 {"RequantizationRange"},
481 {"RequantizationRange"},
482 }
483 },
484 {"Requantize"},
485 {"Requantize"},
486 }
487 },
488 {"Const"},
489 {"Const"},
490 },
491 },
492 {"QuantizeV2"},
493 {"QuantizeV2"},
494 {"Const"},
495 {"Const"},
496 }
497 }, // clang-format on
498 [](const NodeMatch& match, const std::set<string>& input_nodes,
499 const std::set<string>& output_nodes,
500 std::vector<NodeDef>* new_nodes) {
501 const NodeDef& fake_requantize_node = match.node;
502 const NodeDef& original_op_node =
503 match.inputs[0].inputs[0].inputs[0].inputs[0].node;
504 const NodeDef& fake_requantize_min_node = match.inputs[3].node;
505 const NodeDef& fake_requantize_max_node = match.inputs[4].node;
506
507 new_nodes->push_back(original_op_node);
508 new_nodes->push_back(fake_requantize_min_node);
509 new_nodes->push_back(fake_requantize_max_node);
510
511 NodeDef requantize_node;
512 requantize_node = fake_requantize_node;
513 requantize_node.mutable_input()->Clear();
514 AddNodeInput(original_op_node.name() + ":0", &requantize_node);
515 AddNodeInput(original_op_node.name() + ":1", &requantize_node);
516 AddNodeInput(original_op_node.name() + ":2", &requantize_node);
517 AddNodeInput(fake_requantize_min_node.name(), &requantize_node);
518 AddNodeInput(fake_requantize_max_node.name(), &requantize_node);
519 new_nodes->push_back(requantize_node);
520
521 return Status::OK();
522 },
523 {}, output_graph_def));
524
525 return Status::OK();
526}

528// Sometimes FakeQuantWithMinMaxVars ops are added at the end of a chain of
529// linear ops like Relu, MaxPool, etc, several steps from the Conv2D or BiasAdd
530// op that we want to apply the trained constant conversions to. This pass tries
531// to move FakeQuant ops up the input chain, so they're as close as possible to
532// the 32-bit conversion, and so can be easily merged into the automatic dynamic
533// Requantizes.
534Status HoistFakeQuants(const GraphDef& input_graph_def,
535 const TransformFuncContext& context,
536 GraphDef* output_graph_def) {
537 GraphDef current_graph_def = input_graph_def;
538 const int max_depth = 3;
539 for (int depth = max_depth; depth > 0; --depth) {
540 OpTypePattern pattern = {"*"};
541 for (int i = 0; i < depth; ++i) {
542 pattern = {"*", {pattern}};
543 }
544 pattern = {"FakeQuantWithMinMaxVars", {pattern, {"Const"}, {"Const"}}};
545 GraphDef hoisted_graph_def;
546 TF_RETURN_IF_ERROR(ReplaceMatchingOpTypes(
547 current_graph_def, pattern,
548 [depth](const NodeMatch& match, const std::set<string>& input_nodes,
549 const std::set<string>& output_nodes,
550 std::vector<NodeDef>* new_nodes) {
551 const NodeDef& fake_quant_node = match.node;
552 const NodeDef& fake_quant_min_node = match.inputs[1].node;
553 const NodeDef& fake_quant_max_node = match.inputs[2].node;
554 std::vector<NodeDef> linear_nodes;
555 NodeMatch current_match = match;
556 for (int i = 0; i <= depth; ++i) {
557 linear_nodes.push_back(current_match.inputs[0].node);
558 current_match = current_match.inputs[0];
559 }
560 NodeDef new_fake_quant_node;
561 new_fake_quant_node = fake_quant_node;
562 new_fake_quant_node.set_name(fake_quant_node.name() + "_hoisted");
563 new_fake_quant_node.set_input(
564 0, linear_nodes[linear_nodes.size() - 2].input(0));
565 new_nodes->push_back(new_fake_quant_node);
566
567 new_nodes->push_back(fake_quant_min_node);
568 new_nodes->push_back(fake_quant_max_node);
569
570 linear_nodes[linear_nodes.size() - 2].set_input(
571 0, new_fake_quant_node.name());
572 linear_nodes.front().set_name(fake_quant_node.name());
573 for (const NodeDef& linear_node : linear_nodes) {
574 new_nodes->push_back(linear_node);
575 }
576
577 return Status::OK();
578 },
579 {}, &hoisted_graph_def));
580 current_graph_def = hoisted_graph_def;
581 }
582 *output_graph_def = current_graph_def;
583
584 return Status::OK();
585}

587// Converts any float ops that have eight-bit equivalents into their quantized
588// forms, so that as much calculation as possible is done in the lower-precision
589// format.
590Status QuantizeNodes(const GraphDef& input_graph_def,
591 const TransformFuncContext& context,
592 GraphDef* output_graph_def) {
593 // Loop through all of the quantizable op types, and replace any occurrences
594 // with equivalent sub-graphs with quantized ops at their core. For example
595 // this one-input operation:
596 //
597 // Input(float)
598 // |
599 // v
600 // Operation
601 // |
602 // v
603 // (float)
604 //
605 // Will be turned into it's quantized equivalent:
606 //
607 // Input(float) ReshapeDims
608 // +------v v-------------+
609 // | Reshape
610 // | |
611 // | | ReductionDims
612 // | +-----+ |
613 // | | +---c---------+
614 // | v v v v-------+
615 // | Min Max
616 // | +----+ |
617 // v v v--------+
618 // Quantize
619 // |
620 // v
621 // QuantizedOperation
622 // | | |
623 // v v v
624 // Dequantize
625 // |
626 // v
627 // (float)
628 //
629 // This keeps the inputs and outputs visible to the rest of the graph in
630 // float
631 // and converts them down to quantized buffers internally for the
632 // computation.
633 // The result will end up with a lot of redundant dequantize/quantize pairs
634 // between adjacent quantized ops, but a later pass removes these where it
635 // can.
636
637 std::set<string> ops_to_ignore;
638 if (context.params.count("ignore_op") > 0) {
639 for (const string& name : context.params.at("ignore_op")) {
640 ops_to_ignore.insert(name);
641 }
642 }
643
644 const std::vector<QuantizedOpInfo>& op_list = GetQuantizedOpList();
645 string op_pattern;
646 bool is_first = true;
647 std::map<string, QuantizedOpInfo> op_map;
648 for (const QuantizedOpInfo& op_info : op_list) {
649 if (ops_to_ignore.count(op_info.float_name) == 0) {
650 strings::StrAppend(&op_pattern, (is_first ? "" : "|"),
651 op_info.float_name);
652 op_map.insert({op_info.float_name, op_info});
653 is_first = false;
654 }
655 }
656
657 // If input_min and input max have been passed in, then we convert all float
658 // Placeholder nodes into quantized versions, with the supplied values as
659 // their range.
660 GraphDef placeholder_graph_def;
661 TF_RETURN_IF_ERROR(
662 QuantizePlaceholders(input_graph_def, context, &placeholder_graph_def));
663 TF_RETURN_IF_ERROR(IsGraphValid(placeholder_graph_def));
664
665 // If there are any FakeQuantWithMinMaxVars at the end of a chain of linear
666 // operations like Relu or MaxPool, move them up so that they're as close as
667 // possible to ops with 32-bit outputs like BiasAdd or Conv2D.
668 GraphDef hoisted_graph_def;
669 TF_RETURN_IF_ERROR(
670 HoistFakeQuants(placeholder_graph_def, context, &hoisted_graph_def));
671 TF_RETURN_IF_ERROR(IsGraphValid(hoisted_graph_def));
672
673 // Convert any FakeQuantWithMinMaxVars, which hold the trained ranges of
674 // activation layers, into Requantize ops with those ranges instead. This
675 // makes it easier to replace the dynamic range calculations that are used
676 // by default.
677 GraphDef converted_graph_def;
678 TF_RETURN_IF_ERROR(ConvertFakeQuantsToRequantize(hoisted_graph_def, context,
679 &converted_graph_def));
680 TF_RETURN_IF_ERROR(IsGraphValid(converted_graph_def));
681
682 // If fallback_min and fallback_max are set, then we'll use hardwired ranges
683 // for all the 32-bit to 8-bit requantizations.
684 float fallback_min;
685 float fallback_max;
686 bool has_fallback_range;
687 TF_RETURN_IF_ERROR(ExtractRangeFromParams(
688 context, "fallback_min", "fallback_max", &fallback_min, &fallback_max,
689 &has_fallback_range));
690
691 // Replace all occurrences of the current float op with its quantized
692 // equivalent.
693 GraphDef quantized_graph_def;
694 TF_RETURN_IF_ERROR(ReplaceMatchingOpTypes(
695 converted_graph_def, {op_pattern},
696 [&op_map, fallback_min, fallback_max, has_fallback_range](
697 const NodeMatch& match, const std::set<string>& input_nodes,
698 const std::set<string>& output_nodes,
699 std::vector<NodeDef>* new_nodes) {
700 const NodeDef& float_node = match.node;
701 const QuantizedOpInfo& op_info = op_map[float_node.op()];
702
703 DataTypeVector input_types;
704 DataTypeVector output_types;
705 TF_RETURN_IF_ERROR(
706 GetInOutTypes(float_node, &input_types, &output_types));
707 bool are_all_float = true;
708 for (int i = 0; i < float_node.input_size(); ++i) {
709 // Skip any known non-float inputs.
710 if (op_info.unquantized_inputs.count(i)) {
711 continue;
712 }
713 if (input_types[i] != DT_FLOAT) {
714 are_all_float = false;
715 }
716 }
717 for (const DataType& output_type : output_types) {
718 if (output_type != DT_FLOAT) {
719 are_all_float = false;
720 }
721 }
722 // This isn't a float op, so don't quantize it.
723 if (!are_all_float) {
724 CopyOriginalMatch(match, new_nodes);
725 return Status::OK();
726 }
727
728 string namespace_prefix = float_node.name() + "_eightbit";
729
730 // Quantize all of the inputs.
731 std::vector<string> quantized_input_names;
732 for (int i = 0; i < float_node.input_size(); ++i) {
733 // Skip any non-float inputs.
734 if (op_info.unquantized_inputs.count(i)) {
735 continue;
736 }
737
738 const string& input_name = float_node.input(i);
739 string unique_input_name =
740 namespace_prefix + "/" + UniqueNodeNameFromInput(input_name);
741
742 // Add some common constants we need for reshaping inputs.
743 NodeDef reshape_dims;
744 reshape_dims.set_op("Const");
745 reshape_dims.set_name(unique_input_name + "/reshape_dims");
746 AddNodeInput("^" + NodeNameFromInput(input_name), &reshape_dims);
747 SetNodeAttr("dtype", DT_INT32, &reshape_dims);
748 Tensor reshape_dims_tensor(DT_INT32, {1});
749 reshape_dims_tensor.flat<int32>()(0) = -1;
750 SetNodeTensorAttr<int32>("value", reshape_dims_tensor, &reshape_dims);
751 new_nodes->push_back(reshape_dims);
752
753 NodeDef reduction_dims;
754 reduction_dims.set_op("Const");
755 reduction_dims.set_name(unique_input_name + "/reduction_dims");
756 AddNodeInput("^" + NodeNameFromInput(input_name), &reduction_dims);
757 SetNodeAttr("dtype", DT_INT32, &reduction_dims);
758 Tensor reduction_dims_tensor(DT_INT32, {1});
759 reduction_dims_tensor.flat<int32>()(0) = 0;
760 SetNodeTensorAttr<int32>("value", reduction_dims_tensor,
761 &reduction_dims);
762 new_nodes->push_back(reduction_dims);
763
764 NodeDef reshape_node;
765 reshape_node.set_op("Reshape");
766 reshape_node.set_name(unique_input_name + "/reshape");
767 SetNodeAttr("T", DT_FLOAT, &reshape_node);
768 AddNodeInput(input_name, &reshape_node);
769 AddNodeInput(reshape_dims.name(), &reshape_node);
770 new_nodes->push_back(reshape_node);
771
772 NodeDef min_node;
773 min_node.set_op("Min");
774 min_node.set_name(unique_input_name + "/min");
775 SetNodeAttr("T", DT_FLOAT, &min_node);
776 SetNodeAttr("keep_dims", false, &min_node);
777 AddNodeInput(reshape_node.name(), &min_node);
778 AddNodeInput(reduction_dims.name(), &min_node);
779 new_nodes->push_back(min_node);
780
781 NodeDef max_node;
782 max_node.set_op("Max");
783 max_node.set_name(unique_input_name + "/max");
784 SetNodeAttr("T", DT_FLOAT, &max_node);
785 SetNodeAttr("keep_dims", false, &max_node);
786 AddNodeInput(reshape_node.name(), &max_node);
787 AddNodeInput(reduction_dims.name(), &max_node);
788 new_nodes->push_back(max_node);
789
790 NodeDef quantize_node;
791 quantize_node.set_op("QuantizeV2");
792 quantize_node.set_name(unique_input_name + "/quantize");
793 SetNodeAttr("T", DT_QUINT8, &quantize_node);
794 SetNodeAttr("mode", "MIN_FIRST", &quantize_node);
795 AddNodeInput(input_name, &quantize_node);
796 AddNodeInput(min_node.name(), &quantize_node);
797 AddNodeInput(max_node.name(), &quantize_node);
798 new_nodes->push_back(quantize_node);
799 quantized_input_names.push_back(quantize_node.name());
800 }
801
802 // Set up the quantized version of the current op.
803 NodeDef quantized_main_node;
804 quantized_main_node.set_op("Quantized" + float_node.op());
805 quantized_main_node.set_name(float_node.name() + "/eightbit");
806 for (const string& attr_to_copy : op_info.attrs_to_copy) {
807 CopyNodeAttr(float_node, attr_to_copy, attr_to_copy,
808 &quantized_main_node);
809 }
810 for (const std::pair<string, DataType>& dtype_to_set :
811 op_info.dtypes_to_set) {
812 SetNodeAttr(dtype_to_set.first, dtype_to_set.second,
813 &quantized_main_node);
814 }
815 int quantized_input_index = 0;
816 for (int i = 0; i < float_node.input_size(); ++i) {
817 if (op_info.unquantized_inputs.count(i)) {
818 AddNodeInput(float_node.input(i), &quantized_main_node);
819 } else {
820 const string& quantized_input_name =
821 quantized_input_names[quantized_input_index];
822 AddNodeInput(quantized_input_name + ":0", &quantized_main_node);
823 ++quantized_input_index;
824 }
825 }
826 if (op_info.min_max_order == QuantizedOpInfo::CONTIGUOUS_MIN_MAX) {
827 for (const string& quantized_input_name : quantized_input_names) {
828 AddNodeInput(quantized_input_name + ":1", &quantized_main_node);
829 AddNodeInput(quantized_input_name + ":2", &quantized_main_node);
830 }
831 } else {
832 for (const string& quantized_input_name : quantized_input_names) {
833 AddNodeInput(quantized_input_name + ":1", &quantized_main_node);
834 }
835 for (const string& quantized_input_name : quantized_input_names) {
836 AddNodeInput(quantized_input_name + ":2", &quantized_main_node);
837 }
838 }
839 new_nodes->push_back(quantized_main_node);
840
841 string eight_bit_node_name;
842 if (op_info.output_bit_depth == DT_QINT32) {
843 // Shrink the range of the output down from 32 bits to 8.
844 string requantize_min_input;
845 string requantize_max_input;
846 if (has_fallback_range) {
847 // Use constant values for the min/max range if they were given.
848 NodeDef fallback_min_node;
849 fallback_min_node.set_op("Const");
850 fallback_min_node.set_name(quantized_main_node.name() +
851 "/fallback_min");
852 SetNodeAttr("dtype", DT_FLOAT, &fallback_min_node);
853 Tensor fallback_min_tensor(DT_FLOAT, {});
854 fallback_min_tensor.flat<float>()(0) = fallback_min;
855 SetNodeTensorAttr<float>("value", fallback_min_tensor,
856 &fallback_min_node);
857 new_nodes->push_back(fallback_min_node);
858
859 NodeDef fallback_max_node;
860 fallback_max_node.set_op("Const");
861 fallback_max_node.set_name(quantized_main_node.name() +
862 "/fallback_max");
863 SetNodeAttr("dtype", DT_FLOAT, &fallback_max_node);
864 Tensor fallback_max_tensor(DT_FLOAT, {});
865 fallback_max_tensor.flat<float>()(0) = fallback_max;
866 SetNodeTensorAttr<float>("value", fallback_max_tensor,
867 &fallback_max_node);
868 new_nodes->push_back(fallback_max_node);
869
870 requantize_min_input = fallback_min_node.name();
871 requantize_max_input = fallback_max_node.name();
872 } else {
873 // Otherwise dynamically measure the range each time.
874 NodeDef requant_range_node;
875 requant_range_node.set_op("RequantizationRange");
876 requant_range_node.set_name(quantized_main_node.name() +
877 "/requant_range");
878 SetNodeAttr("Tinput", DT_QINT32, &requant_range_node);
879 AddNodeInput(quantized_main_node.name() + ":0",
880 &requant_range_node);
881 AddNodeInput(quantized_main_node.name() + ":1",
882 &requant_range_node);
883 AddNodeInput(quantized_main_node.name() + ":2",
884 &requant_range_node);
885 new_nodes->push_back(requant_range_node);
886
887 requantize_min_input = requant_range_node.name() + ":0";
888 requantize_max_input = requant_range_node.name() + ":1";
889 }
890 NodeDef requantize_node;
891 requantize_node.set_op("Requantize");
892 requantize_node.set_name(quantized_main_node.name() + "/requantize");
893 SetNodeAttr("Tinput", DT_QINT32, &requantize_node);
894 SetNodeAttr("out_type", DT_QUINT8, &requantize_node);
895 AddNodeInput(quantized_main_node.name() + ":0", &requantize_node);
896 AddNodeInput(quantized_main_node.name() + ":1", &requantize_node);
897 AddNodeInput(quantized_main_node.name() + ":2", &requantize_node);
898 AddNodeInput(requantize_min_input, &requantize_node);
899 AddNodeInput(requantize_max_input, &requantize_node);
900 new_nodes->push_back(requantize_node);
901 eight_bit_node_name = requantize_node.name();
902 } else {
903 eight_bit_node_name = quantized_main_node.name();
904 }
905
906 // Convert the 8-bit result back into float for the final output.
907 NodeDef dequantize_node;
908 dequantize_node.set_op("Dequantize");
909 dequantize_node.set_name(float_node.name());
910 SetNodeAttr("T", DT_QUINT8, &dequantize_node);
911 SetNodeAttr("mode", "MIN_FIRST", &dequantize_node);
912 AddNodeInput(eight_bit_node_name + ":0", &dequantize_node);
913 AddNodeInput(eight_bit_node_name + ":1", &dequantize_node);
914 AddNodeInput(eight_bit_node_name + ":2", &dequantize_node);
915 new_nodes->push_back(dequantize_node);
916
917 return Status::OK();
918 },
919 {}, &quantized_graph_def));
920 TF_RETURN_IF_ERROR(IsGraphValid(quantized_graph_def));
921
922 // If we've ended up with two Requantize ops in a row (for example if there
923 // was a Conv2D feeding into a FakeQuantWithMinMaxVars) merge them together,
924 // using the trained range from the second op.
925 GraphDef merged_graph_def;
926 TF_RETURN_IF_ERROR(MergeAdjacentRequantizes(quantized_graph_def, context,
927 &merged_graph_def));
928 TF_RETURN_IF_ERROR(IsGraphValid(merged_graph_def));
929
930 // There can be duplicate quantize nodes if multiple ops pull from a single
931 // input, which makes it harder to remove redundant ones, so strip them out.
932 GraphDef deduped_graph_def;
933 TF_RETURN_IF_ERROR(
934 MergeDuplicateNodes(merged_graph_def, context, &deduped_graph_def));
935 TF_RETURN_IF_ERROR(IsGraphValid(deduped_graph_def));
936
937 // Look for Dequantizes that immediately go into Quantizes, and remove them
938 // since the two together cancel each other out. This allows us to keep the
939 // data flow in eight bit where two adjacent ops are in eight bit, but still
940 // keep interoperability with float ops.
941 TF_RETURN_IF_ERROR(RemoveRedundantQuantizations(deduped_graph_def, context,
942 output_graph_def));
943 TF_RETURN_IF_ERROR(IsGraphValid(*output_graph_def));
944
945 return Status::OK();
946}
