30Status FuseResizePadAndConv(const GraphDef& input_graph_def,
31 const TransformFuncContext& context,
32 GraphDef* output_graph_def) {
33 GraphDef replaced_graph_def;
34 TF_RETURN_IF_ERROR(ReplaceMatchingOpTypes(
35 input_graph_def, // clang-format off
36 {"Conv2D",
37 {
38 {"MirrorPad",
39 {
40 {"ResizeBilinear"},
41 {"*"}
42 }
43 },
44 {"*"}
45 }
46 }, // clang-format on
47 [](const NodeMatch& match, const std::set<string>& input_nodes,
48 const std::set<string>& output_nodes,
49 std::vector<NodeDef>* new_nodes) {
50 // Find all the nodes we expect in the subgraph.
51 const NodeDef& conv_node = match.node;
52 const NodeDef& mirror_pad_node = match.inputs[0].node;
53 const NodeDef& weights_node = match.inputs[1].node;
54 const NodeDef& resize_node = match.inputs[0].inputs[0].node;
55 const NodeDef& pad_dims_node = match.inputs[0].inputs[1].node;
56
57 // We'll be reusing the old weights and pad dimensions.
58 new_nodes->push_back(weights_node);
59 new_nodes->push_back(pad_dims_node);
60
61 // Set up the new fused version of the convolution op.
62 NodeDef fused_conv;
63 fused_conv.set_op("FusedResizeAndPadConv2D");
64 fused_conv.set_name(match.node.name());
65 AddNodeInput(resize_node.input(0), &fused_conv);
66 AddNodeInput(resize_node.input(1), &fused_conv);
67 AddNodeInput(mirror_pad_node.input(1), &fused_conv);
68 AddNodeInput(conv_node.input(1), &fused_conv);
69 CopyNodeAttr(resize_node, "align_corners", "resize_align_corners",
70 &fused_conv);
71 CopyNodeAttr(mirror_pad_node, "mode", "mode", &fused_conv);
72 CopyNodeAttr(conv_node, "T", "T", &fused_conv);
73 CopyNodeAttr(conv_node, "padding", "padding", &fused_conv);
74 CopyNodeAttr(conv_node, "strides", "strides", &fused_conv);
75 new_nodes->push_back(fused_conv);
76
77 return Status::OK();
78 },
79 {}, &replaced_graph_def));
80 *output_graph_def = replaced_graph_def;
81 return Status::OK();
82}

84Status FuseResizeAndConv(const GraphDef& input_graph_def,
85 const TransformFuncContext& context,
86 GraphDef* output_graph_def) {
87 GraphDef replaced_graph_def;
88 TF_RETURN_IF_ERROR(ReplaceMatchingOpTypes(
89 input_graph_def, // clang-format off
90 {"Conv2D",
91 {
92 {"ResizeBilinear"},
93 {"*"}
94 }
95 }, // clang-format on
96 [](const NodeMatch& match, const std::set<string>& input_nodes,
97 const std::set<string>& output_nodes,
98 std::vector<NodeDef>* new_nodes) {
99 // Find all the nodes we expect in the subgraph.
100 const NodeDef& conv_node = match.node;
101 const NodeDef& resize_node = match.inputs[0].node;
102 const NodeDef& weights_node = match.inputs[1].node;
103
104 // We'll be reusing the old weights.
105 new_nodes->push_back(weights_node);
106
107 // Create a 'no-op' mirror padding node that has no effect.
108 NodeDef pad_dims_node;
109 pad_dims_node.set_op("Const");
110 pad_dims_node.set_name(conv_node.name() + "_dummy_paddings");
111 SetNodeAttr("dtype", DT_INT32, &pad_dims_node);
112 SetNodeTensorAttr<int32>("value", {4, 2}, {0, 0, 0, 0, 0, 0, 0, 0},
113 &pad_dims_node);
114 new_nodes->push_back(pad_dims_node);
115
116 // Set up the new fused version of the convolution op.
117 NodeDef fused_conv;
118 fused_conv.set_op("FusedResizeAndPadConv2D");
119 fused_conv.set_name(match.node.name());
120 AddNodeInput(resize_node.input(0), &fused_conv);
121 AddNodeInput(resize_node.input(1), &fused_conv);
122 AddNodeInput(pad_dims_node.name(), &fused_conv);
123 AddNodeInput(conv_node.input(1), &fused_conv);
124 CopyNodeAttr(resize_node, "align_corners", "resize_align_corners",
125 &fused_conv);
126 SetNodeAttr("mode", "REFLECT", &fused_conv);
127 CopyNodeAttr(conv_node, "T", "T", &fused_conv);
128 CopyNodeAttr(conv_node, "padding", "padding", &fused_conv);
129 CopyNodeAttr(conv_node, "strides", "strides", &fused_conv);
130 new_nodes->push_back(fused_conv);
131
132 return Status::OK();
133 },
134 {}, &replaced_graph_def));
135 *output_graph_def = replaced_graph_def;
136 return Status::OK();
137}

139Status FusePadAndConv(const GraphDef& input_graph_def,
140 const TransformFuncContext& context,
141 GraphDef* output_graph_def) {
142 GraphDef replaced_graph_def;
143 TF_RETURN_IF_ERROR(ReplaceMatchingOpTypes(
144 input_graph_def, // clang-format off
145 {"Conv2D",
146 {
147 {"MirrorPad",
148 {
149 {"*"},
150 {"*"},
151 }
152 },
153 {"*"}
154 }
155 }, // clang-format on
156 [](const NodeMatch& match, const std::set<string>& input_nodes,
157 const std::set<string>& output_nodes,
158 std::vector<NodeDef>* new_nodes) {
159 // Find all the nodes we expect in the subgraph.
160 const NodeDef& conv_node = match.node;
161 CHECK_EQ("Conv2D", conv_node.op());
162 const NodeDef& mirror_pad_node = match.inputs[0].node;
163 CHECK_EQ("MirrorPad", mirror_pad_node.op());
164 const NodeDef& weights_node = match.inputs[1].node;
165 const NodeDef& input_node = match.inputs[0].inputs[0].node;
166 const NodeDef& pad_dims_node = match.inputs[0].inputs[1].node;
167
168 // We'll be reusing the old weights and pad dimensions.
169 new_nodes->push_back(weights_node);
170 new_nodes->push_back(input_node);
171 new_nodes->push_back(pad_dims_node);
172
173 // Set up the new fused version of the convolution op.
174 NodeDef fused_conv;
175 fused_conv.set_op("FusedPadConv2D");
176 fused_conv.set_name(match.node.name());
177 AddNodeInput(mirror_pad_node.input(0), &fused_conv);
178 AddNodeInput(mirror_pad_node.input(1), &fused_conv);
179 AddNodeInput(conv_node.input(1), &fused_conv);
180 CopyNodeAttr(mirror_pad_node, "mode", "mode", &fused_conv);
181 CopyNodeAttr(conv_node, "T", "T", &fused_conv);
182 CopyNodeAttr(conv_node, "padding", "padding", &fused_conv);
183 CopyNodeAttr(conv_node, "strides", "strides", &fused_conv);
184 new_nodes->push_back(fused_conv);
185
186 return Status::OK();
187 },
188 {}, &replaced_graph_def));
189 *output_graph_def = replaced_graph_def;
190 return Status::OK();
191}
