31PyRecordReader::PyRecordReader() {}
32
33PyRecordReader* PyRecordReader::New(const string& filename, uint64 start_offset,
34 const string& compression_type_string,
35 TF_Status* out_status) {
36 std::unique_ptr<RandomAccessFile> file;
37 Status s = Env::Default()->NewRandomAccessFile(filename, &file);
38 if (!s.ok()) {
39 Set_TF_Status_from_Status(out_status, s);
40 return nullptr;
41 }
42 PyRecordReader* reader = new PyRecordReader;
43 reader->offset_ = start_offset;
44 reader->file_ = file.release();
45
46 RecordReaderOptions options =
47 RecordReaderOptions::CreateRecordReaderOptions(compression_type_string);
48
49 reader->reader_ = new RecordReader(reader->file_, options);
50 return reader;
51}
52
53PyRecordReader::~PyRecordReader() {
54 delete reader_;
55 delete file_;
56}
57
58void PyRecordReader::GetNext(TF_Status* status) {
59 if (reader_ == nullptr) {
60 Set_TF_Status_from_Status(status,
61 errors::FailedPrecondition("Reader is closed."));
62 return;
63 }
64 Status s = reader_->ReadRecord(&offset_, &record_);
65 Set_TF_Status_from_Status(status, s);
66}
67
68void PyRecordReader::Close() {
69 delete reader_;
70 delete file_;
71 file_ = nullptr;
72 reader_ = nullptr;
73}
