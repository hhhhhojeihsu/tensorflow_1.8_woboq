25// py.__class__.__name__
26const char* ClassName(PyObject* py) {
27/* PyPy doesn't have a separate C API for old-style classes. */
28#if PY_MAJOR_VERSION < 3 && !defined(PYPY_VERSION)
29 if (PyClass_Check(py))
30 return PyString_AS_STRING(
31 CHECK_NOTNULL(reinterpret_cast<PyClassObject*>(py)->cl_name));
32 if (PyInstance_Check(py))
33 return PyString_AS_STRING(CHECK_NOTNULL(
34 reinterpret_cast<PyInstanceObject*>(py)->in_class->cl_name));
35#endif
36 if (Py_TYPE(py) == &PyType_Type) {
37 return reinterpret_cast<PyTypeObject*>(py)->tp_name;
38 }
39 return Py_TYPE(py)->tp_name;
40}

44// Returns a PyObject containing a string, or null
45void TryAppendTraceback(PyObject* ptype, PyObject* pvalue, PyObject* ptraceback,
46 string* out) {
47 // The "traceback" module is assumed to be imported already by script_ops.py.
48 PyObject* tb_module = PyImport_AddModule("traceback");
49
50 if (!tb_module) {
51 return;
52 }
53
54 PyObject* format_exception =
55 PyObject_GetAttrString(tb_module, "format_exception");
56
57 if (!format_exception) {
58 return;
59 }
60
61 if (!PyCallable_Check(format_exception)) {
62 Py_DECREF(format_exception);
63 return;
64 }
65
66 PyObject* ret_val = PyObject_CallFunctionObjArgs(format_exception, ptype,
67 pvalue, ptraceback, nullptr);
68 Py_DECREF(format_exception);
69
70 if (!ret_val) {
71 return;
72 }
73
74 if (!PyList_Check(ret_val)) {
75 Py_DECREF(ret_val);
76 return;
77 }
78
79 Py_ssize_t n = PyList_GET_SIZE(ret_val);
80 for (Py_ssize_t i = 0; i < n; ++i) {
81 PyObject* v = PyList_GET_ITEM(ret_val, i);
82#if PY_MAJOR_VERSION < 3
83 strings::StrAppend(out, PyString_AS_STRING(v), "\n");
84#else
85 strings::StrAppend(out, PyUnicode_AsUTF8(v), "\n");
86#endif
87 }
88
89 // Iterate through ret_val.
90 Py_DECREF(ret_val);
91}

93string PyExceptionFetch() {
94 CHECK(PyErr_Occurred())
95 << "Must only call PyExceptionFetch after an exception.";
96 PyObject* ptype;
97 PyObject* pvalue;
98 PyObject* ptraceback;
99 PyErr_Fetch(&ptype, &pvalue, &ptraceback);
100 PyErr_NormalizeException(&ptype, &pvalue, &ptraceback);
101 string err = ClassName(ptype);
102 if (pvalue) {
103 PyObject* str = PyObject_Str(pvalue);
104
105 if (str) {
106#if PY_MAJOR_VERSION < 3
107 strings::StrAppend(&err, ": ", PyString_AS_STRING(str), "\n");
108#else
109 strings::StrAppend(&err, ": ", PyUnicode_AsUTF8(str), "\n");
110#endif
111 Py_DECREF(str);
112 } else {
113 strings::StrAppend(&err, "(unknown error message)\n");
114 }
115
116 TryAppendTraceback(ptype, pvalue, ptraceback, &err);
117
118 Py_DECREF(pvalue);
119 }
120 Py_DECREF(ptype);
121 Py_XDECREF(ptraceback);
122 return err;
123}
