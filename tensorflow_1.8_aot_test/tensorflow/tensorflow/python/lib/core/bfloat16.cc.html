78// Representation of a Python bfloat16 object.
79struct PyBfloat16 {
80 PyObject_HEAD; // Python object header
81 bfloat16 value;
82};

84// Returns true if 'object' is a PyBfloat16.
85bool PyBfloat16_Check(PyObject* object) {
86 return PyObject_IsInstance(object,
87 reinterpret_cast<PyObject*>(&PyBfloat16_Type));
88}

90// Extracts the value of a PyBfloat16 object.
91bfloat16 PyBfloat16_Bfloat16(PyObject* object) {
92 return reinterpret_cast<PyBfloat16*>(object)->value;
93}

95// Constructs a PyBfloat16 object from a bfloat16.
96Safe_PyObjectPtr PyBfloat16_FromBfloat16(bfloat16 x) {
97 Safe_PyObjectPtr ref =
98 make_safe(PyBfloat16_Type.tp_alloc(&PyBfloat16_Type, 0));
99 PyBfloat16* p = reinterpret_cast<PyBfloat16*>(ref.get());
100 if (p) {
101 p->value = x;
102 }
103 return ref;
104}

106// Converts a Python object to a bfloat16 value. Returns true on success,
107// returns false and reports a Python error on failure.
108bool AsBfloat16(PyObject* arg, bfloat16* output) {
109 if (PyBfloat16_Check(arg)) {
110 *output = PyBfloat16_Bfloat16(arg);
111 return true;
112 }
113 if (PyFloat_Check(arg)) {
114 double d = PyFloat_AsDouble(arg);
115 if (PyErr_Occurred()) {
116 return false;
117 }
118 // TODO(phawkins): check for overflow
119 *output = bfloat16(d);
120 return true;
121 }
122 if (TfPyInt_Check(arg)) {
123 long l = TfPyInt_AsLong(arg); // NOLINT
124 if (PyErr_Occurred()) {
125 return false;
126 }
127 // TODO(phawkins): check for overflow
128 *output = bfloat16(static_cast<float>(l));
129 return true;
130 }
131 if (PyArray_IsScalar(arg, Float)) {
132 float f;
133 PyArray_ScalarAsCtype(arg, &f);
134 *output = bfloat16(f);
135 return true;
136 }
137 PyErr_Format(PyExc_TypeError, "expected number, got %s",
138 arg->ob_type->tp_name);
139 return false;
140}

162#define BFLOAT16_BINOP(name, op) \
163 PyObject* PyBfloat16_##name(PyObject* a, PyObject* b) { \

232// Constructs a new PyBfloat16.
233PyObject* PyBfloat16_New(PyTypeObject* type, PyObject* args, PyObject* kwds) {
234 if (kwds && PyDict_Size(kwds)) {
235 PyErr_SetString(PyExc_TypeError, "constructor takes no keyword arguments");
236 return nullptr;
237 }
238 Py_ssize_t size = PyTuple_Size(args);
239 if (size != 1) {
240 PyErr_SetString(PyExc_TypeError,
241 "expected number as argument to bfloat16 constructor");
242 return nullptr;
243 }
244 PyObject* arg = PyTuple_GetItem(args, 0);
245
246 if (PyBfloat16_Check(arg)) {
247 Py_INCREF(arg);
248 return arg;
249 } else {
250 bfloat16 value;
251 if (!AsBfloat16(arg, &value)) {
252 return nullptr;
253 }
254 return PyBfloat16_FromBfloat16(value).release();
255 }
256}

258// Comparisons on PyBfloat16s.
259PyObject* PyBfloat16_RichCompare(PyObject* a, PyObject* b, int op) {
260 bfloat16 x, y;
261 if (!AsBfloat16(a, &x) || !AsBfloat16(b, &y)) return nullptr;
262 bool result;
263 switch (op) {
264 case Py_LT:
265 result = x < y;
266 break;
267 case Py_LE:
268 result = x <= y;
269 break;
270 case Py_EQ:
271 result = x == y;
272 break;
273 case Py_NE:
274 result = x != y;
275 break;
276 case Py_GT:
277 result = x > y;
278 break;
279 case Py_GE:
280 result = x >= y;
281 break;
282 default:
283 LOG(FATAL) << "Invalid op type " << op;
284 }
285 return PyBool_FromLong(result);
286}

406void ByteSwap16(void* value) {
407 char* p = reinterpret_cast<char*>(value);
408 std::swap(p[0], p[1]);
409}
410
411void NPyBfloat16_CopySwapN(void* dstv, npy_intp dstride, void* srcv,
412 npy_intp sstride, npy_intp n, int swap, void* arr) {
413 char* dst = reinterpret_cast<char*>(dstv);
414 char* src = reinterpret_cast<char*>(srcv);
415 if (!src) {
416 return;
417 }
418 if (swap) {
419 for (npy_intp i = 0; i < n; i++) {
420 char* r = dst + dstride * i;
421 memcpy(r, src + sstride * i, sizeof(uint16_t));
422 ByteSwap16(r);
423 }
424 } else if (dstride == sizeof(uint16_t) && sstride == sizeof(uint16_t)) {
425 memcpy(dst, src, n * sizeof(uint16_t));
426 } else {
427 for (npy_intp i = 0; i < n; i++) {
428 memcpy(dst + dstride * i, src + sstride * i, sizeof(uint16_t));
429 }
430 }
431}
432
433void NPyBfloat16_CopySwap(void* dst, void* src, int swap, void* arr) {
434 if (!src) {
435 return;
436 }
437 memcpy(dst, src, sizeof(uint16_t));
438 if (swap) {
439 ByteSwap16(dst);
440 }
441}

451// Performs a NumPy array cast from type 'From' to 'To'.
452template <typename From, typename To>
453void NPyCast(void* from_void, void* to_void, npy_intp n, void* fromarr,
454 void* toarr) {
455 const From* from = reinterpret_cast<From*>(from_void);
456 To* to = reinterpret_cast<To*>(to_void);
457 for (npy_intp i = 0; i < n; ++i) {
458 to[i] = static_cast<To>(from[i]);
459 }
460}

462// Registers a cast between bfloat16 and type 'T'. 'numpy_type' is the NumPy
463// type corresponding to 'T'. If 'cast_is_safe', registers that bfloat16 can be
464// safely coerced to T.
465template <typename T>
466bool RegisterBfloat16Cast(int numpy_type, bool cast_is_safe) {
467 if (PyArray_RegisterCastFunc(PyArray_DescrFromType(numpy_type), npy_bfloat16_,
468 NPyCast<T, bfloat16>) < 0) {
469 return false;
470 }
471 if (PyArray_RegisterCastFunc(&NPyBfloat16_Descr, numpy_type,
472 NPyCast<bfloat16, T>) < 0) {
473 return false;
474 }
475 if (cast_is_safe && PyArray_RegisterCanCast(&NPyBfloat16_Descr, numpy_type,
476 NPY_NOSCALAR) < 0) {
477 return false;
478 }
479 return true;
480}

482template <typename InType, typename OutType, typename Functor>
483void BinaryUFunc(char** args, npy_intp* dimensions, npy_intp* steps,
484 void* data) {
485 const char* i0 = args[0];
486 const char* i1 = args[1];
487 char* o = args[2];
488 for (npy_intp k = 0; k < *dimensions; k++) {
489 InType x = *reinterpret_cast<const InType*>(i0);
490 InType y = *reinterpret_cast<const InType*>(i1);
491 *reinterpret_cast<OutType*>(o) = Functor()(x, y);
492 i0 += steps[0];
493 i1 += steps[1];
494 o += steps[2];
495 }
496}
497
498template <typename Functor>
499void CompareUFunc(char** args, npy_intp* dimensions, npy_intp* steps,
500 void* data) {
501 BinaryUFunc<bfloat16, npy_bool, Functor>(args, dimensions, steps, data);
502}

504struct Bfloat16EqFunctor {
505 npy_bool operator()(bfloat16 a, bfloat16 b) { return a == b; }

510struct Bfloat16LtFunctor {
511 npy_bool operator()(bfloat16 a, bfloat16 b) { return a < b; }

513struct Bfloat16GtFunctor {

523// Initializes the module.
524bool Initialize() {
525 // It's critical to import umath to avoid crash in open source build.
526 import_umath1(false);
527
528 Safe_PyObjectPtr numpy_str = make_safe(MakePyString("numpy"));
529 if (!numpy_str) {
530 return false;
531 }
532 Safe_PyObjectPtr numpy = make_safe(PyImport_Import(numpy_str.get()));
533 if (!numpy) {
534 return false;
535 }
536
537 // We hit a mysterious crash if we haven't initialized numpy before this:
538 PyBfloat16_Type.tp_base = &PyGenericArrType_Type;
539
540 if (PyType_Ready(&PyBfloat16_Type) < 0) {
541 return false;
542 }
543
544 // Initializes the NumPy descriptor.
545 PyArray_InitArrFuncs(&NPyBfloat16_ArrFuncs);
546 NPyBfloat16_ArrFuncs.getitem = NPyBfloat16_GetItem;
547 NPyBfloat16_ArrFuncs.setitem = NPyBfloat16_SetItem;
548 NPyBfloat16_ArrFuncs.copyswapn = NPyBfloat16_CopySwapN;
549 NPyBfloat16_ArrFuncs.copyswap = NPyBfloat16_CopySwap;
550 NPyBfloat16_ArrFuncs.nonzero = NPyBfloat16_NonZero;
551
552 Py_TYPE(&NPyBfloat16_Descr) = &PyArrayDescr_Type;
553 npy_bfloat16_ = PyArray_RegisterDataType(&NPyBfloat16_Descr);
554 if (npy_bfloat16_ < 0) return false;
555
556 // Support dtype(bfloat16)
557 if (PyDict_SetItemString(PyBfloat16_Type.tp_dict, "dtype",
558 reinterpret_cast<PyObject*>(&NPyBfloat16_Descr)) <
559 0) {
560 return false;
561 }
562
563 // Register casts
564
565 // We lie shamelessly and say that a cast from half to bfloat16 is safe.
566 // Numpy frequently uses the smallest legal representation type for small
567 // float constants (e.g., 1.0), which is often float16. Things break if these
568 // cannot be converted transparently to bfloat16.
569 if (!RegisterBfloat16Cast<Eigen::half>(NPY_HALF, /*cast_is_safe=*/true)) {
570 return false;
571 }
572
573 if (!RegisterBfloat16Cast<float>(NPY_FLOAT, /*cast_is_safe=*/true)) {
574 return false;
575 }
576 if (!RegisterBfloat16Cast<double>(NPY_DOUBLE, /*cast_is_safe=*/true)) {
577 return false;
578 }
579 if (!RegisterBfloat16Cast<int32>(NPY_INT32, /*cast_is_safe=*/false)) {
580 return false;
581 }
582 if (!RegisterBfloat16Cast<int64>(NPY_INT64, /*cast_is_safe=*/false)) {
583 return false;
584 }
585 // Following the numpy convention. imag part is dropped when converting to
586 // float.
587 if (!RegisterBfloat16Cast<complex64>(NPY_COMPLEX64, /*cast_is_safe=*/true)) {
588 return false;
589 }
590 if (!RegisterBfloat16Cast<complex128>(NPY_COMPLEX128,
591 /*cast_is_safe=*/true)) {
592 return false;
593 }

595 // Register ufuncs
596 auto register_ufunc = [&](const char* name, PyUFuncGenericFunction fn,
597 const std::array<int, 3>& types) {
598 Safe_PyObjectPtr ufunc_obj =
599 make_safe(PyObject_GetAttrString(numpy.get(), name));
600 if (!ufunc_obj) {
601 return false;
602 }
603 PyUFuncObject* ufunc = reinterpret_cast<PyUFuncObject*>(ufunc_obj.get());
604 if (types.size() != ufunc->nargs) {
605 PyErr_Format(PyExc_AssertionError,
606 "ufunc %s takes %d arguments, loop takes %lu", name,
607 ufunc->nargs, types.size());
608 return false;
609 }
610 if (PyUFunc_RegisterLoopForType(ufunc, npy_bfloat16_, fn,
611 const_cast<int*>(types.data()),
612 nullptr) < 0) {
613 return false;
614 }
615 return true;
616 };
617
618 // Comparisons
619 const std::array<int, 3> compare_types = {npy_bfloat16_, npy_bfloat16_,
620 NPY_BOOL};
621
622 if (!register_ufunc("equal", CompareUFunc<Bfloat16EqFunctor>,
623 compare_types)) {
624 return false;
625 }
626 if (!register_ufunc("not_equal", CompareUFunc<Bfloat16NeFunctor>,
627 compare_types)) {
628 return false;
629 }
630 if (!register_ufunc("less", CompareUFunc<Bfloat16LtFunctor>, compare_types)) {
631 return false;
632 }
633 if (!register_ufunc("greater", CompareUFunc<Bfloat16GtFunctor>,
634 compare_types)) {
635 return false;
636 }
637 if (!register_ufunc("less_equal", CompareUFunc<Bfloat16LeFunctor>,
638 compare_types)) {
639 return false;
640 }
641 if (!register_ufunc("greater_equal", CompareUFunc<Bfloat16GeFunctor>,
642 compare_types)) {
643 return false;
644 }
645 return true;
646}
