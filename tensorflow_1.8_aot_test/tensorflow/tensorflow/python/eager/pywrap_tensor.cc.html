32TFE_Context* GetContext(PyObject* ctx) {
33 TFE_Context* context =
34 reinterpret_cast<TFE_Context*>(PyCapsule_GetPointer(ctx, nullptr));
35 if (context == nullptr) {
36 PyErr_SetString(PyExc_TypeError,
37 tensorflow::strings::StrCat(
38 "Expecting a PyCapsule encoded context handle. Got ",
39 Py_TYPE(ctx)->tp_name)
40 .c_str());
41 }
42 return context;
43}

45// Convert a Python numpy.ndarray object to a TFE_TensorHandle.
46// The two may share underlying storage so changes to one may reflect in the
47// other.
48TFE_TensorHandle* NumpyToTensorHandle(PyObject* obj) {
49 tensorflow::Tensor t;
50 auto cppstatus = tensorflow::NdarrayToTensor(obj, &t);
51 if (cppstatus.ok()) {
52 return TFE_NewTensorHandle(t);
53 } else {
54 PyErr_SetString(PyExc_ValueError,
55 tensorflow::strings::StrCat(
56 "Failed to convert numpy ndarray to a Tensor (",
57 cppstatus.error_message(), ").")
58 .c_str());
59 return nullptr;
60 }
61}

63// Casts data referred to by `handle` from type `src_type_enum` to type
64// `dst_type_enum`.
65TFE_TensorHandle* EagerCast(TFE_Context* ctx, TFE_TensorHandle* handle,
66 TF_DataType src_type_enum,
67 TF_DataType dst_type_enum, TF_Status* out_status) {
68 if (ctx == nullptr) return nullptr;
69 const char* op_name = "Cast";
70 const char* device_name = "/job:localhost/replica:0/task:0/device:CPU:0";
71 TFE_Op* op = TFE_NewOp(ctx, op_name, out_status);
72#define RETURN_ERROR \
73 { \
74 TFE_DeleteOp(op); \
75 return nullptr; \
76 }
77 if (TF_GetCode(out_status) != TF_OK) RETURN_ERROR
78 TFE_OpSetDevice(op, device_name, out_status);
79 if (TF_GetCode(out_status) != TF_OK) RETURN_ERROR
80 TFE_OpAddInput(op, handle, out_status);
81 if (TF_GetCode(out_status) != TF_OK) RETURN_ERROR
82 TFE_OpSetAttrType(op, "SrcT", src_type_enum);
83 TFE_OpSetAttrType(op, "DstT", dst_type_enum);
84 TFE_TensorHandle* output = nullptr;
85 int num_outputs = 1;
86 TFE_Execute(op, &output, &num_outputs, out_status);
87 if (TF_GetCode(out_status) != TF_OK || num_outputs != 1 ||
88 output == nullptr) {
89 if (output != nullptr) {
90 TFE_DeleteTensorHandle(output);
91 }
92 RETURN_ERROR
93 }
94 TFE_DeleteOp(op);
95 return output;
96#undef RETURN_ERROR
97}

99TFE_TensorHandle* CopyToDevice(TFE_TensorHandle* handle, PyObject* ctx,
100 PyObject* dev) {
101 const char* device = "";
102 if (dev != nullptr && dev != Py_None) {
103 device = PyBytes_AsString(dev);
104#if PY_MAJOR_VERSION >= 3
105 if (device == nullptr) {
106 PyErr_Clear();
107 device = PyUnicode_AsUTF8(dev);
108 }
109#endif
110 if (device == nullptr) {
111 PyErr_SetString(PyExc_TypeError,
112 "Error parsing device argument to CopyToDevice");
113 return nullptr;
114 }
115 }
116 TFE_Context* context = GetContext(ctx);
117 if (context == nullptr) { // PyErr already set by GetContext
118 return nullptr;
119 }
120 auto status = tensorflow::make_safe(TF_NewStatus());
121 TFE_TensorHandle* new_handle =
122 TFE_TensorHandleCopyToDevice(handle, context, device, status.get());
123 if (TF_GetCode(status.get()) != TF_OK) {
124 PyErr_SetString(
125 PyExc_RuntimeError,
126 tensorflow::strings::StrCat("Error copying tensor to device: ", device,
127 ". ", TF_Message(status.get()))
128 .c_str());
129 return nullptr;
130 }
131 return new_handle;
132}

134// Helper function to convert `v` to an int and store it in `*out`. Returns true
135// on success, false otherwise.
136// Note that we assume that v is a python int (not long) representing a
137// TF_DataType value.
138bool PyIntToDataType(PyObject* v, int* out) {
139#if PY_MAJOR_VERSION < 3
140 if (PyInt_Check(v)) {
141 *out = PyInt_AS_LONG(v);
142 return true;
143 }
144#else
145 if (PyLong_Check(v)) {
146 *out = PyLong_AsLong(v);
147 return true;
148 }
149#endif
150 return false;
151}

156 return PyInt_FromLong(l);
157#else
158 return PyLong_FromLong(l);
159#endif

166static const int kMaxEagerTensorParentSize = 64;

168// TODO(agarwal): store context handle in EagerTensor.
169typedef struct EagerTensor {
170 PyObject_HEAD;
171 // Note that we leave kMaxEagerTensorParentSize bytes here for use by the
172 // parent class. The parent class is set at runtime, so we don't know the
173 // exact size at compile time.
174 char unused[kMaxEagerTensorParentSize];
175 TFE_TensorHandle* handle;
176 int64_t id;
177 // This mirrors tensorflow.core.framework.ops.Tensor._handle_data Which will
178 // be None for tensors of type other than DT_REOSURCE. For DT_RESOURCE
179 // tensors, this will contain a serialized HandleData proto with shape
180 // inference metadata about shapes and dtypes of resources accessible from
181 // this handle.
182 // Note that we assume that handle_data cannot participate in reference
183 // cycles, and hence don't provide GC support for it.
184 PyObject* handle_data;
185
186 // This stores `_keras_mask` object and is set by Tensorflow layers.
187 PyObject* keras_mask;
188
189 // This stores `_tensor_shape`, a cached `TensorShape` object, and is set the
190 // first time that `_EagerTensorBase`'s `shape` property is called.
191 PyObject* tensor_shape;
192
193 // We store a status object here as an optimization to avoid allocating a new
194 // Status objects on different functions that operate on EagerTensor and need
195 // to use a TF_Status object. However note that accesses to `status` are not
196 // thread-safe.
197 TF_Status* status;
198} EagerTensor;

200// tp_init for EagerTensor.
201int EagerTensor_init(EagerTensor* self, PyObject* args, PyObject* kwds) {
202 self->id = get_uid();
203 self->handle = nullptr;
204 Py_INCREF(Py_None);
205 self->handle_data = Py_None;
206 Py_INCREF(Py_None);
207 self->keras_mask = Py_None;
208 Py_INCREF(Py_None);
209 self->tensor_shape = Py_None;
210 self->status = TF_NewStatus();
211 PyObject* value;
212 PyObject* context = nullptr;
213 PyObject* device = nullptr;
214 PyObject* dtype = Py_None;
215 const char* kwlist[] = {"value", "context", "device", "dtype", nullptr};
216 if (!PyArg_ParseTupleAndKeywords(args, kwds, "OOO|O",
217 const_cast<char**>(kwlist), &value, &context,
218 &device, &dtype)) {
219 return -1;
220 }
221 // Extract dtype
222 int desired_dtype = -1;
223 if (dtype != Py_None) {
224 if (!PyIntToDataType(dtype, &desired_dtype)) {
225 PyErr_SetString(PyExc_TypeError,
226 tensorflow::strings::StrCat(
227 "Expecting a DataType value for dtype. Got ",
228 Py_TYPE(dtype)->tp_name)
229 .c_str());
230 return -1;
231 }
232 }
233 tensorflow::Safe_TFE_TensorHandlePtr handle =
234 tensorflow::make_safe(static_cast<TFE_TensorHandle*>(nullptr));
235 PyErr_Clear();
236 if (PyArray_Check(value)) {
237 int desired_np_dtype = -1;
238 if (desired_dtype >= 0) {
239 if (!tensorflow::TF_DataType_to_PyArray_TYPE(
240 static_cast<TF_DataType>(desired_dtype), &desired_np_dtype)
241 .ok()) {
242 PyErr_SetString(PyExc_TypeError,
243 tensorflow::strings::StrCat(
244 "Invalid dtype argument value ", desired_dtype)
245 .c_str());
246 return -1;
247 }
248 }
249 PyArrayObject* array = reinterpret_cast<PyArrayObject*>(value);
250 int current_np_dtype = PyArray_TYPE(array);
251 auto safe_value = tensorflow::make_safe(static_cast<PyObject*>(nullptr));
252 if ((desired_np_dtype >= 0 && desired_np_dtype != current_np_dtype) ||
253 !PyArray_ISCARRAY(array)) {
254 int new_dtype =
255 desired_np_dtype >= 0 ? desired_np_dtype : current_np_dtype;
256 safe_value = tensorflow::make_safe(
257 PyArray_FromAny(value, PyArray_DescrFromType(new_dtype), 0, 0,
258 NPY_ARRAY_CARRAY | NPY_ARRAY_FORCECAST, nullptr));
259 if (PyErr_Occurred()) return -1;
260 if (safe_value == nullptr) {
261 PyErr_SetString(PyExc_ValueError, "Error while casting a numpy value");
262 return -1;
263 }
264 value = safe_value.get();
265 }
266 handle = tensorflow::make_safe(NumpyToTensorHandle(value));
267 } else {
268 tensorflow::Tensor t;
269 // TODO(josh11b): Have PySeqToTensor set python errors instead of
270 // returning Status.
271 auto cppstatus = tensorflow::PySeqToTensor(value, dtype, &t);
272 if (!cppstatus.ok()) {
273 PyErr_SetString(PyExc_ValueError, cppstatus.error_message().c_str());
274 return -1;
275 }
276 handle = tensorflow::make_safe(TFE_NewTensorHandle(t));
277 }
278 if (PyErr_Occurred()) return -1;
279 if (handle == nullptr) {
280 PyErr_SetString(PyExc_ValueError, "Error while creating an EagerTensor");
281 return -1;
282 }
283 TF_DataType handle_dtype = TFE_TensorHandleDataType(handle.get());
284 if (desired_dtype >= 0 && desired_dtype != handle_dtype) {
285 handle = tensorflow::make_safe(
286 EagerCast(GetContext(context), handle.get(), handle_dtype,
287 static_cast<TF_DataType>(desired_dtype), self->status));
288 if (TF_GetCode(self->status) != TF_OK) {
289 PyErr_SetString(PyExc_ValueError,
290 tensorflow::strings::StrCat(
291 "Error while casting from DataType ", handle_dtype,
292 " to ", desired_dtype, ". ", TF_Message(self->status))
293 .c_str());
294 // Cleanup self->status before returning.
295 TF_SetStatus(self->status, TF_OK, "");
296 return -1;
297 }
298 handle_dtype = TFE_TensorHandleDataType(handle.get());
299 }

301 // Almost all TensorFlow kernels for GPU devices keep int32 tensors in host
302 // memory. We approximate the same behavior for eager execution - keeping
303 // int32 tensors in host memory.
304 //
305 // We do so to preclude the need for callers into such kernels from having to
306 // explicitly place the int32 tensors in host memory. For example, without
307 // this, one needed:
308 //
309 // with tf.device('/gpu:0'):
310 // ...// code here
311 // with tf.device('/cpu:0'):
312 // shape = tf.constant(...)
313 // y = tf.random_uniform(shape)
314 //
315 // Without the CPU device block, tfe.ops.random_uniform would fail since the
316 // kernel expects the shape in host memory.
317 //
318 // With this support, we simplify the code:
319 //
320 // with tf.device('/gpu:0'):
321 // y = tf.random_uniform(...)
322 //
323 // The approximation is not exact there are GPU kernels which do not require
324 // host memory for int32 tensors. This will lead to a discrepancy between
325 // eager and graph execution.
326 // TODO(ashankar): Fix this.
327 if (handle_dtype != TF_INT32) {
328 // Note that this is a shallow copy and will share the underlying buffer
329 // if copying to the same device.
330 handle = tensorflow::make_safe(CopyToDevice(handle.get(), context, device));
331 if (handle == nullptr) return -1;
332 }
333 self->handle = handle.release();
334 return 0;
335}

337// tp_dealloc for EagerTensor.
338void EagerTensor_dealloc(EagerTensor* self) {
339 TF_DeleteStatus(self->status);
340 Py_DECREF(self->handle_data);
341 Py_DECREF(self->keras_mask);
342 Py_DECREF(self->tensor_shape);
343 if (self->handle != nullptr) {
344 TFE_DeleteTensorHandle(self->handle);
345 self->handle = nullptr;
346 }
347 // We have the global interpreter lock, so use this chance to perform delayed
348 // refcount decrements.
349 tensorflow::ClearDecrefCache();
350 auto id = self->id;
351 Py_TYPE(self)->tp_free(self);
352 TFE_Py_TapeSetDeleteTrace(id);
353}

365// Getter for `_shape_tuple`.
366static PyObject* EagerTensor_shape_tuple(EagerTensor* self) {
367 auto handle = self->handle;
368 int n = TFE_TensorHandleNumDims(handle, self->status);
369 if (MaybeRaiseExceptionFromTFStatus(self->status, PyExc_ValueError)) {
370 // Cleanup self->status before returning.
371 TF_SetStatus(self->status, TF_OK, "");
372 return nullptr;
373 }
374 PyObject* shape = PyTuple_New(n);
375 if (PyErr_Occurred()) return nullptr;
376 for (int i = 0; i < n; ++i) {
377 PyObject* dim =
378 PyLong_FromLongLong(TFE_TensorHandleDim(handle, i, self->status));
379 if (MaybeRaiseExceptionFromTFStatus(self->status, PyExc_ValueError) ||
380 dim == nullptr || PyTuple_SetItem(shape, i, dim) != 0) {
381 // Cleanup self->status before returning.
382 TF_SetStatus(self->status, TF_OK, "");
383 Py_DECREF(shape);
384 if (dim != nullptr) Py_DECREF(dim);
385 PyErr_SetString(PyExc_RuntimeError, "Error while creating shape");
386 return nullptr;
387 }
388 }
389 return shape;
390}

460// Function `_numpy`.
461// Convert an EagerTensor to a Python numpy.ndarray object.
462// The two may share underlying storage so changes to one may reflect in the
463// other.
464// Note that if `self` is not on CPU, we raise an Exception.
465static PyObject* EagerTensor_numpy(EagerTensor* self) {
466 auto status = tensorflow::make_safe(TF_NewStatus());
467 const tensorflow::Tensor* t =
468 TFE_TensorHandleUnderlyingTensorInHostMemory(self->handle, status.get());
469 if (TF_GetCode(status.get()) != TF_OK) {
470 PyErr_SetString(PyExc_RuntimeError, TF_Message(status.get()));
471 return nullptr;
472 }
473 PyObject* ret = nullptr;
474 auto cppstatus = tensorflow::TensorToNdarray(*t, &ret);
475 if (MaybeRaiseExceptionFromStatus(cppstatus, PyExc_RuntimeError)) {
476 Py_XDECREF(ret);
477 return nullptr;
478 } else {
479 return ret;
480 }
481}

528// Note that here we are trying to dynamically create a new class as a subclass
529// of a "HEAPTYPE" class that is itself created in python code and passed in at
530// runtime. This is fairly atypical and undocumented.
531//
532// We use the following strategy for this. Unfortunately, we have to use
533// different approaches for python2.x vs python3.x
534// For python2.x, we create the class as a static type and set its tp_base to
535// the passed in type. Unfortunately setting tp_flags to include
536// Py_TPFLAGS_HEAPTYPE does not work by itself since it needs some more
537// initialization of the underlying PyHeapTypeObject and not doing that leads to
538// some random crashes especially during garbage collection.
539// python3.x explicitly disables a static subclass of a HEAPTYPE base class.
540// However it provides a new function, PyType_FromSpecWithBases, to create
541// types dynamically.
542
543// Type object for EagerTensor. This is set by TFE_Py_InitEagerTensor.
544PyTypeObject* EagerTensorType = nullptr;

546#if PY_MAJOR_VERSION >= 3

550 {Py_tp_getset, reinterpret_cast<void*>(EagerTensor_getseters)},
551 {Py_tp_init, reinterpret_cast<void*>(EagerTensor_init)},
552 {0, nullptr},
553};

611TFE_TensorHandle* EagerTensor_Handle(const PyObject* o) {
612 return reinterpret_cast<const EagerTensor*>(o)->handle;
613}

615PyObject* EagerTensorFromHandle(TFE_TensorHandle* handle) {
616 if (handle == nullptr) {
617 return nullptr;
618 }
619 EagerTensor* t = reinterpret_cast<EagerTensor*>(
620 EagerTensorType->tp_new(EagerTensorType, Py_None, Py_None));
621 if (t != nullptr) {
622 t->id = get_uid();
623 Py_INCREF(Py_None);
624 t->handle_data = Py_None;
625 Py_INCREF(Py_None);
626 t->keras_mask = Py_None;
627 Py_INCREF(Py_None);
628 t->tensor_shape = Py_None;
629 t->handle = handle;
630 t->status = TF_NewStatus();
631 }
632 return reinterpret_cast<PyObject*>(t);
633}

635tensorflow::int64 EagerTensor_id(const PyObject* tensor) {
636 CHECK(EagerTensor_CheckExact(tensor));
637 return reinterpret_cast<const EagerTensor*>(tensor)->id;
638}

640PyObject* TFE_Py_InitEagerTensor(PyObject* base_class) {
641 if (!PyType_Check(base_class)) {
642 PyErr_SetString(
643 PyExc_TypeError,
644 tensorflow::strings::StrCat(
645 "Expecting a class definition for `base_class` passed to ",
646 "TFE_InitEagerTensor. Got ", Py_TYPE(base_class)->tp_name)
647 .c_str());
648 return nullptr;
649 }
650 // Note that we allocated kMaxEagerTensorParentSize bytes of unused space in
651 // EagerTensor to allow for the space usage of the base class.
652 PyTypeObject* base_class_type = reinterpret_cast<PyTypeObject*>(base_class);
653 if (base_class_type->tp_basicsize > kMaxEagerTensorParentSize) {
654 PyErr_SetString(
655 PyExc_TypeError,
656 tensorflow::strings::StrCat(
657 "Unable to create subclass EagerTensor from base class ",
658 Py_TYPE(base_class)->tp_name,
659 ". Need its size to be <= ", kMaxEagerTensorParentSize)
660 .c_str());
661 return nullptr;
662 }
663 if (base_class_type->tp_itemsize != 0) {
664 PyErr_SetString(
665 PyExc_TypeError,
666 tensorflow::strings::StrCat(
667 "Unable to create subclass EagerTensor from base class ",
668 Py_TYPE(base_class)->tp_name,
669 " which supports variable length instances.")
670 .c_str());
671 return nullptr;
672 }
673 Py_INCREF(base_class);
674#if PY_MAJOR_VERSION >= 3
675 PyObject* bases = PyTuple_New(1);
676 PyTuple_SET_ITEM(bases, 0, base_class);
677 EagerTensorType = reinterpret_cast<PyTypeObject*>(
678 PyType_FromSpecWithBases(&EagerTensor_Type_spec, bases));
679 if (PyErr_Occurred()) {
680 return nullptr;
681 }
682 if (EagerTensorType == nullptr) {
683 PyErr_SetString(PyExc_RuntimeError, "Error while creating EagerTensorType");
684 return nullptr;
685 }
686#else
687 _EagerTensorType.tp_base = reinterpret_cast<PyTypeObject*>(base_class);
688
689 if (PyType_Ready(&_EagerTensorType) < 0) {
690 if (PyErr_Occurred()) return nullptr;
691 PyErr_SetString(PyExc_RuntimeError,
692 "Error while creating EagerTensor type.");
693 return nullptr;
694 }
695 EagerTensorType = &_EagerTensorType;
696 Py_INCREF(EagerTensorType);
697#endif
698 // We disable instance based attribute lookup. Its not clear if these
699 // dictionaries are correctly initialized in the first place.
700 EagerTensorType->tp_dictoffset = 0;
701 return reinterpret_cast<PyObject*>(EagerTensorType);
702}

704PyObject* TFE_Py_TensorShapeSlice(PyObject* tensor_list, int slice_dim) {
705 if (!PyList_Check(tensor_list)) {
706 PyErr_SetString(PyExc_TypeError,
707 tensorflow::strings::StrCat(
708 "tensor_list argument must be a list. Got \"",
709 Py_TYPE(tensor_list)->tp_name, "\"")
710 .c_str());
711 return nullptr;
712 }
713 if (slice_dim < 0) {
714 PyErr_SetString(
715 PyExc_ValueError,
716 tensorflow::strings::StrCat("Slice dimension must be non-negative. "
717 "Got ",
718 slice_dim)
719 .c_str());
720 return nullptr;
721 }
722
723 Py_ssize_t num_tensors = PyList_Size(tensor_list);
724 int64_t num_tensors_int = static_cast<int64_t>(num_tensors);
725 auto tensor = tensorflow::make_safe(TF_AllocateTensor(
726 TF_INT32, &num_tensors_int, /*num_dims=*/1, /*len=*/4 * num_tensors_int));
727 int32_t* data = reinterpret_cast<int32_t*>(TF_TensorData(tensor.get()));
728 auto status = tensorflow::make_safe(TF_NewStatus());
729 for (Py_ssize_t i = 0; i < num_tensors; ++i) {
730 PyObject* tensor_obj = PyList_GET_ITEM(tensor_list, i);
731 if (!EagerTensor_CheckExact(tensor_obj)) {
732 PyErr_SetString(PyExc_TypeError,
733 tensorflow::strings::StrCat(
734 "Expected a list of EagerTensors but "
735 "element ",
736 i, " has type \"", Py_TYPE(tensor_obj)->tp_name, "\"")
737 .c_str());
738 return nullptr;
739 }
740
741 EagerTensor* t = reinterpret_cast<EagerTensor*>(tensor_obj);
742 TFE_TensorHandle* handle = t->handle;
743 int num_dims = TFE_TensorHandleNumDims(handle, status.get());
744 if (MaybeRaiseExceptionFromTFStatus(status.get(), PyExc_ValueError)) {
745 return nullptr;
746 }
747 if (slice_dim >= num_dims) {
748 PyErr_SetString(
749 PyExc_IndexError,
750 tensorflow::strings::StrCat("Slice dimension (", slice_dim,
751 ") must be smaller than rank of all "
752 "tensors, but tensor at index ",
753 i, " has rank ", num_dims)
754 .c_str());
755 return nullptr;
756 }
757 int64_t dim = TFE_TensorHandleDim(handle, slice_dim, status.get());
758 if (MaybeRaiseExceptionFromTFStatus(status.get(), PyExc_ValueError)) {
759 return nullptr;
760 }
761 data[i] = dim;
762 }
763
764 TFE_TensorHandle* handle = TFE_NewTensorHandle(tensor.get(), status.get());
765 if (TF_GetCode(status.get()) != TF_OK) {
766 PyErr_SetString(
767 PyExc_RuntimeError,
768 tensorflow::strings::StrCat("Failed to construct new tensor handle: ",
769 TF_Message(status.get()))
770 .c_str());
771 return nullptr;
772 }
773
774 return EagerTensorFromHandle(handle);
775}
