30// Container types for the various arguments and temporary values used

33// A NameVector is a vector of tensor or operation names, as borrowed
34// C strings.
35typedef tensorflow::gtl::InlinedVector<const char*, 8> NameVector;

37// A PyObjectVector is a vector of borrowed pointers to PyObjects.
38typedef tensorflow::gtl::InlinedVector<PyObject*, 8> PyObjectVector;

40// A TF_TensorVector is a vector of borrowed pointers to TF_Tensors.

43// Run the graph associated with the session starting with the
44// supplied inputs[]. Regardless of success or failure, inputs[] are
45// stolen by the implementation (i.e. the implementation will
46// eventually call Py_DECREF on each array input).
47//
48// The PyObject* feed_dict must be a dictionary mapping strings to
49// NumPy arrays. This function does not modify its reference count.
50//
51// On success, the tensors corresponding to output_names[0,noutputs-1]
52// are placed in out_values[], and these outputs[] become the property
53// of the caller (the caller must eventually call Py_DECREF on them).
54//
55// On failure, out_status contains a tensorflow::Status with an error
56// message.
57void TF_Run_wrapper(TF_DeprecatedSession* session, const TF_Buffer* run_options,
58 PyObject* feed_dict, const NameVector& output_names,
59 const NameVector& target_nodes, TF_Status* out_status,
60 PyObjectVector* out_values, TF_Buffer* run_outputs);

100// Continue to run the graph with additional feeds and fetches. The
101// execution state is uniquely identified by the handle.

141// Runs the graph associated with the session starting with the supplied inputs.
142// On success, `py_outputs` is populated with a numpy ndarray for each output
143// (the caller must decref these ndarrays, although this will likely be handled
144// by the Python gc). `session`, `out_status`, and `py_outputs` must be
145// non-null. `py_outputs` should be empty.
146void TF_SessionRun_wrapper(TF_Session* session, const TF_Buffer* run_options,
147 const std::vector<TF_Output>& inputs,
148 const std::vector<PyObject*>& input_ndarrays,
149 const std::vector<TF_Output>& outputs,
150 const std::vector<TF_Operation*>& targets,
151 TF_Buffer* run_metadata, TF_Status* status,
152 std::vector<PyObject*>* py_outputs);

154// Set up the graph with the intended feeds (inputs) and fetches (output) for
155// a sequence of partial run calls.
156//
157// On success, returns a handle that can be used for subsequent PRun calls. The
158// handle is owned by the caller and should be deleted with TF_DeletePRunHandle

160//
161// On failure, out_status contains a tensorflow::Status with an error
162// message.

164 const std::vector<TF_Output>& inputs,
165 const std::vector<TF_Output>& outputs,
166 const std::vector<TF_Operation*>& targets,
167 const char** out_handle, TF_Status* status);

211// Set the shapes and types for the output's handle.
212//
213// The sizes of 'shapes', 'ranks', and 'types' must be equal; `shapes[i]`
214// contains the shape of the handle's i-th value, `ranks[i]` contains the i-th
215// shape's rank, and `types[i]` contains the i-th value's dtype. If the i-th
216// shape is unknown, then `ranks[i]` must be equal to -1.
217//
218// The space between the double angle brackets below looks extraneous, but
219// our version of SWIG cannot parse ">>".
220void TF_GraphSetOutputHandleShapesAndTypes_wrapper(
221 TF_Graph* graph, TF_Output output,
222 const std::vector<std::vector<int64_t> >& shapes,
223 const std::vector<int>& ranks, const std::vector<TF_DataType>& types,
224 TF_Status* status);

229 const std::vector<int64_t>& dims,
230 bool unknown_shape, TF_Status* status);
