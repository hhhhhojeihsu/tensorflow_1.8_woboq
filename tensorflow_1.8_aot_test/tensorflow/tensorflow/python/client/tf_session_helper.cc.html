42 "feed_dict must be a dictionary mapping strings to NumPy arrays.";

45void TF_Run_wrapper_helper(TF_DeprecatedSession* session, const char* handle,
46 const TF_Buffer* run_options, PyObject* feed_dict,
47 const NameVector& output_names,
48 const NameVector& target_nodes,
49 TF_Status* out_status, PyObjectVector* out_values,
50 TF_Buffer* run_outputs) {
51 // 1. Convert the feed inputs to the appropriate form for TF_Run.
52 if (!PyDict_Check(feed_dict)) {
53 Set_TF_Status_from_Status(out_status,
54 errors::InvalidArgument(kFeedDictErrorMsg));
55 return;
56 }
57
58 NameVector input_names;
59 std::vector<Safe_TF_TensorPtr> inputs_safe; // Used to delete tensors.
60 TF_TensorVector inputs_unsafe; // Used to contain the arg to TF_Run.
61
62 PyObject* key;
63 PyObject* value;
64 Py_ssize_t pos = 0;
65 int index = 0;
66 Status s;
67
68 while (PyDict_Next(feed_dict, &pos, &key, &value)) {
69 char* key_string = PyBytes_AsString(key);
70 if (!key_string) {
71 Set_TF_Status_from_Status(out_status,
72 errors::InvalidArgument(kFeedDictErrorMsg));
73 return;
74 }
75 input_names.push_back(key_string);
76
77 inputs_safe.emplace_back(make_safe(static_cast<TF_Tensor*>(nullptr)));
78 s = PyArrayToTF_Tensor(value, &inputs_safe.back());
79 if (!s.ok()) {
80 Set_TF_Status_from_Status(out_status, s);
81 return;
82 }
83 inputs_unsafe.push_back(inputs_safe.back().get());
84 ++index;
85 }
86
87 // 2. Allocate a container for the output data.
88 TF_TensorVector outputs(output_names.size());
89
90 // In case any tensors were leftover from previous runs we might as well clear
91 // them here.
92 ClearDecrefCache();
93
94 // 3. Actually call TF_Run().
95 Py_BEGIN_ALLOW_THREADS;
96 if (handle == nullptr) {
97 TF_Run(session, run_options, input_names.data(), inputs_unsafe.data(),
98 input_names.size(), const_cast<const char**>(output_names.data()),
99 outputs.data(), output_names.size(),
100 const_cast<const char**>(target_nodes.data()), target_nodes.size(),
101 run_outputs, out_status);
102 } else {
103 TF_PRun(session, handle, input_names.data(), inputs_unsafe.data(),
104 input_names.size(), const_cast<const char**>(output_names.data()),
105 outputs.data(), output_names.size(),
106 const_cast<const char**>(target_nodes.data()), target_nodes.size(),
107 out_status);
108 }
109
110 Py_END_ALLOW_THREADS;
111
112 // Decref any numpy arrays we are not using anymore.
113 ClearDecrefCache();
114
115 if (TF_GetCode(out_status) != TF_OK) {
116 return;
117 }
118
119 // 4. We now own the fetched tensors, so set up a safe container to
120 // delete them when we exit this scope.
121 std::vector<Safe_TF_TensorPtr> tf_outputs_safe;
122 for (const auto& output : outputs) {
123 tf_outputs_safe.emplace_back(make_safe(output));
124 }
125
126 // 5. Convert the fetched tensors into numpy ndarrays. Store them in a safe
127 // container so that we do not leak
128 std::vector<Safe_PyObjectPtr> py_outputs_safe;
129 for (size_t i = 0; i < output_names.size(); ++i) {
130 PyObject* py_array;
131 s = TF_TensorToPyArray(std::move(tf_outputs_safe[i]), &py_array);
132 if (!s.ok()) {
133 Set_TF_Status_from_Status(out_status, s);
134 return;
135 }
136 py_outputs_safe.emplace_back(make_safe(py_array));
137 }
138
139 // 6. If we reach this point, we have successfully built a list of objects
140 // so we can release them from the safe container.
141 for (auto& output : py_outputs_safe) {
142 out_values->push_back(output.release());
143 }
144}

159void MakeCallableHelper(tensorflow::Session* session,
160 const TF_Buffer* callable_options, int64_t* out_handle,
161 TF_Status* out_status) {
162 tensorflow::CallableOptions callable_options_proto;
163 if (callable_options != nullptr &&
164 !callable_options_proto.ParseFromArray(callable_options->data,
165 callable_options->length)) {
166 Set_TF_Status_from_Status(
167 out_status,
168 errors::InvalidArgument("Unparseable CallableOptions proto"));
169 return;
170 }
171 tensorflow::Session::CallableHandle handle;
172 Status s = session->MakeCallable(callable_options_proto, &handle);
173 if (!s.ok()) {
174 Set_TF_Status_from_Status(out_status, s);
175 return;
176 }
177 *out_handle = handle;
178}

196void RunCallableHelper(tensorflow::Session* session, int64_t handle,
197 PyObject* feed_values, TF_Status* out_status,
198 PyObjectVector* out_values, TF_Buffer* run_metadata) {
199 // Convert feed values to a vector of tensorflow::Tensor objects.
200 std::vector<Tensor> input_tensors;
201 Status s;
202 {
203 feed_values =
204 PySequence_Fast(feed_values, "feed_values must be a sequence");
205 if (feed_values == nullptr) return;
206 Safe_PyObjectPtr feed_values_holder(make_safe(feed_values));
207 Py_ssize_t len = PySequence_Fast_GET_SIZE(feed_values);
208 input_tensors.reserve(len);
209 for (Py_ssize_t i = 0; i < len; ++i) {
210 PyObject* elem = PySequence_Fast_GET_ITEM(feed_values, i);
211 if (!elem) {
212 Set_TF_Status_from_Status(
213 out_status, errors::Internal("Could not get feed value ", i));
214 return;
215 }
216 Tensor t;
217 s = NdarrayToTensor(elem, &t);
218 if (!s.ok()) {
219 Set_TF_Status_from_Status(out_status, s);
220 return;
221 }
222 input_tensors.push_back(std::move(t));
223 }
224 }
225
226 // Allocate a RunMetadata protobuf object to receive the metadata,
227 // if the caller is expecting any.
228 std::unique_ptr<RunMetadata> run_metadata_proto;
229 if (run_metadata != nullptr) {
230 run_metadata_proto.reset(new RunMetadata);
231 }
232
233 // Run the callable.
234 std::vector<Tensor> output_tensors;
235 Py_BEGIN_ALLOW_THREADS;
236 s = session->RunCallable(handle, input_tensors, &output_tensors,
237 run_metadata_proto.get());
238 Py_END_ALLOW_THREADS;
239
240 if (!s.ok()) {
241 Set_TF_Status_from_Status(out_status, s);
242 return;
243 }
244
245 // If requested, serialize the RunMetadata to pass it back to the caller.
246 if (run_metadata != nullptr) {
247 s = MessageToBuffer(*run_metadata_proto, run_metadata);
248 if (!s.ok()) {
249 Set_TF_Status_from_Status(out_status, s);
250 return;
251 }
252 }
253
254 // Convert results to NumPy arrays. Since this can fail, stage the
255 // results via a safe container that takes care of decreasing the
256 // reference count on failure.
257 std::vector<Safe_PyObjectPtr> py_outputs_safe;
258 py_outputs_safe.reserve(output_tensors.size());
259 for (const Tensor& output : output_tensors) {
260 PyObject* py_array;
261 s = TensorToNdarray(output, &py_array);
262 if (!s.ok()) {
263 Set_TF_Status_from_Status(out_status, s);
264 return;
265 }
266 py_outputs_safe.push_back(make_safe(py_array));
267 }
268
269 // If we reach this point, we have successfully built a list of objects
270 // so we can release them from the safe container.
271 out_values->reserve(py_outputs_safe.size());
272 for (auto& output : py_outputs_safe) {
273 out_values->push_back(output.release());
274 }
275}

342void TF_SessionRun_wrapper_helper(TF_Session* session, const char* handle,
343 const TF_Buffer* run_options,
344 const std::vector<TF_Output>& inputs,
345 const std::vector<PyObject*>& input_ndarrays,
346 const std::vector<TF_Output>& outputs,
347 const std::vector<TF_Operation*>& targets,
348 TF_Buffer* run_metadata,
349 TF_Status* out_status,
350 std::vector<PyObject*>* py_outputs) {
351 DCHECK_EQ(inputs.size(), input_ndarrays.size());
352 DCHECK(py_outputs != nullptr);
353 DCHECK(py_outputs->empty());
354 Status s;
355
356 // Convert input ndarray PyObjects to TF_Tensors. We maintain a continuous
357 // array of TF_Tensor*s as well as scoped containers to make sure they're
358 // cleaned up properly.
359 //
360 // Memory management:
361 // PyArrayToTF_Tensor() creates a new ndarray PyObject from the input
362 // ndarray. We manage the new ndarray's lifetime in order to keep the
363 // underlying data buffer alive (the new ndarray also guarantees a contiguous
364 // data buffer). The new ndarray's data buffer is used to create the
365 // corresponding TF_Tensor. The TF_Tensor's deallocator will queue the new
366 // ndarray to be decref'd by the next ClearDecrefCache() call (we can't call
367 // Py_DECREF in the deallocator directly because the GIL must be held).
368 //
369 // Note that TF_Tensor may directly delegate its data and deallocator to a
370 // TensorBuffer, which may outlive the TF_Tensor (e.g. if the tensor gets
371 // queued or assigned to a variable).
372 TF_TensorVector input_vals;
373 std::vector<Safe_TF_TensorPtr> input_vals_safe;
374 for (PyObject* ndarray : input_ndarrays) {
375 input_vals_safe.emplace_back(make_safe(static_cast<TF_Tensor*>(nullptr)));
376 s = PyArrayToTF_Tensor(ndarray, &input_vals_safe.back());
377 if (!s.ok()) {
378 Set_TF_Status_from_Status(out_status, s);
379 return;
380 }
381 input_vals.push_back(input_vals_safe.back().get());
382 }
383
384 // Allocate space for output TF_Tensor*s
385 TF_TensorVector output_vals(outputs.size());
386
387 // Clear up any unused memory leftover from previous runs
388 ClearDecrefCache();
389
390 // Call TF_SessionRun() (and release GIL during execution)
391 Py_BEGIN_ALLOW_THREADS;
392 if (handle == nullptr) {
393 TF_SessionRun(session, run_options, inputs.data(), input_vals.data(),
394 inputs.size(), outputs.data(), output_vals.data(),
395 outputs.size(), targets.data(), targets.size(), run_metadata,
396 out_status);
397 } else {
398 TF_SessionPRun(session, handle, inputs.data(), input_vals.data(),
399 inputs.size(), outputs.data(), output_vals.data(),
400 outputs.size(), targets.data(), targets.size(), out_status);
401 }
402 Py_END_ALLOW_THREADS;
403
404 // Create scoped containers for output tensors
405 std::vector<Safe_TF_TensorPtr> output_vals_safe;
406 for (TF_Tensor* output : output_vals) {
407 output_vals_safe.emplace_back(make_safe(output));
408 }
409
410 // Convert outputs to ndarrays (in scoped containers)
411 std::vector<Safe_PyObjectPtr> py_outputs_safe;
412 for (size_t i = 0; i < outputs.size(); ++i) {
413 PyObject* py_array;
414 s = TF_TensorToPyArray(std::move(output_vals_safe[i]), &py_array);
415 if (!s.ok()) {
416 Set_TF_Status_from_Status(out_status, s);
417 return;
418 }
419 py_outputs_safe.emplace_back(make_safe(py_array));
420 }
421
422 // If we reach this point, we have successfully built a list of objects so we
423 // can release them from the safe container into the return vector.
424 for (size_t i = 0; i < outputs.size(); ++i) {
425 py_outputs->push_back(py_outputs_safe[i].release());
426 }
427}

444string EqualGraphDefWrapper(const string& actual, const string& expected) {
445 GraphDef actual_def;
446 if (!actual_def.ParseFromString(actual)) {
447 return "actual is not a valid serialized GraphDef";
448 }
449 GraphDef expected_def;
450 if (!expected_def.ParseFromString(expected)) {
451 return "expected is not a valid serialized GraphDef";
452 }
453 string diff;
454 return EqualGraphDef(actual_def, expected_def, &diff) ? "" : diff;
455}
456
457string EqualAttrValueWrapper(const string& actual, const string& expected) {
458 AttrValue actual_attr_value;
459 if (!actual_attr_value.ParseFromString(actual)) {
460 return "actual is not a valid serialized AttrValue";
461 }
462
463 AttrValue expected_attr_value;
464 if (!expected_attr_value.ParseFromString(expected)) {
465 return "expected is not a valid serialized AttrValue";
466 }
467
468 string diff;
469 if (!AreAttrValuesEqual(actual_attr_value, expected_attr_value)) {
470 diff = strings::Printf(
471 "Actual AttrValue %s does not match Expected AttrValue %s.",
472 SummarizeAttrValue(actual_attr_value).c_str(),
473 SummarizeAttrValue(expected_attr_value).c_str());
474 }
475 return diff;
476}

478// Return value set to 6 inlined elements so it fits in a 64-byte cache line.
479tensorflow::gtl::InlinedVector<int64_t, 6> TF_GraphGetTensorShapeHelper(
480 TF_Graph* graph, TF_Output output, TF_Status* out_status,
481 bool* unknown_shape) {
482 // Allocate a single variable for holding the result for RVO.
483 tensorflow::gtl::InlinedVector<int64_t, 6> result;
484 *unknown_shape = false;
485 int num_dims = TF_GraphGetTensorNumDims(graph, output, out_status);
486 if (TF_GetCode(out_status) != TF_OK) {
487 return result;
488 }
489 // If shape is unknown, set boolean and return.
490 if (num_dims == -1) {
491 *unknown_shape = true;
492 return result;
493 }
494
495 // If shape is a scalar, avoid another C call and just return {}.
496 if (num_dims == 0) {
497 return result;
498 }
499
500 result.resize(num_dims);
501 TF_GraphGetTensorShape(graph, output, result.data(), num_dims, out_status);
502 return result;
503}

575TF_Function* TF_GraphToFunction_wrapper(
576 const TF_Graph* fn_body, const char* fn_name, bool append_hash_to_fn_name,
577 const std::vector<TF_Operation*>* opers,
578 const std::vector<TF_Output>& inputs, const std::vector<TF_Output>& outputs,
579 const NameVector& output_names, const TF_FunctionOptions* opts,
580 const char* description, TF_Status* out_status) {
581 if (!output_names.empty() && output_names.size() != outputs.size()) {
582 Set_TF_Status_from_Status(
583 out_status,
584 errors::InvalidArgument(
585 "output names must be either empty or equal in size to outputs. ",
586 "output names size = ", output_names.size(),
587 " outputs size = ", outputs.size()));
588 return nullptr;
589 }
590
591 int nopers = -1;
592 const TF_Operation* const* opers_array = nullptr;
593 if (opers != nullptr) {
594 nopers = opers->size();
595 opers_array = opers->data();
596 }
597
598 const char** output_names_ptr =
599 output_names.empty() ? nullptr
600 : const_cast<const char**>(output_names.data());
601
602 return TF_GraphToFunction(fn_body, fn_name, append_hash_to_fn_name, nopers,
603 opers_array, inputs.size(), inputs.data(),
604 outputs.size(), outputs.data(), output_names_ptr,
605 opts, description, out_status);
606}

632std::vector<string> TF_ImportGraphDefResultsMissingUnusedInputMappings_wrapper(
633 TF_ImportGraphDefResults* results) {
634 int num_missing_unused_input_mappings;
635 const char** src_names;
636 int* src_indexes;
