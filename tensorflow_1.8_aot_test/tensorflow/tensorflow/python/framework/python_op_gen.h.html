28// hidden_ops should be a vector of Op names that should get a leading _ in the
29// output.
30// The Print* version prints the output to stdout, Get* version returns the
31// output as a string.
32void PrintPythonOps(const OpList& ops, const ApiDefMap& api_defs,
33 const std::vector<string>& hidden_ops, bool require_shapes);
34string GetPythonOps(const OpList& ops, const ApiDefMap& api_defs,
35 const std::vector<string>& hidden_ops, bool require_shapes);
36string GetPythonOp(const OpDef& op_def, const ApiDef& api_def,
37 const string& function_name);

39// Get the python wrappers for a list of ops in a OpList.
40// `op_list_buf` should be a pointer to a buffer containing
41// the binary encoded OpList proto, and `op_list_len` should be the
42// length of that buffer.
43string GetPythonWrappers(const char* op_list_buf, size_t op_list_len);
