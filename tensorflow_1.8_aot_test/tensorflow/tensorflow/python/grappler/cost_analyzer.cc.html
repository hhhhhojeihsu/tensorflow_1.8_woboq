26CostAnalyzer::CostAnalyzer(const GrapplerItem& item, Cluster* cluster,
27 const string& suffix)
28 : item_(&item),
29 measure_estimator_(cluster, 10, 0),
30 analytical_estimator_(cluster, false),
31 suffix_(suffix) {}
32
33Status CostAnalyzer::GenerateReport(std::ostream& os, bool per_node_report,
34 bool verbose) {
35 GatherCosts();
36 PreprocessCosts();
37 AnalyzeCosts();
38 PrintAnalysis(os, per_node_report, verbose);
39 return Status::OK();
40}

42void CostAnalyzer::PredictCosts(CostEstimator* cost_estimator,
43 CostGraphDef* cost_graph, int64* total_time) {
44 TF_CHECK_OK(cost_estimator->Initialize(*item_));
45 Costs costs;
46 const Status status =
47 cost_estimator->PredictCosts(item_->graph, cost_graph, &costs);
48 *total_time = costs.execution_time.count();
49 if (!status.ok()) {
50 LOG(ERROR) << "Could not estimate the cost for item " << item_->id << ": "
51 << status.error_message();
52 return;
53 }
54}

56void CostAnalyzer::GatherCosts() {
57 CostGraphDef cost_graph_measured;
58 PredictCosts(&measure_estimator_, &cost_graph_measured,
59 &total_time_measured_);
60 VLOG(1) << "Graph size: " << item_->graph.node_size();
61 VLOG(1) << "cost_graph_measured size: " << cost_graph_measured.node_size();
62
63 CostGraphDef cost_graph_analytical;
64 PredictCosts(&analytical_estimator_, &cost_graph_analytical,
65 &total_time_analytical_);
66 VLOG(1) << "cost_graph_analytical size: "
67 << cost_graph_analytical.node_size();
68
69 CostGraphDef cost_graph_analytical_filtered;
70 CostGraphDef cost_graph_measured_filtered;
71 std::map<string, const CostGraphDef_Node*> measured_nodes;
72 for (const auto& node : cost_graph_measured.node()) {
73 measured_nodes[node.name()] = &node;
74 }
75 for (const auto& node : cost_graph_analytical.node()) {
76 auto it = measured_nodes.find(node.name());
77 // Filter the nodes that are not the cost nodes returned by
78 // MeasuringCostEstimator.
79 if (it == measured_nodes.end()) {
80 continue;
81 }
82 auto added_node_analytical = cost_graph_analytical_filtered.add_node();
83 auto added_node_measured = cost_graph_measured_filtered.add_node();
84 *added_node_analytical = node;
85 *added_node_measured = *(it->second);
86 }
87 VLOG(1) << "cost_graph_analytical_filtered size: "
88 << cost_graph_analytical_filtered.node_size();
89
90 // TODO(yaozhang): add a test to make sure that op_perf_analytical_ and
91 // op_perf_ cover the same set of nodes.
92 op_perf_analytical_ = CostGraphToOpPerformanceData(
93 cost_graph_analytical_filtered, item_->graph);
94 op_perf_ =
95 CostGraphToOpPerformanceData(cost_graph_measured_filtered, item_->graph);
96}

98void CostAnalyzer::PreprocessCosts() {
99 for (int i = 0; i < op_perf_.op_performance_size(); i++) {
100 OpPerformance* perf = op_perf_.mutable_op_performance(i);
101 const OpPerformance& analytical = op_perf_analytical_.op_performance(i);
102 perf->set_compute_time(analytical.compute_time());
103 perf->set_memory_time(analytical.memory_time());
104 double measured_cost = perf->compute_cost();
105
106 double analytical_compute_cost = analytical.compute_time();
107 if (analytical_compute_cost == 0) {
108 // Negative infinity indidates unavailable data.
109 perf->set_compute_efficiency(-INFINITY);
110 } else {
111 perf->set_compute_efficiency(analytical_compute_cost / measured_cost);
112 }
113
114 double analytical_memory_cost = analytical.memory_time();
115 if (analytical_memory_cost == 0) {
116 // Negative infinity indidates unavailable data.
117 perf->set_memory_efficiency(-INFINITY);
118 } else {
119 perf->set_memory_efficiency(analytical_memory_cost / measured_cost);
120 }
121 }
122}

125void CostAnalyzer::SortOpsByTime(std::map<string, OpPerfSummary> ops) {
126 for (const auto& op : ops) {
127 ops_.push_back(op.second);
128 }
129 struct CompareByTime {
130 bool operator()(const OpPerfSummary& a, const OpPerfSummary& b) const {
131 return a.time > b.time;
132 }
133 };
134 std::stable_sort(ops_.begin(), ops_.end(), CompareByTime());
135}

137void CostAnalyzer::AnalyzeCosts() {
138 std::map<string, OpPerfSummary> ops;
139 for (const auto& op_perf : op_perf_.op_performance()) {
140 string op_name = op_perf.op().op();
141 ops[op_name].count++;
142 ops[op_name].time += op_perf.compute_cost();
143 ops[op_name].compute_time += op_perf.compute_time();
144 ops[op_name].memory_time += op_perf.memory_time();
145 ops[op_name].time_upper += op_perf.compute_time() + op_perf.memory_time();
146 ops[op_name].time_lower +=
147 std::max(op_perf.compute_time(), op_perf.memory_time());
148 ops[op_name].name = op_name;
149 }
150 SortOpsByTime(ops);
151
152 total_time_measured_serialized_ = 0;
153 total_time_analytical_upper_ = 0;
154 total_time_analytical_lower_ = 0;
155 for (const auto& op : ops_) {
156 total_time_measured_serialized_ += op.time;
157 total_time_analytical_upper_ += op.time_upper;
158 total_time_analytical_lower_ += op.time_lower;
159 }
160}

162void CostAnalyzer::PrintAnalysis(std::ostream& os, bool per_node_report,
163 bool verbose) const {
164 os << std::endl;
165 os << std::left << std::setw(50)
166 << "Total time measured in ns (serialized): " << std::right
167 << std::setw(20) << total_time_measured_serialized_ << std::endl;
168 os << std::left << std::setw(50)
169 << "Total time measured in ns (actual): " << std::right << std::setw(20)
170 << total_time_measured_ << std::endl;
171 os << std::left << std::setw(50)
172 << "Total time analytical in ns (upper bound): " << std::right
173 << std::setw(20) << total_time_analytical_upper_ << std::endl;
174 os << std::left << std::setw(50)
175 << "Total time analytical in ns (lower bound): " << std::right
176 << std::setw(20) << total_time_analytical_lower_ << std::endl;
177 double efficiency_upper = static_cast<double>(total_time_analytical_upper_) /
178 static_cast<double>(total_time_measured_);
179 os << std::left << std::setw(50)
180 << "Overall efficiency (analytical upper/actual): " << std::right
181 << std::setw(20) << efficiency_upper << std::endl;
182 double efficiency_lower = static_cast<double>(total_time_analytical_lower_) /
183 static_cast<double>(total_time_measured_);
184 os << std::left << std::setw(50)
185 << "Overall efficiency (analytical lower/actual): " << std::right
186 << std::setw(20) << efficiency_lower << std::endl;
187 os << std::endl;
188
189 int width = 35;
190 int width_narrow = 15;
191 int width_wide = 20;
192 os << std::setw(width + 1) << "Op,";
193 os << std::setw(width_narrow + 1) << "Count,";
194 os << std::setw(width_wide + 1) << "Measured time (ns),";
195 os << std::setw(width_narrow + 2) << "Time percent,";
196 os << std::setw(width_narrow + 2) << "Acc percent,";
197 os << std::setw(width_wide + 1) << "Analytical upper,";
198 os << std::setw(width_wide + 1) << "Analytical lower,";
199 os << std::setw(width_narrow + 2) << "Overall eff";
200 os << std::setw(width_narrow + 2) << "Compute eff";
201 os << std::setw(width_narrow + 2) << "Memory eff" << std::endl;
202 float acc_percent = 0;
203 for (const auto& op : ops_) {
204 double percent = static_cast<double>(op.time) /
205 static_cast<double>(total_time_measured_serialized_);
206 double eff =
207 static_cast<double>(op.time_upper) / static_cast<double>(op.time);
208 double compute_eff =
209 static_cast<double>(op.compute_time) / static_cast<double>(op.time);
210 double memory_eff =
211 static_cast<double>(op.memory_time) / static_cast<double>(op.time);
212 os << std::setw(width) << op.name << ",";
213 os << std::setw(width_narrow) << op.count << ",";
214 os << std::setw(width_wide) << op.time << ",";
215 os << std::setw(width_narrow) << std::setprecision(2) << percent * 100
216 << "%,";
217 acc_percent += percent;
218 os << std::setw(width_narrow) << std::setprecision(2) << acc_percent * 100
219 << "%,";
220 os << std::setw(width_wide) << op.time_upper << ",";
221 os << std::setw(width_wide) << op.time_lower << ",";
222 os << std::setw(width_narrow) << std::setprecision(2) << eff * 100 << "%,";
223 os << std::setw(width_narrow) << std::setprecision(2) << compute_eff * 100
224 << "%,";
225 os << std::setw(width_narrow) << std::setprecision(2) << memory_eff * 100
226 << "%,";
227 os << std::endl;
228 }
229 os << std::endl;
230
231 if (per_node_report) {
232 if (verbose) {
233 os << "Below is the full per-node report:" << std::endl;
234 os << op_perf_.DebugString();
235 } else {
236 os << "Below is the per-node report summary:" << std::endl;
237 int width = 35;
238 int width_narrow = 15;
239 int width_wide = 20;
240 os << std::setw(width + 1) << "Op,";
241 os << std::setw(width_wide + 1) << "Measured time (ns),";
242 os << std::setw(width_wide + 1) << "Compute time (ns),";
243 os << std::setw(width_wide + 1) << "Memory time (ns),";
244 os << std::setw(width_narrow + 2) << "Compute eff,";
245 os << std::setw(width_narrow + 2) << "Memory eff,";
246 os << " Inputs" << std::endl;
247 for (int i = 0; i < op_perf_.op_performance_size(); i++) {
248 const auto& perf = op_perf_.op_performance(i);
249 string op_name = perf.op().op();
250 os << std::setw(width) << op_name << ",";
251 os << std::setw(width_wide) << perf.compute_cost() << ",";
252 os << std::setw(width_wide) << perf.compute_time() << ",";
253 os << std::setw(width_wide) << perf.memory_time() << ",";
254 os << std::setw(width_narrow) << std::setprecision(2)
255 << perf.compute_efficiency() * 100 << "%,";
256 os << std::setw(width_narrow) << std::setprecision(2)
257 << perf.memory_efficiency() * 100 << "%,";
258 os << " [";
259 for (int j = 0; j < perf.op().inputs_size(); j++) {
260 const auto& shape = perf.op().inputs(j).shape();
261 if (shape.dim_size() > 0) {
262 os << "(";
263 std::vector<int> dims;
264 for (int k = 0; k < shape.dim_size(); k++) {
265 os << shape.dim(k).size();
266 if (k < shape.dim_size() - 1) {
267 os << ", ";
268 }
269 }
270 os << ")";
271 if (j < perf.op().inputs_size() - 1) {
272 os << ", ";
273 }
274 }
275 }
276 os << "]" << std::endl;
277 }
278 os << std::endl;
279 }
280 }
