33// Gets tensor names from tensor_info and inserts them into the set of tensor
34// names.
35void GetTensorNamesFromTensorInfo(const TensorInfo& tensor_info,
36 std::unordered_set<string>* tensor_names) {
37 if (tensor_info.has_coo_sparse()) {
38 // If the tensor is sparse we have to add all three tensors of the sparse
39 // representations.
40 const TensorInfo_CooSparse& coo_sparse = tensor_info.coo_sparse();
41 tensor_names->insert(coo_sparse.values_tensor_name());
42 tensor_names->insert(coo_sparse.indices_tensor_name());
43 tensor_names->insert(coo_sparse.dense_shape_tensor_name());
44 } else {
45 tensor_names->insert(tensor_info.name());
46 }
47}

49// Gets the union of all inputs and outputs of all SignatureDefs in the bundle
50void GetSignatureDefsInputsAndOutputs(
51 const SavedModelBundle& saved_model_bundle,
52 std::unordered_set<string>* inputs, std::unordered_set<string>* outputs) {
53 for (auto& sigdef_elem : saved_model_bundle.meta_graph_def.signature_def()) {
54 const SignatureDef& signature_def = sigdef_elem.second;
55 for (auto& input_elem : signature_def.inputs()) {
56 GetTensorNamesFromTensorInfo(input_elem.second, inputs);
57 }
58 for (auto& output_elem : signature_def.outputs()) {
59 GetTensorNamesFromTensorInfo(output_elem.second, outputs);
60 }
61 }
62}

64// Gets a map from string node name to NodeDef.
65void GetNodeNameToNodeDefMap(
66 GraphDef* graph_def,
67 std::unordered_map<string, NodeDef*>* name_to_node_map) {
68 for (size_t i = 0; i < graph_def->node_size(); i++) {
69 NodeDef* node = graph_def->mutable_node(i);
70 (*name_to_node_map)[node->name()] = node;
71 }
72}

74// Gets the set of node names needed by `outputs` and the corresponding set of
75// variable nodes to convert.
76void GetReachableNodesAndVariables(
77 GraphDef* graph_def, const std::unordered_set<string>& outputs,
78 const std::unordered_map<string, NodeDef*>& name_to_node_map,
79 std::unordered_set<string>* reachable_node_names,
80 std::unordered_set<string>* variable_node_names) {
81 // TODO(suharshs): Add support for ResourceVariables.
82 static const std::unordered_set<string>* kVariableTypes =
83 new std::unordered_set<string>({"Variable", "VariableV2", "VarHandleOp"});
84
85 std::queue<string> nodes_to_visit;
86 for (const string& tensor_name : outputs) {
87 // We need to strip off the tensor part to get the node name.
88 std::vector<string> tensor_name_parts = str_util::Split(tensor_name, ':');
89 nodes_to_visit.push(tensor_name_parts[0]);
90 }
91 // We do a traversal backwards from the outputs specified in the MetaGraphDef.
92 while (!nodes_to_visit.empty()) {
93 const string node_name = nodes_to_visit.front();
94 nodes_to_visit.pop();
95 if (reachable_node_names->find(node_name) != reachable_node_names->end()) {
96 continue;
97 }
98 reachable_node_names->insert(node_name);
99 NodeDef* node = name_to_node_map.at(node_name);
100 if (kVariableTypes->find(node->op()) != kVariableTypes->end()) {
101 variable_node_names->insert(node->name());
102 }
103 for (const string& input : node->input()) {
104 nodes_to_visit.push(input);
105 }
106 }
107}

109// Gets a map from variable name to variable value.
110Status GetVariableNameToTensorMap(
111 Session* session,
112 const std::unordered_map<string, NodeDef*>& name_to_node_map,
113 std::unordered_set<string> variable_names_set,
114 std::unordered_map<string, Tensor>* variable_name_to_value_map) {
115 if (variable_names_set.empty()) {
116 return Status::OK();
117 }
118 std::vector<string> variable_names;
119 std::vector<string> tensor_names;
120 for (const string& node_name : variable_names_set) {
121 variable_names.push_back(node_name);
122 NodeDef* node_def = name_to_node_map.at(node_name);
123 if (node_def->op() == "VarHandleOp") {
124 // If this is a resource variable, we have to run the corresponding
125 // ReadVariableOp.
126 tensor_names.push_back(node_name + "/Read/ReadVariableOp:0");
127 } else {
128 tensor_names.push_back(node_name + ":0");
129 }
130 }
131 std::vector<Tensor> outputs;
132 TF_RETURN_IF_ERROR(
133 session->Run(/* inputs */ {}, tensor_names, /* targets */ {}, &outputs));
134 for (size_t i = 0; i < variable_names.size(); i++) {
135 (*variable_name_to_value_map)[variable_names[i]] = outputs[i];
136 }
137 return Status::OK();
138}

140// Converts a Variable NodeDef into a Constant NodeDef.
141void ConvertVariableToConstant(const NodeDef& variable_node,
142 const Tensor& variable_value,
143 NodeDef* const_node) {
144 const_node->set_name(variable_node.name());
145 const_node->set_op("Const");
146 (*const_node->mutable_attr())["dtype"] = variable_node.attr().at("dtype");
147 variable_value.AsProtoTensorContent(
148 (*const_node->mutable_attr())["value"].mutable_tensor());
149}
150
151// Converts a ReadVariableOp NodeDef to an Identity NodeDef.
152void ConvertReadVariableOpToIdentity(const NodeDef& node,
153 NodeDef* identity_node) {
154 identity_node->set_name(node.name());
155 identity_node->set_op("Identity");
156 (*identity_node->mutable_attr())["T"] = node.attr().at("dtype");
157 identity_node->add_input(node.input(0));
158}

160// Freezes the subgraph of all nodes needed by `outputs`.
161Status FreezeGraphDef(const SavedModelBundle& saved_model_bundle,
162 const std::unordered_set<string>& outputs,
163 GraphDef* frozen_graph_def) {
164 GraphDef graph_def = saved_model_bundle.meta_graph_def.graph_def();
165 // Copy versions and library as-is from original graph.
166 *frozen_graph_def->mutable_versions() = graph_def.versions();
167 *frozen_graph_def->mutable_library() = graph_def.library();
168 // If the graph is empty there is nothing left to do.
169 if (graph_def.node_size() == 0) {
170 return Status::OK();
171 }
172 // name_to_node_map is needed to get the inputs from the NodeDef corresponding
173 // the a string node name. These inputs are used when doing our backwards
174 // traversal.
175 std::unordered_map<string, NodeDef*> name_to_node_map;
176 GetNodeNameToNodeDefMap(&graph_def, &name_to_node_map);
177 std::unordered_set<string> reachable_node_names;
178 std::unordered_set<string> variable_node_names;
179 GetReachableNodesAndVariables(&graph_def, outputs, name_to_node_map,
180 &reachable_node_names, &variable_node_names);
181 std::unordered_map<string, Tensor> variable_to_value_map;
182 TF_RETURN_IF_ERROR(GetVariableNameToTensorMap(
183 saved_model_bundle.session.get(), name_to_node_map, variable_node_names,
184 &variable_to_value_map));
185 // We copy the nodes in the same order they were in the original graph_def.
186 for (const NodeDef& node : graph_def.node()) {
187 if (reachable_node_names.find(node.name()) == reachable_node_names.end()) {
188 continue;
189 }
190 if (variable_node_names.find(node.name()) != variable_node_names.end()) {
191 ConvertVariableToConstant(node, variable_to_value_map[node.name()],
192 frozen_graph_def->add_node());
193 } else if (node.op() == "ReadVariableOp" &&
194 variable_node_names.find(node.input(0)) !=
195 variable_node_names.end()) {
196 // If the node is a ReadVariableOp, its input VarHandleOp will be
197 // converted to a Constant, so we will need to convert it to an Identity.
198 ConvertReadVariableOpToIdentity(node, frozen_graph_def->add_node());
199 } else {
200 // If the node isn't a variable, just copy the node as-is.
201 *frozen_graph_def->add_node() = node;
202 }
203 }
204 return Status::OK();
205}

209Status FreezeSavedModel(const SavedModelBundle& saved_model_bundle,
210 GraphDef* frozen_graph_def,
211 std::unordered_set<string>* inputs,
212 std::unordered_set<string>* outputs) {
213 GetSignatureDefsInputsAndOutputs(saved_model_bundle, inputs, outputs);
214 TF_RETURN_IF_ERROR(
215 FreezeGraphDef(saved_model_bundle, *outputs, frozen_graph_def));
216 return Status::OK();
217}
