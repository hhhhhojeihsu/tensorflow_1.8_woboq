214std::unordered_set<string> Scope::Impl::GetColocationConstraints(
215 const Operation& colocate_with_op) const {
216 std::unordered_set<string> current_constraints(colocation_constraints_);
217 const AttrSlice attrs = colocate_with_op.node()->attrs();
218 std::vector<string> node_constraints;
219 if (GetNodeAttr(attrs, kColocationAttrName, &node_constraints).ok()) {
220 for (const string& entry : node_constraints) {
221 StringPiece s(entry);
222 if (str_util::ConsumePrefix(&s, kColocationGroupPrefix)) {
223 current_constraints.insert(s.ToString());
224 }
225 }
226 } else {
227 current_constraints.insert(colocate_with_op.node()->name());
228 }
229 return current_constraints;
230}

271void Scope::UpdateBuilder(NodeBuilder* builder) const {
272 std::vector<Node*> control_inputs;
273 for (const auto& op : impl()->control_deps_) {
274 control_inputs.push_back(op.node());
275 }
276 builder->ControlInputs(control_inputs);
277
278 if (!impl()->kernel_label_.empty()) {
279 builder->Attr("_kernel", impl()->kernel_label_);
280 }
281
282 if (!impl()->colocation_constraints_.empty()) {
283 std::vector<string> constraints(impl()->colocation_constraints_.begin(),
284 impl()->colocation_constraints_.end());
285 // Sort the set.
286 std::sort(constraints.begin(), constraints.end());
287 // Add loc:@ prefix
288 std::transform(constraints.begin(), constraints.end(), constraints.begin(),
289 [](const string& s) {
290 return strings::StrCat(kColocationGroupPrefix, s);
291 });
292 builder->Attr(kColocationAttrName, constraints);
293 }
294 if (!impl()->device_.empty()) {
295 builder->Device(impl()->device_);
296 }
297}

299string Scope::Impl::GetUniqueName(const string& prefix,
300 bool check_single_use) const {
301 if (check_single_use && single_use_scope()) {
302 if (*scope_used_) {
303 *status_ =
304 errors::AlreadyExists(prefix, " already exists in the current scope");
305 return "";
306 }
307 *scope_used_ = true;
308 return prefix;
309 }
310 auto entry = name_map_->find(prefix);
311 string unique_name = prefix;
312 if (entry == name_map_->end()) {
313 name_map_->insert({prefix, 0});
314 } else {
315 unique_name = strings::StrCat(unique_name, "_", ++entry->second);
316 }
317 return unique_name;
318}

320string Scope::Impl::GetNameForOp(const string& default_name) const {
321 const string unique_name =
322 GetUniqueName(default_name, true /* check_single_use */);
323 const string sep = name_.empty() || unique_name.empty() ? "" : "/";
324 return strings::StrCat(name_, sep, unique_name);
325}
326
327string Scope::GetUniqueNameForOp(const string& default_name) const {
328 if (impl()->single_use_scope()) {
329 if (impl()->op_name_.empty() || *impl()->scope_used_) {
330 *impl()->status_ =
331 errors::InvalidArgument("Cannot get a unique name in this scope");
332 return "";
333 }
334 *impl()->scope_used_ = true;
335 return impl()->op_name_;
336 }
337 return impl()->op_name_.empty() ? impl()->GetNameForOp(default_name)
338 : impl()->GetNameForOp(impl()->op_name_);
339}

341Scope Scope::NewSubScope(const string& child_scope_name) const {
342 if (child_scope_name.empty()) {
343 return Scope(new Impl(*this, Impl::Tags::ScopeName(), impl()->name_,
344 true /* copy_names */));
345 }
346 const string unique_name =
347 impl()->GetUniqueName(child_scope_name, false /* check_single_use */);
348 const string sep = impl()->name_.empty() || unique_name.empty() ? "" : "/";
349 return Scope(new Impl(*this, Impl::Tags::ScopeName(),
350 strings::StrCat(impl()->name_, sep, unique_name),
351 false /* copy_names */));
352}

405CompositeOpScopes Scope::GetCompositeOpScopes(
406 const string& composite_op_name) const {
407 if (impl()->op_name_.empty() && composite_op_name.empty()) {
408 UpdateStatus(errors::InvalidArgument(
409 "Cannot create composite op scopes with empty name"));
410 return {*this, *this};
411 }
412 if (!impl()->single_use_scope()) {
413 Scope child = NewSubScope(impl()->op_name_.empty() ? composite_op_name
414 : impl()->op_name_);
415 const string child_op_sep = impl()->name_.empty() ? "" : "_";
416 const string child_name =
417 strings::StrCat(impl()->name_, child_op_sep, child.impl()->name_);
418 return {child,
419 Scope(new Impl(child, Impl::Tags::SingleUseScope(), child_name))};
420 } else {
421 return {Scope(new Impl(*this, Impl::Tags::ScopeName(), impl()->op_name_,
422 true /* copy_names */)),
423 *this};
424 }
425}

432class InternalScope {
433 public:
434 // NewScope doesn't take ownership of the inputs.
435 static Scope NewScope(Graph* graph, Status* status, ShapeRefiner* refiner) {
436 Scope::Impl::NameMap* name_map = new Scope::Impl::NameMap;
437 for (const Node* node : graph->nodes()) {
438 (*name_map)[node->name()] = 0;
439 }
440 // We provide null destructors for these shared ptrs (except for name_map)
441 // since the caller owns them and doesn't want the scope to destroy them.
442 return Scope(new Scope::Impl(
443 std::shared_ptr<Graph>(graph, [](Graph*) {}),
444 std::shared_ptr<Status>(status, [](Status*) {}),
445 std::shared_ptr<Scope::Impl::NameMap>(name_map),
446 std::shared_ptr<ShapeRefiner>(refiner, [](ShapeRefiner*) {})));
447 }
448};
449
450Scope NewInternalScope(Graph* graph, Status* status, ShapeRefiner* refiner) {
451 return InternalScope::NewScope(graph, status, refiner);
452}
