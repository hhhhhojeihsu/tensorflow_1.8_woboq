30// TODO(pauldonnelly): Add support for double inputs and scales to this Op,
31// (currently Attr does not support double).
32
33REGISTER_OP("FusedConv2DBiasActivation")
34 .Input("conv_input: T")
35 .Input("filter: T")
36 .Input("bias: Tbias")
37 .Input("side_input: T")
38 .Input("conv_input_scale: float")
39 .Input("side_input_scale: float")
40 .Output("output: T")
41 .Attr("T: {float, half, qint8}")
42 .Attr("Tbias: {float, half}")
43 .Attr("strides: list(int)")
44 .Attr(GetPaddingAttrString())
45 .Attr("data_format: {'NHWC', 'NCHW', 'NCHW_VECT_C'} = 'NHWC'")
46 .Attr("filter_format: {'HWIO', 'OIHW', 'OIHW_VECT_I'} = 'HWIO'")
47 .Attr("activation_mode: {'Relu'} = 'Relu'")
48 .Attr("dilations: list(int) = [1, 1, 1, 1]")
49 .SetShapeFn([](shape_inference::InferenceContext* c) {
50 using shape_inference::ShapeHandle;
51 using shape_inference::DimensionHandle;
52 TF_RETURN_IF_ERROR(shape_inference::Conv2DShape(c));
53
54 string data_format_str, filter_format_str;
55 TF_RETURN_IF_ERROR(c->GetAttr("data_format", &data_format_str));
56 TF_RETURN_IF_ERROR(c->GetAttr("filter_format", &filter_format_str));
57
58 TensorFormat data_format;
59 FormatFromString(data_format_str, &data_format);
60 FilterTensorFormat filter_format;
61 FilterFormatFromString(filter_format_str, &filter_format);
62
63 constexpr int num_spatial_dims = 2;
64 const int rank =
65 GetTensorDimsFromSpatialDims(num_spatial_dims, data_format);
66 ShapeHandle filter_shape;
67 TF_RETURN_IF_ERROR(c->WithRank(c->input(1), rank, &filter_shape));
68
69 DimensionHandle output_depth_dim =
70 c->Dim(filter_shape,
71 GetFilterDimIndex<num_spatial_dims>(filter_format, 'O'));
72 int64 output_depth_dim_val = c->Value(output_depth_dim);
73
74 ShapeHandle bias_shape;
75 // Bias should be a 1-D tensor.
76 TF_RETURN_IF_ERROR(c->WithRank(c->input(2), 1, &bias_shape));
77 DimensionHandle bias_dim = c->Dim(bias_shape, 0);
78 int64 bias_dim_val = c->Value(bias_dim);
79
80 if (output_depth_dim_val != bias_dim_val) {
81 return errors::InvalidArgument(
82 "Output depth dimension (", output_depth_dim_val,
83 ") and bias dimension (", bias_dim_val, ") do not match.");
84 }
85
86 // Check side input shape matches the output shape.
87 ShapeHandle side_input_shape;
88 TF_RETURN_IF_ERROR(c->WithRankAtLeast(c->input(3), 1, &side_input_shape));
89 if (c->Rank(side_input_shape) > 1) {
90 ShapeHandle unused;
91 TF_RETURN_IF_ERROR(c->Merge(side_input_shape, c->output(0), &unused));
92 }
93
94 // Check that conv_input_scale and side_input_scale are scalar tensors.
95 ShapeHandle unused;
96 TF_RETURN_IF_ERROR(c->WithRank(c->input(4), 0, &unused));
97 TF_RETURN_IF_ERROR(c->WithRank(c->input(5), 0, &unused));
98
99 return Status::OK();
100 })
101 .Doc(R"doc(
102 Computes a fused kernel which implements: 2-D convolution, adds side input,
103 with separate scaling on convolution and side inputs, then adds bias and
104 applies the RELU activation function to the result. Supports both float and
105 qint8 data formats. In the case of qint8, the output is clipped to [0..127].
106
107 conv_input: A tensor with format as specified by `data_format` (see below).
108 filter: A tensor with format depending on `data_format` as follows:
109 "NHWC", "NCHW":
110 `float [ filter_height, filter_width, in_channels, out_channels ]`
111 "NCHW_VECT_C":
112 `qint8 [ out_channels, in_channels, filter_height, filter_width ]`
113 bias: 1-D float tensor with size matching the `out_channels` dimension of
114 `filter`.
115 Note: this tensor is still float, even if other inputs are qint8.
116 side_input: A tensor with format as specified by `data_format` (see below).
117 This tensor will be ignored and can be [] if side_input_scale == 0.
118 Otherwise, the size of each dimension must match the `output` tensor.
119 conv_input_scale: scalar float value to be multiplied by `conv_input`.
120 (conceptually.. in reality it is applied after convolution).
121 side_input_scale: scalar float value to be multiplied by `side_input`.
122 output: A tensor with format as specified by `data_format` (see below).
123 The dimension sizes are determined automatically based on other inputs
124 and attributes.
125 T: The element data type of `conv_input`, `side_input` and `output` tensors.
126 Note: must match with the `data_format`.
127 Tbias: The element data type of `bias`.
128 strides: 1-D tensor of length 4. The stride of the sliding window for each
129 dimension of `input`. The dimension order is determined by the value of
130 `data_format`, see below for details.
131 Note: the stride for batch and channel dimensions must be 1.
132 padding: The type of padding algorithm to use.
133 data_format: A string specifying the data format of `conv_input`,
134 `side_input` and `output` tensors with the following options:
135 "NHWC": `float [ batch, height, width, channels ]`
136 "NCHW": `float [ batch, channels, height, width ]`
137 "NCHW_VECT_C":
138 `qint8 [ batch, channels / 4, height, width, channels % 4 ]`
139 Note: for "NCHW_VECT_C", `channels` must be a multiple of 4.
140 filter_format: A string specifying the data format of `filter`,
141 "HWIO": `float [ kernel_height, kernel_width, input_channels,
142 output_channels ]`
143 "OIHW_VECT_I":
144 `qint8 [ output_channels, input_channels / 4,
145 kernel_height, kernel_width, input_channels % 4 ]`
146 activation_mode: The activation applied to the output.
147 Currently must be "Relu".
148 dilations: 1-D tensor of length 4. The dilation factor for each dimension
149 of `input`. If set to k > 1, there will be k-1 skipped cells between
150 each filter element on that dimension. The dimension order is determined
151 by the value of `data_format`, see above for details. Dilations in the
152 batch and depth dimensions must be 1.
153)doc");
