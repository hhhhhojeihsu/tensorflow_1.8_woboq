38template <typename T>
39struct Resampler2DFunctor<CPUDevice, T> {
40 void operator()(::tensorflow::OpKernelContext* ctx, const CPUDevice& d,
41 const T* __restrict__ data, const T* __restrict__ warp,
42 T* __restrict__ output, const int batch_size,
43 const int data_height, const int data_width,
44 const int data_channels, const int num_sampling_points) {
45 const int warp_batch_stride = num_sampling_points * 2;
46 const int data_batch_stride = data_height * data_width * data_channels;
47 const int output_batch_stride = num_sampling_points * data_channels;
48 const T zero = static_cast<T>(0.0);
49 const T one = static_cast<T>(1.0);
50
51 auto resample_batches = [&](const int start, const int limit) {
52 for (int batch_id = start; batch_id < limit; ++batch_id) {
53 // Utility lambda to access data point and set output values.
54 // The functions take care of performing the relevant pointer
55 // arithmetics abstracting away the low level details in the
56 // main loop over samples. Note that data is stored in NHWC format.
57 auto set_output = [&](const int sample_id, const int channel,
58 const T value) {
59 output[batch_id * output_batch_stride + sample_id * data_channels +
60 channel] = value;
61 };
62
63 auto get_data_point = [&](const int x, const int y, const int chan) {
64 const bool point_is_in_range =
65 (x >= 0 && y >= 0 && x <= data_width - 1 && y <= data_height - 1);
66 return point_is_in_range
67 ? data[batch_id * data_batch_stride +
68 data_channels * (y * data_width + x) + chan]
69 : zero;
70 };
71
72 for (int sample_id = 0; sample_id < num_sampling_points; ++sample_id) {
73 const T x = warp[batch_id * warp_batch_stride + sample_id * 2];
74 const T y = warp[batch_id * warp_batch_stride + sample_id * 2 + 1];
75 // The interpolation function:
76 // a) implicitly pads the input data with 0s (hence the unusual checks
77 // with {x,y} > -1)
78 // b) returns 0 when sampling outside the (padded) image.
79 // The effect is that the sampled signal smoothly goes to 0 outside
80 // the original input domain, rather than presenting a jump
81 // discontinuity at the image boundaries.
82 if (x > static_cast<T>(-1.0) && y > static_cast<T>(-1.0) &&
83 x < static_cast<T>(data_width) &&
84 y < static_cast<T>(data_height)) {
85 // Precompute floor (f) and ceil (c) values for x and y.
86 const int fx = std::floor(static_cast<float>(x));
87 const int fy = std::floor(static_cast<float>(y));
88 const int cx = fx + 1;
89 const int cy = fy + 1;
90 const T dx = static_cast<T>(cx) - x;
91 const T dy = static_cast<T>(cy) - y;
92
93 for (int chan = 0; chan < data_channels; ++chan) {
94 const T img_fxfy = dx * dy * get_data_point(fx, fy, chan);
95 const T img_cxcy =
96 (one - dx) * (one - dy) * get_data_point(cx, cy, chan);
97 const T img_fxcy = dx * (one - dy) * get_data_point(fx, cy, chan);
98 const T img_cxfy = (one - dx) * dy * get_data_point(cx, fy, chan);
99 set_output(sample_id, chan,
100 img_fxfy + img_cxcy + img_fxcy + img_cxfy);
101 }
102 } else {
103 for (int chan = 0; chan < data_channels; ++chan) {
104 set_output(sample_id, chan, zero);
105 }
106 }
107 }
108 }
109 };
110 // Rough estimate of work for each batch entry.
111 // From third_party/tensorflow/core/util/work_sharder.cc we gather that an
112 // estimate of the cost of each work unit is needed to correctly shard the
113 // workload. Shard assumes each cost unit is 1ns, minimum cost per shard
114 // being 10us.
115 const int64 cost =
116 static_cast<int64>(num_sampling_points) * data_channels * 1000;
117 auto worker_threads = *(ctx->device()->tensorflow_cpu_worker_threads());
118 ::tensorflow::Shard(worker_threads.num_threads, worker_threads.workers,
119 batch_size, cost, resample_batches);
120 }
121};

123} // namespace functor

125template <typename Device, typename T>
126class ResamplerOp : public ::tensorflow::OpKernel {
127 public:
128 explicit ResamplerOp(::tensorflow::OpKernelConstruction* context)
129 : ::tensorflow::OpKernel(context) {}
130
131 void Compute(::tensorflow::OpKernelContext* ctx) override {
132 const ::tensorflow::Tensor& data = ctx->input(0);
133 const ::tensorflow::Tensor& warp = ctx->input(1);
134
135 const ::tensorflow::TensorShape& data_shape = data.shape();
136 OP_REQUIRES(ctx, data_shape.dims() == 4,
137 ::tensorflow::errors::Unimplemented(
138 "Only bilinear interpolation is currently supported. The "
139 "input data shape must be [batch_size, data_height, "
140 "data_width, data_channels], but is: ",
141 data_shape.DebugString()));
142 const ::tensorflow::TensorShape& warp_shape = warp.shape();
143 OP_REQUIRES(ctx,
144 ::tensorflow::TensorShapeUtils::IsMatrixOrHigher(warp_shape),
145 ::tensorflow::errors::InvalidArgument(
146 "warp should be at least a matrix, got shape ",
147 warp_shape.DebugString()));
148 OP_REQUIRES(ctx, warp_shape.dim_size(warp_shape.dims() - 1) == 2,
149 ::tensorflow::errors::Unimplemented(
150 "Only bilinear interpolation is supported, warping "
151 "coordinates must be 2D; warp shape last entry should be "
152 "2, but shape vector is: ",
153 warp_shape.DebugString()));
154 OP_REQUIRES(ctx, data_shape.dim_size(0) == warp_shape.dim_size(0),
155 ::tensorflow::errors::InvalidArgument(
156 "Batch size of data and warp tensor must be the same, but "
157 "input shapes are: ",
158 data_shape.DebugString(), ", ", warp_shape.DebugString()));
159 const int batch_size = data_shape.dim_size(0);
160 const int data_height = data_shape.dim_size(1);
161 const int data_width = data_shape.dim_size(2);
162 const int data_channels = data_shape.dim_size(3);
163 ::tensorflow::TensorShape output_shape = warp.shape();
164 output_shape.set_dim(output_shape.dims() - 1, data_channels);
165 const int num_sampling_points = warp.NumElements() / batch_size / 2;
166 ::tensorflow::Tensor* output = nullptr;
167 OP_REQUIRES_OK(ctx, ctx->allocate_output(0, output_shape, &output));
168
169 // Execute kernel only for nonempty output; otherwise Eigen crashes on GPU.
170 if (num_sampling_points > 0) {
171 functor::Resampler2DFunctor<Device, T>()(
172 ctx, ctx->eigen_device<Device>(), data.flat<T>().data(),
173 warp.flat<T>().data(), output->flat<T>().data(), batch_size,
174 data_height, data_width, data_channels, num_sampling_points);
175 }
176 }
177
178 private:
179 TF_DISALLOW_COPY_AND_ASSIGN(ResamplerOp);
180};

182#define REGISTER(TYPE) \
183 REGISTER_KERNEL_BUILDER( \
184 Name("Resampler").Device(DEVICE_CPU).TypeConstraint<TYPE>("T"), \
185 ResamplerOp<CPUDevice, TYPE>);
186
187TF_CALL_half(REGISTER);
188TF_CALL_float(REGISTER);
189TF_CALL_double(REGISTER);
190#undef REGISTER

192#if GOOGLE_CUDA
193#define REGISTER(TYPE) \
194 REGISTER_KERNEL_BUILDER( \

196 ResamplerOp<GPUDevice, TYPE>)

199#undef REGISTER
200#endif // GOOGLE_CUDA

204template <typename T>
205struct ResamplerGrad2DFunctor<CPUDevice, T> {
206 void operator()(::tensorflow::OpKernelContext* ctx, const CPUDevice& d,
207 const T* __restrict__ data, const T* __restrict__ warp,
208 const T* __restrict__ grad_output, T* __restrict__ grad_data,
209 T* __restrict__ grad_warp, const int batch_size,
210 const int data_height, const int data_width,
211 const int data_channels, const int num_sampling_points) {
212 // Set gradients to 0, because the kernel incrementally updates the
213 // tensor entries by adding partial contributions.
214 const int resampler_output_size =
215 batch_size * num_sampling_points * data_channels;
216 const int grad_warp_size = resampler_output_size / data_channels * 2;
217 const int grad_data_size =
218 data_height * data_width * data_channels * batch_size;
219 memset(grad_data, 0, sizeof(T) * grad_data_size);
220 memset(grad_warp, 0, sizeof(T) * grad_warp_size);
221
222 const auto&& data_batch_stride = data_height * data_width * data_channels;
223 const auto&& warp_batch_stride = num_sampling_points * 2;
224 const int output_batch_stride = num_sampling_points * data_channels;
225 const T zero = static_cast<T>(0.0);
226 const T one = static_cast<T>(1.0);
227
228 auto update_grads_for_batches = [&](const int start, const int limit) {
229 for (int batch_id = start; batch_id < limit; ++batch_id) {
230 // Utility lambdas to access data and update gradient tensors.
231 // The functions take care of performing the relevant pointer
232 // arithmetics abstracting away the low level details in the
233 // main loop over samples. Note that data is stored in NHWC format.
234 auto get_data_point = [&](const int x, const int y, const int chan) {
235 const bool point_is_in_range =
236 (x >= 0 && y >= 0 && x <= data_width - 1 && y <= data_height - 1);
237 return point_is_in_range
238 ? data[batch_id * data_batch_stride +
239 data_channels * (y * data_width + x) + chan]
240 : zero;
241 };
242
243 auto update_grad_data = [&](const int x, const int y, const int chan,
244 const T value) {
245 const bool point_is_in_range =
246 (x >= 0 && y >= 0 && x <= data_width - 1 && y <= data_height - 1);
247 if (point_is_in_range) {
248 grad_data[batch_id * data_batch_stride +
249 data_channels * (y * data_width + x) + chan] += value;
250 }
251 };
252
253 auto update_grad_warp = [&](const int sample_id, const int channel,
254 const T value) {
255 grad_warp[batch_id * warp_batch_stride + sample_id * 2 + channel] +=
256 value;
257 };
258
259 for (int sample_id = 0; sample_id < num_sampling_points; ++sample_id) {
260 const T x = warp[batch_id * warp_batch_stride + sample_id * 2];
261 const T y = warp[batch_id * warp_batch_stride + sample_id * 2 + 1];
262 // The interpolation function whose gradient this function implements:
263 // a) implicitly pads the input data with 0s (hence the unusual checks
264 // with {x,y} > -1)
265 // b) returns 0 when sampling outside the (padded) image.
266 // The effect is that the sampled signal smoothly goes to 0 outside
267 // the original input domain, rather than presenting a jump
268 // discontinuity at the image boundaries.
269 if (x > static_cast<T>(-1.0) && y > static_cast<T>(-1.0) &&
270 x < static_cast<T>(data_width) &&
271 y < static_cast<T>(data_height)) {
272 // Precompute floor (f) and ceil (c) values for x and y.
273 const int fx = std::floor(static_cast<float>(x));
274 const int fy = std::floor(static_cast<float>(y));
275 const int cx = fx + 1;
276 const int cy = fy + 1;
277 const T dx = static_cast<T>(cx) - x;
278 const T dy = static_cast<T>(cy) - y;
279
280 for (int chan = 0; chan < data_channels; ++chan) {
281 const T grad_output_value =
282 grad_output[batch_id * output_batch_stride +
283 sample_id * data_channels + chan];
284 const T img_fxfy = get_data_point(fx, fy, chan);
285 const T img_cxcy = get_data_point(cx, cy, chan);
286 const T img_fxcy = get_data_point(fx, cy, chan);
287 const T img_cxfy = get_data_point(cx, fy, chan);
288
289 // Update partial gradients wrt relevant warp field entries
290 update_grad_warp(
291 sample_id, 0,
292 grad_output_value * ((one - dy) * (img_cxcy - img_fxcy) +
293 dy * (img_cxfy - img_fxfy)));
294
295 update_grad_warp(
296 sample_id, 1,
297 grad_output_value * ((one - dx) * (img_cxcy - img_cxfy) +
298 dx * (img_fxcy - img_fxfy)));
299
300 // Update partial gradients wrt sampled data
301 update_grad_data(fx, fy, chan, grad_output_value * dx * dy);
302 update_grad_data(cx, cy, chan,
303 grad_output_value * (one - dx) * (one - dy));
304 update_grad_data(fx, cy, chan,
305 grad_output_value * dx * (one - dy));
306 update_grad_data(cx, fy, chan,
307 grad_output_value * (one - dx) * dy);
308 }
309 }
310 }
311 }
312 };
313 // Rough estimate of work for each batch entry.
314 // From third_party/tensorflow/core/util/work_sharder.cc we gather that an
315 // estimate of the cost of each work unit is needed to correctly shard the
316 // workload. Shard assumes each cost unit is 1ns, minimum cost per shard
317 // being 10us.
318 // TODO(fviola): Check out if there is a better way of doing this.
319 auto worker_threads = *(ctx->device()->tensorflow_cpu_worker_threads());
320 const int64 cost =
321 static_cast<int64>(num_sampling_points) * data_channels * 1000;
322 ::tensorflow::Shard(worker_threads.num_threads, worker_threads.workers,
323 batch_size, cost, update_grads_for_batches);
324 }
325};

329template <typename Device, typename T>
330class ResamplerGradOp : public ::tensorflow::OpKernel {
331 public:
332 explicit ResamplerGradOp(::tensorflow::OpKernelConstruction* context)
333 : ::tensorflow::OpKernel(context) {}
334
335 void Compute(::tensorflow::OpKernelContext* ctx) override {
336 const ::tensorflow::Tensor& data = ctx->input(0);
337 const ::tensorflow::Tensor& warp = ctx->input(1);
338 const ::tensorflow::Tensor& grad_output = ctx->input(2);
339
340 const ::tensorflow::TensorShape& data_shape = data.shape();
341 OP_REQUIRES(ctx, data_shape.dims() == 4,
342 ::tensorflow::errors::Unimplemented(
343 "Only bilinear interpolation is supported, the input data "
344 "tensor must be a batch of 2d data; data shape should have "
345 "4 entries corresponding to [batch_size, data_height, "
346 "data_width, data_channels], but is: ",
347 data_shape.DebugString()));
348 const int batch_size = data_shape.dim_size(0);
349 const int data_height = data_shape.dim_size(1);
350 const int data_width = data_shape.dim_size(2);
351 const int data_channels = data_shape.dim_size(3);
352 const ::tensorflow::TensorShape& warp_shape = warp.shape();
353 OP_REQUIRES(ctx,
354 ::tensorflow::TensorShapeUtils::IsMatrixOrHigher(warp_shape),
355 ::tensorflow::errors::InvalidArgument(
356 "warp should be at least a matrix, got shape ",
357 warp_shape.DebugString()));
358 OP_REQUIRES(ctx, warp_shape.dim_size(warp_shape.dims() - 1) == 2,
359 ::tensorflow::errors::Unimplemented(
360 "Only bilinear interpolation is supported, warping "
361 "coordinates must be 2D; warp shape last entry should be "
362 "2, but shape vector is: ",
363 warp_shape.DebugString()));
364 const ::tensorflow::TensorShape& grad_output_shape = grad_output.shape();
365 ::tensorflow::TensorShape resampler_output_shape = warp.shape();
366 resampler_output_shape.set_dim(resampler_output_shape.dims() - 1,
367 data_channels);
368 OP_REQUIRES(ctx, grad_output_shape == resampler_output_shape,
369 ::tensorflow::errors::InvalidArgument(
370 "grad_output shape is not consistent with data and warp "
371 "shapes; it should be ",
372 resampler_output_shape.DebugString(), " but is ",
373 grad_output_shape.DebugString()));
374 const int num_sampling_points = warp.NumElements() / batch_size / 2;
375 ::tensorflow::Tensor* grad_data = nullptr;
376 ::tensorflow::Tensor* grad_warp = nullptr;
377 OP_REQUIRES_OK(ctx, ctx->allocate_output(0, data.shape(), &grad_data));
378 OP_REQUIRES_OK(ctx, ctx->allocate_output(1, warp.shape(), &grad_warp));
379 // Execute kernel only for nonempty output; otherwise Eigen crashes on GPU.
380 if (num_sampling_points > 0) {
381 functor::ResamplerGrad2DFunctor<Device, T>()(
382 ctx, ctx->eigen_device<Device>(), data.flat<T>().data(),
383 warp.flat<T>().data(), grad_output.flat<T>().data(),
384 grad_data->flat<T>().data(), grad_warp->flat<T>().data(), batch_size,
385 data_height, data_width, data_channels, num_sampling_points);
386 }
387 }
388
389 private:
390 TF_DISALLOW_COPY_AND_ASSIGN(ResamplerGradOp);
391};
