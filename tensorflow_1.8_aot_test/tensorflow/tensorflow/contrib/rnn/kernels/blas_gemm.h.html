24namespace tensorflow {
25class OpKernelContext;
26namespace functor {
27
28template <typename T>
29struct TensorCuBlasGemm {
30 void operator()(OpKernelContext* ctx, bool transa, bool transb, uint64 m,
31 uint64 n, uint64 k, T alpha, const T* a, int lda, const T* b,
32 int ldb, T beta, T* c, int ldc);
33};
34
35template <typename Device, typename T, bool USE_CUBLAS>
36struct TensorBlasGemm;
37
38template <typename Device, typename T>
39struct TensorBlasGemm<Device, T, true /* USE_CUBLAS */> {
40 static void compute(OpKernelContext* ctx, const Device& d, bool transa,
41 bool transb, T alpha, typename TTypes<T>::ConstMatrix a,
42 typename TTypes<T>::ConstMatrix b, T beta,
43 typename TTypes<T>::Matrix c) {
44 int64 m = c.dimensions()[0];
45 int64 n = c.dimensions()[1];
46 int64 k = transa ? a.dimensions()[0] : a.dimensions()[1];
47
48 TensorCuBlasGemm<T>()(ctx, transb, transa, n, m, k, alpha, b.data(),
49 transb ? k : n, a.data(), transa ? m : k, beta,
50 c.data(), n);
51 }
52};
53
54template <typename Device, typename T>
55struct TensorBlasGemm<Device, T, false /* USE_CUBLAS */> {
56 static void compute(OpKernelContext* ctx, const Device& d, bool transa,
57 bool transb, T alpha, typename TTypes<T>::ConstMatrix a,
58 typename TTypes<T>::ConstMatrix b, T beta,
59 typename TTypes<T>::Matrix c) {
60 Eigen::array<Eigen::IndexPair<Eigen::DenseIndex>, 1> contract_pairs;
61 contract_pairs[0] =
62 Eigen::IndexPair<Eigen::DenseIndex>(transa == false, transb == true);
63 if (alpha == T(1) && beta == T(0)) {
64 c.device(d) = a.contract(b, contract_pairs);
65 } else if (alpha == T(1) && beta == T(1)) {
66 c.device(d) += a.contract(b, contract_pairs);
67 } else {
68 c.device(d) = c.constant(alpha) * a.contract(b, contract_pairs) +
69 c.constant(beta) * c;
70 }
71 }
72};

74} // namespace functor
