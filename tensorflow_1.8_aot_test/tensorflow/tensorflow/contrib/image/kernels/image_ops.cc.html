45using functor::FillProjectiveTransform;
46using generator::Interpolation;
47using generator::INTERPOLATION_BILINEAR;
48using generator::INTERPOLATION_NEAREST;
49using generator::ProjectiveGenerator;
50
51template <typename Device, typename T>
52class ImageProjectiveTransform : public OpKernel {
53 private:
54 Interpolation interpolation_;
55
56 public:
57 explicit ImageProjectiveTransform(OpKernelConstruction* ctx) : OpKernel(ctx) {
58 string interpolation_str;
59 OP_REQUIRES_OK(ctx, ctx->GetAttr("interpolation", &interpolation_str));
60 if (interpolation_str == "NEAREST") {
61 interpolation_ = INTERPOLATION_NEAREST;
62 } else if (interpolation_str == "BILINEAR") {
63 interpolation_ = INTERPOLATION_BILINEAR;
64 } else {
65 LOG(FATAL) << "Invalid interpolation " << interpolation_str
66 << ". Supported types: NEAREST, BILINEAR";
67 }
68 }
69
70 void Compute(OpKernelContext* ctx) override {
71 const Tensor& images_t = ctx->input(0);
72 const Tensor& transform_t = ctx->input(1);
73 OP_REQUIRES(ctx, images_t.shape().dims() == 4,
74 errors::InvalidArgument("Input images must have rank 4"));
75 OP_REQUIRES(ctx,
76 (TensorShapeUtils::IsMatrix(transform_t.shape()) &&
77 (transform_t.dim_size(0) == images_t.dim_size(0) ||
78 transform_t.dim_size(0) == 1) &&
79 transform_t.dim_size(1) ==
80 ProjectiveGenerator<Device, T>::kNumParameters),
81 errors::InvalidArgument(
82 "Input transform should be num_images x 8 or 1 x 8"));
83 auto images = images_t.tensor<T, 4>();
84 auto transform = transform_t.matrix<float>();
85 Tensor* output_t;
86 OP_REQUIRES_OK(ctx, ctx->allocate_output(0, images_t.shape(), &output_t));
87 auto output = output_t->tensor<T, 4>();
88 (FillProjectiveTransform<Device, T>(interpolation_))(
89 ctx->eigen_device<Device>(), &output, images, transform);
90 }
91};

93#define REGISTER(TYPE) \
94 REGISTER_KERNEL_BUILDER(Name("ImageProjectiveTransform") \
95 .Device(DEVICE_CPU) \
96 .TypeConstraint<TYPE>("dtype"), \
97 ImageProjectiveTransform<CPUDevice, TYPE>)

99TF_CALL_uint8(REGISTER);
100TF_CALL_int32(REGISTER);
101TF_CALL_int64(REGISTER);
102TF_CALL_float(REGISTER);
103TF_CALL_double(REGISTER);
