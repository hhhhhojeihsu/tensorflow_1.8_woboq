24Status BatchFeatures::Initialize(
25 std::vector<Tensor> dense_float_features_list,
26 std::vector<Tensor> sparse_float_feature_indices_list,
27 std::vector<Tensor> sparse_float_feature_values_list,
28 std::vector<Tensor> sparse_float_feature_shapes_list,
29 std::vector<Tensor> sparse_int_feature_indices_list,
30 std::vector<Tensor> sparse_int_feature_values_list,
31 std::vector<Tensor> sparse_int_feature_shapes_list) {
32 // Validate number of feature columns.
33 auto num_dense_float_features = dense_float_features_list.size();
34 auto num_sparse_float_features = sparse_float_feature_indices_list.size();
35 auto num_sparse_int_features = sparse_int_feature_indices_list.size();
36 QCHECK(num_dense_float_features + num_sparse_float_features +
37 num_sparse_int_features >
38 0)
39 << "Must have at least one feature column.";
40
41 // Read dense float features.
42 dense_float_feature_columns_.reserve(num_dense_float_features);
43 for (uint32 dense_feat_idx = 0; dense_feat_idx < num_dense_float_features;
44 ++dense_feat_idx) {
45 auto dense_float_feature = dense_float_features_list[dense_feat_idx];
46 TF_CHECK_AND_RETURN_IF_ERROR(
47 TensorShapeUtils::IsMatrix(dense_float_feature.shape()),
48 errors::InvalidArgument("Dense float feature must be a matrix."));
49 TF_CHECK_AND_RETURN_IF_ERROR(
50 dense_float_feature.dim_size(0) == batch_size_,
51 errors::InvalidArgument(
52 "Dense float vector must have batch_size rows: ", batch_size_,
53 " vs. ", dense_float_feature.dim_size(0)));
54 TF_CHECK_AND_RETURN_IF_ERROR(
55 dense_float_feature.dim_size(1) == 1,
56 errors::InvalidArgument(
57 "Dense float features may not be multivalent: dim_size(1) = ",
58 dense_float_feature.dim_size(1)));
59 dense_float_feature_columns_.emplace_back(dense_float_feature);
60 }
61
62 // Read sparse float features.
63 sparse_float_feature_columns_.reserve(num_sparse_float_features);
64 TF_CHECK_AND_RETURN_IF_ERROR(
65 sparse_float_feature_values_list.size() == num_sparse_float_features &&
66 sparse_float_feature_shapes_list.size() == num_sparse_float_features,
67 errors::InvalidArgument("Inconsistent number of sparse float features."));
68 for (uint32 sparse_feat_idx = 0; sparse_feat_idx < num_sparse_float_features;
69 ++sparse_feat_idx) {
70 auto sparse_float_feature_indices =
71 sparse_float_feature_indices_list[sparse_feat_idx];
72 auto sparse_float_feature_values =
73 sparse_float_feature_values_list[sparse_feat_idx];
74 auto sparse_float_feature_shape =
75 sparse_float_feature_shapes_list[sparse_feat_idx];
76 TF_CHECK_AND_RETURN_IF_ERROR(
77 TensorShapeUtils::IsMatrix(sparse_float_feature_indices.shape()),
78 errors::InvalidArgument(
79 "Sparse float feature indices must be a matrix."));
80 TF_CHECK_AND_RETURN_IF_ERROR(
81 TensorShapeUtils::IsVector(sparse_float_feature_values.shape()),
82 errors::InvalidArgument(
83 "Sparse float feature values must be a vector."));
84 TF_CHECK_AND_RETURN_IF_ERROR(
85 TensorShapeUtils::IsVector(sparse_float_feature_shape.shape()),
86 errors::InvalidArgument(
87 "Sparse float feature shape must be a vector."));
88 auto shape_flat = sparse_float_feature_shape.flat<int64>();
89 TF_CHECK_AND_RETURN_IF_ERROR(
90 shape_flat.size() == 2,
91 errors::InvalidArgument(
92 "Sparse float feature column must be two-dimensional."));
93 TF_CHECK_AND_RETURN_IF_ERROR(
94 shape_flat(0) == batch_size_,
95 errors::InvalidArgument(
96 "Sparse float feature shape incompatible with batch size."));
97 auto tensor_shape = TensorShape({shape_flat(0), shape_flat(1)});
98 auto order_dims = sparse::SparseTensor::VarDimArray({0, 1});
99 sparse_float_feature_columns_.emplace_back(sparse_float_feature_indices,
100 sparse_float_feature_values,
101 tensor_shape, order_dims);
102 }
103
104 // Read sparse int features.
105 sparse_int_feature_columns_.reserve(num_sparse_int_features);
106 TF_CHECK_AND_RETURN_IF_ERROR(
107 sparse_int_feature_values_list.size() == num_sparse_int_features &&
108 sparse_int_feature_shapes_list.size() == num_sparse_int_features,
109 errors::InvalidArgument("Inconsistent number of sparse int features."));
110 for (uint32 sparse_feat_idx = 0; sparse_feat_idx < num_sparse_int_features;
111 ++sparse_feat_idx) {
112 auto sparse_int_feature_indices =
113 sparse_int_feature_indices_list[sparse_feat_idx];
114 auto sparse_int_feature_values =
115 sparse_int_feature_values_list[sparse_feat_idx];
116 auto sparse_int_feature_shape =
117 sparse_int_feature_shapes_list[sparse_feat_idx];
118 TF_CHECK_AND_RETURN_IF_ERROR(
119 TensorShapeUtils::IsMatrix(sparse_int_feature_indices.shape()),
120 errors::InvalidArgument(
121 "Sparse int feature indices must be a matrix."));
122 TF_CHECK_AND_RETURN_IF_ERROR(
123 TensorShapeUtils::IsVector(sparse_int_feature_values.shape()),
124 errors::InvalidArgument("Sparse int feature values must be a vector."));
125 TF_CHECK_AND_RETURN_IF_ERROR(
126 TensorShapeUtils::IsVector(sparse_int_feature_shape.shape()),
127 errors::InvalidArgument("Sparse int feature shape must be a vector."));
128 auto shape_flat = sparse_int_feature_shape.flat<int64>();
129 TF_CHECK_AND_RETURN_IF_ERROR(
130 shape_flat.size() == 2,
131 errors::InvalidArgument(
132 "Sparse int feature column must be two-dimensional."));
133 TF_CHECK_AND_RETURN_IF_ERROR(
134 shape_flat(0) == batch_size_,
135 errors::InvalidArgument(
136 "Sparse int feature shape incompatible with batch size."));
137 auto tensor_shape = TensorShape({shape_flat(0), shape_flat(1)});
138 auto order_dims = sparse::SparseTensor::VarDimArray({0, 1});
139 sparse_int_feature_columns_.emplace_back(sparse_int_feature_indices,
140 sparse_int_feature_values,
141 tensor_shape, order_dims);
142 }
143 return Status::OK();
144}
