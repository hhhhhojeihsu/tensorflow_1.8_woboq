31// Creates a tree ensemble variable.
32class CreateTreeEnsembleVariableOp : public OpKernel {
33 public:
34 explicit CreateTreeEnsembleVariableOp(OpKernelConstruction* context)
35 : OpKernel(context) {}
36
37 void Compute(OpKernelContext* context) override {
38 // Get the stamp token.
39 const Tensor* stamp_token_t;
40 OP_REQUIRES_OK(context, context->input("stamp_token", &stamp_token_t));
41 int64 stamp_token = stamp_token_t->scalar<int64>()();
42
43 // Get the tree ensemble config.
44 const Tensor* tree_ensemble_config_t;
45 OP_REQUIRES_OK(context, context->input("tree_ensemble_config",
46 &tree_ensemble_config_t));
47 auto* result = new boosted_trees::models::DecisionTreeEnsembleResource();
48 if (!result->InitFromSerialized(tree_ensemble_config_t->scalar<string>()(),
49 stamp_token)) {
50 result->Unref();
51 OP_REQUIRES(
52 context, false,
53 errors::InvalidArgument("Unable to parse tree ensemble config."));
54 }
55
56 // Only create one, if one does not exist already. Report status for all
57 // other exceptions.
58 auto status = CreateResource(context, HandleFromInput(context, 0), result);
59 if (!status.ok() && status.code() != tensorflow::error::ALREADY_EXISTS) {
60 OP_REQUIRES(context, false, status);
61 }
62 }
63};

65// Op for retrieving a model stamp token without having to serialize.
66class TreeEnsembleStampTokenOp : public OpKernel {
67 public:
68 explicit TreeEnsembleStampTokenOp(OpKernelConstruction* context)
69 : OpKernel(context) {}
70
71 void Compute(OpKernelContext* context) override {
72 boosted_trees::models::DecisionTreeEnsembleResource* ensemble_resource;
73 OP_REQUIRES_OK(context, LookupResource(context, HandleFromInput(context, 0),
74 &ensemble_resource));
75 tf_shared_lock l(*ensemble_resource->get_mutex());
76 core::ScopedUnref unref_me(ensemble_resource);
77 Tensor* output_stamp_token_t = nullptr;
78 OP_REQUIRES_OK(context, context->allocate_output(0, TensorShape(),
79 &output_stamp_token_t));
80 output_stamp_token_t->scalar<int64>()() = ensemble_resource->stamp();
81 }
82};

84// Op for serializing a model.
85class TreeEnsembleSerializeOp : public OpKernel {
86 public:
87 explicit TreeEnsembleSerializeOp(OpKernelConstruction* context)
88 : OpKernel(context) {}
89
90 void Compute(OpKernelContext* context) override {
91 boosted_trees::models::DecisionTreeEnsembleResource* ensemble_resource;
92 OP_REQUIRES_OK(context, LookupResource(context, HandleFromInput(context, 0),
93 &ensemble_resource));
94 tf_shared_lock l(*ensemble_resource->get_mutex());
95 core::ScopedUnref unref_me(ensemble_resource);
96 Tensor* output_stamp_token_t = nullptr;
97 OP_REQUIRES_OK(context, context->allocate_output(0, TensorShape(),
98 &output_stamp_token_t));
99 output_stamp_token_t->scalar<int64>()() = ensemble_resource->stamp();
100 Tensor* output_config_t = nullptr;
101 OP_REQUIRES_OK(
102 context, context->allocate_output(1, TensorShape(), &output_config_t));
103 output_config_t->scalar<string>()() =
104 ensemble_resource->SerializeAsString();
105 }
106};

108// Op for deserializing a tree ensemble variable from a checkpoint.
109class TreeEnsembleDeserializeOp : public OpKernel {
110 public:
111 explicit TreeEnsembleDeserializeOp(OpKernelConstruction* context)
112 : OpKernel(context) {}
113
114 void Compute(OpKernelContext* context) override {
115 boosted_trees::models::DecisionTreeEnsembleResource* ensemble_resource;
116 OP_REQUIRES_OK(context, LookupResource(context, HandleFromInput(context, 0),
117 &ensemble_resource));
118 mutex_lock l(*ensemble_resource->get_mutex());
119 core::ScopedUnref unref_me(ensemble_resource);
120
121 // Get the stamp token.
122 const Tensor* stamp_token_t;
123 OP_REQUIRES_OK(context, context->input("stamp_token", &stamp_token_t));
124 int64 stamp_token = stamp_token_t->scalar<int64>()();
125
126 // Get the tree ensemble config.
127 const Tensor* tree_ensemble_config_t;
128 OP_REQUIRES_OK(context, context->input("tree_ensemble_config",
129 &tree_ensemble_config_t));
130 // Deallocate all the previous objects on the resource.
131 ensemble_resource->Reset();
132 OP_REQUIRES(
133 context,
134 ensemble_resource->InitFromSerialized(
135 tree_ensemble_config_t->scalar<string>()(), stamp_token),
136 errors::InvalidArgument("Unable to parse tree ensemble config."));
137 }
138};

140class TreeEnsembleUsedHandlersOp : public OpKernel {
141 public:
142 explicit TreeEnsembleUsedHandlersOp(OpKernelConstruction* context)
143 : OpKernel(context) {
144 OP_REQUIRES_OK(context,
145 context->GetAttr("num_all_handlers", &num_handlers_));
146 }
147
148 void Compute(OpKernelContext* context) override {
149 boosted_trees::models::DecisionTreeEnsembleResource* ensemble_resource;
150
151 OP_REQUIRES_OK(context, LookupResource(context, HandleFromInput(context, 0),
152 &ensemble_resource));
153 tf_shared_lock l(*ensemble_resource->get_mutex());
154 core::ScopedUnref unref_me(ensemble_resource);
155
156 // Get the stamp token.
157 const Tensor* stamp_token_t;
158 OP_REQUIRES_OK(context, context->input("stamp_token", &stamp_token_t));
159 int64 stamp_token = stamp_token_t->scalar<int64>()();
160
161 // Only the Chief should run this Op and it is guaranteed to be in
162 // a consistent state so the stamps must always match.
163 CHECK(ensemble_resource->is_stamp_valid(stamp_token));
164
165 Tensor* output_used_handlers_t = nullptr;
166 OP_REQUIRES_OK(
167 context, context->allocate_output("used_handlers_mask", {num_handlers_},
168 &output_used_handlers_t));
169 auto output_used_handlers = output_used_handlers_t->vec<bool>();
170
171 Tensor* output_num_used_handlers_t = nullptr;
172 OP_REQUIRES_OK(context,
173 context->allocate_output("num_used_handlers", {},
174 &output_num_used_handlers_t));
175 int handler_idx = 0;
176 std::vector<int64> used_handlers = ensemble_resource->GetUsedHandlers();
177 output_num_used_handlers_t->scalar<int64>()() = used_handlers.size();
178 for (int64 i = 0; i < num_handlers_; ++i) {
179 if (handler_idx >= used_handlers.size() ||
180 used_handlers[handler_idx] > i) {
181 output_used_handlers(i) = false;
182 } else {
183 OP_REQUIRES(context, used_handlers[handler_idx] == i,
184 errors::InvalidArgument("Handler IDs should be sorted."));
185 ++handler_idx;
186 output_used_handlers(i) = true;
187 }
188 }
189 }
190
191 private:
192 int64 num_handlers_;
193};
