56const char* kLearnerConfigAttributeName = "learner_config";
57const char* kSeedTensorName = "seed";
58const char* kApplyDropoutAttributeName = "apply_dropout";
59const char* kApplyAveragingAttributeName = "apply_averaging";
60const char* kDropoutInfoOutputTensorName = "drop_out_tree_indices_weights";
61const char* kPredictionsTensorName = "predictions";
62
63void CalculateTreesToInclude(
64 const boosted_trees::trees::DecisionTreeEnsembleConfig& config,
65 const std::vector<int32>& trees_to_drop, const int32 num_trees,
66 const bool only_finalized, const bool center_bias,
67 std::vector<int32>* trees_to_include) {
68 trees_to_include->reserve(num_trees - trees_to_drop.size());
69
70 int32 index = 0;
71 // This assumes that trees_to_drop is a sorted list of tree ids.
72 for (int32 tree = 0; tree < num_trees; ++tree) {
73 // Skip the tree if tree is in the list of trees_to_drop.
74 if (!trees_to_drop.empty() && index < trees_to_drop.size() &&
75 trees_to_drop[index] == tree) {
76 ++index;
77 continue;
78 }
79 // Or skip if the tree is not finalized and only_finalized is set,
80 // with the exception of centering bias.
81 if (only_finalized && !(center_bias && tree == 0) &&
82 config.tree_metadata_size() > 0 &&
83 !config.tree_metadata(tree).is_finalized()) {
84 continue;
85 }
86 trees_to_include->push_back(tree);
87 }
88}

91class GradientTreesPredictionOp : public OpKernel {
92 public:
93 explicit GradientTreesPredictionOp(OpKernelConstruction* const context)
94 : OpKernel(context) {
95 OP_REQUIRES_OK(context, context->GetAttr("use_locking", &use_locking_));
96
97 OP_REQUIRES_OK(context, context->GetAttr("center_bias", &center_bias_));
98
99 OP_REQUIRES_OK(
100 context, context->GetAttr(kApplyDropoutAttributeName, &apply_dropout_));
101
102 LearnerConfig learner_config;
103 string learner_config_str;
104 OP_REQUIRES_OK(context, context->GetAttr(kLearnerConfigAttributeName,
105 &learner_config_str));
106 OP_REQUIRES(
107 context, ParseProtoUnlimited(&learner_config, learner_config_str),
108 errors::InvalidArgument("Unable to parse learner config config."));
109
110 num_classes_ = learner_config.num_classes();
111 OP_REQUIRES(context, num_classes_ >= 2,
112 errors::InvalidArgument("Number of classes must be >=2"));
113 OP_REQUIRES(
114 context, ParseProtoUnlimited(&learner_config, learner_config_str),
115 errors::InvalidArgument("Unable to parse learner config config."));
116
117 bool reduce_dim;
118 OP_REQUIRES_OK(context, context->GetAttr("reduce_dim", &reduce_dim));
119 prediction_vector_size_ = reduce_dim ? num_classes_ - 1 : num_classes_;
120
121 only_finalized_trees_ =
122 learner_config.growing_mode() == learner_config.WHOLE_TREE;
123 if (learner_config.has_learning_rate_tuner() &&
124 learner_config.learning_rate_tuner().tuner_case() ==
125 LearningRateConfig::kDropout) {
126 dropout_config_ = learner_config.learning_rate_tuner().dropout();
127 has_dropout_ = true;
128 } else {
129 has_dropout_ = false;
130 }
131
132 OP_REQUIRES_OK(context, context->GetAttr(kApplyAveragingAttributeName,
133 &apply_averaging_));
134 apply_averaging_ =
135 apply_averaging_ && learner_config.averaging_config().config_case() !=
136 AveragingConfig::CONFIG_NOT_SET;
137 if (apply_averaging_) {
138 averaging_config_ = learner_config.averaging_config();
139
140 // If there is averaging config, check that the values are correct.
141 switch (averaging_config_.config_case()) {
142 case AveragingConfig::kAverageLastNTreesFieldNumber: {
143 OP_REQUIRES(context, averaging_config_.average_last_n_trees() > 0,
144 errors::InvalidArgument(
145 "Average last n trees must be a positive number"));
146 break;
147 }
148 case AveragingConfig::kAverageLastPercentTreesFieldNumber: {
149 OP_REQUIRES(context,
150 averaging_config_.average_last_percent_trees() > 0 &&
151 averaging_config_.average_last_percent_trees() <= 1.0,
152 errors::InvalidArgument(
153 "Average last percent must be in (0,1] interval."));
154 break;
155 }
156 case AveragingConfig::CONFIG_NOT_SET: {
157 LOG(QFATAL) << "We should never get here.";
158 break;
159 }
160 }
161 }
162 }
163
164 void Compute(OpKernelContext* const context) override {
165 DecisionTreeEnsembleResource* ensemble_resource;
166 // Gets the resource. Grabs the mutex but releases it.
167 OP_REQUIRES_OK(context, LookupResource(context, HandleFromInput(context, 0),
168 &ensemble_resource));
169 // Release the reference to the resource once we're done using it.
170 core::ScopedUnref unref_me(ensemble_resource);
171 if (use_locking_) {
172 tf_shared_lock l(*ensemble_resource->get_mutex());
173 DoCompute(context, ensemble_resource);
174 } else {
175 DoCompute(context, ensemble_resource);
176 }
177 }
178
179 private:
180 void DoCompute(OpKernelContext* context,
181 DecisionTreeEnsembleResource* ensemble_resource) {
182 // Read dense float features list;
183 OpInputList dense_float_features_list;
184 OP_REQUIRES_OK(context, TensorUtils::ReadDenseFloatFeatures(
185 context, &dense_float_features_list));
186
187 // Read sparse float features list;
188 OpInputList sparse_float_feature_indices_list;
189 OpInputList sparse_float_feature_values_list;
190 OpInputList sparse_float_feature_shapes_list;
191 OP_REQUIRES_OK(context, TensorUtils::ReadSparseFloatFeatures(
192 context, &sparse_float_feature_indices_list,
193 &sparse_float_feature_values_list,
194 &sparse_float_feature_shapes_list));
195
196 // Read sparse int features list;
197 OpInputList sparse_int_feature_indices_list;
198 OpInputList sparse_int_feature_values_list;
199 OpInputList sparse_int_feature_shapes_list;
200 OP_REQUIRES_OK(context, TensorUtils::ReadSparseIntFeatures(
201 context, &sparse_int_feature_indices_list,
202 &sparse_int_feature_values_list,
203 &sparse_int_feature_shapes_list));
204
205 // Infer batch size.
206 const int64 batch_size = TensorUtils::InferBatchSize(
207 dense_float_features_list, sparse_float_feature_shapes_list,
208 sparse_int_feature_shapes_list);
209
210 // Read batch features.
211 boosted_trees::utils::BatchFeatures batch_features(batch_size);
212 OP_REQUIRES_OK(
213 context,
214 batch_features.Initialize(
215 TensorUtils::OpInputListToTensorVec(dense_float_features_list),
216 TensorUtils::OpInputListToTensorVec(
217 sparse_float_feature_indices_list),
218 TensorUtils::OpInputListToTensorVec(
219 sparse_float_feature_values_list),
220 TensorUtils::OpInputListToTensorVec(
221 sparse_float_feature_shapes_list),
222 TensorUtils::OpInputListToTensorVec(
223 sparse_int_feature_indices_list),
224 TensorUtils::OpInputListToTensorVec(sparse_int_feature_values_list),
225 TensorUtils::OpInputListToTensorVec(
226 sparse_int_feature_shapes_list)));
227
228 std::vector<int32> dropped_trees;
229 std::vector<float> original_weights;
230
231 // Do dropout if needed.
232 if (apply_dropout_ && has_dropout_) {
233 // Read in seed and cast to uint64.
234 const Tensor* seed_t;
235 OP_REQUIRES_OK(context, context->input(kSeedTensorName, &seed_t));
236 OP_REQUIRES(context, TensorShapeUtils::IsScalar(seed_t->shape()),
237 errors::InvalidArgument("Seed must be a scalar."));
238 const uint64 seed = seed_t->scalar<int64>()();
239
240 std::unordered_set<int32> trees_not_to_drop;
241 if (center_bias_) {
242 trees_not_to_drop.insert(0);
243 }
244 if (ensemble_resource->decision_tree_ensemble().has_growing_metadata()) {
245 // We are in batch mode, the last tree is the tree that is being built,
246 // we can't drop it during dropout.
247 trees_not_to_drop.insert(ensemble_resource->num_trees() - 1);
248 }
249 const std::vector<float> weights = ensemble_resource->GetTreeWeights();
250 OP_REQUIRES_OK(context, DropoutUtils::DropOutTrees(
251 seed, dropout_config_, trees_not_to_drop,
252 weights, &dropped_trees, &original_weights));
253 }
254
255 // Prepare the list of trees to include in the prediction.
256 std::vector<int32> trees_to_include;
257 CalculateTreesToInclude(
258 ensemble_resource->decision_tree_ensemble(), dropped_trees,
259 ensemble_resource->decision_tree_ensemble().trees_size(),
260 only_finalized_trees_, center_bias_, &trees_to_include);
261
262 // Allocate output predictions matrix.
263 Tensor* output_predictions_t = nullptr;
264 OP_REQUIRES_OK(
265 context, context->allocate_output(kPredictionsTensorName,
266 {batch_size, prediction_vector_size_},
267 &output_predictions_t));
268 auto output_predictions = output_predictions_t->matrix<float>();
269
270 // Run predictor.
271 thread::ThreadPool* const worker_threads =
272 context->device()->tensorflow_cpu_worker_threads()->workers;
273
274 if (apply_averaging_) {
275 DecisionTreeEnsembleConfig adjusted =
276 ensemble_resource->decision_tree_ensemble();
277 const int start_averaging = std::max(
278 0.0,
279 averaging_config_.config_case() ==
280 AveragingConfig::kAverageLastNTreesFieldNumber
281 ? adjusted.trees_size() - averaging_config_.average_last_n_trees()
282 : adjusted.trees_size() *
283 (1.0 - averaging_config_.average_last_percent_trees()));
284 const int num_ensembles = adjusted.trees_size() - start_averaging;
285 for (int i = start_averaging; i < adjusted.trees_size(); ++i) {
286 float weight = adjusted.tree_weights(i);
287 adjusted.mutable_tree_weights()->Set(
288 i, weight * (num_ensembles - i + start_averaging) / num_ensembles);
289 }
290 MultipleAdditiveTrees::Predict(adjusted, trees_to_include, batch_features,
291 worker_threads, output_predictions);
292 } else {
293 MultipleAdditiveTrees::Predict(
294 ensemble_resource->decision_tree_ensemble(), trees_to_include,
295 batch_features, worker_threads, output_predictions);
296 }
297
298 // Output dropped trees and original weights.
299 Tensor* output_dropout_info_t = nullptr;
300 OP_REQUIRES_OK(context, context->allocate_output(
301 kDropoutInfoOutputTensorName,
302 {2, static_cast<int64>(dropped_trees.size())},
303 &output_dropout_info_t));
304 auto output_dropout_info = output_dropout_info_t->matrix<float>();
305
306 for (int32 i = 0; i < dropped_trees.size(); ++i) {
307 output_dropout_info(0, i) = dropped_trees[i];
308 output_dropout_info(1, i) = original_weights[i];
309 }
310 }
311
312 private:
313 LearningRateDropoutDrivenConfig dropout_config_;
314 AveragingConfig averaging_config_;
315 bool only_finalized_trees_;
316 int num_classes_;
317 // What is the size of the output vector for predictions?
318 int prediction_vector_size_;
319 bool apply_dropout_;
320 bool center_bias_;
321 bool apply_averaging_;
322 bool use_locking_;
323 bool has_dropout_;
324};
325
326REGISTER_KERNEL_BUILDER(Name("GradientTreesPrediction").Device(DEVICE_CPU),
327 GradientTreesPredictionOp);

329class GradientTreesPartitionExamplesOp : public OpKernel {
330 public:
331 explicit GradientTreesPartitionExamplesOp(OpKernelConstruction* const context)
332 : OpKernel(context) {
333 OP_REQUIRES_OK(context, context->GetAttr("use_locking", &use_locking_));
334 }
335
336 void Compute(OpKernelContext* const context) override {
337 DecisionTreeEnsembleResource* ensemble_resource;
338 // Gets the resource. Grabs the mutex but releases it.
339 OP_REQUIRES_OK(context, LookupResource(context, HandleFromInput(context, 0),
340 &ensemble_resource));
341 // Release the reference to the resource once we're done using it.
342 core::ScopedUnref unref_me(ensemble_resource);
343 if (use_locking_) {
344 tf_shared_lock l(*ensemble_resource->get_mutex());
345 DoCompute(context, ensemble_resource);
346 } else {
347 DoCompute(context, ensemble_resource);
348 }
349 }
350
351 private:
352 void DoCompute(OpKernelContext* context,
353 DecisionTreeEnsembleResource* ensemble_resource) {
354 // The last non-finalized tree in the ensemble is by convention the
355 // one to partition on. If no such tree exists, a nodeless tree is
356 // created.
357 boosted_trees::trees::DecisionTreeConfig empty_tree_config;
358 const boosted_trees::trees::DecisionTreeConfig& tree_config =
359 (ensemble_resource->num_trees() <= 0 ||
360 ensemble_resource->LastTreeMetadata()->is_finalized())
361 ? empty_tree_config
362 : *ensemble_resource->LastTree();
363
364 // Read dense float features list;
365 OpInputList dense_float_features_list;
366 OP_REQUIRES_OK(context, TensorUtils::ReadDenseFloatFeatures(
367 context, &dense_float_features_list));
368
369 // Read sparse float features list;
370 OpInputList sparse_float_feature_indices_list;
371 OpInputList sparse_float_feature_values_list;
372 OpInputList sparse_float_feature_shapes_list;
373 OP_REQUIRES_OK(context, TensorUtils::ReadSparseFloatFeatures(
374 context, &sparse_float_feature_indices_list,
375 &sparse_float_feature_values_list,
376 &sparse_float_feature_shapes_list));
377
378 // Read sparse int features list;
379 OpInputList sparse_int_feature_indices_list;
380 OpInputList sparse_int_feature_values_list;
381 OpInputList sparse_int_feature_shapes_list;
382 OP_REQUIRES_OK(context, TensorUtils::ReadSparseIntFeatures(
383 context, &sparse_int_feature_indices_list,
384 &sparse_int_feature_values_list,
385 &sparse_int_feature_shapes_list));
386
387 // Infer batch size.
388 const int64 batch_size = TensorUtils::InferBatchSize(
389 dense_float_features_list, sparse_float_feature_shapes_list,
390 sparse_int_feature_shapes_list);
391
392 // Read batch features.
393 boosted_trees::utils::BatchFeatures batch_features(batch_size);
394 OP_REQUIRES_OK(
395 context,
396 batch_features.Initialize(
397 TensorUtils::OpInputListToTensorVec(dense_float_features_list),
398 TensorUtils::OpInputListToTensorVec(
399 sparse_float_feature_indices_list),
400 TensorUtils::OpInputListToTensorVec(
401 sparse_float_feature_values_list),
402 TensorUtils::OpInputListToTensorVec(
403 sparse_float_feature_shapes_list),
404 TensorUtils::OpInputListToTensorVec(
405 sparse_int_feature_indices_list),
406 TensorUtils::OpInputListToTensorVec(sparse_int_feature_values_list),
407 TensorUtils::OpInputListToTensorVec(
408 sparse_int_feature_shapes_list)));
409
410 // Allocate output partitions vector.
411 Tensor* partition_ids_t = nullptr;
412 OP_REQUIRES_OK(context,
413 context->allocate_output(0, {batch_size}, &partition_ids_t));
414 thread::ThreadPool* const worker_threads =
415 context->device()->tensorflow_cpu_worker_threads()->workers;
416 learner::ExamplePartitioner::PartitionExamples(
417 tree_config, batch_features, worker_threads->NumThreads(),
418 worker_threads, partition_ids_t->vec<int32>().data());
419 }
420
421 private:
422 bool use_locking_;
423};
424
425REGISTER_KERNEL_BUILDER(
426 Name("GradientTreesPartitionExamples").Device(DEVICE_CPU),
427 GradientTreesPartitionExamplesOp);
