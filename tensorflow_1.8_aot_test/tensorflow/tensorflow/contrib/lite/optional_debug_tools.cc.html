19void PrintIntVector(const std::vector<int>& v) {
20 for (const auto& it : v) {
21 printf(" %d", it);
22 }
23 printf("\n");
24}
25
26void PrintTfLiteIntVector(const TfLiteIntArray* v) {
27 if (!v) {
28 printf(" (null)");
29 return;
30 }
31 for (int k = 0; k < v->size; k++) {
32 printf(" %d", v->data[k]);
33 }
34 printf("\n");
35}

37const char* TensorTypeName(TfLiteType type) {
38 switch (type) {
39 case kTfLiteNoType:
40 return "kTfLiteNoType";
41 case kTfLiteFloat32:
42 return "kTfLiteFloat32";
43 case kTfLiteInt32:
44 return "kTfLiteInt32";
45 case kTfLiteUInt8:
46 return "kTfLiteUInt8";
47 case kTfLiteInt64:
48 return "kTfLiteInt64";
49 case kTfLiteString:
50 return "kTfLiteString";
51 }
52 return "(invalid)";
53}
54
55const char* AllocTypeName(TfLiteAllocationType type) {
56 switch (type) {
57 case kTfLiteMemNone:
58 return "kTfLiteMemNone";
59 case kTfLiteMmapRo:
60 return "kTfLiteMmapRo";
61 case kTfLiteDynamic:
62 return "kTfLiteDynamic";
63 case kTfLiteArenaRw:
64 return "kTfLiteArenaRw";
65 case kTfLiteArenaRwPersistent:
66 return "kTfLiteArenaRwPersistent";
67 }
68 return "(invalid)";
69}

71// Prints a dump of what tensors and what nodes are in the interpreter.
72void PrintInterpreterState(Interpreter* interpreter) {
73 printf("Interpreter has %d tensors and %d nodes\n",
74 interpreter->tensors_size(), interpreter->nodes_size());
75 printf("Inputs:");
76 PrintIntVector(interpreter->inputs());
77 printf("Outputs:");
78 PrintIntVector(interpreter->outputs());
79 printf("\n");
80 for (int tensor_index = 0; tensor_index < interpreter->tensors_size();
81 tensor_index++) {
82 TfLiteTensor* tensor = interpreter->tensor(tensor_index);
83 printf("Tensor %3d %10s %15s %10zu bytes (%4.1f MB) ", tensor_index,
84 TensorTypeName(tensor->type), AllocTypeName(tensor->allocation_type),
85 tensor->bytes, float(tensor->bytes) / float(1 << 20));
86 PrintTfLiteIntVector(tensor->dims);
87 printf("\n");
88 }
89
90 for (int node_index = 0; node_index < interpreter->nodes_size();
91 node_index++) {
92 const std::pair<TfLiteNode, TfLiteRegistration>* node_and_reg =
93 interpreter->node_and_registration(node_index);
94 const TfLiteNode& node = node_and_reg->first;
95 const TfLiteRegistration& reg = node_and_reg->second;
96 printf("Node %3d Operator Builtin Code %3d\n", node_index,
97 reg.builtin_code);
98 printf(" Inputs:");
99 PrintTfLiteIntVector(node.inputs);
100 printf(" Outputs:");
101 PrintTfLiteIntVector(node.outputs);
102 }
103}
