35// Since std::vector<int32> is in the std namespace, and we are not allowed
36// to add ParseFlag/UnparseFlag to std, we introduce a simple wrapper type
37// to use as the flag type:
38struct IntList {
39 std::vector<int32> elements;
40};
41struct StringMapList {
42 std::vector<std::unordered_map<string, string>> elements;
43};

45// command_line_flags.h don't track whether or not a flag is specified. Arg
46// contains the value (which will be default if not specified) and also
47// whether the flag is specified.
48// TODO(aselle): consider putting doc string and ability to construct the
49// tensorflow argument into this, so declaration of parameters can be less
50// distributed.
51// Every template specialization of Arg is required to implement
52// default_value(), specified(), value(), parse(), bind().
53template <class T>
54class Arg final {
55 public:
56 explicit Arg(T default_ = T()) : value_(default_) {}
57 virtual ~Arg() {}
58
59 // Provide default_value() to arg list
60 T default_value() const { return value_; }
61 // Return true if the command line argument was specified on the command line.
62 bool specified() const { return specified_; }
63 // Const reference to parsed value.
64 const T& value() const { return value_; }
65
66 // Parsing callback for the tensorflow::Flags code
67 bool parse(T value_in) {
68 value_ = value_in;
69 specified_ = true;
70 return true;
71 }
72
73 // Bind the parse member function so tensorflow::Flags can call it.
74 std::function<bool(T)> bind() {
75 return std::bind(&Arg::parse, this, std::placeholders::_1);
76 }
77
78 private:
79 // Becomes true after parsing if the value was specified
80 bool specified_ = false;
81 // Value of the argument (initialized to the default in the constructor).
82 T value_;
83};
84
85template <>
86class Arg<toco::IntList> final {
87 public:
88 // Provide default_value() to arg list
89 string default_value() const { return ""; }
90 // Return true if the command line argument was specified on the command line.
91 bool specified() const { return specified_; }
92 // Bind the parse member function so tensorflow::Flags can call it.
93 bool parse(string text) {
94 parsed_value_.elements.clear();
95 specified_ = true;
96 // strings::Split("") produces {""}, but we need {} on empty input.
97 // TODO(aselle): Moved this from elsewhere, but ahentz recommends we could
98 // use absl::SplitLeadingDec32Values(text.c_str(), &parsed_values_.elements)
99 if (!text.empty()) {
100 int32 element;
101 for (absl::string_view part : absl::StrSplit(text, ',')) {
102 if (!SimpleAtoi(part, &element)) return false;
103 parsed_value_.elements.push_back(element);
104 }
105 }
106 return true;
107 }
108
109 std::function<bool(string)> bind() {
110 return std::bind(&Arg::parse, this, std::placeholders::_1);
111 }
112
113 const toco::IntList& value() const { return parsed_value_; }
114
115 private:
116 toco::IntList parsed_value_;
117 bool specified_ = false;
118};
119
120template <>
121class Arg<toco::StringMapList> final {
122 public:
123 // Provide default_value() to StringMapList
124 string default_value() const { return ""; }
125 // Return true if the command line argument was specified on the command line.
126 bool specified() const { return specified_; }
127 // Bind the parse member function so tensorflow::Flags can call it.
128
129 bool parse(string text) {
130 parsed_value_.elements.clear();
131 specified_ = true;
132
133 if (text.empty()) {
134 return true;
135 }
136
137#if defined(PLATFORM_GOOGLE)
138 std::vector<absl::string_view> outer_vector;
139 absl::string_view text_disposable_copy = text;
140 SplitStructuredLine(text_disposable_copy, ',', "{}", &outer_vector);
141 for (const absl::string_view& outer_member_stringpiece : outer_vector) {
142 string outer_member(outer_member_stringpiece);
143 if (outer_member.empty()) {
144 continue;
145 }
146 string outer_member_copy = outer_member;
147 absl::StripAsciiWhitespace(&outer_member);
148 if (!TryStripPrefixString(outer_member, "{", &outer_member)) return false;
149 if (!TryStripSuffixString(outer_member, "}", &outer_member)) return false;
150 const std::vector<string> inner_fields_vector =
151 absl::StrSplit(outer_member, ',');
152
153 std::unordered_map<string, string> element;
154 for (const string& member_field : inner_fields_vector) {
155 std::vector<string> outer_member_key_value =
156 absl::StrSplit(member_field, ':');
157 if (outer_member_key_value.size() != 2) return false;
158 string& key = outer_member_key_value[0];
159 string& value = outer_member_key_value[1];
160 absl::StripAsciiWhitespace(&key);
161 absl::StripAsciiWhitespace(&value);
162 if (element.count(key) != 0) return false;
163 element[key] = value;
164 }
165 parsed_value_.elements.push_back(element);
166 }
167 return true;
168#else
169 // TODO(aselle): Fix argument parsing when absl supports structuredline
170 fprintf(stderr, "%s:%d StringMapList arguments not supported\n", __FILE__,
171 __LINE__);
172 abort();
173#endif
174 }
175
176 std::function<bool(string)> bind() {
177 return std::bind(&Arg::parse, this, std::placeholders::_1);
178 }
179
180 const toco::StringMapList& value() const { return parsed_value_; }
181
182 private:
183 toco::StringMapList parsed_value_;
184 bool specified_ = false;
185};
