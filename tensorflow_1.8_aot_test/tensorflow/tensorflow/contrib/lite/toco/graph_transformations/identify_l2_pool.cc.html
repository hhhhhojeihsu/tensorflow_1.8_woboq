29std::vector<std::unique_ptr<Operator>>::iterator FindOperator(
30 Model* model, const Operator* op) {
31 auto it = model->operators.begin();
32 for (; it != model->operators.end(); ++it) {
33 if (it->get() == op) {
34 break;
35 }
36 }
37 return it;
38}
39} // namespace

41bool IdentifyL2Pool::Run(Model* model, std::size_t op_index) {
42 const auto sqrt_it = model->operators.begin() + op_index;
43 const auto* sqrt_op = sqrt_it->get();
44 if (sqrt_op->type != OperatorType::kTensorFlowSqrt) {
45 return false;
46 }
47
48 CHECK_EQ(sqrt_op->inputs.size(), 1);
49 CHECK_EQ(sqrt_op->outputs.size(), 1);
50
51 const AveragePoolOperator* avpool_op;
52 const Operator* square_op;
53
54 Operator* prev_to_sqrt_op = GetOpWithOutput(*model, sqrt_op->inputs[0]);
55 if (prev_to_sqrt_op->type != OperatorType::kAveragePool) {
56 AddMessageF(
57 "Giving up trying to identify L2Pool subgraph: "
58 "expected AveragePool op, got %s",
59 LogName(*prev_to_sqrt_op));
60 return false;
61 }
62
63 avpool_op = static_cast<const AveragePoolOperator*>(prev_to_sqrt_op);
64 CHECK_EQ(avpool_op->inputs.size(), 1);
65
66 square_op = GetOpWithOutput(*model, avpool_op->inputs[0]);
67 CHECK_EQ(square_op->inputs.size(), 1);
68 if (square_op->type != OperatorType::kTensorFlowSquare) {
69 AddMessageF(
70 "Giving up trying to identify L2Pool subgraph: "
71 "expected Square op, got %s",
72 LogName(*square_op));
73 return false;
74 }
75
76 // Create and emplace L2Pool node.
77 auto* l2pool_op = new L2PoolOperator;
78
79 l2pool_op->inputs = {square_op->inputs[0]};
80 l2pool_op->outputs = sqrt_op->outputs;
81
82 l2pool_op->padding.type = avpool_op->padding.type;
83 // Note that we do not setup avpool_op->padding.fixed here. This is done by
84 // the PropagateFixedSizes graph transformation.
85
86 l2pool_op->stride_height = avpool_op->stride_height;
87 l2pool_op->stride_width = avpool_op->stride_width;
88 l2pool_op->kheight = avpool_op->kheight;
89 l2pool_op->kwidth = avpool_op->kwidth;
90 model->operators.emplace(sqrt_it, l2pool_op);
91
92 AddMessageF("Creating %s replacing equivalent subgraph", LogName(*l2pool_op));
93
94 // Erase intermediate arrays, keeping input to square op.
95 model->EraseArray(avpool_op->inputs[0]);
96 model->EraseArray(sqrt_op->inputs[0]);
97
98 // Erase three operators being replaced.
99 model->operators.erase(FindOperator(model, square_op));
100 model->operators.erase(FindOperator(model, avpool_op));
101 model->operators.erase(FindOperator(model, sqrt_op));
102
103 return true;
104}
