28bool RemoveFinalDequantizeOp::Run(Model* model, std::size_t op_index) {
29 const auto dequantize_it = model->operators.begin() + op_index;
30 const auto* dequantize_op = dequantize_it->get();
31 if (dequantize_op->type != OperatorType::kDequantize) {
32 return false;
33 }
34 const auto& output = dequantize_op->outputs[0];
35 // We can remove any dequantize op whose output is not consumed by
36 // any op. This is not necessarily equivalent to the output being
37 // one of the model's output arrays, as some intermediate array
38 // in the middle of the graph might be designated as an output
39 // array.
40 if (CountOpsWithInput(*model, output)) {
41 return false;
42 }
43
44 // If one of the model's output arrays was actually the Dequantize op's
45 // output, then we need to update it to point to the Dequantize op's input.
46 for (int i = 0; i < model->flags.output_arrays_size(); i++) {
47 if (output == model->flags.output_arrays(i)) {
48 model->flags.set_output_arrays(i, dequantize_op->inputs[0]);
49 }
50 }
51
52 // Remove the node and its output array.
53 AddMessageF("Removed final %s", LogName(*dequantize_op));
54 model->EraseArray(output);
55 model->operators.erase(dequantize_it);
56 return true;
57}
