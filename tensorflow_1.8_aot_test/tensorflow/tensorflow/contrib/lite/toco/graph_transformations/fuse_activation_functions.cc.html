28bool FuseActivationFunctions::Run(Model* model, std::size_t op_index) {
29 const auto ac_it = model->operators.begin() + op_index;
30 const auto* ac_op = ac_it->get();
31
32 if (ac_op->type != OperatorType::kRelu6 &&
33 ac_op->type != OperatorType::kRelu1 &&
34 ac_op->type != OperatorType::kRelu) {
35 return false;
36 }
37
38 // Find the op producing the array passed to this activation function
39 Operator* op = GetOpWithOutput(*model, ac_op->inputs[0]);
40
41 if (!op) return false;
42
43 if (CountTrueOutputs(*model, *op) > 1) {
44 AddMessageF(
45 "Not fusing activation function %s into %s because it has more than "
46 "one consumed output",
47 LogName(*ac_op), LogName(*op));
48 return false;
49 }
50
51 CHECK_EQ(op->outputs[0], ac_op->inputs[0]);
52
53 int count_ops_consuming_output = CountOpsWithInput(*model, ac_op->inputs[0]);
54 DCHECK_GE(count_ops_consuming_output, 1);
55 if (count_ops_consuming_output > 1) {
56 AddMessageF(
57 "Not fusing activation function into %s because it is consumed by more "
58 "than 1 other operator",
59 LogName(*ac_op), LogName(*op));
60 return false;
61 }
62
63 if (!IsDiscardableArray(*model, op->outputs[0])) {
64 AddMessageF(
65 "Not fusing activation function %s into %s because output %s it is not "
66 "discardable",
67 LogName(*ac_op), LogName(*op), op->outputs[0]);
68 return false;
69 }
70
71 if (op->fused_activation_function != FusedActivationFunctionType::kNone) {
72 AddMessageF(
73 "Not fusing activation function %s into %s because it already has a "
74 "fused activation function",
75 LogName(*ac_op), LogName(*op));
76 return false;
77 }
78
79 if (!OperatorSupportsFusedActivation(op->type)) {
80 AddMessageF(
81 "Not fusing activation function %s because the %s op doesn't support "
82 "it",
83 LogName(*ac_op), LogName(*op));
84 return false;
85 }
86
87 AddMessageF("Fusing activation function %s into the preceding %s",
88 LogName(*ac_op), LogName(*op));
89 if (ac_op->type == OperatorType::kRelu6) {
90 op->fused_activation_function = FusedActivationFunctionType::kRelu6;
91 } else if (ac_op->type == OperatorType::kRelu1) {
92 op->fused_activation_function = FusedActivationFunctionType::kRelu1;
93 } else if (ac_op->type == OperatorType::kRelu) {
94 op->fused_activation_function = FusedActivationFunctionType::kRelu;
95 } else {
96 LOG(FATAL) << "Unhandled activation function type";
97 }
98 model->EraseArray(ac_op->inputs[0]);
99 op->outputs[0] = ac_op->outputs[0];
100 model->operators.erase(ac_it);
101 return true;
102}
