27std::vector<std::unique_ptr<Operator>>::iterator FindOperator(
28 Model* model, const Operator& op) {
29 auto it = model->operators.begin();
30 for (; it != model->operators.end(); ++it) {
31 if (it->get() == &op) {
32 break;
33 }
34 }
35 return it;
36}

38bool GetStateArrayForBackEdge(const Model& model,
39 const string& back_edge_source_array,
40 string* state_array = nullptr) {
41 for (const auto& rnn_state : model.flags.rnn_states()) {
42 if (back_edge_source_array == rnn_state.back_edge_source_array()) {
43 // Found LSTM cell output
44 if (state_array) {
45 *state_array = rnn_state.state_array();
46 }
47 return true;
48 }
49 }
50 return false;
51}

53// Returns true if the given operator has exactly 1 input, and is connected to
54// the given op_type.
55// We use kNone to indicate an input unattached to an operator output. Usually
56// these are the static input arrays.
57bool MatchOperatorInputs(const Operator& op, const Model& model,
58 OperatorType op_type, Operator** connected_op) {
59 // Check for required number of inputs
60 if (op.inputs.size() != 1) {
61 return false;
62 }
63
64 // Check if first input is disconnected/connected to an operator
65 Operator* x = GetOpWithOutput(model, op.inputs[0]);
66 if ((op_type == OperatorType::kNone) && (x != nullptr)) {
67 return false;
68 }
69 if ((op_type != OperatorType::kNone) && (x == nullptr)) {
70 return false;
71 }
72
73 // Check that first operator, if connected, is of correct type
74 if ((x != nullptr) && (x->type != op_type)) {
75 return false;
76 }
77
78 // Successfully matched. Optionally return matching input operators.
79 if (connected_op) {
80 *connected_op = x;
81 }
82
83 return true;
84}

86// Returns true if the given operator has exactly 2 inputs, which are connected
87// to the given op_types.
88// We use kNone to indicate an input unattached to an operator output. Usually
89// these are the static input arrays.
90bool MatchOperatorInputs(const Operator& op, const Model& model,
91 OperatorType a_op_type, Operator** a_op,
92 OperatorType b_op_type, Operator** b_op) {
93 // Check for required number of inputs
94 if (op.inputs.size() != 2) {
95 return false;
96 }
97
98 // Check if first input is disconnected/connected to an operator
99 Operator* x = GetOpWithOutput(model, op.inputs[0]);
100 if ((a_op_type == OperatorType::kNone) && (x != nullptr)) {
101 return false;
102 }
103 if ((a_op_type != OperatorType::kNone) && (x == nullptr)) {
104 return false;
105 }
106
107 // Check that first operator, if connected, is of correct type
108 if ((x != nullptr) && (x->type != a_op_type)) {
109 return false;
110 }
111
112 // Check if second input is disconnected/connected to an operator
113 Operator* y = GetOpWithOutput(model, op.inputs[1]);
114 if ((b_op_type == OperatorType::kNone) && (y != nullptr)) {
115 return false;
116 }
117 if ((b_op_type != OperatorType::kNone) && (y == nullptr)) {
118 return false;
119 }
120
121 // Check that second operator, if connected, is of correct type
122 if ((y != nullptr) && (y->type != b_op_type)) {
123 return false;
124 }
125
126 // Successfully matched. Optionally return matching input operators.
127 if (a_op != nullptr) {
128 *a_op = x;
129 }
130 if (b_op != nullptr) {
131 *b_op = y;
132 }
133 return true;
134}

136// Returns true if the given operator has exactly 3 inputs, which are connected
137// to the given op_types.
138// We use kNone to indicate an input unattached to an operator output. Usually
139// these are the static input arrays.

190

192 if (a_op != nullptr) {

194 }
195 if (b_op != nullptr) {
196 *b_op = y;
197 }
198 if (c_op != nullptr) {
199 *c_op = z;
200 }

206bool IdentifyLstmCell::Run(Model* model, std::size_t op_index) {
207 // This LSTM cell identification method is not invariant to commutation of
208 // commutative operator inputs. For example, if input[0] and input[1] of the
209 // final output multiplication were swapped, this method would not identify it
210 // as an LSTM cell. This is OK in most cases, because
211 // tf.rnn.contrib.BasicLSTMCell always generates LSTM cells the same way.
212
213 // Final output multiply
214 auto op_it = model->operators.begin() + op_index;
215 Operator* final_output_mul = op_it->get();
216 if (final_output_mul->type != OperatorType::kMul) {
217 return false;
218 }
219 Operator *state_output_tanh, *fc_output_sig;
220 if (!MatchOperatorInputs(*final_output_mul, *model, OperatorType::kTanh,
221 &state_output_tanh, OperatorType::kLogistic,
222 &fc_output_sig)) {
223 return false;
224 }
225
226 // State output TanH
227 // (We don't count an operator as ID'd until we verify it has the correct
228 // operator types feeding into it.)
229 Operator* state_combine_add;
230 if (!MatchOperatorInputs(*state_output_tanh, *model, OperatorType::kAdd,
231 &state_combine_add)) {
232 return false;
233 }
234 string prev_state;
235 if (!GetStateArrayForBackEdge(*model, state_output_tanh->inputs[0],
236 &prev_state)) {
237 return false;
238 }
239
240 // State forget & remember addition
241 Operator *state_forget_mul, *state_remember_mul;
242 if (!MatchOperatorInputs(*state_combine_add, *model, OperatorType::kMul,
243 &state_forget_mul, OperatorType::kMul,
244 &state_remember_mul)) {
245 return false;
246 }
247 if (state_forget_mul->inputs[0] != prev_state) {
248 return false;
249 }
250
251 // State forget gate
252 Operator* state_forget_sig;
253 if (!MatchOperatorInputs(*state_forget_mul, *model, OperatorType::kNone,
254 nullptr, OperatorType::kLogistic,
255 &state_forget_sig)) {
256 return false;
257 }
258
259 // State remember gate
260 Operator *state_remember_sig, *state_info_tanh;
261 if (!MatchOperatorInputs(*state_remember_mul, *model, OperatorType::kLogistic,
262 &state_remember_sig, OperatorType::kTanh,
263 &state_info_tanh)) {
264 return false;
265 }
266
267 // State remember "information" activation function
268 Operator* fc_output_split;
269 if (!MatchOperatorInputs(*state_info_tanh, *model,
270 OperatorType::kTensorFlowSplit, &fc_output_split)) {
271 return false;
272 }
273 // State remember gate activation function
274 Operator* tmp;
275 if (!MatchOperatorInputs(*state_remember_sig, *model,
276 OperatorType::kTensorFlowSplit, &tmp) ||
277 (tmp != fc_output_split)) {
278 return false;
279 }
280 // State forget gate activation function
281 if (!MatchOperatorInputs(*state_forget_sig, *model,
282 OperatorType::kTensorFlowSplit, &tmp) ||
283 (tmp != fc_output_split)) {
284 return false;
285 }
286 // Fully connected output activation function
287 if (!MatchOperatorInputs(*fc_output_sig, *model,
288 OperatorType::kTensorFlowSplit, &tmp) ||
289 (tmp != fc_output_split)) {
290 return false;
291 }
292 // Fully connected output split
293 Operator* fully_connected;
294 if (!MatchOperatorInputs(*fc_output_split, *model, OperatorType::kNone,
295 nullptr, OperatorType::kFullyConnected,
296 &fully_connected)) {
297 return false;
298 }
299
300 // Fully connected op
301 Operator* concat_inputs;
302 if (!MatchOperatorInputs(*fully_connected, *model,
303 OperatorType::kConcatenation, &concat_inputs,
304 OperatorType::kNone, nullptr, OperatorType::kNone,
305 nullptr)) {
306 return false;
307 }
308
309 // Emplace a new LSTM cell operator
310 auto* lstm_cell_op = new LstmCellOperator;
311 lstm_cell_op->inputs.resize(LstmCellOperator::NUM_INPUTS);
312 lstm_cell_op->inputs[LstmCellOperator::DATA_INPUT] = concat_inputs->inputs[0];
313 lstm_cell_op->inputs[LstmCellOperator::PREV_ACTIV_INPUT] =
314 concat_inputs->inputs[1];
315 lstm_cell_op->inputs[LstmCellOperator::WEIGHTS_INPUT] =
316 fully_connected->inputs[1];
317 lstm_cell_op->inputs[LstmCellOperator::BIASES_INPUT] =
318 fully_connected->inputs[2];
319 lstm_cell_op->inputs[LstmCellOperator::PREV_STATE_INPUT] = prev_state;
320 lstm_cell_op->outputs.resize(LstmCellOperator::NUM_OUTPUTS);
321 lstm_cell_op->outputs[LstmCellOperator::STATE_OUTPUT] =
322 state_output_tanh->inputs[0];
323 lstm_cell_op->outputs[LstmCellOperator::ACTIV_OUTPUT] =
324 final_output_mul->outputs[0];
325 model->operators.emplace(op_it, lstm_cell_op);
326 AddMessageF("Creating %s replacing equivalent subgraph",
327 LogName(*lstm_cell_op));
328
329 // Create temp arrays used internally during runtime.
330 const string base_name(FindLongestCommonPrefix(
331 lstm_cell_op->outputs[LstmCellOperator::STATE_OUTPUT],
332 lstm_cell_op->outputs[LstmCellOperator::ACTIV_OUTPUT]));
333 const string& concat_temp_array_name =
334 AvailableArrayName(*model, base_name + "concat_temp");
335 model->GetOrCreateArray(concat_temp_array_name);
336 lstm_cell_op->outputs[LstmCellOperator::CONCAT_TEMP] = concat_temp_array_name;
337 const string& activ_temp_array_name =
338 AvailableArrayName(*model, base_name + "activ_temp");
339 model->GetOrCreateArray(activ_temp_array_name);
340 lstm_cell_op->outputs[LstmCellOperator::ACTIV_TEMP] = activ_temp_array_name;
341 AddMessageF("Created temp outputs %s and %s on operator %s",
342 concat_temp_array_name, activ_temp_array_name,
343 LogName(*lstm_cell_op));
344
345 // Delete arrays and operators replaced by the LSTM cell operator. Order is
346 // important - DeleteArrayIfUnused() only succeeds if dependent operators
347 // have been removed first. Start at the output and work towards the input.
348 model->operators.erase(FindOperator(model, *final_output_mul));
349 DeleteArrayIfUnused(state_output_tanh->outputs[0], model);
350 DeleteArrayIfUnused(fc_output_sig->outputs[0], model);
351 model->operators.erase(FindOperator(model, *state_output_tanh));
352 model->operators.erase(FindOperator(model, *fc_output_sig));
353 model->operators.erase(FindOperator(model, *state_combine_add));
354 DeleteArrayIfUnused(state_forget_mul->outputs[0], model);
355 DeleteArrayIfUnused(state_remember_mul->outputs[0], model);
356 model->operators.erase(FindOperator(model, *state_forget_mul));
357 model->operators.erase(FindOperator(model, *state_remember_mul));
358 DeleteArrayIfUnused(state_forget_sig->outputs[0], model);
359 DeleteArrayIfUnused(state_info_tanh->outputs[0], model);
360 DeleteArrayIfUnused(state_remember_sig->outputs[0], model);
361 model->operators.erase(FindOperator(model, *state_forget_sig));
362 model->operators.erase(FindOperator(model, *state_info_tanh));
363 model->operators.erase(FindOperator(model, *state_remember_sig));
364 DeleteArrayIfUnused(fc_output_split->outputs[0], model);
365 DeleteArrayIfUnused(fc_output_split->outputs[1], model);
366 DeleteArrayIfUnused(fc_output_split->outputs[2], model);
367 DeleteArrayIfUnused(fc_output_split->outputs[3], model);
368 string dims_array = fc_output_split->inputs[0];
369 model->operators.erase(FindOperator(model, *fc_output_split));
370 DeleteArrayIfUnused(dims_array, model);
371 DeleteArrayIfUnused(fully_connected->outputs[0], model);
372 model->operators.erase(FindOperator(model, *fully_connected));
373 DeleteArrayIfUnused(concat_inputs->outputs[0], model);
374 model->operators.erase(FindOperator(model, *concat_inputs));
375 return true;
376}
