27// Unrolls a BatchMatMul on the batch dimension.
28// We need to slice each batch out of the inputs, matmul them individually, then
29// stack them all back together at the end.
30//
31// This transform effectively looks like:
32// result_slices = []
33// for bat in B:
34// slice_a = tf.reshape(tf.slice(a, [bat, 0, 0], [1, M, N]), [M, N])
35// slice_b = tf.reshape(tf.slice(b, [bat, 0, 0], [1, M, N]), [M, N])
36// slice_c = tf.matmul(slice_a, slice_b)
37// result_slices[bat] = slice_c
38// result = tf.stack(result_slices)
39bool UnrollBatchMatMul::Run(Model* model, std::size_t op_index) {
40 auto batch_op_it = model->operators.begin() + op_index;
41 if (batch_op_it->get()->type != OperatorType::kBatchMatMul) {
42 return false;
43 }
44 const auto* batch_op =
45 static_cast<const BatchMatMulOperator*>(batch_op_it->get());
46
47 // We must have the shape of at least one input to know our batch size.
48 const auto& input_array_a = model->GetArray(batch_op->inputs[0]);
49 const auto& input_array_b = model->GetArray(batch_op->inputs[1]);
50 if (!input_array_a.has_shape() || !input_array_b.has_shape()) return false;
51
52 // We only support the rank 3 case. If you are batching on rank > 3 you'll
53 // have to figure that out.
54 CHECK_EQ(input_array_a.shape().dimensions_count(),
55 input_array_b.shape().dimensions_count())
56 << "Input dimensions must have the same rank";
57 if (input_array_a.shape().dimensions_count() == 2) {
58 // This is really just a MatMul. This likely means that someone hand-crafted
59 // a graphdef with a BatchMatMul when they really wanted a MatMul.
60 AddMessageF("Replacing non-batch BatchMatMul %s by a MatMul operator",
61 LogName(*batch_op));
62 auto* matmul_op = new TensorFlowMatMulOperator;
63 matmul_op->inputs = batch_op->inputs;
64 matmul_op->outputs = batch_op->outputs;
65 const auto matmul_op_it = model->operators.emplace(batch_op_it, matmul_op);
66 batch_op_it = matmul_op_it + 1;
67 CHECK_EQ(batch_op_it->get(), batch_op);
68 model->operators.erase(batch_op_it);
69 return true;
70 }
71 CHECK_EQ(input_array_a.shape().dimensions_count(), 3)
72 << "Input arrays must have rank 3";
73
74 // Perform the matmul for each slice of the batch.
75 int batch_count = input_array_a.shape().dims(0);
76 AddMessageF("Unrolling BatchMatMul %s %d times", LogName(*batch_op),
77 batch_count);
78 auto tail_it = batch_op_it;
79 std::vector<string> stack_inputs;
80 for (int batch = 0; batch < batch_count; ++batch) {
81 std::string batch_name =
82 std::string(batch_op->outputs[0]) + "_b" + std::to_string(batch);
83
84 // tf.slice(a, ...).
85 auto* slice_a_op = new SliceOperator;
86 slice_a_op->inputs = {
87 batch_op->inputs[0],
88 CreateInt32Array(model, batch_name + "/slice_a/slice/begin",
89 {batch, 0, 0}),
90 CreateInt32Array(
91 model, batch_name + "/slice_a/slice/size",
92 {1, input_array_a.shape().dims(1), input_array_a.shape().dims(2)}),
93 };
94 slice_a_op->outputs = {AvailableArrayName(*model, batch_name + "/slice_a")};
95 auto& slice_a_op_output = model->GetOrCreateArray(slice_a_op->outputs[0]);
96 slice_a_op_output.data_type = input_array_a.data_type;
97 tail_it = model->operators.emplace(tail_it, slice_a_op) + 1;
98
99 // Reshape to remove the first dimension ([1,M,N] -> [M,N]).
100 auto* slice_a_reshape_op = new TensorFlowReshapeOperator;
101 slice_a_reshape_op->inputs = {
102 slice_a_op->outputs[0],
103 CreateInt32Array(model, batch_name + "/slice_a/reshape/shape",
104 {-1, input_array_a.shape().dims(2)})};
105 slice_a_reshape_op->outputs = {
106 AvailableArrayName(*model, batch_name + "/slice_a/reshape")};
107 auto& slice_a_reshape_op_output =
108 model->GetOrCreateArray(slice_a_reshape_op->outputs[0]);
109 slice_a_reshape_op_output.data_type = input_array_a.data_type;
110 tail_it = model->operators.emplace(tail_it, slice_a_reshape_op) + 1;
111
112 // tf.slice(b, ...).
113 auto* slice_b_op = new SliceOperator;
114 slice_b_op->inputs = {
115 batch_op->inputs[1],
116 CreateInt32Array(model, batch_name + "/slice_b/slice/begin", {0, 0, 0}),
117 CreateInt32Array(
118 model, batch_name + "/slice_b/slice/size",
119 {1, input_array_b.shape().dims(1), input_array_b.shape().dims(2)}),
120 };
121 slice_b_op->outputs = {AvailableArrayName(*model, batch_name + "/slice_b")};
122 auto& slice_b_op_output = model->GetOrCreateArray(slice_b_op->outputs[0]);
123 slice_b_op_output.data_type = input_array_b.data_type;
124 tail_it = model->operators.emplace(tail_it, slice_b_op) + 1;
125
126 // Reshape to remove the first dimension ([1,M,N] -> [M,N]).
127 auto* slice_b_reshape_op = new TensorFlowReshapeOperator;
128 slice_b_reshape_op->inputs = {
129 slice_b_op->outputs[0],
130 CreateInt32Array(model, batch_name + "/slice_b/reshape/shape",
131 {-1, input_array_b.shape().dims(2)})};
132 slice_b_reshape_op->outputs = {
133 AvailableArrayName(*model, batch_name + "/slice_b/reshape")};
134 auto& slice_b_reshape_op_output =
135 model->GetOrCreateArray(slice_b_reshape_op->outputs[0]);
136 slice_b_reshape_op_output.data_type = input_array_b.data_type;
137 tail_it = model->operators.emplace(tail_it, slice_b_reshape_op) + 1;
138
139 // tf.matmul(slice_a, slice_b).
140 auto* matmul_op = new TensorFlowMatMulOperator;
141 matmul_op->inputs = {slice_a_reshape_op->outputs[0],
142 slice_b_reshape_op->outputs[0]};
143 matmul_op->outputs = {AvailableArrayName(*model, batch_name)};
144 auto& matmul_op_output = model->GetOrCreateArray(matmul_op->outputs[0]);
145 matmul_op_output.data_type = input_array_a.data_type;
146 tail_it = model->operators.emplace(tail_it, matmul_op) + 1;
147
148 // Add to stack.
149 stack_inputs.push_back(matmul_op->outputs[0]);
150 }
151
152 // The stack that will join all the individual matmul results together.
153 auto* stack_op = new StackOperator;
154 stack_op->inputs = stack_inputs;
155 stack_op->outputs = {batch_op->outputs[0]};
156 stack_op->axis = 0;
157 model->operators.emplace(tail_it, stack_op);
158
159 // Remove the old batch matmul now that we've unrolled.
160 batch_op_it = model->operators.begin();
161 for (; batch_op_it != model->operators.end(); ++batch_op_it) {
162 if (batch_op_it->get() == batch_op) {
163 break;
164 }
165 }
166 CHECK(batch_op_it != model->operators.end());
167 CHECK(batch_op_it->get() == batch_op);
168 model->operators.erase(batch_op_it);
169 return true;
170}
