31bool CopyMinMaxFromFirstInput(const Operator& op, Model* model) {
32 auto& output_array = model->GetArray(op.outputs[0]);
33 if (output_array.minmax) {
34 return false;
35 }
36 const auto& input_array = model->GetArray(op.inputs[0]);
37 if (!input_array.minmax) {
38 return false;
39 }
40 const auto& input_minmax = input_array.GetMinMax();
41 CHECK(!output_array.minmax);
42 auto& output_minmax = output_array.GetOrCreateMinMax();
43 output_minmax.min = input_minmax.min;
44 output_minmax.max = input_minmax.max;
45 return true;
46}

48bool ResolveConstantUnaryOperator::Run(Model* model, std::size_t op_index) {
49 const auto unary_it = model->operators.begin() + op_index;
50 const auto* unary_op = unary_it->get();
51 // Test for unary ops of types that we know how to resolve.
52 switch (unary_op->type) {
53 case OperatorType::kCast:
54 case OperatorType::kLog:
55 case OperatorType::kNeg:
56 case OperatorType::kTensorFlowRsqrt:
57 case OperatorType::kTensorFlowSqrt:
58 case OperatorType::kTensorFlowSquare:
59 case OperatorType::kTensorFlowSum:
60 case OperatorType::kTensorFlowMin:
61 case OperatorType::kTensorFlowMax:
62 case OperatorType::kTensorFlowReshape:
63 case OperatorType::kRelu6:
64 case OperatorType::kRelu1:
65 case OperatorType::kRelu:
66 break;
67 default:
68 return false;
69 }
70
71 // Check if the input is a constant parameter.
72 if (!IsConstantParameterArray(*model, unary_op->inputs[0])) {
73 return false;
74 }
75
76 // if the unary op involves a tensor required by a rnn state, ignore it
77 for (const auto& rnn_state : model->flags.rnn_states()) {
78 if (unary_op->inputs[0] == rnn_state.back_edge_source_array()) {
79 return false;
80 }
81 if (unary_op->inputs[0] == rnn_state.state_array()) {
82 return false;
83 }
84 }
85
86 auto& output_array = model->GetArray(unary_op->outputs[0]);
87 if (!output_array.has_shape()) {
88 // Yield until the output array dims have been resolved.
89 return false;
90 }
91
92 // At the moment we don't want to care about fused activation functions.
93 // The idea is that we should do the present constants-propagation before
94 // activation functions get fused.
95 if (unary_op->fused_activation_function !=
96 FusedActivationFunctionType::kNone) {
97 AddMessageF(
98 "Not resolving constant %s "
99 " because it has a fused activation function",
100 LogName(*unary_op));
101 return false;
102 }
103
104 // The min-max is only copied for ops that copy data without arithmetic.
105 // In future trivial transpose, etc, can be handled here.
106 if (unary_op->type == OperatorType::kTensorFlowReshape) {
107 CopyMinMaxFromFirstInput(*unary_op, model);
108 }
109
110 const auto& input_array = model->GetArray(unary_op->inputs[0]);
111 // We have already tested above for existence of buffers (synonymous to being
112 // a constant param).
113 CHECK(input_array.buffer);
114 std::vector<DataType<ArrayDataType::kFloat>> const* input_float_data;
115 if (unary_op->type == OperatorType::kCast) {
116 CastOperator const* cast_op = static_cast<CastOperator const*>(unary_op);
117 if (cast_op->dst_data_type != ArrayDataType::kFloat) {
118 AddMessageF(
119 "Not resolving constant %s because we currently only support casting "
120 "to float",
121 LogName(*unary_op));
122 return false;
123 }
124 if (cast_op->src_data_type != input_array.buffer->type) {
125 AddMessageF(
126 "Not resolving constant %s because cast op source type does not "
127 "match input type",
128 LogName(*unary_op));
129 }
130 } else {
131 if (input_array.buffer->type != ArrayDataType::kFloat) {
132 return false;
133 }
134 input_float_data = &(input_array.GetBuffer<ArrayDataType::kFloat>().data);
135 }
136
137 // Create a float buffer on the output array, which are always constant.
138 const Shape& output_shape = output_array.shape();
139 const int output_dims_count = output_shape.dimensions_count();
140 const int output_buffer_size = RequiredBufferSizeForShape(output_shape);
141 auto& output_float_data =
142 output_array.GetMutableBuffer<ArrayDataType::kFloat>().data;
143 output_float_data.resize(output_buffer_size);
144
145 const Shape& input_shape = input_array.shape();
146 const int input_buffer_size = RequiredBufferSizeForShape(input_shape);
147 if (unary_op->type == OperatorType::kCast) {
148 for (int i = 0; i < output_buffer_size; i++) {
149 float outval = 0.0f;
150 if (input_array.buffer->type == ArrayDataType::kFloat) {
151 outval = static_cast<float>(
152 input_array.GetBuffer<ArrayDataType::kFloat>().data[i]);
153 } else if (input_array.buffer->type == ArrayDataType::kUint8) {
154 outval = static_cast<float>(
155 input_array.GetBuffer<ArrayDataType::kUint8>().data[i]);
156 } else if (input_array.buffer->type == ArrayDataType::kInt32) {
157 outval = static_cast<float>(
158 input_array.GetBuffer<ArrayDataType::kInt32>().data[i]);
159 } else if (input_array.buffer->type == ArrayDataType::kInt64) {
160 outval = static_cast<float>(
161 input_array.GetBuffer<ArrayDataType::kInt64>().data[i]);
162 } else {
163 LOG(FATAL) << "Unsupported cast op input type";
164 }
165 output_float_data[i] = outval;
166 }
167 } else if (unary_op->type == OperatorType::kTensorFlowReshape) {
168 CHECK(input_buffer_size == output_buffer_size);
169 output_float_data = *input_float_data;
170 } else if (unary_op->type == OperatorType::kTensorFlowSum) {
171 CHECK_EQ(unary_op->inputs.size(), 2) << "Sum needs 2 inputs";
172 if (!IsConstantParameterArray(*model, unary_op->inputs[1])) {
173 AddMessageF("Axis input is non-constant");
174 return false;
175 }
176 auto& axis_array = model->GetArray(unary_op->inputs[1]);
177 CHECK(axis_array.data_type == ArrayDataType::kInt32);
178 int axis = axis_array.GetBuffer<ArrayDataType::kInt32>().data[0];
179 CHECK_LT(axis, input_shape.dimensions_count()) << "Axis out of bounds";
180
181 // We currently only handle reduction on axis 0.
182 CHECK_EQ(axis, 0) << "Only reduction along axis 0 is supported";
183 // We currently only handle 1-D and 2-D input tensors.
184 CHECK_LE(input_shape.dimensions_count(), 2) << "Rank >2 not yet supported";
185 // We only support keep_dims=true; shape prop will need to change otherwise.
186 auto sum_op = static_cast<const TensorFlowSumOperator*>(unary_op);
187 CHECK(sum_op->keep_dims) << "Only keep_dims=true is supported";
188
189 std::vector<int> indices(input_shape.dimensions_count());
190 for (int i = 0; i < input_shape.dims(1); ++i) {
191 indices[1] = i;
192 float sum = 0.f;
193 for (int j = 0; j < input_shape.dims(0); ++j) {
194 indices[0] = j;
195 sum += (*input_float_data)[Offset(input_shape, indices)];
196 }
197 output_float_data[i] = sum;
198 }
199 } else if (unary_op->type == OperatorType::kTensorFlowMin) {
200 // At the moment only full reduction across all dimensions is supported.
201 // TODO(starka): Output should not be padded.
202 for (int i = 0; i < output_dims_count; i++) {
203 CHECK_EQ(output_shape.dims(i), 1);
204 }
205 float min = (*input_float_data)[0];
206 for (int i = 0; i < input_buffer_size; i++) {
207 min = std::min(min, (*input_float_data)[i]);
208 }
209 output_float_data[0] = min;
210 } else if (unary_op->type == OperatorType::kTensorFlowMax) {
211 // At the moment only full reduction across all dimensions is supported.
212 // TODO(starka): Output should not be padded.
213 for (int i = 0; i < output_dims_count; i++) {
214 CHECK_EQ(output_shape.dims(i), 1);
215 }
216 float max = (*input_float_data)[0];
217 for (int i = 0; i < input_buffer_size; i++) {
218 max = std::max(max, (*input_float_data)[i]);
219 }
220 output_float_data[0] = max;
221 } else if (unary_op->type == OperatorType::kNeg ||
222 unary_op->type == OperatorType::kLog ||
223 unary_op->type == OperatorType::kTensorFlowRsqrt ||
224 unary_op->type == OperatorType::kTensorFlowSqrt ||
225 unary_op->type == OperatorType::kTensorFlowSquare) {
226 // Element-wise ops. Should have perfectly matching sizes here.
227 for (int i = 0; i < output_dims_count; i++) {
228 CHECK_EQ(output_shape.dims(i), input_shape.dims(i));
229 }
230
231 for (int i = 0; i < output_buffer_size; i++) {
232 const float val = (*input_float_data)[i];
233 float outval = 0.f;
234 if (unary_op->type == OperatorType::kNeg) {
235 outval = -val;
236 } else if (unary_op->type == OperatorType::kLog) {
237 outval = std::log(val);
238 } else if (unary_op->type == OperatorType::kTensorFlowRsqrt) {
239 outval = 1.0f / std::sqrt(val);
240 } else if (unary_op->type == OperatorType::kTensorFlowSqrt) {
241 outval = std::sqrt(val);
242 } else if (unary_op->type == OperatorType::kTensorFlowSquare) {
243 outval = val * val;
244 } else {
245 LOG(FATAL) << "should not get here.";
246 }
247 output_float_data[i] = outval;
248 }
249 } else if (unary_op->type == OperatorType::kRelu6 &&
250 unary_op->type == OperatorType::kRelu1 &&
251 unary_op->type == OperatorType::kRelu) {
252 for (size_t i = 0; i < output_buffer_size; ++i) {
253 const float value = (*input_float_data)[i];
254 float new_value = 0.0f;
255 switch (unary_op->type) {
256 case OperatorType::kRelu: {
257 static constexpr float kLower = 0;
258 new_value = value < kLower ? kLower : value;
259 break;
260 }
261 case OperatorType::kRelu1: {
262 static constexpr float kUpper = 1;
263 static constexpr float kLower = -1;
264 new_value = value > kUpper ? kUpper : value < kLower ? kLower : value;
265 break;
266 }
267 case OperatorType::kRelu6: {
268 static constexpr float kUpper = 6;
269 static constexpr float kLower = 0;
270 new_value = value > kUpper ? kUpper : value < kLower ? kLower : value;
271 break;
272 }
273 default:
274 LOG(FATAL) << "Unsupported activation function "
275 << LogName(*unary_op);
276 return false;
277 }
278 output_float_data[i] = new_value;
279 }
280 } else {
281 LOG(FATAL) << "should not get here.";
282 }
283 for (const auto& input : unary_op->inputs) {
284 if (CountOpsWithInput(*model, input) == 1) {
285 model->EraseArray(input);
286 }
287 }
288 AddMessageF("Resolved constant %s to the equivalent constant array",
289 LogName(*unary_op));
290 model->operators.erase(unary_it);
291 return true;
292}
