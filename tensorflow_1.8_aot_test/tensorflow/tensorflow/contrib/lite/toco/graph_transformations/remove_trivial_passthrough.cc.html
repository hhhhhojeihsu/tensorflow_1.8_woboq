28// Reroute all edges involving a given discardable array to another
29// array instead. from_array is assumed to be discardable, and consequently
30// this only updates operator edges (since discardable arrays only
31// appear there, and not e.g. in model flags).
32void RerouteEdges(const string& from_array, const string& to_array,
33 Model* model) {
34 for (const auto& op : model->operators) {
35 for (auto& output : op->outputs) {
36 if (output == from_array) {
37 output = to_array;
38 }
39 }
40 for (auto& input : op->inputs) {
41 if (input == from_array) {
42 input = to_array;
43 }
44 }
45 }
46}

50bool RemoveTrivialPassthroughOp(GraphTransformation* transformation,
51 Model* model, std::size_t op_index) {
52 const auto passthru_it = model->operators.begin() + op_index;
53 auto* passthru_op = passthru_it->get();
54 CHECK_EQ(passthru_op->outputs.size(), 1);
55 CHECK_GE(passthru_op->inputs.size(), 1);
56 int count_nonconstant_input_arrays = 0;
57 // We call 'main input' the unique nonconstant input array if there is one,
58 // or else the 0-th input.
59 int main_input_array_index = 0;
60 for (int i = 0; i < passthru_op->inputs.size(); i++) {
61 if (!model->GetArray(passthru_op->inputs[i]).buffer) {
62 count_nonconstant_input_arrays++;
63 if (count_nonconstant_input_arrays == 1) {
64 main_input_array_index = i;
65 }
66 }
67 }
68
69 const string main_input_name = passthru_op->inputs[main_input_array_index];
70 const string output_name = passthru_op->outputs[0];
71
72 // Build the list of all input and output arrays of the passthrough node
73 // that we are considering removing. Any of these arrays is a candidate
74 // for being removed as well, if nothing else references it. Doing that
75 // arrays-removal together with the passthrough-node-removal proved too
76 // error-prone.
77 std::vector<string> removal_candidates;
78 for (const string& input : passthru_op->inputs) {
79 removal_candidates.push_back(input);
80 }
81 removal_candidates.push_back(output_name);
82
83 if (IsDiscardableArray(*model, output_name)) {
84 transformation->AddMessageF(
85 "Removing %s, keeping its non-constant input array",
86 LogName(*passthru_op));
87 for (const string& input : passthru_op->inputs) {
88 if (IsDiscardableArray(*model, input) && input != main_input_name &&
89 CountOpsWithInput(*model, input) == 1) {
90 }
91 }
92 RerouteEdges(output_name, main_input_name, model);
93 } else if (IsDiscardableArray(*model, main_input_name)) {
94 transformation->AddMessageF("Removing %s, keeping its output array",
95 LogName(*passthru_op));
96 for (const string& input : passthru_op->inputs) {
97 if (IsDiscardableArray(*model, input) &&
98 (input == main_input_name || CountOpsWithInput(*model, input) == 1)) {
99 }
100 }
101 RerouteEdges(main_input_name, output_name, model);
102 } else {
103 transformation->AddMessageF(
104 "Cannot remove %s, neither its main input nor its output may be "
105 "discarded",
106 LogName(*passthru_op));
107 return false;
108 }
109
110 // Remove the pass-through node.
111 model->operators.erase(passthru_it);
112
113 // Remove any array that is no longer used.
114 for (const string& removal_candidate : removal_candidates) {
115 bool is_referenced = false;
116 for (const auto& op : model->operators) {
117 for (const string& input : op->inputs) {
118 if (input == removal_candidate) {
119 is_referenced = true;
120 }
121 }
122 for (const string& output : op->outputs) {
123 if (output == removal_candidate) {
124 is_referenced = true;
125 }
126 }
127 }
128 if (!is_referenced) {
129 model->EraseArray(removal_candidate);
130 }
131 }
132
133 return true;
134}
