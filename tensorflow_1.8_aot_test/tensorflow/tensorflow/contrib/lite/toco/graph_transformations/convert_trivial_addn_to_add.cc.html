21// This pass will convert an AddN operator with only 2 inputs into a regular Add
22// operator, to which more optimizations may apply.
23bool ConvertTrivialAddNToAdd::Run(Model* model, std::size_t op_index) {
24 auto addn_it = model->operators.begin() + op_index;
25 if (addn_it->get()->type != OperatorType::kAddN) {
26 return false;
27 }
28 AddNOperator* addn_op = static_cast<AddNOperator*>(addn_it->get());
29 CHECK_GE(addn_op->inputs.size(), 2);
30 CHECK_EQ(addn_op->outputs.size(), 1);
31
32 // We only reduce AddN with N=2 to a regular Add.
33 if (addn_op->inputs.size() != 2) {
34 return false;
35 }
36
37 // Copy inputs & outputs to regular Add.
38 auto* add_op = new AddOperator;
39 add_op->inputs.push_back(addn_op->inputs[0]);
40 add_op->inputs.push_back(addn_op->inputs[1]);
41 add_op->outputs = addn_op->outputs;
42
43 // Replace the AddN operator in the graph.
44 const auto add_it = model->operators.emplace(addn_it, add_op);
45 addn_it = add_it + 1;
46 CHECK_EQ(addn_it->get(), addn_op);
47 model->operators.erase(addn_it);
48 return true;
49}
