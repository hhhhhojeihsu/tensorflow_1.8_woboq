27bool ResolveTensorFlowMatMul::Run(Model* model, std::size_t op_index) {
28 auto matmul_it = model->operators.begin() + op_index;
29 if (matmul_it->get()->type != OperatorType::kTensorFlowMatMul) {
30 return false;
31 }
32 const auto* matmul_op =
33 static_cast<const TensorFlowMatMulOperator*>(matmul_it->get());
34
35 // Reorder the axes on the second input. TensorFlow uses row-major ordering
36 // on both inputs, however this is inefficient for the FullyConnected
37 // operator. We'll transpose the second input to be in column-major order now
38 // and let constant propagation optimize things (if possible).
39 auto* transpose_op = new TransposeOperator;
40 transpose_op->inputs = {
41 matmul_op->inputs[1],
42 CreateInt32Array(
43 model,
44 AvailableArrayName(*model, matmul_op->inputs[1] + "/transpose/perm"),
45 {1, 0})};
46 transpose_op->outputs = {
47 AvailableArrayName(*model, matmul_op->inputs[1] + "/transpose")};
48 model->GetOrCreateArray(transpose_op->outputs[0]);
49 model->operators.emplace(matmul_it, transpose_op);
50
51 // Refresh iterator.
52 matmul_it = model->operators.begin();
53 for (; matmul_it != model->operators.end(); ++matmul_it) {
54 if (matmul_it->get() == matmul_op) {
55 break;
56 }
57 }
58 DCHECK_EQ(matmul_it->get(), matmul_op);
59
60 string input_lhs = matmul_op->inputs[0];
61 string input_rhs = transpose_op->outputs[0];
62
63 // Construct the new FullyConnectedOperator.
64 auto* fc_op = new FullyConnectedOperator;
65 fc_op->outputs = matmul_op->outputs;
66
67 // Insert the newly constructed FullyConnectedOperator.
68 model->operators.emplace(matmul_it, fc_op) + 1;
69
70 // Find the op producing the array passed to this MatMul
71 auto previous_op_it = model->operators.begin();
72 bool found = false;
73 for (; previous_op_it != model->operators.end(); ++previous_op_it) {
74 for (const auto& output : (*previous_op_it)->outputs) {
75 if (output == matmul_op->inputs[0]) {
76 found = true;
77 break;
78 }
79 }
80 if (found) {
81 break;
82 }
83 }
84 Operator* previous_op = (found) ? previous_op_it->get() : nullptr;
85
86 // Refresh iterator.
87 matmul_it = model->operators.begin();
88 for (; matmul_it != model->operators.end(); ++matmul_it) {
89 if (matmul_it->get() == matmul_op) {
90 break;
91 }
92 }
93 DCHECK_EQ(matmul_it->get(), matmul_op);
94
95 // The way that TensorFlow encodes FullyConnected ops is as a pair
96 // (Reshape, MatMul), so we want to remove the Reshape op and rewrite the
97 // MatMul op as a FullyConnected. However, TensorFlow skips the Reshape ops if
98 // the input doesn't need reshaping, so we can't just match (Reshape, MatMul)
99 // pairs.
100 if (previous_op && previous_op->type == OperatorType::kTensorFlowReshape) {
101 AddMessageF("Combining %s and %s into %s", LogName(*previous_op),
102 LogName(*matmul_op), LogName(*fc_op));
103 const auto& previous_op_output = previous_op->outputs[0];
104 if (CountOpsWithInput(*model, previous_op_output) == 1) {
105 model->EraseArray(previous_op_output);
106 }
107 CHECK_EQ(previous_op->inputs.size(), 2);
108 input_lhs = previous_op->inputs[0];
109 // Only remove Reshape node if no other node uses its output.
110 if (CountOpsWithInput(*model, previous_op_output) == 1) {
111 const auto& previous_op_shape = previous_op->inputs[1];
112 if (CountOpsWithInput(*model, previous_op_shape) == 1 &&
113 !GetOpWithOutput(*model, previous_op_shape)) {
114 model->EraseArray(previous_op_shape);
115 }
116 model->operators.erase(previous_op_it);
117 }
118
119 // We may have just invalidated matmul_it, so let's refresh it now.
120 matmul_it = model->operators.begin();
121 for (; matmul_it != model->operators.end(); ++matmul_it) {
122 if (matmul_it->get() == matmul_op) {
123 break;
124 }
125 }
126 CHECK(matmul_it != model->operators.end());
127 CHECK(matmul_it->get() == matmul_op);
128 } else {
129 AddMessageF("Replacing %s by a FullyConnected operator",
130 LogName(*matmul_op));
131 }
132
133 fc_op->inputs = {input_lhs, input_rhs};
134
135 // erase the MatMul operator
136 model->operators.erase(matmul_it);
137 return true;
138}
