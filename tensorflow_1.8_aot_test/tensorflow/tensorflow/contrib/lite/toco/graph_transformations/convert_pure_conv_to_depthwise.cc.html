27bool ConvertPureConvToDepthwise::Run(Model* model, std::size_t op_index) {
28 auto conv_it = model->operators.begin() + op_index;
29 if (conv_it->get()->type != OperatorType::kConv) {
30 return false;
31 }
32 const auto* conv_op = static_cast<ConvOperator*>(conv_it->get());
33 if (conv_op->stride_width != conv_op->stride_height) {
34 return false;
35 }
36 if ((conv_op->dilation_width_factor != 1) ||
37 (conv_op->dilation_height_factor != 1)) {
38 // Depthwise conv does not support dilation
39 return false;
40 }
41 auto& weights_array = model->GetArray(conv_op->inputs[1]);
42 if (!weights_array.buffer) {
43 // Yield until the weights are resolved as a constant array.
44 return false;
45 }
46 if (weights_array.data_type != ArrayDataType::kFloat) {
47 return false;
48 }
49 if (weights_array.shape().dims(3) != 1) {
50 // Not a pure convolution: Conv does accumulation across the depth
51 // dimension.
52 return false;
53 }
54 // At this point we know we have a pure conv. Rewrite it as DepthwiseConv.
55 AddMessageF(
56 "%s is purely convolutional (input/weights depth is 1), replacing it by "
57 "a DepthwiseConv.",
58 LogName(*conv_op));
59 auto* depthwiseconv_op = new DepthwiseConvOperator;
60 // Conv and DepthwiseConv take the same inputs
61 depthwiseconv_op->inputs = conv_op->inputs;
62 // Conv may have a 2nd output for im2col
63 depthwiseconv_op->outputs = {conv_op->outputs[0]};
64 if (conv_op->outputs.size() > 1) {
65 // delete the im2col array.
66 model->EraseArray(conv_op->outputs[1]);
67 }
68 depthwiseconv_op->fused_activation_function =
69 conv_op->fused_activation_function;
70 // Let PropagateFixedSizes recompute fixed padding, just in case some day it
71 // may be different for Conv vs DepthwiseConv.
72 depthwiseconv_op->padding.type = conv_op->padding.type;
73 depthwiseconv_op->stride_height = conv_op->stride_height;
74 depthwiseconv_op->stride_width = conv_op->stride_width;
75 depthwiseconv_op->depth_multiplier = weights_array.shape().dims(0);
76 // Replace the operator in the graph.
77 const auto depthwiseconv_it =
78 model->operators.emplace(conv_it, depthwiseconv_op);
79 conv_it = depthwiseconv_it + 1;
80 CHECK_EQ(conv_it->get(), conv_op);
81 model->operators.erase(conv_it);
82 // Shuffle the weights.
83 const auto& weights_shape = weights_array.shape();
84 auto& weights_buffer =
85 weights_array.GetMutableBuffer<ArrayDataType::kFloat>();
86 const std::vector<float>& conv_weights_data = weights_buffer.data;
87 std::vector<float> depthwise_conv_weights_data(conv_weights_data.size());
88 const int depth = weights_shape.dims(0);
89 const int width = weights_shape.dims(1);
90 const int height = weights_shape.dims(2);
91 const int width_height = width * height;
92 for (int c = 0; c < depth; c++) {
93 for (int xy = 0; xy < width_height; xy++) {
94 depthwise_conv_weights_data[c + depth * xy] =
95 conv_weights_data[xy + width_height * c];
96 }
97 }
98 *weights_array.mutable_shape()->mutable_dims() = {1, width, height, depth};
99 weights_buffer.data = depthwise_conv_weights_data;
100 return true;
101}
