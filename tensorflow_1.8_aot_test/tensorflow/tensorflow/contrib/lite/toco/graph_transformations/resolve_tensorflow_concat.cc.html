28bool ResolveTensorFlowConcat::Run(Model* model, std::size_t op_index) {
29 auto concat_it = model->operators.begin() + op_index;
30 const auto* tf_concat_op = concat_it->get();
31 if (tf_concat_op->type != OperatorType::kTensorFlowConcat &&
32 tf_concat_op->type != OperatorType::kTensorFlowConcatV2) {
33 return false;
34 }
35
36 CHECK_GE(tf_concat_op->inputs.size(), 2);
37 // TensorFlow Concat and ConcatV2 nodes only differ by the ordering
38 // of inputs: in Concat,the axis is the first input, while in
39 // ConcatV2, it is the last input.
40 std::size_t axis_pos = 0;
41 if (tf_concat_op->type == OperatorType::kTensorFlowConcatV2) {
42 axis_pos = tf_concat_op->inputs.size() - 1;
43 }
44 const string axis_name = tf_concat_op->inputs[axis_pos];
45 std::vector<string> concat_input_names;
46 for (std::size_t i = 0; i < tf_concat_op->inputs.size(); i++) {
47 if (i != axis_pos) {
48 concat_input_names.push_back(tf_concat_op->inputs[i]);
49 }
50 }
51 // If the axis array hasn't been resolved to a constant yet,
52 // we need to yield.
53 const auto& axis_array = model->GetArray(axis_name);
54 if (!axis_array.buffer) {
55 AddMessageF("Waiting for the axis of %s to be resolved to a constant",
56 LogName(*tf_concat_op));
57 return false;
58 }
59
60 CHECK(axis_array.data_type == ArrayDataType::kInt32);
61 const auto& axis_data = axis_array.GetBuffer<ArrayDataType::kInt32>().data;
62 CHECK_EQ(axis_data.size(), 1);
63 const int axis = axis_data[0];
64
65 // Create the Concatenation op replacing the TensorFlowConcat op.
66 auto* concatenation_op = new ConcatenationOperator;
67 concatenation_op->axis = axis;
68 concatenation_op->inputs = concat_input_names;
69 concatenation_op->outputs = {tf_concat_op->outputs[0]};
70 auto depth_concat_it = model->operators.emplace(concat_it, concatenation_op);
71 CHECK_EQ(depth_concat_it->get(), concatenation_op);
72 // Update invalidated iterator
73 concat_it = depth_concat_it + 1;
74 CHECK_EQ(concat_it->get(), tf_concat_op);
75
76 // Remove the axis array if it is not used by anything else.
77 if (CountOpsWithInput(*model, axis_name) == 1) {
78 model->EraseArray(axis_name);
79 }
80 // Remove the TensorFlowConcat op
81 model->operators.erase(concat_it);
82 return true;
83}
