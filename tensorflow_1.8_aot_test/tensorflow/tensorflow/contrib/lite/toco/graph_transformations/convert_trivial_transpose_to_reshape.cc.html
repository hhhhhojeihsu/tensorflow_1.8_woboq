26bool TransposeAffectsMemoryOrder(std::vector<int> perm,
27 std::vector<int> in_shape) {
28 CHECK_EQ(perm.size(), in_shape.size());
29 // See what the ordering of the non-unary columns are before and after
30 // transpose permutation. If the major indices stay in the same order (not
31 // just the shape) then the flat buffer representation shouldn't change.
32 std::vector<int> old_major_index_ordering;
33 std::vector<int> new_major_index_ordering;
34 for (int i = 0; i < in_shape.size(); i++) {
35 if (in_shape[i] != 1) {
36 old_major_index_ordering.push_back(i);
37 }
38
39 if (in_shape[perm[i]] != 1) {
40 new_major_index_ordering.push_back(perm[i]);
41 }
42 }
43
44 CHECK_EQ(new_major_index_ordering.size(), old_major_index_ordering.size());
45
46 return old_major_index_ordering != new_major_index_ordering;
47}

51bool ConvertTrivialTransposeToReshape::Run(Model* model, std::size_t op_index) {
52 auto transpose_it = model->operators.begin() + op_index;
53 if (transpose_it->get()->type != OperatorType::kTranspose) {
54 return false;
55 }
56 TransposeOperator* transpose_op =
57 static_cast<TransposeOperator*>(transpose_it->get());
58
59 const auto& input_array = model->GetArray(transpose_op->inputs[0]);
60 const auto& output_array = model->GetArray(transpose_op->outputs[0]);
61 if (!input_array.has_shape() || !output_array.has_shape()) {
62 // Yield until PropagateFixedSizes has been run on this op.
63 return false;
64 }
65 // Note: We can assume we have error checked inputs in PropagateFixedSizes.
66
67 // Check that the permutation has propogated.
68 std::vector<int> const& perm = transpose_op->perm;
69 if (perm.empty()) {
70 return false;
71 }
72
73 // This transpose is trivial if non-unitary dimensions remain in the same
74 // order.
75 std::vector<int> const& input_dims = input_array.shape().dims();
76 std::vector<int> const& output_dims = output_array.shape().dims();
77
78 if (TransposeAffectsMemoryOrder(perm, input_dims)) {
79 return false;
80 }
81
82 // This transpose is trivial. Replace it with a Reshape op.
83 auto* reshape_op = new TensorFlowReshapeOperator;
84
85 // Copy input and output
86 reshape_op->inputs.push_back(transpose_op->inputs[0]);
87 reshape_op->outputs = transpose_op->outputs;
88
89 // Create a new input array for the shape input
90 string perm_array_name = transpose_op->inputs[1];
91 string shape_array_name = toco::AvailableArrayName(*model, perm_array_name);
92 Array& shape_array = model->GetOrCreateArray(shape_array_name);
93 *(shape_array.mutable_shape()->mutable_dims()) = {
94 1, static_cast<int>(output_dims.size())};
95 reshape_op->inputs.push_back(shape_array_name);
96 shape_array.data_type = ArrayDataType::kInt32;
97 auto& shape_buffer = shape_array.GetMutableBuffer<ArrayDataType::kInt32>();
98 shape_buffer.data = output_dims;
99
100 // Delete perm array if unused
101 if (IsDiscardableArray(*model, perm_array_name) &&
102 CountOpsWithInput(*model, perm_array_name) == 1) {
103 model->EraseArray(perm_array_name);
104 }
105
106 // Replace the operator in the graph.
107 const auto reshape_it = model->operators.emplace(transpose_it, reshape_op);
108 transpose_it = reshape_it + 1;
109 CHECK_EQ(transpose_it->get(), transpose_op);
110 model->operators.erase(transpose_it);
111
112 return true;
113}
