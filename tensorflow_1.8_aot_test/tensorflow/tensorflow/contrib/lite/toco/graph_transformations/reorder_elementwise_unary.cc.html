30bool IsElementwiseOperator(OperatorType optype) {
31 switch (optype) {
32 case OperatorType::kCast:
33 case OperatorType::kExp:
34 case OperatorType::kFloor:
35 case OperatorType::kNeg:
36 case OperatorType::kRelu:
37 case OperatorType::kRelu1:
38 case OperatorType::kRelu6:
39 case OperatorType::kTanh:
40 case OperatorType::kTensorFlowSqrt:
41 case OperatorType::kTensorFlowSquare:
42 return true;
43 default:
44 return false;
45 }
46}
47
48bool IsMoveOperator(OperatorType optype) {
49 switch (optype) {
50 case OperatorType::kDepthToSpace:
51 case OperatorType::kExpandDims:
52 case OperatorType::kSpaceToDepth:
53 case OperatorType::kSqueeze:
54 case OperatorType::kTensorFlowReshape:
55 case OperatorType::kTranspose:
56 return true;
57 default:
58 return false;
59 }
60}

64// Swap elementwise operators such that all value operators occur before all
65// element move operators, e.g. negation then transpose.
66bool ReorderElementwiseUnary::Run(Model* model, std::size_t op_index) {
67 const auto element_op_it = model->operators.begin() + op_index;
68 std::unique_ptr<Operator>& element_op = *element_op_it;
69 if (!IsElementwiseOperator(element_op->type)) {
70 return false;
71 }
72
73 const string intermediate_name = element_op->inputs[0];
74 auto it = FindOpWithOutput(*model, intermediate_name);
75 if (it == model->operators.end()) {
76 AddMessageF("No preceding operator");
77 return false;
78 }
79
80 std::unique_ptr<Operator>& move_op = *it;
81 if (!IsMoveOperator(move_op->type)) {
82 AddMessageF("Preceding operator is not a move operator");
83 return false;
84 }
85
86 if (CountOpsWithInput(*model, intermediate_name) != 1) {
87 AddMessageF("Input %s used elsewhere", intermediate_name);
88 return false;
89 }
90
91 // Check that the intermediate is discardable.
92 if (!IsDiscardableArray(*model, intermediate_name)) {
93 AddMessageF(
94 "Cannot swap elementwise as it would invalidate %s which is "
95 "an output array.",
96 intermediate_name);
97 return false;
98 }
99
100 // op->inputs may change so we need to keep a value by copy.
101 const string input_name = move_op->inputs[0];
102 const string output_name = element_op->outputs[0];
103
104 AddMessageF("Swapping around operators with %s and %s", LogName(*element_op),
105 LogName(*move_op));
106
107 // If the output array is an exit node for the graph then we need to retain
108 // the name as an output node. This makes the naming scheme a little confusing
109 // but is required in this rare case.
110 if (!IsDiscardableArray(*model, output_name)) {
111 // The output name of the sequence needs to stay static, so create a new
112 // array new use for the intermediate.
113 const auto new_intermediate_name =
114 AvailableArrayName(*model, element_op->outputs[0] + "_reorder");
115 AddMessageF("Adding new array %s to preserve output array name %s",
116 new_intermediate_name, output_name);
117
118 element_op->inputs[0] = input_name;
119 element_op->outputs[0] = new_intermediate_name;
120 model->EraseArray(intermediate_name);
121 move_op->inputs[0] = new_intermediate_name;
122 move_op->outputs[0] = output_name;
123 } else {
124 // The intermediate array is now the output array.
125 for (int i = 0; i < model->operators.size(); i++) {
126 Operator* consumer = model->operators[i].get();
127 for (int j = 0; j < consumer->inputs.size(); j++) {
128 if (consumer->inputs[j] == output_name) {
129 consumer->inputs[j] = intermediate_name;
130 }
131 }
132 }
133
134 element_op->inputs[0] = input_name;
135 move_op->inputs[0] = output_name;
136 }
137
138 // Reset both arrays as shape, type, min/max, etc can all change because of
139 // the position swap.
140 model->EraseArray(element_op->outputs[0]);
141 model->EraseArray(move_op->outputs[0]);
142
143 // Reconstruct.
144 model->GetOrCreateArray(element_op->outputs[0]);
145 model->GetOrCreateArray(move_op->outputs[0]);
146
147 // Swap the order of the operators.
148 element_op.swap(move_op);
149
150 return true;
151}
