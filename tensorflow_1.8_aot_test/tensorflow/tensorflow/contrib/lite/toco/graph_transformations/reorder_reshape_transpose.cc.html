30bool OperatorReady(const Model& model, const Operator* op) {
31 if (!model.HasArray(op->inputs[0]) || !model.HasArray(op->inputs[1]) ||
32 !model.HasArray(op->outputs[0])) {
33 return false;
34 }
35
36 if (!model.GetArray(op->inputs[0]).has_shape() ||
37 !model.GetArray(op->outputs[0]).has_shape()) {
38 // Input and output needs the shape.
39 return false;
40 }
41
42 if (!model.GetArray(op->inputs[1]).buffer) {
43 // Buffer needs to be a constant.
44 return false;
45 }
46
47 return true;
48}

50// Utility function to filter out a value.
51void Filter(std::vector<int>* vec, int value) {
52 vec->erase(std::remove(vec->begin(), vec->end(), value), vec->end());
53}

55// Computes a new permutation used to swap a reshape-transpose to a
56// transpose-reshape. In this case the permutation operates on the intermediate
57// shape.
58std::vector<int> ComputeNewPerm(std::vector<int> input_dims,
59 std::vector<int> intermediate_dims,
60 std::vector<int> perm) {
61 // These are the major axis of the input.
62 std::vector<int> input_indices;
63 for (int i = 0; i < input_dims.size(); i++) {
64 if (input_dims[i] != 1) {
65 input_indices.push_back(i);
66 }
67 }
68
69 // This maps which indices of the input produced the intermediate indices for
70 // non-unary dimensions.
71 std::unordered_map<int, int> intermediate_to_input_indices_map;
72 for (int i = 0; i < intermediate_dims.size(); i++) {
73 if (intermediate_dims[i] != 1) {
74 intermediate_to_input_indices_map[i] =
75 input_indices[intermediate_to_input_indices_map.size()];
76 }
77 }
78
79 // Translate the transpose permutation to a new permutation starting with the
80 // major indices.
81 std::vector<int> new_perm;
82 new_perm.reserve(input_dims.size());
83 for (int i = 0; i < perm.size(); i++) {
84 if (intermediate_dims[perm[i]] == 1) continue;
85
86 new_perm.push_back(intermediate_to_input_indices_map[perm[i]]);
87 }
88
89 // Fill the rest of the transpose in with the ones.
90 for (int index = 0; index < input_dims.size(); index++) {
91 if (input_dims[index] == 1) {
92 new_perm.push_back(index);
93 }
94 }
95
96 CHECK_EQ(new_perm.size(), input_dims.size());
97 return new_perm;
98}

102// Swaps reshape-transpose to transpose-reshape whenever possible. This is
103// possible when the reshape does not affect memory ordering.
104bool ReorderReshapeTranspose::Run(Model* model, std::size_t op_index) {
105 auto transpose_it = model->operators.begin() + op_index;
106
107 TransposeOperator* transpose_op = ConvertOperator<TransposeOperator*>(
108 transpose_it->get(), OperatorType::kTranspose);
109
110 if (transpose_op == nullptr) {
111 return false;
112 }
113
114 if (!OperatorReady(*model, transpose_op) || transpose_op->perm.empty()) {
115 // Wait for values to propagate.
116 return false;
117 }
118
119 // Find the operator that produces the transpose op.
120 auto reshape_it = FindOpWithOutput(*model, transpose_op->inputs[0]);
121 if (reshape_it == model->operators.end()) {
122 return false;
123 }
124
125 TensorFlowReshapeOperator* reshape_op =
126 ConvertOperator<TensorFlowReshapeOperator*>(
127 reshape_it->get(), OperatorType::kTensorFlowReshape);
128 if (reshape_op == nullptr) {
129 return false;
130 }
131
132 // Ignore if the reshape is uninitialized.
133 if (!OperatorReady(*model, reshape_op) || reshape_op->shape.empty()) {
134 return false;
135 }
136
137 // Need to copy to keep static if permutated.
138 const string input_name = reshape_op->inputs[0];
139 const string intermediate_name = reshape_op->outputs[0];
140 const string output_name = transpose_op->outputs[0];
141
142 // Intermediate should not be consumed by any other operators.
143 if (CountOpsWithInput(*model, intermediate_name) != 1) {
144 AddMessageF("Input %s used elsewhere", intermediate_name);
145 return false;
146 }
147
148 // Check that the intermediate is not an output array.
149 if (!IsDiscardableArray(*model, intermediate_name)) {
150 AddMessageF(
151 "Cannot reorder reshape-transpose as it would invalidate %s which is "
152 "an output array.",
153 intermediate_name);
154 return false;
155 }
156
157 // Get the arrays.
158 const auto& input_array = model->GetArray(input_name);
159 const auto& intermediate_array = model->GetArray(intermediate_name);
160 const auto& output_array = model->GetArray(output_name);
161
162 // Get the shapes of each array.
163 Shape input_shape = input_array.shape();
164 Shape intermediate_shape = intermediate_array.shape();
165 Shape output_shape = output_array.shape();
166
167 // Assign ids to non-unary indices.
168 std::vector<int> input_dims = input_shape.dims();
169 std::vector<int> intermediate_dims = intermediate_shape.dims();
170 std::vector<int> output_dims = output_shape.dims();
171
172 // If the reshape is equivalent to a transpose with fewer/more unary
173 // dimensions then it can be moved between the transpose.
174 if (!ReshapeIsEquivalentToTranspose(*model, reshape_op,
175 true /*allow_extra_unary_dims*/)) {
176 return false;
177 }
178
179 if (!IsDiscardableArray(*model, output_name)) {
180 // The output name of the sequence needs to stay static, so create a new
181 // array new use for the intermediate.
182 const auto new_intermediate_name =
183 AvailableArrayName(*model, transpose_op->outputs[0] + "_exchange");
184 AddMessageF("Adding new array %s to preserve output array name %s",
185 new_intermediate_name, transpose_op->outputs[0]);
186 transpose_op->inputs[0] = input_name;
187 transpose_op->outputs[0] = new_intermediate_name;
188 reshape_op->inputs[0] = new_intermediate_name;
189 reshape_op->outputs[0] = output_name;
190 model->EraseArray(intermediate_name);
191 } else {
192 // The intermediate array is now the output array.
193 for (int i = 0; i < model->operators.size(); i++) {
194 Operator* consumer = model->operators[i].get();
195 for (int j = 0; j < consumer->inputs.size(); j++) {
196 if (consumer->inputs[j] == output_name) {
197 consumer->inputs[j] = intermediate_name;
198 }
199 }
200 }
201
202 transpose_op->inputs[0] = input_name;
203 reshape_op->inputs[0] = output_name;
204 }
205
206 // If transposes constant buffer is used elsewhere, make a new copy.
207 if (CountOpsWithInput(*model, transpose_op->inputs[1]) != 1) {
208 transpose_op->inputs[1] =
209 AvailableArrayName(*model, transpose_op->inputs[1] + "_copy");
210 }
211
212 // Make the new transpose permutation.
213 const std::vector<int> new_perm =
214 ComputeNewPerm(input_dims, intermediate_dims, transpose_op->perm);
215 CHECK_EQ(input_dims.size(), new_perm.size());
216
217 auto& transpose_array = model->GetOrCreateArray(transpose_op->inputs[1]);
218 transpose_array.GetMutableBuffer<ArrayDataType::kInt32>().data = new_perm;
219 *(transpose_array.mutable_shape()->mutable_dims()) = {
220 static_cast<int>(new_perm.size())};
221 transpose_op->perm = new_perm;
222
223 // If the reshape's constant buffer is reused, create a new one.
224 if (CountOpsWithInput(*model, reshape_op->inputs[1]) != 1) {
225 reshape_op->inputs[1] =
226 AvailableArrayName(*model, reshape_op->inputs[1] + "_copy");
227 }
228
229 // We need to modify the reshape input array to target the new output size.
230 auto& reshape_array = model->GetOrCreateArray(reshape_op->inputs[1]);
231 reshape_array.GetMutableBuffer<ArrayDataType::kInt32>().data = output_dims;
232 *(reshape_array.mutable_shape()->mutable_dims()) = {
233 static_cast<int>(output_shape.dimensions_count())};
234 reshape_op->shape.clear();
235
236 AddMessageF("Swapping around operators between %s and %s", input_name,
237 output_name);
238
239 model->GetOrCreateArray(transpose_op->outputs[0]).clear_shape();
240 model->GetOrCreateArray(reshape_op->outputs[0]).clear_shape();
241
242 // Swap the order of the operators.
243 transpose_it->swap(*reshape_it);
244
245 return true;
246}
