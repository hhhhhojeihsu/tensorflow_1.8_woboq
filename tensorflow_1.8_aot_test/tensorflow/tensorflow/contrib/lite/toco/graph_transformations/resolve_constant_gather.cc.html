26// Gathers data from axis 0.
27template <ArrayDataType Type>
28inline void Gather(const Array& input_array, int input_rank,
29 const Array& coords_array, Array* output_array) {
30 const Shape& input_shape = input_array.shape();
31 const std::vector<DataType<Type>>& input_data =
32 input_array.GetBuffer<Type>().data;
33 const Shape& coords_shape = coords_array.shape();
34 const std::vector<int32>& coords_data =
35 coords_array.GetBuffer<ArrayDataType::kInt32>().data;
36
37 const Shape& output_shape = output_array->shape();
38 std::vector<DataType<Type>>& output_data =
39 output_array->GetMutableBuffer<Type>().data;
40 output_data.resize(RequiredBufferSizeForShape(output_shape));
41
42 int rev_input_rank = input_shape.dimensions_count() - 1 - (input_rank - 1);
43 CHECK_EQ(coords_shape.dims(0), output_array->shape().dims(rev_input_rank));
44
45 int stride = 1;
46 for (int i = input_shape.dimensions_count() - 1; i >= input_rank - 1; --i) {
47 stride *= input_shape.dims(i);
48 }
49
50 for (int i = 0; i < coords_shape.dims(0); ++i) {
51 DCHECK_GE(coords_data[i], 0);
52 DCHECK_LT(coords_data[i], input_shape.dims(rev_input_rank));
53 DataType<Type>* out = output_data.data() + i * stride;
54 const DataType<Type>* in = input_data.data() + coords_data[i] * stride;
55 memcpy(out, in, sizeof(DataType<Type>) * stride);
56 }
57}

61// Resolves a constant Gather operation.
62// This simply performs the gather and produces the output array with the
63// appropriate values.
64bool ResolveConstantGather::Run(Model* model, std::size_t op_index) {
65 auto it = model->operators.begin() + op_index;
66 const auto* base_op = it->get();
67 if (base_op->type != OperatorType::kGather) {
68 return false;
69 }
70 const auto* op = static_cast<const GatherOperator*>(base_op);
71
72 CHECK_EQ(op->inputs.size(), 2);
73 CHECK_EQ(op->outputs.size(), 1);
74 auto& output_array = model->GetArray(op->outputs[0]);
75 if (output_array.data_type == ArrayDataType::kNone) {
76 // Yield until the output type has been set by PropagateArrayDataTypes.
77 return false;
78 }
79 if (!output_array.has_shape()) {
80 // Yield until the output shape has been set by PropagateFixedShapes.
81 return false;
82 }
83
84 // Only handling axis=0 for now.
85 if (op->axis != 0) {
86 AddMessageF("%s has axis %d; only axis=0 is supported", LogName(*op),
87 op->axis);
88 return false;
89 }
90
91 // We require constant inputs.
92 if (!IsConstantParameterArray(*model, op->inputs[0]) ||
93 !IsConstantParameterArray(*model, op->inputs[1])) {
94 return false;
95 }
96 const Array& input_array = model->GetArray(op->inputs[0]);
97 const Array& coords_array = model->GetArray(op->inputs[1]);
98 CHECK(coords_array.data_type == ArrayDataType::kInt32)
99 << "Only int32 indices are supported";
100
101 CHECK(!output_array.buffer);
102 switch (output_array.data_type) {
103 case ArrayDataType::kFloat:
104 Gather<ArrayDataType::kFloat>(input_array, op->input_rank, coords_array,
105 &output_array);
106 break;
107 case ArrayDataType::kUint8:
108 Gather<ArrayDataType::kUint8>(input_array, op->input_rank, coords_array,
109 &output_array);
110 break;
111 case ArrayDataType::kInt32:
112 Gather<ArrayDataType::kInt32>(input_array, op->input_rank, coords_array,
113 &output_array);
114 break;
115 case ArrayDataType::kInt64:
116 Gather<ArrayDataType::kInt64>(input_array, op->input_rank, coords_array,
117 &output_array);
118 break;
119 default:
120 LOG(FATAL) << "Unsupported data type given to Gather op with output \""
121 << op->outputs[0] << "\"";
122 break;
123 }
124
125 // Erase input arrays if no longer used after we remove the op.
126 DeleteArrayIfUsedOnce(op->inputs[0], model);
127 DeleteArrayIfUsedOnce(op->inputs[1], model);
128
129 // Erase the operator.
130 model->operators.erase(it);
131 return true;
132}
