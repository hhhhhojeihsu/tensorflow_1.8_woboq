31bool ParseTocoFlagsFromCommandLineFlags(
32 int* argc, char* argv[], string* msg,
33 ParsedTocoFlags* parsed_toco_flags_ptr) {
34 using tensorflow::Flag;
35 ParsedTocoFlags& parsed_flags = *parsed_toco_flags_ptr;
36 std::vector<tensorflow::Flag> flags = {
37 Flag("input_file", parsed_flags.input_file.bind(),
38 parsed_flags.input_file.default_value(),
39 "Input file (model of any supported format). For Protobuf "
40 "formats, both text and binary are supported regardless of file "
41 "extension."),
42 Flag("savedmodel_directory", parsed_flags.savedmodel_directory.bind(),
43 parsed_flags.savedmodel_directory.default_value(),
44 "Full path to the directory containing the SavedModel."),
45 Flag("output_file", parsed_flags.output_file.bind(),
46 parsed_flags.output_file.default_value(),
47 "Output file. "
48 "For Protobuf formats, the binary format will be used."),
49 Flag("input_format", parsed_flags.input_format.bind(),
50 parsed_flags.input_format.default_value(),
51 "Input file format. One of: TENSORFLOW_GRAPHDEF, TFLITE."),
52 Flag("output_format", parsed_flags.output_format.bind(),
53 parsed_flags.output_format.default_value(),
54 "Output file format. "
55 "One of TENSORFLOW_GRAPHDEF, TFLITE, GRAPHVIZ_DOT."),
56 Flag("savedmodel_tagset", parsed_flags.savedmodel_tagset.bind(),
57 parsed_flags.savedmodel_tagset.default_value(),
58 "Comma-separated set of tags identifying the MetaGraphDef within "
59 "the SavedModel to analyze. All tags in the tag set must be "
60 "specified."),
61 Flag("default_ranges_min", parsed_flags.default_ranges_min.bind(),
62 parsed_flags.default_ranges_min.default_value(),
63 "If defined, will be used as the default value for the min bound "
64 "of min/max ranges used for quantization."),
65 Flag("default_ranges_max", parsed_flags.default_ranges_max.bind(),
66 parsed_flags.default_ranges_max.default_value(),
67 "If defined, will be used as the default value for the max bound "
68 "of min/max ranges used for quantization."),
69 Flag("inference_type", parsed_flags.inference_type.bind(),
70 parsed_flags.inference_type.default_value(),
71 "Target data type of arrays in the output file (for input_arrays, "
72 "this may be overridden by inference_input_type). "
73 "One of FLOAT, QUANTIZED_UINT8."),
74 Flag("inference_input_type", parsed_flags.inference_input_type.bind(),
75 parsed_flags.inference_input_type.default_value(),
76 "Target data type of input arrays. "
77 "If not specified, inference_type is used. "

80 parsed_flags.input_type.default_value(),
81 "Deprecated ambiguous flag that set both --input_data_types and "
82 "--inference_input_type."),
83 Flag("input_types", parsed_flags.input_types.bind(),
84 parsed_flags.input_types.default_value(),
85 "Deprecated ambiguous flag that set both --input_data_types and "
86 "--inference_input_type. Was meant to be a "
87 "comma-separated list, but this was deprecated before "
88 "multiple-input-types was ever properly supported."),
89
90 Flag("drop_fake_quant", parsed_flags.drop_fake_quant.bind(),
91 parsed_flags.drop_fake_quant.default_value(),
92 "Ignore and discard FakeQuant nodes. For instance, to "
93 "generate plain float code without fake-quantization from a "
94 "quantized graph."),
95 Flag(
96 "reorder_across_fake_quant",
97 parsed_flags.reorder_across_fake_quant.bind(),
98 parsed_flags.reorder_across_fake_quant.default_value(),
99 "Normally, FakeQuant nodes must be strict boundaries for graph "
100 "transformations, in order to ensure that quantized inference has "

103 "the first place. "
104 "However, that entails subtle requirements on where exactly "

106 "have FakeQuant nodes at unexpected locations, that prevent graph "
107 "transformations that are necessary in order to generate inference "

142// Defines the requirements for a given flag. kUseDefault means the default
143// should be used in cases where the value isn't specified by the user.
144enum class FlagRequirement {
145 kNone,
146 kMustBeSpecified,
147 kMustNotBeSpecified,
148 kUseDefault,
149};
150
151// Enforces the FlagRequirements are met for a given flag.
152template <typename T>
153void EnforceFlagRequirement(const T& flag, const string& flag_name,
154 FlagRequirement requirement) {
155 if (requirement == FlagRequirement::kMustBeSpecified) {
156 QCHECK(flag.specified()) << "Missing required flag " << flag_name;
157 }
158 if (requirement == FlagRequirement::kMustNotBeSpecified) {
159 QCHECK(!flag.specified())
160 << "Given other flags, this flag should not have been specified: "
161 << flag_name;
162 }
163}
164
165// Gets the value from the flag if specified. Returns default if the
166// FlagRequirement is kUseDefault.
167template <typename T>
168absl::optional<T> GetFlagValue(const Arg<T>& flag,
169 FlagRequirement requirement) {
170 if (flag.specified()) return flag.value();
171 if (requirement == FlagRequirement::kUseDefault) return flag.default_value();
172 return absl::optional<T>();
173}

177void ReadTocoFlagsFromCommandLineFlags(const ParsedTocoFlags& parsed_toco_flags,
178 TocoFlags* toco_flags) {
179 namespace port = toco::port;
180 port::CheckInitGoogleIsDone("InitGoogle is not done yet");
181
182#define READ_TOCO_FLAG(name, requirement) \
183 do { \
184 EnforceFlagRequirement(parsed_toco_flags.name, #name, requirement); \
185 auto flag_value = GetFlagValue(parsed_toco_flags.name, requirement); \
186 if (flag_value.has_value()) { \
187 toco_flags->set_##name(flag_value.value()); \
188 } \
189 } while (false)
190
191#define PARSE_TOCO_FLAG(Type, name, requirement) \
192 do { \
193 EnforceFlagRequirement(parsed_toco_flags.name, #name, requirement); \
194 auto flag_value = GetFlagValue(parsed_toco_flags.name, requirement); \
195 if (flag_value.has_value()) { \
196 Type x; \
197 QCHECK(Type##_Parse(flag_value.value(), &x)) \
198 << "Unrecognized " << #Type << " value " \
199 << parsed_toco_flags.name.value(); \
200 toco_flags->set_##name(x); \
201 } \
202 } while (false)
203
204 PARSE_TOCO_FLAG(FileFormat, input_format, FlagRequirement::kUseDefault);
205 PARSE_TOCO_FLAG(FileFormat, output_format, FlagRequirement::kUseDefault);
206 PARSE_TOCO_FLAG(IODataType, inference_type, FlagRequirement::kNone);
207 PARSE_TOCO_FLAG(IODataType, inference_input_type, FlagRequirement::kNone);
208 READ_TOCO_FLAG(default_ranges_min, FlagRequirement::kNone);
209 READ_TOCO_FLAG(default_ranges_max, FlagRequirement::kNone);
210 READ_TOCO_FLAG(drop_fake_quant, FlagRequirement::kNone);
211 READ_TOCO_FLAG(reorder_across_fake_quant, FlagRequirement::kNone);
212 READ_TOCO_FLAG(allow_custom_ops, FlagRequirement::kNone);
213 READ_TOCO_FLAG(drop_control_dependency, FlagRequirement::kNone);
214
215 // Deprecated flag handling.
216 if (parsed_toco_flags.input_type.specified()) {
217 LOG(WARNING)
218 << "--input_type is deprecated. It was an ambiguous flag that set both "
219 "--input_data_types and --inference_input_type. If you are trying "
220 "to complement the input file with information about the type of "
221 "input arrays, use --input_data_type. If you are trying to control "
222 "the quantization/dequantization of real-numbers input arrays in "
223 "the output file, use --inference_input_type.";
224 toco::IODataType input_type;
225 QCHECK(toco::IODataType_Parse(parsed_toco_flags.input_type.value(),
226 &input_type));
227 toco_flags->set_inference_input_type(input_type);
228 }
229 if (parsed_toco_flags.input_types.specified()) {
230 LOG(WARNING)
231 << "--input_types is deprecated. It was an ambiguous flag that set "
232 "both --input_data_types and --inference_input_type. If you are "
233 "trying to complement the input file with information about the "
234 "type of input arrays, use --input_data_type. If you are trying to "
235 "control the quantization/dequantization of real-numbers input "
236 "arrays in the output file, use --inference_input_type.";
237 std::vector<string> input_types =
238 absl::StrSplit(parsed_toco_flags.input_types.value(), ',');
239 QCHECK(!input_types.empty());
240 for (int i = 1; i < input_types.size(); i++) {
241 QCHECK_EQ(input_types[i], input_types[0]);
242 }
243 toco::IODataType input_type;
244 QCHECK(toco::IODataType_Parse(input_types[0], &input_type));
245 toco_flags->set_inference_input_type(input_type);
246 }

249#undef PARSE_TOCO_FLAG
