39 int* argc, char* argv[], string* msg,
40 ParsedModelFlags* parsed_model_flags_ptr) {

42 using tensorflow::Flag;
43 std::vector<tensorflow::Flag> flags = {
44 Flag("input_array", parsed_flags.input_array.bind(),

191void ReadModelFlagsFromCommandLineFlags(
192 const ParsedModelFlags& parsed_model_flags, ModelFlags* model_flags) {
193 toco::port::CheckInitGoogleIsDone("InitGoogle is not done yet");
194
195 // Load proto containing the initial model flags.
196 // Additional flags specified on the command line will overwrite the values.
197 if (parsed_model_flags.model_flags_file.specified()) {
198 string model_flags_file_contents;
199 QCHECK(port::file::GetContents(parsed_model_flags.model_flags_file.value(),
200 &model_flags_file_contents,
201 port::file::Defaults())
202 .ok())
203 << "Specified --model_flags_file="
204 << parsed_model_flags.model_flags_file.value()
205 << " was not found or could not be read";
206 QCHECK(ParseFromStringEitherTextOrBinary(model_flags_file_contents,
207 model_flags))
208 << "Specified --model_flags_file="
209 << parsed_model_flags.model_flags_file.value()
210 << " could not be parsed";
211 }
212
213#ifdef PLATFORM_GOOGLE
214 CHECK(!((base::SpecifiedOnCommandLine("batch") &&
215 parsed_model_flags.variable_batch.specified())))
216 << "The --batch and --variable_batch flags are mutually exclusive.";
217#endif
218 CHECK(!(parsed_model_flags.output_array.specified() &&
219 parsed_model_flags.output_arrays.specified()))
220 << "The --output_array and --vs flags are mutually exclusive.";
221
222 if (parsed_model_flags.output_array.specified()) {
223 model_flags->add_output_arrays(parsed_model_flags.output_array.value());
224 }
225
226 if (parsed_model_flags.output_arrays.specified()) {
227 std::vector<string> output_arrays =
228 absl::StrSplit(parsed_model_flags.output_arrays.value(), ',');
229 for (const string& output_array : output_arrays) {
230 model_flags->add_output_arrays(output_array);
231 }
232 }
233
234 const bool uses_single_input_flags =
235 parsed_model_flags.input_array.specified() ||
236 parsed_model_flags.mean_value.specified() ||
237 parsed_model_flags.std_value.specified() ||
238 parsed_model_flags.input_shape.specified();
239
240 const bool uses_multi_input_flags =
241 parsed_model_flags.input_arrays.specified() ||
242 parsed_model_flags.mean_values.specified() ||
243 parsed_model_flags.std_values.specified() ||
244 parsed_model_flags.input_shapes.specified();
245
246 QCHECK(!(uses_single_input_flags && uses_multi_input_flags))
247 << "Use either the singular-form input flags (--input_array, "
248 "--input_shape, --mean_value, --std_value) or the plural form input "
249 "flags (--input_arrays, --input_shapes, --mean_values, --std_values), "
250 "but not both forms within the same command line.";
251
252 if (parsed_model_flags.input_array.specified()) {
253 QCHECK(uses_single_input_flags);
254 model_flags->add_input_arrays()->set_name(
255 parsed_model_flags.input_array.value());
256 }
257 if (parsed_model_flags.input_arrays.specified()) {
258 QCHECK(uses_multi_input_flags);
259 for (const auto& input_array :
260 absl::StrSplit(parsed_model_flags.input_arrays.value(), ',')) {
261 model_flags->add_input_arrays()->set_name(string(input_array));
262 }
263 }
264 if (parsed_model_flags.mean_value.specified()) {
265 QCHECK(uses_single_input_flags);
266 model_flags->mutable_input_arrays(0)->set_mean_value(
267 parsed_model_flags.mean_value.value());
268 }
269 if (parsed_model_flags.mean_values.specified()) {
270 QCHECK(uses_multi_input_flags);
271 std::vector<string> mean_values =
272 absl::StrSplit(parsed_model_flags.mean_values.value(), ',');
273 QCHECK(mean_values.size() == model_flags->input_arrays_size());
274 for (int i = 0; i < mean_values.size(); ++i) {
275 char* last = nullptr;
276 model_flags->mutable_input_arrays(i)->set_mean_value(
277 strtod(mean_values[i].data(), &last));
278 CHECK(last != mean_values[i].data());
279 }
280 }
281 if (parsed_model_flags.std_value.specified()) {
282 QCHECK(uses_single_input_flags);
283 model_flags->mutable_input_arrays(0)->set_std_value(
284 parsed_model_flags.std_value.value());
285 }
286 if (parsed_model_flags.std_values.specified()) {
287 QCHECK(uses_multi_input_flags);
288 std::vector<string> std_values =
289 absl::StrSplit(parsed_model_flags.std_values.value(), ',');
290 QCHECK(std_values.size() == model_flags->input_arrays_size());
291 for (int i = 0; i < std_values.size(); ++i) {
292 char* last = nullptr;
293 model_flags->mutable_input_arrays(i)->set_std_value(
294 strtod(std_values[i].data(), &last));
295 CHECK(last != std_values[i].data());
296 }
297 }
298 if (parsed_model_flags.input_data_type.specified()) {
299 QCHECK(uses_single_input_flags);
300 IODataType type;
301 QCHECK(IODataType_Parse(parsed_model_flags.input_data_type.value(), &type));
302 model_flags->mutable_input_arrays(0)->set_data_type(type);
303 }
304 if (parsed_model_flags.input_data_types.specified()) {
305 QCHECK(uses_multi_input_flags);
306 std::vector<string> input_data_types =
307 absl::StrSplit(parsed_model_flags.input_data_types.value(), ',');
308 QCHECK(input_data_types.size() == model_flags->input_arrays_size());
309 for (int i = 0; i < input_data_types.size(); ++i) {
310 IODataType type;
311 QCHECK(IODataType_Parse(input_data_types[i], &type));
312 model_flags->mutable_input_arrays(i)->set_data_type(type);
313 }
314 }
315 if (parsed_model_flags.input_shape.specified()) {
316 QCHECK(uses_single_input_flags);
317 if (model_flags->input_arrays().empty()) {
318 model_flags->add_input_arrays();
319 }
320 auto* shape = model_flags->mutable_input_arrays(0)->mutable_shape();
321 shape->clear_dims();
322 const IntList& list = parsed_model_flags.input_shape.value();
323 for (auto& dim : list.elements) {
324 shape->add_dims(dim);
325 }
326 }
327 if (parsed_model_flags.input_shapes.specified()) {
328 QCHECK(uses_multi_input_flags);
329 std::vector<string> input_shapes =
330 absl::StrSplit(parsed_model_flags.input_shapes.value(), ':');
331 QCHECK(input_shapes.size() == model_flags->input_arrays_size());
332 for (int i = 0; i < input_shapes.size(); ++i) {
333 auto* shape = model_flags->mutable_input_arrays(i)->mutable_shape();
334 shape->clear_dims();
335 for (const auto& dim_str : absl::StrSplit(input_shapes[i], ',')) {
336 int size;
337 CHECK(absl::SimpleAtoi(dim_str, &size))
338 << "Failed to parse input_shape: " << input_shapes[i];
339 shape->add_dims(size);
340 }
341 }
342 }
343
344#define READ_MODEL_FLAG(name) \
345 do { \
346 if (parsed_model_flags.name.specified()) { \
347 model_flags->set_##name(parsed_model_flags.name.value()); \
348 } \
349 } while (false)
350
351 READ_MODEL_FLAG(variable_batch);
352
353#undef READ_MODEL_FLAG
354
355 for (const auto& element : parsed_model_flags.rnn_states.value().elements) {
356 auto* rnn_state_proto = model_flags->add_rnn_states();
357 for (const auto& kv_pair : element) {
358 const string& key = kv_pair.first;
359 const string& value = kv_pair.second;
360 if (key == "state_array") {
361 rnn_state_proto->set_state_array(value);
362 } else if (key == "back_edge_source_array") {
363 rnn_state_proto->set_back_edge_source_array(value);
364 } else if (key == "size") {
365 int32 size = 0;
366 CHECK(absl::SimpleAtoi(value, &size));
367 CHECK_GT(size, 0);
368 rnn_state_proto->set_size(size);
369 } else {
370 LOG(FATAL) << "Unknown key '" << key << "' in --rnn_states";
371 }
372 }
373 CHECK(rnn_state_proto->has_state_array() &&
374 rnn_state_proto->has_back_edge_source_array() &&
375 rnn_state_proto->has_size())
376 << "--rnn_states must include state_array, back_edge_source_array and "
377 "size.";
378 }
379
380 for (const auto& element : parsed_model_flags.model_checks.value().elements) {
381 auto* model_check_proto = model_flags->add_model_checks();
382 for (const auto& kv_pair : element) {
383 const string& key = kv_pair.first;
384 const string& value = kv_pair.second;
385 if (key == "count_type") {
386 model_check_proto->set_count_type(value);
387 } else if (key == "count_min") {
388 int32 count = 0;
389 CHECK(absl::SimpleAtoi(value, &count));
390 CHECK_GE(count, -1);
391 model_check_proto->set_count_min(count);
392 } else if (key == "count_max") {
393 int32 count = 0;
394 CHECK(absl::SimpleAtoi(value, &count));
395 CHECK_GE(count, -1);
396 model_check_proto->set_count_max(count);
397 } else {
398 LOG(FATAL) << "Unknown key '" << key << "' in --model_checks";
399 }
400 }
401 }
402
403 model_flags->set_allow_nonascii_arrays(
404 parsed_model_flags.allow_nonascii_arrays.value());
405 model_flags->set_allow_nonexistent_arrays(
406 parsed_model_flags.allow_nonexistent_arrays.value());
407 model_flags->set_change_concat_input_ranges(
408 parsed_model_flags.change_concat_input_ranges.value());
409
410 if (parsed_model_flags.arrays_extra_info_file.specified()) {
411 string arrays_extra_info_file_contents;
412 CHECK(port::file::GetContents(
413 parsed_model_flags.arrays_extra_info_file.value(),
414 &arrays_extra_info_file_contents, port::file::Defaults())
415 .ok());
416 ParseFromStringEitherTextOrBinary(arrays_extra_info_file_contents,
417 model_flags->mutable_arrays_extra_info());
418 }
419}

421ParsedModelFlags* UncheckedGlobalParsedModelFlags(bool must_already_exist) {
422 static auto* flags = [must_already_exist]() {
423 if (must_already_exist) {
424 fprintf(stderr, __FILE__
425 ":"
426 "GlobalParsedModelFlags() used without initialization\n");
427 fflush(stderr);
428 abort();
429 }
430 return new toco::ParsedModelFlags;
431 }();
432 return flags;
433}
434
435ParsedModelFlags* GlobalParsedModelFlags() {
436 return UncheckedGlobalParsedModelFlags(true);
437}
438
439void ParseModelFlagsOrDie(int* argc, char* argv[]) {
440 // TODO(aselle): in the future allow Google version to use
441 // flags, and only use this mechanism for open source
442 auto* flags = UncheckedGlobalParsedModelFlags(false);
443 string msg;
444 bool model_success =
445 toco::ParseModelFlagsFromCommandLineFlags(argc, argv, &msg, flags);
446 if (!model_success || !msg.empty()) {
447 // Log in non-standard way since this happens pre InitGoogle.
448 fprintf(stderr, "%s", msg.c_str());
449 fflush(stderr);
450 abort();
451 }
452}
