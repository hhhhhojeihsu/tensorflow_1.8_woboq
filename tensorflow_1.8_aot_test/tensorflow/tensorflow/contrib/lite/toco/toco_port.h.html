80inline bool ParseFromStringOverload(const std::string& in,
81 TFLITE_PROTO_NS::Message* proto) {
82 return TFLITE_PROTO_NS::TextFormat::ParseFromString(in, proto);
83}
84
85template <typename Proto>
86bool ParseFromStringEitherTextOrBinary(const std::string& input_file_contents,
87 Proto* proto) {
88 if (proto->ParseFromString(input_file_contents)) {
89 return true;
90 }
91
92 if (ParseFromStringOverload(input_file_contents, proto)) {
93 return true;
94 }
95
96 return false;
97}
