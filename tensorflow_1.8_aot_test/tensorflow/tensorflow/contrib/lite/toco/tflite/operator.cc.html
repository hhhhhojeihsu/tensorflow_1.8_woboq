29class AveragePool
30 : public BuiltinOperator<AveragePoolOperator, ::tflite::Pool2DOptions,
31 ::tflite::BuiltinOptions_Pool2DOptions> {
32 public:
33 using BuiltinOperator::BuiltinOperator;
34
35 flatbuffers::Offset<TfLiteOptions> WriteOptions(
36 const TocoOperator& op,
37 flatbuffers::FlatBufferBuilder* builder) const override {
38 auto padding = Padding::Serialize(op.padding.type);
39 auto activation_function =
40 ActivationFunction::Serialize(op.fused_activation_function);
41 return ::tflite::CreatePool2DOptions(*builder, padding, op.stride_width,
42 op.stride_height, op.kwidth,
43 op.kheight, activation_function);
44 }
45
46 void ReadOptions(const TfLiteOptions& options,
47 TocoOperator* op) const override {
48 op->padding.type = Padding::Deserialize(options.padding());
49 op->stride_width = options.stride_w();
50 op->stride_height = options.stride_h();
51 op->kwidth = options.filter_width();
52 op->kheight = options.filter_height();
53 op->fused_activation_function =
54 ActivationFunction::Deserialize(options.fused_activation_function());
55 }
56};

665class TensorFlowUnsupported : public BaseOperator {
666 public:
667 using BaseOperator::BaseOperator;
668
669 Options Serialize(const Operator& op,
670 flatbuffers::FlatBufferBuilder* builder) const override {
671 auto fbb =
672 WriteOptions(static_cast<const TensorFlowUnsupportedOperator&>(op));
673 if (fbb) {
674 return Options::Custom(builder->CreateVector(fbb->GetBuffer()));
675 } else {
676 return Options::Custom(0);
677 }
678 }
679
680 std::unique_ptr<Operator> Deserialize(
681 const BuiltinOptions* builtin_options,
682 const CustomOptions* custom_options) const override {
683 auto op = absl::make_unique<TensorFlowUnsupportedOperator>();
684 if (custom_options) {
685 auto flexbuffer_map =
686 flexbuffers::GetRoot(custom_options->data(), custom_options->size())
687 .AsMap();
688 ReadOptions(flexbuffer_map, op.get());
689 }
690 return std::unique_ptr<Operator>(op.release());
691 }

693 std::unique_ptr<flexbuffers::Builder> WriteOptions(
694 const TensorFlowUnsupportedOperator& op) const {
695 auto fbb = absl::make_unique<flexbuffers::Builder>();
696
697 ::tensorflow::NodeDef node_def;
698 if (!node_def.ParseFromString(op.tensorflow_node_def)) {
699 LOG(ERROR) << "Failed to parse TensorFlow NodeDef";
700 return std::unique_ptr<flexbuffers::Builder>();
701 }
702
703 bool has_valid_attr = false;
704 size_t map_start = fbb->StartMap();
705 for (const auto& pair : node_def.attr()) {
706 const char* key = pair.first.c_str();
707 const auto& attr = pair.second;
708 switch (attr.value_case()) {
709 case ::tensorflow::AttrValue::kS:
710 fbb->String(key, attr.s());
711 has_valid_attr = true;
712 break;
713 case ::tensorflow::AttrValue::kI:
714 fbb->Int(key, attr.i());
715 has_valid_attr = true;
716 break;
717 case ::tensorflow::AttrValue::kF:
718 fbb->Float(key, attr.f());
719 has_valid_attr = true;
720 break;
721 case ::tensorflow::AttrValue::kB:
722 fbb->Bool(key, attr.b());
723 has_valid_attr = true;
724 break;
725 default:
726 LOG(WARNING) << "Ignoring unsupported attribute type with key '"
727 << key << "'";
728 break;
729 }
730 }
731 if (!has_valid_attr) {
732 return std::unique_ptr<flexbuffers::Builder>();
733 }
734 fbb->EndMap(map_start);
735 fbb->Finish();
736 return std::unique_ptr<flexbuffers::Builder>(fbb.release());
737 }
738
739 void ReadOptions(const flexbuffers::Map& m,
740 TensorFlowUnsupportedOperator* op) const {
741 ::tensorflow::NodeDef node_def;
742 auto attr = node_def.mutable_attr();
743
744 const auto& keys = m.Keys();
745 for (size_t i = 0; i < keys.size(); ++i) {
746 const auto key = keys[i].AsKey();
747 const auto& value = m[key];
748 switch (value.GetType()) {
749 case flexbuffers::TYPE_STRING:
750 (*attr)[key].set_s(value.AsString().c_str());
751 break;
752 case flexbuffers::TYPE_INT:
753 (*attr)[key].set_i(value.AsInt64());
754 break;
755 case flexbuffers::TYPE_FLOAT:
756 (*attr)[key].set_f(value.AsFloat());
757 break;
758 case flexbuffers::TYPE_BOOL:
759 (*attr)[key].set_b(value.AsBool());
760 break;
761 default:
762 LOG(WARNING) << "Ignoring unsupported attribute type with key '"
763 << key << "'";
764 break;
765 }
766 }
767 node_def.SerializeToString(&op->tensorflow_node_def);
768 }
769};
