33namespace details {

35// A maps from tensor name to its final position in the TF Lite buffer.
36using TensorsMap = std::unordered_map<string, int>;
37
38// A key to identify an operator.
39// Only when `type` is `kTensorFlowUnsupported`, `custom_code` is filled to
40// identify which operation is used.
41struct OperatorKey {
42 OperatorKey(OperatorType type, const std::string& custom_code)
43 : type(type), custom_code(custom_code) {}
44 const OperatorType type;
45 const std::string custom_code;
46
47 bool operator<(const OperatorKey& other) const {
48 if (type < other.type) return true;
49 if (type > other.type) return false;
50 return custom_code < other.custom_code;
51 }
52
53 bool operator==(const OperatorKey& other) const {
54 return type == other.type && custom_code == other.custom_code;
55 }
56
57 struct Hash {
58 std::size_t operator()(const OperatorKey& key) const {
59 return std::hash<size_t>()(static_cast<size_t>(key.type)) ^
60 std::hash<std::string>()(key.custom_code);
61 }
62 };
63};
64
65// A maps from operator type to its final position in the TF Lite buffer.
66using OperatorsMap = std::unordered_map<OperatorKey, int, OperatorKey::Hash>;
67
68void LoadTensorsMap(const Model& model, TensorsMap* tensors_map);
69void LoadOperatorsMap(const Model& model, OperatorsMap* operators_map);
