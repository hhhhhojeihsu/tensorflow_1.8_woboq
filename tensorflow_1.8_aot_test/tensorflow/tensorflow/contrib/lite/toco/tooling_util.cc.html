37// Find the longest common prefix of two strings.
38absl::string_view FindLongestCommonPrefix(absl::string_view a,
39 absl::string_view b) {
40 if (a.empty() || b.empty()) return absl::string_view();
41
42 const char* pa = a.data();
43 const char* pb = b.data();
44 size_t count = 0;
45 const size_t limit = std::min(a.size(), b.size());
46 while (count < limit && *pa == *pb) {
47 ++pa;
48 ++pb;
49 ++count;
50 }
51
52 return absl::string_view(a.data(), count);
53}

106 if (GetOpWithInput(model, name)) {

393void LogSummary(int log_level, const Model& model) {
394 VLOG(log_level) << "Operators summary (" << model.operators.size()
395 << " operators):";
396 std::unordered_multiset<OperatorType> ops_by_type;
397 for (const auto& op : model.operators) {
398 ops_by_type.insert(op->type);
399 }
400 auto it = ops_by_type.begin();
401 while (it != ops_by_type.end()) {
402 int count = ops_by_type.count(*it);
403 VLOG(log_level) << " " << OperatorTypeName(*it) << ": " << count;
404 std::advance(it, count);
405 }
406}

408void LogArray(int log_level, const Model& model, const string& name) {
409 const auto& array = model.GetArray(name);
410 VLOG(log_level) << "Array: " << name;
411 VLOG(log_level) << " Data type: " << ArrayDataTypeName(array.data_type);
412 VLOG(log_level) << " Final type: "
413 << ArrayDataTypeName(array.final_data_type);
414 if (array.buffer) {
415 VLOG(log_level) << " Constant Buffer";
416 }
417 if (array.alloc) {
418 VLOG(log_level) << " Transient Alloc";
419 }
420 if (array.has_shape()) {
421 const Shape& array_shape = array.shape();
422 if (array_shape.dimensions_count() == 0) {
423 VLOG(log_level) << " (Zero dimensions)";
424 } else {
425 string message = " Dims: ";
426 bool first = true;
427 for (const int dim : array_shape.dims()) {
428 if (!first) {
429 message += ", ";
430 }
431 first = false;
432 toco::port::AppendF(&message, "%d", dim);
433 }
434 VLOG(log_level) << message;
435 }
436 }
437 if (array.minmax) {
438 VLOG(log_level) << " MinMax: " << array.minmax->min << " .. "
439 << array.minmax->max;
440 }
441 if (array.quantization_params) {
442 VLOG(log_level) << " QuantizationParams: zero_point="
443 << static_cast<int>(array.quantization_params->zero_point)
444 << ", scale=" << array.quantization_params->scale;
445 }
446}

448void DumpGraphvizVideoFrame(const Model& model) {
449 namespace port = toco::port;
450
451 const auto& dump_options = *GraphVizDumpOptions::singleton();
452 if (!dump_options.dump_graphviz_video) {
453 return;
454 }
455 CHECK(!dump_options.dump_graphviz.empty());
456 // TODO(benoitjacob): the static data here means that this function
457 // is stateful, not reentrant, and effectively leaks memory till exit
458 // (since dump_hashes can only grow in size). It also means that it
459 // really only is intended to be called for a single model during the
460 // process' lifetime. So it's not great design at all. The overriding
461 // design aspect here is to make the video-dumping code as unintrusive
462 // and self-contained as possible. Eventually, we'll want to have that
463 // cleaned-up, but that will require some form of general statefulness
464 // in toco (some kind of 'tooling state' data structure) that does
465 // not exist at present, and would be premature to design here just for
466 // this new video-dumping feature.
467 static int dump_id = 0;
468 static std::unordered_set<std::size_t> dump_hashes;
469 string graphviz_dump;
470 DumpGraphviz(model, &graphviz_dump);
471 std::size_t hash = std::hash<string>{}(graphviz_dump);
472 if (!dump_hashes.count(hash)) {
473 LOG(INFO) << "DUMPING GRAPHVIZ VIDEO FRAME: " << dump_id;
474 dump_hashes.insert(hash);
475 CHECK(port::file::SetContents(
476 port::file::JoinPath(
477 dump_options.dump_graphviz,
478 toco::port::StringF("toco_video_%05d.dot", dump_id)),
479 graphviz_dump, port::file::Defaults())
480 .ok());
481 dump_id++;
482 }
483}

485void LogDump(int log_level, const string& message, const Model& model) {
486 namespace port = toco::port;
487 const auto& dump_options = *GraphVizDumpOptions::singleton();
488
489 DumpGraphvizVideoFrame(model);
490 if (!dump_options.dump_graphviz.empty()) {
491 string graphviz_dump;
492
493 DumpGraphviz(model, &graphviz_dump);
494 CHECK(port::file::SetContents(
495 port::file::JoinPath(
496 dump_options.dump_graphviz,
497 absl::StrCat("toco_",
498 absl::StrReplaceAll(message, {{" ", "_"}}),
499 ".dot")),
500 graphviz_dump, port::file::Defaults())
501 .ok());
502 }
503
504 if (!VLOG_IS_ON(log_level)) {
505 return;
506 }
507 VLOG(log_level) << "BEGIN DUMP OF TOCO MODEL (" << message << ")";
508 LogSummary(log_level, model);
509 std::unordered_set<string> already_printed_arrays;
510 for (const auto& op : model.operators) {
511 for (const auto& input : op->inputs) {
512 if (!already_printed_arrays.count(input)) {
513 already_printed_arrays.insert(input);
514 LogArray(log_level, model, input);
515 }
516 }
517 VLOG(log_level) << HelpfulOperatorTypeName(*op) << " :";
518 VLOG(log_level) << " " << FormatArraysList(model, op->inputs) << " -> "
519 << FormatArraysList(model, op->outputs);
520 if (op->fused_activation_function != FusedActivationFunctionType::kNone) {
521 VLOG(log_level) << " (with fused activation function)";
522 }
523 for (const auto& output : op->outputs) {
524 if (!already_printed_arrays.count(output)) {
525 already_printed_arrays.insert(output);
526 LogArray(log_level, model, output);
527 }
528 }
529 }
530 VLOG(log_level) << "END DUMP OF TOCO MODEL (" << message << ")";
531}

559bool ShapesAgreeUpToBroadcasting(const Shape& shape0, const Shape& shape1) {
560 CheckShapeDimensions(shape0);
561 CheckShapeDimensions(shape1);
562
563 const Shape* longer = &shape0;
564 const Shape* shorter = &shape1;
565 if (shape1.dimensions_count() > shape0.dimensions_count()) {
566 longer = &shape1;
567 shorter = &shape0;
568 }
569
570 // Walk dimensions back to front until we run out of dimensions in the shorter
571 // shape.
572 int longer_index = longer->dimensions_count() - 1;
573 int shorter_index = shorter->dimensions_count() - 1;
574 while (shorter_index >= 0) {
575 const int d_long = longer->dims(longer_index);
576 const int d_short = shorter->dims(shorter_index);
577 // Broadcasting fails if the dimensions are different *and* neither is 1.
578 if ((d_long != d_short) && (d_long != 1) && (d_short != 1)) {
579 return false;
580 }
581 longer_index--;
582 shorter_index--;
583 }
584 return true;
585}

587bool ShapesAgreeUpToExtending(const Shape& shape0, const Shape& shape1) {
588 CheckShapeDimensions(shape0);
589 CheckShapeDimensions(shape1);
590
591 const Shape* longer = &shape0;
592 const Shape* shorter = &shape1;
593 if (shape1.dimensions_count() > shape0.dimensions_count()) {
594 longer = &shape1;
595 shorter = &shape0;
596 }
597
598 // Walk dimensions back to front until we run out of dimensions in the shorter
599 // shape.
600 int longer_index = longer->dimensions_count() - 1;
601 int shorter_index = shorter->dimensions_count() - 1;
602 while (shorter_index >= 0) {
603 const int d_long = longer->dims(longer_index);
604 const int d_short = shorter->dims(shorter_index);
605 // Extending fails if the dimensions are different.
606 if (d_long != d_short) {
607 return false;
608 }
609 longer_index--;
610 shorter_index--;
611 }
612
613 // The remaining dimensions in the longer shape must be 1.
614 while (longer_index >= 0) {
615 const int d_long = longer->dims(longer_index);
616 if (d_long != 1) {
617 return false;
618 }
619 longer_index--;
620 }
621
622 return true;
623}

854 std::unordered_set<string> arrays_behind_us;
855 for (const auto& array_entry : model.GetArrayMap()) {
856 if (!GetOpWithOutput(model, array_entry.first)) {
857 arrays_behind_us.insert(array_entry.first);
858 }
859 }
860 arrays_behind_us.insert(model.optional_arrays.begin(),
861 model.optional_arrays.end());
862 for (const auto& op : model.operators) {
863 for (const auto& input : op->inputs) {
864 if (!IsConstantParameterArray(model, input)) {
865 CHECK(arrays_behind_us.count(input));
866 }
867 }
868 for (const auto& output : op->outputs) {
869 CHECK(!arrays_behind_us.count(output));
870 arrays_behind_us.insert(output);
871 }
872 }
873 for (const string& output_array : model.flags.output_arrays()) {
874 CHECK(arrays_behind_us.count(output_array));
875 }

878void FixOperatorOrdering(Model* model) {
879 std::unordered_set<string> arrays_behind_us;
880 for (const auto& array_entry : model->GetArrayMap()) {
881 if (!GetOpWithOutput(*model, array_entry.first)) {
882 arrays_behind_us.insert(array_entry.first);
883 }
884 }
885 arrays_behind_us.insert(model->optional_arrays.begin(),
886 model->optional_arrays.end());
887 std::vector<std::unique_ptr<Operator>> old_operators;
888 std::swap(old_operators, model->operators);
889 std::set<std::size_t> remaining;
890 for (std::size_t i = 0; i < old_operators.size(); i++) {
891 remaining.insert(i);
892 }
893 std::unordered_map<string, string> reason_why_leftover;
894 while (true) {
895 bool inserted_something = false;
896 for (auto i : remaining) {
897 bool can_insert = true;
898 auto& op = old_operators[i];
899 CHECK(op.get());
900 for (const auto& input : op->inputs) {
901 if (!IsConstantParameterArray(*model, input) &&
902 !arrays_behind_us.count(input)) {
903 for (const string& output : op->outputs) {
904 reason_why_leftover[output] = input;
905 }
906 can_insert = false;
907 break;
908 }
909 }
910 if (can_insert) {
911 model->operators.emplace_back(nullptr);
912 for (const auto& output : op->outputs) {
913 arrays_behind_us.insert(output);
914 }
915 std::swap(op, model->operators.back());
916 remaining.erase(i);
917 inserted_something = true;
918 break;
919 }
920 }
921 if (!inserted_something) {
922 break;
923 }
924 }
925 if (!remaining.empty()) {
926 LOG(ERROR)
927 << "No viable ordering of operators was found. "
928 << "Here is a 'backtrace' of at least one part of the graph that is "
929 << "problematic. It starts with the first operator that has as "
930 << "problematic input array, and then walks back the graph to "
931 << "the operator that produced that input array, etc., until we find "
932 << "the root cause:";
933 LOG(ERROR) << "BEGIN TRACE OF OPERATOR WITH BAD INPUT";
934 LOG(ERROR) << "Here is the first-encountered operator with a bad input: ";
935 const Operator* bad_op = old_operators[*remaining.begin()].get();
936 std::unordered_set<string> bad_inputs_already_traced;
937 // The following while(true) loop should always end with a LOG(FATAL).
938 while (true) {
939 LOG(ERROR) << HelpfulOperatorTypeName(*bad_op) << " : "
940 << FormatArraysList(*model, bad_op->inputs) << " -> "
941 << FormatArraysList(*model, bad_op->outputs);
942 bool found_bad_output = false;
943 string bad_output;
944 for (const string& output : bad_op->outputs) {
945 if (reason_why_leftover.count(output)) {
946 found_bad_output = true;
947 bad_output = output;
948 break;
949 }
950 }
951 CHECK(found_bad_output);
952 const string& bad_input = reason_why_leftover[bad_output];
953 LOG(ERROR) << "The bad input here is: " << bad_input;
954 if (bad_inputs_already_traced.count(bad_input)) {
955 LOG(FATAL)
956 << "Cycle found! We already encountered that "
957 << "input array, " << bad_input << ", earlier in the "
958 << "above trace! We expect graphs to be acyclic, even "
959 << "RNNs. Let us know if some graph actually needs to have "
960 << "cycles, but first, please check if it really is "
961 << "an *inference* graph. *Training* graphs are out-of-scope "
962 << "for toco.";
963 }
964 bad_inputs_already_traced.insert(bad_input);
965 bad_op = nullptr;
966 for (auto i : remaining) {
967 const Operator* op = old_operators[i].get();
968 for (const string& output : op->outputs) {
969 if (bad_input == output) {
970 bad_op = op;
971 break;
972 }
973 }
974 if (bad_op) {
975 break;
976 }
977 }
978 if (!bad_op) {
979 LOG(ERROR) << "And that's the root cause: "
980 << "that array, " << bad_input << ", isn't produced by any "
981 << "operator, or provided in any other way.";
982 LOG(ERROR) << "END TRACE OF OPERATOR WITH BAD INPUT";
983 LOG(FATAL) << "(The above was a multi-line fatal error)";
984 }
985 LOG(ERROR) << "And that array is the output of the following operator:";
986 }
987 }
988 CHECK(remaining.empty())
989 << "Should never get here! In case of bad graph, "
990 << "the above code should have generated a FATAL error already!";
991}

1098void CloneArray(Model* model, const string& source_array_name,
1099 const string& target_array_name) {
1100 CHECK(!model->HasArray(target_array_name));
1101 const Array& source_array = model->GetArray(source_array_name);
1102 Array& target_array = model->GetOrCreateArray(target_array_name);
1103
1104 switch (source_array.data_type) {
1105 case ArrayDataType::kBool:
1106 CopyArrayBuffer<ArrayDataType::kBool>(source_array, &target_array);
1107 break;
1108 case ArrayDataType::kFloat:
1109 CopyArrayBuffer<ArrayDataType::kFloat>(source_array, &target_array);
1110 break;
1111 case ArrayDataType::kInt8:
1112 CopyArrayBuffer<ArrayDataType::kInt8>(source_array, &target_array);
1113 break;
1114 case ArrayDataType::kUint8:
1115 CopyArrayBuffer<ArrayDataType::kUint8>(source_array, &target_array);
1116 break;
1117 case ArrayDataType::kInt16:
1118 CopyArrayBuffer<ArrayDataType::kInt16>(source_array, &target_array);
1119 break;
1120 case ArrayDataType::kUint16:
1121 CopyArrayBuffer<ArrayDataType::kUint16>(source_array, &target_array);
1122 break;
1123 case ArrayDataType::kInt32:
1124 CopyArrayBuffer<ArrayDataType::kInt32>(source_array, &target_array);
1125 break;
1126 case ArrayDataType::kUint32:
1127 CopyArrayBuffer<ArrayDataType::kUint32>(source_array, &target_array);
1128 break;
1129 case ArrayDataType::kInt64:
1130 CopyArrayBuffer<ArrayDataType::kInt64>(source_array, &target_array);
1131 break;
1132 case ArrayDataType::kUint64:
1133 CopyArrayBuffer<ArrayDataType::kUint64>(source_array, &target_array);
1134 break;
1135 case ArrayDataType::kString:
1136 CopyArrayBuffer<ArrayDataType::kString>(source_array, &target_array);
1137 break;
1138 default:
1139 LOG(FATAL) << "Unsupported data type: "
1140 << ArrayDataTypeName(source_array.data_type);
1141 return;
1142 }
1143
1144 if (source_array.minmax) {
1145 const auto& smm = source_array.GetMinMax();
1146 auto& tmm = target_array.GetOrCreateMinMax();
1147 tmm.min = smm.min;
1148 tmm.max = smm.max;
1149 }
1150
1151 if (source_array.quantization_params) {
1152 const auto& sqp = source_array.GetQuantizationParams();
1153 auto& tqp = target_array.GetOrCreateQuantizationParams();
1154 tqp.zero_point = sqp.zero_point;
1155 tqp.scale = sqp.scale;
1156 }
1157
1158 target_array.data_type = source_array.data_type;
1159 target_array.final_data_type = source_array.final_data_type;
1160
1161 target_array.copy_shape(source_array.shape());
1162}

1209void ResolveModelFlags(const ModelFlags& model_flags, Model* model) {
1210 // Merge info about input_arrays from model_flags into model->flags
1211 for (const auto& specified_input_array : model_flags.input_arrays()) {
1212 toco::InputArray* dst_input_array = nullptr;
1213 for (int i = 0; i < model->flags.input_arrays_size(); i++) {
1214 toco::InputArray* candidate_dst_input_array =
1215 model->flags.mutable_input_arrays(i);
1216 if (candidate_dst_input_array->name() == specified_input_array.name()) {
1217 // specified_input_array from model_flags maps to dst_input_array
1218 // in model->flags
1219 dst_input_array = candidate_dst_input_array;
1220 break;
1221 }
1222 }
1223 if (!dst_input_array) {
1224 // Specified_input_array from model_flags is not found in model->flags.
1225 // Match a name-less specified input array when there can be no ambiguity
1226 // as there is only 1 input array.
1227 if (model->flags.input_arrays_size() == 1 &&
1228 model_flags.input_arrays_size() == 1 &&
1229 !specified_input_array.has_name()) {
1230 dst_input_array = model->flags.mutable_input_arrays(0);
1231 }
1232 }
1233 if (!dst_input_array) {
1234 // Still no match, so create a new input array to copy
1235 // specified_input_array into.
1236 dst_input_array = model->flags.add_input_arrays();
1237 dst_input_array->set_name(specified_input_array.name());
1238 }
1239
1240#define RESOLVE_MODEL_FLAG(field_name) \
1241 if (specified_input_array.has_##field_name()) { \
1242 if (dst_input_array->has_##field_name()) { \
1243 QCHECK_EQ(dst_input_array->field_name(), \
1244 specified_input_array.field_name()) \
1245 << "For input array '" << dst_input_array->name() << "', " \
1246 << "specified " #field_name " flag with value: " \
1247 << specified_input_array.field_name() \
1248 << " does not agree with already defined " #field_name \
1249 " of this model, with value: " \
1250 << specified_input_array.field_name(); \
1251 } else { \
1252 dst_input_array->set_##field_name(specified_input_array.field_name()); \
1253 } \
1254 }
1255 RESOLVE_MODEL_FLAG(std_value);
1256 RESOLVE_MODEL_FLAG(mean_value);
1257#undef RESOLVE_MODEL_FLAG
1258
1259 if (specified_input_array.has_shape()) {
1260 if (dst_input_array->has_shape()) {
1261 QCHECK_EQ(specified_input_array.shape().dims_size(),
1262 dst_input_array->shape().dims_size())
1263 << "For input array '" << specified_input_array.name() << "', "
1264 << "size of specified input shape flag with size: "
1265 << specified_input_array.shape().dims_size()
1266 << " does not agree with already defined input shape"
1267 " of this model, with size: "
1268 << dst_input_array->shape().dims_size();
1269 // We treat the first dimension as a special case, since it is often
1270 // a batch size and the input_shape flag is effectively overriding
1271 // the model.
1272 for (int i = 1; i < specified_input_array.shape().dims_size(); i++) {
1273 QCHECK_EQ(specified_input_array.shape().dims(i),
1274 dst_input_array->shape().dims(i))
1275 << "At dimension number " << i << " of input array "
1276 << specified_input_array.name() << ", the specified shape's "
1277 << "dimension flag with dimension: "
1278 << specified_input_array.shape().dims(i)
1279 << " does not agree with already defined shape"
1280 << " of this model, with dimension: "
1281 << dst_input_array->shape().dims(i);
1282 }
1283 } else {
1284 *dst_input_array->mutable_shape() = specified_input_array.shape();
1285 }
1286 }
1287
1288 if (specified_input_array.has_data_type()) {
1289 QCHECK(!dst_input_array->has_data_type());
1290 dst_input_array->set_data_type(specified_input_array.data_type());
1291 }
1292 }
1293
1294 if (model_flags.output_arrays_size() > 0) {
1295 model->flags.mutable_output_arrays()->CopyFrom(model_flags.output_arrays());
1296 }
1297
1298#define RESOLVE_MODEL_FLAG(name) \
1299 if (model_flags.has_##name()) { \
1300 if (model->flags.has_##name()) { \
1301 QCHECK_EQ(model_flags.name(), model->flags.name()) \
1302 << "Specified " #name " flag with value: " << model_flags.name() \
1303 << " does not agree with already defined " #name \
1304 " of this model, with value: " \
1305 << model->flags.name(); \
1306 } else { \
1307 model->flags.set_##name(model_flags.name()); \
1308 } \
1309 }
1310
1311 RESOLVE_MODEL_FLAG(variable_batch)
1312
1313#undef RESOLVE_MODEL_FLAG
1314
1315 if (!model_flags.rnn_states().empty()) {
1316 model->flags.mutable_rnn_states()->CopyFrom(model_flags.rnn_states());
1317 }
1318
1319 if (model->flags.model_checks_size() == 0) {
1320 model->flags.mutable_model_checks()->CopyFrom(model_flags.model_checks());
1321 }
1322
1323 QCHECK_GT(model->flags.output_arrays_size(), 0)
1324 << "This model does not define output arrays, so a "
1325 "--output_arrays flag must be given on the command-line.";
1326
1327 for (auto& input_array_proto : *model->flags.mutable_input_arrays()) {
1328 auto& input_array = model->GetOrCreateArray(input_array_proto.name());
1329 if (input_array_proto.has_data_type()) {
1330 const ArrayDataType specified_type =
1331 ConvertIODataTypeToArrayDataType(input_array_proto.data_type());
1332 QCHECK(specified_type != ArrayDataType::kNone);
1333 if (input_array.data_type != ArrayDataType::kNone) {
1334 QCHECK(specified_type == input_array.data_type)
1335 << "For input array " << input_array_proto.name()
1336 << " the specified input data type "
1337 << IODataType_Name(input_array_proto.data_type())
1338 << " conflicts with the existing type.";
1339 }
1340 input_array.data_type = specified_type;
1341 }
1342
1343 if (input_array.data_type == ArrayDataType::kNone) {
1344 // We start out with a float input array;
1345 // that may get replaced by a uint8 array later, by
1346 // MakeInitialDequantizeOp.
1347 input_array.data_type = ArrayDataType::kFloat;
1348 }
1349
1350 // Compare/merge the model->flags describing the input_shape with
1351 // the actual input array's shape.
1352 if (!input_array.has_shape()) {
1353 if (input_array_proto.has_shape()) {
1354 auto& input_array_dims = *input_array.mutable_shape()->mutable_dims();
1355 for (auto dim : input_array_proto.shape().dims()) {
1356 CHECK_GE(dim, 1);
1357 input_array_dims.push_back(dim);
1358 }
1359 }
1360 } else {
1361 if (input_array_proto.has_shape()) {
1362 // If an input shape was specified on the flags ensure that it matches
1363 // the actual shape in the model.
1364 const auto& input_array_dims =
1365 *input_array.mutable_shape()->mutable_dims();
1366 CHECK_EQ(input_array_dims.size(),
1367 input_array_proto.shape().dims_size());
1368 for (int i = 0; i < input_array_dims.size(); i++) {
1369 CHECK_EQ(input_array_dims[i], input_array_proto.shape().dims(i));
1370 }
1371 } else {
1372 for (int i = 0; i < input_array.shape().dimensions_count(); i++) {
1373 input_array_proto.mutable_shape()->add_dims(
1374 input_array.shape().dims(i));
1375 }
1376 }
1377 }
1378
1379 const float mean_value = input_array_proto.mean_value();
1380 const float std_value = input_array_proto.std_value();
1381 MinMax input_minmax;
1382 float qmin = 0, qmax = 255;
1383 if (input_array.data_type == ArrayDataType::kInt16) {
1384 qmin = -32768;
1385 qmax = 32767;
1386 }
1387 input_minmax.min = (qmin - mean_value) / std_value;
1388 input_minmax.max = (qmax - mean_value) / std_value;
1389 if (input_array.minmax) {
1390 if (input_array_proto.has_mean_value() ||
1391 input_array_proto.has_std_value()) {
1392 const double width = input_minmax.max - input_minmax.min;
1393 const double kMinMaxAllowedDiff = 1e-6 * width;
1394 CHECK(std::abs(input_minmax.min - input_array.minmax->min) <
1395 kMinMaxAllowedDiff &&
1396 std::abs(input_minmax.max - input_array.minmax->max) <
1397 kMinMaxAllowedDiff)
1398 << input_minmax.min << ", " << input_minmax.max
1399 << " != " << input_array.minmax->min << ", "
1400 << input_array.minmax->max;
1401 }
1402 } else {
1403 input_array.GetOrCreateMinMax() = input_minmax;
1404 }
1405 }
1406 // Creation of the RNN state arrays
1407 for (const auto& rnn_state : model->flags.rnn_states()) {
1408 CreateOrCheckRnnStateArray(rnn_state.state_array(), rnn_state.size(),
1409 model);
1410 }
1411
1412 for (const auto& input_array : model->flags.input_arrays()) {
1413 if (input_array.has_shape()) {
1414 CHECK(input_array.shape().dims_size());
1415 }
1416 }
1417 model->flags.set_change_concat_input_ranges(
1418 model_flags.change_concat_input_ranges());
1419 model->flags.set_allow_nonascii_arrays(model_flags.allow_nonascii_arrays());
1420 model->flags.set_allow_nonexistent_arrays(
1421 model_flags.allow_nonexistent_arrays());
1422
1423 CHECK(!model->flags.has_arrays_extra_info());
1424 *model->flags.mutable_arrays_extra_info() = model_flags.arrays_extra_info();
1425}

1427void CheckIsReadyForQuantization(const Model& model) {
1428 for (const auto& op : model.operators) {
1429 for (const auto& input : op->inputs) {
1430 const auto& input_array = model.GetArray(input);
1431 if (input_array.data_type != ArrayDataType::kFloat) {
1432 // The array is not floats, no quantization needed.
1433 continue;
1434 }
1435 if (input_array.minmax) {
1436 // The array has minmax, we're good.
1437 continue;
1438 }
1439 if (input_array.buffer) {
1440 // The array has a constant buffer, so we can
1441 // fall back to computing the minmax from actual array entries
1442 // (with a WARNING about possible accuracy implications).
1443 continue;
1444 }
1445 LOG(FATAL)
1446 << "Array " << input << ", which is an input to the "
1447 << HelpfulOperatorTypeName(*op) << " operator producing the output "
1448 << "array " << op->outputs[0] << ", is lacking min/max data, "
1449 << "which is necessary for quantization. Either target a "
1450 << "non-quantized output format, or change the input graph to "
1451 << "contain min/max information, or pass --default_ranges_min= and "
1452 << "--default_ranges_max= if you do not care about the accuracy of "
1453 << "results.";
1454 }
1455 }

1619bool EstimateArithmeticOpsCount(const Model& model, int64* result) {
1620 int64 total = 0;
1621 for (const auto& op : model.operators) {
1622 switch (op->type) {
1623 case OperatorType::kFullyConnected:
1624 case OperatorType::kConv:
1625 case OperatorType::kDepthwiseConv: {
1626 const auto& output_array = model.GetArray(op->outputs[0]);
1627 const auto& weights_array = model.GetArray(op->inputs[1]);
1628 if (!output_array.has_shape() || !weights_array.has_shape()) {
1629 return false;
1630 }
1631 int cols = 1;
1632 for (int i = 0; i < output_array.shape().dimensions_count() - 1; i++) {
1633 cols *= output_array.shape().dims(i);
1634 }
1635 const int64 cost_per_col =
1636 2 * RequiredBufferSizeForShape(weights_array.shape());
1637 total += cost_per_col * cols;
1638 if (op->inputs.size() > 2) {
1639 // There is a bias vector. One more op per output value.
1640 total += RequiredBufferSizeForShape(output_array.shape());
1641 }
1642 break;
1643 }
1644 case OperatorType::kAdd:
1645 case OperatorType::kSub:
1646 case OperatorType::kMul: {
1647 const auto& output_array = model.GetArray(op->outputs[0]);
1648 if (!output_array.has_shape()) {
1649 return false;
1650 }
1651 total += RequiredBufferSizeForShape(output_array.shape());
1652 break;
1653 }
1654 case OperatorType::kAddN: {
1655 const auto& output_array = model.GetArray(op->outputs[0]);
1656 if (!output_array.has_shape()) {
1657 return false;
1658 }
1659 // AddN cost is roughly the same cost as N-1 Adds.
1660 const int num_adds = op->inputs.size() - 1;
1661 total += num_adds * RequiredBufferSizeForShape(output_array.shape());
1662 break;
1663 }
1664 case OperatorType::kLogistic:
1665 case OperatorType::kSoftmax:
1666 case OperatorType::kLogSoftmax:
1667 case OperatorType::kTanh: {
1668 const auto& output_array = model.GetArray(op->outputs[0]);
1669 if (!output_array.has_shape()) {
1670 return false;
1671 }
1672 // As a very rough ballpark, the cost of evaluating a math function
1673 // such as tanh or logistic is about 32 multiplications, and about as
1674 // many additions/subtractions. (Just a power-of-two order-of-magnitude
1675 // from looking at actual implementations that we use in runtime/ code).
1676 total += 64 * RequiredBufferSizeForShape(output_array.shape());
1677 break;
1678 }
1679 case OperatorType::kMaxPool: {
1680 const auto& maxpool = *static_cast<const MaxPoolOperator*>(op.get());
1681 const auto& output_array = model.GetArray(op->outputs[0]);
1682 if (!output_array.has_shape()) {
1683 return false;
1684 }
1685 total += RequiredBufferSizeForShape(output_array.shape()) *
1686 maxpool.kheight * maxpool.kwidth;
1687 break;
1688 }
1689 case OperatorType::kAveragePool: {
1690 const auto& avgpool =
1691 *static_cast<const AveragePoolOperator*>(op.get());
1692 const auto& output_array = model.GetArray(op->outputs[0]);
1693 if (!output_array.has_shape()) {
1694 return false;
1695 }
1696 total += RequiredBufferSizeForShape(output_array.shape()) *
1697 avgpool.kheight * avgpool.kwidth;
1698 break;
1699 }
1700 case OperatorType::kL2Pool: {
1701 const auto* maxpool = static_cast<const MaxPoolOperator*>(op.get());
1702 const auto& output_array = model.GetArray(op->outputs[0]);
1703 if (!output_array.has_shape()) {
1704 return false;
1705 }
1706 // The sum of squares requires (kheight*kwidth) multiply-adds,
1707 // and then there is the sqrt which we ballpark at 32 ops.
1708 const int64 cost_per_val = 2 * maxpool->kheight * maxpool->kwidth + 32;
1709 total +=
1710 RequiredBufferSizeForShape(output_array.shape()) * cost_per_val;
1711 break;
1712 }
1713 case OperatorType::kL2Normalization: {
1714 const auto& output_array = model.GetArray(op->outputs[0]);
1715 if (!output_array.has_shape()) {
1716 return false;
1717 }
1718 // Computing the squared L2 norm is N multiply-adds so 2N ops,
1719 // then the single inverse-sqrt is negligible, then we multiply each
1720 // value by the resulting multiplier, so an extra N ops. Total 3N ops.
1721 total += 3 * RequiredBufferSizeForShape(output_array.shape());
1722 break;
1723 }
1724 default:
1725 break;
1726 }
1727 }
1728 *result = total;
1729 return true;
1730}

1732void GetShuffleShape(AxesOrder input_axes_order, AxesOrder output_axes_order,
1733 std::vector<int>* shuffle) {
1734 CHECK_EQ(AxesCount(input_axes_order), AxesCount(output_axes_order));
1735 shuffle->resize(4);
1736 for (int i = 0; i < 4; i++) {
1737 (*shuffle)[i] = i;
1738 }
1739 if (input_axes_order == output_axes_order) {
1740 // nothing to do
1741 } else if (AxesCount(input_axes_order) == 2) {
1742 shuffle->resize(2);
1743 (*shuffle)[0] = 1;
1744 (*shuffle)[1] = 0;
1745 } else if (input_axes_order == AxesOrder::kOHWI &&
1746 output_axes_order == AxesOrder::kHWIO) {
1747 // 3210 <- 3210
1748 // HWIO <- OHWI
1749 (*shuffle)[0] = 1;
1750 (*shuffle)[1] = 2;
1751 (*shuffle)[2] = 3;
1752 (*shuffle)[3] = 0;
1753 } else if (input_axes_order == AxesOrder::kHWIO &&
1754 output_axes_order == AxesOrder::kOHWI) {
1755 // 3210 <- 3210
1756 // OHWI <- HWIO
1757 (*shuffle)[0] = 3;
1758 (*shuffle)[1] = 0;
1759 (*shuffle)[2] = 1;
1760 (*shuffle)[3] = 2;
1761 } else {
1762 LOG(FATAL) << "Bad shuffle";
1763 }
1764}

1766void ExtendShuffle(const std::vector<int>& input_shuffle, int newdim,
1767 std::vector<int>* extended_shuffle) {
1768 *extended_shuffle = input_shuffle;
1769 CHECK(newdim >= input_shuffle.size());
1770 const int pad_size = newdim - input_shuffle.size();
1771 extended_shuffle->resize(newdim);
1772 for (int i = 0; i < pad_size; i++) {
1773 (*extended_shuffle)[i] = i;
1774 }
1775 for (int i = pad_size; i < newdim; i++) {
1776 (*extended_shuffle)[i] = input_shuffle[i - pad_size] + pad_size;
1777 }
1778}

1780void ShuffleDims(const Shape& input_shape, AxesOrder input_axes_order,
1781 AxesOrder output_axes_order, Shape* output_shape) {
1782 if (input_axes_order == AxesOrder::kHWIM &&
1783 output_axes_order == AxesOrder::k1HWO) {
1784 // This special case isn't just a permutation, the IM pair of dims get
1785 // merged into the 3 dim, so we have to special-case it.
1786 *output_shape = Shape({1, input_shape.dims(0), input_shape.dims(1),
1787 input_shape.dims(3) * input_shape.dims(2)});
1788 } else {
1789 std::vector<int> shuffle;
1790 GetShuffleShape(input_axes_order, output_axes_order, &shuffle);
1791 std::vector<int>* output_dims = output_shape->mutable_dims();
1792 output_dims->resize(input_shape.dimensions_count());
1793 for (int i = 0; i < input_shape.dimensions_count(); i++) {
1794 (*output_dims)[i] = input_shape.dims(shuffle[i]);
1795 }
1796 }
1797}

1799template <typename T>
1800void ShuffleArrayTemplate(const Shape& input_shape, AxesOrder input_axes_order,
1801 AxesOrder output_axes_order,
1802 const Shape& output_shape, const T* input_data,
1803 T* output_data) {
1804 if (input_axes_order == AxesOrder::kHWIM &&
1805 output_axes_order == AxesOrder::k1HWO) {
1806 // This special case isn't just a permutation, the IM pair of dims get
1807 // merged into the O dim, so we have to special-case it. Fortunately,
1808 // as far as array shuffling is concerned, it's just the identity
1809 // transformation.
1810 memcpy(output_data, input_data,
1811 RequiredBufferSizeForShape(input_shape) * sizeof(output_data[0]));
1812 return;
1813 }
1814 CHECK(input_shape.dimensions_count() == output_shape.dimensions_count());
1815 const int dim = input_shape.dimensions_count();
1816 CHECK_LE(dim, 4);
1817 std::vector<int> shuffle;
1818 GetShuffleShape(input_axes_order, output_axes_order, &shuffle);
1819 CHECK(shuffle.size() >= dim);
1820 for (int i = 0; i < dim; i++) {
1821 CHECK(shuffle[i] >= 0 && shuffle[i] < dim);
1822 CHECK(input_shape.dims(shuffle[i]) == output_shape.dims(i));
1823 }
1824 Shape extended_input_shape = input_shape;
1825 ExtendShape(&extended_input_shape, 4);
1826 Shape extended_output_shape = output_shape;
1827 ExtendShape(&extended_output_shape, 4);
1828 std::vector<int> extended_shuffle;
1829 ExtendShuffle(shuffle, 4, &extended_shuffle);
1830
1831 const std::vector<int>& extended_input_dims = extended_input_shape.dims();
1832 const std::vector<int>& extended_output_dims = extended_output_shape.dims();
1833
1834 // TODO(starka): Rework to handle different numbers of dimensions.
1835 int input_strides[4];
1836 input_strides[3] = 1;
1837 input_strides[2] = extended_input_dims[3];
1838 input_strides[1] = input_strides[2] * extended_input_dims[2];
1839 input_strides[0] = input_strides[1] * extended_input_dims[1];
1840 const int input_stride_0 = input_strides[extended_shuffle[3]];
1841 const int input_stride_1 = input_strides[extended_shuffle[2]];
1842 const int input_stride_2 = input_strides[extended_shuffle[1]];
1843 const int input_stride_3 = input_strides[extended_shuffle[0]];
1844
1845 const int output_size_0 = extended_output_dims[3];
1846 const int output_size_1 = extended_output_dims[2];
1847 const int output_size_2 = extended_output_dims[1];
1848 const int output_size_3 = extended_output_dims[0];
1849 const int output_stride_0 = 1;
1850 const int output_stride_1 = output_size_0;
1851 const int output_stride_2 = output_stride_1 * output_size_1;
1852 const int output_stride_3 = output_stride_2 * output_size_2;
1853
1854 for (int i3 = 0; i3 < output_size_3; i3++) {
1855 const T* const input_ptr_3 = input_data + i3 * input_stride_3;
1856 T* const output_ptr_3 = output_data + i3 * output_stride_3;
1857 for (int i2 = 0; i2 < output_size_2; i2++) {
1858 const T* const input_ptr_2 = input_ptr_3 + i2 * input_stride_2;
1859 T* const output_ptr_2 = output_ptr_3 + i2 * output_stride_2;
1860 for (int i1 = 0; i1 < output_size_1; i1++) {
1861 const T* input_ptr = input_ptr_2 + i1 * input_stride_1;
1862 T* output_ptr = output_ptr_2 + i1 * output_stride_1;
1863 T* const output_ptr_end = output_ptr + output_size_0 * output_stride_0;
1864 while (output_ptr != output_ptr_end) {
1865 *output_ptr = *input_ptr;
1866 input_ptr += input_stride_0;
1867 output_ptr += output_stride_0;
1868 }
1869 }
1870 }

1936bool ReshapeIsEquivalentToTranspose(const Model& model,
1937 const TensorFlowReshapeOperator* op,
1938 bool allow_extra_unary_dims) {
1939 CHECK(!op->shape.empty());
1940 CHECK(model.HasArray(op->inputs[0]));
1941 CHECK(model.HasArray(op->outputs[0]));
1942
1943 const auto& input_array = model.GetArray(op->inputs[0]);
1944 const auto& output_array = model.GetArray(op->outputs[0]);
1945
1946 CHECK(input_array.has_shape());
1947 CHECK(output_array.has_shape());
1948
1949 std::vector<int> in_shape = input_array.shape().dims();
1950 std::vector<int> out_shape = output_array.shape().dims();
1951
1952 // If the reshape changes the number of dimensions so it cannot be interpreted
1953 // as a transpose.
1954 if (!allow_extra_unary_dims && in_shape.size() != out_shape.size()) {
1955 return false;
1956 }
1957
1958 in_shape.erase(std::remove(in_shape.begin(), in_shape.end(), 1),
1959 in_shape.end());
1960 out_shape.erase(std::remove(out_shape.begin(), out_shape.end(), 1),
1961 out_shape.end());
1962 return in_shape == out_shape;
1963}

2015void UseArraysExtraInfo(Model* model, bool quantize_output) {
2016 for (const auto& entry : model->flags.arrays_extra_info().entries()) {
2017 if (!model->HasArray(entry.name())) {
2018 continue;
2019 }
2020 auto& array = model->GetArray(entry.name());
2021 if (entry.has_min() || entry.has_max()) {
2022 CHECK_EQ(entry.has_min(), entry.has_max());
2023 auto& minmax = array.GetOrCreateMinMax();
2024 minmax.min = entry.min();
2025 minmax.max = entry.max();
2026 }
2027 if (entry.has_data_type() && quantize_output) {
2028 array.final_data_type =
2029 ConvertIODataTypeToArrayDataType(entry.data_type());
2030 }
2031 if (entry.has_shape()) {
2032 array.clear_shape();
2033 // Make sure to create the shape even if there are no dims, to
2034 // correctly record 0-D shapes.
2035 array.mutable_shape();
2036 for (int dim : entry.shape().dims()) {
2037 array.mutable_shape()->mutable_dims()->push_back(dim);
2038 }
2039 }
2040 if (entry.has_constant_float_value()) {
2041 CHECK(array.has_shape());
2042 if (array.data_type == ArrayDataType::kFloat) {
2043 auto& data = array.GetMutableBuffer<ArrayDataType::kFloat>().data;
2044 data.resize(RequiredBufferSizeForShape(array.shape()));
2045 for (float& f : data) {
2046 f = entry.constant_float_value();
2047 }
2048 }
2049 }
