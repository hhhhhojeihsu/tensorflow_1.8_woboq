24// Quantization parameters, determining the mapping of quantized values
25// to real values (i.e. determining how quantized values are mathematically
26// interpreted).
27//
28// The correspondence is as follows:
29//
30// real_value = scale * (quantized_value - zero_point);
31//
32// In other words, zero_point designates which quantized value corresponds to
33// the real 0 value, and scale designates the difference between the real values
34// corresponding to consecutive quantized values differing by 1.
35struct QuantizationParams {
36 int32 zero_point = 0;
37 double scale = 0.0;
38};

40template <int N>
41struct Dims {
42 int sizes[N];
43 int strides[N];
44};

46// Gets next index to iterate through a multidimensional array.
47inline bool NextIndex(const int num_dims, const int* dims, int* current) {
48 TFLITE_DCHECK_GT(num_dims, 0);
49 TFLITE_DCHECK(dims != nullptr);
50 TFLITE_DCHECK(current != nullptr);
51 int carry = 1;
52 for (int idx = num_dims - 1; idx >= 0; --idx) {
53 int current_val = current[idx] + carry;
54 TFLITE_DCHECK_GE(dims[idx], current_val);
55 if (dims[idx] == current_val) {
56 current[idx] = 0;
57 } else {
58 current[idx] = current_val;
59 carry = 0;
60 break;
61 }
62 }
63 return (carry == 0);
64}

66// Gets offset of index if reducing on axis. When reducing, the flattened offset
67// will not change, if the input index changes on the given axis. For example,
68// if you have a 3D tensor and you are reducing to 2D by eliminating axis 0,
69// then index (0, 1, 2) and index (1, 1, 2) will map to the same flattened
70// offset.
71// TODO(kanlig): uses Dims to represent dimensions.
72inline size_t ReducedOutputOffset(const int num_dims, const int* dims,
73 const int* index, const int num_axis,
74 const int* axis) {
75 TFLITE_DCHECK_GT(num_dims, 0);
76 TFLITE_DCHECK(dims != nullptr);
77 TFLITE_DCHECK(index != nullptr);
78 size_t offset = 0;
79 for (int idx = 0; idx < num_dims; ++idx) {
80 // if we need to skip this axis
81 bool is_axis = false;
82 if (axis != nullptr) {
83 for (int axis_idx = 0; axis_idx < num_axis; ++axis_idx) {
84 if (idx == axis[axis_idx]) {
85 is_axis = true;
86 break;
87 }
88 }
89 }
90 if (!is_axis) {
91 offset = offset * static_cast<size_t>(dims[idx]) +
92 static_cast<size_t>(index[idx]);
93 }
94 }
95 return offset;
96}

98inline int Offset(const Dims<4>& dims, int i0, int i1, int i2, int i3) {
99 TFLITE_DCHECK(i0 >= 0 && i0 < dims.sizes[0]);
100 TFLITE_DCHECK(i1 >= 0 && i1 < dims.sizes[1]);
101 TFLITE_DCHECK(i2 >= 0 && i2 < dims.sizes[2]);
102 TFLITE_DCHECK(i3 >= 0 && i3 < dims.sizes[3]);
103 return i0 * dims.strides[0] + i1 * dims.strides[1] + i2 * dims.strides[2] +
104 i3 * dims.strides[3];
105}
106
107inline int Offset(const Dims<4>& dims, int* index) {
108 return Offset(dims, index[0], index[1], index[2], index[3]);
109}

111// Get array size, DCHECKing that the dim index is in range.
112template <int N>
113int ArraySize(const Dims<N>& array, int index) {
114 TFLITE_DCHECK(index >= 0 && index < N);
115 return array.sizes[index];
116}
117
118// Get common array size, DCHECKing that they all agree.
119template <typename ArrayType1, typename ArrayType2>
120int MatchingArraySize(const ArrayType1& array1, int index1,
121 const ArrayType2& array2, int index2) {
122 TFLITE_DCHECK_EQ(ArraySize(array1, index1), ArraySize(array2, index2));
123 return ArraySize(array1, index1);
124}
125
126template <typename ArrayType1, typename ArrayType2, typename... Args>
127int MatchingArraySize(const ArrayType1& array1, int index1,
128 const ArrayType2& array2, int index2, Args... args) {
129 TFLITE_DCHECK_EQ(ArraySize(array1, index1), ArraySize(array2, index2));
130 return MatchingArraySize(array1, index1, args...);
131}

133inline int RequiredBufferSizeForDims(const Dims<4>& dims) {
134 int max_offset = 0;
135 for (int i = 0; i < 4; i++) {
136 max_offset += (dims.sizes[i] - 1) * dims.strides[i];
137 }
138 return max_offset + 1;
139}
140
141template <int N>
142bool IsPackedWithoutStrides(const Dims<N>& dims) {
143 int expected_stride = 1;
144 for (int d = 0; d < N; d++) {
145 if (dims.strides[d] != expected_stride) return false;
146 expected_stride *= dims.sizes[d];
147 }
148 return true;
149}
