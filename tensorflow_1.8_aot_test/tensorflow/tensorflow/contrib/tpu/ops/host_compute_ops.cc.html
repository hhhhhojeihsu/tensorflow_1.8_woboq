16#include "tensorflow/core/framework/common_shape_fns.h"

18#include "tensorflow/core/framework/shape_inference.h"

20namespace tensorflow {
21
22REGISTER_OP("_XlaSendFromHost")
23 .Input("inputs: Tinputs")
24 .Input("dynamic_key: string")
25 .Attr("Tinputs: list(type) >= 0")
26 .Attr("key: string")
27 .Attr("device_ordinal: int")
28 .SetIsStateful()
29 .SetShapeFn(::tensorflow::shape_inference::NoOutputs)
30 .Doc(R"doc(
31A placeholder op for multiple values that will be sent from TensorFlow to a
32running XLA computation.
33
34inputs: A list of tensors that will be sent to the XLA computation.
35dynamic_key: The key sent at runtime by the compile node to identify which
36execution the transfer corresponds to.
37Tinputs: The element types of each element in `inputs`.
38key: A key that is unique in the computation and associates the send with the consumer in
39the XLA computation.
40device_ordinal: The device to use.
41)doc");
42
43REGISTER_OP("_XlaRecvAtHost")
44 .Input("dynamic_key: string")
45 .Output("outputs: Toutputs")
46 .Attr("Toutputs: list(type) >= 0")
47 .Attr("key: string")
48 .Attr("device_ordinal: int")
49 .SetIsStateful()
50 .SetShapeFn(::tensorflow::shape_inference::UnknownShape)
51 .Doc(R"doc(
52A placeholder op for multiple values that will be sent to TensorFlow from a
53running XLA computation.
54
55dynamic_key: The key sent at runtime by the compile node to identify which
56execution the transfer corresponds to.
57outputs: A list of tensors that will be received from the XLA computation.
58Toutputs: The element types of each element in `outputs`.
59key: A key that is unique in the computation and associates the send with the consumer in
60the XLA computation.
61device_ordinal: The device to use.
62)doc");
