20namespace tensorflow {

22using shape_inference::InferenceContext;
23using shape_inference::ShapeHandle;

25REGISTER_OP("TPUReplicateMetadata")
26 .Attr("num_replicas: int >= 0")
27 .Attr("topology: string = \"\"")
28 .Attr("device_assignment: list(int) = []")
29 .Attr("computation_shape: list(int) = []")
30 .Attr("host_compute_core: list(string) = []")
31 .SetShapeFn(shape_inference::UnknownShape);
32
33REGISTER_OP("TPUReplicatedInput")
34 .Input("inputs: N * T")
35 .Output("output: T")
36 .Attr("N: int >= 1")
37 .Attr("T: type")
38 .SetShapeFn([](InferenceContext* c) {
39 ShapeHandle cur = c->input(c->num_inputs() - 1);
40 for (int i = c->num_inputs() - 2; i >= 0; --i) {
41 TF_RETURN_WITH_CONTEXT_IF_ERROR(c->Merge(c->input(i), cur, &cur),
42 "From merging shape ", i,
43 " with other shapes.");
44 }
45 c->set_output(0, cur);
46 return Status::OK();
47 })
48 .Doc(
49 "Operator that connects N unreplicated inputs to an N-way "
50 "replicated TPU computation.");
51
52REGISTER_OP("TPUReplicatedOutput")
53 .Input("input: T")
54 .Output("outputs: num_replicas * T")
55 .Attr("num_replicas: int >= 1")
56 .Attr("T: type")
57 .SetShapeFn([](InferenceContext* c) {
58 for (int i = 0; i < c->num_outputs(); ++i) {
59 c->set_output(i, c->input(0));
60 }
61 return Status::OK();
62 })
63 .Doc(
64 "Operator that connects the output of an N-way replicated TPU "
65 "computation to N separate outputs.");
66
67REGISTER_OP("TPUReplicate")
68 .Attr("computation: func")
69 .Attr("num_replicas: int >= 1")
70 .Attr("topology: string = \"\"")
71 .Attr("device_assignment: list(int) = []")
72 .Attr("host_compute_core: list(string) = []")
73 .Attr("computation_shape: list(int) = []")
74 .Attr("Tinputs: list(type) >= 0")
75 .Attr("Tbroadcast_inputs: list(type) >= 0")
76 .Attr("NumVariables: int >= 0")
77 .Attr("Tguaranteed_constants: list(type) >= 0")
78 .Attr("output_types: list(type) >= 0")
79 .Input("inputs: Tinputs")
80 .Input("broadcast_inputs: Tbroadcast_inputs")
81 .Input("variables: NumVariables * resource")
82 .Input("guaranteed_constants: Tguaranteed_constants")
83 .Output("outputs: output_types")
84 .SetShapeFn(shape_inference::UnknownShape)
85 .Doc(R"doc(
86Runs replicated computations on a distributed TPU system.
87
88computation: a function containing the computation to run.
89num_replicas: the number of replicas of the computation to run.
90topology: A serialized tensorflow.tpu.TopologyProto that describes the TPU
91topology.
92computation_shape: a [mesh_dimension] array describing the shape of each
93 computation replica in numbers of cores in the TPU mesh.
94device_assignment: a flattened array with shape
95 [replica] + computation_shape + [mesh_dimension] that maps the coordinates of
96 logical cores in each replica of a computation to physical coordinates in
97 the TPU topology.
98Tinputs: the types of the arguments to 'computation'.
99inputs: the inputs to 'computation', flattened, in replica-major order.
100Tbroadcast_inputs: the types of the additional arguments to broadcast to all
101 replicas.
102Tguaranteed_constants: the types of the arguments to 'guaranteed_constants'.
103broadcast_inputs: additional arguments to broadcast to all replicas. The
104 broadcast inputs are appended to the per-replica inputs when calling
105 computation.
106guaranteed_constants: arguments which have been guaranteed to not
107change their values during the session lifetime. These contain tensors marked as
108constant using the GuaranteeConstOp.
109output_types: the types of the outputs of 'computation'.
110outputs: the outputs of 'computation'.
111)doc");
