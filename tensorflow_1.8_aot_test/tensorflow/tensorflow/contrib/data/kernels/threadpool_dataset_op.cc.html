24class ThreadPoolResource : public ResourceBase {
25 public:
26 ThreadPoolResource(Env* env, const ThreadOptions& thread_options,
27 const string& name, int num_threads, bool low_latency_hint)
28 : thread_pool_(env, thread_options, name, num_threads, low_latency_hint) {
29 }
30
31 // Schedules fn() for execution in the pool of threads.
32 void Schedule(std::function<void()> fn) {
33 thread_pool_.Schedule(std::move(fn));
34 }
35
36 string DebugString() override { return "ThreadPoolResource"; }
37
38 private:
39 thread::ThreadPool thread_pool_;
40};
41
42// Creates a handle to a ThreadPool resource. Note that we don't use
43// ResourceOpKernel here because the ThreadPoolResource constructor requires
44// access to `OpKernelContext::env()`, which isn't provided by
45// `ResourceOpKernel<T>::CreateResource()`.
46class ThreadPoolHandleOp : public OpKernel {
47 public:
48 explicit ThreadPoolHandleOp(OpKernelConstruction* ctx) : OpKernel(ctx) {
49 OP_REQUIRES_OK(ctx, ctx->GetAttr("display_name", &display_name_));
50 OP_REQUIRES_OK(ctx, ctx->GetAttr("num_threads", &num_threads_));
51 OP_REQUIRES(
52 ctx, num_threads_ > 0,
53 errors::InvalidArgument("`num_threads` must be greater than zero."));
54 }
55
56 // The resource is deleted from the resource manager only when it is private
57 // to kernel. Ideally the resource should be deleted when it is no longer held
58 // by anyone, but it would break backward compatibility.
59 ~ThreadPoolHandleOp() override {
60 if (cinfo_.resource_is_private_to_kernel()) {
61 if (!cinfo_.resource_manager()
62 ->Delete<ThreadPoolResource>(cinfo_.container(), cinfo_.name())
63 .ok()) {
64 // Do nothing; the resource can have been deleted by session resets.
65 }
66 }
67 }
68
69 void Compute(OpKernelContext* ctx) override LOCKS_EXCLUDED(mu_) {
70 mutex_lock l(mu_);
71 if (!initialized_) {
72 ResourceMgr* mgr = ctx->resource_manager();
73 OP_REQUIRES_OK(ctx, cinfo_.Init(mgr, def()));
74 ThreadPoolResource* resource;
75 OP_REQUIRES_OK(ctx, mgr->LookupOrCreate<ThreadPoolResource>(
76 cinfo_.container(), cinfo_.name(), &resource,
77 [this, ctx](ThreadPoolResource** ret)
78 EXCLUSIVE_LOCKS_REQUIRED(mu_) {
79 *ret = new ThreadPoolResource(
80 ctx->env(), {}, display_name_,
81 num_threads_,
82 false /* low_latency_hint */);
83 return Status::OK();
84 }));
85 initialized_ = true;
86 }
87 OP_REQUIRES_OK(ctx, MakeResourceHandleToOutput(
88 ctx, 0, cinfo_.container(), cinfo_.name(),
89 MakeTypeIndex<ThreadPoolResource>()));
90 }
91
92 private:
93 mutex mu_;
94 ContainerInfo cinfo_ GUARDED_BY(mu_);
95 bool initialized_ GUARDED_BY(mu_) = false;
96 string display_name_;
97 int num_threads_;
98};

100class ThreadPoolDatasetOp : public UnaryDatasetOpKernel {
101 public:
102 explicit ThreadPoolDatasetOp(OpKernelConstruction* ctx)
103 : UnaryDatasetOpKernel(ctx) {}
104
105 void MakeDataset(OpKernelContext* ctx, DatasetBase* input,
106 DatasetBase** output) override {
107 ThreadPoolResource* threadpool_resource;
108 OP_REQUIRES_OK(ctx, LookupResource(ctx, HandleFromInput(ctx, 1),
109 &threadpool_resource));
110 core::ScopedUnref unref_iterator(threadpool_resource);
111
112 *output = new Dataset(ctx, input, threadpool_resource);
113 }
114
115 private:
116 class Dataset : public GraphDatasetBase {
117 public:
118 Dataset(OpKernelContext* ctx, const DatasetBase* input,
119 ThreadPoolResource* threadpool)
120 : GraphDatasetBase(ctx), input_(input), threadpool_(threadpool) {
121 input_->Ref();
122 threadpool_->Ref();
123 }
124
125 ~Dataset() override {
126 input_->Unref();
127 threadpool_->Unref();
128 }
129
130 std::unique_ptr<IteratorBase> MakeIterator(
131 const string& prefix) const override {
132 return std::unique_ptr<IteratorBase>(
133 new Iterator({this, strings::StrCat(prefix, "::ThreadPool")}));
134 }
135
136 const DataTypeVector& output_dtypes() const override {
137 return input_->output_dtypes();
138 }
139 const std::vector<PartialTensorShape>& output_shapes() const override {
140 return input_->output_shapes();
141 }
142
143 string DebugString() override { return "ThreadPoolDatasetOp::Dataset"; }
144
145 protected:
146 Status AsGraphDefInternal(OpKernelContext* ctx, DatasetGraphDefBuilder* b,
147 Node** output) const override {
148 return errors::Unimplemented(
149 "Cannot currently serialize the thread pool for a "
150 "ThreadPoolDataset.");
151 }
152
153 private:
154 class Iterator : public DatasetIterator<Dataset> {
155 public:
156 explicit Iterator(const Params& params)
157 : DatasetIterator<Dataset>(params),
158 input_impl_(params.dataset->input_->MakeIterator(params.prefix)) {}
159
160 Status GetNextInternal(IteratorContext* ctx,
161 std::vector<Tensor>* out_tensors,
162 bool* end_of_sequence) override {
163 ThreadPoolResource* pool = dataset()->threadpool_;
164 IteratorContext::Params params;
165 params.env = ctx->env();
166 params.runner = [pool](std::function<void()> c) {
167 pool->Schedule(std::move(c));
168 };
169 params.stats_aggregator_getter = ctx->stats_aggregator_getter();
170 params.lib = ctx->lib();
171 params.function_library = ctx->function_library();
172 params.allocator_getter = ctx->allocator_getter();
173 IteratorContext threadpool_ctx(params);
174 return input_impl_->GetNext(&threadpool_ctx, out_tensors,
175 end_of_sequence);
176 }
177
178 private:
179 std::unique_ptr<IteratorBase> input_impl_;
180 };
181
182 const DatasetBase* const input_;
183 ThreadPoolResource* const threadpool_;
184 };
185};
