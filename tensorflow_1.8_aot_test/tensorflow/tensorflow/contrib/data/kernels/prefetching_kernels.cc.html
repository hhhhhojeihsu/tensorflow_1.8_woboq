27struct BufferElement {
28 // The producer sets `status` if getting the input element fails.
29 Status status;
30 // The buffered data element.
31 std::vector<Tensor> value;
32};
33
34using FunctionBufferCallback = std::function<void(const BufferElement&)>;
35
36class FunctionBufferingResource : public ResourceBase {
37 public:
38 FunctionBufferingResource(FunctionLibraryRuntime* lib,
39 std::unique_ptr<ProcessFunctionLibraryRuntime> pflr,
40 const NameAttrList& func, int64 buffer_size,
41 const string& source_device,
42 const string& target_device,
43 const std::vector<Tensor>& func_args)
44 : lib_(lib),
45 pflr_(std::move(pflr)),
46 func_(func),
47 buffer_size_(buffer_size),
48 source_device_(source_device),
49 target_device_(target_device),
50 func_args_(func_args),
51 handle_(kInvalidHandle),
52 is_buffering_(false),
53 end_of_sequence_(false),
54 cancelled_(false) {}
55
56 ~FunctionBufferingResource() override {
57 Cancel();
58 }
59
60 string DebugString() override {
61 return strings::StrCat("FunctionBufferingResource. Size: ", buffer_size_,
62 "; target_device: ", target_device_);
63 }
64
65 // Instantiates the function the first time it's called. After that it caches
66 // the handle.
67 Status Instantiate() LOCKS_EXCLUDED(mu_) {
68 mutex_lock l(mu_);
69 // Re-use existing handle if it's been set, effectively caching it.
70 if (handle_ != kInvalidHandle) {
71 return Status::OK();
72 }
73 AttrValueMap attr_values = func_.attr();
74 FunctionLibraryRuntime::InstantiateOptions opts;
75 opts.target = target_device_;
76 return lib_->Instantiate(func_.name(), AttrSlice(&attr_values), opts,
77 &handle_);
78 }
79
80 // Returns true if we've got to the end of the sequence and exhausted the
81 // buffer.
82 bool Finished() LOCKS_EXCLUDED(mu_) {
83 mutex_lock l(mu_);
84 return end_of_sequence_ && buffer_.empty();
85 }
86
87 // Cancels any buffering / prefetching going on.
88 void Cancel() LOCKS_EXCLUDED(mu_) {
89 mutex_lock l(mu_);
90 cancelled_ = true;
91 while (is_buffering_) {
92 cond_var_.wait(l);
93 }
94 }
95
96 // Cancels all pending operations and then clears out the state.
97 void Reset() LOCKS_EXCLUDED(mu_) {
98 Cancel();
99 mutex_lock l(mu_);
100 buffer_.clear();
101 requests_.clear();
102 is_buffering_ = false;
103 end_of_sequence_ = false;
104 cancelled_ = false;
105 }
106
107 // If the buffer has anything, runs `callback` on the first element in the
108 // buffer, else schedules the `callback` to be called. Requires `args` and
109 // `lib` in case more function calls need to be scheduled.
110 void MaybeGet(FunctionBufferCallback callback) LOCKS_EXCLUDED(mu_) {
111 bool start_buffering = false;
112 bool produced_output = false;
113 BufferElement buffer_element;
114 {
115 mutex_lock l(mu_);
116 if (!is_buffering_ && !end_of_sequence_) {
117 start_buffering = true;
118 }
119 if (!buffer_.empty()) {
120 produced_output = true;
121 std::swap(buffer_element, buffer_.front());
122 buffer_.pop_front();
123 } else {
124 produced_output = false;
125 requests_.push_back(std::move(callback));
126 }
127 }
128 if (produced_output) {
129 callback(buffer_element);
130 }
131 if (start_buffering) {
132 FillBuffer();
133 }
134 }
135
136 private:
137 void FillBuffer() LOCKS_EXCLUDED(mu_) {
138 FunctionLibraryRuntime::Handle handle;
139 std::vector<FunctionBufferCallback> cancellation_callbacks;
140 std::vector<BufferElement> cancellation_buffer_elements;
141 bool cancelled = false;
142 {
143 mutex_lock l(mu_);
144 handle = handle_;
145 if (cancelled_) {
146 cancelled = true;
147 // Run through and fulfill all pending requests, if possible.
148 while (!requests_.empty()) {
149 if (!buffer_.empty()) {
150 cancellation_buffer_elements.push_back(std::move(buffer_.front()));
151 buffer_.pop_front();
152 cancellation_callbacks.push_back(std::move(requests_.front()));
153 requests_.pop_front();
154 } else {
155 LOG(ERROR) << "Buffer ran out of elements and we couldn't satisfy: "
156 << requests_.size() << " requests";
157 break;
158 }
159 }
160 is_buffering_ = false;
161 } else {
162 is_buffering_ = true;
163 }
164 }
165 if (cancelled) {
166 for (int i = 0; i < cancellation_callbacks.size(); ++i) {
167 cancellation_callbacks[i](cancellation_buffer_elements[i]);
168 }
169 cond_var_.notify_all();
170 return;
171 }
172 FunctionLibraryRuntime::Options opts;
173 // Copied from CapturedFunction::generate_step_id();
174 opts.step_id = -std::abs(static_cast<int64>(random::New64()));
175 opts.source_device = source_device_;
176 AllocatorAttributes arg_alloc_attr;
177 arg_alloc_attr.set_on_host(true);
178 opts.args_alloc_attrs.push_back(arg_alloc_attr);
179 if (opts.source_device != target_device_) {
180 opts.remote_execution = true;
181 }
182 opts.create_rendezvous = true;
183 auto* rets = new std::vector<Tensor>;
184 lib_->Run(opts, handle, func_args_, rets,
185 [this, rets](const Status& status) {
186 FunctionBufferCallback callback = nullptr;
187 BufferElement buffer_front;
188 bool restart_buffering = false;
189 {
190 mutex_lock l(mu_);
191 BufferElement buffer_element;
192 buffer_element.status = status;
193 if (status.ok()) {
194 buffer_element.value.swap(*rets);
195 } else {
196 end_of_sequence_ = true;
197 is_buffering_ = false;
198 }
199 buffer_.push_back(std::move(buffer_element));
200 if (!requests_.empty()) {
201 buffer_front = std::move(buffer_.front());
202 buffer_.pop_front();
203 callback = std::move(requests_.front());
204 requests_.pop_front();
205 }
206 if (buffer_.size() < buffer_size_ && !end_of_sequence_) {
207 restart_buffering = true;
208 } else {
209 // When the buffer is full, we don't want to call
210 // FillBuffer() unless we're in cancellation phase in which
211 // case FillBuffer() will do the final cleanup post
212 // cancellation.
213 if (cancelled_) {
214 restart_buffering = true;
215 }
216 is_buffering_ = false;
217 }
218 }
219 if (callback != nullptr) {
220 callback(buffer_front);
221 }
222 if (restart_buffering) {
223 FillBuffer();
224 }
225 });
226 }
227
228 mutex mu_;
229 FunctionLibraryRuntime* lib_;
230 std::unique_ptr<ProcessFunctionLibraryRuntime> pflr_;
231 NameAttrList func_;
232 const int64 buffer_size_;
233 const string source_device_;
234 const string target_device_;
235 const std::vector<Tensor> func_args_;
236 FunctionLibraryRuntime::Handle handle_ GUARDED_BY(mu_);
237 std::deque<BufferElement> buffer_ GUARDED_BY(mu_);
238 std::deque<FunctionBufferCallback> requests_ GUARDED_BY(mu_);
239 bool is_buffering_ GUARDED_BY(mu_);
240 bool end_of_sequence_ GUARDED_BY(mu_);
241 bool cancelled_ GUARDED_BY(mu_);
242 condition_variable cond_var_;
243};

245class FunctionBufferResourceHandleOp : public OpKernel {
246 public:
247 explicit FunctionBufferResourceHandleOp(OpKernelConstruction* ctx)
248 : OpKernel(ctx), flib_def_(nullptr) {
249 OP_REQUIRES_OK(ctx, ctx->GetAttr("f", &func_));
250 OP_REQUIRES_OK(ctx, ctx->GetAttr("buffer_size", &buffer_size_));
251 OP_REQUIRES_OK(ctx, ctx->GetAttr("container", &container_));
252 OP_REQUIRES_OK(ctx, ctx->GetAttr("shared_name", &name_));
253 }
254
255 ~FunctionBufferResourceHandleOp() override {
256 if (cinfo_.resource_is_private_to_kernel()) {
257 if (!cinfo_.resource_manager()
258 ->Delete<FunctionBufferingResource>(cinfo_.container(),
259 cinfo_.name())
260 .ok()) {
261 // Do nothing; the resource can have been deleted by session resets.
262 }
263 }
264 }
265
266 void Compute(OpKernelContext* ctx) override {
267 const Tensor* string_arg;
268 OP_REQUIRES_OK(ctx, ctx->input("string_arg", &string_arg));
269 std::vector<Tensor> func_args;
270 func_args.push_back(*string_arg);
271
272 // Obtain and canonicalize target_device.
273 const Tensor* target_arg;
274 OP_REQUIRES_OK(ctx, ctx->input("target_device", &target_arg));
275 const string& target_device =
276 DeviceNameUtils::CanonicalizeDeviceName(target_arg->scalar<string>()());
277
278 FunctionLibraryRuntime* lib = ctx->function_library();
279 OP_REQUIRES(ctx, lib != nullptr,
280 errors::Internal("No function library is provided."));
281
282 const string& source_device = ctx->device()->name();
283
284 mutex_lock l(mu_);
285 if (!initialized_) {
286 OP_REQUIRES_OK(ctx, cinfo_.Init(ctx->resource_manager(), def()));
287 FunctionLibraryRuntime* clone_lib;
288 std::unique_ptr<ProcessFunctionLibraryRuntime> pflr;
289 OP_REQUIRES_OK(ctx, lib->Clone(&flib_def_, &pflr, &clone_lib));
290 // Create the resource.
291 FunctionBufferingResource* buffer;
292 OP_REQUIRES_OK(
293 ctx,
294 ctx->resource_manager()->LookupOrCreate<FunctionBufferingResource>(
295 cinfo_.container(), cinfo_.name(), &buffer,
296 [clone_lib, &pflr, &source_device, &target_device, func_args,
297 this](FunctionBufferingResource** ptr) {
298 *ptr = new FunctionBufferingResource(
299 clone_lib, std::move(pflr), func_, buffer_size_,
300 source_device, target_device, func_args);
301 return Status::OK();
302 }));
303 core::ScopedUnref s(buffer);
304 OP_REQUIRES_OK(ctx, buffer->Instantiate());
305 initialized_ = true;
306 }
307
308 OP_REQUIRES_OK(ctx, MakeResourceHandleToOutput(
309 ctx, 0, cinfo_.container(), cinfo_.name(),
310 MakeTypeIndex<FunctionBufferingResource>()));
311 }
312
313 private:
314 mutex mu_;
315 ContainerInfo cinfo_ GUARDED_BY(mu_);
316 bool initialized_ GUARDED_BY(mu_) = false;
317 std::unique_ptr<FunctionLibraryDefinition> flib_def_;
318 NameAttrList func_;
319 int64 buffer_size_;
320 string container_;
321 string name_;
322};

345// Prefetches and fills up a buffer by calling a function that provides the
346// elements to buffer.
347class FunctionBufferingResourceGetNextOp : public AsyncOpKernel {
348 public:
349 explicit FunctionBufferingResourceGetNextOp(OpKernelConstruction* ctx)
350 : AsyncOpKernel(ctx) {}
351
352 ~FunctionBufferingResourceGetNextOp() override {}
353
354 void ComputeAsync(OpKernelContext* ctx, DoneCallback done) override {
355 ResourceHandle handle;
356 OP_REQUIRES_OK_ASYNC(
357 ctx, HandleFromInput(ctx, "function_buffer_resource", &handle), done);
358 FunctionBufferingResource* buffer = nullptr;
359 OP_REQUIRES_OK_ASYNC(
360 ctx, LookupResource<FunctionBufferingResource>(ctx, handle, &buffer),
361 done);
362
363 if (buffer->Finished()) {
364 buffer->Unref();
365 ctx->SetStatus(errors::OutOfRange("end_of_sequence"));
366 done();
367 return;
368 }
369
370 FunctionBufferCallback callback =
371 [ctx, buffer, done](const BufferElement& buffer_element) {
372 Status s = buffer_element.status;
373 if (!s.ok()) {
374 ctx->SetStatus(s);
375 buffer->Unref();
376 done();
377 return;
378 }
379 for (size_t i = 0; i < buffer_element.value.size(); ++i) {
380 ctx->set_output(i, buffer_element.value[i]);
381 }
382 buffer->Unref();
383 done();
384 };
385 buffer->MaybeGet(std::move(callback));
386 }
387};
388

391 .HostMemory("function_buffer_resource"),
392 FunctionBufferingResourceGetNextOp);

404// Resets the FunctionBufferingResource, cancelling all pending requests and
405// clearing out the buffer.
406class FunctionBufferingResourceResetOp : public OpKernel {
407 public:
408 explicit FunctionBufferingResourceResetOp(OpKernelConstruction* ctx)
409 : OpKernel(ctx) {}
410
411 ~FunctionBufferingResourceResetOp() override {}
412
413 void Compute(OpKernelContext* ctx) override {
414 ResourceHandle handle;
415 OP_REQUIRES_OK(ctx,
416 HandleFromInput(ctx, "function_buffer_resource", &handle));
417 FunctionBufferingResource* buffer = nullptr;
418 OP_REQUIRES_OK(
419 ctx, LookupResource<FunctionBufferingResource>(ctx, handle, &buffer));
420 core::ScopedUnref s(buffer);
421
422 buffer->Reset();
423 }
424};

441class IteratorGetDeviceOp : public OpKernel {
442 public:
443 using OpKernel::OpKernel;
444
445 void Compute(OpKernelContext* ctx) override {
446 // NOTE(mrry): We do not currently Validate that the handle
447 // corresponds to a real IteratorResource, because that symbol is
448 // not exposed from the framework library.
449 Tensor* device_name_t;
450 OP_REQUIRES_OK(ctx,
451 ctx->allocate_output(0, TensorShape({}), &device_name_t));
452 // NOTE(mrry): Since the operation's input is a resource, we must be
453 // colocated with it, and so we can simply return the current device's
454 // name without looking at the input.
455 device_name_t->scalar<string>()() = ctx->device()->name();
456 }
457};
