50// The term 'routing function' is synonymous with 'the probability
51// that an instance is routed to each leaf node.' It is defined in
52// 'Deep Neural Decision Forests' by Kontschieder et al.
53REGISTER_OP("StochasticHardRoutingFunction")
54 .Attr("tree_depth: int")
55 .Attr("random_seed: int")
56 .Input("input_data: float")
57 .Input("tree_parameters: float")
58 .Input("tree_biases: float")
59 .Output("path_probability: float")
60 .Output("path: int32")
61 .SetShapeFn([](InferenceContext* c) {
62 ShapeHandle input;
63 TF_RETURN_IF_ERROR(c->WithRankAtLeast(c->input(0), 1, &input));
64 int64 tree_depth;
65 TF_RETURN_IF_ERROR(c->GetAttr("tree_depth", &tree_depth));
66
67 auto out = c->Matrix(c->Dim(input, 0), tree_depth);
68 c->set_output(0, out);
69 c->set_output(1, out);
70 return Status::OK();
71 })
72 .Doc(R"doc(
73 Samples a path for each instance in `input_data` and returns the
74 probability of the path and the path taken.
75
76 tree_depth: The depth of the decision tree.
77 random_seed: The base random seed.
78
79 input_data: The training batch's features as a 2-d tensor; `input_data[i][j]`
80 gives the j-th feature of the i-th input.
81 tree_parameters: `tree_parameters[i]` gives the weight of
82 the logistic regression model that translates from node features to
83 probabilities.
84 tree_biases: `tree_biases[i]` gives the bias of the logistic
85 regression model that translates from node features to
86 probabilities.
87
88 path_probility: `path_probability[i]` gives the probability of reaching each
89 node in `path[i]`.
90 path: `path[i][j]` gives the jth node in the path taken by the ith data
91 instance.
92)doc");
93
94class StochasticHardRoutingFunction : public OpKernel {
95 public:
96 explicit StochasticHardRoutingFunction(OpKernelConstruction* context)
97 : OpKernel(context) {
98 OP_REQUIRES_OK(context, context->GetAttr("tree_depth", &tree_depth_));
99 OP_REQUIRES_OK(context, context->GetAttr("random_seed", &random_seed_));
100 single_rand_ = std::unique_ptr<random::PhiloxRandom>(
101 new random::PhiloxRandom(random_seed_));
102 rng_ = std::unique_ptr<random::SimplePhilox>(
103 new random::SimplePhilox(single_rand_.get()));
104 }
105
106 void Compute(OpKernelContext* context) override {
107 VLOG(1) << "stochastic routing start";
108 const Tensor& input_data = context->input(0);
109 const Tensor& tree_parameters_tensor = context->input(1);
110 const Tensor& tree_biases_tensor = context->input(2);
111
112 if (input_data.shape().dim_size(0) > 0) {
113 OP_REQUIRES(
114 context, input_data.shape().dims() == 2,
115 errors::InvalidArgument("input_data should be two-dimensional"));
116 }
117
118 // Check tensor bounds.
119 if (!CheckTensorBounds(context, input_data)) return;
120
121 const int32 num_data = static_cast<int32>(input_data.shape().dim_size(0));
122 const int32 num_features =
123 static_cast<int32>(input_data.shape().dim_size(1));
124 const int32 num_nodes =
125 static_cast<int32>(tree_parameters_tensor.shape().dim_size(0));
126
127 Tensor* output_probability = nullptr;
128 TensorShape output_probability_shape;
129 output_probability_shape.AddDim(num_data);
130 output_probability_shape.AddDim(tree_depth_);
131
132 Tensor* output_path = nullptr;
133 TensorShape output_path_shape;
134 output_path_shape.AddDim(num_data);
135 output_path_shape.AddDim(tree_depth_);
136
137 OP_REQUIRES_OK(context,
138 context->allocate_output(0, output_probability_shape,
139 &output_probability));
140 OP_REQUIRES_OK(
141 context, context->allocate_output(1, output_path_shape, &output_path));
142
143 auto out_probability = output_probability->tensor<float, 2>();
144 auto out_path = output_path->tensor<int32, 2>();
145 const auto tree_biases = tree_biases_tensor.tensor<float, 1>();
146
147 // Stochastically traverse the tree to a leaf.
148
149 for (int i = 0; i < num_data; i++) {
150 const Tensor point = input_data.Slice(i, i + 1);
151
152 int32 node = 0;
153 out_probability(i, 0) = 1.0;
154 out_path(i, 0) = 0;
155 for (int j = 0; j < tree_depth_ - 1; j++) {
156 int32 left_child = 2 * node + 1;
157 int32 right_child = left_child + 1;
158
159 float left_prob =
160 LeftProbability(point, tree_parameters_tensor.Slice(j, j + 1),
161 tree_biases(j), num_features);
162
163 if (left_prob < rng_->RandFloat()) {
164 CHECK_LT(i, num_data);
165 CHECK_LT(j + 1, tree_depth_);
166 CHECK_LT(left_child, num_nodes);
167
168 out_probability(i, j + 1) = left_prob * out_probability(i, j);
169 out_path(i, j + 1) = left_child;
170 node = left_child;
171 } else {
172 CHECK_LT(i, num_data);
173 CHECK_LT(j + 1, tree_depth_);
174 CHECK_LT(right_child, num_nodes);
175
176 out_probability(i, j + 1) = (1.0 - left_prob) * out_probability(i, j);
177 out_path(i, j + 1) = right_child;
178 node = right_child;
179 }
180 }
181 }
182 VLOG(1) << "stochastic routing end";
183 }
184
185 private:
186 std::unique_ptr<random::PhiloxRandom> single_rand_;
187 std::unique_ptr<random::SimplePhilox> rng_;
188 int32 tree_depth_;
189 int32 random_seed_;
190};
191
192REGISTER_KERNEL_BUILDER(
193 Name("StochasticHardRoutingFunction").Device(DEVICE_CPU),
194 StochasticHardRoutingFunction);
