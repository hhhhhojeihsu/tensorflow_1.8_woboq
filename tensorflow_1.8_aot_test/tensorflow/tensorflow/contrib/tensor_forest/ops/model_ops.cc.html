84REGISTER_OP("TreePredictionsV4")
85 .Attr("input_spec: string")
86 .Attr("params: string")
87 .Input("tree_handle: resource")
88 .Input("input_data: float")
89 .Input("sparse_input_indices: int64")
90 .Input("sparse_input_values: float")
91 .Input("sparse_input_shape: int64")
92 .Output("predictions: float")
93 .Output("tree_paths: string")
94 .SetShapeFn([](InferenceContext* c) {
95 DimensionHandle num_points = c->UnknownDim();
96
97 if (c->RankKnown(c->input(1)) && c->Rank(c->input(1)) > 0 &&
98 c->Value(c->Dim(c->input(1), 0)) > 0) {
99 num_points = c->Dim(c->input(1), 0);
100 }
101
102 c->set_output(0, c->Matrix(num_points, c->UnknownDim()));
103 c->set_output(1, c->Vector(c->UnknownDim()));
104 return Status::OK();
105 })
106 .Doc(R"doc(
107Outputs the predictions for the given input data.
108
109params: A serialized TensorForestParams proto.
110tree_handle: The handle to the tree.
111input_data: The training batch's features as a 2-d tensor; `input_data[i][j]`
112 gives the j-th feature of the i-th input.
113sparse_input_indices: The indices tensor from the SparseTensor input.
114sparse_input_values: The values tensor from the SparseTensor input.
115sparse_input_shape: The shape tensor from the SparseTensor input.
116predictions: `predictions[i][j]` is the probability that input i is class j.
117tree_paths: `tree_paths[i]` is a serialized TreePath proto for example i.
118)doc");

120REGISTER_OP("TraverseTreeV4")
121 .Attr("input_spec: string")
122 .Attr("params: string")
123 .Input("tree_handle: resource")
124 .Input("input_data: float")
125 .Input("sparse_input_indices: int64")
126 .Input("sparse_input_values: float")
127 .Input("sparse_input_shape: int64")
128 .Output("leaf_ids: int32")
129 .SetShapeFn([](InferenceContext* c) {
130 DimensionHandle num_points = c->UnknownDim();
131
132 if (c->RankKnown(c->input(1)) && c->Rank(c->input(1)) > 0 &&
133 c->Value(c->Dim(c->input(1), 0)) > 0) {
134 num_points = c->Dim(c->input(1), 0);
135 }
136
137 c->set_output(0, c->Vector(num_points));
138 return Status::OK();
139 })
140 .Doc(R"doc(
141Outputs the leaf ids for the given input data.
142
143params: A serialized TensorForestParams proto.
144tree_handle: The handle to the tree.
145input_data: The training batch's features as a 2-d tensor; `input_data[i][j]`
146 gives the j-th feature of the i-th input.
147sparse_input_indices: The indices tensor from the SparseTensor input.
148sparse_input_values: The values tensor from the SparseTensor input.
149sparse_input_shape: The shape tensor from the SparseTensor input.
150leaf_ids: `leaf_ids[i]` is the leaf id for input i.
151)doc");

170)doc");
