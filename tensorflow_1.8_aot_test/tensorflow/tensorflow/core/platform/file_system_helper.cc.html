35constexpr int kNumThreads = 8;
36
37// Run a function in parallel using a ThreadPool, but skip the ThreadPool
38// on the iOS platform due to its problems with more than a few threads.
39void ForEach(int first, int last, const std::function<void(int)>& f) {
40#if TARGET_OS_IPHONE
41 for (int i = first; i < last; i++) {
42 f(i);
43 }
44#else
45 int num_threads = std::min(kNumThreads, last - first);
46 thread::ThreadPool threads(Env::Default(), "ForEach", num_threads);
47 for (int i = first; i < last; i++) {
48 threads.Schedule([f, i] { f(i); });
49 }
50#endif
51}

55Status GetMatchingPaths(FileSystem* fs, Env* env, const string& pattern,
56 std::vector<string>* results) {
57 results->clear();
58 // Find the fixed prefix by looking for the first wildcard.
59 string fixed_prefix = pattern.substr(0, pattern.find_first_of("*?[\\"));
60 string eval_pattern = pattern;
61 std::vector<string> all_files;
62 string dir = io::Dirname(fixed_prefix).ToString();
63 // If dir is empty then we need to fix up fixed_prefix and eval_pattern to
64 // include . as the top level directory.
65 if (dir.empty()) {
66 dir = ".";
67 fixed_prefix = io::JoinPath(dir, fixed_prefix);
68 eval_pattern = io::JoinPath(dir, pattern);
69 }
70
71 // Setup a BFS to explore everything under dir.
72 std::deque<string> dir_q;
73 dir_q.push_back(dir);
74 Status ret; // Status to return.
75 // children_dir_status holds is_dir status for children. It can have three
76 // possible values: OK for true; FAILED_PRECONDITION for false; CANCELLED
77 // if we don't calculate IsDirectory (we might do that because there isn't
78 // any point in exploring that child path).
79 std::vector<Status> children_dir_status;
80 while (!dir_q.empty()) {
81 string current_dir = dir_q.front();
82 dir_q.pop_front();
83 std::vector<string> children;
84 Status s = fs->GetChildren(current_dir, &children);
85 ret.Update(s);
86 if (children.empty()) continue;
87 // This IsDirectory call can be expensive for some FS. Parallelizing it.
88 children_dir_status.resize(children.size());
89 ForEach(0, children.size(),
90 [fs, &current_dir, &children, &fixed_prefix,
91 &children_dir_status](int i) {
92 const string child_path = io::JoinPath(current_dir, children[i]);
93 // In case the child_path doesn't start with the fixed_prefix then
94 // we don't need to explore this path.
95 if (!str_util::StartsWith(child_path, fixed_prefix)) {
96 children_dir_status[i] = Status(tensorflow::error::CANCELLED,
97 "Operation not needed");
98 } else {
99 children_dir_status[i] = fs->IsDirectory(child_path);
100 }
101 });
102 for (int i = 0; i < children.size(); ++i) {
103 const string child_path = io::JoinPath(current_dir, children[i]);
104 // If the IsDirectory call was cancelled we bail.
105 if (children_dir_status[i].code() == tensorflow::error::CANCELLED) {
106 continue;
107 }
108 // If the child is a directory add it to the queue.
109 if (children_dir_status[i].ok()) {
110 dir_q.push_back(child_path);
111 }
112 all_files.push_back(child_path);
113 }
114 }
115
116 // Match all obtained files to the input pattern.
117 for (const auto& f : all_files) {
118 if (env->MatchPath(f, eval_pattern)) {
119 results->push_back(f);
120 }
121 }
122 return ret;
123}
