42/// A generic interface for accessing a file system. Implementations
43/// of custom filesystem adapters must implement this interface,
44/// RandomAccessFile, WritableFile, and ReadOnlyMemoryRegion classes.
45class FileSystem {
46 public:
47 /// \brief Creates a brand new random access read-only file with the
48 /// specified name.
49 ///
50 /// On success, stores a pointer to the new file in
51 /// *result and returns OK. On failure stores NULL in *result and
52 /// returns non-OK. If the file does not exist, returns a non-OK
53 /// status.
54 ///
55 /// The returned file may be concurrently accessed by multiple threads.
56 ///
57 /// The ownership of the returned RandomAccessFile is passed to the caller
58 /// and the object should be deleted when is not used.
59 virtual Status NewRandomAccessFile(
60 const string& fname, std::unique_ptr<RandomAccessFile>* result) = 0;
61
62 /// \brief Creates an object that writes to a new file with the specified
63 /// name.
64 ///
65 /// Deletes any existing file with the same name and creates a
66 /// new file. On success, stores a pointer to the new file in
67 /// *result and returns OK. On failure stores NULL in *result and
68 /// returns non-OK.
69 ///
70 /// The returned file will only be accessed by one thread at a time.
71 ///
72 /// The ownership of the returned WritableFile is passed to the caller
73 /// and the object should be deleted when is not used.
74 virtual Status NewWritableFile(const string& fname,
75 std::unique_ptr<WritableFile>* result) = 0;
76
77 /// \brief Creates an object that either appends to an existing file, or
78 /// writes to a new file (if the file does not exist to begin with).
79 ///
80 /// On success, stores a pointer to the new file in *result and
81 /// returns OK. On failure stores NULL in *result and returns
82 /// non-OK.
83 ///
84 /// The returned file will only be accessed by one thread at a time.
85 ///
86 /// The ownership of the returned WritableFile is passed to the caller
87 /// and the object should be deleted when is not used.
88 virtual Status NewAppendableFile(const string& fname,
89 std::unique_ptr<WritableFile>* result) = 0;
90
91 /// \brief Creates a readonly region of memory with the file context.
92 ///
93 /// On success, it returns a pointer to read-only memory region
94 /// from the content of file fname. The ownership of the region is passed to
95 /// the caller. On failure stores nullptr in *result and returns non-OK.
96 ///
97 /// The returned memory region can be accessed from many threads in parallel.
98 ///
99 /// The ownership of the returned ReadOnlyMemoryRegion is passed to the caller
100 /// and the object should be deleted when is not used.
101 virtual Status NewReadOnlyMemoryRegionFromFile(
102 const string& fname, std::unique_ptr<ReadOnlyMemoryRegion>* result) = 0;
103
104 /// Returns OK if the named path exists and NOT_FOUND otherwise.
105 virtual Status FileExists(const string& fname) = 0;
106
107 /// Returns true if all the listed files exist, false otherwise.
108 /// if status is not null, populate the vector with a detailed status
109 /// for each file.
110 virtual bool FilesExist(const std::vector<string>& files,
111 std::vector<Status>* status);
112
113 /// \brief Returns the immediate children in the given directory.
114 ///
115 /// The returned paths are relative to 'dir'.
116 virtual Status GetChildren(const string& dir,
117 std::vector<string>* result) = 0;
118
119 /// \brief Given a pattern, stores in *results the set of paths that matches
120 /// that pattern. *results is cleared.
121 ///
122 /// pattern must match all of a name, not just a substring.
123 ///
124 /// pattern: { term }
125 /// term:
126 /// '*': matches any sequence of non-'/' characters
127 /// '?': matches a single non-'/' character
128 /// '[' [ '^' ] { match-list } ']':
129 /// matches any single character (not) on the list
130 /// c: matches character c (c != '*', '?', '\\', '[')
131 /// '\\' c: matches character c
132 /// character-range:
133 /// c: matches character c (c != '\\', '-', ']')
134 /// '\\' c: matches character c
135 /// lo '-' hi: matches character c for lo <= c <= hi
136 ///
137 /// Typical return codes:
138 /// * OK - no errors
139 /// * UNIMPLEMENTED - Some underlying functions (like GetChildren) are not
140 /// implemented
141 virtual Status GetMatchingPaths(const string& pattern,
142 std::vector<string>* results) = 0;
143
144 /// \brief Obtains statistics for the given path.
145 virtual Status Stat(const string& fname, FileStatistics* stat) = 0;
146
147 /// \brief Deletes the named file.
148 virtual Status DeleteFile(const string& fname) = 0;
149
150 /// \brief Creates the specified directory.
151 /// Typical return codes:
152 /// * OK - successfully created the directory.
153 /// * ALREADY_EXISTS - directory with name dirname already exists.
154 /// * PERMISSION_DENIED - dirname is not writable.
155 virtual Status CreateDir(const string& dirname) = 0;
156
157 /// \brief Creates the specified directory and all the necessary
158 /// subdirectories.
159 /// Typical return codes:
160 /// * OK - successfully created the directory and sub directories, even if
161 /// they were already created.
162 /// * PERMISSION_DENIED - dirname or some subdirectory is not writable.
163 virtual Status RecursivelyCreateDir(const string& dirname);
164
165 /// \brief Deletes the specified directory.
166 virtual Status DeleteDir(const string& dirname) = 0;
167
168 /// \brief Deletes the specified directory and all subdirectories and files
169 /// underneath it. undeleted_files and undeleted_dirs stores the number of
170 /// files and directories that weren't deleted (unspecified if the return
171 /// status is not OK).
172 /// REQUIRES: undeleted_files, undeleted_dirs to be not null.
173 /// Typical return codes:
174 /// * OK - dirname exists and we were able to delete everything underneath.
175 /// * NOT_FOUND - dirname doesn't exist
176 /// * PERMISSION_DENIED - dirname or some descendant is not writable
177 /// * UNIMPLEMENTED - Some underlying functions (like Delete) are not
178 /// implemented
179 virtual Status DeleteRecursively(const string& dirname,
180 int64* undeleted_files,
181 int64* undeleted_dirs);
182
183 /// \brief Stores the size of `fname` in `*file_size`.
184 virtual Status GetFileSize(const string& fname, uint64* file_size) = 0;
185
186 /// \brief Overwrites the target if it exists.
187 virtual Status RenameFile(const string& src, const string& target) = 0;
188
189 /// \brief Copy the src to target.
190 virtual Status CopyFile(const string& src, const string& target);
191
192 /// \brief Translate an URI to a filename for the FileSystem implementation.
193 ///
194 /// The implementation in this class cleans up the path, removing
195 /// duplicate /'s, resolving .. and . (more details in
196 /// tensorflow::lib::io::CleanPath).
197 virtual string TranslateName(const string& name) const;
198
199 /// \brief Returns whether the given path is a directory or not.
200 ///
201 /// Typical return codes (not guaranteed exhaustive):
202 /// * OK - The path exists and is a directory.
203 /// * FAILED_PRECONDITION - The path exists and is not a directory.
204 /// * NOT_FOUND - The path entry does not exist.
205 /// * PERMISSION_DENIED - Insufficient permissions.
206 /// * UNIMPLEMENTED - The file factory doesn't support directories.
207 virtual Status IsDirectory(const string& fname);
208
209 /// \brief Flushes any cached filesystem objects from memory.
210 virtual void FlushCaches();
211
212 FileSystem() {}
213
214 virtual ~FileSystem();
215};

217/// A file abstraction for randomly reading the contents of a file.
218class RandomAccessFile {
219 public:
220 RandomAccessFile() {}
221 virtual ~RandomAccessFile();
222
223 /// \brief Reads up to `n` bytes from the file starting at `offset`.
224 ///
225 /// `scratch[0..n-1]` may be written by this routine. Sets `*result`
226 /// to the data that was read (including if fewer than `n` bytes were
227 /// successfully read). May set `*result` to point at data in
228 /// `scratch[0..n-1]`, so `scratch[0..n-1]` must be live when
229 /// `*result` is used.
230 ///
231 /// On OK returned status: `n` bytes have been stored in `*result`.
232 /// On non-OK returned status: `[0..n]` bytes have been stored in `*result`.
233 ///
234 /// Returns `OUT_OF_RANGE` if fewer than n bytes were stored in `*result`
235 /// because of EOF.
236 ///
237 /// Safe for concurrent use by multiple threads.
238 virtual Status Read(uint64 offset, size_t n, StringPiece* result,
239 char* scratch) const = 0;
240
241 private:
242 TF_DISALLOW_COPY_AND_ASSIGN(RandomAccessFile);
243};
244
245/// \brief A file abstraction for sequential writing.
246///
247/// The implementation must provide buffering since callers may append
248/// small fragments at a time to the file.
249class WritableFile {
250 public:
251 WritableFile() {}
252 virtual ~WritableFile();
253
254 /// \brief Append 'data' to the file.
255 virtual Status Append(const StringPiece& data) = 0;
256
257 /// \brief Close the file.
258 ///
259 /// Flush() and de-allocate resources associated with this file
260 ///
261 /// Typical return codes (not guaranteed to be exhaustive):
262 /// * OK
263 /// * Other codes, as returned from Flush()
264 virtual Status Close() = 0;
265
266 /// \brief Flushes the file and optionally syncs contents to filesystem.
267 ///
268 /// This should flush any local buffers whose contents have not been
269 /// delivered to the filesystem.
270 ///
271 /// If the process terminates after a successful flush, the contents
272 /// may still be persisted, since the underlying filesystem may
273 /// eventually flush the contents. If the OS or machine crashes
274 /// after a successful flush, the contents may or may not be
275 /// persisted, depending on the implementation.
276 virtual Status Flush() = 0;
277
278 /// \brief Syncs contents of file to filesystem.
279 ///
280 /// This waits for confirmation from the filesystem that the contents
281 /// of the file have been persisted to the filesystem; if the OS
282 /// or machine crashes after a successful Sync, the contents should
283 /// be properly saved.
284 virtual Status Sync() = 0;
285
286 private:
287 TF_DISALLOW_COPY_AND_ASSIGN(WritableFile);
288};
289
290/// \brief A readonly memmapped file abstraction.
291///
292/// The implementation must guarantee that all memory is accessible when the
293/// object exists, independently from the Env that created it.
294class ReadOnlyMemoryRegion {
295 public:
296 ReadOnlyMemoryRegion() {}
297 virtual ~ReadOnlyMemoryRegion() = default;
298
299 /// \brief Returns a pointer to the memory region.
300 virtual const void* data() = 0;
301
302 /// \brief Returns the length of the memory region in bytes.
303 virtual uint64 length() = 0;
304};
305
306/// \brief A registry for file system implementations.
307///
308/// Filenames are specified as an URI, which is of the form
309/// [scheme://]<filename>.
310/// File system implementations are registered using the REGISTER_FILE_SYSTEM
311/// macro, providing the 'scheme' as the key.
312class FileSystemRegistry {
313 public:
314 typedef std::function<FileSystem*()> Factory;
315
316 virtual ~FileSystemRegistry();
317 virtual Status Register(const string& scheme, Factory factory) = 0;
318 virtual FileSystem* Lookup(const string& scheme) = 0;
319 virtual Status GetRegisteredFileSystemSchemes(
320 std::vector<string>* schemes) = 0;
321};
