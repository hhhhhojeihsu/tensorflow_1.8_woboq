16#include <sys/time.h>
17#include <time.h>
18
19#include "tensorflow/core/platform/env_time.h"
20
21namespace tensorflow {
22
23namespace {
24
25class PosixEnvTime : public EnvTime {
26 public:
27 PosixEnvTime() {}
28
29 uint64 NowMicros() override {
30 struct timeval tv;
31 gettimeofday(&tv, nullptr);
32 return static_cast<uint64>(tv.tv_sec) * 1000000 + tv.tv_usec;
33 }
34};
35
36} // namespace
37
38#if defined(PLATFORM_POSIX) || defined(__ANDROID__)
39EnvTime* EnvTime::Default() {
40 static EnvTime* default_env_time = new PosixEnvTime;
41 return default_env_time;
42}
43#endif
