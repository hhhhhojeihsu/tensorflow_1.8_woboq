44// pread() based random-access
45class PosixRandomAccessFile : public RandomAccessFile {
46 private:
47 string filename_;
48 int fd_;
49
50 public:
51 PosixRandomAccessFile(const string& fname, int fd)
52 : filename_(fname), fd_(fd) {}
53 ~PosixRandomAccessFile() override { close(fd_); }
54
55 Status Read(uint64 offset, size_t n, StringPiece* result,
56 char* scratch) const override {
57 Status s;
58 char* dst = scratch;
59 while (n > 0 && s.ok()) {
60 ssize_t r = pread(fd_, dst, n, static_cast<off_t>(offset));
61 if (r > 0) {
62 dst += r;
63 n -= r;
64 offset += r;
65 } else if (r == 0) {
66 s = Status(error::OUT_OF_RANGE, "Read less bytes than requested");
67 } else if (errno == EINTR || errno == EAGAIN) {
68 // Retry
69 } else {
70 s = IOError(filename_, errno);
71 }
72 }
73 *result = StringPiece(scratch, dst - scratch);
74 return s;
75 }
76};

78class PosixWritableFile : public WritableFile {
79 private:
80 string filename_;
81 FILE* file_;
82
83 public:
84 PosixWritableFile(const string& fname, FILE* f)
85 : filename_(fname), file_(f) {}
86
87 ~PosixWritableFile() override {
88 if (file_ != nullptr) {
89 // Ignoring any potential errors
90 fclose(file_);
91 }
92 }
93
94 Status Append(const StringPiece& data) override {
95 size_t r = fwrite(data.data(), 1, data.size(), file_);
96 if (r != data.size()) {
97 return IOError(filename_, errno);
98 }
99 return Status::OK();
100 }
101
102 Status Close() override {
103 Status result;
104 if (fclose(file_) != 0) {
105 result = IOError(filename_, errno);
106 }
107 file_ = nullptr;
108 return result;
109 }
110
111 Status Flush() override {
112 if (fflush(file_) != 0) {
113 return IOError(filename_, errno);
114 }
115 return Status::OK();
116 }
117
118 Status Sync() override {
119 Status s;
120 if (fflush(file_) != 0) {
121 s = IOError(filename_, errno);
122 }
123 return s;
124 }
125};

127class PosixReadOnlyMemoryRegion : public ReadOnlyMemoryRegion {
128 public:
129 PosixReadOnlyMemoryRegion(const void* address, uint64 length)
130 : address_(address), length_(length) {}
131 ~PosixReadOnlyMemoryRegion() override {
132 munmap(const_cast<void*>(address_), length_);
133 }
134 const void* data() override { return address_; }
135 uint64 length() override { return length_; }
136
137 private:
138 const void* const address_;
139 const uint64 length_;
140};

181Status PosixFileSystem::NewReadOnlyMemoryRegionFromFile(
182 const string& fname, std::unique_ptr<ReadOnlyMemoryRegion>* result) {
183 string translated_fname = TranslateName(fname);
184 Status s = Status::OK();
185 int fd = open(translated_fname.c_str(), O_RDONLY);
186 if (fd < 0) {
187 s = IOError(fname, errno);
188 } else {
189 struct stat st;
190 ::fstat(fd, &st);
191 const void* address =
192 mmap(nullptr, st.st_size, PROT_READ, MAP_PRIVATE, fd, 0);
193 if (address == MAP_FAILED) {
194 s = IOError(fname, errno);
195 } else {
196 result->reset(new PosixReadOnlyMemoryRegion(address, st.st_size));
197 }
198 close(fd);
199 }
200 return s;
201}

210Status PosixFileSystem::GetChildren(const string& dir,
211 std::vector<string>* result) {
212 string translated_dir = TranslateName(dir);
213 result->clear();
214 DIR* d = opendir(translated_dir.c_str());
215 if (d == nullptr) {
216 return IOError(dir, errno);
217 }
218 struct dirent* entry;
219 while ((entry = readdir(d)) != nullptr) {
220 StringPiece basename = entry->d_name;
221 if ((basename != ".") && (basename != "..")) {
222 result->push_back(entry->d_name);
223 }
224 }
225 closedir(d);
226 return Status::OK();
227}

291Status PosixFileSystem::CopyFile(const string& src, const string& target) {
292 string translated_src = TranslateName(src);
293 struct stat sbuf;
294 if (stat(translated_src.c_str(), &sbuf) != 0) {
295 return IOError(src, errno);
296 }
297 int src_fd = open(translated_src.c_str(), O_RDONLY);
298 if (src_fd < 0) {
299 return IOError(src, errno);
300 }
301 string translated_target = TranslateName(target);
302 // O_WRONLY | O_CREAT:
303 // Open file for write and if file does not exist, create the file.
304 // S_IRUSR | S_IWUSR | S_IRGRP | S_IROTH:
305 // Create the file with permission of 0644
306 int target_fd = open(translated_target.c_str(), O_WRONLY | O_CREAT,
307 S_IRUSR | S_IWUSR | S_IRGRP | S_IROTH);
308 if (target_fd < 0) {
309 close(src_fd);
310 return IOError(target, errno);
311 }
312 int rc = 0;
313 off_t offset = 0;
314 std::unique_ptr<char[]> buffer(new char[kPosixCopyFileBufferSize]);
315 while (offset < sbuf.st_size) {
316 // Use uint64 for safe compare SSIZE_MAX
317 uint64 chunk = sbuf.st_size - offset;
318 if (chunk > SSIZE_MAX) {
319 chunk = SSIZE_MAX;
320 }
321#if defined(__linux__) && !defined(__ANDROID__)
322 rc = sendfile(target_fd, src_fd, &offset, static_cast<size_t>(chunk));
323#else
324 if (chunk > kPosixCopyFileBufferSize) {
325 chunk = kPosixCopyFileBufferSize;
326 }
327 rc = read(src_fd, buffer.get(), static_cast<size_t>(chunk));
328 if (rc <= 0) {
329 break;
330 }
331 rc = write(target_fd, buffer.get(), static_cast<size_t>(chunk));
332 offset += chunk;
333#endif
334 if (rc <= 0) {
335 break;
336 }
337 }
338
339 Status result = Status::OK();
340 if (rc < 0) {
341 result = IOError(target, errno);
342 }
343
344 // Keep the error code
345 rc = close(target_fd);
346 if (rc < 0 && result == Status::OK()) {
347 result = IOError(target, errno);
348 }
349 rc = close(src_fd);
350 if (rc < 0 && result == Status::OK()) {
351 result = IOError(target, errno);
352 }
353
354 return result;
355}
