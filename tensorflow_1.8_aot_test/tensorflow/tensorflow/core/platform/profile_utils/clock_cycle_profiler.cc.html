20namespace tensorflow {
21
22void ClockCycleProfiler::DumpStatistics(const string& tag) {
23 CHECK(!IsStarted());
24 const double average_clock_cycle = GetAverageClockCycle();
25 const double count = GetCount();
26 const std::chrono::duration<double> average_time =
27 profile_utils::CpuUtils::ConvertClockCycleToTime(
28 static_cast<int64>(average_clock_cycle + 0.5));
29 LOG(INFO) << tag << ": average = "
30 << std::chrono::duration_cast<std::chrono::microseconds>(
31 average_time)
32 .count()
33 << " us (" << average_clock_cycle << " cycles)"
34 << ", count = " << count;
35}
