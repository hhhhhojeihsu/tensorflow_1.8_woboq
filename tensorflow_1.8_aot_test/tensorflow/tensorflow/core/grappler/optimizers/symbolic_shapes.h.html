16#ifndef TENSORFLOW_CORE_GRAPPLER_OPTIMIZERS_SYMBOLIC_SHAPES_H_
17#define TENSORFLOW_CORE_GRAPPLER_OPTIMIZERS_SYMBOLIC_SHAPES_H_
18
19#include "tensorflow/core/framework/tensor_shape.pb.h"
20#include "tensorflow/core/grappler/costs/op_performance_data.pb.h"
21
22namespace tensorflow {
23namespace grappler {
24
25bool IsKnown(const TensorShapeProto::Dim& dim);
26bool IsKnownSymbolically(const TensorShapeProto::Dim& dim);
27bool IsUnknown(const TensorShapeProto::Dim& dim);
28
29// Shape is symbolically defined, if it has a known rank, and each dimension is
30// known (dim_size >= 0), or is a symbolic dimension size (dim_size <= -2).
31bool ShapeIsSymbolicallyDefined(const TensorShapeProto& shape);
32bool ShapeIsSymbolicallyDefined(const OpInfo::TensorProperties& properties);
33
34// Shapes are symbolically equal, if they have the same rank, they are known or
35// symbolically defined, and have matching dimensions.
36bool ShapesSymbolicallyEqual(const TensorShapeProto& left,
37 const TensorShapeProto& right);
38bool ShapesSymbolicallyEqual(const OpInfo::TensorProperties& left,
39 const OpInfo::TensorProperties& right);
40
41// Check if two shapes can be broadcasted to each other. Both shapes must be at
42// least symbolically defined, and the have valid BCast instance.
43bool ShapesBroadcastable(const TensorShapeProto& left,
44 const TensorShapeProto& right);
45bool ShapesBroadcastable(const OpInfo::TensorProperties& left,
46 const OpInfo::TensorProperties& right);
47
48// Return true if can prove, that tensor of size 'left' is smaller than tensor
49// of size 'right'. Return false if it's larger or equal, or it's impossible to
50// compare because of unknown dimensions, or mismatch in symbolic dimensions.
51bool CompareSymbolicallyShapedTensorSizes(const TensorShapeProto& left,
52 const TensorShapeProto& right);
53bool CompareSymbolicallyShapedTensorSizes(
54 const OpInfo::TensorProperties& left,
55 const OpInfo::TensorProperties& right);
56
57} // namespace grappler
58} // end namespace tensorflow
