19#include "tensorflow/core/framework/graph.pb.h"
20#include "tensorflow/core/lib/core/status.h"

22namespace tensorflow {
23namespace grappler {
24
25class Cluster;
26struct GrapplerItem;
27
28// An abstract interface for an algorithm for generating a candidate
29// optimization of a GrapplerItem for running on a cluster.
30class GraphOptimizer {
31 public:
32 virtual ~GraphOptimizer() {}
33
34 virtual string name() const = 0;
35
36 // Routine called to allow an algorithm to propose a rewritten graph
37 // for the graph, feeds and fetches in "item" to run more efficiently
38 // on "cluster".
39 // Returns true iff it managed to generate a solution, false otherwise.
40 virtual Status Optimize(Cluster* cluster, const GrapplerItem& item,
41 GraphDef* optimized_graph) = 0;
42
43 // Method invoked by the framework so that it can provide feedback
44 // on how well the "optimized_graph" (produced as *optimized_graph from a
45 // call to Optimize) performed. Lower "result" scores are better.
46 virtual void Feedback(Cluster* cluster, const GrapplerItem& item,
47 const GraphDef& optimized_graph, double result) = 0;
48};
49
50} // end namespace grappler
51} // end namespace tensorflow
