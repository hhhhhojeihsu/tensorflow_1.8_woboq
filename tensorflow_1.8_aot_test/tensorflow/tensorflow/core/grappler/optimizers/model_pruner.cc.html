29bool IsTrivialOp(const NodeDef& node, const GraphRewriter& rewriter) {
30 // Remove the stop gradient nodes since they serve no purpose once the graph
31 // is built. Also remove Identity ops.
32 if (IsStopGradient(node)) {
33 return true;
34 }
35 if (IsIdentity(node)) {
36 if (rewriter.FeedsMerge(node) || rewriter.IsDrivenBySwitch(node) ||
37 rewriter.IsDrivenByControlDependency(node) ||
38 rewriter.DrivesControlDependency(node)) {
39 return false;
40 } else {
41 return true;
42 }
43 }
44 if (IsAddN(node) && NumNonControlInputs(node) <= 1) {
45 return true;
46 }
47
48 return false;
49}

51Status ModelPruner::Optimize(Cluster* cluster, const GrapplerItem& item,
52 GraphDef* pruned_graph) {
53 const std::unordered_set<string> nodes_to_preserve = item.NodesToPreserve();
54
55 // Prune all the nodes that won't be executed, ie all the nodes that aren't in
56 // the fanin of a fetch node. If fetch nodes aren't specified, we'll assume
57 // the whole graph might be executed.
58 GrapplerItem runnable_item;
59 if (!nodes_to_preserve.empty()) {
60 std::vector<string> terminal_nodes(nodes_to_preserve.begin(),
61 nodes_to_preserve.end());
62 std::sort(terminal_nodes.begin(), terminal_nodes.end());
63 bool ill_formed = false;
64 std::vector<const NodeDef*> keep =
65 ComputeTransitiveFanin(item.graph, terminal_nodes, &ill_formed);
66 if (ill_formed) {
67 // Some graph edges are invalid, or some of the feeds/fetch don't exist:
68 // let's be conservative and preserve the graph as is.
69 return errors::InvalidArgument("Invalid input graph.");
70 }
71 // Try to keep the nodes ordered somewhat topologically since this helps
72 // further optimizations perform better.
73 for (int i = keep.size() - 1; i >= 0; --i) {
74 *runnable_item.graph.add_node() = *keep[i];
75 }
76 } else {
77 runnable_item = item;
78 }
79
80 GraphRewriter rewriter(runnable_item);
81
82 // Check if we can further prune the graph, by removing the trivial ops.
83 std::unordered_set<const NodeDef*> nodes_to_delete;
84 for (auto& node : runnable_item.graph.node()) {
85 if (!IsTrivialOp(node, rewriter)) {
86 continue;
87 }
88
89 // Don't remove nodes that must be preserved.
90 if (nodes_to_preserve.find(node.name()) != nodes_to_preserve.end()) {
91 continue;
92 }
93
94 // - Don't remove nodes that drive control dependencies.
95 // - Don't remove nodes that are driven by control dependencies either since
96 // we can't ensure (yet) that we won't increase the number of control
97 // dependency edges by deleting them (for example, removing a node driven
98 // by 10 control edges and driving 10 control edges would result in the
99 // creation of 100 edges).
100 // - Don't modify nodes that are connected to functions since that can
101 // result in inlining failures later on.
102 // - Don't prune nodes that are driven by another device since these could
103 // be used to reduce cross device communication.
104 // - Don't remove nodes that receive reference values, as those can be
105 // converting references to non-references. It is important to preserve
106 // these non-references since the partitioner will avoid sending
107 // non-references across partitions more than once.
108 if (!rewriter.RemovalIncreasesEdgeCount(node) &&
109 !rewriter.IsConnectedToFunction(node) &&
110 !rewriter.IsDrivenByAnotherDevice(node) &&
111 !rewriter.ReceivesRefValue(node)) {
112 nodes_to_delete.insert(&node);
113 }
114 }
115
116 *pruned_graph->mutable_library() = item.graph.library();
117 *pruned_graph->mutable_versions() = item.graph.versions();
118
119 if (nodes_to_delete.empty()) {
120 pruned_graph->mutable_node()->Swap(runnable_item.graph.mutable_node());
121 return Status::OK();
122 }
123
124 const bool fetches_are_known = !item.fetch.empty();
125 for (auto& node : runnable_item.graph.node()) {
126 if (!fetches_are_known ||
127 nodes_to_delete.find(&node) == nodes_to_delete.end()) {
128 NodeDef* new_node = pruned_graph->add_node();
129 *new_node = node;
130 new_node->clear_input();
131 rewriter.ForwardInputs(node, nodes_to_delete, new_node);
132 }
133 }
134 VLOG(1) << "Pruned " << nodes_to_delete.size()
135 << " nodes from the graph. The graph now contains "
136 << pruned_graph->node_size() << " nodes.";
137
138 return Status::OK();
139}
