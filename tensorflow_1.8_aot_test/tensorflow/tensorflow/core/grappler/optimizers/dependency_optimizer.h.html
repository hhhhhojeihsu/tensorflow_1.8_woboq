24namespace tensorflow {
25namespace grappler {
26
27// Optimize TF computations by removing control dependencies or re-arranging
28// them to shorten the critical path for a model step or enable other
29// optimizations, such as removing nodes that are effectively noops.
30class DependencyOptimizer : public GraphOptimizer {
31 public:
32 DependencyOptimizer() {}
33 explicit DependencyOptimizer(RewriterConfig::Toggle opt_level) {}
34 ~DependencyOptimizer() override {}
35
36 string name() const override { return "dependency_optimizer"; };
37
38 Status Optimize(Cluster* cluster, const GrapplerItem& item,
39 GraphDef* optimized_graph) override;
40
41 void Feedback(Cluster* cluster, const GrapplerItem& item,
42 const GraphDef& optimized_graph, double result) override;
43
44 private:
45 // Returns true if node is not an Identity node or if it is an Identity
46 // that is safe to remove.
47 bool SafeToRemoveIdentity(const NodeDef& node);
48 // Returns true if it is safe to convert node to NoOp.
49 bool SafeToConvertToNoOp(const NodeDef& node);
50 // Removes all duplicate control dependencies.
51 void CleanControlInputs();
52 // Builds a map from the &optimized_graph_->node(i) to i.
53 void BuildNodeToIdx();
54 // Tries to optimize the node with the given index, possibly additional
55 // optimizations by inserting nodes in nodes_to_simplify, and pruning nodes by
56 // inserting them in nodes_to_delete.
57 void OptimizeNode(int node_idx, SetVector<int>* nodes_to_simplify,
58 std::set<int>* nodes_to_delete);
59 // Eliminates redundant control dependencies by computing the transitive
60 // reduction of the graph.
61 Status TransitiveReduction();
62 // Main driver of dependency optimizations.
63 Status OptimizeDependencies();
64
65 bool fetch_nodes_known_;
66 std::unordered_set<string> nodes_to_preserve_;
67 std::unique_ptr<NodeMap> node_map_;
68 std::unordered_map<const NodeDef*, int> node_to_idx_;
69 GraphDef* optimized_graph_; // Not owned.
70};
71
72} // end namespace grappler
