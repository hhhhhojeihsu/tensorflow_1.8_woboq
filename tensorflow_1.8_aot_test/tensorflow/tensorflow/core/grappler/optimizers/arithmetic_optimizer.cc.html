52// Extract values from a Const op to `values`. Returns true if succeeds.
53template <typename T>
54bool ValuesFromConstNode(const NodeDef& node, std::vector<T>* values) {
55 if (node.op() != "Const") {
56 return false;
57 }
58
59 if (node.attr().at("dtype").type() != DataTypeToEnum<T>::value) {
60 return false;
61 }
62
63 // TensorProto represents the content of the tensor in either <type>_val or
64 // tensor_content.
65 const TensorProto& tensor = node.attr().at("value").tensor();
66 typename checkpoint::SaveTypeTraits<T>::RepeatedField* tensor_values =
67 checkpoint::MutableTensorProtoData<T>(const_cast<TensorProto*>(&tensor));
68
69 if (!tensor_values->empty() && tensor.has_tensor_shape()) {
70 // When tensor_shape is set, theoretically the representation of the data
71 // could be compressed. So, before copying values to the returned vector,
72 // make sure no compression happens.
73 const TensorShapeProto& shape = tensor.tensor_shape();
74 if (shape.dim_size() == 1 && shape.dim(0).size() == tensor_values->size()) {
75 values->insert(values->end(), tensor_values->begin(),
76 tensor_values->end());
77 return true;
78 }
79 }
80
81 const auto tensor_content_size = tensor.tensor_content().size();
82 if (tensor_content_size > 0) {
83 CHECK_EQ(0, tensor_content_size % sizeof(T))
84 << "tensor_content_size (" << tensor_content_size
85 << ") is not a multiple of " << sizeof(T);
86 values->resize(tensor_content_size / sizeof(T));
87 port::CopyToArray(tensor.tensor_content(),
88 reinterpret_cast<char*>(values->data()));
89 return true;
90 }
91
92 return false;
93}

95template <typename T>
96bool IsInnerMatrixTranspose(const std::vector<T>& perm) {
97 const T n = perm.size();
98 if (n < 2) {
99 return false;
100 }
101 for (T i = 0; i < n - 2; ++i) {
102 if (perm[i] != i) {
103 return false;
104 }
105 }
106 return perm[n - 1] == n - 2 && perm[n - 2] == n - 1;
107}
108
109bool IsInnerMatrixTransposeNode(const NodeDef& transpose_node,
110 const NodeMap* node_map) {
111 if (transpose_node.op() != "Transpose" &&
112 transpose_node.op() != "ConjugateTranspose") {
113 return false;
114 }
115 const NodeDef* perm_node = node_map->GetNode(transpose_node.input(1));
116 std::vector<int> perm32;
117 if (ValuesFromConstNode(*perm_node, &perm32)) {
118 return IsInnerMatrixTranspose(perm32);
119 }
120 std::vector<int64> perm64;
121 if (ValuesFromConstNode(*perm_node, &perm64)) {
122 return IsInnerMatrixTranspose(perm64);
123 }
124 return false;
125}

127bool MaybeAddControlInput(const string& new_input, NodeDef* node,
128 GraphDef* graph, NodeMap* node_map) {
129 bool already_exists = false;
130 for (const string& input : node->input()) {
131 if (input == new_input || AsControlDependency(input) == new_input) {
132 already_exists = true;
133 break;
134 }
135 }
136 if (!already_exists) {
137 const string ctrl_dep =
138 ConstantFolding::AddControlDependency(new_input, graph, node_map);
139 node->add_input(ctrl_dep);
140 node_map->AddOutput(NodeName(new_input), node->name());
141 }
142 return !already_exists;
143}

145int CopyControlInputs(const NodeDef& from, NodeDef* to, GraphDef* graph,
146 NodeMap* node_map) {
147 int num_copied = 0;
148 for (const string& input : from.input()) {
149 if (IsControlInput(input) &&
150 MaybeAddControlInput(input, to, graph, node_map)) {
151 ++num_copied;
152 }
153 }
154 return num_copied;
155}

202// Returns whether `reshape` is an identity op. The tensor that `reshape`
203// reshapes is the `output_pos`-th output of node `input`.
204bool ReshapeIsIdentity(const NodeDef& reshape, const NodeDef& input,
205 const int output_pos,
206 const GraphProperties& graph_properties) {
207 const std::vector<OpInfo::TensorProperties>& reshape_props =
208 graph_properties.GetOutputProperties(reshape.name());
209 const std::vector<OpInfo::TensorProperties>& input_props =
210 graph_properties.GetOutputProperties(input.name());
211 if (reshape_props.empty() || input_props.empty() ||
212 input_props.size() <= output_pos) {
213 return false;
214 }
215
216 const PartialTensorShape& src_shape = input_props[output_pos].shape();
217 const PartialTensorShape& dst_shape = reshape_props[0].shape();
218
219 if (src_shape.unknown_rank() || dst_shape.unknown_rank()) {
220 return false;
221 }
222
223 if (!dst_shape.IsCompatibleWith(src_shape)) {
224 return false;
225 }
226
227 // Returns false when src_shape or dst_shape has >=2 dimensions with unknown
228 // sizes.
229 auto num_unknown_dim_sizes = [](const PartialTensorShape& partial_shape) {
230 auto dim_sizes = partial_shape.dim_sizes();
231 return std::count_if(dim_sizes.begin(), dim_sizes.end(),
232 [](int dim) { return dim < 0; });
233 };
234 int src_num_unknown_dim_sizes = num_unknown_dim_sizes(src_shape);
235 int dst_num_unknown_dim_sizes = num_unknown_dim_sizes(dst_shape);
236 if (src_num_unknown_dim_sizes > 1 || dst_num_unknown_dim_sizes > 1) {
237 return false;
238 }
239
240 // If dst_num_unknown_dim_sizes != src_num_unknown_dim_sizes we would weaken
241 // shape inference in subsequent passes if we removed this reshape.
242 if (src_num_unknown_dim_sizes != dst_num_unknown_dim_sizes) {
243 return false;
244 }
245
246 // Remove the reshape if both are fully defined or partially defined and the
247 // unknown or symbolic shape appears on the same dimension, i.e., if
248 // IsIdenticalTo returns true.
249 return dst_shape.IsIdenticalTo(src_shape);
250}

329 // Input name with a statically inferred shape from GraphProperties
330 struct InputAndShape {
331 InputAndShape(const string& input, const TensorShapeProto& shape)
332 : input(input), shape(shape) {}
333 string input;
334 TensorShapeProto shape;
335 };
336
337 // Subgraph (subtree) of nodes, that we want to optimize in "one shot" (e.g.
338 // all the Add nodes that we plan to rewrite with a single AddN). Subgraph is
339 // obtained by graph traversal, starting from a root node.
340 struct OptimizedNodesGroup {
341 NodeDef* root_node;
342 TensorShapeProto root_shape;
343 // Optimized nodes that will be updated or removed by rewrite
344 std::vector<NodeDef*> optimized_nodes;
345 // Inputs to optimized nodes
346 std::vector<InputAndShape> inputs;
347 };
348
349 Status TrySimplify(NodeDef* node, string* simplified_node_name) override {
350 TF_RETURN_IF_ERROR(EnsureNodeIsSupported(node));
351
352 OptimizedNodesGroup group;
353 TF_RETURN_IF_ERROR(CreateOptimizedNodesGroup(node, &group));
354
355 if (!group.optimized_nodes.empty()) {
356 *simplified_node_name = RewriteOptimizedNodesGroup(group);
357 }
358
359 return Status::OK();
360 }
361
362 protected:
363 // Modify the optimized graph after nodes group was successfully identified
364 virtual string RewriteOptimizedNodesGroup(
365 const OptimizedNodesGroup& group) = 0;
366
367 // Check if input can become a part of current optimized nodes group.
368 virtual bool IsAbsorbableByOptimizedNodesGroup(
369 const OptimizedNodesGroup& group, const string& input) const = 0;
370
371 Status AbsorbInputByOptimizedNodesGroup(const string& input,
372 OptimizedNodesGroup* group) const {
373 NodeDef* node;
374 TF_RETURN_IF_ERROR(GetInputNode(input, &node));
375
376 if (IsAbsorbableByOptimizedNodesGroup(*group, input)) {
377 for (int i = 0; i < node->input_size(); ++i) {
378 const string& input_i = node->input(i);
379 if (!IsControlInput(input)) {
380 TF_RETURN_IF_ERROR(AbsorbInputByOptimizedNodesGroup(input_i, group));
381 }
382 }
383 group->optimized_nodes.push_back(node);
384 } else {
385 // If node can't be absorbed, add it to OptimizedNodesGroup input
386 OpInfo::TensorProperties properties;
387 TF_RETURN_IF_ERROR(GetTensorProperties(input, &properties));
388 group->inputs.emplace_back(input, properties.shape());
389 }
390 return Status::OK();
391 }
392
393 Status CreateOptimizedNodesGroup(NodeDef* root_node,
394 OptimizedNodesGroup* group) const {
395 OpInfo::TensorProperties root_node_output_properties;
396 TF_RETURN_IF_ERROR(
397 GetTensorProperties(root_node->name(), &root_node_output_properties));
398
399 group->root_node = root_node;
400 group->root_shape = root_node_output_properties.shape();
401
402 group->optimized_nodes.reserve(root_node->input_size());
403 for (int i = 0; i < root_node->input_size(); ++i) {
404 const string& input_i = root_node->input(i);
405 if (!IsControlInput(input_i)) {
406 TF_RETURN_IF_ERROR(AbsorbInputByOptimizedNodesGroup(input_i, group));
407 }
408 }
409
410 return Status::OK();
411 }

478// Rewrite a tree of Add/AddN with a single AddN operation, consuming all the
479// original inputs of absorbed nodes.
480//
481// 1) All nodes must have the same device placement.
482//
483// 2) If All nodes in a Add/AddN subgraph have symbolically equal shape, tree is
484// optimized to a single AddN node.
485//
486// AddN_1
487// / | \
488// Add_1 z Add_2 -> AddN(x, y, z, w, q, e)
489// / \ / \
490// x y w Add_3
491// / \
492// q e
493//
494// 3) If some nodes have different shape (it needs to be broadcastable to the
495// shape of a "root), tree is optimized to AddNs for symbolically equal
496// shapes, and a tree of Add ops, that minimize broadcasts.
497//
498// AddN_1 Add
499// / | \ / \
500// Add_1 z Add_2 -> Add w
501// / \ / \ / \
502// x y w Add_3 AddN(x, y, q, e) z
503// / \
504// q e
505class AddOpsRewriteStage : public ArithmeticNodesGroupOptimizerStage {
506 public:
507 explicit AddOpsRewriteStage(const GraphOptimizerContext& ctx,
508 const ArithmeticOptimizerContext& ctx_ext)
509 : ArithmeticNodesGroupOptimizerStage("AddOpsRewrite", ctx, ctx_ext) {}
510 ~AddOpsRewriteStage() override = default;
511
512 // Check if a node can become a root of AddOpsGroup
513 bool IsSupported(const NodeDef* node) const override {
514 if (!CanOptimize(node)) return false;
515
516 // shape must be symbolically defined and all inputs compatible with it
517 OpInfo::TensorProperties properties;
518 Status has_properties = GetTensorProperties(node->name(), &properties);
519 return has_properties.ok() && ShapeIsSymbolicallyDefined(properties) &&
520 HasAllInputsBroadcastableToShape(*node, properties);
521 }
522
523 protected:
524 // Check if a node can be absorbed by current OptimizedNodesGroup
525 bool IsAbsorbableByOptimizedNodesGroup(const OptimizedNodesGroup& group,
526 const string& input) const override {
527 NodeDef* node;
528 Status node_status = GetInputNode(input, &node);
529 if (!node_status.ok() || !CanOptimize(node)) return false;
530
531 if (!IsOnTheSameDevice(group, *node)) {
532 return false;
533 }
534 // with a single output data consumer (presumably if we reach this node from
535 // previously absorbed or a root node, it means that this node is not used
536 // as an input to any other op, outside of the group)
537 if (NumNonControlDataOutputs(*node, *ctx_.node_map) != 1) {
538 return false;
539 }
540 // All input shapes must be broadcastable to the node shape
541 OpInfo::TensorProperties properties;
542 Status has_properties = GetTensorProperties(input, &properties);
543 return has_properties.ok() &&
544 HasAllInputsBroadcastableToShape(*node, properties);
545 }

547 // Node requirements both for a root node and an absorbed node
548 bool CanOptimize(const NodeDef* node) const {
549 // TODO(ezhulenev): check if AccumulateNV2 can be supported too
550 if (!IsAdd(*node) && !IsAddN(*node)) {
551 return false;
552 }
553 if (IsInPreserveSet(*node) || IsAlreadyOptimized(*node)) {
554 return false;
555 }
556 // it must not be created by this stage at any of previous optimization runs
557 if (str_util::StrContains(node->name(), stage_name_)) {
558 return false;
559 }
560 // TODO(ezhulenev): relax this condition for root node
561 return !(IsDrivenByControlDependency(*node) ||
562 DrivesControlDependency(*node));
563 }

565 // Rewrite a group of add ops into a single AddN if all input shapes are
566 // symbolically equal. If not, create AddN for equal shapes first, and then
567 // build an Add tree, minimizing the cost of broadcasts.
568 string RewriteOptimizedNodesGroup(const OptimizedNodesGroup& group) override {
569 // all new nodes will be placed under the scope of a root node
570 auto root_scope_and_name = ParseNodeScopeAndName(group.root_node->name());
571
572 // Find what shapes are present in the inputs of absorbed nodes
573 std::unordered_map<string, std::vector<InputAndShape>> shape_sig_to_inputs;
574 for (const auto& input : group.inputs) {
575 shape_sig_to_inputs[ShapeSignature(input.shape)].push_back(input);
576 }
577
578 // Collect all the shapes from representative elements
579 std::vector<TensorShapeProto> shapes;
580 shapes.reserve(shape_sig_to_inputs.size());
581 for (const auto& el : shape_sig_to_inputs)
582 shapes.push_back(el.second[0].shape);
583
584 // If all inputs have the same shape, rewrite whole group with a single AddN
585 if (shapes.size() == 1) {
586 string node_name = OptimizedNodeName(root_scope_and_name);
587 AddInputsOfSymbolicallyEqualShape(*group.root_node, node_name,
588 group.inputs);
589 return node_name;
590 }
591
592 // For inputs of different shapes:
593 // 1. Rewrite inputs of the same shape using AddN (leaf nodes)
594 // 2. Build a tree of Add nodes, minimizing cost of broadcast
595 std::sort(shapes.begin(), shapes.end(),
596 [](const TensorShapeProto& left, const TensorShapeProto& right) {
597 return CompareSymbolicallyShapedTensorSizes(left, right);
598 });
599
600 // optimized name for leaf AddN nodes
601 auto leaf_node_name = [&root_scope_and_name, this](int i) {
602 return OptimizedNodeName(root_scope_and_name,
603 strings::StrCat("Leaf_", i));
604 };
605 // optimized name for internal nodes of a tree built up from AddN leaves
606 auto internal_node_name = [&root_scope_and_name, this](int i) {
607 return OptimizedNodeName(root_scope_and_name,
608 strings::StrCat("Internal_", i));
609 };
610
611 // Add/AddN nodes that must be added to the tree
612 std::deque<InputAndShape> add_ops;
613
614 // Prepare leaf AddN nodes for inputs of equal shape
615 for (int i = 0; i < shapes.size(); ++i) {
616 const auto node_name = leaf_node_name(i);
617 const auto& inputs = shape_sig_to_inputs[ShapeSignature(shapes[i])];
618 add_ops.push_back(AddInputsOfSymbolicallyEqualShape(*group.root_node,
619 node_name, inputs));
620 }
621
622 // Build up a tree of Add ops
623 int internal_nodes = 0;
624 do {
625 const InputAndShape lhs = add_ops.front();
626 add_ops.pop_front();
627 const InputAndShape rhs = add_ops.front();
628 add_ops.pop_front();
629 string name = add_ops.empty() ? OptimizedNodeName(root_scope_and_name)
630 : internal_node_name(internal_nodes++);
631 InputAndShape add = AddAggregatedInputs(*group.root_node, name, lhs, rhs);
632 add_ops.push_front(add);
633 } while (add_ops.size() > 1);
634
635 InputAndShape optimized_root_node = add_ops.front();
636 return optimized_root_node.input;
637 }
638
639 // Add 'AddN' node to aggregate inputs of symbolically equal shape
640 InputAndShape AddInputsOfSymbolicallyEqualShape(
641 const NodeDef& root_node, const string& node_name,
642 const std::vector<InputAndShape>& inputs) {
643 CHECK(!inputs.empty()) << "Inputs must be non-empty";
644
645 // Do not create redundant AddN nodes
646 if (inputs.size() == 1) {
647 return inputs[0];
648 }
649
650 // get shape from representative element
651 auto shape = inputs[0].shape;
652
653 // copy attributes from a root node

664 ctx_.node_map->AddOutput(inputAndShape.input, node_name);
665 node->add_input(inputAndShape.input);
666 }
667
668 AddToOptimizedNodes(node);
669 return InputAndShape(node_name, shape);
670 }

698// Use the commutativity and (left- and right-) distributive property of
699// multiplication over addition to hoist common factors out of aggregate nodes
700// where all the inputs are Mul nodes. This pattern occurs frequently in
701// regularization terms for the gradients during training.
702//
703// For example, we can rewrite an expression of the form:
704// AddN(Mul(x, y1), Mul(y2, x), Mul(x, y3), ... Mul(x, yn))
705// to the following:
706// Mul(x, AddN(y1, y2, y3, ... yn))
707class HoistCommonFactorOutOfAggregation : public ArithmeticOptimizerStage {
708 public:
709 explicit HoistCommonFactorOutOfAggregation(
710 const GraphOptimizerContext& ctx,
711 const ArithmeticOptimizerContext& ctx_ext)
712 : ArithmeticOptimizerStage("HoistCommonFactor", ctx, ctx_ext) {}
713 ~HoistCommonFactorOutOfAggregation() override = default;
714
715 bool IsSupported(const NodeDef* node) const override {
716 return IsAggregate(*node) && NumNonControlInputs(*node) > 1 &&
717 !IsRewritten(node);
718 }
719
720 Status TrySimplify(NodeDef* node, string* simplified_node_name) override {
721 TF_RETURN_IF_ERROR(EnsureNodeIsSupported(node));
722
723 std::set<string> common_factors;
724 std::vector<string> ctrl_deps;
725 TF_RETURN_IF_ERROR(GetCommonFactors(node, &common_factors, &ctrl_deps));
726
727 if (common_factors.size() == 1) {
728 const string& common_factor = *common_factors.begin();
729
730 // Gather up the non-shared factors
731 bool shapes_match = true;
732 std::vector<string> unique_factors;
733 TF_RETURN_IF_ERROR(GetUniqueFactors(node, common_factor, &shapes_match,
734 &unique_factors));
735
736 if (shapes_match) {
737 NodeDef* input_0;
738 TF_RETURN_IF_ERROR(GetInputNode(node->input(0), &input_0));
739
740 // Use a copy of the first Mul node for the outer multiplication.
741 NodeDef* new_mul_node = AddCopyNode(OuterMulNodeName(node), input_0);
742 // And a copy of aggregation node as one of the inner operands
743 NodeDef* new_add_node = AddCopyNode(InnerAddNodeName(node), node);
744
745 new_mul_node->set_device(node->device());
746 new_mul_node->set_input(0, common_factor);
747 new_mul_node->set_input(1, new_add_node->name());
748
749 ctx_.node_map->AddOutput(common_factor, new_mul_node->name());
750 ctx_.node_map->AddOutput(new_add_node->name(), new_mul_node->name());
751
752 // Hoist non-shared factors up into the new AddN node.
753 for (int i = 0; i < unique_factors.size(); ++i) {
754 const string& unique_factor_i = unique_factors[i];
755 new_add_node->set_input(i, unique_factor_i);
756 ctx_.node_map->AddOutput(unique_factor_i, new_add_node->name());
757 }
758
759 // Add control deps on add node
760 for (const string& ctrl_dep : ctrl_deps) {
761 *new_add_node->add_input() = ctrl_dep;
762 ctx_.node_map->AddOutput(NodeName(ctrl_dep), new_add_node->name());
763 }
764
765 // optimize new inner aggregation node
766 AddToOptimizationQueue(new_add_node);
767 // do not optimize the same node twice
768 rewritten_nodes_.insert(node->name());
769 *simplified_node_name = new_mul_node->name();
770 }
771 }
772 return Status::OK();
773 }
774
775 private:
776 // Get a name for new outer Mul node
777 string OuterMulNodeName(const NodeDef* node) const {
778 auto scope_and_name = ParseNodeScopeAndName(node->name());
779 return OptimizedNodeName(scope_and_name, "Mul");
780 }
781
782 // Get a name new inner Add node
783 string InnerAddNodeName(const NodeDef* node) const {
784 auto scope_and_name = ParseNodeScopeAndName(node->name());
785 return OptimizedNodeName(scope_and_name, "Add");
786 }

788 // Determine the set of common factors if the input nodes are all Mul nodes.
789 Status GetCommonFactors(const NodeDef* node, std::set<string>* common_factors,
790 std::vector<string>* ctrl_deps) const {
791 CHECK(common_factors->empty());
792
793 for (int i = 0; i < node->input_size(); ++i) {
794 if (i > 0 && common_factors->empty()) break;
795 if (IsControlInput(node->input(i))) {
796 ctrl_deps->push_back(node->input(i));
797 continue;
798 }
799 NodeDef* input;
800 TF_RETURN_IF_ERROR(GetInputNode(node->input(i), &input));
801
802 if (!IsMul(*input)) {
803 common_factors->clear();
804 break;
805 }
806
807 std::set<string> factors_i{input->input(0), input->input(1)};
808 if (i == 0) {
809 std::swap(*common_factors, factors_i);
810 } else {
811 std::set<string> intersection;
812 std::set_intersection(
813 factors_i.begin(), factors_i.end(), common_factors->begin(),
814 common_factors->end(),
815 std::inserter(intersection, intersection.begin()));
816 std::swap(*common_factors, intersection);
817 }
818 for (int i = 2; i < input->input_size(); ++i) {
819 ctrl_deps->push_back(input->input(i));
820 }
821 }
822 return Status::OK();
823 }
824
825 // Gather up the non-shared factors (the y's in the example).
826 // Unless the aggregation is Add, we have to make sure that all the y's
827 // have the same shape since the other aggregation ops do not support
828 // broadcasting.
829 Status GetUniqueFactors(const NodeDef* node, const string& common_factor,
830 bool* shapes_match,
831 std::vector<string>* unique_factors) const {
832 *shapes_match = true;
833 unique_factors->reserve(node->input_size());
834
835 for (int i = 0; i < node->input_size() && shapes_match; ++i) {
836 const string& input = node->input(i);
837 if (IsControlInput(input)) {
838 break;
839 }
840 NodeDef* mul_node;
841 TF_RETURN_IF_ERROR(GetInputNode(input, &mul_node));
842 const int unique_factor_index =
843 mul_node->input(0) == common_factor ? 1 : 0;
844 unique_factors->push_back(mul_node->input(unique_factor_index));
845 if (i > 0 && !IsAdd(*node)) {
846 OpInfo::TensorProperties lhs;
847 OpInfo::TensorProperties rhs;
848 TF_RETURN_IF_ERROR(GetTensorProperties(unique_factors->front(), &lhs));
849 TF_RETURN_IF_ERROR(GetTensorProperties(unique_factors->back(), &rhs));
850 *shapes_match = ShapesSymbolicallyEqual(lhs, rhs);
851 }
852 }
853 return Status::OK();
854 }

867// Binary associative ops can be re-ordered to minimize the number of broadcasts
868// and the size of a temporary tensors.
869//
870// Example: [a, c] - scalars, [b, d] - matrices
871// @ - binary associative op (Add or Mul)
872// @* - broadcast
873//
874// @ @*
875// / \ / \
876// @* @* -> @ @
877// / \ / \ / \ / \
878// a b c d a c b d
879class MinimizeBroadcasts : public ArithmeticNodesGroupOptimizerStage {
880 public:
881 explicit MinimizeBroadcasts(const GraphOptimizerContext& ctx,
882 const ArithmeticOptimizerContext& ctx_ext)
883 : ArithmeticNodesGroupOptimizerStage("MinimizeBroadcasts", ctx, ctx_ext) {
884 }
885 ~MinimizeBroadcasts() override = default;
886
887 bool IsSupported(const NodeDef* node) const override {
888 if (!IsBinaryAssociative(*node)) return false;
889
890 // has a symbolically defined shape with broadcastable inputs
891 OpInfo::TensorProperties properties;
892 Status has_properties = GetTensorProperties(node->name(), &properties);
893 return has_properties.ok() && ShapeIsSymbolicallyDefined(properties) &&
894 HasAllInputsBroadcastableToShape(*node, properties);
895 }
896
897 protected:
898 bool IsBinaryAssociative(const NodeDef& node) const {
899 return IsMul(node) || IsAdd(node);
900 }
901
902 bool IsSameOp(const OptimizedNodesGroup& group, const NodeDef& node) const {
903 return group.root_node->op() == node.op();
904 }
905
906 // Check if a node can be absorbed by current OptimizedNodesGroup
907 bool IsAbsorbableByOptimizedNodesGroup(const OptimizedNodesGroup& group,
908 const string& input) const override {
909 NodeDef* node;
910 Status node_status = GetInputNode(input, &node);
911 if (!node_status.ok()) return false;
912
913 if (!IsSameOp(group, *node)) {
914 return false;
915 }
916 if (IsInPreserveSet(*node) || IsAlreadyOptimized(*node)) {
917 return false;
918 }
919 if (IsDrivenByControlDependency(*node) || DrivesControlDependency(*node)) {
920 return false;
921 }
922 if (!IsOnTheSameDevice(group, *node)) {
923 return false;
924 }
925 // Optimized nodes updated in place, and that would break the graph, if the
926 // node has multiple output consumers
927 if (NumNonControlOutputs(*node, *ctx_.node_map) != 1) {
928 return false;
929 }
930 // All input shapes must be broadcastable to the node shape
931 OpInfo::TensorProperties properties;
932 Status has_properties = GetTensorProperties(input, &properties);
933 return has_properties.ok() &&
934 HasAllInputsBroadcastableToShape(*node, properties);
935 }

937 std::size_t CountUniqueShapes(const std::vector<InputAndShape>& inputs) {
938 std::set<string> sigs;
939 for (const auto& ias : inputs) {
940 sigs.insert(ShapeSignature(ias.shape));
941 }
942 return sigs.size();
943 }
944
945 string RewriteOptimizedNodesGroup(const OptimizedNodesGroup& group) override {
946 if (CountUniqueShapes(group.inputs) <= 1) {
947 // nothing to optimize when all shapes are the same
948 return group.root_node->name();
949 }
950
951 auto num_nodes = /*root*/ 1 + group.optimized_nodes.size();
952 auto num_inputs = group.inputs.size();
953 CHECK_EQ(num_nodes, num_inputs - 1)
954 << "Can't build a tree with " << num_inputs << " inputs, using "
955 << num_nodes << "binary op nodes.";
956
957 std::deque<InputAndShape> add_ops(group.inputs.begin(), group.inputs.end());
958 std::deque<NodeDef*> optimized_nodes(group.optimized_nodes.begin(),
959 group.optimized_nodes.end());
960
961 // sort inputs by it's shape from smallest to largest
962 std::stable_sort(add_ops.begin(), add_ops.end(),
963 [](const InputAndShape& lhs, const InputAndShape& rhs) {
964 return CompareSymbolicallyShapedTensorSizes(lhs.shape,
965 rhs.shape);
966 });
967
968 // If there is an odd number of inputs, last one is the largest, and we want
969 // to attach it to the root node, to build a well balanced tree.
970 std::deque<InputAndShape> add_ops_leftover;
971 if (add_ops.size() % 2 != 0) {
972 add_ops_leftover.push_back(add_ops.back());
973 add_ops.pop_back();
974 }
975
976 // At this point it's guaranteed that add_ops have even number of inputs.
977 do {
978 const InputAndShape lhs = add_ops.front();
979 add_ops.pop_front();
980 const InputAndShape rhs = add_ops.front();
981 add_ops.pop_front();
982
983 NodeDef* node;
984 if (!optimized_nodes.empty()) {
985 // re-purpose optimized nodes to build a new tree
986 node = optimized_nodes.front();
987 optimized_nodes.pop_front();
988 } else {
989 // or use root node if none optimized nodes left
990 node = group.root_node;
991 }
992 InputAndShape updated_node = UpdateInputs(lhs.input, rhs.input, node);
993
994 // Pushing updated node to the back of a deque will create a wide and
995 // short tree, pushing to the front will create a tall tree. We prefer to
996 // get a wide tree, it minimizes the potential number of temporary tensors
997 // required to keep in memory, though sometimes we can go up to prevent
998 // propagating a brodcast from leaves to the root. Example:
999 //
1000 // inputs: [s, s, s, M] (s - scalar, M - matrix)
1001 // @* - op with broadcast
1002 //
1003 // (only push_back) @* (push_front first op)
1004 // / \
1005 // @* @ M
1006 // / \ / \
1007 // @ @* -> @ s
1008 // / \ / \ / \
1009 // s s s M s s
1010 if (add_ops.size() >= 2 &&
1011 CompareSymbolicallyShapedTensorSizes(add_ops.at(0).shape,
1012 add_ops.at(1).shape)) {
1013 add_ops.push_front(updated_node);
1014 } else {
1015 add_ops.push_back(updated_node);
1016 }
1017 } while (add_ops.size() > 1);
1018 CHECK_EQ(1, add_ops.size());
1019
1020 // attach the largest tensor to the root op
1021 if (!add_ops_leftover.empty()) {
1022 const InputAndShape lhs = add_ops.front();
1023 add_ops.pop_front();
1024 const InputAndShape rhs = add_ops_leftover.front();
1025 InputAndShape updated_node =
1026 UpdateInputs(lhs.input, rhs.input, group.root_node);
1027 add_ops.push_back(updated_node);
1028 }
1029
1030 return add_ops.front().input;
1031 }
1032
1033 InputAndShape UpdateInputs(const string& input_0, const string& input_1,
1034 NodeDef* node) {
1035 string old_input_0 = node->input(0);
1036 string old_input_1 = node->input(1);
1037
1038 // Update inputs only if they changed
1039 if (old_input_0 != input_0 || old_input_1 != input_1) {
1040 node->set_input(0, input_0);
1041 node->set_input(1, input_1);
1042 // Invalidate node properties (shape)
1043 ctx_.graph_properties->ClearOutputProperties(node->name());
1044 ctx_.graph_properties->ClearInputProperties(node->name());
1045 // Update the node map
1046 ctx_.node_map->RemoveOutput(NodeName(old_input_0), node->name());
1047 ctx_.node_map->RemoveOutput(NodeName(old_input_1), node->name());
1048 ctx_.node_map->AddOutput(NodeName(input_0), node->name());
1049 ctx_.node_map->AddOutput(NodeName(input_1), node->name());
1050 // Add updated node to optimization queue
1051 AddToOptimizationQueue(node);
1052 }
1053
1054 // Do not add updated node to any other group
1055 AddToOptimizedNodes(node);
1056
1057 TensorShapeProto shape; // shape is not important at this point
1058 return InputAndShape(node->name(), shape);
1059 }

1077 TF_RETURN_IF_ERROR(EnsureNodeIsSupported(node));
1078
1079 NodeDef* input;
1080 TF_RETURN_IF_ERROR(GetInputNode(node->input(0), &input));
1081 NodeDef* node_perm;
1082 TF_RETURN_IF_ERROR(GetInputNode(node->input(1), &node_perm));
1083 if (!IsConstant(*node_perm)) {

1085 }

1090 NodeDef* input_perm;

1092 if (!IsConstant(*input_perm)) {
1093 return Status::OK();

1096 TF_RETURN_IF_ERROR(GetPermutation(*input_perm, &input_perm_values));
1097 if (AreInversePermutations(node_perm_values, input_perm_values)) {
1098 *simplified_node_name = input->input(0);
1099 }
1100 } else {
1101 // Remove simple identity transposes.
1102 if (IsIdentityPermutation(node_perm_values)) {
1103 *simplified_node_name = node->input(0);
1104 }

1106 return Status::OK();
1107 }

1114 perm64->reserve(perm32.size());
1115 for (int val : perm32) {
1116 perm64->push_back(static_cast<int64>(val));
1117 }
1118 return Status::OK();
1119 }
1120 if (ValuesFromConstNode(node_perm, perm64)) {
1121 return Status::OK();
1122 }
1123 return errors::InvalidArgument("Couldn't extract permutation from ",
1124 node_perm.name());
1125 }

1129 if (a.size() != b.size()) {
1130 return false;
1131 }
1132 for (int i = 0; i < a.size(); ++i) {
1133 if (a[b[i]] != i) {
1134 return false;
1135 }
1136 }
1137 return true;
1138 }

1141 for (int64 i = 0; i < perm.size(); ++i) {
1142 if (i != perm[i]) {
1143 return false;
1144 }
1145 }
1146 return true;
1147 }

1289std::size_t UniqueNodes::ComputeSignature(const NodeDef& node) const {
1290 std::size_t h = std::hash<string>{}(node.op());
1291 h ^= std::hash<string>{}(node.device());
1292 for (const auto& input : node.input()) {
1293 int pos;
1294 string node_name = ParseNodeName(input, &pos);
1295 h ^= std::hash<string>{}(node_name);
1296 h ^= static_cast<std::size_t>(pos);
1297 }
1298 for (const auto& attr : node.attr()) {
1299 h ^= std::hash<string>{}(attr.first);
1300 string tmp;
1301 attr.second.AppendToString(&tmp);
1302 h ^= std::hash<string>{}(tmp);
1303 }
1304 return h;
1305}

1307bool UniqueNodes::SameNode(const NodeDef& node1, const NodeDef& node2) const {
1308 if (node1.op() != node2.op()) {
1309 return false;
1310 }
1311 if (node1.device() != node2.device()) {
1312 return false;
1313 }
1314 if (node1.input_size() != node2.input_size()) {
1315 return false;
1316 }
1317 if (node1.attr_size() != node2.attr_size()) {
1318 return false;
1319 }
1320
1321 // Compare inputs.
1322 if (IsCommutative(node1)) {
1323 std::vector<string> inputs1(node1.input().begin(), node1.input().end());
1324 std::vector<string> inputs2(node2.input().begin(), node2.input().end());
1325 std::sort(inputs1.begin(), inputs1.end());
1326 std::sort(inputs2.begin(), inputs2.end());
1327 return inputs1 == inputs2;
1328 } else {
1329 std::vector<string> regular_inputs1;
1330 std::vector<string> regular_inputs2;
1331 std::vector<string> ctrl_inputs1;
1332 std::vector<string> ctrl_inputs2;
1333 for (int index = 0; index < node1.input_size(); ++index) {
1334 if (IsControlInput(node1.input(index))) {
1335 ctrl_inputs1.push_back(node1.input(index));
1336 ctrl_inputs2.push_back(node2.input(index));
1337 } else {
1338 regular_inputs1.push_back(node1.input(index));
1339 regular_inputs2.push_back(node2.input(index));
1340 }
1341 }
1342 if (regular_inputs1 != regular_inputs2) {
1343 return false;
1344 }
1345 std::sort(ctrl_inputs1.begin(), ctrl_inputs1.end());
1346 std::sort(ctrl_inputs2.begin(), ctrl_inputs2.end());
1347 if (ctrl_inputs1 != ctrl_inputs2) {
1348 return false;
1349 }
1350 }
1351
1352 // Compare attributes.
1353 if (node1.attr().size() != node2.attr().size()) {
1354 return false;
1355 }
1356 for (const auto& attr1 : node1.attr()) {
1357 auto it = node2.attr().find(attr1.first);
1358 if (it == node2.attr().end()) {
1359 return false;
1360 }
1361 const auto& attr2 = *it;
1362 string val1;
1363 attr1.second.AppendToString(&val1);
1364 string val2;
1365 attr2.second.AppendToString(&val2);
1366 if (val1 != val2) {
1367 return false;
1368 }
1369 }
1370
1371 return true;
1372}

1436
1437void ArithmeticOptimizer::DedupComputations() {
1438 bool stop = true;
1439 SimpleGraphView graph_view;
1440 if (!graph_view.Initialize(*optimized_graph_).ok()) {
1441 LOG(WARNING) << "Failed to build SimpleGraphView.";
1442 return;
1443 }
1444 std::set<int> duplicates;
1445 do {
1446 stop = true;
1447 UniqueNodes nodes;
1448 for (int i = 0; i < optimized_graph_->node_size(); ++i) {
1449 if (duplicates.find(i) != duplicates.end()) {
1450 continue;
1451 }
1452 NodeDef* node = optimized_graph_->mutable_node(i);
1453 if (!CanDedup(*node)) {
1454 continue;
1455 }
1456 NodeDef* rep = nodes.FindOrAddRepresentative(node);
1457 if (rep == node) {
1458 continue;
1459 }
1460 // If either node feeds an inplace op, deduping them may cause data races.
1461 // For example: If we dedup nodes initializing two independent inplace
1462 // accumulations, they will write to the same buffer, clobbering each
1463 // other's results.
1464 if (FeedsInPlaceOp(graph_view, *rep) ||
1465 FeedsInPlaceOp(graph_view, *node)) {
1466 continue;
1467 }
1468 const std::set<NodeDef*>& fanouts = node_map_->GetOutputs(node->name());
1469 for (NodeDef* fanout : fanouts) {
1470 for (int i = 0; i < fanout->input_size(); ++i) {
1471 string* name = fanout->mutable_input(i);
1472 int position;
1473 const string nodename = ParseNodeName(*name, &position);
1474 if (nodename == node->name()) {
1475 // Update name in-place.
1476 if (position > 0) {
1477 *name = StrCat(rep->name(), ":", position);
1478 } else if (position == 0) {
1479 *name = rep->name();
1480 } else {
1481 *name = StrCat("^", rep->name());
1482 }
1483 node_map_->AddOutput(rep->name(), fanout->name());
1484 }
1485 }
1486 }
1487 duplicates.insert(i);
1488 stop = false;
1489 }
1490 } while (!stop);
1491
1492 // Delete duplicates
1493 if (fetch_nodes_known_ && !duplicates.empty()) {
1494 int last = optimized_graph_->node_size() - 1;
1495 for (auto it = duplicates.rbegin(); it != duplicates.rend(); ++it) {
1496 int index = *it;
1497 optimized_graph_->mutable_node()->SwapElements(index, last);
1498 last--;
1499 }
1500 optimized_graph_->mutable_node()->DeleteSubrange(last + 1,
1501 duplicates.size());
1502 // Rebuild the NodeMap which was invalidated by the node swapping above.
1503 node_map_.reset(new NodeMap(optimized_graph_));
1504 }
1505}

1515 break;
1516 }

1521// TODO(ezhulenev): extract each individual simplify rewrite into separate
1522// ArithmeticOptimizerStage
1523string ArithmeticOptimizer::TrySimplifyAndReplaceUses(
1524 const NodeDef* node, SetVector<NodeDef*>* nodes_to_simplify) {
1525 // Remove involutions applied twice.
1526 if (IsInvolution(*node)) {
1527 // An involution is an element-wise function f(x) that is its own inverse,
1528 // i.e. f(f(x)) = x. If we can find a chain of ops
1529 // f->op1->op2->...opn->f
1530 // where op1 through opn preserve the values of their inputs, we can remove
1531 // the two instances of the involution from the graph, since they cancel
1532 // each other.
1533 NodeDef* tail =
1534 GetTailOfValuePreservingChain(*node, *node_map_, nodes_to_preserve_);
1535 NodeDef* involution = node_map_->GetNode(tail->input(0));
1536 if (involution->op() == node->op()) {
1537 // Skip both *node and *involution since they cancel each other.
1538 if (tail == node) {
1539 // The two nodes to eliminate are adjacent.
1540 return involution->input(0);
1541 } else {
1542 tail->set_input(0, involution->input(0));
1543 node_map_->UpdateInput(tail->name(), involution->name(),
1544 involution->input(0));
1545 return node->input(0);
1546 }
1547 }
1548 }
1549
1550 if (node->op() == "Reshape") {
1551 // Reshape
1552 // ^
1553 // |
1554 // Reshape
1555 // ^
1556 // |
1557 // input
1558 //
1559 // becomes
1560 //
1561 // Reshape <-+
1562 // |
1563 // Reshape |
1564 // ^ |
1565 // | |
1566 // input ---+
1567 NodeDef* reshape = const_cast<NodeDef*>(node);
1568 int output_pos = 0;
1569 string input_node_name = ParseNodeName(reshape->input(0), &output_pos);
1570 const NodeDef* input = node_map_->GetNode(input_node_name);
1571 if (input->op() == "Reshape" && !HasControlInputs(*input)) {
1572 reshape->set_input(0, input->input(0));
1573 node_map_->UpdateInput(reshape->name(), input->name(), input->input(0));
1574 nodes_to_simplify->PushBack(reshape);
1575 return reshape->name();
1576 }
1577
1578 // If the reshape is a no-op, forward its input to its consumers, unless it
1579 // anchors a control dependency since we want to make sure that control
1580 // dependency is triggered.
1581 if (ReshapeIsIdentity(*reshape, *input, output_pos, *graph_properties_) &&
1582 !HasControlInputs(*reshape)) {
1583 return reshape->input(0);
1584 }
1585 }
1586
1587 if (node->op() == "Transpose") {
1588 // Reorder Cast and Transpose if beneficial.
1589 //
1590 // A common pattern after the layout optimizer is casting an uint8 NHWC
1591 // image to float before transposing it to NCHW. It is beneficial to reorder
1592 // the cast and the transpose to make the transpose process smaller amount
1593 // of data. This optimization converts
1594 // Transpose(Cast(image, dst_type), perm)
1595 // to
1596 // Cast(Transpose(image, perm), dst_type)
1597 // when sizeof(image.type) < sizeof(dst_type).
1598 //
1599 // TODO(jingyue): This optimization can be generalized to a cast followed by
1600 // a chain of ops that merely reorder elements (e.g. Reshape and
1601 // DepthToSpace).
1602 const NodeDef* transpose = node;
1603 string dontcare;
1604 string device;
1605 // This optimization can be dangerous on devices other than CPU and GPU. The
1606 // transpose might not be implemented for image.type, or might be slower
1607 // with image.type than with dst_type.
1608 if (DeviceNameUtils::SplitDeviceName(transpose->device(), &dontcare,
1609 &device) &&
1610 (str_util::StrContains(device, DEVICE_CPU) ||
1611 str_util::StrContains(device, DEVICE_GPU))) {
1612 const NodeDef* cast = node_map_->GetNode(transpose->input(0));
1613 if (cast->op() == "Cast") {
1614 const NodeDef* input = node_map_->GetNode(cast->input(0));
1615 const DataType src_type = GetSourceDataType(*cast);
1616 const DataType dst_type = GetDestinationDataType(*cast);
1617 if (IsNumberType(src_type) && IsNumberType(dst_type) &&
1618 DataTypeSize(src_type) < DataTypeSize(dst_type) &&
1619 !OptimizedNodeExists(*cast, DataTypeString(dst_type)) &&
1620 !OptimizedNodeExists(*transpose, DataTypeString(src_type))) {
1621 NodeDef* new_transpose = AddNode(*transpose, DataTypeString(src_type),
1622 /*copy_node=*/true);
1623 (*new_transpose->mutable_attr())["T"].set_type(src_type);
1624 new_transpose->set_input(0, cast->input(0));
1625 node_map_->AddOutput(input->name(), new_transpose->name());
1626 node_map_->AddOutput(NodeName(new_transpose->input(1)),
1627 new_transpose->name());
1628
1629 NodeDef* new_cast =
1630 AddNode(*cast, DataTypeString(dst_type), /*copy_node=*/true);
1631 new_cast->set_input(0, new_transpose->name());
1632 node_map_->AddOutput(new_transpose->name(), new_cast->name());
1633
1634 nodes_to_simplify->PushBack(new_transpose);
1635 ForwardControlDependencies(new_transpose, {cast, node});
1636 return new_cast->name();
1637 }
1638 }
1639 }
1640 }
1641
1642 // Fold a multiply of a scalar into the following convolution. This folding
1643 // can jump across nodes that merely reorders data (such as reshape and
1644 // transpose). For example, we can optimize
1645 //
1646 //
1647 // Conv2D
1648 // / \
1649 // Transpose weights
1650 // |
1651 // Mul
1652 // / \
1653 // inputs 255.0
1654 //
1655 // to
1656 //
1657 // Conv2D
1658 // / \
1659 // Transpose Mul
1660 // | / \
1661 // | weights 255.0
1662 // |
1663 // inputs
1664 //
1665 // when `weights` are constant. `Mul` in the optimized graph can be
1666 // constant-folded.
1667 //
1668 // TODO(jingyue): Fold scalar multiplies to Conv?DBackpropFilter and
1669 // Conv?DBackpropInput.
1670 if (node->op() == "Conv2D" || node->op() == "Conv3D") {
1671 NodeDef* conv = const_cast<NodeDef*>(node);
1672 const NodeDef* weights = node_map_->GetNode(NodeName(conv->input(1)));
1673 // Fold the multiply to conv only when the weights are constant, so the
1674 // multiply can be constant-folded. TODO(jingyue): When the weights aren't
1675 // constant, this should also help performance a bit and memory usage a lot,
1676 // since the weights tend to be smaller than the activations.
1677 if (weights->op() == "Const" &&
1678 !OptimizedNodeExists(*weights, StrCat("scaled_", conv->name()))) {
1679 const NodeDef* source = node_map_->GetNode(
1680 GetTailOfValuePreservingChain(*node, *node_map_, nodes_to_preserve_)
1681 ->input(0));
1682 if (source->op() == "Mul" &&
1683 node_map_->GetOutputs(source->name()).size() == 1) {
1684 const NodeDef* mul = source;
1685 // `scale` is the scalar multiplier, and `other` is the other operand.
1686 // TODO(jingyue): handle the case where `scale` is 0-th operand.
1687 const NodeDef* scale = node_map_->GetNode(mul->input(1));
1688 const NodeDef* other = node_map_->GetNode(mul->input(0));
1689 if (scale->op() == "Const" && scale->attr().at("dtype").type() ==
1690 weights->attr().at("dtype").type()) {
1691 const TensorProto& scale_tensor = scale->attr().at("value").tensor();
1692 // Test whether `scale` is a scalar.
1693 if (scale_tensor.has_tensor_shape() &&
1694 scale_tensor.tensor_shape().dim_size() == 0) {
1695 // Create new node `scaled_weights`.
1696 NodeDef* scaled_weights = AddNode(
1697 *weights, StrCat("scaled_", conv->name()), /*copy_node=*/false);
1698 scaled_weights->set_op("Mul");
1699 scaled_weights->set_device(weights->device());
1700 (*scaled_weights->mutable_attr())["T"] =
1701 weights->attr().at("dtype");
1702 nodes_to_simplify->PushBack(scaled_weights);
1703
1704 // Link in its inputs.
1705 scaled_weights->add_input(conv->input(1));
1706 node_map_->AddOutput(weights->name(), scaled_weights->name());
1707 scaled_weights->add_input(mul->input(1));
1708 node_map_->AddOutput(scale->name(), scaled_weights->name());
1709 ForwardControlDependencies(scaled_weights, {source});
1710
1711 // Update `conv`'s weights to `scaled_weights`.
1712 conv->set_input(1, scaled_weights->name());
1713 node_map_->UpdateInput(conv->name(), weights->name(),
1714 scaled_weights->name());
1715 nodes_to_simplify->PushBack(conv);
1716
1717 // Update `mul`'s consumer to bypass `mul` because it's folded to
1718 // the weights.
1719 CHECK_EQ(node_map_->GetOutputs(mul->name()).size(), 1);
1720 NodeDef* consumer_of_mul =
1721 *node_map_->GetOutputs(mul->name()).begin();
1722 consumer_of_mul->set_input(0, mul->input(0));
1723 node_map_->UpdateInput(consumer_of_mul->name(), mul->name(),
1724 other->name());
1725 nodes_to_simplify->PushBack(consumer_of_mul);
1726 return conv->name();
1727 }
1728 }
1729 }
1730 }
1731 }
1732
1733 if (node->op() == "Mul" && node->input(0) == node->input(1) &&
1734 !OptimizedNodeExists(*node, "square")) {
1735 const DataType type = GetDataTypeFromAttr(*node, "T");
1736 bool is_complex = (type == DT_COMPLEX64) || (type == DT_COMPLEX128);
1737 string dontcare;
1738 string device;
1739 bool is_on_cpu =
1740 DeviceNameUtils::SplitDeviceName(node->device(), &dontcare, &device) &&
1741 str_util::StrContains(device, DEVICE_CPU);
1742 if (!is_complex || is_on_cpu) {
1743 NodeDef* new_square_node = AddNode(*node, "square", /*copy_node=*/true);
1744 new_square_node->set_op("Square");
1745 for (int i = 1; i < new_square_node->input_size(); ++i) {
1746 new_square_node->set_input(i - 1, new_square_node->input(i));
1747 }
1748 new_square_node->mutable_input()->RemoveLast();
1749 return new_square_node->name();
1750 }
1751 }
1752
1753 if (IsAggregate(*node) && NumNonControlInputs(*node) > 0) {
1754 // Discard aggregate nodes with a single input and no control dependencies.
1755 if (node->input_size() == 1) {
1756 return node->input(0);
1757 }
1758
1759 // Try to rewrite aggregations of N >= 2 identical terms (possibly due
1760 // to deduping or other rewrites) so we can get rid of the sum entirely.
1761 // The expression (using AddN as an example of an aggregate op):
1762 // AddN(x, x, x, ... ,x)
1763 // <-- N terms -->
1764 // can be rewritten to
1765 // Mul(Const(N), x))
1766 //
1767 bool all_equal = true;
1768 int num_inputs = 1;
1769 for (int i = 1; i < node->input_size(); ++i) {
1770 if (IsControlInput(node->input(i))) {
1771 break;
1772 }
1773 ++num_inputs;
1774 if (node->input(i) != node->input(0)) {
1775 all_equal = false;
1776 break;
1777 }
1778 }
1779 if (all_equal && !OptimizedNodeExists(*node, "const") &&
1780 !OptimizedNodeExists(*node, "mul")) {
1781 // 1. Create constant node with value N.
1782 const auto type = GetDataTypeFromAttr(*node, "T");
1783 Tensor t(type, TensorShape({}));
1784 Status status = SetTensorValue(type, num_inputs, &t);
1785 if (!status.ok()) {
1786 LOG(WARNING) << "Failed to create const node: "
1787 << status.error_message();
1788 return "";
1789 }
1790 TensorValue value(&t);
1791 NodeDef* new_const_node = AddNode(*node, "const", /*copy_node=*/false);
1792 status = ConstantFolding::CreateNodeDef(new_const_node->name(), value,
1793 new_const_node);
1794 if (!status.ok()) {
1795 LOG(WARNING) << "Failed to create const node: "
1796 << status.error_message();
1797 return "";
1798 }
1799 new_const_node->set_device(node->device());
1800 MaybeAddControlInput(NodeName(node->input(0)), new_const_node,
1801 optimized_graph_, node_map_.get());
1802 nodes_to_simplify->PushBack(new_const_node);
1803
1804 // 2. Replace the aggregate node with Mul(Const(N), x).
1805 NodeDef* new_mul_node = AddNode(*node, "mul", /*copy_node=*/false);
1806 new_mul_node->set_op("Mul");
1807 new_mul_node->set_device(node->device());
1808 SetDataTypeToAttr(type, "T", new_mul_node);
1809 new_mul_node->add_input(new_const_node->name());
1810 node_map_->AddOutput(new_const_node->name(), new_mul_node->name());
1811 new_mul_node->add_input(node->input(0));
1812 node_map_->AddOutput(node->input(0), new_mul_node->name());
1813
1814 ForwardControlDependencies(new_mul_node, {node});
1815 return new_mul_node->name();
1816 }
1817 }
1818
1819 // Fold Transpose into matrix multiplication.
1820 if ((node->op() == "MatMul" || node->op() == "SparseMatMul" ||
1821 node->op() == "BatchMatMul") &&
1822 !OptimizedNodeExists(*node, "fused")) {
1823 const NodeDef* a = node_map_->GetNode(node->input(0));
1824 const NodeDef* b = node_map_->GetNode(node->input(1));
1825 bool is_complex = false;
1826 if (node->op() != "SparseMatMul") {
1827 const DataType type = GetDataTypeFromAttr(*node, "T");
1828 is_complex = (type == DT_COMPLEX64) || (type == DT_COMPLEX128);
1829 }
1830 const std::set<string> foldable_transpose_ops =
1831 !is_complex ? std::set<string>{"ConjugateTranspose", "Transpose"}
1832 : (node->op() == "BatchMatMul"
1833 ? std::set<string>{"ConjugateTranspose"}
1834 : std::set<string>{"Transpose"});
1835 const bool a_is_foldable = foldable_transpose_ops.count(a->op()) > 0 &&
1836 IsInnerMatrixTransposeNode(*a, node_map_.get());
1837 const bool b_is_foldable = foldable_transpose_ops.count(b->op()) > 0 &&
1838 IsInnerMatrixTransposeNode(*b, node_map_.get());
1839 if (a_is_foldable || b_is_foldable) {
1840 NodeDef* new_op = AddNode(*node, "fused", /*copy_node=*/true);
1841 if (a_is_foldable) {
1842 const string attr_a =
1843 node->op() == "BatchMatMul" ? "adj_x" : "transpose_a";
1844 FlipBooleanAttr(attr_a, new_op);
1845 new_op->set_input(0, a->input(0));
1846 node_map_->UpdateInput(new_op->name(), a->name(), a->input(0));
1847 }
1848 if (b_is_foldable) {
1849 const string attr_b =
1850 node->op() == "BatchMatMul" ? "adj_y" : "transpose_b";
1851 FlipBooleanAttr(attr_b, new_op);
1852 new_op->set_input(1, b->input(0));
1853 node_map_->UpdateInput(new_op->name(), b->name(), b->input(0));
1854 }
1855 std::vector<const NodeDef*> deps_to_forward({node});
1856 if (a_is_foldable) {
1857 deps_to_forward.push_back(a);
1858 }
1859 if (b_is_foldable) {
1860 deps_to_forward.push_back(b);
1861 }
1862 ForwardControlDependencies(new_op, deps_to_forward);
1863 }
1864 }
1865
1866 // Fold Conj into Transpose or ConjugateTranspose.
1867 if ((node->op() == "Conj" || node->op() == "Transpose" ||
1868 node->op() == "ConjugateTranspose") &&
1869 !OptimizedNodeExists(*node, "fused")) {
1870 const NodeDef* input = node_map_->GetNode(node->input(0));
1871 const NodeDef* transpose_op = node->op() == "Conj" ? input : node;
1872 const NodeDef* conj_op = node->op() == "Conj" ? node : input;
1873
1874 if ((transpose_op->op() == "Transpose" ||
1875 transpose_op->op() == "ConjugateTranspose") &&
1876 conj_op->op() == "Conj") {
1877 NodeDef* new_op =
1878 AddNode(OptimizedNodeName(*node, "fused"), transpose_op);
1879 // Flip the type of transpose op to absorb the conjugation.
1880 new_op->set_op(transpose_op->op() == "Transpose" ? "ConjugateTranspose"
1881 : "Transpose");
1882 new_op->set_input(0, input->input(0));
1883 node_map_->UpdateInput(new_op->name(), node->name(), input->input(0));
1884 ForwardControlDependencies(new_op, {node, input});
1885 return new_op->name();
1886 }
1887 }
1888
1889 return "";
1890}

1892Status ArithmeticOptimizer::SimplifyArithmeticOps(bool can_use_shapes) {
1893 SetVector<NodeDef*> nodes_to_simplify;
1894 nodes_to_simplify.Reserve(optimized_graph_->node_size());
1895 for (int i = 0; i < optimized_graph_->node_size(); ++i) {
1896 nodes_to_simplify.PushBack(optimized_graph_->mutable_node(i));
1897 }
1898
1899 const GraphOptimizerContext ctx(&nodes_to_preserve_, optimized_graph_,
1900 graph_properties_.get(), node_map_.get());
1901 const ArithmeticOptimizerContext ctx_ext(&nodes_to_simplify);
1902
1903 // Stop pipeline after first stage returning non-empty simplified tensor name.
1904 const auto stop = [](const string& result) { return !result.empty(); };
1905 GraphOptimizerStagePipeline<string> pipeline(stop);
1906
1907 if (options_.combine_add_to_addn && can_use_shapes)
1908 pipeline.AddStage<AddOpsRewriteStage>(ctx, ctx_ext);
1909 if (options_.hoist_common_factor_out_of_aggregation && can_use_shapes)
1910 pipeline.AddStage<HoistCommonFactorOutOfAggregation>(ctx, ctx_ext);
1911 if (options_.minimize_broadcasts && can_use_shapes)
1912 pipeline.AddStage<MinimizeBroadcasts>(ctx, ctx_ext);
1913 if (options_.remove_identity_transpose && can_use_shapes)
1914 pipeline.AddStage<RemoveIdentityTranspose>(ctx, ctx_ext);
1915 if (options_.remove_redundant_bitcast)
1916 pipeline.AddStage<RemoveRedundantBitcastStage>(ctx, ctx_ext);
1917 if (options_.remove_redundant_cast)
1918 pipeline.AddStage<RemoveRedundantCastStage>(ctx, ctx_ext);
1919 if (options_.remove_negation)
1920 pipeline.AddStage<RemoveNegationStage>(ctx, ctx_ext);
1921
1922 VLOG(1) << "Simplify arithmetic ops using " << pipeline.NumStages()
1923 << " arithmetic optimization stages";
1924
1925 while (!nodes_to_simplify.Empty()) {
1926 NodeDef* node = nodes_to_simplify.PopBack();
1927
1928 // TODO(ezhulenev): move all rewrites into separate stages
1929 string simplified_tensor = "";
1930 if (options_.enable_try_simplify_and_replace) {
1931 simplified_tensor = TrySimplifyAndReplaceUses(node, &nodes_to_simplify);
1932 }
1933
1934 // if it was not simplified try to run it through all configured stages
1935 if (!stop(simplified_tensor)) {
1936 bool optimized = pipeline.PassThroughAllStages(node, &simplified_tensor);
1937 if (!optimized) {
1938 continue;
1939 }
1940 }
1941
1942 // re-wire consumers of an old node to the new one
1943 if (NodeName(simplified_tensor) != node->name()) {
1944 // Always consider simplified_tensor for further optimizations.
1945 NodeDef* simplified_node = node_map_->GetNode(simplified_tensor);
1946 if (simplified_node != nullptr) {
1947 nodes_to_simplify.PushBack(simplified_node);
1948 }
1949 // When `node` is simplified to another node rather than in-place, the
1950 // consumers of `node` are already redirected to `simplified_tensor`.
1951 // Re-push the consumers into `nodes_to_simplify` for further
1952 // optimizations.
1953 const std::set<NodeDef*> outputs = node_map_->GetOutputs(node->name());
1954 std::vector<NodeDef*> consumers(outputs.begin(), outputs.end());
1955 std::sort(consumers.begin(), consumers.end(),
1956 [](const NodeDef* n1, const NodeDef* n2) {
1957 return n1->name() < n2->name();
1958 });
1959 for (NodeDef* consumer : consumers) {
1960 // Update `consumer`'s use of `node` to `input`'s operand.
1961 for (int i = 0; i < consumer->input_size(); ++i) {
1962 int operand_pos;
1963 string operand_node_name =
1964 ParseNodeName(consumer->input(i), &operand_pos);
1965 if (operand_node_name == node->name()) {
1966 *consumer->mutable_input(i) =
1967 (operand_pos < 0
1968 ? AsControlDependency(NodeName(simplified_tensor))
1969 : simplified_tensor);
1970 }
1971 }
1972 node_map_->UpdateInput(consumer->name(), node->name(),
1973 simplified_tensor);
1974 nodes_to_simplify.PushBack(consumer);
1975 }
1976 }
1977 }
1978 return Status::OK();
1979}

1981Status ArithmeticOptimizer::Optimize(Cluster* /*cluster*/,
1982 const GrapplerItem& item,
1983 GraphDef* optimized_graph) {
1984 GrapplerItem optimized_item(item);
1985 optimized_graph_ = &optimized_item.graph;
1986
1987 // Set up helper data structures.
1988 nodes_to_preserve_ = item.NodesToPreserve();
1989 fetch_nodes_known_ = !item.fetch.empty();
1990 node_map_.reset(new NodeMap(optimized_graph_));
1991
1992 DedupComputations();
1993
1994 // Perform topological sort on the graph in order to help AddOpsRewrite to
1995 // optimize larger subgraphs starting from the roots with more inputs.
1996 TF_RETURN_IF_ERROR(TopologicalSort(optimized_graph_));
1997
1998 // Shapes are only needed in aggressive mode.
1999 graph_properties_.reset(new GraphProperties(item));
2000 const Status status = graph_properties_->InferStatically(false);
2001 const bool can_use_shapes = status.ok();
2002 if (!can_use_shapes) {
2003 VLOG(1) << "Shape inference failed." << status.error_message();
2004 }
2005
2006 // Perform the optimizations.
2007 TF_RETURN_IF_ERROR(SimplifyArithmeticOps(can_use_shapes));
2008
2009 optimized_graph->Swap(optimized_graph_);
2010 return Status::OK();
2011}
