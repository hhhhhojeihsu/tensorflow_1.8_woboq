25namespace tensorflow {
26namespace grappler {

28constexpr char kArithmeticOptimizer[] = "ArithmeticOptimizer";
29
30// Optimize TF computations by reducing the arithmetic complexity required to
31// run a model.
32class ArithmeticOptimizer : public GraphOptimizer {
33 public:
34 ArithmeticOptimizer()
35 : opt_level_(RewriterConfig::ON),
36 options_(ArithmeticOptimizerOptions::Default(RewriterConfig::ON)) {}
37
38 explicit ArithmeticOptimizer(RewriterConfig::Toggle opt_level)
39 : opt_level_(opt_level),
40 options_(ArithmeticOptimizerOptions::Default(opt_level)) {}
41
42 ~ArithmeticOptimizer() override {}
43
44 string name() const override { return "arithmetic_optimizer"; };
45
46 Status Optimize(Cluster* cluster, const GrapplerItem& item,
47 GraphDef* optimized_graph) override;
48
49 void Feedback(Cluster* cluster, const GrapplerItem& item,
50 const GraphDef& optimized_graph, double result) override;
51
52 private:
53 friend class ArithmeticOptimizerTest;
54
55 // Granular control for arithmetic optimizer stages
56 struct ArithmeticOptimizerOptions {
57 // TODO(ezhulenev): flag do disable TrySimplifyAndReplaceUses in tests.
58 // Remove when all optimizers will be migrated to separate stages.
59 bool enable_try_simplify_and_replace = true;
60 bool combine_add_to_addn = false;
61 bool hoist_common_factor_out_of_aggregation = true;
62 bool minimize_broadcasts = false;
63 bool remove_identity_transpose = true;
64 bool remove_redundant_bitcast = true;
65 bool remove_redundant_cast = true;
66 bool remove_negation = true;
67
68 // Choose which arithmetic optimizer stages will be enabled for a given
69 // optimization level by default.
70 static ArithmeticOptimizerOptions Default(
71 RewriterConfig::Toggle opt_level) {
72 ArithmeticOptimizerOptions options;
73 // TODO(ezhulenev): enable by default after 1.8 release cut
74 if (opt_level == RewriterConfig::AGGRESSIVE) {
75 options.combine_add_to_addn = true;
76 options.minimize_broadcasts = true;
77 }
78 return options;
79 }
80 };
81
82 // Returns true is a node with given name and the optimizer prefix already
83 // exists.
84 string OptimizedNodeName(const NodeDef& node, StringPiece suffix) const;
85 bool OptimizedNodeExists(const NodeDef& node, StringPiece suffix) const;
86
87 // Creates a new node in the graph, with name equal to that of node, prefixed
88 // with "ArithmeticOptimizer/" and the given suffix. Also updates node_map_,
89 // and optionally copies node into the new node if copy_node is true.
90 NodeDef* AddNode(const NodeDef& node, StringPiece suffix, bool copy_node);
91
92 // Creates a new node in the graph, prefixed with "ArithmeticOptimizer/",
93 // updates node_map_, and optionally copies *node_to_copy into the new
94 // node, if node_to_copy is not nullptr.
95 NodeDef* AddNode(const string& name, const NodeDef* node_to_copy);
96
97 // Returns true if it is safe to dedup node from the graph.
98 bool CanDedup(const NodeDef& node) const;
99
100 // Dedup redundant nodes in the graph.
101 void DedupComputations();
102
103 // Forward the control dependencies anchored on src_nodes to the target_nodes.
104 void ForwardControlDependencies(NodeDef* target_node,
105 const std::vector<const NodeDef*>& src_nodes);
106
107 // Runs peep-hole optimizations on `optimized_graph`, e.g., removing inverse
108 // transposes.
109 Status SimplifyArithmeticOps(bool can_use_shapes);
110 // Tries to simplify the expression that roots at `node` and replaces the uses
111 // of `node` to the simplified expression. Returns the name of the simplified
112 // tensor (e.g. "split:1") or an emtpy string if no simplification is
113 // performed.
114 //
115 // `node_map` stores the mapping from node names to NodeDef*, and will be
116 // updated according to the rewrite.
117 //
118 // `new_nodes` will be populated with the new nodes this function creates and
119 // updates. The caller can push these nodes into the simplification queue to
120 // optimize them further.
121 //
122 // TODO(jingyue): This interface is not suitable for optimizing nodes with
123 // multiple output tensors. We should pass in a tensor name instead of a
124 // NodeDef.
125 string TrySimplifyAndReplaceUses(const NodeDef* node,
126 SetVector<NodeDef*>* nodes_to_simplify);
127
128 RewriterConfig::Toggle opt_level_;
129 ArithmeticOptimizerOptions options_;
130
131 bool fetch_nodes_known_ = false;
132 std::unordered_set<string> nodes_to_preserve_;
133 std::unique_ptr<NodeMap> node_map_;
134 std::unique_ptr<GraphProperties> graph_properties_;
135 GraphDef* optimized_graph_ = nullptr; // Not owned.
136};
