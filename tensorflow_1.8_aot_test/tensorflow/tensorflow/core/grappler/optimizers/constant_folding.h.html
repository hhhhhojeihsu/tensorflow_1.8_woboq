30const char kConstantFoldingConst[] = "ConstantFolding";
31const char kConstantFoldingCtrl[] = "ConstantFoldingCtrl";
32
33// Constant folding optimization for a graph.
34class ConstantFolding : public GraphOptimizer {
35 public:
36 static Status CreateNodeDef(const string& name, const TensorValue& tensor,
37 NodeDef* node);
38 static string AddControlDependency(const string& input_name, GraphDef* graph,
39 NodeMap* node_map);
40
41 explicit ConstantFolding(DeviceBase* cpu_device);
42 ConstantFolding(RewriterConfig::Toggle opt_level, DeviceBase* cpu_device);
43
44 ~ConstantFolding() override {}
45
46 string name() const override { return "constant folding"; };
47
48 Status Optimize(Cluster* cluster, const GrapplerItem& item,
49 GraphDef* output) override;
50
51 void Feedback(Cluster* cluster, const GrapplerItem& item,
52 const GraphDef& optimize_output, double result) override;
53
54 private:
55 string OptimizedNodeName(const NodeDef& node, StringPiece suffix) const;
56 bool OptimizedNodeExists(const NodeDef& node, StringPiece suffix) const;
57
58 bool IsReallyConstant(const NodeDef& node) const;
59
60 Status MaterializeShapes(const GraphProperties& properties);
61
62 Status MaterializeBroadcastGradientArgs(const NodeDef& node,
63 const GraphProperties& properties);
64 Status MaterializeReductionIndices(NodeDef* node,
65 const GraphProperties& properties);
66
67 Status MaterializeConstants(const GraphProperties& properties);
68 bool IsFoldable(const NodeDef& node) const;
69
70 Status EvaluateNode(const NodeDef& node,
71 const gtl::InlinedVector<TensorValue, 4>& inputs,
72 gtl::InlinedVector<TensorValue, 4>* output) const;
73
74 Status EvaluateOneFoldable(const NodeDef& node,
75 std::vector<NodeDef>* outputs);
76
77 Status FoldNode(NodeDef* node, GraphDef* output_graph);
78
79 bool IsOnes(const NodeDef& node) const;
80 bool IsZeros(const NodeDef& node) const;
81 void ReplaceOperationWithIdentity(int input_to_forward, NodeDef* node,
82 GraphDef* graph);
83 void ReplaceOperationWithSnapshot(int input_to_forward, NodeDef* node,
84 GraphDef* graph);
85 void ReplaceSubtractionFromZeroByNegation(NodeDef* node, GraphDef* graph);
86 Status ReplaceOperationWithConstant(double value, const AttrValue& dtype_attr,
87 const TensorShapeProto& shape,
88 NodeDef* node, GraphDef* graph);
89 void ReplaceDivisionOfOnesByReciprocal(NodeDef* node, GraphDef* graph);
90 Status FoldGraph(GraphDef* output);
91
92 bool IsSimplifiableReduction(const NodeDef& node) const;
93 bool IsSimplifiableReshape(const NodeDef& node,
94 const GraphProperties& properties) const;
95 Status SimplifyGraph(GraphDef* output, GraphProperties* properties,
96 bool use_shape_info);
97
98 Status RunOptimizationPass(Cluster* cluster, const GrapplerItem& item,
99 GraphDef* output);
100
101 // Points to an externally provided device or to owned_device_;
102 RewriterConfig::Toggle opt_level_;
103 DeviceBase* cpu_device_;
104 std::unique_ptr<DeviceBase> owned_device_;
105
106 std::unique_ptr<ResourceMgr> resource_mgr_;
107 GraphDef* graph_;
108 std::unique_ptr<NodeMap> node_map_;
109 std::unordered_set<string> nodes_to_preserve_;
110 std::unordered_set<string> nodes_whitelist_;
111 std::unordered_set<string> feed_nodes_;
112 bool has_fetch_;
113 bool graph_modified_;
114};
