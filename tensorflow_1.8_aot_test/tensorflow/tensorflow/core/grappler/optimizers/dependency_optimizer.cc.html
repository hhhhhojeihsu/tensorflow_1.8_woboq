40bool RemoveInput(NodeDef* node, const string& input, NodeMap* node_map) {
41 bool removed_input = false;
42 int pos = 0;
43 while (pos < node->input_size()) {
44 if (node->input(pos) == input) {
45 node->mutable_input()->SwapElements(pos, node->input_size() - 1);
46 node->mutable_input()->RemoveLast();
47 node_map->RemoveOutput(NodeName(input), node->name());
48 removed_input = true;
49 } else {
50 ++pos;
51 }
52 }
53 return removed_input;
54}

56void DeleteNodes(const std::set<int>& nodes_to_delete, GraphDef* graph) {
57 int last = graph->node_size() - 1;
58 for (auto it = nodes_to_delete.rbegin(); it != nodes_to_delete.rend(); ++it) {
59 const int index = *it;
60 graph->mutable_node()->SwapElements(index, last);
61 last--;
62 }
63 graph->mutable_node()->DeleteSubrange(last + 1, nodes_to_delete.size());
64}

68bool DependencyOptimizer::SafeToRemoveIdentity(const NodeDef& node) {
69 if (!IsIdentity(node)) {
70 return true;
71 }
72
73 if (nodes_to_preserve_.find(node.name()) != nodes_to_preserve_.end()) {
74 return false;
75 }
76 if (!fetch_nodes_known_) {
77 // The output values of this node may be needed.
78 return false;
79 }
80 const NodeDef* input = node_map_->GetNode(NodeName(node.input(0)));
81 CHECK(input != nullptr) << "node = " << node.name()
82 << " input = " << node.input(0);
83 // Don't remove Identity nodes corresponding to Variable reads or following
84 // Recv.
85 if (IsVariable(*input) || IsRecv(*input)) {
86 return false;
87 } else if (IsSwitch(*input)) {
88 // Don't turn Identity nodes following Switch into NoOp or remove them
89 // if it requires anchoring a control dependencies the Switch node, which
90 // is not valid.
91 if (str_util::StartsWith(node.name(), kConstantFoldingCtrl)) {
92 // TODO(rmlarsen): Try to remove this artificial contraint.
93 return false;
94 }
95 }
96 for (auto consumer : node_map_->GetOutputs(node.name())) {
97 if (node.input_size() > 1 && IsMerge(*consumer)) {
98 return false;
99 }
100 if (IsSwitch(*input)) {
101 for (const string& consumer_input : consumer->input()) {
102 if (consumer_input == AsControlDependency(node.name())) {
103 return false;
104 }
105 }
106 }
107 }
108 return true;
109}

111bool DependencyOptimizer::SafeToConvertToNoOp(const NodeDef& node) {
112 if (nodes_to_preserve_.find(node.name()) != nodes_to_preserve_.end()) {
113 return false;
114 }
115 if (!fetch_nodes_known_ || NumNonControlOutputs(node, *node_map_) > 0) {
116 // The output values of this node may be needed.
117 return false;
118 }
119 if (IsMerge(node) || IsSwitch(node)) {
120 return false;
121 }
122 if (ModifiesFrameInfo(node)) {
123 return false;
124 }
125 if (!IsFreeOfSideEffect(node)) {
126 return false;
127 }
128 if (node.op() == "ControlTrigger") {
129 return false;
130 }
131 if (node.op().rfind("Submodel", 0) == 0) {
132 return false;
133 }
134 const OpDef* op_def = nullptr;
135 Status status = OpRegistry::Global()->LookUpOpDef(node.op(), &op_def);
136 if (!status.ok() || op_def->output_arg_size() == 0) {
137 return false;
138 }
139
140 if (!SafeToRemoveIdentity(node)) {
141 return false;
142 }
143
144 const std::unordered_set<string> do_not_rewrite_ops{
145 "Assert", "CheckNumerics", "_Retval",
146 "_Arg", "_ParallelConcatUpdate", "_TPUExecute",
147 "_TPUCompile"};
148 return do_not_rewrite_ops.find(node.op()) == do_not_rewrite_ops.end();
149}

151void DependencyOptimizer::OptimizeNode(int node_idx,
152 SetVector<int>* nodes_to_simplify,
153 std::set<int>* nodes_to_delete) {
154 NodeDef* node = optimized_graph_->mutable_node(node_idx);
155 const bool is_noop = IsNoOp(*node);
156 const bool is_identity = IsIdentity(*node);
157 const string node_name = node->name();
158 // Constant nodes with no input control dependency are always executed early,
159 // so we can prune all their output control dependencies.
160 if (IsConstant(*node) && node->input_size() == 0) {
161 const std::set<NodeDef*> output_nodes = node_map_->GetOutputs(node_name);
162 for (NodeDef* fanout : output_nodes) {
163 bool optimize_fanout = false;
164 bool data_connection = false;
165 for (int i = fanout->input_size() - 1; i >= 0; --i) {
166 int pos;
167 string input_name = ParseNodeName(fanout->input(i), &pos);
168 if (input_name == node_name) {
169 if (pos < 0) {
170 fanout->mutable_input()->SwapElements(i, fanout->input_size() - 1);
171 fanout->mutable_input()->RemoveLast();
172 optimize_fanout = true;
173 } else {
174 data_connection = true;
175 }
176 }
177 }
178 if (optimize_fanout) {
179 nodes_to_simplify->PushBack(node_to_idx_[fanout]);
180 if (!data_connection) {
181 node_map_->RemoveOutput(node_name, fanout->name());
182 }
183 }
184 }
185 if (node_map_->GetOutputs(node_name).empty() && fetch_nodes_known_ &&
186 nodes_to_preserve_.find(node_name) == nodes_to_preserve_.end()) {
187 // Mark the node for deletion.
188 nodes_to_delete->insert(node_to_idx_[node]);
189 }
190 return;
191 }
192
193 // Change ops that only have control dependencies as outputs to NoOps.
194 if (!is_noop && SafeToConvertToNoOp(*node)) {
195 VLOG(1) << "***** Replacing " << node_name << " (" << node->op()
196 << ") with NoOp.";
197 // The outputs of this node are not consumed. Replace its inputs with
198 // control dependencies and replace the op itself with the NoOp op.
199 std::unordered_set<string> ctrl_inputs;
200 int pos = 0;
201 while (pos < node->input_size()) {
202 const string old_input = node->input(pos);
203 if (IsControlInput(old_input)) {
204 if (!ctrl_inputs.insert(old_input).second) {
205 // We found a duplicate control input. Remove it.
206 node->mutable_input()->SwapElements(pos, node->input_size() - 1);
207 node->mutable_input()->RemoveLast();
208 } else {
209 ++pos;
210 }
211 continue;
212 }
213 const string ctrl_input = ConstantFolding::AddControlDependency(
214 old_input, optimized_graph_, node_map_.get());
215 if (ctrl_inputs.insert(ctrl_input).second) {
216 node->set_input(pos, ctrl_input);
217 node_map_->UpdateInput(node_name, old_input, ctrl_input);
218 const NodeDef* old_input_node = node_map_->GetNode(old_input);
219 nodes_to_simplify->PushBack(node_to_idx_[old_input_node]);
220 }
221 ++pos;
222 }
223 node->set_op("NoOp");
224 node->clear_attr();
225 nodes_to_simplify->PushBack(node_to_idx_[node]);
226 return;
227 }
228
229 // Remove NoOp nodes if the product of their fan-in and fan-out is less than
230 // or equal to the sum of the fan-in and fan-out. The non-trivial rewrites
231 // take the following form:
232 //
233 // Case a)
234 // x --^> +------+ x --^> +---+
235 // y --^> | NoOp | --^> a ==> y --^> | a |
236 // ... | | ... | |
237 // z --^> +------+ z --^> +---+
238 //
239 // Case b)
240 // +------+ --^> a +---+ --^> a
241 // x --^> | NoOp | --^> b ==> | x | --^> b
242 // | | ... | | ...
243 // +------+ --^> c +---+ --^> c
244 // Case c)
245 // +------+ x ---^> a
246 // x --^> | NoOp | --^> a ==> \/
247 // y --^> | | --^> b /\
248 // +------+ y ---^> b
249 //
250 // We only apply this optimization if we don't increase the number of control
251 // edges across device boundaries, e.g. in cases a) and b) if NoOp and
252 // a and x, respectively, are on the same device. Control edges across device
253 // boundaries require inter-device communication (Send/Recv pairs to be
254 // inserted in the graph), which is very costly.
255 //
256 // We also remove identity nodes, subject to the same constraints on number of
257 // resulting control edges and device boundary crossings:
258 //
259 // Case a)
260 // +----------+ ---> a +---+ ---> a
261 // x --> | Identity | --^> b ==> | x | --^> b
262 // | | ... | | ...
263 // +----------+ --^> c +---+ --^> c
264 //
265 // Case b)
266 // x ---> +----------+ ---> a x ---> +---+
267 // y --^> | Identity | ==> y --^> | a |
268 // ... | | ... | |
269 // z --^> +----------+ z --^> +---+
270 //
271 // Case c)
272 // +----------+ x ---> +---+
273 // x ---> | Identity | ---> a ==> \--^> | a |
274 // y --^> | | --^> b /\ +---+
275 // +----------+ y --^> b
276
277 if (is_noop || is_identity) {
278 if (is_identity && !SafeToRemoveIdentity(*node)) {
279 return;
280 }
281
282 const auto& output_node_set = node_map_->GetOutputs(node_name);
283 const std::vector<NodeDef*> output_nodes(output_node_set.begin(),
284 output_node_set.end());
285 const int num_outputs = output_nodes.size();
286 const int num_inputs = node->input_size();
287
288 // Don't increase the number of edges in the graph.
289 if (num_inputs * num_outputs > num_inputs + num_outputs) {
290 return;
291 }
292 std::vector<NodeDef*> input_nodes;
293 for (int i = 0; i < num_inputs; ++i) {
294 NodeDef* input_node = node_map_->GetNode(node->input(i));
295 if (input_node == nullptr) {
296 LOG(ERROR) << "Invalid input " << node->input(i);
297 return;
298 }
299 input_nodes.push_back(input_node);
300 }
301
302 // Make sure that we don't increase the number of edges that cross
303 // device boundaries.
304 if ((num_inputs == 1 && num_outputs > 1 &&
305 input_nodes[0]->device() != node->device()) ||
306 (num_inputs > 1 && num_outputs == 1 &&
307 output_nodes[0]->device() != node->device())) {
308 return;
309 }
310
311 // TODO(rmlarsen): Not all device crossings are equally expensive.
312 // Assign a cost to each based on device affinity and compute a
313 // cost before and after.
314 const string& node_dev = node->device();
315 int num_cross_in = 0;
316 for (NodeDef* input_node : input_nodes) {
317 num_cross_in += static_cast<int>(input_node->device() != node_dev);
318 }
319 int num_cross_out = 0;
320 for (NodeDef* output_node : output_nodes) {
321 num_cross_out += static_cast<int>(output_node->device() != node_dev);
322 }
323 if (is_identity && num_cross_in > 0 && num_cross_out > 0) {
324 // This identity node follows a device crossing, so it might be
325 // following a _Recv node after partioning. Do not remove such nodes,
326 // unless they only have consumers on the same device as themselves.
327 return;
328 }
329
330 // Make sure we do not increase the number of device crossings.
331 const int num_cross_before = num_cross_in + num_cross_out;
332 int num_cross_after = 0;
333 for (NodeDef* input_node : input_nodes) {
334 for (NodeDef* output_node : output_nodes) {
335 num_cross_after +=
336 static_cast<int>(input_node->device() != output_node->device());
337 }
338 }
339 if (num_cross_after > num_cross_before) {
340 return;
341 }
342
343 VLOG(1) << "***** Rerouting input around\n" << node->DebugString();
344 // Now remove the node and re-wire its inputs to its outputs.
345 for (auto consumer : output_nodes) {
346 bool updated_consumer = false;
347 VLOG(1) << "consumer before:\n" << consumer->DebugString();
348 for (int i = 0; i < num_inputs; ++i) {
349 const NodeDef* input = input_nodes[i];
350 // Forward dependency from input to consumer if it doesn't already
351 // depend on it.
352 if (is_identity && i == 0) {
353 // Replace regular input from Identity node.
354 bool found_input = false;
355 string new_input;
356 const string& input_to_forward = node->input(0);
357 CHECK(!IsControlInput(input_to_forward));
358 for (int j = 0; j < consumer->input_size(); ++j) {
359 const string& old_input = consumer->input(j);
360 int old_input_pos;
361 string old_input_node_name =
362 ParseNodeName(old_input, &old_input_pos);
363 if (old_input_node_name == node_name) {
364 if (old_input_pos >= 0) {
365 // Regular input
366 new_input = input_to_forward;
367 node_map_->UpdateInput(consumer->name(), old_input, new_input);
368 consumer->set_input(j, new_input);
369 found_input = true;
370 } else {
371 // Control dependency
372 new_input = AsControlDependency(NodeName(input_to_forward));
373 node_map_->UpdateInput(consumer->name(), old_input, new_input);
374 consumer->set_input(j, new_input);
375 found_input = true;
376 }
377 }
378 }
379 CHECK(found_input);
380 updated_consumer = true;
381 } else {
382 // Forward dependency from input to consumer if it doesn't already
383 // depend on it.
384 if (node_map_->GetOutputs(input->name()).count(consumer) == 0) {
385 consumer->add_input(AsControlDependency(input->name()));
386 node_map_->AddOutput(input->name(), consumer->name());
387 nodes_to_simplify->PushBack(node_to_idx_[input]);
388 updated_consumer = true;
389 }
390 }
391 }
392 // Remove dependency on node from consumer.
393 updated_consumer |= RemoveInput(consumer, AsControlDependency(node_name),
394 node_map_.get());
395 if (updated_consumer) {
396 nodes_to_simplify->PushBack(node_to_idx_[consumer]);
397 }
398 VLOG(1) << "consumer after:\n" << consumer->DebugString();
399 }
400 node_map_->RemoveOutputs(node_name);
401 if (fetch_nodes_known_ &&
402 nodes_to_preserve_.find(node_name) == nodes_to_preserve_.end()) {
403 // Mark the node for deletion.
404 nodes_to_delete->insert(node_idx);
405
406 // Disconnect the node from its inputs to enable further optimizations.
407 node_map_->RemoveInputs(node_name);
408 node->clear_input();
409 }
410 }

419Status DependencyOptimizer::OptimizeDependencies() {
420 SetVector<int> nodes_to_simplify;
421 std::set<int> nodes_to_delete;
422 for (int i = 0; i < optimized_graph_->node_size(); ++i) {
423 const NodeDef& node = optimized_graph_->node(i);
424 if (IsNoOp(node) || IsIdentity(node) || IsConstant(node) ||
425 SafeToConvertToNoOp(node)) {
426 nodes_to_simplify.PushBack(i);
427 }
428 }
429 while (!nodes_to_simplify.Empty()) {
430 int node_to_simplify = nodes_to_simplify.PopBack();
431 // Discard nodes that were marked for deletion already.
432 while (nodes_to_delete.find(node_to_simplify) != nodes_to_delete.end()) {
433 node_to_simplify = nodes_to_simplify.PopBack();
434 }
435 OptimizeNode(node_to_simplify, &nodes_to_simplify, &nodes_to_delete);
436 }
437
438 if (fetch_nodes_known_) {
439 VLOG(1) << "Deleted " << nodes_to_delete.size() << " out of "
440 << optimized_graph_->node_size() << " nodes.";
441 DeleteNodes(nodes_to_delete, optimized_graph_);
442 node_map_.reset(new NodeMap(optimized_graph_));
443 BuildNodeToIdx();
444 }
445 return Status::OK();
446}

448Status DependencyOptimizer::TransitiveReduction() {
449 // PRECONDITION: optimized_graph_ must be sorted topologically.
450 const int num_nodes = optimized_graph_->node_size();
451 // Set up a compressed version of the graph to save a constant factor in the
452 // expensive algorithm below. Also cache the set of control outputs and the
453 // highest index of a target of any control output from each node.
454 int num_controls = 0;
455 std::vector<gtl::InlinedVector<int, 4>> inputs(num_nodes);
456 std::vector<gtl::InlinedVector<std::pair<int, int>, 2>> control_outputs(
457 num_nodes);
458 for (int node_idx = 0; node_idx < num_nodes; ++node_idx) {
459 const NodeDef& node = optimized_graph_->node(node_idx);
460 if (ModifiesFrameInfo(node) || !HasOpDef(node)) {
461 // Ignore function nodes and nodes that modify frame info.
462 continue;
463 }
464 for (int input_slot = 0; input_slot < node.input_size(); ++input_slot) {
465 const string& input = node.input(input_slot);
466 const NodeDef* input_node = node_map_->GetNode(input);
467 if (ModifiesFrameInfo(*input_node) || IsMerge(*input_node)) {
468 // Ignore edges from nodes that modify frame info and from Merge nodes,
469 // because we cannot know which of it's input paths executes.
470 continue;
471 }
472 const int input_node_idx = node_to_idx_[input_node];
473 inputs[node_idx].push_back(input_node_idx);
474 if (IsControlInput(input)) {
475 ++num_controls;
476 control_outputs[input_node_idx].emplace_back(node_idx, input_slot);
477 }
478 }
479 }
480
481 // Run the longest path in DAG algorithm for each source node that has control
482 // outputs. If, for any target node of a control output, there exists a path
483 // of length > 1, we can drop that control dependency.
484 int num_controls_removed = 0;
485 std::vector<int> longest_distance(num_nodes);
486 // Map from target_index -> set of (input_slot, source_index), representing
487 // the control edges to remove. We sort them in reverse order by input slot,
488 // such that when we swap them out so we don't clobber the
489 // node(target).input() repeated field.
490 typedef std::pair<int, int> InputSlotAndSource;
491 std::unordered_map<
492 int, std::set<InputSlotAndSource, std::greater<InputSlotAndSource>>>
493 control_edges_to_remove;
494 for (int source = 0; source < num_nodes; ++source) {
495 int highest_control_target = -1;
496 for (const auto& control_output : control_outputs[source]) {
497 if (control_output.first > highest_control_target) {
498 highest_control_target = control_output.first;
499 }
500 }
501 if (highest_control_target <= source) {
502 continue;
503 }
504 std::fill(longest_distance.begin() + source,
505 longest_distance.begin() + highest_control_target + 1, 0);
506 for (int target = source + 1; target <= highest_control_target; ++target) {
507 for (int input : inputs[target]) {
508 // If the input node is before source in the topo order, no path
509 // source -> input -> target can exits and we can skip it.
510 // Also only extend a path from the source itself or from nodes that
511 // have a path from source, indicated by longest_distance[input] > 0.
512 if (input == source ||
513 (input > source && longest_distance[input] > 0)) {
514 // If source -> input -> target is longer than the longest
515 // path so far from source -> target, update the longest_distance.
516 int candidate_longest_distance = longest_distance[input] + 1;
517 if (candidate_longest_distance > longest_distance[target]) {
518 longest_distance[target] = candidate_longest_distance;
519 }
520 }
521 }
522 }
523
524 // If the longest path from source to target of a control dependency is
525 // longer than 1, there exists an alternate path, and we can eliminate the
526 // redundant direct control dependency.
527 for (const auto& control_output : control_outputs[source]) {
528 const int target = control_output.first;
529 if (longest_distance[target] > 1) {
530 const int input_slot = control_output.second;
531 control_edges_to_remove[target].emplace(input_slot, source);
532 }
533 }
534 }
535
536 for (const auto& it : control_edges_to_remove) {
537 const int target = it.first;
538 NodeDef* target_node = optimized_graph_->mutable_node(target);
539 for (const InputSlotAndSource& slot_and_source : it.second) {
540 const int input_slot = slot_and_source.first;
541 const int source = slot_and_source.second;
542 const NodeDef& source_node = optimized_graph_->node(source);
543 CHECK_LT(input_slot, target_node->input_size());
544 target_node->mutable_input()->SwapElements(input_slot,
545 target_node->input_size() - 1);
546 node_map_->RemoveOutput(source_node.name(), target_node->name());
547 target_node->mutable_input()->RemoveLast();
548 ++num_controls_removed;
549 }
550 }
551 VLOG(1) << "Removed " << num_controls_removed << " out of " << num_controls
552 << " control dependencies";
553 return Status::OK();
554}

556void DependencyOptimizer::BuildNodeToIdx() {
557 // Set up &node -> index map.
558 node_to_idx_.clear();
559 for (int i = 0; i < optimized_graph_->node_size(); ++i) {
560 const NodeDef& node = optimized_graph_->node(i);
561 node_to_idx_[&node] = i;
562 }
563}

565Status DependencyOptimizer::Optimize(Cluster* cluster, const GrapplerItem& item,
566 GraphDef* optimized_graph) {
567 optimized_graph_ = optimized_graph;
568 *optimized_graph_ = item.graph;
569 nodes_to_preserve_ = item.NodesToPreserve();
570 fetch_nodes_known_ = !item.fetch.empty();
571 CleanControlInputs();
572
573 const int num_iterations = 2;
574 for (int iteration = 0; iteration < num_iterations; ++iteration) {
575 Status topo_sort_status;
576 // Perform topological sort to prepare the graph for transitive reduction.
577 topo_sort_status = TopologicalSort(optimized_graph_);
578 // Set up index-based graph datastructures to speed up analysis steps below.
579 node_map_.reset(new NodeMap(optimized_graph_));
580 BuildNodeToIdx();
581
582 if (topo_sort_status.ok()) {
583 // Remove redundant control dependencies.
584 TF_RETURN_IF_ERROR(TransitiveReduction());
585 } else {
586 LOG(ERROR) << "Iteration = " << iteration
587 << ", topological sort failed with message: "
588 << topo_sort_status.error_message();
589 }
590 // Turn nodes with only control outputs into NoOps, prune NoOp and Identity
591 // nodes.
592 TF_RETURN_IF_ERROR(OptimizeDependencies());
593
594 // Dedup control inputs.
595 CleanControlInputs();
596 }
597
598 return Status::OK();
599}
