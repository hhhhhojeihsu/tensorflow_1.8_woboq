37class FunctionInliningContext {
38 public:
39 explicit FunctionInliningContext(const GrapplerItem& item,
40 RewriterConfig::Toggle opt_level)
41 : library_(&item.graph.library()),
42 opt_level_(opt_level),
43 functions_(InliningCandidates(item)) {}
44
45 const FunctionDefLibrary& Library() const { return *library_; }
46
47 bool HasInlinedFunctions() const { return !functions_.empty(); }
48
49 // Find inlining candidate by name. Return nullptr if not found.
50 const FunctionDef* FindInlinedFunction(const string& name) const {
51 auto it = functions_.find(name);
52 if (it != functions_.end()) {
53 return it->second;
54 } else {
55 return nullptr;
56 }
57 }
58
59 private:
60 std::unordered_map<string, const FunctionDef*> InliningCandidates(
61 const GrapplerItem& item) const {
62 std::unordered_map<string, const FunctionDef*> functions;
63 for (const FunctionDef& func : item.graph.library().function()) {
64 // Don't inline functions marked as noinline
65 if (func.attr().count("_noinline") != 0 &&
66 func.attr().at("_noinline").b() &&
67 opt_level_ != RewriterConfig::AGGRESSIVE) {
68 continue;
69 }
70 // Can't create IdentityN nodes with no input or output: skip these
71 // functions for now.
72 if (func.signature().input_arg_size() == 0 ||
73 func.signature().output_arg_size() == 0) {
74 continue;
75 }
76 functions[func.signature().name()] = &func;
77 }
78 return functions;
79 }
80
81 const FunctionDefLibrary* library_;
82 RewriterConfig::Toggle opt_level_;
83 std::unordered_map<string, const FunctionDef*> functions_;
84
85 TF_DISALLOW_COPY_AND_ASSIGN(FunctionInliningContext);
86};

88// Copy input/output argument type to the type_list. Return error if argument
89// type is not explicitly defined, and not specified in function attributes.
90Status CopyArgType(const NodeDef& func_node,
91 const std::unordered_map<string, AttrValue>& func_attr,
92 const string& arg_kind, const OpDef::ArgDef& arg,
93 AttrValue::ListValue* type_list) {
94 if (arg.type() != DT_INVALID) {
95 type_list->add_type(arg.type());
96 } else {
97 auto it = func_attr.find(arg.type_attr());
98 if (it == func_attr.end() || it->second.type() == DT_INVALID) {
99 return errors::InvalidArgument(
100 "Invalid ", arg_kind, " argument ", arg.name(), " for function ",
101 func_node.op(), " instantiated by ", func_node.name());
102 }
103 type_list->add_type(it->second.type());
104 }
105 return Status::OK();
106}

108// Add an IdentityN op to hook the function inputs to: this ensures that
109// they're all evaluated before the evaluation of the function body starts.
110Status HookInlinedFunctionInputs(
111 const NodeDef& func_node, const FunctionDef& func,
112 const std::unordered_map<string, AttrValue>& func_attr, NodeDef* inputs) {
113 inputs->set_name(strings::StrCat(func_node.name(), "/", "inlined_inputs"));
114 inputs->set_op("IdentityN");
115 inputs->set_device(func_node.device());
116 *inputs->mutable_input() = func_node.input();
117 AttrValue::ListValue* type_list =
118 (*inputs->mutable_attr())["T"].mutable_list();
119 for (const OpDef::ArgDef& arg : func.signature().input_arg()) {
120 TF_RETURN_IF_ERROR(
121 CopyArgType(func_node, func_attr, "input", arg, type_list));
122 }
123 return Status::OK();
124}
125
126// Add an IdentityN op to hook the function outputs to: this ensures that the
127// function body is fully evaluated before its fanout gets scheduled.
128Status HookInlinedFunctionOutputs(
129 const NodeDef& func_node, const FunctionDef& func,
130 const std::unordered_map<string, AttrValue>& func_attr,
131 const gtl::ArraySlice<string> fetch, NodeDef* outputs) {
132 outputs->set_name(func_node.name());
133 outputs->set_op("IdentityN");
134 outputs->set_device(func_node.device());
135 AttrValue::ListValue* type_list =
136 (*outputs->mutable_attr())["T"].mutable_list();
137 for (int i = 0; i < func.signature().output_arg_size(); ++i) {
138 const OpDef::ArgDef& arg = func.signature().output_arg(i);
139 TF_RETURN_IF_ERROR(
140 CopyArgType(func_node, func_attr, "output", arg, type_list));
141 // Use the fetch names since they take into account the output mapping.
142 outputs->add_input(strings::StrCat(func_node.name(), "/", fetch[i]));
143 }
144 return Status::OK();
145}

147Status InlineFunction(const NodeDef& func_node, const FunctionDef& func,
148 const FunctionInliningContext& ctx,
149 GraphDef* optimized_graph) {
150 const std::unordered_map<string, AttrValue> func_attr(
151 func_node.attr().begin(), func_node.attr().end());
152
153 std::unique_ptr<GrapplerItem> item =
154 GrapplerItemFromFunctionDef(func, func_attr, ctx.Library());
155 if (!item) {
156 return errors::InvalidArgument("Failed to inline function ", func_node.op(),
157 " instantiated by ", func_node.name());
158 }
159
160 std::unordered_map<string, int> input_nodes;
161 for (int i = 0; i < func.signature().input_arg_size(); ++i) {
162 const OpDef::ArgDef& arg = func.signature().input_arg(i);
163 input_nodes[arg.name()] = i;
164 }
165
166 // Hook inlined function inputs to IdentityN node
167 NodeDef* func_inputs = optimized_graph->add_node();
168 TF_RETURN_IF_ERROR(
169 HookInlinedFunctionInputs(func_node, func, func_attr, func_inputs));
170
171 for (NodeDef& func_body_node : *item->graph.mutable_node()) {
172 if (input_nodes.find(func_body_node.name()) != input_nodes.end()) {
173 CHECK_EQ(0, func_body_node.input_size());
174 // Turn input placeholders into identity nodes
175 if (IsPlaceholder(func_body_node)) {
176 func_body_node.set_op("Identity");
177 }
178 int input_id = input_nodes[func_body_node.name()];
179 func_body_node.add_input(
180 strings::StrCat(func_inputs->name(), ":", input_id));
181 } else {
182 // Update the input names if any.
183 for (string& input : *func_body_node.mutable_input()) {
184 input = AddPrefixToNodeName(input, /*prefix=*/func_node.name());
185 }
186 // If the node has no input, make hook it up to the func_inputs node to
187 // ensure it runs in the same frame as the other nodes of the function
188 // body.
189 if (func_body_node.input_size() == 0) {
190 *func_body_node.add_input() = AsControlDependency(func_inputs->name());
191 }
192 }
193
194 // Add the node name as a prefix to avoid collisions after inlining
195 func_body_node.set_name(
196 strings::StrCat(func_node.name(), "/", func_body_node.name()));
197
198 // Make sure the node is placed
199 func_body_node.set_device(func_node.device());
200
201 // Check if a body node is itself a function
202 const FunctionDef* func_body_node_func =
203 ctx.FindInlinedFunction(func_body_node.op());
204 if (func_body_node_func != nullptr) {
205 // Recursively inline function calls
206 TF_RETURN_IF_ERROR(InlineFunction(func_body_node, *func_body_node_func,
207 ctx, optimized_graph));
208 } else {
209 // Annotate the node with the function attributes.
210 for (const auto& attr : func.attr()) {
211 func_body_node.mutable_attr()->insert(attr);
212 }
213 // Move the node to the main graph
214 optimized_graph->add_node()->Swap(&func_body_node);
215 }
216 }
217
218 // Hook inlined function outputs to IdentityN node
219 NodeDef* func_outputs = optimized_graph->add_node();
220 TF_RETURN_IF_ERROR(HookInlinedFunctionOutputs(func_node, func, func_attr,
221 item->fetch, func_outputs));
222
223 return Status::OK();
224}

232class SymbolicGradientEnv {
233 public:
234 SymbolicGradientEnv(int graph_version, const FunctionDefLibrary& library)
235 : graph_version_(graph_version), library_(library) {}
236
237 FunctionLibraryDefinition* function_library() {
238 InitializeIfNeeded();
239 return fld_.get();
240 }
241 FunctionLibraryRuntime* function_library_runtime() {
242 InitializeIfNeeded();
243 return flr_;
244 }
245
246 private:
247 // This initialization is expensive. Do it lazily to avoid paying for it
248 // unless it's needed.
249 void InitializeIfNeeded() {
250 if (flr_) {
251 return;
252 }
253 Env* env = Env::Default();
254 DeviceAttributes attr;
255 attr.set_name("/device:CPU:0");
256 attr.set_device_type("CPU");
257 FakeCPUDevice* dev = new FakeCPUDevice(env, attr);
258 std::vector<Device*> devices;
259 devices.push_back(dev);
260 dvc_mgr_.reset(new DeviceMgr(devices));
261 fld_.reset(new FunctionLibraryDefinition(OpRegistry::Global(), library_));
262 OptimizerOptions optimizer_opts;
263 optimizer_opts.set_do_function_inlining(true);
264 pflr_.reset(new ProcessFunctionLibraryRuntime(
265 dvc_mgr_.get(), env, graph_version_, fld_.get(), optimizer_opts));
266 flr_ = pflr_->GetFLR(dev->name());
267 }
268
269 const int graph_version_;
270 const FunctionDefLibrary& library_;
271 std::unique_ptr<DeviceMgr> dvc_mgr_;
272 std::unique_ptr<FunctionLibraryDefinition> fld_;
273 std::unique_ptr<ProcessFunctionLibraryRuntime> pflr_;
274 FunctionLibraryRuntime* flr_ = nullptr;
275};

277Status InlineSymbolicGradient(const NodeDef& node, SymbolicGradientEnv* env,
278 GraphDef* inlined_graph) {
279 GraphDef graph_def;
280
281 // Create a node to anchor the gradient inputs
282 NodeDef* inlined_input = graph_def.add_node();
283 inlined_input->set_name("FunctionInputs");
284 inlined_input->set_op("IdentityN");
285 AttrValue::ListValue* type_list =
286 (*inlined_input->mutable_attr())["T"].mutable_list();
287 for (const auto& type : node.attr().at("Tin").list().type()) {
288 type_list->add_type(static_cast<DataType>(type));
289 }
290
291 // Add the gradient node
292 NodeDef* inlined = graph_def.add_node();
293 *inlined = node;
294 inlined->clear_input();
295 for (int i = 0; i < node.attr().at("Tin").list().type_size(); ++i) {
296 inlined->add_input(strings::StrCat(inlined_input->name(), ":", i));
297 }
298
299 // Create a node to anchor the gradient outputs
300 NodeDef* inlined_output = graph_def.add_node();
301 inlined_output->set_name("FunctionOutputs");
302 inlined_output->set_op("IdentityN");
303 type_list = (*inlined_output->mutable_attr())["T"].mutable_list();
304 for (const auto& type : node.attr().at("Tout").list().type()) {
305 type_list->add_type(static_cast<DataType>(type));
306 }
307 for (int i = 0; i < node.attr().at("Tout").list().type_size(); ++i) {
308 inlined_output->add_input(strings::StrCat(inlined->name(), ":", i));
309 }
310
311 // Convert the graphdef to a graph
312 GraphConstructorOptions graph_ctor_opts;
313 graph_ctor_opts.allow_internal_ops = true;
314 graph_ctor_opts.expect_device_spec = false;
315 Graph graph(env->function_library());
316 TF_RETURN_IF_ERROR(
317 ConvertGraphDefToGraph(graph_ctor_opts, graph_def, &graph));
318
319 // Recursively inline the functions until there is nothing more to inline. We
320 // should at least expand one function.
321 int counter = 0;
322 while (counter < 50 &&
323 ExpandInlineFunctions(env->function_library_runtime(), &graph)) {
324 ++counter;
325 }
326
327 GraphDef inlined_graph_def;
328 graph.ToGraphDef(&inlined_graph_def);
329
330 // Add the default values of attributes to the nodes that have been inlined.
331 TF_RETURN_IF_ERROR(AddDefaultAttrsToGraphDef(&inlined_graph_def,
332 *graph.op_registry(), 0, true));
333
334 // Add the inlined nodes to the graph
335 for (NodeDef& inlined_node : *inlined_graph_def.mutable_node()) {
336 if (inlined_node.name() == "FunctionOutputs") {
337 inlined_node.set_name(node.name());
338 for (int i = 0; i < inlined_node.input_size(); ++i) {
339 inlined_node.set_input(
340 i, AddPrefixToNodeName(inlined_node.input(i), node.name()));
341 }
342 } else if (inlined_node.name() == "FunctionInputs") {
343 inlined_node.set_name(
344 AddPrefixToNodeName(inlined_node.name(), node.name()));
345 inlined_node.clear_input();
346 for (int i = 0; i < node.input_size(); ++i) {
347 inlined_node.add_input(node.input(i));
348 }
349 } else {
350 inlined_node.set_name(
351 AddPrefixToNodeName(inlined_node.name(), node.name()));
352 for (int i = 0; i < inlined_node.input_size(); ++i) {
353 inlined_node.set_input(
354 i, AddPrefixToNodeName(inlined_node.input(i), node.name()));
355 }
356 // If the node has no input, hook it up to the function input node to make
357 // sure it runs in the same frame as the other nodes of the function body.
358 if (inlined_node.input_size() == 0) {
359 *inlined_node.add_input() = AsControlDependency(
360 AddPrefixToNodeName("FunctionInputs", node.name()));
361 }
362 }
363 inlined_node.set_device(node.device());
364 inlined_graph->add_node()->Swap(&inlined_node);
365 }
366
367 return Status::OK();
368}

372Status FunctionOptimizer::Optimize(Cluster* cluster, const GrapplerItem& item,
373 GraphDef* optimized_graph) {
374 FunctionInliningContext function_inlining_ctx(item, opt_level_);
375
376 // Nothing to do here.
377 if (!function_inlining_ctx.HasInlinedFunctions()) {
378 *optimized_graph = item.graph;
379 return Status::OK();
380 }
381
382 SymbolicGradientEnv env(item.graph.versions().producer(),
383 item.graph.library());
384
385 for (const NodeDef& node : item.graph.node()) {
386 if (node.op() == "SymbolicGradient") {
387 TF_RETURN_IF_ERROR(InlineSymbolicGradient(node, &env, optimized_graph));
388 continue;
389 }
390
391 const FunctionDef* func =
392 function_inlining_ctx.FindInlinedFunction(node.op());
393 if (func != nullptr) {
394 TF_RETURN_IF_ERROR(
395 InlineFunction(node, *func, function_inlining_ctx, optimized_graph));
396 } else {
397 *optimized_graph->add_node() = node;
398 }
399 }
400
401 // TODO(bsteiner): specialize the implementation of functions that can't be
402 // inlined based on the context in which they're instantiated.
403
404 // TODO(bsteiner): trim the library to remove unused function definitions
405 *optimized_graph->mutable_versions() = item.graph.versions();
406 *optimized_graph->mutable_library() = item.graph.library();
407
408 return Status::OK();
409}
