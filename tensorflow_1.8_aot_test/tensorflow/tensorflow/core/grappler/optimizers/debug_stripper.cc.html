29Status DebugStripper::Optimize(Cluster* cluster, const GrapplerItem& item,
30 GraphDef* output) {
31 *output = item.graph;
32 for (NodeDef& node : *output->mutable_node()) {
33 if (IsAssert(node)) {
34 // Convert this node into a no-op.
35 node.set_op("NoOp");
36 node.clear_attr();
37 // Convert all its inputs into control dependency, which will then
38 // be optimized away by dependency optimizer.
39 for (string& inp : *node.mutable_input()) {
40 if (!IsControlInput(inp)) {
41 inp = AsControlDependency(inp);
42 }
43 }
44 } else if (IsCheckNumerics(node) || IsPrint(node)) {
45 // Replace with Identity op which will be pruned later.
46 node.set_op("Identity");
47 // Only preserve T attribute.
48 protobuf::Map<string, AttrValue> new_attr;
49 if (node.attr().find("T") != node.attr().end()) {
50 new_attr.insert({"T", node.attr().at("T")});
51 }
52 node.mutable_attr()->swap(new_attr);
53 // As Identity op only takes one input, mark redundant inputs as control
54 // input.
55 for (size_t i = 1; i < node.input_size(); ++i) {
56 if (!IsControlInput(node.input(i))) {
57 *node.mutable_input(i) = AsControlDependency(node.input(i));
58 }
59 }
60 }
61 }
62 return Status::OK();
63}

67 // Takes no feedback.
