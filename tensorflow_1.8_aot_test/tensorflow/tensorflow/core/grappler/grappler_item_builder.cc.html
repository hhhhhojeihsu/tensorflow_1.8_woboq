53void InitializeTensor(DataType type, Tensor* tensor) {
54 const int period = 7;
55 if (type == DT_FLOAT) {
56 auto flat = tensor->flat<float>();
57 // Populate numbers 0, 0.1, 0.2, ..., 0.5, 0.6, 0, 0.1, 0.2, ...
58 for (int i = 0; i < flat.size(); i++) {
59 flat(i) = static_cast<float>(i % period) / 10.0f;
60 }
61 } else if (type == DT_INT64) {
62 auto flat = tensor->flat<int64>();
63 // Populate numbers 0, 1, 2, ..., 5, 6, 0, 1, 2, ...
64 for (int i = 0; i < flat.size(); i++) {
65 flat(i) = i % period;
66 }
67 } else {
68 memset(const_cast<char*>(tensor->tensor_data().data()), 0,
69 tensor->tensor_data().size());
70 }
71}

73// Optimize the graph def (including function inlining and other optimizations).
74// This is a temporary change that optimizes the graph in context of a single
75// gpu machine. Down the line, we may want to make grappler_item_builder aware
76// of the cluster type (E.g: single cpu, multiple gpu, etc) being simulated in
77// order to get the correct session options and environment, and performing the
78// correct optimizations.
79Status OptimizeGraph(const GraphDef& graph_def_arg, GraphDef* output_graph_def,
80 const ItemConfig& cfg) {
81 if (!cfg.apply_optimizations && !cfg.erase_noinline_attributes) {
82 return Status::OK();
83 }
84
85 // Create a session option for a single GPU device.
86 SessionOptions options;
87
88 // Make a local copy of graph def, because we need to change some things.
89 GraphDef graph_def(graph_def_arg);
90
91 if (cfg.erase_noinline_attributes) {
92 // TF optimizer doesn't inline functions with "_noinline" attribute,
93 // so let's go over the function library and erase it.
94 for (auto& func : *graph_def.mutable_library()->mutable_function()) {
95 func.mutable_attr()->erase("_noinline");
96 }
97 }
98
99 // Instantiate all variables for function library runtime creation.
100 std::vector<Device*> devices;
101 TF_RETURN_IF_ERROR(DeviceFactory::AddDevices(
102 options, "/job:localhost/replica:0/task:0", &devices));
103 std::unique_ptr<DeviceMgr> dvc_mgr(new DeviceMgr(devices));
104 FunctionLibraryDefinition function_library(OpRegistry::Global(),
105 graph_def.library());
106 Env* env = Env::Default();
107
108 // Optimizer options: L1 and inlining. L1 is default.
109 OptimizerOptions* optimizer_opts =
110 options.config.mutable_graph_options()->mutable_optimizer_options();
111 if (cfg.apply_optimizations) {
112 optimizer_opts->set_opt_level(::tensorflow::OptimizerOptions_Level_L1);
113 } else {
114 optimizer_opts->set_opt_level(::tensorflow::OptimizerOptions_Level_L0);
115 }
116
117 // Create the function library runtime.
118 std::unique_ptr<ProcessFunctionLibraryRuntime> pflr(
119 new ProcessFunctionLibraryRuntime(dvc_mgr.get(), env,
120 graph_def.versions().producer(),
121 &function_library, *optimizer_opts));
122 FunctionLibraryRuntime* flr = pflr->GetFLR(devices[0]->name());
123
124 // Create the GraphOptimizer to optimize the graph def.
125 GraphConstructorOptions graph_ctor_opts;
126 graph_ctor_opts.allow_internal_ops = true;
127 graph_ctor_opts.expect_device_spec = false;
128 std::unique_ptr<Graph> graphptr(new Graph(function_library));
129
130 TF_RETURN_IF_ERROR(
131 ConvertGraphDefToGraph(graph_ctor_opts, graph_def, graphptr.get()));
132
133 // Optimize the graph.
134 ::tensorflow::GraphOptimizer optimizer(*optimizer_opts);
135 optimizer.Optimize(flr, env, devices[0], &graphptr, /*shape_map=*/nullptr);
136 graphptr->ToGraphDef(output_graph_def);
137
138 // The default values of attributes might have been stripped by the optimizer.
139 // Add them back.
140 return AddDefaultAttrsToGraphDef(output_graph_def, *graphptr->op_registry(),
141 0, true);
142}

144// Applies the same graph pruning logic to the graph as Session.Run in TF.
145// If the returned status is not OK, item state may be inconsistent.
146Status PruneGraph(GrapplerItem* item) {
147 ModelPruner pruner;
148 GraphDef pruned_graph;
149 Cluster* cluster = nullptr; // ModelPruner doesn't check cluster.
150 TF_RETURN_IF_ERROR(pruner.Optimize(cluster, *item, &pruned_graph));
151 item->graph = std::move(pruned_graph);
152 return Status::OK();
153}

155// Replace any unknown dimensions in a shape with
156// cfg.placeholder_unknown_output_shape_dim if it is no less than 0.
157Status ReplaceUnknownShapeDim(const ItemConfig& cfg,
158 const TensorShapeProto& shape_pb_in,
159 TensorShapeProto* shape_pb_out,
160 TensorShape* shape_out) {
161 std::vector<int32> dims;
162 for (const auto& dim_proto : shape_pb_in.dim()) {
163 if (cfg.placeholder_unknown_output_shape_dim >= 0 &&
164 dim_proto.size() == -1) {
165 dims.push_back(cfg.placeholder_unknown_output_shape_dim);
166 shape_pb_out->add_dim()->set_size(
167 cfg.placeholder_unknown_output_shape_dim);
168 } else {
169 dims.push_back(std::max<int32>(1, dim_proto.size()));
170 shape_pb_out->add_dim()->set_size(dim_proto.size());
171 }
172 }
173 return TensorShapeUtils::MakeShape(dims.data(), dims.size(), shape_out);
174}

178// static
179std::unique_ptr<GrapplerItem> GrapplerItemFromMetaGraphDef(
180 const string& id, const MetaGraphDef& meta_graph, const ItemConfig& cfg) {
181 if (id.empty()) {
182 LOG(ERROR) << "id must be non-empty.";
183 return nullptr;
184 }
185 std::unique_ptr<GrapplerItem> new_item(new GrapplerItem());
186 new_item->id = id;
187 new_item->graph = meta_graph.graph_def();
188
189 // Fill in feed nodes from config, if any provided.
190 for (const auto& feed_node : cfg.feed_nodes) {
191 const string feed_name = NodeName(feed_node);
192 new_item->feed.emplace_back(feed_name, Tensor());
193 }
194
195 // Attempt to detect the fetch node(s).
196 if (meta_graph.collection_def().count("train_op") > 0) {
197 const CollectionDef& nodes = meta_graph.collection_def().at("train_op");
198 if (nodes.has_node_list()) {
199 for (const auto& node : nodes.node_list().value()) {
200 new_item->fetch.push_back(NodeName(node));
201 }
202 }
203 }
204
205 // Detect feed and fetch nodes from signature defs. Signatures may share same
206 // inputs or outputs.
207 std::unordered_set<string> signature_feed_nodes;
208 std::unordered_set<string> signature_fetch_nodes;
209 for (const auto& name_and_signature : meta_graph.signature_def()) {
210 for (const auto& name_and_input : name_and_signature.second.inputs()) {
211 const TensorInfo& input = name_and_input.second;
212 if (input.has_coo_sparse()) {
213 // Define the shapes following the comment of CooSparse.
214 // TODO(yuefengz): we probably want to use different dim values for the
215 // three tensors of a SparseTensor.
216 int64 dim = std::max(1, cfg.placeholder_unknown_output_shape_dim);
217 TensorShape shape_1d({dim});
218 TensorShape shape_2d({dim, dim});
219
220 if (gtl::InsertIfNotPresent(
221 &signature_feed_nodes,
222 NodeName(input.coo_sparse().values_tensor_name()))) {
223 Tensor value_tensor(input.dtype(), shape_1d);
224 InitializeTensor(input.dtype(), &value_tensor);
225 new_item->feed.emplace_back(
226 NodeName(input.coo_sparse().values_tensor_name()), value_tensor);
227 }
228 if (gtl::InsertIfNotPresent(
229 &signature_feed_nodes,
230 NodeName(input.coo_sparse().indices_tensor_name()))) {
231 Tensor indices_tensor(DT_INT64, shape_2d);
232 InitializeTensor(input.dtype(), &indices_tensor);
233 new_item->feed.emplace_back(
234 NodeName(input.coo_sparse().indices_tensor_name()),
235 indices_tensor);
236 }
237 if (gtl::InsertIfNotPresent(
238 &signature_feed_nodes,
239 NodeName(input.coo_sparse().dense_shape_tensor_name()))) {
240 Tensor dense_shape_tensor(DT_INT64, shape_1d);
241 InitializeTensor(input.dtype(), &dense_shape_tensor);
242 new_item->feed.emplace_back(
243 NodeName(input.coo_sparse().dense_shape_tensor_name()),
244 dense_shape_tensor);
245 }
246 } else {
247 if (gtl::InsertIfNotPresent(&signature_feed_nodes,
248 NodeName(input.name()))) {
249 TensorShape shape;
250 TensorShapeProto shape_proto;
251 Status s = ReplaceUnknownShapeDim(cfg, input.tensor_shape(),
252 &shape_proto, &shape);
253 if (!s.ok()) {
254 LOG(ERROR) << "Invalid shape for signature input " << input.name()
255 << ": " << s << ", skipping this input";
256 return nullptr;
257 }
258
259 Tensor fake_input(input.dtype(), shape);
260 InitializeTensor(input.dtype(), &fake_input);
261 new_item->feed.emplace_back(NodeName(input.name()), fake_input);
262 }
263 }
264 }
265 for (const auto& name_and_output : name_and_signature.second.outputs()) {
266 const TensorInfo& output = name_and_output.second;
267 if (output.has_coo_sparse()) {
268 if (gtl::InsertIfNotPresent(
269 &signature_fetch_nodes,
270 NodeName(output.coo_sparse().values_tensor_name()))) {
271 new_item->fetch.push_back(
272 NodeName(output.coo_sparse().values_tensor_name()));
273 }
274 if (gtl::InsertIfNotPresent(
275 &signature_fetch_nodes,
276 NodeName(output.coo_sparse().indices_tensor_name()))) {
277 new_item->fetch.push_back(
278 NodeName(output.coo_sparse().indices_tensor_name()));
279 }
280 if (gtl::InsertIfNotPresent(
281 &signature_fetch_nodes,
282 NodeName(output.coo_sparse().dense_shape_tensor_name()))) {
283 new_item->fetch.push_back(
284 NodeName(output.coo_sparse().dense_shape_tensor_name()));
285 }
286 } else {
287 if (gtl::InsertIfNotPresent(&signature_fetch_nodes,
288 NodeName(output.name()))) {
289 new_item->fetch.push_back(NodeName(output.name()));
290 }
291 }
292 }
293 }
294
295 for (const auto& feed : new_item->feed) {
296 if (feed.first.empty()) {
297 LOG(ERROR) << "Invalid feed node name skipping this input";
298 return nullptr;
299 } else {
300 VLOG(1) << "Will use feed node " << feed.first;
301 }
302 }
303
304 for (const auto& fetch : new_item->fetch) {
305 if (fetch.empty()) {
306 LOG(ERROR) << "Invalid fetch node name skipping this input";
307 return nullptr;
308 } else {
309 VLOG(1) << "Will use fetch node " << fetch;
310 }
311 }
312
313 if (new_item->fetch.empty()) {
314 LOG(ERROR) << "Failed to detect the fetch node(s), skipping this input";
315 return nullptr;
316 }
317
318 // TODO(yuefengz): consider handling saved_model_main_op and legacy_init_op.
319 // The reason why they are difficult to handle is because they may not intend
320 // to initialize all variables that are required to run fetch nodes. We may
321 // have to run restore op first.
322
323 // Try to find initializers from variables and tables as init ops.
324 for (const string& var_collection :
325 {"variables", "local_variables", "model_variables",
326 "trainable_variables"}) {
327 if (meta_graph.collection_def().count(var_collection) == 0) {
328 continue;
329 }
330 const CollectionDef& vars = meta_graph.collection_def().at(var_collection);
331 for (const auto& raw_var : vars.bytes_list().value()) {
332 VariableDef var;
333 var.ParseFromString(raw_var);
334 if (!var.initializer_name().empty()) {
335 new_item->init_ops.push_back(NodeName(var.initializer_name()));
336 }
337 }
338 }
339
340 if (meta_graph.collection_def().count("table_initializer") > 0) {
341 const CollectionDef& inits =
342 meta_graph.collection_def().at("table_initializer");
343 if (inits.has_node_list()) {
344 for (const auto& node : inits.node_list().value()) {
345 new_item->init_ops.push_back(NodeName(node));
346 // Tables are initialized from files, which can take a long time. Add
347 // 30 minutes to the initialization time for each table to avoid
348 // timing out.
349 // TODO(bsteiner): adjust the timeout based on the file size.
350 new_item->expected_init_time += 30 * 60;
351 }
352 }
353 }
354
355 // We keep the mapping from asset node to asset files. This should have been
356 // used as feed but since asset node is usually a constant node, we will fill
357 // the values of these constant nodes with their actual asset file paths.
358 std::unordered_map<string, string> asset_node_to_value;
359
360 // Assets file may have changed their directory, we assemble their new paths
361 // if assets_directory_override is set. We also make sure we still can
362 // access these asset files.
363 if (!cfg.assets_directory_override.empty()) {
364 if (meta_graph.collection_def().count("saved_model_assets") > 0) {
365 const CollectionDef& collection =
366 meta_graph.collection_def().at("saved_model_assets");
367 const auto& any_assets = collection.any_list().value();
368 for (const auto& any_asset : any_assets) {
369 AssetFileDef asset_file_def;
370 if (!ParseAny(any_asset, &asset_file_def, "tensorflow.AssetFileDef")
371 .ok()) {
372 LOG(ERROR) << "Failed to parse AssetFile.";
373 continue;
374 }
375 string asset_filepath = io::JoinPath(cfg.assets_directory_override,
376 asset_file_def.filename());
377 if (!FilesExist({asset_filepath}, nullptr)) {
378 LOG(ERROR) << "Can't access one or more of the asset files "
379 << asset_filepath << ", skipping this input";
380 return nullptr;
381 }
382 asset_node_to_value[NodeName(asset_file_def.tensor_info().name())] =
383 asset_filepath;
384 }
385 }
386 } else if (meta_graph.collection_def().count("asset_filepaths") > 0) {
387 const CollectionDef& file_paths =
388 meta_graph.collection_def().at("asset_filepaths");
389 std::vector<string> paths;
390 for (const auto& raw_path : file_paths.bytes_list().value()) {
391 paths.push_back(raw_path);
392 }
393 if (!FilesExist(paths, nullptr)) {
394 LOG(ERROR) << "Can't access one or more of the asset files, skipping "
395 "this input";
396 return nullptr;
397 }
398 }
399
400 if (meta_graph.collection_def().count("queue_runners") > 0) {
401 const CollectionDef& vars = meta_graph.collection_def().at("queue_runners");
402 for (const auto& raw : vars.bytes_list().value()) {
403 QueueRunnerDef queue_runner;
404 if (!queue_runner.ParseFromString(raw)) {
405 LOG(ERROR) << "Could not parse queue_runners, skipping this input";
406 return nullptr;
407 }
408 if (queue_runner.cancel_op_name().empty()) {
409 LOG(ERROR) << "Queue without a cancel op, skipping this input";
410 return nullptr;
411 }
412 new_item->queue_runners.push_back(queue_runner);
413 }
414 }
415
416 // Add each node referenced in a collection to the list of nodes to keep.
417 for (const auto& col : meta_graph.collection_def()) {
418 const CollectionDef& collection = col.second;
419 for (const string& node : collection.node_list().value()) {
420 new_item->keep_ops.push_back(NodeName(node));
421 }
422 }
423
424 for (auto& node : *new_item->graph.mutable_node()) {
425 if (IsPlaceholder(node) && node.op() != "PlaceholderWithDefault") {
426 if (node.attr().count("dtype") == 0) {
427 LOG(ERROR) << "Unknown type for placeholder " << node.name()
428 << ", skipping this input";
429 return nullptr;
430 }
431 DataType type = node.attr().at("dtype").type();
432
433 if (node.attr().count("shape") == 0) {
434 LOG(INFO) << "Unknown shape for placeholder " << node.name()
435 << ", skipping this input";
436 return nullptr;
437 }
438
439 // Replace all unknown dimensions in the placeholder's tensorshape proto
440 // with cfg.placeholder_unknown_output_shape_dim and create a tensorshape
441 // from it. We do this because in newer protos, the input placeholder
442 // shape is not empty if the shape is partially defined.
443 TensorShape shape;
444 TensorShapeProto shape_proto;
445 Status make_shape_status = ReplaceUnknownShapeDim(
446 cfg, node.attr().at("shape").shape(), &shape_proto, &shape);
447 if (!make_shape_status.ok()) {
448 LOG(ERROR) << "Invalid shape for placeholder " << node.name() << ": "
449 << make_shape_status << ", skipping this input";
450 return nullptr;
451 }
452
453 // Some placeholder nodes have a mis-match between the node
454 // attribute "shape" and a different node attribute "_output_shapes".
455 // Specifically, a shape with shape.dims() == 0 could indicate either
456 // a scalar or an unknown shape. In those cases, we check _output_shapes
457 // for additional information.
458 // This case is observed in the bnmt graphs. Have not observed any
459 // cases where there was more than 1 _output_shapes, so limit it
460 // to cases where there is only 1 _output_shapes.
461 // We only do this if cfg.placeholder_unknown_output_shape_dim has
462 // been set to avoid crashing non-BNMT graphs.
463 if ((cfg.placeholder_unknown_output_shape_dim >= 0) &&
464 (shape.dims() == 0) && (node.attr().count("_output_shapes") == 1)) {
465 const auto& output_shapes =
466 node.attr().at("_output_shapes").list().shape(0);
467
468 if (output_shapes.dim_size() != 0) {
469 shape.Clear();
470 shape_proto.clear_dim();
471
472 for (const auto& dim : output_shapes.dim()) {
473 auto size = dim.size();
474 if (size == -1) size = cfg.placeholder_unknown_output_shape_dim;
475 shape.AddDim(size);
476 shape_proto.add_dim()->set_size(size);
477 }
478 }
479 }
480
481 Tensor fake_input(type, shape);
482 InitializeTensor(type, &fake_input);
483
484 if (cfg.feed_nodes.empty()) {
485 // No specific feed nodes were given. Assume all placeholders are fed.
486 if (signature_feed_nodes.count(node.name()) == 0) {
487 new_item->feed.emplace_back(node.name(), fake_input);
488 }
489 } else if (cfg.feed_nodes.count(node.name()) > 0) {
490 // If specific feed nodes were given, only update their tensors.
491 auto it = find_if(new_item->feed.begin(), new_item->feed.end(),
492 [&node](std::pair<string, Tensor>& f) {
493 return f.first == node.name();
494 });
495 QCHECK(it != new_item->feed.end());
496 it->second = fake_input;
497 }
498
499 // Set the shape of the node in the graph. This is needed for statically
500 // inferring shapes and is a no-op when dynamically inferring shapes as
501 // the Placeholder shape will match the shape passed from new_item->feed.
502 *(node.mutable_attr()->at("shape").mutable_shape()) = shape_proto;
503 } else if (IsConstant(node)) {
504 auto it = asset_node_to_value.find(node.name());
505 if (it != asset_node_to_value.end()) {
506 auto iter = node.mutable_attr()->find("value");
507 if (iter == node.attr().end()) {
508 LOG(ERROR) << "Value attribute expected in const op for asset files";
509 return nullptr;
510 }
511 if (!iter->second.has_tensor() ||
512 iter->second.tensor().string_val_size() != 1) {
513 LOG(INFO) << "Unexected AttrValue proto: "
514 << iter->second.DebugString();
515 return nullptr;
516 }
517 LOG(INFO) << "Using asset file " << it->second << " for node "
518 << node.name();
519 *(iter->second.mutable_tensor()->mutable_string_val(0)) = it->second;
520 }
521 }
522
523 // Erase the recorded result of any previous shape inference to start again
524 // from scratch.
525 node.mutable_attr()->erase("_output_shapes");
526
527 // Delete user specified placement if requested.
528 if (cfg.ignore_user_placement) {
529 node.clear_device();
530 }
531 // Delete colocation constraints if requested.
532 if (cfg.ignore_colocation) {
533 auto attr = node.mutable_attr();
534 auto it = attr->find("_class");
535 if (it != attr->end()) {
536 attr->erase(it);
537 }
538 }
539 }
540
541 if (meta_graph.collection_def().count("savers") > 0) {
542 const CollectionDef& savers = meta_graph.collection_def().at("savers");
543 for (const auto& raw : savers.bytes_list().value()) {
544 SaverDef saver;
545 // Skip bad savers since we don't need saves/restores to be able to run a
546 // graph.
547 if (!saver.ParseFromString(raw)) {
548 continue;
549 }
550 if (saver.filename_tensor_name().empty()) {
551 continue;
552 }
553 new_item->save_op = saver.save_tensor_name();
554 new_item->restore_op = saver.restore_op_name();
555 new_item->save_restore_loc_tensor = saver.filename_tensor_name();
556 // Only use the first saver since it's not clear what to do if there's
557 // more than one.
558 break;
559 }
560 } else {
561 const SaverDef& saver = meta_graph.saver_def();
562 new_item->save_op = saver.save_tensor_name();
563 new_item->restore_op = saver.restore_op_name();
564 new_item->save_restore_loc_tensor = saver.filename_tensor_name();
565 }
566
567 // Instantiate all the missing attributes with their default values.
568 Status attr_status = AddDefaultAttrsToGraphDef(
569 &new_item->graph,
570 FunctionLibraryDefinition(OpRegistry::Global(),
571 new_item->graph.library()),
572 0, true);
573 if (!attr_status.ok()) {
574 LOG(ERROR) << "Failed to instantiate default attribute values: "
575 << attr_status.error_message();
576 return nullptr;
577 }
578
579 // Optimize the graph (function inlining, l1 optimizations, etc).
580 VLOG(1) << "Number of nodes in graph before OptimizeGraph: "
581 << new_item->graph.node_size();
582 Status optimize_status =
583 OptimizeGraph(new_item->graph, &new_item->graph, cfg);
584 if (!optimize_status.ok()) {
585 LOG(ERROR) << "Graph preprocessing failed: " << optimize_status;
586 return nullptr;
587 }
588 VLOG(1) << "Number of nodes in graph after OptimizeGraph: "
589 << new_item->graph.node_size();
590
591 if (cfg.prune_graph) {
592 VLOG(1) << "Pruning graph...";
593 auto status = PruneGraph(new_item.get());
594 if (!status.ok()) {
595 LOG(ERROR) << "Pruning failed: " << status.error_message();
596 return nullptr;
597 }
598 VLOG(1) << "Number of nodes in graph after pruning: "
599 << new_item->graph.node_size();
600 }
601
602 // Validate feed, fetch and init nodes
603 std::unordered_set<string> nodes;
604 for (const auto& node : new_item->graph.node()) {
605 nodes.insert(node.name());
606 }
607 for (const auto& feed : new_item->feed) {
608 if (nodes.find(feed.first) == nodes.end()) {
609 LOG(ERROR) << "Feed node " << feed.first << " doesn't exist in graph";
610 return nullptr;
611 }
612 }
613 for (const auto& fetch : new_item->fetch) {
614 if (nodes.find(fetch) == nodes.end()) {
615 LOG(ERROR) << "Fetch node " << fetch << " doesn't exist in graph";
616 return nullptr;
617 }
