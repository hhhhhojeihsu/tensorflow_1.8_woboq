390bool IsFreeOfSideEffect(const NodeDef& node) {
391 // Placeholders must be preserved to keep the graph feedable.
392 if (IsPlaceholder(node)) {
393 return false;
394 }
395 const OpDef* op_def = nullptr;
396 const string& op_name = node.op();
397 Status status = OpRegistry::Global()->LookUpOpDef(op_name, &op_def);
398 if (!status.ok()) {
399 return false;
400 }
401 if (op_def->is_stateful()) {
402 return false;
403 }
404 // Nodes such as Assign or AssignAdd modify one of their inputs.
405 for (const auto& input : op_def->input_arg()) {
406 if (input.is_ref()) {
407 return false;
408 }
409 }
410 return !ModifiesInputsInPlace(node);
411}
412
413bool ModifiesInputsInPlace(const NodeDef& node) {
414 // Some nodes do in-place updates on regular tensor inputs.
415 string op_name = node.op();
416
417 // Ops that modify resource variables effectively modify one of their inputs.
418 if (op_name == "AssignVariableOp" || op_name == "AssignAddVariableOp" ||
419 op_name == "AssignSubVariableOp" || op_name == "ResourceScatterUpdate" ||
420 op_name == "ResourceScatterAdd" || op_name == "ResourceScatterSub" ||
421 op_name == "ResourceScatterMul" || op_name == "ResourceScatterDiv" ||
422 op_name == "ResourceScatterMin" || op_name == "ResourceScatterMax") {
423 return false;
424 }
425
426 std::transform(op_name.begin(), op_name.end(), op_name.begin(), ::tolower);
427 if (str_util::StrContains(op_name, "inplace")) {
428 return true;
429 }
430 return GetBoolAttr(node, "in_place") || GetBoolAttr(node, "inplace");
431}

437#define OPDEF_PROPERTY_HELPER(PROPERTY_CAP, PROPERTY) \
438 bool Is##PROPERTY_CAP(const NodeDef& node) { \
439 if (node.op() == "Add") { \
440 /* Workaround for "Add" not being marked is_commutative and */ \
441 /* is_aggregate. (See cl/173915048). */ \
442 const auto type = GetDataTypeFromAttr(node, "T"); \
443 return type != DT_INVALID && type != DT_STRING; \
444 } \
445 const OpDef* op_def = nullptr; \
446 Status status = OpRegistry::Global()->LookUpOpDef(node.op(), &op_def); \
447 return status.ok() && op_def->is_##PROPERTY(); \
448 }

453bool IsInvolution(const NodeDef& node) {
454 const std::unordered_set<string> involution_ops{
455 "Conj", "Reciprocal", "Invert", "Neg", "LogicalNot"};
456 return involution_ops.count(node.op()) > 0;
457}

459bool IsValuePreserving(const NodeDef& node) {
460 if (NumNonControlInputs(node) == 1 && IsAggregate(node)) {
461 return true;
462 }
463 const std::unordered_set<string> value_preserving_ops{
464 "Transpose", "Reshape", "Identity", "InvertPermutation",
465 "Reverse", "StopGradient", "PreventGradient", "CheckNumerics",
466 "ExpandDims", "Squeeze"};
467 return value_preserving_ops.count(node.op()) > 0;
468}

470bool HasOpDef(const NodeDef& node) {
471 const OpDef* op_def = nullptr;
