33MeasuringCostEstimator::MeasuringCostEstimator(Cluster* cluster,
34 int measurement_steps,
35 int measurement_threads)
36 : measurement_steps_(measurement_steps),
37 measurement_threads_(measurement_threads) {
38 CHECK_GE(measurement_steps, 1);
39 if (measurement_threads > 0) {
40 thread_pool_.reset(new thread::ThreadPool(
41 Env::Default(), SanitizeThreadSuffix("measurements"),
42 measurement_threads));
43 }
44 cluster_ = cluster;
45}
46
47Status MeasuringCostEstimator::Initialize(const GrapplerItem& item) {
48 feed_ = item.feed;
49 fetch_ = item.fetch;
50 return cluster_->Initialize(item);
51}

53Status MeasuringCostEstimator::PredictCosts(const GraphDef& optimized_graph,
54 CostGraphDef* cost_graph,
55 Costs* costs) const {
56 const bool running_simulation = (cluster_->type() == "virtual");
57
58 std::vector<double> times(measurement_steps_);
59 BlockingCounter barrier(measurement_steps_);
60
61 mutex status_mu;
62 Status status;
63
64 auto measurement_fn = [&](const int step) {
65 const Costs::MicroSeconds start = Env::Default()->NowMicros();
66
67 RunMetadata metadata;
68 const Status local_status =
69 cluster_->Run(optimized_graph, feed_, fetch_, &metadata);
70 {
71 mutex_lock lock(status_mu);
72 status.Update(local_status);
73 }
74 if (step < 0) {
75 // Discard the first iteration as it triggers the warmup, and therefore
76 // takes much longer than a normal step.
77 return;
78 }
79 if (!local_status.ok()) {
80 // Discard the data if the run wasn't successful.
81 barrier.DecrementCount();
82 return;
83 }
84
85 const Costs::MicroSeconds finish = Env::Default()->NowMicros();
86 if (running_simulation) {
87 // When running simulation, return the estimated runtime, not the time it
88 // takes to run the simulation.
89 double time = 0.0;
90 for (const DeviceStepStats& stepstats :
91 metadata.step_stats().dev_stats()) {
92 for (const NodeExecStats& node_stats : stepstats.node_stats()) {
93 const double completion_time =
94 node_stats.all_end_rel_micros() + node_stats.all_start_micros();
95 time = std::max(time, completion_time * 1e3);
96 }
97 }
98 times[step] = time;
99 } else {
100 const double time = (finish - start).count() * 1e3;
101 times[step] = time;
102 }
103 if (cost_graph && (step + 1 == measurement_steps_)) {
104 metadata.mutable_cost_graph()->Swap(cost_graph);
105 }
106
107 barrier.DecrementCount();
108 };
109
110 // Initialize the computation and warm up TensorFlow.
111 measurement_fn(-1);
112
113 if (!status.ok()) {
114 LOG(ERROR) << "Failed to run start measurements: "
115 << status.error_message();
116 costs->execution_time = Costs::Duration::max();
117 return status;
118 }
119
120 // Run "measurement_steps_" and measure the time.
121 VLOG(1) << "Number of measurement steps: " << measurement_steps_;
122 if (measurement_threads_ > 0) {
123 for (int i = 0; i < measurement_steps_; ++i) {
124 thread_pool_->Schedule([i, &measurement_fn]() { measurement_fn(i); });
125 }
126 barrier.Wait();
127 } else {
128 for (int i = 0; i < measurement_steps_ && status.ok(); ++i) {
129 measurement_fn(i);
130 }
131 }
132
133 if (!status.ok()) {
134 LOG(ERROR) << "Failed to measure graph performance: "
135 << status.error_message();
136 costs->execution_time = Costs::Duration::max();
137 return status;
138 }
139
140 // Compute the average time of the measure steps. Use Huber statistics
141 // to filter out outliers.
142 RobustStats stats(times);
143 costs->execution_time = Costs::Duration(stats.mean());
144
145 return Status::OK();
146}
