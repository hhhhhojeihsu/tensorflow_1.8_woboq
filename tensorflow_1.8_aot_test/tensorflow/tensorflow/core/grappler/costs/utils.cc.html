57static OpInfo::TensorProperties UnknownInput() {
58 OpInfo::TensorProperties input;
59 input.set_dtype(DataType::DT_INVALID);
60 input.mutable_shape()->set_unknown_rank(true);
61 return input;
62}

64static std::vector<TensorProto> ExtractTensors(const AttrValue& attr_value) {
65 std::vector<TensorProto> tensors;
66 switch (attr_value.value_case()) {
67 case AttrValue::kTensor: {
68 tensors.push_back(attr_value.tensor());
69 break;
70 }
71 case AttrValue::kList: {
72 for (const auto& tensor_proto : attr_value.list().tensor()) {
73 tensors.push_back(tensor_proto);
74 }
75 break;
76 }
77 default: {}
78 }
79 return tensors;
80}

82// Annotate the op_info inputs with extra information when possible (e.g. the
83// input value if it's known statically).
84static void ExtractExtraProperties(
85 const NodeDef& node,
86 const std::unordered_map<string, const NodeDef*>& name_to_node,
87 OpInfo* op_info) {
88 OpRegistry* op_registry = OpRegistry::Global();
89 const OpDef* op_def = nullptr;
90 auto s = op_registry->LookUpOpDef(node.op(), &op_def);
91 if (!s.ok()) {
92 op_def = nullptr;
93 }
94
95 for (int i = 0; i < node.input_size(); ++i) {
96 const string input_name = node.input(i);
97 CHECK(!input_name.empty());
98 if (IsControlInput(input_name)) {
99 continue;
100 }
101 TensorId input_tensor_id = ParseTensorName(input_name);
102 const string input_node_name = input_tensor_id.first.ToString();
103
104 auto iter = name_to_node.find(input_node_name);
105 if (iter == name_to_node.end()) continue;
106 const NodeDef* input_node = iter->second;
107
108 if (i >= op_info->inputs_size()) {
109 LOG(ERROR) << "OpInfo's inputs doesn't match the graph! OpInfo: "
110 << op_info->DebugString()
111 << "\nCurrent node: " << node.DebugString()
112 << "\nInput node: " << input_node->DebugString();
113 }
114
115 // The value attribute in Const input is useful for cost prediction.
116 if (input_node->op() == "Const" && i < op_info->inputs_size()) {
117 auto it = input_node->attr().find("value");
118 if (it == input_node->attr().end()) continue;
119
120 const AttrValue& attr_value = it->second;
121 std::vector<TensorProto> tensors = ExtractTensors(attr_value);
122 if (tensors.empty()) continue;
123
124 const TensorProto& t = tensors[0];
125 OpInfo::TensorProperties* input = op_info->mutable_inputs(i);
126 *(input->mutable_value()) = t;
127
128 // For filename input, the file size can also be useful.
129 if (op_def && i < op_def->input_arg_size() &&
130 op_def->input_arg(i).name().find("filename") != std::string::npos) {
131 Tensor tensor;
132 if (!tensor.FromProto(t)) {
133 continue;
134 }
135 if (tensor.NumElements() != 1) {
136 continue;
137 }
138 const string filename = tensor.scalar<string>()();
139
140 Env* env = Env::Default();
141 FileStatistics stat;
142 Status s = env->Stat(filename, &stat);
143 if (!s.ok()) {
144 continue;
145 }
146 AttrValue attr;
147 attr.set_i(stat.length);
148 string attr_key = strings::StrCat("input_", i, "_filesize");
149 (*op_info->mutable_attr())[attr_key] = attr;
150 }
151 }
152
153 // When the input is a handle (e.g. look up table handle), the information
154 // in the op itself is not sufficient to predict the op memory.
155 if (op_def && i < op_def->input_arg_size() &&
156 op_def->input_arg(i).name().find("handle") != std::string::npos) {
157 string new_key = strings::StrCat("parent_", i, "_op");
158 AttrValue attr;
159 attr.set_s(input_node->op());
160 (*op_info->mutable_attr())[new_key] = attr;
161 // TODO(yuefengz): Only parent node's op name is copied. Copy inputs
162 // and attributes when necessary.
163 }
164 }
165}

167std::vector<OpInfo::TensorProperties> FindInputFeatures(
168 const NodeDef& node,
169 const std::unordered_map<string, const CostGraphDef::Node*>& name_to_cost,
170 const std::unordered_map<string, const NodeDef*>& name_to_node) {
171 std::vector<OpInfo::TensorProperties> inputs;
172 for (const auto& input_name : node.input()) {
173 CHECK(!input_name.empty());
174 TensorId input_tensor_id = ParseTensorName(input_name);
175 const string input_node_name = input_tensor_id.first.ToString();
176 const int output_index = input_tensor_id.second;
177
178 // Skip control inputs.
179 if (output_index == Graph::kControlSlot) {
180 continue;
181 }
182
183 auto it = name_to_cost.find(input_node_name);
184 if (it == name_to_cost.end() || output_index < 0) {
185 inputs.push_back(UnknownInput());
186 } else {
187 const CostGraphDef::Node* input_cost = it->second;
188 if (input_cost->output_info_size() == 0) {
189 inputs.push_back(UnknownInput());
190 } else {
191 const CostGraphDef::Node::OutputInfo& output =
192 input_cost->output_info(output_index);
193 OpInfo::TensorProperties input;
194 input.set_dtype(output.dtype());
195 *input.mutable_shape() = output.shape();
196 inputs.push_back(input);
197 }
198 }
199 }
200
201 return inputs;
202}

204DeviceProperties GetDeviceInfo(const string& device_str) {
205 DeviceProperties unknown;
206 unknown.set_type("UNKNOWN");
207
208 DeviceNameUtils::ParsedName parsed;
209 if (DeviceNameUtils::ParseFullName(device_str, &parsed)) {
210 if (parsed.type == "GPU") {
211 TfGpuId tf_gpu_id(parsed.id);
212 CudaGpuId cuda_gpu_id;
213 Status s = GpuIdManager::TfToCudaGpuId(tf_gpu_id, &cuda_gpu_id);
214 if (!s.ok()) {
215 // We are probably running simulation without linking cuda libraries.
216 cuda_gpu_id = CudaGpuId(parsed.id);
217 }
218 return GetLocalGPUInfo(cuda_gpu_id);
219 } else if (parsed.type == "CPU") {
220 return GetLocalCPUInfo();
221 }
222 }
223 return unknown;
224}

226DeviceProperties GetDeviceInfo(const CostGraphDef::Node& node) {
227 return GetDeviceInfo(node.device());

230OpInfo BuildOpInfoWithoutDevice(
231 const NodeDef& node,
232 const std::unordered_map<string, const NodeDef*>& name_to_node,
233 const std::vector<OpInfo::TensorProperties>& inputs) {
234 OpInfo op_info;
235 op_info.set_op(node.op());
236 *op_info.mutable_attr() = node.attr();
237 for (auto& input : inputs) {
238 *op_info.add_inputs() = input;
239 }
240 ExtractExtraProperties(node, name_to_node, &op_info);
241 return op_info;
242}

244string GetOpDescription(const OpInfo& op_info) {
245 string description = "[";
246 description += "Op=" + op_info.op() + ", ";
247 description += "input_shapes=[";
248 for (auto const& input : op_info.inputs()) {
249 description += PartialTensorShape::DebugString(input.shape());
250 }
251 description += "]";
252 return description;
253}

255OpPerformanceList CostGraphToOpPerformanceData(const CostGraphDef& cost_graph,
256 const GraphDef& graph) {
257 OpPerformanceList ret;
258 std::unordered_map<string, const CostGraphDef::Node*> name_to_cost;
259 std::unordered_map<string, const NodeDef*> name_to_node;
260 for (auto& node : cost_graph.node()) {
261 name_to_cost[node.name()] = &node;
262 }
263 for (auto& node : graph.node()) {
264 name_to_node[node.name()] = &node;
265 }
266
267 for (const auto& node : graph.node()) {
268 // Skip the nodes that are not in the cost graph: these are nodes that
269 // aren't run, because they aren't in the intersection of transitive
270 // fan-in of a fetch node and the transitive fan-out of an input, or nodes
271 // that were optimized away by the optimizer. Since they don't contribute
272 // to the execution time we simply discard them.
273 auto it = name_to_cost.find(node.name());
274 if (it == name_to_cost.end()) {
275 continue;
276 }
277 const CostGraphDef::Node* cost_node = it->second;
278
279 OpPerformance* perf = ret.add_op_performance();
280 perf->set_node(node.name());
281
282 std::vector<OpInfo::TensorProperties> inputs =
283 FindInputFeatures(node, name_to_cost, name_to_node);
284 *perf->mutable_op() = BuildOpInfoWithoutDevice(node, name_to_node, inputs);
285 *perf->mutable_op()->mutable_device() = GetDeviceInfo(cost_node->device());
286
287 perf->set_temporary_memory_size(cost_node->temporary_memory_size());
288 // Note that CostGraphDef::Node::compute_cost is microseconds, while
289 // OpPerformance.compute_cost is nanoseconds.
290 perf->set_compute_cost(cost_node->compute_cost() * 1000);
291 perf->set_compute_time(cost_node->compute_time() * 1000);
292 perf->set_memory_time(cost_node->memory_time() * 1000);
293
294 for (const auto& output_info : cost_node->output_info()) {
295 perf->mutable_op_memory()->add_output_memory(output_info.size());
296 }
297
298 perf->mutable_op_memory()->set_temp_memory(
299 cost_node->temporary_memory_size());
300 perf->mutable_op_memory()->set_persistent_memory(
301 cost_node->persistent_memory_size());
302 }
303 return ret;
304}

306void TensorSizeHistogram::Add(const uint64 value) {
307 num_elem_++;
308 sum_elem_ += value;
309 min_ = std::min(min_, value);
310 max_ = std::max(max_, value);
311 buckets_[Index(value)]++;
312}
313
314void TensorSizeHistogram::Merge(const TensorSizeHistogram& src) {
315 num_elem_ += src.num_elem_;
316 sum_elem_ += src.sum_elem_;
317 min_ = std::min(min_, src.min_);
318 max_ = std::max(max_, src.max_);
319 std::transform(buckets_.begin(), buckets_.end(), src.buckets_.begin(),
320 buckets_.begin(), std::plus<uint64>());
321}

323std::string TensorSizeHistogram::ToString() const {
324 std::string r;
325 char buf[200];
326 snprintf(buf, sizeof(buf), "Count: %lld, Average: ", num_elem_);
327 r.append(buf);
328 r.append(strings::HumanReadableNumBytes(Average()));
329 r.append(", Min: ");
330 r.append(strings::HumanReadableNumBytes(min_));
331 r.append(", Max: ");
332 r.append(strings::HumanReadableNumBytes(max_));
333 r.append("\n------------------------------------------------------\n");
334 const double mult = num_elem_ > 0 ? 100.0 / num_elem_ : 0.0;
335 uint64 cumul_sum = 0;
336
337 const int size_string_width = 12;
338 for (int i = 0; i < buckets_.size(); i++) {
339 if (buckets_[i] == 0) continue;
340 cumul_sum += buckets_[i];
341 r.append("[ ");
342 if (i == 0) {
343 r.append(size_string_width - 2, ' ');
344 r.append("0B");
345 } else {
346 uint64 left = 1ULL << (i - 1);
347 const auto left_string = strings::HumanReadableNumBytes(left);
348 r.append(size_string_width - left_string.size(), ' ');
349 r.append(left_string);
350 }
351 r.append(", ");
352 uint64 right = 1ULL << i;
353 const auto right_string = strings::HumanReadableNumBytes(right);
354 r.append(size_string_width - right_string.size(), ' ');
355 r.append(right_string);
356 snprintf(buf, sizeof(buf), ") %7lld %7.3f%% %7.3f%% ",
357 buckets_[i], // count
358 mult * buckets_[i], // percentage
359 mult * cumul_sum); // cum percentage
360 r.append(buf);
361
362 // Add hash marks based on percentage; 40 marks for 100%.
363 auto marks = static_cast<int>(
364 (static_cast<double>(40 * buckets_[i] + (num_elem_ >> 1)) / num_elem_));
365 r.append(marks, '#');
366 r.push_back('\n');
367 }
368 return r;
369}

371const int TensorSizeHistogram::Index(const uint64 value) const {
372 // Log2Floor64 returns -1 for 0, 0 for 1, 1 for 2-3, 2 for 4-7, ...
373 const auto index = Log2Floor64(value) + 1;
374 return std::min(index, kMaxBuckets - 1);
375}

377string GetDeviceClassForNonChannelDevice(const string& device_name) {
378 DeviceNameUtils::ParsedName parsed_name;
379 bool parsed = DeviceNameUtils::ParseFullName(device_name, &parsed_name);
380 if (!parsed) {
381 string name = str_util::StringReplace(device_name, "/job_", "/job:", true);
382 name = str_util::StringReplace(name, "/replica_", "/replica:", true);
383 name = str_util::StringReplace(name, "/task_", "/task:", true);
384 name = str_util::StringReplace(name, "/device_", "/device:", true);
385 name = str_util::StringReplace(name, "GPU_", "GPU:", true);
386 name = str_util::StringReplace(name, "CPU_", "CPU:", true);
387 name = str_util::StringReplace(name, "gpu_", "gpu:", true);
388 name = str_util::StringReplace(name, "cpu_", "cpu:", true);
389 parsed = DeviceNameUtils::ParseFullName(name, &parsed_name);
390 }
391 if (parsed) {
392 const string jobname = parsed_name.has_job ? parsed_name.job : "";
393 return strings::StrCat("/", jobname, "/", parsed_name.type);
394 } else {
395 return "Unclassified";
396 }
397}

399string GetDeviceClass(const string& device_name) {
400 // TODO(dyoon): channel device name follows the convention we currently have
401 // in VirtualScheduler. This should be revised with VirtualScheduler as well
402 // as VirtualPlacer in the future.
403 if (device_name.find("Channel") != string::npos) {
404 const string from = "_from_";
405 const string to = "_to_";
406 const auto from_loc = device_name.find(from);
407 const auto to_loc = device_name.find(to);
408 const auto src_device_full = device_name.substr(
409 from_loc + from.size(), to_loc - (from_loc + from.size()));
410 const auto dst_device_full = device_name.substr(to_loc + to.size());
411 return strings::StrCat(
412 "Channel", ": ", GetDeviceClassForNonChannelDevice(src_device_full),
413 " -> ", GetDeviceClassForNonChannelDevice(dst_device_full));
414 } else {
415 return GetDeviceClassForNonChannelDevice(device_name);
416 }
417}

419string GetStatsStringFromRunMetadata(const RunMetadata& run_metadata,
420 bool verbosity) {
421 // TODO(dyoon): print out other stats as needed.
422 std::ostringstream output;
423
424 // Tensor size histogram:
425 // if verbosity, it outputs per-device histogram,
426 // otherwise, only per-class histogram.
427 std::unordered_map<string, TensorSizeHistogram> device_to_hist_map;
428 const auto& step_stats = run_metadata.step_stats();
429 for (const auto& dev_stat : step_stats.dev_stats()) {
430 const auto& device_name = dev_stat.device();
431 auto& hist = device_to_hist_map[device_name];
432 for (const auto& node_stat : dev_stat.node_stats()) {
433 for (const auto& node_output : node_stat.output()) {
434 // TODO(dyoon): Calculate tensor size from tensor_description's dtype
435 // and shape, instead of using optional allocation_description.
436 const auto size = node_output.tensor_description()
437 .allocation_description()
438 .allocated_bytes();
439 hist.Add(size);
440 }
441 }
442 }
443 if (verbosity) {
444 output << "\n";
445 output << "Per device tensor size histogram.\n";
446 }
447
448 std::unordered_map<string, TensorSizeHistogram> device_class_to_hist_map;
449 for (const auto& device_hist : device_to_hist_map) {
450 const auto& device_name = device_hist.first;
451 const auto& hist = device_hist.second;
452 if (verbosity) {
453 output << "Device: " << device_name << "\n" << hist.ToString() << "\n";
454 }
455 const auto device_class = GetDeviceClass(device_name);
456 auto it = device_class_to_hist_map.find(device_class);
457 if (it == device_class_to_hist_map.end()) {
458 device_class_to_hist_map.emplace(device_class, TensorSizeHistogram(hist));
459 } else {
460 it->second.Merge(hist);
461 }
462 }
463 output << "\n";
464 output << "Aggregated per device / channel type tensor size histogram:\n";
465 for (const auto& device_hist : device_class_to_hist_map) {
466 const auto& device_name = device_hist.first;
467 const auto& hist = device_hist.second;
468 output << "Device: " << device_name << "\n" << hist.ToString() << "\n";
469 }
470 output << "\n";
471
472 return output.str();
473}
