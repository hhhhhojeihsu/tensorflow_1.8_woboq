32Status GraphMemory::InferStatically(
33 const std::unordered_map<string, DeviceProperties>& devices) {
34 VirtualCluster cluster(devices);
35 TF_RETURN_IF_ERROR(cluster.Provision());
36 TF_RETURN_IF_ERROR(cluster.Initialize(item_));
37 RunMetadata metadata;
38 Status s = cluster.Run(item_.graph, item_.feed, item_.fetch, &metadata);
39 // The virtual cluster returns the RESOURCE_EXHAUSTED error when it detects
40 // that the model would run out of memory. We still get the metadata we need
41 // out of the simulation, so we just ignore this error.
42 if (!s.ok() && s.code() != error::RESOURCE_EXHAUSTED) {
43 return s;
44 }
45 InferFromTrace(metadata.step_stats());
46 return Status::OK();
47}

49Status GraphMemory::InferDynamically(Cluster* cluster) {
50 if (!cluster->DetailedStatsEnabled()) {
51 return errors::Unavailable("Detailed stats collection must be enabled");
52 }
53
54 TF_RETURN_IF_ERROR(cluster->Initialize(item_));
55 RunMetadata metadata;
56 TF_RETURN_IF_ERROR(
57 cluster->Run(item_.graph, item_.feed, item_.fetch, &metadata));
58 InferFromTrace(metadata.step_stats());
59 return Status::OK();
60}

62int64 GraphMemory::GetWorstCaseMemoryUsage() const {
63 int64 worst_case = -1;
64 for (const auto& peak_usage : peak_usage_) {
65 worst_case = std::max(worst_case, peak_usage.second.used_memory);
66 }
67 return worst_case;
68}

70void GraphMemory::InferMemUsageForNodes(
71 const std::vector<const NodeDef*>& nodes, GraphProperties* properties,
72 int64* worst_case_memory_usage, int64* best_case_memory_usage) const {
73 // TODO(bsteiner) refine this: we should consider the multidevice case.
74 *worst_case_memory_usage = 0;
75 *best_case_memory_usage = 0;
76 for (const auto& node : item_.graph.node()) {
77 // Estimate the memory required to store the tensors generated by the node.
78 std::vector<OpInfo::TensorProperties> outputs =
79 properties->GetOutputProperties(node.name());
80 int64 node_memory_usage = InferMemUsageForNeighbors(outputs);
81
82 // Worst case memory usage corresponds to the case where all the nodes are
83 // alive.
84 *worst_case_memory_usage += node_memory_usage;
85
86 // Estimate the memory required to store the input tensors needed by the
87 // node.
88 std::vector<OpInfo::TensorProperties> inputs =
89 properties->GetInputProperties(node.name());
90 node_memory_usage += InferMemUsageForNeighbors(inputs);
91
92 *best_case_memory_usage =
93 std::max(*best_case_memory_usage, node_memory_usage);
94 }
95}
96
97int64 GraphMemory::InferMemUsageForNeighbors(
98 const std::vector<OpInfo::TensorProperties>& props) const {
99 int64 neighbors_memory_usage = 0;
100 for (const auto& prop : props) {
101 DataType dtype = prop.dtype();
102 int size = DataTypeSize(dtype);
103 TensorShapeProto shape = prop.shape();
104 if (shape.unknown_rank()) {
105 // Can't infer the size if the rank is unknown, just skip.
106 continue;
107 }
108 // If one of the dimensions is unknown statically, assume it's one.
109 for (int i = 0; i < shape.dim_size(); ++i) {
110 if (shape.dim(i).size() < 0) {
111 shape.mutable_dim(i)->set_size(1);
112 }
113 }
114 int num_elems = TensorShape(shape).num_elements();
115 neighbors_memory_usage += num_elems * size;
116 }
117 return neighbors_memory_usage;
118}

120static GraphMemory::LiveTensor* FindOrCreateLiveTensor(
121 const string& node_name, int output_id,
122 std::unordered_map<string, GraphMemory::LiveTensor*>* live_tensors,
123 std::list<GraphMemory::LiveTensor>* device_tensors) {
124 string name = strings::StrCat(node_name, ":", output_id);
125 GraphMemory::LiveTensor* live;
126 auto it = live_tensors->find(name);
127 if (it == live_tensors->end()) {
128 GraphMemory::LiveTensor temp;
129 temp.node = node_name;
130 temp.output_id = output_id;
131 temp.allocation_time = 0;
132 temp.deallocation_time = 0;
133 device_tensors->push_front(temp);
134 live = &device_tensors->front();
135 (*live_tensors)[name] = live;
136 } else {
137 live = it->second;
138 }
139 return live;
140}

142namespace {
143struct Event {
144 int64 timestamp;
145 bool allocated;
146 const GraphMemory::LiveTensor* tensor;
147
148 bool operator<(const Event& other) const {
149 return timestamp < other.timestamp;
150 }
151};

154void GraphMemory::InferFromTrace(const StepStats& timeline) {
155 std::unordered_map<string, string> node_placement;
156 for (const auto& dev_stats : timeline.dev_stats()) {
157 for (const auto& node_stats : dev_stats.node_stats()) {
158 node_placement[node_stats.node_name()] = dev_stats.device();
159 }
160 }
161
162 std::unordered_map<string, LiveTensor*> live_tensors;
163 std::unordered_map<string, std::list<LiveTensor>> live_tensors_per_device;
164
165 NodeMap node_map(&item_.graph);
166 for (const auto& dev_stats : timeline.dev_stats()) {
167 const string& device_name = dev_stats.device();
168 const bool is_gpu = (device_name.find("GPU:") || device_name.find("gpu:"));
169 std::list<LiveTensor>& device_tensors =
170 live_tensors_per_device[dev_stats.device()];
171 for (const auto& node_stats : dev_stats.node_stats()) {
172 for (int i = 0; i < node_stats.output_size(); ++i) {
173 const auto& output = node_stats.output(i);
174
175 LiveTensor* live = FindOrCreateLiveTensor(
176 node_stats.node_name(), i, &live_tensors, &device_tensors);
177 live->memory_used = output.tensor_description()
178 .allocation_description()
179 .allocated_bytes();
180
181 // Allocations typically take place at the very beginning of the op
182 // execution.
183 live->allocation_time =
184 Costs::MicroSeconds(node_stats.all_start_micros());
185 // Add one nanosecond to the completion time of the ops to account for
186 // TF overhead that slightly delays deallocations.
187 live->deallocation_time = std::max<Costs::Duration>(
188 live->deallocation_time,
189 Costs::NanoSeconds(1) +
190 Costs::MicroSeconds(node_stats.all_start_micros() +
191 node_stats.op_end_rel_micros()));
192 }
193
194 const NodeDef* node = node_map.GetNode(node_stats.node_name());
195 if (!node) {
196 // Skip nodes inserted by TF since they don't exist in the original
197 // graph (e.g _Send/_Recv nodes).
198 continue;
199 }
200 std::unordered_set<int> swapped_inputs;
201 if (is_gpu) {
202 auto it = node->attr().find("_swap_to_host");
203 if (it != node->attr().end()) {
204 const AttrValue& val = it->second;
205 for (int port_id : val.list().i()) {
206 swapped_inputs.insert(port_id);
207 }
208 }
209 }
210 for (int i = 0; i < node->input_size(); ++i) {
211 if (swapped_inputs.find(i) != swapped_inputs.end()) {
212 // The memory of swapped inputs will be released as early as possible:
213 // therefore ignore this input when determining the deallocation time
214 // of the tensor.
215 continue;
216 }
217 const string& input = node->input(i);
218 int position;
219 string input_node = ParseNodeName(input, &position);
220 if (position < 0) {
221 // Skip control dependencies
222 continue;
223 }
224 LiveTensor* live = FindOrCreateLiveTensor(
225 input_node, position, &live_tensors,
226 &live_tensors_per_device[node_placement[input_node]]);
227 live->deallocation_time = std::max<Costs::Duration>(
228 live->deallocation_time,
229 Costs::NanoSeconds(1) +
230 Costs::MicroSeconds(node_stats.all_start_micros() +
231 node_stats.op_end_rel_micros()));
232 }
233 }
234 }
235
236 for (const auto& live_per_device : live_tensors_per_device) {
237 std::vector<Event> events;
238 events.reserve(2 * live_per_device.second.size());
239 for (const auto& live : live_per_device.second) {
240 events.push_back(Event{live.allocation_time.count(), true, &live});
241 events.push_back(Event{live.deallocation_time.count(), false, &live});
242 }
243 std::stable_sort(events.begin(), events.end());
244 size_t peak = 0;
245 std::set<const LiveTensor*> live_at_peak;
246 size_t current = 0;
247 std::set<const LiveTensor*> currently_live;
248 for (int i = 0; i < events.size(); ++i) {
249 const auto& event = events[i];
250
251 if (event.allocated) {
252 VLOG(1) << "At time " << event.timestamp << " allocated "
253 << event.tensor->memory_used << " for tensor "
254 << event.tensor->node << ":" << event.tensor->output_id;
255 current += event.tensor->memory_used;
256 currently_live.insert(event.tensor);
257 } else {
258 VLOG(1) << "At time " << event.timestamp << " deallocated "
259 << event.tensor->memory_used << " for tensor "
260 << event.tensor->node << ":" << event.tensor->output_id;
261 current -= event.tensor->memory_used;
262 currently_live.erase(event.tensor);
263 }
264 if (i + 1 == events.size() ||
265 event.timestamp != events[i + 1].timestamp) {
266 if (current > peak) {
267 peak = current;
268 live_at_peak = currently_live;
269 }
270 }
271 }
272 MemoryUsage& peak_mem_usage = peak_usage_[live_per_device.first];
273 peak_mem_usage.used_memory = peak;
274 peak_mem_usage.live_tensors.clear();
275 peak_mem_usage.live_tensors.reserve(live_at_peak.size());
276 for (const auto& live : live_at_peak) {
277 peak_mem_usage.live_tensors.push_back(*live);
278 }
279 }
280}
