16#ifndef TENSORFLOW_CORE_GRAPPLER_COSTS_VIRTUAL_PLACER_H_
17#define TENSORFLOW_CORE_GRAPPLER_COSTS_VIRTUAL_PLACER_H_
18
19#include <unordered_map>
20#include "tensorflow/core/platform/types.h"
21#include "tensorflow/core/protobuf/device_properties.pb.h"
22
23namespace tensorflow {
24class NodeDef;
25
26namespace grappler {
27class Cluster;
28
29// The virtual placer emulates the behavior of the TF placer.
30class VirtualPlacer {
31 public:
32 VirtualPlacer(const Cluster* cluster);
33
34 const DeviceProperties& get_device(const NodeDef& node) const;
35
36 // Returns device name from cluster, which best matches the node.device()
37 // specification. Returns default device if no match was found or the
38 // node.device() could not be parsed.
39 string get_canonical_device_name(const NodeDef& node) const;
40
41 private:
42 // Converts given device name to Lowercase Fully-Qualified Name (LFQN) string.
43 // This helps us disambiguate device names internally and simplify matching.
44 // If device_name couldn't be parsed successfully, returns empty string.
45 string to_lfqn_or_empty(const string& device_name) const;
46
47 // Map based on the cluster info: cluster device name -> device properties.
48 std::unordered_map<string, DeviceProperties> devices_;
49
50 // Maps LFQN to original device name as it was declared in cluster.
51 std::unordered_map<string, string> lfqn_map_;
52
53 string default_device_name_;
54 string default_job_name_lowercase_;
55};
56
57} // namespace grappler
58} // end namespace tensorflow
59
60#endif // TENSORFLOW_CORE_GRAPPLER_COSTS_VIRTUAL_PLACER_H_
