29class SymbolicShapeRefiner;
30class TopoQueue;

32// Infer OpInfo::TensorProperties for graph nodes inputs/outputs.
33//
34// Typical use case, is to infer tensor properties from a graph, before doing
35// optimization pass. Nodes modified during optimization pass have to be
36// invalidated, to prevent further incorrect optimizations based on wrong shape
37// and data type properties.
38class GraphProperties {
39 public:
40 explicit GraphProperties(const GrapplerItem& item) : item_(item) {}
41
42 // Infer the shapes through abstract interpretation. Feed information can be
43 // incorrect so it should be discarded to ensure correctness of the analysis.
44 // However, it can help infer shapes in the fanout of fed nodes (even though
45 // the correctness of these shapes can't be guaranteed), so in some cases
46 // (such as simulation or scheduling) it makes sense of keep these shapes.
47 Status InferStatically(bool assume_valid_feeds);
48 // Infer the shape by running the graph on the specified cluster and recording
49 // the shapes of the processed tensors.
50 Status InferDynamically(Cluster* cluster);
51 // Extract the properties from a cost graph. For testing only since there is
52 // no way to ensure that the cost graph match the item.
53 Status InferFromCostGraph(const CostGraphDef& cost_graph);
54
55 // Stores `item_.graph` with the inferred output shapes to `output_graph_def`.
56 Status AnnotateOutputShapes(GraphDef* output_graph_def) const;
57
58 // Return the properties of node inputs/outputs, including data types and
59 // shapes. Note that the dimensions in the shapes can be negative. We use the
60 // -1 value to denote that we don't know anything about a dimension. We use
61 // values strictly less than -1 to encode symbolic dimensions: although we
62 // don't know the actual value of the symbolic dimension, we know that all the
63 // dimensions denoted by the same negative value are the equal.
64 bool HasInputProperties(const string& name) const;
65 bool HasOutputProperties(const string& name) const;
66 const std::vector<OpInfo::TensorProperties>& GetInputProperties(
67 const string& node_name) const;
68 const std::vector<OpInfo::TensorProperties>& GetOutputProperties(
69 const string& node_name) const;
70 // Invalidate input/output properties for nodes modified during graph
71 // optimization pass, to prevent potential optimizations, based on incorrect
72 // shape information.
73 void ClearInputProperties(const string& node_name);
74 void ClearOutputProperties(const string& node_name);
75
76 private:
77 // Merges shapes <shapes_and_types>, determined from an EnqueueV2 node, into
78 // <*queue_shapes_and_types>.
79 static Status MergeEnqueueShapesAndTypes(
80 SymbolicShapeRefiner* shape_refiner, const Node* qnode,
81 const std::vector<shape_inference::ShapeAndType>& shapes_and_types,
82 std::vector<shape_inference::ShapeAndType>* queue_shapes_and_types);
83 // Relaxes shapes <shapes_and_types>, determined from an EnqueueV2 node, into
84 // <*queue_shapes_and_types>.
85 static Status RelaxEnqueueShapesAndMergeTypes(
86 SymbolicShapeRefiner* shape_refiner, const Node* qnode,
87 const std::vector<shape_inference::ShapeAndType>& shapes_and_types,
88 std::vector<shape_inference::ShapeAndType>* queue_shapes_and_types);
89
90 // Update the shapes for qnode. If output shapes of qnode have changed,
91 // enqueue its fanout in 'new_shapes'.
92 static Status UpdateResource(
93 const Node* qnode, const std::unordered_set<const Node*>& queue_inputs,
94 SymbolicShapeRefiner* shape_refiner, bool relax, TopoQueue* new_shapes);
95
96 // Update the output shapes of a Merge node, and enqueue its fanout in
97 // new_shapes if needed.
98 static Status UpdateMergeNode(SymbolicShapeRefiner* shape_refiner,
99 const Node* node, bool relax,
100 TopoQueue* new_shapes);
101 // Process the Enter node, and enqueue its fanout in new_shapes if needed.
102 static Status UpdateEnter(SymbolicShapeRefiner* shape_refiner,
103 const Node* node, bool relax,
104 TopoQueue* new_shapes);
105 // Process a node that is used to feed the model.
106 Status OverwriteFedPorts(
107 SymbolicShapeRefiner* shape_refiner,
108 const std::unordered_map<string, std::unordered_set<int>>& fed_ports,
109 const Node* node, TopoQueue* new_shapes) const;
110 // Update the shapes for node 'n'. If output shapes for n have changed,
111 // enqueue its fanout in 'new_shapes'.
112 Status UpdateShapes(
113 SymbolicShapeRefiner* shape_refiner, bool relax,
114 const std::unordered_map<string, std::unordered_set<int>>& fed_ports,
115 const Node* n, TopoQueue* new_shapes) const;
116 // Propagate the shapes for the nodes enqueued in new_shapes and their
117 // transitive fanout until a fixed point is reached.
118 Status PropagateShapes(
119 SymbolicShapeRefiner* shape_refiner, bool relax, TopoQueue* new_shapes,
120 const std::unordered_map<const Node*, std::unordered_set<const Node*>>&
121 resources,
122 const std::unordered_map<string, std::unordered_set<int>>& fed_ports,
123 int num_loops) const;
124
125 // Data members
126 GrapplerItem item_;
127 std::map<string, std::vector<OpInfo::TensorProperties>> input_properties_;
128 std::map<string, std::vector<OpInfo::TensorProperties>> output_properties_;
129 const std::vector<OpInfo::TensorProperties> missing_properties_;
130};
