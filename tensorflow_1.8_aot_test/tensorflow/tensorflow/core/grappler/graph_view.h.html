19#include <unordered_map>
20#include <unordered_set>

25namespace tensorflow {
26namespace grappler {
27
28// A utility class to simplify the traversal of a GraphDef.
29class GraphView {
30 public:
31 struct Port {
32 NodeDef* node = nullptr;
33 int port_id = -1;
34
35 bool operator==(const Port& other) const {
36 return node == other.node && port_id == other.port_id;
37 }
38 };
39 struct InputPort : public Port {};
40 struct OutputPort : public Port {};
41
42 struct HashPort {
43 std::size_t operator()(const Port& port) const {
44 return reinterpret_cast<std::size_t>(port.node) + port.port_id;
45 }
46 };
47
48 explicit GraphView(GraphDef* graph);
49 GraphDef* GetGraph() const { return graph_; }
50 NodeDef* GetNode(const string& node_name) const;
51 // Get the specified input port. Note that the special '-1' port_id can be
52 // used to access the controlling nodes (i.e. the nodes connected to node_name
53 // through an incoming control dependency).
54 InputPort GetInputPort(const string& node_name, int port_id) const;
55 // Get the specified output port. Note that the special '-1' port_id can be
56 // used to access the controlled nodes (i.e. the nodes connected to node_name
57 // through an outgoing control dependency).
58 OutputPort GetOutputPort(const string& node_name, int port_id) const;
59
60 // Get the input (resp. output) port(s) in the immediate fanout (resp. fanin)
61 // of an output (resp. input) port.
62 const std::unordered_set<InputPort, HashPort>& GetFanout(
63 const OutputPort& port) const;
64 std::unordered_set<OutputPort, HashPort> GetFanin(
65 const InputPort& port) const;
66 // Special case: regular (i.e. non-control) input ports can only have one
67 // fanin.
68 const OutputPort GetRegularFanin(const InputPort& port) const;
69
70 // Get all the input (resp. output) ports in the immediate fanout (resp fanin)
71 // of a node. Include the controlling nodes iff include_controlling_nodes is
72 // true.
73 std::unordered_set<InputPort, HashPort> GetFanouts(
74 const NodeDef& node, bool include_controlled_nodes) const;
75 std::unordered_set<OutputPort, HashPort> GetFanins(
76 const NodeDef& node, bool include_controlling_nodes) const;
77
78 // Get the number of ports in the immediate fanin of a node. Count the
79 // controlling nodes iff include_controlling_nodes is true.
80 int NumFanins(const NodeDef& node, bool include_controlling_nodes) const;
81
82 private:
83 GraphDef* graph_;
84 std::unordered_map<string, NodeDef*> nodes_;
85 std::unordered_set<InputPort, HashPort> empty_set_;
86 std::unordered_map<OutputPort, std::unordered_set<InputPort, HashPort>,
87 HashPort>
88 fanouts_;
89 std::unordered_map<const NodeDef*, int> num_regular_outputs_;
90};
