30GrapplerItem::GrapplerItem(const GrapplerItem& other, GraphDef&& graphDef) {
31 id = other.id;
32 feed = other.feed;
33 fetch = other.fetch;
34 init_ops = other.init_ops;
35 keep_ops = other.keep_ops;
36 expected_init_time = other.expected_init_time;
37 save_op = other.save_op;
38 restore_op = other.restore_op;
39 save_restore_loc_tensor = other.save_restore_loc_tensor;
40 queue_runners = other.queue_runners;
41 graph.Swap(&graphDef);
42}

44std::vector<const NodeDef*> GrapplerItem::MainOpsFanin() const {
45 return ComputeTransitiveFanin(graph, fetch);
46}

48std::vector<const NodeDef*> GrapplerItem::EnqueueOpsFanin() const {
49 std::vector<string> enqueue_ops;
50 for (const auto& queue_runner : queue_runners) {
51 for (const string& enqueue_op : queue_runner.enqueue_op_name()) {
52 enqueue_ops.push_back(enqueue_op);
53 }
54 }
55 return ComputeTransitiveFanin(graph, enqueue_ops);
56}
57
58std::vector<const NodeDef*> GrapplerItem::InitOpsFanin() const {
59 return ComputeTransitiveFanin(graph, init_ops);
60}

62std::vector<const NodeDef*> GrapplerItem::MainVariables() const {
63 std::vector<const NodeDef*> fanin = ComputeTransitiveFanin(graph, init_ops);
64 std::vector<const NodeDef*> vars;
65 for (const NodeDef* node : fanin) {
66 if (IsVariable(*node)) {
67 vars.push_back(node);
68 }
69 }
70 return vars;
71}

73std::unordered_set<string> GrapplerItem::NodesToPreserve() const {
74 std::unordered_set<string> result;
75 for (const string& f : fetch) {
76 VLOG(1) << "Add fetch " << f;
77 result.insert(NodeName(f));
78 }
79 for (const auto& f : feed) {
80 VLOG(1) << "Add feed " << f.first;
81 result.insert(NodeName(f.first));
82 }
83 for (const auto& node : init_ops) {
84 result.insert(NodeName(node));
85 }
86 for (const auto& node : keep_ops) {
87 result.insert(NodeName(node));
88 }
89 if (!save_op.empty()) {
90 result.insert(NodeName(save_op));
91 }
92 if (!restore_op.empty()) {
93 result.insert(NodeName(restore_op));
94 }
95 if (!save_restore_loc_tensor.empty()) {
96 result.insert(NodeName(save_restore_loc_tensor));
97 }
98
99 for (const auto& queue_runner : queue_runners) {
100 for (const string& enqueue_op : queue_runner.enqueue_op_name()) {
101 result.insert(NodeName(enqueue_op));
102 }
103 if (!queue_runner.close_op_name().empty()) {
104 result.insert(NodeName(queue_runner.close_op_name()));
105 }
106 if (!queue_runner.cancel_op_name().empty()) {
107 result.insert(NodeName(queue_runner.cancel_op_name()));
108 }
109 }
110 return result;
111}

113std::vector<const NodeDef*> ComputeTransitiveFanin(
114 const GraphDef& graph, const std::vector<string>& terminal_nodes) {
115 bool ill_formed = false;
116 std::vector<const NodeDef*> result =
117 ComputeTransitiveFanin(graph, terminal_nodes, &ill_formed);
118 CHECK(!ill_formed);
119 return result;
120}
121
122std::vector<const NodeDef*> ComputeTransitiveFanin(
123 const GraphDef& graph, const std::vector<string>& terminal_nodes,
124 bool* ill_formed) {
125 *ill_formed = false;
126 std::unordered_map<string, const NodeDef*> name_to_node;
127 std::unordered_map<string, const NodeDef*> name_to_send;
128 for (const auto& node : graph.node()) {
129 name_to_node[node.name()] = &node;
130 if (node.op() == "_Send") {
131 const auto& attr = node.attr();
132 name_to_send[attr.at("tensor_name").s()] = &node;
133 }
134 }
135
136 std::vector<const NodeDef*> queue;
137 for (const string& root : terminal_nodes) {
138 const NodeDef* node = name_to_node[NodeName(root)];
139 if (!node) {
140 *ill_formed = true;
141 VLOG(2) << "ComputeTransitiveFanin: problem with root node: " << root;
142 return {};
143 }
144 queue.push_back(node);
145 }
146
147 std::vector<const NodeDef*> result;
148 std::unordered_set<const NodeDef*> visited;
149
150 while (!queue.empty()) {
151 const NodeDef* node = queue.back();
152 queue.pop_back();
153 if (!visited.insert(node).second) {
154 // The node has already been visited.
155 continue;
156 }
157 result.push_back(node);
158 for (const string& input : node->input()) {
159 const NodeDef* in = name_to_node[NodeName(input)];
160 if (!in) {
161 VLOG(2) << "ComputeTransitiveFanin: problem with node: " << input;
162 *ill_formed = true;
163 return {};
164 }
165 queue.push_back(in);
166 }
167 if (node->op() == "_Recv") {
168 const auto& attr = node->attr();
169 const NodeDef* send = name_to_send[attr.at("tensor_name").s()];
170 if (send) {
171 queue.push_back(send);
172 }
173 // Subgraph after partitioning may have either _Send or _Recv, not both.
174 // So, we do not set ill_formed for missing _Send.
175 }
176 }
177 return result;
178}
