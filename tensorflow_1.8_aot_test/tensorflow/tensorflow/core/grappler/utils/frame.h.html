27using FrameMap = std::unordered_map<const NodeDef*, std::vector<int>>;

29// Returns the number of frames present in the graph, and populates
30// the 'frames' argument with the collection of frames (denoted by their
31// frame ids) in the outermost-to-innermost order. Frame ids are arbitrary.
32Status IdentifyFrames(const GraphDef& graph, FrameMap* frame_map,
33 int* num_frames);
34
35// As above, but use an existing NodeMap for graph instead of building it
36// from scratch.
37Status IdentifyFramesWithNodeMap(const GraphDef& graph, const NodeMap& node_map,
38 FrameMap* frame_map, int* num_frames);
