32void Cluster::SetNumWarmupSteps(int num_steps) {

41void Cluster::DisableDetailedStats(bool disable) {
42 if (disable) {
43 options_.config.mutable_graph_options()->set_build_cost_model(0);
44 run_options_.set_trace_level(RunOptions::NO_TRACE);
45 } else {
46 options_.config.mutable_graph_options()->set_build_cost_model(1);
47 run_options_.set_trace_level(RunOptions::HARDWARE_TRACE);
48 }
49}
50
51bool Cluster::DetailedStatsEnabled() const {
52 return options_.config.graph_options().build_cost_model() != 0;
53}
54
55void Cluster::DisableOptimizer(bool disable) {
56 OptimizerOptions* options =
57 options_.config.mutable_graph_options()->mutable_optimizer_options();
58 if (disable) {
59 options->set_opt_level(OptimizerOptions::L0);
60 // Disable Grappler optimizations.
61 auto rewriter_config =
62 options_.config.mutable_graph_options()->mutable_rewrite_options();
63 rewriter_config->set_layout_optimizer(RewriterConfig::OFF);
64 rewriter_config->set_disable_model_pruning(true);
65 rewriter_config->set_function_optimization(RewriterConfig::OFF);
66 rewriter_config->set_arithmetic_optimization(RewriterConfig::OFF);
67 rewriter_config->set_loop_optimization(RewriterConfig::OFF);
68 rewriter_config->set_dependency_optimization(RewriterConfig::OFF);
69 rewriter_config->set_constant_folding(RewriterConfig::OFF);
70 rewriter_config->set_memory_optimization(RewriterConfig::NO_MEM_OPT);
71 rewriter_config->mutable_auto_parallel()->set_enable(false);
72 rewriter_config->clear_optimizers();
73 } else {
74 options->set_opt_level(OptimizerOptions::L1);
75 auto rewriter_config =
76 options_.config.mutable_graph_options()->mutable_rewrite_options();
77 rewriter_config->set_constant_folding(RewriterConfig::DEFAULT);
78 rewriter_config->set_memory_optimization(RewriterConfig::DEFAULT_MEM_OPT);
79 }
80}

82const std::vector<string> Cluster::GetDeviceNames() const {
83 std::vector<string> device_names;
84 device_names.reserve(devices_.size());
85 for (const auto& device : devices_) {
86 device_names.push_back(device.first);
87 }
88 std::sort(device_names.begin(), device_names.end());
89 return device_names;
90}
