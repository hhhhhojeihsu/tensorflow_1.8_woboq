25namespace tensorflow {
26namespace grappler {
27
28// Create a simple cluster that lists the devices (and their properties)
29// available in a TensorFlow session. This cluster simulates the execution of
30// actual graphs.
31class VirtualCluster : public Cluster {
32 public:
33 VirtualCluster(const std::unordered_map<string, DeviceProperties>& devices);
34 VirtualCluster(const std::unordered_map<string, DeviceProperties>& devices,
35 OpLevelCostEstimator* node_estimator,
36 ReadyNodeManager* node_manager);
37
38 ~VirtualCluster() override;
39
40 string type() const override { return "virtual"; }
41
42 Status Provision() override;
43 Status Initialize(const GrapplerItem& item) override;
44 Status Run(const GraphDef& item,
45 const std::vector<std::pair<string, Tensor>>& feed,
46 const std::vector<string>& fetch, RunMetadata* metadata) override;
47
48 private:
49 std::unique_ptr<OpLevelCostEstimator> node_estimator_;
50 std::unique_ptr<ReadyNodeManager> node_manager_;
51};

53} // end namespace grappler
