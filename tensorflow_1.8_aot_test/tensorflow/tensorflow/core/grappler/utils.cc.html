33template <typename T>
34bool SafeSetScalarTensorValue(double value, Tensor* tensor) {
35 using RealType = typename Eigen::NumTraits<T>::Real;
36 if (value > static_cast<double>(std::numeric_limits<RealType>::max()) ||
37 value < static_cast<double>(std::numeric_limits<RealType>::min())) {
38 return false;
39 }
40 tensor->flat<T>()(0) = static_cast<T>(value);
41 return true;
42}

44// Is 'node' an operator that consumes only the shape of its input, not the
45// data itself?
46// TODO(ezhulenev): move to op_types.h. Requires to break circular dependency.
47// TODO(ezhulenev): what about Identity passing tensor to Shape consumer?
48bool IsShapeConsumer(const NodeDef& node) {
49 const string& op = node.op();
50 return op == "Shape" || op == "ShapeN" || op == "Rank" || op == "Size";
51}

55NodeMap::NodeMap(GraphDef* graph) {
56 CHECK(graph != nullptr);
57 for (int i = 0; i < graph->node_size(); i++) {
58 NodeDef* node = graph->mutable_node(i);
59 const string& node_name = node->name();
60 auto rslt = nodes_.emplace(node_name, node);
61 // Check that the graph doesn't contain multiple nodes with the same name.
62 if (!rslt.second) {
63 LOG(WARNING) << "Duplicated node in the graph: " << node_name;
64 }
65 for (const auto& input : node->input()) {
66 outputs_[NodeName(input)].insert(nodes_[node_name]);
67 }
68 }
69}

76NodeDef* NodeMap::GetNode(const string& name) const {
77 const string node_name = NodeName(name);
78 auto it = nodes_.find(node_name);

140bool IsSameInput(const string& name1, const string& name2) {
141 if (name1 == name2) {
142 return true;
143 }
144 int position1;
145 string node1 = ParseNodeName(name1, &position1);
146 int position2;
147 string node2 = ParseNodeName(name2, &position2);
148 return (position1 == position2) && (node1 == node2);
149}
150
151string ParseNodeName(const string& name, int* position) {
152 // Strip the prefix '^' (if any), and strip the trailing ":{digits} (if any)
153 // to get a node name.
154 strings::Scanner scan(name);
155 scan.ZeroOrOneLiteral("^")
156 .RestartCapture()
157 .One(strings::Scanner::LETTER_DIGIT_DOT_UNDERSCORE)
158 .Any(strings::Scanner::LETTER_DIGIT_DASH_DOT_SLASH_UNDERSCORE);
159 StringPiece capture;
160 StringPiece remaining;
161 if (scan.Peek(':') != ':' || !scan.GetResult(&remaining, &capture)) {
162 *position = 0;
163 return "";
164 } else {
165 if (name[0] == '^') {
166 *position = -1;
167 } else if (remaining.empty()) {
168 *position = 0;
169 } else {
170 // Skip the first ':' character.
171 CHECK(strings::safe_strto32(remaining.substr(1), position));
172 }
173 return capture.ToString();
174 }
175}

177bool IsControlInput(const string& name) {
178 return !name.empty() && name[0] == '^';
179}

206bool ExecuteWithTimeout(std::function<void()> fn, const int64 timeout_in_ms,
207 thread::ThreadPool* const thread_pool) {
208 if (timeout_in_ms <= 0) {
209 fn();
210 return true;
211 }
212 auto done = std::make_shared<Notification>();
213 thread_pool->Schedule([done, fn]() {
214 fn();
215 done->Notify();
216 });
217 const bool notified =
218 WaitForNotificationWithTimeout(done.get(), timeout_in_ms * 1000);
219 return notified;
220}

233int NumOutputs(const NodeDef& node, GraphDef* graph) {
234 int num_outputs = 0;
235 const OpDef* op_def = nullptr;
236 auto status = OpRegistry::Global()->LookUpOpDef(node.op(), &op_def);
237 if (status.ok()) {
238 for (const auto& output : op_def->output_arg()) {
239 if (!output.type_list_attr().empty()) {
240 num_outputs +=
241 node.attr().at(output.type_list_attr()).list().type_size();
242 } else if (!output.number_attr().empty()) {
243 num_outputs += node.attr().at(output.number_attr()).i();
244 } else {
245 num_outputs++;
246 }
247 }
248 } else {
249 FunctionLibraryDefinition fdef(OpRegistry::Global(), graph->library());
250 auto status = fdef.LookUpOpDef(node.op(), &op_def);
251 if (status.ok()) {
252 num_outputs = op_def->output_arg_size();
253 }
254 }
255 return num_outputs;
256}

258bool HasControlInputs(const NodeDef& node) {
259 int num_inputs = node.input_size();
260 if (num_inputs > 0 && IsControlInput(node.input(num_inputs - 1))) {
261 return true;

263 return false;

266int NumNonControlInputs(const NodeDef& node) {
267 int num_inputs = node.input_size();
268 for (const string& input : node.input()) {
269 if (IsControlInput(input)) {
270 --num_inputs;
271 }
272 }
273 return num_inputs;
274}

276int NumNonControlOutputs(const NodeDef& node, const NodeMap& node_map) {
277 int num_outputs = 0;
278 for (const NodeDef* output : node_map.GetOutputs(node.name())) {
279 for (const string& node_as_input : output->input()) {
280 if (IsControlInput(node_as_input)) {
281 break;
282 }
283 if (NodeName(node_as_input) == node.name()) {
284 ++num_outputs;
285 }
286 }
287 }
288 return num_outputs;
289}

291int NumNonControlDataOutputs(const NodeDef& node, const NodeMap& node_map) {
292 int num_data_outputs = 0;
293 for (const NodeDef* output : node_map.GetOutputs(node.name())) {
294 if (IsShapeConsumer(*output)) continue;
295
296 for (int i = 0; i < output->input_size(); ++i) {
297 const string& input = output->input(i);
298 if (!IsControlInput(input) && NodeName(input) == node.name()) {
299 ++num_data_outputs;
300 break;
301 }
302 }
303 }
304 return num_data_outputs;
305}

307// Returns the data type in attribute `attr_name` of `node`. If that attribute
308// doesn't exist, returns DT_INVALID.
309DataType GetDataTypeFromAttr(const NodeDef& node, const string& attr_name) {
310 if (!node.attr().count(attr_name)) {
311 return DT_INVALID;
312 }
313 const auto& attr = node.attr().at(attr_name);
314 if (attr.value_case() != AttrValue::kType) {
315 return DT_INVALID;
316 }
317 return attr.type();
318}

320NodeDef* GetTailOfChain(const NodeDef& source, const NodeMap& node_map,
321 bool follow_control_input,
322 const std::function<bool(const NodeDef&)>& pred_fn) {
323 const NodeDef* current = &source;
324 const NodeDef* next = current;
325 while (next == &source || (next != nullptr && pred_fn(*next))) {
326 current = next;
327 if (current->input_size() == 0 ||
328 (!follow_control_input && IsControlInput(current->input(0)))) {
329 break;
330 }
331 next = node_map.GetNode(current->input(0));
332 if (next == nullptr) {
333 LOG(ERROR) << "Node not found: " << current->input(0);
334 }
335 }
336 return const_cast<NodeDef*>(current);
337}

339// Every permutation is a product of one or more cycles. Iterate over the cycles
340// in the permutation, and convert each of those into a product of
341// transpositions (swaps): https://en.wikipedia.org/wiki/Cyclic_permutation
342void PermuteNodesInPlace(GraphDef* graph, std::vector<int>* permutation,
343 bool invert_permutation) {
344 CHECK_EQ(graph->node_size(), permutation->size());
345 std::vector<int> inv_perm(permutation->size(), 0);
346 if (invert_permutation) {
347 for (size_t n = 0; n < permutation->size(); ++n) {
348 inv_perm[(*permutation)[n]] = n;
349 }
350 permutation->swap(inv_perm);
351 }
352 for (std::size_t n = 0; n + 1 < permutation->size(); ++n) {
353 while (n != (*permutation)[n]) {
354 std::size_t r = (*permutation)[n];
355 graph->mutable_node()->SwapElements(n, r);
356 std::swap((*permutation)[n], (*permutation)[r]);
357 }
358 }
359}

361void DedupControlInputs(NodeDef* node) {
362 std::unordered_set<string> inputs;
363 int pos = 0;
364 while (pos < node->input_size()) {
365 const string& input = node->input(pos);
366 if (!inputs.insert(NodeName(input)).second && IsControlInput(input)) {
367 node->mutable_input()->SwapElements(pos, node->input_size() - 1);
368 node->mutable_input()->RemoveLast();
369 } else {
370 ++pos;
371 }
372 }
373}

376template <typename T>
377inline void STLSortAndRemoveDuplicates(T* v) {
378 std::sort(v->begin(), v->end());
379 v->erase(std::unique(v->begin(), v->end()), v->end());
380}

383Status SimpleGraphView::Initialize(const GraphDef& graph, bool dedup_inputs,
384 bool dedup_outputs) {
385 graph_ = &graph;
386 const int num_nodes = graph.node_size();
387 inputs_.clear();
388 inputs_.resize(num_nodes);
389 outputs_.clear();
390 outputs_.resize(num_nodes);
391 name_to_index_.clear();
392 name_to_index_.reserve(num_nodes);
393 index_to_name_.clear();
394 index_to_name_.reserve(num_nodes);
395
396 // Build map from name to index and vice versa.
397 for (int node_idx = 0; node_idx < num_nodes; ++node_idx) {
398 const NodeDef& node = graph.node(node_idx);
399 name_to_index_.emplace(node.name(), node_idx);
400 index_to_name_.push_back(node.name());
401 }
402
403 // Build forward and reverse adjacency lists.
404 for (int node_idx = 0; node_idx < num_nodes; ++node_idx) {
405 const NodeDef& node = graph.node(node_idx);
406 inputs_[node_idx].reserve(node.input_size());
407 for (const string& input : node.input()) {
408 auto it = name_to_index_.find(NodeName(input));
409 if (it == name_to_index_.end()) {
410 return errors::InvalidArgument("Non-existent input ", input,
411 " for node ", node.name());
412 }
413 const int input_idx = it->second;
414 inputs_[node_idx].push_back(input_idx);
415 outputs_[input_idx].push_back(node_idx);
416 }
417 if (dedup_inputs) {
418 // Dedup the input list while it's still hot in cache.
419 STLSortAndRemoveDuplicates(&inputs_[node_idx]);
420 }
421 }
422
423 // Dedup outputs.
424 if (dedup_outputs) {
425 for (int node_idx = 0; node_idx < num_nodes; ++node_idx) {
426 STLSortAndRemoveDuplicates(&outputs_[node_idx]);
427 }
428 }
429 return Status::OK();
430}

432void SimpleGraphView::DepthFirstSearch(
433 const std::unordered_set<string>& op_types_to_traverse, int root_node,
434 std::set<int>* nodes_found) const {
435 nodes_found->clear();
436 const string& op_type = graph_->node(root_node).op();
437 if (op_types_to_traverse.find(op_type) == op_types_to_traverse.end()) {
438 return;
439 }
440 std::vector<int> stack;
441 stack.reserve(32);
442 stack.push_back(root_node);
443 while (!stack.empty()) {
444 const int node_idx = stack.back();
445 stack.pop_back();
446 nodes_found->insert(node_idx);
447 const string& op_type = graph_->node(node_idx).op();
448 if (op_types_to_traverse.find(op_type) != op_types_to_traverse.end()) {
449 for (auto output_idx : this->outputs(node_idx)) {
450 if (nodes_found->find(output_idx) == nodes_found->end()) {
451 stack.push_back(output_idx);
452 }
453 }
454 }
455 }
456}

458string SimpleGraphView::PrintToString() const {
459 string str;
460 for (int i = 0; i < num_nodes(); ++i) {
461 strings::StrAppend(&str, "Node ", i, "'", node_name(i), "'\n", "Inputs: [");
462 for (int input : inputs(i)) {
463 strings::StrAppend(&str, input, " '", node_name(input), "', ");
464 }
465 strings::StrAppend(&str, "]\n", "Outputs: [");
466 for (int j = 0; j < outputs(i).size(); ++j) {
467 const int output = outputs(i)[j];
468 if (j > 0) {
469 strings::StrAppend(&str, ", ");
470 }
471 strings::StrAppend(&str, output, " '", node_name(output), "'");
472 }
473 strings::StrAppend(&str, "]\n");
474 }
475 return str;
476}

478#define HANDLE_CASE(DTYPE) \
479 case DTYPE: \
480 if (!SafeSetScalarTensorValue<EnumToDataType<DTYPE>::Type>( \
481 static_cast<double>(value), tensor)) { \
482 return errors::InvalidArgument("Cannot store value ", value, \
483 " in tensor of type " #DTYPE); \
484 } \
485 break
486
487Status SetTensorValue(DataType dtype, int value, Tensor* tensor) {
488 // TODO(rmlarsen): Support more general shapes.
489 if (tensor->NumElements() != 1) {
490 return errors::InvalidArgument(
491 "Expected scalar tensor, got num_elements = ", tensor->NumElements());
492 }
493 switch (dtype) {
494 HANDLE_CASE(DT_HALF);
495 HANDLE_CASE(DT_BFLOAT16);
496 HANDLE_CASE(DT_BOOL);
497 HANDLE_CASE(DT_FLOAT);
498 HANDLE_CASE(DT_DOUBLE);
499 HANDLE_CASE(DT_UINT8);
500 HANDLE_CASE(DT_INT8);
501 HANDLE_CASE(DT_UINT16);
502 HANDLE_CASE(DT_INT16);
503 HANDLE_CASE(DT_INT32);
504 HANDLE_CASE(DT_INT64);
505 HANDLE_CASE(DT_COMPLEX64);
506 HANDLE_CASE(DT_COMPLEX128);
507 default:
508 return errors::InvalidArgument("Unsupported type ",
509 DataTypeString(dtype));
510 }
511 return Status::OK();
512}
