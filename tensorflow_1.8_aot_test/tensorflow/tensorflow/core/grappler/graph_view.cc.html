22GraphView::GraphView(GraphDef* graph) : graph_(graph) {
23 for (int i = 0; i < graph_->node_size(); i++) {
24 auto node = graph_->mutable_node(i);
25 auto rslt = nodes_.insert(std::make_pair(node->name(), node));
26 // Check that the graph doesn't contain multiple nodes with the same name.
27 CHECK(rslt.second) << "Non unique node name detected: " << node->name();
28 }
29 for (NodeDef& node : *graph_->mutable_node()) {
30 for (int i = 0; i < node.input_size(); ++i) {
31 OutputPort fanin;
32 string fanin_name = ParseNodeName(node.input(i), &fanin.port_id);
33 fanin.node = nodes_[fanin_name];
34
35 InputPort input;
36 input.node = &node;
37 if (fanin.port_id < 0) {
38 input.port_id = -1;
39 } else {
40 input.port_id = i;
41 num_regular_outputs_[fanin.node] =
42 std::max(num_regular_outputs_[fanin.node], fanin.port_id);
43 }
44
45 fanouts_[fanin].insert(input);
46 }
47 }
48}

50NodeDef* GraphView::GetNode(const string& node_name) const {
51 auto it = nodes_.find(node_name);
52 if (it == nodes_.end()) {
53 return nullptr;
54 }
55 return it->second;
56}

76const std::unordered_set<GraphView::InputPort, GraphView::HashPort>&
77GraphView::GetFanout(const GraphView::OutputPort& port) const {
78 auto it = fanouts_.find(port);
79 if (it == fanouts_.end()) {
80 return empty_set_;
81 }
82 return it->second;
83}

85std::unordered_set<GraphView::OutputPort, GraphView::HashPort>
86GraphView::GetFanin(const GraphView::InputPort& port) const {
87 std::unordered_set<GraphView::OutputPort, GraphView::HashPort> result;
88 if (port.port_id >= 0) {
89 result.insert(GetRegularFanin(port));
90 } else {
91 for (int i = port.node->input_size() - 1; i >= 0; --i) {
92 OutputPort fanin;
93 string fanin_name = ParseNodeName(port.node->input(i), &fanin.port_id);
94 if (fanin.port_id < 0) {
95 auto it = nodes_.find(fanin_name);
96 if (it != nodes_.end()) {
97 fanin.node = it->second;
98 result.insert(fanin);
99 }
100 } else {
101 break;
102 }
103 }
104 }
105 return result;
106}

108const GraphView::OutputPort GraphView::GetRegularFanin(
109 const GraphView::InputPort& port) const {
110 CHECK_LE(0, port.port_id);
111 OutputPort fanin;
112 string fanin_name =
113 ParseNodeName(port.node->input(port.port_id), &fanin.port_id);
114 auto it = nodes_.find(fanin_name);
115 if (it == nodes_.end()) {
116 fanin.node = nullptr;
117 } else {
118 fanin.node = it->second;
119 }
120 return fanin;
121}

123std::unordered_set<GraphView::InputPort, GraphView::HashPort>
124GraphView::GetFanouts(const NodeDef& node,
125 bool include_controlled_nodes) const {
126 std::unordered_set<InputPort, HashPort> result;
127 OutputPort port;
128 port.node = const_cast<NodeDef*>(&node);
129 const int first_port_id = include_controlled_nodes ? -1 : 0;
130 auto it = num_regular_outputs_.find(&node);
131 const int last_port_id = (it != num_regular_outputs_.end()) ? it->second : -1;
132
133 for (int i = first_port_id; i <= last_port_id; ++i) {
134 port.port_id = i;
135 auto it = fanouts_.find(port);
136 if (it != fanouts_.end()) {
137 result.insert(it->second.begin(), it->second.end());
138 }
139 }
140 return result;
141}

143std::unordered_set<GraphView::OutputPort, GraphView::HashPort>
144GraphView::GetFanins(const NodeDef& node,
145 bool include_controlling_nodes) const {
146 std::unordered_set<OutputPort, HashPort> result;
147 for (int i = 0; i < node.input_size(); ++i) {
148 OutputPort fanin;
149 string fanin_name = ParseNodeName(node.input(i), &fanin.port_id);
150 if (fanin.port_id < 0) {
151 if (!include_controlling_nodes) {
152 break;
153 }
154 }
155 auto it = nodes_.find(fanin_name);
156 if (it != nodes_.end()) {
157 fanin.node = it->second;
158 result.insert(fanin);
159 }
160 }
161 return result;
162}
163
164int GraphView::NumFanins(const NodeDef& node,
165 bool include_controlling_nodes) const {
166 int count = 0;
167 for (const string& input : node.input()) {
168 if (!include_controlling_nodes && IsControlInput(input)) {
169 break;
170 }
171 count += 1;
172 }
173 return count;
174}
