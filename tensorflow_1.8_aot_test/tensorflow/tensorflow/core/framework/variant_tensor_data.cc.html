25VariantTensorData::VariantTensorData(const VariantTensorDataProto& proto) {
26 FromProto(proto);

29VariantTensorData::~VariantTensorData() {}

31int VariantTensorData::tensors_size() const { return tensors_.size(); }
32
33const Tensor& VariantTensorData::tensors(int index) const {
34 return tensors_[index];
35}
36
37const std::vector<Tensor>& VariantTensorData::tensors() const {
38 return tensors_;
39}
40
41Tensor* VariantTensorData::add_tensors() {
42 tensors_.emplace_back();
43 return &(tensors_[tensors_.size() - 1]);
44}
45
46void VariantTensorData::ToProto(VariantTensorDataProto* proto) const {
47 proto->set_type_name(type_name());
48 proto->set_metadata(metadata_);
49 proto->clear_tensors();
50 for (const auto& tensor : tensors_) {
51 tensor.AsProtoField(proto->mutable_tensors()->Add());
52 }
53}
54
55bool VariantTensorData::FromProto(const VariantTensorDataProto& proto) {
56 set_type_name(proto.type_name());
57 set_metadata(proto.metadata());
58 for (const auto& tensor : proto.tensors()) {
59 Tensor tmp;
60 if (!tmp.FromProto(tensor)) return false;
61 tensors_.push_back(tmp);
62 }
63 return true;
64}
65
66string VariantTensorData::SerializeAsString() const {
67 VariantTensorDataProto proto;
68 ToProto(&proto);
69 return proto.SerializeAsString();
70}
71
72bool VariantTensorData::SerializeToString(string* buf) {
73 VariantTensorDataProto proto;
74 ToProto(&proto);
75 return proto.SerializeToString(buf);
76}
77
78bool VariantTensorData::ParseFromString(const string& s) {
79 VariantTensorDataProto proto;
80 const bool status = proto.ParseFromString(s);
81 if (status) FromProto(proto);
82 return status;
83}

85string VariantTensorData::DebugString() const {
86 string repeated_field = "";
87 for (const auto& t : tensors_) {
88 repeated_field =
89 strings::StrCat(repeated_field, " tensors: ", t.DebugString());
90 }
91 return strings::StrCat("type_name: ", type_name(), " metadata: ", metadata_,
92 repeated_field);
93}
