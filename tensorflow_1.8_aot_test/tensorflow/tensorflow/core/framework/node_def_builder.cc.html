39NodeDefBuilder::NodeDefBuilder(StringPiece name, StringPiece op_name,
40 const OpRegistryInterface* op_registry) {
41 node_def_.set_name(name.ToString());
42 const Status status = op_registry->LookUpOpDef(op_name.ToString(), &op_def_);
43 if (status.ok()) {
44 Initialize();
45 } else {
46 errors_.push_back(status.error_message());
47 inputs_specified_ = 0;
48 }
49}

58 inputs_specified_ = 0;

62const OpDef::ArgDef* NodeDefBuilder::NextArgDef() {
63 if (!NextArgAvailable()) return nullptr;
64 return &op_def_->input_arg(inputs_specified_++);
65}
66
67bool NodeDefBuilder::NextArgAvailable() {
68 if (op_def_ == nullptr) {
69 return false;
70 } else if (inputs_specified_ >= op_def_->input_arg_size()) {
71 errors_.push_back(strings::StrCat("More Input() calls than the ",
72 op_def_->input_arg_size(),
73 " input_args"));
74 return false;
75 }
76 return true;
77}

79NodeDefBuilder& NodeDefBuilder::Input(FakeInputFunctor fake_input) {
80 if (NextArgAvailable()) {
81 Status status = fake_input(*op_def_, inputs_specified_, node_def_, this);
82 if (!status.ok()) errors_.push_back(status.error_message());
83 }
84 return *this;
85}

106void NodeDefBuilder::SingleInput(const OpDef::ArgDef* input_arg,
107 StringPiece src_node, int src_index,
108 DataType dt) {
109 AddInput(src_node, src_index);
110
111 if (!input_arg->number_attr().empty() ||
112 !input_arg->type_list_attr().empty()) {
113 errors_.push_back(strings::StrCat("Single tensor passed to '",
114 input_arg->name(), "', expected list"));
115 return;
116 }
117
118 if (input_arg->type() != DT_INVALID) {
119 const DataType expected = MaybeAddRef(input_arg, input_arg->type());
120 VerifyInputType(input_arg, expected, dt);
121 } else {
122 VerifyInputRef(input_arg, dt);
123 Attr(input_arg->type_attr(), BaseType(dt));
124 }
125}
126
127void NodeDefBuilder::ListInput(const OpDef::ArgDef* input_arg,
128 gtl::ArraySlice<NodeOut> src_list) {
129 for (const auto& node_out : src_list) {
130 AddInput(node_out.node, node_out.index);
131 }
132
133 if (!input_arg->number_attr().empty()) {
134 Attr(input_arg->number_attr(), static_cast<int64>(src_list.size()));
135 if (input_arg->type() != DT_INVALID) {
136 const DataType expected = MaybeAddRef(input_arg, input_arg->type());
137 for (const auto& node_out : src_list) {
138 VerifyInputType(input_arg, expected, node_out.data_type);
139 }
140 } else if (!src_list.empty()) {
141 const DataType base = BaseType(src_list[0].data_type);
142 Attr(input_arg->type_attr(), base);
143 const DataType expected = MaybeAddRef(input_arg, base);
144 for (const auto& node_out : src_list) {
145 VerifyInputType(input_arg, expected, node_out.data_type);
146 }
147 }
148 } else if (!input_arg->type_list_attr().empty()) {
149 DataTypeVector type_vec;
150 type_vec.reserve(src_list.size());
151 for (const auto& node_out : src_list) {
152 const DataType dt = node_out.data_type;
153 VerifyInputRef(input_arg, dt);
154 type_vec.push_back(BaseType(dt));
155 }
156 Attr(input_arg->type_list_attr(), type_vec);
157 } else {
158 errors_.push_back(strings::StrCat("List provided to input '",
159 input_arg->name(),
160 "' when single Tensor expected"));
161 }
162}

164void NodeDefBuilder::AddInput(StringPiece src_node, int src_index) {
165 if (src_node.empty()) {
166 errors_.push_back("Empty input node name");
167 } else if (src_node[0] == '^') {
168 errors_.push_back(
169 strings::StrCat("Non-control input starting with ^: ", src_node));
170 } else if (src_index > 0) {
171 node_def_.add_input(strings::StrCat(src_node, ":", src_index));
172 } else {
173 node_def_.add_input(src_node.ToString());
174 }
175}

177void NodeDefBuilder::VerifyInputType(const OpDef::ArgDef* input_arg,
178 DataType expected, DataType dt) {
179 if (!TypesCompatible(expected, dt)) {
180 errors_.push_back(strings::StrCat("Input '", input_arg->name(), "' passed ",
181 DataTypeString(dt), " expected ",
182 DataTypeString(expected)));
183 }
184}
185
186void NodeDefBuilder::VerifyInputRef(const OpDef::ArgDef* input_arg,
187 DataType dt) {
188 if (input_arg->is_ref() && !IsRefType(dt)) {
189 errors_.push_back(strings::StrCat("Input '", input_arg->name(), "' passed ",
190 DataTypeString(dt),
191 " expected ref type"));
192 }
193}

205Status NodeDefBuilder::Finalize(NodeDef* node_def) const {
206 const std::vector<string>* errors_ptr = &errors_;
207 std::vector<string> errors_storage;
208 if (op_def_ != nullptr && inputs_specified_ < op_def_->input_arg_size()) {
209 // Since this is a const method, to add an error, we have to make
210 // a copy of the existing errors.
211 errors_storage = errors_;
212 errors_storage.push_back(
213 strings::StrCat(inputs_specified_, " inputs specified of ",
214 op_def_->input_arg_size(), " inputs in Op"));
215 errors_ptr = &errors_storage;
216 }
217
218 if (!errors_ptr->empty()) {
219 if (errors_ptr->size() == 1) {
220 if (op_def_ == nullptr) {
221 return errors::InvalidArgument((*errors_ptr)[0],
222 " while building NodeDef '",
223 node_def_.name(), "'");
224 }
225 return errors::InvalidArgument(
226 (*errors_ptr)[0], " while building NodeDef '", node_def_.name(),
227 "' using ", SummarizeOpDef(*op_def_));
228 } else {
229 return errors::InvalidArgument(
230 errors_ptr->size(), " errors while building NodeDef '",
231 node_def_.name(), "' using ", SummarizeOpDef(*op_def_), ":\n",
232 str_util::Join(*errors_ptr, "\n"));
233 }
234 } else {
235 NodeDef node_def_backup;
236 if (node_def == nullptr) node_def = &node_def_backup;
237 *node_def = node_def_;
238
239 // Add control inputs after the regular inputs.
240 for (const auto& control_input : control_inputs_) {
241 node_def->add_input(strings::StrCat("^", control_input));
242 }
243
244 // Add default values for unspecified attrs.
245 AddDefaultsToNodeDef(*op_def_, node_def);
246
247 return Status::OK();
248 }
249}

251NodeDefBuilder& NodeDefBuilder::Attr(StringPiece name, const AttrValue& value) {
252 if (const AttrValue* found = AttrSlice(node_def_).Find(name)) {
253 if (!AreAttrValuesEqual(*found, value)) {
254 errors_.push_back(strings::StrCat("Inconsistent values for attr '", name,
255 "' ", SummarizeAttrValue(*found),
256 " vs. ", SummarizeAttrValue(value)));
257 }
258 } else {
259 AddNodeAttr(name, value, &node_def_);
260 }
261 return *this;
262}
