25bool Variant::TryDecode(Variant* out) const {
26 const VariantTensorDataProto* p = get<VariantTensorDataProto>();
27 if (p == nullptr) return false;
28 VariantTensorData data(*p);
29 return out->Decode(data);
30}
31
32template <>
33void* Variant::get() {
34 if (is_empty()) {
35 return nullptr;
36 }
37 return value_->RawPtr();
38}
39
40template <>
41const void* Variant::get() const {
42 if (is_empty()) {
43 return nullptr;
44 }
45 return value_->RawPtr();
46}

48template <>
49string TypeNameVariant(const VariantTensorDataProto& value) {
50 return value.type_name();
51}
52
53template <>
54void EncodeVariant(const VariantTensorDataProto& value,
55 VariantTensorData* data) {
56 data->FromProto(value);
57}
58
59template <>
60bool DecodeVariant(const VariantTensorData& data,
61 VariantTensorDataProto* value) {
62 data.ToProto(value);
63 return true;
64}
65
66template <>
67void EncodeVariant(const VariantTensorDataProto& value, string* buf) {
68 value.SerializeToString(buf);
69}
70
71template <>
72bool DecodeVariant(const string& buf, VariantTensorDataProto* value) {
73 return value->ParseFromString(buf);
74}
