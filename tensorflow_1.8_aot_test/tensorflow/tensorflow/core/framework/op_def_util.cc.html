40bool HasAttrStyleType(const OpDef::ArgDef& arg) {
41 return arg.type() != DT_INVALID || !arg.type_attr().empty() ||
42 !arg.type_list_attr().empty();
43}

45Status AllowedTypeValue(DataType dt, const OpDef::AttrDef& attr) {
46 const AttrValue& allowed_values(attr.allowed_values());
47 for (auto allowed : allowed_values.list().type()) {
48 if (dt == allowed) {
49 return Status::OK();
50 }
51 }
52 string allowed_str;
53 for (int i = 0; i < allowed_values.list().type_size(); ++i) {
54 if (!allowed_str.empty()) {
55 strings::StrAppend(&allowed_str, ", ");
56 }
57 strings::StrAppend(&allowed_str,
58 DataTypeString(allowed_values.list().type(i)));
59 }
60 return errors::InvalidArgument(
61 "Value for attr '", attr.name(), "' of ", DataTypeString(dt),
62 " is not in the list of allowed values: ", allowed_str);
63}
64
65Status AllowedStringValue(const string& str, const OpDef::AttrDef& attr) {
66 const AttrValue& allowed_values(attr.allowed_values());
67 for (const auto& allowed : allowed_values.list().s()) {
68 if (str == allowed) {
69 return Status::OK();
70 }
71 }
72 string allowed_str;
73 for (const string& allowed : allowed_values.list().s()) {
74 if (!allowed_str.empty()) {
75 strings::StrAppend(&allowed_str, ", ");
76 }
77 strings::StrAppend(&allowed_str, "\"", allowed, "\"");
78 }
79 return errors::InvalidArgument(
80 "Value for attr '", attr.name(), "' of \"", str,
81 "\" is not in the list of allowed values: ", allowed_str);
82}

86// Requires: attr has already been validated.
87Status ValidateAttrValue(const AttrValue& attr_value,
88 const OpDef::AttrDef& attr) {
89 // Is it a valid value?
90 TF_RETURN_WITH_CONTEXT_IF_ERROR(AttrValueHasType(attr_value, attr.type()),
91 " for attr '", attr.name(), "'");
92
93 // Does the value satisfy the minimum constraint in the AttrDef?
94 if (attr.has_minimum()) {
95 if (attr.type() == "int") {
96 if (attr_value.i() < attr.minimum()) {
97 return errors::InvalidArgument(
98 "Value for attr '", attr.name(), "' of ", attr_value.i(),
99 " must be at least minimum ", attr.minimum());
100 }
101 } else {
102 int length = -1;
103 if (attr.type() == "list(string)") {
104 length = attr_value.list().s_size();
105 } else if (attr.type() == "list(int)") {
106 length = attr_value.list().i_size();
107 } else if (attr.type() == "list(float)") {
108 length = attr_value.list().f_size();
109 } else if (attr.type() == "list(bool)") {
110 length = attr_value.list().b_size();
111 } else if (attr.type() == "list(type)") {
112 length = attr_value.list().type_size();
113 } else if (attr.type() == "list(shape)") {
114 length = attr_value.list().shape_size();
115 } else if (attr.type() == "list(tensor)") {
116 length = attr_value.list().tensor_size();
117 }
118 if (length < attr.minimum()) {
119 return errors::InvalidArgument(
120 "Length for attr '", attr.name(), "' of ", length,
121 " must be at least minimum ", attr.minimum());
122 }
123 }
124 }
125
126 // Does the value satisfy the allowed_value constraint in the AttrDef?
127 if (attr.has_allowed_values()) {
128 if (attr.type() == "type") {
129 TF_RETURN_IF_ERROR(AllowedTypeValue(attr_value.type(), attr));
130 } else if (attr.type() == "list(type)") {
131 for (int dt : attr_value.list().type()) {
132 TF_RETURN_IF_ERROR(AllowedTypeValue(static_cast<DataType>(dt), attr));
133 }
134 } else if (attr.type() == "string") {
135 TF_RETURN_IF_ERROR(AllowedStringValue(attr_value.s(), attr));
136 } else if (attr.type() == "list(string)") {
137 for (const string& str : attr_value.list().s()) {
138 TF_RETURN_IF_ERROR(AllowedStringValue(str, attr));
139 }
140 } else {
141 return errors::Unimplemented(
142 "Support for allowed_values not implemented for type ", attr.type());
143 }
144 }
145 return Status::OK();
146}

148const OpDef::AttrDef* FindAttr(StringPiece name, const OpDef& op_def) {
149 for (int i = 0; i < op_def.attr_size(); ++i) {
150 if (op_def.attr(i).name() == name) {
151 return &op_def.attr(i);
152 }
153 }
154 return nullptr;
155}

160 return op_def->mutable_attr(i);

162 }
163 return nullptr;

176 do { \
177 if (!(EXPR)) { \
178 return errors::InvalidArgument( \

183static Status ValidateArg(const OpDef::ArgDef& arg, const OpDef& op_def,
184 bool output, std::set<string>* names) {
185 const string suffix = strings::StrCat(
186 output ? " for output '" : " for input '", arg.name(), "'");
187 VALIDATE(gtl::InsertIfNotPresent(names, arg.name()),
188 "Duplicate name: ", arg.name());
189 VALIDATE(HasAttrStyleType(arg), "Missing type", suffix);
190
191 if (!arg.number_attr().empty()) {
192 const OpDef::AttrDef* attr = FindAttr(arg.number_attr(), op_def);
193 VALIDATE(attr != nullptr, "No attr with name '", arg.number_attr(), "'",
194 suffix);
195 VALIDATE(attr->type() == "int", "Attr '", attr->name(), "' used as length",
196 suffix, " has type ", attr->type(), " != int");
197 VALIDATE(attr->has_minimum(), "Attr '", attr->name(), "' used as length",
198 suffix, " must have minimum");
199 VALIDATE(attr->minimum() >= 0, "Attr '", attr->name(), "' used as length",
200 suffix, " must have minimum >= 0");
201 VALIDATE(arg.type_list_attr().empty(),
202 "Can't have both number_attr and type_list_attr", suffix);
203 VALIDATE((arg.type() != DT_INVALID ? 1 : 0) +
204 (!arg.type_attr().empty() ? 1 : 0) ==
205 1,
206 "Exactly one of type, type_attr must be set", suffix);
207 } else {
208 const int num_type_fields = (arg.type() != DT_INVALID ? 1 : 0) +
209 (!arg.type_attr().empty() ? 1 : 0) +
210 (!arg.type_list_attr().empty() ? 1 : 0);
211 VALIDATE(num_type_fields == 1,
212 "Exactly one of type, type_attr, type_list_attr must be set",
213 suffix);
214 }
215
216 if (!arg.type_attr().empty()) {
217 const OpDef::AttrDef* attr = FindAttr(arg.type_attr(), op_def);
218 VALIDATE(attr != nullptr, "No attr with name '", arg.type_attr(), "'",
219 suffix);
220 VALIDATE(attr->type() == "type", "Attr '", attr->name(),
221 "' used as type_attr", suffix, " has type ", attr->type(),
222 " != type");
223 } else if (!arg.type_list_attr().empty()) {
224 const OpDef::AttrDef* attr = FindAttr(arg.type_list_attr(), op_def);
225 VALIDATE(attr != nullptr, "No attr with name '", arg.type_list_attr(), "'",
226 suffix);
227 VALIDATE(attr->type() == "list(type)", "Attr '", attr->name(),
228 "' used as type_list_attr", suffix, " has type ", attr->type(),
229 " != list(type)");
230 } else {
231 // All argument types should be non-reference types at this point.
232 // ArgDef.is_ref is set to true for reference arguments.
233 VALIDATE(!IsRefType(arg.type()), "Illegal use of ref type '",
234 DataTypeString(arg.type()), "'. Use 'Ref(type)' instead", suffix);
235 }
236
237 return Status::OK();
238}

240Status ValidateOpDef(const OpDef& op_def) {
241 using ::tensorflow::strings::Scanner;
242
243 if (!str_util::StartsWith(op_def.name(), "_")) {
244 VALIDATE(Scanner(op_def.name())
245 .One(Scanner::UPPERLETTER)
246 .Any(Scanner::LETTER_DIGIT)
247 .Eos()
248 .GetResult(),
249 "Invalid name: ", op_def.name(), " (Did you use CamelCase?)");
250 }
251
252 std::set<string> names; // for detecting duplicate names
253 for (const auto& attr : op_def.attr()) {
254 // Validate name
255 VALIDATE(gtl::InsertIfNotPresent(&names, attr.name()),
256 "Duplicate name: ", attr.name());
257 DataType dt;
258 VALIDATE(!DataTypeFromString(attr.name(), &dt), "Attr can't have name ",
259 attr.name(), " that matches a data type");
260
261 // Validate type
262 StringPiece type(attr.type());
263 bool is_list = str_util::ConsumePrefix(&type, "list(");
264 bool found = false;
265 for (StringPiece valid : {"string", "int", "float", "bool", "type", "shape",
266 "tensor", "func"}) {
267 if (str_util::ConsumePrefix(&type, valid)) {
268 found = true;
269 break;
270 }
271 }
272 VALIDATE(found, "Unrecognized type '", type, "' in attr '", attr.name(),
273 "'");
274 if (is_list) {
275 VALIDATE(str_util::ConsumePrefix(&type, ")"),
276 "'list(' is missing ')' in attr ", attr.name(), "'s type ",
277 attr.type());
278 }
279 VALIDATE(type.empty(), "Extra '", type, "' at the end of attr ",
280 attr.name(), "'s type ", attr.type());
281
282 // Validate minimum
283 if (attr.has_minimum()) {
284 VALIDATE(attr.type() == "int" || is_list, "Attr '", attr.name(),
285 "' has minimum for unsupported type ", attr.type());
286 if (is_list) {
287 VALIDATE(attr.minimum() >= 0, "Attr '", attr.name(),
288 "' with list type must have a non-negative minimum, not ",
289 attr.minimum());
290 }
291 } else {
292 VALIDATE(attr.minimum() == 0, "Attr '", attr.name(),
293 "' with has_minimum = false but minimum ", attr.minimum(),
294 " not equal to default of 0");
295 }
296
297 // Validate allowed_values
298 if (attr.has_allowed_values()) {
299 const string list_type =
300 is_list ? attr.type() : strings::StrCat("list(", attr.type(), ")");
301 TF_RETURN_WITH_CONTEXT_IF_ERROR(
302 AttrValueHasType(attr.allowed_values(), list_type), " for attr '",
303 attr.name(), "' in Op '", op_def.name(), "'");
304 }
305
306 // Validate default_value (after we have validated the rest of the attr,
307 // so we can use ValidateAttrValue()).
308 if (attr.has_default_value()) {
309 TF_RETURN_WITH_CONTEXT_IF_ERROR(
310 ValidateAttrValue(attr.default_value(), attr), " in Op '",
311 op_def.name(), "'");
312 }
313 }
314
315 for (const auto& arg : op_def.input_arg()) {
316 TF_RETURN_IF_ERROR(ValidateArg(arg, op_def, false, &names));
317 }
318
319 for (const auto& arg : op_def.output_arg()) {
320 TF_RETURN_IF_ERROR(ValidateArg(arg, op_def, true, &names));
321 }
322
323 return Status::OK();
324}

328Status CheckOpDeprecation(const OpDef& op_def, int graph_def_version) {
329 if (op_def.has_deprecation()) {
330 const OpDeprecation& dep = op_def.deprecation();
331 if (graph_def_version >= dep.version()) {
332 return errors::Unimplemented(
333 "Op ", op_def.name(), " is not available in GraphDef version ",
334 graph_def_version, ". It has been removed in version ", dep.version(),
335 ". ", dep.explanation(), ".");
336 } else {
337 // Warn only once for each op name, and do it in a threadsafe manner.
338 static mutex mu(LINKER_INITIALIZED);
339 static std::unordered_set<string> warned;
340 bool warn;
341 {
342 mutex_lock lock(mu);
343 warn = warned.insert(op_def.name()).second;
344 }
345 if (warn) {
346 LOG(WARNING) << "Op " << op_def.name() << " is deprecated."
347 << " It will cease to work in GraphDef version "
348 << dep.version() << ". " << dep.explanation() << ".";
349 }
350 }
351 }
352 return Status::OK();
353}

357string SummarizeArgs(const protobuf::RepeatedPtrField<OpDef::ArgDef>& args) {
358 string ret;
359 for (const OpDef::ArgDef& arg : args) {
360 if (!ret.empty()) strings::StrAppend(&ret, ", ");
361 strings::StrAppend(&ret, arg.name(), ":");
362 if (arg.is_ref()) strings::StrAppend(&ret, "Ref(");
363 if (!arg.number_attr().empty()) {
364 strings::StrAppend(&ret, arg.number_attr(), "*");
365 }
366 if (arg.type() != DT_INVALID) {
367 strings::StrAppend(&ret, DataTypeString(arg.type()));
368 } else {
369 strings::StrAppend(&ret, arg.type_attr());
370 }
371 if (arg.is_ref()) strings::StrAppend(&ret, ")");
372 }
373 return ret;
374}

378string SummarizeOpDef(const OpDef& op_def) {
379 string ret = strings::StrCat("Op<name=", op_def.name());
380 strings::StrAppend(&ret, "; signature=", SummarizeArgs(op_def.input_arg()),
381 " -> ", SummarizeArgs(op_def.output_arg()));
382 for (int i = 0; i < op_def.attr_size(); ++i) {
383 strings::StrAppend(&ret, "; attr=", op_def.attr(i).name(), ":",
384 op_def.attr(i).type());
385 if (op_def.attr(i).has_default_value()) {
386 strings::StrAppend(&ret, ",default=",
387 SummarizeAttrValue(op_def.attr(i).default_value()));
388 }
389 if (op_def.attr(i).has_minimum()) {
390 strings::StrAppend(&ret, ",min=", op_def.attr(i).minimum());
391 }
392 if (op_def.attr(i).has_allowed_values()) {
393 strings::StrAppend(&ret, ",allowed=",
394 SummarizeAttrValue(op_def.attr(i).allowed_values()));
395 }
396 }
397 if (op_def.is_commutative()) {
398 strings::StrAppend(&ret, "; is_commutative=true");
399 }
400 if (op_def.is_aggregate()) {
401 strings::StrAppend(&ret, "; is_aggregate=true");
402 }
403 if (op_def.is_stateful()) {
404 strings::StrAppend(&ret, "; is_stateful=true");
405 }
406 if (op_def.allows_uninitialized_input()) {
407 strings::StrAppend(&ret, "; allows_uninitialized_input=true");
408 }
409 strings::StrAppend(&ret, ">");
410 return ret;
411}

415// Returns true if every element of `sub` is contained in `super`.
416template <class T>
417bool IsSubsetOf(const T& sub, const T& super) {
418 for (const auto& o : sub) {
419 bool found = false;
420 for (const auto& n : super) {
421 if (o == n) {
422 found = true;
423 break;
424 }
425 }
426 if (!found) return false;
427 }
428 return true;
429}

431bool MoreRestrictive(const OpDef::AttrDef& old_attr,
432 const OpDef::AttrDef& new_attr) {
433 // Anything -> no restriction : not more restrictive.
434 if (!new_attr.has_allowed_values()) return false;
435 // No restriction -> restriction : more restrictive.
436 if (!old_attr.has_allowed_values()) return true;
437 // If anything that was previously allowed is no longer allowed:
438 // more restrictive.
439 if (!IsSubsetOf(old_attr.allowed_values().list().type(),
440 new_attr.allowed_values().list().type())) {
441 return true;
442 }
443 if (!IsSubsetOf(old_attr.allowed_values().list().s(),
444 new_attr.allowed_values().list().s())) {
445 return true;
446 }
447 return false;
448}

450string AllowedStr(const OpDef::AttrDef& attr) {
451 if (!attr.has_allowed_values()) return "no restriction";
452 return SummarizeAttrValue(attr.allowed_values());
453}

455string DefaultAttrStr(const OpDef::AttrDef& attr) {
456 if (!attr.has_default_value()) return "no default";
457 return SummarizeAttrValue(attr.default_value());
458}

460bool HigherMinimum(const OpDef::AttrDef& old_attr,
461 const OpDef::AttrDef& new_attr) {
462 // Anything -> no restriction : not more restrictive.
463 if (!new_attr.has_minimum()) return false;
464 // No restriction -> restriction : more restrictive.
465 if (!old_attr.has_minimum()) return true;
466 // If anything that was previously allowed is no longer allowed:
467 // more restrictive.
468 return new_attr.minimum() > old_attr.minimum();
469}

471string MinStr(const OpDef::AttrDef& attr) {
472 if (!attr.has_minimum()) return "no minimum";
473 return strings::StrCat(attr.minimum());

483// Add a comma to *s every call but the first (*add_comma should be
484// initialized to false).
485void AddComma(string* s, bool* add_comma) {
486 if (*add_comma) {
487 strings::StrAppend(s, ", ");
488 } else {
489 *add_comma = true;
490 }

493// Will add the `name` from arg if name is true.
494void AddName(string* s, bool name, const OpDef::ArgDef& arg) {
495 if (name) {
496 strings::StrAppend(s, arg.name(), ":");
497 }
498}
499
500// Compute a signature for either inputs or outputs that will be the
501// same for both the old and new OpDef if they are compatible. We
502// assume that new_attrs is a superset of old_attrs, and that any attr
503// in the difference has a default. Our strategy is to make a list of
504// types, where the types are things like:
505// * "int32", "float", etc.,
506// * "T" for some attr "T" in old_attrs, or
507// * "N * type" for "N" either some attr in old_attrs.
508//
509// We get the types by either using the attrs in args if they are in
510// old_attrs, or substituting the default value from new_attrs.
511string ComputeArgSignature(
512 const protobuf::RepeatedPtrField<OpDef::ArgDef>& args,
513 const AttrMap& old_attrs, const AttrMap& new_attrs, std::vector<bool>* ref,
514 bool names) {
515 string s;
516 bool add_comma = false;
517 for (const OpDef::ArgDef& arg : args) {
518 if (!arg.type_list_attr().empty()) {
519 const OpDef::AttrDef* old_attr =
520 gtl::FindPtrOrNull(old_attrs, arg.type_list_attr());
521 if (old_attr) {
522 // Both old and new have the list(type) attr, so can use it directly.
523 AddComma(&s, &add_comma);
524 AddName(&s, names, arg);
525 strings::StrAppend(&s, arg.type_list_attr());
526 ref->push_back(arg.is_ref());
527 } else {
528 // Missing the list(type) attr in the old, so use the default
529 // value for the attr from new instead.
530 const OpDef::AttrDef* new_attr =
531 gtl::FindPtrOrNull(new_attrs, arg.type_list_attr());
532 const auto& type_list = new_attr->default_value().list().type();
533 if (type_list.empty()) continue;
534 for (int i = 0; i < type_list.size(); ++i) {
535 AddComma(&s, &add_comma);
536 AddName(&s, names, arg);
537 strings::StrAppend(
538 &s, DataTypeString(static_cast<DataType>(type_list.Get(i))));
539 ref->push_back(arg.is_ref());
540 }
541 }
542 } else {
543 int num = 1; // How many input/outputs does this represent?
544 string type; // What is the type of this arg?
545 AddName(&type, names, arg);
546 if (!arg.number_attr().empty()) {
547 // N * type case.
548 const OpDef::AttrDef* old_attr =
549 gtl::FindPtrOrNull(old_attrs, arg.number_attr());
550 if (old_attr) {
551 // Both old and new have the number attr, so can use it directly.
552 strings::StrAppend(&type, arg.number_attr(), " * ");
553 } else {
554 // Missing the number attr in the old, so use the default
555 // value for the attr from new instead.
556 const OpDef::AttrDef* new_attr =
557 gtl::FindPtrOrNull(new_attrs, arg.number_attr());
558 num = new_attr->default_value().i();
559 }
560 }
561
562 if (arg.type() != DT_INVALID) {
563 // int32, float, etc. case
564 strings::StrAppend(&type, DataTypeString(arg.type()));
565 } else {
566 const OpDef::AttrDef* old_attr =
567 gtl::FindPtrOrNull(old_attrs, arg.type_attr());
568 if (old_attr) {
569 // Both old and new have the type attr, so can use it directly.
570 strings::StrAppend(&type, arg.type_attr());
571 } else {
572 // Missing the type attr in the old, so use the default
573 // value for the attr from new instead.
574 const OpDef::AttrDef* new_attr =
575 gtl::FindPtrOrNull(new_attrs, arg.type_attr());
576 strings::StrAppend(&type,
577 DataTypeString(new_attr->default_value().type()));
578 }
579 }
580
581 // Record `num` * `type` in the signature.
582 for (int i = 0; i < num; ++i) {
583 AddComma(&s, &add_comma);
584 strings::StrAppend(&s, type);
585 ref->push_back(arg.is_ref());
586 }
587 }
588 }
589
590 return s;
591}

595Status OpDefCompatible(const OpDef& old_op, const OpDef& new_op) {
596#define VALIDATE(CONDITION, ...) \
597 if (!(CONDITION)) { \
598 return errors::InvalidArgument("Incompatible Op change: ", __VA_ARGS__, \
599 "; old: ", SummarizeOpDef(old_op), \
600 "; new: ", SummarizeOpDef(new_op)); \
601 }
602
603 VALIDATE(old_op.name() == new_op.name(), "Name mismatch");
604
605 AttrMap new_attrs, old_attrs;
606 FillAttrMap(old_op, &old_attrs);
607 FillAttrMap(new_op, &new_attrs);
608 for (const auto& old_attr : old_op.attr()) {
609 const OpDef::AttrDef* new_attr =
610 gtl::FindPtrOrNull(new_attrs, old_attr.name());
611 VALIDATE(new_attr != nullptr, "Attr '", old_attr.name(), "' removed");
612 VALIDATE(old_attr.type() == new_attr->type(), "Attr '", old_attr.name(),
613 "' changed type '", old_attr.type(), "' -> '", new_attr->type(),
614 "'");
615 VALIDATE(!MoreRestrictive(old_attr, *new_attr), "Attr '", old_attr.name(),
616 "' has a stricter set of allowed values; from ",
617 AllowedStr(old_attr), " to ", AllowedStr(*new_attr));
618 VALIDATE(!HigherMinimum(old_attr, *new_attr), "Attr '", old_attr.name(),
619 "' has a higher minimum; from ", MinStr(old_attr), " to ",
620 MinStr(*new_attr));
621 }
622
623 for (const auto& new_attr : new_op.attr()) {
624 const OpDef::AttrDef* old_attr =
625 gtl::FindPtrOrNull(old_attrs, new_attr.name());
626 VALIDATE(old_attr != nullptr || new_attr.has_default_value(), "Attr '",
627 new_attr.name(), "' added without default");
628 }
629
630 std::vector<bool> old_in_ref, new_in_ref, old_out_ref, new_out_ref;
631 const string old_in_sig = ComputeArgSignature(
632 old_op.input_arg(), old_attrs, new_attrs, &old_in_ref, false /* names */);
633 const string new_in_sig = ComputeArgSignature(
634 new_op.input_arg(), old_attrs, new_attrs, &new_in_ref, false /* names */);
635 VALIDATE(old_in_sig == new_in_sig, "Input signature mismatch '", old_in_sig,
636 "' vs. '", new_in_sig, "'");
637 VALIDATE(old_in_ref.size() == new_in_ref.size(), // Should not happen
638 "Unexpected change in input ref lists.");
639 for (int i = 0; i < old_in_ref.size(); ++i) {
640 // Allowed to remove "ref" from an input (or leave it unchanged).
641 VALIDATE(old_in_ref[i] || !new_in_ref[i], "Input ", i,
642 " changed from non-ref to ref");
643 }
644
645 const string old_out_sig =
646 ComputeArgSignature(old_op.output_arg(), old_attrs, new_attrs,
647 &old_out_ref, true /* names */);
648 const string new_out_sig =
649 ComputeArgSignature(new_op.output_arg(), old_attrs, new_attrs,
650 &new_out_ref, true /* names */);
651 VALIDATE(old_out_sig == new_out_sig, "Output signature mismatch '",
652 old_out_sig, "' vs. '", new_out_sig, "'");
653 VALIDATE(old_out_ref.size() == new_out_ref.size(), // Should not happen
654 "Unexpected change in output ref lists");
655 for (int i = 0; i < old_out_ref.size(); ++i) {
656 // Allowed to add "ref" to an output (or leave it unchanged).
657 VALIDATE(!old_out_ref[i] || new_out_ref[i], "Output ", i,
658 " changed from ref to non-ref");
659 }
660
661 return Status::OK();
662}

664Status OpDefAddedDefaultsUnchanged(const OpDef& old_op,
665 const OpDef& penultimate_op,
666 const OpDef& new_op) {
667 AttrMap new_attrs, old_attrs;
668 FillAttrMap(old_op, &old_attrs);
669 FillAttrMap(new_op, &new_attrs);
670
671 for (const auto& penultimate_attr : penultimate_op.attr()) {
672 const OpDef::AttrDef* old_attr =
673 gtl::FindPtrOrNull(old_attrs, penultimate_attr.name());
674 if (old_attr != nullptr) continue; // attr wasn't added
675 const OpDef::AttrDef* new_attr =
676 gtl::FindPtrOrNull(new_attrs, penultimate_attr.name());
677
678 // These shouldn't happen if the op passed OpDefCompatible().
679 if (new_attr == nullptr) {
680 return errors::InvalidArgument("Missing attr '", penultimate_attr.name(),
681 "' in op: ", SummarizeOpDef(new_op));
682 }
683 if (!penultimate_attr.has_default_value() ||
684 !new_attr->has_default_value()) {
685 return errors::InvalidArgument("Missing default for attr '",
686 penultimate_attr.name(),
687 "' in op: ", SummarizeOpDef(new_op));
688 }
689
690 // Actually test that the attr's default value hasn't changed.
691 if (!AreAttrValuesEqual(penultimate_attr.default_value(),
692 new_attr->default_value())) {
693 return errors::InvalidArgument(
694 "Can't change default value for attr '", penultimate_attr.name(),
695 "' from ", SummarizeAttrValue(penultimate_attr.default_value()),
696 " in op: ", SummarizeOpDef(new_op));
697 }
698 }
699
700 return Status::OK();
701}

703Status OpDefAttrDefaultsUnchanged(const OpDef& old_op, const OpDef& new_op) {
704 AttrMap new_attrs, old_attrs;
705 FillAttrMap(old_op, &old_attrs);
706 FillAttrMap(new_op, &new_attrs);
707
708 for (const auto& old_attr : old_op.attr()) {
709 const OpDef::AttrDef* new_attr =
710 gtl::FindPtrOrNull(new_attrs, old_attr.name());
711 if (new_attr == nullptr) continue;
712 if (old_attr.has_default_value() != new_attr->has_default_value()) {
713 return errors::InvalidArgument(
714 "Attr '", old_attr.name(), "' has added/removed it's default; ",
715 "from ", DefaultAttrStr(old_attr), " to ", DefaultAttrStr(*new_attr));
716 }
717 if (old_attr.has_default_value() &&
718 !AreAttrValuesEqual(old_attr.default_value(),
719 new_attr->default_value())) {
720 return errors::InvalidArgument(
721 "Attr '", old_attr.name(), "' has changed it's default value; ",
722 "from ", DefaultAttrStr(old_attr), " to ", DefaultAttrStr(*new_attr));
723 }
724 }
725
726 return Status::OK();
727}

757bool AttrDefEqual(const OpDef::AttrDef& a1, const OpDef::AttrDef& a2) {
758#ifndef TENSORFLOW_LITE_PROTOS
759 DCHECK_EQ(7, a1.GetDescriptor()->field_count())
760 << "Please modify these equality and hash functions to reflect the "
761 "changes to the AttrDef protobuf";
762#endif // TENSORFLOW_LITE_PROTOS
763
764 if (a1.name() != a2.name()) return false;
765 if (a1.type() != a2.type()) return false;
766 if (a1.description() != a2.description()) return false;
767 if (a1.has_minimum() != a2.has_minimum()) return false;
768 if (a1.has_minimum() && a1.minimum() != a2.minimum()) return false;
769 if (!AreAttrValuesEqual(a1.default_value(), a2.default_value())) return false;
770 if (!AreAttrValuesEqual(a1.allowed_values(), a2.allowed_values()))
771 return false;
772 return true;
773}
774
775uint64 AttrDefHash(const OpDef::AttrDef& a) {
776 uint64 h = Hash64(a.name());
777 h = Hash64(a.type().data(), a.type().size(), h);
778 h = Hash64Combine(AttrValueHash(a.default_value()), h);
779 h = Hash64(a.description().data(), a.description().size(), h);
780 h = Hash64Combine(static_cast<uint64>(a.has_minimum()), h);
781 h = Hash64Combine(static_cast<uint64>(a.minimum()), h);
782 h = Hash64Combine(AttrValueHash(a.allowed_values()), h);
783 return h;
784}
785
786bool RepeatedAttrDefEqual(
787 const protobuf::RepeatedPtrField<OpDef::AttrDef>& a1,
788 const protobuf::RepeatedPtrField<OpDef::AttrDef>& a2) {
789 std::unordered_map<string, const OpDef::AttrDef*> a1_set;
790 for (const OpDef::AttrDef& def : a1) {
791 DCHECK(a1_set.find(def.name()) == a1_set.end())
792 << "AttrDef names must be unique, but '" << def.name()
793 << "' appears more than once";
794 a1_set[def.name()] = &def;
795 }
796 for (const OpDef::AttrDef& def : a2) {
797 auto iter = a1_set.find(def.name());
798 if (iter == a1_set.end()) return false;
799 if (!AttrDefEqual(*iter->second, def)) return false;
800 a1_set.erase(iter);
801 }
802 if (!a1_set.empty()) return false;
803 return true;
804}
805
806uint64 RepeatedAttrDefHash(
807 const protobuf::RepeatedPtrField<OpDef::AttrDef>& a) {
808 // Insert AttrDefs into map to deterministically sort by name
809 std::map<string, const OpDef::AttrDef*> a_set;
810 for (const OpDef::AttrDef& def : a) {
811 a_set[def.name()] = &def;
812 }
813 // Iterate and combines hashes of keys and values
814 uint64 h = 0xDECAFCAFFE;
815 for (const auto& pair : a_set) {
816 h = Hash64(pair.first.data(), pair.first.size(), h);
817 h = Hash64Combine(AttrDefHash(*pair.second), h);
818 }
819 return h;
820}

822bool OpDefEqual(const OpDef& o1, const OpDef& o2) {
823 // attr order doesn't matter.
824 // Compare it separately here instead of serializing below.
825 if (!RepeatedAttrDefEqual(o1.attr(), o2.attr())) return false;
826
827 // Clear attr field, serialize, and compare serialized strings
828 OpDef o1_copy = o1;
829 OpDef o2_copy = o2;
830 o1_copy.clear_attr();
831 o2_copy.clear_attr();
832 string s1, s2;
833 SerializeToStringDeterministic(o1_copy, &s1);
834 SerializeToStringDeterministic(o2_copy, &s2);
835 if (s1 != s2) return false;
836 return true;
837}
838
839uint64 OpDefHash(const OpDef& o) {
840 uint64 h = RepeatedAttrDefHash(o.attr());
841 OpDef o_copy = o;
842 o_copy.clear_attr();
843 string s;
844 SerializeToStringDeterministic(o_copy, &s);
845 return Hash64(s.data(), s.size(), h);
846}
