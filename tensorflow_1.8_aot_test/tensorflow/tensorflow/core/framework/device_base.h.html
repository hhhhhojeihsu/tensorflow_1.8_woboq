58// A wrapper for an Eigen Gpu Device that includes per-op state. The
59// class is defined even for non-GPU devices since the
60// OpKernelContext::Params structure wants to fill it in.
61class PerOpGpuDevice {
62 public:
63 virtual ~PerOpGpuDevice() {}
64 virtual const Eigen::GpuDevice& device() const = 0;
65};
66
67// A class that devices can subclass to pass around
68// Device-specific context to OpKernels.
69class DeviceContext : public core::RefCounted {
70 public:
71 ~DeviceContext() override {}
72 virtual perftools::gputools::Stream* stream() const { return nullptr; }
73 virtual void MaintainLifetimeOnStream(
74 const Tensor* t, perftools::gputools::Stream* stream) const {}
75
76 // "cpu_tensor" is a tensor on a CPU. Copies "cpu_tensor" into
77 // "device_tensor" which is on a GPU device "device". "device_tensor"
78 // must be allocated to be of the same size as "cpu_tensor".
79 virtual void CopyCPUTensorToDevice(const Tensor* cpu_tensor, Device* device,
80 Tensor* device_tensor,
81 StatusCallback done) const {
82 done(errors::Internal("Unrecognized device type in CPU-to-device Copy"));
83 }
84
85 // "device_tensor" is a tensor on a non-CPU device. Copies
86 // device_tensor into "cpu_tensor". "cpu_tensor" must be allocated
87 // to be of the same size as "device_tensor".
88 virtual void CopyDeviceTensorToCPU(const Tensor* device_tensor,
89 StringPiece tensor_name, Device* device,
90 Tensor* cpu_tensor, StatusCallback done) {
91 done(errors::Internal("Unrecognized device type in device-to-CPU Copy"));
92 }
93};

98class DeviceBase {
99 public:
100 explicit DeviceBase(Env* env) : env_(env) {}
101 virtual ~DeviceBase();
102
103 Env* env() const { return env_; }
104
105 // Override this to return true for devices that require an Op's
106 // compute method to save references to the temporary tensors it
107 // allocates until the Op execution completes
108 virtual bool RequiresRecordingAccessedTensors() const { return false; }
109
110 struct CpuWorkerThreads {
111 int num_threads = 0;
112 thread::ThreadPool* workers = nullptr;
113 };
114
115 // Does not take ownership.
116 void set_tensorflow_cpu_worker_threads(CpuWorkerThreads* t) {
117 cpu_worker_threads_ = t;
118 }
119
120 virtual const CpuWorkerThreads* tensorflow_cpu_worker_threads() const {
121 CHECK(cpu_worker_threads_ != nullptr);
122 return cpu_worker_threads_;
123 }
124
125 // "stream" is used in special circumstances (such as the
126 // constructors of Ops) where there is no available OpKernelContext.
127 // "default_context" is used by OpKernelContext whenever a device does not
128 // supply a DeviceContext for an op in FillContextMap (e.g. when only
129 // using a single stream.)
130 // "event_mgr" is used to delay deallocation of temporary GPU buffers.
131 // TODO(pbar) Work out how to move this out of DeviceBase.
132 // GpuDeviceInfo name is an unfortunate legacy, it is used not only by GPUs
133 // but also by TPU devices (to provide default device context).
134 struct GpuDeviceInfo {
135 // Make sure all the defaults are NULL, so we can spot missing assignments.
136 perftools::gputools::Stream* stream = nullptr;
137 DeviceContext* default_context = nullptr;
138 EventMgr* event_mgr = nullptr;
139 int gpu_id = -1;
140 };
141
142 // Does not take ownership.
143 void set_tensorflow_gpu_device_info(GpuDeviceInfo* g) {
144 gpu_device_info_ = g;
145 }
146
147 virtual const GpuDeviceInfo* tensorflow_gpu_device_info() const {
148 return gpu_device_info_;
149 }
150
151 // The preferred thread pool for this device. If it is nullptr, the system
152 // automatically assigns a thread pool for execution.
153 virtual thread::ThreadPool* tensorflow_device_thread_pool() {
154 return device_thread_pool_;
155 }
156
157 // Does not take ownership.
158 void set_eigen_cpu_device(Eigen::ThreadPoolDevice* d) {
159 eigen_cpu_device_ = d;
160 }
161
162#ifdef TENSORFLOW_USE_SYCL
163 void set_eigen_sycl_device(Eigen::SyclDevice* d) { eigen_sycl_device_ = d; }
164#endif

166 // Return the Allocator implementation to use based on the allocator
167 // attributes requested. See allocator.h for more details.
168 virtual Allocator* GetAllocator(AllocatorAttributes /*attr*/) {
169 LOG(FATAL) << "GetAllocator() is not implemented.";
170 return nullptr;
171 }
172
173 // Return the Allocator implementation to use based on the allocator
174 // attributes requested and the supplied resource manager. By
175 // default this ignores the resource manager and calls the base
176 // implementation but devices can override if they want to consult
177 // the resource manager when choosing the allocator.
178 virtual Allocator* GetStepAllocator(AllocatorAttributes attr,
179 ResourceMgr* /*step_resource_manager*/) {
180 return GetAllocator(attr);
181 }
182
183 // Return an Allocator prepared for use in particular places by graph
184 // optimization
185 virtual Allocator* GetScopedAllocator(AllocatorAttributes attr,
186 int64 step_id) {
187 LOG(FATAL) << "Device does not implement GetScopedAllocator()";
188 return nullptr;
189 }
190
191 virtual ScopedAllocatorMgr* GetScopedAllocatorMgr() const { return nullptr; }
192
193 virtual const Eigen::ThreadPoolDevice* eigen_cpu_device() {
194 CHECK(eigen_cpu_device_ != nullptr);
195 return eigen_cpu_device_;
196 }
197
198#ifdef TENSORFLOW_USE_SYCL
199 virtual const Eigen::SyclDevice* eigen_sycl_device() const {
200 CHECK(eigen_sycl_device_ != nullptr);
201 return eigen_sycl_device_;
202 }
203#endif
204
205 // Caller owns the return value. The OpKernelContext calls this even
206 // for devices that do not implement an eigen_gpu_device. Overridden
207 // by GPU devices to return a derived type.
208 virtual PerOpGpuDevice* MakeGpuDevice() { return nullptr; }
209
210 virtual DeviceBase* UnderlyingDevice() { return this; }
211 virtual const DeviceBase* UnderlyingDevice() const { return this; }
212
213 // This is overridden by GPU devices to reinitialize the derived
214 // type returned by MakeGpuDevice.
215 virtual void ReinitializeGpuDevice(OpKernelContext* /*context*/,
216 PerOpGpuDevice* /*device*/,
217 DeviceContext* /*dc*/,
218 Allocator* /*allocator*/) {}
219
220 // Unimplemented by default
221 virtual const DeviceAttributes& attributes() const;
222 virtual const string& name() const;
223
224 // Materializes the given TensorProto into 'tensor' stored in Device
225 // memory. Most devices will want to override this.
226 //
227 // TODO(vrv): We should be able to put this function into
228 // OpKernelContext and handle the copies from device memory via send
229 // and receive nodes, instead of requiring that each device handle
230 // the copies here as well as in copy ops.
231 virtual Status MakeTensorFromProto(const TensorProto& tensor_proto,
232 const AllocatorAttributes alloc_attrs,
233 Tensor* tensor) {
234 return errors::Internal("Device does not implement MakeTensorFromProto()");
235 }
236
237 protected:
238 // Does not take ownership.
239 void set_tensorflow_device_thread_pool(thread::ThreadPool* thread_pool) {
240 device_thread_pool_ = thread_pool;
241 }
242
243 private:
244 Env* const env_;
245 CpuWorkerThreads* cpu_worker_threads_ = nullptr;
246 // Set by GPUs as well as by TPU devices.
247 GpuDeviceInfo* gpu_device_info_ = nullptr;
248 thread::ThreadPool* device_thread_pool_ = nullptr;
249 Eigen::ThreadPoolDevice* eigen_cpu_device_ = nullptr;
250#ifdef TENSORFLOW_USE_SYCL
251 Eigen::SyclDevice* eigen_sycl_device_ = nullptr;
252#endif
253};
