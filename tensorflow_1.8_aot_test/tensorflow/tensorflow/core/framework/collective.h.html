41 REDUCTION_COLLECTIVE = 0,
42 BROADCAST_COLLECTIVE,

46// Data common to all members of a device group.
47// All members share the same device set but its order is
48// particular to an instance so it is stored there.
49struct CollGroupParams {
50 int32 group_key;
51 int32 group_size;
52 DeviceType device_type;
53 int32 num_tasks; // number of distinct tasks in group
54 string ToString() const;
55 CollGroupParams() : device_type(DEVICE_CPU) {}
56};
57
58// The best implementation of a collective op depends on many factors
59// including the number of devices involved, the topology of
60// interconnects between them and the sizes of inputs. This structure
61// is used in generating and representing data movement choreography
62// for each specific algorithm, hence it does not have a single, fixed
63// interpretation. On first execution the runtime will update this
64// structure with decisions that will guide all subsequent executions.
65struct CollImplDetails {
66 std::vector<std::vector<int>> subdiv_permutations;
67 std::vector<int> subdiv_offsets;
68 // broadcast only: rank of source in each subdiv
69 std::vector<int> subdiv_source_rank;
70};
71
72// Data common to all members of a collective instance.
73struct CollInstanceParams {
74 int32 instance_key; // Identifies all participating graph nodes.
75 CollectiveType type;
76 DataType data_type;
77 TensorShape shape;
78 // Fully qualified name of device for each member, in default rank order.
79 std::vector<string> device_names;
80 // Task name prefix of corresponding device name.
81 std::vector<string> task_names;
82 CollImplDetails impl_details;
83 string ToString() const;
84 CollInstanceParams& operator=(const struct CollInstanceParams& other);
85};
86
87// Data common to all instance members in the same task.
88struct CollTaskParams {
89 // True for devices that are local to the process, i.e. no RPC needed.
90 std::vector<bool> is_local;
91 string ToString() const;
92};
93
94// Unique to a single CollectiveOp node.
95struct CollectiveParams {
96 CollGroupParams group;
97 CollInstanceParams instance;
98 CollTaskParams task;
99
100 string name; // node name used only for log or error messages
101 int default_rank; // index of this op within device_names
102 bool is_source; // broadcast only
103 // Rank of this device in each subdivision permutation.
104 std::vector<int> subdiv_rank;
105 std::vector<int> subdiv_source_rank;
106 std::unique_ptr<OpKernel> merge_op; // reduction only
107 std::unique_ptr<OpKernel> final_op; // reduction only
108 string ToString() const;
109};

113// Interface that provides resolution of device localities.
114class DeviceResolverInterface {
115 public:
116 virtual ~DeviceResolverInterface() {}
117
118 // Collects DeviceLocality protobufs from all of the devices identified
119 // in 'col_params'.
120 virtual void GetDeviceLocalitiesAsync(const CollInstanceParams& inst_params,
121 std::vector<DeviceLocality>* localities,
122 const StatusCallback& done) = 0;
123
124 // Populate *locality with the DeviceLocality of the specified
125 // device.
126 virtual void GetLocalityAsync(const string& device, const string& task,
127 DeviceLocality* locality,
128 const StatusCallback& done) = 0;
129
130 // Clear the cache of device data belonging
131 // to the specified task.
132 virtual void ClearTask(const string& task) = 0;
133};

135// Interface that provides resolution of shared CollectiveParams fields.
136class ParamResolverInterface {
137 public:
138 virtual ~ParamResolverInterface() {}
139
140 // Called by each collective op at first execution in order to fill out
141 // the CollectiveParams structure with data gathered from the full
142 // (maybe distributed) collection of peer nodes.
143 virtual void CompleteParamsAsync(const string& device, CollectiveParams* cp,
144 CancellationManager* cancel_mgr,
145 const StatusCallback& done) = 0;
146
147 // Used within a distributed implementation to discover/verify
148 // data shared across a device group.
149 virtual void CompleteGroupAsync(const CompleteGroupRequest* request,
150 CompleteGroupResponse* response,
151 CancellationManager* cancel_mgr,
152 const StatusCallback& done) = 0;
153
154 // Used within a distributed implementation to discover/verify data
155 // shared across an instance group.
156 virtual void CompleteInstanceAsync(const CompleteInstanceRequest* request,
157 CompleteInstanceResponse* response,
158 CancellationManager* cancel_mgr,
159 const StatusCallback& done) = 0;
160};

162// Graphs which utilize Collective Ops in a common instance must
163// execute with identical step_ids even if they are disjoint graphs
164// run by otherwise independent tasks. This interface supplies
165// coordinated step_ids to use in such cases.
166class StepSequenceInterface {
167 public:
168 virtual ~StepSequenceInterface() {}
169
170 // Used with a distributed implementation to coordinate step_id
171 // sequences across tasks.
172 virtual void GetStepSequenceAsync(const GetStepSequenceRequest* request,
173 GetStepSequenceResponse* response,
174 const StatusCallback& done) = 0;
175
176 // Refresh the local per-graph_key step_id sequence from collective
177 // group leader, if applicable.
178 virtual void RefreshStepIdSequenceAsync(int64 graph_key,
179 const StatusCallback& done) = 0;
180
181 // Returns the the step_id that should be used for initiating a new execution
182 // on the specified graph. May return the same step_id multiple times if
183 // RetireStepId or RefreshStepIdReservation is not called.
184 virtual int64 NextStepId(int64 graph_key) = 0;
185
186 // Reports that execution of the given step has completed successfully.
187 // Should be called immediately after a step completes with OK status,
188 // prior to calling NextStepId(). If the step fails, don't call.
189 virtual void RetireStepId(int64 graph_key, int64 step_id) = 0;
190};

192// Interface that provides access to per-step CollectiveExecutor
193// instances and various distributed resolution capabilities.
194class CollectiveExecutorMgrInterface : public StepSequenceInterface {
195 public:
196 virtual ~CollectiveExecutorMgrInterface() {}
197
198 // Returns the step-specific CollectiveExecutor, creating if one does not
199 // already exist. The caller assumes ownership of one Ref on the object.
200 virtual CollectiveExecutor* FindOrCreate(int64 step_id) = 0;
201
202 // If there is a CollectiveExecutor for step_id, remove it from the
203 // table.
204 virtual void Cleanup(int64 step_id) = 0;
205
206 virtual ParamResolverInterface* GetParamResolver() const = 0;
207
208 virtual DeviceResolverInterface* GetDeviceResolver() const = 0;
209};

238// A step-specific object that can execute a collective operation completely
239// described by a CollectiveParams object.
240class CollectiveExecutor : public PeerAccessInterface, public core::RefCounted {
241 public:
242 virtual void StartAbort(const Status& s) {}
243
244 virtual void ExecuteAsync(OpKernelContext* ctx,
245 const CollectiveParams& col_params,
246 const string& exec_key, StatusCallback done) {
247 done(errors::Internal(
248 "A collective Op has been called in a context in which "
249 "a CollectiveExecutor has not been provided."));
250 }
251
252 virtual void CompleteParamsAsync(const string& device, CollectiveParams* cp,
253 CancellationManager* cancel_mgr,
254 StatusCallback done) {
255 cem_->GetParamResolver()->CompleteParamsAsync(device, cp, cancel_mgr, done);
256 }
257
258 virtual PerStepCollectiveRemoteAccess* remote_access() { return nullptr; }
259
260 // Used to designate an invalid group or instance key.
261 static int64 kInvalidId;
262
263 // Lexically scoped handle for Ref.
264 class Handle {
265 public:
266 explicit Handle(CollectiveExecutor* ce, bool inherit_ref) : ce_(ce) {
267 if (!inherit_ref) ce->Ref();
268 }
269 ~Handle() { ce_->Unref(); }
270 CollectiveExecutor* get() const { return ce_; }
271
272 private:
273 CollectiveExecutor* ce_;
274 };
275
276 protected:
277 explicit CollectiveExecutor(CollectiveExecutorMgrInterface* cem)
278 : cem_(cem) {}
279
280 // For use only by derived classes
281 static OpKernelContext::Params* CtxParams(OpKernelContext* ctx);
282 CollectiveExecutorMgrInterface* cem_;
283
284 TF_DISALLOW_COPY_AND_ASSIGN(CollectiveExecutor);
285};
