32// Type used for tag-dispatch of the Encode/Decode Variant implementations. This
33// template can determine whether the first type parameter `T` is one of the
34// following:
35//
36// * A POD type (TypeResolver<T, true>)
37// * A tensorflow::Tensor (TypeResolver<T, false, true>)
38// * A protocol buffer (TypeResolver<T, false, false, true>)
39// * None of the above (TypeResolver<T, false, false, false>)
40//
41template <typename T, bool = std::is_pod<typename std::decay<T>::type>::value,
42 bool = std::is_same<typename std::decay<T>::type,
43 ::tensorflow::Tensor>::value,
44 bool = std::is_base_of<protobuf::MessageLite,
45 typename std::decay<T>::type>::value>
46struct TypeResolver {};
47
48// Specialization for POD type
49template <typename T>
50void EncodeVariantImpl(const T& value, TypeResolver<T, true /* is_pod */>,
51 VariantTensorData* data) {
52 data->set_metadata(value);
53}
54
55// Specialization for tensorflow::Tensor
56template <typename T>
57void EncodeVariantImpl(const T& value,
58 TypeResolver<T, false /* is_pod */, true /* Tensor */>,
59 VariantTensorData* data) {
60 data->tensors_.clear();
61 data->tensors_.push_back(value);
62}
63
64// Specialization for protobuf
65template <typename T>
66void EncodeVariantImpl(const T& value,
67 TypeResolver<T, false /* is_pod */, false /* Tensor */,
68 true /* protobuf */>,
69 VariantTensorData* data) {
70 value.SerializeToString(&data->metadata_);
71}
72
73// Specialization for other types
74template <typename T>
75void EncodeVariantImpl(const T& value,
76 TypeResolver<T, false /* is_pod */, false /* Tensor */,
77 false /* protobuf */>,
78 VariantTensorData* data) {
79 value.Encode(data);
80}
81
82// Specialization for POD type
83template <typename T>
84bool DecodeVariantImpl(const VariantTensorData& data,
85 TypeResolver<T, true /* is_pod */, false /* Tensor */,
86 false /* protobuf */>,
87 T* value) {
88 return data.get_metadata(value);
89}
90
91// Specialization for tensorflow::Tensor
92template <typename T>
93bool DecodeVariantImpl(const VariantTensorData& data,
94 TypeResolver<T, false /* is_pod */, true /* Tensor */,
95 false /* protobuf */>,
96 T* value) {
97 *value = data.tensors(0);
98 return true;
99}
100
101// Specialization for protobuf
102template <typename T>
103bool DecodeVariantImpl(const VariantTensorData& data,
104 TypeResolver<T, false /* is_pod */, false /* Tensor */,
105 true /* protobuf */>,
106 T* value) {
107 string metadata;
108 data.get_metadata(&metadata);
109 return value->ParseFromString(std::move(metadata));
110}
111
112// Specialization for other types
113template <typename T>
114bool DecodeVariantImpl(const VariantTensorData& data,
115 TypeResolver<T, false /* is_pod */, false /* Tensor */,
116 false /* protobuf */>,
117 T* value) {
118 return value->Decode(data);
119}

121template <typename C, typename = void>
122struct has_type_name : std::false_type {};
123
124template <typename C>
125struct has_type_name<
126 C, typename std::enable_if<std::is_same<
127 decltype(std::declval<C>().TypeName()), string>::value>::type>
128 : std::true_type {};
129
130template <typename T, bool = has_type_name<typename std::decay<T>::type>::value,
131 bool = std::is_same<typename std::decay<T>::type,
132 ::tensorflow::Tensor>::value,
133 bool = std::is_base_of<protobuf::MessageLite,
134 typename std::decay<T>::type>::value>
135struct TypeNameResolver {};
136
137template <typename T>
138string TypeNameVariantImpl(const T& value,
139 TypeNameResolver<T, true /* has_type_name */>) {
140 return value.TypeName();
141}
142
143template <typename T>
144string TypeNameVariantImpl(
145 const T& value,
146 TypeNameResolver<T, false /* has_type_name */, true /* Tensor */>) {
147 return "tensorflow::Tensor";
148}
149
150template <typename T>
151string TypeNameVariantImpl(
152 const T& value, TypeNameResolver<T, false /* has_type_name */,
153 false /* Tensor */, true /* protobuf */>) {
154 return value.GetTypeName();
155}
156
157template <typename T>
158string TypeNameVariantImpl(
159 const T& value,
160 TypeNameResolver<T, false /* has_type_name */, false /* Tensor */,
161 false /* protobuf */>) {
162 return port::MaybeAbiDemangle(MakeTypeIndex<T>().name());
163}
164
165template <typename T>
166string TypeNameVariant(const T& value) {
167 return TypeNameVariantImpl(value, TypeNameResolver<T>());
168}

220template <typename T>
221void EncodeVariant(const T& value, VariantTensorData* data) {
222 EncodeVariantImpl(value, TypeResolver<T>(), data);
223 data->set_type_name(TypeNameVariant(value));
224}
225
226template <typename T>
227bool DecodeVariant(const VariantTensorData& data, T* value) {
228 return DecodeVariantImpl(data, TypeResolver<T>(), value);
229}
230
231template <typename T>
232void EncodeVariant(const T& value, string* buf) {
233 VariantTensorData data;
234 EncodeVariantImpl(value, TypeResolver<T>(), &data);
235 data.set_type_name(TypeNameVariant(value));
236 DCHECK(buf != nullptr);
237 data.SerializeToString(buf);
238}
239
240template <typename T>
241bool DecodeVariant(const string& buf, T* value) {
242 VariantTensorData data;
243 if (!data.ParseFromString(buf)) return false;
244 if (!DecodeVariantImpl(data, TypeResolver<T>(), value)) return false;
245 return true;
246}
