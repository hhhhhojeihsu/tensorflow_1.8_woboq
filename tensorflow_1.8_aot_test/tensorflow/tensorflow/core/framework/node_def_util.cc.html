53static string SummarizeAttrsHelper(AttrSlice attrs, StringPiece device) {
54 string ret;
55
56 // We sort the attrs so the output is deterministic.
57 std::vector<string> attr_names;
58 attr_names.reserve(attrs.size());
59 for (const auto& attr : attrs) {
60 attr_names.push_back(attr.first);
61 }
62 std::sort(attr_names.begin(), attr_names.end());
63 bool first = true;
64 for (const string& attr_name : attr_names) {
65 if (!first) strings::StrAppend(&ret, ", ");
66 first = false;
67 strings::StrAppend(&ret, attr_name, "=",
68 SummarizeAttrValue(*attrs.Find(attr_name)));
69 }
70
71 // Consider the device to be a final attr with name "_device".
72 if (!device.empty()) {
73 if (!first) strings::StrAppend(&ret, ", ");
74 first = false;
75 strings::StrAppend(&ret, "_device=\"", device, "\"");
76 }
77 return ret;
78}

80string AttrSlice::SummarizeNode() const {
81 return ndef_ ? SummarizeNodeDef(*ndef_)
82 : strings::StrCat(
83 "[", SummarizeAttrsHelper(*this, StringPiece()), "]");
84}

86string SummarizeNode(const Node& node) { return SummarizeNodeDef(node.def()); }
87
88string SummarizeNodeDef(const NodeDef& node_def) {
89 string ret = strings::StrCat(node_def.name(), " = ", node_def.op(), "[");
90 strings::StrAppend(&ret, SummarizeAttrsHelper(node_def, node_def.device()));
91 strings::StrAppend(&ret, "](");
92
93 // Output inputs, including control inputs, verbatim.
94 bool first = true;
95 for (const string& input : node_def.input()) {
96 if (!first) strings::StrAppend(&ret, ", ");
97 first = false;
98 strings::StrAppend(&ret, input);
99 }
100 strings::StrAppend(&ret, ")");
101 return ret;
102}

104const AttrValue* AttrSlice::Find(StringPiece attr_name) const {
105 // Currently, the collection used for NodeDef::attr() (google::protobuf::Map)
106 // requires that the keys used for lookups have type 'const string&'. Because
107 // this method takes a StringPiece, it is necessary to allocate a temporary
108 // string, copy attr_name to it, and then use that temporary string for the
109 // lookup. This causes an excessive number of short-lived allocations, and for
110 // large graphs, this can be a significant cost.
111 //
112 // Because most nodes have a small number of attributes, a simple linear scan
113 // is generally more efficient than a hashed lookup. If google::protobuf::Map
114 // changes so that it supports efficient lookups using StringPiece instead of
115 // const string&, then this code could be changed to use attrs_->find() again.
116
117 for (const auto& attr : *attrs_) {
118 if (attr.first == attr_name) {
119 return &attr.second;
120 }
121 }
122 return nullptr;
123}
124
125Status AttrSlice::Find(StringPiece attr_name,
126 const AttrValue** attr_value) const {
127 *attr_value = Find(attr_name);
128 if (*attr_value != nullptr) {
129 return Status::OK();
130 }
131 Status s = errors::NotFound("No attr named '", attr_name, "' in NodeDef:");
132 // Skip AttachDef for internal attrs since it is a little bit
133 // expensive and it is common for them to correctly not be included
134 // in a NodeDef.
135 if (!str_util::StartsWith(attr_name, "_") && ndef_ != nullptr) {
136 s = AttachDef(s, *ndef_);
137 }
138 return s;
139}

141bool AttrSlice::EqualAttrs(AttrSlice other, Scratch* scratch) const {
142 if (size() != other.size()) return false;
143
144 for (const auto& attr : *other.attrs_) {
145 auto iter = attrs_->find(attr.first);
146 if (iter == attrs_->end()) return false;
147 // TODO(irving): Comparing AttrValues by proto is slightly buggy, since
148 // TensorProto is a nonunique representation of Tensor. This bug will go
149 // away once AttrSlice switches over to NodeInfo.
150 iter->second.SerializeToString(&scratch->a);
151 attr.second.SerializeToString(&scratch->b);
152 if (scratch->a != scratch->b) return false;
153 }
154 return true;
155}

159#define DEFINE_GET_ATTR(TYPE, FIELD, ATTR_TYPE, APPEND_OP, CAST, ...) \
160 Status GetNodeAttr(const AttrSlice& attrs, StringPiece attr_name, \
161 TYPE* value) { \
162 const AttrValue* attr_value; \
163 TF_RETURN_IF_ERROR(attrs.Find(attr_name, &attr_value)); \
164 TF_RETURN_IF_ERROR(AttrValueHasType(*attr_value, ATTR_TYPE)); \
165 const auto& v = attr_value->FIELD(); \
166 __VA_ARGS__; \
167 *value = CAST; \
168 return Status::OK(); \
169 } \
170 Status GetNodeAttr(const AttrSlice& attrs, StringPiece attr_name, \
171 std::vector<TYPE>* value) { \
172 const AttrValue* attr_value; \
173 TF_RETURN_IF_ERROR(attrs.Find(attr_name, &attr_value)); \
174 TF_RETURN_IF_ERROR(AttrValueHasType(*attr_value, "list(" ATTR_TYPE ")")); \
175 for (const auto& v : attr_value->list().FIELD()) { \
176 __VA_ARGS__; \
177 value->APPEND_OP(CAST); \
178 } \
179 return Status::OK(); \
180 }

183 bool GetNodeAttrSimple(const AttrSlice& attrs, StringPiece attr_name, \
184 TYPE* value) { \
185 const AttrValue* attr_value = attrs.Find(attr_name); \
186 if (attr_value == nullptr) { \
187 return false; \
188 } \
189 Status s = AttrValueHasType(*attr_value, ATTR_TYPE); \
190 if (!s.ok()) { \
191 return false; \
192 } \
193 const auto& v = attr_value->FIELD(); \
194 __VA_ARGS__; \
195 *value = CAST; \
196 return true; \
197 } \
198 bool GetNodeAttrSimple(const AttrSlice& attrs, StringPiece attr_name, \
199 std::vector<TYPE>* value) { \
200 const AttrValue* attr_value = attrs.Find(attr_name); \
201 if (attr_value == nullptr) { \
202 return false; \
203 } \
204 Status s = AttrValueHasType(*attr_value, "list(" ATTR_TYPE ")"); \
205 if (!s.ok()) { \
206 return false; \
207 } \
208 for (const auto& v : attr_value->list().FIELD()) { \
209 __VA_ARGS__; \
210 value->APPEND_OP(CAST); \
211 } \
212 return true; \
213 }

253const string& GetNodeAttrString(const AttrSlice& attrs, StringPiece attr_name) {
254 const AttrValue* attr_value = attrs.Find(attr_name);
255 if (attr_value == nullptr) {
256 return kEmptyString;
257 }
258 Status s = AttrValueHasType(*attr_value, "string");
259 if (!s.ok()) {
260 return kEmptyString;

265Status GetNodeAttr(const AttrSlice& attrs, StringPiece attr_name,
266 DataTypeVector* value) {
267 const AttrValue* attr_value;
268 TF_RETURN_IF_ERROR(attrs.Find(attr_name, &attr_value));
269 TF_RETURN_IF_ERROR(AttrValueHasType(*attr_value, "list(type)"));
270 for (const auto& v : attr_value->list().type()) {
271 value->push_back(static_cast<DataType>(v));
272 }
273 return Status::OK();
274}

296Status AddArgToSig(const NodeDef& node_def, const OpDef::ArgDef& arg_def,
297 DataTypeVector* sig) {
298 const int original_size = sig->size();
299 if (!arg_def.number_attr().empty()) {
300 // Same type repeated "repeats" times.
301 int32 repeats = -1;
302 TF_RETURN_IF_ERROR(GetNodeAttr(node_def, arg_def.number_attr(), &repeats));
303 if (repeats < 0) {
304 return errors::InvalidArgument("Value for number_attr() ", repeats,
305 " < 0");
306 }
307
308 if (!arg_def.type_attr().empty()) {
309 DataType dtype;
310 TF_RETURN_IF_ERROR(GetNodeAttr(node_def, arg_def.type_attr(), &dtype));
311 for (int i = 0; i < repeats; ++i) {
312 sig->push_back(dtype);
313 }
314 } else if (arg_def.type() != DT_INVALID) {
315 for (int i = 0; i < repeats; ++i) {
316 sig->push_back(arg_def.type());
317 }
318 } else {
319 return errors::InvalidArgument("Missing type or type_attr field in ",
320 ProtoShortDebugString(arg_def));
321 }
322 } else if (!arg_def.type_attr().empty()) {
323 const AttrValue* attr_value;
324 TF_RETURN_IF_ERROR(
325 AttrSlice(node_def).Find(arg_def.type_attr(), &attr_value));
326 sig->push_back(attr_value->type());
327 } else if (!arg_def.type_list_attr().empty()) {
328 const AttrValue* attr_value;
329 TF_RETURN_IF_ERROR(
330 AttrSlice(node_def).Find(arg_def.type_list_attr(), &attr_value));
331 for (int dtype : attr_value->list().type()) {
332 sig->push_back(static_cast<DataType>(dtype));
333 }
334 } else if (arg_def.type() != DT_INVALID) {
335 sig->push_back(arg_def.type());
336 } else {
337 return errors::InvalidArgument("No type fields in ",
338 ProtoShortDebugString(arg_def));
339 }
340 if (arg_def.is_ref()) {
341 // For all types that were added by this function call, make them refs.
342 for (size_t i = original_size; i < sig->size(); ++i) {
343 (*sig)[i] = MakeRefType((*sig)[i]);
344 }
345 }
346 return Status::OK();
347}

351Status InputTypeForNode(const NodeDef& node_def, const OpDef& op_def,
352 int input_port, DataType* input_type) {
353 DataTypeVector input_types;
354 for (const auto& arg : op_def.input_arg()) {
355 TF_RETURN_IF_ERROR(AddArgToSig(node_def, arg, &input_types));
356 if (input_types.size() > input_port) {
357 const DataType dtype = input_types[input_port];
358 *input_type = dtype;
359 return Status::OK();
360 }
361 }
362 return errors::InvalidArgument("Input ", input_port, " not found for node ",
363 node_def.name());
364}
365
366Status OutputTypeForNode(const NodeDef& node_def, const OpDef& op_def,
367 int output_port, DataType* output_type) {
368 DataTypeVector output_types;
369 for (const auto& arg : op_def.output_arg()) {
370 TF_RETURN_IF_ERROR(AddArgToSig(node_def, arg, &output_types));
371 if (output_types.size() > output_port) {
372 const DataType dtype = output_types[output_port];
373 *output_type = dtype;
374 return Status::OK();
375 }
376 }
377 return errors::InvalidArgument("Output ", output_port, " not found for node ",
378 node_def.name());
379}

381Status InOutTypesForNode(const NodeDef& node_def, const OpDef& op_def,
382 DataTypeVector* inputs, DataTypeVector* outputs) {
383 for (const auto& arg : op_def.input_arg()) {
384 TF_RETURN_IF_ERROR(AddArgToSig(node_def, arg, inputs));
385 }
386 for (const auto& arg : op_def.output_arg()) {
387 TF_RETURN_IF_ERROR(AddArgToSig(node_def, arg, outputs));
388 }
389 return Status::OK();
390}

392Status ValidateNodeDef(const NodeDef& node_def, const OpDef& op_def) {
393 if (node_def.op() != op_def.name()) {
394 return errors::InvalidArgument("NodeDef op '", node_def.op(),
395 "' does not match ", SummarizeOpDef(op_def),
396 "; NodeDef: ", SummarizeNodeDef(node_def));
397 }
398
399 bool seen_control = false;
400 size_t num_inputs = 0;
401 // TODO(josh11b): Unify the input field validation.
402 for (const string& input : node_def.input()) {
403 if (str_util::StartsWith(input, "^")) {
404 seen_control = true;
405 if (input.find(':') != string::npos) {
406 return errors::InvalidArgument(
407 "Control input '", input,
408 "' must not have ':' in NodeDef: ", SummarizeNodeDef(node_def));
409 }
410 } else if (seen_control) {
411 return errors::InvalidArgument(
412 "Non-control input '", input,
413 "' after control input in NodeDef: ", SummarizeNodeDef(node_def));
414 } else {
415 ++num_inputs;
416 }
417 }
418
419 std::unordered_map<string, const OpDef::AttrDef*> op_attrs;
420 for (const auto& attr : op_def.attr()) {
421 if (!gtl::InsertIfNotPresent(&op_attrs, attr.name(), &attr)) {
422 return errors::InvalidArgument("OpDef has duplicate attr name '",
423 attr.name(),
424 "': ", SummarizeOpDef(op_def));
425 }
426 }
427 for (const auto& attr : node_def.attr()) {
428 // Allow internal optional attributes with names starting with "_".
429 if (str_util::StartsWith(attr.first, "_")) {
430 continue;
431 }
432 auto iter = op_attrs.find(attr.first);
433 if (iter == op_attrs.end()) {
434 // A common cause of this error is that TensorFlow has made a
435 // backwards-compatible change to the NodeDef (e.g., adding a
436 // new attr with a default value), but the binary consuming the
437 // NodeDef does not know about the new attribute; the solution
438 // in these cases is to ensure that the binary consuming the
439 // NodeDef is built with a version of TensorFlow no earlier than
440 // the binary producing it.
441 return errors::InvalidArgument(
442 "NodeDef mentions attr '", attr.first, "' not in ",
443 SummarizeOpDef(op_def), "; NodeDef: ", SummarizeNodeDef(node_def),
444 ". (Check whether your GraphDef-interpreting binary is up to date "
445 "with your GraphDef-generating binary.).");
446 }
447 TF_RETURN_WITH_CONTEXT_IF_ERROR(
448 ValidateAttrValue(attr.second, *iter->second),
449 "; NodeDef: ", SummarizeNodeDef(node_def), "; ",
450 SummarizeOpDef(op_def));
451 // Keep track of which attr names have (not) been found in the NodeDef.
452 op_attrs.erase(iter);
453 }
454
455 // Were all attrs in the OpDef found in the NodeDef?
456 if (!op_attrs.empty()) {
457 string attrs;
458 for (const auto& attr_pair : op_attrs) {
459 if (!attrs.empty()) strings::StrAppend(&attrs, "', '");
460 strings::StrAppend(&attrs, attr_pair.first);
461 }
462 return errors::InvalidArgument("NodeDef missing attr",
463 op_attrs.size() == 1 ? " '" : "s '", attrs,
464 "' from ", SummarizeOpDef(op_def),
465 "; NodeDef: ", SummarizeNodeDef(node_def));
466 }
467
468 // Validate the number of inputs.
469 DataTypeVector inputs, outputs;
470 TF_RETURN_IF_ERROR(InOutTypesForNode(node_def, op_def, &inputs, &outputs));
471
472 if (num_inputs != inputs.size()) {
473 return errors::InvalidArgument(
474 "NodeDef expected inputs '", DataTypeVectorString(inputs),
475 "' do not match ", num_inputs, " inputs specified; ",
476 SummarizeOpDef(op_def), "; NodeDef: ", SummarizeNodeDef(node_def));
477 }
478
479 return Status::OK();
480}

484Status ComputeArgRange(const NodeDef& node_def, const OpDef::ArgDef& arg_def,
485 const OpDef& op_def, int* num) {
486 if (!arg_def.number_attr().empty()) {
487 // Same type repeated "num" times.
488 return GetNodeAttr(node_def, arg_def.number_attr(), num);
489 } else if (!arg_def.type_list_attr().empty()) {
490 const AttrValue* attr_value;
491 TF_RETURN_IF_ERROR(
492 AttrSlice(node_def).Find(arg_def.type_list_attr(), &attr_value));
493 *num = attr_value->list().type_size();
494 } else if (!arg_def.type_attr().empty() || arg_def.type() != DT_INVALID) {
495 *num = 1;
496 } else {
497 return errors::InvalidArgument(
498 "Argument '", arg_def.name(),
499 "' incorrectly specified in op definition: ", SummarizeOpDef(op_def));
500 }
501 return Status::OK();
502}

504Status NameRangesHelper(const NodeDef& node_def,
505 const protobuf::RepeatedPtrField<OpDef::ArgDef>& args,
506 const OpDef& op_def, NameRangeMap* result) {
507 int start = 0;
508 int num;
509 for (const auto& arg : args) {
510 TF_RETURN_IF_ERROR(ComputeArgRange(node_def, arg, op_def, &num));
511 (*result)[arg.name()] = std::make_pair(start, start + num);
512 start += num;
513 }
514 return Status::OK();
515}

549bool IsValidOpName(StringPiece sp) {
550 return Scanner(sp)
551 .One(Scanner::LETTER_DIGIT_DOT)
552 .Any(Scanner::LETTER_DIGIT_DASH_DOT_SLASH_UNDERSCORE)
553 .Eos()
554 .GetResult();
555}
556
557bool IsValidDataInputName(StringPiece sp) {
558 // Data inputs are op_name, op_name:0, or op_name:12345.
559 Scanner scan(sp);
560 scan.One(Scanner::LETTER_DIGIT_DOT)
561 .Any(Scanner::LETTER_DIGIT_DASH_DOT_SLASH_UNDERSCORE);
562 if (scan.Peek() == ':') {
563 scan.OneLiteral(":");
564 if (scan.Peek() == '0') {
565 scan.OneLiteral("0"); // :0
566 } else {
567 scan.Many(Scanner::DIGIT); // :[1-9][0-9]*
568 }
569 }
570 scan.Eos();
571
572 return scan.GetResult();
573}
574
575bool IsValidControlInputName(StringPiece sp) {
576 return Scanner(sp)
577 .OneLiteral("^")
578 .One(Scanner::LETTER_DIGIT_DOT)
579 .Any(Scanner::LETTER_DIGIT_DASH_DOT_SLASH_UNDERSCORE)
580 .Eos()
581 .GetResult();
582}

586Status ValidateOpInput(const string& input_name, bool* is_control_input) {
587 *is_control_input = false;
588 if (IsValidDataInputName(input_name)) {
589 return Status::OK();
590 } else if (IsValidControlInputName(input_name)) {
591 *is_control_input = true;
592 return Status::OK();
593 } else {
594 return errors::InvalidArgument("Illegal op input name '", input_name, "'");
595 }
596}
597
598Status ValidateOpName(const string& op_name) {
599 if (IsValidOpName(op_name)) {
600 return Status::OK();
601 } else {
602 return errors::InvalidArgument("Illegal op name '", op_name, "'");
603 }
604}
605
606Status ValidateExternalNodeDefSyntax(const NodeDef& node_def) {
607 Status s = ValidateOpName(node_def.name());
608 if (!s.ok()) {
609 return AttachDef(s, node_def);
610 }
611 bool in_control_inputs = false;
612 for (const string& input_name : node_def.input()) {
613 bool is_control_input;
614 s = ValidateOpInput(input_name, &is_control_input);
615 if (!s.ok()) {
616 return AttachDef(s, node_def);
617 }
618
619 if (in_control_inputs && !is_control_input) {
620 return AttachDef(errors::InvalidArgument(
621 "All control inputs must follow all data inputs"),
622 node_def);
623 }
624 in_control_inputs = is_control_input;
625 }
626 return Status::OK();
627}
