28string WordWrap(StringPiece prefix, StringPiece str, int width) {
29 const string indent_next_line = "\n" + Spaces(prefix.size());
30 width -= prefix.size();
31 string result;
32 strings::StrAppend(&result, prefix);
33
34 while (!str.empty()) {
35 if (static_cast<int>(str.size()) <= width) {
36 // Remaining text fits on one line.
37 strings::StrAppend(&result, str);
38 break;
39 }
40 auto space = str.rfind(' ', width);
41 if (space == StringPiece::npos) {
42 // Rather make a too-long line and break at a space.
43 space = str.find(' ');
44 if (space == StringPiece::npos) {
45 strings::StrAppend(&result, str);
46 break;
47 }
48 }
49 // Breaking at character at position <space>.
50 StringPiece to_append = str.substr(0, space);
51 str.remove_prefix(space + 1);
52 // Remove spaces at break.
53 while (str_util::EndsWith(to_append, " ")) {
54 to_append.remove_suffix(1);
55 }
56 while (str_util::ConsumePrefix(&str, " ")) {
57 }
58
59 // Go on to the next line.
60 strings::StrAppend(&result, to_append);
61 if (!str.empty()) strings::StrAppend(&result, indent_next_line);
62 }
63
64 return result;
65}

67bool ConsumeEquals(StringPiece* description) {
68 if (str_util::ConsumePrefix(description, "=")) {
69 while (str_util::ConsumePrefix(description,
70 " ")) { // Also remove spaces after "=".
71 }
72 return true;
73 }
74 return false;
75}

77// Split `*orig` into two pieces at the first occurrence of `split_ch`.
78// Returns whether `split_ch` was found. Afterwards, `*before_split`
79// contains the maximum prefix of the input `*orig` that doesn't
80// contain `split_ch`, and `*orig` contains everything after the
81// first `split_ch`.
82static bool SplitAt(char split_ch, StringPiece* orig,
83 StringPiece* before_split) {
84 auto pos = orig->find(split_ch);
85 if (pos == StringPiece::npos) {
86 *before_split = *orig;
87 *orig = StringPiece();
88 return false;
89 } else {
90 *before_split = orig->substr(0, pos);
91 orig->remove_prefix(pos + 1);
92 return true;
93 }
94}

96// Does this line start with "<spaces><field>:" where "<field>" is
97// in multi_line_fields? Sets *colon_pos to the position of the colon.
98static bool StartsWithFieldName(StringPiece line,
99 const std::vector<string>& multi_line_fields) {
100 StringPiece up_to_colon;
101 if (!SplitAt(':', &line, &up_to_colon)) return false;
102 while (str_util::ConsumePrefix(&up_to_colon, " "))
103 ; // Remove leading spaces.
104 for (const auto& field : multi_line_fields) {
105 if (up_to_colon == field) {
106 return true;
107 }
108 }
109 return false;
110}
111
112static bool ConvertLine(StringPiece line,
113 const std::vector<string>& multi_line_fields,
114 string* ml) {
115 // Is this a field we should convert?
116 if (!StartsWithFieldName(line, multi_line_fields)) {
117 return false;
118 }
119 // Has a matching field name, so look for "..." after the colon.
120 StringPiece up_to_colon;
121 StringPiece after_colon = line;
122 SplitAt(':', &after_colon, &up_to_colon);
123 while (str_util::ConsumePrefix(&after_colon, " "))
124 ; // Remove leading spaces.
125 if (!str_util::ConsumePrefix(&after_colon, "\"")) {
126 // We only convert string fields, so don't convert this line.
127 return false;
128 }
129 auto last_quote = after_colon.rfind('\"');
130 if (last_quote == StringPiece::npos) {
131 // Error: we don't see the expected matching quote, abort the conversion.
132 return false;
133 }
134 StringPiece escaped = after_colon.substr(0, last_quote);
135 StringPiece suffix = after_colon.substr(last_quote + 1);
136 // We've now parsed line into '<up_to_colon>: "<escaped>"<suffix>'
137
138 string unescaped;
139 if (!str_util::CUnescape(escaped, &unescaped, nullptr)) {
140 // Error unescaping, abort the conversion.
141 return false;
142 }
143 // No more errors possible at this point.
144
145 // Find a string to mark the end that isn't in unescaped.
146 string end = "END";
147 for (int s = 0; unescaped.find(end) != string::npos; ++s) {
148 end = strings::StrCat("END", s);
149 }
150
151 // Actually start writing the converted output.
152 strings::StrAppend(ml, up_to_colon, ": <<", end, "\n", unescaped, "\n", end);
153 if (!suffix.empty()) {
154 // Output suffix, in case there was a trailing comment in the source.
155 strings::StrAppend(ml, suffix);
156 }
157 strings::StrAppend(ml, "\n");
158 return true;
159}

161string PBTxtToMultiline(StringPiece pbtxt,
162 const std::vector<string>& multi_line_fields) {
163 string ml;
164 // Probably big enough, since the input and output are about the
165 // same size, but just a guess.
166 ml.reserve(pbtxt.size() * (17. / 16));
167 StringPiece line;
168 while (!pbtxt.empty()) {
169 // Split pbtxt into its first line and everything after.
170 SplitAt('\n', &pbtxt, &line);
171 // Convert line or output it unchanged
172 if (!ConvertLine(line, multi_line_fields, &ml)) {
173 strings::StrAppend(&ml, line, "\n");
174 }
175 }
176 return ml;
177}

179// Given a single line of text `line` with first : at `colon`, determine if
180// there is an "<<END" expression after the colon and if so return true and set
181// `*end` to everything after the "<<".
182static bool FindMultiline(StringPiece line, size_t colon, string* end) {
183 if (colon == StringPiece::npos) return false;
184 line.remove_prefix(colon + 1);
185 while (str_util::ConsumePrefix(&line, " ")) {
186 }
187 if (str_util::ConsumePrefix(&line, "<<")) {
188 *end = line.ToString();
189 return true;
190 }
191 return false;
192}

194string PBTxtFromMultiline(StringPiece multiline_pbtxt) {
195 string pbtxt;
196 // Probably big enough, since the input and output are about the
197 // same size, but just a guess.
198 pbtxt.reserve(multiline_pbtxt.size() * (33. / 32));
199 StringPiece line;
200 while (!multiline_pbtxt.empty()) {
201 // Split multiline_pbtxt into its first line and everything after.
202 if (!SplitAt('\n', &multiline_pbtxt, &line)) {
203 strings::StrAppend(&pbtxt, line);
204 break;
205 }
206
207 string end;
208 auto colon = line.find(':');
209 if (!FindMultiline(line, colon, &end)) {
210 // Normal case: not a multi-line string, just output the line as-is.
211 strings::StrAppend(&pbtxt, line, "\n");
212 continue;
213 }
214
215 // Multi-line case:
216 // something: <<END
217 // xx
218 // yy
219 // END
220 // Should be converted to:
221 // something: "xx\nyy"
222
223 // Output everything up to the colon (" something:").
224 strings::StrAppend(&pbtxt, line.substr(0, colon + 1));
225
226 // Add every line to unescaped until we see the "END" string.
227 string unescaped;
228 bool first = true;
229 string suffix;
230 while (!multiline_pbtxt.empty()) {
231 SplitAt('\n', &multiline_pbtxt, &line);
232 if (str_util::ConsumePrefix(&line, end)) break;
233 if (first) {
234 first = false;
235 } else {
236 unescaped.push_back('\n');
237 }
238 strings::StrAppend(&unescaped, line);
239 line = StringPiece();
240 }
241
242 // Escape what we extracted and then output it in quotes.
243 strings::StrAppend(&pbtxt, " \"", str_util::CEscape(unescaped), "\"", line,
244 "\n");
245 }
246 return pbtxt;
247}

249static void StringReplace(const string& from, const string& to, string* s) {
250 // Split *s into pieces delimited by `from`.
251 std::vector<string> split;
252 string::size_type pos = 0;
253 while (pos < s->size()) {
254 auto found = s->find(from, pos);
255 if (found == string::npos) {
256 split.push_back(s->substr(pos));
257 break;
258 } else {
259 split.push_back(s->substr(pos, found - pos));
260 pos = found + from.size();
261 if (pos == s->size()) { // handle case where `from` is at the very end.
262 split.push_back("");
263 }
264 }
265 }
266 // Join the pieces back together with a new delimiter.
267 *s = str_util::Join(split, to.c_str());
268}

270static void RenameInDocs(const string& from, const string& to,
271 ApiDef* api_def) {
272 const string from_quoted = strings::StrCat("`", from, "`");
273 const string to_quoted = strings::StrCat("`", to, "`");
274 for (int i = 0; i < api_def->in_arg_size(); ++i) {
275 if (!api_def->in_arg(i).description().empty()) {
276 StringReplace(from_quoted, to_quoted,
277 api_def->mutable_in_arg(i)->mutable_description());
278 }
279 }
280 for (int i = 0; i < api_def->out_arg_size(); ++i) {
281 if (!api_def->out_arg(i).description().empty()) {
282 StringReplace(from_quoted, to_quoted,
283 api_def->mutable_out_arg(i)->mutable_description());
284 }
285 }
286 for (int i = 0; i < api_def->attr_size(); ++i) {
287 if (!api_def->attr(i).description().empty()) {
288 StringReplace(from_quoted, to_quoted,
289 api_def->mutable_attr(i)->mutable_description());
290 }
291 }
292 if (!api_def->summary().empty()) {
293 StringReplace(from_quoted, to_quoted, api_def->mutable_summary());
294 }
295 if (!api_def->description().empty()) {
296 StringReplace(from_quoted, to_quoted, api_def->mutable_description());
297 }

302// Initializes given ApiDef with data in OpDef.
303void InitApiDefFromOpDef(const OpDef& op_def, ApiDef* api_def) {
304 api_def->set_graph_op_name(op_def.name());
305 api_def->set_visibility(ApiDef::VISIBLE);
306
307 auto* endpoint = api_def->add_endpoint();
308 endpoint->set_name(op_def.name());
309 if (op_def.has_deprecation()) {
310 endpoint->set_deprecation_version(op_def.deprecation().version());
311 }
312
313 for (const auto& op_in_arg : op_def.input_arg()) {
314 auto* api_in_arg = api_def->add_in_arg();
315 api_in_arg->set_name(op_in_arg.name());
316 api_in_arg->set_rename_to(op_in_arg.name());
317 api_in_arg->set_description(op_in_arg.description());
318
319 *api_def->add_arg_order() = op_in_arg.name();
320 }
321 for (const auto& op_out_arg : op_def.output_arg()) {
322 auto* api_out_arg = api_def->add_out_arg();
323 api_out_arg->set_name(op_out_arg.name());
324 api_out_arg->set_rename_to(op_out_arg.name());
325 api_out_arg->set_description(op_out_arg.description());
326 }
327 for (const auto& op_attr : op_def.attr()) {
328 auto* api_attr = api_def->add_attr();
329 api_attr->set_name(op_attr.name());
330 api_attr->set_rename_to(op_attr.name());
331 if (op_attr.has_default_value()) {
332 *api_attr->mutable_default_value() = op_attr.default_value();
333 }
334 api_attr->set_description(op_attr.description());
335 }
336 api_def->set_summary(op_def.summary());
337 api_def->set_description(op_def.description());
338}

340// Updates base_arg based on overrides in new_arg.
341void MergeArg(ApiDef::Arg* base_arg, const ApiDef::Arg& new_arg) {
342 if (!new_arg.rename_to().empty()) {
343 base_arg->set_rename_to(new_arg.rename_to());
344 }
345 if (!new_arg.description().empty()) {
346 base_arg->set_description(new_arg.description());
347 }
348}

350// Updates base_attr based on overrides in new_attr.
351void MergeAttr(ApiDef::Attr* base_attr, const ApiDef::Attr& new_attr) {
352 if (!new_attr.rename_to().empty()) {
353 base_attr->set_rename_to(new_attr.rename_to());
354 }
355 if (new_attr.has_default_value()) {
356 *base_attr->mutable_default_value() = new_attr.default_value();
357 }
358 if (!new_attr.description().empty()) {
359 base_attr->set_description(new_attr.description());
360 }

363// Updates base_api_def based on overrides in new_api_def.
364Status MergeApiDefs(ApiDef* base_api_def, const ApiDef& new_api_def) {
365 // Merge visibility
366 if (new_api_def.visibility() != ApiDef::DEFAULT_VISIBILITY) {
367 base_api_def->set_visibility(new_api_def.visibility());
368 }
369 // Merge endpoints
370 if (new_api_def.endpoint_size() > 0) {
371 base_api_def->clear_endpoint();
372 std::copy(
373 new_api_def.endpoint().begin(), new_api_def.endpoint().end(),
374 protobuf::RepeatedFieldBackInserter(base_api_def->mutable_endpoint()));
375 }
376 // Merge args
377 for (const auto& new_arg : new_api_def.in_arg()) {
378 bool found_base_arg = false;
379 for (int i = 0; i < base_api_def->in_arg_size(); ++i) {
380 auto* base_arg = base_api_def->mutable_in_arg(i);
381 if (base_arg->name() == new_arg.name()) {
382 MergeArg(base_arg, new_arg);
383 found_base_arg = true;
384 break;
385 }
386 }
387 if (!found_base_arg) {
388 return errors::FailedPrecondition("Argument ", new_arg.name(),
389 " not defined in base api for ",
390 base_api_def->graph_op_name());
391 }
392 }
393 for (const auto& new_arg : new_api_def.out_arg()) {
394 bool found_base_arg = false;
395 for (int i = 0; i < base_api_def->out_arg_size(); ++i) {
396 auto* base_arg = base_api_def->mutable_out_arg(i);
397 if (base_arg->name() == new_arg.name()) {
398 MergeArg(base_arg, new_arg);
399 found_base_arg = true;
400 break;
401 }
402 }
403 if (!found_base_arg) {
404 return errors::FailedPrecondition("Argument ", new_arg.name(),
405 " not defined in base api for ",
406 base_api_def->graph_op_name());
407 }
408 }
409 // Merge arg order
410 if (new_api_def.arg_order_size() > 0) {
411 // Validate that new arg_order is correct.
412 if (new_api_def.arg_order_size() != base_api_def->arg_order_size()) {
413 return errors::FailedPrecondition(
414 "Invalid number of arguments ", new_api_def.arg_order_size(), " for ",
415 base_api_def->graph_op_name(),
416 ". Expected: ", base_api_def->arg_order_size());
417 }
418 if (!std::is_permutation(new_api_def.arg_order().begin(),
419 new_api_def.arg_order().end(),
420 base_api_def->arg_order().begin())) {
421 return errors::FailedPrecondition(
422 "Invalid arg_order: ", str_util::Join(new_api_def.arg_order(), ", "),
423 " for ", base_api_def->graph_op_name(),
424 ". All elements in arg_order override must match base arg_order: ",
425 str_util::Join(base_api_def->arg_order(), ", "));
426 }
427
428 base_api_def->clear_arg_order();
429 std::copy(
430 new_api_def.arg_order().begin(), new_api_def.arg_order().end(),
431 protobuf::RepeatedFieldBackInserter(base_api_def->mutable_arg_order()));
432 }
433 // Merge attributes
434 for (const auto& new_attr : new_api_def.attr()) {
435 bool found_base_attr = false;
436 for (int i = 0; i < base_api_def->attr_size(); ++i) {
437 auto* base_attr = base_api_def->mutable_attr(i);
438 if (base_attr->name() == new_attr.name()) {
439 MergeAttr(base_attr, new_attr);
440 found_base_attr = true;
441 break;
442 }
443 }
444 if (!found_base_attr) {
445 return errors::FailedPrecondition("Attribute ", new_attr.name(),
446 " not defined in base api for ",
447 base_api_def->graph_op_name());
448 }
449 }
450 // Merge summary
451 if (!new_api_def.summary().empty()) {
452 base_api_def->set_summary(new_api_def.summary());
453 }
454 // Merge description
455 auto description = new_api_def.description().empty()
456 ? base_api_def->description()
457 : new_api_def.description();
458
459 if (!new_api_def.description_prefix().empty()) {
460 description =
461 strings::StrCat(new_api_def.description_prefix(), "\n", description);
462 }
463 if (!new_api_def.description_suffix().empty()) {
464 description =
465 strings::StrCat(description, "\n", new_api_def.description_suffix());
466 }
467 base_api_def->set_description(description);
468 return Status::OK();
469}

489Status ApiDefMap::LoadFile(Env* env, const string& filename) {
490 if (filename.empty()) return Status::OK();
491 string contents;
492 TF_RETURN_IF_ERROR(ReadFileToString(env, filename, &contents));
493 TF_RETURN_IF_ERROR(LoadApiDef(contents));
494 return Status::OK();

497Status ApiDefMap::LoadApiDef(const string& api_def_file_contents) {
498 const string contents = PBTxtFromMultiline(api_def_file_contents);
499 ApiDefs api_defs;
500 protobuf::TextFormat::ParseFromString(contents, &api_defs);
501 for (const auto& api_def : api_defs.op()) {
502 // Check if the op definition is loaded. If op definition is not
503 // loaded, then we just skip this ApiDef.
504 if (map_.find(api_def.graph_op_name()) != map_.end()) {
505 // Overwrite current api def with data in api_def.
506 TF_RETURN_IF_ERROR(MergeApiDefs(&map_[api_def.graph_op_name()], api_def));
507 }
508 }
509 return Status::OK();
510}

512void ApiDefMap::UpdateDocs() {
513 for (auto& name_and_api_def : map_) {
514 auto& api_def = name_and_api_def.second;
515 CHECK_GT(api_def.endpoint_size(), 0);
516 const string canonical_name = api_def.endpoint(0).name();
517 if (api_def.graph_op_name() != canonical_name) {
518 RenameInDocs(api_def.graph_op_name(), canonical_name, &api_def);
519 }
520 for (const auto& in_arg : api_def.in_arg()) {
521 if (in_arg.name() != in_arg.rename_to()) {
522 RenameInDocs(in_arg.name(), in_arg.rename_to(), &api_def);
523 }
524 }
525 for (const auto& out_arg : api_def.out_arg()) {
526 if (out_arg.name() != out_arg.rename_to()) {
527 RenameInDocs(out_arg.name(), out_arg.rename_to(), &api_def);
528 }
529 }
530 for (const auto& attr : api_def.attr()) {
531 if (attr.name() != attr.rename_to()) {
532 RenameInDocs(attr.name(), attr.rename_to(), &api_def);
533 }
534 }
535 }
