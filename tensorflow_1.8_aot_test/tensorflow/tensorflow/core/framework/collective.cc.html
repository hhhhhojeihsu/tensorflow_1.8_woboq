24string CollGroupParams::ToString() const {

28 " num_tasks=", num_tasks, "}");

31CollInstanceParams& CollInstanceParams::operator=(
32 const CollInstanceParams& other) {
33 if (this != &other) {
34 instance_key = other.instance_key;
35 type = other.type;
36 data_type = other.data_type;
37 shape = other.shape;
38 device_names.clear();
39 device_names.assign(other.device_names.begin(), other.device_names.end());
40 task_names.assign(other.task_names.begin(), other.task_names.end());
41 impl_details.subdiv_offsets.assign(
42 other.impl_details.subdiv_offsets.begin(),
43 other.impl_details.subdiv_offsets.end());
44 impl_details.subdiv_permutations.clear();
45 for (auto p : other.impl_details.subdiv_permutations) {
46 impl_details.subdiv_permutations.push_back(
47 std::vector<int>(p.begin(), p.end()));
48 }
49 impl_details.subdiv_source_rank.assign(
50 other.impl_details.subdiv_source_rank.begin(),
51 other.impl_details.subdiv_source_rank.end());
52 }
53 return *this;
54}

56string CollInstanceParams::ToString() const {
57 string v = strings::StrCat("CollInstanceParams { instance_key=", instance_key,
58 " type=", type, " data_type=", data_type,
59 " shape=", shape.DebugString(), " devices {");
60 for (const auto& d : device_names) {
61 strings::StrAppend(&v, d, ",");
62 }
63 strings::StrAppend(&v, "} task_names={");
64 for (const auto& n : task_names) {
65 strings::StrAppend(&v, n, ", ");
66 }
67 strings::StrAppend(&v, "}, subdiv_offsets={");
68 for (const auto& d : impl_details.subdiv_offsets) {
69 strings::StrAppend(&v, d, ",");
70 }
71 strings::StrAppend(&v, "}, subdiv_perms={");
72 for (const auto& p : impl_details.subdiv_permutations) {
73 strings::StrAppend(&v, "{");
74 for (const auto& i : p) {
75 strings::StrAppend(&v, i, ",");
76 }
77 strings::StrAppend(&v, "}"); // one subdiv
78 }
79 strings::StrAppend(&v, "}"); // all subdivs
80 return v;
81}

92string CollectiveParams::ToString() const {
93 string v = strings::StrCat("CollectiveParams ", name, " {", group.ToString());
94 strings::StrAppend(&v, " ", instance.ToString());
95 strings::StrAppend(&v, " ", task.ToString());
96 strings::StrAppend(&v, " default_rank=", default_rank,
97 " is_source=", is_source, " subdiv_rank={");
98 for (const auto& r : subdiv_rank) {
99 strings::StrAppend(&v, r, ",");
100 }
101 if (!subdiv_source_rank.empty()) {
102 strings::StrAppend(&v, " subdiv_rank={");
103 for (const auto& r : subdiv_source_rank) {
104 strings::StrAppend(&v, r, ",");
105 }
106 strings::StrAppend(&v, "}");
107 }
108 strings::StrAppend(&v, "}}");
109 return v;
110}
