26namespace tensorflow {

28// Identifier for a tensor within a step.
29// first == operation_name, second == output_index
30// Note: does not own backing storage for name.
31struct TensorId : public std::pair<StringPiece, int> {
32 typedef std::pair<StringPiece, int> Base;
33
34 // Inherit the set of constructors.
35 using Base::pair;
36
37 string ToString() const {
38 if (second == Graph::kControlSlot) return strings::StrCat("^", first);
39 return strings::StrCat(first, ":", second);
40 }
41
42 struct Hasher {
43 public:
44 std::size_t operator()(const TensorId& x) const {
45 return Hash32(x.first.data(), x.first.size(), x.second);
46 }
47 };
48};
