49// Rewrite graph by replacing the output tensors specified in
50// "fed_outputs" with special feed nodes for each specified output
51// tensor, and removing any nodes that are now disconnected from the
52// part of the graph that reaches the sink node. The set of special
53// feed nodes added to the graph are returned in "*feed_nodes".
54//
55// Return true on success. On error, return false and sets *error to
56// an appropriate error message (and *g is left in an indeterminate
57// state).
58Status FeedInputs(
59 Graph* g, const std::vector<std::unique_ptr<PruneRewrite>>& feed_rewrites,
60 NameIndex* name_index, DataTypeVector* out_feed_types) {
61 out_feed_types->clear();
62 out_feed_types->reserve(feed_rewrites.size());
63 for (size_t i = 0; i < feed_rewrites.size(); ++i) {
64 const string& t = feed_rewrites[i]->endpoint_name();
65 TensorId id(ParseTensorName(t));
66
67 auto iter = name_index->find(id.first);
68 if (iter == name_index->end()) {
69 return errors::NotFound("FeedInputs: unable to find feed output ", t);
70 }
71 Node* n = iter->second;
72 DCHECK_EQ(n->name(), id.first);
73 if (id.second >= n->num_outputs()) {
74 return errors::InvalidArgument(
75 "FeedInputs: ", t, " should have output index < ", n->num_outputs());
76 }
77
78 Node* feed_node;
79 TF_RETURN_IF_ERROR(
80 feed_rewrites[i]->AddNode(g, {n, id.second}, &feed_node));
81
82 // Update name_index
83 (*name_index)[feed_node->name()] = feed_node;
84 g->AddControlEdge(g->source_node(), feed_node);
85
86 // Look through edges coming out of "n" for edges whose src_output() index
87 // matches "output_index". If found, replace the edges with a connection
88 // from the special feed node.
89 std::vector<const Edge*> to_remove;
90 for (const Edge* e : n->out_edges()) {
91 if (e->src_output() == id.second) {
92 to_remove.emplace_back(e);
93 } else if (e->src_output() == Graph::kControlSlot &&
94 (n->type_string() == "Placeholder" ||
95 n->type_string() == "PlaceholderV2")) {
96 // When feeding a Placeholder node, any outgoing control edges
97 // will be replaced with a control edge from the replacement
98 // feed_node.
99 // TODO(josh11b,mrry): Come up with a more elegant way of addressing
100 // the general version of this problem.
101 to_remove.emplace_back(e);
102 }
103 }
104
105 for (const Edge* e : to_remove) {
106 if (e->src_output() == id.second) {
107 g->AddEdge(feed_node, 0, e->dst(), e->dst_input());
108 } else {
109 CHECK_EQ(Graph::kControlSlot, e->src_output());
110 g->AddControlEdge(feed_node, e->dst());
111 }
112 g->RemoveEdge(e);
113 }
114 out_feed_types->push_back(BaseType(n->output_type(id.second)));
115 }
116 return Status::OK();
117}

119Status FetchOutputs(
120 Graph* g, const std::vector<std::unique_ptr<PruneRewrite>>& fetch_rewrites,
121 NameIndex* name_index, std::vector<Node*>* out_fetch_nodes,
122 DataTypeVector* out_fetch_types) {
123 out_fetch_nodes->clear();
124 out_fetch_nodes->reserve(fetch_rewrites.size());
125 for (size_t i = 0; i < fetch_rewrites.size(); ++i) {
126 const string& t = fetch_rewrites[i]->endpoint_name();
127
128 // Parse t into node_name and output_index.
129 TensorId id(ParseTensorName(t));
130
131 // Find node in graph with that name.
132 auto iter = name_index->find(id.first);
133 if (iter == name_index->end()) {
134 return errors::NotFound("FetchOutputs node ", t, ": not found");
135 }
136 Node* n = iter->second;
137 DCHECK_EQ(n->name(), id.first);
138 VLOG(2) << "Found fetch node for " << t;
139
140 // Validate output_index
141 if (n->num_outputs() == 0) {
142 return errors::InvalidArgument(
143 "Tried to fetch data for '", t,
144 "', which produces no output. To run to a node but not fetch any "
145 "data, pass '",
146 t,
147 "' as an argument to the 'target_node_names' argument of the "
148 "Session::Run API.");
149 } else if (id.second >= n->num_outputs()) {
150 return errors::InvalidArgument("FetchOutputs ", t,
151 ": output index too large, must be < ",
152 n->num_outputs());
153 }
154
155 // Create the fetch Node and connect it up
156 Node* fetch_node;
157 TF_RETURN_IF_ERROR(
158 fetch_rewrites[i]->AddNode(g, {n, id.second}, &fetch_node));
159
160 // Update the index.
161 (*name_index)[fetch_node->name()] = fetch_node;
162
163 g->AddControlEdge(fetch_node, g->sink_node());
164 out_fetch_nodes->push_back(fetch_node);
165 out_fetch_types->push_back(BaseType(n->output_type(id.second)));
166 }
167
168 return Status::OK();
169}

171bool AddNodeToTargets(const string& node_or_tensor_name,
172 const NameIndex& name_index,
173 std::unordered_set<const Node*>* targets) {
174 TensorId id = ParseTensorName(node_or_tensor_name);
175 auto iter = name_index.find(id.first);
176 if (iter == name_index.end()) {
177 return false;
178 }
179 const Node* n = iter->second;
180 CHECK_EQ(n->name(), id.first);
181 targets->insert(n);
182 return true;
183}
184
185Status PruneForTargets(Graph* g, const NameIndex& name_index,
186 const std::vector<Node*>& fetch_nodes,
187 const gtl::ArraySlice<string>& target_nodes) {
188 string not_found;
189 std::unordered_set<const Node*> targets;
190 for (Node* n : fetch_nodes) {
191 if (!AddNodeToTargets(n->name(), name_index, &targets)) {
192 strings::StrAppend(&not_found, n->name(), " ");
193 }
194 }
195 for (const string& s : target_nodes) {
196 if (!AddNodeToTargets(s, name_index, &targets)) {
197 strings::StrAppend(&not_found, s, " ");
198 }
199 }
200 if (!not_found.empty()) {
201 return errors::NotFound("PruneForTargets: Some target nodes not found: ",
202 not_found);
203 }
204 PruneForReverseReachability(g, targets);
205
206 // Reconnect nodes with no outgoing edges to the sink node
207 FixupSourceAndSinkEdges(g);
208
209 return Status::OK();
210}

214Status ArgFeedRewrite::AddNode(Graph* g, NodeBuilder::NodeOut feed_tensor,
215 Node** out_node) {
216 // NOTE(mrry): We must include the index as part of the node
217 // name, because _Arg is a "stateful" kernel and therefore
218 // its name must uniquely identify a kernel instance across all
219 // graphs in the same session.
220 TF_RETURN_IF_ERROR(
221 NodeBuilder(strings::StrCat("_arg_", feed_tensor.node->name(), "_",
222 feed_tensor.index, "_", arg_index_),
223 "_Arg")
224 .Attr("T", BaseType(feed_tensor.node->output_type(feed_tensor.index)))
225 .Attr("index", arg_index_)
226 .Finalize(g, out_node));
227 (*out_node)->set_assigned_device_name(device_info().name());
228 return Status::OK();
229}

231Status RecvFeedRewrite::AddNode(Graph* g, NodeBuilder::NodeOut feed_tensor,
232 Node** out_node) {
233 TF_RETURN_IF_ERROR(
234 NodeBuilder(strings::StrCat("_recv_", feed_tensor.node->name(), "_",
235 feed_tensor.index),
236 "_Recv")
237 .Attr("tensor_type",
238 BaseType(feed_tensor.node->output_type(feed_tensor.index)))
239 .Attr("tensor_name", endpoint_name())
240 .Attr("send_device", device_info().name())
241 .Attr("recv_device", device_info().name())
242 .Attr("send_device_incarnation",
243 static_cast<int64>(device_info().incarnation()))
244 .Attr("client_terminated", true)
245 .Finalize(g, out_node));
246
247 (*out_node)->set_assigned_device_name(device_info().name());
248 return Status::OK();
249}

288Status RewriteGraphForExecution(
289 Graph* g, const gtl::ArraySlice<string>& fed_outputs,
290 const gtl::ArraySlice<string>& fetch_outputs,
291 const gtl::ArraySlice<string>& target_node_names,
292 const DeviceAttributes& device_info, bool use_function_convention,
293 RewriteGraphMetadata* out_metadata) {
294 std::vector<std::unique_ptr<PruneRewrite>> feed_rewrites;
295 feed_rewrites.reserve(fed_outputs.size());
296 if (use_function_convention) {
297 for (size_t i = 0; i < fed_outputs.size(); ++i) {
298 feed_rewrites.emplace_back(new ArgFeedRewrite(
299 &fed_outputs[i], &device_info, static_cast<int32>(i)));
300 }
301 } else {
302 for (const string& fed_output : fed_outputs) {
303 feed_rewrites.emplace_back(
304 new RecvFeedRewrite(&fed_output, &device_info));
305 }
306 }
307
308 std::vector<std::unique_ptr<PruneRewrite>> fetch_rewrites;
309 fetch_rewrites.reserve(fetch_outputs.size());
310 if (use_function_convention) {
311 for (size_t i = 0; i < fetch_outputs.size(); ++i) {
312 fetch_rewrites.emplace_back(new RetvalFetchRewrite(
313 &fetch_outputs[i], &device_info, static_cast<int32>(i)));
314 }
315 } else {
316 for (const string& fetch_output : fetch_outputs) {
317 fetch_rewrites.emplace_back(
318 new SendFetchRewrite(&fetch_output, &device_info));
319 }
320 }
321
322 return RewriteGraphForExecution(g, feed_rewrites, fetch_rewrites,
323 target_node_names, out_metadata);
324}

333Status RewriteGraphForExecution(
334 Graph* g, const std::vector<std::unique_ptr<PruneRewrite>>& feed_rewrites,
335 const std::vector<std::unique_ptr<PruneRewrite>>& fetch_rewrites,
336 const gtl::ArraySlice<string>& target_node_names,
337 RewriteGraphMetadata* out_metadata) {
338 if (fetch_rewrites.empty() && target_node_names.empty()) {
339 return errors::InvalidArgument(
340 "Must specify at least one target to fetch or execute.");
341 }
342
343 std::unordered_set<string> endpoints;
344 for (const auto& feed_rewrite : feed_rewrites) {
345 auto result = endpoints.insert(feed_rewrite->endpoint_name());
346 if (!result.second) {
347 return errors::InvalidArgument("Endpoint \"",
348 feed_rewrite->endpoint_name(),
349 "\" fed more than once.");
350 }
351 }
352
353 for (const auto& fetch_rewrite : fetch_rewrites) {
354 if (endpoints.count(fetch_rewrite->endpoint_name()) > 0) {
355 return errors::InvalidArgument(fetch_rewrite->endpoint_name(),
356 " is both fed and fetched.");
357 }
358 }
359
360 // A separate index mapping name to Node*, for use by FeedInputs,
361 // FetchOutputs, and PruneForTargets
362 NameIndex name_index;
363 name_index.reserve(g->num_nodes());
364 for (Node* n : g->nodes()) {
365 name_index[n->name()] = n;
366 }
367
368 // Add the feeds. This may replace nodes in the graph, including the nodes
369 // currently listed in "fetch_rewrites". We pass "name_index" so the index is
370 // kept up to date.
371 if (!feed_rewrites.empty()) {
372 TF_RETURN_IF_ERROR(
373 FeedInputs(g, feed_rewrites, &name_index, &out_metadata->feed_types));
374 }
375
376 // Add the fetch nodes, also updating "name_index".
377 std::vector<Node*> fetch_nodes;
378 if (!fetch_rewrites.empty()) {
379 TF_RETURN_IF_ERROR(FetchOutputs(g, fetch_rewrites, &name_index,
380 &fetch_nodes, &out_metadata->fetch_types));
381 }
382
383 // Prune the graph to only compute what is needed for the fetch nodes and the
384 // target nodes.
385 if (!fetch_nodes.empty() || !target_node_names.empty()) {
386 TF_RETURN_IF_ERROR(
387 PruneForTargets(g, name_index, fetch_nodes, target_node_names));
388 }
389
390 return Status::OK();
391}
