55struct DupRecvKey {
56 int src_node_id; // Edge's src node id
57 int src_output_slot; // Edge's src node output slot
58 GraphDef* dst_graph; // Edge's dst node is in this subgraph
59 bool recv_output_on_host; // The output of recv is on host
60};
61
62struct DupRecvKeyHash {
63 size_t operator()(const DupRecvKey& k) const {
64 size_t h = Hash64(reinterpret_cast<const char*>(&k.src_node_id),
65 sizeof(k.src_node_id), k.src_output_slot);
66 h = Hash64(reinterpret_cast<const char*>(&k.dst_graph), sizeof(k.dst_graph),
67 h);
68 h = Hash64(reinterpret_cast<const char*>(&k.recv_output_on_host),
69 sizeof(k.recv_output_on_host), h);
70 return h;
71 }
72};
73
74struct DupRecvKeyEq {
75 bool operator()(const DupRecvKey& x, const DupRecvKey& y) const {
76 return (x.src_node_id == y.src_node_id) &&
77 (x.src_output_slot == y.src_output_slot) &&
78 (x.dst_graph == y.dst_graph) &&
79 (x.recv_output_on_host == y.recv_output_on_host);
80 }
81};

121// Return true iff we need to add the same device send/recv for 'edge'.
122bool NeedSameDeviceSendRecv(const Edge* edge, const GraphInfo& info) {
123 if (edge->IsControlEdge()) {
124 return false;
125 }
126
127 const Node* src = edge->src();
128 const Node* dst = edge->dst();
129 if (src->assigned_device_name() == dst->assigned_device_name()) {
130 int src_port = edge->src_output();
131 int dst_port = edge->dst_input();
132 if (info.device_types[src->id()] != DEVICE_CPU) {
133 auto src_it = info.output_types.find({src->id(), src_port});
134 DCHECK(src_it != info.output_types.end());
135 auto dst_it = info.input_types.find({dst->id(), dst_port});
136 DCHECK(dst_it != info.input_types.end());
137 return src_it->second != dst_it->second;
138 }
139 }
140 return false;
141}

143// Return true iff (dst, dst_input) is specified on host memory.
144bool IsDstInputOnHost(const Edge* edge, const GraphInfo& info) {
145 const Node* dst = edge->dst();
146 int dst_port = edge->dst_input();
147 if (info.device_types[dst->id()] != DEVICE_CPU) {
148 if (edge->IsControlEdge()) return false;
149 auto dst_it = info.input_types.find({dst->id(), dst_port});
150 DCHECK(dst_it != info.input_types.end());
151 return dst_it->second == HOST_MEMORY;
152 }
153 return true;
154}

190NodeDef* AddSend(const PartitionOptions& opts, const GraphInfo& g_info,
191 GraphDef* gdef, const Edge* edge,
192 NodeDefBuilder::NodeOut send_from, int64 start_time,
193 Status* status) {
194 const DataType dtype = send_from.data_type;
195 const DataType cast_dtype = opts.should_cast ? opts.should_cast(edge) : dtype;
196 const Node* src = edge->src();
197 const int src_port = edge->src_output();
198
199 // host_memory = true iff we need to use HostSend/HostCast.
200 bool host_memory = false;
201 if (!edge->IsControlEdge()) {
202 auto src_it = g_info.output_types.find({src->id(), src_port});
203 DCHECK(src_it != g_info.output_types.end());
204 host_memory = (src_it->second == HOST_MEMORY);
205 }
206
207 // Add a cast node that casts dtype to cast_dtype.
208 // NOTE(yuanbyu): Only cast for cross-device send/recv.
209 if (dtype != cast_dtype && !NeedSameDeviceSendRecv(edge, g_info)) {
210 const string cast_op = (host_memory) ? "_HostCast" : "Cast";
211 NodeDefBuilder cast_builder(opts.new_name(src->name()), cast_op);
212 cast_builder.Device(src->assigned_device_name()).Input(send_from);
213 if (opts.scheduling_for_recvs) {
214 cast_builder.Attr("_start_time", start_time);
215 }
216 cast_builder.Attr("DstT", cast_dtype);
217 NodeDef* cast = gdef->add_node();
218 *status = cast_builder.Finalize(cast);
219 if (!status->ok()) return nullptr;
220
221 // Connect the Send op to the cast.
222 send_from.Reset(cast->name(), 0, cast_dtype);
223 }
224
225 // Add the send node.
226 const string send_op = (host_memory) ? "_HostSend" : "_Send";
227 NodeDefBuilder send_builder(opts.new_name(src->name()), send_op);
228 SetSendRecvAttrs(opts, edge, &send_builder);
229 send_builder.Device(src->assigned_device_name()).Input(send_from);
230 if (opts.scheduling_for_recvs) {
231 send_builder.Attr("_start_time", start_time);
232 }
233 NodeDef* send = gdef->add_node();
234 *status = send_builder.Finalize(send);
235 return send;
236}

238NodeDef* AddRecv(const PartitionOptions& opts, const GraphInfo& g_info,
239 GraphDef* gdef, const Edge* edge, NodeDef** real_recv,
240 Status* status) {
241 const DataType dtype = EdgeType(edge);
242 const Node* src = edge->src();
243 const Node* dst = edge->dst();
244 const int dst_port = edge->dst_input();
245 DataType cast_dtype = dtype;
246
247 // NOTE(yuanbyu): Only cast for cross-device send/recv.
248 if (opts.should_cast && !NeedSameDeviceSendRecv(edge, g_info)) {
249 cast_dtype = opts.should_cast(edge);
250 }
251
252 // host_memory = true iff we need to use HostRecv/HostCast.
253 bool host_memory = false;
254 if (!edge->IsControlEdge()) {
255 auto dst_it = g_info.input_types.find({dst->id(), dst_port});
256 DCHECK(dst_it != g_info.input_types.end());
257 host_memory = (dst_it->second == HOST_MEMORY);
258 }
259
260 // Add the recv node.
261 const string recv_op = (host_memory) ? "_HostRecv" : "_Recv";
262 NodeDefBuilder recv_builder(opts.new_name(src->name()), recv_op);
263 SetSendRecvAttrs(opts, edge, &recv_builder);
264 recv_builder.Device(dst->assigned_device_name())
265 .Attr("tensor_type", cast_dtype);
266 NodeDef* recv = gdef->add_node();
267 *status = recv_builder.Finalize(recv);
268 if (!status->ok()) return nullptr;
269 *real_recv = recv;
270
271 // Add the cast node (from cast_dtype to dtype) or an Identity node.
272 if (dtype != cast_dtype) {
273 const string cast_op = (host_memory) ? "_HostCast" : "Cast";
274 NodeDefBuilder cast_builder(opts.new_name(src->name()), cast_op);
275 cast_builder.Attr("DstT", dtype);
276 cast_builder.Device(dst->assigned_device_name())
277 .Input(recv->name(), 0, cast_dtype);
278 NodeDef* cast = gdef->add_node();
279 *status = cast_builder.Finalize(cast);
280 if (!status->ok()) return nullptr;
281 return cast;
282 } else if (edge->IsControlEdge()) {
283 // An Identity is only needed for control edges.
284 NodeDefBuilder id_builder(opts.new_name(src->name()), "Identity");
285 id_builder.Device(dst->assigned_device_name())
286 .Input(recv->name(), 0, cast_dtype);
287 NodeDef* id = gdef->add_node();
288 *status = id_builder.Finalize(id);
289 if (!status->ok()) return nullptr;
290 return id;
291 } else {
292 return recv;
293 }
294}

322// Optimize colocation for control flow nodes. For cond, we want the
323// switch nodes to colocate with its data input. This is particularly
324// needed for conditional reading of a remote variable. It may also
325// reduce the number of devices involved in a loop.
326// TODO(yuanbyu): In this case, we don't respect the requested device in
327// the GraphDef for these nodes. Ideally, the placer would enforce the
328// colocation to render this unnecessary.
329void OptimizeControlFlowColocation(Graph* graph) {
330 auto visit = [](Node* node) {
331 if (IsSwitch(node)) {
332 for (const Edge* in_edge : node->in_edges()) {
333 if (in_edge->dst_input() == 0) {
334 // Colocate with the data input.
335 node->set_assigned_device_name(
336 in_edge->src()->assigned_device_name());
337 return;
338 }
339 }
340 } else if (IsExit(node)) {
341 for (const Edge* in_edge : node->in_edges()) {
342 if (!in_edge->IsControlEdge()) {
343 // Colocate with upstream node.
344 node->set_assigned_device_name(
345 in_edge->src()->assigned_device_name());
346 return;
347 }
348 }
349 } else {
350 if ((IsEnter(node) && !IsRefType(node->input_type(0))) ||
351 IsNextIteration(node)) {
352 const Edge* data_edge = nullptr;
353 for (const Edge* out_edge : node->out_edges()) {
354 if (!out_edge->IsControlEdge()) {
355 data_edge = out_edge;
356 break;
357 }
358 }
359 // Colocate with the first downstream data node.
360 if (data_edge) {
361 node->set_assigned_device_name(
362 data_edge->dst()->assigned_device_name());
363 }
364 }
365 }
366 };
367 DFS(*graph, visit, {});
368}

457// Add the control flow info of a new node added during partitioning.
458// The new node has the same control flow info as src.
459void AddControlFlowInfo(const Node* node, const Node* src,
460 std::vector<ControlFlowInfo>* cf_info) {
461 int id = node->id();
462 if (static_cast<size_t>(id) >= cf_info->size()) {
463 cf_info->resize(id + 1);
464 }
465 const ControlFlowInfo& src_info = (*cf_info)[src->id()];
466 ControlFlowInfo* info = &(*cf_info)[id];
467 info->frame = src_info.frame;
468 info->parent_frame = src_info.parent_frame;
469 info->frame_name = src_info.frame_name;
470}

472// Constructs a control loop. Returns a struct containing the newly created
473// enter, merge, and switch nodes. The enter and merge nodes are used in the
474// recursive construction of control loops for nested frames (loops). The
475// switch node will be connected to the LoopCond node. The merge node will
476// be connected to all the recvs of the same frame by control edges when
477// the actual partitioning happens.
478Status AddControlLoop(const PartitionOptions& opts, Graph* g, const Node* src,
479 const Edge* edge, Node* loop_cond,
480 std::vector<ControlFlowInfo>* cf_info,
481 ControlLoop* loop) {
482 Status status;
483 GraphDefBuilder::Options bopts(g, &status);
484 const ControlFlowInfo& src_info = (*cf_info)[src->id()];
485 const string& device_name = edge->dst()->assigned_device_name();
486 const string& frame_name = src_info.frame_name;
487 int parallel_iterations;
488 status = GetNodeAttr(src_info.frame->attrs(), "parallel_iterations",
489 &parallel_iterations);
490 if (!status.ok()) return status;
491
492 // The names of the nodes to be added.
493 const string& enter_name =
494 ControlLoopName(opts.new_name(edge->dst()->name()));
495 const string& merge_name =
496 ControlLoopName(opts.new_name(edge->dst()->name()));
497 const string& switch_name =
498 ControlLoopName(opts.new_name(edge->dst()->name()));
499 const string& next_name = ControlLoopName(opts.new_name(edge->dst()->name()));
500
501 // Add the nodes to the graph g.
502 Node* enter = AddControlEnter(g, enter_name, device_name, frame_name,
503 parallel_iterations, &status);
504 if (!status.ok()) return status;
505 Node* merge = AddControlMerge(enter_name, next_name, g, merge_name,
506 device_name, &status);
507 if (!status.ok()) return status;
508 Node* switch_node = AddControlSwitch(merge, loop_cond, device_name,
509 bopts.WithName(switch_name));
510 if (!status.ok()) return status;
511 Node* next =
512 AddControlNext({switch_node, 1}, device_name, bopts.WithName(next_name));
513 if (!status.ok()) return status;
514
515 // Add control flow info for these new nodes:
516 AddControlFlowInfo(enter, src, cf_info);
517 AddControlFlowInfo(merge, src, cf_info);
518 AddControlFlowInfo(switch_node, src, cf_info);
519 AddControlFlowInfo(next, src, cf_info);
520
521 // Add input edges for the newly created merge node:
522 g->AddEdge(enter, 0, merge, 0);
523 g->AddEdge(next, 0, merge, 1);
524
525 loop->enter = enter;
526 loop->merge = merge;
527 loop->switch_node = switch_node;
528 return Status::OK();
529}

531// Build memory and device type info for every node in the graph.
532// TODO(yuanbyu): It might be simpler if we convert MemoryType to
533// DeviceType for the inputs/outputs of each node.
534Status BuildMemoryDeviceInfo(const Graph& g, GraphInfo* info) {
535 MemoryTypeVector input_memory_types;
536 MemoryTypeVector output_memory_types;
537
538 info->device_types.resize(g.num_node_ids(), DEVICE_CPU);
539 for (const Node* node : g.op_nodes()) {
540 DeviceNameUtils::ParsedName parsed;
541 if (!DeviceNameUtils::ParseFullName(node->assigned_device_name(),
542 &parsed)) {
543 return errors::Internal("Malformed assigned device '",
544 node->assigned_device_name(), "'");
545 }
546
547 TF_RETURN_IF_ERROR(MemoryTypesForNode(
548 g.op_registry(), DeviceType(parsed.type), node->def(),
549 &input_memory_types, &output_memory_types));
550
551 int node_id = node->id();
552 info->device_types[node_id] = DeviceType(parsed.type);
553 for (size_t i = 0; i < input_memory_types.size(); ++i) {
554 info->input_types[{node_id, i}] = input_memory_types[i];
555 }
556 for (size_t i = 0; i < output_memory_types.size(); ++i) {
557 info->output_types[{node_id, i}] = output_memory_types[i];
558 }
559 }
560 return Status::OK();
561}

563const Node* InputFrame(const Node* node,
564 const std::vector<ControlFlowInfo>& cf_info) {
565 // An input is in the same frame as the node except for Enter nodes.
566 // The input of Enter is in the parent frame of the Enter node.
567 if (!node->IsEnter()) {
568 return node;
569 }
570 return cf_info[node->id()].parent_frame;
571}

573const Node* OutputFrame(const Node* node,
574 const std::vector<ControlFlowInfo>& cf_info) {
575 // An output is in the same frame as the node except for Exit nodes.
576 // The output of Exit is in the parent frame of the Exit node.
577 if (!node->IsExit()) {
578 return node;
579 }
580 return cf_info[node->id()].parent_frame;

583// Each participating device needs to decide a) if there is a next iteration,
584// and b) if the loop terminates. We take the approach to encode this control
585// flow logic in the dataflow graph. There are at least two possible encodings.
586// In a completely decentralized encoding, the participants communicate peer
587// to peer. The other encoding uses a frame leader (the participant who owns
588// the pivot termination predicate) to broadcast the termination condition to
589// all the participants. For now we take the latter because it is simpler.
590//
591// TODO(yuanbyu): The correctness of this construction is rather subtle. I got
592// it wrong many times so it would be nice to write a proof to be sure.
593Status AddControlFlow(const PartitionOptions& opts, Graph* g,
594 GraphInfo* g_info) {
595 Status status;
596 GraphDefBuilder::Options bopts(g, &status);
597 std::vector<ControlFlowInfo>& cf_info = g_info->cf_info;
598
599 // Build the control flow info for every node.
600 status = BuildControlFlowInfo(g, &cf_info);
601 if (!status.ok()) return status;
602
603 OptimizeControlFlowColocation(g);
604
605 // The map from frames to their LoopCond nodes.
606 std::unordered_map<string, Node*> frame_cond_map;
607 int num_node_ids = g->num_node_ids();
608 for (int i = 0; i < num_node_ids; ++i) {
609 Node* node = g->FindNodeId(i);
610 if (node == nullptr) continue;
611
612 if (IsLoopCond(node)) {
613 const string& frame_name = cf_info[node->id()].frame_name;
614 DCHECK(!frame_name.empty());
615 frame_cond_map[frame_name] = node;
616 }
617 }
618
619 // Add all control loops for cross-device frames.
620 // A control loop is added only when there is a cross-device edge in a
621 // non-root frame. Nothing is added if there is no loops. We also don't
622 // add anything for a frame that is completely local to a device. For
623 // nested loops, we stack the control loops together by connecting
624 // the merge of the outer loop to the enter of the inner loop.
625 //
626 // A map from <frame_name, device_name> to ControlLoop.
627 std::unordered_map<string, ControlLoop> control_loops;
628 int num_edge_ids = g->num_edge_ids();
629 for (int i = 0; i < num_edge_ids; ++i) {
630 const Edge* edge = g->FindEdgeId(i);
631 if (edge == nullptr) continue;
632
633 const Node* src = edge->src();
634 const Node* dst = edge->dst();
635 // Skip Sink/Source nodes.
636 if (!src->IsOp() || !dst->IsOp()) continue;
637
638 const string& src_device = src->assigned_device_name();
639 const string& dst_device = dst->assigned_device_name();
640 // Skip local edges.
641 if (src_device == dst_device) continue;
642
643 const Node* src_frame = OutputFrame(src, cf_info);
644 const Node* dst_frame = InputFrame(dst, cf_info);
645 const string& src_frame_name = cf_info[src_frame->id()].frame_name;
646 const string& dst_frame_name = cf_info[dst_frame->id()].frame_name;
647 // Skip if src and dst are not in the same frame.
648 if (src_frame_name.empty() || src_frame_name != dst_frame_name) {
649 continue;
650 }
651
652 // Add the control loop. Start by adding the control loop for the
653 // current frame if needed, and recursively adding the control loop
654 // for its outer frame when nested.
655 ControlLoop child_loop;
656 while (true) {
657 const string& curr_frame_name = cf_info[src_frame->id()].frame_name;
658 if (curr_frame_name.empty()) {
659 // We have reached the root frame.
660 if (child_loop.merge != nullptr) {
661 const string& node_name = opts.new_name(edge->dst()->name());
662 const string& device_name = edge->dst()->assigned_device_name();
663 Node* const_node =
664 AddControlConst(device_name, bopts.WithName(node_name));
665 if (!status.ok()) return status;
666 AddControlFlowInfo(const_node, src_frame, &cf_info);
667 g->AddEdge(const_node, 0, child_loop.enter, 0);
668 }
669 break;
670 }
671
672 const string& cl_key = strings::StrCat(curr_frame_name, "$$", dst_device);
673 auto it = control_loops.find(cl_key);
674 if (it != control_loops.end()) {
675 if (child_loop.enter != nullptr) {
676 g->AddEdge(it->second.merge, 0, child_loop.enter, 0);
677 }
678 break;
679 }
680
681 // Get the frame's LoopCond.
682 auto cond_it = frame_cond_map.find(curr_frame_name);
683 if (cond_it == frame_cond_map.end()) {
684 return errors::InvalidArgument(
685 "A cross-device loop must have a pivot predicate: ",
686 curr_frame_name);
687 }
688 Node* loop_cond = cond_it->second;
689
690 // Add the control loop.
691 ControlLoop curr_loop;
692 status = AddControlLoop(opts, g, src_frame, edge, loop_cond, &cf_info,
693 &curr_loop);
694 if (!status.ok()) return status;
695 control_loops[cl_key] = curr_loop;
696
697 if (child_loop.enter != nullptr) {
698 // Connect the merge of the outer loop to the enter of the inner.
699 g->AddEdge(curr_loop.merge, 0, child_loop.enter, 0);
700 }
701 src_frame = cf_info[src_frame->id()].parent_frame;
702 child_loop = curr_loop;
703 }
704 }
705
706 // For a cross-device edge, on the dst device, add a control edge
707 // from the merge node of the control loop to dst. If a send/recv is
708 // introduced for this edge in future partitioning, we delete this
709 // control edge and add a new control edge from the merge to the recv.
710 num_edge_ids = g->num_edge_ids();
711 for (int i = 0; i < num_edge_ids; ++i) {
712 const Edge* edge = g->FindEdgeId(i);
713 if (edge == nullptr) continue;
714
715 const Node* src = edge->src();
716 Node* dst = edge->dst();
717 // Skip Sink/Source nodes.
718 if (!src->IsOp() || !dst->IsOp()) continue;
719
720 const string& src_device = src->assigned_device_name();
721 const string& dst_device = dst->assigned_device_name();
722 if (src_device != dst_device) {
723 const Node* src_frame = OutputFrame(src, cf_info);
724 const Node* dst_frame = InputFrame(dst, cf_info);
725 const string& src_frame_name = cf_info[src_frame->id()].frame_name;
726 const string& dst_frame_name = cf_info[dst_frame->id()].frame_name;
727 if (!src_frame_name.empty() && src_frame_name == dst_frame_name) {
728 const string& cl_key =
729 strings::StrCat(dst_frame_name, "$$", dst_device);
730 ControlLoop loop = control_loops[cl_key];
731 DCHECK(loop.enter != nullptr);
732 // Note that we'll create multiple duplicate edges if dst has multiple
733 // cross-device inputs. This is expected by the logic in Partition(), so
734 // it can add control edges to the recv nodes once they're created.
735 g->AddControlEdge(loop.merge, dst, /*allow_duplicates=*/true);
736 }
737 }
738 }
739 return Status::OK();
740}

742struct PriorityTopoSortNode {
743 PriorityTopoSortNode(const NodeDef* n, int64 st) : node(n), start_time(st) {}
744
745 const NodeDef* node;
746 int64 start_time;
747};
748
749struct PriorityTopoSortNodeGreater {
750 bool operator()(const PriorityTopoSortNode& left,
751 const PriorityTopoSortNode& right) {
752 return left.start_time > right.start_time;
753 }
754};

758// Returns in <nodes> the nodes that should participate in epoch-based recv
759// scheduling, along with their times; <nodes> is ordered by increasing
760// start_time. Returns in <node_to_start_time_out> the timing for all nodes,
761// even those not in <nodes>.
762//
763// Comparing to sorting on the node's start time only, this also processes the
764// nodes in dependency order, and updates start times to ensure a node's
765// start_time > the start time for all dependencies.
766//
767// Note that graph_partition_test.cc accesses this function for testing, even
768// though it's not declared in the header.
769Status TopologicalSortNodesWithTimePriority(
770 const GraphDef* gdef, std::vector<std::pair<const NodeDef*, int64>>* nodes,
771 std::unordered_map<const NodeDef*, int64>* node_to_start_time_out) {
772 // Queue of nodes to process; lowest start time is returned first.
773 std::priority_queue<PriorityTopoSortNode, std::vector<PriorityTopoSortNode>,
774 PriorityTopoSortNodeGreater>
775 q;
776 std::unordered_map<const NodeDef*, int64> node_to_start_time;
777 auto enqueue = [&q, &node_to_start_time](const NodeDef* node) {
778 const int64 start_time = node_to_start_time[node];
779 q.emplace(node, start_time);
780 };
781
782 // Build initial structures, initial contents of queue.
783 std::unordered_map<string, std::vector<const NodeDef*>> node_to_output_nodes;
784 std::unordered_map<const NodeDef*, int> inputs_needed;
785 for (int n = 0; n < gdef->node_size(); ++n) {
786 const NodeDef* ndef = &gdef->node(n);
787 for (int i = 0; i < ndef->input_size(); ++i) {
788 node_to_output_nodes[ParseTensorName(ndef->input(i)).first.ToString()]
789 .push_back(ndef);
790 }
791 int64 start_time;
792 TF_RETURN_IF_ERROR(GetNodeAttr(*ndef, "_start_time", &start_time));
793 node_to_start_time[ndef] = start_time;
794 inputs_needed[ndef] = ndef->input_size();
795 if (ndef->input_size() == 0) {
796 enqueue(ndef);
797 }
798 }
799
800 // Determine which merge nodes are parts of loops; these
801 // need to happen in the traversal after all non-NextIteration inputs
802 // are run.
803 for (int n = 0; n < gdef->node_size(); ++n) {
804 const NodeDef* ndef = &gdef->node(n);
805 if (IsNextIteration(*ndef)) {
806 for (const NodeDef* n : node_to_output_nodes[ndef->name()]) {
807 if (IsMerge(*n)) {
808 // n is a merge that is part of a loop structure.
809 // It doesn't need to wait for this NextIteration loop
810 // when doing the traversal.
811 --inputs_needed[n];
812 }
813 }
814 }
815 }
816
817 // Traverse.
818 std::vector<std::pair<const NodeDef*, int64>> start_times;
819 start_times.reserve(gdef->node_size());
820 while (!q.empty()) {
821 PriorityTopoSortNode cur = q.top();
822 q.pop();
823
824 start_times.emplace_back(cur.node, cur.start_time);
825
826 for (const NodeDef* n : node_to_output_nodes[cur.node->name()]) {
827 auto& output_start_time = node_to_start_time[n];
828 if (output_start_time <= cur.start_time) {
829 output_start_time = cur.start_time + 1;
830 }
831 if (--inputs_needed[n] == 0) {
832 enqueue(n);
833 }
834 }
835 }
836
837 // Done.
838 nodes->swap(start_times);
839 node_to_start_time_out->swap(node_to_start_time);
840 return Status::OK();
841}

843Status AddControlEdges(const PartitionOptions& opts,
844 std::unordered_map<string, GraphDef>* partitions) {
845 Status status;
846 // TODO(yuanbyu): Very naive for now. To be improved.
847 const int num_epochs = 100;
848 const int prefetch = 6;
849
850 for (auto& part : *partitions) {
851 GraphDef* gdef = &part.second;
852 std::vector<std::pair<const NodeDef*, int64>> start_times;
853 std::unordered_map<const NodeDef*, int64> node_to_start_time;
854 status = TopologicalSortNodesWithTimePriority(gdef, &start_times,
855 &node_to_start_time);
856 if (!status.ok()) {
857 return status;
858 }
859
860 // Add a dummy node for every epoch, and add a control edge from the
861 // "last" node in the preceding epoch to the dummy node.
862 string device_name = gdef->node(0).device();
863 int64 makespan = start_times.back().second;
864 int64 resolution = (makespan / num_epochs) + 1;
865
866 int i = 0;
867 int j = 0;
868 std::vector<NodeDef*> dummys;
869 while (i < num_epochs && static_cast<size_t>(j) < start_times.size()) {
870 if (i * resolution > start_times[j].second) {
871 j++;
872 } else {
873 NodeDef* dummy = AddControlTrigger(opts, gdef, device_name, i,
874 i * resolution, &status);
875 if (!status.ok()) {
876 return status;
877 }
878 dummys.push_back(dummy);
879 if (j > 0) {
880 string src_name = start_times[j - 1].first->name();
881 AddInput(dummy, src_name, Graph::kControlSlot);
882 }
883 i++;
884 }
885 }
886
887 // Finally, add the control edges to recvs.
888 for (int n = 0; n < gdef->node_size(); ++n) {
889 NodeDef* ndef = gdef->mutable_node(n);
890 if (ndef->op() == "_Recv") {
891 const int64 start_time = node_to_start_time[ndef];
892 const int recv_epoch = start_time / resolution;
893 if (recv_epoch >= prefetch) {
894 NodeDef* dummy = dummys[recv_epoch - prefetch];
895 AddInput(ndef, dummy->name(), Graph::kControlSlot);
896 }
897 }
898 }
899 }
900 return Status::OK();
901}

938Status Partition(const PartitionOptions& opts, Graph* g,
939 std::unordered_map<string, GraphDef>* partitions) {
940 Status status;
941 partitions->clear();
942
943 GraphInfo g_info;
944 if (!opts.control_flow_added) {
945 // Add the "code" for distributed execution of control flow. Code is
946 // added only for the frames that are placed on multiple devices. The
947 // new graph is an equivalent transformation of the original graph and
948 // has the property that it can be subsequently partitioned arbitrarily
949 // (down to the level of individual device) for distributed execution.
950 status = AddControlFlow(opts, g, &g_info);
951 if (!status.ok()) return status;
952 }
953
954 // At this point, all the graph mutations have been done. Build memory
955 // and device type info for every node and edge in the graph.
956 status = BuildMemoryDeviceInfo(*g, &g_info);
957 if (!status.ok()) return status;
958
959 string dstp;
960 std::vector<const Edge*> inputs;
961 DupRecvTable dup_recv(3);
962 // For a node dst, 'ref_recvs' remembers the recvs introduced by a ref
963 // edge to dst. 'ref_control_inputs' remembers the inputs by a non-ref
964 // edge to dst. We will add a control edge for every pair in
965 // (ref_recvs x ref_control_inputs).
966 std::vector<NodeDef*> ref_recvs;
967 std::vector<string> ref_control_inputs;
968
969 int32 num_data = 0;
970 int32 num_control = 0;
971 for (const Node* dst : g->op_nodes()) {
972 dstp = opts.node_to_loc(dst);
973 GraphDef* dst_graph = &(*partitions)[dstp];
974 NodeDef* dst_def = dst_graph->add_node();
975 *dst_def = dst->def();
976 dst_def->set_device(dst->assigned_device_name());
977 dst_def->clear_input(); // Inputs are filled below
978 if (opts.need_to_record_start_times) {
979 int64 start_time;
980 status = GetNodeAttr(*dst_def, "_start_time", &start_time);
981 if (errors::IsNotFound(status)) {
982 start_time = opts.start_times[dst->id()].value();
983 AddNodeAttr("_start_time", start_time, dst_def);
984 } else if (!status.ok()) {
985 return status;
986 }
987 }
988
989 // Arrange the incoming edges to dst so that input[i] holds the
990 // input flowing into slot numbered i. Trailing entries in input[]
991 // hold control edges.
992 inputs.clear();
993 inputs.resize(dst->num_inputs(), nullptr);
994 ref_recvs.clear();
995 ref_control_inputs.clear();
996 const Edge* control_flow_edge = nullptr;
997 int32 num_control_flow_edges = 0;
998 int32 num_input_edges = 0;
999 for (const Edge* edge : dst->in_edges()) {
1000 if (edge->IsControlEdge()) {
1001 if (IsMerge(edge->src()) && IsControlLoop(edge->src())) {
1002 // This is one of the control edges added for control flow. There
1003 // can be multiple such edges as the dest node may have multiple
1004 // remote inputs. We keep track of the number of such edges.
1005 control_flow_edge = edge;
1006 ++num_control_flow_edges;
1007 } else {
1008 inputs.push_back(edge);
1009 }
1010 } else {
1011 DCHECK(inputs[edge->dst_input()] == nullptr);
1012 inputs[edge->dst_input()] = edge;
1013 ++num_input_edges;
1014 }
1015 }
1016
1017 if (num_input_edges != dst->num_inputs()) {
1018 return errors::InvalidArgument("Incomplete graph, missing ",
1019 (dst->num_inputs() - num_input_edges),
1020 " inputs for ", dst->name());
1021 }
1022
1023 // Process in order so that all data edges are added as inputs to
1024 // dst in Edge::dst_input() order.
1025 for (const Edge* edge : inputs) {
1026 const Node* src = edge->src();
1027 if (!src->IsOp()) continue; // Skip Sink/Source nodes.
1028
1029 GraphDef* src_graph = &(*partitions)[opts.node_to_loc(src)];
1030 if (src_graph == dst_graph && !NeedSameDeviceSendRecv(edge, g_info)) {
1031 // Same partition and compatible memory types:
1032 AddInput(dst_def, src->name(), edge->src_output());
1033 if (edge->IsControlEdge() ||
1034 !IsRefType(src->output_type(edge->src_output()))) {
1035 ref_control_inputs.push_back(src->name());
1036 }
1037 continue;
1038 }
1039
1040 int64 send_start_time = 0;
1041 int64 recv_start_time = 0;
1042 if (opts.scheduling_for_recvs) {
1043 status = GetNodeAttr(src->attrs(), "_start_time", &send_start_time);
1044 if (errors::IsNotFound(status) && opts.need_to_record_start_times) {
1045 send_start_time = opts.start_times[src->id()].value();
1046 } else if (!status.ok()) {
1047 return status;
1048 }
1049
1050 status = GetNodeAttr(dst->attrs(), "_start_time", &recv_start_time);
1051 if (errors::IsNotFound(status) && opts.need_to_record_start_times) {
1052 recv_start_time = opts.start_times[dst->id()].value();
1053 } else if (!status.ok()) {
1054 return status;
1055 }
1056 }
1057
1058 // Check whether there is already a send/recv pair transferring
1059 // the same tensor/control from the src to dst partition.
1060 const bool on_host = IsDstInputOnHost(edge, g_info);
1061 DupRecvKey key{src->id(), edge->src_output(), dst_graph, on_host};
1062 auto iter = dup_recv.find(key);
1063 if (iter != dup_recv.end()) {
1064 // We found one. Reuse the data/control transferred already.
1065 const string& recv_node_name = iter->second.recv->name();
1066 if (edge->IsControlEdge()) {
1067 AddInput(dst_def, recv_node_name, Graph::kControlSlot);
1068 } else {
1069 AddInput(dst_def, recv_node_name, 0);
1070 }
1071 ref_control_inputs.push_back(recv_node_name);
1072
1073 // We want the start_time for the recv to be the smallest of the start
1074 // times of it's consumers. So we update this whenever we use a recv,
1075 // and write it out to the attribute at the end of the subroutine
1076 if (iter->second.start_time > recv_start_time) {
1077 iter->second.start_time = recv_start_time;
1078 }
1079 continue;
1080 }
1081
1082 NodeDefBuilder::NodeOut send_from;
1083 if (edge->IsControlEdge()) {
1084 // Insert a dummy const node that will generate a tiny
1085 // data element to be sent from send to recv.
1086 VLOG(1) << "Send/Recv control: " << src->assigned_device_name() << "["
1087 << src->name() << "] -> " << dst->assigned_device_name() << "["
1088 << dst->name() << "]";
1089 NodeDef* dummy = AddDummyConst(opts, src_graph, edge, &status);
1090 if (!status.ok()) return status;
1091 // Set the start time for this dummy node.
1092 if (opts.scheduling_for_recvs) {
1093 AddNodeAttr("_start_time", send_start_time, dummy);
1094 }
1095 AddInput(dummy, src->name(), Graph::kControlSlot);
1096 send_from.Reset(dummy->name(), 0, DT_FLOAT);
1097 } else {
1098 send_from.Reset(src->name(), edge->src_output(), EdgeType(edge));
1099 }
1100
1101 // Need to split edge by placing matching send/recv nodes on
1102 // the src/dst sides of the edge.
1103 NodeDef* send = AddSend(opts, g_info, src_graph, edge, send_from,
1104 send_start_time, &status);
1105 if (!status.ok()) return status;
1106
1107 NodeDef* real_recv = nullptr;
1108 NodeDef* recv =
1109 AddRecv(opts, g_info, dst_graph, edge, &real_recv, &status);
1110 if (!status.ok()) return status;
1111
1112 // Fix up the control flow edge.
1113 // NOTE(yuanbyu): 'real_recv' must be the real recv node.
1114 if (src_graph == dst_graph) {
1115 // For same device send/recv, add a control edge from send to recv.
1116 // This prevents the asynchronous recv kernel from being scheduled
1117 // before the data is available.
1118 AddInput(real_recv, send->name(), Graph::kControlSlot);
1119 } else if (control_flow_edge != nullptr) {
1120 // Redirect control edge to the real recv since this is not the same
1121 // device send/recv.
1122 --num_control_flow_edges;
1123 AddInput(real_recv, control_flow_edge->src()->name(),
1124 Graph::kControlSlot);
1125 }
1126
1127 if (!edge->IsControlEdge() &&
1128 IsRefType(src->output_type(edge->src_output()))) {
1129 AddNodeAttr("_start_time", recv_start_time, recv);
1130 if (real_recv != recv) {
1131 AddNodeAttr("_start_time", recv_start_time, real_recv);
1132 }
1133 // If src is of ref type and the edge is not a control edge, dst has
1134 // read semantics and therefore we must control the recv.
1135 ref_recvs.push_back(real_recv);
1136 } else {
1137 // Memorize the send/recv pair, only if this is not a "ref" edge.
1138 // NOTE(yuanbyu): Collapsing ref edges requires extreme care so
1139 // for now we don't do it.
1140 dup_recv[key] = {recv, real_recv, recv_start_time};
1141 ref_control_inputs.push_back(recv->name());
1142 }
1143
1144 if (edge->IsControlEdge()) {
1145 ++num_control;
1146 AddInput(dst_def, recv->name(), Graph::kControlSlot);
1147 } else {
1148 ++num_data;
1149 AddInput(dst_def, recv->name(), 0);
1150 }
1151 }
1152
1153 // Add control edges from 'ref_control_inputs' to 'ref_recvs'.
1154 // NOTE(yuanbyu): Adding these control edges should not introduce
1155 // deadlocks. 'dst' has implicit "read" nodes that, when we split
1156 // across devices, are made explicit; Retargeting the dependencies
1157 // to 'dst' to those nodes would not introduce cycles if there isn't
1158 // one before the transformation.
1159 // NOTE(yuanbyu): This may impact performance because it defers the
1160 // execution of recvs until all the other inputs become available.
1161 AddReadControl(ref_recvs, ref_control_inputs);
1162
1163 // Add back the control edges for control flow that are not used.
1164 if (control_flow_edge != nullptr) {
1165 for (int i = 0; i < num_control_flow_edges; ++i) {
1166 AddInput(dst_def, control_flow_edge->src()->name(),
1167 Graph::kControlSlot);
1168 }
1169 }
1170 }
1171
1172 const FunctionLibraryDefinition* flib_def = opts.flib_def;
1173 if (flib_def == nullptr) {
1174 flib_def = &g->flib_def();
1175 }
1176
1177 // Set versions, function library and send/recv incarnation.
1178 for (auto& it : *partitions) {
1179 GraphDef* gdef = &it.second;
1180 *gdef->mutable_versions() = g->versions();
1181 *gdef->mutable_library() = flib_def->ToProto();
1182
1183 // Traverse the graph to fill every send/recv op's incarnation
1184 // information.
1185 SetIncarnation(opts, gdef);
1186 }
1187
1188 // Set the start times for recvs at the very end.
1189 if (opts.scheduling_for_recvs) {
1190 for (auto& it : dup_recv) {
1191 AddNodeAttr("_start_time", it.second.start_time, it.second.recv);
1192 if (it.second.real_recv != it.second.recv) {
1193 AddNodeAttr("_start_time", it.second.start_time, it.second.real_recv);
1194 }
1195 }
1196 }
1197
1198 VLOG(1) << "Added send/recv: controls=" << num_control
1199 << ", data=" << num_data;
1200 return Status::OK();
1201}
