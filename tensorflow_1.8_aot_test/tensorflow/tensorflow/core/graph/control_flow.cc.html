27Status BuildControlFlowInfo(const Graph* g,
28 std::vector<ControlFlowInfo>* info) {
29 info->clear();
30 info->resize(g->num_node_ids());
31
32 std::vector<const Node*> parent_nodes;
33 parent_nodes.resize(g->num_node_ids());
34
35 const Node* src_node = g->source_node();
36 ControlFlowInfo& src_info = (*info)[src_node->id()];
37 src_info.frame = src_node;
38 src_info.parent_frame = src_node;
39
40 string frame_name;
41 std::deque<const Node*> ready;
42 ready.push_back(src_node);
43 while (!ready.empty()) {
44 const Node* curr_node = ready.front();
45 ready.pop_front();
46 const ControlFlowInfo& curr_info = (*info)[curr_node->id()];
47 const Node* frame = curr_info.frame;
48 const Node* parent = curr_info.parent_frame;
49 frame_name = curr_info.frame_name;
50
51 if (IsExit(curr_node)) {
52 // Exit to the parent frame.
53 const ControlFlowInfo& parent_info = (*info)[parent->id()];
54 frame = parent_info.frame;
55 parent = parent_info.parent_frame;
56 frame_name = parent_info.frame_name;
57 }
58
59 for (const Edge* out_edge : curr_node->out_edges()) {
60 const Node* out = out_edge->dst();
61 int out_id = out->id();
62 ControlFlowInfo* out_info = &(*info)[out_id];
63 const Node* out_parent = out_info->parent_frame;
64 bool is_visited = (parent_nodes[out_id] != nullptr);
65
66 // Skip Sink/Source nodes.
67 if (!out->IsOp()) continue;
68
69 // Add to ready queue if not seen.
70 if (!is_visited) {
71 parent_nodes[out->id()] = curr_node;
72 ready.push_back(out);
73 }
74
75 // Process the node 'out'.
76 if (IsEnter(out)) {
77 if (is_visited) {
78 const string& parent_frame = (*info)[out_parent->id()].frame_name;
79 if (parent_frame != frame_name) {
80 return errors::InvalidArgument(
81 "The node '", out->name(),
82 "' has inputs from different "
83 "frames. The input '",
84 curr_node->name(), "' is in frame '", frame_name,
85 "'. The input '", parent_nodes[out->id()]->name(),
86 "' is in frame '", parent_frame, "'.");
87 }
88 } else {
89 out_info->frame = out;
90 out_info->parent_frame = frame;
91 TF_RETURN_IF_ERROR(
92 GetNodeAttr(out->attrs(), "frame_name", &out_info->frame_name));
93 if (out_info->frame_name.empty()) {
94 return errors::InvalidArgument("The Enter node ", out->name(),
95 " must have a frame name.");
96 }
97 }
98 } else {
99 if (is_visited) {
100 if (out_info->frame_name != frame_name) {
101 return errors::InvalidArgument(
102 "The node '", out->name(),
103 "' has inputs from different "
104 "frames. The input '",
105 curr_node->name(), "' is in frame '", frame_name,
106 "'. The input '", parent_nodes[out->id()]->name(),
107 "' is in frame '", out_info->frame_name, "'.");
108 }
109 } else {
110 out_info->frame = frame;
111 out_info->parent_frame = parent;
112 out_info->frame_name = frame_name;
113 }
114 }
115 }
116 }
117 return Status::OK();
118}
