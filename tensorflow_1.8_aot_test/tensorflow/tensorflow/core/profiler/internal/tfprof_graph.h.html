38namespace tensorflow {
39namespace tfprof {
40
41// Organize tensorflow ops in a graph structure, pointing from output ops
42// to input ops.
43class TFGraph : public TFShow {
44 public:
45 explicit TFGraph(checkpoint::CheckpointReader* ckpt_reader)
46 : TFShow(ckpt_reader), root_(nullptr) {}
47 ~TFGraph() override {}
48
49 void AddNode(TFGraphNode* node) override;
50
51 void Build() override;
52
53 private:
54 const ShowNode* ShowInternal(const Options& opts,
55 Timeline* timeline) override;
56
57 bool ShouldShowIfExtra(const ShowNode* node, const Options& opts,
58 int depth) const override {
59 return true;
60 }
61
62 GraphNode* CreateParentNode(const string& name);
63
64 std::vector<GraphNode*> SearchRoot(const std::vector<GraphNode*>& roots,
65 const std::vector<string>& regexes,
66 std::set<string>* visited);
67
68 std::vector<GraphNode*> PrintGraph(const std::vector<GraphNode*> roots,
69 const Options& opts, int depth,
70 int last_ident, std::set<string>* visits);
71
72 std::vector<GraphNode*> Account(const std::vector<GraphNode*>& roots,
73 const Options& opts,
74 std::set<string>* visits);
75
76 void Format(const std::vector<GraphNode*> roots, string* display_str,
77 GraphNodeProto* proto);
78
79 MemoryTracker memory_tracker_;
80 GraphNode* root_;
81 std::vector<std::unique_ptr<NodeDef>> node_defs_;
82 std::map<string, std::unique_ptr<TFGraphNode>> parent_nodes_;
83 std::map<string, std::unique_ptr<GraphNode>> nodes_map_;
84};
