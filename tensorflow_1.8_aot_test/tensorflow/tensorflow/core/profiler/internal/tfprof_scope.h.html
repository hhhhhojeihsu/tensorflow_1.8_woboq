16// Build a tree structure based on the TensorFlow op names.
17// For example, 'name1/name2' is a child of 'name1'.
18// Stats are aggregated from descendants to ancestors.
19

37namespace tensorflow {
38namespace tfprof {
39
40class TFScope : public TFShow {
41 public:
42 explicit TFScope(checkpoint::CheckpointReader* ckpt_reader)
43 : TFShow(ckpt_reader), root_(nullptr) {}
44 ~TFScope() override {}
45
46 void AddNode(TFGraphNode* node) override;
47
48 void Build() override;
49
50 private:
51 const ShowNode* ShowInternal(const Options& opts,
52 Timeline* timeline) override;
53
54 ScopeNode* CreateParentNode(const string& name);
55
56 std::vector<ScopeNode*> SearchRoot(std::vector<ScopeNode*> roots,
57 const std::vector<string>& regexes);
58
59 std::vector<ScopeNode*> PrintScope(const std::vector<ScopeNode*> roots,
60 const Options& opts, int depth,
61 int last_ident);
62
63 std::vector<ScopeNode*> Account(const std::vector<ScopeNode*>& roots,
64 const Options& opts);
65
66 void Format(const std::vector<ScopeNode*> roots, string* display_str,
67 GraphNodeProto* proto);
68
69 ScopeNode* root_;
70 std::vector<std::unique_ptr<NodeDef>> node_defs_;
71 std::map<string, std::unique_ptr<TFGraphNode>> parent_nodes_;
72 std::map<string, std::unique_ptr<ScopeNode>> nodes_map_;
73};
