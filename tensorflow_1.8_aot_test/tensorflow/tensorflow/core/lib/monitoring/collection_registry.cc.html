24void Collector::CollectMetricValues(
25 const CollectionRegistry::CollectionInfo& info) {
26 info.collection_function(MetricCollectorGetter(
27 this, info.metric_def, info.registration_time_millis));
28}

30std::unique_ptr<CollectedMetrics> Collector::ConsumeCollectedMetrics() {
31 mutex_lock l(mu_);
32 return std::move(collected_metrics_);
33}
34
35void Collector::CollectMetricDescriptor(
36 const AbstractMetricDef* const metric_def) {
37 auto* const metric_descriptor = [&]() {
38 mutex_lock l(mu_);
39 return collected_metrics_->metric_descriptor_map
40 .insert(std::make_pair(
41 metric_def->name().ToString(),
42 std::unique_ptr<MetricDescriptor>(new MetricDescriptor())))
43 .first->second.get();
44 }();
45 metric_descriptor->name = metric_def->name().ToString();
46 metric_descriptor->description = metric_def->description().ToString();
47
48 for (const StringPiece label_name : metric_def->label_descriptions()) {
49 metric_descriptor->label_names.push_back(label_name.ToString());
50 }
51
52 metric_descriptor->metric_kind = metric_def->kind();
53 metric_descriptor->value_type = metric_def->value_type();
54}

65CollectionRegistry::CollectionRegistry(Env* const env) : env_(env) {}
66
67std::unique_ptr<CollectionRegistry::RegistrationHandle>
68CollectionRegistry::Register(const AbstractMetricDef* const metric_def,
69 const CollectionFunction& collection_function) {
70 CHECK(collection_function)
71 << "Requires collection_function to contain an implementation.";
72
73 mutex_lock l(mu_);
74
75 const auto found_it = registry_.find(metric_def->name());
76 if (found_it != registry_.end()) {
77 LOG(FATAL) << "Cannot register 2 metrics with the same name: "
78 << metric_def->name();
79 }
80 registry_.insert(
81 {metric_def->name(),
82 {metric_def, collection_function, env_->NowMicros() / 1000}});
83
84 return std::unique_ptr<RegistrationHandle>(
85 new RegistrationHandle(this, metric_def));
86}
87
88void CollectionRegistry::Unregister(const AbstractMetricDef* const metric_def) {
89 mutex_lock l(mu_);
90 registry_.erase(metric_def->name());
91}
92
93std::unique_ptr<CollectedMetrics> CollectionRegistry::CollectMetrics(
94 const CollectMetricsOptions& options) const {
95 internal::Collector collector(env_->NowMicros() / 1000);
96
97 mutex_lock l(mu_);
98 for (const auto& registration : registry_) {
99 if (options.collect_metric_descriptors) {
100 collector.CollectMetricDescriptor(registration.second.metric_def);
101 }
102
103 collector.CollectMetricValues(registration.second /* collection_info */);
104 }
105 return collector.ConsumeCollectedMetrics();
106}
