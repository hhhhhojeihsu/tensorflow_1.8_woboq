23uint32 SimplePhilox::Uniform(uint32 n) {
24 return ExactUniformInt<uint32>(n, [this]() { return Rand32(); });
25}
26
27uint64 SimplePhilox::Uniform64(uint64 n) {
28 return ExactUniformInt<uint64>(n, [this]() { return Rand64(); });
29}
30
31uint32 SimplePhilox::Skewed(int max_log) {
32 CHECK(0 <= max_log && max_log <= 32);
33 const int shift = Rand32() % (max_log + 1);
34 const uint32 mask = shift == 32 ? ~static_cast<uint32>(0) : (1 << shift) - 1;
35 return Rand32() & mask;
36}
