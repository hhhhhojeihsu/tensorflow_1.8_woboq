16#ifndef TENSORFLOW_LIB_IO_BLOCK_BUILDER_H_
17#define TENSORFLOW_LIB_IO_BLOCK_BUILDER_H_
18
19#include <vector>
20
21#include <stdint.h>
22#include "tensorflow/core/lib/core/stringpiece.h"
23
24namespace tensorflow {
25namespace table {
26
27struct Options;
28
29class BlockBuilder {
30 public:
31 explicit BlockBuilder(const Options* options);
32
33 // Reset the contents as if the BlockBuilder was just constructed.
34 void Reset();
35
36 // REQUIRES: Finish() has not been called since the last call to Reset().
37 // REQUIRES: key is larger than any previously added key
38 void Add(const StringPiece& key, const StringPiece& value);
39
40 // Finish building the block and return a slice that refers to the
41 // block contents. The returned slice will remain valid for the
42 // lifetime of this builder or until Reset() is called.
43 StringPiece Finish();
44
45 // Returns an estimate of the current (uncompressed) size of the block
46 // we are building.
47 size_t CurrentSizeEstimate() const;
48
49 // Return true iff no entries have been added since the last Reset()
50 bool empty() const { return buffer_.empty(); }
51
52 private:
53 const Options* options_;
54 string buffer_; // Destination buffer
55 std::vector<uint32> restarts_; // Restart points
56 int counter_; // Number of entries emitted since restart
57 bool finished_; // Has Finish() been called?
58 string last_key_;
59
60 // No copying allowed
61 BlockBuilder(const BlockBuilder&);
62 void operator=(const BlockBuilder&);
63};
64
65} // namespace table
66} // namespace tensorflow
67
68#endif // TENSORFLOW_LIB_IO_BLOCK_BUILDER_H_
