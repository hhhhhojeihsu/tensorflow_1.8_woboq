30// The AlphaNum type was designed to be used as the parameter type for StrCat().
31// Any routine accepting either a string or a number may accept it.
32// The basic idea is that by accepting a "const AlphaNum &" as an argument
33// to your function, your callers will automatically convert bools, integers,
34// and floating point values to strings for you.
35//
36// NOTE: Use of AlphaNum outside of the //strings package is unsupported except
37// for the specific case of function parameters of type "AlphaNum" or "const
38// AlphaNum &". In particular, instantiating AlphaNum directly as a stack
39// variable is not supported.
40//
41// Conversion from 8-bit values is not accepted because if it were, then an
42// attempt to pass ':' instead of ":" might result in a 58 ending up in your
43// result.
44//
45// Bools convert to "0" or "1".
46//
47// Floating point values are converted to a string which, if passed to strtod(),
48// would produce the exact same original double (except in case of NaN; all NaNs
49// are considered the same value). We try to keep the string short but it's not
50// guaranteed to be as short as possible.
51//
52// You can convert to Hexadecimal output rather than Decimal output using Hex.
53// To do this, pass strings::Hex(my_int) as a parameter to StrCat. You may
54// specify a minimum field width using a separate parameter, so the equivalent
55// of Printf("%04x", my_int) is StrCat(Hex(my_int, strings::ZERO_PAD_4))
56//
57// This class has implicit constructors.
58namespace tensorflow {
59namespace strings {
60
61enum PadSpec {
62 NO_PAD = 1,
63 ZERO_PAD_2,
64 ZERO_PAD_3,
65 ZERO_PAD_4,
66 ZERO_PAD_5,
67 ZERO_PAD_6,
68 ZERO_PAD_7,
69 ZERO_PAD_8,
70 ZERO_PAD_9,
71 ZERO_PAD_10,
72 ZERO_PAD_11,
73 ZERO_PAD_12,
74 ZERO_PAD_13,
75 ZERO_PAD_14,
76 ZERO_PAD_15,
77 ZERO_PAD_16,
78};
79
80struct Hex {
81 uint64 value;
82 enum PadSpec spec;
83 template <class Int>
84 explicit Hex(Int v, PadSpec s = NO_PAD) : spec(s) {
85 // Prevent sign-extension by casting integers to
86 // their unsigned counterparts.
87 static_assert(
88 sizeof(v) == 1 || sizeof(v) == 2 || sizeof(v) == 4 || sizeof(v) == 8,
89 "Unknown integer type");
90 value = sizeof(v) == 1
91 ? static_cast<uint8>(v)
92 : sizeof(v) == 2 ? static_cast<uint16>(v)
93 : sizeof(v) == 4 ? static_cast<uint32>(v)
94 : static_cast<uint64>(v);
95 }
96};
97
98class AlphaNum {
99 public:
100 // No bool ctor -- bools convert to an integral type.
101 // A bool ctor would also convert incoming pointers (bletch).
102
103 AlphaNum(int i32) // NOLINT(runtime/explicit)
104 : piece_(digits_, FastInt32ToBufferLeft(i32, digits_)) {}
105 AlphaNum(unsigned int u32) // NOLINT(runtime/explicit)
106 : piece_(digits_, FastUInt32ToBufferLeft(u32, digits_)) {}
107 AlphaNum(long x) // NOLINT(runtime/explicit)
108 : piece_(digits_, FastInt64ToBufferLeft(x, digits_)) {}
109 AlphaNum(unsigned long x) // NOLINT(runtime/explicit)
110 : piece_(digits_, FastUInt64ToBufferLeft(x, digits_)) {}
111 AlphaNum(long long int i64) // NOLINT(runtime/explicit)
112 : piece_(digits_, FastInt64ToBufferLeft(i64, digits_)) {}
113 AlphaNum(unsigned long long int u64) // NOLINT(runtime/explicit)
114 : piece_(digits_, FastUInt64ToBufferLeft(u64, digits_)) {}
115
116 AlphaNum(float f) // NOLINT(runtime/explicit)
117 : piece_(digits_, FloatToBuffer(f, digits_)) {}
118 AlphaNum(double f) // NOLINT(runtime/explicit)
119 : piece_(digits_, DoubleToBuffer(f, digits_)) {}
120
121 AlphaNum(Hex hex); // NOLINT(runtime/explicit)
122
123 AlphaNum(const char *c_str) : piece_(c_str) {} // NOLINT(runtime/explicit)
124 AlphaNum(const StringPiece &pc) : piece_(pc) {} // NOLINT(runtime/explicit)
125 AlphaNum(const tensorflow::string &str) // NOLINT(runtime/explicit)
126 : piece_(str) {}
127
128 StringPiece::size_type size() const { return piece_.size(); }
129 const char *data() const { return piece_.data(); }
130 StringPiece Piece() const { return piece_; }
131
132 private:
133 StringPiece piece_;
134 char digits_[kFastToBufferSize];
135
136 // Use ":" not ':'
137 AlphaNum(char c); // NOLINT(runtime/explicit)
138
139 TF_DISALLOW_COPY_AND_ASSIGN(AlphaNum);
140};

142// ----------------------------------------------------------------------
143// StrCat()
144// This merges the given strings or numbers, with no delimiter. This
145// is designed to be the fastest possible way to construct a string out
146// of a mix of raw C strings, StringPieces, strings, bool values,
147// and numeric values.
148//
149// Don't use this for user-visible strings. The localization process
150// works poorly on strings built up out of fragments.
151//
152// For clarity and performance, don't use StrCat when appending to a
153// string. In particular, avoid using any of these (anti-)patterns:
154// str.append(StrCat(...))
155// str += StrCat(...)
156// str = StrCat(str, ...)
157// where the last is the worse, with the potential to change a loop
158// from a linear time operation with O(1) dynamic allocations into a
159// quadratic time operation with O(n) dynamic allocations. StrAppend
160// is a better choice than any of the above, subject to the restriction
161// of StrAppend(&str, a, b, c, ...) that none of the a, b, c, ... may
162// be a reference into str.
163// ----------------------------------------------------------------------
164
165// For performance reasons, we have specializations for <= 4 args.
166string StrCat(const AlphaNum &a) TF_MUST_USE_RESULT;
167string StrCat(const AlphaNum &a, const AlphaNum &b) TF_MUST_USE_RESULT;
168string StrCat(const AlphaNum &a, const AlphaNum &b,
169 const AlphaNum &c) TF_MUST_USE_RESULT;
170string StrCat(const AlphaNum &a, const AlphaNum &b, const AlphaNum &c,
171 const AlphaNum &d) TF_MUST_USE_RESULT;

173namespace internal {

175// Do not call directly - this is not part of the public API.
176string CatPieces(std::initializer_list<StringPiece> pieces);
177void AppendPieces(string *dest, std::initializer_list<StringPiece> pieces);

179} // namespace internal

181// Support 5 or more arguments
182template <typename... AV>
183string StrCat(const AlphaNum &a, const AlphaNum &b, const AlphaNum &c,
184 const AlphaNum &d, const AlphaNum &e,
185 const AV &... args) TF_MUST_USE_RESULT;
186
187template <typename... AV>
188string StrCat(const AlphaNum &a, const AlphaNum &b, const AlphaNum &c,
189 const AlphaNum &d, const AlphaNum &e, const AV &... args) {
190 return internal::CatPieces({a.Piece(), b.Piece(), c.Piece(), d.Piece(),
191 e.Piece(),
192 static_cast<const AlphaNum &>(args).Piece()...});
193}
194
195// ----------------------------------------------------------------------
196// StrAppend()
197// Same as above, but adds the output to the given string.
198// WARNING: For speed, StrAppend does not try to check each of its input
199// arguments to be sure that they are not a subset of the string being
200// appended to. That is, while this will work:
201//
202// string s = "foo";
203// s += s;
204//
205// This will not (necessarily) work:
206//
207// string s = "foo";
208// StrAppend(&s, s);
209//
210// Note: while StrCat supports appending up to 26 arguments, StrAppend
211// is currently limited to 9. That's rarely an issue except when
212// automatically transforming StrCat to StrAppend, and can easily be
213// worked around as consecutive calls to StrAppend are quite efficient.
214// ----------------------------------------------------------------------
215
216void StrAppend(string *dest, const AlphaNum &a);
217void StrAppend(string *dest, const AlphaNum &a, const AlphaNum &b);
218void StrAppend(string *dest, const AlphaNum &a, const AlphaNum &b,
219 const AlphaNum &c);
220void StrAppend(string *dest, const AlphaNum &a, const AlphaNum &b,
221 const AlphaNum &c, const AlphaNum &d);
222
223// Support 5 or more arguments
224template <typename... AV>
225inline void StrAppend(string *dest, const AlphaNum &a, const AlphaNum &b,
226 const AlphaNum &c, const AlphaNum &d, const AlphaNum &e,
227 const AV &... args) {
228 internal::AppendPieces(dest,
229 {a.Piece(), b.Piece(), c.Piece(), d.Piece(), e.Piece(),
230 static_cast<const AlphaNum &>(args).Piece()...});
231}
