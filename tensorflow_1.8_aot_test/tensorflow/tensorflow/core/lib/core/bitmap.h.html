16#ifndef TENSORFLOW_CORE_LIB_CORE_BITMAP_H_
17#define TENSORFLOW_CORE_LIB_CORE_BITMAP_H_
18
19#include "tensorflow/core/platform/logging.h"
20#include "tensorflow/core/platform/types.h"
21
22namespace tensorflow {
23namespace core {
24
25class Bitmap {
26 public:
27 // Create a bitmap that holds 0 bits.
28 Bitmap();
29
30 // Create a bitmap that holds n bits, all initially zero.
31 explicit Bitmap(size_t n);
32
33 ~Bitmap();
34
35 Bitmap(const Bitmap&) = delete;
36 Bitmap& operator=(const Bitmap&) = delete;
37
38 // Return the number of bits that the bitmap can hold.
39 size_t bits() const;
40
41 // Replace contents of *this with a bitmap of n bits, all set to zero.
42 void Reset(size_t n);
43
44 // Return the contents of the ith bit.
45 // REQUIRES: i < bits()
46 bool get(size_t i) const;
47
48 // Set the contents of the ith bit to true.
49 // REQUIRES: i < bits()
50 void set(size_t i);
51
52 // Set the contents of the ith bit to false.
53 // REQUIRES: i < bits()
54 void clear(size_t i);
55
56 // Return the smallest i such that i >= start and !get(i).
57 // Returns bits() if no such i exists.
58 size_t FirstUnset(size_t start) const;
59
60 // Returns the bitmap as an ascii string of '0' and '1' characters, bits()
61 // characters in length.
62 string ToString() const;
63
64 private:
65 typedef uint32 Word;
66 static const size_t kBits = 32;
67
68 // Return the number of words needed to store n bits.
69 static size_t NumWords(size_t n) { return (n + kBits - 1) / kBits; }
70
71 // Return the mask to use for the ith bit in a word.
72 static Word Mask(size_t i) { return 1ull << i; }
73
74 size_t nbits_; // Length of bitmap in bits.
75 Word* word_;
76};
77
78// Implementation details follow. Clients should ignore.
79
80inline Bitmap::Bitmap() : nbits_(0), word_(nullptr) {}
81
82inline Bitmap::Bitmap(size_t n) : Bitmap() { Reset(n); }
83
84inline Bitmap::~Bitmap() { delete[] word_; }
85
86inline size_t Bitmap::bits() const { return nbits_; }
87
88inline bool Bitmap::get(size_t i) const {
89 DCHECK_LT(i, nbits_);
90 return word_[i / kBits] & Mask(i % kBits);
91}
92
93inline void Bitmap::set(size_t i) {
94 DCHECK_LT(i, nbits_);
95 word_[i / kBits] |= Mask(i % kBits);
96}
97
98inline void Bitmap::clear(size_t i) {
99 DCHECK_LT(i, nbits_);
100 word_[i / kBits] &= ~Mask(i % kBits);
101}
102
103} // namespace core
104} // namespace tensorflow
