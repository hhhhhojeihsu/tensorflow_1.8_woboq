23std::ostream& operator<<(std::ostream& o, StringPiece piece) {
24 o.write(piece.data(), piece.size());
25 return o;
26}

28bool StringPiece::contains(StringPiece s) const {
29 return std::search(begin(), end(), s.begin(), s.end()) != end();
30}
31
32size_t StringPiece::find(char c, size_t pos) const {
33 if (pos >= size_) {
34 return npos;
35 }
36 const char* result =
37 reinterpret_cast<const char*>(memchr(data_ + pos, c, size_ - pos));
38 return result != nullptr ? result - data_ : npos;
39}
40
41// Search range is [0..pos] inclusive. If pos == npos, search everything.
42size_t StringPiece::rfind(char c, size_t pos) const {
43 if (size_ == 0) return npos;
44 for (const char* p = data_ + std::min(pos, size_ - 1); p >= data_; p--) {
45 if (*p == c) {
46 return p - data_;
47 }
48 }
49 return npos;
50}
51
52StringPiece StringPiece::substr(size_t pos, size_t n) const {
53 if (pos > size_) pos = size_;
54 if (n > size_ - pos) n = size_ - pos;
55 return StringPiece(data_ + pos, n);
56}
