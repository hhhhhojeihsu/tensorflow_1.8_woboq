33// The DECLARE_ERROR macro below only supports types that can be converted
34// into StrCat's AlphaNum. For the other types we rely on a slower path
35// through std::stringstream. To add support of a new type, it is enough to
36// make sure there is an operator<<() for it:
37//
38// std::ostream& operator<<(std::ostream& os, const MyType& foo) {
39// os << foo.ToString();
40// return os;
41// }
42// Eventually absl::strings will have native support for this and we will be
43// able to completely remove PrepareForStrCat().
44template <typename T>
45typename std::enable_if<!std::is_convertible<T, strings::AlphaNum>::value,
46 string>::type
47PrepareForStrCat(const T& t) {
48 std::stringstream ss;
49 ss << t;
50 return ss.str();
51}
52inline const strings::AlphaNum& PrepareForStrCat(const strings::AlphaNum& a) {
53 return a;
54}

58// Append some context to an error message. Each time we append
59// context put it on a new line, since it is possible for there
60// to be several layers of additional context.
61template <typename... Args>
62void AppendToMessage(::tensorflow::Status* status, Args... args) {
63 *status = ::tensorflow::Status(
64 status->code(),
65 ::tensorflow::strings::StrCat(status->error_message(), "\n\t", args...));
66}

68// For propagating errors when calling a function.
69#define TF_RETURN_IF_ERROR(...) \
70 do { \
71 const ::tensorflow::Status _status = (__VA_ARGS__); \
72 if (TF_PREDICT_FALSE(!_status.ok())) return _status; \
73 } while (0)
74
75#define TF_RETURN_WITH_CONTEXT_IF_ERROR(expr, ...) \
76 do { \
77 ::tensorflow::Status _status = (expr); \
78 if (TF_PREDICT_FALSE(!_status.ok())) { \
79 ::tensorflow::errors::AppendToMessage(&_status, __VA_ARGS__); \
80 return _status; \
81 } \
82 } while (0)

84// Convenience functions for generating and using error status.
85// Example usage:
86// status.Update(errors::InvalidArgument("The ", foo, " isn't right."));
87// if (errors::IsInvalidArgument(status)) { ... }
88// switch (status.code()) { case error::INVALID_ARGUMENT: ... }
89
90#define DECLARE_ERROR(FUNC, CONST) \
91 template <typename... Args> \
92 ::tensorflow::Status FUNC(Args... args) { \
93 return ::tensorflow::Status( \
94 ::tensorflow::error::CONST, \
95 ::tensorflow::strings::StrCat( \
96 ::tensorflow::errors::internal::PrepareForStrCat(args)...)); \
97 } \
98 inline bool Is##FUNC(const ::tensorflow::Status& status) { \
99 return status.code() == ::tensorflow::error::CONST; \
100 }
