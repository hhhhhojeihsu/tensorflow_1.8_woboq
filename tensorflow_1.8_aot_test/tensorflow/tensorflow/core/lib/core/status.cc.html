21Status::Status(tensorflow::error::Code code, StringPiece msg) {
22 assert(code != tensorflow::error::OK);
23 state_ = std::unique_ptr<State>(new State);
24 state_->code = code;
25 state_->msg = msg.ToString();
26}
27
28void Status::Update(const Status& new_status) {
29 if (ok()) {
30 *this = new_status;
31 }
32}
33
34void Status::SlowCopyFrom(const State* src) {
35 if (src == nullptr) {
36 state_ = nullptr;
37 } else {
38 state_ = std::unique_ptr<State>(new State(*src));
39 }
40}

42const string& Status::empty_string() {

47string Status::ToString() const {
48 if (state_ == nullptr) {
49 return "OK";
50 } else {
51 char tmp[30];
52 const char* type;
53 switch (code()) {
54 case tensorflow::error::CANCELLED:
55 type = "Cancelled";
56 break;
57 case tensorflow::error::UNKNOWN:
58 type = "Unknown";
59 break;
60 case tensorflow::error::INVALID_ARGUMENT:
61 type = "Invalid argument";
62 break;
63 case tensorflow::error::DEADLINE_EXCEEDED:
64 type = "Deadline exceeded";
65 break;
66 case tensorflow::error::NOT_FOUND:
67 type = "Not found";
68 break;
69 case tensorflow::error::ALREADY_EXISTS:
70 type = "Already exists";
71 break;
72 case tensorflow::error::PERMISSION_DENIED:
73 type = "Permission denied";
74 break;
75 case tensorflow::error::UNAUTHENTICATED:
76 type = "Unauthenticated";
77 break;
78 case tensorflow::error::RESOURCE_EXHAUSTED:
79 type = "Resource exhausted";
80 break;
81 case tensorflow::error::FAILED_PRECONDITION:
82 type = "Failed precondition";
83 break;
84 case tensorflow::error::ABORTED:
85 type = "Aborted";
86 break;
87 case tensorflow::error::OUT_OF_RANGE:
88 type = "Out of range";
89 break;
90 case tensorflow::error::UNIMPLEMENTED:
91 type = "Unimplemented";
92 break;
93 case tensorflow::error::INTERNAL:
94 type = "Internal";
95 break;
96 case tensorflow::error::UNAVAILABLE:
97 type = "Unavailable";
98 break;
99 case tensorflow::error::DATA_LOSS:
100 type = "Data loss";
101 break;
102 default:
103 snprintf(tmp, sizeof(tmp), "Unknown code(%d)",
104 static_cast<int>(code()));
105 type = tmp;
106 break;
107 }
108 string result(type);
109 result += ": ";
110 result += state_->msg;
111 return result;
112 }
113}

124string* TfCheckOpHelperOutOfLine(const ::tensorflow::Status& v,
125 const char* msg) {
126 string r("Non-OK-status: ");
127 r += msg;
128 r += " status: ";
129 r += v.ToString();
130 // Leaks string but this is only to be used in a fatal error message
131 return new string(r);
132}
