16// StringPiece is a simple structure containing a pointer into some external
17// storage and a size. The user of a StringPiece must ensure that the slice
18// is not used after the corresponding external storage has been
19// deallocated.
20//
21// Multiple threads can invoke const methods on a StringPiece without
22// external synchronization, but if any of the threads may call a
23// non-const method, all threads accessing the same StringPiece must use
24// external synchronization.
25
26#ifndef TENSORFLOW_LIB_CORE_STRINGPIECE_H_
27#define TENSORFLOW_LIB_CORE_STRINGPIECE_H_
28
29#include <assert.h>
30#include <stddef.h>
31#include <string.h>
32#include <iosfwd>
33#include <string>
34#include "tensorflow/core/platform/types.h"
35
36namespace tensorflow {
37
38class StringPiece {
39 public:
40 typedef size_t size_type;
41
42 // Create an empty slice.
43 StringPiece() : data_(nullptr), size_(0) {}
44
45 // Create a slice that refers to d[0,n-1].
46 StringPiece(const char* d, size_t n) : data_(d), size_(n) {}
47
48 // Create a slice that refers to the contents of "s"
49 StringPiece(const string& s) : data_(s.data()), size_(s.size()) {}
50
51 // Create a slice that refers to s[0,strlen(s)-1]
52 StringPiece(const char* s) : data_(s), size_(strlen(s)) {}
53
54 // Return a pointer to the beginning of the referenced data
55 const char* data() const { return data_; }
56
57 // Return the length (in bytes) of the referenced data
58 size_t size() const { return size_; }
59
60 // Return true iff the length of the referenced data is zero
61 bool empty() const { return size_ == 0; }
62
63 typedef const char* const_iterator;
64 typedef const char* iterator;
65 iterator begin() const { return data_; }
66 iterator end() const { return data_ + size_; }
67
68 static const size_t npos = size_type(-1);
69
70 // Return the ith byte in the referenced data.
71 // REQUIRES: n < size()
72 char operator[](size_t n) const {
73 assert(n < size());
74 return data_[n];
75 }
76
77 // Drop the first "n" bytes from this slice.
78 void remove_prefix(size_t n) {
79 assert(n <= size());
80 data_ += n;
81 size_ -= n;
82 }
83
84 void remove_suffix(size_t n) {
85 assert(size_ >= n);
86 size_ -= n;
87 }
88
89 size_t find(char c, size_t pos = 0) const;
90 size_t rfind(char c, size_t pos = npos) const;
91 // DEPRECATED: Use tensorflow::str_util::StrContains instead.
92 bool contains(StringPiece s) const;
93
94 // Checks whether StringPiece starts with x and if so advances the beginning
95 // of it to past the match. It's basically a shortcut for starts_with
96 // followed by remove_prefix.
97 // DEPRECATED: Use tensorflow::str_util::ConsumePrefix instead.
98 bool Consume(StringPiece x) {
99 if (starts_with(x)) {
100 remove_prefix(x.size_);
101 return true;
102 }
103 return false;
104 }
105
106 StringPiece substr(size_t pos, size_t n = npos) const;
107
108 // Return a string that contains the copy of the referenced data.
109 std::string ToString() const { return std::string(data_, size_); }
110
111 // Three-way comparison. Returns value:
112 // < 0 iff "*this" < "b",
113 // == 0 iff "*this" == "b",
114 // > 0 iff "*this" > "b"
115 int compare(StringPiece b) const;
116
117 // Return true iff "x" is a prefix of "*this"
118 // DEPRECATED: Use tensorflow::str_util::StartsWith instead.
119 bool starts_with(StringPiece x) const {
120 return ((size_ >= x.size_) && (memcmp(data_, x.data_, x.size_) == 0));
121 }
122 // Return true iff "x" is a suffix of "*this"
123 // DEPRECATED: Use tensorflow::str_util::EndsWith instead.
124 bool ends_with(StringPiece x) const {
125 return ((size_ >= x.size_) &&
126 (memcmp(data_ + (size_ - x.size_), x.data_, x.size_) == 0));
127 }
128
129 private:
130 const char* data_;
131 size_t size_;
132
133 // Intentionally copyable
134};
135
136inline bool operator==(StringPiece x, StringPiece y) {
137 return ((x.size() == y.size()) &&
138 (memcmp(x.data(), y.data(), x.size()) == 0));
139}
140
141inline bool operator!=(StringPiece x, StringPiece y) { return !(x == y); }
142
143inline bool operator<(StringPiece x, StringPiece y) { return x.compare(y) < 0; }
144inline bool operator>(StringPiece x, StringPiece y) { return x.compare(y) > 0; }
145inline bool operator<=(StringPiece x, StringPiece y) {
146 return x.compare(y) <= 0;
147}
148inline bool operator>=(StringPiece x, StringPiece y) {
149 return x.compare(y) >= 0;
150}
151
152inline int StringPiece::compare(StringPiece b) const {
153 const size_t min_len = (size_ < b.size_) ? size_ : b.size_;
154 int r = memcmp(data_, b.data_, min_len);
155 if (r == 0) {
156 if (size_ < b.size_)
157 r = -1;
158 else if (size_ > b.size_)
159 r = +1;
160 }
161 return r;
162}

165extern std::ostream& operator<<(std::ostream& o, tensorflow::StringPiece piece);
