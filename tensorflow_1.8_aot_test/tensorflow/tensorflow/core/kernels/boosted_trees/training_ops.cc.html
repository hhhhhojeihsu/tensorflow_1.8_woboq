22namespace {
23constexpr float kLayerByLayerTreeWeight = 1.0;
24
25// TODO(nponomareva, youngheek): consider using vector.
26struct SplitCandidate {
27 SplitCandidate() {}
28
29 // Index in the list of the feature ids.
30 int64 feature_idx;
31
32 // Index in the tensor of node_ids for the feature with idx feature_idx.
33 int64 candidate_idx;
34
35 float gain;
36};
37
38enum PruningMode { kNoPruning = 0, kPrePruning = 1, kPostPruning = 2 };
39
40} // namespace
41
42class BoostedTreesUpdateEnsembleOp : public OpKernel {
43 public:
44 explicit BoostedTreesUpdateEnsembleOp(OpKernelConstruction* const context)
45 : OpKernel(context) {
46 OP_REQUIRES_OK(context, context->GetAttr("num_features", &num_features_));
47
48 int32 pruning_index;
49 OP_REQUIRES_OK(context, context->GetAttr("pruning_mode", &pruning_index));
50 pruning_mode_ = static_cast<PruningMode>(pruning_index);
51 }
52
53 void Compute(OpKernelContext* const context) override {
54 // Get decision tree ensemble.
55 BoostedTreesEnsembleResource* ensemble_resource;
56 OP_REQUIRES_OK(context, LookupResource(context, HandleFromInput(context, 0),
57 &ensemble_resource));
58 core::ScopedUnref unref_me(ensemble_resource);
59 mutex_lock l(*ensemble_resource->get_mutex());
60 // Increase the ensemble stamp.
61 ensemble_resource->set_stamp(ensemble_resource->stamp() + 1);
62
63 // Read node ids, gains, thresholds and node contribs.
64 OpInputList node_ids_list;
65 OpInputList gains_list;
66 OpInputList thresholds_list;
67 OpInputList left_node_contribs;
68 OpInputList right_node_contribs;
69 OP_REQUIRES_OK(context, context->input_list("node_ids", &node_ids_list));
70 OP_REQUIRES_OK(context, context->input_list("gains", &gains_list));
71 OP_REQUIRES_OK(context,
72 context->input_list("thresholds", &thresholds_list));
73 OP_REQUIRES_OK(context, context->input_list("left_node_contribs",
74 &left_node_contribs));
75 OP_REQUIRES_OK(context, context->input_list("right_node_contribs",
76 &right_node_contribs));
77
78 const Tensor* feature_ids_t;
79 OP_REQUIRES_OK(context, context->input("feature_ids", &feature_ids_t));
80 const auto feature_ids = feature_ids_t->vec<int32>();
81
82 const Tensor* max_depth_t;
83 OP_REQUIRES_OK(context, context->input("max_depth", &max_depth_t));
84 const auto max_depth = max_depth_t->scalar<int32>()();
85
86 const Tensor* learning_rate_t;
87 OP_REQUIRES_OK(context, context->input("learning_rate", &learning_rate_t));
88 const auto learning_rate = learning_rate_t->scalar<float>()();
89
90 // Find best splits for each active node.
91 std::map<int32, SplitCandidate> best_splits;
92 FindBestSplitsPerNode(context, node_ids_list, gains_list, &best_splits);
93
94 int32 current_tree =
95 UpdateGlobalAttemptsAndRetrieveGrowableTree(ensemble_resource);
96
97 // No-op if no new splits can be considered.
98 if (best_splits.empty()) {
99 LOG(WARNING) << "Not growing tree ensemble as no good splits were found.";
100 return;
101 }
102
103 const int32 new_num_layers =
104 ensemble_resource->GetNumLayersGrown(current_tree) + 1;
105 VLOG(1) << "Adding layer #" << new_num_layers - 1 << " to tree #"
106 << current_tree << " of ensemble of " << current_tree + 1
107 << " trees.";
108 bool split_happened = false;
109 int32 node_id_start = ensemble_resource->GetNumNodes(current_tree);
110 // Add the splits to the tree.
111 for (auto& split_entry : best_splits) {
112 const int32 node_id = split_entry.first;
113 const SplitCandidate& candidate = split_entry.second;
114
115 const int64 feature_idx = candidate.feature_idx;
116 const int64 candidate_idx = candidate.candidate_idx;
117
118 const int32 feature_id = feature_ids(feature_idx);
119 const int32 threshold =
120 thresholds_list[feature_idx].vec<int32>()(candidate_idx);
121 const float gain = gains_list[feature_idx].vec<float>()(candidate_idx);
122
123 if (pruning_mode_ == kPrePruning) {
124 // Don't consider negative splits if we're pre-pruning the tree.
125 // Note that zero-gain splits are acceptable.
126 if (gain < 0) {
127 continue;
128 }
129 }
130 // For now assume that the weights vectors are one dimensional.
131 // TODO(nponomareva): change here for multiclass.
132 const float left_contrib =
133 learning_rate *
134 left_node_contribs[feature_idx].matrix<float>()(candidate_idx, 0);
135 const float right_contrib =
136 learning_rate *
137 right_node_contribs[feature_idx].matrix<float>()(candidate_idx, 0);
138
139 // unused.
140 int32 left_node_id;
141 int32 right_node_id;
142
143 ensemble_resource->AddBucketizedSplitNode(
144 current_tree, node_id, feature_id, threshold, gain, left_contrib,
145 right_contrib, &left_node_id, &right_node_id);
146 split_happened = true;
147 }
148 int32 node_id_end = ensemble_resource->GetNumNodes(current_tree);
149 if (split_happened) {
150 // Update growable tree metadata.
151 ensemble_resource->SetNumLayersGrown(current_tree, new_num_layers);
152 // Finalize the tree if needed.
153 if (ensemble_resource->GetNumLayersGrown(current_tree) >= max_depth) {
154 // If the tree is finalized, next growing will start from node 0;
155 node_id_start = 0;
156 node_id_end = 1;
157 ensemble_resource->SetIsFinalized(current_tree, true);
158 if (pruning_mode_ == kPostPruning) {
159 ensemble_resource->PostPruneTree(current_tree);
160 }
161 if (ensemble_resource->num_trees() > 0) {
162 // Create a dummy new tree with an empty node.
163 ensemble_resource->AddNewTree(kLayerByLayerTreeWeight);
164 }
165 }
166 // If we managed to split, update the node range. If we didn't, don't
167 // update as we will try to split the same nodes with new instances.
168 ensemble_resource->UpdateLastLayerNodesRange(node_id_start, node_id_end);
169 }
170 }
171
172 private:
173 int32 UpdateGlobalAttemptsAndRetrieveGrowableTree(
174 BoostedTreesEnsembleResource* const ensemble_resource) {
175 int32 num_trees = ensemble_resource->num_trees();
176 int32 current_tree = num_trees - 1;
177
178 // Increment global attempt stats.
179 ensemble_resource->UpdateGrowingMetadata();
180
181 // Note we don't set tree weight to be equal to learning rate, since we
182 // apply learning rate to leaf weights instead, when doing layer-by-layer
183 // boosting.
184 if (num_trees <= 0) {
185 // Create a new tree with a no-op leaf.
186 current_tree = ensemble_resource->AddNewTree(kLayerByLayerTreeWeight);
187 }
188 return current_tree;
189 }
190
191 // Helper method which effectively does a reduce over all split candidates
192 // and finds the best split for each node.
193 void FindBestSplitsPerNode(
194 OpKernelContext* const context, const OpInputList& node_ids_list,
195 const OpInputList& gains_list,
196 std::map<int32, SplitCandidate>* best_split_per_node) {
197 // Find best split per node going through every feature candidate.
198 for (int64 feature_idx = 0; feature_idx < num_features_; ++feature_idx) {
199 const auto& node_ids = node_ids_list[feature_idx].vec<int32>();
200 const auto& gains = gains_list[feature_idx].vec<float>();
201
202 for (size_t candidate_idx = 0; candidate_idx < node_ids.size();
203 ++candidate_idx) {
204 // Get current split candidate.
205 const auto& node_id = node_ids(candidate_idx);
206 const auto& gain = gains(candidate_idx);
207
208 auto best_split_it = best_split_per_node->find(node_id);
209 SplitCandidate candidate;
210 candidate.feature_idx = feature_idx;
211 candidate.candidate_idx = candidate_idx;
212 candidate.gain = gain;
213
214 if (best_split_it == best_split_per_node->end() ||
215 gain > best_split_it->second.gain) {
216 (*best_split_per_node)[node_id] = candidate;
217 }
218 }
219 }
220 }
221
222 private:
223 int32 num_features_;
224 PruningMode pruning_mode_;
225};
