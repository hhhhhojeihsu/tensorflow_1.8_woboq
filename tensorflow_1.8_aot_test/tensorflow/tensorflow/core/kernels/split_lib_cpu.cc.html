27template <typename T, int NDims>
28void Split<Eigen::ThreadPoolDevice, T, NDims>::operator()(
29 const Eigen::ThreadPoolDevice& d, typename TTypes<T, NDims>::Tensor output,
30 typename TTypes<T, NDims>::ConstTensor input,
31 const Eigen::DSizes<Eigen::DenseIndex, NDims>& slice_indices,
32 const Eigen::DSizes<Eigen::DenseIndex, NDims>& slice_sizes) {
33 if (output.size() < 131072) {
34 output = input.slice(slice_indices, slice_sizes);
35 } else {
36 output.device(d) = input.slice(slice_indices, slice_sizes);
37 }
38}
39
40#define DEFINE_CPU_KERNELS(T) \
41 template struct Split<Eigen::ThreadPoolDevice, T, 2>; \
42 template struct Split<Eigen::ThreadPoolDevice, T, 3>;
43
44TF_CALL_ALL_TYPES(DEFINE_CPU_KERNELS)
45DEFINE_CPU_KERNELS(quint8)
