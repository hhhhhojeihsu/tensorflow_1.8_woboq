30typedef Eigen::ThreadPoolDevice CPUDevice;
31typedef Eigen::GpuDevice GPUDevice;
32#ifdef TENSORFLOW_USE_SYCL
33typedef Eigen::SyclDevice SYCLDevice;

36// Partial specialization for a CPUDevice, that uses the Eigen implementation
37// from SoftmaxEigenImpl.
38namespace functor {
39template <typename Device, typename T>
40struct SoftmaxFunctorBase {
41 void operator()(const Device& d, typename TTypes<T>::ConstMatrix logits,
42 typename TTypes<T>::Matrix softmax, const bool log) {
43 SoftmaxEigenImpl<Device, T>::Compute(d, logits, softmax, log);
44 }
45};
46template <typename T>
47struct SoftmaxFunctor<CPUDevice, T> : SoftmaxFunctorBase<CPUDevice, T> {};
48
49#ifdef TENSORFLOW_USE_SYCL
50template <typename T>
51struct SoftmaxFunctor<SYCLDevice, T> : SoftmaxFunctorBase<SYCLDevice, T> {};
52#endif // TENSORFLOW_USE_SYCL
53} // namespace functor
54
55template <typename Device, typename T>
56class SoftmaxOp : public OpKernel {
57 public:
58 explicit SoftmaxOp(OpKernelConstruction* context) : OpKernel(context) {
59 log_ = str_util::StartsWith(type_string(), "Log");
60 }
61
62 void Compute(OpKernelContext* context) override {
63 const Tensor& logits_in = context->input(0);
64 OP_REQUIRES(context, TensorShapeUtils::IsMatrix(logits_in.shape()),
65 errors::InvalidArgument("logits must be 2-dimensional"));
66 Tensor* softmax_out = nullptr;
67 OP_REQUIRES_OK(context, context->forward_input_or_allocate_output(
68 {0}, 0, logits_in.shape(), &softmax_out));
69 if (logits_in.NumElements() > 0) {
70 functor::SoftmaxFunctor<Device, T> functor;
71 functor(context->eigen_device<Device>(), logits_in.matrix<T>(),
72 softmax_out->matrix<T>(), log_);
73 }
74 }
75
76 private:
77 bool log_;
78};

80#define REGISTER_CPU(T) \
81 REGISTER_KERNEL_BUILDER( \
82 Name("Softmax").Device(DEVICE_CPU).TypeConstraint<T>("T"), \
83 SoftmaxOp<CPUDevice, T>);
84TF_CALL_half(REGISTER_CPU);
85TF_CALL_float(REGISTER_CPU);
86TF_CALL_double(REGISTER_CPU);
87
88#undef REGISTER_CPU
89#define REGISTER_CPU(T) \
90 REGISTER_KERNEL_BUILDER( \
91 Name("LogSoftmax").Device(DEVICE_CPU).TypeConstraint<T>("T"), \
92 SoftmaxOp<CPUDevice, T>);
