19#include "third_party/eigen3/unsupported/Eigen/CXX11/Tensor"

24namespace tensorflow {
25
26// ConstantOp returns a tensor specified by ConstantOpDef.
27class ConstantOp : public OpKernel {
28 public:
29 explicit ConstantOp(OpKernelConstruction* ctx);
30 void Compute(OpKernelContext* ctx) override;
31 bool IsExpensive() override { return false; }
32 ~ConstantOp() override;
33
34 private:
35 Tensor tensor_;
36 TF_DISALLOW_COPY_AND_ASSIGN(ConstantOp);
37};
38
39// HostConstantOp differs from ConstantOp in that its output is always
40// in host memory.
41class HostConstantOp : public OpKernel {
42 public:
43 explicit HostConstantOp(OpKernelConstruction* ctx);
44 void Compute(OpKernelContext* ctx) override;
45 bool IsExpensive() override { return false; }
46 ~HostConstantOp() override {}
47
48 private:
49 Tensor tensor_;
50 TF_DISALLOW_COPY_AND_ASSIGN(HostConstantOp);
51};
52
53class PlaceholderOp : public OpKernel {
54 public:
55 explicit PlaceholderOp(OpKernelConstruction* ctx);
56 void Compute(OpKernelContext* ctx) override;
57
58 private:
59 PartialTensorShape expected_shape_;
60};
61
62} // namespace tensorflow
