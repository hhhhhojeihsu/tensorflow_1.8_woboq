35// Helper method to copy using memcpy.
36template <typename T, typename Index, typename SliceIndex,
37 SliceIndex static_slice_elems>
38SliceIndex HandleCopies(OpKernelContext* ctx,
39 typename TTypes<T, 3>::ConstTensor params,
40 typename TTypes<Index>::ConstFlat indices,
41 SliceIndex slice_elems,
42 typename TTypes<T, 3>::Tensor out) {
43 const SliceIndex indices_size = static_cast<SliceIndex>(indices.dimension(0));
44 const SliceIndex batch_size = static_cast<SliceIndex>(params.dimension(0));
45 const Index limit = static_cast<Index>(params.dimension(1));
46 T* out_base = &out(0, 0, 0);
47 const T* params_base = &params(0, 0, 0);
48 if (static_slice_elems >= 0) {
49 // Give compiler static knowledge of the number of elements/bytes
50 slice_elems = static_slice_elems;
51 }
52 // Compute slice_bytes here so that static knowledge is available
53 const size_t slice_bytes = slice_elems * sizeof(T);
54 auto* worker_threads = ctx->device()->tensorflow_cpu_worker_threads();
55 mutex mu;
56 // Store the value of invalidate index for printing error information, it's a
57 // shared variable.
58 SliceIndex result = -1;
59 auto work = [&](int64 start, int64 end) {
60 SliceIndex batch_idx = static_cast<SliceIndex>(start / indices_size);
61 SliceIndex indices_idx = static_cast<SliceIndex>(start % indices_size);
62 SliceIndex batch_idx_end = static_cast<SliceIndex>(end / indices_size);
63 SliceIndex indices_idx_end = static_cast<SliceIndex>(end % indices_size);
64
65 while ((batch_idx < batch_idx_end) ||
66 (batch_idx == batch_idx_end && indices_idx < indices_idx_end)) {
67 SliceIndex i_next = indices_idx + 1;
68 SliceIndex b_next = batch_idx + 1;
69 if ((batch_idx == batch_idx_end && i_next < indices_idx_end) ||
70 (i_next < indices_size)) {
71 port::prefetch<port::PREFETCH_HINT_T0>(
72 &params(batch_idx, indices(i_next), 0));
73 port::prefetch<port::PREFETCH_HINT_T0>(&out(batch_idx, i_next, 0));
74 b_next = batch_idx;
75 } else if (b_next <= batch_idx_end) {
76 port::prefetch<port::PREFETCH_HINT_T0>(&params(b_next, indices(0), 0));
77 port::prefetch<port::PREFETCH_HINT_T0>(&out(b_next, 0, 0));
78 i_next = 0;
79 }
80 const Index index = internal::SubtleMustCopy(indices(indices_idx));
81 if (!FastBoundsCheck(index, limit)) {
82 mutex_lock l(mu);
83 result = indices_idx;
84 return;
85 }
86 // Copy using memcpy if possible, otherwise an Eigen loop
87 // TODO(cwhipkey): avoid linking to framework to get Allocator (to improve
88 // ahead-of-time compilation binary size).
89 if (is_simple_type<T>::value) {
90 // Avoid auto-promotion to Index from SliceIndex by casting.
91 memcpy(
92 out_base + (batch_idx * indices_size + indices_idx) * slice_elems,
93 params_base + (batch_idx * static_cast<SliceIndex>(limit) +
94 static_cast<SliceIndex>(index)) *
95 slice_elems,
96 slice_bytes);
97 } else {
98 // For non-"simple" types (e.g. strings).
99 out.template chip<1>(indices_idx) = params.template chip<1>(index);
100 }
101 indices_idx = i_next;
102 batch_idx = b_next;
103 }
104 };
105
106 Shard(worker_threads->num_threads, worker_threads->workers,
107 batch_size * indices_size, slice_elems * sizeof(T), work);
108 return result;
109}

111template <typename T, typename Index>
112struct GatherFunctorCPU {
113 int64 operator()(OpKernelContext* ctx,
114 typename TTypes<T, 3>::ConstTensor params,
115 typename TTypes<Index>::ConstFlat indices,
116 typename TTypes<T, 3>::Tensor out) {
117 const int64 N = indices.size();
118 const int64 slice_size = out.dimension(2);
119 int64 bad_i;
120
121 bool use_large = (slice_size > std::numeric_limits<int32>::max() ||
122 params.size() > std::numeric_limits<int32>::max() ||
123 N > std::numeric_limits<int32>::max());
124#define CALL(elems) \
125 do { \
126 if (use_large) { \
127 bad_i = HandleCopies<T, Index, int64, elems>(ctx, params, indices, \
128 slice_size, out); \
129 } else { \
130 const int32 small_slice = static_cast<int32>(slice_size); \
131 bad_i = HandleCopies<T, Index, int32, elems>(ctx, params, indices, \
132 small_slice, out); \
133 } \
134 } while (0)
135
136 if (slice_size == 10)
137 CALL(10);
138 else if (slice_size == 20)
139 CALL(20);
140 else
141 CALL(-1);
142#undef CALL
143
144 return bad_i;
145 }
146};

148template <typename Device, typename T, typename Index>
149struct GatherFunctor {
150 int64 operator()(OpKernelContext* ctx,
151 typename TTypes<T, 3>::ConstTensor params,
152 typename TTypes<Index>::ConstFlat indices,
153 typename TTypes<T, 3>::Tensor out);
154};

156template <typename T, typename Index>
157struct GatherFunctor<CPUDevice, T, Index> {
158 int64 operator()(OpKernelContext* ctx,
159 typename TTypes<T, 3>::ConstTensor params,
160 typename TTypes<Index>::ConstFlat indices,
161 typename TTypes<T, 3>::Tensor out) {
162 return GatherFunctorCPU<T, Index>()(ctx, params, indices, out);
163 }
164};

166template <typename Index>

169 typename TTypes<Variant, 3>::ConstTensor params,
170 typename TTypes<Index>::ConstFlat indices,
171 typename TTypes<Variant, 3>::Tensor out) {
172 return GatherFunctorCPU<Variant, Index>()(ctx, params, indices, out);
173 }
