47std::unique_ptr<const NodeDef> StripTensorDataFromNodeDef(
48 OpKernelConstruction* ctx) {
49#ifndef __ANDROID__
50 DCHECK_EQ(NodeDef::descriptor()->field_count(), 5)
51 << "The NodeDef format has changed, and the attr-stripping code may need "
52 << "to be updated.";
53#endif
54 const NodeDef& original = ctx->def();
55 NodeDef* ret = new NodeDef;
56 ret->set_name(original.name());
57 ret->set_op(original.op());
58 ret->set_device(original.device());
59 // Strip the "value" attr from the returned NodeDef.
60 // NOTE(mrry): The present implementation of `OpKernel::OpKernel()` only uses
61 // attrs that affect the cardinality of list-typed inputs and outputs, so it
62 // is safe to drop other attrs from the NodeDef.
63 AddNodeAttr("dtype", ctx->output_type(0), ret);
64 return std::unique_ptr<const NodeDef>(ret);
65}

69ConstantOp::ConstantOp(OpKernelConstruction* ctx)
70 : OpKernel(ctx, StripTensorDataFromNodeDef(ctx)),
71 tensor_(ctx->output_type(0)) {
72 const TensorProto* proto = nullptr;
73 OP_REQUIRES_OK(ctx, ctx->GetAttr("value", &proto));
74 OP_REQUIRES_OK(ctx, ctx->device()->MakeTensorFromProto(
75 *proto, AllocatorAttributes(), &tensor_));
76 OP_REQUIRES(
77 ctx, ctx->output_type(0) == tensor_.dtype(),
78 errors::InvalidArgument("Type mismatch between value (",
79 DataTypeString(tensor_.dtype()), ") and dtype (",
80 DataTypeString(ctx->output_type(0)), ")"));
81}
82
83void ConstantOp::Compute(OpKernelContext* ctx) {
84 ctx->set_output(0, tensor_);
85 if (TF_PREDICT_FALSE(ctx->track_allocations())) {
86 ctx->record_persistent_memory_allocation(tensor_.AllocatedBytes());
87 }
88}

143HostConstantOp::HostConstantOp(OpKernelConstruction* ctx)
144 : OpKernel(ctx), tensor_(ctx->output_type(0)) {
145 const TensorProto* proto = nullptr;
146 AllocatorAttributes alloc_attr;
147 alloc_attr.set_on_host(true);
148 OP_REQUIRES_OK(ctx, ctx->GetAttr("value", &proto));
149 OP_REQUIRES_OK(
150 ctx, ctx->device()->MakeTensorFromProto(*proto, alloc_attr, &tensor_));
151 OP_REQUIRES(
152 ctx, ctx->output_type(0) == tensor_.dtype(),
153 errors::InvalidArgument("Type mismatch between value (",
154 DataTypeString(tensor_.dtype()), ") and dtype (",
155 DataTypeString(ctx->output_type(0)), ")"));
156}

187template <typename Device, typename T, typename Index>
188class FillOp : public OpKernel {
189 public:
190 explicit FillOp(OpKernelConstruction* context) : OpKernel(context) {}
191
192 void Compute(OpKernelContext* context) override {
193 const Tensor& Tdims = context->input(0);
194 OP_REQUIRES(context, IsLegacyVector(Tdims.shape()),
195 errors::InvalidArgument("dims must be a vector, got shape ",
196 Tdims.shape().DebugString()));
197 const Tensor& Tvalue = context->input(1);
198 OP_REQUIRES(context, IsLegacyScalar(Tvalue.shape()),
199 errors::InvalidArgument("value must be a scalar, got shape ",
200 Tvalue.shape().DebugString()));
201 auto dims = Tdims.flat<Index>();
202 TensorShape shape;
203 OP_REQUIRES_OK(context, TensorShapeUtils::MakeShape(
204 reinterpret_cast<const Index*>(dims.data()),
205 dims.size(), &shape));
206 Tensor* out = nullptr;
207 OP_REQUIRES_OK(context, context->allocate_output(0, shape, &out));
208 functor::FillFunctor<Device, T> functor;
209 functor(context->eigen_device<Device>(), out->flat<T>(),
210 Tvalue.scalar<T>());
211 }
212};

286template <typename Device, typename T>
287class ZerosLikeOp : public OpKernel {
288 public:
289 explicit ZerosLikeOp(OpKernelConstruction* ctx) : OpKernel(ctx) {}
290
291 void Compute(OpKernelContext* ctx) override {
292 const Tensor& input = ctx->input(0);
293 const Device& d = ctx->eigen_device<Device>();
294 if (std::is_same<T, Variant>::value) {
295 OP_REQUIRES(
296 ctx, input.dims() == 0,
297 errors::InvalidArgument("ZerosLike non-scalar Tensor with "
298 "dtype=DT_VARIANT is not supported."));
299 const Variant& v = input.scalar<Variant>()();
300 Tensor out(cpu_allocator(), DT_VARIANT, TensorShape({}));
301 Variant* out_v = &(out.scalar<Variant>()());
302 OP_REQUIRES_OK(ctx, UnaryOpVariant<Device>(
303 ctx, ZEROS_LIKE_VARIANT_UNARY_OP, v, out_v));
304 ctx->set_output(0, out);
305 } else {
306 Tensor* out = nullptr;
307 OP_REQUIRES_OK(ctx, ctx->forward_input_or_allocate_output(
308 {0}, 0, input.shape(), &out));
309 functor::SetZeroFunctor<Device, T> f;
310 f(d, out->flat<T>());
311 }
312 }
313};

408PlaceholderOp::PlaceholderOp(OpKernelConstruction* ctx) : OpKernel(ctx) {
409 OP_REQUIRES_OK(ctx, ctx->GetAttr("shape", &expected_shape_));
410}
411
412void PlaceholderOp::Compute(OpKernelContext* ctx) {
413 if (expected_shape_.dims() > 0) {
414 OP_REQUIRES(ctx, false,
415 errors::InvalidArgument(
416 "You must feed a value for placeholder tensor '", name(),
417 "' with dtype ", DataTypeString(output_type(0)),
418 " and shape ", expected_shape_.DebugString()));
419 } else {
420 OP_REQUIRES(ctx, false,
421 errors::InvalidArgument(
422 "You must feed a value for placeholder tensor '", name(),
423 "' with dtype ", DataTypeString(output_type(0))));
424 }
425}
