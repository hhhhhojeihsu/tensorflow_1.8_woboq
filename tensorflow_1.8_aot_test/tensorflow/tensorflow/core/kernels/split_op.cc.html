44template <typename Device, typename T>
45class SplitOpBase : public OpKernel {
46 public:
47 explicit SplitOpBase(OpKernelConstruction* c) : OpKernel(c) {}
48
49 void ComputeEasyCases(OpKernelContext* context, bool* done) {
50 const Tensor& input = context->input(1);
51 const TensorShape& input_shape = input.shape();
52 const int32 split_dim_orig = context->input(0).flat<int32>()(0);
53 const int32 split_dim =
54 split_dim_orig < 0 ? split_dim_orig + input.dims() : split_dim_orig;
55 const int32 num_split = num_outputs();
56
57 OP_REQUIRES(
58 context, 0 <= split_dim && split_dim < input_shape.dims(),
59 errors::InvalidArgument("-input rank(-", input.dims(),
60 ") <= split_dim < input rank (", input.dims(),
61 "), but got ", split_dim_orig));
62
63 OP_REQUIRES(
64 context, num_split > 0,
65 errors::InvalidArgument(
66 "Number of ways to split should be > 0, but got ", num_split));
67
68 OP_REQUIRES(context, input_shape.dim_size(split_dim) % num_split == 0,
69 errors::InvalidArgument(
70 "Number of ways to split should evenly divide the split "
71 "dimension, but got split_dim ",
72 split_dim, " (size = ", input_shape.dim_size(split_dim),
73 ") ", "and num_split ", num_split));
74 // Special case 1: num_split == 1. Nothing to do.
75 if (num_split == 1) {
76 VLOG(1) << "Split identity";
77 context->set_output(0, context->input(1));
78 *done = true;
79 return;
80 }
81
82 // Special case 2: split along the 1st dimension. We can share the
83 // underlying buffer.
84 //
85 // Apply this optimization conservatively: if input is aligned,
86 // the resulting tensors must be aligned. It's conservative
87 // because if the immediate consumer of the resulting tensors are
88 // not using eigen for computation, its perfectly fine to avoid
89 // the copying.
90 if ((split_dim == 0) && IsInnerDimsSizeAligned<T>(input_shape)) {
91 VLOG(1) << "Slice dim 0: " << input_shape.DebugString();
92 const int64 delta = input_shape.dim_size(0) / num_split;
93 for (int i = 0; i < num_split; ++i) {
94 context->set_output(i, input.Slice(i * delta, (i + 1) * delta));
95 }
96 *done = true;
97 return;
98 }
99 }
100
101 template <typename IndexType>
102 std::tuple<IndexType, IndexType, IndexType> SetDims(
103 const TensorShape& input_shape, int32 split_dim) const {
104 static_assert(std::is_integral<IndexType>::value,
105 "IndexType must be an integer type");
106 int32 prefix_dim_size = 1;
107 for (int i = 0; i < split_dim; ++i) {
108 prefix_dim_size *= input_shape.dim_size(i);
109 }
110
111 // Caller must ensure that dim_size and suffix_dim_size are <
112 // std::numeric_limits<IndexType>::max()
113 IndexType split_dim_size =
114 static_cast<IndexType>(input_shape.dim_size(split_dim));
115
116 IndexType suffix_dim_size = 1;
117 for (int i = split_dim + 1; i < input_shape.dims(); ++i) {
118 suffix_dim_size *= static_cast<IndexType>(input_shape.dim_size(i));
119 }
120 return std::make_tuple(prefix_dim_size, split_dim_size, suffix_dim_size);
121 }
122};

124template <typename T, typename InputReshapedType, int NDims>
125class SplitOpCPUImpl {
126 public:
127 template <typename MakeSizesType, typename ReshapeResultType>
128 void operator()(OpKernelContext* context,
129 const InputReshapedType& input_reshaped,
130 const TensorShape& input_shape, int32 split_dim,
131 Eigen::DenseIndex prefix_dim_size,
132 Eigen::DenseIndex split_dim_size,
133 Eigen::DenseIndex suffix_dim_size,
134 const MakeSizesType& make_sizes,
135 const ReshapeResultType& reshape_result, int32 num_split,
136 int64 split_dim_output_size) const {
137 const auto num_threads =
138 context->device()->tensorflow_cpu_worker_threads()->num_threads;
139 // TODO(jewillco): Tune heuristic further.
140 const auto input_element_count = input_shape.num_elements();
141 const bool use_parallelism_between_outputs =
142 (num_split >= 4 &&
143 input_element_count >= std::max(num_threads, num_split) * 4096 &&
144 input_element_count < num_split * 180 * 1024);
145 Eigen::DSizes<Eigen::DenseIndex, NDims> indices;
146 for (int i = 0; i < NDims; ++i) {
147 indices[i] = 0;
148 }
149 auto sizes = make_sizes(split_dim_output_size);
150 TensorShape output_shape(input_shape);
151 output_shape.set_dim(split_dim, split_dim_output_size);
152
153 auto range_output_func = [&indices, context, &output_shape, prefix_dim_size,
154 split_dim_output_size, suffix_dim_size, &sizes,
155 use_parallelism_between_outputs, &input_reshaped,
156 &reshape_result](int64 start, int64 limit) {
157 for (int64 i = start; i < limit; ++i) {
158 Tensor* result = nullptr;
159 OP_REQUIRES_OK(context,
160 context->allocate_output(i, output_shape, &result));
161 if (prefix_dim_size * split_dim_output_size * suffix_dim_size > 0) {
162 Eigen::DSizes<Eigen::DenseIndex, NDims> slice_indices;
163 Eigen::DSizes<Eigen::DenseIndex, NDims> slice_sizes;
164 for (int j = 0; j < NDims; ++j) {
165 slice_indices[j] =
166 (j == NDims - 2 ? i * split_dim_output_size : indices[j]);
167 slice_sizes[j] = sizes[j];
168 }
169
170 auto result_shaped = reshape_result(result, split_dim_output_size);
171
172 if (use_parallelism_between_outputs) {
173 // Use sequential implementation for single output.
174 result_shaped = input_reshaped.slice(slice_indices, slice_sizes);
175 } else {
176 // This implementation may be parallel internally.
177 functor::Split<CPUDevice, T, NDims>()(
178 context->eigen_device<CPUDevice>(), result_shaped,
179 input_reshaped, slice_indices, slice_sizes);
180 }
181 }
182 }
183 };
184 if (use_parallelism_between_outputs) {
185 // Run in parallel, disabling parallelism in functor.
186 context->device()->tensorflow_cpu_worker_threads()->workers->ParallelFor(
187 num_split, input_element_count / num_split, range_output_func);
188 } else {
189 // Run sequentially, but allow internal parallelism in functor.
190 range_output_func(0, num_split);
191 }
192 }
193};

195template <typename T>
196class SplitOpCPU : public SplitOpBase<CPUDevice, T> {
197 public:
198 typedef SplitOpBase<CPUDevice, T> Base;
199 explicit SplitOpCPU(OpKernelConstruction* c) : Base(c) {}
200
201 void Compute(OpKernelContext* context) override {
202 bool done = false;
203 Base::ComputeEasyCases(context, &done);
204 if (!context->status().ok() || done) {
205 return;
206 }
207 const int32 num_split = Base::num_outputs();
208 const Tensor& input = context->input(1);
209 const TensorShape& input_shape = input.shape();
210 const int32 split_dim_orig = context->input(0).flat<int32>()(0);
211 const int32 split_dim =
212 split_dim_orig < 0 ? split_dim_orig + input.dims() : split_dim_orig;
213
214 // Android also uses int32 indexing, so check here also.
215 OP_REQUIRES(
216 context,
217 FastBoundsCheck(input.NumElements(),
218 std::numeric_limits<Eigen::DenseIndex>::max()),
219 errors::InvalidArgument("Split requires input size < ",
220 std::numeric_limits<Eigen::DenseIndex>::max()));
221
222 Eigen::DenseIndex prefix_dim_size;
223 Eigen::DenseIndex split_dim_size;
224 Eigen::DenseIndex suffix_dim_size;
225
226 std::tie(prefix_dim_size, split_dim_size, suffix_dim_size) =
227 Base::template SetDims<Eigen::DenseIndex>(input_shape, split_dim);
228
229 const int64 split_dim_output_size = split_dim_size / num_split;
230
231 if (prefix_dim_size == 1) {
232 auto input_reshaped =
233 input.shaped<T, 2>({split_dim_size, suffix_dim_size});
234 auto make_sizes = [&](Eigen::DenseIndex split_size) {
235 return Eigen::DSizes<Eigen::DenseIndex, 2>{split_size, suffix_dim_size};
236 };
237 auto reshape_result = [&](Tensor* result, Eigen::DenseIndex split_size) {
238 return result->shaped<T, 2>({split_size, suffix_dim_size});
239 };
240 SplitOpCPUImpl<T, decltype(input_reshaped), 2>{}(
241 context, input_reshaped, input_shape, split_dim, prefix_dim_size,
242 split_dim_size, suffix_dim_size, make_sizes, reshape_result,
243 num_split, split_dim_output_size);
244 } else {
245 auto input_reshaped = input.shaped<T, 3>(
246 {prefix_dim_size, split_dim_size, suffix_dim_size});
247 auto make_sizes = [&](Eigen::DenseIndex split_size) {
248 return Eigen::DSizes<Eigen::DenseIndex, 3>{prefix_dim_size, split_size,
249 suffix_dim_size};
250 };
251 auto reshape_result = [&](Tensor* result, Eigen::DenseIndex split_size) {
252 return result->shaped<T, 3>(
253 {prefix_dim_size, split_size, suffix_dim_size});
254 };
255 SplitOpCPUImpl<T, decltype(input_reshaped), 3>{}(
256 context, input_reshaped, input_shape, split_dim, prefix_dim_size,
257 split_dim_size, suffix_dim_size, make_sizes, reshape_result,
258 num_split, split_dim_output_size);
259 }
260 }
261};

272// Partial specialization for GPU
273template <typename T>
274class SplitOpGPU : public SplitOpBase<GPUDevice, T> {
275 public:
276 typedef SplitOpBase<GPUDevice, T> Base;
277 explicit SplitOpGPU(OpKernelConstruction* c) : Base(c) {}
278
279 void Compute(OpKernelContext* context) override {
280 bool done = false;
281 Base::ComputeEasyCases(context, &done);
282 if (!context->status().ok() || done) {
283 return;
284 }
285 const Tensor& input = context->input(1);
286 const TensorShape& input_shape = input.shape();
287 const int32 split_dim_orig = context->input(0).flat<int32>()(0);
288 const int32 split_dim =
289 split_dim_orig < 0 ? split_dim_orig + input.dims() : split_dim_orig;
290 const int32 num_split = Base::num_outputs();
291 OP_REQUIRES(
292 context,
293 FastBoundsCheck(input.NumElements(), std::numeric_limits<int32>::max()),
294 errors::InvalidArgument("Split on GPU requires input size "
295 "< max int32"));
296 int32 prefix_dim_size;
297 int32 split_dim_size;
298 int32 suffix_dim_size;
299 std::tie(prefix_dim_size, split_dim_size, suffix_dim_size) =
300 Base::template SetDims<int32>(input_shape, split_dim);
301
302 const int32 split_dim_output_size = split_dim_size / num_split;
303 TensorShape output_shape(input_shape);
304 output_shape.set_dim(split_dim, split_dim_output_size);
305
306 CudaDeviceArrayOnHost<T*> ptrs(context, num_split);
307 OP_REQUIRES_OK(context, ptrs.Init());
308
309 for (int i = 0; i < num_split; ++i) {
310 Tensor* result = nullptr;
311 OP_REQUIRES_OK(context,
312 context->allocate_output(i, output_shape, &result));
313 ptrs.Set(i, result->flat<T>().data());
314 }
315 if (prefix_dim_size * split_dim_output_size * suffix_dim_size == 0) {
316 return;
317 }
318 OP_REQUIRES_OK(context, ptrs.Finalize());
319
320 SplitOpGPULaunch<T>().Run(context->eigen_device<GPUDevice>(),
321 input.flat<T>().data(), prefix_dim_size,
322 split_dim_size, suffix_dim_size, ptrs.data());
323 OP_REQUIRES(context, context->op_device_context()->stream()->ok(),
324 errors::Internal("Launch of gpu kernel for SplitOp failed"));
325 }
326};
327#endif // GOOGLE_CUDA

336 void Compute(OpKernelContext* context) override {
337 bool done = false;
338 Base::ComputeEasyCases(context, &done);
339 if (!context->status().ok() || done) {
340 return;
341 }
342 const Tensor& input = context->input(1);
343 const TensorShape& input_shape = input.shape();
344 const int32 split_dim_orig = context->input(0).flat<int32>()(0);
345 const int32 split_dim =
346 split_dim_orig < 0 ? split_dim_orig + input.dims() : split_dim_orig;
347 const int32 num_split = Base::num_outputs();
348
349 // Android also uses int32 indexing, so check here also.
350 OP_REQUIRES(
351 context,
352 FastBoundsCheck(input.NumElements(),
353 std::numeric_limits<Eigen::DenseIndex>::max()),
354 errors::InvalidArgument("Split requires input size < ",
355 std::numeric_limits<Eigen::DenseIndex>::max()));
356
357 Eigen::DenseIndex prefix_dim_size;
358 Eigen::DenseIndex split_dim_size;
359 Eigen::DenseIndex suffix_dim_size;
360
361 std::tie(prefix_dim_size, split_dim_size, suffix_dim_size) =
362 Base::template SetDims<Eigen::DenseIndex>(input_shape, split_dim);
363 auto input_reshaped =
364 input.shaped<T, 3>({prefix_dim_size, split_dim_size, suffix_dim_size});
365
366 const int64 split_dim_output_size = split_dim_size / num_split;
367 TensorShape output_shape(input_shape);
368 output_shape.set_dim(split_dim, split_dim_output_size);
369
370 Eigen::DSizes<Eigen::DenseIndex, 3> indices{0, 0, 0};
371 Eigen::DSizes<Eigen::DenseIndex, 3> sizes{
372 prefix_dim_size, split_dim_output_size, suffix_dim_size};
373
374 for (int i = 0; i < num_split; ++i) {
375 Tensor* result = nullptr;
376 OP_REQUIRES_OK(context,
377 context->allocate_output(i, output_shape, &result));
378 if (prefix_dim_size * split_dim_output_size * suffix_dim_size > 0) {
379 Eigen::DSizes<Eigen::DenseIndex, 3> slice_indices;
380 Eigen::DSizes<Eigen::DenseIndex, 3> slice_sizes;
381 for (int j = 0; j < 3; ++j) {
382 slice_indices[j] = indices[j];
383 slice_sizes[j] = sizes[j];
384 }
385
386 auto result_shaped = result->shaped<T, 3>(
387 {prefix_dim_size, split_dim_output_size, suffix_dim_size});
388
389 functor::Split<SYCLDevice, T>()(context->eigen_device<SYCLDevice>(),
390 result_shaped, input_reshaped,
391 slice_indices, slice_sizes);
392 }
393 indices[1] += split_dim_output_size;
394 }
395 }
