40// Variant compatible type for a list of tensors. This is mutable but instances
41// should never be mutated after stored in a variant tensor.
42TensorList::TensorList(const TensorList& other)
43 : tensors(other.tensors),
44 element_shape(other.element_shape),
45 element_dtype(other.element_dtype) {}
46
47void TensorList::Encode(VariantTensorData* data) const {
48 data->set_type_name(TypeName());
49 for (const Tensor& t : tensors) {
50 *data->add_tensors() = t;
51 }
52 string metadata;
53 core::PutVarint64(&metadata, static_cast<uint64>(element_dtype));
54 if (!element_shape.unknown_rank()) {
55 for (TensorShapeDim dim : element_shape) {
56 if (dim.size > 0) {
57 core::PutVarint64(&metadata, dim.size);
58 } else {
59 core::PutVarint64(&metadata, std::numeric_limits<uint64>::max());
60 }
61 }
62 }
63 data->set_metadata(metadata);
64}

66static Status TensorListDeviceCopy(
67 const TensorList& from, TensorList* to,
68 const UnaryVariantOpRegistry::AsyncTensorDeviceCopyFn& copy) {
69 to->element_shape = from.element_shape;
70 to->element_dtype = from.element_dtype;
71 to->tensors.reserve(from.tensors.size());
72 for (const Tensor& t : from.tensors) {
73 Tensor tmp(t.dtype());
74 TF_RETURN_IF_ERROR(copy(t, &tmp));
75 to->tensors.push_back(tmp);
76 }
77 return Status::OK();
78}

98bool TensorList::Decode(const VariantTensorData& data) {
99 tensors = data.tensors();
100 string metadata;
101 data.get_metadata(&metadata);
102 uint64 scratch;
103 StringPiece iter(metadata);
104 core::GetVarint64(&iter, &scratch);
105 element_dtype = static_cast<DataType>(scratch);
106 std::vector<int64> dims;
107 while (!iter.empty()) {
108 core::GetVarint64(&iter, &scratch);
109 if (scratch == std::numeric_limits<uint64>::max()) {
110 dims.push_back(-1);
111 } else {
112 dims.push_back(scratch);
113 }
114 }
115 element_shape = PartialTensorShape(dims);
116 return true;
117}
118
119Status TensorShapeFromTensor(const Tensor& t, PartialTensorShape* out) {
120 if (t.shape() == TensorShape({})) {
121 if ((t.dtype() == DT_INT32 && t.scalar<int32>()() == -1) ||
122 (t.dtype() == DT_INT64 && t.scalar<int64>()() == -1)) {
123 return Status::OK();
124 }
125 return errors::InvalidArgument(
126 "The only valid scalar shape tensor is the fully unknown shape "
127 "specified as -1.");
128 }
129 if (t.dtype() == DT_INT32) {
130 return PartialTensorShape::MakePartialShape(t.vec<int32>().data(),
131 t.NumElements(), out);
132 } else if (t.dtype() == DT_INT64) {
133 return PartialTensorShape::MakePartialShape(t.vec<int64>().data(),
134 t.NumElements(), out);
135 }
136 return errors::InvalidArgument(
137 "Expected an int32 or int64 shape tensor; found ",
138 DataTypeString(t.dtype()));
139}

141class EmptyTensorList : public OpKernel {
142 public:
143 explicit EmptyTensorList(OpKernelConstruction* ctx) : OpKernel(ctx) {
144 OP_REQUIRES_OK(ctx, ctx->GetAttr("element_dtype", &element_dtype_));
145 }
146
147 void Compute(OpKernelContext* ctx) override {
148 Tensor* result;
149 AllocatorAttributes attr;
150 attr.set_on_host(true);
151 OP_REQUIRES_OK(ctx, ctx->allocate_output(0, TensorShape{}, &result, attr));
152 TensorList empty;
153 empty.element_dtype = element_dtype_;
154 PartialTensorShape element_shape;
155 OP_REQUIRES_OK(ctx, TensorShapeFromTensor(ctx->input(0), &element_shape));
156 empty.element_shape = element_shape;
157 result->scalar<Variant>()() = std::move(empty);
158 }
159
160 private:
161 DataType element_dtype_;
162};

177class TensorListPushBack : public OpKernel {
178 public:
179 explicit TensorListPushBack(OpKernelConstruction* c) : OpKernel(c) {
180 OP_REQUIRES_OK(c, c->GetAttr("element_dtype", &element_dtype_));
181 }
182
183 ~TensorListPushBack() override {}
184
185 void Compute(OpKernelContext* c) override {
186 const Tensor& input = c->input(1);
187 OP_REQUIRES(c, element_dtype_ == input.dtype(),
188 errors::InvalidArgument("Invalid data types; list elements ",
189 DataTypeString(element_dtype_),
190 " but tried to append ",
191 DataTypeString(input.dtype())));
192
193 const TensorList* l = c->input(0).scalar<Variant>()().get<TensorList>();
194 OP_REQUIRES(c, l != nullptr,
195 errors::InvalidArgument(
196 "Input handle is not a list. Saw: '",
197 c->input(0).scalar<Variant>()().DebugString(), "'"));
198 OP_REQUIRES(c, l->element_shape.IsCompatibleWith(input.shape()),
199 errors::InvalidArgument(
200 "Tried to append a tensor with incompatible shape to a "
201 "list. Op element shape: ",
202 input.shape().DebugString(),
203 " list shape: ", l->element_shape.DebugString()));
204 OP_REQUIRES(c, element_dtype_ == l->element_dtype,
205 errors::InvalidArgument("Invalid data types; op elements ",
206 DataTypeString(element_dtype_),
207 " but list elements ",
208 DataTypeString(l->element_dtype)));
209
210 TensorList output;
211 output = *l;
212 output.tensors.push_back(input);
213 Tensor* result;
214 AllocatorAttributes attr;
215 attr.set_on_host(true);
216 OP_REQUIRES_OK(c, c->allocate_output(0, TensorShape{}, &result, attr));
217 result->scalar<Variant>()() = std::move(output);
218 }
219
220 private:
221 DataType element_dtype_;
222};

263class TensorListElementShape : public OpKernel {
264 public:
265 explicit TensorListElementShape(OpKernelConstruction* c) : OpKernel(c) {}
266
267 void Compute(OpKernelContext* c) override {
268 OP_REQUIRES(
269 c, c->input(0).shape().num_elements() == 1,
270 errors::InvalidArgument("List tensors are supposed to be scalars."));
271 const TensorList* l = c->input(0).scalar<Variant>()().get<TensorList>();
272 OP_REQUIRES(c, l != nullptr,
273 errors::InvalidArgument(
274 "TensorListElementShape received a variant which is not a "
275 "list. Saw: '",
276 c->input(0).scalar<Variant>()().DebugString(), "'"));
277 Tensor* result;
278 OP_REQUIRES_OK(c, c->allocate_output(
279 0, TensorShape{l->element_shape.dims()}, &result));
280 for (int i = 0; i < l->element_shape.dims(); ++i) {
281 if (result->dtype() == DT_INT32) {
282 result->flat<int32>()(i) = l->element_shape.dim_size(i);
283 } else {
284 result->flat<int64>()(i) = l->element_shape.dim_size(i);
285 }
286 }
287 }
288};

307
308 ~TensorListPopBack() override {}
309
310 void Compute(OpKernelContext* c) override {
311 const TensorList* l = c->input(0).scalar<Variant>()().get<TensorList>();
312 OP_REQUIRES(c, l != nullptr,
313 errors::InvalidArgument(
314 "Input handle is not a list. Saw: '",
315 c->input(0).scalar<Variant>()().DebugString(), "'"));
316 OP_REQUIRES(c, element_dtype_ == l->element_dtype,
317 errors::InvalidArgument("Invalid data types; op elements ",
318 DataTypeString(element_dtype_),
319 " but list elements ",
320 DataTypeString(l->element_dtype)));
321
322 OP_REQUIRES(c, !l->tensors.empty(),
323 errors::InvalidArgument("Trying to pop from an empty list."));
324
325 c->set_output(1, l->tensors.back());
326 TensorList output;
327 output = *l;
328 output.tensors.pop_back();
329 Tensor* result;
330 AllocatorAttributes attr;
331 attr.set_on_host(true);
332 OP_REQUIRES_OK(c, c->allocate_output(0, TensorShape{}, &result, attr));
333 result->scalar<Variant>()() = std::move(output);
334 }
335
336 private:
337 DataType element_dtype_;
338};

388class TensorListGetItem : public OpKernel {
389 public:
390 explicit TensorListGetItem(OpKernelConstruction* c) : OpKernel(c) {
391 OP_REQUIRES_OK(c, c->GetAttr("element_dtype", &element_dtype_));
392 }
393
394 void Compute(OpKernelContext* c) override {
395 OP_REQUIRES(
396 c, c->input(0).shape().num_elements() == 1,
397 errors::InvalidArgument("List tensors are supposed to be scalars."));
398 const TensorList* l = c->input(0).scalar<Variant>()().get<TensorList>();
399 OP_REQUIRES(c, l != nullptr,
400 errors::InvalidArgument(
401 "Input handle is not a list. Saw: '",
402 c->input(0).scalar<Variant>()().DebugString(), "'"));
403 OP_REQUIRES(c, element_dtype_ == l->element_dtype,
404 errors::InvalidArgument("Invalid data types; op elements ",
405 DataTypeString(element_dtype_),
406 " but list elements ",
407 DataTypeString(l->element_dtype)));
408 int32 index = c->input(1).scalar<int32>()();
409 OP_REQUIRES(c, index < l->tensors.size(),
410 errors::InvalidArgument("Trying to access element ", index,
411 " in a list with ", l->tensors.size(),
412 " elements."));
413 c->set_output(0, l->tensors[index]);
414 }
415
416 private:
417 DataType element_dtype_;
418};
