28// See documentation in ../ops/dataset_ops.cc for a high-level

30class GroupByWindowDatasetOp : public UnaryDatasetOpKernel {
31 public:
32 explicit GroupByWindowDatasetOp(OpKernelConstruction* ctx)
33 : UnaryDatasetOpKernel(ctx),
34 graph_def_version_(ctx->graph_def_version()) {
35 OP_REQUIRES_OK(ctx, ctx->GetAttr("key_func", &key_func_));
36 OP_REQUIRES_OK(ctx, ctx->GetAttr("reduce_func", &reduce_func_));
37 OP_REQUIRES_OK(ctx, ctx->GetAttr("window_size_func", &window_size_func_));
38 OP_REQUIRES_OK(ctx, ctx->GetAttr("output_types", &output_types_));
39 OP_REQUIRES_OK(ctx, ctx->GetAttr("output_shapes", &output_shapes_));
40 }
41
42 void MakeDataset(OpKernelContext* ctx, DatasetBase* input,
43 DatasetBase** output) override {
44 // Get captured inputs for the key, reduce, and window_size functions.
45 OpInputList key_func_other_argument_inputs;
46 OP_REQUIRES_OK(ctx, ctx->input_list("key_func_other_arguments",
47 &key_func_other_argument_inputs));
48 std::vector<Tensor> key_func_other_arguments;
49 key_func_other_arguments.reserve(key_func_other_argument_inputs.size());
50 for (const Tensor& t : key_func_other_argument_inputs) {
51 key_func_other_arguments.push_back(t);
52 }
53 OpInputList reduce_func_other_argument_inputs;
54 OP_REQUIRES_OK(ctx, ctx->input_list("reduce_func_other_arguments",
55 &reduce_func_other_argument_inputs));
56 std::vector<Tensor> reduce_func_other_arguments;
57 reduce_func_other_arguments.reserve(
58 reduce_func_other_argument_inputs.size());
59 for (const Tensor& t : reduce_func_other_argument_inputs) {
60 reduce_func_other_arguments.push_back(t);
61 }
62 OpInputList window_size_func_other_argument_inputs;
63 OP_REQUIRES_OK(ctx,
64 ctx->input_list("window_size_func_other_arguments",
65 &window_size_func_other_argument_inputs));
66 std::vector<Tensor> window_size_func_other_arguments;
67 window_size_func_other_arguments.reserve(
68 window_size_func_other_argument_inputs.size());
69 for (const Tensor& t : window_size_func_other_argument_inputs) {
70 window_size_func_other_arguments.push_back(t);
71 }
72 // TODO(mrry): Refactor CapturedFunction to share the runtime
73 // state between multiple functions?
74 std::unique_ptr<CapturedFunction> captured_key_func;
75 OP_REQUIRES_OK(ctx, CapturedFunction::Create(
76 key_func_, std::move(key_func_other_arguments),
77 &captured_key_func));
78 std::unique_ptr<CapturedFunction> captured_reduce_func;
79 OP_REQUIRES_OK(
80 ctx, CapturedFunction::Create(reduce_func_,
81 std::move(reduce_func_other_arguments),
82 &captured_reduce_func));
83 std::unique_ptr<CapturedFunction> captured_window_size_func;
84 OP_REQUIRES_OK(
85 ctx, CapturedFunction::Create(
86 window_size_func_, std::move(window_size_func_other_arguments),
87 &captured_window_size_func));
88
89 *output = new Dataset(
90 ctx, input, key_func_, reduce_func_, window_size_func_,
91 std::move(captured_key_func), std::move(captured_reduce_func),
92 std::move(captured_window_size_func), output_types_, output_shapes_);
93 }

96 class Dataset : public GraphDatasetBase {
97 public:
98 Dataset(OpKernelContext* ctx, const DatasetBase* input,
99 const NameAttrList& key_func, const NameAttrList& reduce_func,
100 const NameAttrList& window_size_func,
101 std::unique_ptr<CapturedFunction> captured_key_func,
102 std::unique_ptr<CapturedFunction> captured_reduce_func,
103 std::unique_ptr<CapturedFunction> captured_window_size_func,
104 const DataTypeVector& output_types,
105 const std::vector<PartialTensorShape>& output_shapes)
106 : GraphDatasetBase(ctx),
107 input_(input),
108 key_func_(key_func),
109 reduce_func_(reduce_func),
110 window_size_func_(window_size_func),
111 captured_key_func_(std::move(captured_key_func)),
112 captured_reduce_func_(std::move(captured_reduce_func)),
113 captured_window_size_func_(std::move(captured_window_size_func)),
114 output_types_(output_types),
115 output_shapes_(output_shapes) {
116 input_->Ref();
117 }
118
119 ~Dataset() override { input_->Unref(); }
120
121 std::unique_ptr<IteratorBase> MakeIterator(
122 const string& prefix) const override {
123 return std::unique_ptr<IteratorBase>(
124 new Iterator({this, strings::StrCat(prefix, "::GroupByWindow")}));
125 }
126
127 const DataTypeVector& output_dtypes() const override {
128 return output_types_;
129 }

131 return output_shapes_;
132 }

134 string DebugString() override { return "GroupByWindowDatasetOp::Dataset"; }

136 protected:
137 Status AsGraphDefInternal(OpKernelContext* ctx, DatasetGraphDefBuilder* b,
138 Node** output) const override {
139 TF_RETURN_IF_ERROR(b->AddFunction(ctx, key_func_.name()));
140 TF_RETURN_IF_ERROR(b->AddFunction(ctx, reduce_func_.name()));
141 TF_RETURN_IF_ERROR(b->AddFunction(ctx, window_size_func_.name()));
142 Node* input_graph_node = nullptr;
143 TF_RETURN_IF_ERROR(b->AddParentDataset(ctx, input_, &input_graph_node));
144
145 std::vector<Node*> key_func_other_arguments_node;
146 DataTypeVector key_func_other_arguments_types;
147 TF_RETURN_IF_ERROR(OtherArgumentsNodeAndType(
148 b, captured_key_func_, &key_func_other_arguments_node,
149 &key_func_other_arguments_types));
150
151 std::vector<Node*> reduce_func_other_arguments_node;
152 DataTypeVector reduce_func_other_arguments_types;
153 TF_RETURN_IF_ERROR(OtherArgumentsNodeAndType(
154 b, captured_reduce_func_, &reduce_func_other_arguments_node,
155 &reduce_func_other_arguments_types));
156
157 std::vector<Node*> window_size_func_other_arguments_node;
158 DataTypeVector window_size_func_other_arguments_types;
159 TF_RETURN_IF_ERROR(OtherArgumentsNodeAndType(
160 b, captured_window_size_func_, &window_size_func_other_arguments_node,
161 &window_size_func_other_arguments_types));
162
163 AttrValue key_func;
164 b->BuildAttrValue(key_func_, &key_func);
165 AttrValue reduce_func;
166 b->BuildAttrValue(reduce_func_, &reduce_func);
167 AttrValue window_size_func;
168 b->BuildAttrValue(window_size_func_, &window_size_func);
169
170 AttrValue key_func_other_arguments_types_attr;
171 b->BuildAttrValue(key_func_other_arguments_types,
172 &key_func_other_arguments_types_attr);
173 AttrValue reduce_func_other_arguments_types_attr;
174 b->BuildAttrValue(reduce_func_other_arguments_types,
175 &reduce_func_other_arguments_types_attr);
176 AttrValue window_size_func_other_arguments_types_attr;
177 b->BuildAttrValue(window_size_func_other_arguments_types,
178 &window_size_func_other_arguments_types_attr);
179
180 TF_RETURN_IF_ERROR(b->AddDataset(
181 this, {{0, input_graph_node}},
182 {{1, key_func_other_arguments_node},
183 {2, reduce_func_other_arguments_node},
184 {3, window_size_func_other_arguments_node}},
185 {{"key_func", key_func},
186 {"reduce_func", reduce_func},
187 {"window_size_func", window_size_func},
188 {"Tkey_func_other_arguments", key_func_other_arguments_types_attr},
189 {"Treduce_func_other_arguments",
190 reduce_func_other_arguments_types_attr},
191 {"Twindow_size_func_other_arguments",
192 window_size_func_other_arguments_types_attr}},
193 output));
194 return Status::OK();
195 }

197 private:
198 class Iterator : public DatasetIterator<Dataset> {
199 public:
200 explicit Iterator(const Params& params)
201 : DatasetIterator<Dataset>(params),
202 input_impl_(params.dataset->input_->MakeIterator(params.prefix)) {}
203
204 Status GetNextInternal(IteratorContext* ctx,
205 std::vector<Tensor>* out_tensors,
206 bool* end_of_sequence) override {
207 mutex_lock l(mu_);
208 do {
209 if (current_group_iterator_) {
210 // We are currently processing a group, so try to get the
211 // next element.
212 bool end_of_group;
213 TF_RETURN_IF_ERROR(current_group_iterator_->GetNext(
214 ctx, out_tensors, &end_of_group));
215 if (!end_of_group) {
216 // Produce the subelement as output.
217 *end_of_sequence = false;
218 return Status::OK();
219 }
220 // We have reached the end of the current group, so maybe move on
221 // to the next group.
222 current_group_iterator_.reset();
223 groups_.erase(current_key_);
224 }
225
226 // Iterate through the input dataset until we get a full
227 // group, or reach the end.
228 while (!end_of_input_) {
229 std::vector<Tensor> next_input_element;
230 TF_RETURN_IF_ERROR(
231 input_impl_->GetNext(ctx, &next_input_element, &end_of_input_));
232
233 if (!end_of_input_) {
234 // Run the key function on the input element to identify its
235 // group.
236 std::vector<Tensor> key_func_output;
237 TF_RETURN_IF_ERROR(
238 dataset()->captured_key_func_->RunWithBorrowedArgs(
239 ctx, next_input_element, &key_func_output));
240
241 if (key_func_output.size() != 1 ||
242 key_func_output[0].dtype() != DT_INT64 ||
243 key_func_output[0].NumElements() != 1) {
244 // TODO(mrry): Support non-int64 keys.
245 return errors::InvalidArgument(
246 "`key_func` must return a scalar int64.");
247 }
248 const int64 key = key_func_output[0].scalar<int64>()();
249
250 if (window_sizes_.find(key) == window_sizes_.end()) {
251 // Run the window size function on the key to identify its
252 // window size.
253 std::vector<Tensor> window_size_func_output;
254 TF_RETURN_IF_ERROR(dataset()->captured_window_size_func_->Run(
255 ctx, std::move(key_func_output), &window_size_func_output));
256
257 if (window_size_func_output.size() != 1 ||
258 window_size_func_output[0].dtype() != DT_INT64 ||
259 window_size_func_output[0].NumElements() != 1) {
260 // TODO(mrry): Support non-int64 window sizes.
261 return errors::InvalidArgument(
262 "`window_size_func` must return a scalar int64.");
263 }
264 const int64 window_size =
265 window_size_func_output[0].scalar<int64>()();
266 if (window_size <= 0) {
267 return errors::InvalidArgument(
268 "Window size must be greater than zero, but got ",
269 window_size, ".");
270 }
271 window_sizes_[key] = window_size;
272 }
273
274 const int64 window_size = window_sizes_[key];
275
276 std::vector<std::vector<Tensor>>& group = groups_[key];
277 group.push_back(std::move(next_input_element));
278
279 if (group.size() == window_size) {
280 current_key_ = key;
281 TF_RETURN_IF_ERROR(StartFlushingGroup(ctx, key));
282 break;
283 }
284 }
285 }
286
287 if (end_of_input_) {
288 if (!groups_.empty()) {
289 // We have consumed all of the input, so flush an
290 // arbitrarily chosen group.
291 current_key_ = groups_.begin()->first;
292 TF_RETURN_IF_ERROR(
293 StartFlushingGroup(ctx, groups_.begin()->first));
294 }
295 }
296 } while (current_group_iterator_ || !end_of_input_);
297
298 *end_of_sequence = true;
299 return Status::OK();
300 }
301
302 protected:
303 Status SaveInternal(IteratorStateWriter* writer) override {
304 mutex_lock l(mu_);
305 TF_RETURN_IF_ERROR(SaveParent(writer, input_impl_));
306
307 if (end_of_input_) {
308 TF_RETURN_IF_ERROR(
309 writer->WriteScalar(full_name("end_of_input"), ""));
310 }
311
312 // Saving groups_
313 if (!groups_.empty()) {
314 TF_RETURN_IF_ERROR(
315 writer->WriteScalar(full_name("groups_size"), groups_.size()));
316 int idx = 0;
317 for (auto it = groups_.begin(); it != groups_.end(); it++) {
318 int64 key = it->first;
319 TF_RETURN_IF_ERROR(writer->WriteScalar(
320 full_name(strings::StrCat("groups_[", idx, "]->key")), key));
321 TF_RETURN_IF_ERROR(SaveGroup(
322 writer, full_name(strings::StrCat("groups_[", idx, "]")),
323 it->second));
324 idx++;
325 }
326 }
327
328 // Saving window_sizes_
329 if (!window_sizes_.empty()) {
330 TF_RETURN_IF_ERROR(writer->WriteScalar(full_name("window_sizes_size"),
331 window_sizes_.size()));
332 int idx = 0;
333 for (auto it = window_sizes_.begin(); it != window_sizes_.end();
334 it++) {
335 TF_RETURN_IF_ERROR(writer->WriteScalar(
336 full_name(strings::StrCat("window_sizes_[", idx, "]->key")),
337 it->first));
338 TF_RETURN_IF_ERROR(writer->WriteScalar(
339 full_name(strings::StrCat("window_sizes_[", idx, "]->value")),
340 it->second));
341 idx++;
342 }
343 }
344
345 if (current_group_iterator_) {
346 TF_RETURN_IF_ERROR(SaveParent(writer, current_group_iterator_));
347
348 // Saving current_key_
349 TF_RETURN_IF_ERROR(
350 writer->WriteScalar(full_name("current_key"), current_key_));
351 } else {
352 TF_RETURN_IF_ERROR(writer->WriteScalar(
353 full_name("current_iterator_not_initialized"), ""));
354 }
355
356 return Status::OK();
357 }
358
359 Status RestoreInternal(IteratorContext* ctx,
360 IteratorStateReader* reader) override {
361 mutex_lock l(mu_);
362 TF_RETURN_IF_ERROR(RestoreParent(ctx, reader, input_impl_));
363
364 if (reader->Contains(full_name("end_of_input"))) end_of_input_ = true;
365
366 // Restoring groups
367 if (reader->Contains(full_name("groups_size"))) {
368 int64 size;
369 TF_RETURN_IF_ERROR(
370 reader->ReadScalar(full_name("groups_size"), &size));
371 for (int idx = 0; idx < size; idx++) {
372 int64 key;
373 TF_RETURN_IF_ERROR(reader->ReadScalar(
374 full_name(strings::StrCat("groups_[", idx, "]->key")), &key));
375 std::vector<std::vector<Tensor>> group;
376 TF_RETURN_IF_ERROR(RestoreGroup(
377 reader, full_name(strings::StrCat("groups_[", idx, "]")),
378 &group));
379 groups_[key] = group;
380 }
381 }
382
383 // Restoring Windows
384 if (reader->Contains(full_name("window_sizes_size"))) {
385 int64 size;
386 TF_RETURN_IF_ERROR(
387 reader->ReadScalar(full_name("window_sizes_size"), &size));
388 for (int idx = 0; idx < size; idx++) {
389 int64 key;
390 TF_RETURN_IF_ERROR(reader->ReadScalar(
391 full_name(strings::StrCat("window_sizes_[", idx, "]->key")),
392 &key));
393 TF_RETURN_IF_ERROR(reader->ReadScalar(
394 full_name(strings::StrCat("window_sizes_[", idx, "]->value")),
395 &window_sizes_[key]));
396 }
397 }
398
399 if (reader->Contains(full_name("current_iterator_not_initialized"))) {
400 current_group_iterator_.reset();
401 } else {
402 // Restore current_key_
403 TF_RETURN_IF_ERROR(
404 reader->ReadScalar(full_name("current_key"), &current_key_));
405
406 // Initialize current_group_iterator_
407 TF_RETURN_IF_ERROR(StartFlushingGroup(ctx, current_key_));
408 // Restore current_group_iterator_ state
409 TF_RETURN_IF_ERROR(
410 RestoreParent(ctx, reader, current_group_iterator_));
411 }
412 return Status::OK();
413 }
414
415 private:
416 Status SaveGroup(IteratorStateWriter* writer, const string& name,
417 const std::vector<std::vector<Tensor>>& group)
418 EXCLUSIVE_LOCKS_REQUIRED(mu_) {
419 TF_RETURN_IF_ERROR(
420 writer->WriteScalar(strings::StrCat(name, "_size"), group.size()));
421 for (int i = 0; i < group.size(); i++) {
422 TF_RETURN_IF_ERROR(writer->WriteScalar(
423 strings::StrCat(name, "[", i, "]_size"), group[i].size()));
424 for (int j = 0; j < group[i].size(); j++) {
425 TF_RETURN_IF_ERROR(writer->WriteTensor(
426 strings::StrCat(name, "[", i, "][", j, "]"), group[i][j]));
427 }
428 }
429 return Status::OK();
430 }
431
432 Status RestoreGroup(IteratorStateReader* reader, const string& name,
433 std::vector<std::vector<Tensor>>* group)
434 EXCLUSIVE_LOCKS_REQUIRED(mu_) {
435 int64 group_size;
436 TF_RETURN_IF_ERROR(
437 reader->ReadScalar(strings::StrCat(name, "_size"), &group_size));
438 group->resize(group_size);
439 for (int i = 0; i < group_size; i++) {
440 int64 vector_size;
441 TF_RETURN_IF_ERROR(reader->ReadScalar(
442 strings::StrCat(name, "[", i, "]_size"), &vector_size));
443 group->at(i).resize(vector_size);
444 for (int j = 0; j < vector_size; j++) {
445 TF_RETURN_IF_ERROR(reader->ReadTensor(
446 strings::StrCat(name, "[", i, "][", j, "]"), &group->at(i)[j]));
447 }
448 }
449 return Status::OK();
450 }
451
452 Status StartFlushingGroup(IteratorContext* ctx, int64 key)
453 EXCLUSIVE_LOCKS_REQUIRED(mu_) {
454 DatasetBase* group_dataset;
455 TF_RETURN_IF_ERROR(NewWindowDataset(
456 groups_[key], dataset()->input_->output_dtypes(),
457 dataset()->input_->output_shapes(), &group_dataset));
458
459 Tensor key_arg(DT_INT64, TensorShape({}));
460 key_arg.scalar<int64>()() = key;
461
462 Tensor group_dataset_arg(DT_VARIANT, TensorShape({}));
463 TF_RETURN_IF_ERROR(
464 StoreDatasetInVariantTensor(group_dataset, &group_dataset_arg));
465
466 std::vector<Tensor> args(
467 {std::move(key_arg), std::move(group_dataset_arg)});
468 std::vector<Tensor> return_values;
469 TF_RETURN_IF_ERROR(dataset()->captured_reduce_func_->Run(
470 ctx, std::move(args), &return_values));
471
472 if (!(return_values.size() == 1 &&
473 return_values[0].dtype() == DT_VARIANT &&
474 TensorShapeUtils::IsScalar(return_values[0].shape()))) {
475 return errors::InvalidArgument(
476 "`reduce_func` must return a single scalar of dtype "
477 "DT_VARIANT.");
478 }
479
480 // Retrieve the dataset that was created in `f`.
481 // `returned_dataset` is borrowed from the `return_values[0]`.
482 DatasetBase* returned_dataset;
483 TF_RETURN_IF_ERROR(
484 GetDatasetFromVariantTensor(return_values[0], &returned_dataset));
485
486 // Create an iterator for the dataset that was returned by `f`.
487 current_group_iterator_ = returned_dataset->MakeIterator(prefix());
488 return Status::OK();
489 }
490
491 mutex mu_;
492 std::unique_ptr<IteratorBase> input_impl_ GUARDED_BY(mu_);
493 // TODO(mrry): Optimize for dense key space if appropriate.
494 bool end_of_input_ GUARDED_BY(mu_) = false;
495 int64 current_key_ GUARDED_BY(mu_);
496 std::map<int64, std::vector<std::vector<Tensor>>> groups_ GUARDED_BY(mu_);
497 std::unique_ptr<IteratorBase> current_group_iterator_ GUARDED_BY(mu_);
498 std::map<int64, int64> window_sizes_ GUARDED_BY(mu_);
499 };
500
501 Status OtherArgumentsNodeAndType(
502 DatasetGraphDefBuilder* b,
503 const std::unique_ptr<CapturedFunction>& captured_func,
504 std::vector<Node*>* other_arguments_node,
505 DataTypeVector* other_arguments_types) const {
506 other_arguments_node->reserve(captured_func->captured_inputs().size());
507 other_arguments_types->reserve(captured_func->captured_inputs().size());
508 for (const Tensor& t : captured_func->captured_inputs()) {
509 Node* node;
510 TF_RETURN_IF_ERROR(b->AddTensor(t, &node));
511 other_arguments_node->emplace_back(node);
512 other_arguments_types->emplace_back(t.dtype());
513 }
514 return Status::OK();
515 }
516
517 const DatasetBase* const input_;
518 const NameAttrList key_func_;
519 const NameAttrList reduce_func_;
520 const NameAttrList window_size_func_;
521 const std::unique_ptr<CapturedFunction> captured_key_func_;
522 const std::unique_ptr<CapturedFunction> captured_reduce_func_;
523 const std::unique_ptr<CapturedFunction> captured_window_size_func_;
524 const DataTypeVector output_types_;
525 const std::vector<PartialTensorShape> output_shapes_;
526 };
527
528 const int graph_def_version_;
529 DataTypeVector output_types_;
530 std::vector<PartialTensorShape> output_shapes_;
531 NameAttrList key_func_;
532 NameAttrList reduce_func_;
533 NameAttrList window_size_func_;
534};
