35// A `CapturedFunction` encapsulates a TensorFlow function and all of
36// the runtime support required to execute it.
37//
38// The `Dataset`-related classes use `CapturedFunction` to execute
39// TensorFlow functions outside a the normal `OpKernel::Compute()`
40// context.
41class CapturedFunction {
42 public:
43 // NOTE(mrry): The `captured_inputs` are passed by value. For
44 // efficiency, you are recommended to move this argument into the call.
45 static Status Create(const NameAttrList& func,
46 std::vector<Tensor> captured_inputs,
47 std::unique_ptr<CapturedFunction>* out_function);
48
49 ~CapturedFunction();
50
51 // Runs the "Captured function" using the given FLR and caches the lib and
52 // handle generated during instantiation. If Run is called with a different
53 // lib afterwards, generates an error. This method takes ownership of the
54 // tensors in `args`, in order to be able to deallocate them as early as
55 // possible. Use `RunWithBorrowedArgs()` if the caller needs to retain
56 // ownership of the `args`.
57 Status Run(IteratorContext* ctx, std::vector<Tensor>&& args,
58 std::vector<Tensor>* rets);
59
60 // Synchronously runs the captured function on the given `args`, and stores
61 // the results in `*rets`. Prefer to use `Run()` or `RunAsync()` when
62 // possible.
63 Status RunWithBorrowedArgs(IteratorContext* ctx,
64 const std::vector<Tensor>& args,
65 std::vector<Tensor>* rets);
66
67 // Explicitly instantiate this function for use in the given
68 // context. This method, and the context-less overload
69 // `RunInstantiated()` below can be useful for calling a captured
70 // function in cases where an `IteratorContext*` is not available
71 // (such as a destructor).
72 Status Instantiate(IteratorContext* ctx);
73
74 // Synchronously runs the captured function on the given `args`, and stores
75 // the results in `*rets`. Prefer to use `Run()` or `RunAsync()` when
76 // possible.
77 //
78 // REQUIRES: `this->Instantiate()` must have been called before this method.
79 Status RunInstantiated(const std::vector<Tensor>& args,
80 std::vector<Tensor>* rets);
81
82 // Asynchronously runs the captured function on the given `args`, stores
83 // the results in `*rets`, and calls the given `done` callback when the
84 // function returns. This method takes ownership of the tensors in `args`,
85 // in order to be able to deallocate them as early as possible.
86 void RunAsync(IteratorContext* ctx, std::vector<Tensor>&& args,
87 std::vector<Tensor>* rets,
88 FunctionLibraryRuntime::DoneCallback done);
89
90 // Returns that additional captured inputs that will be passed to the function
91 // when `Run*()` is called.
92 const std::vector<Tensor>& captured_inputs() { return captured_inputs_; }
93
94 // Returns a step ID for use when running a `CapturedFunction`.
95 static int64 generate_step_id() {
96 // Choose a step ID that is guaranteed not to clash with any
97 // Session-generated step ID. DirectSession only generates
98 // non-negative step IDs (contiguous, starting from 0), and
99 // MasterSession generates 56-bit random step IDs whose MSB is
100 // always 0, so a negative random step ID should suffice.
101 return -std::abs(static_cast<int64>(random::New64()));
102 }
103
104 private:
105 CapturedFunction(const NameAttrList& func,
106 std::vector<Tensor> captured_inputs);
107
108 Status MaybeInstantiate(IteratorContext* ctx,
109 FunctionLibraryRuntime::Handle* out_handle);
110
111 mutex mu_;
112 const NameAttrList func_;
113 FunctionLibraryRuntime* lib_ GUARDED_BY(mu_);
114 FunctionLibraryRuntime::Handle f_handle_ GUARDED_BY(mu_);
115 const std::vector<Tensor> captured_inputs_;
116 DataTypeSlice ret_types_;
117 std::function<void(std::function<void()>)> captured_runner_ = nullptr;
118
119 TF_DISALLOW_COPY_AND_ASSIGN(CapturedFunction);
120};
