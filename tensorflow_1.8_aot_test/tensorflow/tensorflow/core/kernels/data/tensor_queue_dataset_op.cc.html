30bool IsGreaterEqualToOrCompatibleWith(const PartialTensorShape& a,
31 const PartialTensorShape& b) {
32 // Returns true if dims[a] >= dims[b], or are compatible.
33 if (a.unknown_rank()) return true;
34 if (a.dims() != b.dims()) return false;
35 for (int d = 0; d < a.dims(); ++d) {
36 if (a.dim_size(d) == -1 || b.dim_size(d) == -1) continue;
37 if (a.dim_size(d) < b.dim_size(d)) return false;
38 }
39 return true;
40}

42DataTypeVector PrependQueueType(const DataTypeVector& dtypes) {
43 DataTypeVector out;
44 out.reserve(dtypes.size() + 1);
45 out.push_back(DT_VARIANT); // The queue component.
46 for (const DataType& d : dtypes) out.push_back(d);
47 return out;
48}
49
50std::vector<PartialTensorShape> PrependQueueShapeWithBatch(
51 const std::vector<PartialTensorShape>& shapes) {
52 std::vector<PartialTensorShape> out;
53 out.reserve(shapes.size() + 1);
54 out.emplace_back(PartialTensorShape({-1})); // The queue component.
55 for (PartialTensorShape s : shapes) {
56 s.InsertDim(0, -1); // Unknown batch size.
57 out.push_back(std::move(s));
58 }
59 return out;
60}

64class PrependFromQueueAndPaddedBatchDataset : public GraphDatasetBase {
65 public:
66 PrependFromQueueAndPaddedBatchDataset(
67 OpKernelContext* ctx, const int64 batch_size, const DatasetBase* input,
68 const DataTypeVector& dtypes,
69 const std::vector<PartialTensorShape>& shapes,
70 std::vector<Tensor> padding_values)
71 : GraphDatasetBase(ctx),
72 batch_size_(batch_size),
73 input_(input),
74 dtypes_(dtypes),
75 shapes_(shapes),
76 padding_values_(std::move(padding_values)),
77 dtypes_with_queue_(PrependQueueType(dtypes)),
78 batched_shapes_with_queue_(PrependQueueShapeWithBatch(shapes)) {
79 input_->Ref();
80 }
81
82 ~PrependFromQueueAndPaddedBatchDataset() override { input_->Unref(); }
83
84 std::unique_ptr<IteratorBase> MakeIterator(
85 const string& prefix) const override {
86 return std::unique_ptr<IteratorBase>(new Iterator(
87 {this, strings::StrCat(prefix, "::PrependFromQueueAndPaddedBatch")}));
88 }
89
90 const DataTypeVector& output_dtypes() const override {
91 return dtypes_with_queue_;
92 }
93 const std::vector<PartialTensorShape>& output_shapes() const override {
94 return batched_shapes_with_queue_;
95 }
96
97 string DebugString() override {
98 return "PrependFromQueueAndPaddedBatchDatasetOp::Dataset";
99 }
100
101 protected:
102 Status AsGraphDefInternal(OpKernelContext* ctx, DatasetGraphDefBuilder* b,
103 Node** output) const override {
104 Node* input_graph = nullptr;
105 TF_RETURN_IF_ERROR(b->AddParentDataset(ctx, input_, &input_graph));
106 Node* batch_size = nullptr;
107 TF_RETURN_IF_ERROR(b->AddScalar(batch_size_, &batch_size));
108
109 std::vector<Node*> padded_shapes;
110 padded_shapes.reserve(shapes_.size());
111 for (int i = 0; i < shapes_.size(); i++) {
112 Node* node;
113 Tensor t(DT_INT64, TensorShape({shapes_[i].dims()}));
114 for (int j = 0; j < shapes_[i].dims(); j++) {
115 t.vec<int64>()(j) = shapes_[i].dim_size(j);
116 }
117 TF_RETURN_IF_ERROR(b->AddTensor(t, &node));
118 padded_shapes.emplace_back(node);
119 }
120
121 std::vector<Node*> padding_values;
122 padding_values.reserve(padding_values_.size());
123 for (const Tensor& t : padding_values_) {
124 Node* node;
125 TF_RETURN_IF_ERROR(b->AddTensor(t, &node));
126 padding_values.emplace_back(node);
127 }
128
129 AttrValue output_types;
130 b->BuildAttrValue(dtypes_, &output_types);
131
132 AttrValue output_shapes;
133 b->BuildAttrValue(batched_shapes_with_queue_, &output_shapes);
134
135 AttrValue N;
136 b->BuildAttrValue<int64>(shapes_.size(), &N);
137
138 TF_RETURN_IF_ERROR(b->AddDataset(this, {{0, input_graph}, {1, batch_size}},
139 {{2, padded_shapes}, {3, padding_values}},
140 {{"Toutput_types", output_types},
141 {"output_shapes", output_shapes},
142 {"N", N}},
143 output));
144
145 return Status::OK();
146 }

151 class Iterator
152 : public DatasetIterator<PrependFromQueueAndPaddedBatchDataset> {
153 public:
154 explicit Iterator(const Params& params)
155 : DatasetIterator<PrependFromQueueAndPaddedBatchDataset>(params),
156 queue_(new TensorQueue(/*input_impl*/
157 params.dataset->input_->MakeIterator(
158 params.prefix),
159 params.dataset->dtypes_,
160 params.dataset->shapes_)) {}
161
162 ~Iterator() override { queue_->Unref(); }
163
164 Status GetNextInternal(IteratorContext* ctx,
165 std::vector<Tensor>* out_tensors,
166 bool* end_of_sequence) override {
167 std::vector<std::vector<Tensor>> batch;
168 TF_RETURN_IF_ERROR(queue_->GetNext(ctx, dataset()->batch_size_, &batch,
169 end_of_sequence));
170 const auto& dtypes = dataset()->dtypes_;
171 const auto& shapes = dataset()->shapes_;
172 const auto& input_shapes = dataset()->input_->output_shapes();
173 const auto& padding_values = dataset()->padding_values_;
174 const int64 batch_size = batch.size();
175 out_tensors->reserve(dtypes.size());
176
177 std::vector<TensorShape> max_shapes; // Of non-queue components.
178 for (int i = 0; i < dtypes.size(); ++i) {
179 const PartialTensorShape& shape = shapes[i];
180 TensorShape out_shape({batch_size});
181 for (int r = 0; r < shape.dims(); ++r) {
182 if (shape.dim_size(r) >= 0) {
183 // padded_shape[r] is known.
184 out_shape.AddDim(shape.dim_size(r));
185 } else {
186 // padded_shape[r] is unknown, find the maximum across
187 // the batch.
188 int64 dim = 0;
189 for (int b = 0; b < batch.size(); ++b) {
190 dim = std::max(dim, batch[b][i].dim_size(r));
191 }
192 out_shape.AddDim(dim);
193 }
194 }
195 max_shapes.push_back(std::move(out_shape));
196 }
197
198 Tensor queues_t(cpu_allocator(), DT_VARIANT, TensorShape({batch_size}));
199 if (!batch.empty()) {
200 auto queues = queues_t.flat<Variant>();
201 Variant& queue_inserter = queues(0);
202 queue_inserter = TensorQueueInserter();
203 queue_inserter.get<TensorQueueInserter>()->set_queue(queue_);
204 for (int b = 1; b < batch.size(); ++b) {
205 // Copy the TensorQueueInserter. Each copy increments the
206 // Ref on the queue_.
207 queues(b) = queues(0);
208 }
209 }
210 out_tensors->push_back(std::move(queues_t));
211
212 for (int i = 0; i < max_shapes.size(); ++i) {
213 Tensor component(cpu_allocator(), dtypes[i], max_shapes[i]);
214 // Try hard to take the fast path.
215 if (shapes[i].IsFullyDefined() &&
216 shapes[i].IsIdenticalTo(input_shapes[i])) {
217 // Take the fast path if we know all the shapes statically.
218 for (int64 b = 0; b < batch.size(); ++b) {
219 TF_RETURN_IF_ERROR(
220 batch_util::CopyElementToSlice(batch[b][i], &component, b));
221 }
222 } else {
223 TF_RETURN_IF_ERROR(
224 batch_util::SetElementZero(&component, padding_values[i]));
225 for (int64 b = 0; b < batch.size(); ++b) {
226 if (batch[b][i].shape() == max_shapes[i]) {
227 TF_RETURN_IF_ERROR(
228 batch_util::CopyElementToSlice(batch[b][i], &component, b));
229 } else {
230 TF_RETURN_IF_ERROR(batch_util::CopyElementToLargerSlice(
231 batch[b][i], &component, b));
232 }
233 }
234 }
235 out_tensors->push_back(std::move(component));
236 }
237
238 // end_of_sequence was set before we populated out_tensors, so
239 // it's ok to return now.
240 return Status::OK();
241 }

249 class TensorQueue : public core::RefCounted {
250 public:
251 TensorQueue(std::unique_ptr<IteratorBase> input_impl,
252 const DataTypeVector& dtypes,
253 const std::vector<PartialTensorShape>& shapes)
254 : dtypes_(dtypes),
255 shapes_(shapes),
256 input_impl_(std::move(input_impl)) {}
257
258 void MaybeWaitForNotificationLocked(mutex_lock* lock)
259 EXCLUSIVE_LOCKS_REQUIRED(mu_) {
260 // This essentially just releases the lock and immediately relocks.
261 cv_.wait_for(*lock, std::chrono::milliseconds(0));
262 }
263
264 void NotifyLocked() EXCLUSIVE_LOCKS_REQUIRED(mu_) { cv_.notify_all(); }
265
266 Status GetNext(IteratorContext* ctx, const int64 batch_size,
267 std::vector<std::vector<Tensor>>* batch,
268 bool* end_of_sequence) {
269 mutex_lock lock(mu_);
270
271 *end_of_sequence = false;
272
273 for (int64 b = 0; b < batch_size;) {
274 if (!entries_.empty()) {
275 batch->push_back(std::move(entries_.front()));
276 entries_.pop_front();
277 ++b;
278 continue;
279 } else {
280 if (input_impl_) {
281 // There's still input coming in.
282 std::vector<Tensor> tensors;
283 bool input_end;
284 TF_RETURN_IF_ERROR(
285 input_impl_->GetNext(ctx, &tensors, &input_end));
286 if (!input_end) {
287 batch->push_back(std::move(tensors));
288 ++b;
289 continue;
290 } else {
291 input_impl_.reset();
292 }
293 }
294 if (!input_impl_) {
295 // There's no more input coming in.
296 if (RefCountIsOne()) {
297 // No TensorQueueInserters in the wild.
298 if (batch->empty()) {
299 *end_of_sequence = true;
300 }
301 break;
302 } else {
303 MaybeWaitForNotificationLocked(&lock);
304 // If there's data available, try to add entries again.
305 // Otherwise return a smaller batch and hope the next
306 // iterator request has a non-empty or unused queue_.
307 if (entries_.empty()) {
308 break;
309 }
310 }
311 }
312 }
313 } // for (int64 b = ... batch_size)
314 return Status::OK();
315 }
316
317 Status Insert(const std::vector<Tensor>& tensors) {
318 if (tensors.size() != dtypes_.size()) {
319 return errors::InvalidArgument(
320 "TensorQueue::Insert: mismatched number of tensors. Queue "
321 "expects ",
322 dtypes_.size(), " tensors but tried to insert ", tensors.size());
323 }
324 for (int i = 0; i < tensors.size(); ++i) {
325 if (tensors[i].dtype() != dtypes_[i]) {
326 return errors::InvalidArgument(
327 "TensorQueue::Insert: mismatched dtypes at component ", i,
328 ". Attempted "
329 "to insert tensor of type ",
330 DataTypeString(tensors[i].dtype()),
331 " but queue expected type: ", DataTypeString(dtypes_[i]));
332 }
333 if (!shapes_[i].IsCompatibleWith(tensors[i].shape())) {
334 return errors::InvalidArgument(
335 "TensorQueue::Insert: mismatched shapes at component ", i,
336 ". Attempted "
337 "to insert tensor with shape ",
338 tensors[i].shape().DebugString(),
339 " but queue expected shape: ", shapes_[i].DebugString());
340 }
341 }
342 mutex_lock lock(mu_);
343 entries_.push_back(tensors);
344 NotifyLocked();
345 return Status::OK();
346 }
347
348 Status Save(Iterator* iter, IteratorStateWriter* writer) {
349 mutex_lock lock(mu_);
350 if (input_impl_) {
351 TF_RETURN_IF_ERROR(iter->SaveParent(writer, input_impl_));
352 } else {
353 TF_RETURN_IF_ERROR(
354 writer->WriteScalar(iter->full_name("input_exhausted"), ""));
355 }
356 TF_RETURN_IF_ERROR(writer->WriteScalar(iter->full_name("entries_size"),
357 entries_.size()));
358 for (int64 b = 0; b < entries_.size(); ++b) {
359 for (int i = 0; i < dtypes_.size(); ++i) {
360 TF_RETURN_IF_ERROR(
361 writer->WriteTensor(strings::StrCat(iter->full_name("entries"),
362 "[", b, "][", i, "]"),
363 entries_[b][i]));
364 }
365 }
366 return Status::OK();
367 }
368
369 Status Restore(Iterator* iter, IteratorContext* ctx,
370 IteratorStateReader* reader) {
371 mutex_lock l(mu_);
372 if (reader->Contains(iter->full_name("input_exhausted"))) {
373 input_impl_.reset();
374 } else {
375 input_impl_ = iter->dataset_input()->MakeIterator(iter->prefix());
376 TF_RETURN_IF_ERROR(iter->RestoreParent(ctx, reader, input_impl_));
377 }
378 entries_.clear();
379 int64 entries_size = -1;
380 TF_RETURN_IF_ERROR(
381 reader->ReadScalar(iter->full_name("entries_size"), &entries_size));
382 if (entries_size < 0) {
383 return errors::DataLoss(
384 "Expected entries_size key '", iter->full_name("entries_size"),
385 "' to have nonnegative value, but saw: ", entries_size);
386 }
387 for (int64 b = 0; b < entries_size; ++b) {
388 std::vector<Tensor> entry;
389 for (int i = 0; i < dtypes_.size(); ++i) {
390 Tensor value;
391 TF_RETURN_IF_ERROR(
392 reader->ReadTensor(strings::StrCat(iter->full_name("entries"),
393 "[", b, "][", i, "]"),
394 &value));
395 entry.push_back(std::move(value));
396 }
397 entries_.push_back(std::move(entry));
398 }
399 return Status::OK();
400 }
401
402 mutex* mu() { return &mu_; }

404 private:
405 DataTypeVector dtypes_;
406 std::vector<PartialTensorShape> shapes_;
407
408 mutex mu_;
409 std::unique_ptr<IteratorBase> input_impl_ GUARDED_BY(mu_);
410 std::deque<std::vector<Tensor>> entries_ GUARDED_BY(mu_);
411 condition_variable cv_ GUARDED_BY(mu_);
412 };

485class PrependFromQueueAndPaddedBatchDatasetOp : public UnaryDatasetOpKernel {
486 public:
487 explicit PrependFromQueueAndPaddedBatchDatasetOp(OpKernelConstruction* ctx)
488 : UnaryDatasetOpKernel(ctx) {
489 OP_REQUIRES_OK(ctx, ctx->GetAttr("Toutput_types", &output_types_));
490 }
491
492 void MakeDataset(OpKernelContext* ctx, DatasetBase* input,
493 DatasetBase** output) override {
494 int64 batch_size = 0;
495 OP_REQUIRES_OK(ctx,
496 ParseScalarArgument<int64>(ctx, "batch_size", &batch_size));
497 OP_REQUIRES(
498 ctx, batch_size > 0,
499 errors::InvalidArgument("Batch size must be greater than zero."));
500
501 OpInputList padded_shape_tensors;
502 OP_REQUIRES_OK(ctx,
503 ctx->input_list("padded_shapes", &padded_shape_tensors));
504 std::vector<PartialTensorShape> padded_shapes;
505 padded_shapes.reserve(padded_shape_tensors.size());
506 OP_REQUIRES(ctx,
507 padded_shape_tensors.size() == input->output_shapes().size(),
508 errors::InvalidArgument("Number of padded shapes (",
509 padded_shape_tensors.size(),
510 ") must match the number of components "
511 "in the input dataset's elements (",
512 input->output_shapes().size(), ")"));
513 for (const Tensor& padded_shape_t : padded_shape_tensors) {
514 OP_REQUIRES(ctx, TensorShapeUtils::IsVector(padded_shape_t.shape()),
515 errors::InvalidArgument("All padded shapes must be vectors"));
516 PartialTensorShape padded_shape;
517 OP_REQUIRES_OK(ctx, PartialTensorShape::MakePartialShape(
518 padded_shape_t.vec<int64>().data(),
519 padded_shape_t.NumElements(), &padded_shape));
520 padded_shapes.push_back(std::move(padded_shape));
521 }
522
523 OP_REQUIRES(
524 ctx, input->output_dtypes() == output_types_,
525 errors::InvalidArgument("Input dataset and this dataset "
526 "have different output_types: ",
527 DataTypeVectorString(input->output_dtypes()),
528 " and ", DataTypeVectorString(output_types_)));
529
530 for (int i = 0; i < input->output_shapes().size(); ++i) {
531 // Exclude the queue from the tensor_shapes calculation.
532 const PartialTensorShape& tensor_shape = padded_shapes[i];
533 OP_REQUIRES(
534 ctx,
535 IsGreaterEqualToOrCompatibleWith(tensor_shape,
536 input->output_shapes()[i]),
537 errors::InvalidArgument("Incompatible input shapes at component ", i,
538 " between input dataset this dataset: ",
539 input->output_shapes()[i].DebugString(),
540 " vs. ", tensor_shape.DebugString()));
541 }
542
543 OpInputList padding_values_list;
544 OP_REQUIRES_OK(ctx,
545 ctx->input_list("padding_values", &padding_values_list));
546 std::vector<Tensor> padding_values;
547 OP_REQUIRES(ctx,
548 padding_values_list.size() == input->output_shapes().size(),
549 errors::InvalidArgument(
550 "Number of padding values (", padding_values_list.size(),
551 ") must match the number of components in the input "
552 "dataset's elements (",
553 input->output_shapes().size(), ")"));
554 for (int i = 0; i < padding_values_list.size(); ++i) {
555 const Tensor& padding_value_t = padding_values_list[i];
556 OP_REQUIRES(
557 ctx, TensorShapeUtils::IsScalar(padding_value_t.shape()),
558 errors::InvalidArgument(
559 "All padding values must be scalars; but at component ", i,
560 " saw shape: ", padding_value_t.shape().DebugString()));
561 OP_REQUIRES(ctx, padding_value_t.dtype() == input->output_dtypes()[i],
562 errors::InvalidArgument(
563 "Mismatched type between padding value ", i,
564 " and input dataset's component ", i, ": ",
565 DataTypeString(padding_value_t.dtype()), " vs. ",
566 DataTypeString(input->output_dtypes()[i])));
567 padding_values.push_back(padding_value_t);
568 }
569
570 *output = new PrependFromQueueAndPaddedBatchDataset(
571 ctx, batch_size, input, output_types_, padded_shapes,
572 std::move(padding_values));
573 }
574
575 private:
576 DataTypeVector output_types_;
577};

583class EnqueueInQueueDatasetOp : public OpKernel {
584 public:
585 explicit EnqueueInQueueDatasetOp(OpKernelConstruction* ctx) : OpKernel(ctx) {}
586 void Compute(OpKernelContext* ctx) override {
587 using TensorQueueInserter =
588 PrependFromQueueAndPaddedBatchDataset::Iterator::TensorQueueInserter;
589
590 // TODO(ebrevdo): accept list of sequence lengths to do proper
591 // sub-slicing of tensors for placement into the queue?
592 const Tensor& tensor_queue_t = ctx->input(0);
593 OP_REQUIRES(ctx, TensorShapeUtils::IsVector(tensor_queue_t.shape()),
594 errors::InvalidArgument("queue must be a vector, saw shape: ",
595 tensor_queue_t.shape().DebugString()));
596 std::vector<const TensorQueueInserter*> inserters;
597 const int64 batch_size = tensor_queue_t.NumElements();
598 inserters.reserve(batch_size);
599 const Variant* variants = tensor_queue_t.flat<Variant>().data();
600 for (int i = 0; i < batch_size; ++i) {
601 const auto* inserter = variants[i].get<TensorQueueInserter>();
602 OP_REQUIRES(ctx, inserter != nullptr,
603 errors::InvalidArgument(
604 "Could not access TensorQueueInserter from queue[", i,
605 "]. Received variant: ", variants[i].DebugString()));
606 inserters.push_back(inserter);
607 }
608
609 OpInputList components;
610 OP_REQUIRES_OK(ctx, ctx->input_list("components", &components));
611 for (int i = 0; i < components.size(); ++i) {
612 OP_REQUIRES(
613 ctx,
614 components[i].dims() > 0 && components[i].dim_size(0) == batch_size,
615 errors::InvalidArgument(
616 "Expected component ", i, " to have batched shape [", batch_size,
617 ",...], but saw shape: ", components[i].shape().DebugString()));
618 }
619 std::vector<TensorShape> element_shapes;
620 for (int i = 0; i < components.size(); ++i) {
621 TensorShape element_shape = components[i].shape();
622 element_shape.RemoveDim(0);
623 element_shapes.push_back(std::move(element_shape));
624 }
625 for (int64 b = 0; b < batch_size; ++b) {
626 std::vector<Tensor> tensors;
627 tensors.reserve(components.size());
628 for (int i = 0; i < components.size(); ++i) {
629 Tensor t(components[i].dtype(), element_shapes[i]);
630 OP_REQUIRES_OK(ctx,
631 batch_util::CopySliceToElement(components[i], &t, b));
632 tensors.push_back(std::move(t));
633 }
634 // TODO(ebrevdo): Acquire the lock once for all inserters with
635 // the same underlying queue? Add InsertLocked?
636 OP_REQUIRES_OK(ctx, inserters[b]->Insert(tensors));
637 }
638 }
639};
