24// See documentation in ../ops/dataset_ops.cc for a high-level
25// description of the following op.
26
27class BatchDatasetOp : public UnaryDatasetOpKernel {
28 public:
29 explicit BatchDatasetOp(OpKernelConstruction* ctx)
30 : UnaryDatasetOpKernel(ctx) {}
31
32 void MakeDataset(OpKernelContext* ctx, DatasetBase* input,
33 DatasetBase** output) override {
34 int64 batch_size = 0;
35 OP_REQUIRES_OK(ctx,
36 ParseScalarArgument<int64>(ctx, "batch_size", &batch_size));
37 OP_REQUIRES(
38 ctx, batch_size > 0,
39 errors::InvalidArgument("Batch size must be greater than zero."));
40
41 *output = new Dataset(ctx, batch_size, input);
42 }
43
44 private:
45 class Dataset : public GraphDatasetBase {
46 public:
47 Dataset(OpKernelContext* ctx, int64 batch_size, const DatasetBase* input)
48 : GraphDatasetBase(ctx), batch_size_(batch_size), input_(input) {
49 input_->Ref();
50
51 // NOTE(mrry): Currently we implement "batch up to" semantics. If
52 // we could tell statically that the input dataset is infinite,
53 // then we could always report `batch_size` as the 0th dimension.
54 const auto& input_shapes = input_->output_shapes();
55 output_shapes_.reserve(input_shapes.size());
56 for (const auto& input_shape : input_shapes) {
57 output_shapes_.emplace_back(
58 PartialTensorShape({-1}).Concatenate(input_shape));
59 }
60 }
61
62 ~Dataset() override { input_->Unref(); }
63
64 std::unique_ptr<IteratorBase> MakeIterator(
65 const string& prefix) const override {
66 return std::unique_ptr<IteratorBase>(new Iterator(
67 Iterator::Params{this, strings::StrCat(prefix, "::Batch")}));
68 }
69
70 const DataTypeVector& output_dtypes() const override {
71 return input_->output_dtypes();
72 }
73
74 const std::vector<PartialTensorShape>& output_shapes() const override {
75 return output_shapes_;
76 }
77
78 string DebugString() override {
79 return strings::StrCat("BatchDatasetOp(", batch_size_, ")::Dataset");
80 }
81
82 protected:
83 Status AsGraphDefInternal(OpKernelContext* ctx, DatasetGraphDefBuilder* b,
84 Node** output) const override {
85 Node* input_graph_node = nullptr;
86 TF_RETURN_IF_ERROR(b->AddParentDataset(ctx, input_, &input_graph_node));
87 Node* batch_size = nullptr;
88 TF_RETURN_IF_ERROR(b->AddScalar(batch_size_, &batch_size));
89 TF_RETURN_IF_ERROR(
90 b->AddDataset(this, {input_graph_node, batch_size}, output));
91 return Status::OK();
92 }
93
94 private:
95 class Iterator : public DatasetIterator<Dataset> {
96 public:
97 explicit Iterator(const Params& params)
98 : DatasetIterator<Dataset>(params),
99 input_impl_(params.dataset->input_->MakeIterator(params.prefix)) {}
100
101 Status GetNextInternal(IteratorContext* ctx,
102 std::vector<Tensor>* out_tensors,
103 bool* end_of_sequence) override {
104 // Each row of `batch_elements` is a tuple of tensors from the
105 // input iterator.
106 std::vector<std::vector<Tensor>> batch_elements;
107 {
108 mutex_lock l(mu_);
109 if (!input_impl_) {
110 *end_of_sequence = true;
111 return Status::OK();
112 }
113 batch_elements.reserve(dataset()->batch_size_);
114 *end_of_sequence = false;
115 for (int i = 0; i < dataset()->batch_size_ && !*end_of_sequence;
116 ++i) {
117 std::vector<Tensor> batch_element_tuple;
118 TF_RETURN_IF_ERROR(input_impl_->GetNext(ctx, &batch_element_tuple,
119 end_of_sequence));
120 if (!*end_of_sequence) {
121 batch_elements.emplace_back(std::move(batch_element_tuple));
122 } else {
123 input_impl_.reset();
124 }
125 }
126 }
127
128 if (batch_elements.empty()) {
129 DCHECK(*end_of_sequence);
130 return Status::OK();
131 }
132
133 // Copy the retrieved batch elements into one output tensor
134 // per tuple component.
135 // NOTE(mrry): If the input or output sizes are statically
136 // known, we could potentially read the input values in-place
137 // into their respective slice locations. This would require a
138 // different GetNext() overload that supports zero-copy, and might
139 // make sense in an optimization pass.
140 const size_t num_tuple_components = batch_elements[0].size();
141 const int64 num_batch_elements = batch_elements.size();
142 for (size_t component_index = 0; component_index < num_tuple_components;
143 ++component_index) {
144 const Tensor& first_element = batch_elements[0][component_index];
145 TensorShape batch_component_shape({num_batch_elements});
146 batch_component_shape.AppendShape(first_element.shape());
147 Tensor batch_component(ctx->allocator({}), first_element.dtype(),
148 batch_component_shape);
149 // Build the output tuple component by copying one slice
150 // from each input element in the batch.
151 for (size_t i = 0; i < num_batch_elements; ++i) {
152 if (batch_elements[i][component_index].shape() !=
153 first_element.shape()) {
154 return errors::InvalidArgument(
155 "Cannot batch tensors with different shapes in component ",
156 component_index, ". First element had shape ",
157 first_element.shape().DebugString(), " and element ", i,
158 " had shape ",
159 batch_elements[i][component_index].shape().DebugString(),
160 ".");
161 }
162 TF_RETURN_IF_ERROR(batch_util::CopyElementToSlice(
163 std::move(batch_elements[i][component_index]), &batch_component,
164 i));
165 }
166 out_tensors->emplace_back(std::move(batch_component));
167 }
168 *end_of_sequence = false;
169 return Status::OK();
170 }
171
172 protected:
173 Status SaveInternal(IteratorStateWriter* writer) override {
174 mutex_lock l(mu_);
175 if (!input_impl_) {
176 TF_RETURN_IF_ERROR(
177 writer->WriteScalar(full_name("input_impl_empty"), ""));
178 } else {
179 TF_RETURN_IF_ERROR(SaveParent(writer, input_impl_));
180 }
181 return Status::OK();
182 }
183
184 Status RestoreInternal(IteratorContext* ctx,
185 IteratorStateReader* reader) override {
186 mutex_lock l(mu_);
187 if (!reader->Contains(full_name("input_impl_empty"))) {
188 TF_RETURN_IF_ERROR(RestoreParent(ctx, reader, input_impl_));
189 } else {
190 input_impl_.reset();
191 }
192 return Status::OK();
193 }
194
195 private:
196 mutex mu_;
197 std::unique_ptr<IteratorBase> input_impl_ GUARDED_BY(mu_);
198 };
199
200 const int64 batch_size_;
201 const DatasetBase* const input_;
202 std::vector<PartialTensorShape> output_shapes_;
203 };
204};
205
206REGISTER_KERNEL_BUILDER(Name("BatchDataset").Device(DEVICE_CPU),
207 BatchDatasetOp);
