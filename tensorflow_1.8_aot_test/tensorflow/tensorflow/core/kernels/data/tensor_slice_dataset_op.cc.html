24// See documentation in ../ops/dataset_ops.cc for a high-level
25// description of the following op.
26
27class TensorSliceDatasetOp : public DatasetOpKernel {
28 public:
29 explicit TensorSliceDatasetOp(OpKernelConstruction* ctx)
30 : DatasetOpKernel(ctx) {}
31
32 void MakeDataset(OpKernelContext* ctx, DatasetBase** output) override {
33 // Create a new TensorDatasetOp::Dataset, insert it in the step
34 // container, and return it as the output.
35 OpInputList inputs;
36 OP_REQUIRES_OK(ctx, ctx->input_list("components", &inputs));
37 std::vector<Tensor> components;
38 components.reserve(inputs.size());
39 OP_REQUIRES(ctx, inputs[0].dims() > 0,
40 errors::InvalidArgument(
41 "All components must be at least 1-dimensional"));
42 const int64 num_slices = inputs[0].dim_size(0);
43 for (const Tensor& t : inputs) {
44 components.push_back(t);
45 OP_REQUIRES(ctx, t.dims() > 0,
46 errors::InvalidArgument(
47 "All components must be at least 1-dimensional"));
48 OP_REQUIRES(
49 ctx, t.dim_size(0) == num_slices,
50 errors::InvalidArgument(
51 "All components must have the same size in the 0th dimension"));
52 }
53 *output = new Dataset(ctx, std::move(components));
54 }
55
56 private:
57 class Dataset : public GraphDatasetBase {
58 public:
59 explicit Dataset(OpKernelContext* ctx, std::vector<Tensor> tensors)
60 : GraphDatasetBase(ctx), tensors_(std::move(tensors)) {
61 for (const Tensor& t : tensors_) {
62 dtypes_.push_back(t.dtype());
63 gtl::InlinedVector<int64, 4> partial_dim_sizes;
64 // Handle scalar here. Check that everyone matches here? Or fail
65 // at runtime?
66 for (int i = 1; i < t.dims(); ++i) {
67 partial_dim_sizes.push_back(t.dim_size(i));
68 }
69 shapes_.emplace_back(std::move(partial_dim_sizes));
70 }
71 }
72
73 std::unique_ptr<IteratorBase> MakeIterator(
74 const string& prefix) const override {
75 return std::unique_ptr<IteratorBase>(
76 new Iterator({this, strings::StrCat(prefix, "::TensorSlice")}));
77 }
78
79 const DataTypeVector& output_dtypes() const override { return dtypes_; }
80 const std::vector<PartialTensorShape>& output_shapes() const override {
81 return shapes_;
82 }
83
84 string DebugString() override { return "TensorSliceDatasetOp::Dataset"; }
85
86 protected:
87 Status AsGraphDefInternal(DatasetGraphDefBuilder* b,
88 Node** output) const override {
89 std::vector<Node*> components;
90 components.reserve(tensors_.size());
91 for (const Tensor& t : tensors_) {
92 Node* node;
93 TF_RETURN_IF_ERROR(b->AddTensor(t, &node));
94 components.emplace_back(node);
95 }
96 AttrValue dtypes;
97 b->BuildAttrValue(dtypes_, &dtypes);
98 TF_RETURN_IF_ERROR(b->AddDataset(this, {}, {{0, components}},
99 {{"Toutput_types", dtypes}}, output));
100 return Status::OK();
101 }
102
103 private:
104 class Iterator : public DatasetIterator<Dataset> {
105 public:
106 explicit Iterator(const Params& params)
107 : DatasetIterator<Dataset>(params),
108 i_(0),
109 n_(params.dataset->tensors_[0].dim_size(0)) {}
110
111 Status GetNextInternal(IteratorContext* ctx,
112 std::vector<Tensor>* out_tensors,
113 bool* end_of_sequence) override {
114 mutex_lock l(mu_);
115 if (i_ < n_) {
116 out_tensors->clear();
117 out_tensors->reserve(dataset()->tensors_.size());
118 for (int i = 0; i < dataset()->tensors_.size(); ++i) {
119 const Tensor& t = dataset()->tensors_[i];
120 Tensor t_slice(ctx->allocator({}), t.dtype(),
121 TensorShape(dataset()->shapes_[i].dim_sizes()));
122 TF_RETURN_IF_ERROR(batch_util::CopySliceToElement(t, &t_slice, i_));
123 out_tensors->emplace_back(std::move(t_slice));
124 }
125 ++i_;
126 *end_of_sequence = false;
127 } else {
128 *end_of_sequence = true;
129 }
130 return Status::OK();
131 }
132
133 protected:
134 Status SaveInternal(IteratorStateWriter* writer) override {
135 mutex_lock l(mu_);
136 TF_RETURN_IF_ERROR(writer->WriteScalar(full_name("i"), i_));
137 return Status::OK();
138 }
139
140 Status RestoreInternal(IteratorContext* ctx,
141 IteratorStateReader* reader) override {
142 mutex_lock l(mu_);
143 TF_RETURN_IF_ERROR(reader->ReadScalar(full_name("i"), &i_));
144 return Status::OK();
145 }
146
147 private:
148 mutex mu_;
149 int64 i_ GUARDED_BY(mu_);
150 const int64 n_;
151 };
152
153 const std::vector<Tensor> tensors_;
154 DataTypeVector dtypes_;
155 std::vector<PartialTensorShape> shapes_;
156 };
157};
158
159REGISTER_KERNEL_BUILDER(Name("TensorSliceDataset").Device(DEVICE_CPU),
160 TensorSliceDatasetOp);
