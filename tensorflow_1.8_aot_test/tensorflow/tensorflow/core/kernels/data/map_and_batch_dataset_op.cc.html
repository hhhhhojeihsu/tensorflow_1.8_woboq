35class MapAndBatchDatasetOp : public UnaryDatasetOpKernel {
36 public:
37 explicit MapAndBatchDatasetOp(OpKernelConstruction* ctx)
38 : UnaryDatasetOpKernel(ctx),
39 graph_def_version_(ctx->graph_def_version()) {
40 OP_REQUIRES_OK(ctx, ctx->GetAttr("f", &func_));
41 OP_REQUIRES_OK(ctx, ctx->GetAttr("output_types", &output_types_));
42 OP_REQUIRES_OK(ctx, ctx->GetAttr("output_shapes", &output_shapes_));
43 }
44
45 protected:
46 void MakeDataset(OpKernelContext* ctx, DatasetBase* input,
47 DatasetBase** output) override {
48 OpInputList inputs;
49 OP_REQUIRES_OK(ctx, ctx->input_list("other_arguments", &inputs));
50 std::vector<Tensor> other_arguments;
51 other_arguments.reserve(inputs.size());
52 for (const Tensor& t : inputs) {
53 other_arguments.push_back(t);
54 }
55
56 int64 batch_size;
57 OP_REQUIRES_OK(ctx, ParseScalarArgument(ctx, "batch_size", &batch_size));
58 OP_REQUIRES(
59 ctx, batch_size > 0,
60 errors::InvalidArgument("batch_size must be greater than zero."));
61
62 int64 num_parallel_batches;
63 OP_REQUIRES_OK(ctx, ParseScalarArgument(ctx, "num_parallel_batches",
64 &num_parallel_batches));
65 OP_REQUIRES(ctx, num_parallel_batches > 0,
66 errors::InvalidArgument(
67 "num_parallel_batches must be greater than zero."));
68
69 bool drop_remainder;
70 OP_REQUIRES_OK(ctx,
71 ParseScalarArgument(ctx, "drop_remainder", &drop_remainder));
72
73 std::unique_ptr<CapturedFunction> captured_func;
74 OP_REQUIRES_OK(ctx, CapturedFunction::Create(
75 func_, std::move(other_arguments), &captured_func));
76
77 *output = new Dataset(input, batch_size, num_parallel_batches,
78 drop_remainder, output_types_, output_shapes_,
79 std::move(captured_func), &ctx->eigen_cpu_device());
80 }
81
82 private:
83 class Dataset : public DatasetBase {
84 public:
85 Dataset(const DatasetBase* input, int64 batch_size,
86 int64 num_parallel_batches, bool drop_remainder,
87 const DataTypeVector& output_types,
88 const std::vector<PartialTensorShape>& output_shapes,
89 std::unique_ptr<CapturedFunction> captured_func,
90 const Eigen::ThreadPoolDevice* device)
91 : input_(input),
92 batch_size_(batch_size),
93 num_parallel_batches_(num_parallel_batches),
94 drop_remainder_(drop_remainder),
95 output_types_(output_types),
96 output_shapes_(output_shapes),
97 captured_func_(std::move(captured_func)),
98 device_(device) {
99 input_->Ref();
100 }
101
102 ~Dataset() override { input_->Unref(); }
103
104 std::unique_ptr<IteratorBase> MakeIterator(
105 const string& prefix) const override {
106 return std::unique_ptr<IteratorBase>(
107 new Iterator({this, strings::StrCat(prefix, "::MapAndBatch")}));
108 }
109
110 const DataTypeVector& output_dtypes() const override {
111 return output_types_;
112 }
113
114 const std::vector<PartialTensorShape>& output_shapes() const override {
115 return output_shapes_;
116 }
117
118 string DebugString() override { return "MapAndBatchDatasetOp::Dataset"; }
119
120 private:
121 class Iterator : public DatasetIterator<Dataset> {
122 public:
123 explicit Iterator(const Params& params)
124 : DatasetIterator<Dataset>(params),
125 input_impl_(params.dataset->input_->MakeIterator(params.prefix)),
126 invocation_results_(params.dataset->batch_size_ *
127 params.dataset->num_parallel_batches_),
128 batch_results_(params.dataset->num_parallel_batches_) {}
129
130 ~Iterator() override {
131 // TODO(mrry): Replace this cancellation logic with a
132 // CancellationManager. The syntax would be more heavyweight,
133 // but it would be possible to thread a cancellation manager
134 // through the IteratorContext to upstream,
135 // potentially-blocking iterators, when we add these.
136 mutex_lock l(mu_);
137 if (current_batch_index_ != -1) {
138 for (size_t batch_index = 0;
139 batch_index < dataset()->num_parallel_batches_; ++batch_index) {
140 int64 num_elements;
141 WaitForBatch(batch_index, &num_elements).IgnoreError();
142 // Deallocate tensors allocated for the output.
143 batch_results_[batch_index].output.clear();
144 }
145 }
146 }
147
148 // TODO(jsimsa): Implement and profile the following alternative design:
149 //
150 // 0. Set the number of in-flight batches and invocations independently
151 // (though obviously the max number of in-flight invocations must be <
152 // batch_size * num_parallel_batches). Maintain a current producing batch
153 // index and offset.
154 // 1. Issue invocations in order of batch and offset, as you do currently.
155 // 2. When an invocation finishes, increment the current producing batch
156 // and offset. If that invocation would start a new batch and give more
157 // than num_parallel_batches in-flight, block; else start the new
158 // invocation into that location.
159 // 3. When a GetNext() call arrives, block until there's a full batch.
160 // Before returning the batch, if the number of pending invocations is
161 // less than the max, issue that number of invocations.
162 Status GetNextInternal(IteratorContext* ctx,
163 std::vector<Tensor>* out_tensors,
164 bool* end_of_sequence) override {
165 mutex_lock l(mu_);
166
167 // One-time initialization.
168 if (current_batch_index_ == -1) {
169 current_batch_index_ = 0;
170 for (size_t i = 0; i < dataset()->num_parallel_batches_; ++i) {
171 StartInvocationBatch(ctx, i);
172 }
173 }
174
175 int64 num_elements = 0;
176 Status status = WaitForBatch(current_batch_index_, &num_elements);
177 if (num_elements == 0) {
178 *end_of_sequence = true;
179 return Status::OK();
180 }
181 if (!status.ok()) {
182 // Deallocate tensors allocated for the output.
183 batch_results_[current_batch_index_].output.clear();
184 } else {
185 if (num_elements < dataset()->batch_size_) {
186 if (dataset()->drop_remainder_) {
187 // Deallocate tensors allocated for the output.
188 batch_results_[current_batch_index_].output.clear();
189 *end_of_sequence = true;
190 return Status::OK();
191 }
192 const std::vector<Tensor>& output =
193 batch_results_[current_batch_index_].output;
194 for (size_t i = 0; i < output.size(); ++i) {
195 TensorShape component_shape(
196 batch_results_[current_batch_index_].output[i].shape());
197 component_shape.set_dim(0, num_elements);
198 AllocatorAttributes attr;
199 attr.set_gpu_compatible(true);
200 Tensor component(ctx->allocator(attr), output[i].dtype(),
201 component_shape);
202 TF_RETURN_IF_ERROR(
203 CopyPartialBatch(&component, output[i], num_elements));
204 out_tensors->emplace_back(std::move(component));
205 }
206 // Deallocate tensors allocated for the output.
207 batch_results_[current_batch_index_].output.clear();
208 } else {
209 *out_tensors =
210 std::move(batch_results_[current_batch_index_].output);
211 }
212 *end_of_sequence = false;
213 }
214 StartInvocationBatch(ctx, current_batch_index_);
215 current_batch_index_ =
216 (current_batch_index_ + 1) % dataset()->num_parallel_batches_;
217 return status;
218 }
219
220 private:
221 struct BatchResult {
222 mutex mu;
223 bool output_allocated GUARDED_BY(mu);
224 std::vector<Tensor> output;
225 std::unique_ptr<BlockingCounter> counter;
226 };
227
228 struct InvocationResult {
229 Status status;
230 bool end_of_input;
231 std::vector<Tensor> return_values;
232 };
233
234 int64 ComputeInvocationIndex(int64 batch_index, int64 offset) {
235 return batch_index * dataset()->batch_size_ + offset;
236 }
237
238 Status CopyPartialBatch(Tensor* output, const Tensor& value,
239 int64 num_elements) {
240 switch (value.dtype()) {
241#define CASE(type) \
242 case DataTypeToEnum<type>::value: { \
243 auto output_t = output->flat_outer_dims<type>(); \
244 auto value_t = value.flat_outer_dims<type>(); \
245 for (size_t i = 0; i < num_elements; i++) { \
246 output_t.template chip<0>(i) = value_t.template chip<0>(i); \
247 } \
248 return Status::OK(); \
249 }
250 TF_CALL_NUMBER_TYPES(CASE);
251 TF_CALL_string(CASE);
252 TF_CALL_variant(CASE);
253#undef CASE
254 default:
255 return errors::InvalidArgument("Unsupported data type: ",
256 value.dtype());
257 }
258 return Status::OK();
259 }
260
261 void EnsureOutputAllocated(IteratorContext* ctx,
262 BatchResult* batch_result,
263 const std::vector<Tensor>& return_values) {
264 mutex_lock l(batch_result->mu);
265 if (batch_result->output_allocated) {
266 return;
267 }
268 const size_t num_components = return_values.size();
269 for (size_t i = 0; i < num_components; ++i) {
270 TensorShape component_shape({dataset()->batch_size_});
271 component_shape.AppendShape(return_values[i].shape());
272 AllocatorAttributes attr;
273 attr.set_gpu_compatible(true);
274 Tensor component(ctx->allocator(attr), return_values[i].dtype(),
275 component_shape);
276 batch_result->output.emplace_back(std::move(component));
277 }
278 batch_result->output_allocated = true;
279 }
280
281 void InvokeFunctionLocked(IteratorContext* ctx, int64 batch_index,
282 int64 offset) EXCLUSIVE_LOCKS_REQUIRED(mu_) {
283 size_t index = ComputeInvocationIndex(batch_index, offset);
284 InvocationResult* result = &invocation_results_[index];
285 BatchResult* batch_result = &batch_results_[batch_index];
286
287 // Get the next input element.
288 std::vector<Tensor> input_element;
289 result->status =
290 input_impl_->GetNext(ctx, &input_element, &result->end_of_input);
291 if (result->end_of_input || !result->status.ok()) {
292 batch_result->counter->DecrementCount();
293 return;
294 }
295
296 // Call `captured_func_(input_element)`, store the result in
297 // `result->return_values`, and notify `batch_result->counter`
298 // to unblock a consumer.
299 (*ctx->runner())(std::bind(
300 [this, result, batch_result, offset](
301 IteratorContext* ctx, std::vector<Tensor> input_element) {
302 dataset()->captured_func_->RunAsync(
303 ctx, std::move(input_element), &result->return_values,
304 [this, ctx, result, batch_result, offset](Status ret_status) {
305 result->status.Update(ret_status);
306 if (ret_status.ok()) {
307 EnsureOutputAllocated(ctx, batch_result,
308 result->return_values);
309 const size_t num_components =
310 result->return_values.size();
311 for (size_t i = 0; i < num_components; ++i) {
312 const Tensor& tensor = result->return_values[i];
313 Tensor* batch = &(batch_result->output)[i];
314 if (tensor.NumElements() !=
315 (batch->NumElements() / batch->dim_size(0))) {
316 TensorShape batch_shape = batch->shape();
317 batch_shape.RemoveDim(0);
318 result->status.Update(errors::InvalidArgument(
319 "Cannot add tensor to the batch: number of "
320 "elements does not match. Shapes are: [tensor]: ",
321 tensor.shape().DebugString(),
322 ", [batch]: ", batch_shape.DebugString()));
323 break;
324 }
325 // TODO(mrry): Add a version of DoParallelConcat that
326 // allows us to move `tensor` where possible, to speed
327 // up string tensor batching.
328 Status copy_status =
329 ::tensorflow::functor::DoParallelConcat(
330 *dataset()->device_, tensor, offset, batch);
331 if (!copy_status.ok()) {
332 result->status.Update(copy_status);
333 break;
334 }
335 }
336 }
337 delete ctx;
338 // NOTE(mrry): We clear the return values here to release
339 // any memory associated with them and to paralellize the
340 // destruction of the tensors (which can be surprisingly
341 // expensive for map functions with large numbers of return
342 // values).
343 result->return_values.clear();
344 batch_result->counter->DecrementCount();
345 });
346 },
347 new IteratorContext(*ctx), std::move(input_element)));
348 }
349
350 void StartInvocationBatch(IteratorContext* ctx, int64 batch_index)
351 EXCLUSIVE_LOCKS_REQUIRED(mu_) {
352 port::Tracing::TraceMe activity(strings::StrCat(prefix(), "::Start"));
353 // Initialize batch result.
354 {
355 mutex_lock l(batch_results_[batch_index].mu);
356 batch_results_[batch_index].output_allocated = false;
357 batch_results_[batch_index].counter.reset(
358 new BlockingCounter(dataset()->batch_size_));
359 }
360 // Initialize invocation results.
361 for (size_t i = 0; i < dataset()->batch_size_; ++i) {
362 size_t index = ComputeInvocationIndex(batch_index, i);
363 InvocationResult* result = &invocation_results_[index];
364 // Reset the state of `result`; `result->return_values` was cleared
365 // when the previous invocation completed.
366 result->end_of_input = false;
367 result->status = Status::OK();
368 }
369 // Start individual invocations.
370 for (size_t i = 0; i < dataset()->batch_size_; ++i) {
371 InvokeFunctionLocked(ctx, batch_index, i);
372 }
373 }
374
375 Status WaitForBatch(int64 batch_index, int64* num_elements)
376 EXCLUSIVE_LOCKS_REQUIRED(mu_) {
377 port::Tracing::TraceMe activity(strings::StrCat(prefix(), "::Wait"));
378 batch_results_[batch_index].counter->Wait();
379 Status status = Status::OK();
380 for (size_t i = 0; i < dataset()->batch_size_; ++i, ++*num_elements) {
381 size_t index = ComputeInvocationIndex(batch_index, i);
382 InvocationResult* result = &invocation_results_[index];
383 if (result->end_of_input) {
384 VLOG(3) << "end of input encountered at element[" << i << "]: ";
385 return Status::OK();
386 }
387 if (!result->status.ok()) {
388 VLOG(3) << "failed to process element[" << i
389 << "]: " << result->status;
390 status.Update(result->status);
391 }
392 }
393 return status;
394 }
395
396 mutex mu_;
397 int32 current_batch_index_ GUARDED_BY(mu_) = -1;
398 const std::unique_ptr<IteratorBase> input_impl_ GUARDED_BY(mu_);
399 std::vector<InvocationResult> invocation_results_ GUARDED_BY(mu_);
400 std::vector<BatchResult> batch_results_ GUARDED_BY(mu_);
401 };
402
403 const DatasetBase* const input_;
404 const NameAttrList func_;
405 const int64 batch_size_;
406 const int64 num_parallel_batches_;
407 const bool drop_remainder_;
408 const DataTypeVector output_types_;
409 const std::vector<PartialTensorShape> output_shapes_;
410 const std::unique_ptr<CapturedFunction> captured_func_;
411 const Eigen::ThreadPoolDevice* device_; // not owned
412 };
413
414 const int graph_def_version_;
415 DataTypeVector output_types_;
416 std::vector<PartialTensorShape> output_shapes_;
417 NameAttrList func_;
418};
