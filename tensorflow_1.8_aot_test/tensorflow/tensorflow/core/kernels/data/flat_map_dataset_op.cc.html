27// See documentation in ../ops/dataset_ops.cc for a high-level
28// description of the following op.
29
30class FlatMapDatasetOp : public UnaryDatasetOpKernel {
31 public:
32 explicit FlatMapDatasetOp(OpKernelConstruction* ctx)
33 : UnaryDatasetOpKernel(ctx),
34 graph_def_version_(ctx->graph_def_version()) {
35 OP_REQUIRES_OK(ctx, ctx->GetAttr("f", &func_));
36 OP_REQUIRES_OK(ctx, ctx->GetAttr("output_types", &output_types_));
37 OP_REQUIRES_OK(ctx, ctx->GetAttr("output_shapes", &output_shapes_));
38 }
39
40 void MakeDataset(OpKernelContext* ctx, DatasetBase* input,
41 DatasetBase** output) override {
42 OpInputList inputs;
43 OP_REQUIRES_OK(ctx, ctx->input_list("other_arguments", &inputs));
44 std::vector<Tensor> other_arguments;
45 other_arguments.reserve(inputs.size());
46 for (const Tensor& t : inputs) {
47 other_arguments.push_back(t);
48 }
49
50 std::unique_ptr<CapturedFunction> captured_func;
51 OP_REQUIRES_OK(ctx, CapturedFunction::Create(
52 func_, std::move(other_arguments), &captured_func));
53
54 *output = new Dataset(ctx, input, func_, std::move(captured_func),
55 output_types_, output_shapes_);
56 }
57
58 private:
59 class Dataset : public GraphDatasetBase {
60 public:
61 Dataset(OpKernelContext* ctx, const DatasetBase* input,
62 const NameAttrList& func,
63 std::unique_ptr<CapturedFunction> captured_func,
64 const DataTypeVector& output_types,
65 const std::vector<PartialTensorShape>& output_shapes)
66 : GraphDatasetBase(ctx),
67 input_(input),
68 func_(func),
69 captured_func_(std::move(captured_func)),
70 output_types_(output_types),
71 output_shapes_(output_shapes) {
72 input_->Ref();
73 }
74
75 ~Dataset() override { input_->Unref(); }
76
77 std::unique_ptr<IteratorBase> MakeIterator(
78 const string& prefix) const override {
79 return std::unique_ptr<IteratorBase>(
80 new Iterator({this, strings::StrCat(prefix, "::FlatMap")}));
81 }
82
83 const DataTypeVector& output_dtypes() const override {
84 return output_types_;
85 }
86
87 const std::vector<PartialTensorShape>& output_shapes() const override {
88 return output_shapes_;
89 }
90
91 string DebugString() override { return "FlatMapDatasetOp::Dataset"; }
92
93 protected:
94 Status AsGraphDefInternal(OpKernelContext* ctx, DatasetGraphDefBuilder* b,
95 Node** output) const override {
96 TF_RETURN_IF_ERROR(b->AddFunction(ctx, func_.name()));
97 Node* input_graph_node = nullptr;
98 TF_RETURN_IF_ERROR(b->AddParentDataset(ctx, input_, &input_graph_node));
99
100 DataTypeVector other_arguments_types;
101 other_arguments_types.reserve(captured_func_->captured_inputs().size());
102 std::vector<Node*> other_arguments;
103 other_arguments.reserve(captured_func_->captured_inputs().size());
104 for (const Tensor& t : captured_func_->captured_inputs()) {
105 Node* node;
106 TF_RETURN_IF_ERROR(b->AddTensor(t, &node));
107 other_arguments.emplace_back(node);
108 other_arguments_types.emplace_back(t.dtype());
109 }
110 AttrValue f;
111 b->BuildAttrValue(func_, &f);
112 AttrValue other_arguments_types_attr;
113 b->BuildAttrValue(other_arguments_types, &other_arguments_types_attr);
114
115 TF_RETURN_IF_ERROR(b->AddDataset(
116 this, {std::make_pair(0, input_graph_node)}, // Single tensor inputs.
117 {std::make_pair(1, other_arguments)}, // Tensor list inputs.
118 {std::make_pair("f", f),
119 std::make_pair("Targuments", other_arguments_types_attr)}, // Attrs
120 output));
121 return Status::OK();
122 }
123
124 private:
125 class Iterator : public DatasetIterator<Dataset> {
126 public:
127 explicit Iterator(const Params& params)
128 : DatasetIterator<Dataset>(params),
129 input_impl_(params.dataset->input_->MakeIterator(params.prefix)) {}
130
131 Status GetNextInternal(IteratorContext* ctx,
132 std::vector<Tensor>* out_tensors,
133 bool* end_of_sequence) override {
134 mutex_lock l(mu_);
135 do {
136 if (!input_impl_) {
137 *end_of_sequence = true;
138 return Status::OK();
139 }
140 if (current_element_iterator_) {
141 // We are currently precessing a mapped element, so try to get the
142 // next subelement.
143 bool end_of_element;
144 TF_RETURN_IF_ERROR(current_element_iterator_->GetNext(
145 ctx, out_tensors, &end_of_element));
146 if (!end_of_element) {
147 // Produce the subelement as output.
148 *end_of_sequence = false;
149 return Status::OK();
150 }
151
152 // We have reached the end of the current element, so maybe move on
153 // to the next element.
154 current_element_iterator_.reset();
155 }
156
157 // Get the next element from the input dataset.
158 captured_func_inputs_.clear();
159 TF_RETURN_IF_ERROR(input_impl_->GetNext(ctx, &captured_func_inputs_,
160 end_of_sequence));
161 if (*end_of_sequence) {
162 input_impl_.reset();
163 return Status::OK();
164 }
165
166 TF_RETURN_IF_ERROR(BuildCurrentElementIteratorLocked(ctx));
167 } while (true);
168 }
169
170 protected:
171 Status SaveInternal(IteratorStateWriter* writer) override {
172 mutex_lock l(mu_);
173 if (input_impl_) {
174 TF_RETURN_IF_ERROR(SaveParent(writer, input_impl_));
175 TF_RETURN_IF_ERROR(
176 writer->WriteScalar(full_name("element_index"), element_index_));
177 if (current_element_iterator_) {
178 TF_RETURN_IF_ERROR(
179 writer->WriteScalar(full_name("captured_func_inputs_size"),
180 captured_func_inputs_.size()));
181 for (int i = 0; i < captured_func_inputs_.size(); i++) {
182 TF_RETURN_IF_ERROR(writer->WriteTensor(
183 full_name(strings::StrCat("captured_func_inputs[", i, "]")),
184 captured_func_inputs_[i]));
185 }
186 TF_RETURN_IF_ERROR(SaveParent(writer, current_element_iterator_));
187 } else {
188 TF_RETURN_IF_ERROR(writer->WriteScalar(
189 full_name("current_element_iterator_uninitialized"), ""));
190 }
191 } else {
192 TF_RETURN_IF_ERROR(writer->WriteScalar(full_name("exhausted"), ""));
193 }
194 return Status::OK();
195 }
196
197 Status RestoreInternal(IteratorContext* ctx,
198 IteratorStateReader* reader) override {
199 mutex_lock l(mu_);
200 input_impl_.reset();
201 element_index_ = 0;
202 current_element_iterator_.reset();
203 captured_func_inputs_.clear();
204 if (!reader->Contains(full_name("exhausted"))) {
205 input_impl_ = dataset()->input_->MakeIterator(prefix());
206 TF_RETURN_IF_ERROR(RestoreParent(ctx, reader, input_impl_));
207 {
208 int64 temp;
209 TF_RETURN_IF_ERROR(
210 reader->ReadScalar(full_name("element_index"), &temp));
211 element_index_ = temp;
212 }
213 if (!reader->Contains(
214 full_name("current_element_iterator_uninitialized"))) {
215 size_t captured_func_inputs_size;
216 {
217 int64 temp;
218 TF_RETURN_IF_ERROR(reader->ReadScalar(
219 full_name("captured_func_inputs_size"), &temp));
220 captured_func_inputs_size = static_cast<size_t>(temp);
221 }
222 captured_func_inputs_.reserve(captured_func_inputs_size);
223 for (int i = 0; i < captured_func_inputs_size; i++) {
224 captured_func_inputs_.emplace_back();
225 TF_RETURN_IF_ERROR(reader->ReadTensor(
226 full_name(strings::StrCat("captured_func_inputs[", i, "]")),
227 &captured_func_inputs_.back()));
228 }
229 element_index_--;
230 TF_RETURN_IF_ERROR(BuildCurrentElementIteratorLocked(ctx));
231 TF_RETURN_IF_ERROR(
232 RestoreParent(ctx, reader, current_element_iterator_));
233 }
234 }
235 return Status::OK();
236 }
237
238 private:
239 Status BuildCurrentElementIteratorLocked(IteratorContext* ctx)
240 EXCLUSIVE_LOCKS_REQUIRED(mu_) {
241 return dataset::MakeIteratorFromInputElement(
242 ctx, captured_func_inputs_, element_index_++,
243 dataset()->captured_func_.get(), prefix(),
244 &current_element_iterator_);
245 }
246
247 Status BuildCurrentElementIteratorLocked(OpKernelContext* ctx)
248 EXCLUSIVE_LOCKS_REQUIRED(mu_) {
249 IteratorContext::Params params;
250 params.env = ctx->env();
251 params.runner = *(ctx->runner());
252 params.lib = ctx->function_library();
253 IteratorContext iter_ctx(std::move(params));
254 return BuildCurrentElementIteratorLocked(&iter_ctx);
255 }
256
257 mutex mu_;
258 size_t element_index_ GUARDED_BY(mu_) = 0;
259 std::unique_ptr<IteratorBase> input_impl_ GUARDED_BY(mu_);
260 std::unique_ptr<IteratorBase> current_element_iterator_ GUARDED_BY(mu_);
261 std::vector<Tensor> captured_func_inputs_ GUARDED_BY(mu_);
262 };
263
264 const DatasetBase* const input_;
265 const NameAttrList func_;
266 const std::unique_ptr<CapturedFunction> captured_func_;
267 const DataTypeVector output_types_;
268 const std::vector<PartialTensorShape> output_shapes_;
269 };
270
271 const int graph_def_version_;
272 DataTypeVector output_types_;
273 std::vector<PartialTensorShape> output_shapes_;
274 NameAttrList func_;
275};

278 FlatMapDatasetOp);
