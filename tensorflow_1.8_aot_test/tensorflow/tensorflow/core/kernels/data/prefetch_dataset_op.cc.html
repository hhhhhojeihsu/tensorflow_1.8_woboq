27// See documentation in ../ops/dataset_ops.cc for a high-level
28// description of the following op.
29
30class PrefetchDatasetOp : public UnaryDatasetOpKernel {
31 public:
32 explicit PrefetchDatasetOp(OpKernelConstruction* ctx)
33 : UnaryDatasetOpKernel(ctx) {}
34
35 protected:
36 void MakeDataset(OpKernelContext* ctx, DatasetBase* input,
37 DatasetBase** output) override {
38 int64 buffer_size;
39 OP_REQUIRES_OK(
40 ctx, ParseScalarArgument<int64>(ctx, "buffer_size", &buffer_size));
41 OP_REQUIRES(ctx,
42 buffer_size > 0 || buffer_size == PrefetchAutotuner::kAutoTune,
43 errors::InvalidArgument("buffer_size must be > 0"));
44
45 *output = new Dataset(ctx, input, buffer_size);
46 }
47
48 private:
49 class Dataset : public GraphDatasetBase {
50 public:
51 Dataset(OpKernelContext* ctx, const DatasetBase* input, int64 buffer_size)
52 : GraphDatasetBase(ctx), input_(input), buffer_size_(buffer_size) {
53 input_->Ref();
54 }
55
56 ~Dataset() override { input_->Unref(); }
57
58 std::unique_ptr<IteratorBase> MakeIterator(
59 const string& prefix) const override {
60 return std::unique_ptr<IteratorBase>(
61 new Iterator({this, strings::StrCat(prefix, "::Prefetch")}));
62 }
63
64 const DataTypeVector& output_dtypes() const override {
65 return input_->output_dtypes();
66 }
67 const std::vector<PartialTensorShape>& output_shapes() const override {
68 return input_->output_shapes();
69 }
70
71 string DebugString() override { return "PrefetchDatasetOp::Dataset"; }
72
73 protected:
74 Status AsGraphDefInternal(OpKernelContext* ctx, DatasetGraphDefBuilder* b,
75 Node** output) const override {
76 Node* input_graph_node = nullptr;
77 TF_RETURN_IF_ERROR(b->AddParentDataset(ctx, input_, &input_graph_node));
78 Node* buffer_size = nullptr;
79 TF_RETURN_IF_ERROR(b->AddScalar(buffer_size_, &buffer_size));
80 TF_RETURN_IF_ERROR(
81 b->AddDataset(this, {input_graph_node, buffer_size}, output));
82 return Status::OK();
83 }
84
85 private:
86 class Iterator : public DatasetIterator<Dataset> {
87 public:
88 explicit Iterator(const Params& params)
89 : DatasetIterator<Dataset>(params),
90 input_impl_(params.dataset->input_->MakeIterator(params.prefix)),
91 auto_tuner_(params.dataset->buffer_size_) {}
92
93 ~Iterator() override {
94 // Signal the prefetch thread to terminate it. We will then
95 // join that thread when we delete `this->prefetch_thread_`.
96 //
97 // TODO(mrry): Replace this cancellation logic with a
98 // CancellationManager. The syntax would be more heavyweight,
99 // but it would be possible to thread a cancellation manager
100 // through the IteratorContext to upstream,
101 // potentially-blocking iterators, when we add these.
102 {
103 mutex_lock l(mu_);
104 cancelled_ = true;
105 cond_var_.notify_all();
106 }
107 }
108
109 Status GetNextInternal(IteratorContext* ctx,
110 std::vector<Tensor>* out_tensors,
111 bool* end_of_sequence) override {
112 mutex_lock l(mu_);
113 TF_RETURN_IF_ERROR(EnsurePrefetchThreadStarted(ctx));
114
115 while (true) {
116 // Wait until the next element in the buffer has been
117 // produced, or we are shutting down.
118 while (!cancelled_ && !prefetch_thread_finished_ && buffer_.empty()) {
119 auto_tuner_.RecordEmpty();
120 cond_var_.wait(l);
121 }
122
123 if (cancelled_) {
124 return errors::Cancelled(
125 "PrefetchDatasetOp::Dataset::Iterator::GetNext");
126 }
127
128 if (!buffer_.empty()) {
129 // A new element is available. Forward the status from
130 // computing it, and (if we successfully got an element)
131 // the output values.
132 Status s = buffer_.front().status;
133 if (s.ok()) {
134 *out_tensors = std::move(buffer_.front().value);
135 }
136 auto_tuner_.RecordConsumption(buffer_.size());
137 buffer_.pop_front();
138 *end_of_sequence = false;
139
140 // Wake the prefetch thread, in case it has been waiting
141 // for space in the buffer.
142 // Also wake up threads from other calls to GetNext.
143 // TODO(mrry): Consider using different condition variables
144 // for GetNext and Prefetch.
145 cond_var_.notify_all();
146 return s;
147 } else if (prefetch_thread_finished_) {
148 *end_of_sequence = true;
149 return Status::OK();
150 }
151 }
152 }
153
154 protected:
155 Status SaveInternal(IteratorStateWriter* writer) override {
156 // Acquire both locks to ensure that the prefetch thread and
157 // all GetNext threads are blocked.
158 mutex_lock parent_l(parent_mu_);
159 mutex_lock l(mu_);
160 TF_RETURN_IF_ERROR(SaveParent(writer, input_impl_));
161 TF_RETURN_IF_ERROR(
162 writer->WriteScalar(full_name("buffer_size"), buffer_.size()));
163 for (size_t i = 0; i < buffer_.size(); i++) {
164 auto& buffer_element = buffer_[i];
165 TF_RETURN_IF_ERROR(WriteStatus(writer, i, buffer_element.status));
166 if (buffer_element.status.ok()) {
167 TF_RETURN_IF_ERROR(writer->WriteScalar(
168 full_name(strings::StrCat("buffer[", i, "].size")),
169 buffer_element.value.size()));
170 for (size_t j = 0; j < buffer_element.value.size(); j++) {
171 TF_RETURN_IF_ERROR(writer->WriteTensor(
172 full_name(strings::StrCat("buffer[", i, "][", j, "]")),
173 buffer_element.value[j]));
174 }
175 }
176 }
177 return Status::OK();
178 }
179
180 Status RestoreInternal(IteratorContext* ctx,
181 IteratorStateReader* reader) override {
182 mutex_lock parent_l(parent_mu_);
183 mutex_lock l(mu_);
184 buffer_.clear();
185 TF_RETURN_IF_ERROR(RestoreParent(ctx, reader, input_impl_));
186 size_t buffer_size;
187 {
188 int64 temp;
189 TF_RETURN_IF_ERROR(
190 reader->ReadScalar(full_name("buffer_size"), &temp));
191 buffer_size = static_cast<size_t>(temp);
192 }
193 for (size_t i = 0; i < buffer_size; i++) {
194 buffer_.emplace_back();
195 auto& buffer_element = buffer_.back();
196 TF_RETURN_IF_ERROR(ReadStatus(reader, i, &buffer_element.status));
197 if (buffer_element.status.ok()) {
198 size_t value_size;
199 {
200 int64 temp;
201 TF_RETURN_IF_ERROR(reader->ReadScalar(
202 full_name(strings::StrCat("buffer[", i, "].size")), &temp));
203 value_size = static_cast<size_t>(temp);
204 }
205 buffer_element.value.reserve(value_size);
206 for (size_t j = 0; j < value_size; j++) {
207 buffer_element.value.emplace_back();
208 TF_RETURN_IF_ERROR(reader->ReadTensor(
209 full_name(strings::StrCat("buffer[", i, "][", j, "]")),
210 &buffer_element.value.back()));
211 }
212 }
213 }
214 return Status::OK();
215 }
216
217 private:
218 // A buffer element comprises a status and (if that status is
219 // OK) a vector of tensors, representing an element of the input dataset.
220 struct BufferElement {
221 // The producer sets `status` if getting the input element fails.
222 Status status;
223 // The buffered data element.
224 std::vector<Tensor> value;
225 };
226
227 Status EnsurePrefetchThreadStarted(IteratorContext* ctx)
228 EXCLUSIVE_LOCKS_REQUIRED(mu_) {
229 if (!prefetch_thread_) {
230 prefetch_thread_.reset(
231 ctx->env()->StartThread({}, "prefetch_thread",
232 std::bind(&Iterator::PrefetchThread, this,
233 new IteratorContext(*ctx))));
234 }
235 return Status::OK();
236 }
237
238 // Prefetches elements of the input, storing results in an internal
239 // buffer.
240 //
241 // It owns the iterator context passed to it.
242 void PrefetchThread(IteratorContext* ctx) {
243 std::unique_ptr<IteratorContext> cleanup(ctx);
244 while (true) {
245 std::vector<Tensor> value;
246
247 // 1. Wait for a slot in the buffer.
248 {
249 mutex_lock l(mu_);
250 while (!cancelled_ &&
251 buffer_.size() == auto_tuner_.buffer_limit()) {
252 cond_var_.wait(l);
253 }
254
255 if (cancelled_) {
256 return;
257 }
258 }
259
260 // 2. Read the next element.
261 // Acquire the parent lock since we will be reading an element
262 // from the input iterator. Note that we do not wish to release
263 // this lock till we have added the fetched element to the
264 // `buffer_` else there will be local state that may be missed
265 // by SaveInternal.
266 mutex_lock parent_l(parent_mu_);
267 bool end_of_sequence;
268 BufferElement buffer_element;
269 buffer_element.status = input_impl_->GetNext(
270 ctx, &buffer_element.value, &end_of_sequence);
271 if (buffer_element.status.ok() && end_of_sequence) {
272 mutex_lock l(mu_);
273 prefetch_thread_finished_ = true;
274 cond_var_.notify_all();
275 return;
276 }
277
278 // 3. Signal that the element has been produced.
279 {
280 mutex_lock l(mu_);
281 buffer_.push_back(std::move(buffer_element));
282 cond_var_.notify_all();
283 }
284 }
285 }
286
287 Status WriteStatus(IteratorStateWriter* writer, size_t index,
288 const Status& status) EXCLUSIVE_LOCKS_REQUIRED(mu_) {
289 TF_RETURN_IF_ERROR(writer->WriteScalar(
290 CodeKey(index), static_cast<int64>(status.code())));
291 if (!status.ok()) {
292 TF_RETURN_IF_ERROR(writer->WriteScalar(ErrorMessageKey(index),
293 status.error_message()));
294 }
295 return Status::OK();
296 }
297
298 Status ReadStatus(IteratorStateReader* reader, size_t index,
299 Status* status) EXCLUSIVE_LOCKS_REQUIRED(mu_) {
300 int64 code_int;
301 TF_RETURN_IF_ERROR(reader->ReadScalar(CodeKey(index), &code_int));
302 error::Code code = static_cast<error::Code>(code_int);
303
304 if (code != error::Code::OK) {
305 string error_message;
306 TF_RETURN_IF_ERROR(
307 reader->ReadScalar(ErrorMessageKey(index), &error_message));
308 *status = Status(code, error_message);
309 } else {
310 *status = Status::OK();
311 }
312 return Status::OK();
313 }
314
315 string CodeKey(size_t index) {
316 return full_name(strings::StrCat("status[", index, "].code"));
317 }
318
319 string ErrorMessageKey(size_t index) {
320 return full_name(strings::StrCat("status[", index, "].error_message"));
321 }
322
323 // This mutex is used to ensure exclusivity between multiple threads
324 // reading/writing this iterator's local state.
325 mutex mu_;
326 // This mutex is used to ensure exclusivity between multiple threads
327 // accessing the parent iterator. We keep this separate from `mu_` to
328 // allow prefetching to run in parallel with GetNext calls.
329 mutex parent_mu_ ACQUIRED_BEFORE(mu_);
330 const std::unique_ptr<IteratorBase> input_impl_ GUARDED_BY(parent_mu_);
331 condition_variable cond_var_;
332 PrefetchAutotuner auto_tuner_ GUARDED_BY(mu_);
333 std::deque<BufferElement> buffer_ GUARDED_BY(mu_);
334 std::unique_ptr<Thread> prefetch_thread_ GUARDED_BY(mu_);
335 bool cancelled_ GUARDED_BY(mu_) = false;
336 bool prefetch_thread_finished_ GUARDED_BY(mu_) = false;
337 };
338
339 const DatasetBase* const input_;
340 const int64 buffer_size_;
341 };
342};
