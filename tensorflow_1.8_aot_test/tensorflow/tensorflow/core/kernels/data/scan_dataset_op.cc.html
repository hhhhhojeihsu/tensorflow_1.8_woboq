29// See documentation in ../ops/dataset_ops.cc for a high-level
30// description of the following op.
31
32class ScanDatasetOp : public UnaryDatasetOpKernel {
33 public:
34 explicit ScanDatasetOp(OpKernelConstruction* ctx)
35 : UnaryDatasetOpKernel(ctx),
36 graph_def_version_(ctx->graph_def_version()) {
37 OP_REQUIRES_OK(ctx, ctx->GetAttr("f", &func_));
38 OP_REQUIRES_OK(ctx, ctx->GetAttr("Tstate", &state_types_));
39 OP_REQUIRES_OK(ctx, ctx->GetAttr("output_types", &output_types_));
40 OP_REQUIRES_OK(ctx, ctx->GetAttr("output_shapes", &output_shapes_));
41 }
42
43 void MakeDataset(OpKernelContext* ctx, DatasetBase* input,
44 DatasetBase** output) override {
45 OpInputList initial_state_inputs;
46 OP_REQUIRES_OK(ctx,
47 ctx->input_list("initial_state", &initial_state_inputs));
48 std::vector<Tensor> initial_state;
49 initial_state.reserve(initial_state_inputs.size());
50 for (const Tensor& t : initial_state_inputs) {
51 initial_state.push_back(t);
52 }
53
54 OpInputList inputs;
55 OP_REQUIRES_OK(ctx, ctx->input_list("other_arguments", &inputs));
56 std::vector<Tensor> other_arguments;
57 other_arguments.reserve(inputs.size());
58 for (const Tensor& t : inputs) {
59 other_arguments.push_back(t);
60 }
61
62 std::unique_ptr<CapturedFunction> captured_func;
63 OP_REQUIRES_OK(ctx, CapturedFunction::Create(
64 func_, std::move(other_arguments), &captured_func));
65
66 *output = new Dataset(ctx, input, func_, std::move(initial_state),
67 std::move(captured_func), state_types_, output_types_,
68 output_shapes_);
69 }
70
71 private:
72 class Dataset : public GraphDatasetBase {
73 public:
74 Dataset(OpKernelContext* ctx, const DatasetBase* input,
75 const NameAttrList& func, std::vector<Tensor> initial_state,
76 std::unique_ptr<CapturedFunction> captured_func,
77 const DataTypeVector& state_types,
78 const DataTypeVector& output_types,
79 const std::vector<PartialTensorShape>& output_shapes)
80 : GraphDatasetBase(ctx),
81 input_(input),
82 func_(func),
83 initial_state_(std::move(initial_state)),
84 captured_func_(std::move(captured_func)),
85 state_types_(state_types),
86 output_types_(output_types),
87 output_shapes_(output_shapes) {
88 input_->Ref();
89 }
90
91 ~Dataset() override { input_->Unref(); }
92
93 std::unique_ptr<IteratorBase> MakeIterator(
94 const string& prefix) const override {
95 return std::unique_ptr<IteratorBase>(
96 new Iterator({this, strings::StrCat(prefix, "::Scan")}));
97 }
98
99 const DataTypeVector& output_dtypes() const override {
100 return output_types_;
101 }
102 const std::vector<PartialTensorShape>& output_shapes() const override {
103 return output_shapes_;
104 }
105
106 string DebugString() override { return "ScanDatasetOp::Dataset"; }
107
108 protected:
109 Status AsGraphDefInternal(OpKernelContext* ctx, DatasetGraphDefBuilder* b,
110 Node** output) const override {
111 TF_RETURN_IF_ERROR(b->AddFunction(ctx, func_.name()));
112 Node* input_node;
113 TF_RETURN_IF_ERROR(b->AddParentDataset(ctx, input_, &input_node));
114 std::vector<Node*> initial_state_nodes;
115 initial_state_nodes.reserve(initial_state_.size());
116 for (const Tensor& t : initial_state_) {
117 Node* node;
118 TF_RETURN_IF_ERROR(b->AddTensor(t, &node));
119 initial_state_nodes.emplace_back(node);
120 }
121 std::vector<Node*> other_arguments;
122 other_arguments.reserve(captured_func_->captured_inputs().size());
123 DataTypeVector other_arguments_types;
124 other_arguments_types.reserve(captured_func_->captured_inputs().size());
125 for (const Tensor& t : captured_func_->captured_inputs()) {
126 Node* node;
127 TF_RETURN_IF_ERROR(b->AddTensor(t, &node));
128 other_arguments.emplace_back(node);
129 other_arguments_types.emplace_back(t.dtype());
130 }
131 AttrValue f;
132 b->BuildAttrValue(func_, &f);
133 AttrValue state_types;
134 b->BuildAttrValue(state_types_, &state_types);
135 AttrValue other_arguments_types_attr;
136 b->BuildAttrValue(other_arguments_types, &other_arguments_types_attr);
137 TF_RETURN_IF_ERROR(
138 b->AddDataset(this, {{0, input_node}},
139 {{1, initial_state_nodes}, {2, other_arguments}},
140 {{"f", f},
141 {"Tstate", state_types},
142 {"Targuments", other_arguments_types_attr}},
143 output));
144 return Status::OK();
145 }
146
147 private:
148 class Iterator : public DatasetIterator<Dataset> {
149 public:
150 explicit Iterator(const Params& params)
151 : DatasetIterator<Dataset>(params),
152 input_impl_(params.dataset->input_->MakeIterator(params.prefix)),
153 state_(params.dataset->initial_state_) {}
154
155 Status GetNextInternal(IteratorContext* ctx,
156 std::vector<Tensor>* out_tensors,
157 bool* end_of_sequence) override {
158 mutex_lock l(mu_);
159
160 std::vector<Tensor> next_element;
161 TF_RETURN_IF_ERROR(
162 input_impl_->GetNext(ctx, &next_element, end_of_sequence));
163 if (*end_of_sequence) {
164 return Status::OK();
165 }
166
167 std::vector<Tensor> args;
168 args.reserve(state_.size() + next_element.size());
169 std::copy(state_.begin(), state_.end(), std::back_inserter(args));
170 std::copy(next_element.begin(), next_element.end(),
171 std::back_inserter(args));
172
173 std::vector<Tensor> state_and_output;
174 state_and_output.reserve(dataset()->state_types_.size() +
175 output_dtypes().size());
176
177 Status s = dataset()->captured_func_->Run(ctx, std::move(args),
178 &state_and_output);
179 if (s.ok()) {
180 state_.clear();
181 size_t i = 0;
182 for (; i < dataset()->state_types_.size(); ++i) {
183 if (state_and_output[i].dtype() != dataset()->state_types_[i]) {
184 return errors::InvalidArgument(
185 "Got wrong type for scan_func return value ", i,
186 " (expected ", DataTypeString(dataset()->state_types_[i]),
187 ", got ", DataTypeString(state_and_output[i].dtype()), ").");
188 }
189 state_.push_back(std::move(state_and_output[i]));
190 }
191 for (; i < state_and_output.size(); ++i) {
192 const size_t output_index = i - dataset()->state_types_.size();
193 if (state_and_output[i].dtype() != output_dtypes()[output_index]) {
194 return errors::InvalidArgument(
195 "Got wrong type for scan_func return value ", i,
196 " (expected ",
197 DataTypeString(dataset()->state_types_[output_index]),
198 ", got ", DataTypeString(state_and_output[i].dtype()), ").");
199 }
200 if (!output_shapes()[output_index].IsCompatibleWith(
201 state_and_output[i].shape())) {
202 return errors::InvalidArgument(
203 "Got wrong shape for scan_func return value ", i,
204 " (expected ", output_shapes()[output_index].DebugString(),
205 ", got ", state_and_output[i].shape().DebugString(), ").");
206 }
207
208 out_tensors->push_back(std::move(state_and_output[i]));
209 }
210 } else if (errors::IsOutOfRange(s)) {
211 // `f` may deliberately raise `errors::OutOfRange` to indicate
212 // that we should terminate the iteration early.
213 *end_of_sequence = true;
214 return Status::OK();
215 }
216 return s;
217 }
218
219 protected:
220 Status SaveInternal(IteratorStateWriter* writer) override {
221 mutex_lock l(mu_);
222 TF_RETURN_IF_ERROR(SaveParent(writer, input_impl_));
223 if (!state_.empty()) {
224 TF_RETURN_IF_ERROR(
225 writer->WriteScalar(full_name("state_size"), state_.size()));
226 for (int idx = 0; idx < state_.size(); idx++) {
227 TF_RETURN_IF_ERROR(writer->WriteTensor(
228 full_name(strings::StrCat("state[", idx, "]")), state_[idx]));
229 }
230 }
231 return Status::OK();
232 }
233
234 Status RestoreInternal(IteratorContext* ctx,
235 IteratorStateReader* reader) override {
236 mutex_lock l(mu_);
237 TF_RETURN_IF_ERROR(RestoreParent(ctx, reader, input_impl_));
238 if (reader->Contains(full_name("state_size"))) {
239 int64 size;
240 TF_RETURN_IF_ERROR(
241 reader->ReadScalar(full_name("state_size"), &size));
242 state_.resize(size);
243 for (int idx = 0; idx < size; idx++) {
244 TF_RETURN_IF_ERROR(reader->ReadTensor(
245 full_name(strings::StrCat("state[", idx, "]")), &state_[idx]));
246 }
247 }
248 return Status::OK();
249 }
250
251 private:
252 mutex mu_;
253 const std::unique_ptr<IteratorBase> input_impl_ GUARDED_BY(mu_);
254 std::vector<Tensor> state_ GUARDED_BY(mu_);
255 };
256
257 const DatasetBase* const input_;
258 const NameAttrList func_;
259 const std::vector<Tensor> initial_state_;
260 const std::unique_ptr<CapturedFunction> captured_func_;
261 const DataTypeVector state_types_;
262 const DataTypeVector output_types_;
263 const std::vector<PartialTensorShape> output_shapes_;
264 };
265
266 const int graph_def_version_;
267 DataTypeVector state_types_;
268 DataTypeVector output_types_;
269 std::vector<PartialTensorShape> output_shapes_;
270 NameAttrList func_;
271};
