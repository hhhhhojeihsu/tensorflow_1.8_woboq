36class Mutex : public ResourceBase {
37 public:
38 explicit Mutex(OpKernelContext* c, const string& name)
39 : locked_(false),
40 thread_pool_(new thread::ThreadPool(
41 c->env(), ThreadOptions(),
42 strings::StrCat("mutex_lock_thread_", SanitizeThreadSuffix(name)),
43 1 /* num_threads */, false /* low_latency_hint */)),
44 name_(name) {
45 VLOG(2) << "Creating mutex with name " << name << ": " << this;
46 }
47
48 string DebugString() override { return strings::StrCat("Mutex ", name_); }
49
50 class LockReleaser {
51 public:
52 explicit LockReleaser(Mutex* mutex) : mutex_(mutex) {}
53
54 LockReleaser(const LockReleaser&) = delete;
55 LockReleaser& operator=(const LockReleaser&) = delete;
56
57 virtual ~LockReleaser() {
58 VLOG(3) << "Destroying LockReleaser " << this << " for mutex: " << mutex_;
59 if (mutex_) {
60 mutex_lock lock(mutex_->mu_);
61 mutex_->locked_ = false;
62 mutex_->cv_.notify_all();
63 VLOG(3) << "Destroying LockReleaser " << this
64 << ": sent notifications.";
65 }
66 }
67
68 private:
69 Mutex* mutex_;
70 };
71
72 struct SharedLockReleaser {
73 std::shared_ptr<LockReleaser> shared_lock;
74
75 explicit SharedLockReleaser(std::shared_ptr<LockReleaser>&& lock)
76 : shared_lock(std::forward<decltype(lock)>(lock)) {
77 VLOG(3) << "Creating shared_ptr of " << shared_lock.get()
78 << " count is: " << shared_lock.use_count();
79 }
80
81 SharedLockReleaser(SharedLockReleaser&& rhs)
82 : shared_lock(std::move(rhs.shared_lock)) {
83 VLOG(3) << "Moving SharedLockReleaser of " << shared_lock.get()
84 << " count is: " << shared_lock.use_count();
85 }
86
87 SharedLockReleaser(const SharedLockReleaser& rhs)
88 : shared_lock(rhs.shared_lock) {
89 VLOG(3) << "Copying SharedLockReleaser of " << shared_lock.get()
90 << " count is: " << shared_lock.use_count();
91 }
92
93 ~SharedLockReleaser() {
94 VLOG(3) << "Destroying SharedLockReleaser of " << shared_lock.get()
95 << " count is: " << shared_lock.use_count();
96 }

107 void AcquireAsync(
108 OpKernelContext* c,
109 std::function<void(const Status& s, SharedLockReleaser lock)> fn) {
110 CancellationManager* cm = c->cancellation_manager();
111 CancellationToken token{};
112 bool* cancelled = nullptr;
113 if (cm) {
114 cancelled = new bool(false); // GUARDED_BY(mu_);
115 token = cm->get_cancellation_token();
116 const bool already_cancelled =
117 !cm->RegisterCallback(token, [this, cancelled]() {
118 mutex_lock lock(mu_);
119 *cancelled = true;
120 cv_.notify_all();
121 });
122 if (already_cancelled) {
123 delete cancelled;
124 fn(errors::Cancelled("Lock acquisition cancelled."),
125 SharedLockReleaser{nullptr});
126 return;
127 }
128 }
129 thread_pool_->Schedule(std::bind(
130 [this, cm, cancelled,
131 token](std::function<void(const Status& s, SharedLockReleaser&& lock)>
132 fn_) {
133 bool local_locked;
134 {
135 mutex_lock lock(mu_);
136 while (locked_ && !(cancelled && *cancelled)) {
137 cv_.wait(lock);
138 }
139 local_locked = locked_ = !(cancelled && *cancelled);
140 }
141 if (cm) {
142 cm->DeregisterCallback(token);
143 delete cancelled;
144 }
145 if (local_locked) { // Not cancelled.
146 fn_(Status::OK(),
147 SharedLockReleaser{std::make_shared<LockReleaser>(this)});
148 } else {
149 fn_(errors::Cancelled("Lock acqusition cancelled."),
150 SharedLockReleaser{nullptr});
151 }
152 },
153 std::move(fn)));
154 }
155
156 private:
157 mutex mu_;
158 condition_variable cv_ GUARDED_BY(mu_);
159 bool locked_ GUARDED_BY(mu_);
160 std::unique_ptr<thread::ThreadPool> thread_pool_;
161 string name_;
162};

166class MutexLockOp : public AsyncOpKernel {
167 public:
168 explicit MutexLockOp(OpKernelConstruction* c) : AsyncOpKernel(c) {}
169
170 public:
171 void ComputeAsync(OpKernelContext* c, DoneCallback done) override {
172 Mutex* mutex = nullptr;
173 OP_REQUIRES_OK_ASYNC(
174 c,
175 LookupOrCreateResource<Mutex>(c, HandleFromInput(c, 0), &mutex,
176 [c](Mutex** ptr) {
177 *ptr = new Mutex(
178 c, HandleFromInput(c, 0).name());
179 return Status::OK();
180 }),
181 done);
182
183 Tensor* variant;
184 OP_REQUIRES_OK_ASYNC(c, c->allocate_output(0, TensorShape({}), &variant),
185 done);
186
187 mutex->AcquireAsync(
188 c, std::bind(
189 [c, variant, mutex](DoneCallback done_,
190 // End of bound arguments.
191 const Status& s,
192 Mutex::SharedLockReleaser&& lock) {
193 VLOG(2) << "Finished locking mutex " << mutex
194 << " with lock: " << lock.shared_lock.get()
195 << " status: " << s.ToString();
196 if (s.ok()) {
197 variant->scalar<Variant>()() = std::move(lock);
198 } else {
199 c->SetStatus(s);
200 }
201 mutex->Unref();
202 done_();
203 },
204 std::move(done), std::placeholders::_1, std::placeholders::_2));
205 }
206};

208class ConsumeMutexLockOp : public OpKernel {
209 public:
210 explicit ConsumeMutexLockOp(OpKernelConstruction* context)
211 : OpKernel(context) {}
212
213 void Compute(OpKernelContext* c) override {
214 VLOG(2) << "Executing ConsumeMutexLockOp";
215 const Tensor& lock_t = c->input(0);
216 OP_REQUIRES(
217 c, lock_t.dims() == 0,
218 errors::InvalidArgument("Expected input to be a scalar, saw shape: ",
219 lock_t.shape().DebugString()));
220 OP_REQUIRES(
221 c, lock_t.dtype() == DT_VARIANT,
222 errors::InvalidArgument("Expected input to be a variant, saw type: ",
223 DataTypeString(lock_t.dtype())));
224 const auto* lock =
225 lock_t.scalar<Variant>()().get<Mutex::SharedLockReleaser>();
226 OP_REQUIRES(c, lock,
227 errors::InvalidArgument(
228 "Expected input to contain a SharedLockReleaser "
229 "object, but saw variant: '",
230 lock_t.scalar<Variant>()().DebugString(), "'"));
231 const int use_count = lock->shared_lock.use_count();
232 OP_REQUIRES(
233 c, use_count == 1,
234 errors::InvalidArgument("Expected use count of lock to be 1, but saw: ",
235 use_count));
236 }
237
238 bool IsExpensive() override { return false; }
239};
