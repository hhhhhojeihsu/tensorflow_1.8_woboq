36// Helper to instantiate function "func" in the library "lib".
37Status Instantiate(FunctionLibraryRuntime* lib, const NameAttrList& func,
38 FunctionLibraryRuntime::Handle* handle) {
39 return lib->Instantiate(func.name(), AttrSlice(&func.attr()), handle);

42// If "t" is a scalar of a supported type, returns t != 0 in "*v".
43Status ToBool(gtl::ArraySlice<Tensor> t, bool* v) {
44 if (t.size() != 1) {
45 return errors::InvalidArgument(
46 "Expected a single scalar which can be converted to a boolean, got ",
47 t.size(), " tensors.");
48 }
49 if (TensorShapeUtils::IsScalar(t[0].shape())) {
50 switch (t[0].dtype()) {
51#define CASE(T) \
52 case DataTypeToEnum<T>::value: \
53 *v = t[0].scalar<T>()() != 0; \
54 break;
55
56 CASE(float);
57 CASE(double);
58 CASE(int32);
59 CASE(uint8);
60 CASE(int16);
61 CASE(int8);
62 CASE(int64);
63#undef CASE
64 case DT_BOOL:
65 *v = t[0].scalar<bool>()();
66 break;
67 case DT_STRING:
68 *v = !t[0].scalar<string>()().empty();
69 break;
70 default:
71 return errors::InvalidArgument(DataTypeString(t[0].dtype()),
72 " cannot be converted to a boolean");
73 }
74 } else {
75 *v = t[0].NumElements() > 0;
76 }
77 return Status::OK();
78}

80// Sets "rets" to be the output of "ctx". Validates rets' types based
81// on "kernel".
82Status SetOutputs(const OpKernel* kernel, OpKernelContext* ctx,
83 gtl::ArraySlice<Tensor> rets) {
84 if (rets.size() != ctx->num_outputs()) {
85 return errors::Internal("Expect to produce ", ctx->num_outputs(),
86 " tensors, but only get ", rets.size());
87 }
88 for (int i = 0; i < rets.size(); ++i) {
89 if (rets[i].dtype() != kernel->output_type(i)) {
90 return errors::Internal("Expect ", i, "-th output is of type ",
91 DataTypeString(kernel->output_type(i)),
92 " but get ", DataTypeString(rets[i].dtype()));
93 }
94 ctx->set_output(i, rets[i]);
95 }
96 return Status::OK();
97}

99void SetRunOptions(OpKernelContext* ctx, FunctionLibraryRuntime::Options* opts,
100 bool always_collect_stats) {
101 opts->step_id = ctx->step_id();
102 opts->rendezvous = ctx->rendezvous();
103 opts->cancellation_manager = ctx->cancellation_manager();
104 if (always_collect_stats) {
105 opts->stats_collector = ctx->stats_collector();
106 }
107 opts->runner = ctx->runner();
108}

110class IfOp : public AsyncOpKernel {
111 public:
112 explicit IfOp(OpKernelConstruction* ctx) : AsyncOpKernel(ctx) {
113 auto lib = ctx->function_library();
114 OP_REQUIRES(ctx, lib != nullptr, errors::Internal("No function library"));
115 const NameAttrList* func;
116 OP_REQUIRES_OK(ctx, ctx->GetAttr("then_branch", &func));
117 OP_REQUIRES_OK(ctx, Instantiate(lib, *func, &then_handle_));
118 OP_REQUIRES_OK(ctx, ctx->GetAttr("else_branch", &func));
119 OP_REQUIRES_OK(ctx, Instantiate(lib, *func, &else_handle_));
120 }
121
122 ~IfOp() override {}
123
124 void ComputeAsync(OpKernelContext* ctx, DoneCallback done) override {
125 bool cond;
126 OP_REQUIRES_OK(ctx, ToBool({ctx->input(0)}, &cond));
127 (new State(this, ctx, cond, done))->Start();
128 }
129
130 private:
131 FHandle then_handle_;
132 FHandle else_handle_;
133
134 class State {
135 public:
136 State(IfOp* kernel, OpKernelContext* ctx, bool cond, DoneCallback done)
137 : kernel_(kernel),
138 ctx_(ctx),
139 cond_(cond),
140 done_(done),
141 lib_(CHECK_NOTNULL(ctx_->function_library())) {
142 SetRunOptions(ctx_, &opts_, true /* always_collect_stats */);
143 for (int i = 1; i < ctx_->num_inputs(); ++i) {
144 args_.push_back(ctx_->input(i));
145 }
146 }
147
148 ~State() {}
149
150 void Start() {
151 FHandle handle = cond_ ? kernel_->then_handle_ : kernel_->else_handle_;
152 rets_.clear();
153 lib_->Run(
154 // Evaluate one of the branch.
155 opts_, handle, args_, &rets_,
156 // Done callback
157 [this](Status s) {
158 if (s.ok()) {
159 s = SetOutputs(kernel_, ctx_, rets_);
160 }
161 ctx_->SetStatus(s);
162 auto done = done_;
163 delete this;
164 done();
165 });
166 }
167
168 private:
169 IfOp* const kernel_;
170 OpKernelContext* const ctx_;
171 const bool cond_;
172 const DoneCallback done_;
173 FunctionLibraryRuntime* const lib_;
174 FunctionLibraryRuntime::Options opts_;
175 TensorVec args_;
176 TensorVec rets_;
177 };
178};

188class WhileOp : public AsyncOpKernel {
189 public:
190 explicit WhileOp(OpKernelConstruction* ctx) : AsyncOpKernel(ctx) {
191 OP_REQUIRES_OK(ctx, ctx->GetAttr("cond", &cond_func_));
192 OP_REQUIRES_OK(ctx, ctx->GetAttr("body", &body_func_));
193 }
194
195 ~WhileOp() override {}
196
197 void ComputeAsync(OpKernelContext* ctx, DoneCallback done) override {
198 auto lib = ctx->function_library();
199 OP_REQUIRES_ASYNC(ctx, lib != nullptr,
200 errors::Internal("No function library"), done);
201
202 // TODO(b/37549631): Because this op has `SetIsStateful()` in its
203 // op registration, this kernel may be shared by multiple
204 // subgraphs, which have different associated
205 // `FunctionLibraryRuntime` objects and hence different `FHandle`
206 // namespaces. We currently work around this by caching the map
207 // from `FunctionLibraryRuntime*` to `FHandle` pairs for the two
208 // functions this op uses.
209 FHandle cond_handle;
210 FHandle body_handle;
211 {
212 mutex_lock l(mu_);
213 const auto iter = handles_.find(lib);
214 if (iter == handles_.end()) {
215 OP_REQUIRES_OK_ASYNC(ctx, Instantiate(lib, cond_func_, &cond_handle),
216 done);
217 OP_REQUIRES_OK_ASYNC(ctx, Instantiate(lib, body_func_, &body_handle),
218 done);
219 handles_[lib] = {cond_handle, body_handle};
220 } else {
221 cond_handle = iter->second.first;
222 body_handle = iter->second.second;
223 }
224 }
225
226 (new State(this, ctx, cond_handle, body_handle, done))->Start();
227 }
228
229 private:
230 NameAttrList cond_func_;
231 NameAttrList body_func_;
232
233 mutex mu_;
234 std::unordered_map<FunctionLibraryRuntime*, std::pair<FHandle, FHandle>>
235 handles_ GUARDED_BY(mu_);
236
237 class State {
238 public:
239 State(WhileOp* kernel, OpKernelContext* ctx, FHandle cond_handle,
240 FHandle body_handle, DoneCallback done)
241 : kernel_(kernel),
242 ctx_(ctx),
243 cond_handle_(cond_handle),
244 body_handle_(body_handle),
245 done_(done),
246 lib_(CHECK_NOTNULL(ctx_->function_library())) {
247 SetRunOptions(ctx_, &opts_, false /* always_collect_stats */);
248 for (int i = 0; i < ctx_->num_inputs(); ++i) {
249 args_.push_back(ctx_->input(i));
250 }
251 }
252
253 ~State() {}
254
255 void Start() { EvalCond(); }
256
257 private:
258 WhileOp* const kernel_;
259 OpKernelContext* const ctx_;
260 const FHandle cond_handle_;
261 const FHandle body_handle_;
262 const DoneCallback done_;
263 FunctionLibraryRuntime* const lib_;
264 FunctionLibraryRuntime::Options opts_;
265 TensorVec args_;
266 TensorVec rets_;
267
268 void EvalCond() {
269 lib_->Run(
270 // Evaluate the condition.
271 opts_, cond_handle_, args_, &rets_,
272 // Done cb.
273 [this](const Status& s) {
274 if (!s.ok()) {
275 return Finish(s);
276 }
277 StartBody();
278 });
279 }
280
281 void StartBody() {
282 bool cond;
283 Status s = ToBool(rets_, &cond);
284 if (!s.ok()) {
285 return Finish(s);
286 }
287 if (!cond) {
288 return Finish(Status::OK());
289 }
290 rets_.clear();
291 lib_->Run(
292 // Evaluate the body.
293 opts_, body_handle_, args_, &rets_,
294 // Done callback
295 [this](const Status& s) {
296 if (!s.ok()) {
297 return Finish(s);
298 }
299 if (args_.size() != rets_.size()) {
300 return Finish(errors::InvalidArgument(
301 "While loop body returned ", rets_.size(),
302 " arguments. Expected: ", args_.size()));
303 }
304 args_.clear();
305 using std::swap;
306 swap(args_, rets_);
307 EvalCond();
308 });
309 }
310
311 void Finish(Status s) {
312 if (s.ok()) {
313 s = SetOutputs(kernel_, ctx_, args_);
314 }
315 ctx_->SetStatus(s);
316 done_();
317 delete this;
318 }
319 };
320};

328Status GetScalar(OpKernelContext* ctx, int index, int32* value,
329 const char* label) {
330 Tensor t = ctx->input(index);
331 if (!TensorShapeUtils::IsScalar(t.shape())) {
332 return errors::InvalidArgument(label, " must be a scalar, but ",
333 t.shape().DebugString());
334 }
335 *value = t.scalar<int32>()();
336 return Status::OK();
337}

339class ForOp : public AsyncOpKernel {
340 public:
341 explicit ForOp(OpKernelConstruction* ctx) : AsyncOpKernel(ctx) {
342 auto lib = ctx->function_library();
343 OP_REQUIRES(ctx, lib != nullptr, errors::Internal("No function library"));
344 const NameAttrList* func;
345 OP_REQUIRES_OK(ctx, ctx->GetAttr("body", &func));
346 OP_REQUIRES_OK(ctx, Instantiate(lib, *func, &body_handle_));
347 }
348
349 ~ForOp() override {}
350
351 void ComputeAsync(OpKernelContext* ctx, DoneCallback done) override {
352 (new State(this, ctx, done))->Start();
353 }
354
355 private:
356 FHandle body_handle_;
357
358 class State {
359 public:
360 State(ForOp* kernel, OpKernelContext* ctx, DoneCallback done)
361 : kernel_(kernel),
362 ctx_(ctx),
363 done_(std::move(done)),
364 lib_(CHECK_NOTNULL(ctx_->function_library())),
365 args_(1 + ctx_->num_inputs() - 3) {
366 args_[0] = Tensor(DT_INT32, {});
367 iter_ = &args_[0].scalar<int32>()();
368
369 const int32 num_loop_inputs = ctx_->num_inputs() - 3;
370 rets_.reserve(num_loop_inputs);
371 for (int i = 0; i < num_loop_inputs; ++i) {
372 rets_.push_back(ctx_->input(3 + i));
373 }
374 }
375
376 ~State() {}
377
378 void Start() {
379 Status s = StartLoop();
380 if (!s.ok()) Finish(s);
381 }
382
383 private:
384 ForOp* const kernel_;
385 OpKernelContext* const ctx_;
386 const DoneCallback done_;
387 FunctionLibraryRuntime* const lib_;
388 FunctionLibraryRuntime::Options opts_;
389 TensorVec args_;
390 TensorVec rets_;
391
392 int32* iter_; // points to args_[0].
393 int32 limit_;
394 int32 delta_;
395
396 // If an error e is returned, caller must call Finish(e).
397 // If OK is returned, the async loop execution has been started.
398 Status StartLoop() {
399 SetRunOptions(ctx_, &opts_, false /* always_collect_stats */);
400
401 TF_RETURN_IF_ERROR(GetScalar(ctx_, 0, iter_, "start"));
402 TF_RETURN_IF_ERROR(GetScalar(ctx_, 1, &limit_, "limit"));
403 TF_RETURN_IF_ERROR(GetScalar(ctx_, 2, &delta_, "delta"));
404
405 if ((delta_ > 0 && *iter_ <= limit_) ||
406 (delta_ < 0 && *iter_ >= limit_) ||
407 (delta_ == 0 && *iter_ == limit_)) {
408 RunNext();
409 return Status::OK();
410 } else {
411 return errors::InvalidArgument("Invalid start/limit/delta: ", *iter_,
412 " ", limit_, " ", delta_);
413 }
414 }
415
416 void RunNext() {
417 bool done_loop;
418 if (delta_ > 0) {
419 done_loop = *iter_ >= limit_;
420 } else {
421 done_loop = *iter_ <= limit_;
422 }
423 if (done_loop) {
424 Finish(Status::OK());
425 return;
426 }
427
428 if (rets_.size() >= args_.size()) {
429 Finish(errors::InvalidArgument(
430 "For loop body returned ", rets_.size(),
431 " arguments. Expected: ", args_.size() - 1));
432 return;
433 }
434 for (int i = 0; i < rets_.size(); ++i) {
435 args_[1 + i] = std::move(rets_[i]);
436 }
437 rets_.clear();
438 lib_->Run(opts_, kernel_->body_handle_, args_, &rets_,
439 [this](const Status& s) {
440 if (s.ok()) {
441 *iter_ += delta_;
442 RunNext();
443 } else {
444 Finish(s);
445 }
446 });
447 }
448
449 void Finish(Status s) {
450 if (s.ok()) {
451 s = SetOutputs(kernel_, ctx_, rets_);
452 }
453 ctx_->SetStatus(s);
454 done_();
455 delete this;
456 }
457 };
458};
