24// IRemoteFusedGraphOpsDefinitions is an interface class which provides
25// APIs to provide information about op types supported by SOC.
26// TODO(satok): Provide ways to transfer graph definitions into SOC
27class IRemoteFusedGraphOpsDefinitions {
28 public:
29 // op id which is not supported by SOC
30 static constexpr int INVALID_OP_ID = -1;
31
32 IRemoteFusedGraphOpsDefinitions() = default;
33 virtual ~IRemoteFusedGraphOpsDefinitions() = default;
34 // Return total ops count supported by SOC
35 virtual int GetTotalOpsCount() const = 0;
36 // Return op id for given string op name
37 virtual int GetOpIdFor(const string& op_type,
38 const DataTypeVector& dt) const = 0;
39
40 private:
41 TF_DISALLOW_COPY_AND_ASSIGN(IRemoteFusedGraphOpsDefinitions);
42};
