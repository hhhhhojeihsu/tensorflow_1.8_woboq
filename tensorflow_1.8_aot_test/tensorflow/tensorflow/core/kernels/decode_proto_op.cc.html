16// DecodeProto is a TensorFlow Op which extracts arbitrary fields
17// from protos serialized as strings.
18//
19// See docs in ../ops/decode_proto_op.cc.
20//
21// This implementation reads the serialized format using a handful of
22// calls from the WireFormatLite API used by generated proto code.
23// WireFormatLite is marked as an "internal" proto API but is widely
24// used in practice and highly unlikely to change.
25// This will be much faster than the previous implementation based on
26// constructing a temporary dynamic message in memory and using the
27// proto reflection api to read it.
28// It can be used with any proto whose descriptors are available at
29// runtime but should be competitive in speed with approaches that
30// compile in the proto definitions.
31

61// Returns true if the proto field type can be converted to the
62// tensorflow::DataType.
63bool CheckOutputType(FieldDescriptor::Type field_type, DataType output_type) {
64 switch (field_type) {
65 case WireFormatLite::TYPE_DOUBLE:
66 return output_type == tensorflow::DT_DOUBLE;
67 case WireFormatLite::TYPE_FLOAT:
68 return output_type == tensorflow::DT_FLOAT ||
69 output_type == tensorflow::DT_DOUBLE;
70 case WireFormatLite::TYPE_INT64:
71 return output_type == tensorflow::DT_INT64;
72 case WireFormatLite::TYPE_UINT64:
73 return output_type == tensorflow::DT_INT64;
74 case WireFormatLite::TYPE_INT32:
75 return output_type == tensorflow::DT_INT32;
76 case WireFormatLite::TYPE_FIXED64:
77 return output_type == tensorflow::DT_INT64;
78 case WireFormatLite::TYPE_FIXED32:
79 return output_type == tensorflow::DT_INT32 ||
80 output_type == tensorflow::DT_INT64;
81 case WireFormatLite::TYPE_BOOL:
82 return output_type == tensorflow::DT_BOOL;
83 case WireFormatLite::TYPE_STRING:
84 return output_type == tensorflow::DT_STRING;
85 case WireFormatLite::TYPE_GROUP:
86 return output_type == tensorflow::DT_STRING;
87 case WireFormatLite::TYPE_MESSAGE:
88 return output_type == tensorflow::DT_STRING;
89 case WireFormatLite::TYPE_BYTES:
90 return output_type == tensorflow::DT_STRING;
91 case WireFormatLite::TYPE_UINT32:
92 return output_type == tensorflow::DT_INT32 ||
93 output_type == tensorflow::DT_INT64;
94 case WireFormatLite::TYPE_ENUM:
95 return output_type == tensorflow::DT_INT32;
96 case WireFormatLite::TYPE_SFIXED32:
97 return output_type == tensorflow::DT_INT32;
98 case WireFormatLite::TYPE_SFIXED64:
99 return output_type == tensorflow::DT_INT64;
100 case WireFormatLite::TYPE_SINT32:
101 return output_type == tensorflow::DT_INT32;
102 case WireFormatLite::TYPE_SINT64:
103 return output_type == tensorflow::DT_INT64;
104 // default: intentionally omitted in order to enable static checking.
105 }
106}

108// A FieldInfo holds a handful of information from the FieldDescriptor
109// and user attributes.
110struct FieldInfo {
111 FieldInfo(const FieldDescriptor* field_desc, int user_index)
112 : output_index(user_index) {
113 // Without this intermediate data structure, the profile had hotspots
114 // calling methods of FieldDescriptor.
115 number = field_desc->number();
116
117 // The wire format library defines the same constants used in
118 // descriptor.proto. This static_cast is safe because they
119 // are guaranteed to stay in sync.
120 // We need the field type from the FieldDescriptor here
121 // because the wire format doesn't tell us anything about
122 // what happens inside a packed repeated field: there is
123 // enough information in the wire format to skip the
124 // whole field but not enough to know how to parse what's
125 // inside. For that we go to the schema.
126 type = static_cast<WireFormatLite::FieldType>(field_desc->type());
127 is_repeated = field_desc->is_repeated();
128 }
129
130 // Disable copy and move.
131 FieldInfo(const FieldInfo&) = delete;
132 FieldInfo& operator=(const FieldInfo&) = delete;
133
134 // Internally we sort field descriptors by wire number for
135 // fast lookup. In general this is different from the order
136 // given by the user. Output_index gives the index into
137 // the field_names and output_types attributes and into
138 // the output tensor list.
139 int output_index = -1;
140
141 // This is a cache of the relevant fields from `FieldDescriptorProto`.
142 // This was added after noticing that FieldDescriptor->type() was
143 // using 6% of the cpu profile.
144 WireFormatLite::FieldType type;
145 int number;
146 bool is_repeated;
147};

149// A CountCollector counts sizes of repeated and optional fields in a proto.
150//
151// Each field is tracked by a single CountCollector instance. The
152// instance manages a single count, which is stored as a pointer (it
153// is intended to be a reference to the `sizes` output which is being
154// filled in). The pointer is passed in at initialization.
155//
156// Counting is done as a separate pass in order to allocate output tensors
157// all at once. This allows the TensorFlow runtime to optimize allocation
158// for the consumer, while removing the need for copying inside this op.
159// After this pass, the DenseCollector class (below) gathers the data:
160// It is more complex and provides better motivation for the API here.
161class CountCollector {
162 public:
163 // Default constructor allows the collector to be a vector element.
164 CountCollector() = default;
165
166 // The count may be stored inside an Eigen Tensor to eliminate copying.
167 explicit CountCollector(int32* count) : count_ptr_(count) {}
168
169 // Reads (in this case counts) a single value.
170 Status ReadValue(CodedInputStream* input, const FieldInfo& field) {
171 // Only repeated fields can have count > 1.
172 if (*count_ptr_ == 0 || field.is_repeated) {
173 (*count_ptr_)++;
174 }
175 // We expect a wire type based on the schema field_type, to allow
176 // a little more checking.
177 if (!SkipValue(input, field)) {
178 return errors::DataLoss("ReadValue: Failed skipping field when counting");
179 }
180 return Status::OK();
181 }
182
183 // Reads (in this case counts) a length-delimited list of values.
184 Status ReadPackedValues(CodedInputStream* input, const FieldInfo& field,
185 size_t buf_size) {
186 if (buf_size == 0) {
187 return Status::OK();
188 }
189
190 const void* tmpbuf;
191 int unused_max_buf_size;
192
193 input->GetDirectBufferPointerInline(&tmpbuf, &unused_max_buf_size);
194 // This is safe because the underlying storage for the CodedInputStream is
195 // owned by the input tensor. If it were a Cord or file-backed stream this
196 // pointer would go stale after the bytes were skipped.
197 const uint8* buf = reinterpret_cast<const uint8*>(tmpbuf);
198
199 // Important: we skipped the input->{Push,Pop}Limit() calls for speed,
200 // so the bounds check on buf_size inside Skip() is critical, and
201 // must be done before scanning the contents.
202 if (!input->Skip(buf_size)) {
203 return errors::DataLoss("ReadPackedValues: Skipping packed field failed");
204 }
205
206 // Dispatch to the appropriately typed field reader based on the
207 // schema type.
208 Status st;
209 switch (field.type) {
210 case WireFormatLite::TYPE_DOUBLE:
211 st = CountPackedFixed<double>(buf, buf_size);
212 break;
213 case WireFormatLite::TYPE_FLOAT:
214 st = CountPackedFixed<float>(buf, buf_size);
215 break;
216 case WireFormatLite::TYPE_INT64:
217 st = CountPackedVarint(buf, buf_size);
218 break;
219 case WireFormatLite::TYPE_UINT64:
220 st = CountPackedVarint(buf, buf_size);
221 break;
222 case WireFormatLite::TYPE_INT32:
223 st = CountPackedVarint(buf, buf_size);
224 break;
225 case WireFormatLite::TYPE_FIXED64:
226 st = CountPackedFixed<uint64>(buf, buf_size);
227 break;
228 case WireFormatLite::TYPE_FIXED32:
229 st = CountPackedFixed<uint32>(buf, buf_size);
230 break;
231 case WireFormatLite::TYPE_BOOL:
232 st = CountPackedVarint(buf, buf_size);
233 break;
234 case WireFormatLite::TYPE_STRING:
235 st = errors::DataLoss("TYPE_STRING encountered as packed");
236 break;
237 case WireFormatLite::TYPE_GROUP:
238 st = errors::DataLoss("TYPE_GROUP encountered as packed");
239 break;
240 case WireFormatLite::TYPE_MESSAGE:
241 st = errors::DataLoss("TYPE_MESSAGE encountered as packed");
242 break;
243 case WireFormatLite::TYPE_BYTES:
244 st = errors::DataLoss("TYPE_BYTES encountered as packed");
245 break;
246 case WireFormatLite::TYPE_UINT32:
247 st = CountPackedVarint(buf, buf_size);
248 break;
249 case WireFormatLite::TYPE_ENUM:
250 st = CountPackedVarint(buf, buf_size);
251 break;
252 case WireFormatLite::TYPE_SFIXED32:
253 st = CountPackedFixed<int32>(buf, buf_size);
254 break;
255 case WireFormatLite::TYPE_SFIXED64:
256 st = CountPackedFixed<int64>(buf, buf_size);
257 break;
258 case WireFormatLite::TYPE_SINT32:
259 st = CountPackedVarint(buf, buf_size);
260 break;
261 case WireFormatLite::TYPE_SINT64:
262 st = CountPackedVarint(buf, buf_size);
263 break;
264 // default: intentionally omitted in order to enable static checking.
265 }
266 if (!st.ok()) {
267 return st;
268 }
269
270 if (!field.is_repeated && *count_ptr_ > 1) {
271 *count_ptr_ = 1;
272 }
273 return Status::OK();
274 }
275
276 private:
277 // Skips a length-delimited value.
278 static bool SkipBytes(CodedInputStream* input) {
279 uint32 length;
280 if (!input->ReadVarint32(&length)) {
281 return false;
282 }
283 return input->Skip(length);
284 }
285
286 // Counts the number of packed varints in an array.
287 // The end of a varint is signaled by a value < 0x80,
288 // so counting them requires parsing the bytestream.
289 // It is the caller's responsibility to ensure that len > 0.
290 Status CountPackedVarint(const uint8* buf, size_t len) {
291 const uint8* bound = buf + len;
292 int count;
293
294 // The last byte in a valid encoded varint is guaranteed to have
295 // the high bit unset. We rely on this property to prevent
296 // ReadVarint64FromArray from going out of bounds, so validate
297 // the end of the buf before scanning anything.
298 if (bound[-1] & 0x80) {
299 return errors::DataLoss("Corrupt packed varint");
300 }
301
302 // Now we can trust ReadVarint64FromArray to stay in bounds.
303 for (count = 0; buf < bound; ++count) {
304 uint64 temp;
305 bool ok;
306 buf = internal::ReadVarint64FromArray(buf, &ok, &temp);
307 if (!ok) {
308 return errors::DataLoss("Corrupt packed varint");
309 }
310 }
311
312 *count_ptr_ += count;
313 return Status::OK();
314 }
315
316 // Counts the number of fixed-size values in a packed field.
317 // This can be done without actually parsing anything.
318 template <typename T>
319 Status CountPackedFixed(const uint8* unused_buf, size_t len) {
320 int count = len / sizeof(T);
321 if (count * sizeof(T) != len) {
322 return errors::DataLoss(
323 "Illegal data length for packed fixed-size type: ", len);
324 }
325 *count_ptr_ += len / sizeof(T);
326 return Status::OK();
327 }
328
329 // Skips a single value in the input stream.
330 // Dispatches to the appropriately typed field skipper based on the
331 // schema type tag.
332 // This is not as permissive as just handling the wire type.
333 static bool SkipValue(CodedInputStream* input, const FieldInfo& field) {
334 uint32 tmp32;
335 protobuf_uint64 tmp64;
336 switch (field.type) {
337 case WireFormatLite::TYPE_DOUBLE:
338 return input->ReadLittleEndian64(&tmp64);
339 case WireFormatLite::TYPE_FLOAT:
340 return input->ReadLittleEndian32(&tmp32);
341 case WireFormatLite::TYPE_INT64:
342 return input->ReadVarint64(&tmp64);
343 case WireFormatLite::TYPE_UINT64:
344 return input->ReadVarint64(&tmp64);
345 case WireFormatLite::TYPE_INT32:
346 return input->ReadVarint32(&tmp32);
347 case WireFormatLite::TYPE_FIXED64:
348 return input->ReadLittleEndian64(&tmp64);
349 case WireFormatLite::TYPE_FIXED32:
350 return input->ReadLittleEndian32(&tmp32);
351 case WireFormatLite::TYPE_BOOL:
352 return input->ReadVarint32(&tmp32);
353 case WireFormatLite::TYPE_STRING:
354 return SkipBytes(input);
355 case WireFormatLite::TYPE_GROUP:
356 return WireFormatLite::SkipField(
357 input, WireFormatLite::MakeTag(
358 field.number, WireFormatLite::WIRETYPE_START_GROUP));
359 case WireFormatLite::TYPE_MESSAGE:
360 return SkipBytes(input);
361 case WireFormatLite::TYPE_BYTES:
362 return SkipBytes(input);
363 case WireFormatLite::TYPE_UINT32:
364 return input->ReadVarint32(&tmp32);
365 case WireFormatLite::TYPE_ENUM:
366 return input->ReadVarint32(&tmp32);
367 case WireFormatLite::TYPE_SFIXED32:
368 return input->ReadLittleEndian32(&tmp32);
369 case WireFormatLite::TYPE_SFIXED64:
370 return input->ReadLittleEndian64(&tmp64);
371 case WireFormatLite::TYPE_SINT32:
372 return input->ReadVarint32(&tmp32);
373 case WireFormatLite::TYPE_SINT64:
374 return input->ReadVarint64(&tmp64);
375 // default: intentionally omitted in order to enable static checking.
376 }
377 }
378
379 int32* count_ptr_ = nullptr;
380};

382// A DenseCollector accumulates values from a proto into a tensor.
383//
384// There is an instance of DenseCollector for each field of each
385// proto. The DenseCollector deserializes the value from the wire
386// directly into the preallocated output Tensor.
387//
388// This class is named DenseCollector because in the future there should
389// be a SparseCollector that accumulates field data into sparse tensors if
390// the user requests it.
391class DenseCollector {
392 public:
393 // Default constructor allows the collector to be a vector element.
394 DenseCollector() = default;
395
396 // A DenseCollector applies to one field of a serialized message.
397 DenseCollector(uint8* datap, DataType dtype, int max_repeat_count)
398 : datap_(datap), dtype_(dtype), max_repeat_count_(max_repeat_count) {}
399
400 // Reads a value from the input stream and stores it.
401 //
402 // Always inlining gave a ~50% speedup on microbenchmarks at one point.
403 // TODO(nix): try removing it to see if that still holds.
404 // TODO(jsimsa): ABSL_ATTRIBUTE_ALWAYS_INLINE
405 Status ReadValue(CodedInputStream* input, const FieldInfo& field) {
406 // For required and optional fields, we overwrite values[0] with
407 // the latest one in the wire stream.
408 // See https://developers.google.com/protocol-buffers/docs/encoding#optional
409 // Only for repeated fields do we advance the next_repeat_index_ past 1.
410 // TODO(nix): to handle oneof we must also zero out any previous values
411 // seen on the wire.
412 int32 index = 0;
413 if (field.is_repeated) {
414 index = next_repeat_index_;
415 }
416 next_repeat_index_ = index + 1;
417
418 return internal::ReadValue(input, field.type, field.number, dtype_, index,
419 datap_);
420 }
421
422 // Reads and stores a length-delimited list of values.
423 Status ReadPackedValues(CodedInputStream* input, const FieldInfo& field,
424 const size_t buf_size) {
425 const void* buf;
426 int unused_max_buf_size;
427 input->GetDirectBufferPointerInline(&buf, &unused_max_buf_size);
428 // This is safe because the underlying storage for the CodedInputStream is
429 // owned by the input tensor. If it were a Cord or file-backed stream this
430 // pointer would go stale after the bytes were skipped.
431 if (!input->Skip(buf_size)) {
432 return errors::DataLoss(
433 "ReadPackedValues: Skipping packed field failed. Field tag: ",
434 field.number);
435 }
436
437 // Setting stride=0 causes new values to overwrite old ones for
438 // non-repeated fields.
439 const int stride = field.is_repeated ? 1 : 0;
440
441 if (next_repeat_index_ >= max_repeat_count_) {
442 return errors::DataLoss(
443 "ReadPackedValues: Tried to write more entries than allowed. "
444 "Field tag: ",
445 field.number, ", Max entries allowed: ", max_repeat_count_);
446 } else {
447 return internal::ReadPackedFromArray(buf, buf_size, field.type,
448 field.number, dtype_, stride,
449 &next_repeat_index_, datap_);
450 }
451 }
452
453 // Fills in any missing values in the output array with defaults.
454 // Dispatches to the appropriately typed field default based on the
455 // runtime type tag.
456 Status FillWithDefaults() {
457 switch (dtype_) {
458 case DataType::DT_FLOAT:
459 return FillDefault<float>();
460 case DataType::DT_DOUBLE:
461 return FillDefault<double>();
462 case DataType::DT_INT32:
463 return FillDefault<int32>();
464 case DataType::DT_UINT8:
465 return FillDefault<uint8>();
466 case DataType::DT_INT8:
467 return FillDefault<int8>();
468 case DataType::DT_STRING:
469 return FillDefault<string>();
470 case DataType::DT_INT64:
471 return FillDefault<int64>();
472 case DataType::DT_BOOL:
473 return FillDefault<bool>();
474 default:
475 // There are many tensorflow dtypes not handled here, but they
476 // should not come up unless type casting is added to the Op.
477 // Chaining with tf.cast() should do the right thing until then.
478 return errors::DataLoss(
479 "Failed filling defaults in unknown tf::DataType");
480 }
481 }
482
483 private:
484 // Fills empty values in the dense representation with a
485 // default value. This uses next_repeat_index_ which counts the number
486 // of parsed values for the field.
487 template <class T>
488 Status FillDefault() {
489 for (int i = next_repeat_index_; i < max_repeat_count_; i++) {
490 reinterpret_cast<T*>(datap_)[i] = T();
491 }
492 return Status::OK();
493 }
494
495 int32 next_repeat_index_ = 0;
496
497 // This is a pointer to data_[message_index_].
498 // There is no bounds checking at this level: we computed the max
499 // repeat size for each field in CountCollector and use the same
500 // code to traverse it here, so we are guaranteed not to be called
501 // for more items than we have allocated space.
502 void* const datap_ = nullptr;
503
504 const DataType dtype_ = DataType::DT_INVALID;
505 const int max_repeat_count_ = 0;
506};

508class DecodeProtoOp : public OpKernel {
509 public:
510 explicit DecodeProtoOp(OpKernelConstruction* context) : OpKernel(context) {
511 string descriptor_source;
512 OP_REQUIRES_OK(context,
513 context->GetAttr("descriptor_source", &descriptor_source));
514
515 // We always get back a desc_pool, but we may not own it. If we own it,
516 // owned_desc_pool_ will be filled in.
517 DescriptorPool const* desc_pool;
518 OP_REQUIRES_OK(context, GetDescriptorPool(context->env(), descriptor_source,
519 &desc_pool, &owned_desc_pool_));
520
521 string message_type;
522 OP_REQUIRES_OK(context, context->GetAttr("message_type", &message_type));
523
524 const Descriptor* message_desc =
525 desc_pool->FindMessageTypeByName(message_type);
526 OP_REQUIRES(context, message_desc != nullptr,
527 errors::InvalidArgument("No descriptor found for message type ",
528 message_type));
529
530 std::vector<string> field_names;
531 OP_REQUIRES_OK(context, context->GetAttr("field_names", &field_names));
532 std::vector<DataType> output_types;
533 OP_REQUIRES_OK(context, context->GetAttr("output_types", &output_types));
534 OP_REQUIRES(
535 context, field_names.size() == output_types.size(),
536 errors::InvalidArgument("field_names and output_types attributes must "
537 "have the same length"));
538
539 // Gather the field descriptors and check that requested output types match.
540
541 int field_index = 0;
542 std::vector<const FieldDescriptor*> field_descs;
543 for (const string& name : field_names) {
544 auto fd = message_desc->FindFieldByName(name);
545 OP_REQUIRES(context, fd != nullptr,
546 errors::InvalidArgument("Unknown field: ", name,
547 " in message type ", message_type));
548 OP_REQUIRES(context,
549 CheckOutputType(fd->type(), output_types[field_index]),
550 // Many TensorFlow types don't have corresponding proto types
551 // and the user will get an error if they are requested. It
552 // would be nice to allow conversions here, but tf.cast
553 // already exists so we don't duplicate the functionality.
554 // Known unhandled types:
555 // DT_INT16 DT_COMPLEX64 DT_QINT8 DT_QUINT8 DT_QINT32
556 // DT_BFLOAT16 DT_QINT16 DT_QUINT16 DT_UINT16
557 errors::InvalidArgument("Unexpected output type for ",
558 fd->full_name(), ": ", fd->cpp_type(),
559 " to ", output_types[field_index]));
560
561 field_index++;
562 field_descs.push_back(fd);
563 }
564
565 // Internally we want the field_descs sorted by their number on the wire.
566 // But the output tensors are allocated in the order given by the caller.
567 // Build a mapping i->j, where field_descs[i] corresponds to outputs[j].
568 std::vector<int> output_indices;
569 output_indices.reserve(field_names.size());
570 for (int i = 0; i < field_names.size(); i++) {
571 output_indices.push_back(i);
572 }
573 std::sort(output_indices.begin(), output_indices.end(),
574 [field_descs](int a, int b) {
575 return field_descs[a]->number() < field_descs[b]->number();
576 });
577
578 // Now store the fields in sorted order.
579 for (int i = 0; i < field_names.size(); i++) {
580 fields_.push_back(MakeUnique<FieldInfo>(field_descs[output_indices[i]],
581 output_indices[i]));
582 }
583
584 message_prototype_ = message_factory_.GetPrototype(message_desc);
585 OP_REQUIRES(context, message_prototype_ != nullptr,
586 errors::InvalidArgument("Couldn't get prototype message: ",
587 message_desc->full_name()));
588 string format;
589 OP_REQUIRES_OK(context, context->GetAttr("message_format", &format));
590 OP_REQUIRES(
591 context, format == "binary" || format == "text",
592 errors::InvalidArgument("format must be one of binary or text"));
593 is_binary_ = format == "binary";
594
595 // Enable the initial protobuf sanitizer, which is much
596 // more expensive than the decoder.
597 // TODO(nix): Remove this once the fast decoder
598 // has passed security review.
599 OP_REQUIRES_OK(context, context->GetAttr("sanitize", &sanitize_));
600 }
601
602 void Compute(OpKernelContext* ctx) override {
603 const Tensor& buf_tensor = ctx->input(0);
604 int message_count = buf_tensor.NumElements();
605 OP_REQUIRES(ctx, message_count >= 1,
606 errors::InvalidArgument(
607 "Bufs argument must contain at least one value"));
608
609 int field_count = fields_.size();
610
611 // Save the argument shape for later, then flatten the input
612 // Tensor since we are working componentwise. We will restore
613 // the same shape in the returned Tensor.
614 const TensorShape& shape_prefix = buf_tensor.shape();
615
616 TensorShape sizes_shape = shape_prefix;
617 sizes_shape.AddDim(field_count);
618 Tensor* sizes_tensor = nullptr;
619 OP_REQUIRES_OK(ctx, ctx->allocate_output(0, sizes_shape, &sizes_tensor));
620
621 // This is used to allocate binary bufs if used. It serves only
622 // to define memory ownership.
623 std::vector<string> tmp_binary_bufs(message_count);
624
625 // These are the actual buffers to use, which may be in tmp_binary_bufs
626 // or may be pointers into the buf_tensor. Either way they are not owned
627 // here.
628 std::vector<const string*> bufs;
629
630 if (is_binary_ && !sanitize_) {
631 // Fast path.
632 for (int mi = 0; mi < message_count; ++mi) {
633 const string* buf = &buf_tensor.flat<string>()(mi);
634 bufs.push_back(buf);
635 }
636 } else {
637 // We will have to allocate a copy, either to convert from text to
638 // binary or to sanitize a binary proto.
639 for (int mi = 0; mi < message_count; ++mi) {
640 ReserializeMessage(ctx, buf_tensor.flat<string>()(mi),
641 &tmp_binary_bufs[mi]);
642 if (!ctx->status().ok()) {
643 return;
644 }
645 bufs.push_back(&tmp_binary_bufs[mi]);
646 }
647 }
648
649 // Walk through all the strings in the input tensor, counting
650 // the number of fields in each.
651 // We can't allocate our actual output Tensor until we know the
652 // maximum repeat count, so we do a first pass through the serialized
653 // proto just counting fields.
654 // We always allocate at least one value so that optional fields
655 // are populated with default values - this avoids a TF
656 // conditional when handling the output data.
657 // The caller can distinguish between real data and defaults
658 // using the repeat count matrix that is returned by decode_proto.
659 std::vector<int32> max_sizes(field_count, 1);
660 for (int mi = 0; mi < message_count; ++mi) {
661 CountFields(ctx, mi, *bufs[mi], sizes_tensor, &max_sizes);
662 if (!ctx->status().ok()) {
663 return;
664 }
665 }
666
667 // Allocate the output tensors now that we've seen the max size.
668 // TODO(nix): Use allocate_output_or_forward_input for the largest
669 // output tensor. This can avoid one large allocation by re-using
670 // the memory of the input tensor.
671 std::vector<Tensor*> outputs(field_count);
672 for (int fi = 0; fi < field_count; ++fi) {
673 TensorShape flat_shape = {static_cast<int64>(message_count),
674 max_sizes[fi]};
675 TensorShape out_shape = shape_prefix;
676 out_shape.AddDim(max_sizes[fi]);
677
678 // Surprisingly we don't specify the types from the output_types
679 // attribute: that is done for us based on the Op declaration:
680 // REGISTER_OP(...)
681 // .Attr("output_types: list(type) >= 0")
682 // .Output("values: output_types")
683 OP_REQUIRES_OK(ctx,
684 // ctx->allocate_output(output_indices_[fi] + 1,
685 ctx->allocate_output(fields_[fi]->output_index + 1,
686 out_shape, &outputs[fi]));
687 }
688
689 // Make the second pass through the serialized proto, decoding
690 // into preallocated tensors.
691 AccumulateFields(ctx, bufs, outputs);
692 }

694 private:
695 // Copy a serialized message to binary, e.g. to handle text proto inputs.
696 void ReserializeMessage(OpKernelContext* ctx, const string& buf,
697 string* binary_buf) {
698 // Handle text protos by translating them to binary.
699 std::unique_ptr<Message> message(message_prototype_->New());
700 OP_REQUIRES(ctx, message, errors::DataLoss("Initializing message failed"));
701
702 if (is_binary_) {
703 // If we get here we are sanitizing the input protobuf by parsing
704 // and reserializing it with a trusted (but very slow) library.
705 OP_REQUIRES(ctx, message->ParseFromString(buf),
706 errors::DataLoss("Unable to parse binary protobuf"));
707 } else {
708 OP_REQUIRES(ctx, TextFormat::ParseFromString(buf, message.get()),
709 errors::DataLoss("Unable to parse text protobuf"));
710 }
711
712 OP_REQUIRES(ctx, message->SerializeToString(binary_buf),
713 errors::DataLoss("Unable to reserialize text proto as binary"));
714 }

716 // Count the number of occurrences of each requested field in a message batch.
717 void CountFields(OpKernelContext* ctx, int message_index, const string& buf,
718 Tensor* sizes_tensor, std::vector<int32>* max_sizes) {
719 int field_count = fields_.size();
720
721 CodedInputStream input(reinterpret_cast<const uint8*>(buf.c_str()),
722 buf.size());
723
724 std::vector<int32> field_sizes(field_count, 0);
725 std::vector<CountCollector> counters;
726 counters.reserve(field_count);
727 for (int i = 0; i < field_count; i++) {
728 counters.emplace_back(&field_sizes[i]);
729 }
730
731 Status st = Collect(&input, &counters);
732 if (st.ok() && !input.ConsumedEntireMessage()) {
733 st = errors::DataLoss("CountFields: Failed to consume entire buffer");
734 }
735 if (kFailOnDecodeError) {
736 OP_REQUIRES_OK(ctx, st); // NOLINT
737 }
738 if (!st.ok()) {
739 // This code suppresses the corrupt proto, treating it as empty
740 // to avoid crashing the process.
741 LOG(WARNING) << "Proto counting error for message type " << message_type_
742 << ": " << st;
743
744 for (int fi = 0; fi < field_count; fi++) {
745 field_sizes[fi] = 0;
746 }
747 // Finished decoding this message.
748 return;
749 }
750
751 // Update the size tensor and max repeat size for each field.
752 auto sizes = sizes_tensor->flat_inner_dims<int32>();
753 for (int fi = 0; fi < field_count; fi++) {
754 int32 size = field_sizes[fi];
755 sizes(message_index, fields_[fi]->output_index) = size;
756 if ((*max_sizes)[fi] < size) {
757 (*max_sizes)[fi] = size;
758 }
759 }
760 }

762 // Parse fields from a serialized message into preallocated tensors.
763 void AccumulateFields(OpKernelContext* ctx,
764 const std::vector<const string*>& bufs,
765 std::vector<Tensor*> outputs) {
766 struct TensorInfo {
767 explicit TensorInfo(Tensor* tensor) {
768 // Note that we can decode only max_repeat_count values before overflow.
769 // No other bounds checking is done for repeated fields. For
770 // optional fields there is a check to make sure that only the last
771 // value on the wire appears in the output tensor.
772 dtype = tensor->dtype();
773 last_dim_size = tensor->dim_size(tensor->dims() - 1);
774
775 if (dtype != DT_STRING) {
776 const int element_size = DataTypeSize(dtype);
777 CHECK_GT(element_size, 0);
778 stride = last_dim_size * element_size;
779
780 const int64 flatshape[1] = {tensor->NumElements() * element_size};
781 data = tensor->bit_casted_shaped<uint8, 1>(flatshape).data();
782 } else {
783 // DataTypeSize() returns 0 for string types.
784 stride = last_dim_size * sizeof(string);
785 data = reinterpret_cast<uint8*>(tensor->flat<string>().data());
786 }
787 }
788
789 DataType dtype;
790 int last_dim_size;
791 int stride;
792 uint8* data;
793 };
794
795 int field_count = fields_.size();
796
797 std::vector<TensorInfo> tensors;
798 tensors.reserve(field_count);
799 for (int fi = 0; fi < field_count; fi++) {
800 tensors.emplace_back(outputs[fi]);
801 }
802
803 for (int message_index = 0; message_index < bufs.size(); ++message_index) {
804 const string& buf = *bufs[message_index];
805
806 std::vector<DenseCollector> collectors;
807 collectors.reserve(field_count);
808 for (const TensorInfo& info : tensors) {
809 collectors.emplace_back(info.data + message_index * info.stride,
810 info.dtype, info.last_dim_size);
811 }
812
813 // Fill in output tensors from the wire.
814 CodedInputStream input(reinterpret_cast<const uint8*>(buf.c_str()),
815 buf.size());
816 Status st = Collect(&input, &collectors);
817 if (st.ok() && !input.ConsumedEntireMessage()) {
818 st = errors::DataLoss(
819 "AccumulateFields: Failed to consume entire buffer");
820 }
821 if (kFailOnDecodeError) {
822 OP_REQUIRES_OK(ctx, st); // NOLINT
823 }
824 if (!st.ok()) {
825 // This code suppresses the corrupt proto, treating it as empty
826 // to avoid crashing training.
827 LOG(WARNING) << "Proto counting error for message type "
828 << message_type_ << ": " << st;
829 }
830
831 // Fill the remainder of the dense outputs with default values.
832 for (auto& collector : collectors) {
833 OP_REQUIRES_OK(ctx, collector.FillWithDefaults());
834 }
835 }
836 }

838 // Look up the FieldDescriptor for a particular field number.
839 bool LookupField(int field_number, int* field_index) {
840 // Look up the FieldDescriptor using linear search.
841 // TODO(nix): this could be sped up with binary search, but we are
842 // already way off the fastpath at this point. If you see a hotspot
843 // here, somebody is sending you very inefficient protos.
844 for (int fi = fields_.size() - 1; fi >= 0; fi--) {
845 if (field_number == fields_[fi]->number) {
846 *field_index = fi;
847 return true;
848 }
849 }
850 return false;
851 }

853 // Traverses a serialized protobuf, dispatching values to the collectors.
854 template <class CollectorClass>
855 Status Collect(CodedInputStream* input,
856 std::vector<CollectorClass>* collectors) {
857 int last_good_field_index = -1;
858 bool fields_disordered = false;
859 int prev_field_number = -1;
860 int field_number = -1;
861 int last_good_field_number = -1;
862 int next_good_field_number = fields_[0]->number;
863
864 // The 'tag' variable should always be treated as tainted.
865 for (uint32 tag = input->ReadTag();
866 tag != 0 && WireFormatLite::GetTagWireType(tag) !=
867 WireFormatLite::WIRETYPE_END_GROUP;
868 tag = input->ReadTag(), prev_field_number = field_number) {
869 field_number = WireFormatLite::GetTagFieldNumber(tag);
870 const FieldInfo* field = nullptr;
871
872 // This takes advantage of the sorted field numbers in most serialized
873 // protos: it tries the next expected field first rather than doing
874 // a lookup by field number.
875 // TODO(nix): haberman@ suggests a hybrid approach with a lookup table
876 // for small field numbers and a hash table for larger ones. This would
877 // be a simpler approach that should offer comparable speed in most
878 // cases.
879 if (field_number == last_good_field_number) {
880 field = fields_[last_good_field_index].get();
881 } else {
882 if (field_number < prev_field_number) {
883 fields_disordered = true;
884 }
885
886 // If fields are out of order, fall back to slow lookup.
887 if (fields_disordered) {
888 int field_index;
889 if (LookupField(field_number, &field_index)) {
890 field = fields_[field_index].get();
891 last_good_field_index = field_index;
892 }
893 } else {
894 // If we see a field that is past the next field we want,
895 // it was empty. Look for the one after that.
896 // Repeat until we run out of fields that we care about.
897 while (field_number >= next_good_field_number) {
898 if (field_number == next_good_field_number) {
899 last_good_field_number = field_number;
900 field = fields_[last_good_field_index + 1].get();
901 }
902
903 // Start looking for the field after the current one.
904 ++last_good_field_index;
905 if (last_good_field_index < fields_.size() - 1) {
906 next_good_field_number =
907 fields_[last_good_field_index + 1]->number;
908 } else {
909 // Saw something past the last field we care about.
910 // Continue parsing the message just in case there
911 // are disordered fields later, but any remaining
912 // ordered fields will have no effect.
913 next_good_field_number = INT_MAX;
914 }
915 }
916 }
917 }
918
919 if (!field) {
920 // Unknown and unrequested fields are skipped.
921 if (!WireFormatLite::SkipField(input, tag)) {
922 return errors::DataLoss("Failed skipping unrequested field");
923 }
924 continue;
925 }
926
927 Status st = CollectField(*field, WireFormatLite::GetTagWireType(tag),
928 input, &(*collectors)[last_good_field_index]);
929 if (!st.ok()) {
930 return st;
931 }
932 }
933 return Status::OK();
934 }

936 // Collects values for a single field.
937 template <class CollectorClass>
938 Status CollectField(const FieldInfo& field,
939 WireFormatLite::WireType wire_type,
940 CodedInputStream* input, CollectorClass* collector) {
941 // The wire format library defines the same constants used in
942 // descriptor.proto. This static_cast is safe because they
943 // are guaranteed to stay in sync.
944 // We need the field type from the FieldDescriptor here
945 // because the wire format doesn't tell us anything about
946 // what happens inside a packed repeated field: there is
947 // enough information in the wire format to skip the
948 // whole field but not enough to know how to parse what's
949 // inside. For that we go to the schema.
950 WireFormatLite::WireType schema_wire_type =
951 WireFormatLite::WireTypeForFieldType(field.type);
952
953 // Handle packed repeated fields. SkipField would skip the
954 // whole length-delimited blob without letting us count the
955 // values, so we have to scan them ourselves.
956 if (wire_type == WireFormatLite::WIRETYPE_LENGTH_DELIMITED &&
957 schema_wire_type != WireFormatLite::WIRETYPE_LENGTH_DELIMITED) {
958 // Handle packed repeated primitives.
959 int length;
960 if (!input->ReadVarintSizeAsInt(&length)) {
961 return errors::DataLoss("CollectField: Failed reading packed size");
962 }
963 Status st = collector->ReadPackedValues(input, field, length);
964 if (!st.ok()) {
965 return st;
966 }
967 return Status::OK();
968 }
969
970 // Read ordinary values, including strings, bytes, and messages.
971 if (wire_type != schema_wire_type) {
972 if (!WireFormatLite::SkipField(
973 input, WireFormatLite::MakeTag(field.number, wire_type))) {
974 return errors::DataLoss(
975 "CollectField: Failed skipping malformed field");
976 }
977 return Status::OK();
978 }
979 return collector->ReadValue(input, field);
980 }
981
982 string message_type_;
983 // Note that fields are sorted by increasing field number,
984 // which is not in general the order given by the user-specified
985 // field_names and output_types Op attributes.
986 std::vector<std::unique_ptr<const FieldInfo>> fields_;
987
988 // Owned_desc_pool_ is null when using descriptor_source=local.
989 std::unique_ptr<DescriptorPool> owned_desc_pool_;
990 DynamicMessageFactory message_factory_;
991 const Message* message_prototype_;
992
993 // True if decoding binary format, false if decoding text format.
994 bool is_binary_;
995
996 // True if the protos should be sanitized before parsing.
997 // Enables the initial protobuf sanitizer, which is much
998 // more expensive than the decoder. The flag defaults to true
999 // but can be set to false for trusted sources.
1000 // TODO(nix): flip the default to false when the fast decoder
1001 // has passed security review.
1002 bool sanitize_;
1003
1004 TF_DISALLOW_COPY_AND_ASSIGN(DecodeProtoOp);
1005};
