47class BinaryOpShared : public OpKernel {
48 public:
49 explicit BinaryOpShared(OpKernelConstruction* ctx, DataType out, DataType in);
50
51 protected:
52 struct BinaryOpState {
53 // Sets up bcast with the shape of in0 and in1, ensures that the bcast
54 // is valid, and if so, set out, either by allocating a new buffer using
55 // ctx->output(...) or by creating an alias for an owned input buffer for
56 // in-place computation.
57 // Caller must check ctx->status() upon return for non-ok status.
58 // If ctx->status().ok() is true, then out is guaranteed to be allocated.
59 BinaryOpState(OpKernelContext* ctx);
60
61 const Tensor& in0;
62 const Tensor& in1;
63
64 BCast bcast;
65 Tensor* out = nullptr;
66 int64 out_num_elements;
67
68 int64 in0_num_elements;
69 int64 in1_num_elements;
70
71 int ndims;
72 };
73
74 void SetUnimplementedError(OpKernelContext* ctx);
75 void SetComputeError(OpKernelContext* ctx);
76};

78// Coefficient-wise binary operations:
79// Device: E.g., CPUDevice, GPUDevice.
80// Functor: defined in cwise_ops.h. E.g., functor::add.
81template <typename Device, typename Functor>
82class BinaryOp : public BinaryOpShared {
83 public:
84 typedef typename Functor::in_type Tin; // Input scalar data type.
85 typedef typename Functor::out_type Tout; // Output scalar data type.
86
87 explicit BinaryOp(OpKernelConstruction* ctx)
88 : BinaryOpShared(ctx, DataTypeToEnum<Tout>::v(),
89 DataTypeToEnum<Tin>::v()) {}
90
91 void Compute(OpKernelContext* ctx) override {
92 // 'state': Shared helper not dependent on T to reduce code size
93 BinaryOpState state(ctx);
94 if (!ctx->status().ok()) return;
95 Tensor* out = state.out;
96 BCast* bcast = &state.bcast;
97 auto& in0 = state.in0;
98 auto& in1 = state.in1;
99 if (state.out_num_elements == 0) {
100 return;
101 }
102 const int ndims = state.ndims;
103 const Device& eigen_device = ctx->eigen_device<Device>();
104 bool error = false;
105 bool* const error_ptr = Functor::has_errors ? &error : nullptr;
106 if (ndims <= 1) {
107 auto out_flat = out->flat<Tout>();
108 if (state.in1_num_elements == 1) {
109 // tensor op scalar
110 functor::BinaryFunctor<Device, Functor, 1>().Right(
111 eigen_device, out_flat, in0.template flat<Tin>(),
112 in1.template scalar<Tin>(), error_ptr);
113 } else if (state.in0_num_elements == 1) {
114 // scalar op tensor
115 functor::BinaryFunctor<Device, Functor, 1>().Left(
116 eigen_device, out_flat, in0.template scalar<Tin>(),
117 in1.template flat<Tin>(), error_ptr);
118 } else {
119 functor::BinaryFunctor<Device, Functor, 1>()(
120 eigen_device, out_flat, in0.template flat<Tin>(),
121 in1.template flat<Tin>(), error_ptr);
122 }
123 } else if (ndims == 2) {
124 functor::BinaryFunctor<Device, Functor, 2>().BCast(
125 eigen_device, out->shaped<Tout, 2>(bcast->result_shape()),
126 in0.template shaped<Tin, 2>(bcast->x_reshape()),
127 BCast::ToIndexArray<2>(bcast->x_bcast()),
128 in1.template shaped<Tin, 2>(bcast->y_reshape()),
129 BCast::ToIndexArray<2>(bcast->y_bcast()), error_ptr);
130 } else if (ndims == 3) {
131 functor::BinaryFunctor<Device, Functor, 3>().BCast(
132 eigen_device, out->shaped<Tout, 3>(bcast->result_shape()),
133 in0.template shaped<Tin, 3>(bcast->x_reshape()),
134 BCast::ToIndexArray<3>(bcast->x_bcast()),
135 in1.template shaped<Tin, 3>(bcast->y_reshape()),
136 BCast::ToIndexArray<3>(bcast->y_bcast()), error_ptr);
137 } else if (ndims == 4) {
138 functor::BinaryFunctor<Device, Functor, 4>().BCast(
139 eigen_device, out->shaped<Tout, 4>(bcast->result_shape()),
140 in0.template shaped<Tin, 4>(bcast->x_reshape()),
141 BCast::ToIndexArray<4>(bcast->x_bcast()),
142 in1.template shaped<Tin, 4>(bcast->y_reshape()),
143 BCast::ToIndexArray<4>(bcast->y_bcast()), error_ptr);
144 } else if (ndims == 5) {
145 functor::BinaryFunctor<Device, Functor, 5>().BCast(
146 eigen_device, out->shaped<Tout, 5>(bcast->result_shape()),
147 in0.template shaped<Tin, 5>(bcast->x_reshape()),
148 BCast::ToIndexArray<5>(bcast->x_bcast()),
149 in1.template shaped<Tin, 5>(bcast->y_reshape()),
150 BCast::ToIndexArray<5>(bcast->y_bcast()), error_ptr);
151 } else {
152 SetUnimplementedError(ctx);
153 }
154 if (Functor::has_errors && error) {
155 SetComputeError(ctx);
156 }
157 }
158};

160template <typename Device, typename T>
161class ApproximateEqualOp : public OpKernel {
162 public:
163 explicit ApproximateEqualOp(OpKernelConstruction* context)
164 : OpKernel(context) {
165 float tolerance;
166 OP_REQUIRES_OK(context, context->GetAttr("tolerance", &tolerance));
167 tolerance_ = T(tolerance);
168 }
169 void Compute(OpKernelContext* context) override {
170 const Tensor& x_input = context->input(0);
171 const Tensor& y_input = context->input(1);
172 OP_REQUIRES(
173 context, x_input.shape() == y_input.shape(),
174 errors::InvalidArgument("x and y must be of the same shape. ",
175 "x shape: ", x_input.shape().DebugString(),
176 ". y shape: ", y_input.shape().DebugString()));
177 Tensor* z_output = nullptr;
178 OP_REQUIRES_OK(context,
179 context->allocate_output(0, x_input.shape(), &z_output));
180 const Device& d = context->eigen_device<Device>();
181 typename TTypes<T>::ConstFlat x(x_input.flat<T>());
182 typename TTypes<T>::ConstFlat y(y_input.flat<T>());
183 typename TTypes<bool>::Flat z(z_output->flat<bool>());
184 functor::ApproximateEqual<Device, T>()(d, x, y, tolerance_, z);
185 }
186
187 private:
188 T tolerance_;
189};

191// Basic coefficient-wise binary operations that are known to not require
192// any broadcasting. This is the case for example of the gradients of
193// unary operations.
194// Device: E.g., CPUDevice, GPUDevice.
195// Functor: defined above. E.g., functor::tanh_grad.
196template <typename Device, typename Functor>
197class SimpleBinaryOp : public OpKernel {
198 public:
199 typedef typename Functor::in_type Tin; // Input scalar data type.
200 typedef typename Functor::out_type Tout; // Output scalar data type.
201
202 explicit SimpleBinaryOp(OpKernelConstruction* ctx) : OpKernel(ctx) {}
203
204 void Compute(OpKernelContext* ctx) override {
205 const Tensor& in0 = ctx->input(0);
206 const Tensor& in1 = ctx->input(1);
207 auto in0_flat = in0.flat<Tin>();
208 auto in1_flat = in1.flat<Tin>();
209 const Device& eigen_device = ctx->eigen_device<Device>();
210
211 Tensor* out = nullptr;
212 if (std::is_same<Tin, Tout>::value) {
213 OP_REQUIRES_OK(ctx, ctx->forward_input_or_allocate_output(
214 {0, 1}, 0, in0.shape(), &out));
215 } else {
216 OP_REQUIRES_OK(ctx, ctx->allocate_output(0, in0.shape(), &out));
217 }
218 auto out_flat = out->flat<Tout>();
219 functor::SimpleBinaryFunctor<Device, Functor>()(eigen_device, out_flat,
220 in0_flat, in1_flat);
221 }
222};

224// Coefficient-wise unary operations:
225// Device: E.g., CPUDevice, GPUDevice.
226// Functor: defined in cwise_ops.h. E.g., functor::sqrt.
227template <typename Device, typename Functor>
228class UnaryOp : public OpKernel {
229 public:
230 typedef typename Functor::in_type Tin; // Input scalar data type.
231 typedef typename Functor::out_type Tout; // Output scalar data type.
232 // Tin may be different from Tout. E.g., abs: complex64 -> float
233
234 explicit UnaryOp(OpKernelConstruction* ctx) : OpKernel(ctx) {
235 auto in = DataTypeToEnum<Tin>::v();
236 auto out = DataTypeToEnum<Tout>::v();
237 OP_REQUIRES_OK(ctx, ctx->MatchSignature({in}, {out}));
238 }
239
240 void Compute(OpKernelContext* ctx) override {
241 const Tensor& inp = ctx->input(0);
242 Tensor* out = nullptr;
243 if (std::is_same<Tin, Tout>::value) {
244 OP_REQUIRES_OK(ctx, ctx->forward_input_or_allocate_output(
245 {0}, 0, inp.shape(), &out));
246 } else {
247 OP_REQUIRES_OK(ctx, ctx->allocate_output(0, inp.shape(), &out));
248 }
249 functor::UnaryFunctor<Device, Functor>()(
250 ctx->eigen_device<Device>(), out->flat<Tout>(), inp.flat<Tin>());
251 }
252};

254template <typename Device, VariantUnaryOp OpEnum>
255class UnaryVariantOp : public OpKernel {
256 public:
257 explicit UnaryVariantOp(OpKernelConstruction* ctx) : OpKernel(ctx) {}
258
259 void Compute(OpKernelContext* ctx) override {
260 const Tensor& inp = ctx->input(0);
261 OP_REQUIRES(
262 ctx, TensorShapeUtils::IsScalar(inp.shape()),
263 errors::InvalidArgument("Non-scalar variants are not supported."));
264 const Variant& v = inp.scalar<Variant>()();
265 Variant v_out;
266 OP_REQUIRES_OK(ctx, UnaryOpVariant<Device>(ctx, OpEnum, v, &v_out));
267 Tensor out(cpu_allocator(), DT_VARIANT, TensorShape());
268 out.scalar<Variant>()() = std::move(v_out);
269 ctx->set_output(0, std::move(out));
270 }
271};

280// Partial specialization of BinaryFunctor<Device=CPUDevice, Functor, NDIMS>
281// for functors with with no error checking.
282template <typename Functor, int NDIMS>
283struct BinaryFunctor<CPUDevice, Functor, NDIMS, false> {
284 void operator()(const CPUDevice& d, typename Functor::tout_type out,
285 typename Functor::tin_type in0,
286 typename Functor::tin_type in1, bool* error) {
287 Assign(d, out, in0.binaryExpr(in1, typename Functor::func()));
288 }
289
290 void Left(const CPUDevice& d, typename Functor::tout_type out,
291 typename Functor::tscalar_type scalar,
292 typename Functor::tin_type in, bool* error) {
293 typedef typename Functor::out_type Tout;
294 typedef typename Functor::in_type Tin;
295 typedef typename Functor::func Binary;
296 typedef typename Eigen::internal::scalar_left<Tout, Tin, Binary> Unary;
297 Assign(d, out, in.unaryExpr(Unary(scalar.data())));
298 }
299
300 void Right(const CPUDevice& d, typename Functor::tout_type out,
301 typename Functor::tin_type in,
302 typename Functor::tscalar_type scalar, bool* error) {
303 typedef typename Functor::out_type Tout;
304 typedef typename Functor::in_type Tin;
305 typedef typename Functor::func Binary;
306 typedef typename Eigen::internal::scalar_right<Tout, Tin, Binary> Unary;
307 Assign(d, out, in.unaryExpr(Unary(scalar.data())));
308 }
309
310 void BCast(const CPUDevice& dev,
311 typename TTypes<typename Functor::out_type, NDIMS>::Tensor out,
312 typename TTypes<typename Functor::in_type, NDIMS>::ConstTensor in0,
313 typename Eigen::array<Eigen::DenseIndex, NDIMS> bcast0,
314 typename TTypes<typename Functor::in_type, NDIMS>::ConstTensor in1,
315 typename Eigen::array<Eigen::DenseIndex, NDIMS> bcast1,
316 bool* error) {
317 typename Functor::func func;
318 if (AllOne<NDIMS>(bcast0) && AllOne<NDIMS>(bcast1)) {
319 Assign(dev, out, in0.binaryExpr(in1, func));
320 } else if (AllOne<NDIMS>(bcast0)) {
321 auto rhs = in1.broadcast(bcast1);
322 Assign(dev, out, in0.binaryExpr(rhs, func));
323 } else if (AllOne<NDIMS>(bcast1)) {
324 auto lhs = in0.broadcast(bcast0);
325 Assign(dev, out, lhs.binaryExpr(in1, func));
326 } else {
327 auto lhs = in0.broadcast(bcast0);
328 auto rhs = in1.broadcast(bcast1);
329 Assign(dev, out, lhs.binaryExpr(rhs, func));
330 }
331 }
332};

366#if !defined(EIGEN_HAS_INDEX_LIST)
367 inline Eigen::DSizes<int, 2> NByOne(int n) {
368 return Eigen::DSizes<int, 2>(n, 1);
369 }
370 inline Eigen::DSizes<int, 2> OneByM(int m) {
371 return Eigen::DSizes<int, 2>(1, m);
372 }
373#else
374 inline Eigen::IndexList<int, Eigen::type2index<1>> NByOne(int n) {
375 Eigen::IndexList<int, Eigen::type2index<1>> ret;
376 ret.set(0, n);
377 return ret;
378 }
379 inline Eigen::IndexList<Eigen::type2index<1>, int> OneByM(int m) {
380 Eigen::IndexList<Eigen::type2index<1>, int> ret;
381 ret.set(1, m);
382 return ret;
383 }
384#endif
385
386 void BCast(const CPUDevice& dev,
387 typename TTypes<typename Functor::out_type, NDIMS>::Tensor out,
388 typename TTypes<typename Functor::in_type, NDIMS>::ConstTensor in0,
389 typename Eigen::array<Eigen::DenseIndex, NDIMS> bcast0,
390 typename TTypes<typename Functor::in_type, NDIMS>::ConstTensor in1,
391 typename Eigen::array<Eigen::DenseIndex, NDIMS> bcast1,
392 bool* error) {
393 typedef typename Functor::in_type T;
394 typename Functor::func func;
395 if (Functor::use_bcast_optimization && use_bcast_optimization<T>::value) {
396 // Optimize for speed by using Eigen::type2index and avoid
397 // .broadcast() when we know its a no-op.
398 //
399 // Here, we need to handle 6 cases depending on how many "1"
400 // exist in in0 and in1's shapes (4 numbers in total). It's not
401 // possible that two shapes have more than 2 1s because those
402 // are simplified to NDIMS==1 case.
403 //
404 // Because this optimization increases the binary size for each
405 // Functor (+, -, *, /, <, <=, etc.), type and ndim combination.
406 // we only apply such optimization for selected ops/types/ndims.
407 //
408 // Because NDIMS, Functor::use_broadcast_optimization and
409 // use_broadcast_optimization<T> are compile-time constant, gcc
410 // does a decent job avoiding generating code when conditions
411 // are not met.
412 const int a = in0.dimension(0); // in0 is shape [a, b]
413 const int b = in0.dimension(1);
414 const int c = in1.dimension(0); // in1 is shape [c, d]
415 const int d = in1.dimension(1);
416 if ((a == 1) && (d == 1)) {
417 auto lhs = in0.reshape(OneByM(b)).broadcast(NByOne(c));
418 auto rhs = in1.reshape(NByOne(c)).broadcast(OneByM(b));
419 Assign(dev, out, lhs.binaryExpr(rhs, func));
420 return;
421 }
422 if ((b == 1) && (c == 1)) {
423 auto lhs = in0.reshape(NByOne(a)).broadcast(OneByM(d));
424 auto rhs = in1.reshape(OneByM(d)).broadcast(NByOne(a));
425 Assign(dev, out, lhs.binaryExpr(rhs, func));
426 return;
427 }
428 if (a == 1) {
429 auto lhs = in0.reshape(OneByM(b)).broadcast(NByOne(c));
430 auto rhs = in1;
431 Assign(dev, out, lhs.binaryExpr(rhs, func));
432 return;
433 }
434 if (b == 1) {
435 auto lhs = in0.reshape(NByOne(a)).broadcast(OneByM(d));
436 auto rhs = in1;
437 Assign(dev, out, lhs.binaryExpr(rhs, func));
438 return;
439 }
440 if (c == 1) {
441 auto lhs = in0;
442 auto rhs = in1.reshape(OneByM(d)).broadcast(NByOne(a));
443 Assign(dev, out, lhs.binaryExpr(rhs, func));
444 return;
445 }
446 if (d == 1) {
447 auto lhs = in0;
448 auto rhs = in1.reshape(NByOne(c)).broadcast(OneByM(b));
449 Assign(dev, out, lhs.binaryExpr(rhs, func));
450 return;
451 }
452
453 const bool bcast0_all_one = AllOne<NDIMS>(bcast0);
454 const bool bcast1_all_one = AllOne<NDIMS>(bcast1);
455 if (bcast0_all_one && !bcast1_all_one) {
456 auto lhs = in0; // No need to do broadcast for in0
457 auto rhs = in1.broadcast(bcast1);
458 Assign(dev, out, lhs.binaryExpr(rhs, func));
459 return;
460 }
461
462 if (!bcast0_all_one && bcast1_all_one) {
463 auto lhs = in0.broadcast(bcast0);
464 auto rhs = in1; // No need to do broadcast for in1
465 Assign(dev, out, lhs.binaryExpr(rhs, func));
466 return;
467 }
468 }
469
470 // Fallback path. Always works and probably slower.
471 auto lhs = in0.broadcast(bcast0);
472 auto rhs = in1.broadcast(bcast1);
473 Assign(dev, out, lhs.binaryExpr(rhs, func));
474 }
475};

555#if defined(__ANDROID_TYPES_SLIM__)

560// code size.
561#define REGISTER2(OP, D, N, F, T0, T1) REGISTER(OP, D, N, F, T0)
562#define REGISTER3(OP, D, N, F, T0, T1, T2) REGISTER(OP, D, N, F, T0)

564#define REGISTER5(OP, D, N, F, T0, T1, T2, T3, T4) REGISTER(OP, D, N, F, T0)
