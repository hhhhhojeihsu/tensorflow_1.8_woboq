40class ArgOp : public OpKernel {
41 public:
42 explicit ArgOp(OpKernelConstruction* ctx) : OpKernel(ctx) {
43 OP_REQUIRES_OK(ctx, ctx->GetAttr("T", &dtype_));
44 OP_REQUIRES_OK(ctx, ctx->GetAttr("index", &index_));
45 }
46
47 void Compute(OpKernelContext* ctx) override {
48 auto frame = ctx->call_frame();
49 OP_REQUIRES(ctx, frame != nullptr, errors::Internal("no call frame"));
50 Tensor val;
51 OP_REQUIRES_OK(ctx, frame->GetArg(index_, &val));
52 OP_REQUIRES(ctx, val.dtype() == dtype_,
53 errors::InvalidArgument(
54 "Type mismatch: actual ", DataTypeString(val.dtype()),
55 " vs. expect ", DataTypeString(dtype_)));
56 ctx->set_output(0, val);
57 }
58
59 bool IsExpensive() override { return false; }
60
61 private:
62 int index_;
63 DataType dtype_;
64
65 TF_DISALLOW_COPY_AND_ASSIGN(ArgOp);
66};

68class RetvalOp : public OpKernel {
69 public:
70 explicit RetvalOp(OpKernelConstruction* ctx) : OpKernel(ctx) {
71 OP_REQUIRES_OK(ctx, ctx->GetAttr("T", &dtype_));
72 OP_REQUIRES_OK(ctx, ctx->GetAttr("index", &index_));
73 }
74
75 void Compute(OpKernelContext* ctx) override {
76 const Tensor& val = ctx->input(0);
77 OP_REQUIRES(ctx, val.dtype() == dtype_,
78 errors::InvalidArgument(
79 "Type mismatch: actual ", DataTypeString(val.dtype()),
80 " vs. expect ", DataTypeString(dtype_)));
81 auto frame = ctx->call_frame();
82 OP_REQUIRES(ctx, frame != nullptr, errors::Internal("no call frame"));
83 OP_REQUIRES_OK(ctx, frame->SetRetval(index_, val));
84 }
85
86 bool IsExpensive() override { return false; }
87
88 private:
89 int index_;
90 DataType dtype_;
91
92 TF_DISALLOW_COPY_AND_ASSIGN(RetvalOp);
93};

154class PassOn : public OpKernel {
155 public:
156 explicit PassOn(OpKernelConstruction* ctx) : OpKernel(ctx) {
157 OP_REQUIRES(ctx, ctx->num_inputs() == ctx->num_outputs(),
158 errors::Internal("#inputs != #outputs : ", ctx->num_inputs(),
159 " vs. ", ctx->num_outputs()));
160 for (int i = 0; i < ctx->num_inputs(); ++i) {
161 OP_REQUIRES(
162 ctx, input_type(i) == output_type(i),
163 errors::Internal("Input and output types for position ", i,
164 " do not match: ", DataTypeString(input_type(i)),
165 " vs. ", DataTypeString(output_type(i))));
166 }
167 }
168
169 void Compute(OpKernelContext* ctx) override {
170 for (int i = 0; i < ctx->num_inputs(); ++i) {
171 ctx->set_output(i, ctx->input(i));
172 }
173 }
174};

234class SymbolicGradientOp : public AsyncOpKernel {
235 public:
236 explicit SymbolicGradientOp(OpKernelConstruction* ctx) : AsyncOpKernel(ctx) {}
237
238 ~SymbolicGradientOp() override {}
239
240 void ComputeAsync(OpKernelContext* ctx, DoneCallback done) override {
241 FunctionLibraryRuntime* lib = ctx->function_library();
242 OP_REQUIRES_ASYNC(ctx, lib != nullptr,
243 errors::Internal("No function library is provided."),
244 done);
245
246 FunctionLibraryRuntime::Handle handle;
247 OP_REQUIRES_OK_ASYNC(
248 ctx, lib->Instantiate(kGradientOp, AttrSlice(def()), &handle), done);
249
250 FunctionLibraryRuntime::Options opts;
251 opts.step_id = ctx->step_id();
252 opts.rendezvous = ctx->rendezvous();
253 opts.cancellation_manager = ctx->cancellation_manager();
254 opts.runner = ctx->runner();
255 opts.stats_collector = ctx->stats_collector();
256 opts.step_container = ctx->step_container();
257 std::vector<Tensor> args;
258 args.reserve(ctx->num_inputs());
259 for (int i = 0; i < ctx->num_inputs(); ++i) {
260 args.push_back(ctx->input(i));
261 }
262 std::vector<Tensor>* rets = new std::vector<Tensor>;
263 lib->Run(opts, handle, args, rets, [ctx, done, rets](const Status& status) {
264 if (!status.ok()) {
265 ctx->SetStatus(status);
266 } else if (rets->size() != ctx->num_outputs()) {
267 ctx->SetStatus(errors::InvalidArgument(
268 "SymGrad expects to return ", ctx->num_outputs(),
269 " tensor(s), but get ", rets->size(), " tensor(s) instead."));
270 } else {
271 for (size_t i = 0; i < rets->size(); ++i) {
272 ctx->set_output(i, (*rets)[i]);
273 }
274 }
275 delete rets;
276 done();
277 });
278 }
279
280 private:
281 TF_DISALLOW_COPY_AND_ASSIGN(SymbolicGradientOp);
282};

294class RemoteCallOp : public AsyncOpKernel {
295 public:
296 explicit RemoteCallOp(OpKernelConstruction* ctx) : AsyncOpKernel(ctx) {
297 OP_REQUIRES_OK(ctx,
298 ctx->GetAttr(FunctionLibraryDefinition::kFuncAttr, &func_));
299 }
300
301 ~RemoteCallOp() override {}
302
303 void ComputeAsync(OpKernelContext* ctx, DoneCallback done) override {
304 const Tensor* target;
305 OP_REQUIRES_OK_ASYNC(ctx, ctx->input("target", &target), done);
306 const string& target_device =
307 DeviceNameUtils::CanonicalizeDeviceName(target->scalar<string>()());
308
309 FunctionLibraryRuntime* lib = ctx->function_library();
310 OP_REQUIRES_ASYNC(ctx, lib != nullptr,
311 errors::Internal("No function library is provided."),
312 done);
313 AttrValueMap attr_values = func_.attr();
314 FunctionLibraryRuntime::InstantiateOptions instantiate_opts;
315 instantiate_opts.target = target_device;
316
317 FunctionTarget function_target = {target_device, lib};
318
319 FunctionLibraryRuntime::Handle handle;
320 {
321 mutex_lock l(mu_);
322 auto cached_entry = handle_cache_.find(function_target);
323 if (cached_entry != handle_cache_.end()) {
324 handle = cached_entry->second;
325 } else {
326 VLOG(1) << "Instantiating " << func_.name() << " on " << target_device;
327 port::Tracing::TraceMe activity(strings::StrCat(
328 "RemoteCall: Instantiate: ", func_.name(), " on ", target_device));
329 OP_REQUIRES_OK_ASYNC(
330 ctx,
331 lib->Instantiate(func_.name(), AttrSlice(&attr_values),
332 instantiate_opts, &handle),
333 done);
334 auto insert_result = handle_cache_.insert({function_target, handle});
335 CHECK(insert_result.second) << "Insert unsuccessful.";
336 VLOG(1) << "Instantiated " << func_.name() << " on " << target_device
337 << ", resulting in handle: " << handle << " flr: " << lib;
338 }
339 }
340
341 OpInputList arguments;
342 OP_REQUIRES_OK_ASYNC(ctx, ctx->input_list("args", &arguments), done);
343
344 FunctionLibraryRuntime::Options opts;
345 opts.step_id = ctx->step_id();
346 opts.runner = ctx->runner();
347 opts.source_device = lib->device()->name();
348 if (opts.source_device != target_device) {
349 opts.remote_execution = true;
350 }
351 opts.create_rendezvous = true;
352 std::vector<Tensor> args;
353 args.reserve(arguments.size());
354 for (const Tensor& argument : arguments) {
355 args.push_back(argument);
356 }
357 auto* rets = new std::vector<Tensor>;
358 auto* trace = new port::Tracing::TraceMe(strings::StrCat(
359 "RemoteCall: Run: ", func_.name(), " on ", target_device));
360 VLOG(1) << "Running " << func_.name() << " on " << target_device
361 << " with handle: " << handle;
362 lib->Run(opts, handle, args, rets,
363 [rets, trace, done, ctx](const Status& status) {
364 if (!status.ok()) {
365 ctx->SetStatus(status);
366 } else {
367 for (size_t i = 0; i < rets->size(); ++i) {
368 ctx->set_output(i, (*rets)[i]);
369 }
370 }
371 delete rets;
372 delete trace;
373 done();
374 });
375 }
376
377 private:
378 NameAttrList func_;
379
380 mutex mu_;
381 typedef std::pair<string, FunctionLibraryRuntime*> FunctionTarget;
382 std::map<FunctionTarget, FunctionLibraryRuntime::Handle> handle_cache_
383 GUARDED_BY(mu_);
384
385 TF_DISALLOW_COPY_AND_ASSIGN(RemoteCallOp);
386};
