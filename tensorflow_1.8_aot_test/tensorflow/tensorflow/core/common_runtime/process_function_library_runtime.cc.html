28Status ProcessFunctionLibraryRuntime::FunctionData::DistributedInit(
29 DistributedFunctionLibraryRuntime* parent, const string& function_name,
30 const FunctionLibraryDefinition& lib_def, AttrSlice attrs,
31 const FunctionLibraryRuntime::InstantiateOptions& options) {
32 mutex_lock l(mu_);
33 if (!init_started_) {
34 init_started_ = true;
35 init_result_ = parent->Instantiate(function_name, lib_def, attrs, options,
36 &local_handle_);
37 }
38 return init_result_;
39}

41ProcessFunctionLibraryRuntime::ProcessFunctionLibraryRuntime(
42 const DeviceMgr* device_mgr, Env* env, int graph_def_version,
43 const FunctionLibraryDefinition* lib_def,
44 const OptimizerOptions& optimizer_options,
45 thread::ThreadPool* default_thread_pool,
46 DistributedFunctionLibraryRuntime* parent)
47 : device_mgr_(device_mgr),
48 lib_def_(lib_def),
49 default_thread_pool_(default_thread_pool),
50 next_handle_(0),
51 parent_(parent) {
52 if (device_mgr == nullptr) {
53 flr_map_[nullptr] = NewFunctionLibraryRuntime(
54 nullptr, env, nullptr, graph_def_version, lib_def, default_thread_pool,
55 optimizer_options, this);
56 return;
57 }
58 for (Device* d : device_mgr->ListDevices()) {
59 flr_map_[d] = NewFunctionLibraryRuntime(
60 device_mgr, env, d, graph_def_version, lib_def, default_thread_pool,
61 optimizer_options, this);
62 }
63}

65ProcessFunctionLibraryRuntime::ProcessFunctionLibraryRuntime(
66 const DeviceMgr* device_mgr, Env* env, int graph_def_version,
67 const FunctionLibraryDefinition* lib_def,
68 const OptimizerOptions& optimizer_options,
69 CustomKernelCreator custom_kernel_creator,
70 thread::ThreadPool* default_thread_pool,
71 DistributedFunctionLibraryRuntime* parent)
72 : device_mgr_(device_mgr),
73 lib_def_(lib_def),
74 default_thread_pool_(default_thread_pool),
75 next_handle_(0),
76 parent_(parent) {
77 if (device_mgr == nullptr) {
78 flr_map_[nullptr] = NewFunctionLibraryRuntime(
79 nullptr, env, nullptr, graph_def_version, lib_def, default_thread_pool,
80 optimizer_options, std::move(custom_kernel_creator), this);
81 return;
82 }

84 flr_map_[d] = NewFunctionLibraryRuntime(
85 device_mgr, env, d, graph_def_version, lib_def, default_thread_pool,
86 optimizer_options, custom_kernel_creator, this);
87 }
88}

90/* static */
91Status ProcessFunctionLibraryRuntime::SendTensors(
92 const string& source_device, const string& target_device,
93 const string& key_prefix, int64 src_incarnation,
94 gtl::ArraySlice<Tensor> tensors_to_send, DeviceContext* device_context,
95 const std::vector<AllocatorAttributes>& alloc_attrs,
96 Rendezvous* rendezvous) {
97 std::vector<string> keys;
98 for (int i = 0; i < tensors_to_send.size(); ++i) {
99 string name = strings::StrCat(key_prefix, i);
100 string key = Rendezvous::CreateKey(source_device, src_incarnation,
101 target_device, name, FrameAndIter(0, 0));
102 keys.push_back(key);
103 }
104 TF_RETURN_IF_ERROR(SendTensorsToRendezvous(
105 rendezvous, device_context, alloc_attrs, keys, tensors_to_send));
106 return Status::OK();
107}

109/* static */
110void ProcessFunctionLibraryRuntime::ReceiveTensorsAsync(
111 const string& source_device, const string& target_device,
112 const string& key_prefix, int64 src_incarnation, int64 num_tensors,
113 DeviceContext* device_context,
114 const std::vector<AllocatorAttributes>& alloc_attrs, Rendezvous* rendezvous,
115 std::vector<Tensor>* received_tensors, const StatusCallback& done) {
116 std::vector<string> keys;
117 for (int64 i = 0; i < num_tensors; ++i) {
118 string name = strings::StrCat(key_prefix, i);
119 string key = Rendezvous::CreateKey(source_device, src_incarnation,
120 target_device, name, FrameAndIter(0, 0));
121 keys.push_back(key);
122 }
123 RecvOutputsFromRendezvousAsync(
124 rendezvous, device_context, alloc_attrs, keys, received_tensors,
125 [done](const Status& status) { done(status); });
126}

128Status ProcessFunctionLibraryRuntime::GetDeviceIncarnation(
129 const string& device_name, int64* incarnation) {
130 FunctionLibraryRuntime* flr = GetFLR(device_name);
131 if (flr == nullptr) {
132 return errors::InvalidArgument("Device name: ", device_name, " not found");
133 }
134 *incarnation = flr->device()->attributes().incarnation();
135 return Status::OK();
136}

138Status ProcessFunctionLibraryRuntime::GetDeviceContext(
139 const string& device_name, DeviceContext** device_context) {
140 *device_context = nullptr;
141 FunctionLibraryRuntime* flr = GetFLR(device_name);
142 if (flr == nullptr) {
143 return errors::InvalidArgument("Device name: ", device_name, " not found.");
144 }
145 Device* device = flr->device();
146 string device_type = device->parsed_name().type;
147 if (device_type == "CPU") return Status::OK();
148 if (device_type == "GPU") {
149 auto* dev_info = flr->device()->tensorflow_gpu_device_info();
150 if (dev_info) {
151 *device_context = dev_info->default_context;
152 return Status::OK();
153 }
154 }
155 return errors::Internal("Device type: ", device_type,
156 " is currently unsupported for remote ",
157 "function executions");
158}
159
160FunctionLibraryRuntime* ProcessFunctionLibraryRuntime::GetFLR(
161 const string& device_name) const {
162 Device* device = nullptr;
163 if (device_name != kDefaultFLRDevice) {
164 if (!device_mgr_->LookupDevice(device_name, &device).ok()) {
165 VLOG(1) << "Could not find device: " << device_name;
166 return nullptr;
167 }
168 }
169 const auto& iter = flr_map_.find(device);
170 if (iter == flr_map_.end()) {
171 LOG(ERROR) << "Could not find device: " << device_name;
172 return nullptr;
173 }
174 return iter->second.get();
175}

177FunctionLibraryRuntime::Handle ProcessFunctionLibraryRuntime::AddHandle(
178 const string& function_key, const string& device_name,
179 FunctionLibraryRuntime::LocalHandle local_handle) {
180 mutex_lock l(mu_);
181 FunctionLibraryRuntime::Handle h =
182 gtl::FindWithDefault(table_, function_key, kInvalidHandle);
183 if (h != kInvalidHandle) {
184 if (function_data_.count(h) != 0) return h;
185 }
186 h = next_handle_;
187 FunctionData* fd = new FunctionData(device_name, local_handle);
188 function_data_[h] = std::unique_ptr<FunctionData>(fd);
189 table_[function_key] = h;
190 next_handle_++;
191 return h;
192}
193
194FunctionLibraryRuntime::Handle ProcessFunctionLibraryRuntime::GetHandle(
195 const string& function_key) const {
196 mutex_lock l(mu_);
197 FunctionLibraryRuntime::Handle h =
198 gtl::FindWithDefault(table_, function_key, kInvalidHandle);
199 if (h != kInvalidHandle) {
200 if (function_data_.count(h) == 0) return kInvalidHandle;
201 }
202 return h;
203}

205bool ProcessFunctionLibraryRuntime::IsInstantiatedOnDevice(
206 const string& device_name, FunctionLibraryRuntime::Handle handle) {
207 return GetHandleOnDevice(device_name, handle) != kInvalidHandle;
208}
209
210FunctionLibraryRuntime::LocalHandle
211ProcessFunctionLibraryRuntime::GetHandleOnDevice(
212 const string& device_name, FunctionLibraryRuntime::Handle handle) {
213 mutex_lock l(mu_);
214 if (function_data_.count(handle) == 0) {
215 return kInvalidLocalHandle;
216 }
217 FunctionData* function_data = function_data_[handle].get();
218 if (function_data->target_device() != device_name) {
219 return kInvalidLocalHandle;
220 }
221 return function_data->local_handle();
222}

224string ProcessFunctionLibraryRuntime::GetDeviceName(
225 FunctionLibraryRuntime::Handle handle) {
226 mutex_lock l(mu_);
227 CHECK_EQ(1, function_data_.count(handle));
228 FunctionData* function_data = function_data_[handle].get();
229 return function_data->target_device();
230}

232Status ProcessFunctionLibraryRuntime::Instantiate(
233 const string& function_name, AttrSlice attrs,
234 const FunctionLibraryRuntime::InstantiateOptions& options,
235 FunctionLibraryRuntime::Handle* handle) {
236 *handle = kInvalidHandle;
237 FunctionLibraryRuntime* flr = GetFLR(options.target);
238 if (flr != nullptr) {
239 return flr->Instantiate(function_name, attrs, options, handle);
240 }
241 if (parent_ == nullptr) {
242 return errors::Internal(
243 "Currently don't support instantiating functions on device: ",
244 options.target);
245 }
246 VLOG(1) << "ProcessFLR Instantiate: " << function_name
247 << " on: " << options.target;
248 string function_key = Canonicalize(function_name, attrs, options);
249 FunctionData* f;
250 {
251 mutex_lock l(mu_);
252 FunctionLibraryRuntime::Handle h =
253 gtl::FindWithDefault(table_, function_key, kInvalidHandle);
254 if (h == kInvalidHandle || function_data_.count(h) == 0) {
255 h = next_handle_;
256 FunctionData* fd = new FunctionData(options.target, kInvalidHandle);
257 function_data_[h] = std::unique_ptr<FunctionData>(fd);
258 table_[function_key] = h;
259 next_handle_++;
260 }
261 f = function_data_[h].get();
262 *handle = h;
263 }
264 TF_RETURN_IF_ERROR(
265 f->DistributedInit(parent_, function_name, *lib_def_, attrs, options));
266 VLOG(1) << "ProcessFLR Instantiate [success]: " << function_name
267 << " on: " << options.target << " with handle: " << *handle
268 << " (this: " << this << ")";
269 return Status::OK();
270}

272Status ProcessFunctionLibraryRuntime::RemoveHandle(
273 FunctionLibraryRuntime::Handle handle) {
274 mutex_lock l(mu_);
275 function_data_.erase(handle);
276 return Status::OK();
277}

279Status ProcessFunctionLibraryRuntime::ReleaseHandle(
280 FunctionLibraryRuntime::Handle handle) {
281 FunctionLibraryRuntime* flr = nullptr;
282 string target_device;
283 {
284 mutex_lock l(mu_);
285 CHECK_EQ(1, function_data_.count(handle)) << " handle: " << handle;
286 target_device = function_data_[handle]->target_device();
287 }
288 flr = GetFLR(target_device);
289 if (flr != nullptr) {
290 return flr->ReleaseHandle(handle);
291 }
292 return errors::InvalidArgument("Handle not found: ", handle);
293}

295void ProcessFunctionLibraryRuntime::Run(
296 const FunctionLibraryRuntime::Options& opts,
297 FunctionLibraryRuntime::Handle handle, gtl::ArraySlice<Tensor> args,
298 std::vector<Tensor>* rets, FunctionLibraryRuntime::DoneCallback done) {
299 if (!opts.remote_execution) {
300 done(errors::InvalidArgument(
301 "ProcessFunctionLibraryRuntime::Run should only be called when there ",
302 "is a remote execution."));
303 return;
304 }
305
306 FunctionLibraryRuntime* flr = nullptr;
307 string target_device;
308 FunctionLibraryRuntime::LocalHandle local_handle;
309 {
310 mutex_lock l(mu_);
311 if (function_data_.count(handle) == 0) {
312 done(errors::NotFound("Handle: ", handle, " not found."));
313 return;
314 }
315 target_device = function_data_[handle]->target_device();
316 local_handle = function_data_[handle]->local_handle();
317 }
318 flr = GetFLR(target_device);
319 if (flr != nullptr) {
320 auto rendezvous = opts.rendezvous;
321 string source_device = opts.source_device;
322 DeviceContext* device_context;
323 Status s = GetDeviceContext(source_device, &device_context);
324 if (!s.ok()) {
325 done(s);
326 return;
327 }
328 int64 src_incarnation, target_incarnation;
329 s = GetDeviceIncarnation(source_device, &src_incarnation);
330 s.Update(GetDeviceIncarnation(target_device, &target_incarnation));
331 if (!s.ok()) {
332 done(s);
333 return;
334 }
335
336 // Send the args over to the target device.
337 s = SendTensors(source_device, target_device, "arg_", src_incarnation, args,
338 device_context, opts.args_alloc_attrs, rendezvous);
339 if (!s.ok()) {
340 done(s);
341 return;
342 }
343 const std::vector<AllocatorAttributes>& rets_alloc_attrs =
344 opts.rets_alloc_attrs;
345 std::vector<Tensor>* remote_rets = new std::vector<Tensor>;
346 flr->Run(opts, handle, args, remote_rets,
347 [source_device, target_device, target_incarnation, rendezvous,
348 device_context, rets_alloc_attrs, remote_rets, rets,
349 done](const Status& status) {
350 if (!status.ok()) {
351 delete remote_rets;
352 done(status);
353 return;
354 }
355 int64 num_returns = remote_rets->size();
356 delete remote_rets;
357 // Now receive the return values from the target.
358 ReceiveTensorsAsync(target_device, source_device, "ret_",
359 target_incarnation, num_returns,
360 device_context, rets_alloc_attrs, rendezvous,
361 rets, done);
362 });
363 return;
364 }
365 if (parent_ != nullptr) {
366 parent_->Run(opts, local_handle, args, rets, done);
367 return;
368 }
369 done(errors::Internal("Could not find device"));
370}

372Status ProcessFunctionLibraryRuntime::Clone(
373 Env* env, int graph_def_version, const OptimizerOptions& optimizer_options,
374 CustomKernelCreator custom_kernel_creator,
375 std::unique_ptr<FunctionLibraryDefinition>* out_lib_def,
376 std::unique_ptr<ProcessFunctionLibraryRuntime>* out_pflr) {
377 out_lib_def->reset(new FunctionLibraryDefinition(*lib_def_));
378 out_pflr->reset(new ProcessFunctionLibraryRuntime(
379 device_mgr_, env, graph_def_version, out_lib_def->get(),
380 optimizer_options, std::move(custom_kernel_creator), default_thread_pool_,
381 parent_));
382 return Status::OK();
383}
