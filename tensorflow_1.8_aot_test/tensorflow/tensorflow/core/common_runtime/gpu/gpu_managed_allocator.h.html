21namespace tensorflow {
22
23// An allocator for CUDA unified memory. Memory allocated with this allocator
24// can be accessed from both host and device. CUDA transparently migrates dirty
25// pages, which can be slow. Therefore, this allocator is intended for
26// convenience in functional tests only.
27class GpuManagedAllocator : public Allocator {
28 public:
29 string Name() override { return "GpuManagedAllocator"; }
30 void* AllocateRaw(size_t alignment, size_t num_bytes) override;
31 void DeallocateRaw(void* ptr) override;
32};
33
34} // namespace tensorflow
