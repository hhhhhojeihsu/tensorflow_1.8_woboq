27GraphOptimizer::GraphOptimizer(const OptimizerOptions& opts) : opts_(opts) {
28 if (opts_.opt_level() >= OptimizerOptions::L1) {
29 opts_.set_do_common_subexpression_elimination(true);
30 opts_.set_do_constant_folding(true);
31 }
32}
33
34GraphOptimizer::~GraphOptimizer() {}
35
36void GraphOptimizer::Optimize(
37 FunctionLibraryRuntime* runtime, Env* env, Device* device,
38 std::unique_ptr<Graph>* graph,
39 const std::unordered_map<string, std::vector<PartialTensorShape>>*
40 shape_map,
41 const std::function<bool(const Node*)>& cse_consider_fn) {
42 Graph* g = graph->get();
43 DumpGraph("Initial", g);
44
45 bool changed = true;
46 const int kMaxRounds = 10;
47 for (int rounds = 0; rounds < kMaxRounds; ++rounds) {
48 changed = false;
49 if (RemoveListArrayConverter(g)) {
50 DumpGraph("RemoveListArrayConverter", g);
51 changed = true;
52 }
53 if (opts_.do_function_inlining() && RemoveDeadNodes(g)) {
54 DumpGraph("RemoveDeadNodes", g);
55 changed = true;
56 }
57 if (opts_.do_function_inlining() && RemoveIdentityNodes(g)) {
58 DumpGraph("RemoveIdentityNodes", g);
59 changed = true;
60 }
61
62 if (opts_.do_constant_folding()) {
63 ConstantFoldingOptions cf_opts;
64 cf_opts.shape_map = shape_map;
65 if (opts_.max_folded_constant_in_bytes() > 0) {
66 cf_opts.max_constant_size_in_bytes =
67 opts_.max_folded_constant_in_bytes();
68 }
69 bool was_mutated;
70 ConstantFold(cf_opts, runtime, env, device, g, &was_mutated)
71 .IgnoreError();
72 if (was_mutated) {
73 RemoveDeadNodes(g);
74 DumpGraph("ConstFolding", g);
75 changed = true;
76 }
77 }
78
79 if (opts_.do_function_inlining() && FixupSourceAndSinkEdges(g)) {
80 DumpGraph("FixupSourceAndSinkEdges", g);
81 changed = true;
82 }
83 if (opts_.do_common_subexpression_elimination() &&
84 OptimizeCSE(g, cse_consider_fn)) {
85 DumpGraph("OptimizeCSE", g);
86 changed = true;
87 }
88 if (opts_.do_function_inlining() && ExpandInlineFunctions(runtime, g)) {
89 DumpGraph("ExpandInlineFunctions", g);
90 changed = true;
91 }
92 if (!changed) break;
93 }
94
95 // Note that we use the Graph constructor that copies the input
96 // FunctionLibraryDefinition, since the original lib def will go out of scope.
97 std::unique_ptr<Graph> copy(new Graph(g->flib_def()));
98 CopyGraph(*g, copy.get());
99 graph->swap(copy);
100
101 DumpGraph("ReCopy", graph->get());
102}
