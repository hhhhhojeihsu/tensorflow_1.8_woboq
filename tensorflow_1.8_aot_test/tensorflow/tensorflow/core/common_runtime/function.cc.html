54// Represents the index-th output of a node.
55struct Endpoint {
56 Node* node;
57 int index;
58
59 // Returns the string name represents this endpoint.
60 string name() const {
61 if (index == 0) {
62 return node->name();
63 } else {
64 return strings::StrCat(node->name(), ":", index);
65 }
66 }
67
68 DataType dtype() const { return node->output_type(index); }
69};
70
71struct EndpointHash {
72 uint64 operator()(const Endpoint& x) const {
73 return Hash64(reinterpret_cast<const char*>(&x.node), sizeof(Node*),
74 x.index);
75 }
76};
77
78struct EndpointEq {
79 bool operator()(const Endpoint& x, const Endpoint& y) const {
80 return (x.node == y.node) && (x.index == y.index);
81 }
82};

96static Node* AddIdentity(Graph* g, Endpoint input) {
97 DCHECK_LT(0, input.dtype());
98 NodeDef ndef;
99 ndef.set_name(g->NewName(kNodeLabel));
100 ndef.set_op("Identity");
101 ndef.add_input(input.name());
102 AddNodeAttr("T", BaseType(input.dtype()), &ndef);
103 Status s;
104 Node* ret = g->AddNode(ndef, &s);
105 TF_CHECK_OK(s);
106 g->AddEdge(input.node, input.index, ret, 0);
107 return ret;
108}

110static Node* AddArg(Graph* g, DataType dtype, int index) {
111 DCHECK_LT(0, dtype);
112 DCHECK_LT(dtype, DT_FLOAT_REF);
113 NodeDef ndef;
114 ndef.set_name(g->NewName(kNodeLabel));
115 ndef.set_op(kArgOp);
116 AddNodeAttr("T", dtype, &ndef);
117 AddNodeAttr("index", index, &ndef);
118 Status s;
119 Node* ret = g->AddNode(ndef, &s);
120 TF_CHECK_OK(s);
121 return ret;
122}

124static Node* AddRet(Graph* g, Endpoint input, int index) {
125 DCHECK_LT(0, input.dtype());
126 DCHECK_LT(input.dtype(), DT_FLOAT_REF);
127 NodeDef ndef;
128 ndef.set_name(g->NewName(kNodeLabel));
129 ndef.set_op(kRetOp);
130 ndef.add_input(input.name());
131 AddNodeAttr("T", input.dtype(), &ndef);
132 AddNodeAttr("index", index, &ndef);
133 Status s;
134 Node* ret = g->AddNode(ndef, &s);
135 TF_CHECK_OK(s);
136 g->AddEdge(input.node, input.index, ret, 0);
137 return ret;

298// An asynchronous op kernel which executes an instantiated function
299// defined in a library.
300class CallOp : public AsyncOpKernel {
301 public:
302 CallOp(FunctionLibraryRuntime::Handle handle, OpKernelConstruction* ctx)
303 : AsyncOpKernel(ctx), handle_(handle) {}
304
305 ~CallOp() override {}
306
307 void ComputeAsync(OpKernelContext* ctx, DoneCallback done) override {
308 FunctionLibraryRuntime* lib = ctx->function_library();
309 OP_REQUIRES_ASYNC(ctx, lib != nullptr,
310 errors::Internal("No function library is provided."),
311 done);
312 FunctionLibraryRuntime::Options opts;
313 opts.step_id = ctx->step_id();
314 opts.rendezvous = ctx->rendezvous();
315 opts.cancellation_manager = ctx->cancellation_manager();
316 opts.step_container = ctx->step_container();
317 opts.stats_collector = ctx->stats_collector();
318 opts.runner = ctx->runner();
319 std::vector<Tensor> args;
320 args.reserve(ctx->num_inputs());
321 for (int i = 0; i < ctx->num_inputs(); ++i) {
322 args.push_back(ctx->input(i));
323 }
324 std::vector<Tensor>* rets = new std::vector<Tensor>;
325 lib->Run(opts, handle_, args, rets,
326 [ctx, done, rets](const Status& status) {
327 if (!status.ok()) {
328 ctx->SetStatus(status);
329 } else {
330 const int ret_size = static_cast<int>(rets->size());
331 CHECK_EQ(ret_size, ctx->num_outputs());
332 for (int i = 0; i < ret_size; ++i) {
333 ctx->set_output(i, (*rets)[i]);
334 }
335 }
336 delete rets;
337 done();
338 });
339 }
340
341 private:
342 FunctionLibraryRuntime::Handle handle_;
343
344 TF_DISALLOW_COPY_AND_ASSIGN(CallOp);
345};

365Status FunctionLibraryRuntimeImpl::CreateKernel(
366 const NodeDef& ndef, const FunctionLibraryDefinition* lib_def,
367 OpKernel** kernel) {
368 // If a custom kernel creator is given, try that.
369 Status s;
370 if (custom_kernel_creator_) {
371 std::unique_ptr<OpKernel> ret;
372 s = custom_kernel_creator_(this, ndef, &ret);
373 if (s.ok()) {
374 *kernel = ret.release();
375 return s;
376 } else {
377 VLOG(2) << "Custom creator error: " << s;
378 // Falls through.
379 s = Status::OK();
380 }
381 }
382
383 if (lib_def->Find(ndef.op()) == nullptr) {
384 // A primitive operation. Creates the registered kernel.
385 return CreateNonCachedKernel(device_, this, ndef, graph_def_version_,
386 kernel);
387 }
388
389 // Try to instantiate this function for the func/attr. Maybe it's
390 // cached already.
391 InstantiateOptions options;
392 if (lib_def != base_lib_def_) {
393 options.overlay_lib = lib_def;
394 }
395 Handle handle;
396 TF_RETURN_IF_ERROR(
397 Instantiate(ndef.op(), AttrSlice(&ndef.attr()), options, &handle));
398
399 const FunctionBody* fbody = GetFunctionBody(handle);
400 CHECK_NOTNULL(fbody);
401
402 // TODO(zhifengc): For now, we assume int32 and resources are always on host
403 // memory and other types are always on device memory. We should do type
404 // inference over function body to derive the correct input/output memory
405 // types.
406 MemoryTypeVector input_memory_types;
407 for (const auto& t : fbody->arg_types) {
408 input_memory_types.push_back(
409 (t == DT_INT32 || t == DT_RESOURCE) ? HOST_MEMORY : DEVICE_MEMORY);
410 }
411 MemoryTypeVector output_memory_types;
412 for (const auto& t : fbody->ret_types) {
413 output_memory_types.push_back(t == DT_INT32 ? HOST_MEMORY : DEVICE_MEMORY);
414 }
415
416 // Constructs a CallOp kernel for running the instantiated function.
417 auto device_type = DeviceType(device_->attributes().device_type());
418 OpKernelConstruction construction(
419 device_type, device_, device_->GetAllocator(AllocatorAttributes()), &ndef,
420 &fbody->fdef.signature(), this, fbody->arg_types, input_memory_types,
421 fbody->ret_types, output_memory_types, graph_def_version_, &s);
422 *kernel = new CallOp(handle, &construction);
423 if (!s.ok()) {
424 delete *kernel;
425 }
426 return s;
427}

442Status FunctionLibraryRuntimeImpl::InstantiateSymbolicGradient(
443 const NameAttrList& func, const FunctionLibraryDefinition* lib_def,
444 FunctionBody** g_body) {
445 const FunctionDef* fdef = lib_def->Find(func.name());
446 if (fdef == nullptr) {
447 // f is a primitive op.
448 gradient::Creator creator;
449 TF_RETURN_IF_ERROR(gradient::GetOpGradientCreator(func.name(), &creator));
450 if (creator == nullptr) {
451 return errors::InvalidArgument("No gradient is defined for ",
452 func.name());
453 }
454 FunctionDef grad_fdef;
455 // TODO(josh11b): Should filter out the attrs from func that aren't used
456 // by the gradient function.
457 TF_RETURN_IF_ERROR(creator(AttrSlice(&func.attr()), &grad_fdef));
458 TF_RETURN_IF_ERROR(
459 FunctionDefToBody(grad_fdef, AttrSlice(&func.attr()), lib_def, g_body));
460 } else {
461 // f is a user-defined function.
462 InstantiateOptions options;
463 if (lib_def != base_lib_def_) {
464 options.overlay_lib = lib_def;
465 }
466 Handle f_handle;
467 TF_RETURN_IF_ERROR(
468 Instantiate(func.name(), AttrSlice(&func.attr()), options, &f_handle));
469 const FunctionBody* f_body = GetFunctionBody(f_handle);
470 CHECK_NOTNULL(f_body);
471 *g_body = SymbolicGradient(*f_body);
472 }
473 return Status::OK();
474}

476bool FunctionLibraryRuntimeImpl::IsLocalTarget(
477 const InstantiateOptions& options) {
478 if (device_ == nullptr) return true;
479 if (options.target.empty()) return true;
480 Device* target_device;
481 if (!device_mgr_->LookupDevice(options.target, &target_device).ok()) {
482 return false;
483 }
484 return target_device == device_;
485}

487Status FunctionLibraryRuntimeImpl::Instantiate(
488 const string& function_name, AttrSlice attrs,
489 const InstantiateOptions& options, Handle* handle) {
490 if (!IsLocalTarget(options)) {
491 return parent_->Instantiate(function_name, attrs, options, handle);
492 }
493
494 // Since this is a local target, ensure that the local `device_name_` appears
495 // in the canonical key.
496 InstantiateOptions options_copy(options);
497 options_copy.target = device_name_;
498 const string key = Canonicalize(function_name, attrs, options_copy);
499
500 {
501 mutex_lock l(mu_);
502 *handle = parent_->GetHandle(key);
503 if (*handle != kInvalidHandle) {
504 FunctionLibraryRuntime::LocalHandle handle_on_device =
505 parent_->GetHandleOnDevice(device_name_, *handle);
506 if (handle_on_device == kInvalidLocalHandle) {
507 return errors::Internal("LocalHandle not found for handle ", *handle,
508 ".");
509 }
510 auto item_handle = items_.find(handle_on_device);
511 if (item_handle == items_.end()) {
512 return errors::Internal("LocalHandle ", handle_on_device,
513 " for handle ", *handle,
514 " not found in items.");
515 }
516 item_handle->second->Ref();
517 return Status::OK();
518 }
519 }
520
521 Status s;
522 const FunctionLibraryDefinition* lib_def =
523 options.overlay_lib ? options.overlay_lib : base_lib_def_;
524 FunctionBody* fbody = nullptr;
525 if (function_name == kGradientOp) {
526 const AttrValue* f = attrs.Find(kFuncAttr);
527 if (f == nullptr) {
528 return errors::InvalidArgument("SymbolicGradient is missing attr: f");
529 }
530 const auto& func = f->func();
531 if (func.name() == kGradientOp) {
532 return errors::InvalidArgument("Can't take gradient of SymbolicGradient");
533 }
534 const string grad = lib_def->FindGradient(func.name());
535 if (!grad.empty()) {
536 return Instantiate(grad, AttrSlice(&func.attr()), options, handle);
537 }
538 TF_RETURN_IF_ERROR(InstantiateSymbolicGradient(func, lib_def, &fbody));
539 } else {
540 const FunctionDef* fdef = lib_def->Find(function_name);
541 if (fdef == nullptr) {
542 return errors::NotFound("Function ", function_name, " is not defined.");
543 }
544 TF_RETURN_IF_ERROR(FunctionDefToBody(*fdef, attrs, lib_def, &fbody));
545 }
546
547 {
548 mutex_lock l(mu_);
549 *handle = parent_->GetHandle(key);
550 if (*handle != kInvalidHandle) {
551 delete fbody;
552 items_[parent_->GetHandleOnDevice(device_name_, *handle)]->Ref();
553 } else {
554 *handle = parent_->AddHandle(key, device_name_, next_handle_);
555 Item* item = new Item;
556 item->func_graph = fbody;
557 item->overlay_lib = options.overlay_lib;
558 items_.insert({next_handle_, item});
559 next_handle_++;
560 }
561 }
562 return Status::OK();
563}

565Status FunctionLibraryRuntimeImpl::ReleaseHandle(Handle handle) {
566 if (!parent_->IsInstantiatedOnDevice(device_name_, handle)) {
567 return parent_->ReleaseHandle(handle);
568 }
569
570 LocalHandle h = parent_->GetHandleOnDevice(device_name_, handle);
571 CHECK_NE(h, kInvalidLocalHandle);
572 mutex_lock l(mu_);
573 CHECK_EQ(1, items_.count(h));
574 Item* item = items_[h];
575 if (item->Unref()) {
576 items_.erase(h);
577 TF_RETURN_IF_ERROR(parent_->RemoveHandle(handle));
578 }
579 return Status::OK();
580}

603// Removes all stateless nodes that do not contribute to a return
604// value from the function body. Unlike `RemoveDeadNodes()`, which is
605// triggered by `OptimizerOptions.do_function_inlining`, this pass
606// ignores the SINK node, from which (by definition) all nodes are
607// reverse reachable.
608void PruneFunctionBody(Graph* g) {
609 VLOG(2) << "Pruning function body";
610 std::unordered_set<const Node*> nodes;
611 for (auto n : g->nodes()) {
612 // NOTE(mrry): "_Retval" nodes are stateful, and so will be added
613 // to the seed set of `nodes`.
614 // TODO(mrry): Investigate whether the `n->IsControlFlow()` test is
615 // still needed. It would be preferable to prune entire loops and/or
616 // conditionals if they are not used in the graph.
617 if (n->IsControlFlow() || n->op_def().is_stateful()) {
618 nodes.insert(n);
619 }
620 }
621 bool changed = PruneForReverseReachability(g, std::move(nodes));
622 if (changed) {
623 FixupSourceAndSinkEdges(g);
624 }
625}

628Status FunctionLibraryRuntimeImpl::CreateItem(Handle handle, Item** item) {
629 const FunctionBody* fbody;
630 const FunctionLibraryDefinition* lib_def;
631 {
632 mutex_lock l(mu_);
633 fbody = (*item)->func_graph;
634 lib_def = (*item)->overlay_lib;
635 }
636 if (!lib_def) {
637 lib_def = base_lib_def_;
638 }
639 std::unique_ptr<Graph> g(new Graph(lib_def));
640 CopyGraph(*fbody->graph, g.get());
641
642 PruneFunctionBody(g.get());
643 optimizer_.Optimize(this, env(), device(), &g, /*shape_map=*/nullptr);
644 TF_RETURN_IF_ERROR(EnsureMemoryTypes(DeviceType(device()->device_type()),
645 device()->name(), g.get()));
646
647 // Creates an executor based on the g. This must be done without
648 // holding mu_ because create_kernel_ calls back into the library.
649 LocalExecutorParams params;
650 params.device = device_;
651 params.function_library = this;
652 if (lib_def == base_lib_def_) {
653 params.create_kernel = create_kernel_;
654 } else {
655 params.create_kernel = [this, lib_def](const NodeDef& ndef,
656 OpKernel** kernel) {
657 return CreateKernel(ndef, lib_def, kernel);
658 };
659 }
660 params.delete_kernel = [](OpKernel* kernel) {
661 DeleteNonCachedKernel(kernel);
662 };
663 Graph* graph = g.get();
664 Executor* exec;
665 TF_RETURN_IF_ERROR(NewLocalExecutor(params, std::move(g), &exec));
666
667 {
668 // Guard item since it is already inserted in items_.
669 mutex_lock l(mu_);
670 if ((*item)->exec) {
671 delete exec;
672 } else {
673 (*item)->graph = graph;
674 (*item)->exec = exec;
675 }
676 }
677 return Status::OK();
678}

680Status FunctionLibraryRuntimeImpl::GetOrCreateItem(Handle handle, Item** item) {
681 LocalHandle local_handle = parent_->GetHandleOnDevice(device_name_, handle);
682 {
683 mutex_lock l(mu_);
684 if (items_.count(local_handle) == 0) {
685 return errors::NotFound("Function handle ", handle,
686 " is not valid. Likely an internal error.");
687 }
688 *item = items_[local_handle];

698void FunctionLibraryRuntimeImpl::RunRemote(const Options& opts, Handle handle,
699 gtl::ArraySlice<Tensor> args,
700 std::vector<Tensor>* rets,
701 Executor::Args* exec_args,
702 Item* item, DoneCallback done) {
703 DCHECK(exec_args->call_frame == nullptr);
704 string target_device = parent_->GetDeviceName(handle);
705 string source_device = opts.source_device;
706 Rendezvous* rendezvous = opts.rendezvous;
707 DeviceContext* device_context;
708 Status s = parent_->GetDeviceContext(target_device, &device_context);
709 if (!s.ok()) {
710 delete exec_args;
711 done(s);
712 return;
713 }
714 int64 src_incarnation, target_incarnation;
715 s = parent_->GetDeviceIncarnation(source_device, &src_incarnation);
716 s.Update(parent_->GetDeviceIncarnation(target_device, &target_incarnation));
717 if (!s.ok()) {
718 delete exec_args;
719 done(s);
720 return;
721 }
722
723 const FunctionBody* fbody = GetFunctionBody(handle);
724 FunctionCallFrame* frame =
725 new FunctionCallFrame(fbody->arg_types, fbody->ret_types);
726 exec_args->call_frame = frame;
727 if (!s.ok()) {
728 delete frame;
729 delete exec_args;
730 done(s);
731 return;
732 }
733
734 // The ProcFLR sends the arguments to the function from the source_device to
735 // the target_device. So here we receive those arguments. Similarly, when the
736 // computation is done and stored in *rets, we send the return values back
737 // to the source_device (caller) so that the ProcFLR can receive them later.
738 std::vector<Tensor>* remote_args = new std::vector<Tensor>;
739 ProcessFunctionLibraryRuntime::ReceiveTensorsAsync(
740 source_device, target_device, "arg_", src_incarnation, args.size(),
741 device_context, {}, rendezvous, remote_args,
742 [frame, remote_args, item, source_device, target_device,
743 target_incarnation, rendezvous, device_context, rets, done,
744 exec_args](const Status& status) {
745 Status s = status;
746 if (s.ok()) {
747 s = frame->SetArgs(*remote_args);
748 }
749 if (!s.ok()) {
750 delete frame;
751 delete remote_args;
752 delete exec_args;
753 done(s);
754 return;
755 }
756 item->exec->RunAsync(
757 *exec_args, [item, frame, rets, done, source_device, target_device,
758 target_incarnation, rendezvous, device_context,
759 remote_args, exec_args](const Status& status) {
760 Status s = status;
761 if (s.ok()) {
762 s = frame->ConsumeRetvals(rets);
763 }
764 delete frame;
765 if (!s.ok()) {
766 delete remote_args;
767 delete exec_args;
768 done(s);
769 return;
770 }
771 s = ProcessFunctionLibraryRuntime::SendTensors(
772 target_device, source_device, "ret_", target_incarnation,
773 *rets, device_context, {}, rendezvous);
774 delete remote_args;
775 delete exec_args;
776 done(s);
777 });
778 });
779}

1016// If 'edges' contains only 1 non-control edge, returns it. Otherwise,
1017// returns a nullptr.
1018const Edge* GetTheOnlyDataEdge(const EdgeSet& edges) {
1019 const Edge* ret = nullptr;
1020 for (const Edge* e : edges) {
1021 if (e->IsControlEdge() || ret) {
1022 // Don't touch it if there is a control edge.
1023 return nullptr;
1024 }
1025 if (IsRefType(e->src()->output_type(e->src_output()))) {
1026 // Don't touch it if the identity node is effectively de-reffing
1027 // a ref.
1028 return nullptr;
1029 }
1030 if (IsRecv(e->src()) || IsSwitch(e->src())) {
1031 // Don't touch it if the identity is introduced for control flow.
1032 // Recv disables all its successors if it receives a dead signal.
1033 // When Recv has an outgoing control edge, the current executor
1034 // would not disable the destination. The current solution (see
1035 // graph_partition.cc) is to add an identity after Recv and change
1036 // the control edge to be from this identity node. So the identity
1037 // can't be removed.
1038 return nullptr;
1039 }
1040 ret = e;
1041 }
1042 return ret;
1043}

1046bool RemoveIdentityNodes(Graph* g) {
1047 VLOG(2) << "Removing identity nodes";
1048 bool removed_any = false;
1049 gtl::InlinedVector<Node*, 8> matches;
1050 for (Node* n : g->nodes()) {
1051 if (!n->IsIdentity()) continue;
1052 if (!GetTheOnlyDataEdge(n->in_edges())) continue;
1053
1054 // Some identity nodes are used as sink nodes to give names to output
1055 // tensors. These nodes are not going to be executed unless they are in the
1056 // fetch set. But if they are in the fetch set we don't want to remove them.
1057 if (n->out_edges().empty()) continue;
1058
1059 matches.push_back(n);
1060 }
1061 if (!matches.empty()) {
1062 for (Node* n : matches) {
1063 const Edge* in = GetTheOnlyDataEdge(n->in_edges());
1064 for (const Edge* out : n->out_edges()) {
1065 if (out->IsControlEdge()) {
1066 g->AddControlEdge(in->src(), out->dst());
1067 } else {
1068 g->AddEdge(in->src(), in->src_output(), out->dst(), out->dst_input());
1069 }
1070 }
1071 VLOG(2) << "Remove Identity: " << n->DebugString();
1072 g->RemoveNode(n);
1073 removed_any = true;
1074 }
1075 }
1076 return removed_any;
1077}

1079bool RemoveListArrayConverter(Graph* g) {
1080 VLOG(2) << "Removing list array converter";
1081 gtl::InlinedVector<Node*, 8> matches;
1082 for (Node* n : g->nodes()) {
1083 if ((n->type_string() == "_ListToArray") ||
1084 (n->type_string() == "_ArrayToList")) {
1085 matches.push_back(n);
1086 }
1087 }
1088 bool removed_any = false;
1089 if (!matches.empty()) {
1090 for (Node* n : matches) {
1091 if (n->num_inputs() != n->num_outputs()) {
1092 continue; // Not expected. Skip.
1093 }
1094 gtl::InlinedVector<Node*, 8> identity_nodes(n->num_inputs(), nullptr);
1095
1096 // Process input edges first.
1097 Node* input_control_node = nullptr;
1098 for (const Edge* e : n->in_edges()) {
1099 if (e->IsControlEdge()) {
1100 if (input_control_node == nullptr) {
1101 // If node "n" has any control dependencies, adds a no-op
1102 // node (input_control_node) which the additional Identity
1103 // nodes depends on and the input_control_node depends on
1104 // the node "n"s control dependencies.
1105 input_control_node = AddNoOp(g);
1106 }
1107 g->AddControlEdge(e->src(), input_control_node);
1108 } else {
1109 const int index = e->dst_input();
1110 Node** id_node = &identity_nodes[index];
1111 if (*id_node != nullptr) {
1112 LOG(ERROR)
1113 << "RemoveListArrayConverter unexpected duplicated input: "
1114 << e->dst_input();
1115 return removed_any;
1116 }
1117 *id_node = AddIdentity(g, {e->src(), e->src_output()});
1118 }
1119 }
1120
1121 // If node "n" has any control dependencies, the added identity
1122 // nodes should have control dependencies on input_control_node.
1123 if (input_control_node != nullptr) {
1124 for (Node* id : identity_nodes) {
1125 g->AddControlEdge(input_control_node, id);
1126 }
1127 }
1128
1129 Node* output_control_node = nullptr;
1130 for (const Edge* e : n->out_edges()) {
1131 if (e->IsControlEdge()) {
1132 if (output_control_node == nullptr) {
1133 // If node "n" is control-depended upon by other nodes,
1134 // adds a no-op node (output_control_node) which those
1135 // nodes will depend on and output_control_node depends on
1136 // all Identity nodes.
1137 output_control_node = AddNoOp(g);
1138 }
1139 g->AddControlEdge(output_control_node, e->dst());
1140 } else {
1141 Node* id_node = identity_nodes[e->src_output()];
1142 if (id_node == nullptr) {
1143 LOG(ERROR) << "RemoveListArrayConverter unexpected missing input: "
1144 << e->src_output();
1145 return removed_any;
1146 }
1147 CHECK(id_node);
1148 g->AddEdge(id_node, 0, e->dst(), e->dst_input());
1149 }
1150 }
1151
1152 // If any nodes have control dependencies on node "n", those
1153 // nodes should have control dependencies on
1154 // output_control_node.
1155 if (output_control_node != nullptr) {
1156 for (Node* id : identity_nodes) {
1157 g->AddControlEdge(id, output_control_node);
1158 }
1159 }
1160
1161 g->RemoveNode(n);
1162 removed_any = true;
1163 }
1164 }
1165 return removed_any;
1166}

1168// Returns true iff the function '*fbody' can be inlined at 'node'
1169// based on the type signature of 'node' and 'fbody'.
1170static bool ValidateInlining(const Node* node, const FunctionBody* fbody) {
1171 if (static_cast<size_t>(node->num_inputs()) != fbody->arg_types.size()) {
1172 return false;
1173 }
1174 if (static_cast<size_t>(node->num_inputs()) != fbody->arg_nodes.size()) {
1175 return false;
1176 }
1177 if (static_cast<size_t>(node->num_outputs()) != fbody->ret_types.size()) {
1178 return false;
1179 }
1180 if (static_cast<size_t>(node->num_outputs()) != fbody->ret_nodes.size()) {
1181 return false;
1182 }
1183 for (int i = 0; i < node->num_inputs(); ++i) {
1184 if (node->input_type(i) != fbody->arg_types[i]) return false;
1185 }
1186 for (int i = 0; i < node->num_outputs(); ++i) {
1187 if (node->output_type(i) != fbody->ret_types[i]) return false;
1188 }
1189 return true;
1190}

1192// Given a "caller" in "graph", which is a function call of a function
1193// to "fbody". Replaces the "caller" with fbody->graph and connects
1194// edges properly.
1195void InlineFunctionBody(const FunctionLibraryDefinition& flib_def, Graph* g,
1196 Node* caller, const FunctionBody* fbody) {
1197 if (!ValidateInlining(caller, fbody)) {
1198 LOG(WARNING) << "Inlining mismatch: " << caller->DebugString() << " vs. "
1199 << DebugString(fbody->graph);
1200 return;
1201 }
1202
1203 // Input edges. For data edges coming into "caller", we first compute the
1204 // <src>:<src_output> for the i-th input in "inputs".
1205 // If "caller" has any input control dependencies, we add a NoOp
1206 // node "input_control_node", which depends on "caller"'s control inputs.
1207 std::vector<Endpoint> inputs(caller->num_inputs());
1208 Node* input_control_node = nullptr;
1209 for (const Edge* e : caller->in_edges()) {
1210 if (e->IsControlEdge()) {
1211 if (input_control_node == nullptr) {
1212 input_control_node = AddNoOp(g);
1213 }
1214 g->AddControlEdge(e->src(), input_control_node);
1215 } else {
1216 inputs[e->dst_input()] = {e->src(), e->src_output()};
1217 }
1218 }
1219
1220 // Duplicate fbody->graph into 'g'. First, we copy the nodes of
1221 // fbody->graph into 'g' except the source and sink nodes. We copy
1222 // edges among nodes in 'fbody->graph'.
1223 //
1224 // If 'x' is a node in fbody->graph and its copy in 'g' is 'y', we
1225 // remember 'y' in node_map[x->id()].
1226 std::vector<Node*> node_map(fbody->graph->num_node_ids());
1227 Status s;
1228 for (Node* n : fbody->graph->op_nodes()) {
1229 NodeDef ndef = n->def();
1230 ndef.set_name(strings::StrCat(caller->name(), "/", ndef.name()));
1231 ndef.set_device(caller->def().device());
1232 Node* clone = g->AddNode(ndef, &s);
1233 TF_CHECK_OK(s);
1234 node_map[n->id()] = clone;
1235
1236 // If there is an input control node, and one of:
1237 // a) the node has no data or control inputs, or
1238 // b) the node is a function call or SymbolicGradient,
1239 // then add a control edge from the input control node to the clone.
1240 //
1241 // We must not execute any nodes if the original function call would not
1242 // have executed. This is especially critical when the function call is
1243 // inside a control-flow construct like tf.cond(). Case (a) ensures that
1244 // such nodes do not run.
1245 //
1246 // The purpose of case (b) is to ensure that instances of case (a) created
1247 // by further inlining steps also receive the control dependency.
1248 if (input_control_node) {
1249 bool has_inputs = false;
1250 for (const Edge* e : n->in_edges()) {
1251 if (!e->src()->IsSource()) {
1252 has_inputs = true;
1253 break;
1254 }
1255 }
1256 if (!has_inputs || flib_def.Find(clone->type_string()) != nullptr ||
1257 clone->type_string() == "SymbolicGradient") {
1258 g->AddControlEdge(input_control_node, clone);
1259 }
1260 }
1261 }
1262 for (const Edge* e : fbody->graph->edges()) {
1263 if (e->src()->IsSource() || e->src()->IsSink() || e->dst()->IsSource() ||
1264 e->dst()->IsSink()) {
1265 continue;
1266 }
1267 Node* src_copy = node_map[e->src()->id()];
1268 Node* dst_copy = node_map[e->dst()->id()];
1269 g->AddEdge(src_copy, e->src_output(), dst_copy, e->dst_input());
1270 }
1271
1272 // Connect input edges.
1273 //
1274 // We create one Identity node for each input. Then, we connect inputs[i] to
1275 // the i-th identity node added. The nodes that previously connected
1276 // to the j-th output of i-th arg node are reconnected to the i-th
1277 // identity node.
1278 //
1279 // The added identity nodes depend on "input_control_node".
1280 for (std::size_t i = 0; i < fbody->arg_nodes.size(); ++i) {
1281 Node* arg = node_map[fbody->arg_nodes[i]->id()];
1282 Node* n = AddIdentity(g, inputs[i]);
1283 if (input_control_node) {
1284 g->AddControlEdge(input_control_node, n);
1285 }
1286 for (const Edge* e : arg->out_edges()) {
1287 if (e->IsControlEdge()) {
1288 g->AddControlEdge(n, e->dst());
1289 } else {
1290 g->AddEdge(n, 0, e->dst(), e->dst_input());
1291 }
1292 }
1293 node_map[fbody->arg_nodes[i]->id()] = n;
1294 g->RemoveNode(arg); // 'arg' is disconnected.
1295 }
1296
1297 // Connect output edges.
1298 //
1299 // For i-th return node in fbody->graph, we add in "g" an identity
1300 // node (outputs[i-th]). We then reconnect every incoming edge into
1301 // the i-th return node to the added identity node.
1302 //
1303 // For every data edge coming out of "callee"s i-th output, we
1304 // reconnect it to the i-th identity added above.
1305 //
1306 // If "callee" is control-depended upon by any other nodes, we add a
1307 // NoOp node "output_control_node". "output_control_node" depends on
1308 // all identity nodes added above. And nodes previously depend on
1309 // "callee" is changed to depend on "output_control_node".
1310 std::vector<Node*> outputs(caller->num_outputs());
1311 for (std::size_t i = 0; i < fbody->ret_nodes.size(); ++i) {
1312 Node* ret = node_map[fbody->ret_nodes[i]->id()];
1313 Endpoint data; // Data input for the ret node.
1314 for (const Edge* e : ret->in_edges()) {
1315 if (!e->IsControlEdge()) {
1316 data = {e->src(), e->src_output()};
1317 break;
1318 }
1319 }
1320 CHECK(data.node != nullptr);
1321 Node* n = AddIdentity(g, data);
1322 outputs[i] = n;
1323 for (const Edge* e : ret->in_edges()) {
1324 if (e->IsControlEdge()) {
1325 g->AddControlEdge(e->src(), n);
1326 }
1327 }
1328 g->RemoveNode(ret); // 'ret' is disconnected.
1329 }
1330 Node* output_control_node = nullptr;
1331 for (const Edge* e : caller->out_edges()) {
1332 if (e->IsControlEdge()) {
1333 if (output_control_node == nullptr) {
1334 output_control_node = AddNoOp(g);
1335 for (Node* n : outputs) {
1336 g->AddControlEdge(n, output_control_node);
1337 }
1338 }
1339 g->AddControlEdge(output_control_node, e->dst());
1340 } else {
1341 g->AddEdge(outputs[e->src_output()], 0, e->dst(), e->dst_input());
1342 }
1343 }
1344 g->RemoveNode(caller); // 'caller' is replaced with inlined nodes.
1345}

1347bool ExpandInlineFunctions(FunctionLibraryRuntime* lib, Graph* graph) {
1348 std::vector<std::pair<Node*, const FunctionBody*>> candidates;
1349 const FunctionLibraryDefinition* fld = lib->GetFunctionLibraryDefinition();
1350 for (Node* node : graph->nodes()) {
1351 VLOG(3) << "Expanding " << node->DebugString();
1352 bool noinline;
1353 if (fld->GetAttr(*node, kNoInlineAttr, &noinline).ok() && noinline) {
1354 VLOG(3) << "noinline: " << node->DebugString();
1355 continue;
1356 }
1357 FunctionLibraryRuntime::Handle handle;
1358 Status s = lib->Instantiate(node->type_string(), node->attrs(), &handle);
1359 if (!s.ok()) {
1360 // Either "node" is a primitive op, or the instantiation failed.
1361 if (errors::IsNotFound(s)) {
1362 VLOG(3) << "ExpandInlineFunctions " << s;
1363 } else {
1364 LOG(ERROR) << "ExpandInlineFunctions " << s;
1365 }
1366 continue;
1367 }
1368 const FunctionBody* fbody = lib->GetFunctionBody(handle);
1369 CHECK_NOTNULL(fbody);
1370 candidates.push_back({node, fbody});
1371 }
1372 for (const auto& p : candidates) {
1373 InlineFunctionBody(*fld, graph, p.first, p.second);
1374 }
1375 return !candidates.empty();
1376}

1387// and stash the original NodeDef name as an attr for documentation
1388// purpose.
1389void ToGraphDef(const Graph* g, GraphDef* gdef, bool pretty) {
1390 // We visit nodes in forward topological sort order, which is a
1391 // possible execution order of the graph.
1392 gtl::InlinedVector<const Edge*, 4> inputs;
1393 gdef->Clear();
1394 gdef->mutable_versions()->CopyFrom(g->versions());
1395
1396 std::vector<Node*> start_nodes;
1397 for (Node* n : g->nodes()) {
1398 if (n->out_edges().empty()) {
1399 start_nodes.push_back(n);
1400 }
1401 }
1402
1403 ReverseDFSFrom(*g, start_nodes, nullptr, [gdef, pretty, &inputs](Node* n) {
1404 if (!n->IsOp()) return;
1405 NodeDef* ndef = gdef->add_node();
1406 ndef->set_name(NewName(n, pretty));
1407 ndef->set_op(n->type_string());
1408 for (const auto& attr : n->attrs()) {
1409 (*ndef->mutable_attr())[attr.first] = attr.second;
1410 }
1411 inputs.clear();
1412 inputs.resize(n->num_inputs());
1413 for (const Edge* e : n->in_edges()) {
1414 if (e->IsControlEdge()) {
1415 inputs.push_back(e);
1416 } else {
1417 if (inputs[e->dst_input()] == nullptr) {
1418 inputs[e->dst_input()] = e;
1419 } else {
1420 LOG(WARNING) << "Malformed graph node. multiple input edges: "
1421 << n->DebugString();
1422 }
1423 }
1424 }
1425 // node->name() is merely NodeDef::name, which are not guaranteed
1426 // to be unique and stable after optimization rewrites. Therefore,
1427 // we use "n<node id>" instead.
1428 for (const Edge* e : inputs) {
1429 if (e == nullptr) {
1430 ndef->add_input("unknown");
1431 continue;
1432 }
1433 const string srcname = NewName(e->src(), pretty);
1434 if (!e->src()->IsOp()) {
1435 } else if (e->IsControlEdge()) {
1436 ndef->add_input(strings::StrCat("^", srcname));
1437 } else if (e->src_output() == 0) {
1438 ndef->add_input(srcname);
1439 } else {
1440 ndef->add_input(strings::StrCat(srcname, ":", e->src_output()));
1441 }
1442 }
1443 });
1444}

1497void SymbolicGradientHelper::Copy() {
1498 const Graph& src = *(fbody_->graph);
1499 gbody_->graph = new Graph(src.op_registry());
1500 Graph* dst = gbody_->graph;
1501
1502 std::vector<Node*> node_map(src.num_node_ids());
1503
1504 // Copy the nodes.
1505 node_map[src.source_node()->id()] = dst->source_node();
1506 node_map[src.sink_node()->id()] = dst->sink_node();
1507 for (Node* n : src.op_nodes()) {
1508 node_map[n->id()] = dst->CopyNode(n);
1509 }
1510
1511 // Copy the edges.
1512 for (const Edge* e : src.edges()) {
1513 Node* src_copy = node_map[e->src()->id()];
1514 Node* dst_copy = node_map[e->dst()->id()];
1515 dst->AddEdge(src_copy, e->src_output(), dst_copy, e->dst_input());
1516 }
1517
1518 // Save inputs in copied graph.
1519 CHECK_EQ(fbody_->arg_types.size(), fbody_->arg_nodes.size());
1520 gbody_->arg_types = fbody_->arg_types;
1521 for (std::size_t i = 0; i < fbody_->arg_nodes.size(); ++i) {
1522 gbody_->arg_nodes.push_back(node_map[fbody_->arg_nodes[i]->id()]);
1523 }
1524
1525 // Save outputs in copied graph.
1526 CHECK_EQ(fbody_->ret_types.size(), fbody_->ret_nodes.size());
1527 gbody_->ret_types = fbody_->ret_types;
1528 for (std::size_t i = 0; i < fbody_->ret_nodes.size(); ++i) {
1529 gbody_->ret_nodes.push_back(node_map[fbody_->ret_nodes[i]->id()]);
1530 }
1531}

1533FunctionBody* SymbolicGradientHelper::Compute() {
1534 CHECK(gbody_ == nullptr);
1535 gbody_ = new FunctionBody;
1536
1537 // Copy fbody_ into gbody_.
1538 Copy();
1539
1540 Graph* g = gbody_->graph;
1541
1542 const int num_y = static_cast<int>(gbody_->ret_nodes.size());
1543
1544 // Populate 'y_node_outputs_' with node function body outputs.
1545 // Populate 'y_grad_nodes' with initial gradient nodes for each return node of
1546 // the original function body (these will be 'arg' nodes in the function
1547 // gradient body).
1548 std::vector<NodeOut> y_node_outputs;
1549 y_node_outputs.reserve(num_y);
1550 std::vector<NodeOut> y_grad_node_outputs;
1551 y_grad_node_outputs.reserve(num_y);
1552 for (int i = 0; i < num_y; ++i) {
1553 Node* y = gbody_->ret_nodes[i];
1554 y_node_outputs.push_back({y, 0});
1555 DCHECK_EQ(y->type_string(), kRetOp);
1556 const DataType dtype = y->input_type(0);
1557 const int index = static_cast<int>(gbody_->arg_nodes.size());
1558 Node* dy = AddArg(g, dtype, index);
1559 gbody_->arg_types.push_back(dtype);
1560 gbody_->arg_nodes.push_back(dy);
1561 y_grad_node_outputs.push_back({dy, 0});
1562 }
1563
1564 // Populate 'x_nodes' with function args (excluding 'y_grad_node_outputs').
1565 const size_t num_x = fbody_->arg_nodes.size();
1566 std::vector<NodeOut> x_node_outputs;
1567 x_node_outputs.reserve(num_x);
1568 for (size_t i = 0; i < fbody_->arg_nodes.size(); ++i) {
1569 x_node_outputs.push_back({gbody_->arg_nodes[i], 0});
1570 }
1571
1572 // Call AddSymbolicGradients which will add nodes to graph 'g' that
1573 // compute the function gradient (adding an entry in 'x_grad_node_outputs' for
1574 // each node in 'x_node_outputs').
1575 std::vector<NodeOut> x_grad_node_outputs;
1576 TF_CHECK_OK(AddSymbolicGradients(y_node_outputs, x_node_outputs,
1577 y_grad_node_outputs, &x_grad_node_outputs,
1578 g));
1579
1580 // Remove the old return nodes from the function body.
1581 for (Node* n : gbody_->ret_nodes) {
1582 g->RemoveNode(n);
1583 }
1584 gbody_->ret_types = fbody_->arg_types;
1585 gbody_->ret_nodes.clear();
1586 // Add new return nodes to the function gradient body for each node
1587 // in 'x_grad_nodes'.
1588 const int arg_types_size = static_cast<int>(fbody_->arg_types.size());
1589 for (int i = 0; i < arg_types_size; ++i) {
1590 Endpoint grad = {x_grad_node_outputs[i].node, x_grad_node_outputs[i].index};
1591 Node* ret = AddRet(g, grad, i);
1592 gbody_->ret_nodes.push_back(ret);
1593 }
1594
1595 auto ret = gbody_;
1596 gbody_ = nullptr;
1597 return ret;
1598}

1601 return SymbolicGradientHelper(f).Compute();
