15#ifndef TENSORFLOW_COMMON_RUNTIME_COLLECTIVE_RMA_LOCAL_ACCESS_H_
16#define TENSORFLOW_COMMON_RUNTIME_COLLECTIVE_RMA_LOCAL_ACCESS_H_
17#include "tensorflow/core/common_runtime/buf_rendezvous.h"
18#include "tensorflow/core/common_runtime/device_mgr.h"
19#include "tensorflow/core/framework/collective.h"
20#include "tensorflow/core/framework/rendezvous.h"
21
22namespace tensorflow {
23
24// Basic implementation of PerStepCollectiveRemoteAccess.
25class CollectiveRemoteAccessLocal : public PerStepCollectiveRemoteAccess {
26 public:
27 CollectiveRemoteAccessLocal(const DeviceMgr* dev_mgr,
28 DeviceResolverInterface* dev_resolver,
29 int64 step_id)
30 : dev_mgr_(dev_mgr),
31 dev_resolver_(dev_resolver),
32 buf_rendezvous_(step_id),
33 step_id_(step_id) {}
34
35 virtual ~CollectiveRemoteAccessLocal() {}
36
37 void StartAbort(const Status& s);
38
39 void RecvFromPeer(const string& peer_device, const string& peer_task,
40 bool peer_is_local, const string& key, Device* to_device,
41 DeviceContext* to_device_ctx,
42 const AllocatorAttributes& to_alloc_attr, Tensor* to_tensor,
43 const DeviceLocality& client_locality,
44 const StatusCallback& done) override;
45
46 void PostToPeer(const string& peer_device, const string& peer_task,
47 const string& key, Device* from_device,
48 DeviceContext* from_device_ctx,
49 const AllocatorAttributes& from_alloc_attr,
50 const Tensor* from_tensor,
51 const DeviceLocality& client_locality,
52 const StatusCallback& done) override;
53
54 void GetDeviceLocalitiesAsync(const CollInstanceParams& ci_params,
55 std::vector<DeviceLocality>* localities,
56 const StatusCallback& done) override {
57 dev_resolver_->GetDeviceLocalitiesAsync(ci_params, localities, done);
58 }
59
60 void GetLocalityAsync(const string& device, const string& task,
61 DeviceLocality* locality,
62 const StatusCallback& done) override {
63 dev_resolver_->GetLocalityAsync(device, task, locality, done);
64 }
65
66 void ClearTask(const string& task) override {
67 dev_resolver_->ClearTask(task);
68 }
69
70 // Copy utility that always copies bytes from src to dst even if
71 // they are on the same device, unlike CopyTensor::ViaDMA which will
72 // just change the dst buffer pointer in that case.
73 static void MemCpyAsync(DeviceContext* src_dev_ctx,
74 DeviceContext* dst_dev_ctx, Device* src_dev,
75 Device* dst_dev, const AllocatorAttributes& src_attr,
76 const AllocatorAttributes& dst_attr,
77 const Tensor* src, Tensor* dst,
78 const StatusCallback& done);
79
80 protected:
81 const DeviceMgr* dev_mgr_; // not owned
82 DeviceResolverInterface* dev_resolver_; // not owned
83 BufRendezvous buf_rendezvous_;
84 int64 step_id_;
85};
86
87} // namespace tensorflow
88#endif // TENSORFLOW_COMMON_RUNTIME_COLLECTIVE_RMA_LOCAL_ACCESS_H_
