15#ifndef TENSORFLOW_COMMON_RUNTIME_COLLECTIVE_EXECUTOR_MGR_H_
16#define TENSORFLOW_COMMON_RUNTIME_COLLECTIVE_EXECUTOR_MGR_H_
17
18#include "tensorflow/core/framework/collective.h"
19#include "tensorflow/core/lib/gtl/flatmap.h"
20
21namespace tensorflow {
22class ConfigProto;
23class DeviceMgr;
24
25class CollectiveExecutorMgr : public CollectiveExecutorMgrInterface {
26 public:
27 CollectiveExecutorMgr(const ConfigProto& config, const DeviceMgr* dev_mgr,
28 DeviceResolverInterface* dev_resolver,
29 ParamResolverInterface* param_resolver);
30
31 virtual ~CollectiveExecutorMgr();
32
33 CollectiveExecutor* FindOrCreate(int64 step_id) override;
34
35 void Cleanup(int64 step_id) override;
36
37 ParamResolverInterface* GetParamResolver() const override {
38 return param_resolver_.get();
39 }
40
41 DeviceResolverInterface* GetDeviceResolver() const override {
42 return dev_resolver_.get();
43 }
44
45 void GetStepSequenceAsync(const GetStepSequenceRequest* request,
46 GetStepSequenceResponse* response,
47 const StatusCallback& done) override;
48
49 void RefreshStepIdSequenceAsync(int64 graph_key,
50 const StatusCallback& done) override;
51
52 int64 NextStepId(int64 graph_key) override {
53 return CollectiveExecutor::kInvalidId;
54 }
55
56 void RetireStepId(int64 graph_key, int64 step_id) override {}
57
58 protected:
59 const DeviceMgr* dev_mgr_;
60 std::unique_ptr<DeviceResolverInterface> dev_resolver_;
61 std::unique_ptr<ParamResolverInterface> param_resolver_;
62 CollectiveRemoteAccess* remote_access_;
63 string task_name_;
64 mutex exec_mu_;
65 // Map from step_id to CollectiveExecutor
66 gtl::FlatMap<int64, CollectiveExecutor*> executor_table_ GUARDED_BY(exec_mu_);
67};
68
69} // namespace tensorflow
70#endif // TENSORFLOW_COMMON_RUNTIME_COLLECTIVE_EXECUTOR_MGR_H_
