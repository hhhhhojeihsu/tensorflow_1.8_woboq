30/* static */
31bool LocalDevice::use_global_threadpool_ = true;
32
33struct LocalDevice::EigenThreadPoolInfo {
34 explicit EigenThreadPoolInfo(const SessionOptions& options) {
35 int32 intra_op_parallelism_threads =
36 options.config.intra_op_parallelism_threads();
37 if (intra_op_parallelism_threads == 0) {
38 intra_op_parallelism_threads = port::NumSchedulableCPUs();
39 }
40 VLOG(1) << "Local device intra op parallelism threads: "
41 << intra_op_parallelism_threads;
42 eigen_worker_threads_.num_threads = intra_op_parallelism_threads;
43 eigen_worker_threads_.workers = new thread::ThreadPool(
44 options.env, "Eigen", intra_op_parallelism_threads);
45 eigen_threadpool_wrapper_.reset(
46 new EigenThreadPoolWrapper(eigen_worker_threads_.workers));
47 eigen_device_.reset(new Eigen::ThreadPoolDevice(
48 eigen_threadpool_wrapper_.get(), eigen_worker_threads_.num_threads));
49 }
50
51 ~EigenThreadPoolInfo() {
52 eigen_threadpool_wrapper_.reset();
53 eigen_device_.reset();
54 delete eigen_worker_threads_.workers;
55 }
56
57 DeviceBase::CpuWorkerThreads eigen_worker_threads_;
58 std::unique_ptr<Eigen::ThreadPoolInterface> eigen_threadpool_wrapper_;
59 std::unique_ptr<Eigen::ThreadPoolDevice> eigen_device_;
60};
61
62LocalDevice::LocalDevice(const SessionOptions& options,
63 const DeviceAttributes& attributes)
64 : Device(options.env, attributes), owned_tp_info_(nullptr) {
65 // Log info messages if TensorFlow is not compiled with instructions that
66 // could speed up performance and are available on the current CPU.
67 port::InfoAboutUnusedCPUFeatures();
68 LocalDevice::EigenThreadPoolInfo* tp_info;
69 if (use_global_threadpool_) {
70 // All ThreadPoolDevices in the process will use this single fixed
71 // sized threadpool for numerical computations.
72 static LocalDevice::EigenThreadPoolInfo* global_tp_info =
73 new LocalDevice::EigenThreadPoolInfo(options);
74 tp_info = global_tp_info;
75 } else {
76 // Each LocalDevice owns a separate ThreadPoolDevice for numerical
77 // computations.
78 owned_tp_info_.reset(new LocalDevice::EigenThreadPoolInfo(options));
79 tp_info = owned_tp_info_.get();
80 }
81 set_tensorflow_cpu_worker_threads(&tp_info->eigen_worker_threads_);
82 set_eigen_cpu_device(tp_info->eigen_device_.get());
83}
84
85LocalDevice::~LocalDevice() {}
