25namespace tensorflow {
26
27class Device;
28struct SessionOptions;
29
30class DeviceFactory {
31 public:
32 virtual ~DeviceFactory() {}
33 static void Register(const string& device_type, DeviceFactory* factory,
34 int priority);
35 static DeviceFactory* GetFactory(const string& device_type);
36
37 // Append to "*devices" all suitable devices, respecting
38 // any device type specific properties/counts listed in "options".
39 //
40 // CPU devices are added first.
41 static Status AddDevices(const SessionOptions& options,
42 const string& name_prefix,
43 std::vector<Device*>* devices);
44
45 // Helper for tests. Create a single device of type "type". The
46 // returned device is always numbered zero, so if creating multiple
47 // devices of the same type, supply distinct name_prefix arguments.
48 static Device* NewDevice(const string& type, const SessionOptions& options,
49 const string& name_prefix);
50
51 // Most clients should call AddDevices() instead.
52 virtual Status CreateDevices(const SessionOptions& options,
53 const string& name_prefix,
54 std::vector<Device*>* devices) = 0;
55
56 // Return the device priority number for a "device_type" string.
57 //
58 // Higher number implies higher priority.
59 //
60 // In standard TensorFlow distributions, GPU device types are
61 // preferred over CPU, and by default, custom devices that don't set
62 // a custom priority during registration will be prioritized lower
63 // than CPU. Custom devices that want a higher priority can set the
64 // 'priority' field when registering their device to something
65 // higher than the packaged devices. See calls to
66 // REGISTER_LOCAL_DEVICE_FACTORY to see the existing priorities used
67 // for built-in devices.
68 static int32 DevicePriority(const string& device_type);
69};
70
71namespace dfactory {
72
73template <class Factory>
74class Registrar {
75 public:
76 // Multiple registrations for the same device type with different priorities
77 // are allowed. Priorities are used in two different ways:
78 //
79 // 1) When choosing which factory (that is, which device
80 // implementation) to use for a specific 'device_type', the
81 // factory registered with the highest priority will be chosen.
82 // For example, if there are two registrations:
83 //
84 // Registrar<CPUFactory1>("CPU", 125);
85 // Registrar<CPUFactory2>("CPU", 150);
86 //
87 // then CPUFactory2 will be chosen when
88 // DeviceFactory::GetFactory("CPU") is called.
89 //
90 // 2) When choosing which 'device_type' is preferred over other
91 // DeviceTypes in a DeviceSet, the ordering is determined
92 // by the 'priority' set during registration. For example, if there
93 // are two registrations:
94 //
95 // Registrar<CPUFactory>("CPU", 100);
96 // Registrar<GPUFactory>("GPU", 200);
97 //
98 // then DeviceType("GPU") will be prioritized higher than
99 // DeviceType("CPU").
100 //
101 // The default priority values for built-in devices is:
102 // GPU: 210
103 // SYCL: 200
104 // GPUCompatibleCPU: 70
105 // ThreadPoolDevice: 60
106 // Default: 50
107 explicit Registrar(const string& device_type, int priority = 50) {
108 DeviceFactory::Register(device_type, new Factory(), priority);
109 }
110};
