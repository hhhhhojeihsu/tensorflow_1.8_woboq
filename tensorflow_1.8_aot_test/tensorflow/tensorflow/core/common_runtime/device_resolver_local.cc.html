21void DeviceResolverLocal::GetDeviceLocalitiesAsync(
22 const CollInstanceParams& ci_params,
23 std::vector<DeviceLocality>* localities, const StatusCallback& done) {
24 localities->clear();
25 for (const string& device_name : ci_params.device_names) {
26 Device* dev;
27 Status s = dev_mgr_->LookupDevice(device_name, &dev);
28 if (!s.ok()) {
29 done(s);
30 return;
31 }
32 localities->push_back(dev->attributes().locality());
33 }
34 done(Status::OK());
35}
36
37void DeviceResolverLocal::GetLocalityAsync(const string& device,
38 const string& task,
39 DeviceLocality* locality,
40 const StatusCallback& done) {
41 Device* dev;
42 Status s = dev_mgr_->LookupDevice(device, &dev);
43 if (s.ok()) {
44 *locality = dev->attributes().locality();
45 }
46 done(s);
47}
