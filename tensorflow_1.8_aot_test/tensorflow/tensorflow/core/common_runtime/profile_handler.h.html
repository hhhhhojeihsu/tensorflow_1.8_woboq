24namespace tensorflow {
25
26// A profile handler collects event stats from a running step.
27class ProfileHandler {
28 public:
29 ProfileHandler() {}
30 virtual ~ProfileHandler() {}
31
32 // Records that a miscellaneous activity occurred in the current step.
33 //
34 // Implementations of this method must be thread-safe.
35 //
36 // Args:
37 // - device: The device on which the activity occurred.
38 // - start: The time at which the activity started.
39 // - limit: The time at which the activity finished.
40 // - label: A label for the op, which may be used in visualization.
41 // - op_type: A type string for the op, which may be used in visualization.
42 // - details: A details string, which may be used in visualization.
43 // from time "start" to "limit" with "op_type" and "details".
44 virtual void RecordActivity(const string& device, Microseconds start,
45 Microseconds limit, StringPiece label,
46 StringPiece op_type, StringPiece details) = 0;
47
48 // Records that a single Op was executed in the current step.
49 //
50 // Implementations of this method must be thread-safe.
51 //
52 // Args:
53 // - device: Device on which the Op was executed.
54 // - stats: Statistics of node execution timing.
55 // - is_copy: True if the op was a copy, send or recv.
56 // - label: Extra content for timeline click text.
57 // - op_type: String name of the Op.
58 // - details: Main content for timeline click text.
59 virtual void RecordOneOp(const string& device, const NodeExecStats& stats,
60 bool is_copy, StringPiece label, StringPiece op_type,
61 StringPiece details) = 0;
62
63 // Records that the current step finished.
64 //
65 // Implementations of this method need not be thread-safe.
66 //
67 // Args:
68 // - start_time: The time at which the step started.
69 // - finish_time: The time at which the step finished.
70 // - cleanup_time: The time at which cleanup for the step finished.
71 // - total_runops: The number of ops that ran during this step.
72 // - final_status: The status that this step finished with.
73 virtual void StepDone(Microseconds start_time, Microseconds finish_time,
74 Microseconds cleanup_time, int total_runops,
75 Status final_status) = 0;
76
77 // Returns true if the caller should collect rpc activity.
78 virtual bool should_collect_rpcs() = 0;
79};
