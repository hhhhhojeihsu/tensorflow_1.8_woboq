38// We hoist the conversion from C-style string literal to StringPiece here,
39// so that we can avoid the many repeated calls to strlen().
40const StringPiece kColocationAttrNameStringPiece(kColocationAttrName);
41const StringPiece kColocationGroupPrefixStringPiece(kColocationGroupPrefix);
42
43// Returns a list of devices sorted by preferred type and then name
44// from 'devices' whose type is in 'supported_device_types'. This
45// function searches the device types in 'supported_device_types' and
46// returns the subset of devices that match.
47std::vector<Device*> FilterSupportedDevices(
48 const std::vector<Device*>& devices,
49 const DeviceTypeVector& supported_device_types) {
50 std::vector<Device*> filtered_devices;
51 for (const DeviceType& d : supported_device_types) {
52 for (Device* device : devices) {
53 if (DeviceType(device->attributes().device_type()) == d) {
54 filtered_devices.emplace_back(device);
55 }
56 }
57 }
58
59 auto device_sort = [](const Device* a, const Device* b) {
60 auto a_priority = DeviceSet::DeviceTypeOrder(DeviceType(a->device_type()));
61 auto b_priority = DeviceSet::DeviceTypeOrder(DeviceType(b->device_type()));
62 // First sort by prioritized device type (higher is preferred) and
63 // then by device name (lexicographically).
64 if (a_priority != b_priority) {
65 return a_priority > b_priority;
66 }
67 return StringPiece(a->name()) < StringPiece(b->name());
68 };
69 std::sort(filtered_devices.begin(), filtered_devices.end(), device_sort);
70 return filtered_devices;
71}

73// This class maintains the connected components of a colocation
74// constraint graph, and uses this information to assign a satisfying
75// device placement to the nodes of the graph.
76//
77// The typical usage pattern is:
78//
79// Graph graph = ...;
80// DeviceSet device_set = ...;
81// ColocationGraph colocation_graph(graph, device_set);
82//
83// // Add all the nodes of graph to colocation_graph.
84// for (Node* node : graph.nodes()) {
85// TF_RETURN_IF_ERROR(colocation_graph.AddNode(*node));
86// }
87//
88// // Add one or more colocation constraint.
89// Node node_1 = *graph.FindNodeId(...);
90// Node node_2 = *graph.FindNodeId(...);
91// TF_RETURN_IF_ERROR(colocation_graph.ColocateNodes(node_1, node_2));
92//
93// // Assign devices based on the accumulated constraints.
94// for (Node* node : graph.nodes()) {
95// TF_RETURN_IF_ERROR(colocation_graph.AssignDevice(node));
96// }
97//
98// The implementation uses the union-find algorithm to maintain the
99// connected components efficiently and incrementally as edges
100// (implied by ColocationGraph::ColocateNodes() invocations) are added.
101class ColocationGraph {
102 public:
103 ColocationGraph(Graph* graph, const DeviceSet* device_set,
104 bool allow_soft_placement)
105 : graph_(graph),
106 device_set_(device_set),
107 device_types_(device_set->PrioritizedDeviceTypeList()),
108 allow_soft_placement_(allow_soft_placement) {
109 members_.resize(graph->num_node_ids());
110 }
111
112 // Adds each node of the Graph to this ColocationGraph as a singleton.
113 //
114 // NOTE: The implementation assumes that the ids of nodes passed to
115 // this method are dense and zero-based; the memory used will be linear in
116 // the largest node ID.
117 // NOTE: If this method returns an error, *this is left in an undefined
118 // state.
119 Status ColocateAllNodes() {
120 // This maps from a colocation group identifier to the 'root' of that
121 // colocation group. Note that the keys in this map are StringPiece; the
122 // actual strings are stored under the NodeDef. The lifetime of this map
123 // is limited to this ColocateAllNodes() method, and no part of the
124 // NodeDef trees are changed during the lifetime of this method, so using
125 // StringPiece as a key is safe.
126 //
127 // Also, as a further optimization, we remove the "loc:@" prefix from
128 // "class" attribute values, when they are used as keys in this table.
129 // This allows us to use StringPiece values that refer to substrings of
130 // 'string' values stored in NodeDef attribute lists, as well as StringPiece
131 // values that refer to 'string' values from NodeDef::name(), without
132 // performing any string allocations.
133 std::unordered_map<StringPiece, const Node*, StringPieceHasher>
134 colocation_group_root;
135
136 for (Node* node : graph_->nodes()) {
137 if (!node->IsOp()) {
138 continue;
139 }
140
141 // When adding the node, identify whether it is part of a
142 // colocation group.
143
144 // This code is effectively the equivalent of GetNodeAttr() for a string
145 // array, but it avoids all internal allocations (the allocation of the
146 // backing store of the std::vector<string> as well as the copies of the
147 // strings within it). Instead, we combine the query of the colocation
148 // attribute with the calls to ColocateNodeToGroup.
149 bool found_spec = false;
150 const AttrValue* attr_value =
151 node->attrs().Find(kColocationAttrNameStringPiece);
152 if (attr_value != nullptr && attr_value->has_list()) {
153 for (const string& class_spec : attr_value->list().s()) {
154 StringPiece spec(class_spec);
155 if (str_util::ConsumePrefix(&spec,
156 kColocationGroupPrefixStringPiece)) {
157 found_spec = true;
158 TF_RETURN_IF_ERROR(
159 ColocateNodeToGroup(&colocation_group_root, node, spec));
160 }
161 }
162 }
163
164 if (!found_spec) {
165 // If the node does not specify a colocation group, then use the
166 // name of this node as the colocation group.
167 TF_RETURN_IF_ERROR(
168 ColocateNodeToGroup(&colocation_group_root, node, node->name()));
169 }
170 }
171
172 return Status::OK();
173 }
174
175 Status ColocateNodeToGroup(
176 std::unordered_map<StringPiece, const Node*, StringPieceHasher>*
177 colocation_group_root,
178 Node* node, StringPiece colocation_group) {
179 const Node*& root_node = (*colocation_group_root)[colocation_group];
180 if (root_node == nullptr) {
181 // This is the first node of the colocation group, so
182 // designate this node as the 'root' of that colocation group.
183 root_node = node;
184 } else {
185 // Try to colocate the node with the root. If there is an
186 // error, return it.
187 Status s = ColocateNodes(*node, *root_node);
188 if (!s.ok()) {
189 return AttachDef(s, *node);
190 }
191 }
192 return Status::OK();
193 }
194
195 // Merge the (possibly disjoint) sets containing nodes "x" and
196 // "y". Returns OK if the all nodes in the union of these sets can
197 // be placed on the same device type.
198 //
199 // NOTE: If this method returns an error, *this is left in an undefined
200 // state.
201 Status ColocateNodes(const Node& x, const Node& y) {
202 int x_root = FindRoot(x.id());
203 int y_root = FindRoot(y.id());
204 return ColocateNodes(x, x_root, y, y_root);
205 }
206
207 // This overload of ColocateNodes() allows a caller to provide the root node
208 // ids for the two nodes. For large graphs, this noticeably reduces the
209 // graph load time.
210 Status ColocateNodes(const Node& x, int x_root, const Node& y, int y_root) {
211 if (x_root == y_root) {
212 return Status::OK();
213 }
214
215 DCHECK_EQ(x_root, FindRoot(x.id()));
216 DCHECK_EQ(y_root, FindRoot(y.id()));
217
218 Member& x_root_member = members_[x_root];
219 Member& y_root_member = members_[y_root];
220
221 // Merge the sets by swinging the parent pointer of the smaller
222 // tree to point to the root of the larger tree. Together with
223 // path compression in ColocationGraph::FindRoot, this ensures
224 // that we do not experience pathological performance on graphs
225 // such as chains.
226 int new_root, old_root;
227 if (x_root_member.rank < y_root_member.rank) {
228 // The tree rooted at x_root is shallower, so connect it to
229 // y_root. The rank of y_root is unchanged because its new
230 // child has strictly less rank.
231 x_root_member.parent = y_root;
232 new_root = y_root;
233 old_root = x_root;
234 } else if (x_root_member.rank > y_root_member.rank) {
235 // The tree rooted at y_root is shallower, so connect it to
236 // x_root. The rank of x_root is unchanged because its new
237 // child has strictly less rank.
238 y_root_member.parent = x_root;
239 new_root = x_root;
240 old_root = y_root;
241 } else {
242 // Both trees have the same rank, so break the tie by choosing
243 // x_root as the new root.
244 y_root_member.parent = x_root;
245 // Increment the rank of the tree rooted at x_root, because it
246 // is now strictly deeper than before.
247 ++x_root_member.rank;
248 new_root = x_root;
249 old_root = y_root;
250 }
251
252 Member& new_root_member = members_[new_root];
253 Member& old_root_member = members_[old_root];
254
255 // Merge the partial device specifications, and ensure that they are
256 // compatible. NULL options_ is treated as allowing soft placement.
257 // TODO(mrry): Consider enriching the error message by pointing
258 // out which nodes have the explicit partial device
259 // specifications that caused this conflict.
260 Status s = DeviceNameUtils::MergeDevNames(&new_root_member.device_name,
261 old_root_member.device_name,
262 allow_soft_placement_);
263 if (!s.ok()) {
264 return errors::InvalidArgument("Cannot colocate nodes '", x.name(),
265 "' and '", y.name(), ": ",
266 s.error_message());
267 }
268
269 // Ensure that the common root has at least one supported device
270 // type, by computing the intersection of
271 // new_root_member.supported_device_types and
272 // old_root_member.supported_device_types.
273 MergeSupportedDevices(&new_root_member.supported_device_types,
274 old_root_member.supported_device_types);
275 if (new_root_member.supported_device_types.empty()) {
276 return errors::InvalidArgument(
277 "Cannot colocate nodes '", x.name(), "' and '", y.name(),
278 "' because no device type supports both of those nodes and the "
279 "other nodes colocated with them.",
280 DebugInfo(x_root), DebugInfo(y_root));
281 }
282
283 return Status::OK();
284 }
285
286 // For the given node, subject to the constraints previously given
287 // to this ColocationGraph, set its assigned_device_name. Returns OK
288 // if a satisfying device can be found, otherwise an error.
289 //
290 // Note: This method returns a pointer to a field within members_.
291 // The caller must not use the returned pointer after there is any possibility
292 // that the members_[i].possible_devices field has been modified.
293 Status GetDevicesForNode(Node* node,
294 std::vector<Device*>** possible_devices) {
295 *possible_devices = nullptr;
296 const int node_root = FindRoot(node->id());
297 if (!members_[node_root].possible_devices.empty()) {
298 *possible_devices = &members_[node_root].possible_devices;
299 return Status::OK();
300 }
301
302 // We have not yet computed the possible devices for the
303 // colocated node set containing 'node', so we do so now using the
304 // constraints on the root node.
305
306 // "devices" will contain the set of feasible placements for the
307 // colocated node set containing 'node'.
308 std::vector<Device*> devices;
309 if (DeviceNameUtils::HasSomeDetails(members_[node_root].device_name)) {
310 // The root node has a (possibly partial) device
311 // specification, so enumerate the physical devices that
312 // conform to it.
313 device_set_->FindMatchingDevices(members_[node_root].device_name,
314 &devices);
315
316 if (!devices.empty()) {
317 // Filter devices into those that are compatible with the root
318 // node (and its children).
319 devices = FilterSupportedDevices(
320 devices, members_[node_root].supported_device_types);
321 }
322
323 // Perform soft placement if allow_soft_placement_ is set.
324 if (devices.empty() && allow_soft_placement_) {
325 // The soft_device_name is the same as the node's device name
326 // without specifying the device type or ID.
327 DeviceNameUtils::ParsedName soft_device_name =
328 members_[node_root].device_name;
329 soft_device_name.type.clear();
330 soft_device_name.has_type = false;
331 soft_device_name.has_id = false;
332 device_set_->FindMatchingDevices(soft_device_name, &devices);
333 if (!devices.empty()) {
334 devices = FilterSupportedDevices(
335 devices, members_[node_root].supported_device_types);
336 }
337 }
338
339 if (devices.empty()) {
340 // Return an error when a physical device that matches an explicit
341 // device specification is not found. This ensures that we don't
342 // assign a node to GPU when the user wanted to force it on CPU.
343 string debug_info = DebugInfo(node_root);
344
345 DeviceNameUtils::ParsedName specified_device_name;
346 if (DeviceNameUtils::ParseFullName(node->requested_device(),
347 &specified_device_name) &&
348 specified_device_name == members_[node_root].device_name) {
349 // The specified device and merged set device match, and
350 // will appear in the GraphDef (for debugging), so just
351 // print the specified device.
352 std::vector<Device*> devices_matching_nodedef;
353 device_set_->FindMatchingDevices(specified_device_name,
354 &devices_matching_nodedef);
355 if (devices_matching_nodedef.empty()) {
356 // Sometimes it is almost impossible to understand the problem
357 // without a list of available devices.
358 std::vector<string> device_names;
359 for (const Device* device : device_set_->devices()) {
360 device_names.push_back(device->name());
361 }
362 std::sort(device_names.begin(), device_names.end());
363
364 return errors::InvalidArgument(
365 "Operation was explicitly assigned to ",
366 node->requested_device(), " but available devices are [ ",
367 str_util::Join(device_names, ", "), " ]. Make sure ",
368 "the device specification refers to a valid device.");
369 } else if (specified_device_name.has_type) {
370 return errors::InvalidArgument(
371 "Could not satisfy explicit device specification '",
372 node->requested_device(), "' because no supported kernel for ",
373 specified_device_name.type, " devices is available.",
374 debug_info, "\nRegistered kernels:\n",
375 KernelsRegisteredForOp(node->type_string()));
376 } else {
377 return errors::InvalidArgument(
378 "Could not satisfy explicit device specification '",
379 node->requested_device(), debug_info);
380 }
381 } else {
382 // The specified device may be a valid device but the
383 // merged set device is different, so print both.
384 return errors::InvalidArgument(
385 "Could not satisfy explicit device specification '",
386 node->requested_device(),
387 "' because the node was colocated with a group of nodes that "
388 "required incompatible device '",
389 DeviceNameUtils::ParsedNameToString(
390 members_[node_root].device_name),
391 "'", debug_info);
392 }
393 }
394 } else {
395 // The device is completely unspecified, so enumerate the devices that
396 // support all of the nodes in the set.
397 if (device_set_->devices().empty()) {
398 return errors::Internal("No devices are registered");
399 }
400 devices = FilterSupportedDevices(
401 device_set_->devices(), members_[node_root].supported_device_types);
402
403 if (devices.empty()) {
404 return errors::InvalidArgument(
405 "Node had no OpKernel registered to support this operation: ",
406 "Operation was ", node->type_string(), " and inputs were ",
407 DataTypeVectorString(node->input_types()), DebugInfo(node_root));
408 }
409 }
410
411 // Cache the result of the possible devices for this node group.
412 members_[node_root].possible_devices = std::move(devices);
413 *possible_devices = &members_[node_root].possible_devices;
414 return Status::OK();
415 }

417 Status InitializeMembers() {
418 for (Node* node : graph_->nodes()) {
419 if (!node->IsOp()) {
420 continue;
421 }
422 Status status = InitializeMember(*node, &members_[node->id()]);
423 if (!status.ok()) {
424 return AttachDef(status, *node);
425 }
426 }
427 return Status::OK();
428 }

430 // Represents a node in the disjoint node set forest, and the
431 // accumulated constraints on the device used by that node.
432 struct Member {
433 Member() = default;
434 // The id of the node that is the parent of this one, or its own
435 // id if it is a root. parent <= 0 indicates that this member is invalid.
436 int parent = -1;
437
438 // A proxy for the depth of the tree that is used to prefer
439 // connecting smaller trees to larger trees when merging disjoint
440 // sets.
441 int rank = 0;
442
443 // The intersection of all device types supported by this node,
444 // and those of all of its children, in priority order
445 // of the preferred device.
446 DeviceTypeVector supported_device_types;
447
448 // The merged form of the device requested for this node, with
449 // those of all of its children.
450 DeviceNameUtils::ParsedName device_name;
451
452 // If this node is a root, stores a list of Devices to which this node
453 // and all of its children have been assigned, or nullptr if this
454 // has not yet been computed.
455 std::vector<Device*> possible_devices;
456 };
457
458 // Returns debugging info for the node referred to by 'node_root'.
459 string DebugInfo(const int node_root) {
460 string text(
461 "\nColocation Debug Info:\n"
462 "Colocation group had the following types and devices: ");
463
464 // If this node is part of a colocation group, then we want to
465 // collect the mapping of ops to supported devices, so that
466 // the user can see why an unsatisfiable placement occurred.
467
468 std::unordered_map<string, string> type_to_devices;
469 std::vector<const Node*> colocation_nodes;
470 int num_nodes_found = 0;
471
472 for (const Node* node : graph_->nodes()) {
473 if (!node->IsOp()) {
474 continue;
475 }
476 int id = node->id();
477 if (FindRoot(id) != node_root) {
478 continue;
479 }
480 ++num_nodes_found;
481 colocation_nodes.push_back(node);
482 const string& op_type = node->type_string();
483 string devices_registered;
484 for (const auto& device_type : members_[id].supported_device_types) {
485 strings::StrAppend(&devices_registered, DeviceTypeString(device_type),
486 " ");
487 }
488
489 type_to_devices[op_type] = std::move(devices_registered);
490 }
491
492 for (const auto& td : type_to_devices) {
493 strings::StrAppend(&text, "\n", td.first, ": ", td.second);
494 }
495 strings::StrAppend(&text,
496 "\n\nColocation members and user-requested devices:");
497 for (const Node* node : colocation_nodes) {
498 strings::StrAppend(&text, "\n ", node->name(), " (", node->type_string(),
499 ") ", node->requested_device());
500 }
501 strings::StrAppend(&text, "\n");
502
503 if (num_nodes_found <= 1) {
504 text.clear();
505 }
506 return text;
507 }

509 Status InitializeMember(const Node& node, Member* member) {
510 const int id = node.id();
511 DCHECK_GE(id, 0);
512 member->parent = id;
513 TF_RETURN_IF_ERROR(SupportedDeviceTypesForNode(
514 device_types_, node.def(), &member->supported_device_types));
515
516 if (node.has_assigned_device_name()) {
517 // This node has already been assigned to a device, so we
518 // respect this placement, after sanity-checking it. The
519 // device_name and supported_device_types for this node reflect
520 // the assigned device, so any nodes colocated with this node
521 // will be assigned to the same device (assuming this is
522 // possible).
523 // NOTE: Since any assignment must have been performed by
524 // the TensorFlow runtime, we consider errors in this branch to
525 // be INTERNAL.
526 const string& assigned_device_name = node.assigned_device_name();
527 if (!DeviceNameUtils::ParseFullName(assigned_device_name,
528 &member->device_name)) {
529 return errors::Internal("Malformed assigned device '",
530 assigned_device_name, "'");
531 }
532 const Device* assigned_device =
533 device_set_->FindDeviceByName(assigned_device_name);
534 if (assigned_device == nullptr) {
535 return errors::Internal("Assigned device '", assigned_device_name,
536 "' does not match any device");
537 }
538
539 for (const DeviceType& d : member->supported_device_types) {
540 if (DeviceType(assigned_device->attributes().device_type()) == d) {
541 return Status::OK();
542 }
543 }
544
545 return errors::Internal("Assigned device '", assigned_device_name,
546 "' does not have registered OpKernel support "
547 "for ",
548 node.type_string());
549 } else {
550 // This node has not yet been assigned to a device, so we
551 // calculate any constraints due to the set of registered
552 // kernels and any (partial) user-provided device specification
553 // in the NodeDef.
554
555 // If no kernels are registered for this op type, fail with an error.
556 if (member->supported_device_types.empty()) {
557 std::set<string> registered_device_types;
558 for (Device* d : device_set_->devices()) {
559 registered_device_types.insert(d->device_type());
560 }
561 return errors::InvalidArgument(
562 "No OpKernel was registered to support Op '", node.type_string(),
563 "' with these attrs. Registered devices: [",
564 str_util::Join(registered_device_types, ","),
565 "], Registered kernels:\n",
566 KernelsRegisteredForOp(node.type_string()));
567 }
568
569 // If the NodeDef contains a device, then we interpret it as a
570 // (partial) device specification.
571 if (!node.requested_device().empty()) {
572 // The user has specified a device in the NodeDef, try to find a
573 // valid device matching their specification in the set of
574 // devices.
575 // NOTE: The full name may specify a device that is not in
576 // n.supported_device_types(), but we check that in AssignDevice().
577 if (!DeviceNameUtils::ParseFullName(node.requested_device(),
578 &member->device_name)) {
579 return errors::InvalidArgument("Malformed device specification '",
580 node.requested_device(), "'");
581 }
582 }
583 }
584 return Status::OK();
585 }

587 // Updates target to contain the intersection of the device types in
588 // "target" and "other".
589 static void MergeSupportedDevices(DeviceTypeVector* target,
590 const DeviceTypeVector& other) {
591 DeviceTypeVector temp = *target;
592 target->clear();
593
594 // Iterate in priority order.
595 for (const DeviceType& device_type : temp) {
596 bool found = false;
597 for (const DeviceType& other_device_type : other) {
598 if (device_type == other_device_type) {
599 found = true;
600 break;
601 }
602 }
603 if (found) {
604 target->push_back(device_type);
605 }
606 }
607 }
608
609 // Returns the root node of the disjoint tree to which the node with the
610 // given id is connected.
611 int FindRoot(int node_id) {
612 Member& member = members_[node_id];
613
614 int parent = member.parent;
615 DCHECK_GE(parent, 0);
616
617 if (parent != node_id) {
618 // NOTE: Compress paths from node_id to its root, so that future
619 // calls to FindRoot and ColocateNodes are more efficient.
620 int root = FindRoot(parent);
621 if (parent != root) {
622 parent = root;
623 member.parent = root;
624 }
625 }
626
627 DCHECK_GE(parent, 0);
628 return parent;
629 }

631 Graph* const graph_; // Not owned.
632 std::vector<Member> members_;
633 const DeviceSet* device_set_; // Not owned.
634 const std::vector<DeviceType> device_types_;
635 const bool allow_soft_placement_;
636};

638// Returns true if the node has no inputs and produces outputs
639// that are consumed by a single node.
640//
641// TODO(vrv): Currently this handles only nodes with one output, but
642// this could be extended to handle the case where a node has many
643// outputs that are connected to nodes in the same colocation group.
644bool IsGeneratorNode(const Node* node) {
645 return node->num_inputs() == 0 && node->num_outputs() == 1 &&
646 !IsRefType(node->output_type(0));
647}

664Status Placer::Run() {
665 if (devices_->devices().empty()) {
666 return errors::FailedPrecondition("No devices are registered");
667 }
668
669 ColocationGraph colocation_graph(
670 graph_, devices_,
671 options_ == nullptr || options_->config.allow_soft_placement());
672
673 TF_RETURN_IF_ERROR(colocation_graph.InitializeMembers());
674
675 // 1. First add all of the nodes. Note that steps (1) and (2)
676 // requires two passes over the nodes because the graph (and hence
677 // the constraints) may not be acyclic.
678 TF_RETURN_IF_ERROR(colocation_graph.ColocateAllNodes());
679
680 // 2. Enumerate the constraint edges, and use them to update the disjoint
681 // node set.
682
683 // If `node` has an input edge with reference type, add an
684 // edge from the source of that edge to `node`.
685 for (const Edge* edge : graph_->edges()) {
686 if (edge->IsControlEdge()) {
687 continue;
688 }
689 Node* src = edge->src();
690 Node* dst = edge->dst();
691 DataType input_type = dst->input_type(edge->dst_input());
692 if (input_type == DT_RESOURCE || IsRefType(input_type)) {
693 int src_root_id = colocation_graph.FindRoot(src->id());
694 int dst_root_id = colocation_graph.FindRoot(dst->id());
695 auto& src_root = colocation_graph.members_[src_root_id];
696 auto& dst_root = colocation_graph.members_[dst_root_id];
697 // If both the source node and this node have partially
698 // specified a device, then 'node's device should be
699 // cleared: the reference edge forces 'node' to be on the
700 // same device as the source node.
701 const auto& source_parsed_name = src_root.device_name;
702 const auto& dest_parsed_name = dst_root.device_name;
703 if (DeviceNameUtils::HasSomeDetails(source_parsed_name) &&
704 DeviceNameUtils::HasSomeDetails(dest_parsed_name)) {
705 // Ignore a specified device for 'dst' if the two names were
706 // incompatible.
707 if (!DeviceNameUtils::AreCompatibleDevNames(source_parsed_name,
708 dest_parsed_name)) {
709 if (log_device_placement_) {
710 LOG(INFO) << "Ignoring device specification "
711 << DeviceNameUtils::ParsedNameToString(dest_parsed_name)
712 << " for node '" << dst->name()
713 << "' because the input edge from '" << src->name()
714 << "' is a reference connection and already has a device "
715 "field set to "
716 << DeviceNameUtils::ParsedNameToString(
717 source_parsed_name);
718 }
719
720 // Make 'dst' colocated with the source
721 dst_root.device_name = source_parsed_name;
722 } else {
723 bool source_subset_of_dest = DeviceNameUtils::IsSpecification(
724 source_parsed_name, dest_parsed_name);
725 bool dest_subset_of_source = DeviceNameUtils::IsSpecification(
726 dest_parsed_name, source_parsed_name);
727
728 if (source_subset_of_dest && !dest_subset_of_source) {
729 src_root.device_name = dest_parsed_name;
730 } else {
731 dst_root.device_name = source_parsed_name;
732 }
733 }
734 }
735
736 Status status =
737 colocation_graph.ColocateNodes(*src, src_root_id, *dst, dst_root_id);
738 if (!status.ok()) {
739 return AttachDef(
740 errors::InvalidArgument("Nodes were connected by a "
741 "reference connection (requiring them to "
742 "be on the same device), but the two nodes "
743 "were assigned two different devices: ",
744 status.error_message()),
745 *dst);
746 }
747 }
748 }
749
750 // 3. For each node, assign a device based on the constraints in the
751 // disjoint node set.
752 std::vector<Node*> second_pass;
753 for (Node* node : graph_->op_nodes()) {
754 // The graph may have come pre-populated by the framework with assigned
755 // devices (e.g., for stateful placements), so the placer should not try to
756 // place nodes that are already placed.
757 if (node->has_assigned_device_name()) {
758 LogDeviceAssignment(node);
759 continue;
760 }
761
762 // Heuristic A: prefer to place "generators" with their only
763 // consumers.
764 //
765 // If this is a node with no inputs and one output, we save
766 // this for a second pass, so that the consumer's placement
767 // is chosen.
768 if (IsGeneratorNode(node)) {
769 second_pass.push_back(node);
770 continue;
771 }
772
773 std::vector<Device*>* devices;
774 Status status = colocation_graph.GetDevicesForNode(node, &devices);
775 if (!status.ok()) {
776 return AttachDef(
777 errors::InvalidArgument("Cannot assign a device for operation '",
778 node->name(), "': ", status.error_message()),
779 *node);
780 }
781
782 // Returns the first device in sorted devices list so we will always
783 // choose the same device.
784 //
785 // TODO(vrv): Factor this assignment out into a pluggable
786 // algorithm, so that Placer is responsible for enforcing
787 // preconditions and we can experiment with other algorithms when
788 // given a choice of devices. Once we have a better idea of the
789 // types of heuristics we want to use and the information needed
790 // to perform good placement we can add an interface for this.
791 int assigned_device = -1;
792
793 // Heuristic B: If the node only operates on metadata, not data,
794 // then it is desirable to place that metadata node with its
795 // input.
796 if (IsMetadata(node)) {
797 // Make sure that the input device type is in the list of supported
798 // device types for this node.
799 const Node* input = (*node->in_edges().begin())->src();
800 // TODO(vrv): if the input is empty, consider postponing this
801 // node's assignment to the second pass, so that we handle the
802 // case where a metadata node's input comes from a backedge
803 // of a loop.
804 if (CanAssignToDevice(input->assigned_device_name(), *devices)) {
805 assigned_device = input->assigned_device_name_index();
806 }
807 }
808
809 // Provide the default, if necessary.
810 if (assigned_device == -1) {
811 assigned_device = graph_->InternDeviceName((*devices)[0]->name());
812 }
813
814 AssignAndLog(assigned_device, node);
815 }
816
817 // 4. Perform a second pass assignment for those nodes explicitly
818 // skipped during the first pass.
819 for (Node* node : second_pass) {
820 std::vector<Device*>* devices;
821 Status status = colocation_graph.GetDevicesForNode(node, &devices);
822 if (!status.ok()) {
823 return AttachDef(
824 errors::InvalidArgument("Cannot assign a device for operation '",
825 node->name(), "': ", status.error_message()),
826 *node);
827 }
828
829 int assigned_device = -1;
830
831 // Heuristic A application.
832 if (IsGeneratorNode(node)) {
833 const Node* output = (*node->out_edges().begin())->dst();
834 int output_device_name = output->assigned_device_name_index();
835
836 const bool consumers_on_same_device = std::all_of(
837 node->out_edges().begin(), node->out_edges().end(),
838 [output_device_name](const Edge* e) {
839 return e->dst()->assigned_device_name_index() == output_device_name;
840 });
841
842 if (consumers_on_same_device &&
843 CanAssignToDevice(output->assigned_device_name(), *devices)) {
844 assigned_device = output_device_name;
845 }
846 }
847
848 // Provide the default, if necessary.
849 if (assigned_device == -1) {
850 assigned_device = graph_->InternDeviceName((*devices)[0]->name());
851 }
852
853 AssignAndLog(assigned_device, node);
854 }
855
856 return Status::OK();
857}
858
859bool Placer::CanAssignToDevice(const string& candidate_device_name,
860 const std::vector<Device*>& devices) const {
861 if (!candidate_device_name.empty()) {
862 // 'devices' lists the set of devices that the placer or the user has
863 // constrained the operation to. "candidate_device_name" must
864 // refer to a concrete Device that is in the list of 'devices'.
865 const Device* other_device =
866 devices_->FindDeviceByName(candidate_device_name);
867 if (std::find(devices.begin(), devices.end(), other_device) !=
868 devices.end()) {
869 return true;
870 }
871 }
872
873 return false;
874}
875
876void Placer::AssignAndLog(int assigned_device, Node* node) const {
877 node->set_assigned_device_name_index(assigned_device);
878 LogDeviceAssignment(node);
879}
880
881void Placer::LogDeviceAssignment(const Node* node) const {
882 // Log placement if log_device_placement is set.
883 if (log_device_placement_) {
884 printf("%s: (%s): %s\n", node->name().c_str(), node->type_string().c_str(),
885 node->assigned_device_name().c_str());
886 LOG(INFO) << node->name() << ": "
887 << "(" << node->type_string() << ")"
888 << node->assigned_device_name();
889 }
890}
