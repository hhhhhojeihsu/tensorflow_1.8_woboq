33static thread::ThreadPool* InitComputePool(const SessionOptions& options) {
34 int32 inter_op_parallelism_threads =
35 options.config.inter_op_parallelism_threads();
36 if (inter_op_parallelism_threads == 0) {
37 // Default to using the number of cores available in the process.
38 inter_op_parallelism_threads = port::NumSchedulableCPUs();
39 }
40
41 return new thread::ThreadPool(Env::Default(), "Compute",
42 inter_op_parallelism_threads);
43}
44
45} // namespace
46
47thread::ThreadPool* ComputePool(const SessionOptions& options) {
48 static thread::ThreadPool* compute_pool = InitComputePool(options);
49 return compute_pool;
50}

52int32 NumInterOpThreadsFromSessionOptions(const SessionOptions& options) {
53 const int32 inter_op = options.config.inter_op_parallelism_threads();
54 if (inter_op != 0) return inter_op;
55#ifdef INTEL_MKL
56 // MKL library executes ops in parallel using OMP threads
57 // Set inter_op conservatively to avoid thread oversubscription that could
58 // lead to severe perf degradations and OMP resource exhaustion
59 const int mkl_intra_op = omp_get_max_threads();
60 CHECK_GE(mkl_intra_op, 1);
61 const int32 mkl_inter_op = std::max(
62 (port::NumSchedulableCPUs() + mkl_intra_op - 1) / mkl_intra_op, 2);
63 VLOG(0) << "Creating new thread pool with default inter op setting: "
64 << mkl_inter_op
65 << ". Tune using inter_op_parallelism_threads for best performance.";
66 return mkl_inter_op;
67#else
68 // Default to using the number of cores available in the process.
69 return port::NumSchedulableCPUs();
70#endif
71}
72
73thread::ThreadPool* NewThreadPoolFromSessionOptions(
74 const SessionOptions& options) {
75 const int32 num_threads = NumInterOpThreadsFromSessionOptions(options);
76 VLOG(1) << "Direct session inter op parallelism threads: " << num_threads;
77 return new thread::ThreadPool(options.env, "Compute", num_threads);
78}

80void SchedClosure(std::function<void()> closure) {
81 if (port::Tracing::IsActive()) {
82 const uint64 id = port::Tracing::UniqueId();
83 port::Tracing::RecordEvent(port::Tracing::EventCategory::kScheduleClosure,
84 id);
85 std::function<void()> wrapper = std::bind(
86 [id](std::function<void()> closure) {
87 port::Tracing::ScopedActivity region(
88 port::Tracing::EventCategory::kRunClosure, id);
89 closure();
90 },
91 std::move(closure));
92 Env::Default()->SchedClosure(std::move(wrapper));
93 } else {
94 Env::Default()->SchedClosure(std::move(closure));
95 }
96}
97
98void SchedNonBlockingClosureAfter(int64 micros, std::function<void()> closure) {
99 Env::Default()->SchedClosureAfter(micros, std::move(closure));
100}
