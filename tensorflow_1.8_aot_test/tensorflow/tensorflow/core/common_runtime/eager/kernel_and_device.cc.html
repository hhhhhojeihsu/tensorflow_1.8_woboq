32// static
33Status KernelAndDevice::InitOp(Device* device, const NodeDef& ndef,
34 KernelAndDevice* out) {
35 OpKernel* k = nullptr;
36 Status s = CreateOpKernel(device->device_type().c_str(), device,
37 device->GetAllocator(AllocatorAttributes()),
38 nullptr, ndef, TF_GRAPH_DEF_VERSION, &k);
39 out->device_ = device;
40 out->kernel_.reset(k);
41 out->flib_ = nullptr;
42 return s;
43}
44
45// static
46Status KernelAndDevice::Init(const NodeDef& ndef, FunctionLibraryRuntime* flib,
47 KernelAndDevice* out) {
48 OpKernel* k = nullptr;
49 Status s = flib->CreateKernel(ndef, &k);
50 out->device_ = flib->device();
51 out->kernel_.reset(k);
52 out->flib_ = flib;
53 return s;
54}

56Status KernelAndDevice::Run(std::vector<Tensor>* input_tensors,
57 std::vector<Tensor>* output_tensors,
58 NodeExecStats* stats) {
59 gtl::InlinedVector<TensorValue, 4> inputs;
60 for (Tensor& t : *input_tensors) {
61 inputs.push_back(TensorValue(&t));
62 }
63
64 std::vector<AllocatorAttributes> out_attrs(kernel_->num_outputs());
65 for (size_t i = 0; i < out_attrs.size(); ++i) {
66 out_attrs[i].set_on_host(kernel_->output_memory_types()[i] ==
67 tensorflow::HOST_MEMORY);
68 }
69
70 OpKernelContext::Params params;
71 params.device = device_;
72 params.frame_iter = FrameAndIter(0, 0);
73 params.inputs = &inputs;
74 params.op_kernel = kernel_.get();
75 params.resource_manager = device_->resource_manager();
76 params.output_attr_array = gtl::vector_as_array(&out_attrs);
77 params.function_library = flib_;
78 params.slice_reader_cache = &slice_reader_cache_;
79 params.rendezvous = rendez_;
80 if (stats != nullptr) {
81 params.track_allocations = true;
82 }
83 // TODO(apassos): use a thread pool.
84 std::function<void(std::function<void()>)> runner =
85 [](std::function<void()> f) { f(); };
86 params.runner = &runner;
87
88 OpKernelContext context(&params);
89
90 if (kernel_->def().op() == "_Recv") {
91 // TODO(apassos) do not special-case _Recv. Currently the GPU device fails
92 // if trying to run _Recv->Compute(), specifically checking for _Recv. To go
93 // around this we call _Recv->ComputeAsync, to mimic graph mode behavior.
94 AsyncOpKernel* async = kernel_->AsAsync();
95 Notification done;
96 device_->ComputeAsync(async, &context, [&done]() { done.Notify(); });
97 done.WaitForNotification();
98 } else {
99 device_->Compute(kernel_.get(), &context);
100 }
101 if (!context.status().ok()) return context.status();
102
103 output_tensors->clear();
104 for (int i = 0; i < context.num_outputs(); ++i) {
105 output_tensors->push_back(Tensor(*context.mutable_output(i)));
106 }
107 if (stats != nullptr) {
108 for (const auto& allocator_pair : context.wrapped_allocators()) {
109 AllocatorMemoryUsed* memory = stats->add_memory();
110 memory->set_allocator_name(allocator_pair.first->Name());
111 auto sizes = allocator_pair.second->GetSizes();
112 memory->set_total_bytes(std::get<0>(sizes));
113 memory->set_peak_bytes(std::get<1>(sizes));
114 memory->set_live_bytes(std::get<2>(sizes));
115
116 AllocatorStats allocator_stats;
117 allocator_pair.first->GetStats(&allocator_stats);
118 memory->set_allocator_bytes_in_use(allocator_stats.bytes_in_use);
119 allocator_pair.second->GetRecordsAndUnRef();
120 }
121 auto* ms = stats->mutable_memory_stats();
122 ms->set_temp_memory_size(context.temp_memory_allocated());
123 for (const auto& alloc_id : context.persistent_alloc_ids()) {
124 ms->mutable_persistent_tensor_alloc_ids()->Add(alloc_id);
125 }
126
127 ms->set_persistent_memory_size(context.persistent_memory_allocated());
128 }
129 return Status::OK();
130}
