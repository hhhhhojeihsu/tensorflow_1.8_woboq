30namespace tensorflow {
31
32class ExecuteNode : public EagerNode {
33 public:
34 ExecuteNode(uint64 id, EagerContext* ctx, Device* op_device,
35 const tensorflow::gtl::InlinedVector<TensorHandle*, 4>& inputs,
36 KernelAndDevice* kernel, NodeExecStats* maybe_stats,
37 const DataTypeVector& output_dtypes,
38 const tensorflow::gtl::InlinedVector<TensorHandle*, 2>& retvals)
39 : EagerNode(id),
40 ctx_(ctx),
41 op_device_(op_device),
42 inputs_(inputs),
43 kernel_(kernel),
44 maybe_stats_(maybe_stats),
45 retvals_(retvals) {
46 for (auto handle : inputs_) {
47 handle->Ref();
48 }
49 for (auto handle : retvals_) {
50 handle->Ref();
51 }
52 }
53
54 ~ExecuteNode() override {
55 for (auto handle : inputs_) {
56 handle->Unref();
57 }
58 for (auto handle : retvals_) {
59 handle->Unref();
60 }
61 }
62
63 tensorflow::Status Run() override {
64 const Status status =
65 EagerExecute(ctx_, op_device_, inputs_, kernel_, maybe_stats_.get(),
66 retvals_.begin(), retvals_.size());
67 if (status.ok()) {
68 return status;
69 } else {
70 return Status(status.code(),
71 strings::StrCat("Got error, \"", status.error_message(),
72 "\" while executing kernel ",
73 kernel_->kernel()->def().DebugString()));
74 }
75 }
76
77 private:
78 tensorflow::EagerContext* ctx_;
79 tensorflow::Device* op_device_;
80 tensorflow::gtl::InlinedVector<TensorHandle*, 4> inputs_;
81 tensorflow::KernelAndDevice* kernel_;
82 std::unique_ptr<NodeExecStats> maybe_stats_;
83 tensorflow::gtl::InlinedVector<TensorHandle*, 2> retvals_;
84};
