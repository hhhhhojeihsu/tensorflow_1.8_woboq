35Status EagerExecute(EagerContext* ctx, Device* device,
36 const gtl::InlinedVector<TensorHandle*, 4>& op_inputs,
37 KernelAndDevice* kernel, NodeExecStats* maybe_stats,
38 TensorHandle** retvals, int num_retvals) {
39 if (device == nullptr) {
40 // TODO(apassos) debug how the assignment below might return a different
41 // device from the one requested above.
42 device = kernel->device();
43 }
44
45 std::vector<Tensor> outputs(1);
46 const MemoryTypeVector* output_memory_types = nullptr;
47 output_memory_types = &kernel->kernel()->output_memory_types();
48 std::vector<Tensor> inputs(op_inputs.size());
49 for (int i = 0; i < op_inputs.size(); ++i) {
50 const Tensor* input_tensor = nullptr;
51 TF_RETURN_IF_ERROR(op_inputs[i]->Tensor(&input_tensor));
52 inputs[i] = *input_tensor;
53 }
54 // WARNING: kernel->Run utilizes the FunctionLibraryRuntime
55 // (ctx->func_lib(device)), which in turn holds a pointer to func_lib_def.
56 // But knowledge of the implementation
57 // of FunctionLibraryRuntime tells us that func_lib_def is not accessed by
58 // FunctionLibraryRuntime::Run(), so there is no thread-safety concern here.
59 // This is quite subtle. Re-work things to make this better? (Would it make
60 // sense for FunctionLibraryRuntime to ensure thread-safe access to
61 // FunctionLibraryDefinition?). TODO(apassos) figure out how to record stats
62 // for ops which are a part of functions.
63 // TODO(agarwal): change Run to take vector of handles ?
64 TF_RETURN_IF_ERROR(kernel->Run(&inputs, &outputs, maybe_stats));
65 if (maybe_stats != nullptr) {
66 maybe_stats->set_op_end_rel_micros(Env::Default()->NowMicros() -
67 maybe_stats->all_start_micros());
68 mutex_lock ml(*ctx->MetadataMu());
69 if (ctx->ShouldStoreMetadata()) {
70 auto* step_stats = ctx->RunMetadataProto()->mutable_step_stats();
71 // Lazily initialize the RunMetadata with information about all devices if
72 // this is the first call.
73 while (step_stats->dev_stats_size() < ctx->devices()->size()) {
74 step_stats->add_dev_stats();
75 }
76 // Find the current device's index.
77 int device_idx = 0;
78 for (int i = 0; i < ctx->devices()->size(); ++i) {
79 if (ctx->devices()->at(i) == device) {
80 device_idx = i;
81 break;
82 }
83 }
84 // Populate the device stats for this device.
85 auto* dev_stats = step_stats->mutable_dev_stats(device_idx);
86 dev_stats->set_device(device->name());
87 *dev_stats->add_node_stats() = *maybe_stats;
88 }
89 }
90 DCHECK_EQ(num_retvals, outputs.size());
91 Device* op_device = device;
92 for (int i = 0; i < num_retvals; ++i) {
93 Device* d = op_device;
94 if (d != nullptr && output_memory_types != nullptr &&
95 (*output_memory_types)[i] == HOST_MEMORY) {
96 d = nullptr;
97 }
98 if (retvals[i] == nullptr) {
99 retvals[i] = new TensorHandle(outputs[i], d, op_device, ctx);
100 } else {
101 retvals[i]->SetTensorAndDevice(outputs[i], d, op_device);
102 }
103 }
104 return Status::OK();
105}

107Status EagerCopyToDevice(TensorHandle* h, EagerContext* ctx,
108 const char* device_name, TensorHandle** result) {
109 TF_RETURN_IF_ERROR(ctx->GetStatus());
110 Device* dstd = ctx->HostCPU();
111 if (device_name != nullptr && strlen(device_name) > 0) {
112 TF_RETURN_IF_ERROR(ctx->device_mgr()->LookupDevice(device_name, &dstd));
113 }
114 if (ctx->Async()) {
115 // Note that `h` may not be currently ready. However execution order will
116 // make sure that `h` is ready before the copy is actually done.
117 CopyToDeviceNode* node = new CopyToDeviceNode(h, dstd, ctx);
118 TensorHandle* output = node->dst();
119 // Note that calling Add makes `node` accessible by the EagerExecutor
120 // thread. So further accesses need to be thread-safe.
121 ctx->ExecutorAdd(node);
122 *result = output;
123 return Status::OK();
124 } else {
125 TF_RETURN_IF_ERROR(h->CopyToDevice(ctx, dstd, result));
126 return Status::OK();
127 }
128}
