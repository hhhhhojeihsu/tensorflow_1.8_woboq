29class Allocator;
30class AllocatorMemoryUsed;
31class CostModelManager;
32class Graph;
33class NodeExecStats;
34class StepStats;
35class TrackingAllocator;
36
37// Wraps NodeExecStats and adds allocation to it.
38class NodeExecStatsWrapper {
39 public:
40 NodeExecStatsWrapper();
41 // Owns 'stats'.
42 NodeExecStatsWrapper(NodeExecStats* stats);
43
44 // Destructor calls Finalize() to release the TrackingAllocators.
45 ~NodeExecStatsWrapper() { Finalize(); }
46
47 NodeExecStats* stats() { return stats_.get(); }
48
49 // "Does not take ownership of the 'allocator'.
50 // Transfers ownership of the 'tracking_allocator' to *this."
51 void AddAllocation(Allocator* allocator,
52 TrackingAllocator* tracking_allocator);
53
54 private:
55 friend class StepStatsCollector;
56
57 // Populates stats_ and releases TrackingAllocator.
58 void Finalize();
59
60 gtl::InlinedVector<std::pair<AllocatorMemoryUsed*, TrackingAllocator*>, 2>
61 allocations_;
62 std::unique_ptr<NodeExecStats> stats_;
63};
64
65// StepStatsCollector manages the collection of a StepStats object.
66// The StepStats object holds multiple DeviceStats.
67// Each DeviceStats object holds multiple NodeExecStats.
68class StepStatsCollector {
69 public:
70 // Does not take ownership of `ss`.
71 explicit StepStatsCollector(StepStats* ss);
72
73 // BuildCostModel builds or updates a CostModel managed by cost_model_manager,
74 // using the currently collected DeviceStats associated with the devices in
75 // device_map.
76 void BuildCostModel(
77 CostModelManager* cost_model_manager,
78 const std::unordered_map<string, const Graph*>& device_map);
79
80 // Save saves nt to the DeviceStats object associated with device.
81 // Should be called before Finalize.
82 void Save(const string& device, NodeExecStats* nt);
83 void Save(const string& device, NodeExecStatsWrapper* stats);
84
85 // Generates a string reporting the currently used memory based
86 // on ResourceExhausted OOM `err` message.
87 // `err` message needs to contain device name and allocator name, E.g.:
88 // "ResourceExhaustedError: OOM when allocating tensor ...
89 // on /job:localhost/replica:0/task:0/device:GPU:0 by allocator GPU_0_bfc"
90 string ReportAllocsOnResourceExhausted(const string& err);
91
92 // The following 2 Finalize methods populate the StepStats passed
93 // from the constructor. Calling it more than once won't have any effect.
94 // User shouldn't call Save() methods after Finalize.
95 void Finalize();
96 // swaps the content of StepStats* from constructor with 'ss'.
97 void FinalizeAndSwap(StepStats* ss);
98
99 private:
100 void FinalizeInternal() EXCLUSIVE_LOCKS_REQUIRED(mu_);
101
102 typedef std::vector<std::unique_ptr<NodeExecStatsWrapper>> NodeExecStatsVec;
103 // TODO(suharshs): Make this configurable if its not possible to find a value
104 // that works for all cases.
105 const uint64 kMaxCollectedNodes = 1 << 20;
106 mutex mu_;
107 bool finalized_ GUARDED_BY(mu_);
108 std::unordered_map<string, NodeExecStatsVec> dev_stats_ GUARDED_BY(mu_);
109 StepStats* step_stats_ GUARDED_BY(mu_);
110 uint64 collectedNodes GUARDED_BY(mu_) = 0;
111};
