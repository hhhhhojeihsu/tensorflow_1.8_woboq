54GraphExecutionState::GraphExecutionState(
55 GraphDef* graph_def, const GraphExecutionStateOptions& options)
56 : stateful_placements_(options.stateful_placements),
57 device_set_(options.device_set),
58 session_options_(options.session_options),
59 flib_def_(new FunctionLibraryDefinition(OpRegistry::Global(),
60 graph_def->library())),
61 graph_(nullptr) {
62 // NOTE(mrry): GraphDef does not have a move constructor, so we pass
63 // a non-const pointer and use `Swap()` to transfer the contents
64 // without copying.
65 original_graph_def_.Swap(graph_def);
66 // TODO(mrry): Publish placement visualizations or handle the log
67 // placement option.
68}

75/* static */ Status GraphExecutionState::MakeForBaseGraph(
76 GraphDef* graph_def, const GraphExecutionStateOptions& options,
77 std::unique_ptr<GraphExecutionState>* out_state) {
78#ifndef __ANDROID__
79 VLOG(1) << "Graph proto is " << graph_def->DebugString();
80#endif // __ANDROID__
81
82 std::unique_ptr<GraphExecutionState> ret(
83 new GraphExecutionState(graph_def, options));
84
85 TF_RETURN_IF_ERROR(
86 AddDefaultAttrsToGraphDef(&ret->original_graph_def_, *ret->flib_def_, 0));
87 // TODO(mrry): Refactor InitBaseGraph() so that we don't have to
88 // pass an empty BuildGraphOptions (that isn't going to be used when
89 // place_pruned_graph is false).
90 if (!ret->session_options_->config.graph_options().place_pruned_graph()) {
91 TF_RETURN_IF_ERROR(ret->InitBaseGraph(BuildGraphOptions()));
92 }
93 *out_state = std::move(ret);
94 return Status::OK();
95}

97/* static */ Status GraphExecutionState::MakeForPrunedGraph(
98 const FunctionDefLibrary& func_def_lib,
99 const GraphExecutionStateOptions& options, const GraphDef& graph_def,
100 const BuildGraphOptions& subgraph_options,
101 std::unique_ptr<GraphExecutionState>* out_state,
102 std::unique_ptr<ClientGraph>* out_client_graph) {
103 DCHECK(options.session_options->config.graph_options().place_pruned_graph());
104 // NOTE(mrry): This makes a copy of `graph_def`, which is
105 // regrettable. We could make `GraphDef` objects sharable between
106 // execution states to optimize pruned graph execution, but since
107 // this case is primarily used for interactive sessions, we make the
108 // bet that graph construction is not performance-critical. (Note
109 // also that the previous version used `Extend()`, which is strictly
110 // more expensive than copying a `GraphDef`.)
111 GraphDef temp(graph_def);
112 std::unique_ptr<GraphExecutionState> ret(
113 new GraphExecutionState(&temp, options));
114 TF_RETURN_IF_ERROR(
115 AddDefaultAttrsToGraphDef(&ret->original_graph_def_, *ret->flib_def_, 0));
116 TF_RETURN_IF_ERROR(ret->InitBaseGraph(subgraph_options));
117 TF_RETURN_IF_ERROR(ret->BuildGraph(subgraph_options, out_client_graph));
118 *out_state = std::move(ret);
119 return Status::OK();
120}

122Status GraphExecutionState::Extend(
123 const GraphDef& extension_def,
124 std::unique_ptr<GraphExecutionState>* out) const {
125 GraphDef gdef;
126
127 // 1. Copy the function library.
128 TF_RETURN_IF_ERROR(flib_def_->AddLibrary(extension_def.library()));
129 *gdef.mutable_library() = flib_def_->ToProto();
130
131 // 2. Build an index of the new node names.
132 std::unordered_set<string> new_names;
133 for (const NodeDef& node : extension_def.node()) {
134 new_names.insert(node.name());
135 }
136
137 // 3. Add the non-duplicates from the old graph to the new graph.
138 // Return an error if the same node name appears in both the
139 // old graph and the extension.
140 for (const NodeDef& node : original_graph_def_.node()) {
141 if (new_names.count(node.name()) == 0) {
142 *gdef.add_node() = node;
143 } else {
144 return errors::InvalidArgument(tensorflow::strings::Printf(
145 "GraphDef argument to Extend includes node '%s', which was created "
146 "by a previous call to Create or Extend in this session.",
147 node.name().c_str()));
148 }
149 }
150
151 // 4. Merge the versions field.
152 int old_node_size = gdef.node_size();
153 gdef.mutable_node()->MergeFrom(extension_def.node());
154 TF_RETURN_IF_ERROR(
155 AddDefaultAttrsToGraphDef(&gdef, *flib_def_, old_node_size));
156 // Merge versions
157 if (gdef.has_versions()) {
158 if (gdef.versions().producer() != extension_def.versions().producer()) {
159 return errors::InvalidArgument(
160 "Can't extend GraphDef at version ", gdef.versions().producer(),
161 " with graph at version ", extension_def.versions().producer());
162 }
163 VersionDef* versions = gdef.mutable_versions();
164 versions->set_min_consumer(std::max(
165 versions->min_consumer(), extension_def.versions().min_consumer()));
166 if (extension_def.versions().bad_consumers_size()) {
167 // Add new bad_consumers that aren't already marked bad.
168 //
169 // Note: This implementation is quadratic time if there are many calls to
170 // ExtendLocked with many bad consumers. Since this is unlikely, and
171 // fixing it would require data structures outside of this routine,
172 // quadratic time it is.
173 auto* bad_consumers = versions->mutable_bad_consumers();
174 const std::unordered_set<int> existing(bad_consumers->begin(),
175 bad_consumers->end());
176 for (const int v : extension_def.versions().bad_consumers()) {
177 if (existing.find(v) == existing.end()) {
178 bad_consumers->Add(v);
179 }
180 }
181 }
182
183 } else {
184 gdef.mutable_versions()->CopyFrom(extension_def.versions());
185 }
186
187 // 5. Validate that the final graphdef is valid.
188 if (gdef.versions().producer() >= 5) {
189 // Validate the graph: we assume that merging two valid graphs
190 // should maintain graph validity.
191 TF_RETURN_IF_ERROR(graph::ValidateGraphDef(gdef, *flib_def_));
192 }
193
194 // 6. Add the extension.
195 GraphExecutionStateOptions combined_options;
196 combined_options.device_set = device_set_;
197 combined_options.session_options = session_options_;
198 combined_options.stateful_placements = stateful_placements_;
199
200 // NOTE(mrry): `gdef` is no longer valid after the constructor
201 // executes.
202 std::unique_ptr<GraphExecutionState> new_execution_state(
203 new GraphExecutionState(&gdef, combined_options));
204
205 TF_RETURN_IF_ERROR(AddDefaultAttrsToGraphDef(
206 &new_execution_state->original_graph_def_, *flib_def_, 0));
207 if (!session_options_->config.graph_options().place_pruned_graph()) {
208 // TODO(mrry): Refactor InitBaseGraph() so that we don't have to
209 // pass an empty BuildGraphOptions (that isn't going to be used
210 // when place_pruned_graph is false).
211 TF_RETURN_IF_ERROR(new_execution_state->InitBaseGraph(BuildGraphOptions()));
212 }
213 *out = std::move(new_execution_state);
214
215 // TODO(mrry): This is likely to be used for non-throughput-sensitive
216 // interactive workloads, but in future we may want to transfer other
217 // parts of the placement and/or cost model.
218 return Status::OK();
219}

221void GraphExecutionState::SaveStatefulNodes(Graph* graph) {
222 for (Node* n : graph->nodes()) {
223 if (n->op_def().is_stateful()) {
224 VLOG(2) << "Saving " << n->DebugString();
225 stateful_placements_[n->name()] = n->assigned_device_name();
226 }
227 }
228}
229
230void GraphExecutionState::RestoreStatefulNodes(Graph* graph) {
231 for (Node* n : graph->nodes()) {
232 if (n->op_def().is_stateful()) {
233 auto iter = stateful_placements_.find(n->name());
234 if (iter != stateful_placements_.end()) {
235 n->set_assigned_device_name(iter->second);
236 VLOG(2) << "Restored " << n->DebugString();
237 }
238 }
239 }

244class TensorConnectionPruneRewrite : public subgraph::PruneRewrite {
245 public:
246 TensorConnectionPruneRewrite(const string* endpoint_name,
247 NodeBuilder::NodeOut from_tensor)
248 : subgraph::PruneRewrite(endpoint_name, nullptr /* device_info */),
249 from_tensor_(std::move(from_tensor)) {}
250
251 Status AddNode(Graph* g, NodeBuilder::NodeOut feed_tensor,
252 Node** out_node) override {
253 Status s;
254 auto check_no_cycle_fn = [this, feed_tensor, &s](Node* n) {
255 if (n == feed_tensor.node) {
256 s.Update(errors::InvalidArgument(
257 "Requested Tensor connection between nodes \"",
258 feed_tensor.node->name(), "\" and \"", from_tensor_.node->name(),
259 "\" would create a cycle."));
260 }
261 };
262 ReverseDFSFrom(*g, {from_tensor_.node}, std::move(check_no_cycle_fn),
263 nullptr);
264 TF_RETURN_IF_ERROR(s);
265
266 TF_RETURN_IF_ERROR(
267 NodeBuilder(strings::StrCat("_identity_", feed_tensor.node->name(), "_",
268 feed_tensor.index),
269 "Identity")
270 .Input(from_tensor_)
271 .Attr("T",
272 BaseType(from_tensor_.node->output_type(from_tensor_.index)))
273 .Finalize(g, out_node));
274
275 (*out_node)->set_assigned_device_name(
276 feed_tensor.node->assigned_device_name());
277 return Status::OK();
278 }
279
280 private:
281 NodeBuilder::NodeOut from_tensor_;
282};

286Status GraphExecutionState::PruneGraph(
287 const BuildGraphOptions& options, Graph* graph,
288 subgraph::RewriteGraphMetadata* out_rewrite_metadata) {
289 std::vector<std::unique_ptr<subgraph::PruneRewrite>> feed_rewrites;
290 feed_rewrites.reserve(options.callable_options.feed_size());
291 std::vector<std::unique_ptr<subgraph::PruneRewrite>> fetch_rewrites;
292 fetch_rewrites.reserve(options.callable_options.fetch_size());
293 const DeviceAttributes* device_info =
294 &device_set_->client_device()->attributes();
295 if (options.use_function_convention) {
296 for (int i = 0; i < options.callable_options.feed_size(); ++i) {
297 feed_rewrites.emplace_back(new subgraph::ArgFeedRewrite(
298 &options.callable_options.feed(i), device_info, i));
299 }
300 for (int i = 0; i < options.callable_options.fetch_size(); ++i) {
301 fetch_rewrites.emplace_back(new subgraph::RetvalFetchRewrite(
302 &options.callable_options.fetch(i), device_info, i));
303 }
304 } else {
305 for (const string& feed : options.callable_options.feed()) {
306 feed_rewrites.emplace_back(
307 new subgraph::RecvFeedRewrite(&feed, device_info));
308 }
309 for (const string& fetch : options.callable_options.fetch()) {
310 fetch_rewrites.emplace_back(
311 new subgraph::SendFetchRewrite(&fetch, device_info));
312 }
313 }
314
315 for (const TensorConnection& tensor_connection :
316 options.callable_options.tensor_connection()) {
317 Node* from_node = nullptr;
318 TensorId from_id(ParseTensorName(tensor_connection.from_tensor()));
319
320 for (Node* n : graph->nodes()) {
321 if (n->name() == from_id.first) {
322 from_node = n;
323 break;
324 }
325 }
326 if (from_node == nullptr) {
327 return errors::InvalidArgument(
328 "Requested tensor connection from unknown node: \"",
329 tensor_connection.to_tensor(), "\".");
330 }
331 if (from_id.second >= from_node->num_outputs()) {
332 return errors::InvalidArgument(
333 "Requested tensor connection from unknown edge: \"",
334 tensor_connection.to_tensor(),
335 "\" (actual number of outputs = ", from_node->num_outputs(), ").");
336 }
337
338 feed_rewrites.emplace_back(new TensorConnectionPruneRewrite(
339 &tensor_connection.to_tensor(), {from_node, from_id.second}));
340 }
341
342 std::vector<string> target_node_names(
343 options.callable_options.target().begin(),
344 options.callable_options.target().end());
345 TF_RETURN_IF_ERROR(subgraph::RewriteGraphForExecution(
346 graph, feed_rewrites, fetch_rewrites, target_node_names,
347 out_rewrite_metadata));
348
349 CHECK_EQ(out_rewrite_metadata->feed_types.size(),
350 options.callable_options.feed_size() +
351 options.callable_options.tensor_connection_size());
352 for (int i = 0; i < options.callable_options.tensor_connection_size(); ++i) {
353 out_rewrite_metadata->feed_types.pop_back();
354 }
355 return Status::OK();
356}

358Status GraphExecutionState::InitBaseGraph(const BuildGraphOptions& options) {
359 const GraphDef* graph_def = &original_graph_def_;
360
361 std::unique_ptr<Graph> new_graph(new Graph(OpRegistry::Global()));
362 GraphConstructorOptions opts;
363 TF_RETURN_IF_ERROR(ConvertGraphDefToGraph(opts, *graph_def, new_graph.get()));
364 for (const Node* n : new_graph->nodes()) {
365 VLOG(2) << "Mapping " << n->name() << " to " << n->cost_id();
366 node_name_to_cost_id_map_[n->name()] = n->cost_id();
367 }
368 if (session_options_ &&
369 session_options_->config.graph_options().place_pruned_graph()) {
370 // Rewrite the graph before placement.
371 rewrite_metadata_.reset(new subgraph::RewriteGraphMetadata);
372 TF_RETURN_IF_ERROR(
373 PruneGraph(options, new_graph.get(), rewrite_metadata_.get()));
374 }
375
376 // Save stateful placements before placing.
377 RestoreStatefulNodes(new_graph.get());
378
379 GraphOptimizationPassOptions optimization_options;
380 optimization_options.session_options = session_options_;
381 optimization_options.graph = &new_graph;
382 optimization_options.flib_def = flib_def_.get();
383 optimization_options.device_set = device_set_;
384
385 TF_RETURN_IF_ERROR(OptimizationPassRegistry::Global()->RunGrouping(
386 OptimizationPassRegistry::PRE_PLACEMENT, optimization_options));
387
388 Placer placer(new_graph.get(), device_set_, session_options_);
389 // TODO(mrry): Consider making the Placer cancelable.
390 TF_RETURN_IF_ERROR(placer.Run());
391
392 TF_RETURN_IF_ERROR(OptimizationPassRegistry::Global()->RunGrouping(
393 OptimizationPassRegistry::POST_PLACEMENT, optimization_options));
394
395 SaveStatefulNodes(new_graph.get());
396 graph_ = new_graph.release();
397 return Status::OK();
398}

400Status GraphExecutionState::OptimizeGraph(
401 const BuildGraphOptions& options, std::unique_ptr<Graph>* optimized_graph) {
402#ifndef IS_MOBILE_PLATFORM
403 if (session_options_->config.graph_options().place_pruned_graph()) {
404 return errors::InvalidArgument("Can't optimize a pruned graph");
405 }
406
407 const RewriterConfig& rewrite_options =
408 session_options_->config.graph_options().rewrite_options();
409
410 if (grappler::MetaOptimizerEnabled(rewrite_options)) {
411 // Adding this functionality in steps. The first step is to make sure
412 // we don't break dependencies. The second step will be to turn the
413 // functionality on by default.
414 grappler::GrapplerItem item;
415 item.id = "tf_graph";
416 graph_->ToGraphDef(&item.graph);
417
418 item.fetch.insert(item.fetch.end(),
419 options.callable_options.fetch().begin(),
420 options.callable_options.fetch().end());
421 item.fetch.insert(item.fetch.end(),
422 options.callable_options.target().begin(),
423 options.callable_options.target().end());
424
425 for (const TensorConnection& tensor_connection :
426 options.callable_options.tensor_connection()) {
427 item.fetch.push_back(tensor_connection.from_tensor());
428 }
429
430 if (!(options.callable_options.feed().empty() &&
431 options.callable_options.tensor_connection().empty())) {
432 std::unordered_set<string> feeds;
433 for (const string& feed : options.callable_options.feed()) {
434 TensorId id = ParseTensorName(feed);
435 if (id.second != 0) {
436 return errors::InvalidArgument("Unsupported feed: ", feed);
437 }
438 feeds.insert(id.first.ToString());
439 }
440 for (const TensorConnection& tensor_connection :
441 options.callable_options.tensor_connection()) {
442 TensorId id = ParseTensorName(tensor_connection.to_tensor());
443 if (id.second != 0) {
444 return errors::InvalidArgument("Unsupported feed: ",
445 tensor_connection.to_tensor());
446 }
447 feeds.insert(id.first.ToString());
448 }
449 for (const NodeDef& node : original_graph_def_.node()) {
450 if (feeds.find(node.name()) == feeds.end()) {
451 continue;
452 }
453 if (node.attr().count("dtype") == 0 ||
454 node.attr().count("shape") == 0) {
455 return errors::InvalidArgument("Missing node shape or type");
456 }
457 TensorShapeProto shape_proto(node.attr().at("shape").shape());
458 // If the shape of the placeholder value is only partially known, we're
459 // free to use any dimension we want to feed the placeholder. We choose
460 // 1 to minimize the memory impact. Note that this only matters if an
461 // optimizer choose to run the graph to build its cost model, which
462 // doesn't happen (yet)
463 if (shape_proto.unknown_rank()) {
464 shape_proto.set_unknown_rank(false);
465 }
466 for (auto& dim : *shape_proto.mutable_dim()) {
467 if (dim.size() < 0) {
468 dim.set_size(1);
469 }
470 }
471 TensorShape shape(shape_proto);
472 DataType type = node.attr().at("dtype").type();
473 Tensor fake_input(type, shape);
474 item.feed.emplace_back(node.name(), fake_input);
475 }
476 }
477
478 std::unordered_map<string, DeviceProperties> device_map;
479 Device* cpu_device = nullptr;
480 for (const auto& device : device_set_->devices()) {
481 DeviceProperties props = grappler::GetDeviceInfo(device->parsed_name());
482 if (props.type() == "UNKNOWN") {
483 continue;
484 }
485 device_map[device->name()] = props;
486 if (device->parsed_name().id == 0 &&
487 StringPiece(device->parsed_name().type) == "CPU" &&
488 device->GetAllocator(AllocatorAttributes()) != nullptr) {
489 cpu_device = device;
490 }
491 }
492 grappler::VirtualCluster cluster(device_map);
493 GraphDef new_graph;
494 TF_RETURN_IF_ERROR(grappler::RunMetaOptimizer(
495 item, rewrite_options, cpu_device, &cluster, &new_graph));
496 GraphConstructorOptions opts;
497 opts.allow_internal_ops = true;
498 optimized_graph->reset(new Graph(OpRegistry::Global()));
499 TF_RETURN_IF_ERROR(
500 ConvertGraphDefToGraph(opts, new_graph, optimized_graph->get()));
501 // The graph conversion sets the requested device names but not the assigned
502 // device names. However, since at this point the graph is placed TF expects
503 // an assigned device name for every node. Therefore we copy the requested
504 // device into the assigned device field.
505 for (Node* node : optimized_graph->get()->nodes()) {
506 node->set_assigned_device_name(node->requested_device());
507 }
508 return Status::OK();
509 } else {
510 return errors::InvalidArgument("Meta Optimizer disabled");
511 }
512#else
513 return errors::InvalidArgument("Mobile platforms not supported");
514#endif // IS_MOBILE_PLATFORM
515}

517Status GraphExecutionState::BuildGraph(const BuildGraphOptions& options,
518 std::unique_ptr<ClientGraph>* out) {
519 VLOG(1) << "BuildGraph";
520 if (!graph_) {
521 // It is only valid to call this method directly when the original graph
522 // was created with the option `place_pruned_graph == false`.
523 return errors::Internal(
524 "Attempted to prune a graph that has not been fully initialized.");
525 }
526
527 std::unique_ptr<Graph> ng;
528 Status s = OptimizeGraph(options, &ng);
529 if (!s.ok()) {
530 // Simply copy the original graph if we couldn't optimize it.
531 ng.reset(new Graph(flib_def_.get()));
532 CopyGraph(*graph_, ng.get());
533 }
534
535 subgraph::RewriteGraphMetadata rewrite_metadata;
536 if (session_options_ == nullptr ||
537 !session_options_->config.graph_options().place_pruned_graph()) {
538 TF_RETURN_IF_ERROR(PruneGraph(options, ng.get(), &rewrite_metadata));
539 } else {
540 // This GraphExecutionState represents a graph that was
541 // pruned when this was constructed, so we copy the metadata from
542 // a member variable.
543 CHECK(rewrite_metadata_);
544 rewrite_metadata = *rewrite_metadata_;
545 }
546
547 CHECK_EQ(options.callable_options.feed_size(),
548 rewrite_metadata.feed_types.size());
549 CHECK_EQ(options.callable_options.fetch_size(),
550 rewrite_metadata.fetch_types.size());
551
552 // Make a fresh copy of the function library for the client graph.
553 std::unique_ptr<FunctionLibraryDefinition> flib(
554 new FunctionLibraryDefinition(*flib_def_));
555
556 // TODO(andydavis): Clarify optimization pass requirements around CostModel.
557 GraphOptimizationPassOptions optimization_options;
558 optimization_options.session_options = session_options_;
559 optimization_options.graph = &ng;
560 optimization_options.flib_def = flib.get();
561 optimization_options.device_set = device_set_;
562
563 TF_RETURN_IF_ERROR(OptimizationPassRegistry::Global()->RunGrouping(
564 OptimizationPassRegistry::POST_REWRITE_FOR_EXEC, optimization_options));
565
566 // Copy the extracted graph in order to make its node ids dense,
567 // since the local CostModel used to record its stats is sized by
568 // the largest node id.
569 std::unique_ptr<ClientGraph> dense_copy(
570 new ClientGraph(std::move(flib), rewrite_metadata.feed_types,
571 rewrite_metadata.fetch_types));
572 CopyGraph(*ng, &dense_copy->graph);
573
574 // TODO(vrv): We should check invariants of the graph here.
575
576 *out = std::move(dense_copy);
577 return Status::OK();
578}
