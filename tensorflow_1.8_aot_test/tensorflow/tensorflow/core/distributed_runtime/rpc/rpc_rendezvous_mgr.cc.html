56// Used only to retrieve tensors from remote processes.
57class RpcRecvTensorCall : public BaseRecvTensorCall {
58 public:
59 RpcRecvTensorCall() : wi_(nullptr), dst_device_(nullptr) {}
60
61 void Init(WorkerInterface* wi, int64 step_id, StringPiece key,
62 AllocatorAttributes alloc_attrs, Device* dst_device,
63 const Rendezvous::Args& recv_args, Rendezvous::DoneCallback done) {
64 wi_ = wi;
65 alloc_attrs_ = alloc_attrs;
66 dst_device_ = dst_device;
67 recv_args_ = recv_args;
68 done_ = std::move(done);
69 req_.set_step_id(step_id);
70 req_.set_rendezvous_key(key.data(), key.size());
71 req_.set_request_id(GetUniqueRequestId());
72 }
73
74 void Reset(WorkerCacheInterface* wc) {
75 wc->ReleaseWorker(src_worker_, wi_);
76 wi_ = nullptr;
77 alloc_attrs_ = AllocatorAttributes();
78 dst_device_ = nullptr;
79 // We don't clear opts_ and assume that Init will set up the state for
80 // opts_ appropriately.
81 req_.Clear();
82 resp_.Clear();
83 {
84 mutex_lock l(mu_);
85 status_ = Status::OK();
86 }
87 done_ = nullptr;
88 }
89
90 ~RpcRecvTensorCall() override {
91 // Since only the RpcRecvTensorFreeList will delete an
92 // RpcRecvTensorCall, and it always sets this->wi_ to null when
93 // a call object is released to it, we can assert that this->wi_ is
94 // always null at the point of deletion.
95 CHECK_EQ(static_cast<WorkerInterface*>(nullptr), wi_)
96 << "Leaking WorkerInterface in RpcRecvTensorCall destructor.";
97 }
98
99 void Start(std::function<void()> recv_done) override {
100 StartRTCall(std::move(recv_done));
101 }
102
103 void StartAbort(const Status& s) override {
104 {
105 mutex_lock l(mu_);
106 status_.Update(s);
107 }
108 opts_.StartCancel();
109 }
110
111 Status status() const override {
112 mutex_lock l(mu_);
113 return status_;
114 }
115
116 const Tensor& tensor() const { return resp_.tensor(); }
117
118 bool is_dead() const { return resp_.metadata().is_dead(); }
119
120 Device* dst_device() const { return dst_device_; }
121 const Rendezvous::Args& recv_args() const { return recv_args_; }
122 const Rendezvous::DoneCallback& done() const { return done_; }
123
124 private:
125 friend class RpcRemoteRendezvous;
126
127 // Start the main RecvTensor call, checking for an async abort.
128 void StartRTCall(std::function<void()> recv_done) {
129 resp_.InitAlloc(dst_device_, alloc_attrs_);
130 using namespace std::placeholders;
131 StatusCallback cb = std::bind(
132 [this](std::function<void()> recv_done,
133 // Begin unbound arguments.
134 const Status& s) {
135 if (!s.ok()) {
136 mutex_lock l(mu_);
137 status_.Update(s);
138 }
139 recv_done();
140 },
141 std::move(recv_done), _1);
142 wi_->RecvTensorAsync(&opts_, &req_, &resp_, std::move(cb));
143 }
144
145 string src_worker_;
146 string src_rel_device_;
147 WorkerInterface* wi_;
148 AllocatorAttributes alloc_attrs_;
149 Device* dst_device_;
150 CallOptions opts_;
151 RecvTensorRequest req_;
152 TensorResponse resp_;
153 Rendezvous::Args recv_args_;
154 Rendezvous::DoneCallback done_;
155
156 mutable mutex mu_;
157 Status status_ GUARDED_BY(mu_);
158
159 TF_DISALLOW_COPY_AND_ASSIGN(RpcRecvTensorCall);
160};

162class RpcRecvTensorFreeList {
163 public:
164 RpcRecvTensorFreeList() {}
165 ~RpcRecvTensorFreeList() {
166 for (size_t i = 0; i < objects_.size(); i++) {
167 delete objects_[i];
168 }
169 }
170
171 RpcRecvTensorCall* New() {
172 {
173 mutex_lock l(mu_);
174 if (!objects_.empty()) {
175 RpcRecvTensorCall* result = objects_.back();
176 objects_.pop_back();
177 return result;
178 }
179 }
180 return new RpcRecvTensorCall;
181 }
182
183 void Release(RpcRecvTensorCall* obj, WorkerCacheInterface* wc) {
184 obj->Reset(wc);
185 {
186 mutex_lock l(mu_);
187 if (objects_.size() < kMaxObjects) {
188 objects_.push_back(obj);
189 return;
190 }
191 }
192 delete obj;
193 }
194
195 private:
196 static const int kMaxObjects = 1000;
197
198 mutex mu_;
199 std::vector<RpcRecvTensorCall*> objects_ GUARDED_BY(mu_);
200};

202static RpcRecvTensorFreeList* get_call_freelist() {
203 static RpcRecvTensorFreeList* call_freelist = new RpcRecvTensorFreeList();
204 return call_freelist;
205}

207void RpcRemoteRendezvous::RecvFromRemoteAsync(
208 const Rendezvous::ParsedKey& parsed, const Rendezvous::Args& recv_args,
209 DoneCallback done) {
210 CHECK(is_initialized());
211 Status s;
212
213 // Prepare a RecvTensor call that can handle being aborted.
214 RpcRecvTensorCall* call = get_call_freelist()->New();
215
216 // key.src_device identifies a remote device.
217 if (!DeviceNameUtils::SplitDeviceName(parsed.src_device, &call->src_worker_,
218 &call->src_rel_device_)) {
219 s = errors::Internal(parsed.src_device,
220 " is invalid remote source device.");
221 }
222 WorkerSession* sess = session();
223 WorkerInterface* rwi = sess->worker_cache->CreateWorker(call->src_worker_);
224 if (s.ok() && rwi == nullptr) {
225 s = errors::Internal("No worker known as ", call->src_worker_);
226 }
227
228 Device* dst_device;
229 if (s.ok()) {
230 s = sess->device_mgr->LookupDevice(parsed.dst_device, &dst_device);
231 }
232 if (!s.ok()) {
233 if (rwi != nullptr) {
234 sess->worker_cache->ReleaseWorker(call->src_worker_, rwi);
235 }
236 get_call_freelist()->Release(call, sess->worker_cache.get());
237 done(s, Args(), recv_args, Tensor{}, false);
238 return;
239 }
240
241 call->Init(rwi, step_id_, parsed.FullKey(), recv_args.alloc_attrs, dst_device,
242 recv_args, std::move(done));
243
244 // Record "call" in active_ so that it can be aborted cleanly.
245 RegisterCall(call);
246
247 // Start "call".
248 Ref();
249 call->Start([this, call]() {
250 // Removes "call" from active_. Prevent StartAbort().
251 DeregisterCall(call);
252 // If StartAbort was called prior to DeregisterCall, then the
253 // current status should be bad.
254 Status s = call->status();
255 call->done()(s, Args(), call->recv_args(), call->tensor(), call->is_dead());
256 session()->worker_cache->ReleaseWorker(call->src_worker_, call->wi_);
257 call->wi_ = nullptr;
258 get_call_freelist()->Release(call, session()->worker_cache.get());
259 Unref();
260 });
261}
