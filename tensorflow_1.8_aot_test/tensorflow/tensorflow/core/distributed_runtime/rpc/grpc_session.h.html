39namespace tensorflow {
40
41class MasterInterface;
42
43// A Session instance lets the caller drive a TensorFlow graph
44// computation on potentially remote sets of devices. This is a thin
45// wrapper around tensorflow::grpc::MasterService.
46//
47// Multiple threads must synchronize their accesses to a single
48// session.
49class GrpcSession : public Session {
50 protected:
51 explicit GrpcSession(const SessionOptions& options);
52
53 public:
54 static Status Create(const SessionOptions& options,
55 std::unique_ptr<GrpcSession>* out_session);
56 // Resets the resource containers.
57 static Status Reset(const SessionOptions& options,
58 const std::vector<string>& containers);
59
60 ~GrpcSession() override;
61
62 // Creates a session with the "target". The session carries out
63 // the graph computation defined by "graph", and will have version
64 // number "initial_version".
65 Status Create(const GraphDef& graph) override;
66 Status Create(const RunOptions& run_options, const GraphDef& graph) override;
67
68 // Runs with and without RunOptions.
69 Status Run(const std::vector<std::pair<string, Tensor> >& inputs,
70 const std::vector<string>& output_tensor_names,
71 const std::vector<string>& target_node_names,
72 std::vector<Tensor>* outputs) override;
73 Status Run(const RunOptions& run_options,
74 const std::vector<std::pair<string, Tensor> >& inputs,
75 const std::vector<string>& output_tensor_names,
76 const std::vector<string>& target_node_names,
77 std::vector<Tensor>* outputs, RunMetadata* run_metadata) override;
78
79 Status Extend(const GraphDef& graph) override;
80 Status Extend(const RunOptions& run_options, const GraphDef& graph) override;
81
82 Status Close() override;
83
84 // NOTE: This API is still experimental and may change.
85 Status PRunSetup(const std::vector<string>& input_names,
86 const std::vector<string>& output_names,
87 const std::vector<string>& target_nodes,
88 string* handle) override;
89
90 // NOTE: This API is still experimental and may change.
91 Status PRun(const string& handle,
92 const std::vector<std::pair<string, Tensor> >& inputs,
93 const std::vector<string>& output_names,
94 std::vector<Tensor>* outputs) override;
95
96 Status ListDevices(std::vector<DeviceAttributes>* response) override;
97
98 Status MakeCallable(const CallableOptions& callable_options,
99 CallableHandle* out_handle) override;
100 Status RunCallable(CallableHandle handle,
101 const std::vector<Tensor>& feed_tensors,
102 std::vector<Tensor>* fetch_tensors,
103 RunMetadata* run_metadata) override;
104 Status ReleaseCallable(CallableHandle handle) override;
105
106 protected:
107 // Takes ownership of `*master`.
108 void SetRemoteMaster(std::unique_ptr<MasterInterface> master);
109
110 private:
111 SessionOptions options_;
112 std::unique_ptr<MasterInterface> master_;
113 mutex mu_;
114
115 // handle_ returned by the master to identify this session.
116 string handle_ GUARDED_BY(mu_);
117
118 // The current version of the graph.
119 int64 current_graph_version_ GUARDED_BY(mu_);
120
121 Status Handle(string* out_handle) LOCKS_EXCLUDED(mu_);
122
123 Status RunHelper(const RunOptions& run_options,
124 const std::vector<std::pair<string, Tensor> >& inputs,
125 const std::vector<string>& output_tensor_names,
126 const std::vector<string>& target_node_names,
127 std::vector<Tensor>* outputs, RunMetadata* run_metadata,
128 const string& prun_handle);
129
130 Status RunProto(CallOptions* call_options, MutableRunStepRequestWrapper* req,
131 MutableRunStepResponseWrapper* resp);
132
133 // Implementations for all the public interfaces.
134 Status CreateImpl(CallOptions* call_options, const GraphDef& graph);
135 Status ExtendImpl(CallOptions* call_options, const GraphDef& graph);
136
137 TF_DISALLOW_COPY_AND_ASSIGN(GrpcSession);
138};
