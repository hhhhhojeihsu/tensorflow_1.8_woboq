37class Device;
38struct MasterEnv;
39
40// A session encapsulates a graph computation (resource allocation,
41// placement, execution, etc.).
42class MasterSession : public core::RefCounted {
43 public:
44 // This session encapsulates the graph computation for a graph.
45 //
46 // The session places nodes on devices in "remote_devs" and executes
47 // operations on these devices.
48 //
49 // The caller takes ownership of all remote devices.
50 MasterSession(
51 const SessionOptions& options, const MasterEnv* env,
52 std::unique_ptr<std::vector<std::unique_ptr<Device>>> remote_devs,
53 std::unique_ptr<WorkerCacheInterface> worker_cache,
54 std::unique_ptr<DeviceSet> device_set,
55 StatsPublisherFactory stats_publisher_factory);
56
57 // Initialize the MasterSession for "def". Must be called before Extend(),
58 // Run(), or Close().
59 //
60 // After this method returns, `def` will no longer be valid.
61 Status Create(GraphDef* def, const WorkerCacheFactoryOptions& options);
62
63 // Returns the session handle.
64 const string& handle() const { return handle_; }
65
66 // Returns the last access time (the number of micro-seconds since
67 // some fixed point in time) of this session.
68 uint64 last_access_time_usec() const { return last_access_time_usec_.load(); }
69
70 // Attempt to extend the graph according to the given "req".
71 // (See master.proto for details of valid extensions.)
72 //
73 // PRECONDITION: The current version of this session's graph
74 // is "req->current_graph_version".
75 //
76 // POSTCONDITION: The current version of this session's graph
77 // is "resp->new_graph_version".
78 //
79 // Extend() may block the caller thread for a long time.
80 Status Extend(const ExtendSessionRequest* req, ExtendSessionResponse* resp);
81
82 // Setup a partial run call.
83 Status PartialRunSetup(const PartialRunSetupRequest* req,
84 PartialRunSetupResponse* resp);
85
86 // Run one step.
87 Status Run(CallOptions* opts, const RunStepRequestWrapper& req,
88 MutableRunStepResponseWrapper* resp);
89
90 Status ListDevices(ListDevicesResponse* resp) const;
91
92 Status MakeCallable(const MakeCallableRequest& req,
93 MakeCallableResponse* resp);
94
95 Status RunCallable(CallOptions* opts, const RunCallableRequest& req,
96 RunCallableResponse* resp);
97
98 Status ReleaseCallable(const ReleaseCallableRequest& req,
99 ReleaseCallableResponse* resp);
100
101 // Close this session and delete "*this". Returns OK if all known
102 // states are cleanup successfully.
103 //
104 // Close() may block the caller thread for a long time.
105 Status Close();
106
107 // Close this session and release a reference on "*this".
108 //
109 // Note that, unlike Close(), this method does not block on the
110 // completion of all work.
111 void GarbageCollect();
112
113 private:
114 SessionOptions session_opts_;
115
116 // Not owned.
117 const MasterEnv* env_;
118
119 // The opaque session handle.
120 const string handle_;
121
122 std::unique_ptr<std::vector<std::unique_ptr<Device>>> remote_devs_;
123
124 // The optional session-specific worker cluster.
125 // TODO(saeta): Convert to std::optional when available.
126 const std::unique_ptr<WorkerCacheInterface> worker_cache_;
127 // Retrieves either worker_cache_ or the env_->worker_cache as appropriate.
128 WorkerCacheInterface* get_worker_cache() const;
129
130 // The device set used by this session.
131 std::unique_ptr<DeviceSet> devices_;
132
133 StatsPublisherFactory stats_publisher_factory_;
134
135 std::atomic_ulong last_access_time_usec_;
136
137 std::atomic<int64> partial_run_handle_counter_ = {0};
138
139 mutex mu_;
140 std::unique_ptr<GraphExecutionState> execution_state_ GUARDED_BY(mu_);
141 int64 graph_version_;
142
143 // We keep a map from a signature of a run request to the
144 // ReffedClientGraph the can execute it. We keep up to one old copy
145 // of each ReffedClientGraph around because if it gets deallocated
146 // before a new substitute has been created, Variables can go out of
147 // scope and lose their state.
148 class ReffedClientGraph;
149 typedef std::unordered_map<uint64, ReffedClientGraph*> RCGMap;
150 RCGMap run_graphs_ GUARDED_BY(mu_);
151 RCGMap partial_run_graphs_ GUARDED_BY(mu_);
152 int64 next_callable_handle_ GUARDED_BY(mu_) = 0;
153 RCGMap callables_ GUARDED_BY(mu_);
154
155 struct PerStepState {
156 bool collect_costs = false;
157 bool collect_timeline = false;
158 bool collect_rpcs = false;
159 bool collect_partition_graphs = false;
160 bool report_tensor_allocations_upon_oom = false;
161 Microseconds start_micros = Microseconds(0);
162 Microseconds end_micros = Microseconds(0);
163 std::vector<StepStats> step_stats; // per partition
164 StepStats rpc_stats; // for RPC layer
165 CostGraphDef cost_graph;
166 };
167
168 struct RunState {
169 std::unordered_map<string, bool> pending_inputs; // true if fed
170 std::unordered_map<string, bool> pending_outputs; // true if fetched
171 ReffedClientGraph* rcg = nullptr;
172 uint64 step_id;
173 int64 count = 0;
174 PerStepState pss;
175 std::unique_ptr<ProfileHandler> ph;
176 bool step_started = false;
177
178 RunState(const std::vector<string>& input_names,
179 const std::vector<string>& output_names, ReffedClientGraph* rcg,
180 const uint64 step_id, const int64 count);
181
182 bool PendingDone() const;
183
184 ~RunState();
185 };
186 std::unordered_map<string, std::unique_ptr<RunState>> partial_runs_
187 GUARDED_BY(mu_);
188
189 // Active RunStep calls.
190 condition_variable num_running_is_zero_;
191 int32 num_running_ GUARDED_BY(mu_) = 0;
192
193 bool closed_ GUARDED_BY(mu_) = false;
194 bool garbage_collected_ GUARDED_BY(mu_) = false;
195
196 std::unordered_map<uint64, int64> subgraph_execution_counts_ GUARDED_BY(mu_);
197
198 // We need to ensure that certain nodes added (e.g., send and recv
199 // nodes) are unique across all sub-graphs within this session.
200 int64 next_node_id_ GUARDED_BY(mu_) = 0;
201
202 // Used to cancel running steps on Close().
203 CancellationManager cancellation_manager_;
204
205 // Private dtor. The client must call Close().
206 virtual ~MasterSession();
207
208 // Creates sessions on all workers.
209 //
210 // If this session is operating using the new ClusterSpec propagation behavior
211 // call this method in order to propagate the cluster membership to all
212 // workers.
213 Status CreateWorkerSessions(const WorkerCacheFactoryOptions& server_def);
214
215 // TODO(b/36574172): Always use Create/DeleteWorkerSession.
216 bool should_delete_worker_sessions_ = false;
217 Status DeleteWorkerSessions();
218
219 Status StartStep(const BuildGraphOptions& opts, bool is_partial,
220 ReffedClientGraph** out_rcg, int64* out_count);
221 void ClearRunsTable(std::vector<ReffedClientGraph*>* to_unref,
222 RCGMap* rcg_map) EXCLUSIVE_LOCKS_REQUIRED(mu_);
223 void FillPerStepState(MasterSession::ReffedClientGraph* rcg,
224 const RunOptions& run_options, uint64 step_id,
225 int64 count, PerStepState* out_pss,
226 std::unique_ptr<ProfileHandler>* out_ph);
227 Status DoRunWithLocalExecution(CallOptions* opts,
228 const RunStepRequestWrapper& req,
229 MutableRunStepResponseWrapper* resp);
230 Status DoPartialRun(CallOptions* opts, const RunStepRequestWrapper& req,
231 MutableRunStepResponseWrapper* resp);
232 Status DoRunCallable(CallOptions* opts, ReffedClientGraph* rcg,
233 const RunCallableRequest& req,
234 RunCallableResponse* resp);
235 Status PostRunCleanup(MasterSession::ReffedClientGraph* rcg, uint64 step_id,
236 const RunOptions& run_options, PerStepState* pss,
237 const std::unique_ptr<ProfileHandler>& ph,
238 const Status& run_status,
239 RunMetadata* out_run_metadata);
240
241 void MarkRunCompletion();
242 void UpdateLastAccessTime();
243
244 Status BuildAndRegisterPartitions(ReffedClientGraph* rcg);
245
246 Status CreateDebuggerState(
247 const DebugOptions& debug_options, const RunStepRequestWrapper& req,
248 int64 rcg_execution_count,
249 std::unique_ptr<DebuggerStateInterface>* debugger_state);
250
251 TF_DISALLOW_COPY_AND_ASSIGN(MasterSession);
252};

254} // end namespace tensorflow
