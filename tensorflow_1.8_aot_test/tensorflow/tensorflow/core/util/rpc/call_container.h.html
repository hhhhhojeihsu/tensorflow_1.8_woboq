16#ifndef TENSORFLOW_CORE_UTIL_RPC_CALL_CONTAINER_H_
17#define TENSORFLOW_CORE_UTIL_RPC_CALL_CONTAINER_H_
18
19#include <list>
20
21#include "tensorflow/core/framework/op_kernel.h"
22#include "tensorflow/core/framework/tensor_types.h"
23#include "tensorflow/core/lib/core/errors.h"
24#include "tensorflow/core/lib/core/refcount.h"
25#include "tensorflow/core/util/reffed_status_callback.h"
26
27namespace tensorflow {
28
29template <typename Call>
30class CallContainer {
31 public:
32 explicit CallContainer(OpKernelContext* ctx, int num_calls, bool fail_fast,
33 bool try_rpc, AsyncOpKernel::DoneCallback done,
34 CancellationToken token)
35 : ctx_(ctx),
36 done_(std::move(done)),
37 token_(token),
38 fail_fast_(fail_fast),
39 try_rpc_(try_rpc) {
40 CHECK_GT(num_calls, 0);
41
42 // This will run when all RPCs are finished.
43 reffed_status_callback_ = new ReffedStatusCallback([this](const Status& s) {
44 ctx_->cancellation_manager()->DeregisterCallback(token_);
45 ctx_->SetStatus(s);
46 done_();
47 delete this;
48 });
49
50 // Subtract reference count from the initial creation.
51 core::ScopedUnref unref(reffed_status_callback_);
52
53 for (int i = 0; i < num_calls; ++i) {
54 // Increase the reference on the callback for each new RPC.
55 reffed_status_callback_->Ref();
56 }
57 }
58
59 std::list<Call>* calls() { return &calls_; }
60
61 void StartCancel() {
62 // Once this loop is done, can no longer assume anything is valid
63 // because "delete this" may have been immediately called.
64 // Nothing should run after this loop.
65 for (auto& call : calls_) {
66 call.StartCancel();
67 }
68 }
69
70 void Done(const Status& s, int index) {
71 if (!try_rpc_) {
72 reffed_status_callback_->UpdateStatus(s);
73 }
74 reffed_status_callback_->Unref();
75 }
76
77 private:
78 OpKernelContext* ctx_;
79 std::list<Call> calls_;
80 const AsyncOpKernel::DoneCallback done_;
81 const CancellationToken token_;
82 const bool fail_fast_;
83 const bool try_rpc_;
84
85 // Performs its own reference counting.
86 ReffedStatusCallback* reffed_status_callback_;
87};
88
89} // namespace tensorflow
90#endif // TENSORFLOW_CORE_UTIL_RPC_CALL_CONTAINER_H_
