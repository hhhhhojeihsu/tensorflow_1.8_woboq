28bool ParseStringFlag(tensorflow::StringPiece arg, tensorflow::StringPiece flag,
29 const std::function<bool(string)>& hook,
30 bool* value_parsing_ok) {
31 *value_parsing_ok = true;
32 if (str_util::ConsumePrefix(&arg, "--") &&
33 str_util::ConsumePrefix(&arg, flag) &&
34 str_util::ConsumePrefix(&arg, "=")) {
35 *value_parsing_ok = hook(arg.ToString());
36 return true;
37 }
38
39 return false;
40}

42bool ParseInt32Flag(tensorflow::StringPiece arg, tensorflow::StringPiece flag,
43 const std::function<bool(int32)>& hook,
44 bool* value_parsing_ok) {
45 *value_parsing_ok = true;
46 if (str_util::ConsumePrefix(&arg, "--") &&
47 str_util::ConsumePrefix(&arg, flag) &&
48 str_util::ConsumePrefix(&arg, "=")) {
49 char extra;
50 int32 parsed_int32;
51 if (sscanf(arg.data(), "%d%c", &parsed_int32, &extra) != 1) {
52 LOG(ERROR) << "Couldn't interpret value " << arg << " for flag " << flag
53 << ".";
54 *value_parsing_ok = false;
55 } else {
56 *value_parsing_ok = hook(parsed_int32);
57 }
58 return true;
59 }
60
61 return false;
62}

86bool ParseBoolFlag(tensorflow::StringPiece arg, tensorflow::StringPiece flag,
87 const std::function<bool(bool)>& hook,
88 bool* value_parsing_ok) {
89 *value_parsing_ok = true;
90 if (str_util::ConsumePrefix(&arg, "--") &&
91 str_util::ConsumePrefix(&arg, flag)) {
92 if (arg.empty()) {
93 *value_parsing_ok = hook(true);
94 return true;
95 }
96
97 if (arg == "=true") {
98 *value_parsing_ok = hook(true);
99 return true;
100 } else if (arg == "=false") {
101 *value_parsing_ok = hook(false);
102 return true;
103 } else {
104 LOG(ERROR) << "Couldn't interpret value " << arg << " for flag " << flag
105 << ".";
106 *value_parsing_ok = false;
107 return true;
108 }
109 }
110
111 return false;
112}

228bool Flag::Parse(string arg, bool* value_parsing_ok) const {
229 bool result = false;
230 if (type_ == TYPE_INT32) {
231 result = ParseInt32Flag(arg, name_, int32_hook_, value_parsing_ok);
232 } else if (type_ == TYPE_INT64) {
233 result = ParseInt64Flag(arg, name_, int64_hook_, value_parsing_ok);
234 } else if (type_ == TYPE_BOOL) {
235 result = ParseBoolFlag(arg, name_, bool_hook_, value_parsing_ok);
236 } else if (type_ == TYPE_STRING) {
237 result = ParseStringFlag(arg, name_, string_hook_, value_parsing_ok);
238 } else if (type_ == TYPE_FLOAT) {
239 result = ParseFloatFlag(arg, name_, float_hook_, value_parsing_ok);
240 }
241 return result;
242}

244/*static*/ bool Flags::Parse(int* argc, char** argv,
245 const std::vector<Flag>& flag_list) {
246 bool result = true;
247 std::vector<char*> unknown_flags;
248 for (int i = 1; i < *argc; ++i) {
249 if (string(argv[i]) == "--") {
250 while (i < *argc) {
251 unknown_flags.push_back(argv[i]);
252 ++i;
253 }
254 break;
255 }
256
257 bool was_found = false;
258 for (const Flag& flag : flag_list) {
259 bool value_parsing_ok;
260 was_found = flag.Parse(argv[i], &value_parsing_ok);
261 if (!value_parsing_ok) {
262 result = false;
263 }
264 if (was_found) {
265 break;
266 }
267 }
268 if (!was_found) {
269 unknown_flags.push_back(argv[i]);
270 }
271 }
272 // Passthrough any extra flags.
273 int dst = 1; // Skip argv[0]
274 for (char* f : unknown_flags) {
275 argv[dst++] = f;
276 }
277 argv[dst++] = nullptr;
278 *argc = unknown_flags.size() + 1;
279 return result && (*argc < 2 || strcmp(argv[1], "--help") != 0);
280}

282/*static*/ string Flags::Usage(const string& cmdline,
283 const std::vector<Flag>& flag_list) {
284 string usage_text;
285 if (!flag_list.empty()) {
286 strings::Appendf(&usage_text, "usage: %s\nFlags:\n", cmdline.c_str());
287 } else {
288 strings::Appendf(&usage_text, "usage: %s\n", cmdline.c_str());
289 }
290 for (const Flag& flag : flag_list) {
291 const char* type_name = "";
292 string flag_string;
293 if (flag.type_ == Flag::TYPE_INT32) {
294 type_name = "int32";
295 flag_string = strings::Printf("--%s=%d", flag.name_.c_str(),
296 flag.int32_default_for_display_);
297 } else if (flag.type_ == Flag::TYPE_INT64) {
298 type_name = "int64";
299 flag_string = strings::Printf(
300 "--%s=%lld", flag.name_.c_str(),
301 static_cast<long long>(flag.int64_default_for_display_));
302 } else if (flag.type_ == Flag::TYPE_BOOL) {
303 type_name = "bool";
304 flag_string =
305 strings::Printf("--%s=%s", flag.name_.c_str(),
306 flag.bool_default_for_display_ ? "true" : "false");
307 } else if (flag.type_ == Flag::TYPE_STRING) {
308 type_name = "string";
309 flag_string = strings::Printf("--%s=\"%s\"", flag.name_.c_str(),
310 flag.string_default_for_display_.c_str());
311 } else if (flag.type_ == Flag::TYPE_FLOAT) {
312 type_name = "float";
313 flag_string = strings::Printf("--%s=%f", flag.name_.c_str(),
314 flag.float_default_for_display_);
315 }
316 strings::Appendf(&usage_text, "\t%-33s\t%s\t%s\n", flag_string.c_str(),
317 type_name, flag.usage_text_.c_str());
318 }
319 return usage_text;
320}
