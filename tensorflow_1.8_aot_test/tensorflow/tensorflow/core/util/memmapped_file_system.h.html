27// A file system that uses a graph saved in memmapped format by
28// MemmappedEnvWriter as a file system.
29//
30// The format supports saved tensors and protos. Tensors are saved at aligned
31// offsets.
32//
33// Format specification:
34// - last 8 bytes of a package is encoded offset to the directory. The encoding
35// is always little endian, independently from the platform, done by functions
36// EncodeUint64LittleEndian/DecodeUint64LittleEndian
37// - the directory starts from the encoded offset and is saved proto
38// MemmappedFileSystemDirectory with names and offsets to the regions.
39// - at the offsets in the directory the file regions are stored. Tensor regions
40// are aligned such way that when the package mapped to RAM they have the right
41// offset to be used by ImmutableConst operator.
42//
43// Region naming:
44// Region naming is up to the application, all of them starts from
45// kMemmappedPackagePrefix. The default graph usually has name
46// kMemmappedPackageDefaultGraphDef; for more details see the conversion
47// utility
48// third_party/tensorflow/contrib/util/convert_graphdef_memmapped_format.cc
49//
50// A "frozen" GraphDef can be converted into this format using
51// tensorflow/contrib/util/convert_graphdef_memmapped_format
52class MemmappedFileSystem : public FileSystem {
53 public:
54 // Memmapped regions use this prefix to distinguish from
55 // the filesystem.
56#if defined(COMPILER_MSVC)
57 static constexpr char* kMemmappedPackagePrefix =
58#else
59 static constexpr char kMemmappedPackagePrefix[] =
60#endif
61 "memmapped_package://";
62
63// The default graphdef in the package.
64#if defined(COMPILER_MSVC)
65 static constexpr char* kMemmappedPackageDefaultGraphDef =
66#else
67 static constexpr char kMemmappedPackageDefaultGraphDef[] =
68#endif
69 "memmapped_package://.";
70
71 MemmappedFileSystem();
72 ~MemmappedFileSystem() override = default;
73 Status FileExists(const string& fname) override;
74 Status NewRandomAccessFile(
75 const string& filename,
76 std::unique_ptr<RandomAccessFile>* result) override;
77 Status NewReadOnlyMemoryRegionFromFile(
78 const string& filename,
79 std::unique_ptr<ReadOnlyMemoryRegion>* result) override;
80
81 // All these functions return Unimplemented error, the memmapped storage is
82 // read only.
83 Status NewWritableFile(const string& fname,
84 std::unique_ptr<WritableFile>* result) override;
85 Status NewAppendableFile(const string& fname,
86 std::unique_ptr<WritableFile>* result) override;
87 Status GetChildren(const string& dir, std::vector<string>* r) override;
88 Status GetMatchingPaths(const string& pattern,
89 std::vector<string>* results) override;
90 Status DeleteFile(const string& f) override;
91 Status CreateDir(const string& d) override;
92 Status DeleteDir(const string& d) override;
93 Status RenameFile(const string& s, const string& t) override;
94
95 // These functions are implemented.
96 Status GetFileSize(const string& f, uint64* s) override;
97 // Currently just returns size.
98 Status Stat(const string& fname, FileStatistics* stat) override;
99
100 // Initializes filesystem from a file in memmapped format.
101 Status InitializeFromFile(Env* env, const string& filename);
102
103 // Checks if the filename has a correct prefix.
104 static bool IsMemmappedPackageFilename(const string& filename);
105
106 static bool IsWellFormedMemmappedPackageFilename(const string& filename);
107
108 private:
109 struct FileRegion {
110 FileRegion(uint64 o, uint64 l) : offset(o), length(l) {}
111
112 uint64 offset; // Offset from the beginning of the file.
113 uint64 length; // Length of the region.
114 };
115
116 using DirectoryType = std::unordered_map<string, FileRegion>;
117
118 const void* GetMemoryWithOffset(uint64 offset) const;
119
120 std::unique_ptr<ReadOnlyMemoryRegion> mapped_memory_;
121 DirectoryType directory_;
122
123 TF_DISALLOW_COPY_AND_ASSIGN(MemmappedFileSystem);
124};
125
126class MemmappedEnv : public EnvWrapper {
127 public:
128 explicit MemmappedEnv(Env* env);
129 ~MemmappedEnv() override = default;
130 Status GetFileSystemForFile(const string& fname,
131 FileSystem** result) override;
132 Status GetRegisteredFileSystemSchemes(std::vector<string>* schemes) override;
133 Status InitializeFromFile(const string& filename);
134
135 protected:
136 std::unique_ptr<MemmappedFileSystem> memmapped_file_system_;
137};
