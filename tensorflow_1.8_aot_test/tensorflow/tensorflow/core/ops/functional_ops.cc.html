24REGISTER_OP("SymbolicGradient")
25 .Input("input: Tin")
26 .Output("output: Tout")
27 .Attr("Tin: list(type)")
28 .Attr("Tout: list(type)")
29 .Attr("f: func")
30 .SetShapeFn([](InferenceContext* c) {
31 if (c->num_inputs() < c->num_outputs()) {
32 return errors::InvalidArgument("len(inputs) < len(outputs)");
33 }
34 // Say, (u, v) = f(x, y, z), _symbolic_gradient(f) is a function of
35 // (x, y, z, du, dv) -> (dx, dy, dz). Therefore, shapes of its
36 // outputs (dx, dy, dz) are the same as (x, y, z).
37 for (int i = 0; i < c->num_outputs(); ++i) {
38 c->set_output(i, c->input(i));
39 }
40 return Status::OK();
41 });

54REGISTER_OP("_If")
55 .Input("cond: Tcond")
56 .Input("input: Tin")
57 .Output("output: Tout")
58 .Attr("Tcond: type")
59 .Attr("Tin: list(type)")
60 .Attr("Tout: list(type)")
61 .Attr("then_branch: func")
62 .Attr("else_branch: func")
63 .SetShapeFn(shape_inference::UnknownShape)
64 .Doc(R"doc(
65output = cond ? then_branch(input) : else_branch(input)
66
67cond: A Tensor. If the tensor is a scalar of non-boolean type, the
68 scalar is converted to a boolean according to the
69 following rule: if the scalar is a numerical value, non-zero means
70 True and zero means False; if the scalar is a string, non-empty
71 means True and empty means False. If the tensor is not a scalar,
72 being empty means False and being non-empty means True.
73input: A list of input tensors.
74then_branch: A function that takes 'inputs' and returns a list of
75 tensors, whose types are the same as what else_branch returns.
76else_branch: A function that takes 'inputs' and returns a list of
77 tensors. whose types are the same as what then_branch returns.
78)doc");

91// TODO(drpng): remove this.
92REGISTER_OP("_While")
93 .Input("input: T")
94 .Output("output: T")
95 .Attr("T: list(type) >= 0")
96 .Attr("cond: func")
97 .Attr("body: func")
98 .SetIsStateful()
99 .SetShapeFn([](shape_inference::InferenceContext* c) {
100 for (int i = 0; i < c->num_outputs(); ++i) {
101 c->set_output(i, c->input(i));
102 }
103 return Status::OK();
104 })
105 .Doc(R"doc(
106output = input; While (Cond(output)) { output = Body(output) }
107
108input: A list of input tensors whose types are T.
109output: A list of output tensors whose types are T.
110cond: A function takes 'input' and returns a tensor. If the tensor is
111 a scalar of non-boolean, the scalar is converted to a boolean
112 according to the following rule: if the scalar is a numerical
113 value, non-zero means True and zero means False; if the scalar is
114 a string, non-empty means True and empty means False. If the
115 tensor is not a scalar, non-emptiness means True and False
116 otherwise.
117body: A function that takes a list of tensors and returns another
118 list of tensors. Both lists have the same types as specified
119 by T.
120)doc");
