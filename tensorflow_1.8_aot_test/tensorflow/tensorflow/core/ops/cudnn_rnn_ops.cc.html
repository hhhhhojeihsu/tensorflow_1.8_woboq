60REGISTER_OP("CudnnRNN")
61 .Input("input: T")
62 .Input("input_h: T")
63 .Input("input_c: T")
64 .Input("params: T")
65 .SetIsStateful()
66 .Output("output: T")
67 .Output("output_h: T")
68 .Output("output_c: T")
69 .Output("reserve_space: T")
70 .Attr("T: {float16, float32, float64}")
71 .Attr(kRNNModeAttrs)
72 .Attr(kRNNInputModeAttrs)
73 .Attr(kRNNDirectionAttrs)
74 .Attr("dropout: float = 0.0")
75 .Attr("seed: int = 0")
76 .Attr("seed2: int = 0")
77 .Attr("is_training: bool = true")
78 .SetShapeFn([](InferenceContext* c) {
79 auto input_shape = c->input(0);
80 auto input_h_shape = c->input(1);
81 auto seq_length = c->Dim(input_shape, 0);
82 auto batch_size = c->Dim(input_shape, 1);
83 auto num_units = c->Dim(input_h_shape, 2);
84 string direction;
85 TF_RETURN_IF_ERROR(c->GetAttr("direction", &direction));
86 string rnn_mode;
87 TF_RETURN_IF_ERROR(c->GetAttr("rnn_mode", &rnn_mode));
88 int dir_count = (direction == "bidirectional") ? 2 : 1;
89 DimensionHandle output_size;
90 TF_RETURN_IF_ERROR(c->Multiply(num_units, dir_count, &output_size));
91 auto output_shape = c->MakeShape({seq_length, batch_size, output_size});
92 auto output_h_shape = input_h_shape;
93 auto output_c_shape TF_ATTRIBUTE_UNUSED =
94 (rnn_mode == "lstm") ? output_h_shape : c->MakeShape({});
95 c->set_output(0, output_shape);
96 c->set_output(1, output_h_shape);
97 c->set_output(2, output_c_shape);
98 c->set_output(3, c->UnknownShape());
99 return Status::OK();
100 });

103REGISTER_OP("CudnnRNNBackprop")
104 .Input("input: T")
105 .Input("input_h: T")
106 .Input("input_c: T")
107 .Input("params: T")
108 .Input("output: T")
109 .Input("output_h: T")
110 .Input("output_c: T")
111 .Input("output_backprop: T")
112 .Input("output_h_backprop: T")
113 .Input("output_c_backprop: T")
114 .Input("reserve_space: T")
115 .SetIsStateful()
116 .Output("input_backprop: T")
117 .Output("input_h_backprop: T")
118 .Output("input_c_backprop: T")
119 .Output("params_backprop: T")
120 .Attr("T: {float16, float32, float64}")
121 .Attr(kRNNModeAttrs)
122 .Attr(kRNNInputModeAttrs)
123 .Attr(kRNNDirectionAttrs)
124 .Attr("dropout: float = 0.0")
125 .Attr("seed: int = 0")
126 .Attr("seed2: int = 0")
127 .SetShapeFn([](InferenceContext* c) {
128 auto input_shape = c->input(0);
129 auto input_h_shape = c->input(1);
130 auto input_c_shape = c->input(2);
131 auto params_shape = c->input(3);
132 c->set_output(0, input_shape);
133 c->set_output(1, input_h_shape);
134 c->set_output(2, input_c_shape);
135 c->set_output(3, params_shape);
136 return Status::OK();
137 });

140REGISTER_OP("CudnnRNNParamsToCanonical")
141 .Input("num_layers: int32")
142 .Input("num_units: int32")
143 .Input("input_size: int32")
144 .Input("params: T")
145 .Output("weights: num_params * T")
146 .Output("biases: num_params * T")
147 .Attr("T: {float16, float32, float64}")
148 .Attr("num_params: int")
149 .Attr(kRNNModeAttrs)
150 .Attr(kRNNInputModeAttrs)
151 .Attr(kRNNDirectionAttrs)
152 .Attr("dropout: float = 0.0")
153 .Attr("seed: int = 0")
154 .Attr("seed2: int = 0")
155 .SetShapeFn([](InferenceContext* c) {
156 ShapeHandle unused;
157 TF_RETURN_IF_ERROR(c->WithRank(c->input(3), 1, &unused));
158 int num_params;
159 TF_RETURN_IF_ERROR(c->GetAttr("num_params", &num_params));
160 // Set shape for weight matrices
161 for (int i = 0; i < num_params; i++) {
162 c->set_output(i, c->Matrix(InferenceContext::kUnknownDim,
163 InferenceContext::kUnknownDim));
164 }
165 // Set shape for bias vectors
166 for (int i = 0; i < num_params; i++) {
167 c->set_output(num_params + i, c->Vector(InferenceContext::kUnknownDim));
168 }
169 return Status::OK();
170 });
