16#include "tensorflow/core/framework/common_shape_fns.h"

18#include "tensorflow/core/framework/shape_inference.h"

20namespace tensorflow {
21
22REGISTER_OP("Assert")
23 .Input("condition: bool")
24 .Input("data: T")
25 .SetIsStateful()
26 .Attr("T: list(type)")
27 .Attr("summarize: int = 3")
28 .SetShapeFn(shape_inference::NoOutputs);
29
30REGISTER_OP("Print")
31 .Input("input: T")
32 .Input("data: U")
33 .Output("output: T")
34 .SetIsStateful()
35 .Attr("T: type")
36 .Attr("U: list(type) >= 0")
37 .Attr("message: string = ''")
38 .Attr("first_n: int = -1")
39 .Attr("summarize: int = 3")
40 .SetShapeFn(shape_inference::UnchangedShape);

42// ----------------------------------------------------------------------------
43// Operators that deal with SummaryProtos (encoded as DT_STRING tensors) as
44// inputs or outputs in various ways.
45
46REGISTER_OP("TensorSummaryV2")
47 .Input("tag: string")
48 .Input("tensor: T")
49 // This serialized summary metadata field describes a summary value,
50 // specifically which plugins may use that summary.
51 .Input("serialized_summary_metadata: string")
52 .Output("summary: string")
53 .Attr("T: type")
54 .SetShapeFn(shape_inference::ScalarShape);
55
56REGISTER_OP("TensorSummary")
57 .Input("tensor: T")
58 .Output("summary: string")
59 .Attr("T: type")
60 .Attr("description: string = ''")
61 .Attr("labels: list(string) = []")
62 .Attr("display_name: string = ''")
63 .SetShapeFn(shape_inference::ScalarShape);
64
65REGISTER_OP("ScalarSummary")
66 .Input("tags: string")
67 .Input("values: T")
68 .Output("summary: string")
69 .Attr("T: realnumbertype")
70 .SetShapeFn(shape_inference::ScalarShape);
71
72REGISTER_OP("HistogramSummary")
73 .Input("tag: string")
74 .Input("values: T")
75 .Output("summary: string")
76 .Attr("T: realnumbertype = DT_FLOAT")
77 .SetShapeFn(shape_inference::ScalarShape);
78
79REGISTER_OP("ImageSummary")
80 .Input("tag: string")
81 .Input("tensor: T")
82 .Output("summary: string")
83 .Attr("max_images: int >= 1 = 3")
84 .Attr("T: {uint8, float, half, float64} = DT_FLOAT")
85 .Attr(
86 "bad_color: tensor = { dtype: DT_UINT8 "
87 "tensor_shape: { dim { size: 4 } } "
88 "int_val: 255 int_val: 0 int_val: 0 int_val: 255 }")
89 .SetShapeFn(shape_inference::ScalarShape);
90
91REGISTER_OP("AudioSummaryV2")
92 .Input("tag: string")
93 .Input("tensor: float")
94 .Input("sample_rate: float")
95 .Output("summary: string")
96 .Attr("max_outputs: int >= 1 = 3")
97 .SetShapeFn(shape_inference::ScalarShape);
98
99REGISTER_OP("AudioSummary")
100 .Input("tag: string")
101 .Input("tensor: float")
102 .Output("summary: string")
103 .Attr("sample_rate: float")
104 .Attr("max_outputs: int >= 1 = 3")
105 .SetShapeFn(shape_inference::ScalarShape)
106 .Deprecated(15, "Use AudioSummaryV2.");
107
108REGISTER_OP("MergeSummary")
109 .Input("inputs: N * string")
110 .Output("summary: string")
111 .Attr("N : int >= 1")
112 .SetShapeFn(shape_inference::ScalarShape);
113
114REGISTER_OP("Timestamp")
115 .Output("ts: float64")
116 .SetIsStateful()
117 .SetShapeFn(shape_inference::ScalarShape);
