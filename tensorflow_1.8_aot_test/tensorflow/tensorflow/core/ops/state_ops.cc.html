33REGISTER_OP("Variable")
34 .Output("ref: Ref(dtype)")
35 .Attr("shape: shape")
36 .Attr("dtype: type")
37 .Attr("container: string = ''")
38 .Attr("shared_name: string = ''")
39 .SetIsStateful()
40 .SetShapeFn([](InferenceContext* c) {
41 PartialTensorShape shape;
42 TF_RETURN_IF_ERROR(c->GetAttr("shape", &shape));
43
44 // Variable has legacy behavior where we cannot tell the difference
45 // between a scalar shape attribute and 'unknown shape'. So if the shape
46 // is a scalar, we return an unknown shape.
47 if (shape.dims() <= 0) {
48 return shape_inference::UnknownShape(c);
49 }
50
51 ShapeHandle out;
52 TF_RETURN_IF_ERROR(c->MakeShapeFromPartialTensorShape(shape, &out));
53 c->set_output(0, out);
54 return Status::OK();
55 });

116Status ScatterUpdateShape(InferenceContext* c) {
117 ShapeHandle var_shape = c->input(0);
118 ShapeHandle indices_shape = c->input(1);
119
120 ShapeHandle unused_updates_shape;
121 ShapeHandle concat;
122 ShapeHandle var_subshape;
123 TF_RETURN_IF_ERROR(c->Subshape(var_shape, 1, &var_subshape));
124 TF_RETURN_IF_ERROR(c->Concatenate(indices_shape, var_subshape, &concat));
125 TF_RETURN_IF_ERROR(
126 InferenceContext::Rank(c->input(2)) == 0
127 ? Status::OK()
128 : c->Merge(c->input(2), concat, &unused_updates_shape));
129
130 c->set_output(0, var_shape);
131 return Status::OK();
132}

245REGISTER_OP("CountUpTo")
246 .Input("ref: Ref(T)")
247 .Output("output: T")
248 .Attr("limit: int")
249 .Attr("T: {int32, int64}")
250 .SetShapeFn([](InferenceContext* c) {
251 ShapeHandle output;
252 TF_RETURN_IF_ERROR(c->WithRank(c->input(0), 0, &output));
253 c->set_output(0, output);
254 return Status::OK();
255 });

257REGISTER_OP("ResourceCountUpTo")
258 .Input("resource: resource")
259 .Output("output: T")
260 .Attr("limit: int")
261 .Attr("T: {int32, int64}")
262 .SetShapeFn([](InferenceContext* c) {
263 auto* handle_data = c->input_handle_shapes_and_types(0);
264 if (handle_data == nullptr || handle_data->empty()) {
265 return errors::InvalidArgument("Handle has no shape/type information.");
266 }
267 shape_inference::ShapeAndType shape_and_type = (*handle_data)[0];
268 DataType value_dtype;
269 TF_RETURN_IF_ERROR(c->GetAttr("T", &value_dtype));
270 if (value_dtype != shape_and_type.dtype) {
271 return errors::InvalidArgument(
272 "Data types do not match: ", DataTypeString(value_dtype), " and ",
273 DataTypeString(shape_and_type.dtype));
274 }
275 ShapeHandle output;
276 TF_RETURN_IF_ERROR(c->WithRank(shape_and_type.shape, 0, &output));
277 c->set_output(0, output);
278 return Status::OK();
279 });
