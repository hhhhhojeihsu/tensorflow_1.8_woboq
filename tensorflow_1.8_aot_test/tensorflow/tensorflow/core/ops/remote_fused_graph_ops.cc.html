16#include "tensorflow/core/framework/common_shape_fns.h"
17#include "tensorflow/core/framework/op.h"
18#include "tensorflow/core/framework/shape_inference.h"
19
20namespace tensorflow {
21
22namespace {
23using shape_inference::InferenceContext;
24
25Status RemoteFusedGraphExecuteShapeFn(InferenceContext* c) {
26 for (int i = 0; i < c->num_outputs(); ++i) {
27 c->set_output(i, c->UnknownShape());
28 }
29 return Status::OK();
30}
31} // namespace
32
33REGISTER_OP("RemoteFusedGraphExecute")
34 .Input("inputs: Tinputs")
35 .Output("outputs: Toutputs")
36 .Attr("Tinputs: list(type) >= 0")
37 .Attr("Toutputs: list(type) >= 0")
38 .Attr("serialized_remote_fused_graph_execute_info: string")
39 .SetShapeFn(RemoteFusedGraphExecuteShapeFn);
40
41} // namespace tensorflow
