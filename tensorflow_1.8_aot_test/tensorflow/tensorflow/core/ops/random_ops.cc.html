91REGISTER_OP("Multinomial")
92 .SetIsStateful()
93 .Input("logits: T")
94 .Input("num_samples: int32")
95 .Output("output: output_dtype")
96 .Attr("seed: int = 0")
97 .Attr("seed2: int = 0")
98 .Attr("T: realnumbertype")
99 .Attr("output_dtype: {int32, int64} = DT_INT64")
100 .SetShapeFn([](InferenceContext* c) {
101 ShapeHandle logits_shape;
102 ShapeHandle unused;
103 DimensionHandle num_samples;
104 TF_RETURN_IF_ERROR(c->WithRank(c->input(0), 2, &logits_shape));
105 TF_RETURN_IF_ERROR(c->WithRank(c->input(1), 0, &unused));
106 TF_RETURN_IF_ERROR(c->MakeDimForScalarInput(1, &num_samples));
107 c->set_output(0, c->Matrix(c->Dim(logits_shape, 0), num_samples));
108 return Status::OK();
109 });

111REGISTER_OP("RandomGamma")
112 .SetIsStateful()
113 .Input("shape: S")
114 .Input("alpha: T")
115 .Output("output: T")
116 .Attr("seed: int = 0")
117 .Attr("seed2: int = 0")
118 .Attr("S: {int32, int64}")
119 .Attr("T: {half, float, double}")
120 .SetShapeFn([](InferenceContext* c) {
121 ShapeHandle out;
122 TF_RETURN_IF_ERROR(c->MakeShapeFromShapeTensor(0, &out));
123 TF_RETURN_IF_ERROR(c->Concatenate(out, c->input(1), &out));
124 c->set_output(0, out);
125 return Status::OK();
126 });
