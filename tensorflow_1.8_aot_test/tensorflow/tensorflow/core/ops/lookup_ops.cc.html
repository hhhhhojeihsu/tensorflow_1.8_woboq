31Status TwoElementVectorInputsAndScalarOutputs(InferenceContext* c) {
32 ShapeHandle handle;
33 DimensionHandle unused_handle;
34 for (int i = 0; i < c->num_inputs(); ++i) {
35 TF_RETURN_IF_ERROR(c->WithRank(c->input(i), 1, &handle));
36 TF_RETURN_IF_ERROR(c->WithValue(c->Dim(handle, 0), 2, &unused_handle));
37 }
38 for (int i = 0; i < c->num_outputs(); ++i) {
39 c->set_output(i, c->Scalar());
40 }
41 return Status::OK();
42}
43
44Status ScalarAndTwoElementVectorInputsAndScalarOutputs(InferenceContext* c) {
45 ShapeHandle handle;
46 DimensionHandle unused_handle;
47 TF_RETURN_IF_ERROR(c->WithRank(c->input(0), 0, &handle));
48 for (int i = 1; i < c->num_inputs(); ++i) {
49 TF_RETURN_IF_ERROR(c->WithRank(c->input(i), 1, &handle));
50 TF_RETURN_IF_ERROR(c->WithValue(c->Dim(handle, 0), 2, &unused_handle));
51 }
52 for (int i = 0; i < c->num_outputs(); ++i) {
53 c->set_output(i, c->Scalar());
54 }
55 return Status::OK();
56}
57
58Status TwoElementOutput(InferenceContext* c) {
59 c->set_output(0, c->Vector(2));
60 return Status::OK();
61}
62
63Status ScalarOutput(InferenceContext* c) {
64 c->set_output(0, c->Scalar());
65 return Status::OK();
66}

69REGISTER_OP("LookupTableFind")
70 .Input("table_handle: Ref(string)")
71 .Input("keys: Tin")
72 .Input("default_value: Tout")
73 .Output("values: Tout")
74 .Attr("Tin: type")
75 .Attr("Tout: type")
76 .SetShapeFn([](InferenceContext* c) {
77 ShapeHandle handle;
78 TF_RETURN_IF_ERROR(c->WithRank(c->input(0), 1, &handle));
79 DimensionHandle unused_dim;
80 TF_RETURN_IF_ERROR(c->WithValue(c->Dim(handle, 0), 2, &unused_dim));
81
82 // Default value must be scalar or vector.
83 ShapeHandle unused;
84 TF_RETURN_IF_ERROR(c->WithRankAtMost(c->input(2), 1, &unused));
85 c->set_output(0, c->UnknownShape());
86 return Status::OK();
87 });

150 .Input("table_handle: Ref(string)")
151 .Output("keys: Tkeys")

154 .Attr("Tvalues: type")
155 .SetShapeFn([](InferenceContext* c) {

158 DimensionHandle unused_dim;

160

162 TF_RETURN_IF_ERROR(c->WithRankAtLeast(values, 1, &values));

307REGISTER_OP("InitializeTable")
308 .Input("table_handle: Ref(string)")
309 .Input("keys: Tkey")
310 .Input("values: Tval")
311 .Attr("Tkey: type")
312 .Attr("Tval: type")
313 .SetShapeFn([](InferenceContext* c) {
314 ShapeHandle handle;
315 TF_RETURN_IF_ERROR(c->WithRank(c->input(0), 1, &handle));
316 DimensionHandle unused_dim;
317 TF_RETURN_IF_ERROR(c->WithValue(c->Dim(handle, 0), 2, &unused_dim));
318
319 ShapeHandle keys;
320 TF_RETURN_IF_ERROR(c->WithRank(c->input(1), 1, &keys));
321 TF_RETURN_IF_ERROR(c->Merge(keys, c->input(2), &keys));
322 return Status::OK();
323 });
