45// Compiles the XLA computation into executable code.
46Status CompileXla(xla::CompileOnlyClient* client,
47 const xla::Computation& computation,
48 const xla::cpu::CpuAotCompilationOptions& aot_opts,
49 CompileResult* compile_result) {
50 // Retrieves arg and result layouts from the computation.
51 // TODO(toddw): Should we let the user choose the major/minor ordering?
52 xla::StatusOr<std::unique_ptr<xla::ProgramShape>> pshape_or =
53 client->GetComputationShape(computation);
54 if (!pshape_or.ok()) {
55 return errors::Unknown("Couldn't get XLA program shape: ",
56 pshape_or.status().error_message());
57 }
58 compile_result->program_shape = *pshape_or.ValueOrDie();
59 xla::ProgramShape* pshape = &compile_result->program_shape;
60 std::vector<const xla::Shape*> arg_layouts;
61 arg_layouts.reserve(pshape->parameters_size());
62 for (int i = 0; i < pshape->parameters_size(); ++i) {
63 arg_layouts.push_back(pshape->mutable_parameters(i));
64 }
65 xla::CompileOnlyClient::AotComputationInstance instance;
66 instance.computation = &computation;
67 instance.argument_layouts = std::move(arg_layouts);
68 instance.result_layout = &pshape->result();
69 xla::StatusOr<std::vector<std::unique_ptr<xla::AotCompilationResult>>>
70 aot_or = client->CompileAheadOfTime({instance}, aot_opts);
71 if (!aot_or.ok()) {
72 return errors::Unknown("XLA compilation failed: ",
73 aot_or.status().error_message());
74 }
75 compile_result->aot =
76 xla::unique_ptr_static_cast<xla::cpu::CpuAotCompilationResult>(
77 std::move(aot_or.ValueOrDie().back()));
78 compile_result->entry_point = aot_opts.entry_point_name();
79 compile_result->pointer_size =
80 xla::CompileOnlyClient::PointerSizeForTriple(aot_opts.triple());
81 return Status::OK();
82}

86Status CompileGraph(const GraphDef& graph_def, const tf2xla::Config& config,
87 const MainFlags& flags, CompileResult* compile_result) {
88 // Converts the graph into an XLA computation, and compiles the
89 // computation.
90 // TODO(toddw): Should we let the user pick the XLA cpu vs. gpu client?
91 namespace gpu = perftools::gputools;
92 gpu::Platform* cpu_platform =
93 gpu::MultiPlatformManager::PlatformWithName("Host").ValueOrDie();
94 xla::CompileOnlyClient* client =
95 xla::ClientLibrary::GetOrCreateCompileOnlyClient(cpu_platform)
96 .ValueOrDie();
97 xla::Computation computation;
98 TF_RETURN_IF_ERROR(
99 ConvertGraphDefToXla(graph_def, config, client, &computation));
100 if (!flags.out_session_module.empty()) {
101 TF_ASSIGN_OR_RETURN(std::unique_ptr<xla::SessionModule> module,
102 computation.Snapshot());
103 // Serialize the SessionModule deterministically so that all the outputs of
104 // a tf_library genrule are deterministic.
105 string proto;
106 TF_RET_CHECK(SerializeToStringDeterministic(*module, &proto));
107 TF_RETURN_IF_ERROR(
108 WriteStringToFile(Env::Default(), flags.out_session_module, proto));
109 }
110 xla::cpu::CpuAotCompilationOptions aot_opts(
111 flags.target_triple, flags.target_cpu, flags.target_features,
112 flags.entry_point,
113 xla::cpu::CpuAotCompilationOptions::RelocationModel::BigPic);
114 return CompileXla(client, computation, aot_opts, compile_result);
115}
