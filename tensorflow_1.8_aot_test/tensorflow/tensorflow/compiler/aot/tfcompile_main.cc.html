47const char kUsageHeader[] =
48 "tfcompile performs ahead-of-time compilation of a TensorFlow graph,\n"
49 "resulting in an object file compiled for your target architecture, and a\n"
50 "header file that gives access to the functionality in the object file.\n"
51 "A typical invocation looks like this:\n"
52 "\n"
53 " $ tfcompile --graph=mygraph.pb --config=myfile.pbtxt "
54 "--cpp_class=\"mynamespace::MyComputation\"\n"
55 "\n";
56
57Status ReadProtoFile(const string& fname, protobuf::Message* proto) {
58 if (str_util::EndsWith(fname, ".pbtxt")) {
59 return ReadTextProto(Env::Default(), fname, proto);
60 } else {
61 return ReadBinaryProto(Env::Default(), fname, proto);
62 }
63}
64
65Status Main(const MainFlags& flags) {
66 // Process config.
67 tf2xla::Config config;
68 if (flags.config.empty()) {
69 return errors::InvalidArgument("Must specify --config");
70 }
71 TF_RETURN_IF_ERROR(ReadProtoFile(flags.config, &config));
72 TF_RETURN_IF_ERROR(ValidateConfig(config));
73 if (flags.dump_fetch_nodes) {
74 std::set<string> nodes;
75 for (const tf2xla::Fetch& fetch : config.fetch()) {
76 nodes.insert(fetch.id().node_name());
77 }
78 std::cout << str_util::Join(nodes, ",");
79 return Status::OK();
80 }
81
82 // Read and initialize the graph.
83 if (flags.graph.empty()) {
84 return errors::InvalidArgument("Must specify --graph");
85 }
86 GraphDef graph_def;
87 TF_RETURN_IF_ERROR(ReadProtoFile(flags.graph, &graph_def));
88 CompileResult compile_result;
89 TF_RETURN_IF_ERROR(CompileGraph(graph_def, config, flags, &compile_result));
90
91 // Write output files.
92 Env* env = Env::Default();
93 const std::vector<char>& obj = compile_result.aot->object_file_data();
94 TF_RETURN_IF_ERROR(WriteStringToFile(env, flags.out_function_object,
95 StringPiece(obj.data(), obj.size())));
96 CodegenOpts codegen_opts;
97 codegen_opts.gen_name_to_index = flags.gen_name_to_index;
98 codegen_opts.gen_program_shape = flags.gen_program_shape;
99 codegen_opts.target_triple = flags.target_triple;
100 if (flags.cpp_class.empty()) {
101 return errors::InvalidArgument("Must specify --cpp_class");
102 }
103 TF_RETURN_IF_ERROR(ParseCppClass(flags.cpp_class, &codegen_opts.class_name,
104 &codegen_opts.namespaces));
105
106 MetadataResult metadata_result;
107 TF_RETURN_IF_ERROR(
108 GenerateMetadata(codegen_opts, compile_result, &metadata_result));
109 TF_RETURN_IF_ERROR(WriteStringToFile(env, flags.out_metadata_object,
110 metadata_result.object_file_data));
111 string header;
112 TF_RETURN_IF_ERROR(GenerateHeader(codegen_opts, config, compile_result,
113 metadata_result, &header));
114 TF_RETURN_IF_ERROR(WriteStringToFile(env, flags.out_header, header));
115 return Status::OK();
116}

121int main(int argc, char** argv) {
122 tensorflow::tfcompile::MainFlags flags;
123 flags.target_triple = "x86_64-pc-linux";
124 flags.out_function_object = "out_model.o";
125 flags.out_metadata_object = "out_helper.o";
126 flags.out_header = "out.h";
127 flags.entry_point = "entry";
128
129 std::vector<tensorflow::Flag> flag_list;
130 AppendMainFlags(&flag_list, &flags);
131 xla::legacy_flags::AppendDebugOptionsFlags(&flag_list);
132
133 tensorflow::string usage = tensorflow::tfcompile::kUsageHeader;
134 usage += tensorflow::Flags::Usage(argv[0], flag_list);
135 bool parsed_flags_ok = tensorflow::Flags::Parse(&argc, argv, flag_list);
136 QCHECK(parsed_flags_ok) << "\n" << usage;
137
138 tensorflow::port::InitMain(usage.c_str(), &argc, &argv);
139 QCHECK(argc == 1) << "\nERROR: This command does not take any arguments "
140 "other than flags\n\n"
141 << usage;
142 tensorflow::Status status = tensorflow::tfcompile::Main(flags);
143 if (status.code() == tensorflow::error::INVALID_ARGUMENT) {
144 std::cerr << "INVALID ARGUMENTS: " << status.error_message() << "\n\n"
145 << usage;
146 return 1;
147 } else {
148 TF_QCHECK_OK(status);
149 }
150 return 0;
151}
