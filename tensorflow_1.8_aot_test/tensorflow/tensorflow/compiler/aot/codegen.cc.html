39bool IsAlpha(char c) {
40 return (c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z');
41}
42
43bool IsAlphaNum(char c) { return IsAlpha(c) || (c >= '0' && c <= '9'); }

45// Convert an XLA type into a C++ type.
46Status XLATypeToCpp(xla::PrimitiveType type, string* str) {
47 switch (type) {
48 case xla::PRED:
49 *str = "bool";
50 break;
51 case xla::S8:
52 *str = "tensorflow::int8";
53 break;
54 case xla::S16:
55 *str = "tensorflow::int16";
56 break;
57 case xla::S32:
58 *str = "tensorflow::int32";
59 break;
60 case xla::S64:
61 *str = "tensorflow::int64";
62 break;
63 case xla::U8:
64 *str = "tensorflow::uint8";
65 break;
66 case xla::U16:
67 *str = "tensorflow::uint16";
68 break;
69 case xla::U32:
70 *str = "tensorflow::uint32";
71 break;
72 case xla::U64:
73 *str = "tensorflow::uint64";
74 break;
75 case xla::F32:
76 *str = "float";
77 break;
78 case xla::F64:
79 *str = "double";
80 break;
81 default:
82 return errors::Unimplemented("XLA type ", xla::PrimitiveType_Name(type),
83 " has no equivalent in C++");
84 }
85 return Status::OK();
86}

88// total_buffer_bytes returns the sum of each size in `sizes`, skipping -1
89// values. There are `n` entries in `sizes`.
90size_t total_buffer_bytes(const intptr_t* sizes, size_t n) {
91 size_t total = 0;
92 for (size_t i = 0; i < n; ++i) {
93 if (sizes[i] != -1) {
94 total += sizes[i];
95 }
96 }
97 return total;
98}

100// Fills in arg_sizes with the byte size of each positional arg.
101Status ComputeArgSizes(const CompileResult& compile_result,
102 std::vector<int64>* arg_sizes) {
103 const xla::ProgramShape& ps = compile_result.program_shape;
104 for (int i = 0; i < ps.parameters_size(); ++i) {
105 arg_sizes->push_back(xla::ShapeUtil::ByteSizeOf(
106 ps.parameters(i), compile_result.pointer_size));
107 }
108 return Status::OK();
109}

111// Add (from,to) rewrite pairs based on the given shape. These rewrite pairs
112// are used to generate methods for args and results.
113Status AddRewritesForShape(int i, const xla::Shape& shape,
114 std::vector<std::pair<string, string>>* rewrites) {
115 string type;
116 TF_RETURN_IF_ERROR(XLATypeToCpp(shape.element_type(), &type));
117 std::vector<string> dim_vars;
118 string dim_sizes, indices;
119 if (xla::ShapeUtil::Rank(shape) == 0 ||
120 (shape.dimensions_size() == 1 && shape.dimensions(0) == 1)) {
121 dim_sizes = "[1]";
122 indices = "[0]";
123 } else {
124 for (int dim = 0; dim < shape.dimensions_size(); ++dim) {
125 dim_vars.push_back(strings::StrCat("size_t dim", dim));
126 dim_sizes += strings::StrCat("[", shape.dimensions(dim), "]");
127 indices += strings::StrCat("[dim", dim, "]");
128 }
129 }
130 rewrites->push_back({"{{I}}", strings::StrCat(i)});
131 rewrites->push_back({"{{TYPE}}", type});
132 rewrites->push_back({"{{DIM_VARS}}", str_util::Join(dim_vars, ", ")});
133 rewrites->push_back({"{{DIM_SIZES}}", dim_sizes});
134 rewrites->push_back({"{{INDICES}}", indices});
135 return Status::OK();
136}
137
138// Returns code rewritten by replacing all rewrite pairs, with an extra rewrite
139// for the name. Note that the rewriting strategy is roughly O(N*M), where N is
140// the size of the code and M is the number of rewrites. It's fine for now
141// since N and M are pretty small.
142//
143// TODO(toddw): If this becomes a problem, we should be able to change the
144// algorithm to O(N) by using a state machine, e.g. regexps or a real
145// text-templating mechanism.
146string RewriteWithName(const string& name, string code,
147 const std::vector<std::pair<string, string>>& rewrites) {
148 str_util::ReplaceAllPairs(&code, rewrites);
149 return str_util::StringReplace(code, "{{NAME}}", name, /*replace_all=*/true);
150}

152// Generate methods for args (inputs).
153Status GenArgMethods(const tf2xla::Config& config, const xla::ProgramShape& ps,
154 const CompileResult& compile_result, string* methods) {
155 size_t num_args = ps.parameters_size();
156 if (config.feed_size() != num_args) {
157 return errors::InvalidArgument("mismatch between feed_size(",
158 config.feed_size(), ") and num_args(",
159 num_args, ")");
160 }
161 for (int i = 0; i < num_args; ++i) {
162 std::vector<std::pair<string, string>> rewrites;
163 TF_RETURN_IF_ERROR(AddRewritesForShape(i, ps.parameters(i), &rewrites));
164 const string code = R"(
165 void set_arg{{NAME}}_data(void* data) {
166 set_arg_data({{I}}, data);
167 }
168 {{TYPE}}* arg{{NAME}}_data() {
169 return static_cast<{{TYPE}}*>(arg_data({{I}}));
170 }
171 {{TYPE}}& arg{{NAME}}({{DIM_VARS}}) {
172 return (*static_cast<{{TYPE}}(*){{DIM_SIZES}}>(
173 arg_data({{I}}))){{INDICES}};
174 }
175 const {{TYPE}}* arg{{NAME}}_data() const {
176 return static_cast<const {{TYPE}}*>(arg_data({{I}}));
177 }
178 const {{TYPE}}& arg{{NAME}}({{DIM_VARS}}) const {
179 return (*static_cast<const {{TYPE}}(*){{DIM_SIZES}}>(
180 arg_data({{I}}))){{INDICES}};
181 }
182)";
183 *methods += RewriteWithName(strings::StrCat(i), code, rewrites);
184 if (!config.feed(i).name().empty()) {
185 *methods += RewriteWithName("_" + config.feed(i).name(), code, rewrites);
186 }
187 }
188 return Status::OK();
189}
190
191// Generate methods for results (outputs).
192Status GenResultMethods(const tf2xla::Config& config,
193 const xla::ProgramShape& ps, string* methods) {
194 if (ps.result().element_type() != xla::TUPLE) {
195 // The XlaCompiler we use to build the xla computation always generates a
196 // tuple result, and we rely on this to simplify code generation.
197 return errors::Internal("codegen requires the XLA result to be a tuple");
198 }
199 if (config.fetch_size() != ps.result().tuple_shapes_size()) {
200 return errors::InvalidArgument("mismatch between fetch_size(",
201 config.feed_size(), ") and tuple_size(",
202 ps.result().tuple_shapes_size(), ")");
203 }
204 for (int i = 0; i < ps.result().tuple_shapes_size(); ++i) {
205 std::vector<std::pair<string, string>> rewrites;
206 TF_RETURN_IF_ERROR(
207 AddRewritesForShape(i, ps.result().tuple_shapes(i), &rewrites));
208 string code = R"(
209 {{TYPE}}* result{{NAME}}_data() {
210 return static_cast<{{TYPE}}*>(result_data({{I}}));
211 }
212 {{TYPE}}& result{{NAME}}({{DIM_VARS}}) {
213 return (*static_cast<{{TYPE}}(*){{DIM_SIZES}}>(
214 result_data({{I}}))){{INDICES}};
215 }
216 const {{TYPE}}* result{{NAME}}_data() const {
217 return static_cast<const {{TYPE}}*>(result_data({{I}}));
218 }
219 const {{TYPE}}& result{{NAME}}({{DIM_VARS}}) const {
220 return (*static_cast<const {{TYPE}}(*){{DIM_SIZES}}>(
221 result_data({{I}}))){{INDICES}};
222 }
223)";
224 *methods += RewriteWithName(strings::StrCat(i), code, rewrites);
225 if (!config.fetch(i).name().empty()) {
226 *methods += RewriteWithName("_" + config.fetch(i).name(), code, rewrites);
227 }
228 }
229 return Status::OK();
230}

232// Generates code implementing {Arg,Result}Names(), where T is one of
233// tf2xla::{Feed,Fetch}. Each feed or fetch name results in a C-style string
234// literal in the array, with nullptr terminating the array.
235template <typename T>
236string GenNameToIndexCode(const T& entries, bool generate) {
237 // No need for a static array if we're not supposed to generate the data.
238 if (!generate) {
239 return "{\n return nullptr;\n }";
240 }
241 // Determine when to stop. We stop emitting string literals after the last
242 // non-empty name.
243 int end = entries.size();
244 for (int i = entries.size() - 1; i >= 0; --i) {
245 if (!entries[i].name().empty()) {
246 break;
247 }
248 end = i;
249 }
250 // Emit string literals up to the last non-empty name.
251 string code = "{\n static const char* kNames[] = {";
252 for (int i = 0; i < end; ++i) {
253 if (i > 0) {
254 code += ", ";
255 }
256 code += "\"";
257 code += entries[i].name();
258 code += "\"";
259 }
260 if (end > 0) {
261 code += ", ";
262 }
263 code += "nullptr};\n return kNames;\n }";
264 return code;
265}

267Status ValidateFeedFetchCppNames(const tf2xla::Config& config) {
268 for (const tf2xla::Feed& feed : config.feed()) {
269 if (!feed.name().empty()) {
270 TF_RETURN_IF_ERROR(ValidateCppIdent(feed.name(), "feed name"));
271 }
272 }
273 for (const tf2xla::Fetch& fetch : config.fetch()) {
274 if (!fetch.name().empty()) {
275 TF_RETURN_IF_ERROR(ValidateCppIdent(fetch.name(), "fetch name"));
276 }
277 }
278 return Status::OK();
279}

283Status GenerateHeader(const CodegenOpts& opts, const tf2xla::Config& config,
284 const CompileResult& compile_result,
285 const MetadataResult& metadata_result, string* header) {
286 TF_RETURN_IF_ERROR(ValidateConfig(config));
287 TF_RETURN_IF_ERROR(ValidateFeedFetchCppNames(config));
288 const int64 result_index = compile_result.aot->result_buffer_index();
289 const xla::BufferSizes& temp_sizes = compile_result.aot->buffer_sizes();
290 if (result_index < 0 || result_index > temp_sizes.size()) {
291 return errors::InvalidArgument("result index: ", result_index,
292 " is outside the range of temp sizes: [0,",
293 temp_sizes.size(), ")");
294 }
295
296 // Compute sizes and generate methods.
297 std::vector<int64> arg_sizes;
298 TF_RETURN_IF_ERROR(ComputeArgSizes(compile_result, &arg_sizes));
299 const xla::ProgramShape& ps = compile_result.program_shape;
300 string methods_arg, methods_result;
301 TF_RETURN_IF_ERROR(GenArgMethods(config, ps, compile_result, &methods_arg));
302 TF_RETURN_IF_ERROR(GenResultMethods(config, ps, &methods_result));
303 const std::vector<intptr_t> iarg(arg_sizes.begin(), arg_sizes.end());
304 const std::vector<intptr_t> itemp(temp_sizes.begin(), temp_sizes.end());
305 const size_t arg_bytes_aligned =
306 runtime::aligned_buffer_bytes(iarg.data(), iarg.size());
307 const size_t arg_bytes_total = total_buffer_bytes(iarg.data(), iarg.size());
308 const size_t temp_bytes_aligned =
309 runtime::aligned_buffer_bytes(itemp.data(), itemp.size());
310 const size_t temp_bytes_total =
311 total_buffer_bytes(itemp.data(), itemp.size());
312
313 // Create rewrite strings for namespace start and end.
314 string ns_start;
315 for (const string& n : opts.namespaces) {
316 ns_start += strings::StrCat("namespace ", n, " {\n");
317 }
318 ns_start += "\n";
319 string ns_end("\n");
320 for (int i = opts.namespaces.size() - 1; i >= 0; --i) {
321 const string& n = opts.namespaces[i];
322 ns_end += strings::StrCat("} // end namespace ", n, "\n");
323 }
324
325 // Generate metadata.
326 const string arg_names_code =
327 GenNameToIndexCode(config.feed(), opts.gen_name_to_index);
328 const string result_names_code =
329 GenNameToIndexCode(config.fetch(), opts.gen_name_to_index);
330 const string include_xla_data_proto =
331 opts.gen_program_shape
332 ?
333 R"(#include "tensorflow/compiler/xla/xla_data.pb.h")"
334 : "";
335
336 // Use a poor-man's text templating mechanism; first populate the full header
337 // with placeholder tokens, and then rewrite the tokens with real values.
338 *header =
339 R"(// Generated by tfcompile, the TensorFlow graph compiler. DO NOT EDIT!
340//
341// This header was generated via ahead-of-time compilation of a TensorFlow
342// graph. An object file corresponding to this header was also generated.
343// This header gives access to the functionality in that object file.
344//
345// clang-format off
346
347#ifndef TFCOMPILE_GENERATED_{{ENTRY}}_H_ // NOLINT(build/header_guard)
348#define TFCOMPILE_GENERATED_{{ENTRY}}_H_ // NOLINT(build/header_guard)
349
350{{INCLUDE_XLA_DATA_PROTO}}
351#include "tensorflow/compiler/tf2xla/xla_compiled_cpu_function.h"
352#include "tensorflow/core/platform/types.h"
353
354namespace Eigen { struct ThreadPoolDevice; }
355namespace xla { class ExecutableRunOptions; }
356
357// (Implementation detail) Entry point to the function in the object file.
358extern "C" void {{ENTRY}}(
359 void* result, const xla::ExecutableRunOptions* run_options,
360 const void** args, void** temps, tensorflow::int64* profile_counters);
361
362{{DECLS_FROM_OBJ_FILE}}
363
364{{NS_START}}
365// {{CLASS}} represents a computation previously specified in a
366// TensorFlow graph, now compiled into executable code. This extends the generic
367// XlaCompiledCpuFunction class with statically type-safe arg and result
368// methods. Usage example:
369//
370// {{CLASS}} computation;
371// // ...set args using computation.argN methods
372// CHECK(computation.Run());
373// // ...inspect results using computation.resultN methods
374//
375// The Run method invokes the actual computation, with inputs read from arg
376// buffers, and outputs written to result buffers. Each Run call may also use
377// a set of temporary buffers for the computation.
378//
379// By default each instance of this class manages its own arg, result and temp
380// buffers. The AllocMode constructor parameter may be used to modify the
381// buffer allocation strategy.
382//
383// Under the default allocation strategy, this class is thread-compatible:
384// o Calls to non-const methods require exclusive access to the object.
385// o Concurrent calls to const methods are OK, if those calls are made while it
386// is guaranteed that no thread may call a non-const method.
387//
388// The logical function signature is:
389// {{PROGRAM_SHAPE}}
390//
391// Memory stats:
392// arg bytes total: {{ARG_BYTES_TOTAL}}
393// arg bytes aligned: {{ARG_BYTES_ALIGNED}}
394// temp bytes total: {{TEMP_BYTES_TOTAL}}
395// temp bytes aligned: {{TEMP_BYTES_ALIGNED}}
396class {{CLASS}} : public tensorflow::XlaCompiledCpuFunction {
397 public:
398 // Number of input arguments for the compiled computation.
399 static constexpr size_t kNumArgs = {{ARG_NUM}};
400
401 // Byte size of each argument buffer. There are kNumArgs entries.
402 static const intptr_t* ArgSizes() {
403 static constexpr intptr_t kArgSizes[kNumArgs] = {{{ARG_SIZES}}};
404 return kArgSizes;
405 }
406
407 // Returns static data used to create an XlaCompiledCpuFunction.
408 static const tensorflow::XlaCompiledCpuFunction::StaticData& StaticData() {
409 static XlaCompiledCpuFunction::StaticData* kStaticData = [](){
410 XlaCompiledCpuFunction::StaticData* data =
411 new XlaCompiledCpuFunction::StaticData;
412 data->raw_function = {{ENTRY}};
413 data->arg_sizes = ArgSizes();
414 data->num_args = kNumArgs;
415 data->temp_sizes = TempSizes();
416 data->num_temps = kNumTemps;
417 data->result_index = kResultIndex;
418 data->arg_names = StaticArgNames();
419 data->result_names = StaticResultNames();
420 data->program_shape = StaticProgramShape();
421 return data;
422 }();
423 return *kStaticData;
424 }
425
426 {{CLASS}}(AllocMode alloc_mode = AllocMode::ARGS_RESULTS_PROFILES_AND_TEMPS)
427 : XlaCompiledCpuFunction(StaticData(), alloc_mode) {}
428
429 {{CLASS}}(const {{CLASS}}&) = delete;
430 {{CLASS}}& operator=(const {{CLASS}}&) = delete;
431
432 // Arg methods for managing input buffers. Buffers are in row-major order.
433 // There is a set of methods for each positional argument, with the following
434 // general form:
435 //
436 // void set_argN_data(void* data)
437 // Sets the buffer of type T for positional argument N. May be called in
438 // any AllocMode. Must be called before Run to have an affect. Must be
439 // called in AllocMode::RESULTS_PROFILES_AND_TEMPS_ONLY for each positional
440 // argument, to set the argument buffers.
441 //
442 // T* argN_data()
443 // Returns the buffer of type T for positional argument N.
444 //
445 // T& argN(...dim indices...)
446 // Returns a reference to the value of type T for positional argument N,
447 // with dim indices specifying which value. No bounds checking is performed
448 // on dim indices.
449{{METHODS_ARG}}
450
451 // Result methods for managing output buffers. Buffers are in row-major order.
452 // Must only be called after a successful Run call. There is a set of methods
453 // for each positional result, with the following general form:
454 //
455 // T* resultN_data()
456 // Returns the buffer of type T for positional result N.
457 //
458 // T& resultN(...dim indices...)
459 // Returns a reference to the value of type T for positional result N,
460 // with dim indices specifying which value. No bounds checking is performed
461 // on dim indices.
462 //
463 // Unlike the arg methods, there is no set_resultN_data method. The result
464 // buffers are managed internally, and may change after each call to Run.
465{{METHODS_RESULT}}
466
467 private:
468 // Number of result and temporary buffers for the compiled computation.
469 static constexpr size_t kNumTemps = {{TEMP_NUM}};
470 // The 0-based index of the result tuple in the temporary buffers.
471 static constexpr size_t kResultIndex = {{RESULT_INDEX}};
472
473 // Byte size of each result / temporary buffer. There are kNumTemps entries.
474 static const intptr_t* TempSizes() {
475 static constexpr intptr_t kTempSizes[kNumTemps] = {{{TEMP_SIZES}}};
476 return kTempSizes;
477 }
478
479 // Array of names of each positional argument, terminated by nullptr.
480 static const char** StaticArgNames() {{ARG_NAMES_CODE}}
481
482 // Array of names of each positional result, terminated by nullptr.
483 static const char** StaticResultNames() {{RESULT_NAMES_CODE}}
484
485 // Shape of the args and results.
486 static const xla::ProgramShape* StaticProgramShape() {
487 static const xla::ProgramShape* kShape = {{PROGRAM_SHAPE_SHIM_EXPRESSION}};
488 return kShape;
489 }
490};
491{{NS_END}}
492
493#endif // TFCOMPILE_GENERATED_{{ENTRY}}_H_
494
495// clang-format on
496)";
497 // The replacement strategy is naive, but good enough for our purposes.
498 const std::vector<std::pair<string, string>> rewrites = {
499 {"{{ARG_BYTES_ALIGNED}}", strings::StrCat(arg_bytes_aligned)},
500 {"{{ARG_BYTES_TOTAL}}", strings::StrCat(arg_bytes_total)},
501 {"{{ARG_NAMES_CODE}}", arg_names_code},
502 {"{{ARG_NUM}}", strings::StrCat(arg_sizes.size())},
503 {"{{ARG_SIZES}}", str_util::Join(arg_sizes, ", ")},
504 {"{{CLASS}}", opts.class_name},
505 {"{{ENTRY}}", compile_result.entry_point},
506 {"{{INCLUDE_XLA_DATA_PROTO}}", include_xla_data_proto},
507 {"{{METHODS_ARG}}\n", methods_arg},
508 {"{{METHODS_RESULT}}\n", methods_result},
509 {"{{NS_END}}\n", ns_end},
510 {"{{NS_START}}\n", ns_start},
511 {"{{PROGRAM_SHAPE}}", xla::ShapeUtil::HumanString(ps)},
512 {"{{RESULT_INDEX}}", strings::StrCat(result_index)},
513 {"{{RESULT_NAMES_CODE}}", result_names_code},
514 {"{{TEMP_BYTES_ALIGNED}}", strings::StrCat(temp_bytes_aligned)},
515 {"{{TEMP_BYTES_TOTAL}}", strings::StrCat(temp_bytes_total)},
516 {"{{TEMP_NUM}}", strings::StrCat(temp_sizes.size())},
517 {"{{TEMP_SIZES}}", str_util::Join(temp_sizes, ", ")},
518 {"{{DECLS_FROM_OBJ_FILE}}",
519 str_util::Join(metadata_result.header_variable_decls, "\n")},
520 {"{{PROGRAM_SHAPE_SHIM_EXPRESSION}}",
521 metadata_result.program_shape_access_shim}};
522 str_util::ReplaceAllPairs(header, rewrites);
523 return Status::OK();
524}

526static string CreateUniqueIdentifierForProgramShape(const CodegenOpts& opts) {
527 string result = "__tfcompile";
528 for (const string& n : opts.namespaces) {
529 strings::StrAppend(&result, "_", n);
530 }
531
532 strings::StrAppend(&result, "_", opts.class_name, "_ProgramShape");
533 return result;
534}

536Status GenerateMetadata(const CodegenOpts& opts,
537 const CompileResult& compile_result,
538 MetadataResult* metadata_result) {
539 std::unique_ptr<xla::ProgramShape> program_shape;
540
541 if (opts.gen_program_shape) {
542 program_shape =
543 tensorflow::MakeUnique<xla::ProgramShape>(compile_result.program_shape);
544 // The parameter names are currently meaningless, and redundant with the
545 // rest of our metadata, so clear them out to avoid confusion and save
546 // space.
547 program_shape->clear_parameter_names();
548 }
549
550 // When asked to serialize a null protobuf, CreateEmbeddedProtocolBuffer gives
551 // a shim that evaluates to nullptr, which is what we want.
552
553 TF_ASSIGN_OR_RETURN(
554 EmbeddedProtocolBuffer embedded_program_shape,
555 CreateEmbeddedProtocolBuffer(opts.target_triple,
556 CreateUniqueIdentifierForProgramShape(opts),
557 "xla::ProgramShape", program_shape.get()));
558
559 metadata_result->program_shape_access_shim =
560 std::move(embedded_program_shape.cpp_shim_expression);
561 metadata_result->header_variable_decls.emplace_back(
562 std::move(embedded_program_shape.cpp_variable_decl));
563 metadata_result->object_file_data =
564 std::move(embedded_program_shape.object_file_data);
565 return Status::OK();
566}

568Status ParseCppClass(const string& cpp_class, string* class_name,
569 std::vector<string>* namespaces) {
570 class_name->clear();
571 namespaces->clear();
572 size_t begin = 0;
573 size_t end = 0;
574 while ((end = cpp_class.find("::", begin)) != string::npos) {
575 const string ns = cpp_class.substr(begin, end - begin);
576 TF_RETURN_IF_ERROR(ValidateCppIdent(
577 ns, "in namespace component of cpp_class: " + cpp_class));
578 namespaces->push_back(ns);
579 begin = end + 2; // +2 to skip the two colons
580 }
581 const string name = cpp_class.substr(begin);
582 TF_RETURN_IF_ERROR(
583 ValidateCppIdent(name, "in class name of cpp_class: " + cpp_class));
584 *class_name = name;
585 return Status::OK();
586}

588Status ValidateCppIdent(StringPiece ident, StringPiece msg) {
589 if (ident.empty()) {
590 return errors::InvalidArgument("empty identifier: ", msg);
591 }
592 // Require that the identifier starts with a nondigit, and is composed of
593 // nondigits and digits, as specified in section [2.11 Identifiers] of the
594 // C++11 Standard. Note that nondigit is defined as [_a-zA-Z] and digit is
595 // defined as [0-9].
596 //
597 // Technically the standard also allows for `universal-character-name`, with a
598 // table of allowed unicode ranges, as well as `other implementation-defined
599 // characters`. We disallow those here to give better error messages, at the
600 // expensive of being more restrictive than the standard.
601 if (ident[0] != '_' && !IsAlpha(ident[0])) {
602 return errors::InvalidArgument("illegal leading char: ", msg);
603 }
604 for (size_t pos = 1; pos < ident.size(); ++pos) {
605 if (ident[pos] != '_' && !IsAlphaNum(ident[pos])) {
606 return errors::InvalidArgument("illegal char: ", msg);
607 }
608 }
609 return Status::OK();
610}
