40static std::unique_ptr<llvm::Module> CreateModuleWithEmbeddedProtocolBuffer(
41 llvm::LLVMContext* llvm_context, llvm::TargetMachine* target_machine,
42 const ::tensorflow::protobuf::MessageLite& proto,
43 StringPiece unique_identifier, string* protobuf_array_symbol_name,
44 int64* protobuf_array_size) {
45 string protobuf_array_contents = proto.SerializeAsString();
46 *protobuf_array_symbol_name =
47 strings::StrCat(unique_identifier, "_protobuf_array_contents");
48 *protobuf_array_size = protobuf_array_contents.size();
49
50 std::unique_ptr<llvm::Module> module =
51 MakeUnique<llvm::Module>("embedded_data_module", *llvm_context);
52
53 llvm::Constant* protobuf_array_initializer =
54 llvm::ConstantDataArray::getString(*llvm_context,
55 AsStringRef(protobuf_array_contents),
56 /*AddNull=*/false);
57 new llvm::GlobalVariable(
58 *module, protobuf_array_initializer->getType(),
59 /*isConstant=*/true, llvm::GlobalValue::ExternalLinkage,
60 protobuf_array_initializer, AsStringRef(*protobuf_array_symbol_name));
61
62 return module;
63}

65static string CreateCPPShimExpression(StringPiece qualified_cpp_protobuf_name,
66 StringPiece protobuf_array_symbol_name,
67 int64 protobuf_array_size) {
68 string code =
69 "[]() {\n"
70 " {{PROTOBUF_NAME}}* proto = new {{PROTOBUF_NAME}};\n"
71 " proto->ParseFromArray(&{{ARRAY_SYMBOL}}[0], {{ARRAY_SIZE}});\n"
72 " return proto;\n"
73 " }()";
74
75 str_util::ReplaceAllPairs(
76 &code,
77 {
78 {"{{ARRAY_SYMBOL}}", strings::StrCat(protobuf_array_symbol_name)},
79 {"{{ARRAY_SIZE}}", strings::StrCat(protobuf_array_size)},
80 {"{{PROTOBUF_NAME}}", strings::StrCat(qualified_cpp_protobuf_name)},
81 });
82 return code;
83}

85static StatusOr<string> CodegenModule(llvm::TargetMachine* target_machine,
86 std::unique_ptr<llvm::Module> module) {
87 llvm::SmallVector<char, 0> stream_buffer;
88 llvm::raw_svector_ostream ostream(stream_buffer);
89 llvm::legacy::PassManager codegen_passes;
90
91 if (target_machine->addPassesToEmitFile(
92 codegen_passes, ostream, llvm::TargetMachine::CGFT_ObjectFile)) {
93 return xla::InternalError(
94 "Could not create pass pipeline to generate object file");
95 }
96
97 codegen_passes.run(*module);
98
99 return string(stream_buffer.begin(), stream_buffer.end());
100}

102static StatusOr<std::unique_ptr<llvm::TargetMachine>>
103GetTargetMachineFromTriple(StringPiece target_triple) {
104 std::string error;
105 std::string normalized_triple =
106 llvm::Triple::normalize(AsStringRef(target_triple));
107 const llvm::Target* target =
108 llvm::TargetRegistry::lookupTarget(normalized_triple, error);
109 if (target == nullptr) {
110 return xla::InternalError("TargetRegistry::lookupTarget failed: %s",
111 error.c_str());
112 }
113
114 return WrapUnique(target->createTargetMachine(
115 normalized_triple, /*CPU=*/"",
116 /*Features=*/"", llvm::TargetOptions(), llvm::None));
117}

119StatusOr<EmbeddedProtocolBuffer> CreateEmbeddedProtocolBuffer(
120 StringPiece target_triple, StringPiece symbol_prefix,
121 StringPiece qualified_cpp_protobuf_name,
122 const ::tensorflow::protobuf::MessageLite* proto) {
123 TF_ASSIGN_OR_RETURN(std::unique_ptr<llvm::TargetMachine> target_machine,
124 GetTargetMachineFromTriple(target_triple));
125
126 llvm::LLVMContext llvm_context;
127 string object_file, cpp_shim, cpp_variable_decl;
128
129 if (proto) {
130 string protobuf_array_symbol_name;
131 int64 protobuf_array_size;
132
133 std::unique_ptr<llvm::Module> module_with_serialized_proto =
134 CreateModuleWithEmbeddedProtocolBuffer(
135 &llvm_context, target_machine.get(), *proto, symbol_prefix,
136 &protobuf_array_symbol_name, &protobuf_array_size);
137 TF_ASSIGN_OR_RETURN(object_file,
138 CodegenModule(target_machine.get(),
139 std::move(module_with_serialized_proto)));
140 cpp_shim = CreateCPPShimExpression(qualified_cpp_protobuf_name,
141 protobuf_array_symbol_name,
142 protobuf_array_size);
143
144 cpp_variable_decl = strings::StrCat("extern \"C\" char ",
145 protobuf_array_symbol_name, "[];");
146 } else {
147 TF_ASSIGN_OR_RETURN(
148 object_file,
149 CodegenModule(target_machine.get(),
150 MakeUnique<llvm::Module>("empty_module", llvm_context)));
151 cpp_shim = "nullptr";
152 }
153
154 return {{cpp_shim, cpp_variable_decl, object_file}};
155}
