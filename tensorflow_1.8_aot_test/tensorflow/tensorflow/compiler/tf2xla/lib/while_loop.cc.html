23xla::StatusOr<std::vector<xla::ComputationDataHandle>> XlaWhileLoop(
24 const LoopConditionFunction& condition_function,
25 const LoopBodyFunction& body_function,
26 gtl::ArraySlice<xla::ComputationDataHandle> initial_values,
27 StringPiece name, xla::ComputationBuilder* builder) {
28 int arity = initial_values.size();
29 std::vector<xla::Shape> var_shapes;
30 var_shapes.reserve(arity);
31 for (const xla::ComputationDataHandle& input : initial_values) {
32 TF_ASSIGN_OR_RETURN(auto shape, builder->GetShape(input));
33 var_shapes.push_back(std::move(*shape));
34 }
35 xla::Shape tuple_shape = xla::ShapeUtil::MakeTupleShape(var_shapes);
36
37 // Unpacks a tuple into its component parts.
38 auto unpack_tuple = [](xla::ComputationDataHandle tuple, int arity,
39 xla::ComputationBuilder* builder) {
40 std::vector<xla::ComputationDataHandle> elements(arity);
41 for (int i = 0; i < arity; ++i) {
42 elements[i] = builder->GetTupleElement(tuple, i);
43 }
44 return elements;
45 };
46
47 // Build the condition.
48 std::unique_ptr<xla::ComputationBuilder> cond_builder =
49 builder->CreateSubBuilder(strings::StrCat(name, "_condition"));
50 {
51 auto parameter = cond_builder->Parameter(0, tuple_shape, "parameter");
52
53 TF_ASSIGN_OR_RETURN(
54 auto result,
55 condition_function(unpack_tuple(parameter, arity, cond_builder.get()),
56 cond_builder.get()));
57 }
58 TF_ASSIGN_OR_RETURN(auto cond, cond_builder->Build());
59
60 // Build the body.
61 std::unique_ptr<xla::ComputationBuilder> body_builder =
62 builder->CreateSubBuilder(strings::StrCat(name, "_body"));
63 {
64 auto parameter = body_builder->Parameter(0, tuple_shape, "parameter");
65
66 TF_ASSIGN_OR_RETURN(
67 auto result,
68 body_function(unpack_tuple(parameter, arity, body_builder.get()),
69 body_builder.get()));
70
71 TF_RET_CHECK(result.size() == initial_values.size());
72 body_builder->Tuple(result);
73 }
74 TF_ASSIGN_OR_RETURN(auto body, body_builder->Build());
75
76 auto outputs = builder->While(cond, body, builder->Tuple(initial_values));
77
78 return unpack_tuple(outputs, arity, builder);
79}

81xla::StatusOr<std::vector<xla::ComputationDataHandle>> XlaForEachIndex(
82 int64 num_iterations, xla::PrimitiveType num_iterations_type,
83 const ForEachIndexBodyFunction& body_function,
84 gtl::ArraySlice<xla::ComputationDataHandle> initial_values,
85 StringPiece name, xla::ComputationBuilder* builder) {
86 auto while_cond_fn = [&](gtl::ArraySlice<xla::ComputationDataHandle> values,
87 xla::ComputationBuilder* cond_builder)
88 -> xla::StatusOr<xla::ComputationDataHandle> {
89 return cond_builder->Lt(
90 values[0],
91 IntegerLiteral(cond_builder, num_iterations_type, num_iterations));
92 };
93 auto while_body_fn = [&](gtl::ArraySlice<xla::ComputationDataHandle> values,
94 xla::ComputationBuilder* body_builder)
95 -> xla::StatusOr<std::vector<xla::ComputationDataHandle>> {
96 xla::ComputationDataHandle iteration = values[0];
97
98 std::vector<xla::ComputationDataHandle> updated_values;
99 updated_values.reserve(values.size());
100 updated_values.push_back(body_builder->Add(
101 iteration,
102 body_builder->ConstantLiteral(xla::Literal::One(num_iterations_type))));
103
104 values.remove_prefix(1);
105 TF_ASSIGN_OR_RETURN(std::vector<xla::ComputationDataHandle> body_outputs,
106 body_function(iteration, values, body_builder));
107 updated_values.insert(updated_values.end(), body_outputs.begin(),
108 body_outputs.end());
109 return updated_values;
110 };
111
112 std::vector<xla::ComputationDataHandle> values;
113 values.reserve(initial_values.size() + 1);
114 values.push_back(
115 builder->ConstantLiteral(xla::Literal::Zero(num_iterations_type)));
116 values.insert(values.end(), initial_values.begin(), initial_values.end());
117
118 TF_ASSIGN_OR_RETURN(values, XlaWhileLoop(while_cond_fn, while_body_fn, values,
119 name, builder));
120 values.erase(values.begin(), values.begin() + 1);
121 return values;
122}
