42const char* const kArgOp = "_Arg";
43const char* const kRetValOp = "_Retval";

45// Information about a loop argument.
46struct Arg {
47 // Every loop argument has an Enter node.
48 Node* enter;
49
50 // Is the loop argument a loop-invariant value? Taken from the `is_constant`
51 // attribute on the Enter node.
52 bool is_loop_invariant;
53
54 // If 'is_loop_invariant' is true, the following are all nullptr. Non-constant
55 // arguments must have all of the following nodes:
56 Node* merge = nullptr;
57 Node* switch_node = nullptr;
58 Node* next_iteration = nullptr;
59 Node* exit = nullptr;
60};
61
62// Information about a loop frame.
63struct Frame {
64 string name;
65
66 // Pointer to the parent frame. The root frame has a pointer to itself.
67 Frame* parent = nullptr;
68 int num_children = 0;
69
70 // Arguments to this loop.
71 std::vector<Arg> args;
72
73 // The loop condition of the loop. There should be exactly one loop condition
74 // in every loop.
75 Node* loop_cond = nullptr;
76
77 // Set of nodes that belong to the loop frame.
78 std::unordered_set<Node*> nodes;
79};

81// Comparison function used for sorting nodes consistently.
82// a) resource variables are last, and
83// b) sort lexicographically by name (for deterministic output).
84struct NodeCmp {
85 bool operator()(const Node* lhs, const Node* rhs) const {
86 bool lhs_is_resource =
87 lhs->num_inputs() > 0 ? (lhs->input_type(0) == DT_RESOURCE) : false;
88 bool rhs_is_resource =
89 rhs->num_inputs() > 0 ? (rhs->input_type(0) == DT_RESOURCE) : false;
90 return std::tie(lhs_is_resource, lhs->name()) <
91 std::tie(rhs_is_resource, rhs->name());
92 }
93};

95// Returns a textual representation of the names of the nodes in the input.
96template <typename T>
97string NodesToString(const T& nodes) {
98 return strings::StrCat("{",
99 str_util::Join(nodes, ",",
100 [](string* output, const Node* node) {
101 strings::StrAppend(output,
102 node->name());
103 }),
104 "}");
105}

107// Copies a subgraph from `graph` to `output` by performing a reverse DFS
108// starting at nodes in vector `stack`.
109// `node_map` is a vector indexed by source node ID to dest nodes.
110// Does not traverse into nodes in `node_map`, so by adding nodes to `node_map`
111// before the traversal clients can cut the graph. If a frame is provided (frame
112// != nullptr), then this functions will return an error if the
113// traversal leaves 'frame'; the client must add enough nodes to `node_map` to
114// cut the graph and prevent the traversal from escaping.
115//
116// `squash_src_outputs` contains a bool for each source node ID. If true, then
117// the source output on that node will be replaced by zero when copied. This is
118// used when replacing a Switch node with an _Arg node. The output we are
119// taking from the Switch node was not necessarily the first output, but _Arg
120// nodes only have one output. By adding the Switch node to `squash_src_outputs`
121// we rewrite the src_output of the corresponding edge to be 0.
122Status CopySubgraph(const Graph& graph, const Frame* frame,
123 std::vector<Node*> stack,
124 const std::vector<bool>& squash_src_outputs,
125 std::vector<Node*>* node_map, Graph* output) {
126 VLOG(3) << "Stack: " << NodesToString(stack);
127 std::vector<bool> visited(graph.num_node_ids(), false);
128 while (!stack.empty()) {
129 Node* n = stack.back();
130 stack.pop_back();
131
132 VLOG(5) << "Copying node " << n->name();
133
134 if (visited[n->id()]) continue;
135 visited[n->id()] = true;
136
137 for (const Edge* e : n->in_edges()) {
138 Node* src = e->src();
139 if (frame != nullptr && frame->nodes.find(src) == frame->nodes.end()) {
140 // We traversed out of the loop frame, without encountering a cut node.
141 return errors::Internal("Graph traversal of loop frame ", frame->name,
142 " escaped frame at ", src->name(),
143 " without encountering an argument node.");
144 }
145 if ((*node_map)[src->id()] == nullptr) {
146 (*node_map)[src->id()] = output->CopyNode(src);
147 stack.push_back(src);
148 }
149 Node* src_copy = (*node_map)[e->src()->id()];
150 int src_output = squash_src_outputs[e->src()->id()] && !e->IsControlEdge()
151 ? 0
152 : e->src_output();
153 Node* dst_copy = (*node_map)[e->dst()->id()];
154 output->AddEdge(src_copy, src_output, dst_copy, e->dst_input());
155 }
156 }
157 return Status::OK();
158}

160StatusOr<Node*> AddNode(const NodeDef& node_def, Graph* graph) {
161 Status status;
162 Node* inserted_node = graph->AddNode(node_def, &status);
163 if (!status.ok()) {
164 return status;
165 }
166 return inserted_node;
167}

169StatusOr<Node*> BuildArgNode(Graph* graph, DataType type, int index) {
170 NodeDef arg_def;
171 NodeDefBuilder builder(strings::StrCat(kArgOp, index), kArgOp);
172 builder.Attr("T", type);
173 builder.Attr("index", index);
174 TF_RETURN_IF_ERROR(builder.Finalize(&arg_def));
175 return AddNode(arg_def, graph);
176}
177
178StatusOr<Node*> BuildRetvalNode(Graph* graph, DataType type, int index) {
179 NodeDef ret_def;
180 ret_def.set_op(kRetValOp);
181 ret_def.set_name(strings::StrCat(kRetValOp, index));
182 AddNodeAttr("T", type, &ret_def);
183 AddNodeAttr("index", index, &ret_def);
184 return AddNode(ret_def, graph);
185}

187// Builds a graph for the loop condition.
188Status BuildLoopCondition(const Graph& graph, Frame* frame,
189 std::unique_ptr<Graph>* cond_output) {
190 VLOG(2) << "Building loop condition for " << frame->name;
191 *cond_output = xla::MakeUnique<Graph>(graph.op_registry());
192 Graph* output = cond_output->get();
193
194 // Map from nodes in the original graph to the condition graph.
195 std::vector<Node*> node_map(graph.num_node_ids(), nullptr);
196 std::vector<bool> squash_src_outputs(graph.num_node_ids(), false);
197
198 // Build one _Arg node for each Enter node.
199 for (int i = 0; i < frame->args.size(); ++i) {
200 const Arg& arg = frame->args[i];
201
202 TF_ASSIGN_OR_RETURN(Node * arg_node,
203 BuildArgNode(output, arg.enter->input_type(0), i));
204 if (arg.is_loop_invariant) {
205 node_map[arg.enter->id()] = arg_node;
206 } else {
207 node_map[arg.merge->id()] = arg_node;
208 }
209 }
210
211 // Build a Retval node for the loop condition. The LoopCond nodes are always
212 // boolean because of the type constraints on the LoopCond op.
213 TF_ASSIGN_OR_RETURN(node_map[frame->loop_cond->id()],
214 BuildRetvalNode(output, DT_BOOL, 0));
215
216 // Performs a reverse DFS, copying nodes and edges to the output graph.
217 // The _Arg and _Retval nodes were added unconditionally above, so we are
218 // guaranteed to get the correct function signature.
219 return CopySubgraph(graph, frame, {frame->loop_cond}, squash_src_outputs,
220 &node_map, output);
221}
222
223// Builds a graph for the loop body.
224Status BuildLoopBody(const Graph& graph, Frame* frame,
225 DataTypeVector* arg_types,
226 std::unique_ptr<Graph>* body_output) {
227 VLOG(2) << "Building loop body for " << frame->name;
228 *body_output = xla::MakeUnique<Graph>(graph.op_registry());
229 Graph* output = body_output->get();
230
231 // Map from nodes in the original graph to the condition graph.
232 std::vector<Node*> node_map(graph.num_node_ids(), nullptr);
233 std::vector<bool> squash_src_outputs(graph.num_node_ids(), false);
234
235 // Build one _Arg node for each Enter node.
236 std::vector<Node*> next_iterations;
237 next_iterations.reserve(frame->args.size());
238 arg_types->reserve(frame->args.size());
239 for (int i = 0; i < frame->args.size(); ++i) {
240 const Arg& arg = frame->args[i];
241
242 DataType dtype = arg.enter->input_type(0);
243 arg_types->push_back(dtype);
244
245 TF_ASSIGN_OR_RETURN(Node * arg_node, BuildArgNode(output, dtype, i));
246
247 if (dtype == DT_RESOURCE) {
248 // The convention of the XLA bridge is that resource variable arguments
249 // are only inputs to the loop body and have no corresponding output.
250 // TODO(b/37741920): change the convention so that DT_RESOURCE variables
251 // are both inputs and outputs, and then remove this case.
252 TF_RET_CHECK(arg.is_loop_invariant);
253 node_map[arg.enter->id()] = arg_node;
254 } else {
255 TF_ASSIGN_OR_RETURN(Node * retval_node,
256 BuildRetvalNode(output, dtype, i));
257
258 if (arg.is_loop_invariant) {
259 // Argument is loop-invariant. Forward it from the Arg to the Retval.
260 node_map[arg.enter->id()] = arg_node;
261 output->AddEdge(arg_node, 0, retval_node, 0);
262 } else {
263 // Argument is loop-varying.
264 node_map[arg.switch_node->id()] = arg_node;
265 // The Switch node has two outputs, but _Arg only has one. This tells
266 // the CopySubgraph function to rewrite the output number of edges from
267 // the _Arg node to be 0 rather than copying the output number from the
268 // Switch node.
269 squash_src_outputs[arg.switch_node->id()] = true;
270 node_map[arg.next_iteration->id()] = retval_node;
271 next_iterations.push_back(arg.next_iteration);
272 }
273 }
274 }
275
276 // Performs a reverse DFS, copying nodes and edges to the output graph.
277 // The _Arg and _Retval nodes were added unconditionally above, so we are
278 // guaranteed to get the correct function signature.
279 TF_RETURN_IF_ERROR(CopySubgraph(graph, frame, std::move(next_iterations),
280 squash_src_outputs, &node_map, output));
281
282 return Status::OK();
283}

285Status FunctionalizeLoop(Graph* graph, Frame* frame,
286 FunctionLibraryDefinition* library) {
287 VLOG(2) << "Frame " << frame->name << " before: "
288 << dump_graph::DumpGraphToFile("functionalize_before", *graph,
289 library);
290
291 // Split loop-varying Enter nodes with multiple successors. If the same
292 // Tensor is fed as input to multiple loop arguments, we may end up with a
293 // shared Enter node. We clone Enter nodes with multiple successors to
294 // maintain the invariant of a unique Enter node per argument of the final
295 // loop.
296 std::vector<Arg> args;
297 for (const Arg& arg : frame->args) {
298 if (arg.is_loop_invariant) {
299 args.push_back(arg);
300 } else {
301 std::vector<const Edge*> edges(arg.enter->out_edges().begin(),
302 arg.enter->out_edges().end());
303 for (int i = 0; i < edges.size(); ++i) {
304 if (edges[i]->IsControlEdge() && edges[i]->dst()->IsSink()) {
305 continue;
306 }
307 TF_RET_CHECK(!edges[i]->IsControlEdge()) << edges[i]->src()->name();
308 Arg new_arg;
309 new_arg.is_loop_invariant = false;
310 if (i == 0) {
311 new_arg.enter = arg.enter;
312 } else {
313 new_arg.enter = graph->CopyNode(arg.enter);
314 frame->nodes.insert(new_arg.enter);
315 for (Edge const* e : arg.enter->in_edges()) {
316 graph->AddEdge(e->src(), e->src_output(), new_arg.enter,
317 e->IsControlEdge() ? Graph::kControlSlot : 0);
318 }
319 Node* dst = edges[i]->dst();
320 int dst_input = edges[i]->dst_input();
321 graph->RemoveEdge(edges[i]);
322 graph->AddEdge(new_arg.enter, 0, dst, dst_input);
323 }
324 args.push_back(new_arg);
325 }
326 }
327 }
328 frame->args = std::move(args);
329
330 std::sort(
331 frame->args.begin(), frame->args.end(),
332 [](const Arg& a, const Arg& b) { return NodeCmp()(a.enter, b.enter); });
333
334 if (frame->loop_cond == nullptr) {
335 return errors::InvalidArgument("Loop ", frame->name,
336 " has no LoopCond node");
337 }
338
339 // Find the set of Switch nodes that are successors of the LoopCond.
340 std::unordered_set<Node*> switches;
341 for (const Edge* edge : frame->loop_cond->out_edges()) {
342 if (!edge->IsControlEdge() && IsSwitch(edge->dst()) &&
343 edge->dst_input() == 1) {
344 switches.insert(edge->dst());
345 }
346 }
347
348 // For each non-constant argument, looks for the following pattern of nodes:
349 // Enter ----> Merge --------> Switch --> Exit
350 // ^ ^
351 // | |
352 // NextIteration LoopCond
353 // ^ ^
354 // | |
355 // ... ...
356 for (Arg& arg : frame->args) {
357 if (!arg.is_loop_invariant) {
358 // Follow the edge from the Enter to Merge.
359 const Edge* enter_merge = nullptr;
360 for (const Edge* e : arg.enter->out_edges()) {
361 // Ignore control-edges to the sink node. These are allowed by the
362 // graph invariants, although probably they should have been stripped
363 // off earlier.
364 if (e->IsControlEdge() && e->dst()->IsSink()) {
365 continue;
366 }
367 if (enter_merge != nullptr) {
368 return errors::Internal(
369 "Enter node for loop-varying argument ", arg.enter->name(),
370 " has multiple successors: ", enter_merge->dst()->name(), " and ",
371 e->dst()->name());
372 }
373 enter_merge = e;
374 }
375 if (enter_merge == nullptr) {
376 return errors::Internal("Enter node for loop-varying argument ",
377 arg.enter->name(), " has zero successors");
378 }
379 arg.merge = enter_merge->dst();
380 if (!IsMerge(arg.merge)) {
381 return errors::InvalidArgument(
382 "Successor of Enter node for loop-varying argument ",
383 arg.merge->name(),
384 " is not a Merge node; got: ", arg.merge->type_string());
385 }
386
387 // Find the NextIteration from the merge. There should be two inputs to
388 // the Merge and the NextIteration should be the other input.
389 if (arg.merge->input_types().size() != 2) {
390 return errors::InvalidArgument(
391 "Unexpected number of inputs to Merge node for loop-varying "
392 "argument ",
393 arg.merge->name(), "; expected 2, got ",
394 arg.merge->input_types().size());
395 }
396 TF_RETURN_IF_ERROR(arg.merge->input_node(1 - enter_merge->dst_input(),
397 &arg.next_iteration));
398 if (!IsNextIteration(arg.next_iteration)) {
399 return errors::InvalidArgument(
400 "Expected NextIteration node as input to Merge node; got node ",
401 arg.next_iteration->name(), " with kind ",
402 arg.next_iteration->type_string());
403 }
404
405 // Find the Switch successor of the Merge. There should be exactly one
406 // Switch node that is a successor of both the Merge and the LoopCond.
407 for (const Edge* edge : arg.merge->out_edges()) {
408 if (edge->dst_input() == 0 && IsSwitch(edge->dst()) &&
409 switches.find(edge->dst()) != switches.end()) {
410 if (arg.switch_node != nullptr) {
411 return errors::InvalidArgument("Duplicate Switch successors to ",
412 arg.merge->name());
413 }
414 arg.switch_node = edge->dst();
415 }
416 }
417 if (arg.switch_node == nullptr) {
418 return errors::InvalidArgument("Missing Switch successor to ",
419 arg.merge->name());
420 }
421
422 // Update the device on the Identity outputs of the switch to match their
423 // target. These Identity outputs do not
424
425 // Loop over the switch node's output to:
426 // - Find the Exit successor.
427 // - Set the sharding on all Identity outputs of the switch. These
428 // identity nodes are values used by the loop body or condition.
429 // The Identity node may have the wrong device so copy the device from
430 // one of its outputs instead.
431 std::deque<const Edge*> possible_exit;
432 for (const Edge* edge : arg.switch_node->out_edges()) {
433 if (edge->src_output() == 0) {
434 possible_exit.push_back(edge);
435 }
436 if (IsIdentity(edge->dst())) {
437 TF_RETURN_IF_ERROR(
438 SetNodeShardingFromNeighbors(edge->dst(), /*out_edges=*/true));
439 }
440 }
441 // TODO(b/67425339): Allow general graph between switch and exit.
442 while (!possible_exit.empty()) {
443 const Edge* edge = possible_exit.front();
444 possible_exit.pop_front();
445 if (IsExit(edge->dst())) {
446 if (arg.exit != nullptr) {
447 return errors::InvalidArgument("Duplicate Exit successors to ",
448 arg.switch_node->name());
449 }
450 arg.exit = edge->dst();
451 } else {
452 if (!IsIdentity(edge->dst())) {
453 return errors::Unimplemented("General graph between switch (",
454 arg.switch_node->name(),
455 ") and exit node of frame ",
456 frame->name, " not supported yet.");
457 }
458 for (const Edge* out : edge->dst()->out_edges()) {
459 possible_exit.push_back(out);
460 }
461 }
462 }
463 }
464 }
465
466 // Builds the condition and body functions.
467 std::unique_ptr<Graph> cond_graph;
468 TF_RETURN_IF_ERROR(BuildLoopCondition(*graph, frame, &cond_graph));
469 DataTypeVector arg_types;
470 std::unique_ptr<Graph> body_graph;
471 TF_RETURN_IF_ERROR(BuildLoopBody(*graph, frame, &arg_types, &body_graph));
472
473 VLOG(2) << "Frame " << frame->name << " condition: "
474 << dump_graph::DumpGraphToFile("loop_condition", *cond_graph, library)
475 << " body: " << dump_graph::DumpGraphToFile("loop_body", *body_graph);
476
477 static std::atomic<int64> sequence_num(0LL);
478 int64 id = ++sequence_num;
479 NameAttrList cond_name;
480 cond_name.set_name(strings::StrCat("_functionalize_cond_", id));
481 NameAttrList body_name;
482 body_name.set_name(strings::StrCat("_functionalize_body_", id));
483 FunctionDef cond_fdef;
484 TF_RETURN_IF_ERROR(
485 GraphToFunctionDef(*cond_graph, cond_name.name(), &cond_fdef));
486 FunctionDef body_fdef;
487 TF_RETURN_IF_ERROR(
488 GraphToFunctionDef(*body_graph, body_name.name(), &body_fdef));
489
490 TF_RETURN_IF_ERROR(library->AddFunctionDef(cond_fdef));
491 TF_RETURN_IF_ERROR(library->AddFunctionDef(body_fdef));
492
493 // Builds a While operator.
494 NodeDef while_def;
495 NodeDefBuilder builder(frame->loop_cond->name(), "XlaWhile");
496 builder.Attr("T", arg_types);
497 builder.Attr("cond", cond_name);
498 builder.Attr("body", body_name);
499 std::vector<NodeDefBuilder::NodeOut> inputs;
500 for (int i = 0; i < frame->args.size(); ++i) {
501 const Arg& arg = frame->args[i];
502 const Edge* in_edge;
503 TF_RETURN_IF_ERROR(arg.enter->input_edge(0, &in_edge));
504 if (in_edge->IsControlEdge()) {
505 builder.ControlInput(in_edge->src()->name());
506 } else {
507 inputs.push_back(NodeDefBuilder::NodeOut(
508 in_edge->src()->name(), in_edge->src_output(), arg_types[i]));
509 }
510 }
511 builder.Input(inputs);
512 TF_RETURN_IF_ERROR(builder.Finalize(&while_def));
513 TF_ASSIGN_OR_RETURN(Node * while_node, AddNode(while_def, graph));
514
515 // Copies edges to the Enter nodes and from the Exit nodes onto the While.
516 for (int i = 0; i < frame->args.size(); ++i) {
517 const Arg& arg = frame->args[i];
518 const Edge* in_edge;
519 TF_RETURN_IF_ERROR(arg.enter->input_edge(0, &in_edge));
520 if (in_edge->IsControlEdge()) {
521 graph->AddControlEdge(in_edge->src(), while_node);
522 } else {
523 graph->AddEdge(in_edge->src(), in_edge->src_output(), while_node, i);
524 }
525
526 if (!arg.is_loop_invariant) {
527 // Add output edges if the output of the loop is consumed.
528 if (arg.exit != nullptr) {
529 std::vector<const Edge*> edges(arg.exit->out_edges().begin(),
530 arg.exit->out_edges().end());
531 for (const Edge* edge : edges) {
532 Node* dst = edge->dst();
533 int dst_input = edge->dst_input();
534 graph->RemoveEdge(edge);
535
536 if (dst_input == Graph::kControlSlot) {
537 graph->AddControlEdge(while_node, dst);
538 } else {
539 graph->AddEdge(while_node, i, dst, dst_input);
540 }
541 }
542 }
543 }
544 }
545
546 // Remove the old nodes from the graph, and add the while node to the parent
547 // frame.
548 for (Node* node : frame->nodes) {
549 graph->RemoveNode(node);
550 }
551 frame->nodes.clear();
552 frame->parent->nodes.insert(while_node);
553
554 VLOG(2) << "Frame " << frame->name << " after: "
555 << dump_graph::DumpGraphToFile("functionalize_after", *graph,
556 library);
557
558 return Status::OK();
559}

561class FunctionalizeCond {
562 public:
563 // All nodes are assumed to be either in no branch, then branch, else branch,
564 // or both branches (such as merge nodes).
565 enum Branch {
566 kElseBranch = 0,
567 kThenBranch = 1,
568 kBoth = 2,
569 kNeither = 3,
570 kNumBranchTypes = 4
571 };
572
573 // Returns a textual representation of the Branch b.
574 static string Branch_Name(FunctionalizeCond::Branch b);
575
576 // Functionalize all the switch-merge nodes of a loop-free graph into XlaIf
577 // nodes. That is, attempt to transform every remaining switch and merge nodes
578 // in the graph into XlaIf nodes.
579 // Precondition: All while loops have been removed from graph.
580 static Status Functionalize(Graph* graph, FunctionLibraryDefinition* library);
581
582 private:
583 // CondArgNode represents a input to the conditional and its corresponding
584 // switch nodes.
585 struct CondArgNode {
586 explicit CondArgNode(Node* src, int src_output)
587 : src(src), src_output(src_output) {}
588 string ToString() const {
589 return strings::StrCat("src=", src->name(), ":", src_output,
590 " switches=", NodesToString(switches));
591 }
592
593 Node* src;
594 int src_output;
595 std::vector<Node*> switches;
596 };
597 using CondArgNodes = std::vector<CondArgNode>;
598
599 struct ForwardFlowNode {
600 explicit ForwardFlowNode(Branch branch = Branch::kNeither)
601 : branch(branch), count(0) {}
602 string ToString() const {
603 return strings::StrCat("branch=", Branch_Name(branch), " count=", count);
604 }
605 Branch branch;
606 int count;
607 };
608
609 // Group of switch nodes that will be part of the same XlaIf.
610 struct SwitchCluster {
611 explicit SwitchCluster(const Edge* predicate_edge)
612 : predicate_edge(predicate_edge) {}
613 string ToString() const {
614 return strings::StrCat(name, " predicate=", predicate_edge->src()->name(),
615 " switches=", NodesToString(switches));
616 }
617
618 string name;
619 const Edge* predicate_edge;
620 std::vector<Node*> switches;
621 };
622
623 FunctionalizeCond(Graph* graph, FunctionLibraryDefinition* library,
624 bool dump_graphs)
625 : library_(library), graph_(graph), dump_graphs_(dump_graphs) {}
626
627 // Perform the actual cond functionalization. Iterate over groups of switch
628 // nodes (linked by common predicate), from innermost to outermost, and
629 // extract into XlaIf nodes.
630 Status FunctionalizeInternal();
631
632 // Determines the branch_map (mapping from node to branch of cond) and
633 // frontier (the nodes where the cond ends).
634 StatusOr<std::pair<std::unordered_map<Node*, ForwardFlowNode>,
635 std::unordered_set<Node*>>>
636 DetermineBranchMapAndFrontier(const SwitchCluster& switch_cluster);
637
638 // Returns XlaIf node created from subgraph of merge and switch nodes. This
639 // encapsulates the process of extracting the bodies needed for the then and
640 // else branch, creates a XlaIf node, removing the nodes of the branches from
641 // the graph and replacing the merge node with a XlaIf.
642 StatusOr<Node*> ConvertToXlaIf(const CondArgNodes& cond_arg_nodes,
643 const SwitchCluster& switch_cluster,
644 const std::vector<Node*>& switches);
645
646 // Builds a XlaIfOp to replace the Switch-Graph-Merge cluster with.
647 StatusOr<Node*> BuildAndAddXlaIfOp(const CondArgNodes& cond_arg_nodes,
648 const SwitchCluster& switch_cluster,

693bool IsDeadSwitch(const Node* node) {
694 for (const Edge* e : node->out_edges()) {
695 const Node* dst = e->dst();
696 if (!dst->IsIdentity()) {
697 return false;
698 }
699 for (const Edge* ee : dst->out_edges()) {
700 if (!ee->IsControlEdge() || !ee->dst()->IsSink()) {
701 return false;
702 }
703 }
704 }
705 return true;
706}

716 branch_map,
717 const std::unordered_set<Node*>& frontier) {
718 std::unordered_set<const Node*> pending[kNumBranchTypes];
719 for (Node* n : frontier) {
720 pending[branch_map.at(n).branch].insert(n);
721 }
722 TF_RET_CHECK(pending[kNeither].empty()) << NodesToString(pending[kNeither]);
723 for (const Node* n : pending[kBoth]) {
724 TF_RET_CHECK(IsMerge(n)) << n->DebugString();
725 // Merge nodes may be in then or else branch too
726 }
727 int index = (pending[kThenBranch].size() <= pending[kElseBranch].size())
728 ? kThenBranch
729 : kElseBranch;
730 int other = 1 - index;
731 for (const Node* n : pending[index]) {
732 if (pending[other].find(n) != pending[other].end()) {
733 return errors::Internal(
734 "Node (", n->DebugString().c_str(),
735 ") in both Else and Then branch should be in Both.");
736 }
737 }
738 // An empty frontier indicates a dead switch. Above we attempt to remove dead
739 // switch nodes, but not all are removed so don't treat it as an error yet.
740 // TODO(jpienaar): Find out why dead switch nodes remain.
741 // if (pending[kBoth].empty() && pending[kThenBranch].empty() &&
742 // pending[kElseBranch].empty()) {
743 // return errors::Internal("Unexpected empty frontier for switch nodes");
744 // }
745 return Status::OK();

748Status FunctionalizeCond::Join(const ForwardFlowNode& src_state,
749 const Node* dst, ForwardFlowNode* dst_state) {
750 TF_RET_CHECK(dst_state->branch != Branch::kBoth &&
751 dst_state->branch != Branch::kNumBranchTypes)
752 << "Unexpected/Invalid branch type: Merging "
753 << Branch_Name(src_state.branch) << " with "
754 << Branch_Name(dst_state->branch);
755 if (dst_state->branch == Branch::kNeither) {
756 dst_state->branch = src_state.branch;
757 } else if (src_state.branch != dst_state->branch &&
758 src_state.branch != Branch::kNeither) {
759 if (IsMerge(dst)) {
760 dst_state->branch = Branch::kBoth;
761 } else {
762 return errors::Internal("Illegal merge:\n", src_state.ToString(),
763 " with ", dst_state->ToString(), " for\n",
764 dst->DebugString());
765 }
766 }
767 ++dst_state->count;
768 return Status::OK();
769}

771StatusOr<std::vector<FunctionalizeCond::SwitchCluster>>
772FunctionalizeCond::DeterminePredicateSwitchOrder() {
773 struct Cluster {
774 bool operator==(const Cluster& other) const {
775 return representative == other.representative;
776 }
777 int representative = -1;
778 };
779
780 // Perform a DFS over the graph and
781 // * Determine the reverse topological order of the nodes (there should be no
782 // cycles at this point so the post-order numbering corresponds to the
783 // reverse topological sorting);
784 // * Identify dead switches;
785 // * Initialize the cluster's representative;
786 std::vector<UnionFind<Cluster>> clusters(graph_->num_node_ids());
787 std::vector<Node*> dead_switches;
788 std::vector<Node*> switch_order;
789 std::vector<Node*> rev_topo_sorted_nodes;
790 DFS(*graph_, nullptr, [&](Node* n) {
791 clusters[n->id()].Get().representative = n->id();
792 if (IsSwitch(n)) {
793 if (IsDeadSwitch(n)) {
794 dead_switches.push_back(n);
795 } else {
796 rev_topo_sorted_nodes.push_back(n);
797 switch_order.push_back(n);
798 }
799 } else if (n->IsOp()) {
800 // Exclude src and sink nodes from further consideration.
801 rev_topo_sorted_nodes.push_back(n);
802 }
803 });
804
805 std::vector<SwitchCluster> switch_clusters;
806 // Return early if there are no switches in the graph.
807 if (switch_order.empty()) {
808 return switch_clusters;
809 }
810
811 // Remove all dead switch nodes.
812 for (Node* n : dead_switches) {
813 VLOG(2) << "Removing dead switch: " << n->DebugString();
814 graph_->RemoveNode(n);
815 }
816
817 // Identify switch nodes that are part of the same control flow context by
818 // considering the operands of operations: an operation is part of the same
819 // control context as its operands unless the operation is a switch. Control
820 // dependencies are considered part of the same control flow context if the
821 // switch depth is the same (see comment below).
822
823 // entry_cluster records the input cluster to a switch node. This is used when
824 // merging with a merge node where the dst's cluster is merged with the entry
825 // cluster of the merge node's cluster (which corresponds to a switch cluster
826 // and so has an entry cluster).
827 std::unordered_map<int, UnionFind<Cluster>*> entry_cluster;
828
829 // Returns the output cluster of a node. Where the output cluster is cluster
830 // where the output of the node is used. For non-merge nodes this is simply
831 // the cluster they are part of, while for merge nodes it is the entry cluster
832 // of the cluster they are part of (this will correspond to the entry node of
833 // a switch node that dominates the merge).
834 auto find_output_cluster = [&](Node* n) {
835 UnionFind<Cluster>* cluster = &clusters[n->id()];
836 if (!IsMerge(n)) return cluster;
837 auto it = entry_cluster.find(clusters[n->id()].Get().representative);
838 // If the cluster is not found in the entry_cluster map then an
839 // instruction not dominated by a switch node has been merged into the
840 // cluster of the merge. This indicates a failure of the clustering.
841 CHECK(it != entry_cluster.end())
842 << "Unable to find entry for n=" << n->id() << " ("
843 << cluster->Get().representative << ")";
844 return it->second;
845 };
846
847 // TODO(jpienaar): This could be combined with DetermineBranchMapAndFrontier.
848 std::vector<int> switch_depth(graph_->num_node_ids());
849 for (auto it = rev_topo_sorted_nodes.rbegin();
850 it != rev_topo_sorted_nodes.rend(); ++it) {
851 Node* n = *it;
852
853 // Compute switch depth.
854 int new_switch_depth = 0;
855 for (const Edge* e : n->in_edges()) {
856 Node* src = e->src();
857 new_switch_depth = std::max(
858 new_switch_depth, switch_depth[src->id()] - (IsMerge(src) ? 1 : 0));
859 }
860 switch_depth[n->id()] = new_switch_depth + (IsSwitch(n) ? 1 : 0);
861
862 // Only merge the input operands of a switch. The switch's clustering itself
863 // is determined by the interaction of the switch's outputs.
864 if (IsSwitch(n)) {
865 Node* input;
866 TF_CHECK_OK(n->input_node(0, &input));
867 entry_cluster[n->id()] = find_output_cluster(input);
868 UnionFind<Cluster>* cluster = entry_cluster[n->id()];
869 int cluster_depth = switch_depth[cluster->Get().representative];
870 // Merge the inputs of the switch node with one another. This results in
871 // predicates and control input residing in the same cluster.
872 for (const Edge* e : n->in_edges()) {
873 Node* src = e->src();
874 UnionFind<Cluster>* src_cluster = find_output_cluster(src);
875 int src_cluster_depth = switch_depth[src_cluster->Get().representative];
876 if (cluster_depth != src_cluster_depth) {
877 return errors::InvalidArgument(
878 "Unable to functionalize control flow in graph: Switch ('",
879 n->name(), "') has operands ('", input->name(), "' and '",
880 src->name(), "') that have different switch depths (",
881 cluster_depth, " != ", src_cluster_depth, ")");
882 }
883 cluster->Merge(src_cluster);
884 }
885 continue;
886 }
887
888 for (const Edge* e : n->in_edges()) {
889 Node* src = e->src();
890 if (!src->IsOp()) continue;
891 UnionFind<Cluster>* cluster = find_output_cluster(src);
892 // Merge a node with its data operands and with its control operands if
893 // the src and dst are in the same ControlContext. The ControlContext is
894 // not explicitly available here, and instead the switch depth is used as
895 // a proxy here. Due to the invariant that control edges can only be from
896 // a containing scope to an inner scope or from the inner scope to its
897 // containing scope (for exit nodes), the switch depth will only match if
898 // the src and dst are in the same ControlContext. Control edges between
899 // ControlContexts are handled during the extraction.
900 int src_id = cluster->Get().representative;
901 int src_depth = switch_depth[src_id];
902 if (!e->IsControlEdge() || new_switch_depth == src_depth) {
903 if (src_depth != new_switch_depth) {
904 // TODO(b/77601805) remove this when outside_compilation supports
905 // control flow.
906 if (str_util::StrContains(src->name(), "outside_compilation") ||
907 str_util::StrContains(n->name(), "outside_compilation")) {
908 return errors::InvalidArgument(
909 "outside_compilation is not yet supported within TensorFlow "
910 "control flow constructs b/77601805");
911 }
912 return errors::InvalidArgument(
913 "Unable to functionalize control flow in graph: Operand ('",
914 src->name(), "') and operator ('", n->name(),
915 "') have different switch depths (", src_depth,
916 " != ", new_switch_depth, ")");
917 }
918 cluster->Merge(&clusters[n->id()]);
919 }
920 }
921 }
922
923 if (dump_graphs_) {
924 // Mark the switch cluster each node is part of.
925 for (Node* n : graph_->nodes()) {
926 n->ClearAttr("_XlaFunctionalizeSwitchGroup");
927 n->AddAttr("_XlaFunctionalizeSwitchGroup",
928 clusters[n->id()].Get().representative);
929 }
930 LOG(INFO) << "FunctionalizeControlFlow (with_clusters): "
931 << dump_graph::DumpGraphToFile("functionalize_clustered", *graph_,
932 library_);
933 }
934
935 // Verify all the nodes of a cluster are at the same depth.
936 std::unordered_map<int, std::pair<int, Node*>> cluster_to_depth_node;
937 for (Node* n : graph_->nodes()) {
938 int depth = switch_depth[n->id()];
939 int cluster_rep = clusters[n->id()].Get().representative;
940 auto it = cluster_to_depth_node.find(cluster_rep);
941 if (it == cluster_to_depth_node.end()) {
942 cluster_to_depth_node[cluster_rep] = std::make_pair(depth, n);
943 } else {
944 if (it->second.first != depth) {
945 return errors::Internal(
946 "Illegal clustering created, mismatch in depths:", "\n\t",
947 n->DebugString(), "(", clusters[n->id()].Get().representative,
948 ") at depth=", depth, " vs\n\t", it->second.second->DebugString(),
949 "(", clusters[n->id()].Get().representative, ") at depth ",
950 it->second.first);
951 }
952 }
953 }
954
955 struct Hash {
956 size_t operator()(const std::pair<Node*, Cluster>& item) const {
957 return Hash64Combine(hash<Node*>()(item.first),
958 std::hash<int>()(item.second.representative));
959 }
960 };
961
962 // Merge Switch nodes with common predicate.
963 std::unordered_map<std::pair<Node*, Cluster>, int, Hash> predicate_index;
964 // The nodes in switch_order are in reverse topological order, but the
965 // clustered switches need not be (i.e., when considered as a cluster one
966 // element of a cluster may be later in the topological order than another
967 // node whose cluster is later in the topological order of clustered
968 // switches).
969 for (auto it = switch_order.rbegin(); it != switch_order.rend(); ++it) {
970 const Edge* pred_edge;
971 TF_CHECK_OK((*it)->input_edge(1, &pred_edge));
972 // The predicate can be preceded by a identity node. Look through identity
973 // nodes to predicate.
974 while (pred_edge->src()->IsIdentity()) {
975 TF_CHECK_OK(pred_edge->src()->input_edge(0, &pred_edge));
976 }
977 auto repr = std::make_pair(pred_edge->src(), clusters[(*it)->id()].Get());
978 if (predicate_index.find(repr) == predicate_index.end()) {
979 predicate_index[repr] = switch_clusters.size();
980 switch_clusters.emplace_back(pred_edge);
981 // Generate a name by concatenating with the cluster representative as
982 // there could be multiple switch clusters with the same predicate.
983 switch_clusters[predicate_index[repr]].name = strings::StrCat(
984 pred_edge->src()->name(), "_", repr.second.representative, "_If");
985 }
986 switch_clusters[predicate_index[repr]].switches.push_back(*it);
987 }
988
989 return switch_clusters;
990}

992StatusOr<std::vector<Node*>>
993FunctionalizeCond::EnsureDominanceAndReturnNonDominatedControlNodes(
994 const std::unordered_map<Node*, ForwardFlowNode>& branch_map,
995 const std::vector<Node*>& switches) {
996 std::vector<Node*> old_control_nodes;
997 for (const auto& kv : branch_map) {
998 if (kv.second.count != kv.first->in_edges().size()) {
999 std::vector<const Edge*> delete_edges;
1000 for (const Edge* in : kv.first->in_edges()) {
1001 auto it = branch_map.find(in->src());
1002 if (it == branch_map.end()) {
1003 if (in->IsControlEdge()) {
1004 old_control_nodes.push_back(in->src());
1005 delete_edges.push_back(in);
1006 } else {
1007 if (IsSwitch(in->src())) {
1008 if (std::find(switches.begin(), switches.end(), in->src()) ==
1009 switches.end()) {
1010 return errors::Internal(
1011 "Unexpected switch node found during flow forward: ",
1012 in->src()->DebugString());
1013 }
1014 continue;
1015 }
1016 return errors::InvalidArgument(
1017 "Value ", kv.first->name(), "'s input, ", in->src()->name(),
1018 ", is not dominated by switch nodes ", NodesToString(switches));
1019 }
1020 }
1021 }
1022 // Remove control edges from nodes that are not dominated by the switch
1023 // nodes. New control dependencies will be added between these nodes and
1024 // the XlaIf node inserted.
1025 for (const Edge* e : delete_edges) {
1026 graph_->RemoveEdge(e);
1027 }
1028 }
1029 }
1030 return old_control_nodes;
1031}

1033StatusOr<
1034 std::pair<std::unordered_map<Node*, FunctionalizeCond::ForwardFlowNode>,
1035 std::unordered_set<Node*>>>
1036FunctionalizeCond::DetermineBranchMapAndFrontier(
1037 const SwitchCluster& switch_cluster) {
1038 std::unordered_map<Node*, ForwardFlowNode> branch_map;
1039 std::unordered_set<Node*> frontier;
1040 std::vector<Node*> stack = switch_cluster.switches;
1041 std::vector<bool> visited(graph_->num_node_ids(), false);
1042 while (!stack.empty()) {
1043 Node* n = stack.back();
1044 stack.pop_back();
1045
1046 if (visited[n->id()]) {
1047 continue;
1048 }
1049 visited[n->id()] = true;
1050
1051 // Propagate branch state along each edge of a switch node.
1052 bool sink_only = true;
1053 for (const Edge* e : n->out_edges()) {
1054 Node* out = e->dst();
1055 if (!out->IsOp()) {
1056 continue;
1057 }
1058 sink_only = false;
1059 // Propagate branch information.
1060 ForwardFlowNode& ffn = branch_map[out];
1061 if (IsSwitch(n)) {
1062 int index = e->IsControlEdge() ? Branch::kNeither : e->src_output();
1063 TF_RETURN_WITH_CONTEXT_IF_ERROR(
1064 Join(ForwardFlowNode(Branch(index)), out, &ffn), " when joining ",
1065 e->DebugString());
1066 } else {
1067 TF_RETURN_WITH_CONTEXT_IF_ERROR(Join(branch_map[n], out, &ffn),
1068 " when joining ", e->DebugString());
1069 }
1070 if (IsMerge(out)) {
1071 if (out->in_edges().size() == ffn.count) {
1072 frontier.insert(out);
1073 }
1074 } else if (!visited[out->id()]) {
1075 stack.push_back(out);
1076 }
1077 }
1078 if (sink_only) {
1079 if (!IsIdentity(n)) {
1080 VLOG(1) << "Feeding into sink: " << n->DebugString();
1081 }
1082 }
1083 }
1084
1085 if (dump_graphs_) {
1086 for (const auto& kv : branch_map) {
1087 // Append attribute to the graph if running with logging to make the
1088 // changes clearer in the visualization.
1089 kv.first->AddAttr("_XlaFunctionalizeBranch",
1090 Branch_Name(kv.second.branch));
1091 }
1092 }
1093 return std::make_pair(std::move(branch_map), std::move(frontier));
1094}

1096Status FunctionalizeCond::FunctionalizeInternal() {
1097 TF_ASSIGN_OR_RETURN(std::vector<SwitchCluster> predicate_switch_order,
1098 DeterminePredicateSwitchOrder());
1099
1100 // Iterate from innermost set of clustered switches to outermost, replacing
1101 // matching switch->merge subgraphs with single XlaIf nodes.
1102 for (auto it = predicate_switch_order.rbegin();
1103 it != predicate_switch_order.rend(); ++it) {
1104 auto& ps = *it;
1105 VLOG(3) << "Flow down from: " << ps.ToString();
1106
1107 std::unordered_map<Node*, ForwardFlowNode> branch_map;
1108 std::unordered_set<Node*> frontier;
1109 TF_ASSIGN_OR_RETURN(std::tie(branch_map, frontier),
1110 DetermineBranchMapAndFrontier(ps));
1111
1112 if (dump_graphs_)
1113 LOG(INFO) << "FunctionalizeControlFlow (before XlaIf conversion): "
1114 << dump_graph::DumpGraphToFile("functionalize_bc", *graph_,
1115 library_);
1116 TF_RETURN_IF_ERROR(ValidateFrontier(branch_map, frontier));
1117
1118 struct Hash {
1119 size_t operator()(const std::pair<Node*, int>& item) const {
1120 return Hash64Combine(hash<Node*>()(item.first),
1121 std::hash<int>()(item.second));
1122 }
1123 };
1124
1125 // Sort the merge and switch nodes using NodeCmp. The switch-nodes are
1126 // further grouped (post sorting) by input to the switch node as in the
1127 // functionalized form each input will be passed in only once. This grouping
1128 // should retain the sorted order.
1129 CondArgNodes cond_arg_nodes;
1130 std::sort(ps.switches.begin(), ps.switches.end(), NodeCmp());
1131 std::unordered_map<std::pair<Node*, int>, int, Hash> input_index;
1132 for (Node* switch_node : ps.switches) {
1133 const Edge* e;
1134 TF_RETURN_IF_ERROR(switch_node->input_edge(0, &e));
1135 std::pair<Node*, int> key = std::make_pair(e->src(), e->src_output());
1136 if (input_index.find(key) == input_index.end()) {
1137 input_index[key] = cond_arg_nodes.size();
1138 cond_arg_nodes.emplace_back(key.first, key.second);
1139 }
1140 cond_arg_nodes.at(input_index.at(key)).switches.push_back(switch_node);
1141 }
1142 std::vector<Node*> merge_nodes(frontier.begin(), frontier.end());
1143 std::sort(merge_nodes.begin(), merge_nodes.end(), NodeCmp());
1144
1145 TF_ASSIGN_OR_RETURN(std::vector<Node*> old_control_nodes,
1146 EnsureDominanceAndReturnNonDominatedControlNodes(
1147 branch_map, ps.switches));
1148
1149 TF_ASSIGN_OR_RETURN(Node * if_node,
1150 ConvertToXlaIf(cond_arg_nodes, ps, merge_nodes));
1151 for (Node* old : old_control_nodes) {
1152 graph_->AddControlEdge(old, if_node);
1153 }
1154
1155 for (auto& del_kv : branch_map) {
1156 graph_->RemoveNode(del_kv.first);
1157 }
1158 for (auto& kv : cond_arg_nodes) {
1159 for (Node* node : kv.switches) {
1160 graph_->RemoveNode(node);
1161 }
1162 }
1163 if (dump_graphs_)
1164 LOG(INFO) << "FunctionalizeControlFlow (after XlaIf conversion): "
1165 << dump_graph::DumpGraphToFile("functionalize_ac", *graph_,
1166 library_);
1167 }
1168 return Status::OK();
1169}
1170
1171StatusOr<Node*> FunctionalizeCond::BuildAndAddXlaIfOp(
1172 const CondArgNodes& cond_arg_nodes, const SwitchCluster& switch_cluster,
1173 const std::vector<Node*>& merge_nodes) {
1174 VLOG(2) << "Build if op for " << switch_cluster.name;
1175
1176 NodeDef if_def;
1177 // Create a new If node using the name of the merge node.
1178 NodeDefBuilder builder(switch_cluster.name, "XlaIf");
1179 string branch[] = {"else_branch", "then_branch"};
1180 for (int i = 0; i < 2; ++i) {
1181 static std::atomic<int64> sequence_num(0LL);
1182 int64 id = ++sequence_num;
1183
1184 NameAttrList body_name;
1185 body_name.set_name(
1186 strings::StrCat("_functionalize_if_", branch[i], "_", id));
1187 auto body = xla::MakeUnique<Graph>(graph_->op_registry());
1188 TF_RETURN_IF_ERROR(ExtractBody(cond_arg_nodes, switch_cluster.switches,
1189 merge_nodes, i, body.get()));
1190 VLOG(3) << "Body " << branch[i] << ": " << DebugString(body.get());
1191 FunctionDef body_fdef;
1192 TF_RETURN_IF_ERROR(GraphToFunctionDef(*body, body_name.name(), &body_fdef));
1193 TF_RETURN_IF_ERROR(library_->AddFunctionDef(body_fdef));
1194 builder.Attr(branch[i], body_name);
1195 }
1196
1197 // Build input type.
1198 std::vector<NodeDefBuilder::NodeOut> inputs;
1199 DataTypeVector in_arg_types;
1200 for (auto& kv : cond_arg_nodes) {
1201 bool inserted = false;
1202 for (const Node* arg : kv.switches) {
1203 const Edge* in_edge;
1204 TF_RETURN_IF_ERROR(arg->input_edge(0, &in_edge));
1205 if (in_edge->IsControlEdge()) {
1206 builder.ControlInput(in_edge->src()->name());
1207 } else {
1208 if (!inserted) {
1209 DataType dtype = arg->input_type(0);
1210 inputs.emplace_back(NodeDefBuilder::NodeOut(
1211 in_edge->src()->name(), in_edge->src_output(), dtype));
1212 in_arg_types.push_back(dtype);
1213 inserted = true;
1214 }
1215 }
1216 }
1217 }
1218 builder.Attr("Tin", in_arg_types);
1219
1220 // Build output type.
1221 DataTypeVector out_type;
1222 for (const Node* merge : merge_nodes) {
1223 DataType dtype = merge->output_type(0);
1224 out_type.push_back(dtype);
1225 }
1226 builder.Attr("Tout", out_type);
1227
1228 builder.Attr("Tcond", DT_BOOL);
1229 builder.Device(switch_cluster.predicate_edge->src()->assigned_device_name());
1230 // Conditional should be the first input ...
1231 builder.Input(NodeDefBuilder::NodeOut(
1232 switch_cluster.predicate_edge->src()->name(),
1233 switch_cluster.predicate_edge->src_output(),
1234 switch_cluster.predicate_edge->src()->output_type(0)));
1235 // ... followed by the other inputs.
1236 builder.Input(inputs);
1237
1238 TF_RETURN_IF_ERROR(builder.Finalize(&if_def));
1239 TF_ASSIGN_OR_RETURN(Node * if_node, AddNode(if_def, graph_));
1240 return if_node;
1241}

1243Status FunctionalizeCond::ExtractBody(const CondArgNodes& cond_arg_nodes,
1244 const std::vector<Node*>& switches,
1245 const std::vector<Node*>& merge_nodes,
1246 int input_edge, Graph* body) {
1247 VLOG(2) << "ExtractBody for " << NodesToString(merge_nodes) << " along edge "
1248 << input_edge;
1249 std::vector<bool> squash_src_outputs(graph_->num_node_ids(), false);
1250 std::vector<Node*> node_map(graph_->num_node_ids(), nullptr);
1251 int arg_count = 0;
1252 for (auto& kv : cond_arg_nodes) {
1253 Node* arg_node = nullptr;
1254 for (const auto* arg : kv.switches) {
1255 DataType dtype = arg->input_type(0);
1256 if (arg_node == nullptr) {
1257 TF_ASSIGN_OR_RETURN(arg_node, BuildArgNode(body, dtype, arg_count++));
1258 }
1259 node_map.at(arg->id()) = arg_node;
1260 squash_src_outputs.at(arg->id()) = true;
1261 }
1262 }
1263
1264 std::vector<Node*> stack;
1265 stack.reserve(merge_nodes.size());
1266 for (int j = 0; j < merge_nodes.size(); ++j) {
1267 Node* node = merge_nodes[j];
1268 TF_ASSIGN_OR_RETURN(node_map.at(node->id()),
1269 BuildRetvalNode(body, node->output_type(0),
1270 /*index=*/j));
1271 const Edge* in_edge;
1272 TF_RETURN_IF_ERROR(node->input_edge(input_edge, &in_edge));
1273 Node* in = in_edge->src();
1274 if (node_map.at(in->id()) == nullptr) {
1275 node_map.at(in->id()) = body->CopyNode(in);
1276 }
1277
1278 if (std::find(switches.begin(), switches.end(), in) == switches.end()) {
1279 body->AddEdge(node_map.at(in->id()), in_edge->src_output(),
1280 node_map.at(node->id()), 0);
1281 } else {
1282 body->AddEdge(node_map.at(in->id()), 0, node_map.at(node->id()), 0);
1283 // Don't include input nodes that are already just returned in stack.
1284 continue;
1285 }
1286 stack.push_back(in);
1287 }
1288
1289 return CopySubgraph(*graph_, nullptr, stack, squash_src_outputs, &node_map,
1290 body);
1291}

1293Status FunctionalizeCond::AddInputEdges(const CondArgNodes& cond_arg_nodes,
1294 const Edge* predicate_edge,
1295 Node* if_node) {
1296 VLOG(3) << "AddInputEdges for " << if_node->name();
1297 int index = 0;
1298 graph_->AddEdge(predicate_edge->src(), predicate_edge->src_output(), if_node,
1299 index++);
1300 for (auto& arg : cond_arg_nodes) {
1301 if (arg.src_output == Graph::kControlSlot) {
1302 graph_->AddControlEdge(arg.src, if_node);
1303 } else {
1304 graph_->AddEdge(arg.src, arg.src_output, if_node, index++);
1305 }
1306 }
1307 return Status::OK();
1308}

1310Status FunctionalizeCond::AddOutputEdges(const std::vector<Node*>& outputs,
1311 Node* if_node) {
1312 VLOG(3) << "AddOutputEdges for " << if_node->name();
1313 for (int i = 0; i < outputs.size(); ++i) {
1314 Node* node = outputs[i];
1315 std::vector<const Edge*> edges(node->out_edges().begin(),
1316 node->out_edges().end());
1317 for (const Edge* edge : edges) {
1318 Node* dst = edge->dst();
1319 int dst_input = edge->dst_input();
1320
1321 if (edge->src_output() > 0) {
1322 return errors::Unimplemented("Output of index (", edge->src_output(),
1323 ") of merge node ", node->name());
1324 }
1325
1326 int src_output =
1327 dst_input == Graph::kControlSlot ? Graph::kControlSlot : i;
1328 graph_->RemoveEdge(edge);
1329 graph_->AddEdge(if_node, src_output, dst, dst_input);
1330 }
1331 }
1332 return Status::OK();
1333}

1361// Transformation that converts TensorFlow's graph control flow constructs into
1362// functional equivalents.
1363Status FunctionalizeControlFlow(Graph* graph,
1364 FunctionLibraryDefinition* library) {
1365 VLOG(2) << "FunctionalizeControlFlow (initial): "
1366 << dump_graph::DumpGraphToFile("functionalize_initial", *graph,
1367 library);
1368
1369 // Note: BuildControlFlowInfo() requires that the graph's source node is
1370 // connected to all source nodes in the graph. Many graphs violate this
1371 // invariant.
1372 std::vector<ControlFlowInfo> cf_info;
1373 TF_RETURN_IF_ERROR(BuildControlFlowInfo(graph, &cf_info));
1374
1375 // Builds Frames, indexed by name.
1376 std::unordered_map<string, Frame> frames;
1377 for (Node* node : graph->op_nodes()) {
1378 const ControlFlowInfo& cf = cf_info[node->id()];
1379
1380 VLOG(2) << "node: " << node->name() << " (" << node->id()
1381 << ") frame_name: " << cf.frame_name
1382 << " frame: " << (cf.frame ? cf.frame->name() : "---")
1383 << " parent_frame: "
1384 << (cf.parent_frame ? cf.parent_frame->name() : "---");
1385 TF_RET_CHECK(cf.frame != nullptr && cf.parent_frame != nullptr);
1386
1387 Frame& frame = frames[cf.frame_name];
1388 Frame* parent = &frames[cf_info[cf.parent_frame->id()].frame_name];
1389 if (frame.parent == nullptr) {
1390 frame.parent = parent;
1391 frame.name = cf.frame_name;
1392 ++parent->num_children;
1393 } else if (frame.parent != parent) {
1394 return errors::InvalidArgument("Mismatched parent frames for ",
1395 cf.frame->id(), ": ", parent->name, " vs ",
1396 frame.parent->name);
1397 }
1398
1399 if (IsEnter(node)) {
1400 Arg arg;
1401 arg.enter = node;
1402 TF_RETURN_IF_ERROR(GetNodeAttr(arg.enter->attrs(), "is_constant",
1403 &arg.is_loop_invariant));
1404 frame.args.push_back(arg);
1405 } else if (IsLoopCond(node)) {
1406 if (frame.loop_cond) {
1407 return errors::InvalidArgument(
1408 "Loop ", cf.frame_name,
1409 " has more than one LoopCond node: ", node->name(), " and ",
1410 frame.loop_cond->name());
1411 }
1412 frame.loop_cond = node;
1413 }
1414 frame.nodes.insert(node);
1415 }
1416
1417 // Adds frames with no children (i.e., the innermost frames) to a worklist.
1418 std::deque<Frame*> worklist;
1419 for (auto& frame : frames) {
1420 if (frame.second.num_children == 0) {
1421 worklist.push_back(&frame.second);
1422 }
1423 }
1424
1425 // Eliminate loops from innermost to outermost.
1426 while (!worklist.empty()) {
1427 Frame* frame = worklist.front();
1428 worklist.pop_front();
1429 if (frame->parent == frame) {
1430 // Skip the root frame.
1431 continue;
1432 }
1433
1434 TF_RETURN_IF_ERROR(FunctionalizeLoop(graph, frame, library));
1435
1436 // If the parent has no remaining children, add it to the worklist.
1437 --frame->parent->num_children;
1438 if (frame->parent->num_children == 0) {
1439 worklist.push_back(frame->parent);
1440 }
1441 }
1442
1443 // FunctionalizeControlFlow is invoked for every function, so the loops's
1444 // bodies and conditionals that were extracted into functions will be handled
1445 // in successive invocations.
1446 TF_RETURN_IF_ERROR(FunctionalizeCond::Functionalize(graph, library));
1447
1448 VLOG(2) << "FunctionalizeControlFlow (final): "
1449 << dump_graph::DumpGraphToFile("functionalize_final", *graph,
1450 library);
1451 return Status::OK();
1452}
