16// Utilities for working with XLA Literals.
17
18#ifndef TENSORFLOW_COMPILER_TF2XLA_LITERAL_UTIL_H_
19#define TENSORFLOW_COMPILER_TF2XLA_LITERAL_UTIL_H_
20
21#include "tensorflow/compiler/xla/literal_util.h"
22#include "tensorflow/compiler/xla/xla_data.pb.h"
23#include "tensorflow/core/framework/tensor.h"
24#include "tensorflow/core/lib/core/status.h"
25
26namespace tensorflow {
27
28// Copies 'host_tensor' to an XLA Literal. Fails if host_tensor is of an
29// unsupported type.
30Status HostTensorToLiteral(const Tensor& host_tensor, xla::Literal* literal);
31
32// Copies 'literal' to freshly allocated 'host_tensor', which is allocated of
33// type <target_type>.
34// Fails if the literal's primitive type !=
35// DataTypeToPrimitiveType(target_type). Note that <target_type> is not
36// derivable from the type of <literal>, because multiple tensorflow types map
37// to the same XLA type (e.g. INT32 and QINT32 both map to INT32 in
38// XLA).
39Status LiteralToHostTensor(const xla::Literal& literal, DataType target_type,
40 Tensor* host_tensor);
41
42// Copies the contents of 'literal' to a previously allocated tensor
43// 'host_tensor'. The tensor and the literal must have the same number of
44// elements and the same type.
45Status CopyLiteralToHostTensor(const xla::Literal& literal,
46 Tensor* host_tensor);
47
48} // namespace tensorflow
