48// Checks that arguments `args` match types `types`.
49Status CheckSignature(const DataTypeVector& types,
50 const std::vector<XlaCompiler::Argument>& args) {
51 if (args.size() != types.size()) {
52 return errors::Internal("Compilation arguments have ", args.size(),
53 " elements while function has ", types.size());
54 }
55 for (int i = 0; i < types.size(); ++i) {
56 if (types[i] != args[i].type && types[i] != DT_RESOURCE) {
57 return errors::Internal(
58 "Argument ", i, " has declared type ", DataTypeString(args[i].type),
59 " but function parameter has type ", DataTypeString(types[i]));
60 }
61 }
62 return Status::OK();
63}

67bool XlaCompiler::Argument::operator==(
68 const XlaCompiler::Argument& other) const {
69 if (std::tie(kind, resource_kind, type, name, initialized, tensor_array_size,
70 tensor_array_gradients) !=
71 std::tie(other.kind, other.resource_kind, other.type, other.name,
72 other.initialized, other.tensor_array_size,
73 other.tensor_array_gradients)) {
74 return false;
75 }
76 if (shape != other.shape) {
77 return false;
78 }
79 if (constant_value.shape() != other.constant_value.shape()) {
80 return false;
81 }
82 return constant_value.tensor_data() == other.constant_value.tensor_data();
83}

85XlaCompiler::XlaCompiler(XlaCompiler::Options options)
86 : options_(options),
87 initialization_status_(Status::OK()),
88 next_step_id_(1),
89 device_(
90 new XlaCompilationDevice(SessionOptions(), *options_.device_type)),
91 device_mgr_({device_}) {
92 // We no longer need the device_type.
93 options_.device_type = nullptr;
94
95 if (options_.populate_resource_manager) {
96 initialization_status_ =
97 (*options_.populate_resource_manager)(device_->resource_manager());
98 }
99
100 local_flib_def_.reset(new FunctionLibraryDefinition(OpRegistry::Global(),
101 FunctionDefLibrary{}));
102 local_pflr_.reset(new ProcessFunctionLibraryRuntime(
103 &device_mgr_, Env::Default(), options.graph_def_version,
104 local_flib_def_.get(), OptimizerOptions(),
105 nullptr /* custom_kernel_creator */));
106 pflr_.reset(new ProcessFunctionLibraryRuntime(
107 &device_mgr_, Env::Default(), options.graph_def_version, options.flib_def,
108 OptimizerOptions(), nullptr /* custom_kernel_creator */));
109
110 local_flib_runtime_ = local_pflr_->GetFLR(device_->name());
111 flib_runtime_ = pflr_->GetFLR(device_->name());
112
113 // The default variable representation shape is the identity function.
114 if (!options_.variable_representation_shape_fn) {
115 options_.variable_representation_shape_fn =
116 [](const TensorShape& shape, DataType type) { return shape; };
117 }
118}

129static Status GetFunctionBody(const NameAttrList& function,
130 FunctionLibraryRuntime* flib_runtime,
131 const FunctionBody** fbody) {
132 FunctionLibraryRuntime::Handle handle;
133 TF_RETURN_IF_ERROR(flib_runtime->Instantiate(
134 function.name(), AttrSlice(&function.attr()), &handle));
135
136 *fbody = flib_runtime->GetFunctionBody(handle);
137 TF_RET_CHECK(*fbody);
138 return Status::OK();
139}
140
141Status XlaCompiler::FindFunctionBody(const NameAttrList& function,
142 const FunctionBody** fbody) {
143 // The function may be in either the local_flib_runtime_ or flib_runtime_.
144 // Look up the function in local first and if it is not found then look up the
145 // function in flib_runtime_.
146 auto status = GetFunctionBody(function, local_flib_runtime_, fbody);
147 if (!status.ok()) {
148 if (!errors::IsNotFound(status)) {
149 return status;
150 }
151 TF_RETURN_WITH_CONTEXT_IF_ERROR(
152 GetFunctionBody(function, flib_runtime_, fbody),
153 "Local lookup failed with: ", status.error_message());
154 }
155 return Status::OK();
156}

158std::unique_ptr<Graph> XlaCompiler::GetGraph(const FunctionBody* fbody) {
159 std::unique_ptr<Graph> graph(new Graph(options_.flib_def));
160 CopyGraph(*fbody->graph, graph.get());
161 OptimizerOptions opts;
162 opts.set_opt_level(OptimizerOptions::L0);
163 opts.set_do_common_subexpression_elimination(false);
164 opts.set_do_function_inlining(true);
165 opts.set_do_constant_folding(true);
166 GraphOptimizer optimizer(opts);
167 optimizer.Optimize(flib_runtime_, flib_runtime_->env(),
168 /*device=*/nullptr, &graph, /*shape_map=*/nullptr);
169
170 return graph;
171}

173Status XlaCompiler::CompileFunction(const XlaCompiler::CompileOptions& options,
174 const NameAttrList& function,
175 std::vector<XlaCompiler::Argument> args,
176 XlaCompiler::CompilationResult* result) {
177 const string function_id =
178 Canonicalize(function.name(), AttrSlice(&function.attr()));
179 VLOG(1) << "XlaCompiler::CompileFunction " << function_id;
180
181 auto it = cache_.find({function_id, args});
182 if (it != cache_.end()) {
183 *result = it->second;
184 return Status::OK();
185 }
186
187 const FunctionBody* fbody;
188 TF_RETURN_IF_ERROR(FindFunctionBody(function, &fbody));
189
190 TF_RETURN_WITH_CONTEXT_IF_ERROR(
191 CheckSignature(fbody->arg_types, args),
192 "Signature check failure while compiling: ", function.name());
193
194 std::unique_ptr<Graph> graph = GetGraph(fbody);
195
196 // _Arg and _Retval nodes don't exist in the stored subgraph for the function;
197 // they are added by the function body looked up. Therefore, they don't have
198 // core assignments here.
199 // Attempt to assign a core to each _Retval and _Arg. Chooses the
200 // lowest-numbered core that consumes the argument. We choose the
201 // lowest-numbered core so the assignment is deterministic.
202 for (Node* n : graph->nodes()) {
203 if (StringPiece(n->type_string()) == "_Arg") {
204 TF_RETURN_IF_ERROR(SetNodeShardingFromNeighbors(n, /*out_edges=*/true));
205 }
206 }
207 // Do _Retval as a second loop, in case the retval's input is an _Arg (which
208 // may have gotten a device assignment from the first loop).
209 for (Node* n : graph->nodes()) {
210 if (StringPiece(n->type_string()) == "_Retval") {
211 TF_RETURN_IF_ERROR(SetNodeShardingFromNeighbors(n, /*out_edges=*/false));
212 }
213 }
214
215 if (VLOG_IS_ON(2)) {
216 VLOG(2) << "XlaCompiler::CompileFunction: "
217 << dump_graph::DumpGraphToFile(
218 strings::StrCat("xla_compile_function_", function_id),
219 *graph);
220 }
221
222 VLOG(1) << "====================================================";
223 TF_RETURN_IF_ERROR(
224 CompileGraph(options, function_id, std::move(graph), args, result));
225 VLOG(1) << "====================================================";
226
227 cache_[{function_id, args}] = *result;
228 return Status::OK();
229}

231// Computes the XLA shape for argument 'arg'.
232Status XlaCompiler::XLAShapeForArgument(const XlaCompiler::Argument& arg,
233 xla::Shape* xla_shape) {
234 switch (arg.kind) {
235 case XlaCompiler::Argument::kConstant:
236 return TensorShapeToXLAShape(arg.type, arg.constant_value.shape(),
237 xla_shape);
238 case XlaCompiler::Argument::kParameter:
239 return TensorShapeToXLAShape(arg.type, arg.shape, xla_shape);
240 case XlaCompiler::Argument::kResource: {
241 TF_RET_CHECK(arg.initialized);
242
243 switch (arg.resource_kind) {
244 case XlaResource::kVariable: {
245 TensorShape representation_shape =
246 options_.variable_representation_shape_fn(arg.shape, arg.type);
247 return TensorShapeToXLAShape(arg.type, representation_shape,
248 xla_shape);
249 }
250 case XlaResource::kTensorArray: {
251 if (arg.tensor_array_size < 0) {
252 return errors::InvalidArgument(
253 "Negative tensor_array_size in XLAShapeForArgument");
254 }
255 TensorShape shape;
256 shape.AddDim(arg.tensor_array_size);
257 shape.AppendShape(arg.shape);
258 TF_RETURN_IF_ERROR(TensorShapeToXLAShape(arg.type, shape, xla_shape));
259
260 if (!arg.tensor_array_gradients.empty()) {
261 std::vector<xla::Shape> tuple_shape(
262 arg.tensor_array_gradients.size() + 1, *xla_shape);
263 *xla_shape = xla::ShapeUtil::MakeTupleShape(tuple_shape);
264 }
265 return Status::OK();
266 }
267 case XlaResource::kStack: {
268 if (arg.tensor_array_size < 0) {
269 return errors::InvalidArgument(
270 "Negative tensor_array_size in XLAShapeForArgument");
271 }
272 TensorShape shape;
273 shape.AddDim(arg.tensor_array_size);
274 shape.AppendShape(arg.shape);
275 xla::Shape buffer_shape;
276 TF_RETURN_IF_ERROR(
277 TensorShapeToXLAShape(arg.type, shape, &buffer_shape));
278 *xla_shape = xla::ShapeUtil::MakeTupleShape(
279 {buffer_shape, xla::ShapeUtil::MakeShape(xla::S32, {})});
280 return Status::OK();
281 }
282
283 case XlaResource::kInvalid:
284 return errors::Internal(
285 "Invalid resource type in XLAShapeForArgument()");
286 }
287 }
288 case XlaCompiler::Argument::kInvalid:
289 return errors::Internal("Invalid argument type in XLAShapeForArgument()");
290 }

295Status ExecuteGraph(XlaContext* xla_context, std::unique_ptr<Graph> graph,
296 XlaCompilationDevice* device, FunctionLibraryRuntime* flib,
297 int64 step_id) {
298 // Resource cleanup is a bit messy. XlaContext is a ref-countd resource; the
299 // resource manager takes ownership via Create, and unrefs via Cleanup. We
300 // explicitly add a reference to ensure the refcount at entry is maintained at
301 // all exit points; Create and Cleanup are always called in this function.
302 //
303 // The Executor requires us to use ScopedStepContainer. We wrap it in a
304 // unique_ptr so we can capture the cleanup status in the end.
305 xla_context->Ref();
306 Status status;
307 auto step_container = xla::MakeUnique<ScopedStepContainer>(
308 step_id, [&status, device](const string& name) {
309 status = device->resource_manager()->Cleanup(name);
310 });
311 TF_RETURN_IF_ERROR(device->resource_manager()->Create(
312 step_container->name(), XlaContext::kXlaContextResourceName,
313 xla_context));
314
315 GraphCompiler graph_compiler(xla_context, device, graph.get(), flib,
316 step_container.get());
317 TF_RETURN_IF_ERROR(graph_compiler.Compile());
318 // Explicitly clean up the step container, to capture the cleanup status.
319 step_container.reset();
320 return Status::OK();
321}

323// Builds the XLA computation.
324//
325// `retvals` is the list of retvals produced by _Retval operators, in index
326// order. `variable_map` is a map from variable ID numbers to XlaOpContext
327// variable states, generated by the symbolic evaluation.
328// If `return_updated_values_for_all_resources` is true, all resources will be
329// included in `resource_updates`, regardless of whether their value changed.
330// Sets `*num_nonconst_outputs` to the number of outputs of the `computation`.
331// Sets `*resource_updates` to a description of resources whose values are
332// written by the computation; the variable writes are the last
333// `resource_updates.size()` return values from the computation. Each entry in
334// `resource_updates` is a (input_index, type) pair, where `input_index` is the
335// index of a resource variable argument to the computation, and `type` is the
336// type of the final output.
337Status BuildComputation(
338 const std::vector<XlaCompiler::Argument>& args,
339 const std::vector<int>& arg_cores,
340 const std::vector<XlaExpression>& retvals,
341 const std::vector<std::unique_ptr<XlaResource>>& resources,
342 bool return_updated_values_for_all_resources,
343 xla::ComputationBuilder* builder, xla::Computation* computation,
344 int* num_computation_outputs, int* num_nonconst_outputs,
345 std::vector<XlaCompiler::ResourceUpdate>* resource_updates) {
346 std::vector<xla::ComputationDataHandle> elems;
347 elems.reserve(retvals.size());
348 for (const XlaExpression& retval : retvals) {
349 if (!retval.has_constant_value()) {
350 elems.push_back(retval.handle());
351 }
352 }
353 *num_nonconst_outputs = elems.size();
354
355 // Add return values for resources whose values have changed.
356 std::vector<const XlaResource*> arg_resources;
357 arg_resources.reserve(resources.size());
358 for (const auto& resource : resources) {
359 if (resource->arg_num() >= 0) {
360 arg_resources.push_back(resource.get());
361 }
362 }
363 std::sort(arg_resources.begin(), arg_resources.end(),
364 [](const XlaResource* a, const XlaResource* b) {
365 return a->arg_num() < b->arg_num();
366 });
367
368 // Attach a common operator name as metadata. This has no semantic effect it
369 // merely makes the HLO graph more readable when visualized via TensorBoard,
370 // since TensorBoard forms groups out of operators with similar names.
371 xla::OpMetadata retval_metadata;
372 retval_metadata.set_op_name("XLA_Retvals");
373 builder->SetOpMetadata(retval_metadata);
374
375 for (const XlaResource* resource : arg_resources) {
376 const XlaCompiler::Argument& arg = args[resource->arg_num()];
377 const int core = arg_cores[resource->arg_num()];
378 DCHECK_LT(resource->arg_num(), arg_cores.size());
379 bool modified =
380 resource->value().handle() != resource->initial_value().handle();
381 // TensorArray gradients were modified if their values changed or there are
382 // any newly created gradients.
383 for (const auto& grad : resource->tensor_array_gradients()) {
384 modified = modified ||
385 grad.second->value().handle() !=
386 grad.second->initial_value().handle() ||
387 arg.tensor_array_gradients.count(grad.first) == 0;
388 }
389 if (return_updated_values_for_all_resources || modified) {
390 resource_updates->emplace_back();
391 XlaCompiler::ResourceUpdate& update = resource_updates->back();
392 update.input_index = resource->arg_num();
393 update.type = resource->type();
394 update.shape = resource->shape();
395 update.modified = modified;
396 for (const auto& grad : resource->tensor_array_gradients()) {
397 update.tensor_array_gradients_accessed.insert(grad.first);
398 }
399
400 // Request that the value be returned on a specific core.
401 xla::ScopedShardingAssignment assign_sharding(
402 builder, core == -1 ? tensorflow::gtl::optional<xla::OpSharding>()
403 : xla::sharding_builder::AssignDevice(core));
404
405 xla::ComputationDataHandle handle;
406 TF_RETURN_IF_ERROR(resource->Pack(&handle, builder));
407
408 // Since we can't change the sharding metadata of <value> as this point,
409 // create a tuple/get-tuple-element combination so that sharding
410 // assignment will be placed on this value, which will cause the resource
411 // update to be returned from the same device that provided the resource.
412 handle = builder->GetTupleElement(builder->Tuple({handle}), 0);
413
414 elems.push_back(handle);
415 }
416 }
417
418 *num_computation_outputs = elems.size();
419
420 // Builds the XLA computation.
421 builder->Tuple(elems);
422 builder->ClearOpMetadata();
423
424 xla::StatusOr<xla::Computation> computation_status = builder->Build();
425 if (!computation_status.ok()) {
426 return computation_status.status();
427 }
428 *computation = computation_status.ConsumeValueOrDie();
429 return Status::OK();
430}

434// Builds XLA computations for each of the arguments to the computation.
435// `args` are the arguments to the computation.
436Status XlaCompiler::BuildArguments(
437 const Graph& graph, const std::vector<XlaCompiler::Argument>& args,
438 bool use_tuple_arg, xla::ComputationBuilder* builder, XlaContext* context,
439 std::vector<int>* arg_cores, std::vector<XlaExpression>* arg_expressions,
440 std::vector<int>* input_mapping, std::vector<xla::Shape>* input_shapes,
441 bool is_entry_computation) {
442 arg_expressions->resize(args.size());
443 *arg_cores = std::vector<int>(args.size(), -1);
444
445 // Argument numbers of arguments and resources that are to be passed to the
446 // XLA computation as runtime parameters.
447 input_mapping->clear();
448 input_mapping->reserve(args.size());
449 std::vector<int> resources;
450 resources.reserve(args.size());
451
452 // Fills in constant arguments, and computes non-constant argument order.
453 for (std::vector<XlaCompiler::Argument>::size_type i = 0; i < args.size();
454 ++i) {
455 const XlaCompiler::Argument& arg = args[i];
456 XlaExpression& arg_expression = (*arg_expressions)[i];
457 switch (arg.kind) {
458 case XlaCompiler::Argument::kResource:
459 TF_RET_CHECK(arg.resource_kind != XlaResource::kInvalid);
460 // TODO(phawkins): this code assumes that resource arguments do not
461 // alias.
462 XlaResource* resource;
463 TF_RETURN_IF_ERROR(context->CreateResource(
464 arg.resource_kind, i, arg.name, arg.type, arg.shape,
465 xla::ComputationDataHandle(),
466 /*tensor_array_size=*/arg.tensor_array_size,
467 /*tensor_array_gradients=*/arg.tensor_array_gradients, &resource));
468 arg_expression.set_resource(resource);
469 if (arg.initialized) {
470 resources.push_back(i);
471 }
472 break;
473 case XlaCompiler::Argument::kParameter: {
474 input_mapping->push_back(i);
475 break;
476 }
477 case XlaCompiler::Argument::kConstant:
478 arg_expression.set_constant_value(arg.constant_value);
479 break;
480 case XlaCompiler::Argument::kInvalid:
481 return errors::Internal("Unreachable case in BuildArguments()");
482 }
483 }
484
485 // Append parameters containing variable values after the other runtime
486 // parameters.
487 input_mapping->insert(input_mapping->end(), resources.begin(),
488 resources.end());
489 if (input_mapping->empty()) {
490 return Status::OK();
491 }
492
493 std::vector<xla::Shape> arg_shapes(input_mapping->size());
494 for (std::vector<int>::size_type i = 0; i < input_mapping->size(); ++i) {
495 // Computes the shapes of non-constant arguments.
496 TF_RETURN_IF_ERROR(
497 XLAShapeForArgument(args[(*input_mapping)[i]], &arg_shapes[i]));
498 }
499
500 if (use_tuple_arg) {
501 input_shapes->push_back(xla::ShapeUtil::MakeTupleShape(arg_shapes));
502 } else {
503 *input_shapes = arg_shapes;
504 }
505
506 // Use the _Arg nodes in the graph to resolve core assignments.
507 for (const Node* n : graph.nodes()) {
508 if (StringPiece(n->type_string()) != "_Arg") continue;
509 int index;
510 TF_RETURN_IF_ERROR(GetNodeAttr(n->attrs(), "index", &index));
511 TF_RET_CHECK(index >= 0 && index < args.size())
512 << "_Arg out of bounds: " << index << " vs " << args.size();
513 TF_ASSIGN_OR_RETURN(
514 auto sharding,
515 ParseShardingFromDevice(*n, std::numeric_limits<int32>::max()));
516 if (sharding.has_value()) {
517 TF_RET_CHECK(sharding.value().type() ==
518 xla::OpSharding::Type::OpSharding_Type_MAXIMAL);
519 const int core = sharding.value().tile_assignment_devices(0);
520 if ((*arg_cores)[index] == -1 || core < (*arg_cores)[index]) {
521 (*arg_cores)[index] = core;
522 }
523 }
524 }
525
526 // Attach a common operator name as metadata. This has no semantic effect it
527 // merely makes the HLO graph more readable when visualized via TensorBoard,
528 // since TensorBoard forms groups out of operators with similar names.
529 xla::OpMetadata arg_metadata;
530 arg_metadata.set_op_name("XLA_Args");
531 builder->SetOpMetadata(arg_metadata);
532
533 // Build parameter handles for non-constant arguments.
534 std::vector<xla::ComputationDataHandle> arg_handles(input_mapping->size());
535 if (use_tuple_arg) {
536 xla::ComputationDataHandle tuple;
537 if (is_entry_computation) {
538 xla::OpSharding tuple_sharding;
539 tuple_sharding.set_type(xla::OpSharding::Type::OpSharding_Type_TUPLE);
540 for (int64 parameter : *input_mapping) {
541 const int core = (*arg_cores)[parameter];
542 const int root_device = 0;
543 *tuple_sharding.add_tuple_shardings() =
544 core == -1 ? xla::sharding_builder::AssignDevice(root_device)
545 : xla::sharding_builder::AssignDevice(core);
546 }
547 xla::ScopedShardingAssignment assign_tuple_sharding(builder,
548 tuple_sharding);
549 tuple = builder->Parameter(0, (*input_shapes)[0], "arg_tuple");
550 } else {
551 tuple = builder->Parameter(0, (*input_shapes)[0], "arg_tuple");
552 }
553 for (std::vector<int>::size_type i = 0; i < input_mapping->size(); ++i) {
554 const int core = (*arg_cores)[input_mapping->at(i)];
555 xla::ScopedShardingAssignment assign_sharding(
556 builder, core == -1 ? tensorflow::gtl::optional<xla::OpSharding>()
557 : xla::sharding_builder::AssignDevice(core));
558 arg_handles[i] = builder->GetTupleElement(tuple, i);
559 }
560 } else {
561 for (std::vector<int>::size_type i = 0; i < input_mapping->size(); ++i) {
562 const int core = (*arg_cores)[input_mapping->at(i)];
563 xla::ScopedShardingAssignment assign_sharding(
564 builder, core == -1 ? tensorflow::gtl::optional<xla::OpSharding>()
565 : xla::sharding_builder::AssignDevice(core));
566 arg_handles[i] =
567 builder->Parameter(i, (*input_shapes)[i], strings::StrCat("arg", i));
568 }
569 }
570
571 builder->ClearOpMetadata();
572
573 // Fill in the handles in non-constant arguments.
574 VLOG(2) << "XLA computation inputs:";
575 for (std::vector<int>::size_type i = 0; i < input_mapping->size(); ++i) {
576 const XlaCompiler::Argument& arg = args[input_mapping->at(i)];
577 VLOG(2) << " XLA arg " << i
578 << " shape: " << xla::ShapeUtil::HumanString(arg_shapes[i])
579 << " name: " << arg.name << " TF arg " << input_mapping->at(i);
580 XlaExpression& arg_expression = (*arg_expressions)[input_mapping->at(i)];
581 switch (arg.kind) {
582 case XlaCompiler::Argument::kResource: {
583 TF_RET_CHECK(arg.initialized);
584 XlaResource* resource = arg_expression.resource();
585 TF_RETURN_IF_ERROR(resource->SetFromPack(arg.tensor_array_gradients,
586 arg_handles[i], builder));
587 VLOG(2) << " resource: num_gradients: "
588 << arg.tensor_array_gradients.size();
589 break;
590 }
591 case XlaCompiler::Argument::kParameter:
592 arg_expression.set_handle(arg_handles[i]);
593 break;
594 case XlaCompiler::Argument::kConstant:
595 case XlaCompiler::Argument::kInvalid:
596 return errors::Internal("Unreachable case in BuildArguments()");
597 }
598 }
599
600 return Status::OK();
601}

603Status XlaCompiler::CompileSingleOp(
604 const XlaCompiler::CompileOptions& options, string const& name,
605 OpKernelContext* ctx, const std::vector<XlaCompiler::Argument>& args,
606 CompilationResult* result) {
607 // TODO(b/74182462): We implement this by creating a new dummy Graph including
608 // _Arg nodes, and let CompileGraph walk it. This could be optimized.
609 std::unique_ptr<Graph> graph(new Graph(OpRegistry::Global()));
610
611 Status status;
612 // First create the actual node we care about computing.
613 Node* main_node = graph->AddNode(ctx->op_kernel().def(), &status);
614 TF_RETURN_IF_ERROR(status);
615
616 // Create dummy _Arg nodes. Link these to `node` and also via a control
617 // dependency edge to the _SOURCE node.
618 for (int64 i = 0; i < ctx->num_inputs(); ++i) {
619 Node* node;
620 string name = strings::StrCat(ctx->op_kernel().name(), "_", i, "_arg");
621 Status status = NodeBuilder(name, "_Arg")
622 .ControlInput(graph->source_node())
623 .Attr("T", ctx->input_dtype(i))
624 .Attr("index", i)
625 .Finalize(graph.get(), &node);
626 TF_RETURN_IF_ERROR(status);
627 graph->AddEdge(node, 0, main_node, i);
628 }
629
630 // Similarly with return values, create dummy _Retval nodes fed by `node`.
631 for (int64 i = 0; i < ctx->num_outputs(); ++i) {
632 Node* node;
633 string name = strings::StrCat(ctx->op_kernel().name(), "_", i, "_retval");
634 Status status = NodeBuilder(name, "_Retval")
635 .Input(main_node, i)
636 .Attr("T", ctx->expected_output_dtype(i))
637 .Attr("index", i)
638 .Finalize(graph.get(), &node);
639 TF_RETURN_IF_ERROR(status);
640 }
641
642 return CompileGraph(options, name, std::move(graph), args, result);
643}

645Status XlaCompiler::CompileGraph(const XlaCompiler::CompileOptions& options,
646 string const& name,
647 std::unique_ptr<Graph> graph,
648 const std::vector<XlaCompiler::Argument>& args,
649 CompilationResult* result) {
650 VLOG(1) << "Executing graph symbolically to populate ComputationBuilder.";
651
652 if (VLOG_IS_ON(2)) {
653 VLOG(2) << "XlaCompiler::CompileGraph: "
654 << dump_graph::DumpGraphToFile(
655 strings::StrCat("xla_compile_graph_", name), *graph);
656 }
657
658 // Report the error here if initialization failed.
659 TF_RETURN_IF_ERROR(initialization_status_);
660
661 // Converts Tensorflow's graph control-flow constructs into functional
662 // control-flow that can be compiled into XLA code.
663 TF_RETURN_IF_ERROR(
664 FunctionalizeControlFlow(graph.get(), local_flib_def_.get()));
665
666 xla::ComputationBuilder builder(client(), name);
667 XlaContext* context =
668 new XlaContext(this, &builder, options_.allow_cpu_custom_calls,
669 options.resolve_compile_time_constants,
670 &options_.variable_representation_shape_fn);
671 core::ScopedUnref context_unref(context);
672
673 std::vector<XlaExpression> arg_expressions;
674 std::vector<int> arg_cores;
675 TF_RETURN_IF_ERROR(
676 BuildArguments(*graph, args, options.use_tuple_arg, &builder, context,
677 &arg_cores, &arg_expressions, &result->input_mapping,
678 &result->xla_input_shapes, options.is_entry_computation));
679 context->set_args(std::move(arg_expressions));
680
681 TF_RETURN_IF_ERROR(ExecuteGraph(context, std::move(graph), device_,
682 flib_runtime_, NextStepId()));
683
684 int num_nonconst_outputs;
685 int num_computation_outputs;
686 result->computation = std::make_shared<xla::Computation>();
687 TF_RETURN_IF_ERROR(BuildComputation(
688 args, arg_cores, context->retvals(), context->resources(),
689 options.return_updated_values_for_all_resources, &builder,
690 result->computation.get(), &num_computation_outputs,
691 &num_nonconst_outputs, &result->resource_updates));
692
693 VLOG(2) << "Outputs: total: " << context->retvals().size()
694 << " nonconstant: " << num_nonconst_outputs;
695 result->outputs.resize(context->retvals().size());
696 for (std::vector<XlaExpression>::size_type i = 0;
697 i < context->retvals().size(); ++i) {
698 const XlaExpression& retval = context->retvals()[i];
699 if (retval.has_constant_value()) {
700 OutputDescription& output = result->outputs[i];
701 output.shape = retval.constant_value().shape();
702 output.is_constant = true;
703 output.constant_value = retval.constant_value();
704 }
705 }
706
707 // Compute the output shapes, if there is a computation with non-constant
708 // outputs.
709 auto computation_shape = client()->GetComputationShape(*result->computation);
710 if (!computation_shape.ok()) {
711 return computation_shape.status();
712 }
713
714 result->xla_output_shape.Swap(
715 computation_shape.ValueOrDie()->mutable_result());
716 VLOG(2) << "XLA output shape: "
717 << xla::ShapeUtil::HumanString(result->xla_output_shape);
718
719 // Copy the host transfer metadata to the result.
720 for (const auto& send : host_compute_sends_) {
721 *result->host_compute_metadata.add_device_to_host() = send.second;
722 }
723 for (const auto& recv : host_compute_recvs_) {
724 *result->host_compute_metadata.add_host_to_device() = recv.second;
725 }
726
727 // Tensorflow expects a major-to-minor order of results.
728 xla::LayoutUtil::SetToDefaultLayout(&result->xla_output_shape);
729
730 // Converts the output shapes to TensorShapes.
731 int computation_output = 0;
732 for (std::vector<XlaExpression>::size_type i = 0;
733 i < context->retvals().size(); ++i) {
734 const XlaExpression& retval = context->retvals()[i];
735 if (!retval.has_constant_value()) {
736 TF_RET_CHECK(computation_output < num_computation_outputs)
737 << "Computation has more outputs than expected";
738 OutputDescription& output = result->outputs[i];
739 output.is_constant = false;
740 TF_RETURN_IF_ERROR(XLAShapeToTensorShape(
741 xla::ShapeUtil::GetTupleElementShape(result->xla_output_shape,
742 computation_output),
743 &output.shape));
744 ++computation_output;
745 }
746 }
747 return Status::OK();
748}

777Status XlaCompiler::SetDeviceToHostMetadata(
778 const string& key, gtl::ArraySlice<DataType> types,
779 gtl::ArraySlice<TensorShape> shapes) {
780 if (host_compute_sends_.find(key) != host_compute_sends_.end()) {
781 return errors::InvalidArgument(
782 "Duplicate calls to SetDeviceToHostMetadata with key ", key);
783 }
784 tf2xla::HostTransferMetadata& transfer = host_compute_sends_[key];
785 SetTransfer(key, types, shapes, &transfer);
786 return Status::OK();
787}
788
789Status XlaCompiler::GetDeviceToHostShapes(
790 const string& key, std::vector<TensorShape>* shapes) const {
791 const auto iter = host_compute_sends_.find(key);
792 if (iter == host_compute_sends_.end()) {
793 return errors::InvalidArgument(
794 "No host compute send shapes registered for key ", key);
795 }
796 shapes->clear();
797 for (int i = 0; i < iter->second.metadata_size(); ++i) {
798 TensorShape shape(iter->second.metadata(i).shape());
799 shapes->push_back(shape);
800 }
801 return Status::OK();
802}

812 SetTransfer(key, types, shapes, &transfer);
813 return Status::OK();
