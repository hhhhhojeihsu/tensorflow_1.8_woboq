36// The XlaCompiler class is responsible for compilation of a self-contained
37// subgraph of a TensorFlow computation using the XLA linear algebra runtime.
38// It does a symbolic execution of the graph starting from specific input
39// shapes, using a JIT device to convert operators into XLA computations.
40//
41// XlaCompiler is typically invoked from an `_XlaLaunch` operator once the
42// shapes of all input parameters to the computation are known. This is
43// because the symbolic execution requires known shapes for all operations.
44//
45// XlaCompiler compiles Tensorflow graphs that received inputs via _Arg nodes,
46// and return outputs via _Retval nodes.
47//
48// The XlaCompiler requires one Argument struct for each _Arg index, that
49// describes each argument. Arguments can be compile-time constants
50// (kind kConstant), run-time parameters (kind kParameter), or resources
51// (kind kResource).
52//
53// Only kParameter and initialized kResource arguments become runtime parameters
54// to the generated XLA computation. The XLA computation will have run-time
55// parameters in the following order:
56// +---------------------+-----------------------------------------+
57// | kParameter values | Initial values of kResource arguments |
58// +---------------------+-----------------------------------------+
59// Within each block, the arguments are arranged by the _Arg index from which
60// they were derived.
61//
62// The run-time outputs of the XLA computation are arranged in the following
63// order:
64// +------------------+-----------------------------------------+
65// | _Retval values | Updated values of kResource arguments |
66// +------------------+-----------------------------------------+
67// _Retval values are ordered by _Retval index, whereas kResource values are
68// ordered by the original _Arg position of the variable.
69//
70// In both inputs and outputs, kResource values are placed the end. When
71// emitting While loop bodies, we must ensure that the loop body has
72// identical input and output signatures. By moving variable values
73// to the end of the argument list and using the
74// `return_updated_values_for_all_variables` option, we can ensure that the
75// input and output values of resources appear at the same positions.
76//
77// Resources are passed as parameters or returned as resource updates in
78// "packed" form.
79// kStack resources are packed as (array, size of stack) XLA tuples.
80// kTensorArray resources without gradients are packed as the array that
81// backs the TensorArray. If gradients are present (`tensor_array_gradients`),
82// the packed representation is a (array, gradient0, gradient1, ...) tuple,
83// where gradient_k is the value of the k-th gradient in the
84// `tensor_array_gradients` ordered set.
85class XlaCompiler {
86 public:
87 // Describes how to derive the value of each _Arg node in the graph/function
88 // being compiled. There must be one Argument for each _Arg index.
89 struct Argument {
90 enum Kind {
91 // Default value; not a valid kind.
92 kInvalid,
93
94 // Argument is a compile-time constant. No associated runtime parameter.
95 kConstant,
96
97 // Argument is a Variable, TensorArray, or Stack resource. Has an
98 // associated runtime parameter iff `initialized` is true.
99 kResource,
100
101 // Argument is a run-time parameter.
102 kParameter,
103 };
104
105 Kind kind = kInvalid;
106
107 // The type of the argument. If the argument is a resource, this
108 // is the type of the variable's value, not DT_RESOURCE.
109 DataType type;
110
111 // The shape of the argument. For:
112 // * a parameter: the shape of the parameter.
113 // * a constant: ignored; the shape given by constant_value is used
114 // instead.
115 // * an uninitialized resource: ignored. We don't yet know the shape of an
116 // uninitialized resource (otherwise we would have initialized it!)
117 // * an initialized variable: the shape of the variable's value.
118 // * an initialized TensorArray or Stack resource: the shape of an entry in
119 // the TensorArray/Stack. Note this is the size of a single entry, not the
120 // XLA data structure that represents the complete stack/array.
121 TensorShape shape;
122
123 // The value of the argument, if it is a compile-time constant. Must be a
124 // host-memory tensor.
125 Tensor constant_value;
126
127 // The name of this argument, used for debugging.
128 string name;
129
130 // For a kResource, what kind of resource is it?
131 XlaResource::Kind resource_kind = XlaResource::kInvalid;
132
133 // For a kResource, has this resource been initialized?
134 bool initialized = false;
135
136 // For a TensorArray or Stack resource, what is the array's declared size?
137 // (Used for lazy initialization.)
138 int64 tensor_array_size = -1;
139
140 // TensorArray resource parameters are passed as (array, gradient array 0,
141 // ..., gradient array k), where the gradient arrays are in the same order
142 // as `tensor_array_gradients`.
143 std::set<string> tensor_array_gradients;
144
145 bool operator==(const Argument& other) const;
146 };
147
148 // Options pertaining to an individual call to CompileGraph() or
149 // CompileFunction().
150 struct CompileOptions {
151 // If `use_tuple_arg` is true, a single tuple parameter will be used for all
152 // arguments; if false, each argument gets its own parameter.
153 bool use_tuple_arg = false;
154
155 // If 'return_updated_values_for_all_resources' is true, then updated
156 // values of all resource arguments will be included in the
157 // 'resource_updates' of the computation, even if the resource was not
158 // modified by the computation. Used when compiling loop bodies to ensure
159 // the input and output signatures match.
160 bool return_updated_values_for_all_resources = false;
161
162 // If 'resolve_compile_time_constants' is true, then outputs of a
163 // computation that are known to be compile-time constants will be returned
164 // as Tensors at compile-time, rather than as run-time outputs of the
165 // computation.
166 bool resolve_compile_time_constants = true;
167
168 // True when compiling the entry computation, false for subcomputations
169 // (while, call, etc.)
170 bool is_entry_computation = true;
171 };
172
173 struct OutputDescription {
174 // Type and shape of the output.
175 DataType type;
176 TensorShape shape;
177
178 // Constant output value, if known to be constant at JIT compilation time.
179 // 'Tensor' is in host memory.
180 bool is_constant = false;
181 Tensor constant_value;
182 };
183
184 // Describes a variable write side effect of the computation.
185 struct ResourceUpdate {
186 // Index of the input that contains the variable resource to write to.
187 int input_index;
188
189 // Type and shape of the tensor to be written back.
190 // The `shape` field has the same meaning as the Argument::shape field.
191 DataType type;
192 TensorShape shape;
193
194 // Was the value of the variable modified by the computation?
195 // (Always true, unless `return_updated_values_for_all_resources` is true.)
196 bool modified;
197
198 // If the resource is a TensorArray, the set of gradients read or written.
199 std::set<string> tensor_array_gradients_accessed;
200 };
201
202 struct CompilationResult {
203 // Vector that maps from the parameters of the XLA computation to their
204 // original argument positions. To handle compile-time constant inputs and
205 // resources, the parameters to the XLA computation may be a subset of the
206 // original arguments, and are not necessarily in the same order.)
207 std::vector<int> input_mapping;
208
209 // Input shapes of the computation.
210 std::vector<xla::Shape> xla_input_shapes;
211
212 // Output shape in XLA format. The output shape is always a tuple.
213 xla::Shape xla_output_shape;
214
215 // TensorFlow shapes of outputs, together with the values of any
216 // constant arguments. Vector indexed by Tensorflow _Retval number,
217 // containing both constant and non-constant results.
218 std::vector<OutputDescription> outputs;
219
220 // TensorFlow shapes and types of sends/recvs from HostCompute Ops to their
221 // matching RecvAtHost/SendFromHost Ops in the outer graph.
222 tf2xla::HostComputeMetadata host_compute_metadata;
223
224 // Resources whose values were updated by the computation, ordered
225 // by return value position. Resource updates follow the non-constant
226 // results in the outputs of XLA computation.
227 std::vector<ResourceUpdate> resource_updates;
228
229 // The XLA computation built from the tensorflow subgraph.
230 std::shared_ptr<xla::Computation> computation;
231 };

233 struct Options {
234 // Name of the compilation device to use. Needs to be live only during
235 // XlaCompiler's constructor.
236 const DeviceType* device_type = nullptr;
237
238 xla::Client* client = nullptr;
239
240 // Function library in which to find function definitions. Must be non-null.
241 const FunctionLibraryDefinition* flib_def = nullptr;
242
243 // The graph def version to be compiled.
244 int graph_def_version = TF_GRAPH_DEF_VERSION;
245
246 // If 'allow_cpu_custom_calls' is true, kernels may make use of CustomCall()
247 // for CPU.
248 bool allow_cpu_custom_calls = false;
249
250 // If set, the XLA representation of variables represented to XLA as the
251 // shape given by this shape function. Variables are reshaped to this shape
252 // on write, and reshaped to their original shape on read.
253 std::function<TensorShape(const TensorShape&, DataType)>
254 variable_representation_shape_fn;
255
256 // If not nullptr, populate_resource_manager is called with the
257 // compilation device's resource manager when the compilation
258 // device is created, and can be used to create metadata objects
259 // that can be accessed by XLA op kernels.
260 std::function<Status(ResourceMgr*)>* populate_resource_manager = nullptr;
261
262 // If not nullptr, this memory allocator can be used by the compiler for
263 // temporary allocations it might want to make during compilation.
264 //
265 // For example, the compiler may want to try out different algorithms and
266 // choose the fastest one, and it might run those algorithms over buffers
267 // created using this allocator.
268 //
269 // The compiler can function correctly without an explicit allocator given
270 // here, but on some devices (notably, GPUs), TensorFlow tends to eagerly
271 // allocate most or all available memory on the device, leaving none for the
272 // compiler to access, unless it can use TensorFlow's allocator.
273 xla::DeviceMemoryAllocator* device_allocator = nullptr;
274 };
