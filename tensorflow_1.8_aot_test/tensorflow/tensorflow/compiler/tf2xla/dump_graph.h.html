16// Helper functions for dumping Graphs, GraphDefs, and FunctionDefs to files for
17// debugging.
18
19#ifndef TENSORFLOW_COMPILER_TF2XLA_DUMP_GRAPH_H_
20#define TENSORFLOW_COMPILER_TF2XLA_DUMP_GRAPH_H_
21
22#include "tensorflow/core/framework/function.h"
23#include "tensorflow/core/framework/graph.pb.h"
24#include "tensorflow/core/graph/graph.h"
25
26namespace tensorflow {
27namespace dump_graph {
28
29// Dumps 'graph_def' to a file, as a GraphDef text proto. Returns the file name
30// chosen.
31//
32// Automatically picks a file name. Prefixes 'name' with the value of the
33// --tf_dump_graph_prefix flag and suffixes it with ".pbtxt" to form a name.
34// If a graph has already been dumped by this process with the same name,
35// suffixes with "_n.pbtxt", where 'n' is a sequence number.
36string DumpGraphDefToFile(const string& name, GraphDef const& graph_def);
37
38// Similar to DumpGraphDefToFile, but builds the GraphDef to dump from a 'graph'
39// and an optional function library 'flib_def'. Returns the file name chosen.
40string DumpGraphToFile(const string& name, Graph const& graph,
41 const FunctionLibraryDefinition* flib_def = nullptr);
42
43// Similar to DumpGraphDefToFile, but dumps a function as a FunctionDef text
44// proto. Returns the file name chosen.
45string DumpFunctionDefToFile(const string& name, FunctionDef const& fdef);
46
47} // namespace dump_graph
48} // namespace tensorflow
49
50#endif // TENSORFLOW_COMPILER_TF2XLA_DUMP_GRAPH_H_
