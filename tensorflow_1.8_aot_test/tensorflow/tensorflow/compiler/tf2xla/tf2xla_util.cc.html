41Status ValidateTensorId(const tf2xla::TensorId& id) {
42 if (id.node_name().empty()) {
43 return errors::InvalidArgument("TensorId node_name must be non-empty");
44 }
45 if (id.output_index() < 0) {
46 return errors::InvalidArgument("TensorId output_index must be positive");
47 }
48 return Status::OK();
49}

51Status CheckNameDuplicates(const string& kind, const string& name,
52 std::set<string>* names) {
53 if (!name.empty()) {
54 if (!names->insert(name).second) {
55 return errors::InvalidArgument("duplicate ", kind, " name: ", name);
56 }
57 }
58 return Status::OK();
59}

61Status CheckFeedFetchNameConflicts(const string& kind,
62 const std::set<string>& names) {
63 // We don't allow the feeds or fetches to contain both "foo" and "foo_data",
64 // since that will cause a collision in codegen symbols.
65 for (const string& name : names) {
66 const string name_data(name + "_data");
67 if (names.find(name_data) != names.end()) {
68 return errors::InvalidArgument("conflicting ", kind, " name: ", name,
69 " and ", name_data);
70 }
71 }
72 return Status::OK();
73}

77Status ValidateConfig(const tf2xla::Config& config) {
78 std::set<string> names;
79 for (const tf2xla::Feed& feed : config.feed()) {
80 TF_RETURN_IF_ERROR(ValidateTensorId(feed.id()));
81 TF_RETURN_IF_ERROR(TensorShape::IsValidShape(feed.shape()));
82 TF_RETURN_IF_ERROR(CheckNameDuplicates("feed", feed.name(), &names));
83 }
84 TF_RETURN_IF_ERROR(CheckFeedFetchNameConflicts("feed", names));
85 names.clear();
86 for (const tf2xla::Fetch& fetch : config.fetch()) {
87 TF_RETURN_IF_ERROR(ValidateTensorId(fetch.id()));
88 TF_RETURN_IF_ERROR(CheckNameDuplicates("fetch", fetch.name(), &names));
89 }
90 TF_RETURN_IF_ERROR(CheckFeedFetchNameConflicts("fetch", names));
91 if (config.fetch().empty()) {
92 return errors::InvalidArgument("fetches must be specified");
93 }
94 return Status::OK();
95}

97Status AddPlaceholdersForFeeds(
98 const tf2xla::Config& config, const OpRegistryInterface* op_registry,
99 std::unordered_map<string, string>* feed_remapping, GraphDef* graph_def) {
100 struct PlaceholderInfo {
101 const tf2xla::Feed* feed = nullptr; // point to Feed in <config>.
102 string placeholder_name;
103 DataType data_type = DT_INVALID;
104 };
105
106 // Put each fed tensor into a map by name:port. A map is used for determinism
107 // when creating placeholders (genrules want deterministic output).
108 std::map<string, PlaceholderInfo> placeholder_info;
109 for (int i = 0; i < config.feed_size(); ++i) {
110 const tf2xla::Feed* feed = &config.feed(i);
111 const string name_port = TensorIdToString(feed->id());
112 PlaceholderInfo& info = placeholder_info[name_port];
113 info.feed = feed;
114 info.placeholder_name = strings::StrCat(
115 "aot_feed_", feed->id().output_index(), "/", feed->id().node_name());
116 (*feed_remapping)[name_port] = info.placeholder_name;
117 }
118
119 // Verify node exists and determine data type.
120 std::unordered_map<string, const NodeDef*> name_to_node;
121 for (int i = 0; i < graph_def->node_size(); ++i) {
122 name_to_node[graph_def->node(i).name()] = &graph_def->node(i);
123 }
124 for (auto it = placeholder_info.begin(); it != placeholder_info.end(); ++it) {
125 PlaceholderInfo& info = it->second;
126 const tf2xla::TensorId& feed_id = info.feed->id();
127
128 // Find the existing node and determine data type.
129 auto node_it = name_to_node.find(feed_id.node_name());
130 if (node_it == name_to_node.end()) {
131 return errors::NotFound("Can't find feed node: ",
132 TensorIdToString(feed_id));
133 }
134 const NodeDef* existing = node_it->second;
135
136 if (info.feed->type() != DT_INVALID) {
137 info.data_type = info.feed->type();
138 } else {
139 // Build the node in order to infer its type.
140
141 // Must first add default attrs as well, so do this in a copied GraphDef.
142 GraphDef gd;
143 *gd.mutable_versions() = graph_def->versions();
144 *gd.add_node() = *existing;
145 TF_RETURN_IF_ERROR(
146 AddDefaultAttrsToGraphDef(&gd, *op_registry, 0 /*node_offset*/));
147
148 // Now build the node from the copied node def.
149 Graph g(op_registry);
150 g.set_versions(graph_def->versions());
151 Status status;
152 Node* feed_node = g.AddNode(gd.node(0), &status);
153 TF_RETURN_IF_ERROR(status);
154
155 if (info.feed->id().output_index() < feed_node->num_outputs()) {
156 info.data_type =
157 BaseType(feed_node->output_type(info.feed->id().output_index()));
158 } else {
159 return errors::InvalidArgument(
160 "Invalid output_index ", info.feed->id().output_index(),
161 " for feed node ", info.feed->id().node_name());
162 }
163 }
164 }
165
166 // Create placeholders. Note that we could avoid creating a placeholder for
167 // feeds which are already placeholders, but we omit that to avoid more cases
168 // in this code.
169 for (auto it = placeholder_info.begin(); it != placeholder_info.end(); ++it) {
170 const PlaceholderInfo& info = it->second;
171 NodeDef* d = graph_def->add_node();
172 d->set_name(info.placeholder_name);
173 d->set_op("PlaceholderV2");
174 auto& attr_map = *d->mutable_attr();
175 attr_map["dtype"].set_type(info.data_type);
176 *attr_map["shape"].mutable_shape() = info.feed->shape();
177 }
178
179 // Rewrite references to the fed tensors to refer to the placeholder.
180 for (int i = 0; i < graph_def->node_size(); ++i) {
181 NodeDef* node_def = graph_def->mutable_node(i);
182 for (int j = 0; j < node_def->input_size(); ++j) {
183 auto id = ParseTensorName(node_def->input(j));
184 auto it = placeholder_info.find(id.ToString());
185 if (it != placeholder_info.end()) {
186 node_def->set_input(j, it->second.placeholder_name);
187 }
188 }
189 }
190
191 return Status::OK();
192}

194Status PruneGraphDefInto(const tf2xla::Config& config, const GraphDef& in,
195 GraphDef* out) {
196 *out = in;
197 out->clear_node();
198
199 // Tensors needed for feeding.
200 std::set<std::pair<string, int>> feed_tensors;
201 for (const tf2xla::Feed& feed : config.feed()) {
202 feed_tensors.insert(
203 std::make_pair(feed.id().node_name(), feed.id().output_index()));
204 }
205
206 // Maps node name to reachability.
207 std::unordered_map<string, std::pair<bool, const NodeDef*>> node_by_name;
208 for (const NodeDef& node : in.node()) {
209 node_by_name[node.name()] = std::pair<bool, const NodeDef*>(false, &node);
210 }
211
212 // Traverse.
213 std::queue<string> name_queue;
214 for (int i = 0; i < config.fetch_size(); ++i) {
215 name_queue.push(config.fetch(i).id().node_name());
216 }
217 while (!name_queue.empty()) {
218 const string name = name_queue.front();
219 name_queue.pop();
220
221 auto find_it = node_by_name.find(name);
222 if (find_it == node_by_name.end()) {
223 return errors::InvalidArgument("While pruning graph, node ", name,
224 " needed but not found in the graph.");
225 }
226 auto& map_entry = find_it->second;
227 if (map_entry.first) {
228 continue;
229 }
230 map_entry.first = true;
231
232 // Push input nodes of the currently visited node to name_queue.
233 for (const string& in_edge : map_entry.second->input()) {
234 auto id = ParseTensorName(in_edge);
235 const string node_name = id.first.ToString();
236 if (feed_tensors.find(std::make_pair(node_name, id.second)) ==
237 feed_tensors.end()) {
238 name_queue.push(node_name);
239 } else {
240 // The input tensor is from an edge that is being fed. Therefore,
241 // we skip recursing down that edge, to avoid requiring nodes that
242 // may not be needed (note that the input node may still be added
243 // to name_queue later if one of its output edges is not being fed).
244 }
245 }
246 }
247
248 // Copy over, preserving order of original and only nodes that are reachable
249 // from the fetches.
250 out->mutable_node()->Reserve(in.node_size());
251 for (const NodeDef& node : in.node()) {
252 if (node_by_name[node.name()].first) {
253 *out->add_node() = node;
254 }
255 }
256 return Status::OK();
257}

260 return strings::StrCat(id.node_name(), ":", id.output_index());

263Status SetNodeShardingFromNeighbors(Node* n, bool out_edges) {
264 int core = -1;
265 const Node* matching_node = nullptr;
266 for (const Edge* edge : (out_edges ? n->out_edges() : n->in_edges())) {
267 if (edge->IsControlEdge()) continue;
268 const Node* possible_match = out_edges ? edge->dst() : edge->src();
269 TF_ASSIGN_OR_RETURN(
270 tensorflow::gtl::optional<xla::OpSharding> sharding,
271 ParseShardingFromDevice(
272 *possible_match,
273 /*num_cores_per_replica=*/std::numeric_limits<int32>::max()));
274 if (sharding.has_value()) {
275 TF_RET_CHECK(sharding.value().type() ==
276 xla::OpSharding::Type::OpSharding_Type_MAXIMAL);
277 const int core_annotation = sharding.value().tile_assignment_devices(0);
278 if (core == -1 || core > core_annotation) {
279 core = core_annotation;
280 matching_node = possible_match;
281 }
282 }
283 }
284 if (matching_node != nullptr) {
285 n->set_assigned_device_name(matching_node->assigned_device_name());
286 n->set_requested_device(matching_node->requested_device());
287 }
288 return Status::OK();
289}
