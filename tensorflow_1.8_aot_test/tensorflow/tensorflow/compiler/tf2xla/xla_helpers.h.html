16// This file defines helper routines for the XLA device.
17
18#ifndef TENSORFLOW_COMPILER_TF2XLA_XLA_HELPERS_H_
19#define TENSORFLOW_COMPILER_TF2XLA_XLA_HELPERS_H_
20
21#include "tensorflow/compiler/tf2xla/xla_context.h"
22#include "tensorflow/compiler/xla/client/computation_builder.h"
23#include "tensorflow/core/framework/tensor.h"
24#include "tensorflow/core/lib/gtl/array_slice.h"
25
26namespace tensorflow {
27
28// Helper methods for building XLA computations.
29class XlaHelpers {
30 public:
31 // Returns a handle representing the minimum value of a scalar
32 // element of data_type.
33 static xla::ComputationDataHandle MinValue(xla::ComputationBuilder* b,
34 DataType data_type);
35
36 // Returns a handle representing the maximum value of a scalar
37 // element of data_type.
38 static xla::ComputationDataHandle MaxValue(xla::ComputationBuilder* b,
39 DataType data_type);
40
41 // Returns a handle representing the zero value of a scalar
42 // element of data_type.
43 static xla::ComputationDataHandle Zero(xla::ComputationBuilder* b,
44 DataType data_type);
45
46 // Returns a handle representing the one value of a scalar
47 // element of data_type.
48 static xla::ComputationDataHandle One(xla::ComputationBuilder* b,
49 DataType data_type);
50
51 // Returns the machine epsilon for floating-point type `data_type`, i.e.,
52 // the difference between 1.0 and the next representable value.
53 static xla::ComputationDataHandle Epsilon(xla::ComputationBuilder* b,
54 DataType data_type);
55
56 // Returns a handle representing the given value of an integer scalar
57 // element of data_type.
58 // Note that unlike One and Zero, does not work on boolean types.
59 static xla::ComputationDataHandle IntegerLiteral(xla::ComputationBuilder* b,
60 DataType data_type,
61 int64 value);
62
63 // Returns a handle representing the given value of a floating-point scalar
64 // element of data_type.
65 static xla::ComputationDataHandle FloatLiteral(xla::ComputationBuilder* b,
66 DataType data_type,
67 double value);
68
69 // Reshapes literal 'input' to have 'shape'. Both the original shape and
70 // 'shape' must contain the same number of elements.
71 static Status ReshapeLiteral(const xla::Literal& input,
72 gtl::ArraySlice<int64> shape,
73 xla::Literal* output);
74
75 // Sets `argmax` to the argmax of `input` along `axis`. `input_shape` and
76 // `input_dtype` are the shape and dtype of `input` respectively, and
77 // `output_type` is the dtype to use for `argmax`.
78 static Status ArgMax(xla::ComputationBuilder* builder,
79 XlaOpKernelContext* ctx,
80 const xla::ComputationDataHandle& input,
81 const TensorShape& input_shape, DataType input_type,
82 DataType output_type, int axis,
83 xla::ComputationDataHandle* argmax);
84
85 // Sets `argmin` to the argmin of `input` along `axis`. `input_shape` and
86 // `input_dtype` are the shape and dtype of `input` respectively, and
87 // `output_type` is the dtype to use for `argmin`.
88 static Status ArgMin(xla::ComputationBuilder* builder,
89 XlaOpKernelContext* ctx,
90 const xla::ComputationDataHandle& input,
91 const TensorShape& input_shape, DataType input_type,
92 DataType output_type, int axis,
93 xla::ComputationDataHandle* argmin);
94
95 // Sets *iota to a rank 1 tensor with values [0, 1, 2, ...] of `dtype`.
96 static Status Iota(xla::ComputationBuilder* builder, DataType dtype,
97 int64 size, xla::ComputationDataHandle* iota);
98
99 // Converts `indices` into a one-hot representation. `depth` is the size
100 // of the new axis to add. `axis` is the position at which to add the new
101 // axis. `indices_shape` is the shape of `indices`. `on_value` and
102 // `off_value` represent the values to use for the on and off positions,
103 // respectively.
104 static Status OneHot(xla::ComputationBuilder* builder, int64 depth, int axis,
105 DataType index_type, const TensorShape& indices_shape,
106 const xla::ComputationDataHandle& indices,
107 const xla::ComputationDataHandle& on_value,
108 const xla::ComputationDataHandle& off_value,
109 xla::ComputationDataHandle* one_hot);
110
111 // Certain DataTypes should use increased precision DataTypes when performing
112 // reductions. This function remaps a given DataType to a higher precision
113 // DataType if needed.
114 static DataType SumAccumulationType(const DataType& dtype);
115
116 // A helper for creating a ConvertElementType xla op given a DataType rather
117 // than the xla::PrimitiveType.
118 static xla::ComputationDataHandle ConvertElementType(
119 xla::ComputationBuilder* const builder,
120 const xla::ComputationDataHandle& operand,
121 const DataType new_element_type);
122};
123
124} // end namespace tensorflow
125
126#endif // TENSORFLOW_COMPILER_TF2XLA_XLA_HELPERS_H_
