23XlaCompiledCpuFunction::XlaCompiledCpuFunction(const StaticData& static_data,
24 AllocMode alloc_mode)
25 : raw_function_(static_data.raw_function),
26 result_index_(static_data.result_index),
27 args_(new void*[static_data.num_args]),
28 temps_(new void*[static_data.num_temps]),
29 arg_names_(static_data.arg_names),
30 result_names_(static_data.result_names),
31 program_shape_(static_data.program_shape),
32 hlo_profile_printer_data_(static_data.hlo_profile_printer_data) {
33 // Allocate arg and temp buffers.
34 if (alloc_mode == AllocMode::ARGS_RESULTS_PROFILES_AND_TEMPS) {
35 alloc_args_ = tensorflow::tfcompile::runtime::MallocContiguousBuffers(
36 static_data.arg_sizes, static_data.num_args, args_,
37 /*annotate_initialized=*/false);
38 }
39 alloc_temps_ = tensorflow::tfcompile::runtime::MallocContiguousBuffers(
40 static_data.temp_sizes, static_data.num_temps, temps_,
41 /*annotate_initialized=*/true);
42
43 // If Hlo profiling is enabled the generated code expects an appropriately
44 // sized buffer to be passed in as the last argument. If Hlo profiling is
45 // disabled the last function argument is still present in the function
46 // signature, but it is ignored by the generated code and we pass in null for
47 // it.
48 if (hlo_profiling_enabled()) {
49 profile_counters_ = new int64[static_data.profile_counters_size]();
50 }
51}
52
53XlaCompiledCpuFunction::~XlaCompiledCpuFunction() {
54 tensorflow::tfcompile::runtime::FreeContiguous(alloc_args_);
55 tensorflow::tfcompile::runtime::FreeContiguous(alloc_temps_);
56 delete[] args_;
57 delete[] temps_;
58 delete[] profile_counters_;
59}

63// Linear search through `names` looking for a match with `name`. Returns -1 if
64// the name isn't found, or is empty.
65//
66// REQUIRES: `names` is a nullptr-terminated array.
67int LookupNameIndex(const string& name, const char** names) {
68 // Hitting this assert means that there is no name-to-index data available;
69 // for AOT try the setting the tfcompile --gen_name_to_index flag.
70 assert(names != nullptr);
71
72 constexpr int kNotFound = -1;
73 if (name.empty()) {
74 return kNotFound;
75 }
76 for (int index = 0; names[index] != nullptr; ++index) {
77 if (name == names[index]) {
78 return index;
79 }
80 }
81 return kNotFound;
82}
83
84} // namespace
85
86int XlaCompiledCpuFunction::LookupArgIndex(const string& name) const {
87 return LookupNameIndex(name, arg_names_);
88}
89
90int XlaCompiledCpuFunction::LookupResultIndex(const string& name) const {
91 return LookupNameIndex(name, result_names_);
92}
