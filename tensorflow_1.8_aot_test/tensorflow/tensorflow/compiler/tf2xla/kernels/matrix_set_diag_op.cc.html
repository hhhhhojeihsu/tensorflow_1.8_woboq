16#include "tensorflow/compiler/tf2xla/xla_helpers.h"
17#include "tensorflow/compiler/tf2xla/xla_op_kernel.h"
18#include "tensorflow/compiler/tf2xla/xla_op_registry.h"
19
20namespace tensorflow {
21
22class MatrixSetDiagOp : public XlaOpKernel {
23 public:
24 explicit MatrixSetDiagOp(OpKernelConstruction* context)
25 : XlaOpKernel(context) {}
26
27 void Compile(XlaOpKernelContext* context) override {
28 const TensorShape input_shape = context->InputShape(0);
29 const TensorShape diag_shape = context->InputShape(1);
30
31 const int rank = input_shape.dims();
32
33 // Preliminary validation of sizes.
34 OP_REQUIRES(context, TensorShapeUtils::IsMatrixOrHigher(input_shape),
35 errors::InvalidArgument(
36 "input must be at least 2-dim, received shape: ",
37 input_shape.DebugString()));
38
39 // Check to make sure the last dimension of diag is equal to the smaller of
40 // the last two dimensions of input.
41 const int64 m = input_shape.dim_size(rank - 2);
42 const int64 n = input_shape.dim_size(rank - 1);
43 const int64 min_dim = std::min(m, n);
44
45 TensorShape batch_shape = input_shape;
46 batch_shape.RemoveLastDims(2);
47
48 TensorShape expected_diag_shape = batch_shape;
49 expected_diag_shape.AddDim(min_dim);
50 OP_REQUIRES(context, expected_diag_shape == diag_shape,
51 errors::InvalidArgument(
52 "must have diagonal.shape == input.shape[:-2] + "
53 "min(input.shape[-2:]), but received input shape: ",
54 input_shape.DebugString(),
55 " and diagonal shape: ", diag_shape.DebugString()));
56
57 xla::ComputationBuilder* builder = context->builder();
58 xla::ComputationDataHandle input = context->Input(0);
59 xla::ComputationDataHandle diag = context->Input(1);
60
61 auto zero = XlaHelpers::Zero(builder, context->input_type(0));
62
63 // Create an indicator tensor that is true only on the diagonal.
64 xla::ComputationDataHandle iota_m;
65 OP_REQUIRES_OK(context, XlaHelpers::Iota(builder, DT_INT32, m, &iota_m));
66 xla::ComputationDataHandle iota_n;
67 OP_REQUIRES_OK(context, XlaHelpers::Iota(builder, DT_INT32, n, &iota_n));
68 auto indicator = builder->Eq(iota_m,
69 builder->Broadcast(iota_n, {m}),
70 /*broadcast_dimensions=*/{0});
71 indicator = builder->Broadcast(indicator, batch_shape.dim_sizes());
72
73 // Broadcast diag up to the input shape. Use an implicit broadcast (Add)
74 // because we need to broadcast on the right.
75 std::vector<int64> diag_broadcast_dims(rank - 1);
76 std::iota(diag_broadcast_dims.begin(), diag_broadcast_dims.end(), 0);
77 if (min_dim != m) {
78 diag_broadcast_dims.back() = rank - 1;
79 }
80 diag = builder->Add(diag, builder->Broadcast(zero, input_shape.dim_sizes()),
81 /*broadcast_dimensions=*/diag_broadcast_dims);
82
83 auto output = builder->Select(indicator, diag, input);
84 context->SetOutput(0, output);
85 }
86
87 private:
88 TF_DISALLOW_COPY_AND_ASSIGN(MatrixSetDiagOp);
89};
90
91REGISTER_XLA_OP(Name("MatrixSetDiag"), MatrixSetDiagOp);
92
93} // namespace tensorflow
