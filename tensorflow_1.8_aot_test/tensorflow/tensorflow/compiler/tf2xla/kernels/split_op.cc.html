31class SplitOp : public XlaOpKernel {
32 public:
33 explicit SplitOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {}
34
35 void Compile(XlaOpKernelContext* ctx) override {
36 const int32 num_split = num_outputs();
37 const TensorShape index_shape = ctx->InputShape(0);
38 const TensorShape input_shape = ctx->InputShape(1);
39
40 xla::Literal literal_index;
41 OP_REQUIRES_OK(ctx, ctx->ConstantInput(0, &literal_index));
42
43 int32 split_dim_orig;
44 if (index_shape.dims() == 0) {
45 split_dim_orig = literal_index.Get<int>({});
46 } else {
47 OP_REQUIRES(
48 ctx, index_shape.dims() == 1,
49 errors::InvalidArgument("split_index input to Split Op must be a "
50 "scalar or a vector with 1 element"));
51 OP_REQUIRES(
52 ctx, index_shape.dim_size(0) == 1,
53 errors::InvalidArgument("split_index input to Split Op must be a "
54 "scalar or a vector with 1 element"));
55 split_dim_orig = literal_index.Get<int>({0});
56 }
57 int32 split_dim = split_dim_orig < 0 ? split_dim_orig + input_shape.dims()
58 : split_dim_orig;
59 OP_REQUIRES(ctx, 0 <= split_dim && split_dim < input_shape.dims(),
60 errors::InvalidArgument("-input rank(-", input_shape.dims(),
61 ") <= split_dim < input rank (",
62 input_shape.dims(), "), but got ",
63 split_dim_orig));
64
65 OP_REQUIRES(
66 ctx, num_split > 0,
67 errors::InvalidArgument(
68 "Number of ways to split should be > 0, but got ", num_split));
69
70 OP_REQUIRES(
71 ctx, input_shape.dim_size(split_dim) % num_split == 0,
72 errors::InvalidArgument(
73 "Number of ways to split should evenly divide the split "
74 "dimension, but got split_dim ",
75 split_dim_orig, " (size = ", input_shape.dim_size(split_dim), ") ",
76 "and num_split ", num_split));
77
78 // All the slices are the same size: this is the size along the
79 // split dimension.
80 const int32 slice_size = input_shape.dim_size(split_dim) / num_split;
81
82 // The vectors we will use to define the slice. The entry for the
83 // split dimensions varies for each output.
84 std::vector<int64> begin(input_shape.dims(), 0);
85 std::vector<int64> limits(input_shape.dims());
86 std::vector<int64> strides(input_shape.dims(), 1);
87 for (int i = 0; i < input_shape.dims(); ++i) {
88 // Initially set up the limits to be the full size of the input:
89 // the split dimension is filled in below.
90 int64 dim = input_shape.dim_size(i);
91 limits[i] = dim;
92 }
93
94 auto input = ctx->Input(1);
95
96 // Create each of the outputs.
97 for (int i = 0; i < num_split; ++i) {
98 // Slice out the ith split from the split dimension.
99 begin[split_dim] = i * slice_size;
100 limits[split_dim] = (i + 1) * slice_size;
101 ctx->SetOutput(i, ctx->builder()->Slice(input, begin, limits, strides));
102 }
103 }
104};
105
106REGISTER_XLA_OP(Name("Split").CompileTimeConstInput("split_dim"), SplitOp);

108class SplitVOp : public XlaOpKernel {
109 public:
110 explicit SplitVOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {}
111
112 void Compile(XlaOpKernelContext* ctx) override {
113 const int32 num_split = num_outputs();
114 const TensorShape input_shape = ctx->InputShape(0);
115 const TensorShape index_shape = ctx->InputShape(2);
116
117 int64 split_dim_orig;
118 OP_REQUIRES_OK(ctx, ctx->ConstantInputAsIntScalar(2, &split_dim_orig));
119 int64 split_dim = split_dim_orig < 0 ? split_dim_orig + input_shape.dims()
120 : split_dim_orig;
121 OP_REQUIRES(ctx, 0 <= split_dim && split_dim < input_shape.dims(),
122 errors::InvalidArgument("-input rank(-", input_shape.dims(),
123 ") <= split_dim < input rank (",
124 input_shape.dims(), "), but got ",
125 split_dim_orig));
126
127 xla::ComputationDataHandle input = ctx->Input(0);
128
129 OP_REQUIRES(ctx, input_shape.dims() > 0,
130 errors::InvalidArgument("Can't split a 0 dimensional input"));
131
132 OP_REQUIRES(
133 ctx, num_split > 0,
134 errors::InvalidArgument(
135 "Number of ways to split should be > 0, but got ", num_split));
136
137 // check that sizes are correct
138 int total_split_size = 0;
139 int neg_one_dim = -1;
140 std::vector<int64> split_sizes_vec(num_split, -1);
141 const TensorShape split_size_shape = ctx->InputShape(1);
142 OP_REQUIRES(ctx,
143 split_size_shape.dims() == 1 &&
144 split_size_shape.num_elements() == num_split,
145 errors::InvalidArgument(
146 "shape of tensor describing "
147 " the output must have dimension 1 and the same "
148 " number of elements as the output. Got ",
149 split_size_shape.dims(), "-D and ",
150 split_size_shape.num_elements(), " elements"));
151 // get the dimension of this split
152 xla::Literal split_size_literal;
153 OP_REQUIRES_OK(ctx, ctx->ConstantInput(1, &split_size_literal));
154
155 for (int i = 0; i < num_split; ++i) {
156 int slice_size;
157 slice_size = split_size_literal.Get<int>({i});
158 if (slice_size == -1) {
159 OP_REQUIRES(
160 ctx, neg_one_dim == -1,
161 errors::InvalidArgument("Only one dimensions can have a value of"
162 "-1. Second one found at dimension ",
163 i));
164 neg_one_dim = i;
165 } else {
166 split_sizes_vec[i] = slice_size;
167 total_split_size += slice_size;
168 }
169 }
170
171 OP_REQUIRES(
172 ctx,
173 (neg_one_dim == -1 &&
174 total_split_size == input_shape.dim_size(split_dim)) ||
175 (neg_one_dim >= 0 &&
176 total_split_size <= input_shape.dim_size(split_dim)),
177 errors::InvalidArgument("Determined shape must either match "
178 "input shape along split_dim exactly if "
179 "fully specified, or be less than the size of "
180 "the input along split_dim if not fully "
181 "specified. Got: ",
182 total_split_size));
183
184 if (neg_one_dim >= 0) {
185 split_sizes_vec[neg_one_dim] =
186 input_shape.dim_size(split_dim) - total_split_size;
187 }
188
189 // The vectors we will use to define the slice. The entry for the
190 // split dimensions varies for each output.
191 std::vector<int64> begin(input_shape.dims(), 0);
192 auto dim_sizes = input_shape.dim_sizes();
193 std::vector<int64> limits(dim_sizes.begin(), dim_sizes.end());
194 std::vector<int64> strides(input_shape.dims(), 1);
195 for (int i = 0; i < num_split; ++i) {
196 TensorShape output_shape(input_shape);
197 int slice_size = split_sizes_vec[i];
198 output_shape.set_dim(split_dim, slice_size);
199
200 // Slice out the ith split from the split dimension.
201 limits[split_dim] = begin[split_dim] + slice_size;
202 ctx->SetOutput(i, ctx->builder()->Slice(input, begin, limits, strides));
203 begin[split_dim] = limits[split_dim];
204 }
205 }
206};
207
208REGISTER_XLA_OP(Name("SplitV")
209 .CompileTimeConstInput("split_dim")
210 .CompileTimeConstInput("size_splits"),
211 SplitVOp);
