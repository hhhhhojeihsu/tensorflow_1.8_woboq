24// Converts 'input' from RGB format to HSV format.
25// 'shape' is the shape of the red/green/blue tensors.
26std::array<xla::ComputationDataHandle, 3> RGBToHSV(
27 XlaOpKernelContext* ctx, xla::ComputationBuilder* b,
28 const std::array<xla::ComputationDataHandle, 3>& rgb, DataType dtype,
29 const TensorShape& shape) {
30 auto zero = XlaHelpers::Zero(b, dtype);
31 auto one = XlaHelpers::One(b, dtype);
32
33 auto red = rgb[0];
34 auto green = rgb[1];
35 auto blue = rgb[2];
36 auto value = b->Max(b->Max(red, green), blue);
37 auto minimum = b->Min(b->Min(red, green), blue);
38 auto range = b->Sub(value, minimum);
39
40 auto zeros = b->Broadcast(zero, shape.dim_sizes());
41 auto saturation = b->Select(b->Gt(value, zero), b->Div(range, value), zeros);
42
43 auto norm = b->Div(XlaHelpers::FloatLiteral(b, dtype, 1.0 / 6.0), range);
44
45 auto hue = b->Select(b->Eq(green, value),
46 b->Add(b->Mul(norm, b->Sub(blue, red)),
47 XlaHelpers::FloatLiteral(b, dtype, 2.0 / 6.0)),
48 b->Add(b->Mul(norm, b->Sub(red, green)),
49 XlaHelpers::FloatLiteral(b, dtype, 4.0 / 6.0)));
50 hue = b->Select(b->Eq(red, value), b->Mul(norm, b->Sub(green, blue)), hue);
51 hue = b->Select(b->Gt(range, zero), hue, zeros);
52 hue = b->Select(b->Lt(hue, zero), b->Add(hue, one), hue);
53 return {hue, saturation, value};
54}

56// Converts 'input' from HSV format to RGB format.
57std::array<xla::ComputationDataHandle, 3> HSVToRGB(
58 xla::ComputationBuilder* b,
59 const std::array<xla::ComputationDataHandle, 3>& hsv, DataType dtype) {
60 xla::ComputationDataHandle hue = hsv[0];
61 xla::ComputationDataHandle saturation = hsv[1];
62 xla::ComputationDataHandle value = hsv[2];
63 auto zero = XlaHelpers::Zero(b, dtype);
64 auto one = XlaHelpers::FloatLiteral(b, dtype, 1.0);
65 auto two = XlaHelpers::FloatLiteral(b, dtype, 2.0);
66 auto three = XlaHelpers::FloatLiteral(b, dtype, 3.0);
67 auto four = XlaHelpers::FloatLiteral(b, dtype, 4.0);
68 auto six = XlaHelpers::FloatLiteral(b, dtype, 6.0);
69
70 auto dh = b->Mul(hue, six);
71 auto dr = b->Clamp(zero, b->Sub(b->Abs(b->Sub(dh, three)), one), one);
72 auto dg = b->Clamp(zero, b->Sub(two, b->Abs(b->Sub(dh, two))), one);
73 auto db = b->Clamp(zero, b->Sub(two, b->Abs(b->Sub(dh, four))), one);
74 auto one_minus_s = b->Sub(one, saturation);
75
76 auto red = b->Mul(b->Add(one_minus_s, b->Mul(saturation, dr)), value);
77 auto green = b->Mul(b->Add(one_minus_s, b->Mul(saturation, dg)), value);
78 auto blue = b->Mul(b->Add(one_minus_s, b->Mul(saturation, db)), value);
79 return {red, green, blue};
80}

82class RGBToHSVOp : public XlaOpKernel {
83 public:
84 explicit RGBToHSVOp(OpKernelConstruction* context) : XlaOpKernel(context) {}
85
86 void Compile(XlaOpKernelContext* context) override {
87 const TensorShape input_shape = context->InputShape(0);
88 OP_REQUIRES(context, input_shape.dims() >= 1,
89 errors::InvalidArgument("input must be at least 1D",
90 input_shape.DebugString()));
91 int channel_dim = input_shape.dims() - 1;
92 int64 channels = input_shape.dim_size(channel_dim);
93 OP_REQUIRES(
94 context, channels == 3,
95 errors::FailedPrecondition("input must have 3 channels but input has ",
96 channels, " channels."));
97
98 xla::ComputationBuilder* b = context->builder();
99 xla::ComputationDataHandle input = context->Input(0);
100
101 xla::ComputationDataHandle red =
102 b->SliceInDim(input, /*start_index=*/0, /*limit_index=*/1, /*stride=*/1,
103 /*dimno=*/channel_dim);
104 xla::ComputationDataHandle green =
105 b->SliceInDim(input, /*start_index=*/1, /*limit_index=*/2, /*stride=*/1,
106 /*dimno=*/channel_dim);
107 xla::ComputationDataHandle blue =
108 b->SliceInDim(input, /*start_index=*/2, /*limit_index=*/3, /*stride=*/1,
109 /*dimno=*/channel_dim);
110 TensorShape channel_shape = input_shape;
111 channel_shape.set_dim(channel_dim, 1);
112 auto hsv = RGBToHSV(context, b, {red, green, blue}, context->input_type(0),
113 channel_shape);
114
115 context->SetOutput(0, b->ConcatInDim(hsv, channel_dim));
116 }
117};
118REGISTER_XLA_OP(Name("RGBToHSV"), RGBToHSVOp);

156class AdjustContrastOpV2 : public XlaOpKernel {
157 public:
158 explicit AdjustContrastOpV2(OpKernelConstruction* context)
159 : XlaOpKernel(context) {}
160
161 void Compile(XlaOpKernelContext* context) override {
162 const TensorShape& input_shape = context->InputShape(0);
163 const TensorShape& factor_shape = context->InputShape(1);
164 OP_REQUIRES(context, input_shape.dims() >= 3,
165 errors::InvalidArgument("input must be at least 3-D, got shape",
166 input_shape.DebugString()));
167 int height_dim = input_shape.dims() - 3;
168 int width_dim = input_shape.dims() - 2;
169 int channel_dim = input_shape.dims() - 1;
170 const int64 height = input_shape.dim_size(height_dim);
171 const int64 width = input_shape.dim_size(width_dim);
172
173 OP_REQUIRES(context, TensorShapeUtils::IsScalar(factor_shape),
174 errors::InvalidArgument("contrast_factor must be scalar: ",
175 factor_shape.DebugString()));
176
177 xla::ComputationBuilder* b = context->builder();
178 xla::ComputationDataHandle input = context->Input(0);
179 xla::ComputationDataHandle factor = context->Input(1);
180
181 DataType type = context->input_type(0);
182
183 const DataType accumulation_type = XlaHelpers::SumAccumulationType(type);
184 auto converted =
185 XlaHelpers::ConvertElementType(b, input, accumulation_type);
186 auto reduce = b->Reduce(converted, XlaHelpers::Zero(b, accumulation_type),
187 *context->GetOrCreateAdd(accumulation_type),
188 {height_dim, width_dim});
189 auto output = XlaHelpers::ConvertElementType(b, reduce, type);
190 output = b->Div(output, XlaHelpers::FloatLiteral(b, type, height * width));
191
192 std::vector<int64> broadcast_dims(input_shape.dims() - 2);
193 std::iota(broadcast_dims.begin(), broadcast_dims.end(), 0);
194 broadcast_dims.back() = channel_dim;
195 output = b->Add(b->Mul(input, factor),
196 b->Mul(output, b->Sub(XlaHelpers::One(b, type), factor)),
197 broadcast_dims);
198 context->SetOutput(0, output);
199 }
200};
201REGISTER_XLA_OP(Name("AdjustContrastv2"), AdjustContrastOpV2);

203class AdjustSaturationOp : public XlaOpKernel {
204 public:
205 explicit AdjustSaturationOp(OpKernelConstruction* context)
206 : XlaOpKernel(context) {}
207
208 void Compile(XlaOpKernelContext* context) override {
209 const TensorShape& input_shape = context->InputShape(0);
210 const TensorShape& scale_shape = context->InputShape(1);
211 OP_REQUIRES(context, input_shape.dims() >= 3,
212 errors::InvalidArgument("input must be at least 3-D, got shape",
213 input_shape.DebugString()));
214 OP_REQUIRES(context, TensorShapeUtils::IsScalar(scale_shape),
215 errors::InvalidArgument("scale must be scalar: ",
216 scale_shape.DebugString()));
217 const int channel_dim = input_shape.dims() - 1;
218 const int64 channels = input_shape.dim_size(channel_dim);
219 OP_REQUIRES(
220 context, channels == 3,
221 errors::InvalidArgument("input must have 3 channels but instead has ",
222 channels, " channels."));
223
224 xla::ComputationBuilder* b = context->builder();
225 xla::ComputationDataHandle input = context->Input(0);
226 xla::ComputationDataHandle scale = context->Input(1);
227
228 DataType type = context->input_type(0);
229
230 xla::ComputationDataHandle red =
231 b->SliceInDim(input, /*start_index=*/0, /*limit_index=*/1, /*stride=*/1,
232 /*dimno=*/channel_dim);
233 xla::ComputationDataHandle green =
234 b->SliceInDim(input, /*start_index=*/1, /*limit_index=*/2, /*stride=*/1,
235 /*dimno=*/channel_dim);
236 xla::ComputationDataHandle blue =
237 b->SliceInDim(input, /*start_index=*/2, /*limit_index=*/3, /*stride=*/1,
238 /*dimno=*/channel_dim);
239 TensorShape channel_shape = input_shape;
240 channel_shape.set_dim(channel_dim, 1);
241 auto hsv = RGBToHSV(context, b, {red, green, blue}, context->input_type(0),
242 channel_shape);
243
244 hsv[1] = b->Clamp(XlaHelpers::Zero(b, type), b->Mul(hsv[1], scale),
245 XlaHelpers::One(b, type));
246
247 auto rgb = HSVToRGB(context->builder(), hsv, context->input_type(0));
248
249 context->SetOutput(0, b->ConcatInDim(rgb, channel_dim));
250 }
251};
252REGISTER_XLA_OP(Name("AdjustSaturation"), AdjustSaturationOp);

254class AdjustHueOp : public XlaOpKernel {
255 public:
256 explicit AdjustHueOp(OpKernelConstruction* context) : XlaOpKernel(context) {}
257
258 void Compile(XlaOpKernelContext* context) override {
259 const TensorShape& input_shape = context->InputShape(0);
260 const TensorShape& delta_shape = context->InputShape(1);
261 OP_REQUIRES(context, input_shape.dims() >= 3,
262 errors::InvalidArgument("input must be at least 3-D, got shape",
263 input_shape.DebugString()));
264 OP_REQUIRES(context, TensorShapeUtils::IsScalar(delta_shape),
265 errors::InvalidArgument("delta must be scalar: ",
266 delta_shape.DebugString()));
267 const int channel_dim = input_shape.dims() - 1;
268 const int64 channels = input_shape.dim_size(channel_dim);
269 OP_REQUIRES(
270 context, channels == 3,
271 errors::InvalidArgument("input must have 3 channels but instead has ",
272 channels, " channels."));
273
274 xla::ComputationBuilder* b = context->builder();
275 xla::ComputationDataHandle input = context->Input(0);
276 xla::ComputationDataHandle delta = context->Input(1);
277
278 DataType type = context->input_type(0);
279
280 xla::ComputationDataHandle red =
281 b->SliceInDim(input, /*start_index=*/0, /*limit_index=*/1, /*stride=*/1,
282 /*dimno=*/channel_dim);
283 xla::ComputationDataHandle green =
284 b->SliceInDim(input, /*start_index=*/1, /*limit_index=*/2, /*stride=*/1,
285 /*dimno=*/channel_dim);
286 xla::ComputationDataHandle blue =
287 b->SliceInDim(input, /*start_index=*/2, /*limit_index=*/3, /*stride=*/1,
288 /*dimno=*/channel_dim);
289 TensorShape channel_shape = input_shape;
290 channel_shape.set_dim(channel_dim, 1);
291 auto hsv = RGBToHSV(context, b, {red, green, blue}, context->input_type(0),
292 channel_shape);
293
294 auto zero = XlaHelpers::Zero(b, type);
295 auto one = XlaHelpers::One(b, type);
296
297 auto& hue = hsv[0];
298 hue = b->Rem(b->Add(hsv[0], delta), one);
299 hue = b->Select(b->Lt(hue, zero), b->Rem(b->Add(one, hue), one), hue);
300
301 auto rgb = HSVToRGB(context->builder(), hsv, context->input_type(0));
302
303 context->SetOutput(0, b->ConcatInDim(rgb, channel_dim));
304 }
305};
306REGISTER_XLA_OP(Name("AdjustHue"), AdjustHueOp);
