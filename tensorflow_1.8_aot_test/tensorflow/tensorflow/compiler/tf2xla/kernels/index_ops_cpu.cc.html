32// The logic below uses a custom-call to implement argmax.
33//
34// Also see b/29507024 for first-class XLA support for indexing ops.
35class ArgMaxCustomCallOp : public XlaOpKernel {
36 public:
37 explicit ArgMaxCustomCallOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {}
38
39 void Compile(XlaOpKernelContext* ctx) override {
40 const TensorShape input_shape = ctx->InputShape(0);
41 const TensorShape dimension_shape = ctx->InputShape(1);
42 OP_REQUIRES(ctx, TensorShapeUtils::IsScalar(dimension_shape),
43 errors::InvalidArgument(
44 "dim must be a scalar, but received tensor of shape: ",
45 dimension_shape.DebugString()));
46
47 // We require that the dimension argument is a constant, since it lets us
48 // dispatch to a specialized custom-call function without any run-time
49 // overhead, when compiling ahead-of-time.
50 xla::Literal literal;
51 OP_REQUIRES_OK(ctx, ctx->ConstantInput(1, &literal));
52 const int32 dim = literal.Get<int32>({});
53 OP_REQUIRES(ctx, dim >= 0, errors::InvalidArgument("dim must be >= 0"));
54 OP_REQUIRES(
55 ctx, dim < input_shape.dims(),
56 errors::InvalidArgument("dim must be < input rank (",
57 input_shape.dims(), "), but got: ", dim));
58 const int64 dim_size = input_shape.dim_size(dim);
59 OP_REQUIRES(ctx, dim_size > 0,
60 errors::InvalidArgument(
61 "Reduction axis ", dim,
62 " is empty in shape: ", input_shape.DebugString()));
63
64 // The output shape is the input shape contracted along dim.
65 TensorShape output_shape;
66 for (int d = 0; d < input_shape.dims() - 1; ++d) {
67 output_shape.AddDim(input_shape.dim_size((d < dim) ? d : d + 1));
68 }
69
70 // For now we use a custom-call, only for the 1d and 2d cases.
71 OP_REQUIRES(ctx, XlaContext::Get(ctx).allow_cpu_custom_calls(),
72 errors::InvalidArgument(
73 "ArgMax implementation requires a CustomCall on CPU"));
74 xla::ComputationBuilder& b = *ctx->builder();
75
76 // XLA passes <out> to the function, so it is not included here.
77 std::vector<xla::ComputationDataHandle> args;
78 args.push_back(ctx->Input(0));
79 args.push_back(b.ConstantLiteral(
80 *xla::Literal::CreateR1<int64>(input_shape.dim_sizes())));
81 if (input_shape.dims() > 1) {
82 // Don't bother passing the output shape and dim for the 1d case, since
83 // the shape is always a scalar and the dim is always 0.
84 args.push_back(b.ConstantLiteral(
85 *xla::Literal::CreateR1<int64>(output_shape.dim_sizes())));
86 args.push_back(b.ConstantLiteral(*xla::Literal::CreateR0<int32>(dim)));
87 }
88
89 xla::Shape xla_shape =
90 xla::ShapeUtil::MakeShape(xla::S64, output_shape.dim_sizes());
91
92 // Tell XLA to call the custom code, defined in
93 // index_ops_kernel_argmax_float_1d.cc.
94 xla::ComputationDataHandle output;
95 switch (input_shape.dims()) {
96 case 1:
97 output = b.CustomCall("argmax_float_1d_xla_impl", args, xla_shape);
98 break;
99 case 2:
100 output = b.CustomCall("argmax_float_2d_xla_impl", args, xla_shape);
101 break;
102 default:
103 OP_REQUIRES(ctx, false,
104 errors::Unimplemented(
105 "Argmax is only implemented for 1d and 2d tensors"
106 ", but got shape: ",
107 input_shape.DebugString()));
108 }
109 ctx->SetOutput(0, output);
110 }
111
112 private:
113 TF_DISALLOW_COPY_AND_ASSIGN(ArgMaxCustomCallOp);
114};
115
116REGISTER_XLA_OP(Name("ArgMax")
117 .TypeConstraint("T", DT_FLOAT)
118 .Device(DEVICE_CPU_XLA_JIT)
119 .CompileTimeConstInput("dimension"),
120 ArgMaxCustomCallOp);
