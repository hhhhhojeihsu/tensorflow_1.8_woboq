30// A subclass of a XlaBinaryOp must build the computation that
31// describes the (tensor,tensor)->tensor function to apply to each element of
32// the input.
33#define XLA_MAKE_BINARY(NAME, HLO) \
34 class NAME##Op : public XlaBinaryOp { \
35 public: \
36 explicit NAME##Op(OpKernelConstruction* ctx) : XlaBinaryOp(ctx) {} \
37 xla::ComputationDataHandle Computation( \
38 XlaOpKernelContext* ctx, const xla::ComputationDataHandle& lhs, \
39 const gtl::ArraySlice<int64>& lhs_shape, \
40 const xla::ComputationDataHandle& rhs, \
41 const gtl::ArraySlice<int64>& rhs_shape, \
42 const BCast& broadcast_helper, \
43 const std::vector<int64>& extend_dimensions) override { \
44 xla::ComputationBuilder* b = ctx->builder(); \
45 return HLO; \
46 } \
47 }; \
48 REGISTER_XLA_OP(Name(#NAME), NAME##Op)

58// Implementation of FloorDiv. Pseudo-code:
59// if ((x < 0) != (y < 0)) {
60// T abs_x = std::abs(x);
61// T abs_y = std::abs(y);
62// return -(abs_x + abs_y - 1) / abs_y;
63// } else {
64// return x / y;
65// }
66static xla::ComputationDataHandle FloorDivImpl(xla::ComputationBuilder* b,
67 DataType dtype,
68 xla::ComputationDataHandle x,
69 xla::ComputationDataHandle y,
70 const BCast& broadcast_helper) {
71 std::tie(x, y) = XlaBinaryOp::Broadcast(b, x, y, broadcast_helper);
72 auto zero = XlaHelpers::Zero(b, dtype);
73 auto one = XlaHelpers::One(b, dtype);
74 auto different_sign = b->Ne(b->Lt(x, zero), b->Lt(y, zero));
75 auto abs_x = b->Abs(x);
76 auto abs_y = b->Abs(y);
77 auto t = b->Neg(b->Sub(b->Add(abs_x, abs_y), one));
78 auto result = b->Select(different_sign, b->Div(t, abs_y), b->Div(x, y));
79 if (DataTypeIsFloating(dtype)) {
80 result = b->Floor(result);
81 }
82 return result;
83}
84XLA_MAKE_BINARY(FloorDiv,
85 FloorDivImpl(b, input_type(0), lhs, rhs, broadcast_helper));
86
87// Implementation of FloorMod. Pseudo-code:
88// T trunc_mod = std::fmod(x, y);
89// return (x < T(0)) == (y < T(0)) ? trunc_mod : std::fmod(trunc_mod + y, y);
90static xla::ComputationDataHandle FloorModImpl(xla::ComputationBuilder* b,
91 DataType dtype,
92 xla::ComputationDataHandle x,
93 xla::ComputationDataHandle y,
94 const BCast& broadcast_helper) {
95 std::tie(x, y) = XlaBinaryOp::Broadcast(b, x, y, broadcast_helper);
96 auto zero = XlaHelpers::Zero(b, dtype);
97 auto same_sign = b->Eq(b->Lt(x, zero), b->Lt(y, zero));
98 auto trunc_mod = b->Rem(x, y);
99 return b->Select(same_sign, trunc_mod, b->Rem(b->Add(trunc_mod, y), y));
100}
101XLA_MAKE_BINARY(FloorMod,
102 FloorModImpl(b, input_type(0), lhs, rhs, broadcast_helper));

130static xla::ComputationDataHandle Square(xla::ComputationBuilder* builder,
131 const xla::ComputationDataHandle& x) {
132 return builder->Mul(x, x);
133}

149// Non-linear ops
150XLA_MAKE_BINARY(SigmoidGrad,
151 b->Mul(b->Mul(rhs, lhs),
152 b->Sub(XlaHelpers::One(b, input_type(0)), lhs)));
153
154XLA_MAKE_BINARY(SoftplusGrad,
155 b->Div(lhs, b->Add(b->Exp(b->Neg(rhs)),
156 XlaHelpers::One(b, input_type(1)))));
157
158// softsigngrad(gradients, features) = gradients / (1 + abs(features)) ** 2
159XLA_MAKE_BINARY(SoftsignGrad,
160 b->Div(lhs, Square(b, b->Add(XlaHelpers::One(b, input_type(0)),
161 b->Abs(rhs)))));
162
163XLA_MAKE_BINARY(TanhGrad, b->Mul(rhs, b->Sub(XlaHelpers::One(b, input_type(0)),
164 b->Mul(lhs, lhs))));

170class ApproximateEqualOp : public XlaOpKernel {
171 public:
172 explicit ApproximateEqualOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {
173 OP_REQUIRES_OK(ctx, ctx->GetAttr("tolerance", &tolerance_));
174 }
175
176 // Computes the max of the scalar input x and 0.
177 void Compile(XlaOpKernelContext* ctx) override {
178 xla::ComputationBuilder* b = ctx->builder();
179 auto abs = b->Abs(b->Sub(ctx->Input(0), ctx->Input(1)));
180 auto abs_shape = b->GetShape(abs);
181 OP_REQUIRES_OK(ctx, abs_shape.status());
182 auto abs_type = abs_shape.ValueOrDie()->element_type();
183 auto result = b->Lt(
184 abs, b->ConvertElementType(b->ConstantR0<float>(tolerance_), abs_type));
185 ctx->SetOutput(0, result);
186 }
187
188 private:
189 float tolerance_;
190};
191REGISTER_XLA_OP(Name("ApproximateEqual"), ApproximateEqualOp);
