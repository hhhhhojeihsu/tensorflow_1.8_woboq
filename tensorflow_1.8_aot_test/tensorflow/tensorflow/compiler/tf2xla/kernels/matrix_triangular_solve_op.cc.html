16#include "tensorflow/compiler/tf2xla/lib/triangular_solve.h"
17#include "tensorflow/compiler/tf2xla/xla_op_kernel.h"
18#include "tensorflow/compiler/tf2xla/xla_op_registry.h"
19
20namespace tensorflow {
21namespace {
22
23class MatrixTriangularSolveOp : public XlaOpKernel {
24 public:
25 explicit MatrixTriangularSolveOp(OpKernelConstruction* ctx)
26 : XlaOpKernel(ctx) {
27 OP_REQUIRES_OK(ctx, ctx->GetAttr("lower", &lower_));
28 OP_REQUIRES_OK(ctx, ctx->GetAttr("adjoint", &adjoint_));
29 }
30
31 void Compile(XlaOpKernelContext* ctx) override {
32 auto result = TriangularSolve(
33 ctx->builder(), ctx->Input(0), ctx->Input(1), /*left_side=*/true,
34 /*lower=*/lower_, /*transpose_a=*/adjoint_, /*conjugate_a=*/adjoint_);
35 if (!result.ok()) {
36 ctx->SetStatus(result.status());
37 return;
38 }
39 ctx->SetOutput(0, result.ValueOrDie());
40 }
41
42 private:
43 bool lower_;
44 bool adjoint_;
45};
46
47REGISTER_XLA_OP(Name("MatrixTriangularSolve"), MatrixTriangularSolveOp);
