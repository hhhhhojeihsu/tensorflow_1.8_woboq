32class DynamicStitchOp : public XlaOpKernel {
33 public:
34 explicit DynamicStitchOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {
35 OP_REQUIRES(
36 ctx, ctx->num_inputs() > 0,
37 errors::InvalidArgument("DynamicStitchOp: Must have some inputs"));
38 OP_REQUIRES(ctx, ctx->num_inputs() % 2 == 0,
39 errors::InvalidArgument(
40 "DynamicStitchOp: Must have even number of arguments"));
41 // Compute expected input signature
42 const int n = ctx->num_inputs() / 2;
43 const DataType dt = ctx->input_type(n);
44 DataTypeVector expected;
45 for (int i = 0; i < n; i++) {
46 expected.push_back(DT_INT32);
47 }
48 for (int i = 0; i < n; i++) {
49 expected.push_back(dt);
50 }
51 OP_REQUIRES_OK(ctx, ctx->MatchSignature(expected, {dt}));
52 }
53
54 void Compile(XlaOpKernelContext* ctx) override {
55 // Validate that data_shape[i] = indices[i].shape() + constant
56 std::vector<xla::Literal> indices_input;
57 OP_REQUIRES_OK(ctx, ctx->ConstantInputList("indices", &indices_input));
58
59 std::vector<xla::ComputationDataHandle> data;
60 std::vector<TensorShape> data_shapes;
61 OP_REQUIRES_OK(ctx, ctx->InputList("data", &data, &data_shapes));
62
63 std::vector<xla::Literal> indices(indices_input.size());
64
65 const TensorShape& data0_shape = data_shapes[0];
66 TensorShape indices0_shape;
67 OP_REQUIRES_OK(
68 ctx, XLAShapeToTensorShape(indices_input[0].shape(), &indices0_shape));
69 for (int input_num = 0; input_num < indices_input.size(); input_num++) {
70 TensorShape indices_shape;
71 OP_REQUIRES_OK(ctx,
72 XLAShapeToTensorShape(indices_input[input_num].shape(),
73 &indices_shape));
74 const TensorShape& data_shape = data_shapes[input_num];
75 OP_REQUIRES(
76 ctx, TensorShapeUtils::StartsWith(data_shape, indices_shape),
77 errors::InvalidArgument("data[", input_num,
78 "].shape = ", data_shape.DebugString(),
79 " does not start with indices[", input_num,
80 "].shape = ", indices_shape.DebugString()));
81 OP_REQUIRES(
82 ctx,
83 input_num == 0 || SameExtraShape(data0_shape, indices0_shape,
84 data_shape, indices_shape),
85 errors::InvalidArgument(
86 "Need data[0].shape[", indices0_shape.dims(), ":] = data[",
87 input_num, "].shape[", indices_shape.dims(),
88 ":], got data[0].shape = ", data0_shape.DebugString(), ", data[",
89 input_num, "].shape = ", data_shape.DebugString(),
90 ", indices[0].shape = ", indices0_shape.DebugString(),
91 ", indices[", input_num,
92 "].shape = ", indices_shape.DebugString()));
93
94 OP_REQUIRES_OK(ctx,
95 XlaHelpers::ReshapeLiteral(indices_input[input_num],
96 {indices_shape.num_elements()},
97 &indices[input_num]));
98 }
99
100 // Find which slice will be used for each index. If the same index
101 // appears in multiple inputs, the last one is used. The logic
102 // here is different from that in third_party/tensorflow because
103 // it is important for XLA that there be a well-formed Concat
104 // operation at the end. The existing CPU/GPU code copies multiple
105 // source slices to the same destination slice if there are
106 // repeated indices, whereas the XLA code works out which
107 // source slice will 'win' and only uses that in the Concat.
108 int max_index = -1;
109 for (int input_num = 0; input_num < indices.size(); input_num++) {
110 for (int i = 0; i < indices[input_num].shape().dimensions(0); ++i) {
111 max_index = std::max(max_index, indices[input_num].Get<int>({i}));
112 }
113 }
114 int number_of_indices = max_index + 1;
115 OP_REQUIRES(ctx, number_of_indices > 0,
116 errors::InvalidArgument("no indices supplied"));
117 // Construct the reverse mapping, for each index, of which slice of which
118 // input it comes from.
119 std::vector<int32> src_input_vector(number_of_indices);
120 std::vector<int32> src_slice_vector(number_of_indices);
121 std::vector<bool> src_index_used(number_of_indices);
122 int index_used_count = 0;
123 for (int input_num = 0; input_num < indices.size(); input_num++) {
124 for (int i = 0; i < indices[input_num].shape().dimensions(0); ++i) {
125 int index = indices[input_num].Get<int>({i});
126 src_input_vector[index] = input_num;
127 src_slice_vector[index] = i;
128 if (!src_index_used[index]) {
129 src_index_used[index] = true;
130 ++index_used_count;
131 }
132 }
133 }
134 OP_REQUIRES(ctx, index_used_count == number_of_indices,
135 errors::InvalidArgument("not all indices are used"));
136
137 // Look up all the children expressions that represent the data
138 // inputs.
139 std::vector<xla::ComputationDataHandle> input(indices.size());
140 for (int input_num = 0; input_num < indices.size(); input_num++) {
141 TensorShape new_shape;
142 // first reshaped dimension is the number of indices for this input.
143 new_shape.AddDim(indices[input_num].shape().dimensions(0));
144 // Then the rest are the common extra shape.
145 for (int d = indices0_shape.dims(); d < data0_shape.dims(); d++) {
146 new_shape.AddDim(data0_shape.dim_size(d));
147 }
148 // Get the data, shaped appropriately.
149 auto handle = data[input_num];
150 if (new_shape == data_shapes[input_num]) {
151 input[input_num] = handle;
152 } else {
153 input[input_num] =
154 ctx->builder()->Reshape(handle, new_shape.dim_sizes());
155 }
156 }
157
158 // Set up the vectors for slicing: the first dimension will vary
159 // slice by slice, and the rest take the full common extra shape.
160 std::vector<int64> slice_start(1 + data0_shape.dims() -
161 indices0_shape.dims());
162 std::vector<int64> slice_limit(1 + data0_shape.dims() -
163 indices0_shape.dims());
164 std::vector<int64> stride(1 + data0_shape.dims() - indices0_shape.dims(),
165 1);
166 for (int d = indices0_shape.dims(); d < data0_shape.dims(); d++) {
167 slice_limit[1 + d - indices0_shape.dims()] = data0_shape.dim_size(d);
168 }
169 std::vector<xla::ComputationDataHandle> to_concat(number_of_indices);
170 for (int index_num = 0; index_num < number_of_indices; index_num++) {
171 const auto& expression = input[src_input_vector[index_num]];
172 // Take the appropriate slice of data.
173 slice_start[0] = src_slice_vector[index_num];
174 slice_limit[0] = src_slice_vector[index_num] + 1;
175 // And place it in the concat list in the place indicated by
176 // the index.
177 to_concat[index_num] =
178 ctx->builder()->Slice(expression, slice_start, slice_limit, stride);
179 }
180
181 ctx->SetOutput(0, ctx->builder()->ConcatInDim(to_concat, 0));
182 }
183
184 private:
185 // Check if data0_shape[indices0.dims():] == data1_shape[indices1.dims():]
186 static bool SameExtraShape(const TensorShape& data0_shape,
187 const TensorShape& indices0,
188 const TensorShape& data1_shape,
189 const TensorShape& indices1) {
190 const int extra0 = data0_shape.dims() - indices0.dims();
191 const int extra1 = data1_shape.dims() - indices1.dims();
192 if (extra0 != extra1) return false;
193 for (int i = 0; i < extra0; i++) {
194 if (data0_shape.dim_size(indices0.dims() + i) !=
195 data1_shape.dim_size(indices1.dims() + i)) {
196 return false;
197 }
198 }
199 return true;
200 }
201};

204 DynamicStitchOp);
205REGISTER_XLA_OP(Name("ParallelDynamicStitch").CompileTimeConstInput("indices"),
206 DynamicStitchOp);
