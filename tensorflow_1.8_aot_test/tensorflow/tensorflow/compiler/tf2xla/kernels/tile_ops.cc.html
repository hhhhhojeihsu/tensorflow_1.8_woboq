34// --------------------------------------------------------------------------
35class TileOp : public XlaOpKernel {
36 public:
37 explicit TileOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {}
38
39 void Compile(XlaOpKernelContext* ctx) override {
40 const TensorShape input_shape = ctx->InputShape(0);
41 const TensorShape multiples_shape = ctx->InputShape(1);
42
43 OP_REQUIRES(
44 ctx, IsLegacyVector(multiples_shape),
45 errors::InvalidArgument("Expected multiples to be 1-D, but got shape ",
46 multiples_shape.DebugString()));
47 OP_REQUIRES(ctx, input_shape.dims() == multiples_shape.num_elements(),
48 errors::InvalidArgument(
49 "Expected multiples argument to be a vector of length ",
50 input_shape.dims(), " but got length ",
51 multiples_shape.dim_size(0)));
52 const int input_dims = input_shape.dims();
53
54 // If input is a scalar then multiples has 0 elements and this is
55 // a NoOp.
56 if (input_dims == 0) {
57 ctx->SetOutput(0, ctx->Input(0));
58 return;
59 }
60
61 xla::Literal literal;
62 OP_REQUIRES_OK(ctx, ctx->ConstantInput(1, &literal));
63
64 // zero_element_result is true if the final shape has 0 elements,
65 // i.e. if any of the input dimensions or multiples is zero.
66 std::vector<int64> multiples_array(input_dims);
67 std::vector<int64> output_shape;
68 bool all_multiples_are_one = true;
69 bool one_dimension_is_broadcasted_without_multiple = true;
70 for (int i = 0; i < input_dims; ++i) {
71 int multiple = literal.Get<int>({i});
72 OP_REQUIRES(ctx, multiple,
73 errors::InvalidArgument("Expected multiples[", i,
74 "] >= 0, but got ", multiple));
75 int64 new_dim = input_shape.dim_size(i) * multiple;
76 output_shape.push_back(new_dim);
77 multiples_array[i] = multiple;
78 all_multiples_are_one = all_multiples_are_one && multiple == 1;
79 // If the multiple of a non-one dimensions is not one, then binary
80 // operation broadcast semantics will not be sufficient to implement the
81 // tile operation.
82 one_dimension_is_broadcasted_without_multiple =
83 one_dimension_is_broadcasted_without_multiple &&
84 ((input_shape.dim_size(i) > 1 && multiple == 1) ||
85 input_shape.dim_size(i) == 1);
86 }
87 auto input = ctx->Input(0);
88 // If all multiples are 1, than the input is the same as the output.
89 if (all_multiples_are_one) {
90 ctx->SetOutput(0, input);
91 return;
92 }
93 if (one_dimension_is_broadcasted_without_multiple) {
94 // Create a constant Zero the size of the output shape to leverage binary
95 // operation broadcast semantics.
96 auto broadcasted_zero = ctx->builder()->Broadcast(
97 XlaHelpers::Zero(ctx->builder(), ctx->input_type(0)), output_shape);
98 ctx->SetOutput(0, ctx->builder()->Add(broadcasted_zero, input));
99 return;
100 }
101
102 // First broadcast the requisite number of multiples along each
103 // dimension. This prepends the broadcasted dimensions, so an
104 // input of shape [2,3,1] broadcast with multiples [5,4,3] will
105 // end up with shape [5,4,3,2,3,1].
106 auto broadcasted = ctx->builder()->Broadcast(input, multiples_array);
107 // Now flatten and reshape. The broadcasted dimensions are
108 // paired with the original dimensions so in the above example
109 // we flatten [0,3,1,4,2,5] then reshape to [10,12,3].
110 std::vector<int64> flattened;
111 for (int i = 0; i < output_shape.size(); ++i) {
112 flattened.push_back(i);
113 flattened.push_back(i + output_shape.size());
114 }
115 xla::ComputationDataHandle output =
116 ctx->builder()->Reshape(broadcasted, flattened, output_shape);
117
118 ctx->SetOutput(0, output);
119 }
120
121 private:
122 TF_DISALLOW_COPY_AND_ASSIGN(TileOp);
123};
124
125REGISTER_XLA_OP(Name("Tile").CompileTimeConstInput("multiples"), TileOp);
