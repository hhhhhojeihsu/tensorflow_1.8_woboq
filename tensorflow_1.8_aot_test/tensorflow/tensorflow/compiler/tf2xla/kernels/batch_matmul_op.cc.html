16#include "tensorflow/compiler/tf2xla/lib/batch_dot.h"
17#include "tensorflow/compiler/tf2xla/xla_op_kernel.h"
18#include "tensorflow/compiler/tf2xla/xla_op_registry.h"
19
20namespace tensorflow {
21namespace {
22
23class BatchMatMulOp : public XlaOpKernel {
24 public:
25 explicit BatchMatMulOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {
26 OP_REQUIRES_OK(ctx, ctx->GetAttr("adj_x", &adj_x_));
27 OP_REQUIRES_OK(ctx, ctx->GetAttr("adj_y", &adj_y_));
28 }
29
30 void Compile(XlaOpKernelContext* ctx) override {
31 auto result = BatchDot(ctx->builder(), ctx->Input(0), ctx->Input(1),
32 /*transpose_x=*/adj_x_, /*transpose_y=*/adj_y_,
33 /*conjugate_x=*/adj_x_, /*conjugate_y=*/adj_y_);
34 OP_REQUIRES_OK(ctx, result.status());
35 ctx->SetOutput(0, result.ValueOrDie());
36 }
37
38 private:
39 bool adj_x_;
40 bool adj_y_;
41};
42
43REGISTER_XLA_OP(Name("BatchMatMul"), BatchMatMulOp);
44
45} // namespace
46} // namespace tensorflow
