16#include "tensorflow/compiler/tf2xla/lib/cholesky.h"
17#include "tensorflow/compiler/tf2xla/xla_op_kernel.h"
18#include "tensorflow/compiler/tf2xla/xla_op_registry.h"
19
20namespace tensorflow {
21namespace {
22
23class CholeskyOp : public XlaOpKernel {
24 public:
25 explicit CholeskyOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {}
26 void Compile(XlaOpKernelContext* ctx) override {
27 auto result = Cholesky(ctx->builder(), ctx->Input(0));
28 if (!result.ok()) {
29 ctx->SetStatus(result.status());
30 return;
31 }
32 ctx->SetOutput(0, result.ValueOrDie());
33 }
34};
35
36REGISTER_XLA_OP(Name("Cholesky").TypeConstraint("T", kFloatTypes), CholeskyOp);
