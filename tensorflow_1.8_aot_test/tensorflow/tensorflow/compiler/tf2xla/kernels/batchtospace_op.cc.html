23void BatchToSpace(XlaOpKernelContext* ctx,
24 const xla::ComputationDataHandle& input, DataType input_dtype,
25 const TensorShape& input_tensor_shape,
26 gtl::ArraySlice<int64> block_shape,
27 const xla::Literal& crops) {
28 const int input_rank = input_tensor_shape.dims();
29 const gtl::InlinedVector<int64, 4> input_shape =
30 input_tensor_shape.dim_sizes();
31 const int block_rank = block_shape.size();
32
33 OP_REQUIRES(
34 ctx, input_rank >= 1 + block_rank,
35 errors::InvalidArgument("input rank should be >= ", 1 + block_rank,
36 " instead of ", input_rank));
37 gtl::ArraySlice<int64> remainder_shape(input_shape);
38 remainder_shape.remove_prefix(1 + block_rank);
39
40 OP_REQUIRES(
41 ctx,
42 xla::ShapeUtil::Rank(crops.shape()) == 2 &&
43 block_rank == xla::ShapeUtil::GetDimension(crops.shape(), 0) &&
44 2 == xla::ShapeUtil::GetDimension(crops.shape(), 1),
45 errors::InvalidArgument("crops should have shape [", block_rank,
46 ", 2] instead of ",
47 xla::ShapeUtil::HumanString(crops.shape())));
48
49 xla::ComputationBuilder* b = ctx->builder();
50 const int64 batch_size = input_shape[0];
51
52 // Compute the product of the block_shape values.
53 int64 block_num_elems = 1;
54 for (int i = 0; i < block_rank; ++i) {
55 block_num_elems *= block_shape[i];
56 }
57 OP_REQUIRES(ctx, block_num_elems > 0,
58 errors::InvalidArgument(
59 "The product of the block dimensions must be positive"));
60
61 // 1. Reshape `input` to `reshaped` of shape:
62 // [block_shape[0], ..., block_shape[M-1],
63 // batch / prod(block_shape),
64 // input_shape[1], ..., input_shape[N-1]]
65
66 OP_REQUIRES(
67 ctx, batch_size % block_num_elems == 0,
68 errors::InvalidArgument("Input batch dimension (", batch_size,
69 ") is not divisible by product of block sizes (",
70 block_num_elems, ")"));
71 std::vector<int64> reshaped_shape(input_rank + block_rank);
72 std::copy(block_shape.begin(), block_shape.end(), reshaped_shape.begin());
73 reshaped_shape[block_rank] = batch_size / block_num_elems;
74 std::copy(input_shape.begin() + 1, input_shape.end(),
75 reshaped_shape.begin() + block_rank + 1);
76 xla::ComputationDataHandle reshaped = b->Reshape(input, reshaped_shape);
77
78 // 2. Permute dimensions of `reshaped` to produce `permuted` of shape
79 // [batch / prod(block_shape),
80 //
81 // input_shape[1], block_shape[0],
82 // ...,
83 // input_shape[M], block_shape[M-1],
84 //
85 // input_shape[M+1], ..., input_shape[N-1]]
86 std::vector<int64> permutation(reshaped_shape.size());
87 permutation[0] = block_rank;
88 for (int i = 0; i < block_rank; ++i) {
89 permutation[1 + 2 * i] = block_rank + 1 + i;
90 permutation[1 + 2 * i + 1] = i;
91 }
92 std::iota(permutation.begin() + 1 + block_rank * 2, permutation.end(),
93 1 + block_rank * 2);
94 xla::ComputationDataHandle permuted = b->Transpose(reshaped, permutation);
95
96 // 3. Reshape `permuted` to produce `reshaped_permuted` of shape
97 // [batch / prod(block_shape),
98 //
99 // input_shape[1] * block_shape[0],
100 // ...,
101 // input_shape[M] * block_shape[M-1],
102 //
103 // input_shape[M+1],
104 // ...,
105 // input_shape[N-1]]
106 std::vector<int64> reshaped_permuted_shape(input_rank);
107 reshaped_permuted_shape[0] = batch_size / block_num_elems;
108 for (int i = 0; i < block_rank; ++i) {
109 reshaped_permuted_shape[1 + i] = block_shape[i] * input_shape[1 + i];
110 }
111 std::copy(remainder_shape.begin(), remainder_shape.end(),
112 reshaped_permuted_shape.begin() + 1 + block_rank);
113
114 xla::ComputationDataHandle reshaped_permuted =
115 b->Reshape(permuted, reshaped_permuted_shape);
116
117 // 4. Crop the start and end of dimensions `[1, ..., M]` of
118 // `reshaped_permuted` according to `crops` to produce the output of shape:
119 // [batch / prod(block_shape),
120 //
121 // input_shape[1] * block_shape[0] - crops[0,0] - crops[0,1],
122 // ...,
123 // input_shape[M] * block_shape[M-1] - crops[M-1,0] - crops[M-1,1],
124 //
125 // input_shape[M+1], ..., input_shape[N-1]]
126 std::vector<int64> start_indices(input_rank, 0);
127 std::vector<int64> end_indices = reshaped_permuted_shape;
128 std::vector<int64> strides(input_rank, 1);
129 for (int i = 0; i < block_rank; ++i) {
130 int64 crop_start = crops.Get<int64>({i, 0});
131 int64 crop_end = crops.Get<int64>({i, 1});
132 OP_REQUIRES(ctx, crop_start >= 0 && crop_end >= 0,
133 errors::InvalidArgument("Crops must be non-negative"));
134 start_indices[1 + i] = crop_start;
135 end_indices[1 + i] -= crop_end;
136 OP_REQUIRES(
137 ctx, start_indices[1 + i] <= end_indices[1 + i],
138 errors::InvalidArgument(
139 "Cropped size must be non-negative: start: ", crop_start,
140 " end: ", crop_end, " size ", reshaped_permuted_shape[1 + i]));
141 }
142 xla::ComputationDataHandle output =
143 b->Slice(reshaped_permuted, start_indices, end_indices, strides);
144 ctx->SetOutput(0, output);
145}

147class BatchToSpaceNDOp : public XlaOpKernel {
148 public:
149 explicit BatchToSpaceNDOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {}
150
151 void Compile(XlaOpKernelContext* ctx) override {
152 std::vector<int64> block_shape;
153 OP_REQUIRES_OK(ctx, ctx->ConstantInputAsIntVector(1, &block_shape));
154
155 xla::Literal crops;
156 OP_REQUIRES_OK(ctx, ctx->ConstantInputAsInt64Literal(2, &crops));
157
158 BatchToSpace(ctx, ctx->Input(0), input_type(0), ctx->InputShape(0),
159 block_shape, crops);
160 }
161};
162REGISTER_XLA_OP(Name("BatchToSpaceND")
163 .CompileTimeConstInput("block_shape")
164 .CompileTimeConstInput("crops"),
165 BatchToSpaceNDOp);
166
167class BatchToSpaceOp : public XlaOpKernel {
168 public:
169 explicit BatchToSpaceOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {
170 OP_REQUIRES_OK(ctx, ctx->GetAttr("block_size", &block_size_));
171 OP_REQUIRES(
172 ctx, block_size_ > 1,
173 errors::InvalidArgument("Block size should be > 1: ", block_size_));
174 }
175
176 void Compile(XlaOpKernelContext* ctx) override {
177 xla::Literal crops;
178 OP_REQUIRES_OK(ctx, ctx->ConstantInputAsInt64Literal(1, &crops));
179
180 BatchToSpace(ctx, ctx->Input(0), input_type(0), ctx->InputShape(0),
181 {block_size_, block_size_}, crops);
182 }
183
184 private:
185 int block_size_;
186};
187REGISTER_XLA_OP(Name("BatchToSpace").CompileTimeConstInput("crops"),
188 BatchToSpaceOp);
