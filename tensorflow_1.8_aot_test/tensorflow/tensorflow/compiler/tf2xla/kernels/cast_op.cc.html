26class CastOp : public XlaOpKernel {
27 public:
28 explicit CastOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {
29 OP_REQUIRES_OK(ctx, ctx->GetAttr("SrcT", &src_dtype_));
30 OP_REQUIRES_OK(ctx, ctx->GetAttr("DstT", &dst_dtype_));
31 OP_REQUIRES_OK(ctx, DataTypeToPrimitiveType(src_dtype_, &src_type_));
32 OP_REQUIRES_OK(ctx, DataTypeToPrimitiveType(dst_dtype_, &dst_type_));
33 }
34
35 void Compile(XlaOpKernelContext* ctx) override {
36 xla::ComputationBuilder* builder = ctx->builder();
37 xla::ComputationDataHandle input = ctx->Input(0);
38 xla::ComputationDataHandle output;
39
40 if (src_dtype_ == dst_dtype_) {
41 output = input;
42 } else if (dst_dtype_ == DT_BOOL) {
43 output = builder->Ne(input, XlaHelpers::Zero(builder, src_dtype_));
44 } else if (xla::primitive_util::IsComplexType(src_type_) &&
45 !xla::primitive_util::IsComplexType(dst_type_)) {
46 // As in cast_op.h, we replicate the numpy behavior of truncating the
47 // imaginary part.
48 output = builder->ConvertElementType(builder->Real(input), dst_type_);
49 } else {
50 output = builder->ConvertElementType(input, dst_type_);
51 }
52
53 ctx->SetOutput(0, output);
54 }
55
56 protected:
57 DataType src_dtype_, dst_dtype_;
58 xla::PrimitiveType src_type_, dst_type_;
59
60 TF_DISALLOW_COPY_AND_ASSIGN(CastOp);
61};
62
63REGISTER_XLA_OP(Name("Cast"), CastOp);

65class BitcastOp : public XlaOpKernel {
66 public:
67 explicit BitcastOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {
68 OP_REQUIRES_OK(ctx, ctx->GetAttr("T", &src_dtype_));
69 OP_REQUIRES_OK(ctx, ctx->GetAttr("type", &dst_dtype_));
70 OP_REQUIRES_OK(ctx, DataTypeToPrimitiveType(src_dtype_, &src_type_));
71 OP_REQUIRES_OK(ctx, DataTypeToPrimitiveType(dst_dtype_, &dst_type_));
72 }
73
74 void Compile(XlaOpKernelContext* ctx) override {
75 xla::ComputationBuilder* builder = ctx->builder();
76 xla::ComputationDataHandle input = ctx->Input(0);
77 xla::ComputationDataHandle output;
78
79 if (src_dtype_ == dst_dtype_) {
80 output = input;
81 } else {
82 // The only complex type in XLA is C64, so error out if the bitcast has a
83 // complex source or destination type and the bitcast is not trivial.
84 OP_REQUIRES(ctx,
85 !xla::primitive_util::IsComplexType(src_type_) &&
86 !xla::primitive_util::IsComplexType(dst_type_),
87 errors::Unimplemented("Complex types not supported."));
88 // XLA bitcast requires that the bit-width of the source and destination
89 // matches, and currently only the simple lowering is performed.
90 OP_REQUIRES(ctx,
91 xla::primitive_util::BitWidth(src_type_) ==
92 xla::primitive_util::BitWidth(dst_type_),
93 errors::Unimplemented(
94 "Only bitcasts between equally sized types supported."));
95 output = builder->BitcastConvertType(input, dst_type_);
96 }
97
98 ctx->SetOutput(0, output);
99 }
100
101 protected:
102 DataType src_dtype_, dst_dtype_;
103 xla::PrimitiveType src_type_, dst_type_;
104
105 TF_DISALLOW_COPY_AND_ASSIGN(BitcastOp);
106};
107
108REGISTER_XLA_OP(Name("Bitcast"), BitcastOp);
