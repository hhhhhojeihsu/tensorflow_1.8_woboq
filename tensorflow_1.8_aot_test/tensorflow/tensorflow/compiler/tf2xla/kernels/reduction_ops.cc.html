28class SumOp : public XlaReductionOp {
29 public:
30 explicit SumOp(OpKernelConstruction* ctx)
31 : XlaReductionOp(ctx,
32 XlaHelpers::SumAccumulationType(ctx->input_type(0))) {}
33 xla::ComputationDataHandle InitialValue(
34 xla::ComputationBuilder* builder) override {
35 return XlaHelpers::Zero(builder, reduction_type_);
36 }
37 void BuildReducer(xla::ComputationBuilder* builder,
38 const xla::ComputationDataHandle& scalar_lhs,
39 const xla::ComputationDataHandle& scalar_rhs) override {
40 builder->Add(scalar_lhs, scalar_rhs);
41 }
42};
43
44REGISTER_XLA_OP(Name("Sum").CompileTimeConstInput("reduction_indices"), SumOp);

105class MeanOp : public XlaReductionOp {
106 public:
107 explicit MeanOp(OpKernelConstruction* ctx)
108 : XlaReductionOp(ctx,
109 XlaHelpers::SumAccumulationType(ctx->input_type(0))) {}
110
111 xla::ComputationDataHandle InitialValue(
112 xla::ComputationBuilder* builder) override {
113 return XlaHelpers::Zero(builder, reduction_type_);
114 }
115 void BuildReducer(xla::ComputationBuilder* builder,
116 const xla::ComputationDataHandle& scalar_lhs,
117 const xla::ComputationDataHandle& scalar_rhs) override {
118 builder->Add(scalar_lhs, scalar_rhs);
119 }
120
121 xla::ComputationDataHandle BuildFinalizer(
122 xla::ComputationBuilder* builder,
123 const xla::ComputationDataHandle& reduce_output,
124 int64 num_elements_reduced) override {
125 auto divisor = XlaHelpers::IntegerLiteral(builder, input_type(0),
126 num_elements_reduced);
127 return builder->Div(reduce_output, divisor);
128 }
129};
130
131REGISTER_XLA_OP(Name("Mean").CompileTimeConstInput("reduction_indices"),
132 MeanOp);
