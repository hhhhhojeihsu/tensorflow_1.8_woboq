32// Struct that represents a possibly-absent Tensor.
33struct OptionalTensor {
34 string name; // A descriptive name
35 bool present = false; // Is the tensor present?
36 Tensor value; // If present, what is the Tensor's value?
37};
38
39// The XlaCompilationCache class caches the results of the XlaCompiler class,
40// which converts a Tensorflow graph into a compiled XLA compilation.
41//
42// Since XLA computations must have static shapes, the cache generates a new
43// XLA computation for each new set of input shapes.
44//
45// Currently no cache eviction policy is implemented and the cache grows without
46// bound.
47class XlaCompilationCache : public ResourceBase {
48 public:
49 XlaCompilationCache(xla::LocalClient* client, DeviceType device_type);
50 ~XlaCompilationCache() override;
51
52 // Compiles a function into a XlaCompiler::CompilationResult that can be used
53 // to execute an XLA Computation. Compilation results are cached.
54 // `function` is the name of a Tensorflow function to compile.
55 // `constant_args` is a map of tensorflow argument number to its constant
56 // value.
57 // `variable_args` is a snapshot of the current values of the
58 // resource variable arguments to `function`; uninitialized variables are
59 // represented by an absent OptionalTensor.
60 // The result of compilation is written to `*compilation_result`, which must
61 // be non-null. If `executable` is non-null, also builds an
62 // xla::LocalExecutable and sets `executable` to point to it. The resulting
63 // executable pointer may be null if the computation has no non-constant
64 // outputs.
65 Status Compile(const XlaCompiler::Options& options,
66 const NameAttrList& function,
67 const std::map<int, Tensor>& constant_args,
68 const std::map<int, OptionalTensor>& variable_args,
69 OpKernelContext* ctx,
70 const XlaCompiler::CompilationResult** compilation_result,
71 xla::LocalExecutable** executable,
72 const XlaCompiler::CompileOptions* compile_options);
73
74 // As above, but calls XlaCompiler::CompileSingleOp instead of
75 // XlaCompiler::CompileFunction.
76 Status CompileSingleOp(
77 const XlaCompiler::Options& options,
78 const std::map<int, Tensor>& constant_args,
79 const std::map<int, OptionalTensor>& variable_args, OpKernelContext* ctx,
80 const XlaCompiler::CompilationResult** compilation_result,
81 xla::LocalExecutable** executable,
82 const XlaCompiler::CompileOptions* compile_options);
83
84 xla::LocalClient* client() const { return client_; }
85 const DeviceType& device_type() const { return device_type_; }
86
87 string DebugString() override;
88
89 private:
90 // Common implementation of Compile and CompileSingleOp.
91 Status CompileImpl(const XlaCompiler::Options& options,
92 const NameAttrList& function,
93 const std::map<int, Tensor>& constant_args,
94 const std::map<int, OptionalTensor>& variable_args,
95 OpKernelContext* ctx,
96 const XlaCompiler::CompilationResult** compilation_result,
97 xla::LocalExecutable** executable,
98 const XlaCompiler::CompileOptions* compile_options,
99 bool compile_single_op);
100
101 // Takes `result` which has been compiled from a Tensorflow subgraph to a
102 // XLA computation already, and generates an XLA LocalExecutable `executable`.
103 Status BuildExecutable(const XlaCompiler::Options& options,
104 const XlaCompiler::CompilationResult& result,
105 std::unique_ptr<xla::LocalExecutable>* executable);
106
107 xla::LocalClient* const client_;
108 const DeviceType device_type_;
109
110 // Describes the types, shapes and any compile-time constant arguments
111 // to a kernel. Key that uniquely identifies a compilation output.
112 struct Signature {
113 string name;
114
115 std::vector<std::pair<DataType, TensorShape>> arg_types;
116
117 // List of Tensor values for compile-time constant arguments to the
118 // compilation, ordered by argument number. Tensors must be in host memory.
119 std::vector<Tensor> arg_values;
120
121 bool operator==(const Signature& other) const;
122
123 struct Hash {
124 uint64 operator()(const Signature& signature) const;
125 };
126 };
127 static string SignatureDebugString(const Signature& sig);
128
129 // Builds the signature for a compilation.
130 Status BuildSignature(const NameAttrList& function,
131 const std::map<int, Tensor>& constant_args,
132 const std::map<int, OptionalTensor>& variable_args,
133 OpKernelContext* ctx, Signature* signature);
134
135 // The value associated with a cache entry.
136 struct Entry {
137 mutex mu;
138
139 // Have we tried compiling this entry?
140 bool compiled = false;
141
142 // Did compilation succeed?
143 Status compilation_status GUARDED_BY(mu);
144
145 // Output of the XlaCompiler.
146 XlaCompiler::CompilationResult compilation_result GUARDED_BY(mu);
147
148 // The XLA executable compiled from <computation>. May be null if no
149 // executable has been built.
150 std::unique_ptr<xla::LocalExecutable> executable GUARDED_BY(mu);
151 };
152
153 mutex mu_;
154 std::unordered_map<Signature, std::unique_ptr<Entry>, Signature::Hash> cache_
155 GUARDED_BY(mu_);
156
157 TF_DISALLOW_COPY_AND_ASSIGN(XlaCompilationCache);
158};
