57// Caches a XlaDeviceAllocator per <backend, device ordinal> pair. A
58// XlaDeviceAllocator is created on demand and is associated with a
59// XlaDevice. It outlives the device itself (for instance, the buffer
60// backing a tensor holds a pointer to the allocator for book-keeping,
61// and this buffer can outlast the device).
62class XlaDeviceAllocatorState {
63 public:
64 // Creates or returns a cached XlaDeviceAllocator for a given
65 // backend and device_ordinal.
66 static XlaDeviceAllocator* GetOrCreateXlaDeviceAllocator(
67 const xla::Backend* backend, int device_ordinal);
68
69 private:
70 // Returns the singleton instance of XlaDeviceAllocatorState.
71 static XlaDeviceAllocatorState& Singleton();
72 XlaDeviceAllocatorState();
73 ~XlaDeviceAllocatorState();
74
75 mutex allocator_mutex_; // Guards the singleton allocator state.
76 std::unordered_map<std::pair<const xla::Backend*, int>,
77 std::unique_ptr<XlaDeviceAllocator>,
78 hash<std::pair<const xla::Backend*, int>>>
79 allocators_ GUARDED_BY(allocator_mutex_);
80
81 TF_DISALLOW_COPY_AND_ASSIGN(XlaDeviceAllocatorState);
82};
83
84/* static */ XlaDeviceAllocatorState& XlaDeviceAllocatorState::Singleton() {
85 static auto a = new XlaDeviceAllocatorState;
86 return *a;
87}
88
89XlaDeviceAllocatorState::XlaDeviceAllocatorState() = default;
90XlaDeviceAllocatorState::~XlaDeviceAllocatorState() = default;
91
92XlaDeviceAllocator* XlaDeviceAllocatorState::GetOrCreateXlaDeviceAllocator(
93 const xla::Backend* backend, int device_ordinal) {
94 XlaDeviceAllocatorState& state = Singleton();
95 mutex_lock lock(state.allocator_mutex_);
96
97 auto it = state.allocators_.find({backend, device_ordinal});
98 if (it != state.allocators_.end()) {
99 return it->second.get();
100 }
101
102 std::unique_ptr<XlaDeviceAllocator> alloc =
103 xla::MakeUnique<XlaDeviceAllocator>();
104 XlaDeviceAllocator* alloc_ptr = alloc.get();
105 state.allocators_[{backend, device_ordinal}] = std::move(alloc);
106 return alloc_ptr;
107}

109/* static */ Status XlaDevice::Create(
110 const string& platform_name, const string& device_name, int device_ordinal,
111 const string& jit_device_name, const SessionOptions& options,
112 const string& name_prefix,
113 const XlaOpRegistry::DeviceRegistration& registration,
114 bool transfer_as_literal, std::unique_ptr<XlaDevice>* device) {
115 VLOG(1) << "XlaDevice::Create " << platform_name << " " << device_name << ":"
116 << device_ordinal;
117
118 // These are no-ops if they have already been done previously for
119 // this device_name/compilation_device_name pair.
120 XlaOpRegistry::RegisterCompilationDevice(device_name, registration);
121
122 auto platform = se::MultiPlatformManager::PlatformWithName(platform_name);
123 if (!platform.ok()) {
124 return StreamExecutorUtil::ConvertStatus(platform.status());
125 }
126
127 const DeviceAttributes attrs = Device::BuildDeviceAttributes(
128 strings::StrCat(name_prefix, "/device:", device_name, ":",
129 device_ordinal),
130 DeviceType(device_name), Bytes(16ULL << 30), DeviceLocality(),
131 strings::StrCat("device: ", device_name, " device"));
132
133 device->reset(new XlaDevice(options, attrs, device_ordinal,
134 DeviceType(jit_device_name),
135 platform.ValueOrDie(), transfer_as_literal));
136 return Status::OK();
137}

158/* static */ Status XlaDevice::GetMetadata(OpKernelContext* ctx,
159 const Metadata** metadata) {
160 *metadata = nullptr;
161 XlaDevice* xla_device =
162 dynamic_cast<XlaDevice*>(ctx->device()->UnderlyingDevice());
163 if (xla_device == nullptr) {
164 return errors::Internal(
165 "Cannot get XLA metadata from non-XLA device \"", ctx->device()->name(),
166 "\". GetMetadata must only be called on an XLA device. Either an "
167 "internal bug has been triggered, or an XLA-specific op has been "
168 "placed on the wrong device.");
169 }
170 *metadata = &(xla_device->xla_metadata_);
171 return Status::OK();
172}

188xla::LocalClient* XlaDevice::client() const {
189 // We lazily create the client because the platform commits to the
190 // details of the host hardware when the client is created, so we
191 // don't want to do it until we get a chance to hook the platform up
192 // to a simulator.
193
194 // For now GetOrCreateLocalClient always returns success when passed
195 // a non-null platform. If that changes we may have to plumb in some
196 // way to pass Status back.
197 return xla::ClientLibrary::GetOrCreateLocalClient(platform_).ValueOrDie();
198}

200Allocator* XlaDevice::GetAllocator(AllocatorAttributes attr) {
201 if (attr.on_host()) {
202 return cpu_allocator();
203 }
204
205 if (xla_allocator_ == nullptr) {
206 xla::Backend* backend = client()->mutable_backend();
207 xla_allocator_ = XlaDeviceAllocatorState::GetOrCreateXlaDeviceAllocator(
208 backend, device_ordinal_);
209 }
210 return xla_allocator_;
211}

213xla::StatusOr<se::Stream*> XlaDevice::GetStream() {
214 if (!stream_) {
215 xla::Backend* backend = client()->mutable_backend();
216 TF_ASSIGN_OR_RETURN(stream_, backend->BorrowStream(device_ordinal_));
217 }
218 return stream_.get();
219}

221Status XlaDevice::FillContextMap(const Graph* graph,
222 DeviceContextMap* device_context_map) {
223 VLOG(1) << "XlaDevice::FillContextMap";
224 device_context_map->resize(graph->num_node_ids());
225 TF_ASSIGN_OR_RETURN(se::Stream * stream, GetStream());
226 // Call GetAllocator for the side-effect of ensuring the allocator and
227 // XlaTensorInfoManager is created.
228 (void)GetAllocator({});
229 auto ctx = new XlaDeviceContext(stream, client(), transfer_as_literal_);
230 for (Node* n : graph->nodes()) {
231 VLOG(2) << n->id() << " : " << n->type_string() << " : " << n->name();
232 ctx->Ref();
233 (*device_context_map)[n->id()] = ctx;
234 }
235 ctx->Unref();
236 return Status::OK();
237}

259Status XlaDevice::MakeTensorFromProto(const TensorProto& tensor_proto,
260 const AllocatorAttributes alloc_attrs,
261 Tensor* tensor) {
262 VLOG(1) << "XlaDevice::MakeTensorFromProto";
263
264 Tensor parsed(tensor_proto.dtype());
265 if (!parsed.FromProto(cpu_allocator(), tensor_proto)) {
266 return errors::InvalidArgument("Cannot parse tensor from proto: ",
267 tensor_proto.DebugString());
268 }
269
270 Status status;
271 if (alloc_attrs.on_host()) {
272 *tensor = parsed;
273 } else {
274 Tensor copy(GetAllocator(alloc_attrs), parsed.dtype(), parsed.shape());
275 Notification n;
276 TF_ASSIGN_OR_RETURN(se::Stream * stream, GetStream());
277 XlaTransferManager manager(stream, client(), transfer_as_literal_);
278 manager.CopyCPUTensorToDevice(&parsed, this, &copy,
279 [&n, &status](const Status& s) {
280 status = s;
281 n.Notify();
282 });
283 n.WaitForNotification();
284 *tensor = copy;
285 }
286 VLOG(2) << "Allocated tensor at " << DMAHelper::base(tensor);
287 return status;
288}

290XlaDeviceOpRegistrations* RegisterXlaDeviceKernels(const char* device,
291 const char* jit_device) {
292 // Any op assigned to the device that isn't rewritten by the graph rewriter
293 // gets executed by a n XlaCompileOnDemandOp, which compiles it and executes
294 // it just-in-time.
295 kernel_factory::OpKernelRegistrar::Factory factory =
296 [](OpKernelConstruction* context) -> OpKernel* {
297 return new XlaCompileOnDemandOp(context);
298 };
299 XlaOpRegistry::RegisterCompilationKernels();
300 XlaDeviceOpRegistrations* registrations = new XlaDeviceOpRegistrations;
301 for (const KernelDef* jit_def : XlaOpRegistry::DeviceKernels(
302 jit_device,
303 /*include_compilation_only_kernels=*/false)) {
304 KernelDef* def = new KernelDef(*jit_def);
305 def->set_device_type(device);
306 registrations->op_kernel_registrars.emplace_back(
307 new kernel_factory::OpKernelRegistrar(def, "XlaCompileOnDemandOp",
308 factory));
309 }
310 return registrations;
311}
