16// The XlaDevice executes a TensorFlow graph using the XLA linear algebra
17// runtime.
18//
19// Operators assigned to an XlaDevice are compiled into XLA computations.
20// Tensors on an XlaDevice are thin wrappers around XLA GlobalDataHandles; state
21// is managed by XLA.
22//
23// XlaDevice is instantiated separately for each XLA backend (e.g., CPU or GPU),
24// under different names (e.g., XLA_CPU or XLA_GPU).
25

44namespace tensorflow {
45
46class XlaDevice : public LocalDevice {
47 public:
48 // Wrapper class to store metadata about the XlaDevice, where it can be
49 // retrieved e.g., when lazily creating the XlaCompilationCache device.
50 class Metadata {
51 public:
52 Metadata(int device_ordinal, perftools::gputools::Platform* platform,
53 const DeviceType& device_type);
54
55 // The index of the device on this host.
56 int device_ordinal() const;
57
58 perftools::gputools::Platform* platform() const;
59 xla::LocalClient* client() const;
60 const DeviceType& jit_device_type() const;
61
62 private:
63 const int device_ordinal_;
64 const DeviceType device_type_;
65 perftools::gputools::Platform* platform_; // Not owned.
66
67 TF_DISALLOW_COPY_AND_ASSIGN(Metadata);
68 };
69
70 // Sets `*metadata` to the XlaDevice Metadata in the XLA device used by `ctx`.
71 static Status GetMetadata(OpKernelContext* ctx, const Metadata** metadata);
72
73 // Factory function. 'platform_name' is the name of the XLA platform.
74 // 'device_name' is the name of the Tensorflow device to create.
75 // 'jit_device_name' is the name of the corresponding JIT device.
76 // 'transfer_as_literal' is true if device<->host transfers must be done using
77 // XLA's TransferLiteral{To,From}Device interface. If false, we can use
78 // ThenMemcpy instead.
79 static Status Create(const string& platform_name, const string& device_name,
80 int device_ordinal, const string& jit_device_name,
81 const SessionOptions& options, const string& name_prefix,
82 const XlaOpRegistry::DeviceRegistration& registration,
83 bool transfer_as_literal,
84 std::unique_ptr<XlaDevice>* device);
85
86 XlaDevice(const SessionOptions& options, const DeviceAttributes& attrs,
87 int device_ordinal, const DeviceType& jit_device_name,
88 ::perftools::gputools::Platform* platform,
89 bool transfer_as_literal);
90 ~XlaDevice() override;
91
92 Allocator* GetAllocator(AllocatorAttributes attr) override;
93 void Compute(OpKernel* op_kernel, OpKernelContext* context) override;
94 void ComputeAsync(AsyncOpKernel* op_kernel, OpKernelContext* context,
95 AsyncOpKernel::DoneCallback done) override;
96 Status Sync() override { return Status::OK(); }
97
98 Status FillContextMap(const Graph* graph,
99 DeviceContextMap* device_context_map) override;
100
101 Status MakeTensorFromProto(const TensorProto& tensor_proto,
102 const AllocatorAttributes alloc_attrs,
103 Tensor* tensor) override;
104
105 xla::LocalClient* client() const;
106 xla::StatusOr<::perftools::gputools::Stream*> GetStream();
107
108 private:
109 // The metadata of this XlaDevice.
110 const Metadata xla_metadata_;
111 // Which hardware device in the client's platform this XlaDevice controls.
112 const int device_ordinal_;
113 // The name of the device that is used to compile Ops for this XlaDevice.
114 DeviceType jit_device_name_;
115 // Memory allocator associated with this device.
116 Allocator* xla_allocator_; // Not owned.
117 ::perftools::gputools::Platform* platform_; // Not owned.
118 // Stream associated with this device. Operations enqueued on this
119 // stream are executed on the device. Operations include data
120 // copying back and forth between CPU and the device, and
121 // computations enqueued by XLA.
122 xla::Backend::StreamPtr stream_;
123 // Must we use XLA's transfer manager for correct host<->device transfers? if
124 // false, we can use ThenMemcpy() instead.
125 bool transfer_as_literal_;
126};
127
128// Builds OpKernel registrations on 'device' for the JIT operators
129// registered on 'jit_device'. Returns ownership of a XlaDeviceOpRegistrations
130// object that encapsulates the kernel registrations.
131struct XlaDeviceOpRegistrations {
132 std::vector<std::unique_ptr<kernel_factory::OpKernelRegistrar>>
133 op_kernel_registrars;
134};
135XlaDeviceOpRegistrations* RegisterXlaDeviceKernels(const char* device,
136 const char* jit_device);
