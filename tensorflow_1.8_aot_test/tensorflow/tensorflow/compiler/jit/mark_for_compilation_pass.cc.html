49bool HasXLAKernel(const Node& node, const DeviceType& jit_device_type) {
50 // There is a SymbolicGradient kernel on the XLA_JIT device, but the gradient
51 // is really a kind of function call and will be handled by
52 // IsCompilableCall().
53 if (node.type_string() == "SymbolicGradient") return false;
54 return FindKernelDef(jit_device_type, node.def(), nullptr, nullptr).ok();
55}

64// Tests whether 'while_node' is a completely compilable loop.
65// Every operator in the condition and body functions must be compilable for a
66// while loop to be compilable.
67bool IsCompilableWhile(const Node& while_node,
68 const DeviceType& jit_device_type, int depth,
69 FunctionLibraryRuntime* lib_runtime) {
70 VLOG(2) << "Loop marking: " << while_node.type_string();
71
72 const NameAttrList* name_attr;
73 NodeDef call;
74 Status status;
75 status = GetNodeAttr(while_node.attrs(), "cond", &name_attr);
76 if (!status.ok()) {
77 VLOG(2) << "Missing 'cond' attribute on While node.";
78 return false;
79 }
80 const string cond_func = name_attr->name();
81 call.set_name("while_cond");
82 call.set_op(cond_func);
83 *call.mutable_attr() = name_attr->attr();
84 if (!IsCompilableCall(call, jit_device_type, depth + 1, lib_runtime)) {
85 VLOG(2) << "Can't compile loop condition: " << cond_func;
86 return false;
87 }
88 status = GetNodeAttr(while_node.attrs(), "body", &name_attr);
89 if (!status.ok()) {
90 VLOG(2) << "Missing 'body' attribute on While node.";
91 return false;
92 }
93 const string body_func = name_attr->name();
94 call.set_name("while_body");
95 call.set_op(body_func);
96 *call.mutable_attr() = name_attr->attr();
97 if (!IsCompilableCall(call, jit_device_type, depth + 1, lib_runtime)) {
98 VLOG(2) << "Can't compile loop body: " << body_func;
99 return false;
100 }
101 VLOG(2) << "Loop is compilable.";
102 return true;
103}

105// Tests whether 'call_def' is a call to a completely compilable function.
106// Every operator in the function must be compilable for a function to be
107// compilable.
108bool IsCompilableCall(const NodeDef& call_def,
109 const DeviceType& jit_device_type, int depth,
110 FunctionLibraryRuntime* lib_runtime) {
111 VLOG(2) << "Function marking: " << call_def.op();
112
113 if (depth > kMaxRecursionDepth) {
114 VLOG(2) << "Function depth limit exceeded";
115 return false;
116 }
117
118 FunctionLibraryRuntime::Handle handle;
119 Status status =
120 lib_runtime->Instantiate(call_def.op(), AttrSlice(call_def), &handle);
121 if (!status.ok()) {
122 VLOG(2) << "Could not instantiate " << call_def.op() << ": " << status;
123 return false;
124 }
125 const FunctionBody* fbody = lib_runtime->GetFunctionBody(handle);
126 CHECK(fbody);
127 const FunctionDef& fdef = fbody->fdef;
128 bool noinline = false;
129 if (GetNodeAttr(AttrSlice(&fdef.attr()), "_noinline", &noinline).ok() &&
130 noinline) {
131 // The underlying mechanism that calls non-inlined functions uses
132 // LocalExecutor, which interacts poorly with the LocalExecutor used by
133 // tf2xla to translate the TF graph into XLA. So we avoid this for now.
134 //
135 // TODO(b/36139787): Create a mechanism to set inlining hints.
136 VLOG(2) << "Can't compile noinline function: " << fdef.DebugString();
137 return false;
138 }
139
140 for (Node* node : fbody->graph->op_nodes()) {
141 if (node->type_string() == "_Arg" || node->type_string() == "_Retval")
142 continue;
143 if (node->type_string() == "While") {
144 // Handle functional While loop (not in open source build).
145 return IsCompilableWhile(*node, jit_device_type, depth + 1, lib_runtime);
146 }
147 if (!HasXLAKernel(*node, jit_device_type) &&
148 !IsCompilableCall(node->def(), jit_device_type, depth + 1,
149 lib_runtime)) {
150 VLOG(2) << "Function marking failed: unsupported op " << node->name()
151 << ": " << node->def().ShortDebugString();
152 return false;
153 }
154 }
155 VLOG(2) << "Function is compilable: " << call_def.op();
156 return true;
157}

336Status FindCompilationCandidates(
337 const Graph& graph, FunctionLibraryDefinition* flib_def, Env* env,
338 const std::function<bool(const Node*, const DeviceType&)>& is_compilable_fn,
339 OrderedNodeSet* candidates) {
340 OptimizerOptions opts;
341 std::unique_ptr<ProcessFunctionLibraryRuntime> pflr(
342 new ProcessFunctionLibraryRuntime(nullptr, env, TF_GRAPH_DEF_VERSION,
343 flib_def, opts));
344 FunctionLibraryRuntime* lib_runtime =
345 pflr->GetFLR(ProcessFunctionLibraryRuntime::kDefaultFLRDevice);
346
347 int64& fuel =
348 legacy_flags::GetMarkForCompilationPassFlags()->tf_xla_clustering_fuel;
349
350 // Iterate over nodes in sorted order so that compiler fuel is deterministic.
351 // We can't simply pass op_nodes().begin() and op_nodes().end to the
352 // std::vector constructor because they're not proper iterators, with
353 // iterator_traits defined and so on.
354 std::vector<Node*> sorted_nodes;
355 for (Node* node : graph.op_nodes()) {
356 sorted_nodes.push_back(node);
357 }
358 std::sort(sorted_nodes.begin(), sorted_nodes.end(), NodeCompare());
359
360 for (Node* node : sorted_nodes) {
361 VLOG(2) << "Fuel: " << fuel;
362 if (fuel <= 0) {
363 VLOG(2)
364 << "Hit fuel limit; not marking any remaining ops as clusterable.";
365 break;
366 }
367
368 VLOG(2) << "FindCompilationCandidates(): Processing "
369 << node->DebugString();
370
371 DeviceType device_type("");
372 TF_RETURN_IF_ERROR(
373 DeviceTypeOfDevice(node->assigned_device_name(), &device_type));
374
375 if (is_compilable_fn && !is_compilable_fn(node, device_type)) continue;
376
377 const XlaOpRegistry::DeviceRegistration* registration;
378 CHECK(
379 XlaOpRegistry::GetCompilationDevice(device_type.type(), &registration));
380 DeviceType jit_device_type(registration->compilation_device_name);
381 if (!HasXLAKernel(*node, jit_device_type) &&
382 !IsCompilableCall(node->def(), jit_device_type, 0, lib_runtime)) {
383 VLOG(2) << "Compilation rejected node: unsupported op " << node->name()
384 << ": " << node->type_string();
385 continue;
386 }
387 if (!registration->compile_resource_ops &&
388 HasResourceInputOrOutput(*node)) {
389 VLOG(2) << "Compilation rejected node: resource input/output "
390 << node->name() << ": " << node->type_string();
391 continue;
392 }
393 if (node->type_string() == "While" &&
394 !IsCompilableWhile(*node, jit_device_type, 0, lib_runtime)) {
395 continue;
396 }
397 // _Arg nodes in a top-level function represent feeds.
398 // Do not compile them.
399 if (node->type_string() == "_Arg") {
400 VLOG(2) << "Skipping jit compilation for '_Arg'-typed node "
401 << node->DebugString();
402 continue;
403 }
404 // _Retval nodes in a top-level function represent fetches.
405 // Do not compile them.
406 if (node->type_string() == "_Retval") {
407 VLOG(2) << "Compilation rejected node: return value " << node->name()
408 << ": " << node->type_string();
409 continue;
410 }
411 candidates->insert(node);
412 --fuel;
413 }
414 VLOG(2) << "candidates->size() = " << candidates->size();
415 return Status::OK();
416}

424// Returns a string describing how an edge from src to dst would
425// create a cycle.
426string DescribeCycle(const GraphCycles& cycles, const Graph& graph, int src,
427 int dst) {
428 int32 max_path_size = graph.num_node_ids() + 1;
429 std::vector<int32> path(max_path_size);
430 int32 path_size = cycles.FindPath(dst, src, max_path_size, path.data());
431 if (path_size == 0) {
432 return "";
433 }
434
435 auto node_name = [&cycles, &graph](int node_id) {
436 if (!FastBoundsCheck(node_id, graph.num_node_ids())) {
437 return string("(null)");
438 }
439 auto* node = graph.FindNodeId(node_id);
440 if (node == nullptr) {
441 return string("(null)");
442 }
443 return node->name();
444 };
445
446 string description;
447 strings::StrAppend(&description, "Edge from ", node_name(src), " to ",
448 node_name(dst), " would create a cycle.\n");
449 path.resize(path_size);
450 for (int32 node_id : path) {
451 string ascii_art;
452 if (node_id == dst) {
453 ascii_art = "+-> ";
454 } else if (node_id != src) {
455 ascii_art = "| ";
456 } else {
457 ascii_art = "+-- ";
458 }
459 strings::StrAppend(&description, ascii_art, node_name(node_id), "\n");
460 }
461 return description;
462}

466bool IsCompilable(FunctionLibraryRuntime* flr, const NodeDef& ndef) {
467 Device* device = flr->device();
468 const XlaOpRegistry::DeviceRegistration* registration;
469 CHECK(XlaOpRegistry::GetCompilationDevice(device->device_type(),
470 &registration));
471 DeviceType jit_device_type(registration->compilation_device_name);
472 return IsCompilableCall(ndef, jit_device_type, 0, flr);
473}

475Status MarkForCompilationPass::Run(
476 const GraphOptimizationPassOptions& options) {
477 // TODO(phawkins): precompute the "GetCompilationDevice" properties of each
478 // device ahead of time.
479 OptimizerOptions::GlobalJitLevel global_jit_level =
480 options.session_options->config.graph_options()
481 .optimizer_options()
482 .global_jit_level();
483 if (global_jit_level == OptimizerOptions::DEFAULT) {
484 // To set compilation to be on by default, change the following line.
485 global_jit_level = OptimizerOptions::OFF;
486 }
487 legacy_flags::MarkForCompilationPassFlags* flags =
488 legacy_flags::GetMarkForCompilationPassFlags();
489 if (flags->tf_xla_auto_jit == -1 ||
490 (1 <= flags->tf_xla_auto_jit && flags->tf_xla_auto_jit <= 2)) {
491 // If the flag tf_xla_auto_jit is a valid, non-zero setting, it overrides
492 // the setting in ConfigProto.
493 global_jit_level =
494 static_cast<OptimizerOptions::GlobalJitLevel>(flags->tf_xla_auto_jit);
495 }
496 bool cpu_global_jit = flags->tf_xla_cpu_global_jit;
497 bool fusion_only = flags->tf_xla_fusion_only;
498
499 VLOG(1) << "flags->tf_xla_cpu_global_jit = " << flags->tf_xla_cpu_global_jit;
500 VLOG(1) << "flags->tf_xla_fusion_only = " << flags->tf_xla_fusion_only;
501 const FunctionLibraryDefinition* fld = options.flib_def;
502
503 auto is_compilable = [global_jit_level, cpu_global_jit, fusion_only, fld](
504 const Node* node, const DeviceType& device_type) {
505 const XlaOpRegistry::DeviceRegistration* registration;
506 if (!XlaOpRegistry::GetCompilationDevice(device_type.type(),
507 &registration)) {
508 return false;
509 }
510
511 // Don't compile control trigger nodes. We won't preserve their deadness
512 // semantics correctly, so it's safest not to compile them.
513 if (node->IsControlTrigger()) return false;
514
515 // If this device requires a JIT, we must say yes.
516 if (registration->requires_compilation) return true;
517
518 // If there is a _XlaCompile annotation, use its value.
519 bool compile = false;
520 Status status = GetNodeAttr(node->attrs(), kXlaCompileAttr, &compile);
521 if (status.ok()) return compile;
522
523 status = fld->GetAttr(*node, kXlaCompileAttr, &compile);
524 if (status.ok()) return compile;
525
526 // Check for fusable ops only if requested.
527 if (global_jit_level > 0 && fusion_only && !IsXlaFusable(node->def())) {
528 return false;
529 }
530
531 // Otherwise use the value of global_jit_level.
532 // Ignore enable_jit_by_default if global jit compilation for CPU
533 // is explicitly requested via tf_xla_cpu_global_jit flag
534 bool ignore_registration = cpu_global_jit && device_type == DEVICE_CPU;
535 return (ignore_registration || registration->enable_jit_by_default) &&
536 global_jit_level > 0;
537 };
538 return RunImpl(options, is_compilable);
539}

551Status MarkForCompilationPass::RunImpl(
552 const GraphOptimizationPassOptions& options,
553 const std::function<bool(const Node*, const DeviceType&)>&
554 is_compilable_fn) {
555 VLOG(1) << "MarkForCompilationPass::Run";
556
557 // Make sure that kernels have been registered on the JIT device.
558 XlaOpRegistry::RegisterCompilationKernels();
559
560 Graph* graph = options.graph->get();
561
562 OrderedNodeSet compilation_candidates;
563 TF_RETURN_IF_ERROR(FindCompilationCandidates(
564 *graph, options.flib_def,
565 (options.session_options != nullptr) ? options.session_options->env
566 : Env::Default(),
567 is_compilable_fn, &compilation_candidates));
568
569 GraphCycles cycles;
570 for (int i = 0; i < graph->num_node_ids(); ++i) {
571 // We rely on the node IDs in the cycle detection graph being consecutive
572 // integers starting from 0.
573 CHECK_EQ(i, cycles.NewNode());
574 }
575
576 // Compute the loop structure of the graph.
577 std::vector<ControlFlowInfo> control_flow_info;
578 TF_RETURN_IF_ERROR(BuildControlFlowInfo(graph, &control_flow_info));
579
580 // The clustering code must avoid adding cycles to the graph to prevent
581 // deadlock. However, the graph may contain loops, which would trigger the
582 // cycle detection code. To handle loops, we alter the structure of the cycle
583 // detection graph, disconnecting each loop from the enclosing graph.
584 // Specifically, we:
585 // * add a new "frame" node for each loop.
586 // * replace edges to "Enter" nodes, and edges from "Exit" nodes with edges
587 // to/from the corresponding frame node. In essence, we collapse the loop
588 // into a single node for the purpose of cycle detection in the enclosing
589 // graph.
590 // * the body of the loop should now be disconnected from the rest of the
591 // graph; we make it acyclic by breaking loop backedges (edges outgoing from
592 // "NextIteration" nodes.
593
594 // Map from frame name strings to node IDs in the cycle detection graph.
595 std::unordered_map<string, int> frame_nodes;
596
597 // Get the cycle graph node ID for frame 'frame_name', or add one if none
598 // exists.
599 auto GetOrAddFrameNodeId = [&frame_nodes, &cycles](const string& frame_name) {
600 int& frame_id = frame_nodes.emplace(frame_name, -1).first->second;
601 if (frame_id < 0) {
602 // The emplace succeeded; we have not allocated a frame node yet.
603 frame_id = cycles.NewNode();
604 }
605 return frame_id;
606 };
607
608 for (Edge const* edge : graph->edges()) {
609 if (edge->dst()->IsEnter()) {
610 // Lift edges to an "Enter" node to the corresponding frame node.
611 const string& frame_name =
612 control_flow_info[edge->dst()->id()].frame_name;
613 int dst = GetOrAddFrameNodeId(frame_name);
614 if (!cycles.InsertEdge(edge->src()->id(), dst)) {
615 return errors::Internal(
616 "Cycle detected when adding enter->frame edge: ",
617 DescribeCycle(cycles, *graph, edge->src()->id(), dst));
618 }
619 continue;
620 }
621 if (edge->src()->IsExit()) {
622 // Lift edges from an "Exit" node to the corresponding frame node.
623 const string& frame_name =
624 control_flow_info[edge->src()->id()].frame_name;
625 int src = GetOrAddFrameNodeId(frame_name);
626 if (!cycles.InsertEdge(src, edge->dst()->id())) {
627 return errors::Internal(
628 "Cycle detected when adding frame->exit edge: ",
629 DescribeCycle(cycles, *graph, src, edge->dst()->id()));
630 }
631 // Drop the original edge.
632 continue;
633 }
634 if (edge->src()->IsNextIteration()) {
635 // Break loop back-edges.
636 continue;
637 }
638 if (!cycles.InsertEdge(edge->src()->id(), edge->dst()->id())) {
639 // This should never happen. All cycles in the graph should contain
640 // a control flow operator.
641 return errors::Internal(
642 "Found cycle in graph without control flow operator during XLA "
643 "compilation: ",
644 DescribeCycle(cycles, *graph, edge->src()->id(), edge->dst()->id()));
645 }
646 }
647
648 // Each compilation candidate belongs to a cluster. The cluster's
649 // representative
650 // names the node in the 'cycles' graph that represents the cluster.
651 std::vector<UnionFind<Cluster>> clusters(graph->num_node_ids());
652 std::deque<UnionFind<Cluster>*> worklist;
653 for (Node* node : compilation_candidates) {
654 Cluster& cluster = clusters[node->id()].Get();
655 cluster.representative = node->id();
656 worklist.push_back(&clusters[node->id()]);
657 }
658
659 legacy_flags::MarkForCompilationPassFlags* flags =
660 legacy_flags::GetMarkForCompilationPassFlags();
661
662 // Repeatedly contract edges between clusters that are on the same device,
663 // provided the contraction would not create a cycle.
664 while (!worklist.empty()) {
665 int from = worklist.front()->Get().representative;
666 worklist.pop_front();
667
668 Node* node_from = graph->FindNodeId(from);
669 if (node_from->IsControlFlow()) {
670 // Control flow nodes aren't compilation candidates and should never
671 // appear.
672 return errors::Internal(
673 "Found control flow node in clustering worklist: ",
674 node_from->type_string());
675 }
676 string from_scope;
677 string to_scope;
678 for (int to : cycles.Successors(from)) {
679 if (to >= graph->num_node_ids()) {
680 // Node is a "frame" node that is present only in the cycle detection
681 // graph. No clustering is possible.
682 continue;
683 }
684 Node* node_to = graph->FindNodeId(to);
685 if (compilation_candidates.find(node_to) ==
686 compilation_candidates.cend()) {
687 continue;
688 }
689 if (node_from->assigned_device_name() !=
690 node_to->assigned_device_name()) {
691 continue;
692 }
693 // Look for an _XlaScope on both nodes. If both nodes have a
694 // scope and the scopes do not match, do not cluster along this
695 // edge. If even one of the nodes lacks an _XlaScope attribute,
696 // then it is treated as a "bridge" and a cluster may be created
697 // along it. We may want to restrict this behavior to require
698 // all nodes marked with _XlaCompile=true to also have a
699 // _XlaScope property set (and raise an error otherwise); but
700 // for now we don't do this.
701 if (GetNodeAttr(node_from->attrs(), kXlaScopeAttr, &from_scope).ok() &&
702 GetNodeAttr(node_to->attrs(), kXlaScopeAttr, &to_scope).ok() &&
703 from_scope != to_scope) {
704 continue;
705 }
706
707 // Ops that consume shapes cannot be the root of a cluster. This is an
708 // optimization.
709 if (clusters[from].Size() == 1 && IsShapeConsumerOp(*node_from)) {
710 continue;
711 }
712
713 // Don't exceed the maximum cluster size.
714 if (clusters[from].Size() + clusters[to].Size() >
715 flags->tf_xla_max_cluster_size) {
716 continue;
717 }
718
719 // If contracting the edge would create a cycle, bail out.
720 // However, just because we can't merge the clusters now does not mean
721 // we won't be able to merge them in the future.
722 // e.g., if we have edges 1->2, 2->3 and 1->3, we cannot contract edge
723 // 1->3. But if we first contract 1->2 then we can later contract 1->3.
724 if (!cycles.ContractEdge(from, to)) continue;
725
726 // Merge the clusters. ContractEdge uses 'from' as the number of the
727 // merged node, so make sure 'from' is the chosen representative.
728 clusters[from].Merge(&clusters[to]);
729
730 worklist.push_back(&clusters[from]);
731 break;
732 }
733 }
734
735 // Count the number of non-trivial elements in each cluster.
736 std::vector<int> effective_cluster_sizes(graph->num_node_ids());
737 for (const Node* n : compilation_candidates) {
738 int cluster = clusters[n->id()].Get().representative;
739 // Identity nodes will be removed if the node gets marked for compilation.
740 // Therefore we don't want to count them towards the effective cluster size.
741 if (n->def().op() != "Identity") {
742 effective_cluster_sizes[cluster]++;
743 }
744 }
745
746 // Names for each cluster.
747 std::unordered_map<int, string> cluster_names;
748
749 // Mark clusters for compilation that:
750 // * are placed on a device that requires compilation (an XlaDevice),
751 // * are explicitly marked for compilation (_XlaCompile=true), or
752 // * have more than flags->tf_xla_min_cluster_size elements (applicable only
753 // if compilation is enabled, otherwise there will be no such candidates).
754 const int min_cluster_size = flags->tf_xla_min_cluster_size;
755 for (Node* n : compilation_candidates) {
756 int cluster = clusters[n->id()].Get().representative;
757
758 // Compile if the user marked this node _XlaCompile=true
759 bool compile_attr = false;
760 bool marked_for_compilation = false;
761 if (GetNodeAttr(n->attrs(), kXlaCompileAttr, &compile_attr).ok()) {
762 marked_for_compilation = compile_attr;
763 } else if (options.flib_def->GetAttr(*n, kXlaCompileAttr, &compile_attr)
764 .ok()) {
765 marked_for_compilation = compile_attr;
766 }
767
768 // Compile if this operator is placed on a device that requires
769 // compilation.
770 DeviceType device_type("");
771 TF_RETURN_IF_ERROR(
772 DeviceTypeOfDevice(n->assigned_device_name(), &device_type));
773 const XlaOpRegistry::DeviceRegistration* registration;
774 XlaOpRegistry::GetCompilationDevice(device_type.type(), &registration);
775
776 // Compile if this is a cluster of >= min_cluster_size compilable operators.
777 // Also, always compile if the operator is placed on a device that requires
778 // compilation, or if it contains at least one op that is marked for
779 // compilation that is not an Identity op.
780 if (effective_cluster_sizes[cluster] >= min_cluster_size ||
781 (effective_cluster_sizes[cluster] > 0 && marked_for_compilation) ||
782 registration->requires_compilation) {
783 string& name = cluster_names[cluster];
784
785 if (name.empty()) {
786 name = strings::StrCat("cluster_", cluster_sequence_num++);
787 }
788 n->AddAttr(kXlaClusterAttr, name);
789 VLOG(3) << "Assigning node " << n->name() << " to cluster " << name;
790 }
791 }
792
793 if (flags->tf_xla_clustering_debug) {
794 dump_graph::DumpGraphToFile("mark_for_compilation", **options.graph,
795 options.flib_def);
796 }
797 return Status::OK();
798}
