19// Legacy flags for the XLA bridge's xla_device module.
20
21#include <vector>
22
23#include "tensorflow/core/platform/types.h"
24#include "tensorflow/core/util/command_line_flags.h"
25
26namespace tensorflow {
27namespace legacy_flags {
28
29// The values of flags associated with the XLA bridge's
30// xla_device module.
31typedef struct {
32 // Switch the CPU device into "on-demand" mode, where instead of
33 // autoclustering ops are compiled one by one just-in-time.
34 // Enabling this mode by a legacy flag is a temporary mechanism. When this
35 // feature is battle-tested, we will switch this to be a session option.
36 bool tf_xla_compile_on_demand;
37} XlaDeviceFlags;
38
39// Return a pointer to the XlaDeviceFlags struct;
40// repeated calls return the same pointer.
41// This should be called only after Flags::Parse() has returned.
42XlaDeviceFlags* GetXlaDeviceFlags();
43
44} // namespace legacy_flags
45} // namespace tensorflow
46
