16// Legacy flags for the XLA bridge's mark_for_compilation_pass module.
17
18#include <mutex>
19#include <vector>
20
21#include "tensorflow/compiler/jit/legacy_flags/mark_for_compilation_pass_flags.h"
22#include "tensorflow/compiler/xla/legacy_flags/parse_flags_from_env.h"
23#include "tensorflow/core/platform/types.h"
24#include "tensorflow/core/util/command_line_flags.h"

26namespace tensorflow {
27namespace legacy_flags {
28
29// Pointers to the parsed value of the flags and flag descriptors, initialized
30// via flags_init.
31static MarkForCompilationPassFlags* flags;
32static std::vector<Flag>* flag_list;
33static std::once_flag flags_init;
34
35// Allocate *flags. Called via call_once(&flags_init,...).
36static void AllocateFlags() {
37 flags = new MarkForCompilationPassFlags;
38 flags->tf_xla_auto_jit = 0;
39 flags->tf_xla_min_cluster_size = 2;
40 flags->tf_xla_max_cluster_size = std::numeric_limits<int32>::max();
41 flags->tf_xla_clustering_debug = false;
42 flags->tf_xla_cpu_global_jit = false;
43 flags->tf_xla_clustering_fuel = std::numeric_limits<int64>::max();
44 flags->tf_xla_fusion_only = false;
45 flag_list = new std::vector<Flag>(
46 {Flag("tf_xla_auto_jit", &flags->tf_xla_auto_jit,
47 "Control compilation of operators into XLA computations on CPU and "
48 "GPU devices. 0 = use ConfigProto setting; -1 = off; 1 = on for "
49 "things very likely to be improved; 2 = on for everything. "
50 "Experimental."),
51 Flag("tf_xla_min_cluster_size", &flags->tf_xla_min_cluster_size,
52 "Minimum number of operators in an XLA compilation. Ignored for "
53 "operators placed on an XLA device or operators explicitly marked "
54 "for compilation."),
55 Flag("tf_xla_max_cluster_size", &flags->tf_xla_max_cluster_size,
56 "Maximum number of operators in an XLA compilation."),
57 Flag("tf_xla_clustering_debug", &flags->tf_xla_clustering_debug,
58 "Dump graphs during XLA compilation."),
59 Flag("tf_xla_cpu_global_jit", &flags->tf_xla_cpu_global_jit,
60 "Enables global JIT compilation for CPU via SessionOptions."),
61 Flag("tf_xla_clustering_fuel", &flags->tf_xla_clustering_fuel,
62 "Places an artificial limit on the number of ops marked as "
63 "eligible for clustering."),
64 Flag("tf_xla_fusion_only", &flags->tf_xla_fusion_only,
65 "enable fusion of element-wise operations only using XLA when "
66 "global_jit_level is ON*.")});
67 xla::legacy_flags::ParseFlagsFromEnv(*flag_list);
68}
69
70// Append to *append_to flag definitions associated with the XLA bridge's
71// mark_for_compilation_pass module.
72void AppendMarkForCompilationPassFlags(std::vector<Flag>* append_to) {
73 std::call_once(flags_init, &AllocateFlags);
74 append_to->insert(append_to->end(), flag_list->begin(), flag_list->end());
75}
76
77// Return a pointer to the MarkForCompilationPassFlags struct;
78// repeated calls return the same pointer.
79// This should be called only after Flags::Parse() has returned.
80MarkForCompilationPassFlags* GetMarkForCompilationPassFlags() {
81 std::call_once(flags_init, &AllocateFlags);
82 return flags;
83}
