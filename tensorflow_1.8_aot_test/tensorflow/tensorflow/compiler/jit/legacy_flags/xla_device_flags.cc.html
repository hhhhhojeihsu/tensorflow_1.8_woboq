16// Legacy flags for the XLA bridge's xla_device module.
17
18#include <mutex>
19#include <vector>
20
21#include "tensorflow/compiler/jit/legacy_flags/xla_device_flags.h"
22#include "tensorflow/compiler/xla/legacy_flags/parse_flags_from_env.h"
23#include "tensorflow/core/platform/types.h"
24#include "tensorflow/core/util/command_line_flags.h"
25
26namespace tensorflow {
27namespace legacy_flags {
28
29// Pointers to the parsed value of the flags and flag descriptors, initialized
30// via flags_init.
31static XlaDeviceFlags* flags;
32static std::vector<Flag>* flag_list;
33static std::once_flag flags_init;
34
35// Allocate *flags. Called via call_once(&flags_init,...).
36static void AllocateFlags() {
37 flags = new XlaDeviceFlags;
38 flags->tf_xla_compile_on_demand = false;
39 flag_list = new std::vector<Flag>({
40 Flag("tf_xla_compile_on_demand", &flags->tf_xla_compile_on_demand,
41 "Switch a device into 'on-demand' mode, where instead of "
42 "autoclustering ops are compiled one by one just-in-time."),
43 });
44 xla::legacy_flags::ParseFlagsFromEnv(*flag_list);
45}
46
47// Return a pointer to the XlaDeviceFlags struct;
48// repeated calls return the same pointer.
49// This should be called only after Flags::Parse() has returned.
50XlaDeviceFlags* GetXlaDeviceFlags() {
51 std::call_once(flags_init, &AllocateFlags);
52 return flags;
53}
