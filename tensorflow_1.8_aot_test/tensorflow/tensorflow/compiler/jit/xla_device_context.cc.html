30// The allocator used for Tensors assigned to the XLA device.
31XlaDeviceAllocator::XlaDeviceAllocator() {}
32XlaDeviceAllocator::~XlaDeviceAllocator() = default;
33
34string XlaDeviceAllocator::Name() { return "xla"; }
35
36void* XlaDeviceAllocator::AllocateRaw(size_t alignment, size_t num_bytes) {
37 // We always return an empty XlaTensor object, encoded as an opaque tagged
38 // pointer. We can return an empty object and ignore num_bytes here because we
39 // have control over all of the uses of this device tensor, and can lazily
40 // allocate memory when used. This allows us to also know the shape of the
41 // allocated Tensor, which is useful if the device's tensor representation
42 // differs from the host.
43 return XlaTensor::ToOpaquePointer(new XlaTensor());
44}
45
46void XlaDeviceAllocator::DeallocateRaw(void* ptr) {
47 delete XlaTensor::FromOpaquePointer(ptr);
48}

52XlaTransferManager::XlaTransferManager(se::Stream* stream,
53 xla::LocalClient* client,
54 bool transfer_as_literal)
55 : stream_(stream),
56 client_(client),
57 transfer_manager_(client->backend().transfer_manager()),
58 transfer_as_literal_(transfer_as_literal) {}
59
60Status XlaTransferManager::TransferLiteralToDevice(
61 const Tensor& host_tensor, Tensor* device_tensor) const {
62 xla::Literal literal;
63 TF_RETURN_IF_ERROR(HostTensorToLiteral(host_tensor, &literal));
64 VLOG(1) << "Transfer to device as literal: " << literal.ToString();
65
66 const xla::ShapedBuffer& shaped_buffer =
67 XlaTensor::FromTensor(device_tensor)->shaped_buffer();
68 return transfer_manager_->TransferLiteralToDevice(stream_->parent(), literal,
69 shaped_buffer);
70}
71
72Status XlaTransferManager::TransferLiteralFromDevice(
73 Tensor* host_tensor, const Tensor& device_tensor) const {
74 const xla::ShapedBuffer& shaped_buffer =
75 XlaTensor::FromTensor(&device_tensor)->shaped_buffer();
76
77 TF_ASSIGN_OR_RETURN(std::unique_ptr<xla::Literal> literal,
78 transfer_manager_->TransferLiteralFromDevice(
79 stream_->parent(), shaped_buffer));
80 VLOG(1) << "Transfer from device as literal: " << literal->ToString();
81 return LiteralToHostTensor(*literal, host_tensor->dtype(), host_tensor);
82}

84void XlaTransferManager::CopyCPUTensorToDevice(const Tensor* cpu_tensor,
85 Device* device,
86 Tensor* device_tensor,
87 StatusCallback done) const {
88 if (cpu_tensor->NumElements() > 0) {
89 VLOG(2) << "CopyCPUTensorToDevice "
90 << reinterpret_cast<const void*>(cpu_tensor->tensor_data().data())
91 << " "
92 << reinterpret_cast<const void*>(
93 device_tensor->tensor_data().data())
94 << " " << cpu_tensor->NumElements();
95
96 void* src_ptr = const_cast<void*>(DMAHelper::base(cpu_tensor));
97 const int64 total_bytes = cpu_tensor->TotalBytes();
98
99 XlaTensor* xla_tensor = XlaTensor::FromTensor(device_tensor);
100 CHECK(xla_tensor);
101 if (!xla_tensor->has_shaped_buffer()) {
102 Status s = xla_tensor->AllocateShapedBuffer(
103 device_tensor->dtype(), device_tensor->shape(), client_,
104 stream_->parent()->device_ordinal());
105 if (!s.ok()) {
106 done(s);
107 return;
108 }
109 }
110
111 se::DeviceMemoryBase dev_dst_ptr =
112 XlaTensor::DeviceMemoryFromTensor(*device_tensor);
113 Status status;
114 if (transfer_as_literal_) {
115 status = TransferLiteralToDevice(*cpu_tensor, device_tensor);
116 } else {
117 stream_->ThenMemcpy(&dev_dst_ptr, src_ptr, total_bytes);
118 // TODO(hpucha): Make this asynchronous.
119 Status block_status = stream_->BlockHostUntilDone();
120 if (!block_status.ok()) {
121 status = xla::InternalError(
122 "Failed to complete data transfer on stream %p: %s", stream_,
123 block_status.error_message().c_str());
124 }
125 }
126 xla_tensor->set_host_tensor(*cpu_tensor);
127
128 done(status);
129 return;
130 }
131
132 VLOG(2) << "CopyCPUTensorToDevice empty tensor";
133 done(Status::OK());
134}

136void XlaTransferManager::CopyDeviceTensorToCPU(const Tensor* device_tensor,
137 StringPiece tensor_name,
138 Device* device,
139 Tensor* cpu_tensor,
140 StatusCallback done) {
141 if (device_tensor->NumElements() > 0) {
142 VLOG(2) << "CopyDeviceTensorToCPU "
143 << reinterpret_cast<const void*>(
144 device_tensor->tensor_data().data())
145 << " "
146 << reinterpret_cast<const void*>(cpu_tensor->tensor_data().data())
147 << device_tensor->NumElements();
148
149 const int64 total_bytes = cpu_tensor->TotalBytes();
150 se::DeviceMemoryBase dev_src_ptr =
151 XlaTensor::DeviceMemoryFromTensor(*device_tensor);
152 void* dst_ptr = DMAHelper::base(cpu_tensor);
153
154 Status status;
155 if (transfer_as_literal_) {
156 status = TransferLiteralFromDevice(cpu_tensor, *device_tensor);
157 } else {
158 stream_->ThenMemcpy(dst_ptr, dev_src_ptr, total_bytes);
159 // TODO(hpucha): Make this asynchronous.
160 Status block_status = stream_->BlockHostUntilDone();
161 if (!block_status.ok()) {
162 status = xla::InternalError(
163 "Failed to complete data transfer on stream %p: %s", stream_,
164 block_status.error_message().c_str());
165 }
166 }
167
168 done(status);
169 return;
170 }
171
172 VLOG(2) << "CopyDeviceTensorToCPU empty tensor";
173 done(Status::OK());
174}
