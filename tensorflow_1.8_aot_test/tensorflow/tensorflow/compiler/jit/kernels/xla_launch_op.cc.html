44XlaLocalLaunchOp::XlaLocalLaunchOp(OpKernelConstruction* ctx)
45 : OpKernel(ctx), device_type_(ctx->device_type()) {
46 const NameAttrList* func;
47 OP_REQUIRES_OK(ctx, ctx->GetAttr("function", &func));
48 function_ = *func;
49 DataTypeVector constant_types;
50 OP_REQUIRES_OK(ctx, ctx->GetAttr("Tconstants", &constant_types));
51 num_constant_args_ = constant_types.size();
52 OP_REQUIRES_OK(ctx, ctx->GetAttr("Nresources", &num_resource_args_));
53 if (device_type_ == DeviceType(DEVICE_CPU)) {
54 platform_id_ = gpu::host::kHostPlatformId;
55 } else if (device_type_ == DeviceType(DEVICE_GPU)) {
56 platform_id_ = gpu::cuda::kCudaPlatformId;
57 } else {
58 platform_id_ = nullptr;
59 }
60}

62Status XlaLocalLaunchOp::BuildCompilationCache(OpKernelContext* ctx,
63 XlaCompilationCache** cache) {
64 const XlaDevice::Metadata* metadata;
65 Status s = XlaDevice::GetMetadata(ctx, &metadata);
66 if (s.ok()) {
67 *cache = new XlaCompilationCache(metadata->client(),
68 metadata->jit_device_type());
69 return Status::OK();
70 }
71
72 auto platform = gpu::MultiPlatformManager::PlatformWithId(platform_id_);
73 if (!platform.ok()) {
74 return StreamExecutorUtil::ConvertStatus(platform.status());
75 }
76 xla::LocalClientOptions client_options;
77 client_options.set_platform(platform.ValueOrDie());
78 client_options.set_intra_op_parallelism_threads(
79 ctx->device()->tensorflow_cpu_worker_threads()->num_threads);
80 auto client = xla::ClientLibrary::GetOrCreateLocalClient(client_options);
81 if (!client.ok()) {
82 return client.status();
83 }
84 const XlaOpRegistry::DeviceRegistration* registration;
85 if (!XlaOpRegistry::GetCompilationDevice(device_type_.type(),
86 &registration)) {
87 return errors::InvalidArgument("No JIT device registered for ",
88 device_type_.type());
89 }
90 *cache = new XlaCompilationCache(
91 client.ValueOrDie(), DeviceType(registration->compilation_device_name));
92 return Status::OK();
93}

95void XlaLocalLaunchOp::Compute(OpKernelContext* ctx) {
96 VLOG(1) << "XlaLocalLaunchOp::Compute "
97 << Canonicalize(function_.name(), AttrSlice(&function_.attr()));
98 // We store information about the JIT-compiled XLA computation
99 // in the ResourceMgr.
100 ResourceMgr* rm = ctx->resource_manager();
101 OP_REQUIRES(ctx, rm, errors::Internal("No resource manager."));
102
103 gpu::Stream* stream =
104 ctx->op_device_context() ? ctx->op_device_context()->stream() : nullptr;
105
106 XlaCompilationCache* cache;
107 OP_REQUIRES_OK(ctx, rm->LookupOrCreate<XlaCompilationCache>(
108 rm->default_container(), "xla_cache", &cache,
109 [this, ctx](XlaCompilationCache** cache) {
110 return BuildCompilationCache(ctx, cache);
111 }));
112 // Hold the reference to the JIT during evaluation. (We could probably
113 // free it sooner because the ResourceMgr will retain a reference, but
114 // this is more obviously correct.)
115 core::ScopedUnref cache_ref(cache);
116
117 const XlaDevice::Metadata* metadata;
118 Status s = XlaDevice::GetMetadata(ctx, &metadata);
119 bool allocate_xla_tensors = s.ok();
120
121 // Get the platform_id_ for XLA_* devices.
122 if (platform_id_ == nullptr) {
123 if (s.ok()) {
124 platform_id_ = metadata->platform()->id();
125 }
126 }
127
128 std::map<int, OptionalTensor> variables =
129 SnapshotResourceVariables(ctx, num_resource_args_);
130
131 xla::LocalClient* client = static_cast<xla::LocalClient*>(cache->client());
132
133 XlaAllocator local_xla_allocator(client->backend().platform(),
134 ctx->device()->GetAllocator({}));
135 xla::DeviceMemoryAllocator* xla_allocator;
136 // If we are on an XlaDevice, use the underlying XLA platform's allocator
137 // directly. We could use the StreamExecutor's allocator which may
138 // theoretically be more correct, but XLA returns a nice OOM message in a
139 // Status and StreamExecutor does not.
140 //
141 // Importantly we can't use ctx->device()->GetAllocator() as the allocator
142 // (which local_xla_allocator above uses) as on an XlaDevice, this is a
143 // dummy allocator that returns XlaTensor objects. The XlaCompiler needs a
144 // real allocator to allocate real buffers.
145 if (allocate_xla_tensors) {
146 xla_allocator = client->backend().memory_allocator();
147 } else {
148 xla_allocator = &local_xla_allocator;
149 }
150
151 XlaCompiler::Options options;
152 options.client = client;
153 options.device_type = &cache->device_type();
154 options.flib_def = ctx->function_library()->GetFunctionLibraryDefinition();
155 options.graph_def_version = ctx->function_library()->graph_def_version();
156 options.allow_cpu_custom_calls = (platform_id_ == gpu::host::kHostPlatformId);
157 options.device_allocator = xla_allocator;
158 // TODO(b/77671268): We don't set variable_representation_shape_fn here. This
159 // is restricted to Variables, but we need something like this to apply to
160 // normal Tensors too.
161
162 const XlaCompiler::CompilationResult* kernel;
163 xla::LocalExecutable* executable;
164
165 std::map<int, Tensor> constant_args;
166 for (int i = 0; i < num_constant_args_; ++i) {
167 constant_args.insert({i, ctx->input(i)});
168 }
169 OP_REQUIRES_OK(ctx, cache->Compile(options, function_, constant_args,
170 variables, ctx, &kernel, &executable,
171 /*compile_options=*/nullptr));
172
173 VLOG(1) << "Executing XLA Computation...";
174
175 XlaComputationLaunchContext launch_context(
176 num_resource_args_, client, xla_allocator, allocate_xla_tensors);
177 launch_context.PopulateInputs(ctx, kernel, variables);
178
179 // Execute the computation.
180 VLOG(2) << "Executing computation.";
181 xla::ExecutableRunOptions run_options;
182 run_options.set_stream(stream);
183 run_options.set_allocator(xla_allocator);
184 run_options.set_intra_op_thread_pool(&ctx->eigen_cpu_device());
185 run_options.set_rng_seed(ctx->step_id());
186 Env* env = Env::Default();
187 auto start_time = env->NowMicros();
188
189 auto run_result = executable->Run(launch_context.arguments(), run_options);
190 OP_REQUIRES(ctx, run_result.ok(), run_result.status());
191
192 auto elapsed = env->NowMicros() - start_time;
193 VLOG(2) << "Elapsed time: " << elapsed << "us";
194
195 launch_context.PopulateOutputs(ctx, kernel, run_result.ConsumeValueOrDie());
196 VLOG(1) << "Done";
197}
