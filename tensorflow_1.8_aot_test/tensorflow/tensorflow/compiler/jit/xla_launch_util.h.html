34// Takes a snapshot of the values of resource variable arguments, which are
35// the last `num_variables` arguments. We snapshot tensors that back
36// resource variables since concurrent updates may modify the shape, and it is
37// important that the shapes used for compilation match the true shapes of the
38// buffers.
39//
40// Returns a map of TensorFlow argument index to resource variable.
41std::map<int, OptionalTensor> SnapshotResourceVariables(OpKernelContext* ctx,
42 int num_variables);

44// Adapter class that wraps a Tensorflow allocator as an XLA allocator.
45// Assumes that the Tensorflow allocator permits asynchronous deallocation:
46// see comment on `AllowsAsynchronousDeallocation()`.
47class XlaAllocator : public xla::DeviceMemoryAllocator {
48 public:
49 XlaAllocator(const perftools::gputools::Platform* platform,
50 Allocator* wrapped);
51 ~XlaAllocator() override;
52 xla::StatusOr<perftools::gputools::DeviceMemoryBase> Allocate(
53 int device_ordinal, uint64 size, bool retry_on_failure) override;
54 Status Deallocate(int device_ordinal,
55 perftools::gputools::DeviceMemoryBase* mem) override;
56
57 // The Tensorflow BFC allocator used on GPU allows host-side deallocation
58 // before GPU execution takes place. Tensorflow uses the ordering of the main
59 // compute stream to enforce a happens-before relationship between a memory
60 // allocation and code that reuses the same memory. If Tensorflow adds
61 // support for multiple GPU streams or allocators with different ordering
62 // requirements, this code may need to change.
63 // (This attribute has no effect on CPU.)
64 bool AllowsAsynchronousDeallocation() const override { return true; }
65
66 private:
67 Allocator* wrapped_;
68};

70// Helper class to perform the marshalling of TensorFlow inputs and outputs to
71// ShapedBuffers suitable for passing to an XLA computation.
72class XlaComputationLaunchContext {
73 public:
74 // Create a new launch context. 'allocate_xla_tensors' is true if allocated
75 // output tensors and variables are always XlaTensors. If false they are
76 // assumed to be "normal" device pointers.
77 XlaComputationLaunchContext(int64 num_resource_args, xla::LocalClient* client,
78 xla::DeviceMemoryAllocator* xla_allocator,
79 bool allocate_xla_tensors);
80
81 // Add all inputs within `ctx` as XLA arguments (returned by arguments()).
82 // `variables` is a map from TensorFlow argument number to resource variable.
83 void PopulateInputs(OpKernelContext* ctx,
84 const XlaCompiler::CompilationResult* kernel,
85 const std::map<int, OptionalTensor>& variables);
86
87 // Given the XLA output in `output`, populate all outputs of `ctx`.
88 void PopulateOutputs(OpKernelContext* ctx,
89 const XlaCompiler::CompilationResult* kernel,
90 std::unique_ptr<xla::ScopedShapedBuffer> output);
91
92 // Return the argument list. Only valid after PopulateInputs() has been
93 // called.
94 const std::vector<xla::ShapedBuffer*>& arguments() const { return arg_ptrs_; }
95
96 private:
97 int64 num_resource_args_;
98 xla::LocalClient* client_;
99 xla::DeviceMemoryAllocator* xla_allocator_;
100 bool allocate_xla_tensors_;
101 std::vector<std::unique_ptr<xla::ShapedBuffer>> arg_buffers_;
102 std::vector<xla::ShapedBuffer*> arg_ptrs_;
103};

105// A simple TensorBuffer implementation that allows us to create Tensors that
106// take ownership of pre-allocated memory.
107class XlaTensorBuffer : public TensorBuffer {
108 public:
109 XlaTensorBuffer(const void* ptr, size_t expected_size, size_t actual_size,
110 Allocator* allocator)
111 : expected_size_(expected_size),
112 actual_size_(actual_size),
113 allocator_(allocator) {
114 data_ = const_cast<void*>(ptr);
115 }
116
117 ~XlaTensorBuffer() override { allocator_->DeallocateRaw(data_); }
118
119 void* data() const override { return data_; }
120 size_t size() const override { return expected_size_; }
121
122 TensorBuffer* root_buffer() override { return this; }
123
124 void FillAllocationDescription(AllocationDescription* proto) const override {
125 proto->set_allocated_bytes(actual_size_);
126 }
127
128 static Tensor MakeTensor(DataType dtype, const TensorShape& shape,
129 perftools::gputools::DeviceMemoryBase buffer,
130 Allocator* allocator) {
131 size_t expected_size = shape.num_elements() * DataTypeSize(dtype);
132 auto* tensor_buffer = new XlaTensorBuffer(buffer.opaque(), expected_size,
133 buffer.size(), allocator);
134 Tensor t(dtype, shape, tensor_buffer);
135 tensor_buffer->Unref();
136 return t;
137 }
138
139 private:
140 void* data_;
141 size_t expected_size_;
142 size_t actual_size_;
143 Allocator* allocator_;
144};
