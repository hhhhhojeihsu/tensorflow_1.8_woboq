49// Compute a string signature which encodes the shapes of the
50// arguments in the supplied list.
51string XlaCompilationCache::SignatureDebugString(const Signature& sig) {
52 string result = sig.name;
53 for (const auto& a : sig.arg_types) {
54 strings::StrAppend(&result, ",", DataTypeString(a.first),
55 a.second.DebugString());
56 }
57
58 for (const auto& v : sig.arg_values) {
59 strings::StrAppend(&result, "; ", v.DebugString());
60 }
61 return result;
62}

64bool XlaCompilationCache::Signature::operator==(const Signature& other) const {
65 if (name != other.name) return false;
66 if (arg_types != other.arg_types) return false;
67
68 if (arg_values.size() != other.arg_values.size()) return false;
69 for (int i = 0; i < arg_values.size(); ++i) {
70 if (arg_values[i].tensor_data() != other.arg_values[i].tensor_data()) {
71 return false;
72 }
73 }
74 return true;
75}

77uint64 XlaCompilationCache::Signature::Hash::operator()(
78 const XlaCompilationCache::Signature& signature) const {
79 uint64 h = std::hash<string>()(signature.name);
80 for (const auto& arg : signature.arg_types) {
81 h = Hash64Combine(h, std::hash<int>()(static_cast<int>(arg.first)));
82 h = Hash64Combine(h, std::hash<int>()(arg.second.dims()));
83 for (int dim : arg.second.dim_sizes()) {
84 h = Hash64Combine(h, std::hash<int>()(dim));
85 }
86 }
87 for (const auto& arg : signature.arg_values) {
88 h = Hash64Combine(
89 h, Hash64(arg.tensor_data().data(), arg.tensor_data().size()));
90 }
91 return h;
92}

94Status XlaCompilationCache::BuildSignature(
95 const NameAttrList& function, const std::map<int, Tensor>& constant_args,
96 const std::map<int, OptionalTensor>& variable_args, OpKernelContext* ctx,
97 Signature* signature) {
98 signature->name = Canonicalize(function.name(), AttrSlice(&function.attr()));
99 signature->arg_values.reserve(constant_args.size());
100
101 signature->arg_types.reserve(ctx->num_inputs() - constant_args.size());
102
103 for (int i = 0; i < ctx->num_inputs(); ++i) {
104 if (constant_args.count(i) > 0) {
105 // Use the values of compile time constants in the signature.
106 signature->arg_values.push_back(constant_args.at(i));
107 } else if (variable_args.count(i) > 0) {
108 const OptionalTensor& variable = variable_args.at(i);
109 if (variable.present) {
110 signature->arg_types.emplace_back(variable.value.dtype(),
111 variable.value.shape());
112 } else {
113 signature->arg_types.emplace_back(DT_INVALID, TensorShape());
114 }
115 } else {
116 signature->arg_types.emplace_back(ctx->input_dtype(i),
117 ctx->input(i).shape());
118 }
119 }
120 return Status::OK();
121}

125// Builds a XlaCompiler::Argument vector from the arguments to the _XlaLaunch
126// op.
127Status BuildArguments(const std::map<int, Tensor>& constant_args,
128 const std::map<int, OptionalTensor>& variable_args,
129 OpKernelContext* ctx,
130 std::vector<XlaCompiler::Argument>* args) {
131 args->resize(ctx->num_inputs());
132
133 for (int64 input_num = 0; input_num < ctx->num_inputs(); ++input_num) {
134 XlaCompiler::Argument& arg = (*args)[input_num];
135 if (constant_args.count(input_num) > 0) {
136 // Handles compile-time constants.
137 const Tensor& input = constant_args.at(input_num);
138 TF_RET_CHECK(input.dtype() != DT_RESOURCE);
139 arg.kind = XlaCompiler::Argument::kConstant;
140 arg.type = input.dtype();
141 arg.shape = input.shape();
142 arg.constant_value = input;
143 } else if (variable_args.count(input_num) == 0) {
144 // Handles the non-constant arguments.
145 const Tensor& input = ctx->input(input_num);
146 TF_RET_CHECK(input.dtype() != DT_RESOURCE);
147 if (input.NumElements() > 0) {
148 arg.kind = XlaCompiler::Argument::kParameter;
149 } else {
150 arg.kind = XlaCompiler::Argument::kConstant;
151 arg.constant_value = input;
152 }
153 arg.type = input.dtype();
154 arg.shape = input.shape();
155 } else {
156 // Handles resource variables.
157 const Tensor& input = ctx->input(input_num);
158 TF_RET_CHECK(input.dtype() == DT_RESOURCE);
159 const OptionalTensor& variable = variable_args.at(input_num);
160 arg.name = variable.name;
161 arg.kind = XlaCompiler::Argument::kResource;
162 arg.resource_kind = XlaResource::kVariable;
163 if (variable.present) {
164 const Tensor& value = variable.value;
165 arg.type = value.dtype();
166 arg.shape = value.shape();
167 arg.initialized = true;
168 } else {
169 // The values of uninitialized variables are not passed as inputs, since
170 // they are meaningless. However, it is legal to assign to a resource
171 // variable for the first time inside the XLA computation, so we do
172 // permit uninitialized variables.
173 arg.initialized = false;
174 arg.type = DT_INVALID;
175 arg.shape = TensorShape();
176 }
177 }
178 }
179
180 return Status::OK();
181}

185Status XlaCompilationCache::BuildExecutable(
186 const XlaCompiler::Options& options,
187 const XlaCompiler::CompilationResult& result,
188 std::unique_ptr<xla::LocalExecutable>* executable) {
189 VLOG(2) << "Compiling to local executable";
190
191 std::vector<const xla::Shape*> argument_layouts(
192 result.xla_input_shapes.size());
193 for (int i = 0; i < result.xla_input_shapes.size(); ++i) {
194 argument_layouts[i] = &result.xla_input_shapes[i];
195 }
196 xla::ExecutableBuildOptions build_options;
197 build_options.set_device_ordinal(client_->default_device_ordinal());
198 build_options.set_result_layout(result.xla_output_shape);
199 build_options.set_device_allocator(options.device_allocator);
200
201 auto compile_result =
202 client_->Compile(*result.computation, argument_layouts, build_options);
203 if (!compile_result.ok()) {
204 return compile_result.status();
205 }
206 *executable = std::move(compile_result.ValueOrDie());
207 return Status::OK();
208}

210Status XlaCompilationCache::Compile(
211 const XlaCompiler::Options& options, const NameAttrList& function,
212 const std::map<int, Tensor>& constant_args,
213 const std::map<int, OptionalTensor>& variable_args, OpKernelContext* ctx,
214 const XlaCompiler::CompilationResult** compilation_result,
215 xla::LocalExecutable** executable,
216 const XlaCompiler::CompileOptions* compile_options) {
217 return CompileImpl(options, function, constant_args, variable_args, ctx,
218 compilation_result, executable, compile_options, false);
219}

221Status XlaCompilationCache::CompileSingleOp(
222 const XlaCompiler::Options& options,
223 const std::map<int, Tensor>& constant_args,
224 const std::map<int, OptionalTensor>& variable_args, OpKernelContext* ctx,
225 const XlaCompiler::CompilationResult** compilation_result,
226 xla::LocalExecutable** executable,
227 const XlaCompiler::CompileOptions* compile_options) {
228 const NodeDef& def = ctx->op_kernel().def();
229 NameAttrList name;
230 name.set_name(def.op());
231 *name.mutable_attr() = def.attr();
232 return CompileImpl(options, name, constant_args, variable_args, ctx,
233 compilation_result, executable, compile_options, true);
234}

236Status XlaCompilationCache::CompileImpl(
237 const XlaCompiler::Options& options, const NameAttrList& function,
238 const std::map<int, Tensor>& constant_args,
239 const std::map<int, OptionalTensor>& variable_args, OpKernelContext* ctx,
240 const XlaCompiler::CompilationResult** compilation_result,
241 xla::LocalExecutable** executable,
242 const XlaCompiler::CompileOptions* compile_options,
243 bool compile_single_op) {
244 VLOG(1) << "XlaCompilationCache::Compile " << DebugString();
245
246 if (VLOG_IS_ON(2)) {
247 VLOG(2) << "num_inputs=" << ctx->num_inputs()
248 << " num_constant_args=" << constant_args.size()
249 << " num_variable_args=" << variable_args.size();
250 for (int i = 0; i < ctx->num_inputs(); i++) {
251 TensorShape shape = ctx->input(i).shape();
252 VLOG(2) << i << ": dtype=" << DataTypeString(ctx->input_dtype(i))
253 << " present=" << ctx->has_input(i)
254 << " shape=" << shape.DebugString();
255 }
256 for (auto& iterator : variable_args) {
257 const OptionalTensor& variable = iterator.second;
258 VLOG(2) << "variable present=" << variable.present
259 << " type=" << DataTypeString(variable.value.dtype())
260 << " shape=" << variable.value.shape().DebugString()
261 << " TF arg= " << iterator.first;
262 }
263 VLOG(2) << "num_outputs = " << ctx->num_outputs();
264 for (int i = 0; i < ctx->num_outputs(); i++) {
265 VLOG(2) << i << ": dtype=" << ctx->expected_output_dtype(i);
266 }
267 }
268
269 TF_RET_CHECK(constant_args.size() + variable_args.size() <=
270 ctx->num_inputs());
271
272 Signature signature;
273 TF_RETURN_IF_ERROR(
274 BuildSignature(function, constant_args, variable_args, ctx, &signature));
275
276 VLOG(2) << "Signature: " << SignatureDebugString(signature);
277 // The outer lock protects the existence of the cache entry. It does not
278 // protect the contents of the cache entry.
279 Entry* entry;
280 {
281 mutex_lock lock(mu_);
282 // Find or create a cache entry.
283 std::unique_ptr<Entry>& e = cache_[signature];
284 if (!e) {
285 e.reset(new Entry);
286 }
287 entry = e.get();
288 }
289
290 // Acquire the cache entry lock and compile, if necessary.
291 // TODO(phawkins): this locking will need to be restructured when we implement
292 // cache eviction.
293 mutex_lock entry_lock(entry->mu);
294 if (!entry->compiled) {
295 VLOG(1) << "Compilation cache miss for signature: "
296 << SignatureDebugString(signature);
297 // Do the actual JIT compilation without holding the lock (it can take
298 // a long time.)
299 std::vector<XlaCompiler::Argument> args;
300 TF_RETURN_IF_ERROR(
301 BuildArguments(constant_args, variable_args, ctx, &args));
302
303 XlaCompiler compiler(options);
304 entry->compiled = true;
305
306 if (compile_single_op) {
307 entry->compilation_status = compiler.CompileSingleOp(
308 compile_options ? *compile_options : XlaCompiler::CompileOptions(),
309 signature.name, ctx, args, &entry->compilation_result);
310 } else {
311 entry->compilation_status = compiler.CompileFunction(
312 compile_options ? *compile_options : XlaCompiler::CompileOptions(),
313 function, args, &entry->compilation_result);
314 }
315 }
316 *compilation_result = &entry->compilation_result;
317 if (entry->compilation_status.ok() && executable) {
318 if (entry->executable == nullptr) {
319 entry->compilation_status = BuildExecutable(
320 options, entry->compilation_result, &entry->executable);
321 }
322 *executable = entry->executable.get();
323 }
324
325 Status status = entry->compilation_status;
326 return status;
327}
