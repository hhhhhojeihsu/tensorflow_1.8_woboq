24SparseIndexArray::SparseIndexArray() : rank_(0), max_indices_(0) {}
25
26SparseIndexArray::SparseIndexArray(int64 max_indices, int64 rank,
27 std::vector<int64> indices)
28 : indices_(std::move(indices)), rank_(rank), max_indices_(max_indices) {
29 CHECK_GT(rank_, 0);
30 CHECK_EQ(indices_.size() % rank_, 0)
31 << "indices_.size(): " << indices_.size() << ", rank_: " << rank_;
32 CHECK_LT(index_count(), max_indices_);
33}
34
35SparseIndexArray::SparseIndexArray(int64 max_indices, int64 rank,
36 tensorflow::gtl::ArraySlice<int64> indices)
37 : SparseIndexArray(max_indices, rank,
38 std::vector<int64>(indices.begin(), indices.end())) {}
39
40SparseIndexArray::SparseIndexArray(int64 max_indices,
41 const Array2D<int64>& indices)
42 : SparseIndexArray(max_indices, indices.n2(),
43 std::vector<int64>(indices.begin(), indices.end())) {}
44
45int64 SparseIndexArray::index_count() const {
46 CHECK_GT(rank_, 0);
47 CHECK_EQ(indices_.size() % rank_, 0);
48 return indices_.size() / rank_;
49}
50
51tensorflow::gtl::ArraySlice<int64> SparseIndexArray::At(
52 int64 sparse_element_number) const {
53 CHECK_GT(rank_, 0);
54 CHECK_GE(sparse_element_number, 0);
55 CHECK_LE(rank_ * sparse_element_number + rank_, indices_.size());
56 return tensorflow::gtl::ArraySlice<int64>(
57 indices_.data() + rank_ * sparse_element_number, rank_);
58}
59
60tensorflow::gtl::MutableArraySlice<int64> SparseIndexArray::At(
61 int64 sparse_element_number) {
62 CHECK_GT(rank_, 0);
63 CHECK_GE(sparse_element_number, 0);
64 CHECK_LE(rank_ * sparse_element_number + rank_, indices_.size());
65 return tensorflow::gtl::MutableArraySlice<int64>(
66 indices_.data() + rank_ * sparse_element_number, rank_);
67}
68
69void SparseIndexArray::Append(tensorflow::gtl::ArraySlice<int64> index) {
70 CHECK_GT(rank_, 0);
71 CHECK_EQ(index.size(), rank_);
72 indices_.insert(indices_.end(), index.begin(), index.end());
73}
74
75void SparseIndexArray::Clear() { indices_.clear(); }
76
77void SparseIndexArray::Resize(int64 num_indices) {
78 CHECK_GT(rank_, 0);
79 indices_.resize(rank_ * num_indices);
80}
81
82bool SparseIndexArray::Validate(const Shape& shape) const {
83 if (rank_ == 0 || rank_ != ShapeUtil::Rank(shape)) {
84 return false;
85 }
86 int64 num_indices = index_count();
87 if (num_indices > LayoutUtil::MaxSparseElements(shape.layout())) {
88 return false;
89 }
90 if (num_indices < 2) {
91 return true;
92 }
93 tensorflow::gtl::ArraySlice<int64> last = At(0);
94 if (!IndexUtil::IndexInBounds(shape, last)) {
95 return false;
96 }
97 for (int64 n = 1; n < num_indices; ++n) {
98 tensorflow::gtl::ArraySlice<int64> next = At(n);
99 if (!IndexUtil::IndexInBounds(shape, next)) {
100 return false;
101 }
102 if (IndexUtil::CompareIndices(last, next) >= 0) {
103 return false;
104 }
105 last = next;
106 }
107 return true;
108}
