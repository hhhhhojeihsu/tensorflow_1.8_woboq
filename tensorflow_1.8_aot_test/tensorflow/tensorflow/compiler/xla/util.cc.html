33Status WithLogBacktrace(const Status& status) {
34 CHECK(!status.ok());
35 VLOG(1) << status.ToString();
36 VLOG(1) << tensorflow::CurrentStackTrace();
37 return status;
38}

40ScopedLoggingTimer::ScopedLoggingTimer(const string& label, bool enabled)
41 : enabled(enabled), label(label) {
42 if (enabled) {
43 start_micros = tensorflow::Env::Default()->NowMicros();
44 }
45}
46
47ScopedLoggingTimer::~ScopedLoggingTimer() {
48 if (enabled) {
49 uint64 end_micros = tensorflow::Env::Default()->NowMicros();
50 double secs = (end_micros - start_micros) / 1000000.0;
51
52 LOG(INFO) << label << " time: "
53 << tensorflow::strings::HumanReadableElapsedTime(secs);
54 }
55}

149string Reindent(tensorflow::StringPiece original,
150 const tensorflow::StringPiece indentation) {
151 std::vector<string> pieces = tensorflow::str_util::Split(
152 tensorflow::StringPiece(original.data(), original.size()), '\n');
153 return tensorflow::str_util::Join(
154 pieces, "\n", [indentation](string* out, string s) {
155 tensorflow::StringPiece piece(s);
156 tensorflow::str_util::RemoveWhitespaceContext(&piece);
157 tensorflow::strings::StrAppend(out, indentation, piece);
158 });
159}

161bool IsPermutation(tensorflow::gtl::ArraySlice<int64> permutation, int64 rank) {
162 if (rank != permutation.size()) {
163 return false;
164 }
165 std::vector<int64> output(permutation.size(), -1);
166 for (auto index : permutation) {
167 CHECK_GE(index, 0);
168 CHECK_LT(index, rank);
169 output[index] = 0;
170 }
171 return std::find(output.begin(), output.end(), -1) == output.end();
172}
173
174std::vector<int64> InversePermutation(
175 tensorflow::gtl::ArraySlice<int64> input_permutation) {
176 DCHECK(IsPermutation(input_permutation, input_permutation.size()));
177 std::vector<int64> output_permutation(input_permutation.size(), -1);
178 for (size_t i = 0; i < input_permutation.size(); ++i) {
179 output_permutation[input_permutation[i]] = i;
180 }
181 return output_permutation;
182}
183
184std::vector<int64> ComposePermutations(tensorflow::gtl::ArraySlice<int64> p1,
185 tensorflow::gtl::ArraySlice<int64> p2) {
186 CHECK_EQ(p1.size(), p2.size());
187 std::vector<int64> output;
188 for (size_t i = 0; i < p1.size(); ++i) {
189 output.push_back(p1[p2[i]]);
190 }
191 return output;
192}
193
194bool IsIdentityPermutation(tensorflow::gtl::ArraySlice<int64> permutation) {
195 for (int64 i = 0; i < permutation.size(); ++i) {
196 if (permutation[i] != i) {
197 return false;
198 }
199 }
200 return true;
201}

203PaddingConfig MakeNoPaddingConfig(int64 rank) {
204 PaddingConfig padding_config;
205 for (int64 dnum = 0; dnum < rank; ++dnum) {
206 auto dimension = padding_config.add_dimensions();
207 dimension->set_edge_padding_low(0);
208 dimension->set_edge_padding_high(0);
209 dimension->set_interior_padding(0);
210 }
211 return padding_config;
212}
213
214PaddingConfig MakeEdgePaddingConfig(
215 tensorflow::gtl::ArraySlice<std::pair<int64, int64>> padding) {
216 PaddingConfig padding_config;
217 for (const std::pair<int64, int64>& dim : padding) {
218 auto dimension = padding_config.add_dimensions();
219 dimension->set_edge_padding_low(dim.first);
220 dimension->set_edge_padding_high(dim.second);
221 dimension->set_interior_padding(0);
222 }
223 return padding_config;
224}
225
226bool HasInteriorPadding(const PaddingConfig& config) {
227 for (const auto& dim : config.dimensions()) {
228 if (dim.interior_padding() != 0) {
229 return true;
230 }
231 }
232 return false;
233}

236string HumanReadableNumOps(double flops, double nanoseconds,
237 tensorflow::StringPiece op_prefix) {
238 if (nanoseconds == 0) {
239 return tensorflow::strings::StrCat("NaN ", op_prefix, "OP/s");
240 }
241 double nano_flops = flops / nanoseconds;
242 string throughput = tensorflow::strings::HumanReadableNum(
243 static_cast<int64>(nano_flops * 1e9));
244 tensorflow::StringPiece sp(throughput);
245 // Use the more common "G(FLOPS)", rather than "B(FLOPS)"
246 if (tensorflow::str_util::EndsWith(sp, "B") || // Ends in 'B', ignoring case
247 tensorflow::str_util::EndsWith(sp, "b")) {
248 *throughput.rbegin() = 'G';
249 }
250 throughput += tensorflow::strings::StrCat(op_prefix, "OP/s");
251 return throughput;
252}
253} // namespace

263void LogLines(int sev, tensorflow::StringPiece text, const char* fname,
264 int lineno) {
265 const int orig_sev = sev;
266 if (sev == tensorflow::FATAL) {
267 sev = tensorflow::ERROR;
268 }
269
270 // Protect calls with a mutex so we don't interleave calls to LogLines from
271 // multiple threads.
272 static tensorflow::mutex log_lines_mu(tensorflow::LINKER_INITIALIZED);
273 tensorflow::mutex_lock lock(log_lines_mu);
274
275 size_t cur = 0;
276 while (cur < text.size()) {
277 size_t eol = text.find('\n', cur);
278 if (eol == tensorflow::StringPiece::npos) {
279 eol = text.size();
280 }
281 auto msg = text.substr(cur, eol - cur);
282 tensorflow::internal::LogString(fname, lineno, sev,
283 string(msg.data(), msg.size()));
284 cur = eol + 1;
285 }
286
287 if (orig_sev == tensorflow::FATAL) {
288 tensorflow::internal::LogString(fname, lineno, orig_sev,
289 "Aborting due to errors.");
290 }
291}

293int64 Product(tensorflow::gtl::ArraySlice<int64> xs) {

298std::vector<std::pair<int64, int64>> CommonFactors(
299 tensorflow::gtl::ArraySlice<int64> a,
300 tensorflow::gtl::ArraySlice<int64> b) {
301 CHECK_EQ(Product(a), Product(b));
302 if (0 == Product(a)) {
303 return {std::make_pair(0, 0), std::make_pair(a.size(), b.size())};
304 }
305
306 std::vector<std::pair<int64, int64>> bounds;
307 for (int64 i = 0, j = 0, prior_i = -1, prior_j = -1, partial_size_a = 1,
308 partial_size_b = 1;
309 ;) {
310 if (partial_size_a == partial_size_b && (i > prior_i || j > prior_j)) {
311 std::tie(prior_i, prior_j) = std::make_pair(i, j);
312 bounds.emplace_back(i, j);
313 continue;
314 }
315 bool in_bounds_i = i < a.size();
316 bool in_bounds_j = j < b.size();
317 if (!(in_bounds_i || in_bounds_j)) {
318 break;
319 }
320 bool next_a =
321 partial_size_a < partial_size_b ||
322 (in_bounds_i &&
323 (!in_bounds_j || (partial_size_a == partial_size_b && a[i] <= b[j])));
324 bool next_b =
325 partial_size_b < partial_size_a ||
326 (in_bounds_j &&
327 (!in_bounds_i || (partial_size_b == partial_size_a && b[j] <= a[i])));
328 if (next_a) {
329 partial_size_a *= a[i];
330 ++i;
331 }
332 if (next_b) {
333 partial_size_b *= b[j];
334 ++j;
335 }
336 }
337 return bounds;
338}

340string SanitizeFileName(string file_name) {
341 for (char& c : file_name) {
342 if (c == '/' || c == '\\' || c == '[' || c == ']' || c == ' ') {
343 c = '_';
344 }
345 }
346 return file_name;
