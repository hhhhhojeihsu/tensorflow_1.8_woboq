19#include "tensorflow/compiler/xla/service/hlo_module.h"
20#include "tensorflow/compiler/xla/service/hlo_pass_interface.h"

22namespace xla {
23
24// A pass which performs common-subexpression elimination. Identical constants
25// and identical instructions with the same operands are commoned. The pass
26// iterates over the instructions in topological order which enables the pass to
27// find arbitrarily large common expressions.
28class HloCSE : public HloPassInterface {
29 public:
30 // If is_layout_sensitive is true, then the simplifier preserves layout during
31 // transformation. Otherwise, layout is ignored.
32 explicit HloCSE(bool is_layout_sensitive)
33 : is_layout_sensitive_(is_layout_sensitive) {}
34 ~HloCSE() override {}
35 tensorflow::StringPiece name() const override { return "cse"; }
36
37 // Run CSE on the given module. Returns whether the module was changed (common
38 // subexpressions were found and eliminated).
39 StatusOr<bool> Run(HloModule* module) override;
40
41 private:
42 bool is_layout_sensitive_;
43};
44
45} // namespace xla
