32StatusOr<std::vector<std::unique_ptr<ShapedBuffer>>>
33Executable::ExecuteOnStreams(
34 ArraySlice<const ServiceExecutableRunOptions> run_options,
35 ArraySlice<ArraySlice<const ShapedBuffer*>> arguments) {
36 TF_RET_CHECK(run_options.size() == arguments.size());
37
38 std::vector<std::unique_ptr<ShapedBuffer>> return_values(run_options.size());
39
40 if (run_options.size() == 1) {
41 TF_ASSIGN_OR_RETURN(return_values[0],
42 ExecuteOnStream(&run_options[0], arguments[0],
43 /*hlo_execution_profile=*/nullptr));
44 return std::move(return_values);
45 }
46
47 for (size_t i = 0; i < run_options.size(); ++i) {
48 // We cannot BlockHostUntilDone() on the already-launched executions in case
49 // of error, since if the executions communicate, the initially launched
50 // executions may never complete if not all executions are running.
51 TF_ASSIGN_OR_RETURN(return_values[i],
52 ExecuteAsyncOnStream(&run_options[i], arguments[i]));
53 }
54 for (const auto& options : run_options) {
55 TF_RET_CHECK(options.stream() != nullptr);
56 TF_RETURN_IF_ERROR(options.stream()->BlockHostUntilDone());
57 }
58 return std::move(return_values);
59}

61StatusOr<std::unique_ptr<ShapedBuffer>> Executable::ExecuteOnStreamWrapper(
62 const ServiceExecutableRunOptions* run_options, ExecutionProfile* profile,
63 ArraySlice<const ShapedBuffer*> arguments) {
64 perftools::gputools::Stream* stream = run_options->stream();
65 std::unique_ptr<perftools::gputools::Timer> timer;
66 if (profile != nullptr) {
67 timer.reset(new perftools::gputools::Timer(stream->parent()));
68 stream->InitTimer(timer.get()).ThenStartTimer(timer.get());
69 }
70
71 VLOG(1) << "enqueueing executable on stream...";
72 // If the profiling flag isn't enabled, we pass nullptr as the profile to
73 // indicate profiling is not requested.
74 std::unique_ptr<HloExecutionProfile> profile_ptr =
75 module_config().debug_options().xla_hlo_profile() &&
76 hlo_profiling_enabled()
77 ? MakeUnique<HloExecutionProfile>(&hlo_profile_printer_data(),
78 &hlo_profile_index_map())
79 : nullptr;
80
81 StatusOr<std::unique_ptr<ShapedBuffer>> return_value =
82 ExecuteOnStream(run_options, arguments, profile_ptr.get());
83 TF_RETURN_IF_ERROR(return_value.status());
84
85 if (profile != nullptr) {
86 VLOG(1) << "enqueueing 'stop timer' and blocking host until done...";
87 stream->ThenStopTimer(timer.get());
88 TF_RETURN_IF_ERROR(stream->BlockHostUntilDone());
89 VLOG(1) << "done with block-host-until-done";
90
91 // Merge in run-time profile information from execution_profile.
92 //
93 // TODO(b/71713097): This is buggy -- even though the mutex takes care of
94 // C++ level races, some other concurrent ExecuteOnStreamWrapper call could
95 // have rewritten the execution_profile before we get to it.
96 profile->MergeFrom(execution_profile());
97
98 // Overall execution time (in nanoseconds) from the executor timer.
99 if (stream->ok()) {
100 // Don't read timer->Nanoseconds() if the stream isn't OK -- that's
101 // illegal.
102 profile->set_compute_and_transfer_time_ns(timer->Nanoseconds());
103 }
104
105 // TODO(b/28123297): On GPU we end up including transfer time in
106 // the compute time this way. Instead, we should get the correct
107 // value by measuring it. Setting the field here at least lets
108 // benchmarks provide *some* value for GPU computations.
109 //
110 // TODO(b/28447609): The value in compute_and_transfer_time_ns is actually
111 // the compute time without the transfer time, so this way we get the
112 // correct compute time. We should instead have the correct value for
113 // compute_and_transfer_time and set compute_time to the compute time.
114 if (profile->compute_time_ns() == 0) {
115 profile->set_compute_time_ns(profile->compute_and_transfer_time_ns());
116 }
117 }
118
119 if (profile_ptr != nullptr) {
120 XLA_LOG_LINES(
121 tensorflow::INFO,
122 profile_ptr->ToString(stream->parent()->GetDeviceDescription()));
123 hlo_graph_dumper::MaybeDumpHloModule(module(), "Service::Execute",
124 profile_ptr.get());
125 }
126
127 return return_value;
128}

130Status Executable::DumpSessionModule() {
131 TF_RET_CHECK(dumping());
132 const string& directory_path =
133 module_config().debug_options().xla_dump_executions_to();
134 VersionedComputationHandle versioned_handle = entry_computation_handle();
135 // This filename does not include the version number because the computation
136 // is only ever executed at one version.
137 string filename = tensorflow::strings::Printf(
138 "computation_%lld__%s__execution_%lld", versioned_handle.handle.handle(),
139 session_module_->entry().name().c_str(), ++execution_count_);
140 return Executable::DumpToDirectory(directory_path, filename,
141 *session_module_);
142}

144/* static */ Status Executable::DumpToDirectory(
145 const string& directory_path, string filename,
146 const SessionModule& session_module) {
147 tensorflow::Env* env = tensorflow::Env::Default();
148 if (!env->IsDirectory(directory_path).ok()) {
149 // NB! CreateDir does not work reliably with multiple XLA threads -- two
150 // threads can race to observe the absence of the dump directory and
151 // simultaneously try to create it, causing the "losing" thread to get a
152 // "directory already exists" error.
153 TF_RETURN_IF_ERROR(env->RecursivelyCreateDir(directory_path));
154 }
155 filename = SanitizeFileName(std::move(filename));
156 string file_path = tensorflow::io::JoinPath(directory_path, filename);
157 string result;
158 TF_RET_CHECK(
159 tensorflow::SerializeToStringDeterministic(session_module, &result));
160 return tensorflow::WriteStringToFile(tensorflow::Env::Default(), file_path,
161 result);
162}
