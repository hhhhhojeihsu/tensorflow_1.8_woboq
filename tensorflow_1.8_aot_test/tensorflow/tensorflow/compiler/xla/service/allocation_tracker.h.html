36// Tracks allocations for the XLA service; allocations can be registered
37// with shape/device/tag and resolved from a handle for later use.
38class AllocationTracker {
39 public:
40 // The allocator is used for deallocating memory when allocations are
41 // deregistered. All registered allocations must have the same platform as the
42 // allocator.
43 AllocationTracker(Backend* backend) : backend_(backend), next_handle_(1) {}
44
45 // Registers a shaped buffer of device memory, and returns a corresponding
46 // handle that can be used for talking to XLA clients. The given shaped buffer
47 // will be treated as the buffer corresponding to the only replica.
48 StatusOr<GlobalDataHandle> Register(
49 std::unique_ptr<ShapedBuffer> shaped_buffer, const string& tag);
50
51 // Registers a vector of shaped buffers of device memory, one per replica, and
52 // returns a corresponding handle that can be used for talking to XLA clients.
53 StatusOr<GlobalDataHandle> RegisterReplicatedBuffers(
54 std::vector<std::unique_ptr<ShapedBuffer>> replicated_buffers,
55 const string& tag);
56
57 // Unregister the allocation for the given data handle.
58 Status Unregister(const GlobalDataHandle& data);
59
60 // Returns a vector of global data handles that point to the tuple elements.
61 StatusOr<std::vector<GlobalDataHandle>> DeconstructTuple(
62 const GlobalDataHandle& Data);
63
64 // Resolve a handle from an XLA client to a vector of shaped buffers, one per
65 // replica, or provide an error status to say whether any of those buffers
66 // were not found (or found, but found deallocated).
67 StatusOr<std::vector<const ShapedBuffer*>> Resolve(
68 const GlobalDataHandle& data);
69
70 // Resolves a handle from an XLA client and replica id to a shaped buffer, or
71 // provide an error status to say whether it was not found (or found, but
72 // found deallocated).
73 StatusOr<const ShapedBuffer*> ResolveForReplica(const GlobalDataHandle& data,
74 int replica_id);
75
76 private:
77 // Data structure encapsulating single memory allocation on the device.
78 struct Allocation {
79 // The pointer to this allocation.
80 perftools::gputools::DeviceMemoryBase device_memory;
81
82 // The device that the memory is allocated on.
83 int device_ordinal;
84
85 // This is the number of times this memory allocation is referred to by
86 // registered data handles.
87 int ref_count;
88 };
89
90 // Internal helper which resolves the given GlobalDataHandle to a
91 // ShapedBuffer.
92 StatusOr<std::vector<const ShapedBuffer*>> ResolveInternal(
93 const GlobalDataHandle& data) EXCLUSIVE_LOCKS_REQUIRED(mutex_);
94
95 // Internal helper which registers a vector of shaped buffers, one per
96 // replica.
97 StatusOr<GlobalDataHandle> RegisterInternal(
98 std::vector<std::unique_ptr<ShapedBuffer>> replicated_buffers,
99 const string& tag) EXCLUSIVE_LOCKS_REQUIRED(mutex_);
100
101 // Resets the shaped buffers corresponding to the given handle.
102 Status Reset(const GlobalDataHandle& data) EXCLUSIVE_LOCKS_REQUIRED(mutex_);
103
104 // Adds the given device address to the allocation tracker, or if it already
105 // exists, then increment it's reference count.
106 void AddAllocationOrIncrementRefCount(
107 perftools::gputools::DeviceMemoryBase device_memory, int device_ordinal)
108 EXCLUSIVE_LOCKS_REQUIRED(mutex_);
109
110 // Decrements the reference count of the given device memory. Then, if it is
111 // zero, deallocate the memory.
112 Status DecrementRefCount(perftools::gputools::DeviceMemoryBase device_memory,
113 int device_ordinal) EXCLUSIVE_LOCKS_REQUIRED(mutex_);
114
115 // A map from device memory opaque value to allocation. One such map is
116 // maintained per device ordinal.
117 using AllocationMap = tensorflow::gtl::FlatMap<const void*, Allocation>;
118
119 tensorflow::mutex mutex_;
120
121 // Backend to use with this tracker. The backend supplies the memory allocator
122 // to use when deallocating memory.
123 Backend* backend_;
124
125 // The next handle to assign to an allocation, guarded by the same mutex as
126 // the mapping as they'll be mutated at the same time.
127 int64 next_handle_ GUARDED_BY(mutex_);
128
129 // A map from device ordinal to AllocationMap.
130 tensorflow::gtl::FlatMap<int, AllocationMap> opaque_to_allocation_map_
131 GUARDED_BY(mutex_);
132
133 // A map from data handle to a vector of shaped buffers that represent the
134 // buffers for different replicas.
135 tensorflow::gtl::FlatMap<int64, std::vector<std::unique_ptr<ShapedBuffer>>>
136 handle_to_shaped_buffers_ GUARDED_BY(mutex_);
137
138 TF_DISALLOW_COPY_AND_ASSIGN(AllocationTracker);
139};
