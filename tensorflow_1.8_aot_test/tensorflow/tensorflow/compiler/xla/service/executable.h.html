42// A given platform's compiler will produce an Executable -- this is a uniform
43// interface that is used for launching compiled programs across platforms.
44class Executable {
45 public:
46 explicit Executable(
47 std::unique_ptr<const HloModule> hlo_module,
48 std::unique_ptr<HloProfilePrinterData> hlo_profile_printer_data,
49 std::unique_ptr<HloProfileIndexMap> hlo_profile_index_map)
50 : hlo_module_(std::move(hlo_module)),
51 hlo_profile_printer_data_(std::move(hlo_profile_printer_data)),
52 hlo_profile_index_map_(std::move(hlo_profile_index_map)) {
53 CHECK_EQ(hlo_profile_printer_data_.get() == nullptr,
54 hlo_profile_index_map_.get() == nullptr);
55 }
56 virtual ~Executable() {}
57
58 // Enqueues the compilation result on the provided stream, passing the given
59 // arguments. This call is blocking and returns after the execution is done.
60 //
61 // If the hlo_execution_profile is provided as non-nullptr, profiling will be
62 // enabled.
63 //
64 // Returns a shaped buffer containing the result of the computation.
65 virtual StatusOr<std::unique_ptr<ShapedBuffer>> ExecuteOnStream(
66 const ServiceExecutableRunOptions* run_options,
67 tensorflow::gtl::ArraySlice<const ShapedBuffer*> arguments,
68 HloExecutionProfile* hlo_execution_profile) = 0;
69
70 // Same as ExecuteOnStream(), but this call is non-blocking and returns as
71 // soon as all of the operations are enqueued for launch on the stream.
72 virtual StatusOr<std::unique_ptr<ShapedBuffer>> ExecuteAsyncOnStream(
73 const ServiceExecutableRunOptions* run_options,
74 tensorflow::gtl::ArraySlice<const ShapedBuffer*> arguments) = 0;
75
76 // Same as ExecuteOnStream(), but runs this executable on multiple
77 // streams. arguments[i] contains the arguments to the execution on
78 // run_options[i]->stream() and the returned value is at index i of the
79 // returned vector.
80 virtual StatusOr<std::vector<std::unique_ptr<ShapedBuffer>>> ExecuteOnStreams(
81 tensorflow::gtl::ArraySlice<const ServiceExecutableRunOptions>
82 run_options,
83 tensorflow::gtl::ArraySlice<
84 tensorflow::gtl::ArraySlice<const ShapedBuffer*>>
85 arguments);
86
87 // Populates `hlo_execution_profile` from `executor`. This is implicit in any
88 // Execute* API call that takes a hlo_execution_profile argument, but must be
89 // called explicitly for other (async, for example) variants after the stream
90 // has completed.
91 virtual Status PopulateExecutionProfile(
92 HloExecutionProfile* hlo_execution_profile,
93 perftools::gputools::StreamExecutor* executor) {
94 return Status::OK();
95 }
96
97 // Convenience wrapper for calling Executable::ExecuteOnStream. Sets up a
98 // timer for the execution, sets up HLO profiling if enabled, and fills in the
99 // given ExecutionProfile if non-null.
100 StatusOr<std::unique_ptr<ShapedBuffer>> ExecuteOnStreamWrapper(
101 const ServiceExecutableRunOptions* run_options, ExecutionProfile* profile,
102 tensorflow::gtl::ArraySlice<const ShapedBuffer*> arguments);
103
104 // Returns the ExecutionProfile from executing on the device. This includes
105 // the number of cycles taken for the computation or the compilation time.
106 ExecutionProfile execution_profile() const {
107 tensorflow::mutex_lock lock(mutex_);
108 return execution_profile_;
109 }
110
111 const HloProfilePrinterData& hlo_profile_printer_data() const {
112 CHECK(hlo_profiling_enabled());
113 return *hlo_profile_printer_data_;
114 }
115
116 const HloProfileIndexMap& hlo_profile_index_map() const {
117 CHECK(hlo_profiling_enabled());
118 return *hlo_profile_index_map_;
119 }
120
121 // Returns whether this executable was compiled with HLO profilings support
122 // enabled. If not, the caller should not expect an hlo_execution_profile
123 // passed to ExecuteOnStream above to be populated during execution.
124 bool hlo_profiling_enabled() const {
125 return hlo_profile_printer_data_ != nullptr;
126 }
127
128 const HloModule& module() const { return *hlo_module_; }
129
130 const bool has_module() const { return hlo_module_ != nullptr; }
131
132 const HloModuleConfig& module_config() const { return hlo_module_->config(); }
133
134 // Returns the versioned computation handle of the computation computed by
135 // this executable.
136 const VersionedComputationHandle& entry_computation_handle() const {
137 return hlo_module_->entry_computation_handle();
138 }
139
140 // The shape (including layout) that results from this execution. This is the
141 // shape of the DeviceMemoryBase result value in ExecuteOnStream above.
142 const Shape& result_shape() const {
143 return hlo_module_->config().entry_computation_layout().result_shape();
144 }
145
146 // Dumping helpers.
147 void set_session_module(std::unique_ptr<xla::SessionModule> session_module) {
148 session_module_ = std::move(session_module);
149 }
150 bool dumping() const { return session_module_ != nullptr; }
151 SessionModule* session_module() const { return session_module_.get(); }
152 Status DumpSessionModule();
153
154 // Dump session_module to directory_path/filename.
155 static Status DumpToDirectory(const string& directory_path, string filename,
156 const SessionModule& session_module);
157
158 protected:
159 mutable tensorflow::mutex mutex_;
160
161 // Execution profile data on the device.
162 ExecutionProfile execution_profile_ GUARDED_BY(mutex_);
163
164 // HloModule this was compiled from. BufferAssignment keeps pointers to
165 // HloInstructions owned by the HloModule so we need to keep the HloModule
166 // around.
167 const std::unique_ptr<const HloModule> hlo_module_;
168
169 // SessionModule this was compiled from. Null if not dumping executions.
170 std::unique_ptr<SessionModule> session_module_;
171
172 // Execution count, used to generate a unique filename for each dumped
173 // execution.
174 int64 execution_count_ = 0;
175
176 std::unique_ptr<HloProfilePrinterData> hlo_profile_printer_data_;
177 std::unique_ptr<HloProfileIndexMap> hlo_profile_index_map_;
178};
179
180} // namespace xla
