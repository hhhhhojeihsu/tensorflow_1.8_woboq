30// Copies `to_hoist` to the computation containing `while_instr`, hoisting its
31// operands as needed. All of its transitive operands are expected to be either
32// in `hoisted_instructions` or `unhoisted_invariant_instructions`. This
33// function hoists the operands in `unhoisted_invariant_instructions` and moves
34// them into `hoisted_instructions`.
35static void CreateLoopInvariantCopy(
36 FlatMap<HloInstruction*, HloInstruction*>* hoisted_instructions,
37 FlatSet<HloInstruction*>* unhoisted_invariant_instructions,
38 HloInstruction* while_instr, HloInstruction* to_hoist) {
39 HloComputation* parent_of_while = while_instr->parent();
40 HloComputation* while_body = while_instr->while_body();
41
42 struct DFSFrame {
43 HloInstruction* instruction;
44 int64 operand_index;
45 };
46
47 InlinedVector<DFSFrame, 8> dfs_stack;
48 dfs_stack.push_back({to_hoist, 0});
49
50 HloInstruction* while_body_param = while_body->parameter_instruction(0);
51 HloInstruction* while_operand = while_instr->mutable_operand(0);
52
53 do {
54 DFSFrame* frame = &dfs_stack.back();
55 if (frame->operand_index == frame->instruction->operand_count()) {
56 HloInstruction* old_instruction = frame->instruction;
57
58 // All of the operands for old_instruction have been cloned, so it is
59 // time to clone old_instruction itself.
60
61 auto get_new_operand = [&](HloInstruction* old_operand) {
62 return old_operand == while_body_param
63 ? while_operand
64 : FindOrDie(*hoisted_instructions, old_operand);
65 };
66
67 InlinedVector<HloInstruction*, 4> new_operands;
68 c_transform(old_instruction->operands(), std::back_inserter(new_operands),
69 get_new_operand);
70
71 HloInstruction* new_instruction =
72 parent_of_while->AddInstruction(old_instruction->CloneWithNewOperands(
73 old_instruction->shape(), new_operands));
74
75 InsertOrDie(hoisted_instructions, old_instruction, new_instruction);
76
77 // Approximately half of the instructions that would normally be present
78 // in unhoisted_invariant_instructions are constants. We save a bit of
79 // compile time by not putting these in the hashtable.
80 CHECK_EQ(unhoisted_invariant_instructions->erase(old_instruction),
81 to_hoist != old_instruction &&
82 old_instruction->opcode() != HloOpcode::kConstant);
83 dfs_stack.pop_back();
84 continue;
85 }
86
87 HloInstruction* next_operand =
88 frame->instruction->mutable_operand(frame->operand_index++);
89 if (hoisted_instructions->count(next_operand) ||
90 next_operand == while_body_param) {
91 continue;
92 }
93
94 dfs_stack.push_back({next_operand, 0});
95 } while (!dfs_stack.empty());
96}

98// Returns true if `instruction` is worth hoisting only if it lets us hoist some
99// instruction using it. The rationale is that hoisting these instructions will
100// prevent simplification and fusion in the while body.
101static bool NotWorthHoistingIndividually(const HloInstruction& instruction) {
102 switch (instruction.opcode()) {
103 default:
104 return false;
105
106 case HloOpcode::kBitcast:
107 case HloOpcode::kBroadcast:
108 case HloOpcode::kConstant:
109 case HloOpcode::kReshape:
110 case HloOpcode::kReverse:
111 case HloOpcode::kSlice:
112 case HloOpcode::kTranspose:
113 case HloOpcode::kTuple:
114 return true;
115 }
116}

118// Populates `gte_set` with the GetTupleElement instructions in `while_body`
119// that access elements in the parameter tuple that don't change across
120// iterations. Assumes `while_body` is the body computation of the while loop
121// in question.
122static void GatherInvariantGTEs(HloComputation* while_body,
123 FlatSet<HloInstruction*>* gte_set) {
124 const HloInstruction::InstructionVector root_operands =
125 while_body->root_instruction()->operands();
126 for (int i = 0; i < root_operands.size(); i++) {
127 HloInstruction* instr = root_operands[i];
128 if (instr->opcode() == HloOpcode::kGetTupleElement &&
129 instr->tuple_index() == i &&
130 instr->operand(0) == while_body->parameter_instruction(0) &&
131 ShapeUtil::IsArray(instr->shape())) {
132 InsertOrDie(gte_set, instr);
133 }
134 }
135}

137static StatusOr<bool> TryHoistingInvariantInstructionsFromWhileBody(
138 HloInstruction* while_instr) {
139 auto print_no_metadata = HloPrintOptions{}.set_print_metadata(false);
140
141 if (!ShapeUtil::IsTuple(while_instr->shape())) {
142 // This restriction leaves one interesting pattern on the table:
143 //
144 // while_body(f32[1024, 1024] %param) {
145 // %value = expensive_op(%param)
146 // outfeed(%value)
147 // ROOT = %param
148 // }
149 //
150 // If we see that pattern in the while, instead of generalizing this
151 // algorithm to work with non-tuples, we should instead add a pass that
152 // canonicalizes while loops like the above to use a tuple state.
153 return false;
154 }
155
156 string while_instr_name = while_instr->ToString(print_no_metadata);
157 VLOG(2) << "Trying to hoist from " << while_instr_name;
158
159 HloComputation* while_body = while_instr->while_body();
160
161 // Maps instructions in the while body to instructions hoisted outside the
162 // while that compute the same value.
163 FlatMap<HloInstruction*, HloInstruction*> hoisted_instructions;
164
165 // Contains instructions that can be legally hoisted, but were deemed to be
166 // unprofitable to be hoisted alone by NotWorthHoistingIndividually. When we
167 // hoist an instruction in this set, we move it from
168 // unhoisted_invariant_instructions to hoisted_instructions.
169 FlatSet<HloInstruction*> unhoisted_invariant_instructions;
170
171 // Invariant GTE's axiomatically satisfy the constraints for
172 // unhoisted_invariant_instructions -- they can be legally hoisted, but there
173 // is no benefit to hoisting them unless something that uses it is also
174 // hoisted.
175 GatherInvariantGTEs(while_body, &unhoisted_invariant_instructions);
176
177 if (unhoisted_invariant_instructions.empty()) {
178 // There are no obviously loop invariant elements in the state being
179 // threaded through the while loop so give up. In theory this precondition
180 // is too strong -- we could have code that e.g. permutes the elements in
181 // the while state but uses a select to pick the same value on every
182 // iteration.
183 return false;
184 }
185
186 // instructions_to_replace[i] is hoisted into a loop invariant instruction
187 // replacement_instructions[i].
188 std::vector<HloInstruction*> instructions_to_replace;
189 std::vector<HloInstruction*> replacement_instructions;
190
191 for (auto* instruction : while_body->MakeInstructionPostOrder()) {
192 if (instruction->HasSideEffect() ||
193 instruction->opcode() == HloOpcode::kParameter ||
194 !instruction->control_predecessors().empty() ||
195 !instruction->control_successors().empty()) {
196 continue;
197 }
198
199 auto is_invariant = [&](HloInstruction* op) {
200 return hoisted_instructions.find(op) != hoisted_instructions.end() ||
201 unhoisted_invariant_instructions.count(op) ||
202 op->opcode() == HloOpcode::kConstant;
203 };
204
205 if (!c_all_of(instruction->operands(), is_invariant)) {
206 continue;
207 }
208
209 if (NotWorthHoistingIndividually(*instruction)) {
210 VLOG(2) << "Adding " << instruction->ToString(print_no_metadata)
211 << " to unhoisted invariant set.";
212 // Approximately half of the instructions that reach this point are
213 // constants. We save a bit of compile time by not putting these in the
214 // hashtable.
215 if (instruction->opcode() != HloOpcode::kConstant) {
216 InsertOrDie(&unhoisted_invariant_instructions, instruction);
217 }
218 continue;
219 }
220
221 VLOG(2) << "Hoisting " << instruction->ToString(print_no_metadata);
222
223 CreateLoopInvariantCopy(&hoisted_instructions,
224 &unhoisted_invariant_instructions, while_instr,
225 instruction);
226
227 instructions_to_replace.push_back(instruction);
228 replacement_instructions.push_back(
229 FindOrDie(hoisted_instructions, instruction));
230 }
231
232 if (instructions_to_replace.empty()) {
233 return false;
234 }
235
236 TF_ASSIGN_OR_RETURN(
237 WhileUtil::MakeInstructionsLiveInResult live_in_instructions_result,
238 WhileUtil::MakeInstructionsLiveIn(while_instr, replacement_instructions));
239
240 HloComputation* new_while_body =
241 live_in_instructions_result.new_while_instr->while_body();
242
243 for (int i = 0; i < instructions_to_replace.size(); i++) {
244 HloInstruction* instruction_to_replace_in_new_while =
245 FindOrDie(live_in_instructions_result.while_body_instruction_map,
246 instructions_to_replace[i]);
247 TF_RETURN_IF_ERROR(new_while_body->ReplaceInstruction(
248 instruction_to_replace_in_new_while,
249 live_in_instructions_result.while_body_live_in_values[i]));
250 }
251
252 VLOG(1) << "Hoisted " << instructions_to_replace.size()
253 << " instructions from " << while_instr_name;
254
255 return true;
256}

258StatusOr<bool> WhileLoopInvariantCodeMotion::Run(HloModule* module) {
259 bool changed = false;
260 std::vector<HloInstruction*> while_instrs;
261 for (auto* comp : module->computations()) {
262 c_copy_if(comp->instructions(), std::back_inserter(while_instrs),
263 [](const HloInstruction* instr) {
264 return instr->opcode() == HloOpcode::kWhile;
265 });
266 }
267
268 for (HloInstruction* while_instr : while_instrs) {
269 // Right now we only hoist computations from the while body, but
270 // TryHoistingInvariantInstructionsFromWhileBody can be generalized to
271 // optimize the condition computation too, if needed.
272 //
273 // The transform we do here is a pessmization for while loops that execute
274 // zero times*, but at this time we expect those to be rare. If this
275 // becomes a problem we can consider using the conditional HLO to avoid
276 // doing extra work for while loops with zero trip count.
277 //
278 // * We delete while loops that have a zero trip count, so this would have
279 // to be a while loop with a somewhat opaque condition expression.
280
281 TF_ASSIGN_OR_RETURN(
282 bool result,
283 TryHoistingInvariantInstructionsFromWhileBody(while_instr));
284 changed |= result;
285 }
286 return changed;
287}
