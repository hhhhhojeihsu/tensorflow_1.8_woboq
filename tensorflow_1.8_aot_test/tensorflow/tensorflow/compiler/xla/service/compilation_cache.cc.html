27std::shared_ptr<Executable> CompilationCache::Insert(
28 std::unique_ptr<Executable> executable,
29 const HloModuleConfig& module_config) {
30 tensorflow::mutex_lock lock(mutex_);
31
32 CacheKey key =
33 BuildKey(executable->entry_computation_handle(), module_config);
34 VLOG(2) << "inserting cache key: " << key;
35 if (cache_.count(key) == 0) {
36 cache_.emplace(key, std::move(executable));
37 } else {
38 // Executable already exists in the cache. This can happen if two Execute
39 // calls for a new computation are received simultaneously by the
40 // service. In this case, we discard the Executable given as a parameter and
41 // return what is in the cache. This is necessary because the service relies
42 // on the cache to keep ownership of the Executable. We only want to store
43 // one Executable for a given computation version and we can't discard the
44 // executable which is in the cache because it may be in use.
45 executable.reset();
46 }
47 return cache_.at(key);
48}
49
50std::shared_ptr<Executable> CompilationCache::LookUp(
51 const VersionedComputationHandle& versioned_handle,
52 const HloModuleConfig& module_config) const {
53 tensorflow::mutex_lock lock(mutex_);
54
55 CacheKey key = BuildKey(versioned_handle, module_config);
56 VLOG(2) << "looking up cache key: " << key;
57 if (cache_.count(key) == 0) {
58 VLOG(2) << "cache key not found: " << key;
59 return nullptr;
60 } else {
61 std::shared_ptr<Executable> result = cache_.at(key);
62 VLOG(2) << "hit executable with module config: "
63 << result->module_config().compilation_cache_key();
64 return result;
65 }
66}
67
68CompilationCache::CacheKey CompilationCache::BuildKey(
69 const VersionedComputationHandle& versioned_handle,
70 const HloModuleConfig& module_config) const {
71 // The computation shape is represented entirely by its ProgramShape member,
72 // so just serialize the proto as part of the key.
73 return tensorflow::strings::StrCat(versioned_handle.handle.handle(), "::",
74 versioned_handle.version, "::",
75 module_config.compilation_cache_key());
76}
