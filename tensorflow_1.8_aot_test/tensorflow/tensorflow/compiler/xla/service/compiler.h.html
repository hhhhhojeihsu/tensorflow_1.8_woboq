41// The following types are used for ahead of time compilation.
42
43// Contains the object file data created as a result of ahead-of-time
44// compuation.
45using ObjectFileData = std::vector<char>;
46
47// Contains the buffer sizes information needed to allocate buffers to execute
48// an ahead-of-time computation. Entries which contain -1 designate a parameter
49// which should be skipped over during allocation.
50using BufferSizes = std::vector<int64>;

52// Abstract superclass describing the result of an ahead-of-time compilation.
53class AotCompilationResult {
54 public:
55 AotCompilationResult(const AotCompilationResult&) = delete;
56 AotCompilationResult& operator=(AotCompilationResult const&) = delete;
57
58 virtual ~AotCompilationResult() = default;
59
60 protected:
61 AotCompilationResult() = default;
62};
63
64// Abstract superclass describing options to an ahead-of-time compilation.
65class AotCompilationOptions {
66 public:
67 AotCompilationOptions(const AotCompilationOptions&) = delete;
68 AotCompilationOptions& operator=(AotCompilationOptions const&) = delete;
69
70 virtual ~AotCompilationOptions() = default;
71
72 // Returns the ID of the platform to which these options apply.
73 virtual perftools::gputools::Platform::Id PlatformId() const = 0;
74
75 // Optional allocator that may be used for allocating temp space on the device
76 // during compilation.
77 DeviceMemoryAllocator* device_allocator() const { return device_allocator_; }
78 void set_device_allocator(DeviceMemoryAllocator* device_allocator) {
79 device_allocator_ = device_allocator;
80 }
81
82 const DebugOptions& debug_options() const { return debug_options_; }
83 DebugOptions* mutable_debug_options() { return &debug_options_; }
84
85 protected:
86 AotCompilationOptions();
87
88 private:
89 DeviceMemoryAllocator* device_allocator_ = nullptr;
90 DebugOptions debug_options_;
91};

93// Abstract compiler interface that is subclassed for compilation on a
94// particular platform.
95//
96// The compiler ties together high level optimization (HLO) and low level
97// optimization (LLO) / codegen (CG) to generate efficient executables for the
98// target platform.
99//
100// The platform-based compiler singletons are registered via module initializers
101// in their corresponding XLA compiler libraries, and are registered via the
102// RegisterCompilerFactory API below.
103//
104// Thread-safety: subclasses of Compiler must be thread-safe, as multiple
105// XLA clients may be requesting compilation concurrently for a given
106// platform.
107class Compiler {
108 public:
109 virtual ~Compiler() {}
110
111 // Returns the ID of the platform that this compiler targets.
112 virtual perftools::gputools::Platform::Id PlatformId() const = 0;
113
114 // Runs Hlo passes to optimize the given Hlo module, returns the optimized
115 // module.
116 //
117 // If device_allocator is not null, the compiler may use it to allocate temp
118 // space on the device for use during compilation. For example, the compiler
119 // may allocate buffers on the device and then run variants of a given
120 // algorithm over those buffers, to see which variant is fastest. Any space
121 // allocated should be deallocated before this function returns.
122 virtual StatusOr<std::unique_ptr<HloModule>> RunHloPasses(
123 std::unique_ptr<HloModule> module,
124 perftools::gputools::StreamExecutor* executor,
125 DeviceMemoryAllocator* device_allocator) = 0;
126
127 // Compiles the HLO module for execution on a device given by the executor,
128 // and returns an executable object or an error status. No HLO passes are
129 // applied to module. Generally a module should be passed through RunHloPasses
130 // prior to calling this method because some HLO passes are required for
131 // correctness. Takes ownership of the HLO module and is free to transform it.
132 //
133 // The compiler may optionally specialize to the individual device
134 // (not just type of device) indicated by the executor.
135 //
136 // device_allocator is optional; see RunHloPasses.
137 //
138 // Use the overload below to compile computations that run in parallel.
139 virtual StatusOr<std::unique_ptr<Executable>> RunBackend(
140 std::unique_ptr<HloModule> module,
141 perftools::gputools::StreamExecutor* executor,
142 DeviceMemoryAllocator* device_allocator) = 0;
143
144 // Compiles a set of HLO modules that can run in parallel, potentially
145 // communicating data between the modules, and returns a corresponding
146 // sequence of executable objects.
147 //
148 // device_allocator is optional; see RunHloPasses.
149 //
150 // TODO(b/68666782): Remove this method after adding support for multiple
151 // modules to RunHloPasses and RunBackends.
152 virtual StatusOr<std::vector<std::unique_ptr<Executable>>> Compile(
153 std::vector<std::unique_ptr<HloModule>> modules,
154 std::vector<std::vector<perftools::gputools::StreamExecutor*>>
155 stream_exec,
156 DeviceMemoryAllocator* device_allocator) = 0;
157
158 // Compiles the HLO module for ahead-of-time execution. This is intended for
159 // use in static compilation.
160 virtual StatusOr<std::vector<std::unique_ptr<AotCompilationResult>>>
161 CompileAheadOfTime(std::vector<std::unique_ptr<HloModule>> modules,
162 const AotCompilationOptions& options) = 0;
163
164 /////
165 // The Compiler class also serves as a point to register compiler objects
166 // for the various platforms.
167
168 using CompilerFactory = std::function<std::unique_ptr<Compiler>()>;
169
170 // Registers the compiler singleton for the platform. This is assumed to
171 // be a singleton, so no ownership is transferred.
172 //
173 // Precondition: a platform kind must not be registered more than once.
174 static void RegisterCompilerFactory(
175 perftools::gputools::Platform::Id platform_id,
176 CompilerFactory compiler_factory);
177
178 // Returns the compiler singleton pointer if it is available for the given
179 // platform, or an error status if it is not.
180 static StatusOr<Compiler*> GetForPlatform(
181 const perftools::gputools::Platform* platform);
182
183 // Returns a function that computes the size in bytes of the logical
184 // buffer that contains a shape.
185 virtual HloCostAnalysis::ShapeSizeFunction ShapeSizeBytesFunction() const = 0;
186
187 // Returns a function that computes the size in bytes of a given
188 // logical buffer.
189 std::function<int64(const LogicalBuffer&)> BufferSizeBytesFunction() {
190 HloCostAnalysis::ShapeSizeFunction shape_size = ShapeSizeBytesFunction();
191 return [shape_size](const LogicalBuffer& buffer) {
192 return shape_size(buffer.shape());
193 };
194 }
195
196 private:
197 // Mutex that guards the platform-compiler map.
198 static tensorflow::mutex platform_compiler_mutex_;
199
200 // Map from platform kind to compiler factory.
201 static std::map<perftools::gputools::Platform::Id, CompilerFactory>*
202 GetPlatformCompilerFactories();
203
204 // Map from platform kind to compiler instance, if we made one already (based
205 // on the factories above).
206 static std::map<perftools::gputools::Platform::Id, std::unique_ptr<Compiler>>*
207 GetPlatformCompilers();
208};
