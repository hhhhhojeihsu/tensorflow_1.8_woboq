27// Base class for HLO passes. These are used with the HloPassPipeline to
28// organize a sequence of passes.
29class HloPassInterface {
30 public:
31 virtual ~HloPassInterface() = default;
32 virtual tensorflow::StringPiece name() const = 0;
33
34 // Run the pass on the given HLO module. Return whether it modified the
35 // module.
36 virtual StatusOr<bool> Run(HloModule* module) = 0;
37};
