31StatusOr<bool> ConvCanonicalization::Run(HloModule* module) {
32 bool changed = false;
33 for (HloInstruction* hlo :
34 module->entry_computation()->MakeInstructionPostOrder()) {
35 if (hlo->opcode() == HloOpcode::kConvolution &&
36 !PotentiallyImplementedAsEigenConvolution(*hlo)) {
37 const ConvolutionDimensionNumbers& dnums =
38 hlo->convolution_dimension_numbers();
39 auto input_batch_dim = dnums.input_batch_dimension();
40 auto input_feature_dim = dnums.input_feature_dimension();
41 auto kernel_input_feature_dim = dnums.kernel_input_feature_dimension();
42 auto kernel_output_feature_dim = dnums.kernel_output_feature_dimension();
43
44 const int64 num_spatial_dims = dnums.output_spatial_dimensions_size();
45 const int64 num_dims = num_spatial_dims + 2;
46
47 // A canonical convolution's dimension numbers need to satisfy the
48 // following conditions (see cs/PotentiallyImplementedAsEigenConvolution).
49 //
50 // - the input is in NHWC order.
51 // - the kernel is in HWIO order.
52 //
53 // For simplicity, as a first step, we reshape the input and filter to
54 // NHWC and HWIO order, respectively. This may lose precision but won't
55 // break the soundness.
56 HloInstruction* input = hlo->mutable_operand(0);
57
58 std::vector<int64> new_input_dim_order(num_dims);
59 std::vector<int64> new_input_dims(num_dims);
60 new_input_dim_order[0] = input_batch_dim;
61 new_input_dims[0] = input->shape().dimensions(input_batch_dim);
62 for (int64 i = 0; i < num_spatial_dims; ++i) {
63 new_input_dim_order[i + 1] = dnums.input_spatial_dimensions(i);
64 new_input_dims[i + 1] =
65 input->shape().dimensions(dnums.input_spatial_dimensions(i));
66 }
67 new_input_dim_order[num_dims - 1] = input_feature_dim;
68 new_input_dims[num_dims - 1] =
69 input->shape().dimensions(input_feature_dim);
70
71 Shape new_input_shape =
72 ShapeUtil::MakeShape(input->shape().element_type(), new_input_dims);
73 HloInstruction* new_input = module->entry_computation()->AddInstruction(
74 HloInstruction::CreateTranspose(new_input_shape, input,
75 new_input_dim_order));
76
77 HloInstruction* kernel = hlo->mutable_operand(1);
78
79 std::vector<int64> new_kernel_dim_order(num_dims);
80 std::vector<int64> new_kernel_dims(num_dims);
81 for (int64 i = 0; i < num_spatial_dims; ++i) {
82 new_kernel_dim_order[i] = dnums.kernel_spatial_dimensions(i);
83 new_kernel_dims[i] =
84 kernel->shape().dimensions(dnums.kernel_spatial_dimensions(i));
85 }
86 new_kernel_dim_order[num_dims - 2] = kernel_input_feature_dim;
87 new_kernel_dims[num_dims - 2] =
88 kernel->shape().dimensions(kernel_input_feature_dim);
89 new_kernel_dim_order[num_dims - 1] = kernel_output_feature_dim;
90 new_kernel_dims[num_dims - 1] =
91 kernel->shape().dimensions(kernel_output_feature_dim);
92
93 Shape new_kernel_shape =
94 ShapeUtil::MakeShape(kernel->shape().element_type(), new_kernel_dims);
95 HloInstruction* new_kernel = module->entry_computation()->AddInstruction(
96 HloInstruction::CreateTranspose(new_kernel_shape, kernel,
97 new_kernel_dim_order));
98
99 std::vector<int64> new_output_dim_order(num_dims);
100 std::vector<int64> new_conv_dims(num_dims);
101 auto output_batch_dim = dnums.output_batch_dimension();
102 auto output_feature_dim = dnums.output_feature_dimension();
103 new_output_dim_order[0] = output_batch_dim;
104 new_conv_dims[0] = hlo->shape().dimensions(output_batch_dim);
105 for (int64 i = 0; i < num_spatial_dims; ++i) {
106 new_output_dim_order[i + 1] = dnums.output_spatial_dimensions(i);
107 new_conv_dims[i + 1] =
108 hlo->shape().dimensions(dnums.output_spatial_dimensions(i));
109 }
110 new_output_dim_order[num_dims - 1] = output_feature_dim;
111 new_conv_dims[num_dims - 1] = hlo->shape().dimensions(output_feature_dim);
112 Shape new_conv_shape =
113 ShapeUtil::MakeShape(hlo->shape().element_type(), new_conv_dims);
114
115 ConvolutionDimensionNumbers new_dnums;
116 new_dnums.set_input_batch_dimension(0);
117 new_dnums.set_output_batch_dimension(0);
118 for (int64 i = 0; i < num_spatial_dims; ++i) {
119 new_dnums.add_input_spatial_dimensions(i + 1);
120 new_dnums.add_kernel_spatial_dimensions(i);
121 new_dnums.add_output_spatial_dimensions(i + 1);
122 }
123 new_dnums.set_input_feature_dimension(num_dims - 1);
124 new_dnums.set_output_feature_dimension(num_dims - 1);
125 new_dnums.set_kernel_input_feature_dimension(num_dims - 2);
126 new_dnums.set_kernel_output_feature_dimension(num_dims - 1);
127
128 // The window of the old convolution is reused, because reshapes only
129 // change the dimension mapping but not the dimension sizes. For
130 // example, input height and width are the same as before the reshapes.
131 HloInstruction* new_conv = module->entry_computation()->AddInstruction(
132 HloInstruction::CreateConvolve(new_conv_shape, new_input, new_kernel,
133 hlo->window(), new_dnums));
134
135 // Reshape the output back to the shape of the original convolution.
136 TF_RETURN_IF_ERROR(module->entry_computation()->ReplaceWithNewInstruction(
137 hlo, HloInstruction::CreateTranspose(
138 hlo->shape(), new_conv,
139 InversePermutation(new_output_dim_order))));
140 changed = true;
141 }
142 }
143
144 return changed;
145}
