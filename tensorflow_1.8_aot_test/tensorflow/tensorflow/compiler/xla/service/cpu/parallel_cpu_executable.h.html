41namespace xla {

44// CPU-targeting parallel implementation of the XLA Executable interface.
45//
46// Wraps a JIT-ed object that can be executed "on device". We JIT for the host
47// architecture, so JIT-ed code and host code share the same ABI.
48class ParallelCpuExecutable : public Executable {
49 public:
50 ParallelCpuExecutable(
51 std::unique_ptr<SimpleOrcJIT> jit,
52 std::unique_ptr<const BufferAssignment> assignment,
53 std::unique_ptr<const HloModule> hlo_module,
54 std::unique_ptr<const HloInstructionMap<string>> function_names,
55 std::unordered_map<const HloInstruction*,
56 std::unique_ptr<unsigned char[]>>
57 aligned_constants,
58 std::unique_ptr<HloProfilePrinterData> hlo_profile_printer_data,
59 std::unique_ptr<HloProfileIndexMap> hlo_profile_index_map);
60 ~ParallelCpuExecutable() override {}
61
62 StatusOr<std::unique_ptr<ShapedBuffer>> ExecuteOnStream(
63 const ServiceExecutableRunOptions* run_options,
64 tensorflow::gtl::ArraySlice<const ShapedBuffer*> arguments,
65 HloExecutionProfile* hlo_execution_profile) override;
66
67 StatusOr<std::unique_ptr<ShapedBuffer>> ExecuteAsyncOnStream(
68 const ServiceExecutableRunOptions* run_options,
69 tensorflow::gtl::ArraySlice<const ShapedBuffer*> arguments) override;
70
71 // This should be called after set_ir_module_string.
72 const string& ir_module_string() const { return ir_module_string_; }
73
74 void set_ir_module_string(const string& ir_module_string) {
75 ir_module_string_ = ir_module_string;
76 }
77
78 static int64 ShapeSizeBytes(const Shape& shape) {
79 // On the cpu, opaques are pointers.
80 if (ShapeUtil::IsOpaque(shape)) {
81 return sizeof(void*);
82 }
83 return ShapeUtil::ByteSizeOf(shape, sizeof(void*));
84 }
85
86 private:
87 // Allocate buffers required for execution and assign them to the elements of
88 // "buffers". "buffers" should be sized to the number of buffers in buffer
89 // assignment. Each vector element corresponds to a particular Index. If
90 // a vector element already contains a non-null DeviceMemoryBase, then no
91 // buffer is assigned for this element.
92 Status AllocateBuffers(
93 DeviceMemoryAllocator* memory_allocator, int device_ordinal,
94 std::vector<perftools::gputools::DeviceMemoryBase>* buffers);
95
96 // Calls the generated functions in 'function_names_', performing the
97 // computation with the given arguments using the supplied buffers.
98 Status ExecuteComputeFunctions(
99 const ServiceExecutableRunOptions* run_options,
100 tensorflow::gtl::ArraySlice<const ShapedBuffer*> arguments,
101 tensorflow::gtl::ArraySlice<perftools::gputools::DeviceMemoryBase>
102 buffers,
103 HloExecutionProfile* hlo_execution_profile);
104
105 // Returns the points-to set of the root instruction of the entry
106 // computation. Uses points-to analysis from buffer assignment.
107 const PointsToSet& GetRootPointsToSet() const;
108
109 // The JIT containing compiled modules.
110 tensorflow::mutex jit_mutex_;
111 const std::unique_ptr<SimpleOrcJIT> jit_ GUARDED_BY(jit_mutex_);
112
113 // Buffer assignment for the buffers we need to allocate.
114 const std::unique_ptr<const BufferAssignment> assignment_;
115
116 // The LLVM IR, in string format, of the unoptimized module generated for this
117 // ParallelCpuExecutable. We save a string instead of an llvm::Module* because
118 // leaving llvm::Module* in a singleton can cause the heap checker to emit
119 // false positives.
120 string ir_module_string_;
121
122 // Map containing the JITted function names for each HLO instruction.
123 const std::unique_ptr<const HloInstructionMap<string>> function_names_;
124
125 // Map from HLO Constant instructions to a pointer to their literal data.
126 // The data stored in the protocol buffer might be insufficiently aligned,
127 // we create a sufficiently aligned copy and store it in this map.
128 const std::unordered_map<const HloInstruction*,
129 std::unique_ptr<unsigned char[]>>
130 aligned_constants_;
131
132 TF_DISALLOW_COPY_AND_ASSIGN(ParallelCpuExecutable);
133};
