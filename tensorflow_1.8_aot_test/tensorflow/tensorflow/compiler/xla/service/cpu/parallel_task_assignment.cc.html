28class SimpleCostModel : public ParallelCostModel {
29 public:
30 SimpleCostModel(const int64 max_parallelism,
31 const HloCostAnalysis::ShapeSizeFunction& shape_size)
32 : max_parallelism_(max_parallelism), shape_size_(shape_size) {}
33 ~SimpleCostModel() override {}
34
35 int64 GetParallelTaskCount(HloInstruction* instruction) override {
36 // Simple cost model based on hlo size and typical L2 cache size.
37 const int64 instruction_cost = shape_size_(instruction->shape());
38 const int64 min_cost_per_thread = 256LL << 10; // 256KB L2 Cache size.
39 // Return target parallel task count in [1, max_parallelism_].
40 return std::min(max_parallelism_,
41 std::max(1LL, instruction_cost / min_cost_per_thread));
42 }
43
44 private:
45 const int64 max_parallelism_;
46 const HloCostAnalysis::ShapeSizeFunction shape_size_;
47};
48
49class DefaultCostModel : public ParallelCostModel {
50 public:
51 DefaultCostModel(const int64 max_parallelism,
52 const HloCostAnalysis::ShapeSizeFunction& shape_size,
53 std::unique_ptr<HloCostAnalysis> cost_analysis)
54 : max_parallelism_(max_parallelism),
55 shape_size_(shape_size),
56 cost_analysis_(std::move(cost_analysis)) {}
57 ~DefaultCostModel() override {}
58
59 int64 GetParallelTaskCount(HloInstruction* instruction) override {
60 // Parameters for parallel task count computation.
61 int64 instruction_cost;
62 int64 min_cost_per_thread;
63 int64 max_parallelism;
64 // Calculate flops-to-bytes-ratio for 'instruction'.
65 const int64 bytes_accessed =
66 std::max(1LL, cost_analysis_->bytes_accessed(*instruction));
67 const float flops_to_bytes_ratio =
68 cost_analysis_->flop_count(*instruction) /
69 static_cast<float>(bytes_accessed);
70 // Check for I/O bound instructions.
71 if (flops_to_bytes_ratio <= 1.0) {
72 // Limit max parallelism for I/O bound instructions by assuming a
73 // sub-linear scaling function (fit based on empirical benchmark results).
74 // TODO(b/29630486) Develop system bandwidth model.
75 max_parallelism =
76 std::ceil(std::sqrt(tensorflow::port::NumSchedulableCPUs()));
77 // Use shape size instruction cost and L2 cache size min per-thread cost.
78 instruction_cost = shape_size_(instruction->shape());
79 min_cost_per_thread = 256LL << 10; // 256KB L2 Cache size.
80 } else {
81 // Use max parallelism for compute bound instructions.
82 max_parallelism = max_parallelism_;
83 // Calculate the instruction cost in cycles.
84 // TODO(b/29630486) Improve on this linear cost model.
85 // Consider making 'min_cost_per_thread' be a function of the target
86 // bandwidth limit for instructions with low arithmetic complexity.
87 instruction_cost =
88 1 * cost_analysis_->flop_count(*instruction) +
89 2 * cost_analysis_->transcendental_count(*instruction) +
90 10 * cost_analysis_->bytes_accessed(*instruction);
91 // Minimum per-thread cost is 100us of work on a 2GHz core.
92 min_cost_per_thread = 100000;
93 }
94 // Return target parallel task count in [1, max_parallelism_].
95 return std::min(max_parallelism,
96 std::max(1LL, instruction_cost / min_cost_per_thread));
97 }
98
99 private:
100 const int64 max_parallelism_;
101 const HloCostAnalysis::ShapeSizeFunction shape_size_;
102 const std::unique_ptr<HloCostAnalysis> cost_analysis_;
103};

105ParallelTaskAssignment::ParallelTaskAssignment(
106 const int64 max_parallelism,
107 const HloCostAnalysis::ShapeSizeFunction& shape_size, HloModule* module) {
108 VLOG(1) << "ParallelTaskAssignment max_parallelism: " << max_parallelism;
109 // Run cost analysis on 'module'.
110 auto cost_analysis = MakeUnique<HloCostAnalysis>(shape_size);
111 HloComputation* computation = module->entry_computation();
112 Status status = computation->root_instruction()->Accept(cost_analysis.get());
113 if (status.ok()) {
114 // Set default cost model based on 'cost_analysis'.
115 cost_model_.reset(new DefaultCostModel(max_parallelism, shape_size,
116 std::move(cost_analysis)));
117 } else {
118 // Fall back to a simple cost model based on hlo size and L2 cache size.
119 // Note that HloCostAnalysis can returns an error status (likely because
120 // HLOs like CustomCall are not yet implemented in the HloCostAnalysis).
121 cost_model_.reset(new SimpleCostModel(max_parallelism, shape_size));
122 }
123}

125int64 ParallelTaskAssignment::GetTargetParallelTaskCount(
126 HloInstruction* instruction) {
127 // Currently, we do not assign parallel tasks to instructions with at least
128 // one of the following properties:
129 // *) Internal threading (library calls to kConv, kDot, kFft, kCustomCall).
130 // *) Emit custom loops (kSelectAndScatter, FusionKind::kTransposeDot).
131 // *) Operations that are not thread safe (like infeed and rng).
132 // *) Tuple-shaped.
133 // TODO(b/27458679) Parallelize instructions which are skipped here.
134 auto opcode = instruction->opcode();
135 if (opcode == HloOpcode::kParameter || opcode == HloOpcode::kConstant ||
136 opcode == HloOpcode::kCall || opcode == HloOpcode::kCustomCall ||
137 opcode == HloOpcode::kDot || opcode == HloOpcode::kSelectAndScatter ||
138 opcode == HloOpcode::kGetTupleElement || opcode == HloOpcode::kBitcast ||
139 opcode == HloOpcode::kFft || opcode == HloOpcode::kInfeed ||
140 opcode == HloOpcode::kOutfeed || opcode == HloOpcode::kRng ||
141 (opcode == HloOpcode::kConvolution &&
142 PotentiallyImplementedAsEigenConvolution(*instruction)) ||
143 PotentiallyImplementedAsEigenDot(*instruction) ||
144 (opcode == HloOpcode::kFusion &&
145 instruction->fusion_kind() != HloInstruction::FusionKind::kLoop) ||
146 ShapeUtil::IsTuple(instruction->shape())) {
147 return 1;
148 }
149
150 // Consult 'cost_model_' to compute target parallel task count.
151 return cost_model_->GetParallelTaskCount(instruction);
152}

154StatusOr<bool> ParallelTaskAssigner::Run(HloModule* module) {
155 XLA_VLOG_LINES(2, "ParallelTaskAssigner ENTRY");
156 XLA_VLOG_LINES(3, module->ToString());
157 // Compute target parallel task counts for all instructions in 'module'.
158 HloToParallelTasks hlo_to_parallel_tasks;
159 ComputeTargetParallelTasks(module, &hlo_to_parallel_tasks);
160
161 // Assign parallel tasks to target specific instructions in 'module'.
162 // TODO(b/27458679) Support inter-op parallelism.
163 bool changed = AssignParallelTasks(module, hlo_to_parallel_tasks);
164
165 XLA_VLOG_LINES(2, "ParallelTaskAssigner EXIT");
166 XLA_VLOG_LINES(3, module->ToString());
167 return changed;
168}

170bool ParallelTaskAssigner::AssignParallelTasks(
171 HloModule* module, const HloToParallelTasks& hlo_to_parallel_tasks) {
172 return AssignParallelTasksHelper(module, module->entry_computation(),
173 hlo_to_parallel_tasks);
174}
175
176bool ParallelTaskAssigner::AssignParallelTasksHelper(
177 HloModule* module, HloComputation* computation,
178 const HloToParallelTasks& hlo_to_parallel_tasks) {
179 bool changed = false;
180 // Snapshot set of instructions because outlining modifies the set below.
181 std::vector<HloInstruction*> instructions(computation->instructions().begin(),
182 computation->instructions().end());
183 for (auto* instruction : instructions) {
184 // Assign parallel tasks to sub-computations for While and Call HLOs.
185 // TODO(b/27458679) Evaluate alternative intra-op parallelsim placement,
186 // and support other callable computations like reduce.
187 if (instruction->opcode() == HloOpcode::kWhile) {
188 changed |= AssignParallelTasksHelper(module, instruction->while_body(),
189 hlo_to_parallel_tasks);
190 continue;
191 } else if (instruction->opcode() == HloOpcode::kCall) {
192 changed |= AssignParallelTasksHelper(module, instruction->to_apply(),
193 hlo_to_parallel_tasks);
194 continue;
195 }
196 // Skip if no parallel tasks were computed in first pass.
197 auto it = hlo_to_parallel_tasks.find(instruction);
198 if (it == hlo_to_parallel_tasks.end()) {
199 continue;
200 }
201 // Get target parallel task count computed for 'instruction'.
202 const int64 target_parallel_task_count = (*it).second;
203 // Assign feasible dimension partitions (based on actual dimension sizes).
204 auto dim_partition_counts = ShapePartitionAssigner(instruction->shape())
205 .Run(target_parallel_task_count);
206 const int64 total_partition_count =
207 ShapePartitionAssigner::GetTotalPartitionCount(dim_partition_counts);
208 if (total_partition_count <= 1) {
209 // Feasible partition calculation resulting in no partitioning, so skip.
210 continue;
211 }
212
213 // Outline 'instruction' in 'computation' for parallel task assignment.
214 auto* call = module->OutlineExpressionFromComputation(
215 {instruction},
216 tensorflow::strings::StrCat("parallel_", instruction->name()),
217 computation);
218
219 // Set assigned dimension partitioning to 'instruction'.
220 auto* new_root = call->to_apply()->root_instruction();
221 new_root->set_outer_dimension_partitions(dim_partition_counts);
222
223 VLOG(2) << "Assigned parallel task count: " << total_partition_count
224 << " to instruction: " << new_root->name()
225 << " parent: " << new_root->parent()->name();
226 changed = true;
227 }
228 return changed;
229}

231void ParallelTaskAssigner::ComputeTargetParallelTasks(
232 HloModule* module, HloToParallelTasks* hlo_to_parallel_tasks) {
233 ParallelTaskAssignment parallel_task_assignment(max_parallelism_,
234 shape_size_function_, module);
235
236 // Compute parallel task counts for all instructions in 'module'.
237 for (auto* computation : module->computations()) {
238 if (computation->IsFusionComputation()) {
239 continue;
240 }
241 for (auto* instruction : computation->instructions()) {
242 // Query ParallelTaskAssignment for target parallel task count.
243 const int64 target_parallel_task_count =
244 parallel_task_assignment.GetTargetParallelTaskCount(instruction);
245 if (target_parallel_task_count > 1) {
246 hlo_to_parallel_tasks->insert(
247 {instruction, target_parallel_task_count});
248 }
249 }
250 }
251}
