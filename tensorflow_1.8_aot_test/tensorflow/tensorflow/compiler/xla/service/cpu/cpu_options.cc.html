22const char* const kXlaParallelCpuOption = "xla_cpu_parallel";
23const char* const kXlaOptimizeForSizeCpuOption = "xla_cpu_optimize_for_size";
24const char* const kXlaDisableVectorizedReduce = "xla_disable_vectorized_reduce";
25const char* const kLlvmIrDotTilingFactor = "xla_llvm_dot_tiling_factor";
26
27} // namespace

29namespace xla {
30namespace cpu {
31namespace options {
32
33bool CpuParallelBackendRequested(const HloModuleConfig& config) {
34 const auto& extra_options_map =
35 config.debug_options().xla_backend_extra_options();
36 return extra_options_map.count(kXlaParallelCpuOption) > 0;
37}
38
39bool OptimizeForSizeRequested(const HloModuleConfig& config) {
40 const auto& extra_options_map =
41 config.debug_options().xla_backend_extra_options();
42 return extra_options_map.count(kXlaOptimizeForSizeCpuOption) > 0;
43}
44
45bool VectorizedReduceDisabled(const HloModuleConfig& config) {
46 const auto& extra_options_map =
47 config.debug_options().xla_backend_extra_options();
48 return extra_options_map.count(kXlaOptimizeForSizeCpuOption) > 0;
49}
50
51tensorflow::gtl::optional<int64> LlvmIrGemvTilingFactor(
52 const HloModuleConfig& config) {
53 const auto& extra_options_map =
54 config.debug_options().xla_backend_extra_options();
55 auto it = extra_options_map.find(kLlvmIrDotTilingFactor);
56 int64 tiling_factor;
57 if (it != extra_options_map.end() &&
58 tensorflow::strings::safe_strto64(it->second, &tiling_factor)) {
59 return tiling_factor;
60 }
61 return tensorflow::gtl::nullopt;
62}
