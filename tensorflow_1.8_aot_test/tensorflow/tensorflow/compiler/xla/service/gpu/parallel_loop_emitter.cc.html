53llvm_ir::IrArray::Index ParallelLoopEmitter::EmitIndexAndSetExitBasicBlock(
54 tensorflow::StringPiece loop_name) {
55 // Emit the following code in LLVM IR:
56 // linear_index = blockIdx.x * blockDim.x + threadIdx.x;
57 // if (linear_index < num_elements) {
58 // array_index = LinearIndexToMultidimensionalIndex(shape_, linear_index);
59 // ...
60 // }
61
62 // Per the PTX documentation:
63 // "It is guaranteed that [...] 0 <= %ctaid.x < %nctaid.x"
64 //
65 // %nctaid.x is currently specified as 2147483647.
66 llvm::Value* block_id = llvm_ir::EmitCallToIntrinsic(
67 llvm::Intrinsic::nvvm_read_ptx_sreg_ctaid_x, {}, {}, ir_builder_);
68 llvm_ir::AddRangeMetadata(0, launch_dimensions_.block_count(),
69 static_cast<llvm::Instruction*>(block_id));
70 block_id =
71 ir_builder_->CreateZExt(block_id, ir_builder_->getInt64Ty(), "block_id");
72
73 // Per the PTX documentation:
74 // "It is guaranteed that [...] 0 <= %tid.x < %ntid.x"
75 //
76 // %ntid.x is currently specified as 1024.
77 llvm::Value* thread_id = llvm_ir::EmitCallToIntrinsic(
78 llvm::Intrinsic::nvvm_read_ptx_sreg_tid_x, {}, {}, ir_builder_);
79 llvm_ir::AddRangeMetadata(0, launch_dimensions_.threads_per_block(),
80 static_cast<llvm::Instruction*>(thread_id));
81 thread_id = ir_builder_->CreateZExt(thread_id, ir_builder_->getInt64Ty(),
82 "thread_id");
83
84 llvm::Value* linear_index = ir_builder_->CreateAdd(
85 ir_builder_->CreateMul(
86 block_id,
87 ir_builder_->getInt64(launch_dimensions_.threads_per_block()), "",
88 /*HasNUW=*/true, /*HasNSW=*/true),
89 thread_id, "linear_index", /*HasNUW=*/true, /*HasNSW=*/true);
90
91 // Add an @llvm.assume(linear_index < threads_per_block * num_blocks).
92 //
93 // This might seem obvious from the computation above, but LLVM does not
94 // currently determine the range of linear_index precisely. InstCombine uses
95 // known-bits, which, when applied to the task of determining a value's range,
96 // is imprecise for everything other than powers of 2. And
97 // CorrelatedValuePropagation is, as a cost-saving measure, disabled for
98 // conditions in the same basic block as their operands.
99 llvm_ir::EmitCallToIntrinsic(
100 llvm::Intrinsic::assume,
101 {ir_builder_->CreateICmpULT(
102 linear_index,
103 ir_builder_->getInt64(launch_dimensions_.threads_per_block() *
104 launch_dimensions_.block_count()),
105 "linear_index_in_range")},
106 {}, ir_builder_);
107
108 auto if_in_bounds = llvm_ir::EmitIfThenElse(
109 ir_builder_->CreateICmpULT(
110 linear_index, ir_builder_->getInt64(ShapeUtil::ElementsIn(shape_))),
111 llvm_ir::IrName(loop_name, "in_bounds"), ir_builder_, false);
112
113 // Set exit_bb_ to the exit block of the if structure.
114 exit_bb_ = if_in_bounds.after_block;
115 CHECK_NE(nullptr, exit_bb_);
116
117 // Set IR builder insertion point to the body of the if structure.
118 llvm_ir::SetToFirstInsertPoint(if_in_bounds.true_block, ir_builder_);
119 return llvm_ir::IrArray::Index(linear_index, shape_, ir_builder_);
120}
