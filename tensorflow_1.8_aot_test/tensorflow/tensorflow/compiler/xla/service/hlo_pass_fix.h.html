25namespace xla {
26
27// Do an HLO pass to a fix point.
28template <typename Pass>
29class HloPassFix : public Pass {
30 public:
31 template <typename... Args>
32 explicit HloPassFix(Args&&... args) : Pass(args...) {}
33
34 StatusOr<bool> Run(HloModule* module) override {
35 bool changed = false;
36 bool changed_this_iteration = true;
37 while (changed_this_iteration) {
38 TF_ASSIGN_OR_RETURN(changed_this_iteration, Pass::Run(module));
39 changed |= changed_this_iteration;
40 }
41 return changed;
42 }
43};
