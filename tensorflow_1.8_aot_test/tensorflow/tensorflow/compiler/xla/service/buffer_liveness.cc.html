38/* static */
39StatusOr<std::unique_ptr<BufferLiveness>> BufferLiveness::Run(
40 const HloModule* module, std::unique_ptr<HloOrdering> hlo_ordering) {
41 std::unique_ptr<BufferLiveness> liveness(
42 new BufferLiveness(module, std::move(hlo_ordering)));
43 TF_RETURN_IF_ERROR(liveness->Analyze());
44 return std::move(liveness);
45}
46
47tensorflow::Status BufferLiveness::Analyze() {
48 TF_ASSIGN_OR_RETURN(points_to_analysis_, TuplePointsToAnalysis::Run(module_));
49 for (auto* computation : module_->computations()) {
50 if (computation->IsFusionComputation()) {
51 continue;
52 }
53 // Gather all instructions whose buffers might alias other instructions into
54 // the set aliased_buffers_. This includes those contained as a tuple
55 // element in other instruction's output.
56 for (const auto& instruction : computation->instructions()) {
57 for (const LogicalBuffer* aliased_buffer :
58 points_to_analysis_->GetPointsToSet(instruction)
59 .CreateFlattenedSet()) {
60 if (aliased_buffer->instruction() != instruction) {
61 aliased_buffers_.insert(aliased_buffer);
62 }
63 }
64 }
65
66 if (computation == module_->entry_computation()) {
67 const HloInstruction* root = computation->root_instruction();
68 maybe_live_out_buffers_ =
69 points_to_analysis_->GetPointsToSet(root).CreateFlattenedSet();
70 }
71 }
72
73 XLA_VLOG_LINES(3, ToString());
74 return tensorflow::Status::OK();
75}

77string BufferLiveness::ToString() const {
78 std::vector<string> pieces;
79 pieces.push_back(tensorflow::strings::Printf("BufferLiveness(module=%s):",
80 module_->name().c_str()));
81 pieces.push_back("HloOrdering:");
82 pieces.push_back(hlo_ordering_->ToString());
83 pieces.push_back(tensorflow::strings::Printf("Aliased buffers:"));
84 for (const LogicalBuffer* buffer : aliased_buffers_) {
85 pieces.push_back(
86 tensorflow::strings::Printf(" %s", buffer->ToString().c_str()));
87 }
88 pieces.push_back(tensorflow::strings::Printf("Live out buffers:"));
89 for (const LogicalBuffer* buffer : maybe_live_out_buffers_) {
90 pieces.push_back(
91 tensorflow::strings::Printf(" %s", buffer->ToString().c_str()));
92 }
93 return tensorflow::str_util::Join(pieces, "\n");
94}

96bool BufferLiveness::live_range_strictly_before(const LogicalBuffer& a,
97 const LogicalBuffer& b) const {
98 TF_CHECK_OK(points_to_analysis_->VerifyBuffer(a));
99 TF_CHECK_OK(points_to_analysis_->VerifyBuffer(b));
100
101 if (!hlo_ordering_->ExecutesBefore(a.instruction(), b.instruction())) {
102 return false;
103 }
104
105 for (const BufferAlias& alias : points_to_analysis_->GetBufferAliases(a)) {
106 // Every user of 'a' must be a predecessor of 'b' or 'b' itself.
107 for (auto user : alias.instruction()->users()) {
108 if (DoesNotUseOperandBuffer(alias.instruction(), alias.index(), user,
109 points_to_analysis())) {
110 continue;
111 }
112 if (user != b.instruction() &&
113 !hlo_ordering_->ExecutesBefore(user, b.instruction())) {
114 return false;
115 }
116 }
117
118 // If the root instruction aliases the buffer 'a', the live range of 'a' is
119 // until the end of the computation and can never be strictly before another
120 // buffer defined in the same computation. This is needed to prevent the
121 // root instruction's buffers from being reused by later instructions even
122 // when the root is not the last instruction in the schedule.
123 if (alias.instruction()->parent()->root_instruction() ==
124 alias.instruction() &&
125 alias.instruction()->parent() == b.instruction()->parent()) {
126 return false;
127 }
128 }
129
130 // If 'b' is a user of 'a' then the buffers interfere unless 'a.instruction'
131 // and 'b.instruction' emit the same shape/layout, and 'b.instruction' meets
132 // the qualifications specified in CanShareOperandBufferWithUser.
133 for (const BufferAlias& alias : points_to_analysis_->GetBufferAliases(a)) {
134 if (b.instruction()->IsUserOf(alias.instruction()) &&
135 !CanShareOperandBufferWithUser(alias.instruction(), alias.index(),
136 b.instruction(), b.index(),
137 points_to_analysis())) {
138 return false;
139 }
140 }
141 return true;
142}

144namespace {
145bool IsEntryParameter(const HloInstruction* instruction) {
146 const HloComputation* computation = instruction->parent();
147 return instruction->opcode() == HloOpcode::kParameter &&
148 computation == computation->parent()->entry_computation();
149}
150} // namespace

152bool BufferLiveness::MayInterfere(const LogicalBuffer& a,
153 const LogicalBuffer& b) const {
154 // Entry parameters live at the entry of the execution, thus always interfere
155 // with all other instructions executing before them in the ordering.
156 const HloInstruction* a_instruction = a.instruction();
157 const HloInstruction* b_instruction = b.instruction();
158 if (IsEntryParameter(a_instruction) &&
159 hlo_ordering_->ExecutesBefore(b_instruction, a_instruction)) {
160 return true;
161 }
162 if (IsEntryParameter(b_instruction) &&
163 hlo_ordering_->ExecutesBefore(a_instruction, b_instruction)) {
164 return true;
165 }
166 // Buffers without disjoint liveness may interfere.
167 return !live_range_strictly_before(a, b) && !live_range_strictly_before(b, a);
168}
169
170bool BufferLiveness::MaybeLiveOut(const LogicalBuffer& buffer) const {
171 // Verify that a buffer is actually defined at the given instruction/index
172 // (eg, its not an alias of another buffer such as occurs with a bitcast).
173 TF_CHECK_OK(points_to_analysis_->VerifyBuffer(buffer));
174 return maybe_live_out_buffers_.count(&buffer);
175}
