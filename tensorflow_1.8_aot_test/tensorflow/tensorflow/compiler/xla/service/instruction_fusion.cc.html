31/*static*/ bool InstructionFusion::IsExpensive(
32 const HloInstruction& instruction) {
33 switch (instruction.opcode()) {
34 // Cheap instructions.
35 case HloOpcode::kAdd:
36 case HloOpcode::kAnd:
37 case HloOpcode::kBitcast:
38 case HloOpcode::kBitcastConvert:
39 case HloOpcode::kBroadcast:
40 case HloOpcode::kBroadcastDimOne:
41 case HloOpcode::kCeil:
42 case HloOpcode::kClamp:
43 case HloOpcode::kComplex:
44 case HloOpcode::kConcatenate:
45 case HloOpcode::kConstant:
46 case HloOpcode::kConvert:
47 case HloOpcode::kCopy:
48 case HloOpcode::kDynamicSlice:
49 case HloOpcode::kDynamicUpdateSlice:
50 case HloOpcode::kEq:
51 case HloOpcode::kFloor:
52 case HloOpcode::kGe:
53 case HloOpcode::kGetTupleElement:
54 case HloOpcode::kGt:
55 case HloOpcode::kImag:
56 case HloOpcode::kInfeed:
57 case HloOpcode::kIsFinite:
58 case HloOpcode::kLe:
59 case HloOpcode::kLt:
60 case HloOpcode::kMaximum:
61 case HloOpcode::kMinimum:
62 case HloOpcode::kMultiply:
63 case HloOpcode::kNe:
64 case HloOpcode::kNegate:
65 case HloOpcode::kNot:
66 case HloOpcode::kOr:
67 case HloOpcode::kOutfeed:
68 case HloOpcode::kPad:
69 case HloOpcode::kReal:
70 case HloOpcode::kReducePrecision:
71 case HloOpcode::kReshape:
72 case HloOpcode::kReverse:
73 case HloOpcode::kRoundNearestAfz:
74 case HloOpcode::kSelect:
75 case HloOpcode::kShiftLeft:
76 case HloOpcode::kShiftRightArithmetic:
77 case HloOpcode::kShiftRightLogical:
78 case HloOpcode::kSlice:
79 case HloOpcode::kSubtract:
80 case HloOpcode::kTranspose:
81 case HloOpcode::kTuple:
82 return false;
83
84 // Cheap instructions for reals, but expensive for complex.
85 case HloOpcode::kAbs:
86 case HloOpcode::kCos:
87 case HloOpcode::kSign:
88 case HloOpcode::kSin:
89 return ShapeUtil::ElementIsComplex(instruction.shape());
90
91 // Expensive instructions.
92 case HloOpcode::kAtan2:
93 case HloOpcode::kBatchNormGrad:
94 case HloOpcode::kBatchNormInference:
95 case HloOpcode::kBatchNormTraining:
96 case HloOpcode::kCall:
97 case HloOpcode::kConditional:
98 case HloOpcode::kConvolution:
99 case HloOpcode::kCrossReplicaSum:
100 case HloOpcode::kCustomCall:
101 case HloOpcode::kDivide:
102 case HloOpcode::kDot:
103 case HloOpcode::kExp:
104 case HloOpcode::kFft:
105 case HloOpcode::kFusion:
106 case HloOpcode::kGather:
107 case HloOpcode::kHostCompute:
108 case HloOpcode::kLog:
109 case HloOpcode::kMap:
110 case HloOpcode::kParameter:
111 case HloOpcode::kPower:
112 case HloOpcode::kRecv:
113 case HloOpcode::kRecvDone:
114 case HloOpcode::kReduce:
115 case HloOpcode::kReduceWindow:
116 case HloOpcode::kRemainder:
117 case HloOpcode::kRng:
118 case HloOpcode::kSelectAndScatter:
119 case HloOpcode::kSend:
120 case HloOpcode::kSendDone:
121 case HloOpcode::kSort:
122 case HloOpcode::kTanh:
123 case HloOpcode::kTrace:
124 case HloOpcode::kWhile:
125 return true;
126 }
127
128 return false;
129}

131// An "effectively unary" operation is one that has one "large"
132// input with the others being negligible in terms of memory usage.
133// We use "has a smaller true rank than the output" as a heuristic
134// for "negligible" memory usage.
135bool InstructionFusion::EffectivelyUnary(HloInstruction* hlo) {
136 int64 output_rank = 0;
137 ShapeUtil::ForEachSubshape(
138 hlo->shape(),
139 [&output_rank](const Shape& subshape, const ShapeIndex& shape_index) {
140 if (ShapeUtil::IsArray(subshape)) {
141 output_rank = std::max(output_rank, ShapeUtil::TrueRank(subshape));
142 }
143 });
144 return std::count_if(hlo->operands().begin(), hlo->operands().end(),
145 [output_rank](HloInstruction* operand) {
146 if (operand->opcode() == HloOpcode::kBroadcast ||
147 operand->opcode() == HloOpcode::kBroadcastDimOne) {
148 return false;
149 }
150 if (operand->opcode() == HloOpcode::kConstant &&
151 ShapeUtil::IsEffectiveScalar(operand->shape())) {
152 return false;
153 }
154 return ShapeUtil::TrueRank(operand->shape()) >=
155 output_rank;
156 }) <= 1;
157}

159bool InstructionFusion::CanFuseOnAllPaths(
160 const HloReachabilityMap& reachability_map, HloInstruction* producer,
161 HloInstruction* consumer, DoNotFuseSet* do_not_fuse) {
162 auto could_fuse_on_all_paths = [&] {
163 // First check to see if we have already marked this producer as infeasible
164 // to fuse into consumer.
165 if (do_not_fuse->count(producer) > 0) {
166 return false;
167 }
168 // Make sure it is possible for producer and consumer to exist in a fusion
169 // node.
170 if (!producer->IsFusable() || !consumer->IsFusable()) {
171 return false;
172 }
173 // We do an upward walk of the graph from consumer towards all paths which
174 // lead to producer to find any unfusable paths.
175 for (int64 i = 0, e = consumer->operand_count(); i < e; ++i) {
176 auto* consumer_operand = consumer->mutable_operand(i);
177 if (consumer_operand == producer) {
178 // This is the base case: our upward crawl ends but we need to make sure
179 // that fusion from consumer can happen.
180 if (!ShouldFuse(consumer, i)) {
181 return false;
182 }
183 } else if (reachability_map.IsReachable(producer, consumer_operand)) {
184 // The reachability map told us that consumer_operand is a node on the
185 // path to producer. We need to further investigate from
186 // consumer_operand.
187
188 // First check if we have already ruled out fusing producer into
189 // consumer_operand.
190 if (do_not_fuse->count(consumer_operand) > 0) {
191 return false;
192 }
193 // Make sure it is possible for consumer_operand to exist in a fusion
194 // node.
195 if (!consumer_operand->IsFusable()) {
196 return false;
197 }
198 // The producer is reachable from consumer_operand which means we need
199 // to be able to fuse consumer_operand into consumer in order for
200 // producer to be fusable into consumer on all paths.
201 if (!ShouldFuse(consumer, i)) {
202 return false;
203 }
204 // Perform the recursive step: make sure producer can be fused into
205 // consumer_operand on all paths.
206 if (!CanFuseOnAllPaths(reachability_map, producer, consumer_operand,
207 do_not_fuse)) {
208 return false;
209 }
210 }
211 }
212 return true;
213 };
214 if (could_fuse_on_all_paths()) {
215 return true;
216 }
217 // We couldn't fuse on all paths, record this result.
218 do_not_fuse->insert(producer);
219 return false;
220}

222StatusOr<bool> InstructionFusion::Run(HloModule* module) {
223 VLOG(2) << "Before instruction fusion:";
224 XLA_VLOG_LINES(2, module->ToString());
225
226 bool changed = false;
227 module_ = module;
228 for (auto* computation : module->MakeNonfusionComputations()) {
229 CHECK(!computation->IsFusionComputation());
230 computation_ = computation;
231
232 // We want to be able to remove arbitrary instructions from the post order
233 // and also compare positions of instructions in the post order. To make
234 // this possible, create vector of instructions in post order and create a
235 // map from HloInstruction* to the instruction's index in the vector. An
236 // instruction is "removed" from the vector by setting it's element to
237 // nullptr.
238 std::list<HloInstruction*> post_order_list =
239 computation_->MakeInstructionPostOrder();
240 std::vector<HloInstruction*> post_order(post_order_list.begin(),
241 post_order_list.end());
242
243 tensorflow::gtl::FlatMap<HloInstruction*, int> post_order_index;
244 for (size_t i = 0; i < post_order.size(); ++i) {
245 InsertOrDie(&post_order_index, post_order[i], i);
246 }
247
248 DoNotFuseSet do_not_fuse;
249 auto reachability = computation->ComputeReachability();
250
251 auto cheap_to_duplicate = [this](HloInstruction* producer) {
252 if (producer->opcode() == HloOpcode::kBroadcast ||
253 producer->opcode() == HloOpcode::kBroadcastDimOne) {
254 return true;
255 }
256 if (producer->opcode() == HloOpcode::kConstant &&
257 ShapeUtil::IsEffectiveScalar(producer->shape())) {
258 return true;
259 }
260 if (EffectivelyUnary(producer)) {
261 return true;
262 }
263 return false;
264 };
265
266 for (HloInstruction* consumer : post_order) {
267 for (HloInstruction* producer : consumer->operands()) {
268 if (cheap_to_duplicate(producer)) {
269 continue;
270 }
271 if (CanFuseOnAllPaths(*reachability, producer, consumer,
272 &do_not_fuse)) {
273 CHECK_EQ(do_not_fuse.count(producer), 0);
274 } else {
275 CHECK_GT(do_not_fuse.count(producer), 0);
276 }
277 }
278 }
279
280 // Instruction fusion effectively fuses edges in the computation graph
281 // (producer instruction -> consumer instruction) so we iterate over all
282 // edges. When we fuse an edge, we create a copy of the producer inside the
283 // fusion instruction.
284 while (!post_order.empty()) {
285 // We want to iterate in reverse post order, so remove from the back of
286 // the vector.
287 HloInstruction* instruction = post_order.back();
288 post_order.pop_back();
289
290 // Instructions are "removed" from the post order by nulling out the
291 // element in the vector, so if the pointer is null, continue to the next
292 // instruction in the sort.
293 if (instruction == nullptr) {
294 continue;
295 }
296
297 // Remove instruction from the index map to ensure the vector and map stay
298 // consistent.
299 post_order_index.erase(instruction);
300
301 if (!instruction->IsFusable() &&
302 instruction->opcode() != HloOpcode::kFusion) {
303 continue;
304 }
305
306 // Consider each operand of this instruction for fusion into this
307 // instruction. We want to consider the operands in a particular order to
308 // avoid creating duplicate instruction clones in the fusion instruction.
309 // For example, consider the following expression:
310 //
311 // A = ...
312 // B = op(A)
313 // C = op(A, B)
314 //
315 // If we are considering the operands of C for fusion into C. We might
316 // fuse A or B first. If we fuse A first, we get:
317 //
318 // A = ...
319 // B = op(A)
320 // C_fusion = { A' = ...
321 // C' = op(A', B) }
322 //
323 // Where A' and C' are clones of A and C, respectively. Now only B is an
324 // operand of the fusion instruction C_fusion, so then we fuse B:
325 //
326 // A = ...
327 // B = op(A)
328 // C_fusion = { A' = ...
329 // B' = op(A)
330 // C' = op(A', B') }
331 //
332 // Now A is an operand of C_fusion again, so we then fuse A (again!):
333 //
334 // A = ...
335 // B = op(A)
336 // C_fusion = { A' = ...
337 // A" = ..
338 // B' = op(A")
339 // C' = op(A', B') }
340 //
341 // We prevent this duplication by considering the operands in the reverse
342 // order they appear in the instruction post order. In the example, this
343 // ensures that B will be considered before A.
344 //
345 // We store the original indices of the operands to pass to ShouldFuse.
346 std::vector<int64> sorted_operand_numbers(instruction->operands().size());
347 std::iota(std::begin(sorted_operand_numbers),
348 std::end(sorted_operand_numbers), 0);
349 std::sort(
350 sorted_operand_numbers.begin(), sorted_operand_numbers.end(),
351 [&](int64 i, int64 j) {
352 // Instructions with higher indices in the post order come
353 // first.
354 return (
355 FindOrDie(post_order_index, instruction->mutable_operand(i)) >
356 FindOrDie(post_order_index, instruction->mutable_operand(j)));
357 });
358
359 for (int64 i : sorted_operand_numbers) {
360 HloInstruction* operand = instruction->mutable_operand(i);
361
362 if (!operand->IsFusable()) {
363 continue;
364 }
365 if (!ShouldFuse(instruction, i)) {
366 continue;
367 }
368 if (do_not_fuse.count(operand) > 0) {
369 continue;
370 }
371 HloInstruction* fusion_instruction = Fuse(operand, instruction);
372
373 // Fusing an instruction into a fusion instruction can change the
374 // operand set of the fusion instruction. For simplicity just push the
375 // instruction to the top of the post_order and reconsider it for
376 // further fusion in the next iteration of the outer loop.
377 post_order.push_back(fusion_instruction);
378 InsertOrDie(&post_order_index, fusion_instruction,
379 post_order.size() - 1);
380 changed = true;
381
382 if (operand->user_count() == 0) {
383 // Operand is now dead. Remove from post order by setting its
384 // location to nullptr.
385 post_order[FindOrDie(post_order_index, operand)] = nullptr;
386 post_order_index.erase(operand);
387
388 // Remove from computation.
389 TF_RETURN_IF_ERROR(computation_->RemoveInstruction(operand));
390 }
391 break;
392 }
393 }
394 }
395
396 VLOG(2) << "After instruction fusion:";
397 XLA_VLOG_LINES(2, module->ToString());
398
399 return changed;
400}

402HloInstruction* InstructionFusion::Fuse(HloInstruction* producer,
403 HloInstruction* consumer) {
404 HloInstruction* fusion_instruction;
405
406 VLOG(2) << "Fusing " << producer->ToString() << " into "
407 << consumer->ToString();
408 auto kind = ChooseKind(producer, consumer);
409 if (consumer->opcode() == HloOpcode::kFusion) {
410 fusion_instruction = consumer;
411 if (kind != fusion_instruction->fusion_kind()) {
412 fusion_instruction->set_fusion_kind(kind);
413 }
414 } else {
415 fusion_instruction = computation_->AddInstruction(
416 HloInstruction::CreateFusion(consumer->shape(), kind, consumer));
417 TF_CHECK_OK(computation_->ReplaceInstruction(consumer, fusion_instruction));
418 }
419
420 fusion_instruction->FuseInstruction(producer);
421 return fusion_instruction;
422}

424bool InstructionFusion::ShouldFuse(HloInstruction* consumer,
425 int64 operand_index) {
426 HloInstruction* producer = consumer->mutable_operand(operand_index);
427 // Cost condition: don't duplicate expensive instructions.
428 if (FusionWouldDuplicate(*producer, *consumer) &&
429 (is_expensive_(*producer) || !may_duplicate_)) {
430 return false;
431 }
432
433 if (consumer->opcode() == HloOpcode::kFusion &&
434 consumer->fusion_kind() != HloInstruction::FusionKind::kLoop &&
435 consumer->fusion_kind() != HloInstruction::FusionKind::kInput &&
436 consumer->fusion_kind() != HloInstruction::FusionKind::kOutput) {
437 return false;
438 }
439
440 if (producer->CouldBeBitcast() &&
441 // We can't fuse parameters anyhow, so we leave the user unfused to become
442 // a bitcast. If the operand is not a parameter, we would break a
443 // potential fusion to make it a bitcast, which is not so clear a win.
444 producer->operand(0)->opcode() == HloOpcode::kParameter) {
445 return false;
446 }
447
448 return true;
449}
450
451HloInstruction::FusionKind InstructionFusion::ChooseKind(
452 const HloInstruction* producer, const HloInstruction* consumer) {
453 return HloInstruction::FusionKind::kLoop;
454}
