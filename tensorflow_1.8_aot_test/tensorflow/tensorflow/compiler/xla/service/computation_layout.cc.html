24namespace xla {
25
26ComputationLayout::ComputationLayout(const ProgramShape& program_shape)
27 : result_layout_(program_shape.result()) {
28 for (auto& shape : program_shape.parameters()) {
29 parameter_layouts_.emplace_back(shape);
30 }
31 SetToDefaultLayout();
32}
33
34void ComputationLayout::SetToDefaultLayout() {
35 for (auto& parameter_layout : parameter_layouts_) {
36 parameter_layout.SetToDefaultLayout();
37 }
38 result_layout_.SetToDefaultLayout();
39}
40
41bool ComputationLayout::LayoutIsSet() const {
42 return std::all_of(parameter_layouts_.begin(), parameter_layouts_.end(),
43 [](const ShapeLayout& s) { return s.LayoutIsSet(); }) &&
44 result_layout_.LayoutIsSet();
45}
46
47string ComputationLayout::ToString() const {
48 std::vector<string> params;
49 for (auto& param_layout : parameter_layouts_) {
50 params.push_back(param_layout.ToString());
51 }
52 return tensorflow::strings::StrCat("(",
53 tensorflow::str_util::Join(params, ", "),
54 ") => ", result_layout_.ToString());
55}
