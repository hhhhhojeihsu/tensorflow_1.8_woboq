25// Visitor which verifies that the output shape is correctly set. Verifies
26// against the inferred shape for the instruction.
27// TODO(b/26024837): Check output shape for all instruction types.
28class ShapeVerifier : public DfsHloVisitor {
29 public:
30 explicit ShapeVerifier() : allow_mixed_precision_(false) {}
31 explicit ShapeVerifier(bool allow_mixed_precision)
32 : allow_mixed_precision_(allow_mixed_precision) {}
33
34 Status HandleElementwiseUnary(HloInstruction* hlo) override;
35 Status HandleElementwiseBinary(HloInstruction* hlo) override;
36 Status HandleClamp(HloInstruction* clamp) override;
37 Status HandleSelect(HloInstruction* select) override;
38 Status HandleConcatenate(HloInstruction* concatenate) override;
39 Status HandleConvert(HloInstruction* convert) override;
40 Status HandleBitcastConvert(HloInstruction* convert) override;
41 Status HandleCopy(HloInstruction* copy) override;
42 Status HandleDot(HloInstruction* dot) override;
43 Status HandleConvolution(HloInstruction* convolution) override;
44 Status HandleFft(HloInstruction* fft) override;
45 Status HandleCrossReplicaSum(HloInstruction* crs) override;
46 Status HandleReducePrecision(HloInstruction* reduce_precision) override;
47 Status HandleInfeed(HloInstruction*) override;
48 Status HandleOutfeed(HloInstruction*) override;
49 Status HandleRng(HloInstruction*) override;
50 Status HandleReverse(HloInstruction* reverse) override;
51 Status HandleSort(HloInstruction* sort) override;
52 Status HandleConstant(HloInstruction* constant) override;
53 Status HandleGetTupleElement(HloInstruction* get_tuple_element) override;
54 Status HandleReduce(HloInstruction* reduce) override;
55 Status HandleBitcast(HloInstruction* bitcast) override;
56 Status HandleBroadcast(HloInstruction* broadcast) override;
57 Status HandleBroadcastDimOne(HloInstruction* broadcastDimOne) override;
58 Status HandleReshape(HloInstruction* reshape) override;
59 Status HandleTranspose(HloInstruction* transpose) override;
60 Status HandleParameter(HloInstruction*) override;
61 Status HandleFusion(HloInstruction*) override;
62 Status HandleCall(HloInstruction* call) override;
63 Status HandleCustomCall(HloInstruction*) override;
64 Status HandleHostCompute(HloInstruction*) override;
65 Status HandleSlice(HloInstruction* slice) override;
66 Status HandleDynamicSlice(HloInstruction* dynamic_slice) override;
67 Status HandleDynamicUpdateSlice(
68 HloInstruction* dynamic_update_slice) override;
69 Status HandleTuple(HloInstruction* tuple) override;
70 Status HandleMap(HloInstruction* map) override;
71 Status HandleReduceWindow(HloInstruction* reduce_window) override;
72 Status HandleSelectAndScatter(HloInstruction* instruction) override;
73 Status HandleWhile(HloInstruction* xla_while) override;
74 Status HandleConditional(HloInstruction* conditional) override;
75 Status HandlePad(HloInstruction* pad) override;
76 Status HandleSend(HloInstruction* send) override;
77 Status HandleSendDone(HloInstruction* send_done) override;
78 Status HandleRecv(HloInstruction* recv) override;
79 Status HandleRecvDone(HloInstruction* recv_done) override;
80 Status HandleBatchNormTraining(HloInstruction* batch_norm_training) override;
81 Status HandleBatchNormInference(
82 HloInstruction* batch_norm_inference) override;
83 Status HandleBatchNormGrad(HloInstruction* batch_norm_grad) override;
84 Status HandleGather(HloInstruction* gather) override;
85
86 Status FinishVisit(HloInstruction*) override {
87 return tensorflow::Status::OK();
88 }
89
90 protected:
91 // Check the instruction's shape against the shape given by ShapeInference
92 // and return an appropriate error if there is a mismatch.
93 Status CheckShape(const HloInstruction* instruction,
94 const Shape& inferred_shape);
95
96 // Overload which takes a StatusOr to reduce boilerplate in the caller.
97 Status CheckShape(const HloInstruction* instruction,
98 const StatusOr<Shape>& inferred_shape_status);
99
100 // Check a unary (binary, etc) instruction's shape against the inferred shape.
101 Status CheckUnaryShape(const HloInstruction* instruction);
102 Status CheckBinaryShape(const HloInstruction* instruction);
103 Status CheckTernaryShape(const HloInstruction* instruction);
104 Status CheckVariadicShape(const HloInstruction* instruction);
105
106 // Checks if the given two instructions shares the same channel id.
107 Status CheckSameChannel(const HloInstruction* instr1,
108 const HloInstruction* instr2);
109
110 private:
111 // Whether the inputs and output of an instruction can contain both F32s and
112 // BF16s. Tuples that include both F32s and BF16s are allowed regardless of
113 // this flag.
114 bool allow_mixed_precision_;
115};

117// HLO pass that verifies invariants of HLO instructions for each computation in
118// the module.
119class HloVerifier : public HloPassInterface {
120 public:
121 using ShapeVerifierFactory = std::function<std::unique_ptr<ShapeVerifier>()>;
122
123 // Uses standard shape inference.
124 explicit HloVerifier()
125 : shape_verifier_factory_(
126 [] { return MakeUnique<ShapeVerifier>(false); }) {}
127
128 explicit HloVerifier(bool allow_mixed_precision)
129 : shape_verifier_factory_([allow_mixed_precision] {
130 return MakeUnique<ShapeVerifier>(allow_mixed_precision);
131 }) {}
132
133 // Uses custom shape verification.
134 explicit HloVerifier(ShapeVerifierFactory shape_verifier_factory)
135 : shape_verifier_factory_(std::move(shape_verifier_factory)) {}
136
137 ~HloVerifier() override = default;
138 tensorflow::StringPiece name() const override { return "verifier"; }
139
140 // Note: always returns false (no instructions are ever modified by this
141 // pass).
142 StatusOr<bool> Run(HloModule* module) override;
143
144 private:
145 // CHECKs various invariants of a fusion instruction.
146 Status CheckFusionInstruction(HloInstruction* fusion) const;
147
148 // Creates a ShapeVerifier that checks that shapes match inferred
149 // expectations. This is a factory function because ShapeVerifier, Note that
150 // ShapeVerifier, being a DfsHloVisitor, is stateful. We want a clean object
151 // for each run of the verifier.
152 ShapeVerifierFactory shape_verifier_factory_;
153};
