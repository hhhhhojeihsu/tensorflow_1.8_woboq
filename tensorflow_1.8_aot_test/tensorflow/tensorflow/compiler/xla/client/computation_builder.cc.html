43void ComputationBuilder::NoteError(const Status& error) {
44 if (die_immediately_on_error_) {
45 LOG(FATAL) << "error building computation: " << error;
46 }
47
48 if (first_error_.ok()) {
49 first_error_ = error;
50 first_error_backtrace_.CreateCurrent(/*skip_count=*/1);

62Status ComputationBuilder::PrepareComputation() {
63 TF_RETURN_IF_ERROR(first_error_);
64
65 if (!computation_.IsNull()) {
66 return Status::OK();
67 }
68
69 ComputationRequest request;
70 request.set_name(name_);
71 ComputationResponse response;
72
73 VLOG(2) << "making computation request";
74 Status s = client_->stub()->Computation(&request, &response);
75 VLOG(2) << "done with computation request";
76
77 if (!s.ok()) {
78 NoteError(s);
79 return first_error_;
80 }
81
82 computation_ = Computation(client_->stub(), response.computation());
83 return Status::OK();
84}

86Status ComputationBuilder::RunOp(OpRequest* op_request,
87 OpResponse* op_response) {
88 TF_RETURN_IF_ERROR(first_error_);
89 TF_RETURN_IF_ERROR(PrepareComputation());
90
91 // Fill in fields that are set on every OpRequest.
92 *op_request->mutable_computation() = computation_.handle();
93 *op_request->mutable_metadata() = metadata_;
94 if (sharding_) {
95 *op_request->mutable_sharding() = *sharding_;
96 }
97
98 const string& op_name =
99 OpRequest::descriptor()->FindFieldByNumber(op_request->op_case())->name();
100 VLOG(2) << "running op request: " << op_name;
101 Status status = client_->stub()->Op(op_request, op_response);
102 VLOG(2) << "done with op request: " << op_name;
103 return status;
104}

106void ComputationBuilder::RunOpAndNoteError(OpRequest* op_request) {
107 OpResponse op_response;
108 Status status = RunOp(op_request, &op_response);
109 if (!status.ok()) {
110 NoteError(status);
111 }

114ComputationDataHandle ComputationBuilder::RunOpAndParseResponse(
115 OpRequest* op_request) {
116 OpResponse op_response;
117 Status status = RunOp(op_request, &op_response);
118 if (!status.ok()) {
119 NoteError(status);
120 return ComputationDataHandle();
121 }
122 if (op_response.output().handle() == 0) {
123 NoteError(InternalError("No output handle"));
124 return ComputationDataHandle();
125 }
126 return op_response.output();
127}

129bool ComputationBuilder::MakeWindow(
130 tensorflow::gtl::ArraySlice<int64> window_dimensions,
131 tensorflow::gtl::ArraySlice<int64> window_strides,
132 tensorflow::gtl::ArraySlice<std::pair<int64, int64>> padding,
133 tensorflow::gtl::ArraySlice<int64> lhs_dilation,
134 tensorflow::gtl::ArraySlice<int64> rhs_dilation, Window* window) {
135 const auto verify_size = [&](const size_t x, const char* x_name) {
136 if (x == 0 || x == window_dimensions.size()) {
137 return true;
138 } else {
139 NoteError(InvalidArgument(
140 "%s", tensorflow::strings::StrCat(
141 "Window has different number of window dimensions than of ",
142 x_name, "\nNumber of window dimensions: ",
143 window_dimensions.size(), "\nNumber of ", x_name, ": ", x,
144 "\n")
145 .c_str())); //
146 return false;
147 }
148 };
149 if (!verify_size(window_strides.size(), "window strides") ||
150 !verify_size(padding.size(), "padding entries") ||
151 !verify_size(lhs_dilation.size(), "lhs dilation factors") ||
152 !verify_size(rhs_dilation.size(), "rhs dilation factors")) {
153 return false;
154 }
155
156 window->Clear();
157 for (size_t i = 0; i < window_dimensions.size(); i++) {
158 auto dim = window->add_dimensions();
159 dim->set_size(window_dimensions[i]);
160 if (!window_strides.empty()) {
161 dim->set_stride(window_strides[i]);
162 } else {
163 dim->set_stride(1);
164 }
165 if (!padding.empty()) {
166 dim->set_padding_low(padding[i].first);
167 dim->set_padding_high(padding[i].second);
168 } else {
169 dim->set_padding_low(0);
170 dim->set_padding_high(0);
171 }
172 if (!lhs_dilation.empty()) {
173 dim->set_base_dilation(lhs_dilation[i]);
174 } else {
175 dim->set_base_dilation(1);
176 }
177 if (!rhs_dilation.empty()) {
178 dim->set_window_dilation(rhs_dilation[i]);
179 } else {
180 dim->set_window_dilation(1);
181 }
182 dim->set_window_reversal(false);
183 }
184 return true;
185}

276ComputationDataHandle ComputationBuilder::SliceInDim(
277 const ComputationDataHandle& operand, int64 start_index, int64 limit_index,
278 int64 stride, int64 dimno) {
279 StatusOr<std::unique_ptr<Shape>> shape_status = GetShape(operand);

290 limits[dimno] = limit_index;
291 strides[dimno] = stride;
292 return Slice(operand, starts, limits, strides);

376 if (!first_error_.ok()) {

378 }

381 if (!shape.ok()) {

383 }

389ComputationDataHandle ComputationBuilder::Collapse(
390 const ComputationDataHandle& operand,
391 tensorflow::gtl::ArraySlice<int64> dimensions) {
392 if (!first_error_.ok()) {
393 return ComputationDataHandle();
394 }
395
396 // Don't support out-of-order collapse here.
397 // Checks that the collapsed dimensions are in order and consecutive.
398 for (tensorflow::gtl::ArraySlice<int64>::size_type i = 1;
399 i < dimensions.size(); ++i) {
400 if (dimensions[i] - 1 != dimensions[i - 1]) {
401 NoteError(InvalidArgument(
402 "Collapsed dimensions are not in order and consecutive."));
403 return ComputationDataHandle();
404 }
405 }
406
407 // Create a new sizes vector from the old shape, replacing the collapsed
408 // dimensions by the product of their sizes.
409 StatusOr<std::unique_ptr<Shape>> shape_or_status = GetShape(operand);
410 if (!shape_or_status.ok()) {
411 return ComputationDataHandle();
412 }
413 std::unique_ptr<Shape> original_shape = shape_or_status.ConsumeValueOrDie();
414
415 VLOG(3) << "original shape: " << ShapeUtil::HumanString(*original_shape);
416 VLOG(3) << "dims to collapse: "
417 << tensorflow::str_util::Join(dimensions, ",");
418
419 if (dimensions.size() <= 1) {
420 // Not collapsing anything, trivially we can return the operand versus
421 // enqueueing a trivial reshape.
422 return operand;
423 }
424
425 std::vector<int64> new_sizes;
426 for (int i = 0; i < ShapeUtil::Rank(*original_shape); ++i) {
427 if (i <= dimensions.front() || i > dimensions.back()) {
428 new_sizes.push_back(original_shape->dimensions(i));
429 } else {
430 new_sizes.back() *= original_shape->dimensions(i);
431 }
432 }
433
434 VLOG(3) << "new sizes: [" << tensorflow::str_util::Join(new_sizes, ",")
435 << "]";
436
437 return Reshape(operand, new_sizes);
438}

554bool ComputationBuilder::VerifyConvolution(
555 const Shape& lhs_shape, const Shape& rhs_shape,
556 const ConvolutionDimensionNumbers& dimension_numbers) {
557 if (ShapeUtil::Rank(lhs_shape) != ShapeUtil::Rank(rhs_shape)) {
558 NoteError(
559 InvalidArgument("Convolution arguments must have same number of "
560 "dimensions. Got: %s and %s",
561 ShapeUtil::HumanString(lhs_shape).c_str(),
562 ShapeUtil::HumanString(rhs_shape).c_str()));
563 return false;
564 }
565 int num_dims = ShapeUtil::Rank(lhs_shape);
566 if (num_dims < 2) {
567 NoteError(InvalidArgument(
568 "Convolution expects argument arrays with >= 3 dimensions. "
569 "Got: %s and %s",
570 ShapeUtil::HumanString(lhs_shape).c_str(),
571 ShapeUtil::HumanString(rhs_shape).c_str()));
572 return false;
573 }
574 int num_spatial_dims = num_dims - 2;
575
576 const auto check_spatial_dimensions =
577 [&](const char* const field_name,
578 const tensorflow::protobuf::RepeatedField<tensorflow::protobuf_int64>&
579 numbers) {
580 if (numbers.size() != num_spatial_dims) {
581 NoteError(InvalidArgument("Expected %d elements for %s, but got %d.",
582 num_spatial_dims, field_name,
583 numbers.size()));
584 return false;
585 }
586 for (int i = 0; i < numbers.size(); ++i) {
587 if (numbers.Get(i) < 0 || numbers.Get(i) >= num_dims) {
588 NoteError(
589 InvalidArgument("Convolution %s[%d] is out of bounds: %lld",
590 field_name, i, numbers.Get(i)));
591 return false;
592 }
593 }
594 return true;
595 };
596 return check_spatial_dimensions(
597 "input_spatial_dimensions",
598 dimension_numbers.input_spatial_dimensions()) &&
599 check_spatial_dimensions(
600 "kernel_spatial_dimensions",
601 dimension_numbers.kernel_spatial_dimensions()) &&
602 check_spatial_dimensions(
603 "output_spatial_dimensions",
604 dimension_numbers.output_spatial_dimensions());
605}

607ComputationDataHandle ComputationBuilder::ConvWithGeneralDimensions(
608 const ComputationDataHandle& lhs, const ComputationDataHandle& rhs,
609 tensorflow::gtl::ArraySlice<int64> window_strides, Padding padding,
610 const ConvolutionDimensionNumbers& dimension_numbers) {
611 if (!first_error_.ok() || !PrepareComputation().ok()) {
612 return ComputationDataHandle();
613 }
614
615 StatusOr<std::unique_ptr<Shape>> lhs_shape_or_status = GetShape(lhs);
616 if (!lhs_shape_or_status.ok()) {
617 return ComputationDataHandle();
618 }
619
620 StatusOr<std::unique_ptr<Shape>> rhs_shape_or_status = GetShape(rhs);
621 if (!rhs_shape_or_status.ok()) {
622 return ComputationDataHandle();
623 }
624
625 std::unique_ptr<Shape> lhs_shape = lhs_shape_or_status.ConsumeValueOrDie();
626 std::unique_ptr<Shape> rhs_shape = rhs_shape_or_status.ConsumeValueOrDie();
627
628 if (!VerifyConvolution(*lhs_shape, *rhs_shape, dimension_numbers)) {
629 NoteError(InternalError("failed to verify convolution"));
630 return ComputationDataHandle();
631 }
632
633 std::vector<int64> base_area_dimensions(
634 dimension_numbers.input_spatial_dimensions_size());
635 for (std::vector<int64>::size_type i = 0; i < base_area_dimensions.size();
636 ++i) {
637 base_area_dimensions[i] =
638 lhs_shape->dimensions(dimension_numbers.input_spatial_dimensions(i));
639 }
640
641 std::vector<int64> window_dimensions(
642 dimension_numbers.kernel_spatial_dimensions_size());
643 for (std::vector<int64>::size_type i = 0; i < window_dimensions.size(); ++i) {
644 window_dimensions[i] =
645 rhs_shape->dimensions(dimension_numbers.kernel_spatial_dimensions(i));
646 }
647
648 return ConvGeneral(lhs, rhs, window_strides,
649 MakePadding(base_area_dimensions, window_dimensions,
650 window_strides, padding),
651 dimension_numbers);
652}

654ComputationDataHandle ComputationBuilder::ConvGeneral(
655 const ComputationDataHandle& lhs, const ComputationDataHandle& rhs,
656 tensorflow::gtl::ArraySlice<int64> window_strides,
657 tensorflow::gtl::ArraySlice<std::pair<int64, int64>> padding,
658 const ConvolutionDimensionNumbers& dimension_numbers) {

663ComputationDataHandle ComputationBuilder::ConvGeneralDilated(
664 const ComputationDataHandle& lhs, const ComputationDataHandle& rhs,
665 tensorflow::gtl::ArraySlice<int64> window_strides,
666 tensorflow::gtl::ArraySlice<std::pair<int64, int64>> padding,
667 tensorflow::gtl::ArraySlice<int64> lhs_dilation,
668 tensorflow::gtl::ArraySlice<int64> rhs_dilation,
669 const ConvolutionDimensionNumbers& dimension_numbers) {
670 if (!first_error_.ok() || !PrepareComputation().ok()) {
671 return ComputationDataHandle();
672 }
673
674 StatusOr<std::unique_ptr<Shape>> lhs_shape_or_status = GetShape(lhs);
675 if (!lhs_shape_or_status.ok()) {
676 return ComputationDataHandle();
677 }
678
679 StatusOr<std::unique_ptr<Shape>> rhs_shape_or_status = GetShape(rhs);
680 if (!rhs_shape_or_status.ok()) {
681 return ComputationDataHandle();
682 }
683
684 std::unique_ptr<Shape> lhs_shape = lhs_shape_or_status.ConsumeValueOrDie();
685 std::unique_ptr<Shape> rhs_shape = rhs_shape_or_status.ConsumeValueOrDie();
686 if (!VerifyConvolution(*lhs_shape, *rhs_shape, dimension_numbers)) {
687 // Error is recorded in VerifyConvolution.
688 return ComputationDataHandle();
689 }
690
691 std::vector<int64> window_dimensions(
692 dimension_numbers.kernel_spatial_dimensions_size());
693 for (std::vector<int64>::size_type i = 0; i < window_dimensions.size(); ++i) {
694 window_dimensions[i] =
695 rhs_shape->dimensions(dimension_numbers.kernel_spatial_dimensions(i));
696 }
697
698 OpRequest op_request;
699 ConvolveRequest* request = op_request.mutable_convolve_request();
700 *request->mutable_lhs() = lhs;
701 *request->mutable_rhs() = rhs;
702 *request->mutable_dimension_numbers() = dimension_numbers;
703
704 if (!MakeWindow(window_dimensions, window_strides, padding, lhs_dilation,
705 rhs_dilation, request->mutable_window())) {
706 // Error is recorded in MakeWindow.
707 return ComputationDataHandle();
708 }
709
710 return RunOpAndParseResponse(&op_request);
711}

1523 int64 input_second_spatial, int64 output_batch, int64 output_feature,
1524 int64 output_first_spatial, int64 output_second_spatial,

1563 dimension_numbers.set_output_feature_dimension(output_feature);
1564 dimension_numbers.add_output_spatial_dimensions(output_first_spatial);
1565 dimension_numbers.add_output_spatial_dimensions(output_second_spatial);
