39StatusOr<std::unique_ptr<Literal>> Client::Transfer(
40 const GlobalData& data, const Shape* shape_with_layout) {
41 TransferToClientRequest request;
42 *request.mutable_data() = data.handle();
43 if (shape_with_layout != nullptr) {
44 *request.mutable_shape_with_layout() = *shape_with_layout;
45 }
46 TransferToClientResponse response;
47
48 VLOG(1) << "making transfer request";
49 VLOG(3) << "TransferToClientRequest: {" << request.DebugString() << "}";
50 Status s = stub_->TransferToClient(&request, &response);
51 VLOG(1) << "done with request";
52
53 if (!s.ok()) {
54 return s;
55 }
56 VLOG(3) << "TransferToClientResponse: {" << response.DebugString() << "}";
57
58 if (!response.has_literal()) {
59 return FailedPrecondition(
60 "server provided response without a literal in "
61 "TransferToClient request");
62 }
63 return Literal::CreateFromProto(*response.mutable_literal());
64}

164StatusOr<std::unique_ptr<Literal>> Client::ExecuteAndTransfer(
165 const Computation& computation,
166 tensorflow::gtl::ArraySlice<GlobalData*> arguments,
167 const ExecutionOptions* execution_options,
168 ExecutionProfile* execution_profile) {
169 TF_ASSIGN_OR_RETURN(
170 std::unique_ptr<GlobalData> data,
171 Execute(computation, arguments, execution_options, execution_profile));
172
173 const Shape* shape_with_output_layout = nullptr;
174 if (execution_options && execution_options->has_shape_with_output_layout()) {
175 shape_with_output_layout = &execution_options->shape_with_output_layout();
176 }
177 return Transfer(*data, shape_with_output_layout);
178}

210StatusOr<std::unique_ptr<GlobalData>> Client::Execute(
211 const Computation& computation,
212 tensorflow::gtl::ArraySlice<GlobalData*> arguments,
213 const ExecutionOptions* execution_options,
214 ExecutionProfile* execution_profile) {
215 ExecuteRequest request;
216 *request.mutable_computation() = computation.handle();
217
218 if (execution_options == nullptr) {
219 *request.mutable_execution_options() = CreateDefaultExecutionOptions();
220 } else {
221 *request.mutable_execution_options() = *execution_options;
222 }
223 for (GlobalData* argument : arguments) {
224 CHECK(argument != nullptr) << "Argument pointers must not be null.";
225 *request.add_arguments() = argument->handle();
226 }
227
228 ExecuteResponse response;
229 VLOG(1) << "making execute request: " << request.ShortDebugString();
230 Status s = stub_->Execute(&request, &response);
231 VLOG(1) << "done with request";
232
233 if (!s.ok()) {
234 return s;
235 }
236
237 if (execution_profile != nullptr) {
238 *execution_profile = response.profile();
239 if (VLOG_IS_ON(1)) {
240 TF_ASSIGN_OR_RETURN(
241 auto execution_stats,
242 ExecutionStatsAsString(computation, response.profile()));
243 VLOG(1) << execution_stats;
244 }
245 }
246
247 return MakeUnique<GlobalData>(stub_, response.output());
248}

290StatusOr<std::vector<std::unique_ptr<GlobalData>>> Client::ExecuteParallel(
291 tensorflow::gtl::ArraySlice<ComputationInstance> computations) {
292 ExecuteParallelRequest request;
293
294 for (const ComputationInstance& computation : computations) {
295 ExecuteRequest single_request;
296 *single_request.mutable_computation() = computation.computation.handle();
297 for (GlobalData* argument : computation.arguments) {
298 *single_request.add_arguments() = argument->handle();
299 }
300 *single_request.mutable_execution_options() = computation.execution_options;
301 *request.add_requests() = single_request;
302 }
303
304 ExecuteParallelResponse response;
305 VLOG(1) << "making execute-parallel request: " << request.ShortDebugString();
306 tensorflow::Status s = stub_->ExecuteParallel(&request, &response);
307 VLOG(1) << "done with request";
308
309 if (!s.ok()) {
310 return s;
311 }
312
313 std::vector<std::unique_ptr<GlobalData>> outputs;
314 for (size_t i = 0; i < computations.size(); ++i) {
315 outputs.push_back(
316 MakeUnique<GlobalData>(stub_, response.responses(i).output()));
317 if (computations[i].execution_profile != nullptr) {
318 *computations[i].execution_profile = response.responses(i).profile();
319 }
320 }
321
322 return std::move(outputs);
323}

375 return s;

377

497StatusOr<string> Client::ExecutionStatsAsString(
498 const Computation& computation, const ExecutionProfile& profile) {
499 TF_ASSIGN_OR_RETURN(
500 auto computation_stats,
501 GetComputationStats(computation,
502 legacy_flags::GetDebugOptionsFromFlags()));
503 int64 total_flops =
504 computation_stats.flop_count() + computation_stats.transcendental_count();
505 if (profile.compute_time_ns() > 0) {
506 int64 nanoseconds = profile.compute_time_ns();
507 int64 cycle_count = profile.compute_cycle_count();
508 double gflops = total_flops / nanoseconds;
509 return tensorflow::strings::StrCat(
510 "[Execution Statistics] flop count: ", computation_stats.flop_count(),
511 ", transcendental count: ", computation_stats.transcendental_count(),
512 ", compute execution time: ", nanoseconds, " nsec",
513 ", compute cycles: ", cycle_count, ", performance: ", gflops,
514 "gflop/s");
515 }
516 return string("[Execution Statistics] not available.");
517}
