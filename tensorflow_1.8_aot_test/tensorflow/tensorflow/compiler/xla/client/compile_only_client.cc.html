24StatusOr<std::vector<std::unique_ptr<AotCompilationResult>>>
25CompileOnlyClient::CompileAheadOfTime(
26 const tensorflow::gtl::ArraySlice<AotComputationInstance> computations,
27 const AotCompilationOptions& options) {
28 std::vector<CompileOnlyService::AotComputationInstance> service_instances;
29 service_instances.reserve(computations.size());
30 for (const AotComputationInstance& instance : computations) {
31 service_instances.push_back({});
32 CompileOnlyService::AotComputationInstance& service_instance =
33 service_instances.back();
34 TF_RET_CHECK(instance.computation != nullptr);
35 service_instance.computation = instance.computation->handle();
36 service_instance.argument_layouts = instance.argument_layouts;
37 service_instance.result_layout = instance.result_layout;
38 }
39 return compiler_service_->CompileAheadOfTime(service_instances, options);
40}

42int64 CompileOnlyClient::PointerSizeForTriple(tensorflow::StringPiece triple) {
43 llvm::Triple llvm_triple(
44 llvm::Triple::normalize(llvm::StringRef(triple.data(), triple.size())));
45 if (llvm_triple.isArch64Bit()) {
46 return 8;
47 } else if (llvm_triple.isArch32Bit()) {
48 return 4;
49 } else {
50 CHECK(llvm_triple.isArch16Bit());
51 return 2;
52 }
53}
