46// Wraps an XLA client with a convenient interface for building up
47// computations. Any errors encountered in building up the computation are
48// deferred from being handled until Build() is called.
49//
50// Thread-compatible.
51class ComputationBuilder {
52 public:
53 // client: client in which to build the computation.
54 // computation_name: name to use for the built computation.
55 ComputationBuilder(Client* client, const string& computation_name);

60 Client* client() const { return client_; }

62 // Returns the computation name.

65 // Sets OpMetadata that will be added to all instructions until cleared.
66 //
67 // OpMetadata is often applied to a series of XLA HLO instructions. As a
68 // result, OpMetadata is set on the Computation Builder. All subsequent
69 // instructions generated via this Computation Builder will have the same
70 // OpMetadata attached until a call to ClearOpMetadata.
71 void SetOpMetadata(const OpMetadata& metadata) { metadata_ = metadata; }
72
73 // Clears the HloMetadata state.
74 void ClearOpMetadata() { metadata_.Clear(); }

76 // Sets an OpSharding that will be attached to all instructions until cleared.

78
79 // Clears the sharding. Ops will be sharded according to the default placement
80 // policy.
81 void ClearSharding() { sharding_ = tensorflow::gtl::nullopt; }

83 // Returns the OpSharding that will be attached to all instructions.
84 const tensorflow::gtl::optional<OpSharding>& sharding() const {
85 return sharding_;
86 }

161 // Adds dimensions to an array by duplicating the data in the array.
162 //
163 // The new dimensions are inserted on the left, i.e. if
164 // broadcast_sizes has values {a0, ..., aN} and the operand shape
165 // has dimensions {b0, ..., bM} then the shape of the output has
166 // dimensions {a0, ..., aN, b0, ..., bM}.
167 //
168 // The new dimensions index into copies of the operand, i.e.
169 //
170 // output[i0, ..., iN, j0, ..., jM] = operand[j0, ..., jM]
171 ComputationDataHandle Broadcast(
172 const ComputationDataHandle& operand,
173 tensorflow::gtl::ArraySlice<int64> broadcast_sizes);

182 // Enqueues an operation onto the computation that flattens the operand based
183 // on the dimension order (major/slowest-varying to minor/fastest-varying)
184 // given, followed by reshaping it into the shape with the given dimension
185 // sizes (also major to minor). Conceptually, this is a limited form of
186 // "shape casting".
187 ComputationDataHandle Reshape(const ComputationDataHandle& operand,
188 tensorflow::gtl::ArraySlice<int64> dimensions,
189 tensorflow::gtl::ArraySlice<int64> new_sizes);
190
191 // Enqueues an operation onto the computation that collapses the operand, from
192 // first to last dimension (C order), then reshapes it to the given dimension
193 // sizes. Conceptually, this is a limited form of "shape casting".
194 ComputationDataHandle Reshape(const ComputationDataHandle& operand,
195 tensorflow::gtl::ArraySlice<int64> new_sizes);
196
197 // Wrapper for Reshape.
198 // Enqueues an operation to collapse the provided dimensions; e.g. an
199 // operand with dimensions {x=256, y=2, z=2, p=32} can be collapsed to
200 // {x=1024, y=32} by collapsing dims {0, 1, 2}. Collapsing dimensions must
201 // be a consecutive, in-order subsequence of the operand dimensions.
202 //
203 // Note that collapsing a single dimension does nothing:
204 //
205 // {256} collapsing {0} => {256}
206 // {1} collapsing {0} => {1}
207 //
208 // Collapsing multiple dimensions produces a single result dimension:
209 //
210 // {256, 2} collapsing {0,1} => {512}
211 // {256, 2, 3} collapsing {0,1} => {512, 3}
212 //
213 // This could potentially cause data to be moved -- it provides a more
214 // structured form of reshaping than an arbitrary Reshape operation.
215 ComputationDataHandle Collapse(const ComputationDataHandle& operand,
216 tensorflow::gtl::ArraySlice<int64> dimensions);
217
218 // Enqueues a slice operation onto the computation that slices the operand
219 // from the start indices to the limit indices; e.g.
220 //
221 // x
222 // [ 0 1 2 3 ]
223 // y [ 4 5 6 7 ] => slice(start={1, 1}, limit={2, 3}) => [ 5 6 ]
224 // [ 8 9 a b ]
225 //
226 // Note that "limit" means up-to-but-not-including; i.e. [start, limit) in 1D
227 // range notation.
228 // The strides parameter determines the stride over the slice
229 ComputationDataHandle Slice(const ComputationDataHandle& operand,
230 tensorflow::gtl::ArraySlice<int64> start_indices,
231 tensorflow::gtl::ArraySlice<int64> limit_indices,
232 tensorflow::gtl::ArraySlice<int64> strides);
233
234 // Enqueues a slice operation in a given dimension, taking all other
235 // dimensions as they are; e.g. if dimno is 1 from start_index 2 to
236 // limit_index 4 by 1, and the shape is f32[7,8,9], this call is short-hand
237 // for:
238 //
239 // array[:, 2:4:1, :]
240 ComputationDataHandle SliceInDim(const ComputationDataHandle& operand,
241 int64 start_index, int64 limit_index,
242 int64 stride, int64 dimno);
243
244 // Enqueues a slice operation onto the computation that slices the 'operand'
245 // from dynamic start indices which are passed in 'start_indices'.
246 // The size of the slice in each dimension is passed in 'slice_sizes',
247 // which specify the end point of exclusive slice intervals in each
248 // dimension [start, start + size).
249 // The shape of 'start_indices' must be rank == 1, with dimension size
250 // equal to the rank of the 'operand'.
251 // Slice index calculations are computed modulo input dimension sizes to
252 // prevent dynamic start indices from generating out-of-bound array accesses.
253 ComputationDataHandle DynamicSlice(
254 const ComputationDataHandle& operand,
255 const ComputationDataHandle& start_indices,
256 tensorflow::gtl::ArraySlice<int64> slice_sizes);
257
258 // Enqueues a dynamic update slice operation onto the computation, which
259 // updates a slice of 'operand' with 'update' at dynamic 'start_indices'.
260 // The shape of 'update' determines the shape of the slice of 'operand'
261 // which is updated.
262 // The indices specified in 'start_indices' specify the offset of the slice
263 // of 'operand' which is updated.
264 //
265 // update = {10, 11} // calculated at runtime.
266 // [1 2 3] start = {1, 1} // calculated at runtime. [1 2 3 ]
267 // [4 5 6] => DynamicUpdateslice(data, update, start) => [4 10 11]
268 // [7 8 9] [7 8 9 ]
269 //
270 // The shape of 'start_indices' must be rank == 1, with dimension size
271 // equal to the rank of the 'operand'.
272 // Slice index calculations are computed modulo update dimension sizes to
273 // prevent dynamic start indices from generating out-of-bound array accesses.
274 ComputationDataHandle DynamicUpdateSlice(
275 const ComputationDataHandle& operand, const ComputationDataHandle& update,
276 const ComputationDataHandle& start_indices);

1033// RAII-style object: sets the current sharding assignment in builder on
1034// construction, and sets back to the previous assignment on destruction.
1035class ScopedShardingAssignment {
1036 public:
1037 ScopedShardingAssignment(xla::ComputationBuilder* builder,
1038 tensorflow::gtl::optional<OpSharding> sharding)
1039 : builder_(builder), prev_sharding_(builder->sharding()) {
1040 SetSharding(sharding);
1041 }
1042
1043 ~ScopedShardingAssignment() { SetSharding(prev_sharding_); }
1044
1045 private:
1046 void SetSharding(const tensorflow::gtl::optional<OpSharding>& sharding) {
1047 if (sharding.has_value()) {
1048 builder_->SetSharding(sharding.value());
1049 } else {
1050 builder_->ClearSharding();
1051 }
1052 }
1053
1054 xla::ComputationBuilder* const builder_;
1055 tensorflow::gtl::optional<OpSharding> prev_sharding_;
1056
1057 TF_DISALLOW_COPY_AND_ASSIGN(ScopedShardingAssignment);
1058};
