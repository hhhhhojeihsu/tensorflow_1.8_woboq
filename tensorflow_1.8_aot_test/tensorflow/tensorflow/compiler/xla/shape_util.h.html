40// An index for specifying a particular nested subshape within a shape. Used in
41// ShapeUtil::GetSubshape and other interfaces. Shapes are recursive data
42// structures (trees) and ShapeIndex defines a path through the tree where each
43// element of ShapeIndex indexes into a tuple (or nested tuple) within the
44// shape. For a non-nested tuple, an index has a single element. For example,
45// given a 3-element tuple (a, b, c) containing arrays a, b, and c, the index
46// {1} corresponds to array b. For a nested tuple, the index can have more than
47// one element. For the nested tuple (a, (b, c, d), e) below are the values
48// corresponding to the given indices:
49//
50// index {0} : array a
51// index {1, 2} : array d
52// index {2} : array e
53// index {0, 0} : invalid index (element at {0} is an array not a tuple)
54//
55// For indexing into array shapes, the index is always trivially empty, ie {}.
56//
57// ShapeIndex is a trivial wrapper around std::vector with a minimum number of
58// methods implemented.
59class ShapeIndex {
60 public:
61 ShapeIndex() = default;
62 ShapeIndex(std::initializer_list<int64> init) : indices_(init) {}
63
64 bool empty() const { return indices_.empty(); }
65 size_t size() const { return indices_.size(); }
66 void push_back(int64 value) { indices_.push_back(value); }
67 void pop_back() { indices_.pop_back(); }
68
69 // push_front is O(n^2), but shapes don't usually have a ton of dimensions.
70 void push_front(int64 value) { indices_.insert(indices_.begin(), value); }
71
72 std::vector<int64>::const_iterator begin() const { return indices_.begin(); }
73 std::vector<int64>::const_iterator end() const { return indices_.end(); }
74 std::vector<int64>::iterator begin() { return indices_.begin(); }
75 std::vector<int64>::iterator end() { return indices_.end(); }
76
77 const int64* data() const { return indices_.data(); }
78
79 int64 back() const { return indices_.back(); }
80 int64& back() { return indices_.back(); }
81
82 const int64& operator[](size_t i) const { return indices_[i]; }
83 int64& operator[](size_t i) { return indices_[i]; }
84
85 bool operator==(const ShapeIndex& other) const {
86 return indices_ == other.indices_;
87 }
88 bool operator!=(const ShapeIndex& other) const { return !(*this == other); }
89 bool operator<(const ShapeIndex& other) const {
90 return indices_ < other.indices_;
91 }
92
93 string ToString() const;
94
95 private:
96 std::vector<int64> indices_;
97};

99// A view into a ShapeIndex as above, with the cheap/easy ability to consume the
100// value at the front of the view.
101//
102// NB! ShapeIndexView does not own the memory backing the index array.
103// The memory backing the index array should be owned by an object
104// that lives longer than the ShapeIndexView instances pointing into
105// it.
106class ShapeIndexView {
107 public:
108 ShapeIndexView(const ShapeIndex& shape_index, int64 offset = 0)
109 : ShapeIndexView(shape_index.data() + offset,
110 shape_index.data() + shape_index.size()) {
111 CHECK_LE(offset, shape_index.size());
112 }
113 ShapeIndexView(std::initializer_list<int64> indices)
114 : ShapeIndexView(indices.begin(), indices.end()) {}
115 ShapeIndexView(const ShapeIndexView& other) = default;
116
117 using iterator = const int64*;
118
119 iterator begin() const { return begin_; }
120 iterator end() const { return end_; }
121 int64 size() const { return std::distance(begin_, end_); }
122 bool empty() const { return begin_ == end_; }
123 int64 front() const {
124 CHECK(!empty());
125 return *begin_;
126 }
127 ShapeIndexView ConsumeFront() const {
128 CHECK(!empty());
129 auto new_begin = begin_;
130 ++new_begin;
131 return ShapeIndexView(new_begin, end_);
132 }
133
134 string ToString() const;
135
136 private:
137 ShapeIndexView(iterator begin, iterator end) : begin_(begin), end_(end) {}
138
139 iterator begin_;
140 iterator end_;
141};

573 // Iterates through all the shape indexes, in minor to major order, starting
574 // from the base indexes, incrementing by the incr steps, up to count
575 // (index[i] < base[i] + count[i]), and calls the visitor_function with the
576 // current index.
577 // The visitor_function visitor function should return true if it wants to
578 // continue, or false otherwise.
579 //
580 // visitor_function must be a callable of type
581 // StatusOr<bool>(ArraySlice<int64>) or compatible.
582 template <typename FnType>
583 static Status ForEachIndexWithStatus(const Shape& shape,
584 tensorflow::gtl::ArraySlice<int64> base,
585 tensorflow::gtl::ArraySlice<int64> count,
586 tensorflow::gtl::ArraySlice<int64> incr,
587 const FnType& visitor_function) {
588 return ForEachIndexInternal(shape, base, count, incr, visitor_function);
589 }

608 static void ForEachIndex(const Shape& shape,
609 tensorflow::gtl::ArraySlice<int64> base,
610 tensorflow::gtl::ArraySlice<int64> count,
611 tensorflow::gtl::ArraySlice<int64> incr,
612 const FnType& visitor_function) {

620 // A parallel version of ForEachIndex(WithStatus). This can only be used if
621 // the visitor_function is thread-safe and the order of iteration does not
622 // matter.
623 //
624 // visitor_function must be a callable of type
625 // void(ArraySlice<int64>) or compatible.
626 template <typename FnType>
627 static void ForEachIndexParallel(const Shape& shape,
628 tensorflow::gtl::ArraySlice<int64> base,
629 tensorflow::gtl::ArraySlice<int64> count,
630 tensorflow::gtl::ArraySlice<int64> incr,
631 const FnType& visitor_function) {
632 // The parallel version of ForEachIndexInternal can never fail.
633 CHECK(ForEachIndexInternal(
634 shape, base, count, incr,
635 [&visitor_function](tensorflow::gtl::ArraySlice<int64> indexes)
636 -> StatusOr<bool> {
637 visitor_function(indexes);
638 return true;
639 },
640 /*parallel=*/true)
641 .ok());
642 }

646 // used by both the layout-optional and layout-required public method.
647 static Status ValidateShapeWithOptionalLayoutInternal(const Shape& shape);
648
649 template <typename FnType>
650 static Status ForEachIndexInternal(const Shape& shape,
651 tensorflow::gtl::ArraySlice<int64> base,
652 tensorflow::gtl::ArraySlice<int64> count,
653 tensorflow::gtl::ArraySlice<int64> incr,
654 const FnType& visitor_function,
655 bool parallel = false) {
656 if (ShapeUtil::HasZeroElements(shape)) {
657 return Status::OK();
658 }
659 CHECK_EQ(Rank(shape), base.size());
660 CHECK_EQ(incr.size(), base.size());
661 CHECK_EQ(count.size(), base.size());
662 const int64 rank = LayoutUtil::MinorToMajor(shape).size();
663 // Allows handling R0 arrays, such that the visitor function will be called
664 // once with the proper empty indexes.
665 int64 n = -1;
666 std::vector<int64> indexes(base.begin(), base.end());
667 const int kNumThreads = tensorflow::port::NumSchedulableCPUs();
668 tensorflow::gtl::optional<tensorflow::thread::ThreadPool> pool;
669 if (parallel) {
670 pool.emplace(tensorflow::Env::Default(), "foreach", kNumThreads);
671 }
672
673 while (n < rank) {
674 if (pool != tensorflow::gtl::nullopt) {
675 pool->Schedule(
676 [indexes, &visitor_function] { visitor_function(indexes); });
677 } else {
678 TF_ASSIGN_OR_RETURN(bool should_continue, visitor_function(indexes));
679 if (!should_continue) {
680 break;
681 }
682 }
683 // Increments dimensions in minor to major order.
684 for (n = 0; n < rank; ++n) {
685 int64 dim = LayoutUtil::Minor(shape.layout(), n);
686 indexes[dim] += incr[dim];
687 if (indexes[dim] < base[dim] + count[dim]) {
688 break;
689 }
690 indexes[dim] = base[dim];
691 }
692 }
693
694 return Status::OK();
695 }
