22namespace perftools {
23namespace gputools {
24namespace host {
25
26using std::chrono::duration_cast;
27
28bool HostTimer::Start(Stream* stream) {
29 return stream->ThenDoHostCallback([this]() { this->StartNow(); }).ok();
30}
31
32bool HostTimer::Stop(Stream* stream) {
33 return stream->ThenDoHostCallback([this]() { this->StopNow(); }).ok();
34}
35
36uint64 HostTimer::Microseconds() const {
37 return duration_cast<std::chrono::microseconds>(duration_).count();
38}
39
40uint64 HostTimer::Nanoseconds() const {
41 return duration_cast<std::chrono::nanoseconds>(duration_).count();
42}
43
44void HostTimer::StartNow() { start_time_ = clock::now(); }
45
46void HostTimer::StopNow() { duration_ = clock::now() - start_time_; }
47
