30using Status = tensorflow::Status;

32#define SE_CHECK_OK(val) TF_CHECK_OK(val)
33#define SE_ASSERT_OK(val) \
34 ASSERT_EQ(::perftools::gputools::port::Status::OK(), (val))

36// Define some canonical error helpers.
37inline Status UnimplementedError(StringPiece message) {
38 return Status(error::UNIMPLEMENTED, message);
39}
40inline Status InternalError(StringPiece message) {
41 return Status(error::INTERNAL, message);
42}
43inline Status FailedPreconditionError(StringPiece message) {
44 return Status(error::FAILED_PRECONDITION, message);
45}
