29// Returns a copy of the input string 'str' with the given 'suffix'
30// removed. If the suffix doesn't match, returns a copy of the original string.
31inline string StripSuffixString(port::StringPiece str, port::StringPiece suffix) {
32 if (tensorflow::str_util::EndsWith(str, suffix)) {
33 str.remove_suffix(suffix.size());
34 }
35 return str.ToString();
36}
