63std::unique_ptr<std::map<string, string>> DeviceDescription::ToMap() const {
64 std::unique_ptr<std::map<string, string>> owned_result{
65 new std::map<string, string>};
66 std::map<string, string> &result = *owned_result;
67 result["Device Vendor"] = device_vendor();
68 result["Platform Version"] = platform_version();
69 result["Driver Version"] = driver_version();
70 result["Runtime Version"] = runtime_version();
71 result["PCI bus ID"] = pci_bus_id_;
72 result["Device Name"] = name_;
73
74 const ThreadDim &thread_dim = thread_dim_limit();
75 result["ThreadDim Limit"] =
76 port::StrCat(thread_dim.x, ",", thread_dim.y, ",", thread_dim.z);
77 const BlockDim &block_dim = block_dim_limit();
78 result["BlockDim Limit"] =
79 port::StrCat(block_dim.x, ",", block_dim.y, ",", block_dim.z);
80
81 result["Threads Per Core Limit"] = port::StrCat(threads_per_core_limit());
82 result["Threads Per Block Limit"] = port::StrCat(threads_per_block_limit());
83 result["Registers Per Block Limit"] =
84 port::StrCat(registers_per_block_limit());
85
86 result["Device Address Bits"] = port::StrCat(device_address_bits());
87 result["Device Memory Size"] =
88 port::HumanReadableNumBytes::ToString(device_memory_size());
89 result["Memory Bandwidth"] = port::StrCat(
90 port::HumanReadableNumBytes::ToString(memory_bandwidth_), "/s");
91
92 result["Shared Memory Per Core"] =
93 port::HumanReadableNumBytes::ToString(shared_memory_per_core_);
94 result["Shared Memory Per Block"] =
95 port::HumanReadableNumBytes::ToString(shared_memory_per_block_);
96
97 result["Clock Rate GHz"] = port::StrCat(clock_rate_ghz());
98
99 result["CUDA Compute Capability"] = port::StrCat(
100 cuda_compute_capability_major_, ".", cuda_compute_capability_minor_);
101
102 result["NUMA Node"] = port::StrCat(numa_node());
103 result["Core Count"] = port::StrCat(core_count());
104 result["ECC Enabled"] = port::StrCat(ecc_enabled());
105 return owned_result;
106}

121bool ThreadDimOk(const DeviceDescription &device_description,
122 const ThreadDim &thread_dim) {
123 auto total_threads = thread_dim.x * thread_dim.y * thread_dim.z;
124 auto threads_per_block_limit = device_description.threads_per_block_limit();
125 if (total_threads > threads_per_block_limit) {
126 VLOG(2) << "exceeded total-thread-per-block limit: " << total_threads
127 << " vs limit " << threads_per_block_limit;
128 return false;
129 }
130
131 const auto &limit = device_description.thread_dim_limit();
132 bool ok = thread_dim.x <= limit.x && thread_dim.y <= limit.y &&
133 thread_dim.z <= limit.z;
134 if (!ok) {
135 VLOG(2) << "thread dim " << thread_dim.ToString()
136 << " exceeds limit contraints of " << limit.ToString();
137 }
138 return ok;
139}

141uint64 DivideCeil(uint64 x, uint64 y) {
142 return port::MathUtil::CeilOfRatio(x, y);
143}

145void CalculateDimensionality(const DeviceDescription &device_description,
146 uint64 element_count, uint64 *threads_per_block,
147 uint64 *block_count) {
148 *threads_per_block = device_description.threads_per_block_limit();
149 *block_count = DivideCeil(element_count, *threads_per_block);
150 if (*block_count == 1) {
151 CHECK_LE(element_count, *threads_per_block);
152 *threads_per_block = element_count;
153 }
154}

156// Round value up to a multiple of n.
157static uint64 RoundUp(uint64 value, uint64 n) {
158 return port::MathUtil::CeilOfRatio(value, n) * n;
159}
160
161// Round value down to a multiple of n.
162static uint64 RoundDown(uint64 value, uint64 n) {
163 return port::MathUtil::FloorOfRatio(value, n) * n;
164}

166uint64 CalculateOccupancy(const DeviceDescription &device_description,
167 uint64 registers_per_thread,
168 uint64 shared_memory_per_block,
169 const ThreadDim &thread_dims) {
170 // Don't try to compute occupancy if necessary values are not initialized.
171 uint64 required_fields[] = { device_description.registers_per_thread_limit(),
172 device_description.threads_per_warp(),
173 device_description.warp_alloc_granularity(),
174 device_description.register_alloc_granularity(),
175 device_description.registers_per_block_limit(),
176 device_description.shared_memory_per_core(),
177 device_description.blocks_per_core_limit() };
178 for (auto value : required_fields) {
179 if (value == kUninitializedUint64) {
180 return 0;
181 }
182 }
183
184 if (registers_per_thread > device_description.registers_per_thread_limit()) {
185 return 0;
186 }
187
188 uint64 warps_per_block =
189 port::MathUtil::CeilOfRatio(thread_dims.x * thread_dims.y * thread_dims.z,
190 device_description.threads_per_warp());
191
192 // Warp resources are allocated at a particular granularity. This value is
193 // the effective number of warps for resource allocation purposes.
194 uint64 alloc_warps_per_block =
195 RoundUp(warps_per_block, device_description.warp_alloc_granularity());
196
197 uint64 alloc_regs_per_warp =
198 RoundUp(device_description.threads_per_warp() * registers_per_thread,
199 device_description.register_alloc_granularity());
200 uint64 regs_per_block = alloc_warps_per_block * alloc_regs_per_warp;
201 uint64 reg_limit =
202 device_description.registers_per_block_limit() / regs_per_block;
203
204 uint64 alloc_smem_per_block = RoundUp(
205 shared_memory_per_block,
206 device_description.shared_memory_alloc_granularity());
207 uint64 smem_limit = alloc_smem_per_block > 0 ?
208 device_description.shared_memory_per_core() / alloc_smem_per_block :
209 device_description.blocks_per_core_limit();
210
211 uint64 thread_limit = device_description.threads_per_core_limit()
212 / (warps_per_block * device_description.threads_per_warp());
213
214 return std::min({ device_description.blocks_per_core_limit(),
215 reg_limit, smem_limit, thread_limit });
216}
217
218uint64 CalculateRegisterLimitForTargetOccupancy(
219 const DeviceDescription &device_description, uint64 shared_memory_per_block,
220 const ThreadDim &thread_dims, uint64 target_blocks_per_core) {
221 // Linear search from maximum number of registers down until the target
222 // blocks per SM is found.
223 // TODO(meheff): Compute this using a closed form solution.
224 int reg_step = device_description.register_alloc_granularity() /
225 device_description.threads_per_warp();
226 for (int r = device_description.registers_per_thread_limit(); r > 0;
227 r = RoundDown(r - 1, reg_step)) {
228 uint64 occupancy = CalculateOccupancy(
229 device_description, r, shared_memory_per_block, thread_dims);
230 if (occupancy >= target_blocks_per_core) {
231 return r;
232 }
233 }
234 return 0;
235}
