152std::tuple<int, int, int> GetDimIndices(const DataLayout& layout,
153 const int data_dims) {
154 int depth_idx, batch_idx, spatial_idx;
155 switch (layout) {
156 case DataLayout::kYXBatchDepth:
157 depth_idx = data_dims - 1;
158 batch_idx = data_dims - 2;
159 spatial_idx = 0;
160 break;
161
162 case DataLayout::kYXDepthBatch:
163 depth_idx = data_dims - 2;
164 batch_idx = data_dims - 1;
165 spatial_idx = 0;
166 break;
167
168 case DataLayout::kBatchYXDepth:
169 depth_idx = data_dims - 1;
170 batch_idx = 0;
171 spatial_idx = 1;
172 break;
173
174 case DataLayout::kBatchDepthYX:
175 case DataLayout::kBatchDepthYX4:
176 depth_idx = 1;
177 batch_idx = 0;
178 spatial_idx = 2;
179 break;
180 }
181
182 return std::make_tuple(depth_idx, batch_idx, spatial_idx);
183}

185std::vector<int64> ReorderDims(const std::vector<int64>& input,
186 const DataLayout& from, const DataLayout& to) {
187 if (from == to) return input;
188
189 int d_idx_from, b_idx_from, spatial_idx_from;
190 int d_idx_to, b_idx_to, spatial_idx_to;
191
192 std::tie(d_idx_from, b_idx_from, spatial_idx_from) =
193 GetDimIndices(from, input.size());
194 std::tie(d_idx_to, b_idx_to, spatial_idx_to) =
195 GetDimIndices(to, input.size());
196
197 std::vector<int64> reordered(input.size());
198 reordered[b_idx_to] = input[b_idx_from];
199 reordered[d_idx_to] = input[d_idx_from];
200
201 for (size_t i = 0; i < input.size() - 2;
202 i++, spatial_idx_from++, spatial_idx_to++) {
203 reordered[spatial_idx_to] = input[spatial_idx_from];
204 }
205
206 return reordered;
207}

218BatchDescriptor::BatchDescriptor(int ndims)
219 : count_(0),
220 feature_map_count_(0),
221 spatial_size_(ndims, 0),
222 value_max_(0.0),
223 value_min_(0.0),
224 layout_(DataLayout::kYXDepthBatch),
225 ndims_(ndims),
226 quantized_activation_mode_(QuantizedActivationMode::k8Bit) {}
227
228BatchDescriptor::BatchDescriptor() : BatchDescriptor(/*ndims=*/2) {}
229
230std::vector<int64> BatchDescriptor::full_dims(const DataLayout& layout) const {
231 std::vector<int64> bdyx_dims(ndims_ + 2);
232 bdyx_dims[0] = count();
233 bdyx_dims[1] = feature_map_count();
234 std::copy(spatial_size_.begin(), spatial_size_.end(), bdyx_dims.begin() + 2);
235 return ReorderDims(bdyx_dims, DataLayout::kBatchDepthYX, layout);
236}
237
238std::vector<int64> BatchDescriptor::full_strides(
239 const DataLayout& layout) const {
240 if (layout_ == DataLayout::kBatchDepthYX4) {
241 LOG(FATAL)
242 << "Cannot compute full strides for batch descriptor " << ToString()
243 << ", because its layout is kBatchDepthYX4. In fact, "
244 "cudnnSetTensorNdDescriptor doesn't work for kBatchDepthYX4 at all. "
245 "Use cudnnSetTensor4DDescriptor to set cudnnTensorDescriptor_t "
246 "instead.";
247 }
248 std::vector<int64> phys_dims = full_dims(layout_);
249 std::vector<int64> phys_strides(phys_dims.size());
250 phys_strides[ndims_ + 1] = 1;
251 for (int i = ndims_; i >= 0; i--) {
252 phys_strides[i] = phys_strides[i + 1] * phys_dims[i + 1];
253 }
254 return ReorderDims(phys_strides, layout_, layout);
255}

268string BatchDescriptor::ToString() const {

272 }
273 return port::Printf(

280string BatchDescriptor::ToShortString() const {
281 // All the constituent strings are less than 15 characters, so the
282 // small string optimization ensures that there will be at most one
283 // heap memory allocation.
284 string depth = port::StrCat("d", feature_map_count());
285 string batch = port::StrCat("b", count());
286
287 string spatial = "s";
288 for (int i = 0; i < ndims_; i++) {
289 port::Appendf(&spatial, "%lld ", spatial_size_[i]);
290 }
291
292 string suffix;
293 if (value_min() != value_max()) {
294 port::StrAppend(&suffix, "[", value_min(), ";", value_max(), "]");
295 }
296 if (quantized_activation_mode() == QuantizedActivationMode::k16Bit) {
297 suffix += "_16bit";
298 }
299
300 switch (layout()) {
301 case DataLayout::kYXDepthBatch:
302 return port::StrCat(spatial, depth, batch, suffix);
303 case DataLayout::kYXBatchDepth:
304 return port::StrCat(spatial, batch, depth, suffix);
305 case DataLayout::kBatchYXDepth:
306 return port::StrCat(batch, spatial, depth, suffix);
307 case DataLayout::kBatchDepthYX:
308 return port::StrCat(batch, depth, spatial, suffix);
309 case DataLayout::kBatchDepthYX4:
310 return port::StrCat(batch, depth, spatial, suffix, "(VECT_C)");
311 default:
312 LOG(FATAL) << "Unknown layout " << static_cast<int32>(layout());
313 return ""; // Avoid return warning (unreachable)
314 }
315}

317int64 BatchDescriptor::NodesPerFeatureMap() const {
318 int64 ret = 1;
319 for (int i = 0; i < ndims_; i++) {
320 ret *= spatial_size_[i];
321 }
322 return ret;
323}
324
325int64 BatchDescriptor::NodesAcrossFeatureMaps() const {
326 return NodesPerFeatureMap() * feature_map_count_;
327}
328
329int64 BatchDescriptor::ElementCount() const {
330 return count_ * feature_map_count_ * NodesPerFeatureMap();
331}
332
333int64 BatchDescriptor::FullyConnectedWeightCount(
334 const BatchDescriptor& input, const BatchDescriptor& output) {
335 return input.NodesAcrossFeatureMaps() * output.NodesAcrossFeatureMaps();
336}
337
338int64 BatchDescriptor::FullyConnectedBiasCount(const BatchDescriptor& output) {
339 return output.NodesAcrossFeatureMaps();
340}

342BatchDescriptor BatchDescriptor::DepthConcatenateOutputDescriptor(
343 port::ArraySlice<dnn::BatchDescriptor> inputs) {
344 if (inputs.empty()) {
345 return BatchDescriptor();
346 }
347 int feature_map_count = 0;
348 for (const auto& dimensions : inputs) {
349 feature_map_count += dimensions.feature_map_count();
350 }
351 BatchDescriptor output = inputs[0];
352 output.set_feature_map_count(feature_map_count);
353 return output;
354}

391string FilterDescriptor::ToShortString() const {
392 // All the constituent strings are less than 15 characters, so the
393 // small string optimization ensures that there will be at most one
394 // heap memory allocation.
395 string od = port::StrCat("od", output_feature_map_count_);
396 string id = port::StrCat("id", input_feature_map_count_);
397
398 string spatial = "s";
399 for (int i = 0; i < ndims_; i++) {
400 port::Appendf(&spatial, "%lld ", input_filter_dims_[i]);
401 }
402
403 switch (layout_) {
404 case FilterLayout::kOutputInputYX:
405 return port::StrCat(od, id, spatial);
406 case FilterLayout::kOutputInputYX4:
407 return port::StrCat(od, id, spatial, "(VECT_C)");
408 case FilterLayout::kInputYXOutput:
409 return port::StrCat(id, spatial, od);
410 case FilterLayout::kYXInputOutput:
411 return port::StrCat(spatial, id, od);
412 default:
413 LOG(FATAL) << "Unknown layout " << static_cast<int32>(layout_);
414 return ""; // Avoid return warning (unreachable)
415 }
416}

418int64 FilterDescriptor::ComputeWeightCount() const {
419 int64 ret = output_feature_map_count_ * input_feature_map_count_;
420 for (int i = 0; i < ndims_; i++) {
421 ret *= input_filter_dims_[i];
422 }
423 return ret;

490 propagate_nans_ = other.propagate_nans_;

492
493string PoolingDescriptor::ToString() const {
494 const char* mode_string =
495 mode_ == dnn::PoolingMode::kMaximum ? "kMaximum" : "kAverage";
496
497 string window, strides, padding;
498 for (int i = 0; i < ndims_; i++) {
499 port::Appendf(&window, "%lld ", window_[i]);
500 port::Appendf(&strides, "%lld ", strides_[i]);
501 port::Appendf(&padding, "%lld", padding_[i]);
502 }
503
504 const char* propagate_string = propagate_nans_ ? "Yes" : "No";
505
506 return port::Printf(
507 "{mode: %s window: %s strides: %s padding: %s propagate NaNs: %s}",
508 mode_string, window.c_str(), strides.c_str(), padding.c_str(),
509 propagate_string);
510}
511
512string PoolingDescriptor::ToShortString() const {
513 string window, strides, padding;
514 for (int i = 0; i < ndims_; i++) {
515 port::Appendf(&window, "_w%d:%lld", i, window_[i]);
516 port::Appendf(&strides, "_s%d:%lld", i, strides_[i]);
517 port::Appendf(&padding, "_p%d:%lld", i, padding_[i]);
518 }
519 return port::StrCat(mode_ == dnn::PoolingMode::kMaximum ? "max" : "avg",
520 window, strides, padding,
521 propagate_nans_ ? "propagate_nans" : "ignore_nans");
522}
