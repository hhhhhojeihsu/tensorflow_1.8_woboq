58#define PARSE_VALUE(fn_name, type, check_fn, parse_fn) \
59 bool fn_name(const string& key, PyObject* py_value, TF_Status* status, \
60 type* value) { \
61 if (check_fn(py_value)) { \
62 *value = static_cast<type>(parse_fn(py_value)); \
63 return true; \
64 } else { \
65 TF_SetStatus(status, TF_INVALID_ARGUMENT, \
66 tensorflow::strings::StrCat( \
67 "Expecting " #type " value for attr ", key, ", got ", \
68 py_value->ob_type->tp_name) \
69 .c_str()); \
70 return false; \
71 } \
72 }
73
74#if PY_MAJOR_VERSION >= 3
75PARSE_VALUE(ParseIntValue, int, PyLong_Check, PyLong_AsLong)
76PARSE_VALUE(ParseInt64Value, int64_t, PyLong_Check, PyLong_AsLong)
77#else
78PARSE_VALUE(ParseIntValue, int, PyInt_Check, PyInt_AsLong)
79PARSE_VALUE(ParseInt64Value, int64_t, PyInt_Check, PyInt_AsLong)
80PARSE_VALUE(ParseInt64LongValue, int64_t, PyLong_Check, PyLong_AsLong)
81#endif
82PARSE_VALUE(ParseFloatValue, float, PyFloat_Check, PyFloat_AsDouble)
83#undef PARSE_VALUE

85Py_ssize_t TensorShapeNumDims(PyObject* value) {
86 const auto size = PySequence_Size(value);
87 if (size == -1) {
88 // TensorShape.__len__ raises an error in the scenario where the shape is an
89 // unknown, which needs to be cleared.
90 // TODO(nareshmodi): ensure that this is actually a TensorShape.
91 PyErr_Clear();
92 }
93 return size;
94}

96bool IsInteger(PyObject* py_value) {
97#if PY_MAJOR_VERSION >= 3
98 return PyLong_Check(py_value);
99#else
100 return PyInt_Check(py_value);
101#endif

104bool ParseDimensionValue(const string& key, PyObject* py_value,
105 TF_Status* status, int64_t* value) {
106 if (IsInteger(py_value)) {
107 return ParseInt64Value(key, py_value, status, value);
108 }
109
110 tensorflow::Safe_PyObjectPtr dimension_value(
111 PyObject_GetAttrString(py_value, "_value"));
112 if (dimension_value == nullptr) {
113 TF_SetStatus(
114 status, TF_INVALID_ARGUMENT,
115 tensorflow::strings::StrCat("Expecting a Dimension for attr ", key,
116 ", got ", py_value->ob_type->tp_name)
117 .c_str());
118 return false;
119 }
120
121 return ParseInt64Value(key, dimension_value.get(), status, value);
122}

124bool ParseStringValue(const string& key, PyObject* py_value, TF_Status* status,
125 const char** value) {
126 if (PyBytes_Check(py_value)) {
127 *value = PyBytes_AsString(py_value);
128 return true;
129 }
130#if PY_MAJOR_VERSION >= 3
131 if (PyUnicode_Check(py_value)) {
132 *value = PyUnicode_AsUTF8(py_value);
133 return true;
134 }
135#endif
136 TF_SetStatus(
137 status, TF_INVALID_ARGUMENT,
138 tensorflow::strings::StrCat("Expecting a string value for attr ", key,
139 ", got ", py_value->ob_type->tp_name)
140 .c_str());
141 return false;
142}

144bool ParseBoolValue(const string& key, PyObject* py_value, TF_Status* status,
145 unsigned char* value) {
146 *value = PyObject_IsTrue(py_value);
147 return true;
148}

150// The passed in py_value is expected to be an object of the python type
151// dtypes.DType or an int.
152bool ParseTypeValue(const string& key, PyObject* py_value, TF_Status* status,
153 int* value) {
154 if (IsInteger(py_value)) {
155 return ParseIntValue(key, py_value, status, value);
156 }
157
158 tensorflow::Safe_PyObjectPtr py_type_enum(
159 PyObject_GetAttrString(py_value, "_type_enum"));
160 if (py_type_enum == nullptr) {
161 TF_SetStatus(
162 status, TF_INVALID_ARGUMENT,
163 tensorflow::strings::StrCat("Expecting a DType.dtype for attr ", key,
164 ", got ", py_value->ob_type->tp_name)
165 .c_str());
166 return false;
167 }
168
169 return ParseIntValue(key, py_type_enum.get(), status, value);
170}

172bool SetOpAttrList(
173 TFE_Op* op, const char* key, PyObject* py_list, TF_AttrType type,
174 tensorflow::gtl::FlatMap<string, tensorflow::int64>* attr_list_sizes,
175 TF_Status* status) {
176 if (!PySequence_Check(py_list)) {
177 TF_SetStatus(
178 status, TF_INVALID_ARGUMENT,
179 tensorflow::strings::StrCat("Expecting sequence value for attr ", key,
180 ", got ", py_list->ob_type->tp_name)
181 .c_str());
182 return false;
183 }
184 const int num_values = PySequence_Size(py_list);
185 if (attr_list_sizes != nullptr) (*attr_list_sizes)[key] = num_values;
186
187#define PARSE_LIST(c_type, parse_fn) \
188 std::unique_ptr<c_type[]> values(new c_type[num_values]); \
189 for (int i = 0; i < num_values; ++i) { \
190 tensorflow::Safe_PyObjectPtr py_value(PySequence_ITEM(py_list, i)); \
191 if (!parse_fn(key, py_value.get(), status, &values[i])) return false; \
192 }
193
194 if (type == TF_ATTR_STRING) {
195 PARSE_LIST(const char*, ParseStringValue);
196 TFE_OpSetAttrStringList(op, key, values.get(), num_values);
197 } else if (type == TF_ATTR_INT) {
198 PARSE_LIST(int64_t, ParseInt64Value);
199 TFE_OpSetAttrIntList(op, key, values.get(), num_values);
200 } else if (type == TF_ATTR_FLOAT) {
201 PARSE_LIST(float, ParseFloatValue);
202 TFE_OpSetAttrFloatList(op, key, values.get(), num_values);
203 } else if (type == TF_ATTR_BOOL) {
204 PARSE_LIST(unsigned char, ParseBoolValue);
205 TFE_OpSetAttrBoolList(op, key, values.get(), num_values);
206 } else if (type == TF_ATTR_TYPE) {
207 PARSE_LIST(int, ParseTypeValue);
208 TFE_OpSetAttrTypeList(op, key,
209 reinterpret_cast<const TF_DataType*>(values.get()),
210 num_values);
211 } else if (type == TF_ATTR_SHAPE) {
212 // Make one pass through the input counting the total number of
213 // dims across all the input lists.
214 int total_dims = 0;
215 for (int i = 0; i < num_values; ++i) {
216 tensorflow::Safe_PyObjectPtr py_value(PySequence_ITEM(py_list, i));
217 if (py_value.get() != Py_None) {
218 if (!PySequence_Check(py_value.get())) {
219 TF_SetStatus(
220 status, TF_INVALID_ARGUMENT,
221 tensorflow::strings::StrCat(
222 "Expecting None or sequence value for element", i,
223 " of attr ", key, ", got ", py_value->ob_type->tp_name)
224 .c_str());
225 return false;
226 }
227 const auto size = TensorShapeNumDims(py_value.get());
228 if (size >= 0) {
229 total_dims += size;
230 }
231 }
232 }
233 // Allocate a buffer that can fit all of the dims together.
234 std::unique_ptr<int64_t[]> buffer(new int64_t[total_dims]);
235 // Copy the input dims into the buffer and set dims to point to
236 // the start of each list's dims.
237 std::unique_ptr<const int64_t* []> dims(new const int64_t*[num_values]);
238 std::unique_ptr<int[]> num_dims(new int[num_values]);
239 int64_t* offset = buffer.get();
240 for (int i = 0; i < num_values; ++i) {
241 tensorflow::Safe_PyObjectPtr py_value(PySequence_ITEM(py_list, i));
242 if (py_value.get() == Py_None) {
243 dims[i] = nullptr;
244 num_dims[i] = -1;
245 } else {
246 const auto size = TensorShapeNumDims(py_value.get());
247 if (size == -1) {
248 dims[i] = nullptr;
249 num_dims[i] = -1;
250 continue;
251 }
252 dims[i] = offset;
253 num_dims[i] = size;
254 for (int j = 0; j < size; ++j) {
255 tensorflow::Safe_PyObjectPtr inner_py_value(
256 PySequence_ITEM(py_value.get(), j));
257 if (inner_py_value.get() == Py_None) {
258 *offset = -1;
259 } else if (!ParseDimensionValue(key, inner_py_value.get(), status,
260 offset)) {
261 return false;
262 }
263 ++offset;
264 }
265 }
266 }
267 TFE_OpSetAttrShapeList(op, key, dims.get(), num_dims.get(), num_values,
268 status);
269 if (TF_GetCode(status) != TF_OK) return false;
270 } else {
271 TF_SetStatus(status, TF_UNIMPLEMENTED,
272 tensorflow::strings::StrCat("Attr ", key,
273 " has unhandled list type ", type)
274 .c_str());
275 return false;
276 }
277#undef PARSE_LIST
278 return true;
279}

281TFE_Op* GetFunc(TFE_Context* ctx, const tensorflow::NameAttrList& func,
282 TF_Status* status) {
283 TFE_Op* func_op = TFE_NewOp(ctx, func.name().data(), status);
284 for (const auto& attr : func.attr()) {
285 if (TF_GetCode(status) != TF_OK) return nullptr;
286 SetOpAttrValueScalar(ctx, func_op, attr.second, attr.first.data(), status);
287 if (TF_GetCode(status) != TF_OK) return nullptr;
288 }
289 return func_op;
290}

292void SetOpAttrListDefault(
293 TFE_Context* ctx, TFE_Op* op, const tensorflow::OpDef::AttrDef& attr,
294 const char* key, TF_AttrType type,
295 tensorflow::gtl::FlatMap<string, tensorflow::int64>* attr_list_sizes,
296 TF_Status* status) {
297 if (type == TF_ATTR_STRING) {
298 int num_values = attr.default_value().list().s_size();
299 std::unique_ptr<const char* []> values(new const char*[num_values]);
300 (*attr_list_sizes)[key] = num_values;
301 for (int i = 0; i < num_values; i++) {
302 values[i] = attr.default_value().list().s(i).data();
303 }
304 TFE_OpSetAttrStringList(op, key, values.get(), num_values);
305 } else if (type == TF_ATTR_INT) {
306 int num_values = attr.default_value().list().i_size();
307 std::unique_ptr<int64_t[]> values(new int64_t[num_values]);
308 (*attr_list_sizes)[key] = num_values;
309 for (int i = 0; i < num_values; i++) {
310 values[i] = attr.default_value().list().i(i);
311 }
312 TFE_OpSetAttrIntList(op, key, values.get(), num_values);
313 } else if (type == TF_ATTR_FLOAT) {
314 int num_values = attr.default_value().list().f_size();
315 std::unique_ptr<float[]> values(new float[num_values]);
316 (*attr_list_sizes)[key] = num_values;
317 for (int i = 0; i < num_values; i++) {
318 values[i] = attr.default_value().list().f(i);
319 }
320 TFE_OpSetAttrFloatList(op, key, values.get(), num_values);
321 } else if (type == TF_ATTR_BOOL) {
322 int num_values = attr.default_value().list().b_size();
323 std::unique_ptr<unsigned char[]> values(new unsigned char[num_values]);
324 (*attr_list_sizes)[key] = num_values;
325 for (int i = 0; i < num_values; i++) {
326 values[i] = attr.default_value().list().b(i);
327 }
328 TFE_OpSetAttrBoolList(op, key, values.get(), num_values);
329 } else if (type == TF_ATTR_TYPE) {
330 int num_values = attr.default_value().list().type_size();
331 std::unique_ptr<int[]> values(new int[num_values]);
332 (*attr_list_sizes)[key] = num_values;
333 for (int i = 0; i < num_values; i++) {
334 values[i] = attr.default_value().list().type(i);
335 }
336 TFE_OpSetAttrTypeList(op, key,
337 reinterpret_cast<const TF_DataType*>(values.get()),
338 attr.default_value().list().type_size());
339 } else if (type == TF_ATTR_SHAPE) {
340 int num_values = attr.default_value().list().shape_size();
341 (*attr_list_sizes)[key] = num_values;
342 int total_dims = 0;
343 for (int i = 0; i < num_values; ++i) {
344 if (!attr.default_value().list().shape(i).unknown_rank()) {
345 total_dims += attr.default_value().list().shape(i).dim_size();
346 }
347 }
348 // Allocate a buffer that can fit all of the dims together.
349 std::unique_ptr<int64_t[]> buffer(new int64_t[total_dims]);
350 // Copy the input dims into the buffer and set dims to point to
351 // the start of each list's dims.
352 std::unique_ptr<const int64_t* []> dims(new const int64_t*[num_values]);
353 std::unique_ptr<int[]> num_dims(new int[num_values]);
354 int64_t* offset = buffer.get();
355 for (int i = 0; i < num_values; ++i) {
356 const auto& shape = attr.default_value().list().shape(i);
357 if (shape.unknown_rank()) {
358 dims[i] = nullptr;
359 num_dims[i] = -1;
360 } else {
361 for (int j = 0; j < shape.dim_size(); j++) {
362 *offset = shape.dim(j).size();
363 ++offset;
364 }
365 }
366 }
367 TFE_OpSetAttrShapeList(op, key, dims.get(), num_dims.get(), num_values,
368 status);
369 } else if (type == TF_ATTR_FUNC) {
370 int num_values = attr.default_value().list().func_size();
371 (*attr_list_sizes)[key] = num_values;
372 std::unique_ptr<const TFE_Op* []> funcs(new const TFE_Op*[num_values]);
373 for (int i = 0; i < num_values; i++) {
374 funcs[i] = GetFunc(ctx, attr.default_value().list().func(i), status);
375 }
376 TFE_OpSetAttrFunctionList(op, key, funcs.get(), num_values);
377 } else {
378 TF_SetStatus(status, TF_UNIMPLEMENTED,
379 "Lists of tensors are not yet implemented for default valued "
380 "attributes for an operation.");
381 }
382}

384bool SetOpAttrScalar(
385 TFE_Context* ctx, TFE_Op* op, const char* key, PyObject* py_value,
386 TF_AttrType type,
387 tensorflow::gtl::FlatMap<string, tensorflow::int64>* attr_list_sizes,
388 TF_Status* status) {
389 if (type == TF_ATTR_STRING) {
390 const char* value;
391 if (!ParseStringValue(key, py_value, status, &value)) return false;
392 TFE_OpSetAttrString(op, key, value);
393 } else if (type == TF_ATTR_INT) {
394 int64_t value;
395 if (!ParseInt64Value(key, py_value, status, &value)) return false;
396 TFE_OpSetAttrInt(op, key, value);
397 // attr_list_sizes is set for all int attributes (since at this point we are
398 // not aware if that attribute might be used to calculate the size of an
399 // output list or not).
400 if (attr_list_sizes != nullptr) (*attr_list_sizes)[key] = value;
401 } else if (type == TF_ATTR_FLOAT) {
402 float value;
403 if (!ParseFloatValue(key, py_value, status, &value)) return false;
404 TFE_OpSetAttrFloat(op, key, value);
405 } else if (type == TF_ATTR_BOOL) {
406 unsigned char value;
407 if (!ParseBoolValue(key, py_value, status, &value)) return false;
408 TFE_OpSetAttrBool(op, key, value);
409 } else if (type == TF_ATTR_TYPE) {
410 int value;
411 if (!ParseTypeValue(key, py_value, status, &value)) return false;
412 TFE_OpSetAttrType(op, key, static_cast<TF_DataType>(value));
413 } else if (type == TF_ATTR_SHAPE) {
414 if (py_value == Py_None) {
415 TFE_OpSetAttrShape(op, key, nullptr, -1, status);
416 } else {
417 if (!PySequence_Check(py_value)) {
418 TF_SetStatus(status, TF_INVALID_ARGUMENT,
419 tensorflow::strings::StrCat(
420 "Expecting None or sequence value for attr", key,
421 ", got ", py_value->ob_type->tp_name)
422 .c_str());
423 return false;
424 }
425 const auto num_dims = TensorShapeNumDims(py_value);
426 if (num_dims == -1) {
427 TFE_OpSetAttrShape(op, key, nullptr, -1, status);
428 return true;
429 }
430 std::unique_ptr<int64_t[]> dims(new int64_t[num_dims]);
431 for (int i = 0; i < num_dims; ++i) {
432 tensorflow::Safe_PyObjectPtr inner_py_value(
433 PySequence_ITEM(py_value, i));
434 if (inner_py_value.get() == Py_None) {
435 dims[i] = -1;
436 } else if (!ParseDimensionValue(key, inner_py_value.get(), status,
437 &dims[i])) {
438 return false;
439 }
440 }
441 TFE_OpSetAttrShape(op, key, dims.get(), num_dims, status);
442 }
443 if (TF_GetCode(status) != TF_OK) return false;
444 } else if (type == TF_ATTR_FUNC) {
445 // Allow:
446 // (1) String function name, OR
447 // (2) A Python object with a .name attribute
448 // (A crude test for being a
449 // tensorflow.python.framework.function._DefinedFunction)
450 // (which is what the various "defun" or "Defun" decorators do).
451 // And in the future also allow an object that can encapsulate
452 // the function name and its attribute values.
453 const char* func_name = nullptr;
454 if (!ParseStringValue(key, py_value, status, &func_name)) {
455 PyObject* name_attr = PyObject_GetAttrString(py_value, "name");
456 if (name_attr == nullptr ||
457 !ParseStringValue(key, name_attr, status, &func_name)) {
458 TF_SetStatus(
459 status, TF_INVALID_ARGUMENT,
460 tensorflow::strings::StrCat(
461 "unable to set function value attribute from a ",
462 py_value->ob_type->tp_name,
463 " object. If you think this is an error, please file an issue "
464 "at https://github.com/tensorflow/tensorflow/issues/new")
465 .c_str());
466 return false;
467 }
468 }
469 TFE_Op* func = TFE_NewOp(ctx, func_name, status);
470 if (TF_GetCode(status) != TF_OK) return false;
471 TFE_OpSetAttrFunction(op, key, func);
472 TFE_DeleteOp(func);
473 } else {
474 TF_SetStatus(
475 status, TF_UNIMPLEMENTED,
476 tensorflow::strings::StrCat("Attr ", key, " has unhandled type ", type)
477 .c_str());
478 return false;
479 }
480 return true;
481}

494// start_index is the index at which the Tuple/List attrs will start getting
495// processed.
496void SetOpAttrs(TFE_Context* ctx, TFE_Op* op, PyObject* attrs, int start_index,
497 TF_Status* out_status) {
498 if (attrs == Py_None) return;
499 Py_ssize_t len = PyTuple_GET_SIZE(attrs) - start_index;
500 if ((len & 1) != 0) {
501 TF_SetStatus(out_status, TF_INVALID_ARGUMENT,
502 "Expecting attrs tuple to have even length.");
503 return;
504 }
505 // Parse attrs
506 for (Py_ssize_t i = 0; i < len; i += 2) {
507 PyObject* py_key = PyTuple_GET_ITEM(attrs, start_index + i);
508 PyObject* py_value = PyTuple_GET_ITEM(attrs, start_index + i + 1);
509#if PY_MAJOR_VERSION >= 3
510 const char* key = PyBytes_Check(py_key) ? PyBytes_AsString(py_key)
511 : PyUnicode_AsUTF8(py_key);
512#else
513 const char* key = PyBytes_AsString(py_key);
514#endif
515 unsigned char is_list = 0;
516 const TF_AttrType type = TFE_OpGetAttrType(op, key, &is_list, out_status);
517 if (TF_GetCode(out_status) != TF_OK) return;
518 if (is_list != 0) {
519 if (!SetOpAttrList(op, key, py_value, type, nullptr, out_status)) return;
520 } else {
521 if (!SetOpAttrScalar(ctx, op, key, py_value, type, nullptr, out_status))
522 return;
523 }
524 }
525}

527// This function will set the op attrs required. If an attr has the value of
528// None, then it will read the AttrDef to get the default value and set that
529// instead. Any failure in this function will simply fall back to the slow
530// path.
531void SetOpAttrWithDefaults(
532 TFE_Context* ctx, TFE_Op* op, const tensorflow::OpDef::AttrDef& attr,
533 const char* attr_name, PyObject* attr_value,
534 tensorflow::gtl::FlatMap<string, tensorflow::int64>* attr_list_sizes,
535 TF_Status* status) {
536 unsigned char is_list = 0;
537 const TF_AttrType type = TFE_OpGetAttrType(op, attr_name, &is_list, status);
538 if (TF_GetCode(status) != TF_OK) return;
539 if (attr_value == Py_None) {
540 if (is_list != 0) {
541 SetOpAttrListDefault(ctx, op, attr, attr_name, type, attr_list_sizes,
542 status);
543 } else {
544 SetOpAttrScalarDefault(ctx, op, attr.default_value(), attr_name,
545 attr_list_sizes, status);
546 }
547 } else {
548 if (is_list != 0) {
549 SetOpAttrList(op, attr_name, attr_value, type, attr_list_sizes, status);
550 } else {
551 SetOpAttrScalar(ctx, op, attr_name, attr_value, type, attr_list_sizes,
552 status);
553 }
554 }

574void TFE_Py_Execute(TFE_Context* ctx, const char* device_name,
575 const char* op_name, TFE_InputTensorHandles* inputs,
576 PyObject* attrs, TFE_OutputTensorHandles* outputs,
577 TF_Status* out_status) {
578 TFE_Op* op = TFE_NewOp(ctx, op_name, out_status);
579 if (TF_GetCode(out_status) != TF_OK) return;
580 TFE_OpSetDevice(op, device_name, out_status);
581 if (TF_GetCode(out_status) == TF_OK) {
582 for (int i = 0; i < inputs->size() && TF_GetCode(out_status) == TF_OK;
583 ++i) {
584 TFE_OpAddInput(op, inputs->at(i), out_status);
585 }
586 }
587 if (TF_GetCode(out_status) == TF_OK) {
588 SetOpAttrs(ctx, op, attrs, 0, out_status);
589 }
590 Py_BEGIN_ALLOW_THREADS;
591 if (TF_GetCode(out_status) == TF_OK) {
592 int num_outputs = outputs->size();
593 TFE_Execute(op, outputs->data(), &num_outputs, out_status);
594 outputs->resize(num_outputs);
595 }
596 if (TF_GetCode(out_status) != TF_OK) {
597 TF_SetStatus(out_status, TF_GetCode(out_status),
598 tensorflow::strings::StrCat(TF_Message(out_status),
599 " [Op:", op_name, "]")
600 .c_str());
601 }
602 TFE_DeleteOp(op);
603 Py_END_ALLOW_THREADS;
604}

1011static tensorflow::eager::TapeTensor TapeTensorFromTensor(PyObject* tensor) {
1012 if (EagerTensor_CheckExact(tensor)) {
1013 TFE_TensorHandle* t = EagerTensor_Handle(tensor);
1014 tensorflow::int64 id = EagerTensor_id(tensor);
1015 const tensorflow::Tensor* tensor = nullptr;
1016 const tensorflow::Status status = t->handle->Tensor(&tensor);
1017 if (MaybeRaiseExceptionFromStatus(status, nullptr)) {
1018 return tensorflow::eager::TapeTensor{id, t->handle->dtype,
1019 tensorflow::TensorShape({})};
1020 } else {
1021 return tensorflow::eager::TapeTensor{id, t->handle->dtype,
1022 tensor->shape()};
1023 }
1024 }
1025 tensorflow::int64 id = FastTensorId(tensor);
1026 if (PyErr_Occurred()) {
1027 return tensorflow::eager::TapeTensor{
1028 id, static_cast<tensorflow::DataType>(0), tensorflow::TensorShape({})};
1029 }
1030 PyObject* dtype_object = PyObject_GetAttrString(tensor, "dtype");
1031 PyObject* dtype_enum = PyObject_GetAttrString(dtype_object, "_type_enum");
1032 Py_DECREF(dtype_object);
1033 tensorflow::DataType dtype =
1034 static_cast<tensorflow::DataType>(MakeInt(dtype_enum));
1035 Py_DECREF(dtype_enum);
1036 if (PyErr_Occurred() != nullptr) {
1037 return tensorflow::eager::TapeTensor{id, dtype,
1038 tensorflow::TensorShape({})};
1039 }
1040 static char _shape_tuple[] = "_shape_tuple";
1041 PyObject* shape_tuple = PyObject_CallMethod(tensor, _shape_tuple, nullptr);
1042 if (PyErr_Occurred() != nullptr) {
1043 return tensorflow::eager::TapeTensor{id, dtype,
1044 tensorflow::TensorShape({})};
1045 }
1046 auto l = MakeIntList(shape_tuple);
1047 Py_DECREF(shape_tuple);
1048 // Replace -1, which represents accidental Nones which can occur in graph mode
1049 // and can cause errors in shape cosntruction with 0s.
1050 for (auto& c : l) {
1051 if (c < 0) {
1052 c = 0;
1053 }
1054 }
1055 tensorflow::TensorShape shape(l);
1056 return tensorflow::eager::TapeTensor{id, dtype, shape};
1057}

1124 if (py_str == nullptr) return;

1126 Py_DECREF(py_str);
1127#endif

1130 return;

1250 tensorflow::Status CallBackwardFunction(
1251 PyObject* backward_function,
1252 tensorflow::gtl::ArraySlice<PyObject*> output_gradients,
1253 std::vector<PyObject*>* result) const final {
1254 PyObject* grads = PyTuple_New(output_gradients.size());
1255 for (int i = 0; i < output_gradients.size(); ++i) {
1256 if (output_gradients[i] == nullptr) {
1257 Py_INCREF(Py_None);
1258 PyTuple_SET_ITEM(grads, i, Py_None);
1259 } else {
1260 PyTuple_SET_ITEM(grads, i,
1261 reinterpret_cast<PyObject*>(output_gradients[i]));
1262 }
1263 }
1264 PyObject* py_result = PyEval_CallObject(
1265 reinterpret_cast<PyObject*>(backward_function), grads);
1266 Py_DECREF(grads);
1267 if (py_result == nullptr) {
1268 return tensorflow::errors::Internal("gradient function threw exceptions");
1269 }
1270 result->clear();
1271 PyObject* seq =
1272 PySequence_Fast(py_result, "expected a sequence of gradients");
1273 if (seq == nullptr) {
1274 return tensorflow::errors::InvalidArgument(
1275 "gradient function did not return a list");
1276 }
1277 int len = PySequence_Fast_GET_SIZE(seq);
1278 VLOG(1) << "Gradient length is " << len;
1279 result->reserve(len);
1280 for (int i = 0; i < len; ++i) {
1281 PyObject* item = PySequence_Fast_GET_ITEM(seq, i);
1282 if (item == Py_None) {
1283 result->push_back(nullptr);
1284 } else {
1285 Py_INCREF(item);
1286 result->push_back(item);
1287 }
1288 }
1289 Py_DECREF(seq);
1290 Py_DECREF(py_result);
1291 return tensorflow::Status::OK();
1292 }

1324PyObject* TFE_Py_TapeGradient(PyObject* tape, PyObject* vspace,
1325 PyObject* target, PyObject* sources,
1326 PyObject* output_gradients, TF_Status* status) {
1327 TFE_Py_Tape* tape_obj = reinterpret_cast<TFE_Py_Tape*>(tape);
1328 if (!tape_obj->tape->IsPersistent()) {
1329 auto* tape_set = GetTapeSet();
1330 if (tape_set->find(tape_obj) != tape_set->end()) {
1331 PyErr_SetString(PyExc_RuntimeError,
1332 "Trying to call tape.gradient on a non-persistent tape "
1333 "while it is still active.");
1334 return nullptr;
1335 }
1336 }
1337 PyVSpace c_vspace(vspace);
1338 if (!c_vspace.Initialize().ok()) {
1339 return nullptr;
1340 }
1341
1342 std::vector<tensorflow::int64> target_vec = MakeTensorIDList(target);
1343 if (PyErr_Occurred()) {
1344 return nullptr;
1345 }
1346 std::vector<tensorflow::int64> sources_vec = MakeTensorIDList(sources);
1347 if (PyErr_Occurred()) {
1348 return nullptr;
1349 }
1350 std::vector<PyObject*> outgrad_vec;
1351 if (output_gradients != Py_None) {
1352 outgrad_vec = MakeTensorList(output_gradients);
1353 if (PyErr_Occurred()) {
1354 return nullptr;
1355 }
1356 for (PyObject* tensor : outgrad_vec) {
1357 // Calling the backward function will eat a reference to the tensors in
1358 // outgrad_vec, so we need to increase their reference count.
1359 Py_INCREF(tensor);
1360 }
1361 }
1362 std::vector<PyObject*> result;
1363 status->status = tape_obj->tape->ComputeGradient(
1364 c_vspace, target_vec, sources_vec, outgrad_vec, &result);
1365 if (!status->status.ok()) {
1366 if (PyErr_Occurred()) {
1367 // Do not propagate the erroneous status as that would swallow the
1368 // exception which caused the problem.
1369 status->status = tensorflow::Status::OK();
1370 }
1371 return nullptr;
1372 }
1373 if (!result.empty()) {
1374 PyObject* py_result = PyList_New(result.size());
1375 tensorflow::gtl::FlatSet<PyObject*> seen_results(result.size());
1376 for (int i = 0; i < result.size(); ++i) {
1377 if (result[i] == nullptr) {
1378 Py_INCREF(Py_None);
1379 result[i] = Py_None;
1380 } else if (seen_results.find(result[i]) != seen_results.end()) {
1381 Py_INCREF(result[i]);
1382 }
1383 seen_results.insert(result[i]);
1384 PyList_SET_ITEM(py_result, i, reinterpret_cast<PyObject*>(result[i]));
1385 }
1386 return py_result;
1387 }
1388 return PyList_New(0);
1389}

1558PyObject* RecordGradient(PyObject* op_name, PyObject* inputs, PyObject* attrs,
1559 PyObject* results, PyObject* name) {
1560 std::vector<tensorflow::int64> input_ids = MakeTensorIDList(inputs);
1561 if (PyErr_Occurred()) return nullptr;
1562
1563 bool should_record = false;
1564 for (TFE_Py_Tape* tape : SafeTapeSet()) {
1565 if (tape->tape->ShouldRecord(input_ids)) {
1566 should_record = true;
1567 break;
1568 }
1569 }
1570 if (!should_record) Py_RETURN_NONE;
1571
1572 string c_op_name = TFE_GetPythonString(op_name);
1573 PyObject* op_outputs;
1574 if (OpDoesntRequireOutput(c_op_name)) {
1575 op_outputs = Py_None;
1576 } else {
1577 op_outputs = results;
1578 }
1579
1580 PyObject* op_inputs;
1581 if (OpDoesntRequireInput(c_op_name)) {
1582 op_inputs = Py_None;
1583 } else {
1584 op_inputs = inputs;
1585 }
1586
1587 PyObject* num_inputs = PyLong_FromLong(PySequence_Size(inputs));
1588 PyObject* callback_args =
1589 Py_BuildValue("OOOOO", op_name, attrs, num_inputs, op_inputs, op_outputs);
1590
1591 PyObject* backward_function =
1592 PyObject_CallObject(backward_function_getter, callback_args);
1593 Py_DECREF(callback_args);
1594 if (backward_function == nullptr) return nullptr;
1595
1596 TapeSetRecordOperation(op_name, results, input_ids, backward_function);
1597
1598 Py_DECREF(backward_function);
1599
1600 Py_RETURN_NONE;
1601}

1613bool ReadVariableOp(const FastPathOpExecInfo& parent_op_exec_info,
1614 PyObject* input, tensorflow::Safe_PyObjectPtr* output,
1615 TF_Status* status) {
1616 MaybeWatchVariable(input);
1617
1618 TFE_Op* op = TFE_NewOp(parent_op_exec_info.ctx, "ReadVariableOp", status);
1619 auto cleaner = tensorflow::gtl::MakeCleanup([op] { TFE_DeleteOp(op); });
1620 if (MaybeRaiseExceptionFromTFStatus(status, nullptr)) return false;
1621
1622 // Set dtype
1623 DCHECK(PyObject_HasAttrString(input, "_dtype"));
1624 tensorflow::Safe_PyObjectPtr dtype(PyObject_GetAttrString(input, "_dtype"));
1625 int value;
1626 if (!ParseTypeValue("_dtype", dtype.get(), status, &value)) {
1627 return false;
1628 }
1629 TFE_OpSetAttrType(op, "dtype", static_cast<TF_DataType>(value));
1630
1631 TFE_OpSetDevice(op, parent_op_exec_info.device_name, status);
1632 if (MaybeRaiseExceptionFromTFStatus(status, nullptr)) return false;
1633
1634 // Get handle
1635 tensorflow::Safe_PyObjectPtr handle(PyObject_GetAttrString(input, "_handle"));
1636 if (!EagerTensor_CheckExact(handle.get())) return false;
1637 TFE_OpAddInput(op, EagerTensor_Handle(handle.get()), status);
1638 if (MaybeRaiseExceptionFromTFStatus(status, nullptr)) return false;
1639
1640 int num_retvals = 1;
1641 TFE_TensorHandle* output_handle;
1642 TFE_Execute(op, &output_handle, &num_retvals, status);
1643 if (MaybeRaiseExceptionFromTFStatus(status, nullptr)) return false;
1644
1645 // Always create the py object (and correctly DECREF it) from the returned
1646 // value, else the data will leak.
1647 output->reset(EagerTensorFromHandle(output_handle));
1648
1649 // TODO(nareshmodi): Should we run post exec callbacks here?
1650 if (parent_op_exec_info.run_gradient_callback) {
1651 tensorflow::Safe_PyObjectPtr inputs(PyTuple_New(1));
1652 PyTuple_SET_ITEM(inputs.get(), 0, handle.release());
1653
1654 tensorflow::Safe_PyObjectPtr outputs(PyTuple_New(1));
1655 Py_INCREF(output->get()); // stay alive after since tuple steals.
1656 PyTuple_SET_ITEM(outputs.get(), 0, output->get());
1657
1658 if (!RecordGradient(GetPythonObjectFromString("ReadVariableOp"),
1659 inputs.get(), Py_None, outputs.get(), Py_None)) {
1660 return false;
1661 }
1662 }
1663
1664 return true;
1665}

1762bool RunCallbacks(
1763 const FastPathOpExecInfo& op_exec_info, PyObject* args,
1764 const std::vector<tensorflow::Safe_PyObjectPtr>& flattened_inputs,
1765 const std::vector<tensorflow::Safe_PyObjectPtr>& flattened_attrs,
1766 PyObject* flattened_result) {
1767 if (!op_exec_info.run_callbacks) return true;
1768
1769 tensorflow::Safe_PyObjectPtr inputs(PyTuple_New(flattened_inputs.size()));
1770 for (int i = 0; i < flattened_inputs.size(); i++) {
1771 PyObject* input = flattened_inputs[i].get();
1772 Py_INCREF(input);
1773 PyTuple_SET_ITEM(inputs.get(), i, input);
1774 }
1775
1776 int num_non_inferred_attrs = PyTuple_GET_SIZE(args) -
1777 op_exec_info.op_def->input_arg_size() -
1778 kFastPathExecuteInputStartIndex;
1779 int num_attrs = flattened_attrs.size() + num_non_inferred_attrs;
1780 tensorflow::Safe_PyObjectPtr attrs(PyTuple_New(num_attrs));
1781
1782 for (int i = 0; i < num_non_inferred_attrs; i++) {
1783 auto* attr =
1784 PyTuple_GET_ITEM(args, kFastPathExecuteInputStartIndex +
1785 op_exec_info.op_def->input_arg_size() + i);
1786 Py_INCREF(attr);
1787 PyTuple_SET_ITEM(attrs.get(), i, attr);
1788 }
1789 for (int i = num_non_inferred_attrs; i < num_attrs; i++) {
1790 PyObject* attr_or_name =
1791 flattened_attrs.at(i - num_non_inferred_attrs).get();
1792 Py_INCREF(attr_or_name);
1793 PyTuple_SET_ITEM(attrs.get(), i, attr_or_name);
1794 }
1795
1796 if (op_exec_info.run_gradient_callback) {
1797 if (!RecordGradient(op_exec_info.op_name, inputs.get(), attrs.get(),
1798 flattened_result, op_exec_info.name)) {
1799 return false;
1800 }
1801 }
1802
1803 if (op_exec_info.run_post_exec_callbacks) {
1804 tensorflow::Safe_PyObjectPtr callback_args(
1805 Py_BuildValue("OOOOO", op_exec_info.op_name, inputs.get(), attrs.get(),
1806 flattened_result, op_exec_info.name));
1807 for (Py_ssize_t i = 0; i < PyList_Size(op_exec_info.callbacks); i++) {
1808 PyObject* callback_fn = PyList_GET_ITEM(op_exec_info.callbacks, i);
1809 if (!PyCallable_Check(callback_fn)) {
1810 PyErr_SetString(
1811 PyExc_TypeError,
1812 Printf("expected a function for "
1813 "post execution callback in index %ld, got %s instead",
1814 i, callback_fn->ob_type->tp_name)
1815 .c_str());
1816 return false;
1817 }
1818 PyObject* callback_result =
1819 PyObject_CallObject(callback_fn, callback_args.get());
1820 if (!callback_result) {
1821 return false;
1822 }
1823 Py_DECREF(callback_result);
1824 }
1825 }
1826
1827 return true;
1828}

1832PyObject* TFE_Py_FastPathExecute_C(PyObject*, PyObject* args) {
1833 Py_ssize_t args_size = PyTuple_GET_SIZE(args);
1834 if (args_size < kFastPathExecuteInputStartIndex) {
1835 PyErr_SetString(
1836 PyExc_ValueError,
1837 Printf("There must be at least %d items in the input tuple.",

1842
1843 FastPathOpExecInfo op_exec_info;
1844
1845 op_exec_info.ctx = reinterpret_cast<TFE_Context*>(
1846 PyCapsule_GetPointer(PyTuple_GET_ITEM(args, 0), nullptr));
1847
1848 if (op_exec_info.ctx == nullptr) {
1849 // The context hasn't been initialized. It will be in the slow path.
1850 RaiseFallbackException(
1851 "This function does not handle the case of the path where "

1865 // TODO(nareshmodi): Add a benchmark for the fast-path with gradient callbacks
1866 // (similar to benchmark_tf_gradient_function_*). Also consider using an
1867 // InlinedVector for flattened_attrs and flattened_inputs if the benchmarks
1868 // point out problems with heap allocs.
1869 op_exec_info.run_gradient_callback =
1870 !*ThreadTapeIsStopped() && !GetTapeSet()->empty();
1871 op_exec_info.run_post_exec_callbacks =
1872 op_exec_info.callbacks != Py_None &&
1873 PyList_Size(op_exec_info.callbacks) > 0;
1874 op_exec_info.run_callbacks = op_exec_info.run_gradient_callback ||
1875 op_exec_info.run_post_exec_callbacks;
1876
1877 if (args_size < kFastPathExecuteInputStartIndex + op_def->input_arg_size()) {
1878 PyErr_SetString(
1879 PyExc_ValueError,
1880 Printf("Tuple size smaller than intended. Expected to be at least %d, "
1881 "was %ld",
1882 kFastPathExecuteInputStartIndex + op_def->input_arg_size(),
1883 args_size)
1884 .c_str());
1885 return nullptr;
1886 }
1887
1888 if (!CheckInputsOk(args, kFastPathExecuteInputStartIndex, *op_def)) {
1889 RaiseFallbackException(
1890 "This function does not handle the case of the path where "
1891 "all inputs are not already EagerTensors.");
1892 return nullptr;
1893 }
1894
1895 TF_Status* status = TF_NewStatus();
1896 TFE_Op* op = TFE_NewOp(op_exec_info.ctx, op_def->name().c_str(), status);
1897 auto cleaner = tensorflow::gtl::MakeCleanup([status, op] {
1898 TF_DeleteStatus(status);
1899 TFE_DeleteOp(op);
1900 });
1901 if (MaybeRaiseExceptionFromTFStatus(status, nullptr)) {
1902 return nullptr;
1903 }
1904
1905 // Mapping of attr name to size - used to calculate the number of values
1906 // to be expected by the TFE_Execute run.
1907 tensorflow::gtl::FlatMap<string, tensorflow::int64> attr_list_sizes;
1908
1909 // Set non-inferred attrs, including setting defaults if the attr is passed in
1910 // as None.
1911 for (int i = kFastPathExecuteInputStartIndex + op_def->input_arg_size();
1912 i < args_size; i += 2) {
1913 PyObject* py_attr_name = PyTuple_GET_ITEM(args, i);
1914 const tensorflow::StringPiece attr_name(TFE_GetPythonString(py_attr_name));
1915 PyObject* py_attr_value = PyTuple_GET_ITEM(args, i + 1);
1916
1917 // Not creating an index since most of the time there are not more than a
1918 // few attrs.
1919 // TODO(nareshmodi): Maybe include the index as part of the
1920 // OpRegistrationData.
1921 for (const auto& attr : op_def->attr()) {
1922 if (attr_name == attr.name()) {
1923 SetOpAttrWithDefaults(op_exec_info.ctx, op, attr, attr_name.data(),
1924 py_attr_value, &attr_list_sizes, status);
1925
1926 if (TF_GetCode(status) != TF_OK) {
1927 VLOG(1) << "Falling back to slow path for Op \"" << op_def->name()
1928 << "\" since we are unable to set the value for attr \""
1929 << attr.name() << "\" due to: " << TF_Message(status);
1930 RaiseFallbackException(TF_Message(status));
1931 return nullptr;
1932 }
1933
1934 break;
1935 }
1936 }
1937 }
1938
1939 TFE_OpSetDevice(op, op_exec_info.device_name, status);
1940 if (MaybeRaiseExceptionFromTFStatus(status, nullptr)) {
1941 return nullptr;
1942 }
1943
1944 // Flat attrs and inputs as required by the record_gradient call. The attrs
1945 // here only contain inferred attrs (non-inferred attrs are added directly
1946 // from the input args).
1947 // All items in flattened_attrs and flattened_inputs contain
1948 // Safe_PyObjectPtr - any time something steals a reference to this, it must
1949 // INCREF.
1950 // TODO(nareshmodi): figure out why PyList_New/PyList_Append don't work
1951 // directly.
1952 std::unique_ptr<std::vector<tensorflow::Safe_PyObjectPtr>> flattened_attrs =
1953 nullptr;
1954 std::unique_ptr<std::vector<tensorflow::Safe_PyObjectPtr>> flattened_inputs =
1955 nullptr;
1956
1957 // TODO(nareshmodi): Encapsulate callbacks information into a struct.
1958 if (op_exec_info.run_callbacks) {
1959 flattened_attrs.reset(new std::vector<tensorflow::Safe_PyObjectPtr>);
1960 flattened_inputs.reset(new std::vector<tensorflow::Safe_PyObjectPtr>);
1961 }
1962
1963 // Add inferred attrs and inputs.
1964 // The following code might set duplicate type attrs. This will result in
1965 // the CacheKey for the generated AttrBuilder possibly differing from
1966 // those where the type attrs are correctly set. Inconsistent CacheKeys
1967 // for ops means that there might be unnecessarily duplicated kernels.
1968 // TODO(nareshmodi): Fix this.
1969 for (int i = 0; i < op_def->input_arg_size(); i++) {
1970 const auto& input_arg = op_def->input_arg(i);
1971
1972 PyObject* input =
1973 PyTuple_GET_ITEM(args, kFastPathExecuteInputStartIndex + i);
1974 if (!input_arg.number_attr().empty()) {
1975 // The item is a homogeneous list.
1976 if (!RaiseIfNotPySequence(input, input_arg.number_attr())) return nullptr;
1977 Py_ssize_t len = PySequence_Fast_GET_SIZE(input);
1978
1979 TFE_OpSetAttrInt(op, input_arg.number_attr().data(), len);
1980 if (op_exec_info.run_callbacks) {
1981 flattened_attrs->emplace_back(
1982 GetPythonObjectFromString(input_arg.number_attr().data()));
1983 flattened_attrs->emplace_back(PyLong_FromLong(len));
1984 }
1985 attr_list_sizes[input_arg.number_attr()] = len;
1986
1987 if (len > 0) {
1988 // First item adds the type attr.
1989 if (!AddInputToOp(op_exec_info, PySequence_Fast_GET_ITEM(input, 0),
1990 &input_arg, flattened_attrs.get(),
1991 flattened_inputs.get(), op, status)) {
1992 return nullptr;
1993 }
1994
1995 for (Py_ssize_t j = 1; j < len; j++) {
1996 // Since the list is homogeneous, we don't need to re-add the attr.
1997 if (!AddInputToOp(op_exec_info, PySequence_Fast_GET_ITEM(input, j),
1998 nullptr /* input_arg */,
1999 nullptr /* flattened_attrs */,
2000 flattened_inputs.get(), op, status)) {
2001 return nullptr;
2002 }
2003 }
2004 }
2005 } else if (!input_arg.type_list_attr().empty()) {
2006 // The item is a heterogeneous list.
2007 if (!RaiseIfNotPySequence(input, input_arg.type_list_attr())) {
2008 return nullptr;
2009 }
2010 const string& attr_name = input_arg.type_list_attr();
2011 Py_ssize_t len = PySequence_Fast_GET_SIZE(input);
2012 tensorflow::gtl::InlinedVector<TF_DataType, 4> attr_value(len);
2013 PyObject* py_attr_value = nullptr;
2014 if (op_exec_info.run_callbacks) {
2015 py_attr_value = PyTuple_New(len);
2016 }
2017 for (Py_ssize_t j = 0; j < len; j++) {
2018 PyObject* py_input = PySequence_Fast_GET_ITEM(input, j);
2019 tensorflow::Safe_PyObjectPtr py_eager_tensor;
2020 if (!ConvertToTensor(op_exec_info, py_input, &py_eager_tensor,
2021 status)) {
2022 return nullptr;
2023 }
2024
2025 TFE_TensorHandle* input_handle =
2026 EagerTensor_Handle(py_eager_tensor.get());
2027
2028 attr_value[j] = TFE_TensorHandleDataType(input_handle);
2029
2030 TFE_OpAddInput(op, input_handle, status);
2031 if (MaybeRaiseExceptionFromTFStatus(status, nullptr)) {
2032 return nullptr;
2033 }
2034
2035 if (op_exec_info.run_callbacks) {
2036 flattened_inputs->emplace_back(std::move(py_eager_tensor));
2037
2038 PyTuple_SET_ITEM(py_attr_value, j, PyLong_FromLong(attr_value[j]));
2039 }
2040 }
2041 if (op_exec_info.run_callbacks) {
2042 flattened_attrs->emplace_back(
2043 GetPythonObjectFromString(attr_name.data()));
2044 flattened_attrs->emplace_back(py_attr_value);
2045 }
2046 TFE_OpSetAttrTypeList(op, attr_name.data(), attr_value.data(),
2047 attr_value.size());
2048 attr_list_sizes[attr_name] = len;
2049 } else {
2050 // The item is a single item.
2051 if (!AddInputToOp(op_exec_info, input, &input_arg, flattened_attrs.get(),
2052 flattened_inputs.get(), op, status)) {
2053 return nullptr;
2054 }
2055 }
2056 }
2057
2058 int num_retvals = 0;
2059 for (int i = 0; i < op_def->output_arg_size(); i++) {
2060 const auto& output_arg = op_def->output_arg(i);
2061 if (!output_arg.number_attr().empty()) {
2062 num_retvals += attr_list_sizes[output_arg.number_attr()];
2063 } else if (!output_arg.type_list_attr().empty()) {
2064 num_retvals += attr_list_sizes[output_arg.type_list_attr()];
2065 } else {
2066 num_retvals++;
2067 }
2068 }
2069
2070 tensorflow::gtl::InlinedVector<TFE_TensorHandle*, 2> retvals(num_retvals);
2071
2072 Py_BEGIN_ALLOW_THREADS;
2073 TFE_Execute(op, retvals.data(), &num_retvals, status);
2074 Py_END_ALLOW_THREADS;
2075
2076 if (TF_GetCode(status) != TF_OK) {
2077 // Augment the status with the op_name for easier debugging similar to
2078 // TFE_Py_Execute.
2079 TF_SetStatus(status, TF_GetCode(status),
2080 tensorflow::strings::StrCat(
2081 TF_Message(status),
2082 " [Op:", TFE_GetPythonString(op_exec_info.op_name), "]")
2083 .c_str());
2084
2085 MaybeRaiseExceptionFromTFStatus(status, nullptr);
2086 return nullptr;
2087 }
2088
2089 tensorflow::Safe_PyObjectPtr flat_result(PyList_New(num_retvals));
2090 for (int i = 0; i < num_retvals; ++i) {
2091 PyList_SET_ITEM(flat_result.get(), i, EagerTensorFromHandle(retvals[i]));
2092 }
2093
2094 if (!RunCallbacks(op_exec_info, args, *flattened_inputs, *flattened_attrs,
2095 flat_result.get())) {
2096 return nullptr;
2097 }
2098
2099 // Unflatten results.
2100 if (op_def->output_arg_size() == 0) {
2101 Py_RETURN_NONE;
2102 }
2103
2104 if (op_def->output_arg_size() == 1) {
2105 if (!op_def->output_arg(0).number_attr().empty() ||
2106 !op_def->output_arg(0).type_list_attr().empty()) {
2107 return flat_result.release();
2108 } else {
2109 auto* result = PyList_GET_ITEM(flat_result.get(), 0);
2110 Py_INCREF(result);
2111 return result;
2112 }
2113 }
2114
2115 // Correctly output the results that are made into a namedtuple.
2116 PyObject* result = PyList_New(op_def->output_arg_size());
2117 int flat_result_index = 0;
2118 for (int i = 0; i < op_def->output_arg_size(); i++) {
2119 if (!op_def->output_arg(i).number_attr().empty()) {
2120 int list_length = attr_list_sizes[op_def->output_arg(i).number_attr()];
2121 PyObject* inner_list = PyList_New(list_length);
2122 for (int j = 0; j < list_length; j++) {
2123 PyObject* obj = PyList_GET_ITEM(flat_result.get(), flat_result_index++);
2124 Py_INCREF(obj);
2125 PyList_SET_ITEM(inner_list, j, obj);
2126 }
2127 PyList_SET_ITEM(result, i, inner_list);
2128 } else if (!op_def->output_arg(i).type_list_attr().empty()) {
2129 int list_length = attr_list_sizes[op_def->output_arg(i).type_list_attr()];
2130 PyObject* inner_list = PyList_New(list_length);
2131 for (int j = 0; j < list_length; j++) {
2132 PyObject* obj = PyList_GET_ITEM(flat_result.get(), flat_result_index++);
2133 Py_INCREF(obj);
2134 PyList_SET_ITEM(inner_list, j, obj);
2135 }
2136 PyList_SET_ITEM(result, i, inner_list);
2137 } else {
2138 PyObject* obj = PyList_GET_ITEM(flat_result.get(), flat_result_index++);
2139 Py_INCREF(obj);
2140 PyList_SET_ITEM(result, i, obj);
2141 }
2142 }
2143 return result;
2144}
