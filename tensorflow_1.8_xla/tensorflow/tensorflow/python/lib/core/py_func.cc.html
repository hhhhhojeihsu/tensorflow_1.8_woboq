41static mutex mu(LINKER_INITIALIZED);
42static PyObject* py_trampoline GUARDED_BY(mu) = nullptr;
43
44// Returns the py_trampoline that is used to pass the control to the
45// python runtime.
46PyObject* GetPyTrampoline() {
47 mutex_lock l(mu);
48 return py_trampoline;
49}

51// A call to the registered python function.
52struct PyCall {
53 // Passed to python runtime to call the python function registered
54 // with this "token".
55 string token;
56
57 // The device on which Tensors are stored; only used for EagerPyFunc.
58 Device* device;
59
60 // True if and only if the op has been placed on a GPU.
61 bool gpu;
62
63 // True if the call is associated with an EagerPyFunc.
64 bool eager;
65
66 // Inputs and outputs of this function invocation.
67 std::vector<Tensor> ins;
68 std::vector<Tensor> out;
69};
70
71// Givens the 'call', prepares the token and inputs as a python tuple
72// that is appropriate for calling the trampoline.
73Status MakeArgTuple(const PyCall* call, PyObject** tuple) {
74 int64 n = call->ins.size();
75 PyObject* lst = PyList_New(n);
76 CHECK(lst);
77 for (int64 i = 0; i < n; ++i) {
78 PyObject* arg = nullptr;
79 const Tensor& t = call->ins[i];
80 if (call->eager) {
81 if (call->gpu) {
82 arg = EagerTensorFromHandle(
83 new TFE_TensorHandle(t, call->device, call->device));
84 } else {
85 // TFE_TensorHandle assumes that CPU is identified by `nullptr`.
86 arg = EagerTensorFromHandle(new TFE_TensorHandle(t, nullptr, nullptr));
87 }
88 if (arg == nullptr) {
89 return errors::Internal("Unable to procure EagerTensor from Tensor.");
90 }
91 } else {
92 Status s = ConvertTensorToNdarray(t, &arg);
93 if (!s.ok()) {
94 Py_DECREF(lst);
95 return s;
96 }
97 }
98 PyList_SetItem(lst, i, arg);
99 }
100 *tuple = Py_BuildValue("(sON)", call->token.c_str(),
101 call->gpu ? Py_True : Py_False, lst);
102 CHECK(*tuple);
103 return Status::OK();
104}

106// Returns the corresponding tf dtype in 'tf' for numpy data type
107// 'np'. Returns an error if the type is not supported by this
108// module.
109Status NumericNpDTypeToTfDType(const int np, DataType* tf) {
110 switch (np) {
111 case NPY_FLOAT16:
112 *tf = DT_HALF;
113 break;
114 case NPY_FLOAT32:
115 *tf = DT_FLOAT;
116 break;
117 case NPY_FLOAT64:
118 *tf = DT_DOUBLE;
119 break;
120 case NPY_INT32:
121 *tf = DT_INT32;
122 break;
123 case NPY_UINT8:
124 *tf = DT_UINT8;
125 break;
126 case NPY_INT8:
127 *tf = DT_INT8;
128 break;
129 case NPY_INT16:
130 *tf = DT_INT16;
131 break;
132 case NPY_INT64:
133 *tf = DT_INT64;
134 break;
135 case NPY_BOOL:
136 *tf = DT_BOOL;
137 break;
138 case NPY_COMPLEX64:
139 *tf = DT_COMPLEX64;
140 break;
141 case NPY_COMPLEX128:
142 *tf = DT_COMPLEX128;
143 break;
144 default:
145 return errors::Unimplemented("Unsupported numpy type ", np);
146 }
147 return Status::OK();
148}

150bool IsSingleNone(PyObject* obj) {
151 if (!PyArray_Check(obj)) {
152 return false;
153 }
154 PyArrayObject* array_obj = reinterpret_cast<PyArrayObject*>(obj);
155 if (PyArray_NDIM(array_obj) != 0 || PyArray_SIZE(array_obj) != 1) {
156 return false;
157 }
158 std::array<npy_intp, 0> indices;
159 char* item_ptr =
160 static_cast<char*>(PyArray_GetPtr(array_obj, indices.data()));
161 PyObject* item = PyArray_GETITEM(array_obj, item_ptr);
162 CHECK(item);
163 return item == Py_None;
164}

166// Retrieves a Tensor from `eager_tensor` and stores it in `output_tensor`.
167tensorflow::Status ExtractTensorFromEagerTensor(const PyObject* eager_tensor,
168 const Tensor** output_tensor) {
169 return EagerTensor_Handle(eager_tensor)->handle->Tensor(output_tensor);

172// Calls the registered py function through the trampoline.
173Status DoCallPyFunc(PyCall* call, bool* out_log_on_error) {
174 *out_log_on_error = true;
175 PyObject* trampoline = GetPyTrampoline();
176 if (trampoline == nullptr) {
177 return errors::InvalidArgument(
178 "Missing py trampoline. Most likely, it is a link error.");
179 }
180 // Prepare the argument.
181 PyObject* args = nullptr;
182 TF_RETURN_IF_ERROR(MakeArgTuple(call, &args));
183 CHECK(args);
184
185 // Invokes the trampoline.
186 PyObject* result = PyEval_CallObject(trampoline, args);
187 Py_DECREF(args);
188 if (result == nullptr) {
189 if (PyErr_Occurred()) {
190 if (PyErr_ExceptionMatches(PyExc_ValueError) ||
191 PyErr_ExceptionMatches(PyExc_TypeError)) {
192 return errors::InvalidArgument(PyExceptionFetch());
193 } else if (PyErr_ExceptionMatches(PyExc_StopIteration)) {
194 *out_log_on_error = false;
195 return errors::OutOfRange(PyExceptionFetch());
196 } else if (PyErr_ExceptionMatches(PyExc_MemoryError)) {
197 return errors::ResourceExhausted(PyExceptionFetch());
198 } else if (PyErr_ExceptionMatches(PyExc_NotImplementedError)) {
199 return errors::Unimplemented(PyExceptionFetch());
200 } else {
201 // TODO(ebrevdo): Check if exception is an OpError and use the
202 // OpError.error_code property to map it back in the Status.
203 return errors::Unknown(PyExceptionFetch());
204 }
205 } else {
206 return errors::Internal("Failed to run py callback ", call->token,
207 ": see error log.");
208 }
209 }
210
211 // Process the return values and convert them to TF Tensors.
212 Status s = Status::OK();
213 if (PyList_Check(result)) {
214 // `result` is a Python list; if this operation is an `EagerPyFunc`, then
215 // every item in the list must be an `EagerTensor`; otherwise, every element
216 // must be a NumPy array.
217 call->out.clear();
218 for (int i = 0; i < PyList_Size(result); ++i) {
219 Tensor t;
220 if (call->eager) {
221 const PyObject* item = PyList_GetItem(result, i);
222 if (EagerTensor_CheckExact(item)) {
223 const Tensor* tensor = nullptr;
224 s = ExtractTensorFromEagerTensor(item, &tensor);
225 if (s.ok()) t = *tensor;
226 } else {
227 s = errors::FailedPrecondition(
228 "Expected EagerTensor, found PyObject of type: ",
229 Py_TYPE(item)->tp_name);
230 }
231 } else {
232 s = ConvertNdarrayToTensor(PyList_GetItem(result, i), &t);
233 }
234
235 if (!s.ok()) {
236 break;
237 }
238 call->out.push_back(t);
239 }
240 } else if (EagerTensor_CheckExact(result) || result == Py_None) {
241 // result is an `EagerTensor` or `None`.
242 DCHECK(call->eager);
243 if (result != Py_None) {
244 const Tensor* t = nullptr;
245 s = ExtractTensorFromEagerTensor(result, &t);
246 if (s.ok()) call->out.push_back(*t);
247 }
248 } else if (PyArray_Check(result)) {
249 // `result` is a NumPy array.
250 DCHECK(!call->eager);
251 if (!IsSingleNone(result)) {
252 Tensor t;
253 s = ConvertNdarrayToTensor(result, &t);
254 if (s.ok()) {
255 call->out.push_back(t);
256 }
257 }
258 } else {
259 s = errors::Internal("Unexpected PyObject was returned: ",
260 Py_TYPE(result)->tp_name);
261 }
262 Py_DECREF(result);
263 return s;
264}

268// Outside anonymous namespace just to make the friend declaration in
269// tensorflow::Tensor apply.
270class NumpyTensorBuffer : public TensorBuffer {
271 public:
272 NumpyTensorBuffer(PyArrayObject* array, size_t len, void* data)
273 : array_(array), len_(len), data_(data) {}
274
275 ~NumpyTensorBuffer() override {
276 // Note: The session::run wrapper is responsible for freeing this while
277 // holding the GIL.
278 DelayedNumpyDecref(data_, len_, array_);
279 }
280
281 void* data() const override { return data_; }
282 size_t size() const override { return len_; }
283 TensorBuffer* root_buffer() override { return this; }
284 void FillAllocationDescription(AllocationDescription* proto) const override {
285 tensorflow::int64 rb = size();
286 proto->set_requested_bytes(rb);
287 proto->set_allocator_name(tensorflow::cpu_allocator()->Name());
288 }
289 Tensor MakeTensor(DataType dtype, const TensorShape& shape) {
290 CHECK_EQ(len_, shape.num_elements() * DataTypeSize(dtype));
291 return Tensor(dtype, shape, this);
292 }
293
294 // Prevents input forwarding from overwriting this buffer.
295 bool OwnsMemory() const override { return false; }
296
297 private:
298 PyArrayObject* array_;
299 size_t len_;
300 void* data_;
301};

303Status ConvertNdarrayToTensor(PyObject* obj, Tensor* ret) {
304 PyArrayObject* input = reinterpret_cast<PyArrayObject*>(obj);
305 DataType dtype = DT_INVALID;
306 TensorShape shape;
307 for (int i = 0; i < PyArray_NDIM(input); ++i) {
308 shape.AddDim(PyArray_SHAPE(input)[i]);
309 }
310 const int np_type = PyArray_TYPE(input);
311 switch (np_type) {
312 case NPY_OBJECT: {
313 dtype = DT_STRING;
314 Tensor t(dtype, shape);
315 auto tflat = t.flat<string>();
316 PyObject** input_data = reinterpret_cast<PyObject**>(PyArray_DATA(input));
317 for (int i = 0; i < tflat.dimension(0); ++i) {
318 char* el;
319 Py_ssize_t el_size;
320 if (PyBytes_AsStringAndSize(input_data[i], &el, &el_size) == -1) {
321#if PY_MAJOR_VERSION >= 3
322 el = PyUnicode_AsUTF8AndSize(input_data[i], &el_size);
323#else
324 el = nullptr;
325 if (PyUnicode_Check(input_data[i])) {
326 PyObject* unicode = PyUnicode_AsUTF8String(input_data[i]);
327 if (unicode) {
328 if (PyString_AsStringAndSize(unicode, &el, &el_size) == -1) {
329 Py_DECREF(unicode);
330 el = nullptr;
331 }
332 }
333 }
334#endif
335 if (!el) {
336 return errors::Unimplemented("Unsupported object type ",
337 input_data[i]->ob_type->tp_name);
338 }
339 }
340 tflat(i) = string(el, el_size);
341 }
342 *ret = t;
343 break;
344 }
345 case NPY_STRING: {
346 dtype = DT_STRING;
347 Tensor t(dtype, shape);
348 auto tflat = t.flat<string>();
349 char* input_data = PyArray_BYTES(input);
350 Py_ssize_t el_size = PyArray_ITEMSIZE(input);
351 for (int i = 0; i < tflat.dimension(0); ++i) {
352 tflat(i) = string(input_data + i * el_size, el_size);
353 }
354 *ret = t;
355 break;
356 }
357 default: {
358 TF_RETURN_IF_ERROR(NumericNpDTypeToTfDType(PyArray_TYPE(input), &dtype));
359 CHECK(DataTypeCanUseMemcpy(dtype));
360 if (reinterpret_cast<intptr_t>(PyArray_DATA(input)) %
361 EIGEN_MAX_ALIGN_BYTES !=
362 0) {
363 Tensor t(dtype, shape);
364 StringPiece p = t.tensor_data();
365 memcpy(const_cast<char*>(p.data()), PyArray_DATA(input), p.size());
366 *ret = t;
367 } else {
368 // Incref the array as the calling context will decref it when we
369 // return and we want to keep a handle to this memory.
370 Py_INCREF(input);
371 NumpyTensorBuffer* buf = new NumpyTensorBuffer(
372 input, shape.num_elements() * DataTypeSize(dtype),
373 PyArray_DATA(input));
374 *ret = buf->MakeTensor(dtype, shape);
375 buf->Unref();
376 }
377 }
378 }
379 return Status::OK();
380}

382// Creates a numpy array in 'ret' which either aliases the content of 't' or has
383// a copy.
384Status ConvertTensorToNdarray(const Tensor& t, PyObject** ret) {
385 int typenum = -1;
386 TF_RETURN_IF_ERROR(TF_DataType_to_PyArray_TYPE(
387 static_cast<TF_DataType>(t.dtype()), &typenum));
388 PyArray_Descr* descr = PyArray_DescrFromType(typenum);
389 CHECK(descr);
390 std::vector<npy_intp> dims;
391 dims.reserve(t.dims());
392 for (int i = 0; i < t.dims(); ++i) {
393 dims.push_back(t.dim_size(i));
394 }
395 Tensor* copy = new Tensor(t);
396 if (ArrayFromMemory(dims.size(), dims.data(),
397 const_cast<char*>(copy->tensor_data().data()), t.dtype(),
398 [copy]() { delete copy; }, ret)
399 .ok()) {
400 return Status::OK();
401 }
402 delete copy;
403
404 PyObject* obj = PyArray_Empty(dims.size(), dims.data(), descr, 0);
405 if (obj == nullptr) {
406 return errors::Internal("Failed to allocate np array: ",
407 t.shape().DebugString());
408 }
409 PyArrayObject* np_array = reinterpret_cast<PyArrayObject*>(obj);
410 if (typenum == NPY_OBJECT) {
411 CHECK_EQ(DT_STRING, t.dtype());
412 auto tflat = t.flat<string>();
413 PyObject** out = reinterpret_cast<PyObject**>(PyArray_DATA(np_array));
414 for (int i = 0; i < tflat.dimension(0); ++i) {
415 const string& el = tflat(i);
416 out[i] = PyBytes_FromStringAndSize(el.data(), el.size());
417 if (out[i] == nullptr) {
418 for (int j = 0; j < i; ++j) {
419 Py_DECREF(out[j]);
420 }
421 Py_DECREF(obj);
422 return errors::Internal("Failed to allocate a copy of string ", i);
423 }
424 }
425 } else {
426 CHECK(DataTypeCanUseMemcpy(t.dtype()));
427 StringPiece p = t.tensor_data();
428 memcpy(PyArray_DATA(np_array), p.data(), p.size());
429 }
430 *ret = PyArray_Return(np_array);
431 return Status::OK();
432}

434void InitializePyTrampoline(PyObject* trampoline) {
435 mutex_lock l(mu);
436 if (py_trampoline == nullptr) {
437 py_trampoline = trampoline;
438 Py_INCREF(py_trampoline);
439 } else {
440 LOG(WARNING) << "InitializeCallback should only be called once";
441 }
442}

444class PyFuncOp : public OpKernel {
445 public:
446 explicit PyFuncOp(OpKernelConstruction* ctx) : OpKernel(ctx) {
447 OP_REQUIRES_OK(ctx, ctx->GetAttr("token", &token_));
448 eager_ = type_string() == "EagerPyFunc";
449 gpu_ = ctx->device_type().type_string() == DEVICE_GPU;
450 }
451
452 void Compute(OpKernelContext* ctx) override {
453 PyCall call;
454 call.token = token_;
455 call.gpu = gpu_;
456 call.eager = eager_;
457 if (call.eager) {
458 // Eager's C API uses `Device`, whereas `OpKernelContext` stores a
459 // `DeviceBase`; attempt to downcast.
460 call.device = dynamic_cast<Device*>(ctx->device());
461 if (call.device == nullptr) {
462 ctx->CtxFailureWithWarning(
463 errors::Internal("Unrecognized device class"));
464 }
465 }
466
467 for (int i = 0; i < ctx->num_inputs(); ++i) {
468 call.ins.push_back(ctx->input(i));
469 }
470
471 PyGILState_STATE py_threadstate;
472 py_threadstate = PyGILState_Ensure();
473 bool log_on_error;
474 Status s = DoCallPyFunc(&call, &log_on_error);
475 // Sometimes py_funcs can be called without a session and leak memory. This
476 // ensures we clear the decref cache so this doesn't happen.
477 ClearDecrefCache();
478 PyGILState_Release(py_threadstate);
479
480 // Ensures that GIL is released even when !s.ok().
481 if (!s.ok()) {
482 if (log_on_error) {
483 ctx->CtxFailureWithWarning(s);
484 } else {
485 ctx->CtxFailure(s);
486 }
487 return;
488 }
489
490 OP_REQUIRES(ctx, static_cast<int32>(call.out.size()) == ctx->num_outputs(),
491 errors::InvalidArgument(token_, " returns ", call.out.size(),
492 " values, but expects to see ",
493 ctx->num_outputs(), " values."));
494 for (size_t i = 0; i < call.out.size(); ++i) {
495 const auto& t = call.out[i];
496 OP_REQUIRES(
497 ctx, t.dtype() == output_type(i),
498 errors::InvalidArgument(i, "-th value returned by ", token_, " is ",
499 DataTypeString(t.dtype()), ", but expects ",
500 DataTypeString(output_type(i))));
501 ctx->set_output(i, t);
502 }
503 }
504
505 private:
506 string token_;
507
508 // True if and only if this op has been placed on a GPU.
509 bool gpu_;
510
511 // True if and only if this op should execute the python function eagerly,
512 // i.e., if and only if the eager attribute is set.
513 bool eager_;
514
515 TF_DISALLOW_COPY_AND_ASSIGN(PyFuncOp);
516};
