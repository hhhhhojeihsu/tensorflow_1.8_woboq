84namespace {
85enum KernelLabel { DEFAULT_LABEL, OVERLOAD_1_LABEL, OVERLOAD_2_LABEL };
86} // namespace
87
88template <KernelLabel KL>
89class KernelLabelOp : public OpKernel {
90 public:
91 using OpKernel::OpKernel;
92
93 void Compute(OpKernelContext* ctx) override {
94 Tensor* output;
95 OP_REQUIRES_OK(ctx,
96 ctx->allocate_output("result", TensorShape({}), &output));
97 switch (KL) {
98 case DEFAULT_LABEL:
99 output->scalar<string>()() = "My label is: default";
100 break;
101 case OVERLOAD_1_LABEL:
102 output->scalar<string>()() = "My label is: overload_1";
103 break;
104 case OVERLOAD_2_LABEL:
105 output->scalar<string>()() = "My label is: overload_2";
106 break;
107 }
108 }
109};
110
111REGISTER_KERNEL_BUILDER(Name("KernelLabel").Device(DEVICE_CPU),
112 KernelLabelOp<DEFAULT_LABEL>);
113REGISTER_KERNEL_BUILDER(Name("KernelLabel")
114 .Device(DEVICE_CPU)
115 .Label("overload_1"),
116 KernelLabelOp<OVERLOAD_1_LABEL>);
117REGISTER_KERNEL_BUILDER(Name("KernelLabel")
118 .Device(DEVICE_CPU)
119 .Label("overload_2"),
120 KernelLabelOp<OVERLOAD_2_LABEL>);
121
122// All "KernelLabelRequired" kernels have labels
123REGISTER_KERNEL_BUILDER(
124 Name("KernelLabelRequired").Device(DEVICE_CPU).Label("overload_1"),
125 KernelLabelOp<OVERLOAD_1_LABEL>);
126REGISTER_KERNEL_BUILDER(
127 Name("KernelLabelRequired").Device(DEVICE_CPU).Label("overload_2"),
128 KernelLabelOp<OVERLOAD_2_LABEL>);

130class GraphDefVersionOp : public OpKernel {
131 public:
132 explicit GraphDefVersionOp(OpKernelConstruction* ctx)
133 : OpKernel(ctx), graph_def_version_(ctx->graph_def_version()) {}
134
135 void Compute(OpKernelContext* ctx) override {
136 Tensor* output = nullptr;
137 OP_REQUIRES_OK(ctx, ctx->allocate_output(0, TensorShape({}), &output));
138 output->scalar<int>()() = graph_def_version_;
139 }
140
141 private:
142 const int graph_def_version_;
143};
144
145REGISTER_KERNEL_BUILDER(Name("GraphDefVersion").Device(DEVICE_CPU),
146 GraphDefVersionOp);

168class ResourceCreateOp : public OpKernel {
169 public:
170 explicit ResourceCreateOp(OpKernelConstruction* c) : OpKernel(c) {}
171
172 void Compute(OpKernelContext* c) override {
173 OP_REQUIRES_OK(c,
174 CreateResource(c, HandleFromInput(c, 0), new StubResource));
175 }
176};
177
178REGISTER_KERNEL_BUILDER(Name("ResourceCreateOp").Device(DEVICE_CPU),
179 ResourceCreateOp);
