24// See documentation in ../ops/dataset_ops.cc for a high-level
25// description of the following op.
26
27class UniqueDatasetOp : public UnaryDatasetOpKernel {
28 public:
29 explicit UniqueDatasetOp(OpKernelConstruction* ctx)
30 : UnaryDatasetOpKernel(ctx) {}
31
32 void MakeDataset(OpKernelContext* ctx, DatasetBase* input,
33 DatasetBase** output) override {
34 OP_REQUIRES(ctx, input->output_dtypes().size() == 1,
35 errors::InvalidArgument("UniqueDataset only supports "
36 "inputs with a single component."));
37
38 DataType input_dtype = input->output_dtypes()[0];
39 OP_REQUIRES(ctx,
40 input_dtype == DT_INT32 || input_dtype == DT_INT64 ||
41 input_dtype == DT_STRING,
42 errors::InvalidArgument(
43 "UniqueDataset only supports inputs with a single "
44 "`tf.int32`, `tf.int64`, or `tf.string` component."));
45
46 *output = new Dataset(ctx, input);
47 }
48
49 private:
50 class Dataset : public GraphDatasetBase {
51 public:
52 Dataset(OpKernelContext* ctx, const DatasetBase* input)
53 : GraphDatasetBase(ctx), input_(input) {
54 input_->Ref();
55 }
56
57 ~Dataset() override { input_->Unref(); }
58
59 std::unique_ptr<IteratorBase> MakeIterator(
60 const string& prefix) const override {
61 return std::unique_ptr<IteratorBase>(
62 new Iterator({this, strings::StrCat(prefix, "::Unique")}));
63 }
64
65 const DataTypeVector& output_dtypes() const override {
66 return input_->output_dtypes();
67 }
68
69 const std::vector<PartialTensorShape>& output_shapes() const override {
70 return input_->output_shapes();
71 }
72
73 string DebugString() override {
74 return strings::StrCat("UniqueDatasetOp::Dataset");
75 }
76
77 protected:
78 Status AsGraphDefInternal(OpKernelContext* ctx, DatasetGraphDefBuilder* b,
79 Node** output) const override {
80 Node* input_graph_node = nullptr;
81 TF_RETURN_IF_ERROR(b->AddParentDataset(ctx, input_, &input_graph_node));
82 TF_RETURN_IF_ERROR(b->AddDataset(this, {input_graph_node}, output));
83 return Status::OK();
84 }
85
86 private:
87 class Iterator : public DatasetIterator<Dataset> {
88 public:
89 explicit Iterator(const typename Iterator::Params& params)
90 : DatasetIterator<Dataset>(params),
91 input_impl_(params.dataset->input_->MakeIterator(params.prefix)) {}
92
93 Status GetNextInternal(IteratorContext* ctx,
94 std::vector<Tensor>* out_tensors,
95 bool* end_of_sequence) override {
96 mutex_lock l(mu_);
97 bool saw_new_value;
98 do {
99 saw_new_value = false;
100 out_tensors->clear();
101 TF_RETURN_IF_ERROR(
102 input_impl_->GetNext(ctx, out_tensors, end_of_sequence));
103 if (*end_of_sequence) {
104 break;
105 }
106 DCHECK_EQ(1, out_tensors->size());
107 saw_new_value = unique_elements_.insert((*out_tensors)[0]).second;
108 } while (!saw_new_value);
109 return Status::OK();
110 }
111
112 protected:
113 Status SaveInternal(IteratorStateWriter* writer) override {
114 mutex_lock l(mu_);
115 if (input_impl_) {
116 TF_RETURN_IF_ERROR(SaveParent(writer, input_impl_));
117 } else {
118 TF_RETURN_IF_ERROR(
119 writer->WriteScalar(full_name("input_impl_empty"), ""));
120 }
121 TF_RETURN_IF_ERROR(writer->WriteScalar(
122 full_name("unique_elements_size"), unique_elements_.size()));
123 size_t i = 0;
124 for (const Tensor& t : unique_elements_) {
125 TF_RETURN_IF_ERROR(writer->WriteTensor(
126 full_name(strings::StrCat("unique_elements[", i++, "]")), t));
127 }
128 return Status::OK();
129 }
130
131 Status RestoreInternal(IteratorContext* ctx,
132 IteratorStateReader* reader) override {
133 mutex_lock l(mu_);
134 if (!reader->Contains(full_name("input_impl_empty"))) {
135 TF_RETURN_IF_ERROR(RestoreParent(ctx, reader, input_impl_));
136 } else {
137 input_impl_.reset();
138 }
139 int64 num_unique_elements;
140 unique_elements_.clear();
141 TF_RETURN_IF_ERROR(reader->ReadScalar(full_name("unique_elements_size"),
142 &num_unique_elements));
143 for (int64 i = 0; i < num_unique_elements; ++i) {
144 Tensor unique_element;
145 TF_RETURN_IF_ERROR(reader->ReadTensor(
146 full_name(strings::StrCat("unique_elements[", i, "]")),
147 &unique_element));
148 auto insert_result = unique_elements_.insert(unique_element);
149 if (!insert_result.second) {
150 return errors::InvalidArgument(
151 "Checkpoint contained two unique elements with the same "
152 "value.");
153 }
154 }
155 return Status::OK();
156 }
157
158 private:
159 struct TensorHash {
160 size_t operator()(const Tensor& t) const {
161 if (t.dtype() == DT_INT32 || t.dtype() == DT_INT64) {
162 return Hash64(t.tensor_data().data(), t.tensor_data().size());
163 } else {
164 DCHECK_EQ(DT_STRING, t.dtype());
165 auto flat_t = t.flat<string>();
166 uint64 hash = 0;
167 for (int64 i = 0; i < t.NumElements(); ++i) {
168 hash = Hash64Combine(hash, Hash64(flat_t(i)));
169 }
170 return static_cast<size_t>(hash);
171 }
172 }
173 };
174
175 struct TensorKeyEqual {
176 bool operator()(const Tensor& lhs, const Tensor& rhs) const {
177 if (lhs.shape() != rhs.shape() || lhs.dtype() != rhs.dtype()) {
178 return false;
179 }
180 switch (lhs.dtype()) {
181#define HANDLE_TYPE(T) \
182 case T: \
183 do { \
184 auto lhs_flat = lhs.flat<EnumToDataType<T>::Type>(); \
185 auto rhs_flat = rhs.flat<EnumToDataType<T>::Type>(); \
186 for (int64 i = 0; i < lhs.NumElements(); ++i) { \
187 if (lhs_flat(i) != rhs_flat(i)) { \
188 return false; \
189 } \
190 } \
191 return true; \
192 } while (0)
193
194 HANDLE_TYPE(DT_INT32);
195 HANDLE_TYPE(DT_INT64);
196 HANDLE_TYPE(DT_STRING);
197 default:
198 LOG(FATAL) << "UniqueDataset unhandled data type: "
199 << DataTypeString(lhs.dtype());
200 }
201 }
202 };
203
204 mutex mu_;
205 std::unique_ptr<IteratorBase> input_impl_ GUARDED_BY(mu_);
206 std::unordered_set<Tensor, TensorHash, TensorKeyEqual> unique_elements_
207 GUARDED_BY(mu_);
208 };
209
210 const DatasetBase* const input_;
211 };
212};

214REGISTER_KERNEL_BUILDER(Name("UniqueDataset").Device(DEVICE_CPU),
215 UniqueDatasetOp);
