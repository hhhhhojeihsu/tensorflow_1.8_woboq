30class SummaryFileWriter : public SummaryWriterInterface {
31 public:
32 SummaryFileWriter(int max_queue, int flush_millis, Env* env)
33 : SummaryWriterInterface(),
34 is_initialized_(false),
35 max_queue_(max_queue),
36 flush_millis_(flush_millis),
37 env_(env) {}
38
39 Status Initialize(const string& logdir, const string& filename_suffix) {
40 const Status is_dir = env_->IsDirectory(logdir);
41 if (!is_dir.ok()) {
42 if (is_dir.code() != tensorflow::error::NOT_FOUND) {
43 return is_dir;
44 }
45 TF_RETURN_IF_ERROR(env_->RecursivelyCreateDir(logdir));
46 }
47 mutex_lock ml(mu_);
48 events_writer_ =
49 tensorflow::MakeUnique<EventsWriter>(io::JoinPath(logdir, "events"));
50 TF_RETURN_WITH_CONTEXT_IF_ERROR(
51 events_writer_->InitWithSuffix(filename_suffix),
52 "Could not initialize events writer.");
53 last_flush_ = env_->NowMicros();
54 is_initialized_ = true;
55 return Status::OK();
56 }
57
58 Status Flush() override {
59 mutex_lock ml(mu_);
60 if (!is_initialized_) {
61 return errors::FailedPrecondition("Class was not properly initialized.");
62 }
63 return InternalFlush();
64 }
65
66 ~SummaryFileWriter() override {
67 (void)Flush(); // Ignore errors.
68 }
69
70 Status WriteTensor(int64 global_step, Tensor t, const string& tag,
71 const string& serialized_metadata) override {
72 std::unique_ptr<Event> e{new Event};
73 e->set_step(global_step);
74 e->set_wall_time(GetWallTime());
75 Summary::Value* v = e->mutable_summary()->add_value();
76 t.AsProtoTensorContent(v->mutable_tensor());
77 v->set_tag(tag);
78 if (!serialized_metadata.empty()) {
79 v->mutable_metadata()->ParseFromString(serialized_metadata);
80 }
81 return WriteEvent(std::move(e));
82 }
83
84 Status WriteScalar(int64 global_step, Tensor t, const string& tag) override {
85 std::unique_ptr<Event> e{new Event};
86 e->set_step(global_step);
87 e->set_wall_time(GetWallTime());
88 TF_RETURN_IF_ERROR(
89 AddTensorAsScalarToSummary(t, tag, e->mutable_summary()));
90 return WriteEvent(std::move(e));
91 }
92
93 Status WriteHistogram(int64 global_step, Tensor t,
94 const string& tag) override {
95 std::unique_ptr<Event> e{new Event};
96 e->set_step(global_step);
97 e->set_wall_time(GetWallTime());
98 TF_RETURN_IF_ERROR(
99 AddTensorAsHistogramToSummary(t, tag, e->mutable_summary()));
100 return WriteEvent(std::move(e));
101 }
102
103 Status WriteImage(int64 global_step, Tensor t, const string& tag,
104 int max_images, Tensor bad_color) override {
105 std::unique_ptr<Event> e{new Event};
106 e->set_step(global_step);
107 e->set_wall_time(GetWallTime());
108 TF_RETURN_IF_ERROR(AddTensorAsImageToSummary(t, tag, max_images, bad_color,
109 e->mutable_summary()));
110 return WriteEvent(std::move(e));
111 }
112
113 Status WriteAudio(int64 global_step, Tensor t, const string& tag,
114 int max_outputs, float sample_rate) override {
115 std::unique_ptr<Event> e{new Event};
116 e->set_step(global_step);
117 e->set_wall_time(GetWallTime());
118 TF_RETURN_IF_ERROR(AddTensorAsAudioToSummary(
119 t, tag, max_outputs, sample_rate, e->mutable_summary()));
120 return WriteEvent(std::move(e));
121 }
122
123 Status WriteGraph(int64 global_step,
124 std::unique_ptr<GraphDef> graph) override {
125 std::unique_ptr<Event> e{new Event};
126 e->set_step(global_step);
127 e->set_wall_time(GetWallTime());
128 graph->SerializeToString(e->mutable_graph_def());
129 return WriteEvent(std::move(e));
130 }
131
132 Status WriteEvent(std::unique_ptr<Event> event) override {
133 mutex_lock ml(mu_);
134 queue_.emplace_back(std::move(event));
135 if (queue_.size() >= max_queue_ ||
136 env_->NowMicros() - last_flush_ > 1000 * flush_millis_) {
137 return InternalFlush();
138 }
139 return Status::OK();
140 }
141
142 string DebugString() override { return "SummaryFileWriter"; }
143
144 private:
145 double GetWallTime() {
146 return static_cast<double>(env_->NowMicros()) / 1.0e6;
147 }
148
149 Status InternalFlush() EXCLUSIVE_LOCKS_REQUIRED(mu_) {
150 for (const std::unique_ptr<Event>& e : queue_) {
151 events_writer_->WriteEvent(*e);
152 }
153 queue_.clear();
154 TF_RETURN_WITH_CONTEXT_IF_ERROR(events_writer_->Flush(),
155 "Could not flush events file.");
156 last_flush_ = env_->NowMicros();
157 return Status::OK();
158 }
159
160 bool is_initialized_;
161 const int max_queue_;
162 const int flush_millis_;
163 uint64 last_flush_;
164 Env* env_;
165 mutex mu_;
166 std::vector<std::unique_ptr<Event>> queue_ GUARDED_BY(mu_);
167 // A pointer to allow deferred construction.
168 std::unique_ptr<EventsWriter> events_writer_ GUARDED_BY(mu_);
169 std::vector<std::pair<string, SummaryMetadata>> registered_summaries_
170 GUARDED_BY(mu_);
171};
172
173} // namespace
174
175Status CreateSummaryFileWriter(int max_queue, int flush_millis,
176 const string& logdir,
177 const string& filename_suffix, Env* env,
178 SummaryWriterInterface** result) {
179 SummaryFileWriter* w = new SummaryFileWriter(max_queue, flush_millis, env);
180 const Status s = w->Initialize(logdir, filename_suffix);
181 if (!s.ok()) {
182 w->Unref();
183 *result = nullptr;
184 return s;
185 }
186 *result = w;
187 return Status::OK();
188}
