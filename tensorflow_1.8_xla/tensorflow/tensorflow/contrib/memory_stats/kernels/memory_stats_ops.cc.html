15#include "tensorflow/core/framework/op_kernel.h"
16
17namespace tensorflow {
18
19// Base class of ops that collects statistics of the allocator of a device.
20class MemoryStatsOp : public OpKernel {
21 public:
22 explicit MemoryStatsOp(OpKernelConstruction* context) : OpKernel(context) {}
23
24 void Compute(OpKernelContext* context) override {
25 Allocator* allocator =
26 context->device()->GetAllocator(AllocatorAttributes());
27 AllocatorStats allocator_stats;
28 allocator->GetStats(&allocator_stats);
29
30 Tensor* output_tensor = nullptr;
31 OP_REQUIRES_OK(
32 context, context->allocate_output(0, TensorShape({}), &output_tensor));
33 output_tensor->scalar<int64>()() = ExtractAllocatorStats(allocator_stats);
34 }
35
36 protected:
37 // Extracts a certain field (determined by subclasses) from an allocator
38 // stats.
39 virtual int64 ExtractAllocatorStats(
40 const AllocatorStats& allocator_stats) const = 0;
41};
42
43// Op that measures current memory in bytes.
44class BytesInUseOp : public MemoryStatsOp {
45 public:
46 explicit BytesInUseOp(OpKernelConstruction* context)
47 : MemoryStatsOp(context) {}
48
49 private:
50 int64 ExtractAllocatorStats(
51 const AllocatorStats& allocator_stats) const override {
52 return allocator_stats.bytes_in_use;
53 }
54};
55
56// Register this op on GPU only, see comment for MaxBytesInUse for reason
57REGISTER_KERNEL_BUILDER(Name("BytesInUse").Device(DEVICE_GPU).HostMemory("out"),
58 BytesInUseOp);
59
60#ifdef TENSORFLOW_USE_SYCL
61REGISTER_KERNEL_BUILDER(
62 Name("BytesInUse").Device(DEVICE_SYCL).HostMemory("out"), BytesInUseOp);
63#endif // TENSORFLOW_USE_SYCL
64
65// Op that measures the total memory (in bytes) of a device.
66class BytesLimitOp : public MemoryStatsOp {
67 public:
68 explicit BytesLimitOp(OpKernelConstruction* context)
69 : MemoryStatsOp(context) {}
70
71 private:
72 int64 ExtractAllocatorStats(
73 const AllocatorStats& allocator_stats) const override {
74 return allocator_stats.bytes_limit;
75 }
76};
77
78REGISTER_KERNEL_BUILDER(Name("BytesLimit").Device(DEVICE_CPU), BytesLimitOp);
79REGISTER_KERNEL_BUILDER(Name("BytesLimit").Device(DEVICE_GPU).HostMemory("out"),
80 BytesLimitOp);
81
82#ifdef TENSORFLOW_USE_SYCL
83REGISTER_KERNEL_BUILDER(
84 Name("BytesLimit").Device(DEVICE_SYCL).HostMemory("out"), BytesLimitOp);
85#endif // TENSORFLOW_USE_SYCL
86
87// Op that measures the peak memory in bytes.
88class MaxBytesInUseOp : public MemoryStatsOp {
89 public:
90 explicit MaxBytesInUseOp(OpKernelConstruction* context)
91 : MemoryStatsOp(context) {}
92
93 private:
94 int64 ExtractAllocatorStats(
95 const AllocatorStats& allocator_stats) const override {
96 return allocator_stats.max_bytes_in_use;
97 }
98};
99
100// MallocExtension_GetAllocatedSize doesn't return the allocated size reliably
101// for CPU allocators, so we register this op on GPU only.
102REGISTER_KERNEL_BUILDER(
103 Name("MaxBytesInUse").Device(DEVICE_GPU).HostMemory("out"),
104 MaxBytesInUseOp);
105
106#ifdef TENSORFLOW_USE_SYCL
107REGISTER_KERNEL_BUILDER(
108 Name("MaxBytesInUse").Device(DEVICE_SYCL).HostMemory("out"),
109 MaxBytesInUseOp);
110#endif // TENSORFLOW_USE_SYCL
111
112} // namespace tensorflow
