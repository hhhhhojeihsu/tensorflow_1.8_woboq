41// Creates a stats variable.
42class CreateFertileStatsVariableOp : public OpKernel {
43 public:
44 explicit CreateFertileStatsVariableOp(OpKernelConstruction* context)
45 : OpKernel(context) {
46 string serialized_params;
47 OP_REQUIRES_OK(context, context->GetAttr("params", &serialized_params));
48 ParseProtoUnlimited(&param_proto_, serialized_params);
49 }
50
51 void Compute(OpKernelContext* context) override {
52 const Tensor* stats_config_t;
53 OP_REQUIRES_OK(context, context->input("stats_config", &stats_config_t));
54 OP_REQUIRES(context, TensorShapeUtils::IsScalar(stats_config_t->shape()),
55 errors::InvalidArgument("Stats config must be a scalar."));
56 auto* result = new FertileStatsResource(param_proto_);
57 FertileStats stats;
58 if (!ParseProtoUnlimited(&stats, stats_config_t->scalar<string>()())) {
59 result->Unref();
60 OP_REQUIRES(context, false,
61 errors::InvalidArgument("Unable to parse stats config."));
62 }
63
64 result->ExtractFromProto(stats);
65 result->MaybeInitialize();
66
67 // Only create one, if one does not exist already. Report status for all
68 // other exceptions.
69 auto status = CreateResource(context, HandleFromInput(context, 0), result);
70 if (!status.ok() && status.code() != tensorflow::error::ALREADY_EXISTS) {
71 OP_REQUIRES(context, false, status);
72 }
73 }
74
75 private:
76 TensorForestParams param_proto_;
77};

144// Try to update a leaf's stats by acquiring its lock. If it can't be
145// acquired, put it in a waiting queue to come back to later and try the next
146// one. Once all leaf_ids have been visited, cycle through the waiting ids
147// until they're gone.
148void UpdateStats(FertileStatsResource* fertile_stats_resource,
149 const std::unique_ptr<TensorDataSet>& data,
150 const TensorInputTarget& target, int num_targets,
151 const Tensor& leaf_ids_tensor,
152 std::unordered_map<int32, std::unique_ptr<mutex>>* locks,
153 mutex* set_lock, int32 start, int32 end,
154 std::unordered_set<int32>* ready_to_split) {
155 const auto leaf_ids = leaf_ids_tensor.unaligned_flat<int32>();
156
157 // Stores leaf_id, leaf_depth, example_id for examples that are waiting
158 // on another to finish.
159 std::queue<std::tuple<int32, int32>> waiting;
160
161 int32 i = start;
162 while (i < end || !waiting.empty()) {
163 int32 leaf_id;
164 int32 example_id;
165 bool was_waiting = false;
166 if (i >= end) {
167 std::tie(leaf_id, example_id) = waiting.front();
168 waiting.pop();
169 was_waiting = true;
170 } else {
171 leaf_id = leaf_ids(i);
172 example_id = i;
173 ++i;
174 }
175 const std::unique_ptr<mutex>& leaf_lock = (*locks)[leaf_id];
176 if (was_waiting) {
177 leaf_lock->lock();
178 } else {
179 if (!leaf_lock->try_lock()) {
180 waiting.emplace(leaf_id, example_id);
181 continue;
182 }
183 }
184
185 bool is_finished;
186 fertile_stats_resource->AddExampleToStatsAndInitialize(
187 data, &target, {example_id}, leaf_id, &is_finished);
188 leaf_lock->unlock();
189 if (is_finished) {
190 set_lock->lock();
191 ready_to_split->insert(leaf_id);
192 set_lock->unlock();
193 }
194 }
195}

197// Update leaves from start through end in the leaf_examples iterator.
198void UpdateStatsCollated(
199 FertileStatsResource* fertile_stats_resource,
200 DecisionTreeResource* tree_resource,
201 const std::unique_ptr<TensorDataSet>& data, const TensorInputTarget& target,
202 int num_targets,
203 const std::unordered_map<int32, std::vector<int>>& leaf_examples,
204 mutex* set_lock, int32 start, int32 end,
205 std::unordered_set<int32>* ready_to_split) {
206 auto it = leaf_examples.begin();
207 std::advance(it, start);
208 auto end_it = leaf_examples.begin();
209 std::advance(end_it, end);
210 while (it != end_it) {
211 int32 leaf_id = it->first;
212 bool is_finished;
213 fertile_stats_resource->AddExampleToStatsAndInitialize(
214 data, &target, it->second, leaf_id, &is_finished);
215 if (is_finished) {
216 set_lock->lock();
217 ready_to_split->insert(leaf_id);
218 set_lock->unlock();
219 }
220 ++it;
221 }
222}

224// Op for traversing the tree with each example, accumulating statistics, and
225// outputting node ids that are ready to split.
226class ProcessInputOp : public OpKernel {
227 public:
228 explicit ProcessInputOp(OpKernelConstruction* context) : OpKernel(context) {
229 string serialized_params;
230 OP_REQUIRES_OK(context, context->GetAttr("params", &serialized_params));
231 ParseProtoUnlimited(&param_proto_, serialized_params);
232
233 OP_REQUIRES_OK(context, context->GetAttr("random_seed", &random_seed_));
234
235 string serialized_proto;
236 OP_REQUIRES_OK(context, context->GetAttr("input_spec", &serialized_proto));
237 input_spec_.ParseFromString(serialized_proto);
238 }
239
240 void Compute(OpKernelContext* context) override {
241 const Tensor& input_data = context->input(2);
242 const Tensor& sparse_input_indices = context->input(3);
243 const Tensor& sparse_input_values = context->input(4);
244 const Tensor& sparse_input_shape = context->input(5);
245 const Tensor& input_labels = context->input(6);
246 const Tensor& input_weights = context->input(7);
247 const Tensor& leaf_ids_tensor = context->input(8);
248
249 std::unique_ptr<TensorDataSet> data_set(new TensorDataSet(input_spec_, 0));
250 data_set->set_input_tensors(input_data, sparse_input_indices,
251 sparse_input_values, sparse_input_shape);
252
253 FertileStatsResource* fertile_stats_resource;
254 OP_REQUIRES_OK(context, LookupResource(context, HandleFromInput(context, 1),
255 &fertile_stats_resource));
256 DecisionTreeResource* tree_resource;
257 OP_REQUIRES_OK(context, LookupResource(context, HandleFromInput(context, 0),
258 &tree_resource));
259 mutex_lock l1(*fertile_stats_resource->get_mutex());
260 mutex_lock l2(*tree_resource->get_mutex());
261
262 core::ScopedUnref unref_stats(fertile_stats_resource);
263 core::ScopedUnref unref_tree(tree_resource);
264
265 const int32 num_data = data_set->NumItems();
266 auto worker_threads = context->device()->tensorflow_cpu_worker_threads();
267 int num_threads = worker_threads->num_threads;
268
269 const auto leaf_ids = leaf_ids_tensor.unaligned_flat<int32>();
270
271 // Create one mutex per leaf. We need to protect access to leaf pointers,
272 // so instead of grouping examples by leaf, we spread examples out among
273 // threads to provide uniform work for each of them and protect access
274 // with mutexes.
275 std::unordered_map<int, std::unique_ptr<mutex>> locks;
276 std::unordered_map<int32, std::vector<int>> leaf_examples;
277 if (param_proto_.collate_examples()) {
278 for (int i = 0; i < num_data; ++i) {
279 leaf_examples[leaf_ids(i)].push_back(i);
280 }
281 } else {
282 for (int i = 0; i < num_data; ++i) {
283 const int32 id = leaf_ids(i);
284 if (FindOrNull(locks, id) == nullptr) {
285 // TODO(gilberth): Consider using a memory pool for these.
286 locks[id] = std::unique_ptr<mutex>(new mutex);
287 }
288 }
289 }
290
291 const int32 num_leaves = leaf_examples.size();
292 const int32 label_dim =
293 input_labels.shape().dims() <= 1
294 ? 0
295 : static_cast<int>(input_labels.shape().dim_size(1));
296 const int32 num_targets =
297 param_proto_.is_regression() ? (std::max(1, label_dim)) : 1;
298
299 // Ids of leaves that can split.
300 std::unordered_set<int32> ready_to_split;
301 mutex set_lock;
302
303 TensorInputTarget target(input_labels, input_weights, num_targets);
304
305 // TODO(gilberth): This is a rough approximation based on measurements
306 // from a digits run on local desktop. Heuristics might be necessary
307 // if it really matters that much.
308 const int64 costPerUpdate = 1000;
309 auto update = [this, &target, &leaf_ids_tensor, &num_targets, &data_set,
310 fertile_stats_resource, &locks, &set_lock, &ready_to_split,
311 num_data](int64 start, int64 end) {
312 CHECK(start <= end);
313 CHECK(end <= num_data);
314 UpdateStats(fertile_stats_resource, data_set, target, num_targets,
315 leaf_ids_tensor, &locks, &set_lock, static_cast<int32>(start),
316 static_cast<int32>(end), &ready_to_split);
317 };
318
319 auto update_collated = [this, &target, &num_targets, fertile_stats_resource,
320 tree_resource, &leaf_examples, &set_lock,
321 &ready_to_split, &data_set,
322 num_leaves](int64 start, int64 end) {
323 CHECK(start <= end);
324 CHECK(end <= num_leaves);
325 UpdateStatsCollated(fertile_stats_resource, tree_resource, data_set,
326 target, num_targets, leaf_examples, &set_lock,
327 static_cast<int32>(start), static_cast<int32>(end),
328 &ready_to_split);
329 };
330
331 if (param_proto_.collate_examples()) {
332 Shard(num_threads, worker_threads->workers, num_leaves, costPerUpdate,
333 update_collated);
334 } else {
335 Shard(num_threads, worker_threads->workers, num_data, costPerUpdate,
336 update);
337 }
338
339 Tensor* output_finished_t = nullptr;
340 TensorShape output_shape;
341 output_shape.AddDim(ready_to_split.size());
342 OP_REQUIRES_OK(
343 context, context->allocate_output(0, output_shape, &output_finished_t));
344 auto output = output_finished_t->unaligned_flat<int32>();
345 std::copy(ready_to_split.begin(), ready_to_split.end(), output.data());
346 }
347
348 private:
349 int32 random_seed_;
350 tensorforest::TensorForestDataSpec input_spec_;
351 TensorForestParams param_proto_;
352};

354// Op for growing finished nodes.
355class GrowTreeOp : public OpKernel {
356 public:
357 explicit GrowTreeOp(OpKernelConstruction* context) : OpKernel(context) {
358 string serialized_params;
359 OP_REQUIRES_OK(context, context->GetAttr("params", &serialized_params));
360 ParseProtoUnlimited(&param_proto_, serialized_params);
361 }
362
363 void Compute(OpKernelContext* context) override {
364 FertileStatsResource* fertile_stats_resource;
365 OP_REQUIRES_OK(context, LookupResource(context, HandleFromInput(context, 1),
366 &fertile_stats_resource));
367 DecisionTreeResource* tree_resource;
368 OP_REQUIRES_OK(context, LookupResource(context, HandleFromInput(context, 0),
369 &tree_resource));
370 mutex_lock l1(*fertile_stats_resource->get_mutex());
371 mutex_lock l2(*tree_resource->get_mutex());
372
373 core::ScopedUnref unref_stats(fertile_stats_resource);
374 core::ScopedUnref unref_tree(tree_resource);
375
376 const Tensor& finished_nodes = context->input(2);
377
378 const auto finished = finished_nodes.unaligned_flat<int32>();
379
380 const int32 num_nodes =
381 static_cast<int32>(finished_nodes.shape().dim_size(0));
382
383 // This op takes so little of the time for one batch that it isn't worth
384 // threading this.
385 for (int i = 0;
386 i < num_nodes &&
387 tree_resource->decision_tree().decision_tree().nodes_size() <
388 param_proto_.max_nodes();
389 ++i) {
390 const int32 node = finished(i);
391 std::unique_ptr<SplitCandidate> best(new SplitCandidate);
392 int32 parent_depth;
393 // TODO(gilberth): Pushing these to an output would allow the complete
394 // decoupling of tree from resource.
395 bool found =
396 fertile_stats_resource->BestSplit(node, best.get(), &parent_depth);
397 if (found) {
398 std::vector<int32> new_children;
399 tree_resource->SplitNode(node, best.get(), &new_children);
400 fertile_stats_resource->Allocate(parent_depth, new_children);
401 // We are done with best, so it is now safe to clear node.
402 fertile_stats_resource->Clear(node);
403 CHECK(tree_resource->get_mutable_tree_node(node)->has_leaf() == false);
404 } else { // reset
405 fertile_stats_resource->ResetSplitStats(node, parent_depth);
406 }
407 }
408 }
409
410 private:
411 tensorforest::TensorForestDataSpec input_spec_;
412 TensorForestParams param_proto_;
413};

415void FinalizeLeaf(bool is_regression, bool drop_final_class,
416 const std::unique_ptr<LeafModelOperator>& leaf_op,
417 decision_trees::Leaf* leaf) {
418 // regression models are already stored in leaf in normalized form.
419 if (is_regression) {
420 return;
421 }
422
423 // TODO(gilberth): Calculate the leaf's sum.
424 float sum = 0;
425 LOG(FATAL) << "FinalizeTreeOp is disabled for now.";
426 if (sum <= 0.0) {
427 LOG(WARNING) << "Leaf with sum " << sum << " has stats "
428 << leaf->ShortDebugString();
429 return;
430 }
431
432 if (leaf->has_vector()) {
433 for (int i = 0; i < leaf->vector().value_size(); i++) {
434 auto* v = leaf->mutable_vector()->mutable_value(i);
435 v->set_float_value(v->float_value() / sum);
436 }
437 if (drop_final_class) {
438 leaf->mutable_vector()->mutable_value()->RemoveLast();
439 }
440 return;
441 }
442
443 if (leaf->has_sparse_vector()) {
444 for (auto& it : *leaf->mutable_sparse_vector()->mutable_sparse_value()) {
445 it.second.set_float_value(it.second.float_value() / sum);
446 }
447 return;
448 }
449
450 LOG(FATAL) << "Unknown leaf type in " << leaf->DebugString();
451}

453// Op for finalizing a tree at the end of training.
454class FinalizeTreeOp : public OpKernel {
455 public:
456 explicit FinalizeTreeOp(OpKernelConstruction* context) : OpKernel(context) {
457 string serialized_params;
458 OP_REQUIRES_OK(context, context->GetAttr("params", &serialized_params));
459 ParseProtoUnlimited(&param_proto_, serialized_params);
460
461 model_op_ = LeafModelOperatorFactory::CreateLeafModelOperator(param_proto_);
462 }
463
464 void Compute(OpKernelContext* context) override {
465 DecisionTreeResource* tree_resource;
466 OP_REQUIRES_OK(context, LookupResource(context, HandleFromInput(context, 0),
467 &tree_resource));
468 FertileStatsResource* fertile_stats_resource;
469 OP_REQUIRES_OK(context, LookupResource(context, HandleFromInput(context, 1),
470 &fertile_stats_resource));
471
472 mutex_lock l1(*fertile_stats_resource->get_mutex());
473 mutex_lock l2(*tree_resource->get_mutex());
474
475 core::ScopedUnref unref_me(tree_resource);
476 core::ScopedUnref unref_stats(fertile_stats_resource);
477
478 // TODO(thomaswc): Add threads
479 int num_nodes = tree_resource->decision_tree().decision_tree().nodes_size();
480 for (int i = 0; i < num_nodes; i++) {
481 auto* node = tree_resource->mutable_decision_tree()
482 ->mutable_decision_tree()
483 ->mutable_nodes(i);
484 if (node->has_leaf()) {
485 FinalizeLeaf(param_proto_.is_regression(),
486 param_proto_.drop_final_class(), model_op_,
487 node->mutable_leaf());
488 }
489 }
490 }
491
492 private:
493 std::unique_ptr<LeafModelOperator> model_op_;
494 TensorForestParams param_proto_;
495};
