20namespace tensorflow {
21namespace tensorforest {
22
23// Returns the smoothed, unweighted Gini impurity.
24float GiniImpurity(const LeafStat& stats, int32 num_classes);
25
26// Returns the smoothed, weighted Gini impurity
27float WeightedGiniImpurity(const LeafStat& stats, int32 num_classes);
28
29// Updates the GiniStats given the old and new values of a class count that
30// was updated.
31void UpdateGini(LeafStat* stats, float old_val, float weight);
32
33// Returns the variance in stats for the given output.
34float Variance(const LeafStat& stats, int output);
35
36// Returns the variance sum for all outputs.
37float TotalVariance(const LeafStat& stats);
38
39// ------- functions used by C++ stats classes -------- //
40// Returns the smoothed gini score given the sum and sum of the squares of the
41// class counts.
42float SmoothedGini(float sum, float square, int num_classes);
43
44// Returns the smoothed gini score weighted by the sum.
45float WeightedSmoothedGini(float sum, float square, int num_classes);
