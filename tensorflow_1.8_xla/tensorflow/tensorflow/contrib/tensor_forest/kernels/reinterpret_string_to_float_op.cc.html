15// Converts strings of arbitrary length to float values by
16// hashing and cramming bits.
17#include <functional>
18
19#include "tensorflow/contrib/tensor_forest/kernels/tree_utils.h"
20
21#include "tensorflow/core/framework/common_shape_fns.h"
22#include "tensorflow/core/framework/op.h"
23#include "tensorflow/core/framework/op_kernel.h"
24#include "tensorflow/core/lib/strings/numbers.h"
25#include "tensorflow/core/lib/strings/strcat.h"
26
27#include "tensorflow/core/util/work_sharder.h"

29namespace tensorflow {
30
31using tensorforest::CheckTensorBounds;
32
33float Convert(const string& in) {
34 const std::size_t intval = std::hash<string>()(in);
35 return static_cast<float>(intval);
36}
37
38void Evaluate(const Tensor& input_data, Tensor output_data, int32 start,
39 int32 end) {
40 auto out_data = output_data.unaligned_flat<float>();
41 const auto in_data = input_data.unaligned_flat<string>();
42
43 for (int32 i = start; i < end; ++i) {
44 out_data(i) = Convert(in_data(i));
45 }
46}
47
48class ReinterpretStringToFloat : public OpKernel {
49 public:
50 explicit ReinterpretStringToFloat(OpKernelConstruction* context)
51 : OpKernel(context) {}
52
53 void Compute(OpKernelContext* context) override {
54 const Tensor& input_data = context->input(0);
55
56 // Check tensor bounds.
57 if (!CheckTensorBounds(context, input_data)) return;
58
59 Tensor* output_data = nullptr;
60 OP_REQUIRES_OK(
61 context, context->allocate_output(0, input_data.shape(), &output_data));
62
63 // Evaluate input data in parallel.
64 const int32 num_data = static_cast<int32>(input_data.NumElements());
65 auto worker_threads = context->device()->tensorflow_cpu_worker_threads();
66 int num_threads = worker_threads->num_threads;
67 if (num_threads <= 1) {
68 Evaluate(input_data, *output_data, 0, num_data);
69 } else {
70 auto work = [&input_data, output_data, num_data](int64 start, int64 end) {
71 CHECK(start <= end);
72 CHECK(end <= num_data);
73 Evaluate(input_data, *output_data, static_cast<int32>(start),
74 static_cast<int32>(end));
75 };
76 Shard(num_threads, worker_threads->workers, num_data, 100, work);
77 }
78 }
79};
80
81REGISTER_KERNEL_BUILDER(Name("ReinterpretStringToFloat").Device(DEVICE_CPU),
82 ReinterpretStringToFloat);
