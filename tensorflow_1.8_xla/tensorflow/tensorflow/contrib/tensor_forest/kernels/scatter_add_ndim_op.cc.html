15// ScatterAddNdim implements a scatter_add that can operate on sparse
16// updates without being limited to the first dimension for indices.
17
18#include "tensorflow/contrib/tensor_forest/kernels/tree_utils.h"
19
20#include "tensorflow/core/framework/op.h"
21#include "tensorflow/core/framework/op_kernel.h"
22#include "tensorflow/core/framework/shape_inference.h"
23#include "tensorflow/core/platform/logging.h"
24
25namespace tensorflow {
26
27using tensorforest::CheckTensorBounds;
28
29class ScatterAddNdim : public OpKernel {
30 public:
31 explicit ScatterAddNdim(OpKernelConstruction* context) : OpKernel(context) {}
32
33 void Compute(OpKernelContext* context) override {
34 Tensor input_tensor = context->mutable_input(0, false);
35 const Tensor& indices_tensor = context->input(1);
36 const Tensor& deltas_tensor = context->input(2);
37
38 if (indices_tensor.shape().dim_size(0) > 0) {
39 OP_REQUIRES(context, indices_tensor.shape().dims() == 2,
40 errors::InvalidArgument("indices should be two-dimensional"));
41 const int32 delta_dims = deltas_tensor.shape().dims();
42 OP_REQUIRES(
43 context,
44 indices_tensor.shape().dim_size(1) + delta_dims ==
45 input_tensor.shape().dims() + 1,
46 errors::InvalidArgument(
47 "Number of indices dimensions should be the same as input "
48 "rank."));
49 OP_REQUIRES(
50 context,
51 indices_tensor.shape().dim_size(0) ==
52 deltas_tensor.shape().dim_size(0),
53 errors::InvalidArgument(
54 "Number of updates should be same as number of indices."));
55 } else {
56 return;
57 }
58
59 // Check tensor bounds.
60 if (!CheckTensorBounds(context, input_tensor)) return;
61 if (!CheckTensorBounds(context, indices_tensor)) return;
62 if (!CheckTensorBounds(context, deltas_tensor)) return;
63
64 auto input = input_tensor.flat<float>();
65
66 const auto indices = indices_tensor.tensor<int32, 2>();
67 const auto deltas = deltas_tensor.unaligned_flat<float>();
68
69 const int32 num_dims =
70 static_cast<int32>(indices_tensor.shape().dim_size(1));
71
72 // Figure out if indices don't specify a complete position in the
73 // input tensor.
74 int32 num_data_per_index = 1;
75 for (int32 i = 0; i < input_tensor.shape().dims() - num_dims; ++i) {
76 num_data_per_index *= input_tensor.shape().dim_size(num_dims + i);
77 }
78
79 // Calculate index multipliers.
80 std::vector<int32> multipliers;
81 OP_REQUIRES(context, input.size() < std::numeric_limits<int32>::max(),
82 errors::InvalidArgument(
83 "Input must contain less than 2^31 total elements"));
84 int32 last_size = static_cast<int32>(input.size());
85
86 for (int32 j = 0; j < num_dims; j++) {
87 const int32 m = last_size / input_tensor.shape().dim_size(j);
88 multipliers.push_back(m);
89 last_size = m;
90 }
91
92 // Perform updates.
93 for (int32 i = 0; i < indices_tensor.shape().dim_size(0); i++) {
94 int32 start_index = 0;
95 for (int32 j = 0; j < num_dims; j++) {
96 start_index += indices(i, j) * multipliers[j];
97 }
98 for (int32 offset = 0; offset < num_data_per_index; ++offset) {
99 const int32 input_index = start_index + offset;
100 const int32 delta_index = i * num_data_per_index + offset;
101 CHECK(input_index < input.size());
102 CHECK(delta_index < deltas.size());
103 input(input_index) += deltas(delta_index);
104 }
105 }
106 }
107};
108
109REGISTER_KERNEL_BUILDER(Name("ScatterAddNdim").Device(DEVICE_CPU),
110 ScatterAddNdim);
111} // namespace tensorflow
