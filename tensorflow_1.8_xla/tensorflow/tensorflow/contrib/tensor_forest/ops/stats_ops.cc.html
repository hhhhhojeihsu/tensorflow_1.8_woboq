74REGISTER_OP("GrowTreeV4")
75 .Attr("params: string")
76 .Input("tree_handle: resource")
77 .Input("stats_handle: resource")
78 .Input("finshed_nodes: int32")
79 .SetShapeFn(tensorflow::shape_inference::NoOutputs)
80 .Doc(R"doc(
81Grows the tree for finished nodes and allocates waiting nodes.
82
83params: A serialized TensorForestParams proto.
84tree_handle: The handle to the tree.
85stats_handle: The handle to the stats.
86finshed_nodes: A 1-d Tensor of finished node ids from ProcessInput.
87)doc");
88
89REGISTER_OP("ProcessInputV4")
90 .Attr("random_seed: int")
91 .Attr("input_spec: string")
92 .Attr("params: string")
93 .Input("tree_handle: resource")
94 .Input("stats_handle: resource")
95 .Input("input_data: float")
96 .Input("sparse_input_indices: int64")
97 .Input("sparse_input_values: float")
98 .Input("sparse_input_shape: int64")
99 .Input("input_labels: float")
100 .Input("input_weights: float")
101 .Input("leaf_ids: int32")
102 .Output("finished_nodes: int32")
103 .SetShapeFn([](InferenceContext* c) {
104 c->set_output(0, c->Vector(c->UnknownDim()));
105 return Status::OK();
106 })
107 .Doc(R"doc(
108Add labels to stats after traversing the tree for each example.
109
110Outputs node ids that are finished.
111
112params: A serialized TensorForestParams proto.
113tree_handle: The handle to the tree.
114stats_handle: The handle to the stats.
115input_data: The training batch's features as a 2-d tensor; `input_data[i][j]`
116 gives the j-th feature of the i-th input.
117sparse_input_indices: The indices tensor from the SparseTensor input.
118sparse_input_values: The values tensor from the SparseTensor input.
119sparse_input_shape: The shape tensor from the SparseTensor input.
120input_labels: The training batch's labels as a 1 or 2-d tensor.
121 'input_labels[i][j]' gives the j-th label/target for the i-th input.
122input_weights: The training batch's eample weights as a 1-d tensor.
123 'input_weights[i]' gives the weight for the i-th input.
124finished_nodes: A 1-d tensor of node ids that have finished and are ready to
125 grow.
126leaf_ids: `leaf_ids[i]` is the leaf id for input i.
127)doc");
128
129REGISTER_OP("FinalizeTree")
130 .Attr("params: string")
131 .Input("tree_handle: resource")
132 .Input("stats_handle: resource")
133 .SetShapeFn([](InferenceContext* c) { return Status::OK(); })
134 .Doc(R"doc(
135Puts the Leaf models inside the tree into their final form.
136
137If drop_final_class is true, the per-class probability prediction of the
138last class is not stored in the leaf models.
139
140params: A serialized TensorForestParams proto.
141tree_handle: The handle to the tree.
142stats_handle: The handle to the stats.
143)doc");
