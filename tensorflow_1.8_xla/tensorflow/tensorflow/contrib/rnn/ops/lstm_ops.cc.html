25REGISTER_OP("LSTMBlockCell")
26 .Input("x: T")
27 .Input("cs_prev: T")
28 .Input("h_prev: T")
29 .Input("w: T")
30 .Input("wci: T")
31 .Input("wcf: T")
32 .Input("wco: T")
33 .Input("b: T")
34 .Output("i: T")
35 .Output("cs: T")
36 .Output("f: T")
37 .Output("o: T")
38 .Output("ci: T")
39 .Output("co: T")
40 .Output("h: T")
41 .Attr("forget_bias: float = 1.0")
42 .Attr("cell_clip: float = 3.0")
43 .Attr("use_peephole: bool = false")
44 .Attr("T: {float}")
45 .SetShapeFn([](InferenceContext* c) {
46 ShapeHandle x, cs_prev;
47 TF_RETURN_IF_ERROR(c->WithRank(c->input(0), 2, &x));
48 TF_RETURN_IF_ERROR(c->WithRank(c->input(1), 2, &cs_prev));
49
50 DimensionHandle batch_size = c->Dim(x, 0);
51 DimensionHandle cell_size = c->Dim(cs_prev, 1);
52 ShapeHandle output = c->Matrix(batch_size, cell_size);
53 for (int i = 0; i < 7; ++i) {
54 c->set_output(i, output);
55 }
56 return tensorflow::Status::OK();
57 })
58 .Doc(R"doc(
59Computes the LSTM cell forward propagation for 1 time step.
60
61This implementation uses 1 weight matrix and 1 bias vector, and there's an
62optional peephole connection.
63
64This kernel op implements the following mathematical equations:
65
66```python
67xh = [x, h_prev]
68[i, f, ci, o] = xh * w + b
69f = f + forget_bias
70
71if not use_peephole:
72 wci = wcf = wco = 0
73
74i = sigmoid(cs_prev * wci + i)
75f = sigmoid(cs_prev * wcf + f)
76ci = tanh(ci)
77
78cs = ci .* i + cs_prev .* f
79cs = clip(cs, cell_clip)
80
81o = sigmoid(cs * wco + o)
82co = tanh(cs)
83h = co .* o
84```
85
86cell_clip: Value to clip the 'cs' value to.
87use_peephole: Whether to use peephole weights.
88forget_bias: The forget gate bias.
89
90x: The input to the LSTM cell, shape (batch_size, num_inputs).
91cs_prev: Value of the cell state at previous time step.
92h_prev: Output of the previous cell at previous time step.
93w: The weight matrix.
94wci: The weight matrix for input gate peephole connection.
95wcf: The weight matrix for forget gate peephole connection.
96wco: The weight matrix for output gate peephole connection.
97b: The bias vector.
98
99i: The input gate.
100cs: The cell state before the tanh.
101f: The forget gate.
102o: The output gate.
103ci: The cell input.
104co: The cell after the tanh.
105h: The output h vector.
106)doc");

125 .Output("cs_prev_grad: T")
126 .Output("dicfo: T")
127 .Output("wci_grad: T")
128 .Output("wcf_grad: T")
129 .Output("wco_grad: T")
130 .Attr("use_peephole: bool")
131 .Attr("T: {float}")
132 .SetShapeFn([](InferenceContext* c) {
133 ShapeHandle x, cs_prev;
134 TF_RETURN_IF_ERROR(c->WithRank(c->input(0), 2, &x));
135 TF_RETURN_IF_ERROR(c->WithRank(c->input(1), 2, &cs_prev));
136
137 DimensionHandle batch_size = c->Dim(x, 0);
138 DimensionHandle cell_size = c->Dim(cs_prev, 1);
139 DimensionHandle cell_size_times_4;
140 TF_RETURN_IF_ERROR(c->Multiply(cell_size, 4, &cell_size_times_4));
141 ShapeHandle cell_size_vec = c->Vector(cell_size);
142
143 c->set_output(0, c->Matrix(batch_size, cell_size));
144 c->set_output(1, c->Matrix(batch_size, cell_size_times_4));
145 c->set_output(2, cell_size_vec);
146 c->set_output(3, cell_size_vec);
147 c->set_output(4, cell_size_vec);
148 return tensorflow::Status::OK();
149 })

179REGISTER_OP("BlockLSTM")
180 .Input("seq_len_max: int64")
181 .Input("x: T")
182 .Input("cs_prev: T")
183 .Input("h_prev: T")
184 .Input("w: T")
185 .Input("wci: T")
186 .Input("wcf: T")
187 .Input("wco: T")
188 .Input("b: T")
189 .Output("i: T")
190 .Output("cs: T")
191 .Output("f: T")
192 .Output("o: T")
193 .Output("ci: T")
194 .Output("co: T")
195 .Output("h: T")
196 .Attr("forget_bias: float = 1.0")
197 .Attr("cell_clip: float = 3.0")
198 .Attr("use_peephole: bool = false")
199 .Attr("T: {float}")
200 .SetShapeFn([](InferenceContext* c) {
201 ShapeHandle x, b;
202 TF_RETURN_IF_ERROR(c->WithRank(c->input(1), 3, &x));
203 TF_RETURN_IF_ERROR(c->WithRank(c->input(c->num_inputs() - 1), 1, &b));
204
205 DimensionHandle timelen = c->Dim(x, 0);
206 DimensionHandle batch_size = c->Dim(x, 1);
207 DimensionHandle cell_size;
208 TF_RETURN_IF_ERROR(
209 c->Divide(c->Dim(b, 0), 4, true /* evenly_divisible */, &cell_size));
210
211 DCHECK_EQ(7, c->num_outputs());
212 ShapeHandle output = c->MakeShape({timelen, batch_size, cell_size});
213 for (int i = 0; i < 7; ++i) {
214 c->set_output(i, output);
215 }
216 return Status::OK();
217 })
218 .Doc(R"doc(
219Computes the LSTM cell forward propagation for all the time steps.
220
221This is equivalent to applying LSTMBlockCell in a loop, like so:
222
223```python
224for x1 in unpack(x):
225 i1, cs1, f1, o1, ci1, co1, h1 = LSTMBlock(
226 x1, cs_prev, h_prev, w, wci, wcf, wco, b)
227 cs_prev = cs1
228 h_prev = h1
229 i.append(i1)
230 cs.append(cs1)
231 f.append(f1)
232 o.append(o1)
233 ci.append(ci1)
234 co.append(co1)
235 h.append(h1)
236return pack(i), pack(cs), pack(f), pack(o), pack(ci), pack(ch), pack(h)
237```
238
239cell_clip: Value to clip the 'cs' value to.
240use_peephole: Whether to use peephole weights.
241forget_bias: The forget gate bias.
242
243seq_len_max: Maximum time length actually used by this input. Outputs are padded
244 with zeros beyond this length.
245x: The sequence input to the LSTM, shape (timelen, batch_size, num_inputs).
246cs_prev: Value of the initial cell state.
247h_prev: Initial output of cell (to be used for peephole).
248w: The weight matrix.
249wci: The weight matrix for input gate peephole connection.
250wcf: The weight matrix for forget gate peephole connection.
251wco: The weight matrix for output gate peephole connection.
252b: The bias vector.
253
254i: The input gate over the whole time sequence.
255cs: The cell state before the tanh over the whole time sequence.
256f: The forget gate over the whole time sequence.
257o: The output gate over the whole time sequence.
258ci: The cell input over the whole time sequence.
259co: The cell after the tanh over the whole time sequence.
260h: The output h vector over the whole time sequence.
261)doc");

280 .Input("cs_grad: T")
281 .Input("h_grad: T")
282 .Output("x_grad: T")
283 .Output("cs_prev_grad: T")
284 .Output("h_prev_grad: T")
285 .Output("w_grad: T")
286 .Output("wci_grad: T")
287 .Output("wcf_grad: T")
288 .Output("wco_grad: T")
289 .Output("b_grad: T")
290 .Attr("use_peephole: bool")
291 .Attr("T: {float}")
292 .SetShapeFn([](InferenceContext* c) {
293 ShapeHandle x, cs_prev, h_prev, w, wci, wco, wcf, b;
294 TF_RETURN_IF_ERROR(c->WithRank(c->input(1), 3, &x));
295 TF_RETURN_IF_ERROR(c->WithRank(c->input(2), 2, &cs_prev));
296 TF_RETURN_IF_ERROR(c->WithRank(c->input(3), 2, &h_prev));
297 TF_RETURN_IF_ERROR(c->WithRank(c->input(4), 2, &w));
298 TF_RETURN_IF_ERROR(c->WithRank(c->input(5), 1, &wci));
299 TF_RETURN_IF_ERROR(c->WithRank(c->input(6), 1, &wco));
300 TF_RETURN_IF_ERROR(c->WithRank(c->input(7), 1, &wcf));
301 TF_RETURN_IF_ERROR(c->WithRank(c->input(8), 1, &b));
302
303 c->set_output(0, x);
304 c->set_output(1, cs_prev);
305 c->set_output(2, h_prev);
306 c->set_output(3, w);
307 c->set_output(4, wci);
308 c->set_output(5, wco);
309 c->set_output(6, wcf);
310 c->set_output(7, b);
311
312 return Status::OK();
313 })
