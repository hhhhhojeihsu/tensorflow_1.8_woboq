15#ifndef TENSORFLOW_CONTRIB_LITE_ERROR_REPORTER_H_
16#define TENSORFLOW_CONTRIB_LITE_ERROR_REPORTER_H_
17
18#include <cstdarg>
19#include "tensorflow/contrib/lite/context.h"
20
21namespace tflite {
22
23// A functor that reports error to supporting system. Invoked similar to
24// printf.
25//
26// Usage:
27// ErrorReporter foo;
28// foo.Report("test %d", 5);
29// or
30// va_list args;
31// foo.Report("test %d", args); // where args is va_list
32//
33// Subclass ErrorReporter to provide another reporting destination.
34// For example, if you have a GUI program, you might redirect to a buffer
35// that drives a GUI error log box.
36class ErrorReporter {
37 public:
38 virtual ~ErrorReporter();
39 virtual int Report(const char* format, va_list args) = 0;
40 int Report(const char* format, ...);
41 int ReportError(void*, const char* format, ...);
42};
43
44// An error reporter that simplify writes the message to stderr.
45struct StderrReporter : public ErrorReporter {
46 int Report(const char* format, va_list args) override;
47};
48
49// Return the default error reporter (output to stderr).
50ErrorReporter* DefaultErrorReporter();
51
52} // namespace tflite
53
54#endif // TENSORFLOW_CONTRIB_LITE_ERROR_REPORTER_H_
