20int TfLiteIntArrayGetSizeInBytes(int size) {
21 static TfLiteIntArray dummy;
22 return sizeof(dummy) + sizeof(dummy.data[0]) * size;
23}
24
25TfLiteIntArray* TfLiteIntArrayCreate(int size) {
26 TfLiteIntArray* ret =
27 (TfLiteIntArray*)malloc(TfLiteIntArrayGetSizeInBytes(size));
28 ret->size = size;
29 return ret;
30}

32void TfLiteIntArrayPrint(const char* s, TfLiteIntArray* a) {
33 printf("%s: length=%d [", s, a->size);
34 if (a->size) printf("%d", a->data[0]);
35 int i = 1;
36 for (; i < a->size; i++) {
37 printf(" %d", a->data[i]);
38 }
39 printf("]\n");
40}

42int TfLiteIntArrayEqual(TfLiteIntArray* a, TfLiteIntArray* b) {
43 if (a == b) return 1;
44 if (a == NULL || b == NULL) return 0;
45 if (a->size != b->size) return 0;
46 int i = 0;
47 for (; i < a->size; i++)
48 if (a->data[i] != b->data[i]) return 0;
49 return 1;
50}

52TfLiteIntArray* TfLiteIntArrayCopy(TfLiteIntArray* src) {
53 if (!src) return NULL;
54 TfLiteIntArray* ret = TfLiteIntArrayCreate(src->size);
55 if (ret) {
56 memcpy(ret->data, src->data, src->size * sizeof(int));
57 }
58 return ret;
59}

63void TfLiteTensorDataFree(TfLiteTensor* t) {
64 if (t->allocation_type == kTfLiteDynamic && t->data.raw) {
65 free(t->data.raw);
66 }
67 t->data.raw = NULL;
68}
69
70void TfLiteTensorFree(TfLiteTensor* t) {
71 TfLiteTensorDataFree(t);
72 if (t->dims) TfLiteIntArrayFree(t->dims);
73 t->dims = NULL;
74}

76void TfLiteTensorReset(TfLiteType type, const char* name, TfLiteIntArray* dims,
77 TfLiteQuantizationParams quantization, char* buffer,
78 size_t size, TfLiteAllocationType allocation_type,
79 const void* allocation, TfLiteTensor* tensor) {
80 TfLiteTensorFree(tensor);
81 tensor->type = type;
82 tensor->name = name;
83 tensor->dims = dims;
84 tensor->params = quantization;
85 tensor->data.raw = buffer;
86 tensor->bytes = size;
87 tensor->allocation_type = allocation_type;
88 tensor->allocation = allocation;
89}
90
91void TfLiteTensorRealloc(size_t num_bytes, TfLiteTensor* tensor) {
92 if (tensor->allocation_type != kTfLiteDynamic) {
93 return;
94 }
95 if (!tensor->data.raw) {
96 tensor->data.raw = malloc(num_bytes);
97 } else if (num_bytes > tensor->bytes) {
98 tensor->data.raw = realloc(tensor->data.raw, num_bytes);
99 }
100 tensor->bytes = num_bytes;
101}
