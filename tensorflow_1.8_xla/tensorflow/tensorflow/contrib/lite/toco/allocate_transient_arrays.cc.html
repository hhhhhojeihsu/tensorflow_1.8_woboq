32// The life span of an array.
33struct ArrayLifespan {
34 // If true, the array is persistent state (as in a RNN). In that case,
35 // its allocation is permanent and the first_op, last_op members are
36 // unused. (The term 'transient' is a misnomer and we should think in
37 // terms of 'workspace' instead).
38 bool persistent = false;
39 // Index of the first op addressing that array. The array must be allocated
40 // just before executing this op.
41 std::size_t first_op = 0;
42 // Index of the last op addressing that array. We want to deallocate the array
43 // immediately after executing this op.
44 std::size_t last_op = 0;
45};
46
47bool StartsAt(const ArrayLifespan& lifespan, std::size_t op_index) {
48 return !lifespan.persistent && lifespan.first_op == op_index;
49}
50
51bool EndsAt(const ArrayLifespan& lifespan, std::size_t op_index) {
52 return !lifespan.persistent && lifespan.last_op == op_index;
53}
54
55// Helper function for ComputeArrayLifespans: updates one ArrayLifespan for
56// one array for one op.
57void UpdateArrayLifespan(
58 const string& array_name, std::size_t op_index,
59 std::unordered_map<string, ArrayLifespan>* array_lifespans) {
60 if (array_lifespans->count(array_name)) {
61 auto& lifespan = array_lifespans->at(array_name);
62 if (!lifespan.persistent) {
63 lifespan.first_op = std::min(lifespan.first_op, op_index);
64 lifespan.last_op = std::max(lifespan.last_op, op_index);
65 }
66 } else {
67 ArrayLifespan lifespan;
68 lifespan.first_op = op_index;
69 lifespan.last_op = op_index;
70 (*array_lifespans)[array_name] = lifespan;
71 }
72}
73
74// Computes the ArrayLifespan for each array.
75void ComputeArrayLifespans(
76 const Model& model,
77 std::unordered_map<string, ArrayLifespan>* array_lifespans) {
78 CHECK(array_lifespans->empty());
79 for (const auto& rnn_state : model.flags.rnn_states()) {
80 ArrayLifespan lifespan;
81 lifespan.persistent = true;
82 (*array_lifespans)[rnn_state.state_array()] = lifespan;
83 }
84 for (std::size_t op_index = 0; op_index < model.operators.size();
85 op_index++) {
86 const auto& op = model.operators[op_index];
87 for (const auto& input : op->inputs) {
88 UpdateArrayLifespan(input, op_index, array_lifespans);
89 }
90 for (const auto& output : op->outputs) {
91 UpdateArrayLifespan(output, op_index, array_lifespans);
92 }
93 }
94}

96inline bool operator==(const Alloc& a, const Alloc& b) {
97 CHECK(a.start != b.start || a.end == b.end);
98 return a.start == b.start;
99}

101// Helper to keep track of total allocation size and of currently live
102// allocations, and containing the core allocation routine.
103class Allocator {
104 public:
105 Allocator() : total_size_(0) {}
106
107 // Core allocation routine.
108 void Allocate(std::size_t size, Alloc* result) {
109 // Naive algorithm: pick the first gap between live allocations,
110 // that is wide enough for the new array.
111 std::size_t pos = 0;
112 for (const auto& a : live_allocs_) {
113 if (a.start >= pos + size) {
114 result->start = pos;
115 result->end = pos + size;
116 live_allocs_.insert(*result);
117 return;
118 }
119 pos = a.end;
120 }
121 // No sufficiently wide gap was found before an existing live allocation,
122 // so we allocate the new array at the end of the allocation space.
123 // We may then have to grow total_size_.
124 total_size_ = std::max(total_size_, pos + size);
125 result->start = pos;
126 result->end = pos + size;
127 live_allocs_.insert(*result);
128 }
129
130 void Deallocate(const Alloc& a) {
131 auto iter = std::lower_bound(live_allocs_.begin(), live_allocs_.end(), a);
132 CHECK(iter != live_allocs_.end());
133 CHECK(*iter == a);
134 live_allocs_.erase(iter);
135 }
136
137 std::size_t total_size() const { return total_size_; }
138
139 private:
140 std::size_t total_size_;
141 std::set<Alloc> live_allocs_;
142};

144// Returns the required transient allocation size (in bytes) for a given array,
145// or 0 if it's not a transient array.
146std::size_t TransientArraySize(const Model& model, const string& array_name,
147 std::size_t transient_data_alignment) {
148 if (!IsAllocatableTransientArray(model, array_name)) {
149 return 0;
150 }
151 const auto& array = &model.GetArray(array_name);
152 CHECK(array->has_shape())
153 << "Array '" << array_name << "' doesn't have a shape";
154 if (array->data_type == ArrayDataType::kNone) {
155 // Catch a typical issue at the moment with RNN states
156 for (const auto& rnn_state : model.flags.rnn_states()) {
157 if (rnn_state.state_array() == array_name) {
158 LOG(FATAL)
159 << "A RNN state array, " << array_name << ", still does not "
160 << "have a known data type after all graph transformations have "
161 << "run.";
162 }
163 }
164 LOG(FATAL) << "An array, " << array_name << ", still does not "
165 << "have a known data type after all graph transformations have "
166 << "run.";
167 }
168 const std::size_t elem_size = ElementSize(array->data_type);
169 const std::size_t raw_size =
170 elem_size * RequiredBufferSizeForShape(array->shape());
171 const std::size_t rounded_size =
172 RoundUpToNextMultipleOf(raw_size, transient_data_alignment);
173 return rounded_size;
174}

176// Allocates an array: call this for every array just before the first
177// op where it is used.
178void AllocateTransientArray(const Model& model, const string& array_name,
179 Allocator* allocator,
180 std::size_t transient_data_alignment) {
181 if (!IsAllocatableTransientArray(model, array_name)) {
182 return;
183 }
184 const std::size_t size =
185 TransientArraySize(model, array_name, transient_data_alignment);
186 const auto& array = &model.GetArray(array_name);
187 CHECK(!array->alloc);
188 allocator->Allocate(size, &array->GetOrCreateAlloc());
189}
190
191// Deallocates an array: call this for every array just after the last
192// op where it is used.
193void DeallocateTransientArray(const Model& model, const string& array_name,
194 Allocator* allocator) {
195 if (!IsAllocatableTransientArray(model, array_name)) {
196 return;
197 }
198 const auto& array = &model.GetArray(array_name);
199 CHECK(!!array->alloc);
200 allocator->Deallocate(*array->alloc);
201}
202
203void PushBackIfNotFound(const string& s, std::vector<string>* v) {
204 if (std::find(v->begin(), v->end(), s) == v->end()) {
205 v->push_back(s);
206 }
207}

211void AllocateTransientArrays(Model* model,
212 std::size_t transient_data_alignment) {
213 // Precompute the lifespans for all arrays.
214 std::unordered_map<string, ArrayLifespan> array_lifespans;
215 ComputeArrayLifespans(*model, &array_lifespans);
216
217 // In case of variable batch, our convention will be to compute the
218 // allocations for batch==1, then let the inference code multiply all
219 // the offsets by the actual runtime batch size. Conveniently,
220 // the variable_batch and batch flags are mutually exclusive, and the default
221 // value of batch is 1, so we have nothing special to do here. Let us
222 // just guard this assumption with a CHECK:
223 bool batchless_input_shapes = true;
224 for (const auto& input_array : model->flags.input_arrays()) {
225 if (!input_array.has_shape() || input_array.shape().dims().empty() ||
226 input_array.shape().dims(0) != 1) {
227 batchless_input_shapes = false;
228 break;
229 }
230 }
231 CHECK(!model->flags.variable_batch() || batchless_input_shapes);
232
233 Allocator allocator;
234
235 // Construct a sorted map of array names, so that other layout engines can
236 // match exactly.
237 std::map<string, const Array*> ordered_arrays_map;
238 for (const auto& pair : model->GetArrayMap()) {
239 ordered_arrays_map[pair.first] = pair.second.get();
240 }
241
242 // Allocate persistent arrays (like RNN states). For them, 'transient'
243 // is a misnormer, should read 'workspace'.
244 for (const auto& array_pair : ordered_arrays_map) {
245 const string& array_name = array_pair.first;
246 auto it = array_lifespans.find(array_name);
247 if (it != array_lifespans.end() && it->second.persistent) {
248 AllocateTransientArray(*model, array_name, &allocator,
249 transient_data_alignment);
250 }
251 }
252
253 for (std::size_t op_index = 0; op_index < model->operators.size();
254 op_index++) {
255 const auto& op = model->operators[op_index];
256 // Allocate those arrays whose lifespan starts exactly here.
257 std::vector<string> arrays_to_allocate;
258 for (const auto& input : op->inputs) {
259 if (StartsAt(array_lifespans[input], op_index)) {
260 PushBackIfNotFound(input, &arrays_to_allocate);
261 }
262 }
263 for (const auto& output : op->outputs) {
264 if (StartsAt(array_lifespans[output], op_index)) {
265 PushBackIfNotFound(output, &arrays_to_allocate);
266 }
267 }
268 for (const string& array : arrays_to_allocate) {
269 AllocateTransientArray(*model, array, &allocator,
270 transient_data_alignment);
271 }
272
273 // Deallocate those arrays whose lifespan ends exactly here.
274 std::vector<string> arrays_to_deallocate;
275 for (const auto& input : op->inputs) {
276 if (EndsAt(array_lifespans[input], op_index)) {
277 PushBackIfNotFound(input, &arrays_to_deallocate);
278 }
279 }
280 for (const auto& output : op->outputs) {
281 if (EndsAt(array_lifespans[output], op_index)) {
282 PushBackIfNotFound(output, &arrays_to_deallocate);
283 }
284 }
285 for (const string& array : arrays_to_deallocate) {
286 DeallocateTransientArray(*model, array, &allocator);
287 }
288 }
289
290 // Just out of curiosity (not used in the actual allocation process)
291 // evaluate the optimal total allocated size.
292 // First, compute the size of persistent arrays.
293 std::size_t optimal_transient_alloc_size = 0;
294 std::size_t persistent_alloc_size = 0;
295 for (const auto& array_pair : ordered_arrays_map) {
296 const string& array_name = array_pair.first;
297 auto it = array_lifespans.find(array_name);
298 if (it != array_lifespans.end() && it->second.persistent) {
299 persistent_alloc_size +=
300 TransientArraySize(*model, array_name, transient_data_alignment);
301 }
302 }
303 for (const auto& op : model->operators) {
304 // for each operator, compute the sum of the sizes of the array that must
305 // be live during the execution of this operator, plus the size of
306 // persistent arrays that must be live at all times.
307 std::vector<string> non_persistent_edges;
308 for (const auto& input : op->inputs) {
309 if (!array_lifespans[input].persistent) {
310 PushBackIfNotFound(input, &non_persistent_edges);
311 }
312 }
313 for (const auto& output : op->outputs) {
314 if (!array_lifespans[output].persistent) {
315 PushBackIfNotFound(output, &non_persistent_edges);
316 }
317 }
318 std::size_t size = persistent_alloc_size;
319 for (const string& edge : non_persistent_edges) {
320 size += TransientArraySize(*model, edge, transient_data_alignment);
321 }
322 // The optimal total size is the maximum of all operator-specific sizes.
323 optimal_transient_alloc_size = std::max(optimal_transient_alloc_size, size);
324 }
325
326 model->transient_data_size = allocator.total_size();
327 model->transient_data_alignment = transient_data_alignment;
328 CHECK_GE(model->transient_data_size, optimal_transient_alloc_size);
329 LOG(INFO) << "Total transient array allocated size: "
330 << model->transient_data_size << " bytes, "
331 << "theoretical optimal value: " << optimal_transient_alloc_size
332 << " bytes.";
333 CheckInvariants(*model);
334}
