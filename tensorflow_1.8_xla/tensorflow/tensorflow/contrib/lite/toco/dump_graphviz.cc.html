37class Color {
38 public:
39 Color() {}
40 Color(uint8 r, uint8 g, uint8 b) : r_(r), g_(g), b_(b) {}
41 // Returns the string serialization of this color in graphviz format,
42 // for use as 'fillcolor' in boxes.
43 string FillColorString() const { return StringF("%.2X%.2X%.2X", r_, g_, b_); }
44 // Returns the serialization in graphviz format of a suitable color to use
45 // 'fontcolor' in the same boxes. It should black or white, whichever offers
46 // the better contrast from FillColorString().
47 string TextColorString() const {
48 // https://en.wikipedia.org/wiki/Relative_luminance
49 const float luminance = 0.2126f * r_ + 0.7152f * g_ + 0.0722f * b_;
50 const uint8 l = luminance > 128.f ? 0 : 255;
51 return StringF("%.2X%.2X%.2X", l, l, l);
52 }
53
54 private:
55 uint8 r_ = 0, g_ = 0, b_ = 0;
56};

58struct NodeProperties {
59 // The text to display inside the box for this node.
60 string label;
61 // The color to use for this node; will be used as 'fillcolor'
62 // for its box. See Color::FillColorString. A suitable, different
63 // color will be chosen for the 'fontcolor' for the inside text
64 // label, see Color::TextColorString.
65 Color color;
66};

69// https://material.io/guidelines/style/color.html

71Color GetColorForArray(const Model& model, const string& array_name) {
72 // Arrays involved in RNN back-edges have a different color
73 for (const auto& rnn_state : model.flags.rnn_states()) {
74 // RNN state, fed by a back-edge. Bold color.
75 if (array_name == rnn_state.state_array()) {
76 return Color(0x0F, 0x9D, 0x58);
77 }
78 // RNN back-edge source, feeding a RNN state.
79 // Light tone of the same color as RNN states.
80 if (array_name == rnn_state.back_edge_source_array()) {
81 return Color(0xB7, 0xE1, 0xCD);
82 }
83 }
84 // Constant parameter arrays have their own bold color
85 if (model.GetArray(array_name).buffer) {
86 return Color(0x42, 0x85, 0xF4);
87 }
88 // Remaining arrays are activations.
89 // We use gray colors for them because they are the majority
90 // of arrays so we want to highlight other arrays instead of them.
91 // First, we use a bolder gray for input/output arrays:
92 const auto& dump_options = *GraphVizDumpOptions::singleton();
93 if (IsInputArray(model, array_name) ||
94 array_name == dump_options.graphviz_first_array ||
95 array_name == dump_options.graphviz_last_array) {
96 return Color(0x9E, 0x9E, 0x9E);
97 }
98 for (const string& output_array : model.flags.output_arrays()) {
99 if (array_name == output_array) {
100 return Color(0x9E, 0x9E, 0x9E);
101 }
102 }
103 // Remaining arrays are intermediate activation arrays.
104 // Lighter tone of the same grey as for input/output arrays:
105 // We want these to be very discrete.
106 return Color(0xF5, 0xF5, 0xF5);
107}

109void AppendArrayVal(string* string, Array const& array, int index) {
110 if (array.buffer->type == ArrayDataType::kFloat) {
111 const auto& data = array.GetBuffer<ArrayDataType::kFloat>().data;
112 if (index >= data.size()) {
113 return;
114 }
115 AppendF(string, "%.3f", data[index]);
116 } else if (array.buffer->type == ArrayDataType::kUint8) {
117 const auto& data = array.GetBuffer<ArrayDataType::kUint8>().data;
118 if (index >= data.size()) {
119 return;
120 }
121 AppendF(string, "%d", data[index]);
122 } else if (array.buffer->type == ArrayDataType::kInt32) {
123 const auto& data = array.GetBuffer<ArrayDataType::kInt32>().data;
124 if (index >= data.size()) {
125 return;
126 }
127 AppendF(string, "%d", data[index]);
128 } else if (array.buffer->type == ArrayDataType::kInt64) {
129 const auto& data = array.GetBuffer<ArrayDataType::kInt64>().data;
130 if (index >= data.size()) {
131 return;
132 }
133 AppendF(string, "%d", data[index]);
134 }
135}

137NodeProperties GetPropertiesForArray(const Model& model,
138 const string& array_name) {
139 NodeProperties node_properties;
140 node_properties.color = GetColorForArray(model, array_name);
141 node_properties.label = absl::StrReplaceAll(array_name, {{"/", "/\\n"}});
142
143 // Append array shape to the label.
144 auto& array = model.GetArray(array_name);
145 AppendF(&node_properties.label, "\\nType: %s",
146 ArrayDataTypeName(array.data_type));
147
148 if (array.has_shape()) {
149 auto& array_shape = array.shape();
150 node_properties.label += "\\n[";
151 for (int id = 0; id < array_shape.dimensions_count(); id++) {
152 if (id == 0) {
153 AppendF(&node_properties.label, "%d", array_shape.dims(id));
154 } else {
155 // 0x00D7 is the unicode multiplication symbol
156 AppendF(&node_properties.label, "\u00D7%d", array_shape.dims(id));
157 }
158 }
159 node_properties.label += "]";
160
161 if (array.buffer) {
162 const auto& array = model.GetArray(array_name);
163 int buffer_size = RequiredBufferSizeForShape(array.shape());
164 if (buffer_size <= 4) {
165 AppendF(&node_properties.label, " = ");
166 if (array.shape().dimensions_count() > 0) {
167 AppendF(&node_properties.label, "{");
168 }
169 for (int i = 0; i < buffer_size; i++) {
170 AppendArrayVal(&node_properties.label, array, i);
171 if (i + 1 < buffer_size) {
172 AppendF(&node_properties.label, ", ");
173 }
174 }
175 } else {
176 AppendF(&node_properties.label, "\\n = ");
177 if (array.shape().dimensions_count() > 0) {
178 AppendF(&node_properties.label, "{");
179 }
180 AppendArrayVal(&node_properties.label, array, 0);
181 AppendF(&node_properties.label, ", ");
182 AppendArrayVal(&node_properties.label, array, 1);
183 // 0x2026 is the unicode ellipsis symbol
184 AppendF(&node_properties.label, " \u2026 ");
185 AppendArrayVal(&node_properties.label, array, buffer_size - 2);
186 AppendF(&node_properties.label, ", ");
187 AppendArrayVal(&node_properties.label, array, buffer_size - 1);
188 }
189 if (array.shape().dimensions_count() > 0) {
190 AppendF(&node_properties.label, "}");
191 }
192 }
193 }
194
195 if (array.minmax) {
196 AppendF(&node_properties.label, "\\nMinMax: [%.7g, %.7g]",
197 array.minmax->min, array.minmax->max);
198 }
199
200 if (array.quantization_params) {
201 AppendF(&node_properties.label, "\\nQuantization: %7g * (x - %d)",
202 array.quantization_params->scale,
203 array.quantization_params->zero_point);
204 }
205
206 if (array.alloc) {
207 AppendF(&node_properties.label, "\\nTransient Alloc: [%d, %d)",
208 array.alloc->start, array.alloc->end);
209 }
210
211 return node_properties;
212}

214NodeProperties GetPropertiesForOperator(const Operator& op) {
215 NodeProperties node_properties;
216 if (op.type == OperatorType::kTensorFlowUnsupported) {
217 node_properties.label =
218 static_cast<const TensorFlowUnsupportedOperator&>(op).tensorflow_op;
219 } else {
220 node_properties.label =
221 string(absl::StripPrefix(OperatorTypeName(op.type), "TensorFlow"));
222 }
223 switch (op.fused_activation_function) {
224 case FusedActivationFunctionType::kRelu:
225 AppendF(&node_properties.label, "\\nReLU");
226 break;
227 case FusedActivationFunctionType::kRelu6:
228 AppendF(&node_properties.label, "\\nReLU6");
229 break;
230 case FusedActivationFunctionType::kRelu1:
231 AppendF(&node_properties.label, "\\nReLU1");
232 break;
233 default:
234 break;
235 }
236 // Additional information for some of the operators.
237 switch (op.type) {
238 case OperatorType::kConv: {
239 const auto& conv_op = static_cast<const ConvOperator&>(op);
240 node_properties.color = Color(0xC5, 0x39, 0x29); // Bolder color
241 AppendF(&node_properties.label, "\\n%dx%d/%s", conv_op.stride_width,
242 conv_op.stride_height,
243 conv_op.padding.type == PaddingType::kSame ? "S" : "V");
244 break;
245 }
246 case OperatorType::kDepthwiseConv: {
247 const auto& conv_op = static_cast<const DepthwiseConvOperator&>(op);
248 node_properties.color = Color(0xC5, 0x39, 0x29); // Bolder color
249 AppendF(&node_properties.label, "\\n%dx%d/%s", conv_op.stride_width,
250 conv_op.stride_height,
251 conv_op.padding.type == PaddingType::kSame ? "S" : "V");
252 break;
253 }
254 case OperatorType::kFullyConnected: {
255 node_properties.color = Color(0xC5, 0x39, 0x29); // Bolder color
256 break;
257 }
258 default:
259 node_properties.color = Color(0xDB, 0x44, 0x37);
260 break;
261 }
262
263 return node_properties;
264}

266std::vector<const Operator*> OperatorsToDump(const Model& model) {
267 const auto& dump_options = *GraphVizDumpOptions::singleton();
268 bool first_specified = !dump_options.graphviz_first_array.empty();
269 bool last_specified = !dump_options.graphviz_last_array.empty();
270 CHECK_EQ(first_specified, last_specified);
271 std::vector<const Operator*> ops_to_dump;
272 if (last_specified) {
273 // Return only the part of the graph between graphviz_first_array
274 // and graphviz_last_array.
275 CHECK(model.HasArray(dump_options.graphviz_first_array));
276 CHECK(model.HasArray(dump_options.graphviz_last_array));
277 std::unordered_set<string> arrays_already_produced;
278 std::vector<string> arrays_to_produce;
279 arrays_to_produce.push_back(dump_options.graphviz_last_array);
280 while (!arrays_to_produce.empty()) {
281 const string array = arrays_to_produce.back();
282 arrays_to_produce.pop_back();
283 CHECK(!arrays_already_produced.count(array));
284 arrays_already_produced.insert(array);
285 const Operator* op = GetOpWithOutput(model, array);
286 if (!op) {
287 continue;
288 }
289 ops_to_dump.push_back(op);
290 for (const string& input : op->inputs) {
291 if (arrays_already_produced.count(input) ||
292 input == dump_options.graphviz_first_array) {
293 continue;
294 }
295 arrays_to_produce.push_back(input);
296 }
297 }
298 } else {
299 // Return the whole graph.
300 for (const auto& op : model.operators) {
301 ops_to_dump.push_back(op.get());
302 }
303 }
304 return ops_to_dump;
305}

309void DumpGraphviz(const Model& model, string* output_file_contents) {
310 AppendF(output_file_contents, "digraph Computegraph {\n");
311
312 constexpr char kNodeFormat[] =
313 "\t \"%s\" [label=\"%s\", shape=%s, style=filled, fillcolor=\"#%s\", "
314 "fontcolor = \"#%sDD\"];\n";
315
316 constexpr char kEdgeFormat[] = "\t \"%s\" -> \"%s\";\n";
317
318 constexpr char kRNNBackEdgeFormat[] =
319 "\t \"%s\" -> \"%s\" [color=\"#0F9D58\"];\n";
320
321 std::vector<const Operator*> ops_to_dump = OperatorsToDump(model);
322 std::set<string> already_added_arrays;
323 for (int op_index = 0; op_index < ops_to_dump.size(); op_index++) {
324 const Operator& op = *ops_to_dump[op_index];
325 // Add node for operator.
326 auto op_properties = GetPropertiesForOperator(op);
327 string operator_id = StringF("op%05d", op_index);
328 AppendF(output_file_contents, kNodeFormat, operator_id, op_properties.label,
329 "box", op_properties.color.FillColorString().c_str(),
330 op_properties.color.TextColorString().c_str());
331 // Add nodes and edges for all inputs of the operator.
332 for (const auto& input : op.inputs) {
333 if (!model.HasArray(input)) {
334 // Arrays should _always_ exist. Except, perhaps, during development.
335 continue;
336 }
337 auto array_properties = GetPropertiesForArray(model, input);
338 if (!already_added_arrays.count(input)) {
339 AppendF(output_file_contents, kNodeFormat, input,
340 array_properties.label, "octagon",
341 array_properties.color.FillColorString().c_str(),
342 array_properties.color.TextColorString().c_str());
343 }
344 AppendF(output_file_contents, kEdgeFormat, input, operator_id);
345 already_added_arrays.insert(input);
346 }
347 // Add nodes and edges for all outputs of the operator.
348 for (const auto& output : op.outputs) {
349 if (!model.HasArray(output)) {
350 // Arrays should _always_ exist. Except, perhaps, during development.
351 continue;
352 }
353 auto array_properties = GetPropertiesForArray(model, output);
354 if (!already_added_arrays.count(output)) {
355 AppendF(output_file_contents, kNodeFormat, output,
356 array_properties.label, "octagon",
357 array_properties.color.FillColorString().c_str(),
358 array_properties.color.TextColorString().c_str());
359 }
360 AppendF(output_file_contents, kEdgeFormat, operator_id, output);
361 already_added_arrays.insert(output);
362 }
363 }
364
365 for (const auto& rnn_state : model.flags.rnn_states()) {
366 AppendF(output_file_contents, kRNNBackEdgeFormat,
367 rnn_state.back_edge_source_array(), rnn_state.state_array());
368 }
369
370 AppendF(output_file_contents, "}\n");
371}
