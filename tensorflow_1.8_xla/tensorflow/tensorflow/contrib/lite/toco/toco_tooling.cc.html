36// CHECK-fails if the model contains a kTensorFlowUnsupported operation.
37void CheckUnsupportedOperations(const Model& model) {
38 std::set<string> unsupported_ops;
39 for (auto& op : model.operators) {
40 if (op->type == OperatorType::kTensorFlowUnsupported) {
41 unsupported_ops.insert(
42 static_cast<const TensorFlowUnsupportedOperator*>(op.get())
43 ->tensorflow_op);
44 }
45 }
46 QCHECK(unsupported_ops.empty())
47 << "These unsupported ops were not removed by graph transformations: "
48 << absl::StrJoin(unsupported_ops, ", ");
49}

115bool SupportsQuantization(FileFormat format) {
116 return (format == GRAPHVIZ_DOT || format == TFLITE);
117}
118
119bool SupportsFusedActivationFunction(FileFormat format) {
120 return (format == GRAPHVIZ_DOT || format == TFLITE);
121}
122
123bool SupportsLstmCell(FileFormat format) {
124 return (format == TENSORFLOW_GRAPHDEF || format == GRAPHVIZ_DOT ||
125 format == TFLITE);
126}
127
128bool SupportsPreallocatedWorkspace(FileFormat format) {
129 return (format == TFLITE);
130}

132bool IsRealValued(toco::ArrayDataType type) {
133 // TODO(benoitjacob) - this is hardcoding that uint8 and int16 are only used
134 // for quantized real-number values, and no other integer type is ever used
135 // for that. This is dirty, should be resolved as part of a more general push
136 // to more explicitly distinguish between true-integers and
137 // integers used as quantized values representing real numbers.
138 return static_cast<bool>(type == toco::ArrayDataType::kFloat ||
139 type == toco::ArrayDataType::kUint8 ||
140 type == toco::ArrayDataType::kInt16);
141}

143void SetFinalDataTypeOnInputs(const TocoFlags& toco_flags, Model* model) {
144 const FileFormat output_format = toco_flags.output_format();
145 ArrayDataType type;
146 if (!SupportsQuantization(output_format)) {
147 // Data type is implicitly float for non-quantized formats
148 type = ArrayDataType::kFloat;
149 } else if (toco_flags.has_inference_input_type()) {
150 type = ConvertIODataTypeToArrayDataType(toco_flags.inference_input_type());
151 } else if (toco_flags.has_inference_type()) {
152 type = ConvertIODataTypeToArrayDataType(toco_flags.inference_type());
153 } else {
154 // Nothing to do. Data types stay as-is.
155 return;
156 }
157
158 for (int i = 0; i < model->flags.input_arrays_size(); i++) {
159 string const& array_name = model->flags.input_arrays(i).name();
160 auto* array = &model->GetArray(array_name);
161 // Note that the notion of changing data types only applies to real-numbers
162 // arrays (see the documentation for inference_input_type).
163 // TODO(benoitjacob) this is assuming that uint8 arrays are quantized,
164 // i.e. represent real numbers by means of quantization parameters,
165 // and not plain integer uint8 input arrays.
166 if (!IsRealValued(array->data_type)) {
167 // Ignore non-real data types.
168 continue;
169 }
170
171 array->final_data_type = type;
172 }
173}

177std::unique_ptr<Model> Import(const TocoFlags& toco_flags,
178 const ModelFlags& model_flags,
179 const string& input_file_contents) {
180 std::unique_ptr<Model> model;
181 switch (toco_flags.input_format()) {
182 case TENSORFLOW_GRAPHDEF: {
183 TensorFlowImportFlags tf_import_flags;
184 tf_import_flags.drop_control_dependency =
185 toco_flags.has_drop_control_dependency()
186 ? toco_flags.drop_control_dependency()
187 : (toco_flags.output_format() != TENSORFLOW_GRAPHDEF);
188 model = ImportTensorFlowGraphDef(model_flags, tf_import_flags,
189 input_file_contents);
190 break;
191 }
192 case TFLITE:
193 model = toco::tflite::Import(model_flags, input_file_contents);
194 ResolveModelFlags(model_flags, model.get());
195 CheckInvariants(*model);
196 break;
197 default:
198 LOG(FATAL) << "Unhandled input_format";
199 }
200
201 LogDump(kLogLevelModelChanged, "AT IMPORT", *model);
202
203 return model;
204}

206void Transform(const TocoFlags& toco_flags, Model* model) {
207 const FileFormat output_format = toco_flags.output_format();
208 const IODataType inference_type = toco_flags.inference_type();
209
210 const bool quantize_output =
211 SupportsQuantization(output_format) &&
212 (inference_type == QUANTIZED_UINT8 || inference_type == QUANTIZED_INT16);
213
214 if (quantize_output) {
215 QCHECK_NE(toco_flags.inference_input_type(), FLOAT)
216 << "Quantized inference is not allowed with float inputs.";
217 }
218
219 // Clean up after import.
220 SetFinalDataTypeOnInputs(toco_flags, model);
221 UseArraysExtraInfo(model, quantize_output);
222 FinishBuildingRNNStates(model);
223
224 // Remove unused ops before performing any other optimizations. This is to
225 // stop optimizations from crossing the input/output boundaries. For example
226 // this will stop BatchNorm fusing if the output node is in between a conv
227 // and BatchNorm layers.
228 RunGraphTransformations(model, "Removing unused ops",
229 {new toco::RemoveUnusedOp});
230
231 GraphTransformationsSet transformations;
232 MakeGeneralGraphTransformationsSet(&transformations);
233 auto* remove_trivial_reshape = new RemoveTrivialReshape;
234 transformations.Add(remove_trivial_reshape);
235 if (SupportsFusedActivationFunction(output_format)) {
236 transformations.Add(new FuseActivationFunctions);
237 } else {
238 transformations.Add(new UnfuseActivationFunctions);
239 }
240 if (toco_flags.drop_fake_quant()) {
241 transformations.Add(new DropFakeQuant);
242 } else {
243 // See the doc for --reorder_across_fake_quant: that flag is needed to
244 // support some existing models, e.g. WordLens, that have FakeQuant
245 // nodes in the wrong places.
246 // TODO(benoitjacob): drop special casing when we can.
247 if ((quantize_output && toco_flags.reorder_across_fake_quant())) {
248 transformations.Add(new DropFakeQuant);
249 }
250 }
251 transformations.Add(new ConvertPureConvToDepthwise);
252 if (SupportsLstmCell(output_format)) {
253 if (!toco_flags.debug_disable_recurrent_cell_fusion()) {
254 transformations.Add(new IdentifyLstmCell);
255 }
256 if (output_format == TFLITE) {
257 transformations.Add(new toco::SplitLstmCellInputs);
258 } else {
259 transformations.Add(new toco::MergeLstmCellInputs);
260 }
261 }
262 transformations.Add(new ResolveConstantConcatenation);
263 RunGraphTransformations(model, "general graph transformations",
264 transformations);
265
266 if (quantize_output) {
267 RunGraphTransformations(model, "pre-quantization graph transformations",
268 {new HardcodeMinMax, new DropFakeQuant});
269 }
270
271 if (quantize_output) {
272 if (toco_flags.has_default_ranges_min() &&
273 toco_flags.has_default_ranges_max()) {
274 UseDefaultMinMaxRangeValues(model, toco_flags.default_ranges_min(),
275 toco_flags.default_ranges_max());
276 // The new MinMax info may need to be propagated a bit.
277 RunGraphTransformations(
278 model, "default min-max range propagation graph transformations",
279 {new HardcodeMinMax});
280 }
281 CheckIsReadyForQuantization(*model);
282 RunGraphTransformations(
283 model, "quantization graph transformations",
284 {new Quantize, new RemoveTrivialQuantizedActivationFunc,
285 new RemoveFinalDequantizeOp});
286 } else {
287 GraphTransformationsSet dequantization_transformations{new Dequantize};
288 // Dequantize creates FakeQuant nodes. We may want to discard
289 // those immediately.
290 if (toco_flags.drop_fake_quant()) {
291 dequantization_transformations.Add(new DropFakeQuant);
292 }
293
294 RunGraphTransformations(model, "dequantization graph transformations",
295 dequantization_transformations);
296 }
297
298 if (output_format == TENSORFLOW_GRAPHDEF) {
299 EncodeConstantArraysMinMaxByWrappingThemInFakeQuantNodes(model);
300 }
301
302 // Fix any issues with IO edges. This must happen after any transform that
303 // may modify the structure of the edges.
304 FixEdgeArrays(model);
305
306 LogDump(kLogLevelModelChanged, "AFTER TRANSFORMATIONS", *model);
307
308 if (output_format != GRAPHVIZ_DOT && output_format != TFLITE) {
309 // By now there shouldn't be any unsupported ops when exporting to
310 // TensorFlow GraphDef.
311 CheckUnsupportedOperations(*model);
312 }
313
314 if (SupportsPreallocatedWorkspace(output_format)) {
315 AllocateTransientArrays(model, kDefaultTransientDataAlignment);
316 LogDump(kLogLevelModelChanged, "AFTER ALLOCATION", *model);
317 }
318
319 CheckModelCounts(*model);
320 CheckFinalDataTypesSatisfied(*model);
321
322 int64 ops_count;
323 if (EstimateArithmeticOpsCount(*model, &ops_count)) {
324 LOG(INFO) << "Estimated count of arithmetic ops: " << 1e-9 * ops_count
325 << " billion (note that a multiply-add is counted as 2 ops).";
326 }
327}

329void Export(const TocoFlags& toco_flags, const Model& model,
330 bool allow_custom_ops, string* output_file_contents) {
331 switch (toco_flags.output_format()) {
332 case TENSORFLOW_GRAPHDEF:
333 ExportTensorFlowGraphDef(model, output_file_contents);
334 break;
335 case TFLITE:
336 toco::tflite::Export(model, allow_custom_ops, output_file_contents);
337 break;
338 case GRAPHVIZ_DOT:
339 DumpGraphviz(model, output_file_contents);
340 break;
341 default:
342 LOG(FATAL) << "Unhandled output_format";
343 }
344}
