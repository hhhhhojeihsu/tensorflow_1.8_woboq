111// Given two shapes with potentially different dimensionality and dimension
112// arrays d0 and d1. Without loss of generality, assume that shape0 may have
113// higher dimensionality (length(d0) >= length(d1)). Then shape0 and shape1
114// "agree up to broadcasting" if:
115// - When walking the d0 and d1 from back to front with indices i0, i1,
116// d0[i0] == d1[i1] or d0[i0] == 1 or d1[i1] == 1, for each dimension until
117// i1 == 0 (inclusive).
118bool ShapesAgreeUpToBroadcasting(const Shape& shape0, const Shape& shape1);
119
120// A stricter constraint than ShapesAgreeUpToBroadcasting().
121//
122// Given two shapes with potentially different dimensionality and dimension
123// arrays d0 and d1. Without loss of generality, assume that shape0 may have
124// higher dimensionality (length(d0) >= length(d1)). Then shape0 and shape1
125// "agree up to extending" if:
126// - When walking the d0 and d1 from back to front with indices i0, i1,
127// d0[i0] == d1[i1] for each dimension until i1 == 0 (inclusive).
128// - For the remaining indices [0..i0), d0[i0] == 1.
129bool ShapesAgreeUpToExtending(const Shape& shape0, const Shape& shape1);

189inline int Offset(const Shape& shape, const std::vector<int>& indices) {
190 DCHECK_EQ(shape.dimensions_count(), indices.size());
191 const int dims_count = shape.dimensions_count();
192 int offset = 0;
193 for (int i = 0; i < dims_count; i++) {
194 const int index = indices[i];
195 DCHECK(index >= 0 && index < shape.dims(i));
196 offset *= shape.dims(i);
197 offset += index;
198 }
199 return offset;
200}
201
202inline std::vector<int> ReverseOffset(const Shape& shape, int index) {
203 DCHECK_GE(index, 0);
204 DCHECK_LT(index, RequiredBufferSizeForShape(shape));
205 const int dims_count = shape.dimensions_count();
206 std::vector<int> indices(dims_count);
207 int residual = index;
208 for (int i = dims_count - 1; i >= 0; i--) {
209 indices[i] = residual % shape.dims(i);
210 residual /= shape.dims(i);
211 }
212 return indices;
213}

271// The process of building models varies according to the import format.
272//
273// (a) In some cases, such as model-proto format, the model should be fully
274// specified. In these cases, no extra action should be taken by this function.
275// (b) In other cases, such as TF graphdef format, the desired types of RNN
276// arrays are not specified directly in the model, neither can they be inferred.
277// However, we can set the types of RNN destination arrays to float. This breaks
278// any cycles such as when resolution of the type of an RNN source array depends
279// on the type of its destination array.
280//
281// This function is applied after the main import, after resolution of flags and
282// after application of ArraysExtraInfo. It only defaults destination RNN arrays
283// to float. If the model is subsequently quantized, it is assumed that the
284// model contains sufficient information for that to be completed. If it is
285// already quantized, then case (a) should hold.
286void FinishBuildingRNNStates(Model* model);
