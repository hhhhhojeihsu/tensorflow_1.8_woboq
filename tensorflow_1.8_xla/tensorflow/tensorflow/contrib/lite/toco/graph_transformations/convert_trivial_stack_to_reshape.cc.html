28bool ConvertTrivialStackToReshape::Run(Model* model, std::size_t op_index) {
29 auto stack_it = model->operators.begin() + op_index;
30 if (stack_it->get()->type != OperatorType::kStack) {
31 return false;
32 }
33 auto* stack_op = static_cast<StackOperator*>(stack_it->get());
34 if (stack_op->inputs.size() > 1) {
35 // Not trivial.
36 return false;
37 }
38 CHECK_EQ(stack_op->outputs.size(), 1);
39
40 const auto& input_array = model->GetArray(stack_op->inputs[0]);
41 if (!input_array.has_shape()) {
42 // Yield until input dims have been resolved.
43 return false;
44 }
45 if (input_array.shape().dimensions_count() == 0) {
46 // Input array cannot be 0-D.
47 // (Unsure if this is TF behavior, but was required to get a test to pass.)
48 return false;
49 }
50
51 AddMessageF("Converting trivial %s to a reshape", LogName(*stack_op));
52
53 // Note that we could convert to ExpandDims but toco prefers reshapes.
54 auto* reshape_op = new TensorFlowReshapeOperator;
55 reshape_op->inputs = {stack_op->inputs[0]};
56 reshape_op->outputs = stack_op->outputs;
57
58 // Create shape param.
59 string shape_array_name =
60 AvailableArrayName(*model, stack_op->outputs[0] + "_shape");
61 Array& shape_array = model->GetOrCreateArray(shape_array_name);
62 *(shape_array.mutable_shape()->mutable_dims()) = {
63 1 + input_array.shape().dimensions_count()};
64 reshape_op->inputs.push_back(shape_array_name);
65 shape_array.data_type = ArrayDataType::kInt32;
66 auto& shape_buffer = shape_array.GetMutableBuffer<ArrayDataType::kInt32>();
67 shape_buffer.data.push_back(1);
68 for (int dim : input_array.shape().dims()) {
69 shape_buffer.data.push_back(dim);
70 }
71
72 // Replace the operator in the graph.
73 const auto reshape_it = model->operators.emplace(stack_it, reshape_op);
74 stack_it = reshape_it + 1;
75 CHECK_EQ(stack_it->get(), stack_op);
76 model->operators.erase(stack_it);
77
78 return true;
79}
