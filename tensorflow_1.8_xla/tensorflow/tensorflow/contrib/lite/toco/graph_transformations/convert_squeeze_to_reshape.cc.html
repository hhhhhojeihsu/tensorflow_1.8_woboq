28// Replaces a tf.squeeze operator with a reshape.
29// Squeeze removes dimensions == 1 (if in the list of squeeze_dims). This
30// means that the data layout will never change with this op, just the shape.
31// By converting these to reshapes once we have run shape propagation we allow
32// standard reshape optimization transforms to do their magic.
33bool ConvertSqueezeToReshape::Run(Model* model, std::size_t op_index) {
34 auto squeeze_it = model->operators.begin() + op_index;
35 if (squeeze_it->get()->type != OperatorType::kSqueeze) {
36 return false;
37 }
38 auto squeeze_op = static_cast<SqueezeOperator*>(squeeze_it->get());
39 CHECK_EQ(squeeze_op->inputs.size(), 1);
40 CHECK_EQ(squeeze_op->outputs.size(), 1);
41
42 const auto& input_array = model->GetArray(squeeze_op->inputs[0]);
43 if (!input_array.has_shape()) {
44 // Yield until input dims have been resolved.
45 return false;
46 }
47 if (input_array.shape().dimensions_count() == 0) {
48 // Input array cannot be 0-D.
49 return false;
50 }
51 if (!model->HasArray(squeeze_op->outputs[0]) ||
52 !model->GetArray(squeeze_op->outputs[0]).has_shape()) {
53 // Yield until shape propagation has set the output shape for us.
54 return false;
55 }
56
57 // We use the output shape that has been calculated by shape propagation.
58 const auto& output_shape = model->GetArray(squeeze_op->outputs[0]).shape();
59
60 // Empty shapes will not work as empty data arrays.
61 if (output_shape.dimensions_count() == 0) {
62 return false;
63 }
64
65 auto* reshape_op = new TensorFlowReshapeOperator;
66 reshape_op->inputs = {
67 squeeze_op->inputs[0],
68 CreateInt32Array(model, squeeze_op->outputs[0] + "_shape",
69 output_shape.dims()),
70 };
71 reshape_op->outputs = squeeze_op->outputs;
72
73 AddMessageF("Replacing %s with %s", LogName(*squeeze_op),
74 LogName(*reshape_op));
75
76 // Replace the operator in the graph.
77 const auto reshape_it = model->operators.emplace(squeeze_it, reshape_op);
78 squeeze_it = reshape_it + 1;
79 CHECK_EQ(squeeze_it->get(), squeeze_op);
80 model->operators.erase(squeeze_it);
81
82 return true;
83}
