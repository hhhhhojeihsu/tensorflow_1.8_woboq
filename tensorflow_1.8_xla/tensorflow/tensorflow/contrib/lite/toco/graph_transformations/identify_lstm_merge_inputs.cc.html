28bool MergeLstmCellInputs::Run(Model* model, std::size_t op_index) {
29 // Find lstm cell.
30 auto op_it = model->operators.begin() + op_index;
31 auto src_op = op_it->get();
32 if (src_op->type != OperatorType::kLstmCell) {
33 return false;
34 }
35
36 // Already a compact LstmCell with LstmCellOperator::NUM_INPUTS of inputs,
37 // do not need to merge cell inputs.
38 if (src_op->inputs.size() == LstmCellOperator::NUM_INPUTS) {
39 return false;
40 }
41
42 // Identify prev_activ_input, prev_state_input as required Op inputs,
43 // using the rnn_states in the model flag.
44 string prev_activ_input;
45 if (!GetMatchingRnnArray(model, src_op->outputs[kOutputTensor],
46 &prev_activ_input)) {
47 return false;
48 }
49 string prev_state_input;
50 if (!GetMatchingRnnArray(model, src_op->outputs[kCellStateTensor],
51 &prev_state_input)) {
52 return false;
53 }
54
55 // Get LstmCell's cell, input, output size.
56 int num_cell = model->GetArray(src_op->inputs[kInputToInputWeightsTensor])
57 .shape()
58 .dims(0);
59 int num_input = model->GetArray(src_op->inputs[kInputToInputWeightsTensor])
60 .shape()
61 .dims(1);
62 int num_output =
63 model->GetArray(src_op->inputs[kRecurrentToInputWeightsTensor])
64 .shape()
65 .dims(1);
66
67 // Make sure n_cell and n_output are equal as there is no projection.
68 CHECK_EQ(num_cell, num_output);
69
70 // Create tensorflow_graphdef style's one big weight tensor.
71 const string base_name(FindLongestCommonPrefix(
72 src_op->outputs[kOutputTensor], src_op->outputs[kCellStateTensor]));
73 string merged_weights = AvailableArrayName(*model, base_name + "weights");
74 auto& array = model->GetOrCreateArray(merged_weights);
75 array.data_type = ArrayDataType::kFloat;
76 int weights_dim1 = 4 * num_cell;
77 int weights_dim2 = num_input + num_output;
78 Shape shape = Shape({weights_dim1, weights_dim2});
79 array.copy_shape(shape);
80 auto& buffer = array.GetMutableBuffer<ArrayDataType::kFloat>();
81 buffer.data.resize(weights_dim1 * weights_dim2);
82
83 // Merge 8 small weight tensors to 1 weight tensor.
84 CopyArrayToSubArray(
85 buffer, weights_dim2,
86 model->GetArray(src_op->inputs[kInputToInputWeightsTensor]), 0, 0);
87 CopyArrayToSubArray(
88 buffer, weights_dim2,
89 model->GetArray(src_op->inputs[kInputToCellWeightsTensor]), num_cell, 0);
90 CopyArrayToSubArray(
91 buffer, weights_dim2,
92 model->GetArray(src_op->inputs[kInputToForgetWeightsTensor]),
93 num_cell * 2, 0);
94 CopyArrayToSubArray(
95 buffer, weights_dim2,
96 model->GetArray(src_op->inputs[kInputToOutputWeightsTensor]),
97 num_cell * 3, 0);
98 CopyArrayToSubArray(
99 buffer, weights_dim2,
100 model->GetArray(src_op->inputs[kRecurrentToInputWeightsTensor]), 0,
101 num_input);
102 CopyArrayToSubArray(
103 buffer, weights_dim2,
104 model->GetArray(src_op->inputs[kRecurrentToCellWeightsTensor]), num_cell,
105 num_input);
106 CopyArrayToSubArray(
107 buffer, weights_dim2,
108 model->GetArray(src_op->inputs[kRecurrentToForgetWeightsTensor]),
109 num_cell * 2, num_input);
110 CopyArrayToSubArray(
111 buffer, weights_dim2,
112 model->GetArray(src_op->inputs[kRecurrentToOutputWeightsTensor]),
113 num_cell * 3, num_input);
114
115 // Create tensorflow_graphdef style's one big bias tensor.
116 string merged_biases = AvailableArrayName(*model, base_name + "biases");
117 auto& bias_array = model->GetOrCreateArray(merged_biases);
118 bias_array.data_type = ArrayDataType::kFloat;
119 bias_array.copy_shape(Shape({weights_dim1}));
120 auto& bias_buffer = bias_array.GetMutableBuffer<ArrayDataType::kFloat>();
121 bias_buffer.data.resize(weights_dim1);
122
123 // Merge 4 small bias tensors into a big one.
124 CopyArrayToSubArray(bias_buffer, weights_dim2,
125 model->GetArray(src_op->inputs[kInputGateBiasTensor]), 0,
126 0);
127 CopyArrayToSubArray(bias_buffer, weights_dim2,
128 model->GetArray(src_op->inputs[kCellGateBiasTensor]),
129 num_cell, 0);
130 CopyArrayToSubArray(bias_buffer, weights_dim2,
131 model->GetArray(src_op->inputs[kForgetGateBiasTensor]),
132 num_cell * 2, 0);
133 CopyArrayToSubArray(bias_buffer, weights_dim2,
134 model->GetArray(src_op->inputs[kOutputGateBiasTensor]),
135 num_cell * 3, 0);
136
137 // Emplace a new LSTM cell operator (use basic 5 inputs kernel).
138 auto lstm_cell_op = absl::make_unique<LstmCellOperator>();
139
140 // Compact LstmCell's 5 inputs.
141 lstm_cell_op->inputs.resize(LstmCellOperator::NUM_INPUTS);
142 lstm_cell_op->inputs[LstmCellOperator::DATA_INPUT] =
143 src_op->inputs[kInputTensor];
144 lstm_cell_op->inputs[LstmCellOperator::WEIGHTS_INPUT] = merged_weights;
145 lstm_cell_op->inputs[LstmCellOperator::BIASES_INPUT] = merged_biases;
146 lstm_cell_op->inputs[LstmCellOperator::PREV_ACTIV_INPUT] = prev_activ_input;
147 lstm_cell_op->inputs[LstmCellOperator::PREV_STATE_INPUT] = prev_state_input;
148
149 // Reorder LstmCell's 4 outputs.
150 lstm_cell_op->outputs.resize(LstmCellOperator::NUM_OUTPUTS);
151 lstm_cell_op->outputs[LstmCellOperator::ACTIV_OUTPUT] =
152 src_op->outputs[kOutputTensor];
153 lstm_cell_op->outputs[LstmCellOperator::STATE_OUTPUT] =
154 src_op->outputs[kCellStateTensor];
155 lstm_cell_op->outputs[LstmCellOperator::CONCAT_TEMP] =
156 src_op->outputs[kScratchBufferTensor];
157 lstm_cell_op->outputs[LstmCellOperator::ACTIV_TEMP] =
158 src_op->outputs[kOutputStateTensor];
159
160 // Add the op into model.
161 model->operators.emplace(op_it, std::move(lstm_cell_op));
162 AddMessageF("Creating compact LstmCell replacing previous lstm cell");
163
164 // Delete arrays and operators replaced by the LSTM cell operator. Order is
165 // important - DeleteArrayIfUnused() only succeeds if dependent operators
166 // have been removed first. Start at the output and work towards the input.
167 // Erase curr lstm op being replaced.
168 DeleteArrayIfUnused(src_op->inputs[kInputToInputWeightsTensor], model);
169 DeleteArrayIfUnused(src_op->inputs[kInputToForgetWeightsTensor], model);
170 DeleteArrayIfUnused(src_op->inputs[kInputToCellWeightsTensor], model);
171 DeleteArrayIfUnused(src_op->inputs[kInputToOutputWeightsTensor], model);
172 DeleteArrayIfUnused(src_op->inputs[kRecurrentToInputWeightsTensor], model);
173 DeleteArrayIfUnused(src_op->inputs[kRecurrentToForgetWeightsTensor], model);
174 DeleteArrayIfUnused(src_op->inputs[kRecurrentToCellWeightsTensor], model);
175 DeleteArrayIfUnused(src_op->inputs[kRecurrentToOutputWeightsTensor], model);
176 DeleteArrayIfUnused(src_op->inputs[kInputGateBiasTensor], model);
177 DeleteArrayIfUnused(src_op->inputs[kForgetGateBiasTensor], model);
178 DeleteArrayIfUnused(src_op->inputs[kCellGateBiasTensor], model);
179 DeleteArrayIfUnused(src_op->inputs[kOutputGateBiasTensor], model);
180 model->operators.erase(FindOp(*model, src_op));
181
182 return true;
183}
