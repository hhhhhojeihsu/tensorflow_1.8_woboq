29bool PropagateActivationFunctionIntoConstants::Run(Model* model,
30 std::size_t op_index) {
31 const auto ac_it = model->operators.begin() + op_index;
32 const auto* ac_op = ac_it->get();
33 if (ac_op->type != OperatorType::kRelu6 &&
34 ac_op->type != OperatorType::kRelu1 &&
35 ac_op->type != OperatorType::kRelu) {
36 return false;
37 }
38
39 // Find the op producing the array passed to this activation function.
40 auto* src_op = GetOpWithOutput(*model, ac_op->inputs[0]);
41 if (!src_op) {
42 return false;
43 }
44
45 // Ensure the src_op is not used without the activation function applied.
46 if (CountTrueOutputs(*model, *src_op) > 1) {
47 AddMessageF(
48 "Not propagating activation function %s into %s because it has more "
49 "than one consumed output",
50 LogName(*ac_op), LogName(*src_op));
51 }
52
53 // Filter to the list of supported ops.
54 string src_op_input;
55 switch (src_op->type) {
56 case OperatorType::kGather:
57 src_op_input = src_op->inputs[0];
58 break;
59 default:
60 return false;
61 }
62 CHECK_EQ(src_op->outputs[0], ac_op->inputs[0]);
63
64 // Ensure the input is constant as otherwise this needs to happen at runtime.
65 // If we bail here, it's still possible that FuseActivationFunctions will fuse
66 // the activation if it's supported by the op.
67 if (!IsConstantParameterArray(*model, src_op_input)) {
68 AddMessageF(
69 "Not propagating activation function %s into %s:%s because it is not "
70 "constant",
71 LogName(*ac_op), LogName(*src_op), src_op_input);
72 return false;
73 }
74
75 // Get the array we'll be working with and ensure it's a compatible type.
76 auto& const_array = model->GetArray(src_op_input);
77 if (const_array.data_type != ArrayDataType::kFloat) {
78 AddMessageF(
79 "Not propagating activation function %s into %s:%s because it is "
80 "non-float data",
81 LogName(*ac_op), LogName(*src_op), src_op_input);
82 return false;
83 }
84 auto& const_array_data =
85 const_array.GetMutableBuffer<ArrayDataType::kFloat>().data;
86
87 // Perform the activation function directly into the constant data array.
88 for (size_t i = 0; i < const_array_data.size(); ++i) {
89 const float value = const_array_data[i];
90 float new_value = value;
91 switch (ac_op->type) {
92 case OperatorType::kRelu: {
93 static constexpr float kLower = 0;
94 new_value = value < kLower ? kLower : value;
95 break;
96 }
97 case OperatorType::kRelu1: {
98 static constexpr float kUpper = 1;
99 static constexpr float kLower = -1;
100 new_value = value > kUpper ? kUpper : value < kLower ? kLower : value;
101 break;
102 }
103 case OperatorType::kRelu6: {
104 static constexpr float kUpper = 6;
105 static constexpr float kLower = 0;
106 new_value = value > kUpper ? kUpper : value < kLower ? kLower : value;
107 break;
108 }
109 default:
110 LOG(FATAL) << "Unsupported activation function " << LogName(*ac_op);
111 return false;
112 }
113 const_array_data[i] = new_value;
114 }
115
116 AddMessageF("Propagated activation function %s into %s:%s", LogName(*ac_op),
117 LogName(*src_op), src_op_input);
118 return RemoveTrivialPassthroughOp(this, model, op_index);
119}
