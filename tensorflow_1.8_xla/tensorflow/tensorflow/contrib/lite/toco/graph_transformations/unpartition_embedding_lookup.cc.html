25bool UnpartitionEmbeddingLookup::Run(Model* model, std::size_t op_index) {
26 // Collapses a partitioned tf.nn.embedding_lookup back into a single Gather.
27 // https://www.tensorflow.org/api_docs/python/tf/nn/embedding_lookup
28 // This transform attempts to identify the len(params) > 1 case and collapse
29 // it to the len(params) = 1 case by concatenating the original params and
30 // reversing the partitioning.
31 //
32 // If len(params) to the tf.nn.embedding_lookup == 1, the whole op becomes
33 // simply a gather:
34 // https://github.com/tensorflow/tensorflow/blob/r1.5/tensorflow/python/ops/embedding_ops.py#L150
35 //
36 // Notes on this implementation:
37 // - only supports partition_strategy='mod'
38 //
39 // A rough graph of a partitioned embedding_lookup looks like:
40 // (ids)--+-->FloorDiv--+-->DynamicPartition-->[[Gather]]--\
41 // \-->FloorMod--/ |
42 // V |
43 // Range-->DynamicPartition-------->DynamicStitch<---------/
44 // (const) V
45 // (embeddings)
46
47 // First look for the final DynamicStitch.
48 auto op_it = model->operators.begin() + op_index;
49 if (op_it->get()->type != OperatorType::kDynamicStitch) {
50 return false;
51 }
52 auto* stitch_op = static_cast<DynamicStitchOperator*>(op_it->get());
53
54 // Split up the DynamicStitch inputs into the indices and data.
55 std::vector<string> stitch_indices_inputs;
56 std::vector<string> stitch_data_inputs;
57 for (size_t i = 0; i < stitch_op->num_partitions; ++i) {
58 stitch_indices_inputs.push_back(stitch_op->inputs[i]);
59 }
60 for (size_t i = stitch_op->num_partitions; i < stitch_op->num_partitions * 2;
61 ++i) {
62 stitch_data_inputs.push_back(stitch_op->inputs[i]);
63 }
64
65 // Validate all indices come from the same DynamicPartition.
66 DynamicPartitionOperator* indices_partition_op = nullptr;
67 for (const string& indices_partition_output_name : stitch_indices_inputs) {
68 auto* op = GetOpWithOutput(*model, indices_partition_output_name);
69 CHECK(op) << "Source of " << indices_partition_output_name << " not found";
70 if (op->type != OperatorType::kDynamicPartition) {
71 AddMessageF(
72 "Skipping because indices input %s into "
73 "%s is unexpected",
74 LogName(*op), LogName(*stitch_op));
75 return false;
76 }
77 if (!indices_partition_op) {
78 indices_partition_op = static_cast<DynamicPartitionOperator*>(op);
79 } else {
80 // Ensure this is the same op as previous ones.
81 if (op != indices_partition_op) {
82 AddMessageF(
83 "Skipping because indices input %s into "
84 "%s is from a different source op than others",
85 LogName(*op), LogName(*stitch_op));
86 return false;
87 }
88 }
89 }
90 CHECK(indices_partition_op) << "No indices inputs";
91
92 // The data for the indices must be a constant range of the array shape.
93 if (!IsConstantParameterArray(*model, indices_partition_op->inputs[0])) {
94 AddMessageF("Skipping because indices partition data is non-constant");
95 return false;
96 }
97 auto& indices_data_array = model->GetArray(indices_partition_op->inputs[0]);
98 if (indices_data_array.data_type == ArrayDataType::kNone) {
99 // Yield until data types are propagated.
100 return false;
101 }
102 CHECK(indices_data_array.data_type == ArrayDataType::kInt32)
103 << "Indices partition inputs must be int32";
104 const auto& indices_data_buffer =
105 indices_data_array.GetBuffer<ArrayDataType::kInt32>().data;
106 for (size_t i = 0; i < indices_data_buffer.size(); ++i) {
107 CHECK_EQ(indices_data_buffer[i], i) << "Indices range must be identity";
108 }
109
110 // Find all of the gathers used for the data inputs.
111 std::vector<GatherOperator*> gather_ops;
112 for (const string& gather_output_name : stitch_data_inputs) {
113 auto* op = GetOpWithOutput(*model, gather_output_name);
114 CHECK(op) << "Source of " << gather_output_name << " not found";
115 if (op->type != OperatorType::kGather) {
116 AddMessageF(
117 "Skipping because data input %s into %s "
118 "is unexpected",
119 LogName(*op), LogName(*stitch_op));
120 return false;
121 }
122 gather_ops.push_back(static_cast<GatherOperator*>(op));
123 }
124
125 // Validate all gathers come from the same DynamicPartition.
126 DynamicPartitionOperator* data_partition_op = nullptr;
127 for (auto* gather_op : gather_ops) {
128 auto* op = GetOpWithOutput(*model, gather_op->inputs[1]);
129 CHECK(op) << "Source of " << gather_op->inputs[1] << " not found";
130 if (op->type != OperatorType::kDynamicPartition) {
131 AddMessageF(
132 "Skipping because data input %s into "
133 "%s is unexpected",
134 LogName(*op), LogName(*gather_op));
135 return false;
136 }
137 if (!data_partition_op) {
138 data_partition_op = static_cast<DynamicPartitionOperator*>(op);
139 } else {
140 // Ensure this is the same op as previous ones.
141 if (op != data_partition_op) {
142 AddMessageF(
143 "Skipping because data input %s into "
144 "%s is from a different source op than others",
145 LogName(*op), LogName(*gather_op));
146 return false;
147 }
148 }
149 }
150 CHECK(data_partition_op) << "No data inputs";
151
152 // Validate the partition ops have the same sizes.
153 CHECK_EQ(indices_partition_op->num_partitions,
154 data_partition_op->num_partitions)
155 << "Indices and data partition ops have differing dimensions";
156 int num_partitions = indices_partition_op->num_partitions;
157
158 // Partition strategy of 'mod' gives us a FloorMod and FloorDiv.
159 // The gather partition uses the FloorDiv as the data and FloorMod as the
160 // partitions and the indices use the FloorMod as their partitions.
161 Operator* div_op = GetOpWithOutput(*model, data_partition_op->inputs[0]);
162 Operator* mod_op = GetOpWithOutput(*model, data_partition_op->inputs[1]);
163 CHECK(div_op && div_op->type == OperatorType::kFloorDiv)
164 << "Unsupported partition strategy";
165 CHECK(mod_op && mod_op->type == OperatorType::kFloorMod)
166 << "Unsupported partition strategy";
167 CHECK_EQ(mod_op, GetOpWithOutput(*model, indices_partition_op->inputs[1]))
168 << "Indices and data partition ops require the same partition strategy "
169 "and inputs";
170
171 // Glob together all of the gather data. This is not yet in the correct order.
172 auto* gather_params_concat_op = new ConcatenationOperator;
173 for (const auto& gather_op : gather_ops) {
174 gather_params_concat_op->inputs.push_back(gather_op->inputs[0]);
175 }
176 gather_params_concat_op->outputs.push_back(
177 AvailableArrayName(*model, gather_ops[0]->inputs[0] + "_unpartitioned"));
178 op_it = model->operators.emplace(op_it, gather_params_concat_op) + 1;
179 model->GetOrCreateArray(gather_params_concat_op->outputs[0]);
180
181 // Permute the gather params to undo the partitioning that was originally
182 // done.
183 auto* gather_params_permute_op = new GatherOperator;
184 gather_params_permute_op->inputs.push_back(
185 gather_params_concat_op->outputs[0]);
186 gather_params_permute_op->inputs.push_back(
187 AvailableArrayName(*model, gather_ops[0]->inputs[0] + "_permuted/perm"));
188 gather_params_permute_op->outputs.push_back(
189 AvailableArrayName(*model, gather_ops[0]->inputs[0] + "_permuted"));
190 op_it = model->operators.emplace(op_it, gather_params_permute_op) + 1;
191 model->GetOrCreateArray(gather_params_permute_op->outputs[0]);
192 const auto& partition_array = model->GetArray(gather_ops[0]->inputs[0]);
193 const auto& partition_array_dims = partition_array.shape().dims();
194 gather_params_permute_op->input_rank =
195 partition_array.shape().dimensions_count();
196 auto& perm_array =
197 model->GetOrCreateArray(gather_params_permute_op->inputs[1]);
198 perm_array.data_type = ArrayDataType::kInt32;
199 perm_array.mutable_shape()->ReplaceDims(
200 {num_partitions * partition_array_dims[0]});
201 auto& perm_data = perm_array.GetMutableBuffer<ArrayDataType::kInt32>().data;
202 perm_data.resize(RequiredBufferSizeForShape(perm_array.shape()));
203 // NOTE: this is what relies on the partition_strategy.
204 for (int i = 0; i < num_partitions * partition_array_dims[0]; ++i) {
205 int p = i % num_partitions;
206 perm_data[i] = p * partition_array_dims[0] + i / num_partitions;
207 }
208
209 // Insert the new unpartitioned gather op.
210 auto* merged_gather_op = new GatherOperator;
211 merged_gather_op->inputs = {gather_params_permute_op->outputs[0],
212 mod_op->inputs[0]};
213 merged_gather_op->outputs = {stitch_op->outputs[0]};
214 merged_gather_op->input_rank = partition_array.shape().dimensions_count();
215 model->operators.emplace(op_it, merged_gather_op);
216
217 AddMessageF(
218 "Replacing suspected partitioned tf.nn.embedding_lookup (starting at %s "
219 "+ %s and ending at %s) with a single unpartitioned gather %s",
220 LogName(*div_op), LogName(*mod_op), LogName(*stitch_op),
221 LogName(*merged_gather_op));
222
223 // Ensure the stitch output array is dead, as we don't want whatever was in it
224 // previously now that we've redefined it. It'll be recreated when needed.
225 model->EraseArray(stitch_op->outputs[0]);
226 model->GetOrCreateArray(merged_gather_op->outputs[0]);
227
228 // Erase all the original ops.
229 DeleteOpAndArraysIfUnused(model, div_op);
230 DeleteOpAndArraysIfUnused(model, mod_op);
231 for (auto* gather_op : gather_ops) {
232 DeleteOpAndArraysIfUnused(model, gather_op);
233 }
234 DeleteOpAndArraysIfUnused(model, indices_partition_op);
235 DeleteOpAndArraysIfUnused(model, data_partition_op);
236 DeleteOpAndArraysIfUnused(model, stitch_op);
237 return true;
238}
