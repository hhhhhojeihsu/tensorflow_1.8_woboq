31bool IsSliceTrivial(const Model& model, const Operator& op,
32 RemoveTrivialSlice* transformation) {
33 CHECK(op.type == OperatorType::kSlice);
34
35 // Slices are trivial if they are slicing the entire input contents.
36 const auto& input_array = model.GetArray(op.inputs[0]);
37 const auto& output_array = model.GetArray(op.outputs[0]);
38 if (input_array.has_shape() && output_array.has_shape()) {
39 if (input_array.shape() == output_array.shape()) {
40 transformation->AddMessageF(
41 "%s is trivial because its input and output shapes are equal",
42 LogName(op));
43 return true;
44 }
45 }
46
47 return false;
48}
49
50} // namespace
51
52bool RemoveTrivialSlice::Run(Model* model, std::size_t op_index) {
53 const auto reshape_it = model->operators.begin() + op_index;
54 auto* slice_op = reshape_it->get();
55 if (slice_op->type != OperatorType::kSlice) {
56 return false;
57 }
58
59 if (!IsSliceTrivial(*model, *slice_op, this)) {
60 return false;
61 }
62
63 AddMessageF("Removing trivial %s", LogName(*slice_op));
64
65 CHECK_EQ(slice_op->inputs.size(), 3);
66 return RemoveTrivialPassthroughOp(this, model, op_index);
67}
