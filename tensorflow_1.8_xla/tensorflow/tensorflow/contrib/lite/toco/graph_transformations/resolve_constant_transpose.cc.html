26// Transposes an array up to rank 4.
27// This is ShuffleArrayTemplate with non-enum permutation.
28template <ArrayDataType Type>
29void Transpose(Model* model, const Array& input_array,
30 const std::vector<int>& perm, Array* output_array) {
31 const Shape& input_shape = input_array.shape();
32 const std::vector<DataType<Type>>& input_data =
33 input_array.GetBuffer<Type>().data;
34
35 const Shape& output_shape = output_array->shape();
36 std::vector<DataType<Type>>& output_data =
37 output_array->GetMutableBuffer<Type>().data;
38 output_data.resize(RequiredBufferSizeForShape(output_shape));
39
40 CHECK(input_shape.dimensions_count() == output_shape.dimensions_count());
41 const int dim = input_shape.dimensions_count();
42 CHECK_LE(dim, 4);
43 CHECK(perm.size() >= dim);
44 for (int i = 0; i < dim; i++) {
45 CHECK(perm[i] >= 0 && perm[i] < dim);
46 CHECK(input_shape.dims(perm[i]) == output_shape.dims(i));
47 }
48 Shape extended_input_shape = input_shape;
49 ExtendShape(&extended_input_shape, 4);
50 Shape extended_output_shape = output_shape;
51 ExtendShape(&extended_output_shape, 4);
52 std::vector<int> extended_perm;
53 ExtendShuffle(perm, 4, &extended_perm);
54
55 const std::vector<int>& extended_input_dims = extended_input_shape.dims();
56 const std::vector<int>& extended_output_dims = extended_output_shape.dims();
57
58 // TODO(starka): Rework to handle different numbers of dimensions.
59 int input_strides[4];
60 input_strides[3] = 1;
61 input_strides[2] = extended_input_dims[3];
62 input_strides[1] = input_strides[2] * extended_input_dims[2];
63 input_strides[0] = input_strides[1] * extended_input_dims[1];
64 const int input_stride_0 = input_strides[extended_perm[3]];
65 const int input_stride_1 = input_strides[extended_perm[2]];
66 const int input_stride_2 = input_strides[extended_perm[1]];
67 const int input_stride_3 = input_strides[extended_perm[0]];
68
69 const int output_size_0 = extended_output_dims[3];
70 const int output_size_1 = extended_output_dims[2];
71 const int output_size_2 = extended_output_dims[1];
72 const int output_size_3 = extended_output_dims[0];
73 const int output_stride_0 = 1;
74 const int output_stride_1 = output_size_0;
75 const int output_stride_2 = output_stride_1 * output_size_1;
76 const int output_stride_3 = output_stride_2 * output_size_2;
77
78 for (int i3 = 0; i3 < output_size_3; i3++) {
79 const DataType<Type>* const input_ptr_3 =
80 input_data.data() + i3 * input_stride_3;
81 DataType<Type>* const output_ptr_3 =
82 output_data.data() + i3 * output_stride_3;
83 for (int i2 = 0; i2 < output_size_2; i2++) {
84 const DataType<Type>* const input_ptr_2 =
85 input_ptr_3 + i2 * input_stride_2;
86 DataType<Type>* const output_ptr_2 = output_ptr_3 + i2 * output_stride_2;
87 for (int i1 = 0; i1 < output_size_1; i1++) {
88 const DataType<Type>* input_ptr = input_ptr_2 + i1 * input_stride_1;
89 DataType<Type>* output_ptr = output_ptr_2 + i1 * output_stride_1;
90 DataType<Type>* const output_ptr_end =
91 output_ptr + output_size_0 * output_stride_0;
92 while (output_ptr != output_ptr_end) {
93 *output_ptr = *input_ptr;
94 input_ptr += input_stride_0;
95 output_ptr += output_stride_0;
96 }
97 }
98 }
99 }
100}

104bool ResolveConstantTranspose::Run(Model* model, std::size_t op_index) {
105 auto it = model->operators.begin() + op_index;
106 const auto* base_op = it->get();
107 if (base_op->type != OperatorType::kTranspose) {
108 return false;
109 }
110 const auto* op = static_cast<const TransposeOperator*>(base_op);
111
112 CHECK_EQ(op->inputs.size(), 2);
113 CHECK_EQ(op->outputs.size(), 1);
114 auto& output_array = model->GetArray(op->outputs[0]);
115 if (output_array.data_type == ArrayDataType::kNone) {
116 // Yield until the output type has been set by PropagateArrayDataTypes.
117 return false;
118 }
119 if (!output_array.has_shape()) {
120 // Yield until the output shape has been set by PropagateFixedShapes.
121 return false;
122 }
123
124 // We require constant inputs.
125 if (!IsConstantParameterArray(*model, op->inputs[0]) ||
126 !IsConstantParameterArray(*model, op->inputs[1])) {
127 return false;
128 }
129 const Array& input_array = model->GetArray(op->inputs[0]);
130
131 if (input_array.minmax) {
132 output_array.GetOrCreateMinMax() = input_array.GetMinMax();
133 }
134
135 if (op->perm.empty()) {
136 // Yield until perm has been populated by ResolveTransposeAttributes.
137 return false;
138 }
139
140 // We currently only support 1-4 dimensions.
141 CHECK_LE(op->perm.size(), 4);
142
143 CHECK(!output_array.buffer);
144 switch (output_array.data_type) {
145 case ArrayDataType::kFloat:
146 Transpose<ArrayDataType::kFloat>(model, input_array, op->perm,
147 &output_array);
148 break;
149 case ArrayDataType::kUint8:
150 Transpose<ArrayDataType::kUint8>(model, input_array, op->perm,
151 &output_array);
152 break;
153 case ArrayDataType::kInt32:
154 Transpose<ArrayDataType::kInt32>(model, input_array, op->perm,
155 &output_array);
156 break;
157 case ArrayDataType::kInt64:
158 Transpose<ArrayDataType::kInt64>(model, input_array, op->perm,
159 &output_array);
160 break;
161 default:
162 LOG(FATAL) << "Unsupported data type given to Transpose op with output \""
163 << op->outputs[0] << "\"";
164 break;
165 }
166
167 // Erase input arrays if no longer used.
168 for (const auto& input : op->inputs) {
169 if (IsDiscardableArray(*model, input) &&
170 CountOpsWithInput(*model, input) == 1) {
171 model->EraseArray(input);
172 }
173 }
174
175 // Erase the operator.
176 model->operators.erase(it);
177 return true;
178}
