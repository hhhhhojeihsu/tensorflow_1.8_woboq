27void SetDataTypeForAllOutputs(Model* model, Operator* op,
28 ArrayDataType data_type) {
29 for (const auto& output : op->outputs) {
30 model->GetArray(output).data_type = data_type;
31 }
32}
33} // namespace
34
35bool PropagateArrayDataTypes::Run(Model* model, std::size_t op_index) {
36 auto it = model->operators.begin() + op_index;
37 auto* op = it->get();
38
39 // If the data type of some input is unknown, we need to yield.
40 for (const auto& input : op->inputs) {
41 if (!model->IsOptionalArray(input) &&
42 model->GetArray(input).data_type == ArrayDataType::kNone) {
43 return false;
44 }
45 }
46 // Record data types of output before processing, so we can see at the
47 // end if we changed anything, and return the correct boolean value.
48 std::unordered_map<string, ArrayDataType> old_output_data_types;
49 for (const auto& output : op->outputs) {
50 old_output_data_types[output] = model->GetArray(output).data_type;
51 }
52 // Do the actual output data types propagation.
53 switch (op->type) {
54 case OperatorType::kDequantize:
55 case OperatorType::kResizeBilinear:
56 // These operators unconditionally produce float outputs
57 SetDataTypeForAllOutputs(model, op, ArrayDataType::kFloat);
58 break;
59 case OperatorType::kTensorFlowLess:
60 case OperatorType::kTensorFlowLessEqual:
61 case OperatorType::kTensorFlowGreater:
62 case OperatorType::kTensorFlowGreaterEqual:
63 // These operators unconditionally produce bool outputs
64 SetDataTypeForAllOutputs(model, op, ArrayDataType::kBool);
65 break;
66 case OperatorType::kRank:
67 case OperatorType::kTensorFlowShape:
68 // These operators only produce int32 outputs.
69 SetDataTypeForAllOutputs(model, op, ArrayDataType::kInt32);
70 break;
71 case OperatorType::kTensorFlowSplit:
72 case OperatorType::kTensorFlowConcat:
73 case OperatorType::kFill: {
74 // These operators produce an output with the same type as their 2nd input
75 CHECK_GE(op->inputs.size(), 2);
76 const ArrayDataType data_type = model->GetArray(op->inputs[1]).data_type;
77 SetDataTypeForAllOutputs(model, op, data_type);
78 break;
79 }
80 case OperatorType::kTransposeConv: {
81 // These operators produce an output with the same type as their 3rd input
82 CHECK_GE(op->inputs.size(), 3);
83 const ArrayDataType data_type = model->GetArray(op->inputs[2]).data_type;
84 SetDataTypeForAllOutputs(model, op, data_type);
85 break;
86 }
87 case OperatorType::kCast: {
88 // Data type of the Cast op is specified.
89 CHECK_EQ(op->outputs.size(), 1);
90 auto* cast_op = static_cast<CastOperator*>(op);
91 model->GetArray(op->outputs[0]).data_type = cast_op->dst_data_type;
92 break;
93 }
94 case OperatorType::kArgMax: {
95 // Data type of the ArgMax op is specified.
96 CHECK_EQ(op->outputs.size(), 1);
97 auto* argmax_op = static_cast<ArgMaxOperator*>(op);
98 model->GetArray(op->outputs[0]).data_type = argmax_op->output_data_type;
99 break;
100 }
101 case OperatorType::kRange: {
102 auto* range_op = static_cast<RangeOperator*>(op);
103 // Output type of the Range op can be set via an attribute
104 ArrayDataType data_type;
105 if (range_op->dtype != ArrayDataType::kNone) {
106 // Use the type if specified
107 data_type = range_op->dtype;
108 } else {
109 // Otherwise use the first input
110 CHECK_GE(op->inputs.size(), 1);
111 data_type = model->GetArray(op->inputs[0]).data_type;
112 }
113 CHECK_EQ(op->outputs.size(), 1);
114 SetDataTypeForAllOutputs(model, op, data_type);
115 break;
116 }
117 case OperatorType::kRandomUniform: {
118 auto* rand_op = static_cast<RandomUniformOperator*>(op);
119 // The output type of RandomUniform is specified with an attribute
120 if (rand_op->dtype == ArrayDataType::kNone) {
121 return false;
122 }
123 CHECK_EQ(op->outputs.size(), 1);
124 SetDataTypeForAllOutputs(model, op, rand_op->dtype);
125 break;
126 }
127 case OperatorType::kTensorFlowUnsupported: {
128 auto* unsupported_op = static_cast<TensorFlowUnsupportedOperator*>(op);
129 // Some output tensors from the op could be eliminated by optimization.
130 // This can make unsupported_op->output_data_types have more elements than
131 // op->outputs.
132 if (unsupported_op->output_data_types.size() < op->outputs.size()) {
133 return false;
134 }
135 for (int i = 0; i < op->outputs.size(); ++i) {
136 auto output = op->outputs[i];
137 auto data_type = unsupported_op->output_data_types[i];
138 model->GetArray(output).data_type = data_type;
139 }
140 break;
141 }
142 case OperatorType::kExpandDims: {
143 // Yield on ExpandDim until it is converted to Reshape
144 return false;
145 }
146 default: {
147 // These operators produce outputs with the same type as their 1st input
148 CHECK_GT(op->inputs.size(), 0);
149 const ArrayDataType data_type = model->GetArray(op->inputs[0]).data_type;
150 SetDataTypeForAllOutputs(model, op, data_type);
151 break;
152 }
153 }
154
155 // Return true if any output data type changed, false if none changed.
156 for (const auto& output : op->outputs) {
157 if (old_output_data_types[output] != model->GetArray(output).data_type) {
158 return true;
159 }
160 }
161 return false;
162}
