32void ComputeConvSizes(const Shape& input_shape, int output_depth, int kwidth,
33 int kheight, int stride_width, int stride_height,
34 int dilation_width_factor, int dilation_height_factor,
35 PaddingType padding_type, Shape* output_shape,
36 FixedPadding* fixed_padding) {
37 const int input_width = input_shape.dims(2);
38 const int input_height = input_shape.dims(1);
39 const int batch = input_shape.dims(0);
40
41 CHECK_GE(input_width, 1);
42 CHECK_GE(input_height, 1);
43 CHECK_GE(batch, 1);
44 CHECK_GE(kwidth, 1);
45 CHECK_GE(kheight, 1);
46 CHECK_GE(stride_width, 1);
47 CHECK_GE(stride_height, 1);
48 CHECK_GE(dilation_width_factor, 1);
49 CHECK_GE(dilation_height_factor, 1);
50
51 int dilated_kwidth = dilation_width_factor * (kwidth - 1) + 1;
52 int dilated_kheight = dilation_height_factor * (kheight - 1) + 1;
53
54 int output_height = 0;
55 int output_width = 0;
56 if (padding_type == PaddingType::kValid) {
57 output_height =
58 (input_height + stride_height - dilated_kheight) / stride_height;
59 output_width = (input_width + stride_width - dilated_kwidth) / stride_width;
60 } else if (padding_type == PaddingType::kSame) {
61 output_height = (input_height + stride_height - 1) / stride_height;
62 output_width = (input_width + stride_width - 1) / stride_width;
63 } else {
64 LOG(FATAL) << "Only supporting SAME or VALID padding";
65 }
66
67 fixed_padding->height = std::max(0, ((output_height - 1) * stride_height +
68 dilated_kheight - input_height) /
69 2);
70 fixed_padding->width = std::max(
71 0,
72 ((output_width - 1) * stride_width + dilated_kwidth - input_width) / 2);
73
74 // Actually had to debug a situation where those were negative due to bad
75 // propagation of placeholder -1 sizes in TensorFlowReshape.
76 CHECK_GT(output_width, 0);
77 CHECK_GT(output_height, 0);
78 output_shape->ReplaceDims({batch, output_height, output_width, output_depth});
79}

81void ComputeBinaryOperatorOutputSize(const Shape& input_shape_x,
82 const Shape& input_shape_y,
83 Array* output_array) {
84 // This matches the code in BroadcastBinaryOpShapeFn from tensorflow.
85 // It zips together the two input shapes and pads with 1 to make them the
86 // same length. For each dimension we broadcast if either dimension is 1 and
87 // otherwise expect them to match.
88 int rank_x = input_shape_x.dimensions_count();
89 int rank_y = input_shape_y.dimensions_count();
90 int rank_out = std::max(rank_x, rank_y);
91 std::vector<int>* dims_out = output_array->mutable_shape()->mutable_dims();
92 dims_out->clear();
93 dims_out->reserve(rank_out);
94 for (int i = 0; i < rank_out; ++i) {
95 int dim_x = i < (rank_out - rank_x)
96 ? 1
97 : input_shape_x.dims(i - (rank_out - rank_x));
98 bool dim_y_is_one = i < (rank_out - rank_y);
99 int dim_y = dim_y_is_one ? 1 : input_shape_y.dims(i - (rank_out - rank_y));
100 if (dim_x == -1 || dim_y == -1) {
101 // One or both dimensions is unknown.
102 QCHECK(false) << "Shapes must be specified";
103 } else if (dim_x == 1 || dim_y == 1) {
104 // Broadcast one dimension to the other that is 1.
105 if (dim_x == 1 && !dim_y_is_one) {
106 // Broadcast dim_y to dim_x (1).
107 dims_out->push_back(dim_y);
108 } else {
109 // Broadcast dim_x to dim_y (1).
110 DCHECK_EQ(dim_y, 1);
111 dims_out->push_back(dim_x);
112 }
113 } else {
114 // Expect the dimensions to match.
115 CHECK_EQ(dim_x, dim_y) << "Dimensions must match";
116 dims_out->push_back(dim_x);
117 }
118 }
119 CHECK(output_array->has_shape());
120}

122int GetOutputDepthFromWeights(const Model& model, const Operator& op) {
123 const string& weights_name = op.inputs[1];
124 const auto& weights_shape = model.GetArray(weights_name).shape();
125 if (op.type == OperatorType::kConv ||
126 op.type == OperatorType::kFullyConnected) {
127 return weights_shape.dims(0);
128 } else if (op.type == OperatorType::kDepthwiseConv) {
129 return weights_shape.dims(3);
130 } else {
131 LOG(FATAL) << "Unhandled operator type";
132 }

135bool EnsureBiasVectorShape(Model* model, Operator* op) {
136 const string& weights_name = op->inputs[1];
137 const auto& weights_array = model->GetArray(weights_name);
138 // Yield until weights shape has been resolved.
139 if (!weights_array.has_shape()) {
140 return false;
141 }
142
143 if (op->inputs.size() < 3) {
144 return false;
145 }
146 auto& bias_array = model->GetArray(op->inputs[2]);
147 if (bias_array.has_shape()) {
148 return true;
149 }
150
151 const int output_depth = GetOutputDepthFromWeights(*model, *op);
152 bias_array.copy_shape(Shape({output_depth}));
153
154 auto& float_buffer = bias_array.GetMutableBuffer<ArrayDataType::kFloat>();
155 float_buffer.data.resize(output_depth, 0);
156
157 return true;
158}

160void ProcessConvOperator(Model* model, ConvOperator* op) {
161 if (!EnsureBiasVectorShape(model, op)) {
162 return;
163 }
164
165 const auto& input_array = model->GetArray(op->inputs[0]);
166 // Yield until input dims have been resolved.
167 if (!input_array.has_shape()) {
168 return;
169 }
170 const auto& input_shape = input_array.shape();
171 CHECK_EQ(input_shape.dimensions_count(), 4);
172
173 const auto& weights_array = model->GetArray(op->inputs[1]);
174 // Yield until weights dims have been resolved.
175 if (!weights_array.has_shape()) {
176 return;
177 }
178 const auto& weights_shape = weights_array.shape();
179 CHECK_EQ(weights_shape.dimensions_count(), 4);
180
181 auto& output_array = model->GetArray(op->outputs[0]);
182 const int output_depth = weights_shape.dims(0);
183 const int kheight = weights_shape.dims(1);
184 const int kwidth = weights_shape.dims(2);
185 ComputeConvSizes(input_shape, output_depth, kwidth, kheight, op->stride_width,
186 op->stride_height, op->dilation_width_factor,
187 op->dilation_height_factor, op->padding.type,
188 output_array.mutable_shape(),
189 &op->padding.GetOrCreateFixedPadding());
190 CHECK_EQ(output_array.shape().dimensions_count(), 4);
191
192 // Set im2col array dimensions if there is one.
193 if (op->outputs.size() == 2) {
194 const auto& output_shape = output_array.shape();
195 const int input_depth = weights_shape.dims(3);
196 auto& im2col_array = model->GetArray(op->outputs[1]);
197 im2col_array.copy_shape(Shape{output_shape.dims(0), output_shape.dims(1),
198 output_shape.dims(2),
199 input_depth * kheight * kwidth});
200 }
201}

203void ProcessTransposeConvOperator(Model* model, TransposeConvOperator* op) {
204 // TransposeConv is unique in that it is specifically given the output shape
205 // as a 1D array on it's 1st input. Theoretically then, resolving the output
206 // shape is as easy as waiting for this input to be resolved. However, we also
207 // have to calculate the padding which requires the weights shape. So, we
208 // might as well calculate the output shape and ensure it matches the
209 // specified one
210
211 // Check if we have already run.
212 auto& output_array = model->GetArray(op->outputs[0]);
213 if (output_array.has_shape()) {
214 return;
215 }
216
217 // SPECIFIED OUTPUT SHAPE
218 // The below is the specified, or prescribed output shape, _given_ to the
219 // operator as an input.
220 auto& specified_output_shape_array =
221 model->GetArray(op->inputs[TransposeConvOperator::OUTPUT_SHAPE]);
222 if (!specified_output_shape_array.has_shape() ||
223 !specified_output_shape_array.buffer) {
224 // Yield until the specified output shape is resolved as a constant
225 return;
226 }
227
228 CHECK(specified_output_shape_array.data_type == ArrayDataType::kInt32)
229 << "TransposeConv input_dims must be int32";
230
231 CHECK(specified_output_shape_array.shape().dimensions_count() == 1 &&
232 specified_output_shape_array.shape().dims(0) == 4)
233 << "TransposeConv requires a 1D, 4 element array on it's 0th input "
234 "specifying the output shape. \""
235 << op->inputs[TransposeConvOperator::OUTPUT_SHAPE] << "\" had shape "
236 << toco::ShapeToString(specified_output_shape_array.shape());
237
238 // COMPUTE PADDING
239 // We require the weights shape to calculate padding.
240 const auto& weights_array =
241 model->GetArray(op->inputs[TransposeConvOperator::WEIGHTS]);
242 if (!weights_array.has_shape()) {
243 // Yield until weights dims have been resolved.
244 return;
245 }
246 const auto& weights_shape = weights_array.shape();
247 CHECK_EQ(weights_shape.dimensions_count(), 4)
248 << "TransposeConv weights must have 4 input dimensions. Input weights \""
249 << op->inputs[TransposeConvOperator::WEIGHTS] << "\" had shape "
250 << toco::ShapeToString(weights_shape) << ".";
251
252 CHECK(weights_shape.dims(0) == 1 && weights_shape.dims(3) == 1)
253 << "TransposeConv weights dimensions must begin and end with 1. Input "
254 "weights \""
255 << op->inputs[TransposeConvOperator::WEIGHTS] << "\" had shape "
256 << toco::ShapeToString(weights_shape) << ".";
257
258 // Compute padding
259 const int kheight = weights_shape.dims(1);
260 const int kwidth = weights_shape.dims(2);
261 op->padding.GetOrCreateFixedPadding();
262 if (op->padding.type == PaddingType::kValid) {
263 op->padding.fixed->height = 0;
264 op->padding.fixed->width = 0;
265 } else if (op->padding.type == PaddingType::kSame) {
266 op->padding.fixed->height = (kheight - 1) / 2;
267 op->padding.fixed->width = (kwidth - 1) / 2;
268 } else {
269 LOG(FATAL) << "TransposeConv only supports SAME or VALID padding";
270 }
271
272 // VALIDATE OUTPUT SHAPE
273 // Compute the output shape from the input and weights shapes to verify it
274 // agrees with the specified output shape.
275 const auto& input_array =
276 model->GetArray(op->inputs[TransposeConvOperator::DATA_INPUT]);
277 if (!input_array.has_shape()) {
278 // Yield until input dims have been resolved.
279 return;
280 }
281 const auto& input_shape = input_array.shape();
282 CHECK_EQ(input_shape.dimensions_count(), 4)
283 << "TransposeConv input shape must have 4 dimensions. Input \""
284 << op->inputs[TransposeConvOperator::WEIGHTS] << "\" had shape "
285 << toco::ShapeToString(weights_shape) << ".";
286
287 // Compute output shape
288 const int input_width = input_shape.dims(2);
289 const int input_height = input_shape.dims(1);
290 int output_height = op->stride_height * (input_height - 1);
291 int output_width = op->stride_width * (input_width - 1);
292 if (op->padding.type == PaddingType::kValid) {
293 output_height += kheight;
294 output_width += kwidth;
295 } else if (op->padding.type == PaddingType::kSame) {
296 output_height += 1;
297 output_width += 1;
298 }
299
300 CHECK(specified_output_shape_array.GetBuffer<ArrayDataType::kInt32>().data ==
301 std::vector<int32>({input_shape.dims(0), output_height, output_width,
302 weights_shape.dims(3)}))
303 << "Specified output shape: " << ShapeToString(output_array.shape())
304 << ", does not agree with shape computed from input data and weights: ["
305 << input_shape.dims(0) << ", " << output_height << ", " << output_width
306 << ", " << weights_shape.dims(3) << "].";
307
308 // SUCCESS: Set the op's output shape according to the specified output shape.
309 *(output_array.mutable_shape()->mutable_dims()) =
310 specified_output_shape_array.GetBuffer<ArrayDataType::kInt32>().data;
311}

313void ProcessDepthwiseConvOperator(Model* model, DepthwiseConvOperator* op) {
314 if (!EnsureBiasVectorShape(model, op)) {
315 return;
316 }
317
318 const auto& input_array = model->GetArray(op->inputs[0]);
319 // Yield until input dims have been resolved.
320 if (!input_array.has_shape()) {
321 return;
322 }
323 const auto& input_shape = input_array.shape();
324 CHECK_EQ(input_shape.dimensions_count(), 4);
325
326 const auto& weights_array = model->GetArray(op->inputs[1]);
327 // Yield until weights dims have been resolved.
328 if (!weights_array.has_shape()) {
329 return;
330 }
331 const auto& weights_shape = weights_array.shape();
332 CHECK_EQ(weights_shape.dimensions_count(), 4);
333
334 const string& output_name = op->outputs[0];
335 const int input_depth = input_shape.dims(3);
336 const int output_depth = weights_shape.dims(3);
337 // TensorFlow doesn't define the depth_multiplier value on DepthwiseConv ops,
338 // instead it has to be inferred from the weights dims. However, once we are
339 // here, weights dims have already been converted to our own internal format,
340 // where the multiplier is no longer readily apparent. So instead we get it
341 // as the quotient of output and input depths. We only want to do that when
342 // depth_multiplier had the zero value: any other value should be checked
343 // as done by the next if() below.
344 if (!op->depth_multiplier) {
345 op->depth_multiplier = output_depth / input_depth;
346 }
347 QCHECK_EQ(output_depth, input_depth * op->depth_multiplier)
348 << "input/output depths and depth_multiplier don't match";
349
350 const int kheight = weights_shape.dims(1);
351 const int kwidth = weights_shape.dims(2);
352 ComputeConvSizes(input_shape, output_depth, kwidth, kheight, op->stride_width,
353 op->stride_height, 1, 1, op->padding.type,
354 model->GetArray(output_name).mutable_shape(),
355 &op->padding.GetOrCreateFixedPadding());
356}

462void ProcessTensorFlowReshapeOperator(Model* model,
463 TensorFlowReshapeOperator* op) {
464 auto& output_array = model->GetArray(op->outputs[0]);
465 if (output_array.has_shape()) {
466 // We have already run
467 return;
468 }
469
470 const auto& input_array = model->GetArray(op->inputs[0]);
471 if (!input_array.has_shape()) {
472 // Yield until input dims have been resolved.
473 return;
474 }
475 const auto& input_shape = input_array.shape();
476
477 auto& shape_array = model->GetArray(op->inputs[1]);
478 if (!shape_array.has_shape()) {
479 // Yield until target_shape shape been resolved.
480 return;
481 }
482 if (!shape_array.buffer) {
483 // Yield until the target_shape is constant
484 return;
485 }
486 CHECK(shape_array.data_type == ArrayDataType::kInt32)
487 << "Reshape dims must be int32";
488
489 // shape_data is the raw array of ints describing the shape
490 // in the TensorFlow node. We intentionally make a copy here, rather than
491 // modify wildcards in-place below, because in some graphs, the same shape
492 // array with a wildcard may be referenced from multiple Reshape nodes, where
493 // the wildcard needs to resolved to distinct values.
494 std::vector<int32> shape_data =
495 shape_array.GetBuffer<ArrayDataType::kInt32>().data;
496 // The Reshape shape may have a wildcard dim, encoded as -1.
497 bool has_wildcard = false;
498 int wildcard_index = 0;
499 int product_non_wildcard_dims = 1;
500 for (int i = 0; i < shape_data.size(); i++) {
501 if (shape_data[i] == -1) {
502 CHECK(!has_wildcard);
503 has_wildcard = true;
504 wildcard_index = i;
505 } else {
506 product_non_wildcard_dims *= shape_data[i];
507 }
508 }
509 const int input_flat_size = RequiredBufferSizeForShape(input_shape);
510 if (has_wildcard) {
511 CHECK_GE(input_flat_size, product_non_wildcard_dims)
512 << "Array not large enough to fill the requested dimensions for "
513 "Reshape op with output \""
514 << op->outputs[0] << "\". Are your input shapes correct?";
515 shape_data[wildcard_index] = input_flat_size / product_non_wildcard_dims;
516 }
517 auto& output_shape = *output_array.mutable_shape();
518 *output_shape.mutable_dims() = shape_data;
519 CHECK_EQ(input_flat_size, RequiredBufferSizeForShape(output_shape))
520 << "Input cannot be reshaped to requested dimensions for Reshape op with "
521 "output \""
522 << op->outputs[0] << "\". Are your input shapes correct?";
523}

555void ProcessAddNOperator(Model* model, Operator* op) {
556 // Yield until all input dims have been resolved.
557 //
558 // TODO(myenik): Since AddN does not support broadcasting, maybe we could
559 // actually use this to improve shape propagation by propagating the shape of
560 // one input to all other inputs once it is resolved instead of just the
561 // output, since all inputs must be the same size and shape for a well-formed
562 // graph.
563 for (const auto& input : op->inputs) {
564 const auto& input_array = model->GetArray(input);
565 if (!input_array.has_shape()) {
566 return;
567 }
568 }
569
570 // AddN does not support broadcasting, all inputs must be the same shape, so
571 // we just take the first input shape and apply it to the output.
572 const auto& input0_array = model->GetArray(op->inputs[0]);
573 auto& output_array = model->GetArray(op->outputs[0]);
574 output_array.copy_shape(input0_array.shape());
575}

577bool KeepDims(const Operator& op) {
578 switch (op.type) {
579 case OperatorType::kTensorFlowMin:
580 return static_cast<const TensorFlowMinOperator&>(op).keep_dims;
581 case OperatorType::kTensorFlowMax:
582 return static_cast<const TensorFlowMaxOperator&>(op).keep_dims;
583 case OperatorType::kTensorFlowSum:
584 return static_cast<const TensorFlowSumOperator&>(op).keep_dims;
585 case OperatorType::kMean:
586 return static_cast<const MeanOperator&>(op).keep_dims;
587 default:
588 LOG(FATAL) << "Not a reduction operator!";
589 return false;
590 }
591}

593void ProcessTensorFlowReductionOperator(Model* model, Operator* op) {
594 CHECK_LE(op->inputs.size(), 2);
595 auto& output_array = model->GetArray(op->outputs[0]);
596 if (output_array.has_shape()) {
597 return;
598 }
599 const auto& input_array = model->GetArray(op->inputs[0]);
600 if (!input_array.has_shape()) {
601 return;
602 }
603 const auto& input_shape = input_array.shape();
604 const bool keep_dims = KeepDims(*op);
605 if (op->inputs.size() == 2) {
606 // There is a reduction_indices input.
607 const auto& reduction_array = model->GetArray(op->inputs[1]);
608 if (!reduction_array.buffer) {
609 return;
610 }

627 }
628 }
629 } else {
630 // No reduction_indices means complete reduction to a single scalar.

634 output_array.copy_shape(Shape({}));

684 // Yield until input dims have been resolved.
685 for (const auto& input_name : op->inputs) {
686 auto& input_array = model->GetArray(input_name);

716 CHECK_LT(axis, output_shape.dimensions_count());

1217void ProcessStackOperator(Model* model, StackOperator* op) {
1218 CHECK_GE(op->inputs.size(), 1);
1219 CHECK_EQ(op->outputs.size(), 1);
1220 auto& output_array = model->GetArray(op->outputs[0]);
1221 if (output_array.has_shape()) {
1222 // Shape already propagated
1223 return;
1224 }
1225
1226 std::unique_ptr<Shape> stacked_shape;
1227 for (const auto& input : op->inputs) {
1228 const auto& input_array = model->GetArray(input);
1229 if (!input_array.has_shape()) {
1230 // Yield until all input dims have been resolved.
1231 return;
1232 }
1233
1234 Shape shape = input_array.shape();
1235 if (shape.dimensions_count() == 0) {
1236 // Convert 0D scalars to 1D scalars of shape {1}.
1237 shape.mutable_dims()->push_back(1);
1238 }
1239 if (!stacked_shape) {
1240 stacked_shape.reset(new Shape(shape));
1241 } else {
1242 CHECK(*stacked_shape == shape) << "All input arrays to Stack operators "
1243 "must have the same shape. Input \""
1244 << input << "\" is different.";
1245 }
1246 }
1247
1248 int axis = op->axis;
1249 if (axis < 0) {
1250 // Handle negative axis
1251 axis += stacked_shape->dims().size() + 1;
1252 }
1253 stacked_shape->mutable_dims()->insert(
1254 stacked_shape->mutable_dims()->begin() + axis, op->inputs.size());
1255 output_array.copy_shape(*stacked_shape);
1256}

1258void ProcessStridedSliceOperator(Model* model, StridedSliceOperator* op) {
1259 CHECK_GE(op->inputs.size(), 1);
1260 CHECK_EQ(op->outputs.size(), 1);
1261 auto& output_array = model->GetArray(op->outputs[0]);
1262 if (output_array.has_shape()) {
1263 // Shape already propagated
1264 return;
1265 }
1266
1267 if (op->start_indices.empty() || op->stop_indices.empty() ||
1268 op->strides.empty()) {
1269 // ResolveStridedSliceAttributes has not run yet.
1270 return;
1271 }
1272
1273 const auto& input_array = model->GetArray(op->inputs[0]);
1274 if (!input_array.has_shape()) {
1275 // Yield until input dims have been resolved.
1276 return;
1277 }
1278
1279 if (op->ellipsis_mask != 0) {
1280 // Something like LOG_FIRST_N(WARNING, 10) would be prefferable to reduce
1281 // log noise. However, the TensorFlow logging library does not appear to
1282 // support this.
1283 LOG(WARNING) << "Skipping StridedSlice op with output \"" << op->outputs[0]
1284 << "\". ellipsis_mask is not supported (mask="
1285 << op->ellipsis_mask << ")";
1286 return;
1287 }
1288 if (op->new_axis_mask != 0) {
1289 LOG(WARNING) << "Skipping StridedSlice op with output \"" << op->outputs[0]
1290 << "\". new_axis_mask is not supported (mask="
1291 << op->new_axis_mask << ")";
1292 return;
1293 }
1294
1295 int dim_count = input_array.shape().dimensions_count();
1296 CHECK(op->start_indices.size() == dim_count)
1297 << ": Incorrect number of start indices supplied to StridedSlice op with "
1298 "output \""
1299 << op->outputs[0] << "\". Op requires " << dim_count << " start indices";
1300 CHECK(op->stop_indices.size() == dim_count)
1301 << ": Incorrect number of stop indices supplied to StridedSlice op with "
1302 "output \""
1303 << op->outputs[0] << "\". Op requires " << dim_count << " stop indices";
1304 CHECK(op->strides.size() == dim_count)
1305 << ": Incorrect number of strides supplied to StridedSlice op with "
1306 " output \""
1307 << op->outputs[0] << "\". Op requires " << dim_count << " strides";
1308
1309 // Create output shape
1310 std::vector<int>* dims = output_array.mutable_shape()->mutable_dims();
1311
1312 // Compute output shape
1313 for (int i = 0; i < dim_count; ++i) {
1314 const int mask = 1 << i;
1315 int start = (op->begin_mask & mask) ? 0 : op->start_indices[i];
1316 if (start < 0) {
1317 // handle negative indices
1318 start += input_array.shape().dims(i);
1319 }
1320 int stop = (op->end_mask & mask) ? input_array.shape().dims(i)
1321 : op->stop_indices[i];
1322 if (stop < 0) {
1323 // handle negative indices
1324 stop += input_array.shape().dims(i);
1325 }
1326
1327 int dim_size = ceil((stop - start) / static_cast<float>(op->strides[i]));
1328 dim_size = dim_size < 0 ? 0 : dim_size;
1329 if (op->shrink_axis_mask & mask) {
1330 CHECK_EQ(dim_size, 1) << "Output size for an axis must compute to 1 when "
1331 "shrinking that axis";
1332 } else {
1333 dims->push_back(dim_size);
1334 }
1335 }
1336}

1365 CHECK(op->inputs.size() == 3 || op->inputs.size() == 4);
1366 const auto& input_array = model->GetArray(op->inputs[0]);
1367 if (!input_array.has_shape()) return;
1368
1369 auto& weights_feature_array = model->GetArray(op->inputs[1]);
1370 if (!weights_feature_array.has_shape()) return;
1371
1372 const auto& weights_time_array = model->GetArray(op->inputs[2]);
1373 if (!weights_time_array.has_shape()) return;
1374

1383 const int memory_size = weights_time_array.shape().dims()[1];

1386 state_array.mutable_shape()->ReplaceDims(
1387 {batch_size, memory_size * num_units});
1388
1389 auto& output_array = model->GetArray(op->outputs[1]);
1390 output_array.mutable_shape()->ReplaceDims({batch_size, num_units});

1393void ProcessTransposeOperator(Model* model, TransposeOperator* op) {
1394 auto& output_array = model->GetArray(op->outputs[0]);
1395 if (output_array.has_shape()) {
1396 // We have already run
1397 return;
1398 }
1399
1400 const auto& input_array = model->GetArray(op->inputs[0]);
1401 if (!input_array.has_shape()) {
1402 // Yield until input dims have been resolved.
1403 return;
1404 }
1405 const auto& input_shape = input_array.shape();
1406
1407 auto& perm_array = model->GetArray(op->inputs[1]);
1408 if (!perm_array.has_shape()) {
1409 // Yield until permutation shape been resolved.
1410 return;
1411 }
1412 if (!perm_array.buffer) {
1413 // Yield until the permutation is constant
1414 return;
1415 }
1416 CHECK(perm_array.data_type == ArrayDataType::kInt32)
1417 << "Transpose permutation input must be int32";
1418
1419 std::vector<int32> const& perm =
1420 perm_array.GetBuffer<ArrayDataType::kInt32>().data;
1421 CHECK_EQ(perm.size(), input_shape.dimensions_count())
1422 << "Transpose permutation input " << op->inputs[1]
1423 << " must be same length as input dimensions";
1424 std::vector<int>* output_dims = output_array.mutable_shape()->mutable_dims();
1425 for (int i = 0; i < perm.size(); i++) {
1426 int axis = perm[i];
1427 CHECK_GE(axis, 0);
1428 CHECK_LT(axis, input_shape.dimensions_count());
1429 output_dims->push_back(input_shape.dims(axis));
1430 }
1431}

1433void ProcessArgMaxOperator(Model* model, ArgMaxOperator* op) {
1434 CHECK_EQ(op->inputs.size(), 2);
1435 const auto& input_array = model->GetArray(op->inputs[0]);
1436 // Yield until input dims have been resolved.
1437 if (!input_array.has_shape()) {
1438 return;
1439 }
1440
1441 // The current ArgMax implementation only supports 4-dimensional inputs with
1442 // the last dimension as the axis to perform ArgMax for.
1443 const std::vector<int>& input_dims = input_array.shape().dims();
1444 CHECK_EQ(input_dims.size(), 4);
1445 std::vector<int> output_dims;
1446

1448 for (int i = 0; i < input_dims.size() - 1; ++i) {
1449 output_dims.push_back(input_dims[i]);

1451 output_dims.push_back(1);
1452 const string& output_name = op->outputs[0];
1453 auto& output_array = model->GetArray(output_name);
1454 if (output_array.has_shape()) {
1455 return;
1456 }
1457 *output_array.mutable_shape()->mutable_dims() = output_dims;

1462bool PropagateFixedSizes::Run(Model* model, std::size_t op_index) {
1463 auto it = model->operators.begin() + op_index;
1464 auto* op = it->get();
1465 std::unordered_map<string, std::vector<int>> old_output_dims;
1466 for (const auto& output : op->outputs) {
1467 if (model->GetArray(output).has_shape()) {
1468 old_output_dims[output] = model->GetArray(output).shape().dims();
1469 }
1470 }
1471
1472 switch (op->type) {
1473 case OperatorType::kBatchNormalization:
1474 case OperatorType::kL2Normalization:
1475 case OperatorType::kDequantize:
1476 case OperatorType::kRelu:
1477 case OperatorType::kRelu1:
1478 case OperatorType::kRelu6:
1479 case OperatorType::kPRelu:
1480 case OperatorType::kSoftmax:
1481 case OperatorType::kLogSoftmax:
1482 case OperatorType::kLog:
1483 case OperatorType::kLogistic:
1484 case OperatorType::kTanh:
1485 case OperatorType::kLocalResponseNormalization:
1486 case OperatorType::kTensorFlowIdentity:
1487 case OperatorType::kFakeQuant:
1488 case OperatorType::kNeg:
1489 case OperatorType::kTensorFlowRsqrt:
1490 case OperatorType::kTensorFlowSqrt:
1491 case OperatorType::kTensorFlowSquare:
1492 case OperatorType::kTensorFlowAll:
1493 case OperatorType::kTensorFlowAssert:
1494 case OperatorType::kCast:
1495 case OperatorType::kFloor:
1496 case OperatorType::kExp:
1497 ProcessSimpleOperator(model, op);
1498 break;
1499 case OperatorType::kGather:
1500 ProcessGatherOperator(model, static_cast<GatherOperator*>(op));
1501 break;
1502 case OperatorType::kTopK_V2:
1503 ProcessTopkV2Operator(model, static_cast<TopKV2Operator*>(op));
1504 break;
1505 case OperatorType::kAdd:
1506 case OperatorType::kSub:
1507 case OperatorType::kMul:
1508 case OperatorType::kDiv:
1509 case OperatorType::kFloorDiv:
1510 case OperatorType::kFloorMod:
1511 case OperatorType::kTensorFlowLess:
1512 case OperatorType::kTensorFlowLessEqual:
1513 case OperatorType::kTensorFlowGreater:
1514 case OperatorType::kTensorFlowMaximum:
1515 case OperatorType::kTensorFlowMinimum:
1516 case OperatorType::kTensorFlowGreaterEqual:
1517 ProcessSimpleBinaryOperator(model, op);
1518 break;
1519 case OperatorType::kAddN:
1520 ProcessAddNOperator(model, op);
1521 break;
1522 case OperatorType::kConv:
1523 ProcessConvOperator(model, static_cast<ConvOperator*>(op));
1524 break;
1525 case OperatorType::kTransposeConv:
1526 ProcessTransposeConvOperator(model,
1527 static_cast<TransposeConvOperator*>(op));
1528 break;
1529 case OperatorType::kDepthwiseConv:
1530 ProcessDepthwiseConvOperator(model,
1531 static_cast<DepthwiseConvOperator*>(op));
1532 break;
1533 case OperatorType::kDepthToSpace:
1534 ProcessDepthToSpaceOperator(model,
1535 static_cast<DepthToSpaceOperator*>(op));
1536 break;
1537 case OperatorType::kSpaceToDepth:
1538 ProcessSpaceToDepthOperator(model,
1539 static_cast<SpaceToDepthOperator*>(op));
1540 break;
1541 case OperatorType::kFill:
1542 CHECK_EQ(op->inputs.size(), 2);
1543 ProcessOpWithShapeInput(model, op);
1544 break;
1545 case OperatorType::kFullyConnected:
1546 ProcessFullyConnectedOperator(model,
1547 static_cast<FullyConnectedOperator*>(op));
1548 break;
1549 case OperatorType::kTensorFlowReshape:
1550 ProcessTensorFlowReshapeOperator(
1551 model, static_cast<TensorFlowReshapeOperator*>(op));
1552 break;
1553 case OperatorType::kAveragePool:
1554 ProcessAveragePoolOperator(model, static_cast<AveragePoolOperator*>(op));
1555 break;
1556 case OperatorType::kMaxPool:
1557 ProcessMaxPoolOperator(model, static_cast<MaxPoolOperator*>(op));
1558 break;
1559 case OperatorType::kL2Pool:
1560 ProcessL2PoolOperator(model, static_cast<L2PoolOperator*>(op));
1561 break;
1562 case OperatorType::kTensorFlowMin:
1563 case OperatorType::kTensorFlowMax:
1564 case OperatorType::kTensorFlowSum:
1565 case OperatorType::kMean:
1566 ProcessTensorFlowReductionOperator(model, op);
1567 break;
1568
1569 case OperatorType::kSlice:
1570 ProcessSliceOperator(model, static_cast<SliceOperator*>(op));
1571 break;
1572
1573 case OperatorType::kTensorFlowTile:
1574 // We don't currently implement the propagation of fixed sizes through
1575 // a TensorFlow Tile.
1576 //
1577 // Fortunately, we don't need to: so far, we have only dealt with Tile
1578 // or Slice ops in subgraphs that are identified as L2Normalization.
1579 // See IdentifyL2Normalization.
1580 break;
1581 case OperatorType::kTensorFlowSwitch:
1582 // We can't know the sizes of the outputs until we have resolved the
1583 // predicate, and once we have resolved the predicate, the whole
1584 // Switch node will get resolved away.
1585 // See ResolveTensorFlowSwitch.
1586 break;
1587 case OperatorType::kTensorFlowMerge:
1588 // No need to bother resolving TensorFlow Merge ops: other graph
1589 // transformations will remove them anyway.
1590 // See ResolveTensorFlowMerge.
1591 break;
1592 case OperatorType::kTensorFlowSplit:
1593 ProcessTensorFlowSplitOperator(model,
1594 static_cast<TensorFlowSplitOperator*>(op));
1595 break;
1596 case OperatorType::kSqueeze:
1597 ProcessSqueezeOperator(model, static_cast<SqueezeOperator*>(op));
1598 break;
1599 case OperatorType::kTensorFlowConcat:
1600 case OperatorType::kTensorFlowConcatV2:
1601 // Unimplemented, hopefully another graph transformation will
1602 // drop it or rewrite it. Concretely, either ResolveTensorFlowConcat
1603 // will resolve this node to a DepthConcatenation, or else we have
1604 // a more general non-depth concatenation that will hopefully be dropped,
1605 // or else at the moment we will abort.
1606 break;
1607 case OperatorType::kExpandDims:
1608 // Yield until ExpandDims is converted to Reshape
1609 break;
1610 case OperatorType::kRange:
1611 ProcessRangeOperator(model, static_cast<RangeOperator*>(op));
1612 break;
1613 case OperatorType::kRank:
1614 ProcessRankOperator(model, static_cast<RankOperator*>(op));
1615 break;
1616 case OperatorType::kTensorFlowShape:
1617 ProcessShapeOperator(model, static_cast<TensorFlowShapeOperator*>(op));
1618 break;
1619 case OperatorType::kStack:
1620 ProcessStackOperator(model, static_cast<StackOperator*>(op));
1621 break;
1622 case OperatorType::kReorderAxes:
1623 ProcessReorderAxesOperator(model, static_cast<ReorderAxesOperator*>(op));
1624 break;
1625 case OperatorType::kConcatenation:
1626 ProcessConcatenationOperator(model,
1627 static_cast<ConcatenationOperator*>(op));
1628 break;
1629 case OperatorType::kResizeBilinear:
1630 ProcessResizeBilinearOperator(model,
1631 static_cast<ResizeBilinearOperator*>(op));
1632 break;
1633 case OperatorType::kLstmCell:
1634 ProcessLstmCellOperator(model, static_cast<LstmCellOperator*>(op));
1635 break;
1636 case OperatorType::kBatchMatMul:
1637 case OperatorType::kTensorFlowMatMul:
1638 // MatMul operators are converted to FullyConnected, after which their
1639 // shapes are propagated.
1640 break;
1641 case OperatorType::kSpaceToBatchND:
1642 ProcessSpaceToBatchNDOperator(model,
1643 static_cast<SpaceToBatchNDOperator*>(op));
1644 break;
1645 case OperatorType::kBatchToSpaceND:
1646 ProcessBatchToSpaceNDOperator(model,
1647 static_cast<BatchToSpaceNDOperator*>(op));
1648 break;
1649 case OperatorType::kPad:
1650 ProcessPadOperator(model, static_cast<PadOperator*>(op));
1651 break;
1652 case OperatorType::kStridedSlice:
1653 ProcessStridedSliceOperator(model,
1654 static_cast<StridedSliceOperator*>(op));
1655 break;
1656 case OperatorType::kArgMax:
1657 ProcessArgMaxOperator(model, static_cast<ArgMaxOperator*>(op));
1658 break;
1659 case OperatorType::kTensorFlowUnsupported:
1660 break;
1661 case OperatorType::kSvdf:
1662 ProcessSvdfOperator(model, static_cast<SvdfOperator*>(op));
1663 break;
1664 case OperatorType::kTranspose:
1665 ProcessTransposeOperator(model, static_cast<TransposeOperator*>(op));
1666 break;
1667 case OperatorType::kDynamicPartition:
1668 case OperatorType::kDynamicStitch:
1669 // DynamicPartition/DynamicStitch are currently only supported for
1670 // transforms that remove them, so we avoid propagating shapes through
1671 // them and let things settle once they've been removed.
1672 break;
1673 case OperatorType::kRandomUniform:
1674 CHECK_EQ(op->inputs.size(), 1);
1675 ProcessOpWithShapeInput(model, op);
1676 break;
1677 default:
1678 // Unimplemented, another graph transformation should drop it.
1679 LOG(FATAL) << "Unhandled operator type " << OperatorTypeName(op->type);
1680 }
1681
1682 // Return true if any output dim changed, false if none changed.
1683 // Assumption: no transformation clears an output shape, they only add shapes.
1684 for (const auto& output : op->outputs) {
1685 if (model->GetArray(output).has_shape() &&
1686 (old_output_dims[output] != model->GetArray(output).shape().dims())) {
1687 AddMessageF("Set shape of %s to [%s]", output,
1688 absl::StrJoin(model->GetArray(output).shape().dims(), ","));
1689 return true;
1690 }
1691 }
1692 return false;
1693}
