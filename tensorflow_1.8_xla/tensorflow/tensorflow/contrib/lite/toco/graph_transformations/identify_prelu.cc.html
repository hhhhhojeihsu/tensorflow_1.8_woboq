25// This transformation rule tries to identify the PRelu structure generated by
26// Keras, and convert it to a single op.
27//
28// The formula of PReLU is:
29// f(x) = alpha * x for x < 0, f(x) = x for x >= 0.
30//
31// `x` is the input, and `alpha` is a trainable tensor which can be broadcasted
32// to the shape of `x`.
33//
34// There's no native PRelu op in TensorFlow, so Keras generates the following
35// structure which does the equivalent calculation:
36// f(x) = Relu(x) + (-alpha * Relu(-x))
37//
38// Practically, alpha is always a constant in the inference graph, and Toco have
39// other graph transformations which fold the activation functions to other ops.
40// Therefore, we're looking for the structure:
41//
42// f(x) = Relu(x) + (negative_alpha * Neg(x, activation=Relu))
43
44namespace toco {
45
46bool IdentifyPRelu::Run(Model* model, std::size_t op_index) {
47 const auto add_op_it = model->operators.begin() + op_index;
48 const auto* add_op = add_op_it->get();
49 if (add_op == nullptr || add_op->type != OperatorType::kAdd ||
50 add_op->inputs.size() != 2 ||
51 add_op->fused_activation_function != FusedActivationFunctionType::kNone) {
52 return false;
53 }
54
55 const auto* relu_input_op = GetOpWithOutput(*model, add_op->inputs[0]);
56 if (relu_input_op == nullptr || relu_input_op->type != OperatorType::kRelu ||
57 relu_input_op->inputs.size() != 1 ||
58 relu_input_op->fused_activation_function !=
59 FusedActivationFunctionType::kNone) {
60 return false;
61 }
62
63 // TODO(ycling): Both Add and Mul are commutative. Support the case where
64 // the position of operands are exchanged.
65 const auto* mul_op = GetOpWithOutput(*model, add_op->inputs[1]);
66 if (mul_op == nullptr || mul_op->type != OperatorType::kMul ||
67 mul_op->inputs.size() != 2 ||
68 mul_op->fused_activation_function != FusedActivationFunctionType::kNone) {
69 return false;
70 }
71
72 const auto neg_alpha_tensor_name = mul_op->inputs[0];
73
74 const auto* relu_neg_input_op = GetOpWithOutput(*model, mul_op->inputs[1]);
75
76 if (relu_neg_input_op == nullptr ||
77 relu_neg_input_op->type != OperatorType::kNeg ||
78 relu_neg_input_op->fused_activation_function !=
79 FusedActivationFunctionType::kRelu ||
80 relu_neg_input_op->inputs.size() != 1) {
81 return false;
82 }
83
84 if (relu_input_op->inputs[0] != relu_neg_input_op->inputs[0]) {
85 return false;
86 }
87
88 const auto input_tensor_name = relu_input_op->inputs[0];
89 const auto output_tensor_name = add_op->outputs[0];
90
91 // Construct a tensor for positive alpha (double negative).
92 const auto alpha_tensor_name =
93 AvailableArrayName(*model, neg_alpha_tensor_name + "_neg");
94 model->GetOrCreateArray(alpha_tensor_name);
95
96 auto* neg_neg_alpha_op = new NegOperator;
97 neg_neg_alpha_op->inputs = {neg_alpha_tensor_name};
98 neg_neg_alpha_op->outputs = {alpha_tensor_name};
99 model->operators.emplace(add_op_it, neg_neg_alpha_op);
100
101 auto* prelu_op = new PReluOperator;
102 prelu_op->inputs = {input_tensor_name, alpha_tensor_name};
103 prelu_op->outputs = {output_tensor_name};
104 model->operators.emplace(add_op_it, prelu_op);
105 AddMessageF("Creating %s replacing equivalent subgraph", LogName(*prelu_op));
106
107 DeleteArrayIfUsedOnce(neg_alpha_tensor_name, model);
108 DeleteArrayIfUsedOnce(add_op->inputs[0], model);
109 DeleteArrayIfUsedOnce(add_op->inputs[1], model);
110 DeleteArrayIfUsedOnce(mul_op->inputs[1], model);
111 // Remove the existing Add op that outputs the final result. If the other
112 // intermediate tensors aren't used by other ops, those will be removed by
113 // other graph transformation rules.
114 model->operators.erase(FindOp(*model, add_op));
115
116 return true;
117}
