22bool ResolveConstantRange::Run(Model* model, std::size_t op_index) {
23 const auto it = model->operators.begin() + op_index;
24 auto* base_op = it->get();
25 if (base_op->type != OperatorType::kRange) {
26 return false;
27 }
28 auto* op = static_cast<RangeOperator*>(base_op);
29
30 CHECK_EQ(op->inputs.size(), 3);
31 const auto& start_array = model->GetArray(op->inputs[0]);
32 if (!start_array.has_shape()) {
33 // Yield until all input dims have been resolved.
34 return false;
35 }
36 const auto& limit_array = model->GetArray(op->inputs[1]);
37 if (!limit_array.has_shape()) {
38 // Yield until all input dims have been resolved.
39 return false;
40 }
41 const auto& delta_array = model->GetArray(op->inputs[2]);
42 if (!delta_array.has_shape()) {
43 // Yield until all input dims have been resolved.
44 return false;
45 }
46
47 for (const auto& input : op->inputs) {
48 if (!IsConstantParameterArray(*model, input)) {
49 // yield if any input is mutable
50 return false;
51 }
52 }
53
54 CHECK_EQ(op->outputs.size(), 1);
55 auto& output_array = model->GetArray(op->outputs[0]);
56 if (output_array.data_type == ArrayDataType::kNone) {
57 // Yield until the output type has been set by PropagateArrayDataTypes
58 return false;
59 }
60
61 CHECK_EQ(RequiredBufferSizeForShape(start_array.shape()), 1)
62 << "Range op inputs must be scalar.";
63 CHECK_EQ(RequiredBufferSizeForShape(limit_array.shape()), 1)
64 << "Range op inputs must be scalar.";
65 CHECK_EQ(RequiredBufferSizeForShape(delta_array.shape()), 1)
66 << "Range op inputs must be scalar.";
67
68 CHECK(start_array.data_type == ArrayDataType::kInt32)
69 << "Range op inputs must be int32.";
70 CHECK(limit_array.data_type == ArrayDataType::kInt32)
71 << "Range op inputs must be int32.";
72 CHECK(delta_array.data_type == ArrayDataType::kInt32)
73 << "Range op inputs must be int32.";
74
75 // Compute buffer contents
76 int start = start_array.GetBuffer<ArrayDataType::kInt32>().data[0];
77 int limit = limit_array.GetBuffer<ArrayDataType::kInt32>().data[0];
78 int delta = delta_array.GetBuffer<ArrayDataType::kInt32>().data[0];
79 auto& buffer = output_array.GetMutableBuffer<ArrayDataType::kInt32>();
80 buffer.data.clear();
81 for (int32 val = start; val < limit; val += delta) {
82 buffer.data.push_back(val);
83 }
84 CHECK_EQ(floor((limit - start) / delta), buffer.data.size());
85 CHECK_EQ(buffer.data.size(), output_array.shape().dims()[0]);
86
87 // Delete the input array if no longer used
88 if (IsDiscardableArray(*model, op->inputs[0]) &&
89 CountOpsWithInput(*model, op->inputs[0]) == 1) {
90 model->EraseArray(op->inputs[0]);
91 }
92 if (IsDiscardableArray(*model, op->inputs[1]) &&
93 CountOpsWithInput(*model, op->inputs[1]) == 1) {
94 model->EraseArray(op->inputs[1]);
95 }
96 if (IsDiscardableArray(*model, op->inputs[2]) &&
97 CountOpsWithInput(*model, op->inputs[2]) == 1) {
98 model->EraseArray(op->inputs[2]);
99 }
100
101 // Delete the operator
102 model->operators.erase(it);
103
104 return true;
105}
