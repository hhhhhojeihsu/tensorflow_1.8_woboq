25// A dilated convolution can be emulated with a regular convolution by chaining
26// SpaceToBatch and BatchToSpace ops before and after it:
27//
28// SpaceToBatchND -> Conv2D -> BatchToSpaceND
29//
30// This method was common before Conv2D fully supported dilated convolution in
31// TensorFlow. This transformation detects this "emulation", and replaces it
32// with a true dilated convolution, eliminating the SpaceToBatch and
33// BatchtoSpace ops.
34//
35// Detecting this alone would be relatively easy. However, in practice some
36// extra ops are used, so we detect the following patterns:
37//
38//
39// SpaceToBatchND -> Expand -> Conv2D -> Squeeze -> BatchToSpaceND -> BiasAdd
40//
41// SpaceToBatchND -> Expand -> Conv2D -> Squeeze -> Pad -> BatchToSpaceND ->
42// BiasAdd
43//
44// SpaceToBatchND -> Expand -> Conv2D -> Squeeze -> BiasAdd -> BatchToSpaceND
45//
46// SpaceToBatchND -> Conv2D -> Pad -> BatchToSpaceND -> BiasAdd
47//
48// SpaceToBatchND -> Conv2D -> BatchToSpaceND -> BiasAdd
49//
50//
51// The Expand/Squeeze combination is used to adapt a 3D array (such as in
52// WaveNet) to the 4D arrays that Conv2D requires. Padding and BiasAdd are
53// thrown in just for the extra headache. Padding adapts non-conforming input
54// sizes, and can be discarded. The bias is necessary, so is kept.
55
56bool IdentifyDilatedConv::Run(Model* model, std::size_t op_index) {
57 const auto it = model->operators.begin() + op_index;
58 auto* stb_op = it->get();
59
60 // 1. IDENTIFY OPERATORS
61 // ***************************************************************************
62 // SpaceToBatch Op.
63 if (stb_op->type != OperatorType::kSpaceToBatchND) {
64 return false;
65 }
66 if (stb_op->inputs.size() != 3) {
67 return false;
68 }
69 CHECK_EQ(stb_op->outputs.size(), 1);
70 // Extract the dilation factor from Input[1] of SpaceToBatch
71 // TODO(mjmatthews): Support 2D dilation factors.
72 const auto& block_shape_array = model->GetArray(stb_op->inputs[1]);
73 if (!block_shape_array.buffer) {
74 return false;
75 }
76 CHECK_EQ(block_shape_array.shape().dimensions_count(), 1);
77 int dilation_factor =
78 block_shape_array.Array::GetBuffer<ArrayDataType::kInt32>().data[0];
79
80 // Expand Op
81 auto* post_stb_op = GetOpWithInput(*model, stb_op->outputs[0]);
82 if (!post_stb_op) {
83 return false;
84 }
85 bool has_expand_op = false;
86 if (post_stb_op->type == OperatorType::kExpandDims) {
87 has_expand_op = true;
88 CHECK_EQ(post_stb_op->inputs.size(), 2);
89 CHECK_EQ(post_stb_op->outputs.size(), 1);
90 }
91
92 // Conv Op
93 ConvOperator* conv_op = dynamic_cast<ConvOperator*>(
94 has_expand_op ? GetOpWithInput(*model, post_stb_op->outputs[0])
95 : GetOpWithInput(*model, stb_op->outputs[0]));
96 if (!conv_op || conv_op->type != OperatorType::kConv) {
97 return false;
98 }
99 if (conv_op->inputs.size() != 2) {
100 // The conv op must only have weights, no bias.
101 return false;
102 }
103 CHECK_EQ(conv_op->outputs.size(), 1);
104
105 // Squeeze Op
106 auto* post_conv_op = GetOpWithInput(*model, conv_op->outputs[0]);
107 if (!post_conv_op) {
108 return false;
109 }
110 if (has_expand_op) {
111 if (post_conv_op->type != OperatorType::kSqueeze) {
112 // If an expand op was used, the post-conv op must be a squeeze op
113 return false;
114 }
115 CHECK_EQ(post_conv_op->inputs.size(), 1);
116 CHECK_EQ(post_conv_op->outputs.size(), 1);
117 }
118
119 // Pad Op
120 const auto* pad_op = has_expand_op
121 ? GetOpWithInput(*model, post_conv_op->outputs[0])
122 : GetOpWithInput(*model, conv_op->outputs[0]);
123 bool has_pad_op = false;
124 if (pad_op->type == OperatorType::kPad) {
125 has_pad_op = true;
126 CHECK_EQ(pad_op->inputs.size(), 2);
127 CHECK_EQ(pad_op->outputs.size(), 1);
128 }
129 // TODO(mjmatthews): Perform validity checking on padding dimensions.
130
131 // Pre-BatchToSpace Bias Op
132 auto* next_op = has_pad_op
133 ? GetOpWithInput(*model, pad_op->outputs[0])
134 : has_expand_op
135 ? GetOpWithInput(*model, post_conv_op->outputs[0])
136 : GetOpWithInput(*model, conv_op->outputs[0]);
137 bool has_bias_before_bts = false;
138 if (next_op->type == OperatorType::kAdd) {
139 has_bias_before_bts = true;
140 }
141 auto final_op = GetOpWithInput(*model, next_op->outputs[0]);
142
143 // BatchToSpace Op
144 const auto* bts_op = has_bias_before_bts ? final_op : next_op;
145 if (bts_op->type != OperatorType::kBatchToSpaceND) {
146 return false;
147 }
148 CHECK_EQ(bts_op->inputs.size(), 3);
149 CHECK_EQ(bts_op->outputs.size(), 1);
150
151 // Post-BatchToSpace Bias Op
152 Operator* bias_add_op = !has_bias_before_bts ? final_op : next_op;
153 if (bias_add_op->type != OperatorType::kAdd) {
154 // Bias op is required before or after BatchToSpace
155 return false;
156 }
157 CHECK_EQ(bias_add_op->inputs.size(), 2);
158 CHECK_EQ(bias_add_op->outputs.size(), 1);
159
160 LOG(INFO) << "Identified sub-network emulating dilated convolution.";
161
162 // 2. RE-WIRE OPERATORS
163 // ***************************************************************************
164 // Re-use the existing Conv2D op.
165 conv_op->dilation_width_factor = dilation_factor;
166 conv_op->dilation_height_factor = dilation_factor;
167 conv_op->padding.type = PaddingType::kSame;
168
169 // Rewire the ops to bypass SpaceToBatch, BatchToSpace, and Pad.
170 bias_add_op->outputs[0] = final_op->outputs[0];
171 if (has_expand_op) {
172 bias_add_op->inputs[0] = post_conv_op->outputs[0];
173 post_conv_op->inputs[0] = conv_op->outputs[0];
174 conv_op->inputs[0] = post_stb_op->outputs[0];
175 post_stb_op->inputs[0] = stb_op->inputs[0];
176 } else {
177 bias_add_op->inputs[0] = conv_op->outputs[0];
178 conv_op->inputs[0] = stb_op->inputs[0];
179 }
180 // TODO(mjmatthews): Connect bias directly into the Conv2D?
181
182 // 3. DELETE LEFTOVER OPERATORS
183 // ***************************************************************************
184 // Order is important. Delete the output array first, then the op, then it's
185 // redundant inputs.
186 // BatchToSpace Op
187 DeleteArrayIfUnused(bts_op->outputs[0], model);
188 std::vector<string> bts_op_inputs = bts_op->inputs;
189 model->operators.erase(FindOp(*model, bts_op));
190 DeleteArrayIfUnused(bts_op_inputs[1], model);
191 DeleteArrayIfUnused(bts_op_inputs[2], model);
192
193 // Pad Op if present
194 if (has_pad_op) {
195 DeleteArrayIfUnused(pad_op->outputs[0], model);
196 std::vector<string> pad_op_inputs = pad_op->inputs;
197 model->operators.erase(FindOp(*model, pad_op));
198 DeleteArrayIfUnused(pad_op_inputs[1], model);
199 }
200
201 // SpaceToBatch Op
202 DeleteArrayIfUnused(stb_op->outputs[0], model);
203 std::vector<string> stb_op_inputs = stb_op->inputs;
204 model->operators.erase(FindOp(*model, stb_op));
205 DeleteArrayIfUnused(stb_op_inputs[1], model);
206 DeleteArrayIfUnused(stb_op_inputs[2], model);
207
208 LOG(INFO) << "Replaced with Dilated Conv2D op outputting \""
209 << conv_op->outputs[0] << "\".";
210 return true;
211}
