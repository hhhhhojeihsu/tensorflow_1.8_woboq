28bool SplitLstmCellInputs::Run(Model* model, std::size_t op_index) {
29 // Find lstm cell.
30 auto op_it = model->operators.begin() + op_index;
31 auto curr_op = op_it->get();
32 if (curr_op->type != OperatorType::kLstmCell) {
33 return false;
34 }
35
36 // Already an extended LstmCell with kExtendedLstmInputCount of inputs,
37 // do not need to split cell inputs.
38 if (curr_op->inputs.size() == kExtendedLstmInputCount) {
39 return false;
40 }
41
42 // Make sure the WEIGHTS_INPUT and BIASES_INPUT are constant arrays,
43 // that are able to be split into smaller weight and bias tensors.
44 if (!IsConstantParameterArray(
45 *model, curr_op->inputs[LstmCellOperator::WEIGHTS_INPUT]) ||
46 !IsConstantParameterArray(
47 *model, curr_op->inputs[LstmCellOperator::BIASES_INPUT])) {
48 return false;
49 }
50
51 // Make sure propagate_fixed_sizes has defined the size of the output.
52 if (!model->GetArray(curr_op->outputs[LstmCellOperator::ACTIV_OUTPUT])
53 .has_shape()) {
54 return false;
55 }
56
57 // Emplace a new LstmCell operator with extended inputs (kernel/lstm.cc).
58 auto lstm_cell_op = absl::make_unique<LstmCellOperator>();
59 lstm_cell_op->inputs.resize(kExtendedLstmInputCount);
60 int num_input = model->GetArray(curr_op->inputs[LstmCellOperator::DATA_INPUT])
61 .shape()
62 .dims(1);
63
64 // n_cell and n_output have the same size when there is no projection.
65 int num_cell =
66 model->GetArray(curr_op->outputs[LstmCellOperator::ACTIV_OUTPUT])
67 .shape()
68 .dims(1);
69 int num_output = num_cell;
70
71 // Data input.
72 lstm_cell_op->inputs[kInputTensor] =
73 curr_op->inputs[LstmCellOperator::ACTIV_OUTPUT];
74
75 // Get original weight tensor and decompose 1 tensor to 8 sub tensors.
76 Array& kernel =
77 model->GetArray(curr_op->inputs[LstmCellOperator::WEIGHTS_INPUT]);
78 const string base_name(FindLongestCommonPrefix(
79 curr_op->outputs[LstmCellOperator::ACTIV_OUTPUT],
80 curr_op->outputs[LstmCellOperator::STATE_OUTPUT]));
81
82 // Input weight tensors of size {n_cell, n_input}.
83 CopySubArrayToArray(
84 model, &(lstm_cell_op->inputs[kInputToInputWeightsTensor]),
85 base_name + "weight_i_i", num_cell, num_input, kernel, 0, 0);
86 CopySubArrayToArray(model, &(lstm_cell_op->inputs[kInputToCellWeightsTensor]),
87 base_name + "weight_c_i", num_cell, num_input, kernel,
88 num_cell, 0);
89 CopySubArrayToArray(
90 model, &(lstm_cell_op->inputs[kInputToForgetWeightsTensor]),
91 base_name + "weight_f_i", num_cell, num_input, kernel, num_cell * 2, 0);
92 CopySubArrayToArray(
93 model, &(lstm_cell_op->inputs[kInputToOutputWeightsTensor]),
94 base_name + "weight_o_i", num_cell, num_input, kernel, num_cell * 3, 0);
95
96 // Recurrent weight tensors of size {n_cell, n_output}.
97 CopySubArrayToArray(
98 model, &(lstm_cell_op->inputs[kRecurrentToInputWeightsTensor]),
99 base_name + "weight_i_r", num_cell, num_output, kernel, 0, num_input);
100 CopySubArrayToArray(model,
101 &(lstm_cell_op->inputs[kRecurrentToCellWeightsTensor]),
102 base_name + "weight_c_r", num_cell, num_output, kernel,
103 num_cell, num_input);
104 CopySubArrayToArray(model,
105 &(lstm_cell_op->inputs[kRecurrentToForgetWeightsTensor]),
106 base_name + "weight_f_r", num_cell, num_output, kernel,
107 num_cell * 2, num_input);
108 CopySubArrayToArray(model,
109 &(lstm_cell_op->inputs[kRecurrentToOutputWeightsTensor]),
110 base_name + "weight_o_r", num_cell, num_output, kernel,
111 num_cell * 3, num_input);
112
113 // Peephole (optional).
114 CreateOptionalArray(model, &(lstm_cell_op->inputs[kCellToInputWeightsTensor]),
115 base_name + "peephole_c_i");
116 CreateOptionalArray(model,
117 &(lstm_cell_op->inputs[kCellToForgetWeightsTensor]),
118 base_name + "peephole_c_f");
119 CreateOptionalArray(model,
120 &(lstm_cell_op->inputs[kCellToOutputWeightsTensor]),
121 base_name + "peephole_c_o");
122
123 // Get original bias tensor and decompose 1 tensor to 4 sub tensors
124 Array& bias =
125 model->GetArray(curr_op->inputs[LstmCellOperator::BIASES_INPUT]);
126 CopySubArrayToArray(model, &(lstm_cell_op->inputs[kInputGateBiasTensor]),
127 base_name + "bias_i", num_cell, 1, bias, 0, 0);
128 CopySubArrayToArray(model, &(lstm_cell_op->inputs[kCellGateBiasTensor]),
129 base_name + "bias_c", num_cell, 1, bias, num_cell, 0);
130 CopySubArrayToArray(model, &(lstm_cell_op->inputs[kForgetGateBiasTensor]),
131 base_name + "bias_f", num_cell, 1, bias, num_cell * 2, 0);
132 CopySubArrayToArray(model, &(lstm_cell_op->inputs[kOutputGateBiasTensor]),
133 base_name + "bias_o", num_cell, 1, bias, num_cell * 3, 0);
134
135 // Projection (optional).
136 CreateOptionalArray(model, &(lstm_cell_op->inputs[kProjectionWeightsTensor]),
137 base_name + "proj_weight");
138 CreateOptionalArray(model, &(lstm_cell_op->inputs[kProjectionBiasTensor]),
139 base_name + "proj_bias");
140
141 // Reorder LstmCell's outputs.
142 lstm_cell_op->outputs.resize(LstmCellOperator::NUM_OUTPUTS);
143 lstm_cell_op->outputs[kScratchBufferTensor] =
144 curr_op->outputs[LstmCellOperator::CONCAT_TEMP];
145 lstm_cell_op->outputs[kOutputStateTensor] =
146 curr_op->outputs[LstmCellOperator::ACTIV_TEMP];
147 lstm_cell_op->outputs[kCellStateTensor] =
148 curr_op->outputs[LstmCellOperator::STATE_OUTPUT];
149 lstm_cell_op->outputs[kOutputTensor] =
150 curr_op->outputs[LstmCellOperator::ACTIV_OUTPUT];
151
152 // Add the op into model.
153 model->operators.emplace(op_it, std::move(lstm_cell_op));
154 AddMessageF("Creating extended LstmCell replacing previous lstm cell");
155
156 // Delete arrays and operators replaced by the LSTM cell operator. Order is
157 // important - DeleteArrayIfUnused() only succeeds if dependent operators
158 // have been removed first. Start at the output and work towards the input.
159 // Erase curr lstm op being replaced.
160 DeleteArrayIfUnused(curr_op->inputs[LstmCellOperator::WEIGHTS_INPUT], model);
161 DeleteArrayIfUnused(curr_op->inputs[LstmCellOperator::BIASES_INPUT], model);
162 DeleteArrayIfUnused(curr_op->inputs[LstmCellOperator::PREV_ACTIV_INPUT],
163 model);
164 DeleteArrayIfUnused(curr_op->inputs[LstmCellOperator::PREV_STATE_INPUT],
165 model);
166 model->operators.erase(FindOp(*model, curr_op));
167
168 return true;
169}
