32bool OperatorReady(const Model& model, const Operator* op) {
33 if (!model.HasArray(op->inputs[0]) || !model.HasArray(op->inputs[1]) ||
34 !model.HasArray(op->outputs[0])) {
35 // Arrays are missing.
36 return false;
37 }
38
39 if (!model.GetArray(op->inputs[0]).has_shape() ||
40 !model.GetArray(op->outputs[0]).has_shape()) {
41 // Input and output needs the shape.
42 return false;
43 }
44
45 if (!model.GetArray(op->inputs[1]).buffer) {
46 // Buffer needs to be a constant.
47 return false;
48 }
49
50 return true;
51}

53// Returns whether the reshape could be a transpose.
54std::vector<int32> ReshapeToTranspose(const Model& model,
55 const TensorFlowReshapeOperator* op) {
56 CHECK(!op->shape.empty());
57 CHECK(model.HasArray(op->inputs[0]));
58 CHECK(model.HasArray(op->outputs[0]));
59
60 const auto& input_array = model.GetArray(op->inputs[0]);
61 const auto& output_array = model.GetArray(op->outputs[0]);
62
63 CHECK(input_array.has_shape());
64 CHECK(output_array.has_shape());
65
66 std::vector<int> in_shape = input_array.shape().dims();
67 std::vector<int> out_shape = output_array.shape().dims();
68
69 std::vector<int> one_indices;
70 std::vector<int> not_one_indices;
71
72 // Separate into one indices and not one indices.
73 for (int i = 0; i < in_shape.size(); i++) {
74 if (in_shape[i] == 1) {
75 one_indices.push_back(i);
76 } else {
77 not_one_indices.push_back(i);
78 }
79 }
80
81 // Reorder the vertices.
82 std::vector<int> perm;
83 perm.reserve(in_shape.size());
84 int one_index = 0;
85 int not_one_index = 0;
86 for (const auto val : out_shape) {
87 if (val == 1) {
88 perm.push_back(one_indices[one_index]);
89 one_index++;
90 } else {
91 perm.push_back(not_one_indices[not_one_index]);
92 not_one_index++;
93 }
94 }
95
96 return perm;
97}

101// When a transpose is fed into a reshape, it is possible for the two operators
102// to be merged if the reshape does not affect memory ordering and does not
103// affects the number of dimensions. This only occurs when only unary dimensions
104// are shifting position.
105bool MergeReshapeIntoPrecedingTranspose::Run(Model* model,
106 std::size_t op_index) {
107 auto it = model->operators.begin() + op_index;
108 auto* reshape_op = ConvertOperator<TensorFlowReshapeOperator*>(
109 it->get(), OperatorType::kTensorFlowReshape);
110
111 if (reshape_op == nullptr) {
112 return false;
113 }
114
115 if (!OperatorReady(*model, reshape_op) || reshape_op->shape.empty()) {
116 return false;
117 }
118
119 const string intermediate_name = reshape_op->inputs[0];
120 const string output_name = reshape_op->outputs[0];
121
122 // Guarantee the input is only consume by the reshape.
123 if (CountOpsWithInput(*model, intermediate_name) != 1) {
124 return false;
125 }
126
127 // Check for the parent operator.
128 const auto& transpose_it = FindOpWithOutput(*model, intermediate_name);
129 if (transpose_it == model->operators.end()) {
130 return false;
131 }
132
133 // Find the parent operator and guarantee it is a transpose.
134 TransposeOperator* transpose_op = ConvertOperator<TransposeOperator*>(
135 transpose_it->get(), OperatorType::kTranspose);
136
137 if (transpose_op == nullptr) {
138 return false;
139 }
140
141 if (!OperatorReady(*model, transpose_op) || transpose_op->perm.empty()) {
142 return false;
143 }
144
145 if (!ReshapeIsEquivalentToTranspose(*model, reshape_op,
146 false /*allow_extra_unary_dimensions*/)) {
147 return false;
148 }
149
150 // Check that the intermediate is not an output array.
151 if (!IsDiscardableArray(*model, intermediate_name)) {
152 AddMessageF(
153 "Cannot fuse %s and %s as it would invalidate the transpose "
154 "output array.",
155 LogName(*transpose_op), LogName(*reshape_op));
156 return false;
157 }
158
159 AddMessageF("Merging operations %s and %s", LogName(*transpose_op),
160 LogName(*reshape_op));
161
162 // const auto& intermediate_array = model->GetArray(intermediate_name);
163 // const auto& output_array = model->GetArray(output_name);
164
165 auto merged_perm = ReshapeToTranspose(*model, reshape_op);
166
167 // Combine the permutations.
168 const auto& transpose_perm = transpose_op->perm;
169 for (int i = 0; i < merged_perm.size(); i++) {
170 merged_perm[i] = transpose_perm[merged_perm[i]];
171 }
172
173 // Remove the reshape as passthrough operation.
174 if (!RemoveTrivialPassthroughOp(this, model, op_index)) {
175 return false;
176 }
177
178 // Update transpose_op's constant buffer to contain the new permutation.
179 model->GetArray(transpose_op->inputs[1])
180 .GetMutableBuffer<ArrayDataType::kInt32>()
181 .data = merged_perm;
182 transpose_op->perm = merged_perm;
183
184 // transpose_ops's shape will likely has changed.
185 model->GetArray(transpose_op->outputs[0]).clear_shape();
186
187 return true;
188}
