28bool CreateIm2colArrays::Run(Model* model, std::size_t op_index) {
29 auto conv_it = model->operators.begin() + op_index;
30 if (conv_it->get()->type != OperatorType::kConv) {
31 return false;
32 }
33 auto* conv_op = static_cast<ConvOperator*>(conv_it->get());
34 if (conv_op->outputs.size() == 2) {
35 // We already have an im2col array
36 return false;
37 }
38 const auto& weights_array = model->GetArray(conv_op->inputs[1]);
39 if (!weights_array.has_shape()) {
40 // We need to yield until weights dims have been resolved, because
41 // from the weights dims we determine whether an im2col array is
42 // needed.
43 return false;
44 }
45 const auto& weights_shape = weights_array.shape();
46 const int kheight = weights_shape.dims(1);
47 const int kwidth = weights_shape.dims(2);
48 if (kwidth == 1 && kheight == 1 && conv_op->stride_width == 1 &&
49 conv_op->stride_height == 1) {
50 // 1x1 unstrided conv does not need an im2col array.
51 return false;
52 }
53
54 // Create the im2col array.
55 CHECK_EQ(conv_op->outputs.size(), 1);
56 const string& im2col_array_name =
57 AvailableArrayName(*model, conv_op->inputs[0] + "_im2col");
58 model->GetOrCreateArray(im2col_array_name);
59 conv_op->outputs.push_back(im2col_array_name);
60 AddMessageF(
61 "Created an im2col array for %s, with %dx%d kernel and stride_width=%d, "
62 "stride_height=%d",
63 LogName(*conv_op), kwidth, kheight, conv_op->stride_width,
64 conv_op->stride_height);
65
66 return true;
67}
