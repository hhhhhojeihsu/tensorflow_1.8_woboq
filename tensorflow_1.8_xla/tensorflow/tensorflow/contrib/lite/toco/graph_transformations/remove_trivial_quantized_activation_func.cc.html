29bool RemoveTrivialQuantizedActivationFunc::Run(Model* model,
30 std::size_t op_index) {
31 const auto it = model->operators.begin() + op_index;
32 auto* op = it->get();
33 if (op->fused_activation_function != FusedActivationFunctionType::kRelu &&
34 op->fused_activation_function != FusedActivationFunctionType::kRelu1 &&
35 op->fused_activation_function != FusedActivationFunctionType::kRelu6) {
36 return false;
37 }
38 const auto& output_array = model->GetArray(op->outputs[0]);
39 if (!output_array.quantization_params) {
40 return false;
41 }
42 if (output_array.data_type != ArrayDataType::kUint8) {
43 return false;
44 }
45 const auto& quantization_params = output_array.GetQuantizationParams();
46
47 double clamp_min;
48 double clamp_max;
49 switch (op->fused_activation_function) {
50 case FusedActivationFunctionType::kRelu:
51 clamp_min = 0.0;
52 clamp_max = std::numeric_limits<double>::infinity();
53 break;
54 case FusedActivationFunctionType::kRelu1:
55 clamp_min = -1.0;
56 clamp_max = 1.0;
57 break;
58 case FusedActivationFunctionType::kRelu6:
59 clamp_min = 0.0;
60 clamp_max = 6.0;
61 break;
62 default:
63 LOG(FATAL) << "Unsupported fused activation type: "
64 << static_cast<int>(op->fused_activation_function);
65 return false;
66 }
67
68 bool has_nontrivial_min_bound = false;
69 bool has_nontrivial_max_bound = false;
70
71 double lowest_representable_output =
72 (0. - quantization_params.zero_point) * quantization_params.scale;
73 if (lowest_representable_output < clamp_min) {
74 has_nontrivial_min_bound = true;
75 AddMessageF(
76 "Quantized activation function is not trivial: "
77 "the lowest representable output value %g"
78 " less than the clamp min bound %g.",
79 lowest_representable_output, clamp_min);
80 }
81 double highest_representable_output =
82 (255. - quantization_params.zero_point) * quantization_params.scale;
83 if (highest_representable_output > clamp_max) {
84 has_nontrivial_max_bound = true;
85 AddMessageF(
86 "Quantized activation function is not trivial: "
87 "the highest representable output value %g"
88 " is greater than the clamp max bound %g.",
89 highest_representable_output, clamp_max);
90 }
91
92 if (has_nontrivial_min_bound || has_nontrivial_max_bound) {
93 return false;
94 }
95
96 op->fused_activation_function = FusedActivationFunctionType::kNone;
97 AddMessageF(
98 "Removing trivial quantized activation function on %s"
99 " because the output quantization parameters imply at least as tight"
100 " a clamp anyway.",
101 LogName(*op));
102 return true;
103}
