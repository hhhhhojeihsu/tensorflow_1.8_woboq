28bool DropFakeQuant::Run(Model* model, std::size_t op_index) {
29 const auto fakequant_it = model->operators.begin() + op_index;
30 auto* fakequant_base_op = fakequant_it->get();
31 if (fakequant_base_op->type != OperatorType::kFakeQuant) {
32 return false;
33 }
34 auto* fakequant_op = static_cast<FakeQuantOperator*>(fakequant_base_op);
35
36 if (!fakequant_op->minmax) {
37 return false;
38 }
39
40 const auto& output_array = model->GetArray(fakequant_op->outputs[0]);
41 if (!output_array.minmax) {
42 return false;
43 }
44
45 // Drop min/max inputs
46 for (int i = 1; i < fakequant_op->inputs.size(); i++) {
47 if (CountOpsWithInput(*model, fakequant_op->inputs[i]) == 1) {
48 model->EraseArray(fakequant_op->inputs[i]);
49 }
50 }
51 fakequant_op->inputs.resize(1);
52
53 return RemoveTrivialPassthroughOp(this, model, op_index);
54}
