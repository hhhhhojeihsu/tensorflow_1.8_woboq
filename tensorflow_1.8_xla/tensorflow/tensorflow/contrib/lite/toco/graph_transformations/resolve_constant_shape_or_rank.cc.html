22bool ResolveConstantShapeOrRank::Run(Model* model, std::size_t op_index) {
23 const auto it = model->operators.begin() + op_index;
24 const auto* op = it->get();
25 if (!(op->type == OperatorType::kTensorFlowShape ||
26 op->type == OperatorType::kRank)) {
27 return false;
28 }
29
30 CHECK_EQ(op->outputs.size(), 1);
31 auto& output_array = model->GetArray(op->outputs[0]);
32 if (output_array.data_type == ArrayDataType::kNone) {
33 // Yield until the output type has been resolved
34 return false;
35 }
36
37 const auto& input_array = model->GetArray(op->inputs[0]);
38 if (!input_array.has_shape()) {
39 // Yield until the input array's shape has been resolved.
40 return false;
41 }
42
43 if (!output_array.has_shape()) {
44 // Yield until the output shape has been resolved.
45 return false;
46 }
47
48 // Compute the output
49 CHECK(!output_array.buffer);
50 auto& output_buffer = output_array.GetMutableBuffer<ArrayDataType::kInt32>();
51 if (op->type == OperatorType::kTensorFlowShape) {
52 // Copy the input shape into the output buffer.
53 output_buffer.data = input_array.shape().dims();
54 } else if (op->type == OperatorType::kRank) {
55 // Copy the dimension count into the output buffer.
56 output_buffer.data.resize(1);
57 output_buffer.data[0] = input_array.shape().dimensions_count();
58 }
59 output_array.mutable_shape()->ReplaceDims(
60 {static_cast<int>(output_buffer.data.size())});
61
62 // Delete the input array if no longer used
63 if (IsDiscardableArray(*model, op->inputs[0]) &&
64 CountOpsWithInput(*model, op->inputs[0]) == 1) {
65 model->EraseArray(op->inputs[0]);
66 }
67
68 model->operators.erase(it);
69 return true;
70}
