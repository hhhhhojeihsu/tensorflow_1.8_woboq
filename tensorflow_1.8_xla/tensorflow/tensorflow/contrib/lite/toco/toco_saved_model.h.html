30// Parses metadata into `toco_flags` and `model_flags`.
31//
32// Stores `inputs` as input_arrays and `outputs` as output_arrays in
33// `model_flags`. Infers input_shapes from the GraphDef and stores it in
34// `model_flags` as part of the input_arrays. Assumes inference_type is FLOAT
35// and stores it in `toco_flags`.
36void ParseMetaData(const tensorflow::GraphDef& graph_def,
37 const std::unordered_set<string>& inputs,
38 const std::unordered_set<string>& outputs,
39 const ParsedTocoFlags& parsed_toco_flags,
40 const ParsedModelFlags& parsed_model_flags,
41 TocoFlags* toco_flags, ModelFlags* model_flags);
42
43// Generates a frozen graph from the SavedModel in the directory specified in
44// `toco_flags`. Reads frozen graph contents into `graph_def_contents`. Parses
45// metadata relating to the GraphDef into `toco_flags` and `model_flags`.
46void GetSavedModelContents(const ParsedTocoFlags& parsed_toco_flags,
47 const ParsedModelFlags& parsed_model_flags,
48 TocoFlags* toco_flags, ModelFlags* model_flags,
49 string* graph_def_contents);
