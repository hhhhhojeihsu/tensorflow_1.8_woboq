33MMAPAllocation::MMAPAllocation(const char* filename,
34 ErrorReporter* error_reporter)
35 : Allocation(error_reporter), mmapped_buffer_(MAP_FAILED) {
36 mmap_fd_ = open(filename, O_RDONLY);
37 if (mmap_fd_ == -1) {
38 error_reporter_->Report("Could not open '%s'.", filename);
39 return;
40 }
41 struct stat sb;
42 fstat(mmap_fd_, &sb);
43 buffer_size_bytes_ = sb.st_size;
44 mmapped_buffer_ =
45 mmap(nullptr, buffer_size_bytes_, PROT_READ, MAP_SHARED, mmap_fd_, 0);
46 if (mmapped_buffer_ == MAP_FAILED) {
47 error_reporter_->Report("Mmap of '%s' failed.", filename);
48 return;
49 }
50}
51
52MMAPAllocation::~MMAPAllocation() {
53 if (valid()) {
54 munmap(const_cast<void*>(mmapped_buffer_), buffer_size_bytes_);
55 }
56 if (mmap_fd_ != -1) close(mmap_fd_);
57}

59const void* MMAPAllocation::base() const { return mmapped_buffer_; }
60
61size_t MMAPAllocation::bytes() const { return buffer_size_bytes_; }
62
63bool MMAPAllocation::valid() const { return mmapped_buffer_ != MAP_FAILED; }

65FileCopyAllocation::FileCopyAllocation(const char* filename,
66 ErrorReporter* error_reporter)
67 : Allocation(error_reporter) {
68 // Obtain the file size, using an alternative method that is does not
69 // require fstat for more compatibility.
70 std::unique_ptr<FILE, decltype(&fclose)> file(fopen(filename, "rb"), fclose);
71 if (!file) {
72 error_reporter_->Report("Could not open '%s'.", filename);
73 return;
74 }
75 // TODO(ahentz): Why did you think using fseek here was better for finding
76 // the size?
77 struct stat sb;
78 if (fstat(fileno(file.get()), &sb) != 0) {
79 error_reporter_->Report("Failed to get file size of '%s'.", filename);
80 return;
81 }
82 buffer_size_bytes_ = sb.st_size;
83 std::unique_ptr<char[]> buffer(new char[buffer_size_bytes_]);
84 if (!buffer) {
85 error_reporter_->Report("Malloc of buffer to hold copy of '%s' failed.",
86 filename);
87 return;
88 }
89 size_t bytes_read =
90 fread(buffer.get(), sizeof(char), buffer_size_bytes_, file.get());
91 if (bytes_read != buffer_size_bytes_) {
92 error_reporter_->Report("Read of '%s' failed (too few bytes read).",
93 filename);
94 return;
95 }
96 copied_buffer_ = std::move(buffer);
97}
98
99FileCopyAllocation::~FileCopyAllocation() {}
100
101const void* FileCopyAllocation::base() const { return copied_buffer_.get(); }
102
103size_t FileCopyAllocation::bytes() const { return buffer_size_bytes_; }
104
105bool FileCopyAllocation::valid() const { return copied_buffer_ != nullptr; }
