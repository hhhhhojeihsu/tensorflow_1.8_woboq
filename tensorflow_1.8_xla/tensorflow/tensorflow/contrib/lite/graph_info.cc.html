22// Provide a range iterable wrapper for TfLiteIntArray* (C lists that TfLite
23// C api uses. Can't use the google array_view, since we can't depend on even
24// absl for embedded device reasons.
25// TODO(aselle): Move this into central utilities.
26class TfLiteIntArrayView {
27 public:
28 // Construct a view of a TfLiteIntArray*. Note, `int_array` should be non-null
29 // and this view does not take ownership of it.
30 explicit TfLiteIntArrayView(const TfLiteIntArray* int_array)
31 : int_array_(int_array) {}
32
33 typedef const int* const_iterator;
34 const_iterator begin() const { return int_array_->data; }
35 const_iterator end() const { return &int_array_->data[int_array_->size]; }
36
37 TfLiteIntArrayView(const TfLiteIntArrayView&) = default;
38 TfLiteIntArrayView& operator=(const TfLiteIntArrayView& rhs) = default;
39
40 private:
41 const TfLiteIntArray* int_array_;
42};

44// Helper class that actually performs partitioning by subgraph.
45// Outputs to a provided `subgraphs` structure.
46//
47// Example usage:
48// PartitionGraphIntoIndependentSubgraphsImpl partitioner(
49// info, nodes_to_part, subgraphs);
50// partitioner.Partition();
51class PartitionGraphIntoIndependentSubgraphsImpl {
52 public:
53 PartitionGraphIntoIndependentSubgraphsImpl(
54 const GraphInfo* info, const TfLiteIntArray* nodes_to_partition,
55 std::vector<Subgraph>* subgraphs)
56 : info_(info),
57 subgraphs_(subgraphs),
58 node_type_(info->num_nodes(), Subgraph::kTfNonPartition) {
59 // Populate the node_type_ map.
60 for (auto node_index : TfLiteIntArrayView(nodes_to_partition)) {
61 node_type_[node_index] = Subgraph::kTfPartition;
62 }
63 }
64
65 // Actually partition the graph.
66 void Partition() {
67 // Initialize here to make Partition() re-entrant.
68 subgraphs_->clear();
69 tensor_epochs_.clear();
70 tensor_epochs_.resize(info_->num_tensors(), kEpochAlwaysReady);
71 node_epochs_.clear();
72 node_epochs_.resize(info_->num_nodes(), kEpochNotReady);
73 // Set computed tensors to be kEpochNotReady (initializer set everything to
74 // AlwaysReady).
75 for (int node_index = 0; node_index < info_->num_nodes(); node_index++) {
76 const TfLiteNode& node = info_->node(node_index);
77 for (int output_tensor_index : TfLiteIntArrayView(node.outputs)) {
78 tensor_epochs_[output_tensor_index] = kEpochNotReady;
79 }
80 }
81
82 // Do a graph traversal where each iteration in the loop is an epoch
83 // that corresponds to a subgraph that only contains nodes that are of
84 // the same node_type_.
85 while (true) {
86 BuildSubgraph();
87 if (subgraphs_->back().nodes.empty()) {
88 subgraphs_->pop_back();
89 break;
90 }
91 }
92
93 // Mark model outputs as subgraph outputs. All the rest have already been
94 // identified.
95 for (int output_index : info_->outputs()) {
96 int output_epoch = tensor_epochs_[output_index];
97 Subgraph& output_subgraph = (*subgraphs_)[output_epoch];
98 output_subgraph.output_tensors.push_back(output_index);
99 }
100 // Make sure every subgraph's inputs and outputs are unique. Since the
101 // list of inputs and outputs is generated in a way that produces
102 // duplicates.
103 for (Subgraph& subgraph : *subgraphs_) {
104 // Sort and uniquefy using standard library algorithms.
105 auto uniquefy = [](std::vector<int>* items) {
106 std::sort(items->begin(), items->end());
107 auto last = std::unique(items->begin(), items->end());
108 items->erase(last, items->end());
109 };
110 uniquefy(&subgraph.input_tensors);
111 uniquefy(&subgraph.output_tensors);
112 }
113 }
114
115 private:
116 // Special integer values needed for tensor_epochs_ and node_epochs_.
117 enum {
118 // The node or tensor is not ready to be assigned an epoch. e.g. a node's
119 // inputs have not all been assigned epochs.
120 kEpochNotReady = -1,
121 // Used for tensor_epochs_. This means that the tensor is always ready.
122 // e.g. an input to the whole model or a constant that has no dependencies.
123 kEpochAlwaysReady = -2
124 };
125
126 // Updates the node `node_index` and returns true if it is assigned to an
127 // epoch. False is returned if the node is already set to an epoch, its inputs
128 // are not all assigned to epochs, or if it cannot be assigned to the current
129 // epoch since the epoch's node_type doesn't match.
130 bool UpdateNode(int node_index) {
131 const TfLiteNode& node = info_->node(node_index);
132 Subgraph& current_subgraph = subgraphs_->back();
133 int current_epoch = subgraphs_->size() - 1;
134 // Check if node is already done.
135 if (node_epochs_[node_index] != kEpochNotReady) {
136 return false;
137 }
138 // See if all dependencies of this node are already assigned to a
139 // subgraph.
140 for (int input_tensor_index : TfLiteIntArrayView(node.inputs)) {
141 if (tensor_epochs_[input_tensor_index] == kEpochNotReady) {
142 return false;
143 }
144 }
145 // When we are starting a new epoch, the first ready node defines
146 // the type of that epoch.
147 if (current_subgraph.type == Subgraph::kTfUnexplored) {
148 current_subgraph.type = node_type_[node_index];
149 }
150 // The node gets assigned to this epoch if it is the same type as
151 // the epoch's assigned type. Note, if this is the current ready
152 // node encountered during this epoch, this condition will be
153 // automatically true.
154 if (current_subgraph.type == node_type_[node_index]) {
155 node_epochs_[node_index] = current_epoch;
156 current_subgraph.nodes.push_back(node_index);
157 // All outputs of this node now are assigned to this epoch as
158 // well.
159 for (int output_tensor_index : TfLiteIntArrayView(node.outputs)) {
160 tensor_epochs_[output_tensor_index] = current_epoch;
161 }
162 // Look at our inputs one more time to update that tensor's
163 // epochs' outputs
164 for (int input_tensor_index : TfLiteIntArrayView(node.inputs)) {
165 int input_epoch = tensor_epochs_[input_tensor_index];
166 int node_epoch = current_epoch;
167 if (input_epoch != node_epoch) {
168 current_subgraph.input_tensors.push_back(input_tensor_index);
169 // Set inputs to be outputs of the subgraph where they reside.
170 // the if condition makes sure inputs to the whole computation
171 // are not included (i.e. those initialized to -2 above).
172 if (input_epoch >= 0) {
173 Subgraph& input_subgraph = (*subgraphs_)[input_epoch];
174 input_subgraph.output_tensors.push_back(input_tensor_index);
175 }
176 }
177 }
178 return true;
179 } else {
180 return false;
181 }
182 }
183
184 // Completely populates the current subgraph by doing graph traversal
185 void BuildSubgraph() {
186 subgraphs_->emplace_back(Subgraph());
187 // loop until no more nodes can be updated.
188 while (true) {
189 bool did_something = false;
190 for (int node_index = 0; node_index < info_->num_nodes(); node_index++) {
191 if (UpdateNode(node_index)) {
192 did_something = true;
193 }
194 }
195 if (!did_something) return;
196 }
197 }
198
199 // Temporary data needed for partitioning.
200 const GraphInfo* info_;
201 // List of subgraphs to populate
202 std::vector<Subgraph>* subgraphs_;
203 std::vector<Subgraph::Type> node_type_;
204 // Maps from tensor index to the epoch in which it is assigned. Also special
205 // negative values of kEpochNotAssigned if not assigned, kEpochNotReady if it
206 // is an input or constant.
207 std::vector<int> tensor_epochs_;
208 // Maps from tensor index to the epoch in which it is assigned. Also special
209 // negative values of kEpochNotAssigned if not assigned.
210 std::vector<int> node_epochs_;
211};
212
213} // namespace
214
215TfLiteStatus PartitionGraphIntoIndependentSubgraphs(
216 const GraphInfo* info, const TfLiteIntArray* nodes_to_partition,
217 std::vector<Subgraph>* subgraphs) {
218 PartitionGraphIntoIndependentSubgraphsImpl(info, nodes_to_partition,
219 subgraphs)
220 .Partition();
221 return kTfLiteOk;
222}
