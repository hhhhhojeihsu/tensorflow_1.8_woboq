53inline void GetActivationMinMax(FusedActivationFunctionType ac,
54 float* output_activation_min,
55 float* output_activation_max) {
56 switch (ac) {
57 case FusedActivationFunctionType::kNone:
58 *output_activation_min = std::numeric_limits<float>::lowest();
59 *output_activation_max = std::numeric_limits<float>::max();
60 break;
61 case FusedActivationFunctionType::kRelu:
62 *output_activation_min = 0.f;
63 *output_activation_max = std::numeric_limits<float>::max();
64 break;
65 case FusedActivationFunctionType::kRelu1:
66 *output_activation_min = -1.f;
67 *output_activation_max = 1.f;
68 break;
69 case FusedActivationFunctionType::kRelu6:
70 *output_activation_min = 0.f;
71 *output_activation_max = 6.f;
72 break;
73 }
74}
75
76inline float ActivationFunctionWithMinMax(float x, float output_activation_min,
77 float output_activation_max) {
78 return std::min(std::max(x, output_activation_min), output_activation_max);
79}
80
81// Legacy function, left for compatibility only.
82template <FusedActivationFunctionType Ac>
83float ActivationFunction(float x) {
84 float output_activation_min, output_activation_max;
85 GetActivationMinMax(Ac, &output_activation_min, &output_activation_max);
86 return ActivationFunctionWithMinMax(x, output_activation_min,
87 output_activation_max);
88}

90inline int32 MultiplyByQuantizedMultiplierSmallerThanOne(
91 int32 x, int32 quantized_multiplier, int right_shift) {
92 using gemmlowp::RoundingDivideByPOT;
93 using gemmlowp::SaturatingRoundingDoublingHighMul;
94 return RoundingDivideByPOT(
95 SaturatingRoundingDoublingHighMul(x, quantized_multiplier), right_shift);
96}
97
98inline int32 MultiplyByQuantizedMultiplierGreaterThanOne(
99 int32 x, int32 quantized_multiplier, int left_shift) {
100 using gemmlowp::SaturatingRoundingDoublingHighMul;
101 return SaturatingRoundingDoublingHighMul(x * (1 << left_shift),
102 quantized_multiplier);
103}
104
105inline int32 MultiplyByQuantizedMultiplier(int32 x, int32 quantized_multiplier,
106 int shift) {
107 using gemmlowp::RoundingDivideByPOT;
108 using gemmlowp::SaturatingRoundingDoublingHighMul;
109 int left_shift = shift > 0 ? shift : 0;
110 int right_shift = shift > 0 ? 0 : -shift;
111 return RoundingDivideByPOT(SaturatingRoundingDoublingHighMul(
112 x * (1 << left_shift), quantized_multiplier),
113 right_shift);
114}
