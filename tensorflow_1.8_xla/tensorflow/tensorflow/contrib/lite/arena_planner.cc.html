21// Memory allocation tuning
22constexpr const int kDefaultArenaAlignment = 64;
23constexpr const int kDefaultTensorAlignment = 4;

27struct AllocationInfo {
28 // The node index requesting this allocation.
29 int node;
30 // The tensor index to be allocated or deallocated.
31 int tensor;
32 // Whether to allocate or deallocate
33 enum { ALLOC, DEALLOC } type;
34};

36ArenaPlanner::ArenaPlanner(TfLiteContext* context,
37 std::unique_ptr<GraphInfo> graph_info)
38 : context_(context),
39 graph_info_(std::move(graph_info)),
40 arena_(kDefaultArenaAlignment),
41 persistent_arena_(kDefaultArenaAlignment) {}

45int64_t ArenaPlanner::BasePointer(TfLiteAllocationType type) {
46 if (type == kTfLiteArenaRwPersistent) {
47 return persistent_arena_.BasePointer();
48 }
49 if (type == kTfLiteArenaRw) {
50 return arena_.BasePointer();
51 }
52 return 0;
53}

55TfLiteStatus ArenaPlanner::ResetAllocations() {
56 TF_LITE_ENSURE_STATUS(arena_.Clear());
57 TF_LITE_ENSURE_STATUS(persistent_arena_.Clear());
58 allocs_.clear();
59 allocs_.resize(graph_info_->num_tensors());
60 return kTfLiteOk;
61}

63TfLiteStatus ArenaPlanner::PlanAllocations() {
64 // Invalidate any existing data.
65 TF_LITE_ENSURE_STATUS(ResetAllocations());
66
67 // Keeps track of references to each tensor.
68 std::vector<int> refcounts(graph_info_->num_tensors(), 0);
69
70 // There will be an entry in alloc_queue_ for the allocation of each tensor
71 // and another for their deallocation.
72 alloc_queue_.reserve(2 * graph_info_->num_tensors());
73
74 // We must make sure the output tensors are never overwritten. We do that by
75 // artificially adding one to their ref-counts so they are never selected
76 // for deallocation.
77 for (int tensor_index : graph_info_->outputs()) {
78 refcounts[tensor_index]++;
79 }
80
81 // Count references to node input tensors.
82 for (int i = 0; i < graph_info_->num_nodes(); ++i) {
83 const TfLiteNode& node = graph_info_->node(i);
84 TfLiteIntArray* node_inputs = node.inputs;
85 for (int j = 0; j < node_inputs->size; ++j) {
86 int tensor_index = node_inputs->data[j];
87 if (tensor_index != kOptionalTensor) {
88 refcounts[tensor_index]++;
89 }
90 }
91 }
92
93 // Queue all graph inputs for allocation.
94 for (int tensor_index : graph_info_->inputs()) {
95 if (tensor_index != kOptionalTensor) {
96 alloc_queue_.push_back({0, tensor_index, AllocationInfo::ALLOC});
97 }
98 }
99
100 // Go through the graph in execution order.
101 for (int i = 0; i < graph_info_->num_nodes(); ++i) {
102 const TfLiteNode& node = graph_info_->node(i);
103
104 // First queue output tensors for allocation.
105 TfLiteIntArray* node_outputs = node.outputs;
106 for (int j = 0; j < node_outputs->size; ++j) {
107 int tensor_index = node_outputs->data[j];
108 alloc_queue_.push_back({i, tensor_index, AllocationInfo::ALLOC});
109 }
110
111 // Then update the ref-counts of the node's inputs, and if necessary queue
112 // them for deallocation.
113 TfLiteIntArray* node_inputs = node.inputs;
114 for (int j = 0; j < node_inputs->size; ++j) {
115 int tensor_index = node_inputs->data[j];
116 if (tensor_index != kOptionalTensor) {
117 refcounts[tensor_index]--;
118 if (refcounts[tensor_index] == 0) {
119 alloc_queue_.push_back({i, tensor_index, AllocationInfo::DEALLOC});
120 }
121 }
122 }
123 }
124
125 // Note that graph outputs will never be scheduled for deallocation. We
126 // could do that here for completeness, but it won't have any effect.
127 return kTfLiteOk;
128}

130TfLiteStatus ArenaPlanner::ExecuteAllocations(int first_node, int last_node) {
131 // Grow the size of `allocs_` if necessary. This allows allocating temporary
132 // tensors in op's `prepare` function.
133 TF_LITE_ENSURE(context_, graph_info_->num_tensors() >= allocs_.size());
134 allocs_.resize(graph_info_->num_tensors());
135
136 TF_LITE_ENSURE_STATUS(CalculateAllocations(first_node, last_node));
137 TF_LITE_ENSURE_STATUS(Commit());
138
139 for (int i = 0; i < graph_info_->num_tensors(); ++i) {
140 // TODO(ahentz): we could do this only for the tensors that were modified
141 // in CalculateAllocations(), instead of redoing it for tensors that
142 // already had proper pointers. However we must be very careful, because
143 // SimpleMemoryArena::Commit() could move the base pointer.
144 TF_LITE_ENSURE_STATUS(ResolveTensorAllocation(i));
145 }
146
147 return kTfLiteOk;
148}

150TfLiteStatus ArenaPlanner::Commit() {
151 TF_LITE_ENSURE_STATUS(arena_.Commit(context_));
152 TF_LITE_ENSURE_STATUS(persistent_arena_.Commit(context_));
153 return kTfLiteOk;
154}

156TfLiteStatus ArenaPlanner::CalculateAllocations(int first_node, int last_node) {
157 int active_node = first_node;
158 // When dynamic tensors are present this method is called multiple times.
159 // The items in the alloc_queue_ referring to nodes before first_node were
160 // processed previously and should be skipped. Entries after last_node are
161 // not yet ready to be handled.
162 for (const auto& alloc_info : alloc_queue_) {
163 if (alloc_info.node < first_node) continue;
164 if (alloc_info.node > last_node) break;
165 if (alloc_info.node == active_node) {
166 // This is the first allocation/deallocation for a given node. It is
167 // time to deallocate the previous temporaries and allocate new ones.
168 if (active_node != first_node) {
169 TF_LITE_ENSURE_STATUS(
170 CalculateDeallocationOfInternalTensors(active_node - 1));
171 }
172 TF_LITE_ENSURE_STATUS(CalculateAllocationOfInternalTensors(active_node));
173 ++active_node;
174 }
175 // Handle the current item.
176 if (alloc_info.type == AllocationInfo::ALLOC) {
177 TF_LITE_ENSURE_STATUS(CalculateTensorAllocation(alloc_info.tensor));
178 } else {
179 TF_LITE_ENSURE_STATUS(CalculateTensorDeallocation(alloc_info.tensor));
180 }
181 }
182
183 // Don't forget to deallocate temporaries of last node.
184 TF_LITE_ENSURE_STATUS(
185 CalculateDeallocationOfInternalTensors(active_node - 1));
186
187 return kTfLiteOk;
188}

190TfLiteStatus ArenaPlanner::ResolveTensorAllocation(int tensor_index) {
191 TfLiteTensor& tensor = *graph_info_->tensor(tensor_index);
192 if (tensor.allocation_type == kTfLiteArenaRw) {
193 // Skip resolution if the size of the tensor is zero, leaving it as a
194 // nullptr.
195 if (allocs_[tensor_index].size != 0) {
196 TF_LITE_ENSURE_STATUS(arena_.ResolveAlloc(context_, allocs_[tensor_index],
197 &tensor.data.raw));
198 }
199 }
200 if (tensor.allocation_type == kTfLiteArenaRwPersistent) {
201 TF_LITE_ENSURE_STATUS(persistent_arena_.ResolveAlloc(
202 context_, allocs_[tensor_index], &tensor.data.raw));
203 }
204 return kTfLiteOk;
205}

207TfLiteStatus ArenaPlanner::CalculateTensorAllocation(int tensor_index) {
208 TfLiteTensor& tensor = *graph_info_->tensor(tensor_index);
209 if (tensor.allocation_type == kTfLiteArenaRw) {
210 TF_LITE_ENSURE_STATUS(arena_.Allocate(context_, kDefaultTensorAlignment,
211 tensor.bytes,
212 &allocs_[tensor_index]));
213 }
214 if (tensor.allocation_type == kTfLiteArenaRwPersistent) {
215 TF_LITE_ENSURE_STATUS(
216 persistent_arena_.Allocate(context_, kDefaultTensorAlignment,
217 tensor.bytes, &allocs_[tensor_index]));
218 }
219 return kTfLiteOk;
220}
221
222TfLiteStatus ArenaPlanner::CalculateTensorDeallocation(int tensor_index) {
223 TfLiteTensor& tensor = *graph_info_->tensor(tensor_index);
224 if (tensor.allocation_type == kTfLiteArenaRw) {
225 TF_LITE_ENSURE_STATUS(arena_.Deallocate(context_, allocs_[tensor_index]));
226 }
227 return kTfLiteOk;
228}

230TfLiteStatus ArenaPlanner::CalculateAllocationOfInternalTensors(
231 int node_index) {
232 if (node_index < graph_info_->num_nodes()) {
233 const TfLiteNode& node = graph_info_->node(node_index);
234 TfLiteIntArray* node_temporaries = node.temporaries;
235 for (int i = 0; i < node_temporaries->size; ++i) {
236 int tensor_index = node_temporaries->data[i];
237 TF_LITE_ENSURE_STATUS(CalculateTensorAllocation(tensor_index));
238 }
239 }
240 return kTfLiteOk;
241}
242
243TfLiteStatus ArenaPlanner::CalculateDeallocationOfInternalTensors(
244 int node_index) {
245 if (node_index < graph_info_->num_nodes()) {
246 const TfLiteNode& node = graph_info_->node(node_index);
247 TfLiteIntArray* node_temporaries = node.temporaries;
248 for (int i = 0; i < node_temporaries->size; ++i) {
249 int tensor_index = node_temporaries->data[i];
250 TF_LITE_ENSURE_STATUS(CalculateTensorDeallocation(tensor_index));
251 }
252 }
253 return kTfLiteOk;
254}
