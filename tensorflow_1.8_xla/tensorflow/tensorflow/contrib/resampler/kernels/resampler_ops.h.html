30// Helper functor for the Resampler Op in 2D
31template <typename Device, typename T>
32struct Resampler2DFunctor {
33 void operator()(::tensorflow::OpKernelContext* ctx, const Device& d,
34 const T* __restrict__ data, const T* __restrict__ warp,
35 T* __restrict__ output, const int batch_size,
36 const int data_height, const int data_width,
37 const int data_channels, const int num_sampling_points);
38};

46 T* __restrict__ grad_warp, const int batch_size,
47 const int data_height, const int data_width,
48 const int data_channels, const int num_sampling_points);
49};
