46class BaseBuildSplitOp : public OpKernel {
47 public:
48 explicit BaseBuildSplitOp(OpKernelConstruction* const context)
49 : OpKernel(context) {
50 OP_REQUIRES_OK(context, context->GetAttr("feature_column_group_id",
51 &feature_column_group_id_));
52 OP_REQUIRES_OK(context,
53 context->GetAttr("l1_regularization", &l1_regularization_));
54 OP_REQUIRES_OK(context,
55 context->GetAttr("l2_regularization", &l2_regularization_));
56 OP_REQUIRES_OK(context, context->GetAttr("tree_complexity_regularization",
57 &tree_complexity_regularization_));
58 OP_REQUIRES_OK(context,
59 context->GetAttr("min_node_weight", &min_node_weight_));
60
61 int strategy;
62 OP_REQUIRES_OK(context, context->GetAttr("multiclass_strategy", &strategy));
63 OP_REQUIRES(
64 context,
65 boosted_trees::learner::LearnerConfig_MultiClassStrategy_IsValid(
66 strategy),
67 errors::InvalidArgument("Wrong multiclass strategy passed."));
68 multiclass_strategy_ = LearnerConfig_MultiClassStrategy(strategy);
69 }
70
71 NodeStats ComputeNodeStats(const GradientStats& grad_stats) {
72 return NodeStats(l1_regularization_, l2_regularization_, min_node_weight_,
73 multiclass_strategy_, grad_stats);
74 }
75
76 void ReadClassId(OpKernelContext* const context, int32* class_id) {
77 const Tensor* class_id_t;
78 OP_REQUIRES_OK(context, context->input("class_id", &class_id_t));
79 OP_REQUIRES(context, TensorShapeUtils::IsScalar(class_id_t->shape()),
80 errors::InvalidArgument("class_id must be a scalar."));
81 *class_id = class_id_t->scalar<int32>()();
82 }
83
84 void FillLeaf(const int class_id, const NodeStats& best_node_stats,
85 boosted_trees::trees::Leaf* leaf) const {
86 if (class_id == -1) {
87 // This would be the case either for TREE_PER_CLASS with only 2 classes,
88 // or for other multiclass strategies.
89 for (float f : best_node_stats.weight_contribution) {
90 leaf->mutable_vector()->add_value(f);
91 }
92 } else {
93 CHECK(best_node_stats.weight_contribution.size() == 1)
94 << "Weight contribution size = "
95 << best_node_stats.weight_contribution.size();
96 leaf->mutable_sparse_vector()->add_index(class_id);
97 leaf->mutable_sparse_vector()->add_value(
98 best_node_stats.weight_contribution[0]);
99 }
100 }
101
102 protected:
103 LearnerConfig_MultiClassStrategy multiclass_strategy_;
104 int32 feature_column_group_id_;
105 float l1_regularization_;
106 float l2_regularization_;
107 float min_node_weight_;
108 float tree_complexity_regularization_;
109};

111class BuildDenseInequalitySplitsOp : public BaseBuildSplitOp {
112 public:
113 explicit BuildDenseInequalitySplitsOp(OpKernelConstruction* const context)
114 : BaseBuildSplitOp(context) {}
115
116 void Compute(OpKernelContext* const context) override {
117 const Tensor* num_minibatches_t;
118 OP_REQUIRES_OK(context,
119 context->input("num_minibatches", &num_minibatches_t));
120 const int64 num_minibatches = num_minibatches_t->scalar<int64>()();
121 const float normalizer_ratio = (1.0f / num_minibatches);
122
123 const Tensor* bucket_boundaries_t;
124 OP_REQUIRES_OK(context,
125 context->input("bucket_boundaries", &bucket_boundaries_t));
126 const auto& bucket_boundaries = bucket_boundaries_t->vec<float>();
127
128 const Tensor* partition_ids_t;
129 OP_REQUIRES_OK(context, context->input("partition_ids", &partition_ids_t));
130 const auto& partition_ids = partition_ids_t->vec<int32>();
131
132 const Tensor* bucket_ids_t;
133 OP_REQUIRES_OK(context, context->input("bucket_ids", &bucket_ids_t));
134 const auto& bucket_ids = bucket_ids_t->matrix<int64>();
135
136 const Tensor* gradients_t;
137 OP_REQUIRES_OK(context, context->input("gradients", &gradients_t));
138
139 const Tensor* hessians_t;
140 OP_REQUIRES_OK(context, context->input("hessians", &hessians_t));
141
142 int class_id;
143 ReadClassId(context, &class_id);
144
145 // Find the number of unique partitions before we allocate the output.
146 std::vector<int32> partition_boundaries;
147 partition_boundaries.push_back(0);
148 for (int i = 1; i < partition_ids.size(); ++i) {
149 if (partition_ids(i) != partition_ids(i - 1)) {
150 // Make sure the input is sorted by partition_ids;
151 OP_REQUIRES(context, partition_ids(i) >= partition_ids(i - 1),
152 errors::InvalidArgument("Input should be sorted."));
153 partition_boundaries.push_back(i);
154 }
155 }
156 if (partition_ids.size() > 0) {
157 partition_boundaries.push_back(partition_ids.size());
158 }
159 int32 num_elements = partition_boundaries.size() - 1;
160
161 // When the handler is inactive, no bucket boundaries are built for it.
162 if (bucket_boundaries.size() == 0) {
163 num_elements = 0;
164 }
165
166 Tensor* output_partition_ids_t = nullptr;
167 OP_REQUIRES_OK(context,
168 context->allocate_output("output_partition_ids",
169 TensorShape({num_elements}),
170 &output_partition_ids_t));
171
172 tensorflow::TTypes<int32>::Vec output_partition_ids =
173 output_partition_ids_t->vec<int32>();
174
175 Tensor* gains_t = nullptr;
176 OP_REQUIRES_OK(
177 context, context->allocate_output("gains", TensorShape({num_elements}),
178 &gains_t));
179
180 tensorflow::TTypes<float>::Vec gains = gains_t->vec<float>();
181
182 Tensor* output_splits_t = nullptr;
183 OP_REQUIRES_OK(context, context->allocate_output(
184 "split_infos", TensorShape({num_elements}),
185 &output_splits_t));
186 tensorflow::TTypes<string>::Vec output_splits =
187 output_splits_t->vec<string>();
188 for (int root_idx = 0; root_idx < num_elements; ++root_idx) {
189 float best_gain = std::numeric_limits<float>::lowest();
190 int start_index = partition_boundaries[root_idx];
191 int end_index = partition_boundaries[root_idx + 1];
192 GradientStats root_gradient_stats;
193 for (int64 bucket_idx = start_index; bucket_idx < end_index;
194 ++bucket_idx) {
195 root_gradient_stats +=
196 GradientStats(*gradients_t, *hessians_t, bucket_idx);
197 }
198 root_gradient_stats *= normalizer_ratio;
199 NodeStats root_stats = ComputeNodeStats(root_gradient_stats);
200 int32 best_bucket_idx = 0;
201 NodeStats best_right_node_stats(0);
202 NodeStats best_left_node_stats(0);
203 GradientStats left_gradient_stats;
204 for (int64 bucket_idx = start_index; bucket_idx < end_index;
205 ++bucket_idx) {
206 GradientStats g(*gradients_t, *hessians_t, bucket_idx);
207 g *= normalizer_ratio;
208 left_gradient_stats += g;
209 NodeStats left_stats = ComputeNodeStats(left_gradient_stats);
210 GradientStats right_gradient_stats =
211 root_gradient_stats - left_gradient_stats;
212 NodeStats right_stats = ComputeNodeStats(right_gradient_stats);
213 if (left_stats.gain + right_stats.gain > best_gain) {
214 best_gain = left_stats.gain + right_stats.gain;
215 best_left_node_stats = left_stats;
216 best_right_node_stats = right_stats;
217 best_bucket_idx = bucket_idx;
218 }
219 }
220 SplitInfo split_info;
221 auto* dense_split =
222 split_info.mutable_split_node()->mutable_dense_float_binary_split();
223 dense_split->set_feature_column(feature_column_group_id_);
224 dense_split->set_threshold(
225 bucket_boundaries(bucket_ids(best_bucket_idx, 0)));
226
227 auto* left_child = split_info.mutable_left_child();
228 auto* right_child = split_info.mutable_right_child();
229
230 FillLeaf(class_id, best_left_node_stats, left_child);
231 FillLeaf(class_id, best_right_node_stats, right_child);
232 split_info.SerializeToString(&output_splits(root_idx));
233 gains(root_idx) =
234 best_gain - root_stats.gain - tree_complexity_regularization_;
235 output_partition_ids(root_idx) = partition_ids(start_index);
236 }
237 }
238};
239REGISTER_KERNEL_BUILDER(Name("BuildDenseInequalitySplits").Device(DEVICE_CPU),
240 BuildDenseInequalitySplitsOp);

242class BuildSparseInequalitySplitsOp : public BaseBuildSplitOp {
243 public:
244 explicit BuildSparseInequalitySplitsOp(OpKernelConstruction* const context)
245 : BaseBuildSplitOp(context) {
246 OP_REQUIRES_OK(context,
247 context->GetAttr("bias_feature_id", &bias_feature_id_));
248 }
249
250 void Compute(OpKernelContext* const context) override {
251 const Tensor* num_minibatches_t;
252 OP_REQUIRES_OK(context,
253 context->input("num_minibatches", &num_minibatches_t));
254 const int64 num_minibatches = num_minibatches_t->scalar<int64>()();
255 const float normalizer_ratio = (1.0f / num_minibatches);
256
257 const Tensor* bucket_boundaries_t;
258 OP_REQUIRES_OK(context,
259 context->input("bucket_boundaries", &bucket_boundaries_t));
260 const auto& bucket_boundaries = bucket_boundaries_t->vec<float>();
261
262 const Tensor* partition_ids_t;
263 OP_REQUIRES_OK(context, context->input("partition_ids", &partition_ids_t));
264 const auto& partition_ids = partition_ids_t->vec<int32>();
265
266 const Tensor* bucket_ids_t;
267 OP_REQUIRES_OK(context, context->input("bucket_ids", &bucket_ids_t));
268 const auto& bucket_ids_and_dimensions = bucket_ids_t->matrix<int64>();
269
270 const int32 tensor_elements = partition_ids.size();
271
272 const Tensor* gradients_t;
273 OP_REQUIRES_OK(context, context->input("gradients", &gradients_t));
274
275 const Tensor* hessians_t;
276 OP_REQUIRES_OK(context, context->input("hessians", &hessians_t));
277
278 int class_id;
279 ReadClassId(context, &class_id);
280
281 // For each partition (tree node), store starting index for each dimension.
282 PartitionAndDimensionBoundaries partition_boundaries;
283 // Stores indices in partition_boundaries for those partitions that are
284 // not empty (have at least one dimension and a bucket apart from catch-all
285 // bucket of -1 bucket id and dimension 0.
286 std::vector<int32> non_empty_partitions;
287 bool non_empty_partition = false;
288
289 for (int i = 0; i < partition_ids.size(); ++i) {
290 // Make sure the input is sorted by partition_ids;
291 if (i > 0) {
292 CHECK_LE(partition_ids(i - 1), partition_ids(i))
293 << "Partition ids should be sorted. Not sorted for " << i;
294 }
295 const int32 dimension = bucket_ids_and_dimensions(i, 1);
296
297 if (i == 0 || (partition_ids(i) != partition_ids(i - 1))) {
298 if (i != 0) {
299 // Not the first entry, so partition has changed.
300 if (non_empty_partition) {
301 // Saves the id of a previous partition in a list of non empty
302 // partitions, since it was non empty (had more than just a bias
303 // bucket -1.
304 non_empty_partitions.push_back(partition_boundaries.size() - 1);
305 }
306 // Add dummy dimension to signify the end for the previous dimension.
307 partition_boundaries.back().emplace_back(DUMMY_FEATURE_DIMENSION, i);
308 }
309 // Allocate for a new partition.
310 partition_boundaries.emplace_back();
311 // Save info about the first dimension for a new partition.
312 partition_boundaries.back().emplace_back(dimension, i);
313
314 // Each partition has dummy -1 bucket with all gradients and then info
315 // for all other dimensions -> if we have >1 elements for a partition,
316 // then it is not empty.
317 non_empty_partition = (i < partition_ids.size() - 1) &&
318 (partition_ids(i) == partition_ids(i + 1));
319 } else if (bucket_ids_and_dimensions(i, 1) !=
320 bucket_ids_and_dimensions(i - 1, 1)) {
321 // Dimension changed.
322 partition_boundaries.back().emplace_back(dimension, i);
323 }
324 }
325 if (tensor_elements > 0) {
326 if (non_empty_partition) {
327 non_empty_partitions.push_back(partition_boundaries.size() - 1);
328 }
329 // Add dummy dimension to signify the end for the previous dimension.
330 partition_boundaries.back().emplace_back(DUMMY_FEATURE_DIMENSION,
331 partition_ids.size());
332 }
333
334 int num_elements = non_empty_partitions.size();
335 Tensor* output_partition_ids_t = nullptr;
336 OP_REQUIRES_OK(context,
337 context->allocate_output("output_partition_ids",
338 TensorShape({num_elements}),
339 &output_partition_ids_t));
340
341 tensorflow::TTypes<int32>::Vec output_partition_ids =
342 output_partition_ids_t->vec<int32>();
343
344 Tensor* gains_t = nullptr;
345 OP_REQUIRES_OK(
346 context, context->allocate_output("gains", TensorShape({num_elements}),
347 &gains_t));
348
349 tensorflow::TTypes<float>::Vec gains = gains_t->vec<float>();
350
351 Tensor* output_splits_t = nullptr;
352 OP_REQUIRES_OK(context, context->allocate_output(
353 "split_infos", TensorShape({num_elements}),
354 &output_splits_t));
355 tensorflow::TTypes<string>::Vec output_splits =
356 output_splits_t->vec<string>();
357 // For each tree node that needs to be split.
358 for (int root_idx = 0; root_idx < num_elements; ++root_idx) {
359 const auto& dimension_boundaries =
360 partition_boundaries[non_empty_partitions[root_idx]];
361
362 float best_gain = std::numeric_limits<float>::lowest();
363 int32 best_dimension_idx = 0;
364 bool default_right = false;
365 int32 best_element_idx = 0;
366
367 NodeStats best_right_node_stats(0);
368 NodeStats best_left_node_stats(0);
369
370 // For each partition, the first bucket is dummy catch all.
371 int32 bias_start_index = dimension_boundaries[0].start_index;
372
373 OP_REQUIRES(
374 context,
375 bucket_ids_and_dimensions(bias_start_index, 0) == bias_feature_id_,
376 errors::InvalidArgument("Bias feature ID missing."));
377
378 // Dimension for bias feature is always 0
379 OP_REQUIRES(
380 context, bucket_ids_and_dimensions(bias_start_index, 1) == 0,
381 errors::InvalidArgument("Bias feature ID must be with dimension 0."));
382
383 // For each root, we do two passes over the quantized feature buckets
384 // accumulating gradients on one side and using the root aggregate
385 // gradients to get the gradients for the other side.
386 // Split gains are evaluated for each pass at every threshold and the best
387 // split is picked.
388 GradientStats root_gradient_stats(*gradients_t, *hessians_t,
389 bias_start_index);
390 root_gradient_stats *= normalizer_ratio;
391 NodeStats root_stats = ComputeNodeStats(root_gradient_stats);
392
393 // Iterate through dimensions.
394 for (int j = 0; j < dimension_boundaries.size() - 1; ++j) {
395 const DimensionBoundary& dimension_and_start = dimension_boundaries[j];
396 const int32 dimension_id = dimension_and_start.dimension_id;
397
398 int start_index = dimension_and_start.start_index;
399 // Even for the last dimension, we always have additional dummy
400 // dimension that we can use to find the end index.
401 const int end_index =
402 partition_boundaries[non_empty_partitions[root_idx]][j + 1]
403 .start_index;
404 CHECK(bucket_ids_and_dimensions(start_index, 1) ==
405 bucket_ids_and_dimensions(end_index - 1, 1))
406 << "For bucket " << bucket_ids_and_dimensions(start_index, 0)
407 << " the dimension was "
408 << bucket_ids_and_dimensions(start_index, 1) << " and for "
409 << bucket_ids_and_dimensions(end_index - 1, 0) << " "
410 << bucket_ids_and_dimensions(end_index - 1, 1);
411 if (bucket_ids_and_dimensions(start_index, 0) == bias_feature_id_) {
412 // 0-dimension case which has a first bucket for catch all feature.
413 CHECK(bucket_ids_and_dimensions(start_index, 1) == 0)
414 << "Dimension of bias feature should be 0";
415 ++start_index;
416 }
417
418 GradientStats present_gradient_stats;
419 for (int64 bucket_idx = start_index; bucket_idx < end_index;
420 ++bucket_idx) {
421 present_gradient_stats +=
422 GradientStats(*gradients_t, *hessians_t, bucket_idx);
423 }
424 present_gradient_stats *= normalizer_ratio;
425
426 GradientStats left_gradient_stats;
427 for (int64 element_idx = start_index; element_idx < end_index;
428 ++element_idx) {
429 // Check that bucket ids are sorted.
430 if (element_idx != start_index) {
431 CHECK(bucket_ids_and_dimensions(element_idx - 1, 0) <
432 bucket_ids_and_dimensions(element_idx, 0))
433 << "Bucket ids must be sorted."
434 << ", problem on " << element_idx << " and dimension is " << j;
435 }
436
437 GradientStats g(*gradients_t, *hessians_t, element_idx);
438 g *= normalizer_ratio;
439 left_gradient_stats += g;
440 // We have the sum of all present gradients. Use that to compute the
441 // backward pass gradients.
442 GradientStats right_gradient_stats =
443 present_gradient_stats - left_gradient_stats;
444 {
445 NodeStats left_stats_default_left =
446 ComputeNodeStats(root_gradient_stats - right_gradient_stats);
447 NodeStats right_stats_default_left =
448 ComputeNodeStats(right_gradient_stats);
449 if (left_stats_default_left.gain + right_stats_default_left.gain >
450 best_gain) {
451 best_gain =
452 left_stats_default_left.gain + right_stats_default_left.gain;
453 best_left_node_stats = left_stats_default_left;
454 best_right_node_stats = right_stats_default_left;
455 best_element_idx = element_idx;
456 default_right = false;
457 best_dimension_idx = dimension_id;
458 }
459 }
460 {
461 NodeStats left_stats_default_right =
462 ComputeNodeStats(left_gradient_stats);
463 NodeStats right_stats_default_right =
464 ComputeNodeStats(root_gradient_stats - left_gradient_stats);
465 if (left_stats_default_right.gain + right_stats_default_right.gain >
466 best_gain) {
467 best_gain = left_stats_default_right.gain +
468 right_stats_default_right.gain;
469 best_left_node_stats = left_stats_default_right;
470 best_right_node_stats = right_stats_default_right;
471 best_element_idx = element_idx;
472 default_right = true;
473 best_dimension_idx = dimension_id;
474 }
475 }
476 }
477 }
478
479 SplitInfo split_info;
480 boosted_trees::trees::DenseFloatBinarySplit* dense_split = nullptr;
481 if (default_right) {
482 dense_split = split_info.mutable_split_node()
483 ->mutable_sparse_float_binary_split_default_right()
484 ->mutable_split();
485 } else {
486 dense_split = split_info.mutable_split_node()
487 ->mutable_sparse_float_binary_split_default_left()
488 ->mutable_split();
489 }
490 dense_split->set_feature_column(feature_column_group_id_);
491 // Set the feature index for the best feature column.
492 const int64 best_dimension_id =
493 bucket_ids_and_dimensions(best_element_idx, 1);
494 const int32 best_bucket_id =
495 bucket_ids_and_dimensions(best_element_idx, 0);
496 dense_split->set_dimension_id(best_dimension_id);
497 dense_split->set_threshold(bucket_boundaries(best_bucket_id));
498
499 auto* left_child = split_info.mutable_left_child();
500 auto* right_child = split_info.mutable_right_child();
501 FillLeaf(class_id, best_left_node_stats, left_child);
502 FillLeaf(class_id, best_right_node_stats, right_child);
503 split_info.SerializeToString(&output_splits(root_idx));
504 gains(root_idx) =
505 best_gain - root_stats.gain - tree_complexity_regularization_;
506 output_partition_ids(root_idx) = partition_ids(bias_start_index);
507 }
508 }
509
510 private:
511 struct DimensionBoundary {
512 DimensionBoundary(const int32 dimension_id, const int32 start_index)
513 : dimension_id(dimension_id), start_index(start_index) {}
514
515 int32 dimension_id;
516 int32 start_index;
517 };
518
519 // For each partition, store start indices of feature column dimensions.
520 typedef std::vector<std::vector<DimensionBoundary>>
521 PartitionAndDimensionBoundaries;
522
523 int64 bias_feature_id_;
524};
525REGISTER_KERNEL_BUILDER(Name("BuildSparseInequalitySplits").Device(DEVICE_CPU),
526 BuildSparseInequalitySplitsOp);

528class BuildCategoricalEqualitySplitsOp : public BaseBuildSplitOp {
529 public:
530 explicit BuildCategoricalEqualitySplitsOp(OpKernelConstruction* const context)
531 : BaseBuildSplitOp(context) {
532 OP_REQUIRES_OK(context,
533 context->GetAttr("bias_feature_id", &bias_feature_id_));
534 }
535
536 void Compute(OpKernelContext* const context) override {
537 const Tensor* num_minibatches_t;
538 OP_REQUIRES_OK(context,
539 context->input("num_minibatches", &num_minibatches_t));
540 const int64 num_minibatches = num_minibatches_t->scalar<int64>()();
541 const float normalizer_ratio = (1.0f / num_minibatches);
542
543 const Tensor* partition_ids_t;
544 OP_REQUIRES_OK(context, context->input("partition_ids", &partition_ids_t));
545 const auto& partition_ids = partition_ids_t->vec<int32>();
546
547 const Tensor* feature_ids_t;
548 OP_REQUIRES_OK(context, context->input("feature_ids", &feature_ids_t));
549 const auto& feature_ids = feature_ids_t->matrix<int64>();
550
551 const Tensor* gradients_t;
552 OP_REQUIRES_OK(context, context->input("gradients", &gradients_t));
553
554 const Tensor* hessians_t;
555 OP_REQUIRES_OK(context, context->input("hessians", &hessians_t));
556
557 int class_id;
558 ReadClassId(context, &class_id);
559
560 // Find the number of unique partitions before we allocate the output.
561 std::vector<int32> partition_boundaries;
562 std::vector<int32> non_empty_partitions;
563 for (int i = 0; i < partition_ids.size() - 1; ++i) {
564 // Make sure the input is sorted by partition_ids;
565 CHECK_LE(partition_ids(i), partition_ids(i + 1));
566 if (i == 0 || partition_ids(i) != partition_ids(i - 1)) {
567 partition_boundaries.push_back(i);
568 // Some partitions might only have bias feature. We don't want to split
569 // those so check that the partition has at least 2 features.
570 if (partition_ids(i) == partition_ids(i + 1)) {
571 non_empty_partitions.push_back(partition_boundaries.size() - 1);
572 }
573 }
574 }
575 if (partition_ids.size() > 0) {
576 partition_boundaries.push_back(partition_ids.size());
577 }
578 int num_elements = non_empty_partitions.size();
579 Tensor* output_partition_ids_t = nullptr;
580 OP_REQUIRES_OK(context,
581 context->allocate_output("output_partition_ids",
582 TensorShape({num_elements}),
583 &output_partition_ids_t));
584
585 tensorflow::TTypes<int32>::Vec output_partition_ids =
586 output_partition_ids_t->vec<int32>();
587
588 Tensor* gains_t = nullptr;
589 OP_REQUIRES_OK(
590 context, context->allocate_output("gains", TensorShape({num_elements}),
591 &gains_t));
592
593 tensorflow::TTypes<float>::Vec gains = gains_t->vec<float>();
594
595 Tensor* output_splits_t = nullptr;
596 OP_REQUIRES_OK(context, context->allocate_output(
597 "split_infos", TensorShape({num_elements}),
598 &output_splits_t));
599 tensorflow::TTypes<string>::Vec output_splits =
600 output_splits_t->vec<string>();
601 for (int root_idx = 0; root_idx < num_elements; ++root_idx) {
602 float best_gain = std::numeric_limits<float>::lowest();
603 int start_index = partition_boundaries[non_empty_partitions[root_idx]];
604 int end_index = partition_boundaries[non_empty_partitions[root_idx] + 1];
605 // First feature ID in each partition should be the bias feature.
606 OP_REQUIRES(context, feature_ids(start_index, 0) == bias_feature_id_,
607 errors::InvalidArgument("Bias feature ID missing."));
608 GradientStats root_gradient_stats(*gradients_t, *hessians_t, start_index);
609 root_gradient_stats *= normalizer_ratio;
610 NodeStats root_stats = ComputeNodeStats(root_gradient_stats);
611 int32 best_feature_idx = 0;
612 NodeStats best_right_node_stats(0);
613 NodeStats best_left_node_stats(0);
614 for (int64 feature_idx = start_index + 1; feature_idx < end_index;
615 ++feature_idx) {
616 GradientStats left_gradient_stats(*gradients_t, *hessians_t,
617 feature_idx);
618 left_gradient_stats *= normalizer_ratio;
619 GradientStats right_gradient_stats =
620 root_gradient_stats - left_gradient_stats;
621 NodeStats left_stats = ComputeNodeStats(left_gradient_stats);
622 NodeStats right_stats = ComputeNodeStats(right_gradient_stats);
623 if (left_stats.gain + right_stats.gain > best_gain) {
624 best_gain = left_stats.gain + right_stats.gain;
625 best_left_node_stats = left_stats;
626 best_right_node_stats = right_stats;
627 best_feature_idx = feature_idx;
628 }
629 }
630 SplitInfo split_info;
631 auto* equality_split = split_info.mutable_split_node()
632 ->mutable_categorical_id_binary_split();
633 equality_split->set_feature_column(feature_column_group_id_);
634 equality_split->set_feature_id(feature_ids(best_feature_idx, 0));
635 auto* left_child = split_info.mutable_left_child();
636 auto* right_child = split_info.mutable_right_child();
637 FillLeaf(class_id, best_left_node_stats, left_child);
638 FillLeaf(class_id, best_right_node_stats, right_child);
639 split_info.SerializeToString(&output_splits(root_idx));
640 gains(root_idx) =
641 best_gain - root_stats.gain - tree_complexity_regularization_;
642 output_partition_ids(root_idx) = partition_ids(start_index);
643 }
644 }
645
646 private:
647 int64 bias_feature_id_;
648};
