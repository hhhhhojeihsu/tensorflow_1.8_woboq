15#include "tensorflow/contrib/boosted_trees/lib/utils/parallel_for.h"
16#include "tensorflow/core/lib/core/blocking_counter.h"

18namespace tensorflow {
19namespace boosted_trees {
20namespace utils {
21
22void ParallelFor(int64 batch_size, int64 desired_parallelism,
23 thread::ThreadPool* thread_pool,
24 std::function<void(int64, int64)> do_work) {
25 // Parallelize work over the batch.
26 if (desired_parallelism <= 0) {
27 do_work(0, batch_size);
28 return;
29 }
30 const int num_shards = std::max<int>(
31 1, std::min(static_cast<int64>(desired_parallelism), batch_size));
32 const int64 block_size = (batch_size + num_shards - 1) / num_shards;
33 CHECK_GT(block_size, 0);
34 const int num_shards_used = (batch_size + block_size - 1) / block_size;
35 BlockingCounter counter(num_shards_used - 1);
36 for (int64 start = block_size; start < batch_size; start += block_size) {
37 auto end = std::min(start + block_size, batch_size);
38 thread_pool->Schedule([&do_work, &counter, start, end]() {
39 do_work(start, end);
40 counter.DecrementCount();
41 });
42 }
43
44 // Execute first shard on main thread.
45 do_work(0, std::min(block_size, batch_size));
46 counter.Wait();
47}
