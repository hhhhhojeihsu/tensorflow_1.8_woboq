27// Represents sparse vector that have a value for some feature indices within
28// the feature column.
29// Allows subscript access [].
30template <class T>
31class SparseMultidimensionalValues {
32 public:
33 void Add(const int32 feature_idx, const T value) {
34 values_.emplace_back(feature_idx, value);
35 }
36
37 void Clear() { values_.clear(); }
38
39 void Reserve(const int32 size) { values_.reserve(size); }
40
41 OptionalValue<T> operator[](int feature_idx) const {
42 auto value_iter =
43 std::find_if(values_.begin(), values_.end(),
44 [&feature_idx](const std::pair<int32, T>& element) {
45 return element.first == feature_idx;
46 });
47
48 if (value_iter == values_.end()) {
49 return OptionalValue<T>();
50 }
51 return OptionalValue<T>(value_iter->second);
52 }
53
54 private:
55 std::vector<std::pair<int32, T>> values_;
56};
57
58// Represents storage for a sparse float feature column. Can store values either
59// for one dimensional or a multivalent (multidimensional) sparse column.
60// Allows subscript operator access [feature_id].
61template <class T>
62class SparseFloatFeatureColumn {
63 public:
64 void Reserve(const int32 size) {
65 if (!single_dimensional_) {
66 multidimensional_values.Reserve(size);
67 }
68 }
69
70 void SetDimension(const int32 dimension) {
71 single_dimensional_ = dimension <= 1;
72 }
73
74 void Add(const int32 feature_idx, const float value) {
75 if (single_dimensional_) {
76 DCHECK_EQ(0, feature_idx);
77 single_value_ = value;
78 } else {
79 multidimensional_values.Add(feature_idx, value);
80 }
81 initialized_ = true;
82 }
83
84 void Clear() {
85 single_dimensional_ = false;
86 initialized_ = false;
87 multidimensional_values.Clear();
88 }
89
90 OptionalValue<T> operator[](int feature_idx) const {
91 if (!initialized_) {
92 return OptionalValue<T>();
93 }
94 if (single_dimensional_) {
95 return OptionalValue<T>(single_value_);
96 } else {
97 return multidimensional_values[feature_idx];
98 }
99 }
100
101 private:
102 bool single_dimensional_;
103 bool initialized_;
104 T single_value_;
105 SparseMultidimensionalValues<T> multidimensional_values;
106};

108// Holds data for one example and enables lookup by feature column.
109struct Example {
110 // Default constructor creates an empty example.
111 Example() : example_idx(-1) {}
112
113 // Example index.
114 int64 example_idx;
115
116 // Dense and sparse float features indexed by feature column.
117 // TODO(salehay): figure out a design to support multivalent float features.
118 std::vector<float> dense_float_features;
119
120 // Sparse float features columns (can be either single or multivalent
121 // (multidimensional).
122 std::vector<SparseFloatFeatureColumn<float>> sparse_float_features;
123
124 // Sparse integer features indexed by feature column.
125 // Note that all integer features are assumed to be categorical, i.e. will
126 // never be compared by order. Also these features can be multivalent.
127 std::vector<std::unordered_set<int64>> sparse_int_features;
128};
