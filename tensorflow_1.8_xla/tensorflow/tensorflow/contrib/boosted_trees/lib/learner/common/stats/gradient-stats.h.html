28const double kEps = 1e-6;
29
30// A data structure for accumulating a Tensor value.
31struct TensorStat {
32 TensorStat() {}
33
34 explicit TensorStat(const float v) : t(DT_FLOAT, TensorShape({1})) {
35 t.flat<float>()(0) = v;
36 }
37
38 explicit TensorStat(const Tensor& rt) : t(tensor::DeepCopy(rt)) {}
39
40 TensorStat(const TensorStat& ts) : t(tensor::DeepCopy(ts.t)) {}
41
42 TensorStat& operator+=(const TensorStat& other) {
43 if (t.NumElements() == 0) {
44 t = tensor::DeepCopy(other.t);
45 return (*this);
46 }
47 CHECK(t.shape() == other.t.shape())
48 << "My shape = " << t.shape().DebugString()
49 << " Other shape = " << other.t.shape().DebugString();
50 auto me_flat = t.unaligned_flat<float>();
51 auto other_flat = other.t.unaligned_flat<float>();
52 for (int i = 0; i < me_flat.size(); i++) {
53 me_flat(i) += other_flat(i);
54 }
55 return (*this);
56 }
57
58 TensorStat& operator-=(const TensorStat& other) {
59 if (other.t.NumElements() == 0) {
60 return (*this);
61 }
62 CHECK(t.shape() == other.t.shape())
63 << "My shape = " << t.shape().DebugString()
64 << " Other shape = " << other.t.shape().DebugString();
65 auto me_flat = t.unaligned_flat<float>();
66 auto other_flat = other.t.unaligned_flat<float>();
67 for (int i = 0; i < me_flat.size(); i++) {
68 me_flat(i) -= other_flat(i);
69 }
70 return (*this);
71 }
72
73 TensorStat& operator*=(float value) {
74 auto me_flat = t.unaligned_flat<float>();
75 for (size_t i = 0; i < me_flat.size(); i++) {
76 me_flat(i) *= value;
77 }
78 return (*this);
79 }
80
81 bool IsZero() const {
82 auto me_flat = t.unaligned_flat<float>();
83 for (int i = 0; i < me_flat.size(); i++) {
84 if (me_flat(i) != 0.0f) {
85 return false;
86 }
87 }
88 return true;
89 }
90
91 // Checks if the L^2 magnitude of the tensor is less than eps.
92 bool IsAlmostZero(const float eps = kEps) const {
93 auto me_flat = t.unaligned_flat<float>();
94 double s = 0.0;
95 for (int i = 0; i < me_flat.size(); i++) {
96 s += me_flat(i) * me_flat(i);
97 if (s > eps * eps) {
98 return false;
99 }
100 }
101 return true;
102 }
103
104 float Magnitude() const {
105 auto me_flat = t.unaligned_flat<float>();
106 double s = 0.0;
107 for (int i = 0; i < me_flat.size(); i++) {
108 s += me_flat(i) * me_flat(i);
109 }
110 return sqrt(s);
111 }
112
113 string DebugString() const { return t.DebugString(); }
114
115 Tensor t;
116};

118// GradientStats holds first and second order gradient stats.
119struct GradientStats {
120 GradientStats() {}
121
122 // Legacy constructor for tests
123 GradientStats(float g, float h) : first(g), second(h) {}
124
125 GradientStats(const Tensor& g, const Tensor& h) : first(g), second(h) {}
126
127 GradientStats(const Tensor& g, const Tensor& h, int64 example_index)
128 : first(g.Slice(example_index, example_index + 1)),
129 second(h.Slice(example_index, example_index + 1)) {}
130
131 GradientStats& operator+=(const GradientStats& other) {
132 first += other.first;
133 second += other.second;
134 return (*this);
135 }
136
137 GradientStats& operator*=(float value) {
138 first *= value;
139 second *= value;
140 return (*this);
141 }
142
143 GradientStats& operator-=(const GradientStats& other) {
144 first -= other.first;
145 second -= other.second;
146 return (*this);
147 }
148
149 bool IsZero() const { return first.IsZero() && second.IsZero(); }
150
151 bool IsAlmostZero(const float eps = kEps) const {
152 return first.IsAlmostZero(eps) && second.IsAlmostZero(eps);
153 }
154
155 float Magnitude() const { return second.Magnitude(); }
156
157 string DebugString() const {
158 return "First = " + first.DebugString() +
159 " Second = " + second.DebugString();
160 }
161
162 TensorStat first;
163 TensorStat second;
164};
165
166struct GradientStatsAccumulator {
167 void operator()(const GradientStats& from, GradientStats* to) const {
168 (*to) += from;
169 }
170};
171
172inline GradientStats operator+(const GradientStats& a, const GradientStats& b) {
173 GradientStats ret(a);
174 ret += b;
175 return ret;
176}
177
178inline GradientStats operator-(const GradientStats& a, const GradientStats& b) {
179 GradientStats ret(a);
180 ret -= b;
181 return ret;
182}
183
184// Helper macro to check gradient stats approximate equality.
185#define EXPECT_GRADIENT_STATS_EQ(val1, val2) \
186 EXPECT_TRUE((val1 - val2).IsAlmostZero());
