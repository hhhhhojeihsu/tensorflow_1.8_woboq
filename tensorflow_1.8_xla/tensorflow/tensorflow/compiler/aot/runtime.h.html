16// This file contains utilities to make it easier to invoke functions generated
17// by tfcompile. Usage of these utilities is optional.
18
19#ifndef TENSORFLOW_COMPILER_AOT_RUNTIME_H_
20#define TENSORFLOW_COMPILER_AOT_RUNTIME_H_
21
22#include "tensorflow/core/platform/types.h"
23
24namespace tensorflow {
25namespace tfcompile {
26namespace runtime {
27
28// Align to 32-bytes, to mimic tensorflow::Allocator::kAllocatorAlignment.
29static constexpr size_t kAlign = 32;
30
31// aligned_buffer_bytes returns the sum of each size in `sizes`, skipping -1
32// values. There are `n` entries in `sizes`. Each buffer is aligned to kAlign
33// byte boundaries.
34size_t aligned_buffer_bytes(const intptr_t* sizes, size_t n);
35
36// MallocContiguousBuffers allocates buffers for use by the entry point
37// generated by tfcompile. `sizes` is an array of byte sizes for each buffer,
38// where -1 causes the buffer pointer to be nullptr. There are `n` entries in
39// `sizes`. If `annotate_initialized` is set, the allocated memory will be
40// annotated as having been initialized - this is useful when allocating
41// temporary buffers.
42//
43// A single contiguous block of memory is allocated, and portions of it are
44// parceled out into `bufs`, which must have space for `n` entries. Returns the
45// head of the allocated contiguous block, which should be passed to
46// FreeContiguous when the buffers are no longer in use.
47void* MallocContiguousBuffers(const intptr_t* sizes, size_t n, void** bufs,
48 bool annotate_initialized);
49
50// FreeContiguous frees the contiguous block of memory allocated by
51// MallocContiguousBuffers.
52void FreeContiguous(void* contiguous);
53
54} // namespace runtime
55} // namespace tfcompile
56} // namespace tensorflow
57
58#endif // TENSORFLOW_COMPILER_AOT_RUNTIME_H_
