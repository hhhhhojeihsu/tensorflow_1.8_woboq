189 foo::bar::MatMulComp matmul;

192 EXPECT_EQ(matmul.arg1_data(), matmul.args()[1]);

194 // Test using the argN() methods.
195 {
196 matmul.arg0(0, 0) = 1;
197 matmul.arg0(0, 1) = 2;

201 matmul.arg0(1, 2) = 6;
202
203 matmul.arg1(0, 0) = 7;
204 matmul.arg1(0, 1) = 8;
205 matmul.arg1(1, 0) = 9;
206 matmul.arg1(1, 1) = 10;

226 EXPECT_EQ(matmul.error_msg(), "");
227 const float results[4] = {5800, 6400, 13900, 15400};

229 EXPECT_EQ(matmul.result0(i / 2, i % 2), results[i]);

403 fn.arg1(0, 1) = 20;

406 EXPECT_TRUE(fn.Run());
407 EXPECT_EQ(fn.error_msg(), "");

409 3.29964742e+12};
410 EXPECT_NEAR(expected[0], fn.result0(0, 0), 1e4);
411 EXPECT_NEAR(expected[1], fn.result0(0, 1), 1e4);
412 EXPECT_NEAR(expected[2], fn.result0(1, 0), 1e4);
413 EXPECT_NEAR(expected[3], fn.result0(1, 1), 1e4);

442TEST(TFCompileTest, ProgramShape) {
443 using xla::ShapeUtil;
444 const xla::Shape f32_2x2 = ShapeUtil::MakeShape(xla::F32, {2, 2});
445
446 // add doesn't have the program shape defined.
447 AddComp add;
448 ASSERT_TRUE(add.ProgramShape() == nullptr);
449
450 // muladd has the program shape defined.
451 MatMulAndAddComp muladd;
452 const xla::ProgramShape* muladd_shape = muladd.ProgramShape();
453 ASSERT_TRUE(muladd_shape != nullptr);
454 ASSERT_EQ(muladd_shape->parameters_size(), 2);
455 EXPECT_TRUE(ShapeUtil::Compatible(muladd_shape->parameters(0), f32_2x2));
456 EXPECT_TRUE(ShapeUtil::Compatible(muladd_shape->parameters(1), f32_2x2));
457
458 const xla::Shape& muladd_result = muladd_shape->result();
459 ASSERT_EQ(muladd_result.element_type(), xla::TUPLE);
460 ASSERT_EQ(ShapeUtil::TupleElementCount(muladd_result), 2);
461 const xla::Shape& muladd_result0 =
462 ShapeUtil::GetTupleElementShape(muladd_result, 0);
463 EXPECT_TRUE(ShapeUtil::Compatible(muladd_result0, f32_2x2));
464 const xla::Shape& muladd_result1 =
465 ShapeUtil::GetTupleElementShape(muladd_result, 1);
466 EXPECT_TRUE(ShapeUtil::Compatible(muladd_result1, f32_2x2));
467}
