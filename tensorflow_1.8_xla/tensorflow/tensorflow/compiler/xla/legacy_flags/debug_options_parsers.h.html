29template <typename T>
30void parse_xla_backend_extra_options(T* extra_options_map,
31 string comma_separated_values) {
32 std::vector<string> extra_options_parts =
33 tensorflow::str_util::Split(comma_separated_values, ',');
34
35 // The flag contains a comma-separated list of options; some options
36 // have arguments following "=", some don't.
37 for (const auto& part : extra_options_parts) {
38 size_t eq_pos = part.find_first_of('=');
39 if (eq_pos == string::npos) {
40 (*extra_options_map)[part] = "";
41 } else {
42 string value = "";
43 if (eq_pos + 1 < part.size()) {
44 value = part.substr(eq_pos + 1);
45 }
46 (*extra_options_map)[part.substr(0, eq_pos)] = value;
47 }
48 }
49}
50
51// The --xla_reduce_precision option has the format "LOCATION=E,M:OPS;NAME",
52// where LOCATION is an HloReducePrecisionOptions::location, E and M are
53// integers for the exponent and matissa bit counts respectively, and OPS and
54// NAMES are comma-separated of the operation types and names to which to
55// attach the reduce-precision operations. The OPS values are matches to the
56// strings produced by HloOpcodeString, while the NAME values are arbitrary
57// strings subject to the requirements that they not contain any of "=,:;".
58// The NAME string (with its preceding semicolon) is optional.
59inline bool parse_xla_reduce_precision_option(
60 HloReducePrecisionOptions* options, string option_string) {
61 // Split off "LOCATION" from remainder of string.
62 std::vector<string> eq_split =
63 tensorflow::str_util::Split(option_string, '=');
64 if (eq_split.size() != 2) {
65 return false;
66 }
67 string& location = eq_split[0];
68 if (location == "OP_INPUTS") {
69 options->set_location(HloReducePrecisionOptions::OP_INPUTS);
70 } else if (location == "OP_OUTPUTS") {
71 options->set_location(HloReducePrecisionOptions::OP_OUTPUTS);
72 } else if (location == "UNFUSED_OP_OUTPUTS") {
73 options->set_location(HloReducePrecisionOptions::UNFUSED_OP_OUTPUTS);
74 } else if (location == "FUSION_INPUTS_BY_CONTENT") {
75 options->set_location(HloReducePrecisionOptions::FUSION_INPUTS_BY_CONTENT);
76 } else if (location == "FUSION_OUTPUTS_BY_CONTENT") {
77 options->set_location(HloReducePrecisionOptions::FUSION_OUTPUTS_BY_CONTENT);
78 } else {
79 return false;
80 }
81
82 // Split off "E,M" from remainder of string.
83 std::vector<string> colon_split =
84 tensorflow::str_util::Split(eq_split[1], ':');
85 if (colon_split.size() != 2) {
86 return false;
87 }
88
89 // Split E and M, and parse.
90 std::vector<int32> bitsizes;
91 if (!tensorflow::str_util::SplitAndParseAsInts(colon_split[0], ',',
92 &bitsizes) ||
93 bitsizes.size() != 2) {
94 return false;
95 }
96 options->set_exponent_bits(bitsizes[0]);
97 options->set_mantissa_bits(bitsizes[1]);
98
99 // Split off OPS comma-separated list from remainder of string, if the
100 // remainder exists.
101 std::vector<string> semicolon_split =
102 tensorflow::str_util::Split(colon_split[1], ';');
103 if (semicolon_split.size() > 2) {
104 return false;
105 }
106 // The opcode values are either 'all' (meaning all opcodes), or matches to
107 // the strings returned by HloOpcodeString. An empty string is also
108 // interpreted as 'all', for convenience. Note that 'all' may not be part
109 // of a comma-separated list; it must stand alone.
110 string& opcode_string = semicolon_split[0];
111 if (opcode_string == "" || opcode_string == "all") {
112 for (int i = 0; i < HloOpcodeCount(); i++) {
113 options->add_opcodes_to_suffix(i);
114 }
115 } else {
116 std::vector<string> opcodes =
117 tensorflow::str_util::Split(opcode_string, ',');
118 for (const string& opcode : opcodes) {
119 bool found = false;
120 for (int i = 0; i < HloOpcodeCount(); i++) {
121 if (opcode == HloOpcodeString(static_cast<HloOpcode>(i))) {
122 options->add_opcodes_to_suffix(i);
123 found = true;
124 break;
125 }
126 }
127 if (!found) {
128 return false;
129 }
130 }
131 }
132
133 // Process the NAMES string, if it exists.
134 if (semicolon_split.size() == 2) {
135 std::vector<string> opnames =
136 tensorflow::str_util::Split(semicolon_split[1], ',');
137 for (const string& opname : opnames) {
138 if (opname.length() > 0) {
139 options->add_opname_substrings_to_suffix(opname);
140 }
141 }
142 }
143
144 return true;
145}
