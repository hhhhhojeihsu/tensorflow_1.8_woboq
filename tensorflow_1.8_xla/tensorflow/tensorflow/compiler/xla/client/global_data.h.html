19#include "tensorflow/compiler/xla/service_interface.h"
20#include "tensorflow/compiler/xla/xla.pb.h"

23
24namespace xla {
25
26// A GlobalData object represents a globally-accessible allocation of
27// data in the associated XLA service.
28class GlobalData {
29 public:
30 // Gives ownership of the global data handle to this object.
31 GlobalData(ServiceInterface* parent, GlobalDataHandle handle);
32
33 // Unregisters the wrapped handle, which causes the service to
34 // deallocate the associated data.
35 ~GlobalData();
36
37 const GlobalDataHandle& handle() const { return handle_; }
38
39 private:
40 GlobalDataHandle handle_; // Handle being wrapped.
41 ServiceInterface* parent_; // Service used to unregister handle_.
42
43 TF_DISALLOW_COPY_AND_ASSIGN(GlobalData);
44};
45
46} // namespace xla
