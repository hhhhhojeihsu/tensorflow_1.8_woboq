35int64 GetUniqueId() {
36 static tensorflow::mutex mu(tensorflow::LINKER_INITIALIZED);
37 static int64 built_counter = 0;
38 tensorflow::mutex_lock loc(mu);
39 const int64 id = built_counter++;
40 return id;
41}

43// Returns true if an instruction with the given opcode can be the root of the
44// computation.
45bool CanBeRoot(HloOpcode opcode) {
46 switch (opcode) {
47 case HloOpcode::kSend:
48 case HloOpcode::kSendDone:
49 case HloOpcode::kOutfeed:
50 case HloOpcode::kTrace:
51 return false;
52 default:
53 return true;
54 }
55}

57StatusOr<std::vector<Shape>> GetOperandShapes(
58 tensorflow::gtl::ArraySlice<XlaOp> operands) {
59 std::vector<Shape> operand_shapes;
60 for (const XlaOp& operand : operands) {
61 TF_ASSIGN_OR_RETURN(const Shape& shape, operand.GetShape());
62 operand_shapes.push_back(shape);
63 }
64 return operand_shapes;
65}

89void XlaBuilder::NoteError(const Status& error) {
90 CHECK(!error.ok());
91 if (die_immediately_on_error_) {
92 LOG(FATAL) << "error building computation: " << error;
93 }
94
95 if (first_error_.ok()) {
96 first_error_ = error;
97 first_error_backtrace_.CreateCurrent(/*skip_count=*/1);
98 }
99}
100
101XlaOp XlaBuilder::NoteErrorOrReturn(
102 const std::function<StatusOr<XlaOp>()>& op_creator) {
103 if (!first_error_.ok()) {
104 return {};
105 }
106 auto op = op_creator();
107 if (!op.ok()) {
108 NoteError(op.status());
109 return {};
110 }
111 return op.ConsumeValueOrDie();
112}

114StatusOr<ProgramShape> XlaBuilder::GetProgramShape(int64* root_id) {
115 TF_RETURN_IF_ERROR(first_error_);
116
117 TF_RET_CHECK(root_id != nullptr);
118 ProgramShape program_shape;
119
120 // Not all instructions can be roots. Walk backwards from the last added
121 // instruction until a valid root is found.
122 int64 index = instructions_.size() - 1;
123 for (; index >= 0; index--) {
124 TF_ASSIGN_OR_RETURN(HloOpcode opcode,
125 StringToHloOpcode(instructions_[index].opcode()));
126 if (CanBeRoot(opcode)) {
127 break;
128 }
129 }
130 if (index < 0) {
131 return FailedPrecondition("no root instruction was found");
132 }
133 *root_id = instructions_[index].id();
134 *program_shape.mutable_result() = instructions_[index].shape();
135
136 // Check that the parameter numbers are continuous from 0, and add parameter
137 // shapes and names to the program shape.
138 const int64 param_count = parameter_numbers_.size();
139 for (int64 i = 0; i < param_count; i++) {
140 program_shape.add_parameters();
141 program_shape.add_parameter_names();
142 }
143 for (const HloInstructionProto& instr : instructions_) {
144 // Parameter number uniqueness is guaranteed in XlaBuilder::Parameter(). So
145 // to verify continuity, we just need to verify that every parameter is in
146 // the right range.
147 if (instr.opcode() == HloOpcodeString(HloOpcode::kParameter)) {
148 const int64 index = instr.parameter_number();
149 TF_RET_CHECK(index >= 0 && index < param_count)
150 << "invalid parameter number: " << index;
151 *program_shape.mutable_parameters(index) = instr.shape();
152 *program_shape.mutable_parameter_names(index) = instr.name();
153 }
154 }
155 return program_shape;
156}
157
158StatusOr<ProgramShape> XlaBuilder::GetProgramShape() {
159 int64 root_id;
160 return GetProgramShape(&root_id);
161}

163XlaComputation XlaBuilder::BuildAndNoteError() {
164 DCHECK(parent_builder_ != nullptr);
165 auto build_status = Build();
166 if (!build_status.ok()) {
167 parent_builder_->NoteError(
168 AddStatus(build_status.status(),
169 tensorflow::strings::StrCat("error from: ", name_)));
170 return {};
171 }
172 return build_status.ConsumeValueOrDie();
173}
174
175StatusOr<XlaComputation> XlaBuilder::Build() {
176 if (!first_error_.ok()) {
177 string backtrace;
178 first_error_backtrace_.Dump(tensorflow::DebugWriteToString, &backtrace);
179 return AppendStatus(first_error_, backtrace);
180 }
181
182 HloComputationProto entry;
183
184 {
185 int64 root_id;
186 ProgramShape program_shape;
187 TF_ASSIGN_OR_RETURN(program_shape, GetProgramShape(&root_id));
188 entry.mutable_program_shape()->Swap(&program_shape);
189 entry.set_root_id(root_id);
190 }
191
192 for (auto& instruction : instructions_) {
193 entry.add_instructions()->Swap(&instruction);
194 }
195
196 entry.set_id(unique_id_);
197 entry.set_name(StrCat(name_, entry.id())); // Ensure that the name is unique.
198 XlaComputation computation(entry.id());
199 HloModuleProto* module = computation.mutable_proto();
200 module->set_name(entry.name());
201 module->set_id(entry.id());
202 module->set_entry_computation_name(entry.name());
203 module->set_entry_computation_id(entry.id());
204 *module->mutable_program_shape() = entry.program_shape();
205 for (auto& e : embedded_) {
206 module->add_computations()->Swap(&e.second);
207 }
208 module->add_computations()->Swap(&entry);
209
210 // Clear data held by this builder.
211 this->instructions_.clear();
212 this->embedded_.clear();
213 this->parameter_numbers_.clear();
214
215 return std::move(computation);
216}

218StatusOr<XlaOp> XlaBuilder::InDimBroadcast(
219 const Shape& shape, const XlaOp& operand,
220 tensorflow::gtl::ArraySlice<int64> broadcast_dimensions) {
221 TF_RETURN_IF_ERROR(first_error_);
222
223 HloInstructionProto instr;
224 *instr.mutable_shape() = shape;
225 for (int64 dim : broadcast_dimensions) {
226 instr.add_dimensions(dim);
227 }
228 return AddInstruction(std::move(instr), HloOpcode::kBroadcast, {operand});
229}
230
231StatusOr<XlaOp> XlaBuilder::AddBroadcastSequence(const Shape& output_shape,
232 const XlaOp& operand) {
233 TF_RETURN_IF_ERROR(first_error_);
234
235 TF_ASSIGN_OR_RETURN(const Shape& operand_shape, operand.GetShape());
236
237 CHECK(ShapeUtil::IsScalar(operand_shape) ||
238 ShapeUtil::Rank(operand_shape) == ShapeUtil::Rank(output_shape));
239 Shape broadcast_shape =
240 ShapeUtil::ChangeElementType(output_shape, operand_shape.element_type());
241
242 // Do explicit broadcast for scalar.
243 if (ShapeUtil::IsScalar(operand_shape)) {
244 return InDimBroadcast(broadcast_shape, operand, {});
245 }
246
247 // Do explicit broadcast for degenerate broadcast.
248 std::vector<int64> broadcast_dimensions;
249 std::vector<int64> reshaped_dimensions;
250 for (int i = 0; i < ShapeUtil::Rank(operand_shape); i++) {
251 if (operand_shape.dimensions(i) == output_shape.dimensions(i)) {
252 broadcast_dimensions.push_back(i);
253 reshaped_dimensions.push_back(operand_shape.dimensions(i));
254 } else {
255 TF_RET_CHECK(operand_shape.dimensions(i) == 1)
256 << "An explicit broadcast sequence requires the broadcasted "
257 "dimensions to be trivial; operand shape: "
258 << operand_shape << "; output_shape: " << output_shape;
259 }
260 }
261 // Eliminate the size one dimensions.
262 TF_ASSIGN_OR_RETURN(XlaOp reshaped_operand,
263 Reshape(ShapeUtil::MakeShape(operand_shape.element_type(),
264 reshaped_dimensions),
265 operand));
266 // Broadcast 'reshape' up to the larger size.
267 return InDimBroadcast(broadcast_shape, reshaped_operand,
268 broadcast_dimensions);
269}

281XlaOp XlaBuilder::BinaryOp(
282 HloOpcode binop, const XlaOp& lhs, const XlaOp& rhs,
283 tensorflow::gtl::ArraySlice<int64> broadcast_dimensions) {
284 return NoteErrorOrReturn([&]() -> StatusOr<XlaOp> {
285 HloInstructionProto instr;
286 TF_ASSIGN_OR_RETURN(const Shape& lhs_shape, lhs.GetShape());
287 TF_ASSIGN_OR_RETURN(const Shape& rhs_shape, rhs.GetShape());
288 TF_ASSIGN_OR_RETURN(*instr.mutable_shape(),
289 ShapeInference::InferBinaryOpShape(
290 binop, lhs_shape, rhs_shape, broadcast_dimensions));
291
292 const int64 lhs_rank = ShapeUtil::Rank(lhs_shape);
293 const int64 rhs_rank = ShapeUtil::Rank(rhs_shape);
294
295 XlaOp updated_lhs = lhs;
296 XlaOp updated_rhs = rhs;
297
298 if (!broadcast_dimensions.empty() && lhs_rank != rhs_rank) {
299 const bool should_broadcast_lhs = lhs_rank < rhs_rank;
300 XlaOp from = should_broadcast_lhs ? lhs : rhs;
301 const Shape& from_shape = should_broadcast_lhs ? lhs_shape : rhs_shape;
302
303 std::vector<int64> to_size;
304 for (int64 size : instr.shape().dimensions()) {
305 to_size.push_back(size);
306 }
307 for (int64 from_dim = 0; from_dim < ShapeUtil::Rank(from_shape);
308 from_dim++) {
309 int64 to_dim = broadcast_dimensions[from_dim];
310 to_size[to_dim] = from_shape.dimensions(from_dim);
311 }
312
313 const Shape& broadcasted_shape =
314 ShapeUtil::MakeShape(from_shape.element_type(), to_size);
315 TF_ASSIGN_OR_RETURN(
316 XlaOp broadcasted_operand,
317 InDimBroadcast(broadcasted_shape, from, broadcast_dimensions));
318
319 updated_lhs = should_broadcast_lhs ? broadcasted_operand : lhs;
320 updated_rhs = !should_broadcast_lhs ? broadcasted_operand : rhs;
321 }
322
323 TF_ASSIGN_OR_RETURN(Shape updated_lhs_shape, updated_lhs.GetShape());
324 if (!ShapeUtil::SameDimensions(instr.shape(), updated_lhs_shape)) {
325 TF_ASSIGN_OR_RETURN(updated_lhs,
326 AddBroadcastSequence(instr.shape(), updated_lhs));
327 }
328 TF_ASSIGN_OR_RETURN(Shape updated_rhs_shape, updated_rhs.GetShape());
329 if (!ShapeUtil::SameDimensions(instr.shape(), updated_rhs_shape)) {
330 TF_ASSIGN_OR_RETURN(updated_rhs,
331 AddBroadcastSequence(instr.shape(), updated_rhs));
332 }
333
334 return AddInstruction(std::move(instr), binop, {updated_lhs, updated_rhs});
335 });
336}

338XlaOp XlaBuilder::TernaryOp(HloOpcode triop, const XlaOp& lhs, const XlaOp& rhs,
339 const XlaOp& ehs) {
340 return NoteErrorOrReturn([&]() -> StatusOr<XlaOp> {
341 HloInstructionProto instr;
342 TF_ASSIGN_OR_RETURN(const Shape& lhs_shape, lhs.GetShape());
343 TF_ASSIGN_OR_RETURN(const Shape& rhs_shape, rhs.GetShape());
344 TF_ASSIGN_OR_RETURN(const Shape& ehs_shape, ehs.GetShape());
345 TF_ASSIGN_OR_RETURN(*instr.mutable_shape(),
346 ShapeInference::InferTernaryOpShape(
347 triop, lhs_shape, rhs_shape, ehs_shape));
348 XlaOp updated_lhs = lhs;
349 XlaOp updated_rhs = rhs;
350 XlaOp updated_ehs = ehs;
351 if (!ShapeUtil::IsTuple(instr.shape())) {
352 if (!ShapeUtil::IsTuple(lhs_shape) &&
353 !ShapeUtil::SameDimensions(instr.shape(), lhs_shape)) {
354 // lhs is being implicitly broadcasted. Change to explicit.
355 TF_ASSIGN_OR_RETURN(updated_lhs,
356 AddBroadcastSequence(instr.shape(), lhs));
357 }
358 if (!ShapeUtil::IsTuple(rhs_shape) &&
359 !ShapeUtil::SameDimensions(instr.shape(), rhs_shape)) {
360 // rhs is being implicitly broadcasted. Change to explicit.
361 TF_ASSIGN_OR_RETURN(updated_rhs,
362 AddBroadcastSequence(instr.shape(), rhs));
363 }
364 if (!ShapeUtil::IsTuple(ehs_shape) &&
365 !ShapeUtil::SameDimensions(instr.shape(), ehs_shape)) {
366 // ehs is being implicitly broadcasted. Change to explicit.
367 TF_ASSIGN_OR_RETURN(updated_ehs,
368 AddBroadcastSequence(instr.shape(), ehs));
369 }
370 }
371 return AddInstruction(std::move(instr), triop,
372 {updated_lhs, updated_rhs, updated_ehs});
373 });
374}

395XlaOp XlaBuilder::Call(const XlaComputation& computation,
396 tensorflow::gtl::ArraySlice<XlaOp> operands) {
397 return NoteErrorOrReturn([&]() -> StatusOr<XlaOp> {
398 HloInstructionProto instr;
399 std::vector<const Shape*> operand_shape_ptrs;
400 TF_ASSIGN_OR_RETURN(const auto& operand_shapes, GetOperandShapes(operands));
401 c_transform(operand_shapes, std::back_inserter(operand_shape_ptrs),
402 [](const Shape& shape) { return &shape; });
403 TF_ASSIGN_OR_RETURN(const ProgramShape& called_program_shape,
404 computation.GetProgramShape());
405 TF_ASSIGN_OR_RETURN(
406 *instr.mutable_shape(),
407 ShapeInference::InferCallShape(operand_shape_ptrs,
408 /*to_apply=*/called_program_shape));
409
410 AddCalledComputation(computation, &instr);
411
412 return AddInstruction(std::move(instr), HloOpcode::kCall, operands);
413 });
414}

416XlaOp XlaBuilder::Parameter(int64 parameter_number, const Shape& shape,
417 const string& name) {
418 return NoteErrorOrReturn([&]() -> StatusOr<XlaOp> {
419 HloInstructionProto instr;
420 if (parameter_numbers_.find(parameter_number) != parameter_numbers_.end()) {
421 return InvalidArgument("parameter %lld already registered",
422 parameter_number);
423 }
424 parameter_numbers_.insert(parameter_number);
425 instr.set_parameter_number(parameter_number);
426 instr.set_name(name);
427 *instr.mutable_shape() = shape;
428 return AddInstruction(std::move(instr), HloOpcode::kParameter);
429 });

432XlaOp XlaBuilder::Broadcast(
433 const XlaOp& operand, tensorflow::gtl::ArraySlice<int64> broadcast_sizes) {
434 return NoteErrorOrReturn([&]() -> StatusOr<XlaOp> {
435 TF_ASSIGN_OR_RETURN(const Shape& operand_shape, operand.GetShape());
436 TF_ASSIGN_OR_RETURN(
437 const Shape& shape,
438 ShapeInference::InferBroadcastShape(operand_shape, broadcast_sizes));
439
440 // The client-level broadcast op just appends dimensions on the left (adds
441 // lowest numbered dimensions). The HLO broadcast instruction is more
442 // flexible and can add new dimensions anywhere. The instruction's
443 // dimensions field maps operand dimensions to dimensions in the broadcast
444 // output, so to append dimensions on the left the instruction's dimensions
445 // should just be the n highest dimension numbers of the output shape where
446 // n is the number of input dimensions.
447 const int64 operand_rank = ShapeUtil::Rank(operand_shape);
448 std::vector<int64> dimensions(operand_rank);
449 for (int i = 0; i < operand_rank; ++i) {
450 dimensions[i] = i + ShapeUtil::Rank(shape) - operand_rank;
451 }
452 return InDimBroadcast(shape, operand, dimensions);
453 });
454}

594XlaOp XlaBuilder::Collapse(const XlaOp& operand,
595 tensorflow::gtl::ArraySlice<int64> dimensions) {
596 return NoteErrorOrReturn([&]() -> StatusOr<XlaOp> {
597 if (dimensions.size() <= 1) {
598 // Not collapsing anything, trivially we can return the operand versus
599 // enqueueing a trivial reshape.
600 return operand;
601 }
602
603 // Out-of-order collapse is not supported.
604 // Checks that the collapsed dimensions are in order and consecutive.
605 for (tensorflow::gtl::ArraySlice<int64>::size_type i = 1;
606 i < dimensions.size(); ++i) {
607 if (dimensions[i] - 1 != dimensions[i - 1]) {
608 return InvalidArgument(
609 "Collapsed dimensions are not in consecutive order.");
610 }
611 }
612
613 // Create a new sizes vector from the old shape, replacing the collapsed
614 // dimensions by the product of their sizes.
615 TF_ASSIGN_OR_RETURN(const Shape& original_shape, GetShape(operand));
616
617 VLOG(3) << "original shape: " << ShapeUtil::HumanString(original_shape);
618 VLOG(3) << "dims to collapse: "
619 << tensorflow::str_util::Join(dimensions, ",");
620
621 std::vector<int64> new_sizes;
622 for (int i = 0; i < ShapeUtil::Rank(original_shape); ++i) {
623 if (i <= dimensions.front() || i > dimensions.back()) {
624 new_sizes.push_back(original_shape.dimensions(i));
625 } else {
626 new_sizes.back() *= original_shape.dimensions(i);
627 }
628 }
629
630 VLOG(3) << "new sizes: [" << tensorflow::str_util::Join(new_sizes, ",")
631 << "]";
632
633 return Reshape(operand, new_sizes);
634 });
635}

1042XlaOp XlaBuilder::RngOp(RandomDistribution distribution,
1043 tensorflow::gtl::ArraySlice<XlaOp> parameters,
1044 const Shape& shape) {
1045 return NoteErrorOrReturn([&]() -> StatusOr<XlaOp> {
1046 HloInstructionProto instr;
1047
1048 // Check the number of parameters per RNG distribution.
1049 switch (distribution) {
1050 case RandomDistribution::RNG_NORMAL:
1051 case RandomDistribution::RNG_UNIFORM:
1052 if (parameters.size() != 2) {
1053 return InvalidArgument(
1054 "RNG distribution (%s) expects 2 parameters, but got %ld",
1055 RandomDistribution_Name(distribution).c_str(), parameters.size());
1056 }
1057 break;
1058 default:
1059 LOG(FATAL) << "unhandled distribution " << distribution;
1060 }
1061
1062 TF_RETURN_IF_ERROR(ShapeUtil::ValidateShapeWithOptionalLayout(shape));
1063 *instr.mutable_shape() = shape;
1064
1065 instr.set_distribution(distribution);
1066
1067 return AddInstruction(std::move(instr), HloOpcode::kRng, parameters);
1068 });
1069}

1083 return NoteErrorOrReturn([&]() -> StatusOr<XlaOp> {

1212void XlaBuilder::Send(const XlaOp& operand, const ChannelHandle& handle) {
1213 NoteErrorOrReturn([&]() -> StatusOr<XlaOp> {
1214 HloInstructionProto instr;
1215
1216 // Send instruction produces a tuple of {aliased operand, U32 context}.
1217 TF_ASSIGN_OR_RETURN(const Shape& shape, GetShape(operand));
1218 *instr.mutable_shape() =
1219 ShapeUtil::MakeTupleShape({shape, ShapeUtil::MakeShape(U32, {})});
1220 instr.set_channel_id(handle.handle());
1221 TF_ASSIGN_OR_RETURN(
1222 XlaOp send,
1223 AddInstruction(std::move(instr), HloOpcode::kSend, {operand}));
1224
1225 HloInstructionProto send_done_instr;
1226 *send_done_instr.mutable_shape() = ShapeUtil::MakeNil();
1227 send_done_instr.set_channel_id(handle.handle());
1228 return AddInstruction(std::move(send_done_instr), HloOpcode::kSendDone,
1229 {send});
1230 });
1231}

1294/* static */ Status XlaBuilder::Validate(
1295 const ConvolutionDimensionNumbers& dnum) {
1296 if (dnum.input_spatial_dimensions_size() < 2) {
1297 return FailedPrecondition("input spacial dimension < 2: %d",
1298 dnum.input_spatial_dimensions_size());
1299 }
1300 if (dnum.kernel_spatial_dimensions_size() < 2) {
1301 return FailedPrecondition("kernel spacial dimension < 2: %d",
1302 dnum.kernel_spatial_dimensions_size());
1303 }
1304 if (dnum.output_spatial_dimensions_size() < 2) {
1305 return FailedPrecondition("output spacial dimension < 2: %d",
1306 dnum.output_spatial_dimensions_size());
1307 }
1308
1309 if (std::set<int64>(
1310 {dnum.input_batch_dimension(), dnum.input_feature_dimension(),
1311 dnum.input_spatial_dimensions(0), dnum.input_spatial_dimensions(1)})
1312 .size() != 4) {
1313 return FailedPrecondition(
1314 "dimension numbers for the input are not unique: (%lld, %lld, %lld, "
1315 "%lld)",
1316 dnum.input_batch_dimension(), dnum.input_feature_dimension(),
1317 dnum.input_spatial_dimensions(0), dnum.input_spatial_dimensions(1));
1318 }
1319 if (std::set<int64>({dnum.kernel_output_feature_dimension(),
1320 dnum.kernel_input_feature_dimension(),
1321 dnum.kernel_spatial_dimensions(0),
1322 dnum.kernel_spatial_dimensions(1)})
1323 .size() != 4) {
1324 return FailedPrecondition(
1325 "dimension numbers for the weight are not unique: (%lld, %lld, %lld, "
1326 "%lld)",
1327 dnum.kernel_output_feature_dimension(),
1328 dnum.kernel_input_feature_dimension(),
1329 dnum.kernel_spatial_dimensions(0), dnum.kernel_spatial_dimensions(1));
1330 }
1331 if (std::set<int64>({dnum.output_batch_dimension(),
1332 dnum.output_feature_dimension(),
1333 dnum.output_spatial_dimensions(0),
1334 dnum.output_spatial_dimensions(1)})
1335 .size() != 4) {
1336 return FailedPrecondition(
1337 "dimension numbers for the output are not unique: (%lld, %lld, %lld, "
1338 "%lld)",
1339 dnum.output_batch_dimension(), dnum.output_feature_dimension(),
1340 dnum.output_spatial_dimensions(0), dnum.output_spatial_dimensions(1));
1341 }
1342 return Status::OK();
1343}

1345StatusOr<XlaOp> XlaBuilder::AddInstruction(
1346 HloInstructionProto&& instr, HloOpcode opcode,
1347 tensorflow::gtl::ArraySlice<XlaOp> operands) {
1348 TF_RETURN_IF_ERROR(first_error_);
1349
1350 const int64 handle = instructions_.size();
1351 instr.set_id(handle);
1352 instr.set_opcode(HloOpcodeString(opcode));
1353 if (instr.name().empty()) {
1354 instr.set_name(StrCat(instr.opcode(), ".", unique_id_, ".", handle));
1355 } else {
1356 // Append the handle to make sure the name is unique.
1357 instr.set_name(StrCat(instr.name(), ".", unique_id_, ".", handle));
1358 }
1359 for (const auto& operand : operands) {
1360 if (operand.builder_ == nullptr) {
1361 return InvalidArgument("invalid XlaOp with handle %lld",
1362 operand.handle());
1363 }
1364 if (operand.builder_ != this) {
1365 return InvalidArgument("Do not add XlaOp from builder %s to builder %s",
1366 operand.builder_->name().c_str(),
1367 this->name().c_str());
1368 }
1369 instr.add_operand_ids(operand.handle());
1370 }
1371
1372 *instr.mutable_metadata() = metadata_;
1373 if (sharding_) {
1374 *instr.mutable_sharding() = *sharding_;
1375 }
1376
1377 instructions_.push_back(instr);
1378
1379 XlaOp op(handle, this);
1380 return op;
1381}

1384 HloInstructionProto* instr) {
1385 instr->add_called_computation_ids(computation.proto().entry_computation_id());
1386 for (const HloComputationProto& e : computation.proto().computations()) {

1391StatusOr<const HloInstructionProto*> XlaBuilder::LookUpInstruction(
1392 const XlaOp& op) const {
1393 TF_RETURN_IF_ERROR(first_error_);
1394
1395 if (op.builder_ != this) {
1396 return InvalidArgument("invalid XlaOp with handle %lld", op.handle());
1397 }
1398
1399 TF_RET_CHECK(op.builder_ == this);
1400 if (op.handle() >= instructions_.size() || op.handle() < 0) {
1401 return InvalidArgument("no XlaOp value %lld", op.handle());
1402 }
1403 return &instructions_[op.handle()];
1404}
