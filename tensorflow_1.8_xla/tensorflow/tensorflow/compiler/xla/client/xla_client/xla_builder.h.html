48// This represents an instruction that has been enqueued using the XlaBuilder.
49// This is used to pass to subsequent computations that depends upon the
50// instruction as an operand.
51//
52// TODO(b/74197823): Replace xla::ComputationDataHandle with this one.
53class XlaOp {
54 public:
55 XlaOp() : handle_(0), builder_(nullptr) {}
56
57 StatusOr<Shape> GetShape() const;
58
59 private:
60 XlaOp(int64 handle, XlaBuilder* builder)
61 : handle_(handle), builder_(builder) {}
62
63 int64 handle() const { return handle_; }
64 friend class XlaBuilder;
65
66 int64 handle_;
67 XlaBuilder* builder_; // Not owned.
68};

177 // Adds dimensions to an array by duplicating the data in the array.
178 //
179 // The new dimensions are inserted on the left, i.e. if
180 // broadcast_sizes has values {a0, ..., aN} and the operand shape
181 // has dimensions {b0, ..., bM} then the shape of the output has
182 // dimensions {a0, ..., aN, b0, ..., bM}.
183 //
184 // The new dimensions index into copies of the operand, i.e.
185 //
186 // output[i0, ..., iN, j0, ..., jM] = operand[j0, ..., jM]
187 XlaOp Broadcast(const XlaOp& operand,
188 tensorflow::gtl::ArraySlice<int64> broadcast_sizes);

190 // Enqueues a pad operation onto the computation that pads the given value on

194 const PaddingConfig& padding_config);

196 // Enqueues an operation onto the computation that flattens the operand based
197 // on the dimension order (major/slowest-varying to minor/fastest-varying)
198 // given, followed by reshaping it into the shape with the given dimension
199 // sizes (also major to minor). Conceptually, this is a limited form of
200 // "shape casting".
201 XlaOp Reshape(const XlaOp& operand,
202 tensorflow::gtl::ArraySlice<int64> dimensions,
203 tensorflow::gtl::ArraySlice<int64> new_sizes);
204
205 // Enqueues an operation onto the computation that collapses the operand, from
206 // first to last dimension (C order), then reshapes it to the given dimension
207 // sizes. Conceptually, this is a limited form of "shape casting".
208 XlaOp Reshape(const XlaOp& operand,
209 tensorflow::gtl::ArraySlice<int64> new_sizes);
210
211 // Wrapper for Reshape.
212 // Enqueues an operation to collapse the provided dimensions; e.g. an
213 // operand with dimensions {x=256, y=2, z=2, p=32} can be collapsed to
214 // {x=1024, y=32} by collapsing dims {0, 1, 2}. Collapsing dimensions must
215 // be a consecutive, in-order subsequence of the operand dimensions.
216 //
217 // Note that collapsing a single dimension does nothing:
218 //
219 // {256} collapsing {0} => {256}
220 // {1} collapsing {0} => {1}
221 //
222 // Collapsing multiple dimensions produces a single result dimension:
223 //
224 // {256, 2} collapsing {0,1} => {512}
225 // {256, 2, 3} collapsing {0,1} => {512, 3}
226 //
227 // This could potentially cause data to be moved -- it provides a more
228 // structured form of reshaping than an arbitrary Reshape operation.
229 XlaOp Collapse(const XlaOp& operand,
230 tensorflow::gtl::ArraySlice<int64> dimensions);
231
232 // Enqueues a slice operation onto the computation that slices the operand
233 // from the start indices to the limit indices; e.g.
234 //
235 // x
236 // [ 0 1 2 3 ]
237 // y [ 4 5 6 7 ] => slice(start={1, 1}, limit={2, 3}) => [ 5 6 ]
238 // [ 8 9 a b ]
239 //
240 // Note that "limit" means up-to-but-not-including; i.e. [start, limit) in 1D
241 // range notation.
242 // The strides parameter determines the stride over the slice
243 XlaOp Slice(const XlaOp& operand,
244 tensorflow::gtl::ArraySlice<int64> start_indices,
245 tensorflow::gtl::ArraySlice<int64> limit_indices,
246 tensorflow::gtl::ArraySlice<int64> strides);
247
248 // Enqueues a slice operation in a given dimension, taking all other
249 // dimensions as they are; e.g. if dimno is 1 from start_index 2 to
250 // limit_index 4 by 1, and the shape is f32[7,8,9], this call is short-hand
251 // for:
252 //
253 // array[:, 2:4:1, :]
254 XlaOp SliceInDim(const XlaOp& operand, int64 start_index, int64 limit_index,
255 int64 stride, int64 dimno);
256
257 // Enqueues a slice operation onto the computation that slices the 'operand'
258 // from dynamic start indices which are passed in 'start_indices'.
259 // The size of the slice in each dimension is passed in 'slice_sizes',
260 // which specify the end point of exclusive slice intervals in each
261 // dimension [start, start + size).
262 // The shape of 'start_indices' must be rank == 1, with dimension size
263 // equal to the rank of the 'operand'.
264 // Slice index calculations are computed modulo input dimension sizes to
265 // prevent dynamic start indices from generating out-of-bound array accesses.
266 XlaOp DynamicSlice(const XlaOp& operand, const XlaOp& start_indices,
267 tensorflow::gtl::ArraySlice<int64> slice_sizes);
268
269 // Enqueues a dynamic update slice operation onto the computation, which
270 // updates a slice of 'operand' with 'update' at dynamic 'start_indices'.
271 // The shape of 'update' determines the shape of the slice of 'operand'
272 // which is updated.
273 // The indices specified in 'start_indices' specify the offset of the slice
274 // of 'operand' which is updated.
275 //
276 // update = {10, 11} // calculated at runtime.
277 // [1 2 3] start = {1, 1} // calculated at runtime. [1 2 3 ]
278 // [4 5 6] => DynamicUpdateslice(data, update, start) => [4 10 11]
279 // [7 8 9] [7 8 9 ]
280 //
281 // The shape of 'start_indices' must be rank == 1, with dimension size
282 // equal to the rank of the 'operand'.
283 // Slice index calculations are computed modulo update dimension sizes to
284 // prevent dynamic start indices from generating out-of-bound array accesses.
285 XlaOp DynamicUpdateSlice(const XlaOp& operand, const XlaOp& update,
286 const XlaOp& start_indices);

737 // The operand must represent a constant value, which in this case

739 // computation that is being built other then the ones specified on the
740 // parameter list. The parameters in the list will be indexed by their
741 // parameter id property so the number of parameters specified should be at
742 // least as many as the largest used parameter index.

744 // `IsConstant` can be used to test whether a computation is a compile-time

757 // will be stored using that layout.
