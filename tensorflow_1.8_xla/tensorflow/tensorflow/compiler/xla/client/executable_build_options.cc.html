53string ExecutableBuildOptions::ToString() const {
54 string result_layout = "nullopt";
55 if (result_layout_set_) {
56 result_layout = ShapeUtil::HumanStringWithLayout(result_layout_);
57 }
58 string generate_hlo_graph = "nullopt";
59 if (generate_hlo_graph_.has_value()) {
60 generate_hlo_graph = generate_hlo_graph_.value();
61 }
62 return tensorflow::strings::Printf(
63 "ExecutableBuildOptions{device_ordinal=%d, result_layout=%s, "
64 "generate_hlo_graph=%s}",
65 device_ordinal_, result_layout.c_str(), generate_hlo_graph.c_str());
66}
