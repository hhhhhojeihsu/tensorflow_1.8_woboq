44// conjunction
45//
46// Performs a compile-time logical AND operation on the passed types (which
47// must have `::value` members convertible to `bool`. Short-circuits if it
48// encounters any `false` members (and does not compare the `::value` members
49// of any remaining arguments).
50//
51// This metafunction is designed to be a drop-in replacement for the C++17
52// `std::conjunction` metafunction.
53template <typename... Ts>
54struct conjunction;
55
56template <typename T, typename... Ts>
57struct conjunction<T, Ts...>
58 : std::conditional<T::value, conjunction<Ts...>, T>::type {};
59
60template <>
61struct conjunction<> : std::true_type {};
62
63// A type trait that is valid when all elements in a parameter pack are of
64// integral type.
65template <typename... T>
66using pack_is_integral = conjunction<std::is_integral<T>...>;
67
68// Compares three same-sized vectors elementwise. For each item in `values`,
69// returns false if any of values[i] is outside the half-open range [starts[i],
70// ends[i]).
71template <typename C1, typename C2, typename C3>
72bool all_inside_range(const C1& values, const C2& range_starts,
73 const C3& range_ends) {
74 for (size_t i = 0, e = values.size(); i < e; ++i) {
75 if (values[i] < range_starts[i] || values[i] >= range_ends[i]) {
76 return false;
77 }
78 }
79 return true;
80}

84// General N dimensional array class with arbitrary value type.

89 // nests, especially if one or more dimensions is one as the compiler just

110 // Creates a 2D array from the given nested initializer list. The outer
111 // initializer list is the first dimension, the inner is the second dimension.
112 // For example, {{1, 2, 3}, {4, 5, 6}} results in an array with n1=2 and n2=3.
113 Array(InitializerList2D values)
114 : Array(ToInt64Vector({values.size(), values.begin()->size()})) {
115 int64 idx = 0;
116 for (const auto& it1 : values) {
117 for (const auto& it2 : it1) {
118 values_[idx] = it2;
119 ++idx;
120 }
121 }
122 CHECK(idx == num_elements());
123 }

283 // Fills the array with random normal variables with the specified mean.
284 void FillRandom(const T& value, const double mean = 0.0,
285 const int seed = 12345) {
286 std::mt19937 g(seed);
287 std::normal_distribution<double> distribution(mean,
288 static_cast<double>(value));
289 for (int64 i = 0; i < num_elements(); ++i) {
290 values_[i] = static_cast<T>(distribution(g));
291 }
292 }

350 // Returns the value at the cell specified by the indexes. The number of
351 // arguments have to match with the number of dimensions for the array.
352 //
353 // The type trait is required to avoid this overload participating too
354 // eagerly; a parameter pack can take zero or more elements, so we must
355 // restrict this to only parameter packs that are all of integral type.
356 template <typename... Dims>
357 typename std::enable_if<array_impl::pack_is_integral<Dims...>::value,
358 const T&>::type
359 operator()(Dims... dims) const {
360 // We are using a std::array to avoid having to allocate memory in this
361 // function for performance reasons.
362 std::array<int64, sizeof...(dims)> indexes{{static_cast<int64>(dims)...}};
363 return values_[calculate_index(indexes)];
364 }

421 bool operator==(const Array<T>& other) const {
422 if (sizes_.size() != other.sizes_.size()) {
423 return false;

440 // Performs the equivalent of a slice operation on this array.
441 Array<T> Slice(tensorflow::gtl::ArraySlice<int64> starts,
442 tensorflow::gtl::ArraySlice<int64> limits) const {
443 CHECK_EQ(starts.size(), num_dimensions());
444 CHECK_EQ(limits.size(), num_dimensions());
445
446 std::vector<int64> sizes;
447 std::transform(starts.begin(), starts.end(), limits.begin(),
448 std::back_inserter(sizes),
449 [](int64 start, int64 limit) { return limit - start; });
450 Array<T> result(sizes);
451
452 std::vector<int64> index(sizes_.size());
453 int64 slice_i = 0;
454 for (int64 i = 0; i < num_elements(); ++i, next_index(&index)) {
455 if (array_impl::all_inside_range(index, starts, limits)) {
456 // Even though the bounds of result are different to our bounds, we're
457 // iterating in the same order. So we can simply write successive linear
458 // indices instead of recalculating a multi-dimensional index.
459 result.values_[slice_i++] = values_[i];
460 }
461 }
462 return result;
463 }

465 // Performs the equivalent of a DynamicUpdateSlice in-place on this array.
466 void UpdateSlice(const Array<T>& from,
467 tensorflow::gtl::ArraySlice<int64> start_indices) {
468 CHECK_EQ(from.num_dimensions(), num_dimensions());
469 std::vector<int64> limit_indices;
470 std::transform(start_indices.begin(), start_indices.end(),
471 from.dimensions().begin(), std::back_inserter(limit_indices),
472 std::plus<int64>{});
473 std::vector<int64> index(sizes_.size());
474 int64 from_i = 0;
475 for (int64 i = 0; i < num_elements(); ++i, next_index(&index)) {
476 if (array_impl::all_inside_range(index, start_indices, limit_indices)) {
477 // Even though the bounds of from are different to our bounds, we're
478 // iterating in the same order. So we can simply write successive linear
479 // indices instead of recalculating a multi-dimensional index.
480 values_[i] = from.values_[from_i++];
481 }
482 }
483 }

493 // Returns a string representation of the array suitable for debugging.
494 string ToString() const {
495 std::vector<string> pieces;
496 std::vector<int64> index(sizes_.size());
497 do {
498 // Emit leading spaces and opening square brackets
499 if (index.back() == 0) {
500 for (int64 i = sizes_.size() - 1; i >= 0; --i) {
501 if (i == 0 || index[i - 1] != 0) {
502 for (int64 j = 0; j < sizes_.size(); ++j) {
503 pieces.push_back(j < i ? " " : "[");
504 }
505 break;
506 }
507 }
508 }
509
510 pieces.push_back(
511 tensorflow::strings::AlphaNum(values_[calculate_index(index)])
512 .data());
513
514 // Emit comma if it isn't the last element
515 if (index.back() != sizes_.back() - 1) {
516 pieces.push_back(", ");
517 }
518
519 // Emit closing square brackets
520 for (int64 i = sizes_.size() - 1; i >= 0; --i) {
521 if (index[i] != sizes_[i] - 1) {
522 break;
523 }
524 pieces.push_back("]");
525 if (i != 0 && index[i - 1] != sizes_[i - 1] - 1) {
526 pieces.push_back(",\n");
527 }
528 }
529 } while (next_index(&index));
530 return tensorflow::str_util::Join(pieces, "");
531 }

534 // Converts an initializer_list of type U to a vector of type int64. Used by
535 // the initializer list based constructors to convert the size type into int64
536 // to be passed to the size based constructor.
537 template <typename U>
538 static std::vector<int64> ToInt64Vector(
539 const std::initializer_list<U>& data) {
540 return std::vector<int64>(data.begin(), data.end());
541 }
542
543 // Returns the linear index from the list of per-dimension indexes. Function
544 // is templated so can be used with an std::array from operator() to avoid
545 // memory allocation.
546 template <typename U>
547 int64 calculate_index(const U& indexes) const {
548 CHECK_EQ(sizes_.size(), indexes.size());
549 int64 index = 0;
550 for (int64 i = 0; i < sizes_.size(); ++i) {
551 index *= sizes_[i];
552 index += indexes[i];
553 }
554 return index;
555 }
556
557 // Advances the specified set of indexes and returns true if we haven't
558 // wrapped around (i.e. result isn't {0, 0, ...}).
559 bool next_index(std::vector<int64>* index) const {
560 CHECK_EQ(index->size(), sizes_.size());
561 for (int64 i = sizes_.size() - 1; i >= 0; --i) {
562 (*index)[i]++;
563 if ((*index)[i] < sizes_[i]) {
564 return true;
565 }
566 (*index)[i] = 0;
567 }
568 return false;
569 }

571 std::vector<int64> sizes_;
572 std::unique_ptr<T[]> values_;
