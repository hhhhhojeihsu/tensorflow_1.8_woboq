33StatusOr<GlobalDataHandle> AllocationTracker::Register(
34 std::unique_ptr<ShapedBuffer> shaped_buffer, const string& tag) {
35 tensorflow::mutex_lock lock(mutex_);
36 VLOG(2) << "Register";
37 std::vector<std::unique_ptr<ShapedBuffer>> replicated_buffers;
38 replicated_buffers.emplace_back(std::move(shaped_buffer));
39 return RegisterInternal(std::move(replicated_buffers), tag);
40}

50StatusOr<GlobalDataHandle> AllocationTracker::RegisterInternal(
51 std::vector<std::unique_ptr<ShapedBuffer>> replicated_buffers,
52 const string& tag) {
53 VLOG(2) << "RegisterInternal("
54 << "tag: \"" << tag << "\" with " << replicated_buffers.size()
55 << " shaped_buffers.";
56 for (const auto& shaped_buffer : replicated_buffers) {
57 VLOG(2) << "shaped_buffer:" << *shaped_buffer;
58 if (shaped_buffer->platform() != backend_->platform()) {
59 return InvalidArgument(
60 "AllocationTracker for platform %s cannot register buffer from "
61 "platform %s",
62 backend_->platform()->Name().c_str(),
63 shaped_buffer->platform()->Name().c_str());
64 }
65 }
66
67 int64 handle = next_handle_++;
68 for (auto& shaped_buffer : replicated_buffers) {
69 std::vector<ShapeIndex> shape_indices;
70 ShapeUtil::ForEachSubshape(shaped_buffer->on_device_shape(),
71 [this, &shape_indices](const Shape& /*subshape*/,
72 const ShapeIndex& index) {
73 shape_indices.push_back(index);
74 });
75 for (const ShapeIndex& index : shape_indices) {
76 AddAllocationOrIncrementRefCount(shaped_buffer->buffer(index),
77 shaped_buffer->device_ordinal());
78 }
79 handle_to_shaped_buffers_[handle].emplace_back(std::move(shaped_buffer));
80 }
81
82 GlobalDataHandle result;
83 result.set_handle(handle);
84 VLOG(2) << "handle: " << handle;
85 return result;
86}

88tensorflow::Status AllocationTracker::Unregister(const GlobalDataHandle& data) {
89 tensorflow::mutex_lock lock(mutex_);
90 VLOG(2) << "Unregister("
91 << "handle: " << data.handle() << ")";
92 TF_ASSIGN_OR_RETURN(std::vector<const ShapedBuffer*> replicated_buffers,
93 ResolveInternal(data));
94 for (const auto& shaped_buffer : replicated_buffers) {
95 std::vector<ShapeIndex> shape_indices;
96 ShapeUtil::ForEachSubshape(shaped_buffer->on_device_shape(),
97 [this, &shape_indices](const Shape& /*subshape*/,
98 const ShapeIndex& index) {
99 shape_indices.push_back(index);
100 });
101 for (const ShapeIndex& index : shape_indices) {
102 TF_RETURN_IF_ERROR(DecrementRefCount(shaped_buffer->buffer(index),
103 shaped_buffer->device_ordinal()));
104 }
105 }
106 return Reset(data);
107}

109Status AllocationTracker::Reset(const GlobalDataHandle& data) {
110 // Keep a nullptr as a tombstone for unregistered handles. This enables
111 // better error messages. That is, "handle has been deallocated" versus
112 // "handle does not exist".
113 auto it = handle_to_shaped_buffers_.find(data.handle());
114 if (it == handle_to_shaped_buffers_.end()) {
115 return NotFound("no allocation record for global data handle: %lld",
116 data.handle());
117 }
118 for (auto& shaped_buffer : it->second) {
119 shaped_buffer.reset();
120 }
121 return tensorflow::Status::OK();
122}

124StatusOr<std::vector<GlobalDataHandle>> AllocationTracker::DeconstructTuple(
125 const GlobalDataHandle& data) {
126 tensorflow::mutex_lock lock(mutex_);
127
128 TF_ASSIGN_OR_RETURN(std::vector<const ShapedBuffer*> replicated_buffers,
129 ResolveInternal(data));
130 // We only need to care about replica id 0 here, since the GlobalDataHandle is
131 // the same for all buffers across replicas.
132 const ShapedBuffer* shaped_buffer = replicated_buffers[0];
133 if (!ShapeUtil::IsTuple(shaped_buffer->on_host_shape())) {
134 return InvalidArgument("global data handle %lld is not a tuple",
135 data.handle());
136 }
137 // If the on-host representation is a tuple, then the on-device one should be
138 // as well.
139 TF_RET_CHECK(ShapeUtil::IsTuple(shaped_buffer->on_device_shape()));
140
141 if (ShapeUtil::IsNestedTuple(shaped_buffer->on_device_shape())) {
142 return Unimplemented("Deconstructing nested tuples is not implemented.");
143 }
144
145 std::vector<GlobalDataHandle> element_handles;
146 for (int i = 0;
147 i < ShapeUtil::TupleElementCount(shaped_buffer->on_device_shape());
148 ++i) {
149 auto element_buffer = MakeUnique<ShapedBuffer>(
150 ShapeUtil::GetTupleElementShape(shaped_buffer->on_host_shape(), i),
151 ShapeUtil::GetTupleElementShape(shaped_buffer->on_device_shape(), i),
152 shaped_buffer->platform(), shaped_buffer->device_ordinal());
153 element_buffer->set_buffer(shaped_buffer->buffer(/*index=*/{i}),
154 /*index=*/{});
155 std::vector<std::unique_ptr<ShapedBuffer>> replicated_buffers;
156 replicated_buffers.emplace_back(std::move(element_buffer));
157 TF_ASSIGN_OR_RETURN(
158 GlobalDataHandle element_handle,
159 RegisterInternal(std::move(replicated_buffers), "deconstructed tuple"));
160
161 element_handles.push_back(element_handle);
162 }
163 return std::move(element_handles);
164}

168 tensorflow::mutex_lock lock(mutex_);
169 return AllocationTracker::ResolveInternal(data);

172StatusOr<const ShapedBuffer*> AllocationTracker::ResolveForReplica(
173 const GlobalDataHandle& data, int replica_id) {
174 tensorflow::mutex_lock lock(mutex_);
175 TF_ASSIGN_OR_RETURN(std::vector<const ShapedBuffer*> replicated_buffers,
176 ResolveInternal(data));
177 if (replica_id >= replicated_buffers.size()) {
178 return InvalidArgument(
179 "Requesting buffer for replica %d, but found buffers only for %lu "
180 "replicas.",
181 replica_id, replicated_buffers.size());
182 }
183 return replicated_buffers[replica_id];
184}
185
186StatusOr<std::vector<const ShapedBuffer*>> AllocationTracker::ResolveInternal(
187 const GlobalDataHandle& data) {
188 VLOG(2) << "resolve:" << data.handle();
189 auto it = handle_to_shaped_buffers_.find(data.handle());
190 if (it == handle_to_shaped_buffers_.end()) {
191 return NotFound("no allocation record for global data handle: %lld",
192 data.handle());
193 }
194 std::vector<const ShapedBuffer*> replicated_buffers;
195 for (const auto& shaped_buffer : it->second) {
196 if (shaped_buffer == nullptr) {
197 return InvalidArgument(
198 "global data handle %lld was previously deallocated", data.handle());
199 }
200 replicated_buffers.push_back(shaped_buffer.get());
201 }
202
203 return replicated_buffers;
204}

206void AllocationTracker::AddAllocationOrIncrementRefCount(
207 perftools::gputools::DeviceMemoryBase device_memory, int device_ordinal) {
208 AllocationMap& allocation_map = opaque_to_allocation_map_[device_ordinal];
209 auto it = allocation_map.find(device_memory.opaque());
210 if (it == allocation_map.end()) {
211 allocation_map[device_memory.opaque()] = {device_memory, device_ordinal,
212 /*ref_count=*/1};
213 } else {
214 it->second.ref_count++;
215 }
216}
217
218Status AllocationTracker::DecrementRefCount(
219 perftools::gputools::DeviceMemoryBase device_memory, int device_ordinal) {
220 AllocationMap& allocation_map = opaque_to_allocation_map_[device_ordinal];
221 auto it = allocation_map.find(device_memory.opaque());
222 TF_RET_CHECK(it != allocation_map.end());
223 Allocation& allocation = it->second;
224 TF_RET_CHECK(allocation.ref_count >= 1);
225 if (allocation.ref_count == 1) {
226 TF_RETURN_IF_ERROR(backend_->memory_allocator()->Deallocate(
227 device_ordinal, &device_memory));
228 allocation_map.erase(it);
229 } else {
230 allocation.ref_count--;
231 }
232 return tensorflow::Status::OK();
233}
