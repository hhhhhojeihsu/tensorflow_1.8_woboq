63// Walk the call graph of the HLO module and place each computation into either
64// thread_local_computations or global_computations depending upon whether the
65// computation requires thread-local allocations or global allocations. The
66// elements in thread_local_computations and global_computations are in post
67// order (if computation A has an instruction which calls computation B, then A
68// will appear after B in the vector).
69Status GatherComputationsByAllocationType(
70 const HloModule* module,
71 std::vector<const HloComputation*>* thread_local_computations,
72 std::vector<const HloComputation*>* global_computations) {
73 // Create a worklist of computations paired with whether the allocation must
74 // be thread-local.
75 std::deque<std::pair<const HloComputation*, bool>> worklist;
76 worklist.push_back(std::make_pair(module->entry_computation(),
77 /*is_thread_local*/ false));
78
79 // Sets for quickly checking membership. Computations are returned in vectors
80 // for stable iteration.
81 FlatSet<const HloComputation*> thread_local_set;
82 FlatSet<const HloComputation*> global_set;
83
84 while (!worklist.empty()) {
85 auto worklist_front = worklist.front();
86 worklist.pop_front();
87 const HloComputation* computation = worklist_front.first;
88 bool is_thread_local = worklist_front.second;
89 bool in_thread_local_set = thread_local_set.count(computation) > 0;
90 bool in_global_set = global_set.count(computation) > 0;
91
92 // If the computation has already been added to the respective set, then
93 // nothing to do.
94 if ((is_thread_local && in_thread_local_set) ||
95 (!is_thread_local && in_global_set)) {
96 continue;
97 }
98
99 // If the computation has already been added to the other set this is an
100 // error condition because the global call to the computation (eg,
101 // while/call) may return a reference to one of the thread-local buffers to
102 // the calling computation which will become a dangling reference when the
103 // thread-local is deallocated with the call return.
104 if ((is_thread_local && in_global_set) ||
105 (!is_thread_local && in_thread_local_set)) {
106 return InvalidArgument(
107 "computation %s has conflicting allocation requirements (global "
108 "and thread-local)",
109 computation->name().c_str());
110 }
111
112 if (is_thread_local) {
113 thread_local_set.insert(computation);
114 } else {
115 global_set.insert(computation);
116 }
117
118 for (auto* instruction : computation->instructions()) {
119 for (HloComputation* subcomputation :
120 instruction->called_computations()) {
121 switch (instruction->opcode()) {
122 case HloOpcode::kCall:
123 case HloOpcode::kConditional:
124 case HloOpcode::kWhile:
125 // Call and while must be called from a computation with global
126 // allocations as they may return references to buffers inside the
127 // called computation which cannot be thread-local.
128 if (is_thread_local) {
129 return InvalidArgument(
130 "computation %s cannot contain call/while op because it "
131 "requires thread-local buffer allocations",
132 computation->name().c_str());
133 }
134 worklist.push_back(std::make_pair(subcomputation,
135 false)); // Not thread local.
136 break;
137 case HloOpcode::kMap:
138 case HloOpcode::kReduce:
139 case HloOpcode::kReduceWindow:
140 case HloOpcode::kSelectAndScatter:
141 case HloOpcode::kFusion:
142 // Map/reduce etc computations are always thread-local.
143 worklist.push_back(std::make_pair(subcomputation,
144 true)); // Thread local.
145 break;
146 default:
147 return InternalError(
148 "Unexpected calling opcode: %s",
149 HloOpcodeString(instruction->opcode()).c_str());
150 }
151 }
152 }
153 }
154
155 // Add the computations to the vectors in post order.
156 for (auto* computation : module->MakeComputationPostOrder()) {
157 if (thread_local_set.count(computation) > 0) {
158 thread_local_computations->push_back(computation);
159 } else if (global_set.count(computation) > 0) {
160 global_computations->push_back(computation);
161 }
162 // If the computation is not reachable from the entry computation, then it
163 // will not appear in either thread_local_set or global_set. We don't bother
164 // assigning buffers for these.
165 }
166 return Status::OK();
167}

169// Checks that points-to set of 'instruction' is unambiguous and distinct
170// (ensured by CopyInsertion), then adds the buffer from the points-to set at
171// 'index' to 'colocated_set'.
172const LogicalBuffer* AddBufferToColocatedSet(
173 const HloInstruction* instruction, const ShapeIndex& index,
174 const TuplePointsToAnalysis& points_to_analysis,
175 std::vector<const LogicalBuffer*>* colocated_set) {
176 // CopyInsertion ensures root points-to set is unambiguous and distinct.
177 const auto& points_to = points_to_analysis.GetPointsToSet(instruction);
178 DCHECK(!points_to.IsAmbiguous());
179 colocated_set->push_back(points_to.element(index)[0]);
180 return colocated_set->back();
181}

183// Given the interference map of a graph (the list of interfering node indices
184// for each node), perform graph coloring such that interfering nodes are
185// assigned to different colors. Returns the assigned color of the nodes, where
186// the colors are represented as integer values [0, color_count).
187std::vector<int64> ColorInterferenceGraph(
188 const std::vector<std::vector<int64>>& interference_map) {
189 const int64 node_count = interference_map.size();
190
191 // Sort the nodes such that we assign nodes with more interference first. This
192 // relies on the common heuristic of assigning the most constrained node
193 // first, but it would be good to investigate other ordering heuristics too.
194 std::vector<int64> nodes(node_count);
195 std::iota(nodes.begin(), nodes.end(), 0);
196 std::sort(nodes.begin(), nodes.end(),
197 [&interference_map](const int64 i, const int64 j) {
198 return interference_map[i].size() > interference_map[j].size();
199 });
200
201 const int64 kColorUnassigned = -1;
202 std::vector<int64> assigned_colors(node_count, kColorUnassigned);
203 for (int64 node : nodes) {
204 // Mark the colors that are already assigned to the neighbors.
205 std::vector<bool> available_colors(node_count, true);
206 for (int64 neighbor : interference_map[node]) {
207 int64 color = assigned_colors[neighbor];
208 if (color != kColorUnassigned) {
209 available_colors[color] = false;
210 }
211 }
212
213 // Find the color that is not yet assigned to the neighbors.
214 int64 color = kColorUnassigned;
215 for (color = 0; color < available_colors.size(); ++color) {
216 if (available_colors[color]) {
217 break;
218 }
219 }
220 CHECK_NE(color, kColorUnassigned);
221 assigned_colors[node] = color;
222 }
223 return assigned_colors;
224}

228size_t BufferAllocation::Slice::Hasher::operator()(Slice s) const {
229 uint64 h = std::hash<int64>()(s.index());
230 h = tensorflow::Hash64Combine(h, std::hash<int64>()(s.offset()));
231 h = tensorflow::Hash64Combine(h, std::hash<int64>()(s.size()));
232 return h;
233}

246void BufferAllocation::AddAssignment(const LogicalBuffer& buffer, int64 offset,
247 int64 size) {
248 VLOG(4) << "Trying to add " << buffer << " to " << this;
249 CHECK(assigned_buffers_.count(&buffer) == 0)
250 << "LogicalBuffer " << buffer << " already assigned to allocation "
251 << index_;
252 CHECK_LE(offset, size_) << "LogicalBuffer " << buffer
253 << " offset out of range";
254 CHECK_LE(offset + size, size_)
255 << "LogicalBuffer " << buffer
256 << " size out of range at offset: " << offset << " with size: " << size;
257 CHECK_EQ(buffer.color(), color())
258 << "Buffer color " << buffer.color() << " for buffer " << buffer
259 << " does not match allocation color " << color() << ".";
260 OffsetSize offset_size;
261 offset_size.offset = offset;
262 offset_size.size = size;
263 assigned_buffers_.emplace(&buffer, offset_size);
264}

266BufferAllocationProto BufferAllocation::ToProto() const {
267 BufferAllocationProto proto;
268 proto.set_index(index_);
269 proto.set_size(size_);
270 proto.set_is_thread_local(is_thread_local_);
271 proto.set_is_reusable(is_reusable_);
272 proto.set_color(color_.value());
273 if (is_entry_computation_parameter_) {
274 proto.set_is_entry_computation_parameter(true);
275 for (int64 idx : param_shape_index()) {
276 proto.add_parameter_shape_index(idx);
277 }
278 proto.set_parameter_number(parameter_number_);
279 }
280 proto.set_maybe_live_out(maybe_live_out_);
281 for (const auto& buffer_offset_size : assigned_buffers_) {
282 BufferAllocationProto::Assigned* proto_assigned = proto.add_assigned();
283 proto_assigned->set_logical_buffer_id(buffer_offset_size.first->id());
284 proto_assigned->set_offset(buffer_offset_size.second.offset);
285 proto_assigned->set_size(buffer_offset_size.second.size);
286 }
287 std::sort(proto.mutable_assigned()->begin(), proto.mutable_assigned()->end(),
288 [](const BufferAllocationProto::Assigned& assign1,
289 const BufferAllocationProto::Assigned& assign2) {
290 return assign1.logical_buffer_id() < assign2.logical_buffer_id();
291 });
292 return proto;
293}

295std::pair<int64, std::vector<const LogicalBuffer*>>
296BufferAllocation::ComputePeakMemoryLogicalBuffers() const {
297 if (HeapTraces().empty()) {
298 // Just return the largest LogicalBuffer in the allocation.
299 const LogicalBuffer* largest_buffer = nullptr;
300 int64 largest_size = 0;
301 for (const auto& pair : assigned_buffers()) {
302 const LogicalBuffer* buffer = pair.first;
303 int64 size = pair.second.size;
304 if (largest_buffer == nullptr) {
305 largest_buffer = buffer;
306 largest_size = size;
307 continue;
308 }
309 // Tie-break with LogicalBuffer::Id so the return value is stable relative
310 // to changing addresses.
311 if (size > largest_size ||
312 ((size == largest_size) && (largest_buffer->id() > buffer->id()))) {
313 largest_buffer = buffer;
314 largest_size = size;
315 }
316 }
317 CHECK(largest_buffer != nullptr)
318 << "No logical buffers in allocation: " << ToString();
319 return {largest_size, {largest_buffer}};
320 }
321
322 // Create a map from LogicalBuffer::Id to LogicalBuffer* for the logical
323 // buffers in this allocation.
324 tensorflow::gtl::FlatMap<LogicalBuffer::Id, const LogicalBuffer*>
325 id_to_buffer;
326 tensorflow::gtl::FlatMap<const LogicalBuffer*, int64> buffer_sizes;
327 for (const auto& pair : assigned_buffers()) {
328 const LogicalBuffer* buffer = pair.first;
329 const OffsetSize& offset_size = pair.second;
330 id_to_buffer[buffer->id()] = buffer;
331 buffer_sizes[buffer] = offset_size.size;
332 }
333
334 // Returns how much the given event increases the total size of live
335 // buffers. Can be negative.
336 auto memory_delta = [this, &id_to_buffer, &buffer_sizes](
337 const HeapSimulatorTrace::Event& event) -> int64 {
338 const LogicalBuffer* buffer = id_to_buffer.at(event.buffer_id());
339 const int64 buffer_size = buffer_sizes.at(buffer);
340 if (event.kind() == HeapSimulatorTrace::Event::ALLOC) {
341 return buffer_size;
342 } else if (event.kind() == HeapSimulatorTrace::Event::SHARE_WITH) {
343 // Sharing a buffer does not change the live set size for the purposes of
344 // the heap simulator. Even though the shared-with buffer may be smaller,
345 // the entire allocation remains live.
346 return 0;
347 } else if (event.kind() == HeapSimulatorTrace::Event::FREE) {
348 return -1 * buffer_size;
349 }
350 LOG(FATAL) << "Unknown event kind: " << event.kind();
351 };
352
353 int64 total_max_live_size = 0;
354 std::vector<const LogicalBuffer*> live_buffers_vector;
355 for (const HeapSimulatorTrace& heap_trace : HeapTraces()) {
356 // First compute the size of the maximal live set.
357 int64 max_live_size = 0;
358 int64 live_size = 0;
359 for (const auto& event : heap_trace.events()) {
360 live_size += memory_delta(event);
361 if (max_live_size < live_size) {
362 max_live_size = live_size;
363 }
364 }
365
366 // Next gather the set of logical buffers live at the earliest point of
367 // maximal live set size.
368 tensorflow::gtl::FlatSet<const LogicalBuffer*> live_buffers;
369 live_size = 0;
370 for (const auto& event : heap_trace.events()) {
371 const LogicalBuffer* buffer = id_to_buffer.at(event.buffer_id());
372 if (event.kind() == HeapSimulatorTrace::Event::ALLOC) {
373 InsertOrDie(&live_buffers, buffer);
374 } else if (event.kind() == HeapSimulatorTrace::Event::SHARE_WITH) {
375 // Nothing to do.
376 } else if (event.kind() == HeapSimulatorTrace::Event::FREE) {
377 CHECK(ContainsKey(live_buffers, buffer));
378 live_buffers.erase(buffer);
379 }
380
381 live_size += memory_delta(event);
382 if (live_size == max_live_size) {
383 break;
384 }
385 }
386 CHECK_EQ(live_size, max_live_size);
387 total_max_live_size += max_live_size;
388
389 live_buffers_vector.insert(live_buffers_vector.end(), live_buffers.begin(),
390 live_buffers.end());
391 }
392
393 // Stabily sort the live buffers.
394 std::sort(live_buffers_vector.begin(), live_buffers_vector.end(),
395 [](const LogicalBuffer* a, const LogicalBuffer* b) {
396 return a->id() < b->id();
397 });
398 return {total_max_live_size, live_buffers_vector};
399}

401string BufferAllocation::ToString() const {
402 string output;
403 Appendf(&output, "allocation %lld: %p, size %lld", index_, this, size());
404 if (color().value() != 0) {
405 StrAppend(&output, ", color ", color().value());
406 }
407 if (is_entry_computation_parameter()) {
408 StrAppend(&output, ", parameter ", parameter_number(), " at ShapeIndex ",
409 param_shape_index().ToString());
410 }
411 if (is_thread_local()) {
412 StrAppend(&output, ", thread-local");
413 }
414 if (maybe_live_out()) {
415 StrAppend(&output, ", maybe-live-out");
416 }
417 if (IsPreallocatedTempBuffer()) {
418 StrAppend(&output, ", preallocated-temp");
419 }
420 StrAppend(&output, ":\n");
421 // Dump the assigned buffers ordered by id.
422 std::vector<const LogicalBuffer*> sorted_buffers;
423 for (const auto& buffer_offset_size : assigned_buffers_) {
424 sorted_buffers.push_back(buffer_offset_size.first);
425 }
426 std::sort(sorted_buffers.begin(), sorted_buffers.end(),
427 [](const LogicalBuffer* a, const LogicalBuffer* b) {
428 return a->id() < b->id();
429 });
430 for (const LogicalBuffer* buffer : sorted_buffers) {
431 const OffsetSize& offset_size = FindOrDie(assigned_buffers_, buffer);
432 StrAppend(&output,
433 tensorflow::strings::Printf(
434 " %s [%lld,%lld]: %s\n", buffer->ToString().c_str(),
435 offset_size.offset, offset_size.size,
436 ShapeUtil::HumanStringWithLayout(buffer->shape()).c_str()));
437 }
438 return output;
439}

511StatusOr<BufferAllocation::Slice> BufferAssignment::GetUniqueSlice(
512 const HloInstruction* instruction, const ShapeIndex& index) const {
513 VLOG(3) << "Trying to find unique slice for " << instruction->name() << " ["
514 << index << "]";
515 BufferAllocation::Slice result;
516 for (const LogicalBuffer* buffer :
517 GetPointsToSet(instruction).element(index)) {
518 VLOG(3) << "Examining buffer " << *buffer;
519 if (HasAllocation(*buffer)) {
520 VLOG(3) << "Has allocation";
521 const BufferAllocation::Slice slice =
522 GetAssignedAllocation(*buffer).GetSlice(*buffer);
523 if (result.allocation() == nullptr) {
524 result = slice;
525 } else if (result != slice) {
526 return FailedPrecondition(
527 "BufferAllocation::Slice for instruction %s at index %s cannot "
528 "be determined at compile-time.",
529 instruction->name().c_str(), index.ToString().c_str());
530 }
531 } else {
532 VLOG(3) << "No allocation";
533 }
534 }
535 if (result.allocation() == nullptr) {
536 return FailedPrecondition(
537 "BufferAllocation::Slice not assigned for instruction %s at index %s",
538 instruction->name().c_str(), index.ToString().c_str());
539 }
540 return result;
541}

555bool BufferAssignment::HaveDisjointSlices(const HloInstruction* hlo_a,
556 const HloInstruction* hlo_b) const {
557 using SliceSet =
558 FlatSet<BufferAllocation::Slice, BufferAllocation::Slice::Hasher>;
559 // Gets the slices all of instr's subshapes. If any subshape doesn't have an
560 // assigned slice, returns the empty set.
561 auto collect_slices = [&](const HloInstruction* instr) -> SliceSet {
562 SliceSet slices;
563 Status status = ShapeUtil::ForEachSubshapeWithStatus(
564 instr->shape(),
565 [&](const Shape& /*subshape*/, const ShapeIndex& index) {
566 auto shape_slices = GetAllSlices(instr, index);
567 if (shape_slices.empty()) {
568 return InvalidArgument("No slices assigned to part of instr.");
569 }
570 slices.insert(shape_slices.begin(), shape_slices.end());
571 return Status::OK();
572 });
573 if (!status.ok()) {
574 return {};
575 }
576 return slices;
577 };
578
579 SliceSet slices_a = collect_slices(hlo_a);
580 SliceSet slices_b = collect_slices(hlo_b);
581 // hlo_a and hlo_b have disjoint slices if collect_slices succeeded (i.e.
582 // didn't return the empty set) for both HLOs, and the two resulting sets of
583 // slices are disjoint.
584 return !slices_a.empty() && !slices_b.empty() &&
585 std::none_of(slices_a.begin(), slices_a.end(),
586 [&](const BufferAllocation::Slice& slice) {
587 return slices_b.count(slice) > 0;
588 });
589}

631// Combines allocations of temporary buffers of the same color into one big
632// BufferAllocation.
633void BufferAssignment::CombineTempAllocations() {
634 VLOG(1) << "CombineTempAllocations()";
635 FlatMap<LogicalBuffer::Color, BufferAllocation, LogicalBuffer::Color::Hasher>
636 combined_allocation_map;
637
638 // Move all temp allocations into a single run at the end of the allocations
639 // vector.
640 const auto first_temp_it =
641 std::partition(allocations_.begin(), allocations_.end(),
642 [](const BufferAllocation& allocation) {
643 return !allocation.IsPreallocatedTempBuffer();
644 });
645
646 // Walk over the run of temp allocations, collecting the allocations belonging
647 // to the same color.
648 if (first_temp_it != allocations_.end()) {
649 for (auto it = first_temp_it; it != allocations_.end(); ++it) {
650 const BufferAllocation& temp_allocation = *it;
651 LogicalBuffer::Color color = temp_allocation.color();
652 auto combined_it = combined_allocation_map.find(color);
653 if (combined_it == combined_allocation_map.end()) {
654 // We have found the first temp allocation of this color. Collect
655 // the other temp allocations of the same color into it.
656 VLOG(1) << "Combined temp allocation for color " << color
657 << " is: " << temp_allocation;
658 combined_allocation_map.emplace(color, temp_allocation);
659 continue;
660 }
661
662 auto* combined_allocation = &combined_it->second;
663 VLOG(1) << "Combined allocation absorbing temp allocation: "
664 << temp_allocation;
665
666 // Each temp allocation is placed end-to-end, accounting for alignment.
667 // The offset of each buffer in the combined allocation is computed from
668 // the base offset of the allocation.
669 int64 alignment = color_alignment_(color);
670 const int64 base =
671 RoundUpToNearest(combined_allocation->size(), alignment);
672 combined_allocation->set_size(base + temp_allocation.size());
673 for (const auto& buffer_offset_size : temp_allocation.assigned_buffers_) {
674 const LogicalBuffer* buffer = buffer_offset_size.first;
675 const int64 offset = buffer_offset_size.second.offset;
676 const int64 size = buffer_offset_size.second.size;
677 combined_allocation->AddAssignment(*buffer, base + offset, size);
678 }
679 if (!temp_allocation.HeapTraces().empty()) {
680 CHECK_EQ(temp_allocation.HeapTraces().size(), 1);
681 combined_allocation->AddHeapTrace(temp_allocation.HeapTraces().front());
682 }
683 }
684 // Replace all existing temporary allocations with the new combined
685 // allocations.
686 allocations_.erase(first_temp_it, allocations_.end());
687 for (auto& combined : combined_allocation_map) {
688 allocations_.push_back(combined.second);
689 temp_allocation_total_size_ += combined.second.size();
690 }
691 }
692
693 // Update allocation indices to their new positions.
694 allocation_index_for_buffer_.clear_no_resize();
695 for (size_t index = 0; index < allocations_.size(); ++index) {
696 BufferAllocation* allocation = &allocations_[index];
697 allocation->set_index(index);
698 for (const auto& buffer_offset_size : allocation->assigned_buffers_) {
699 const LogicalBuffer* buffer = buffer_offset_size.first;
700 allocation_index_for_buffer_[buffer] = index;
701 }
702 }
703}

705Status BufferAssignment::ComputeSummaryStats() {
706 for (auto& allocation : Allocations()) {
707 if (allocation.is_entry_computation_parameter()) {
708 stats_.parameter_allocation_count++;
709 stats_.parameter_allocation_bytes += allocation.size();
710 }
711 if (allocation.maybe_live_out()) {
712 stats_.maybe_live_out_allocation_count++;
713 stats_.maybe_live_out_allocation_bytes += allocation.size();
714 }
715 if (allocation.IsPreallocatedTempBuffer()) {
716 stats_.preallocated_temp_allocation_count++;
717 stats_.preallocated_temp_allocation_bytes += allocation.size();
718 }
719 stats_.total_allocation_count++;
720 stats_.total_allocation_bytes += allocation.size();
721 }
722
723 // Only compute total fragmentation if all computations are sequential.
724 SequentialHloOrdering::HloModuleSequence module_sequence;
725 for (const auto& computation : module_->computations()) {
726 const std::vector<const HloInstruction*>* sequence =
727 liveness_->hlo_ordering().SequentialOrder(*computation);
728 if (sequence != nullptr) {
729 module_sequence.emplace(computation, *sequence);
730 }
731 }
732 if (module_sequence.size() == module_->computation_count()) {
733 TF_ASSIGN_OR_RETURN(
734 const int64 min_size,
735 MinimumMemoryForSequence(module_sequence, buffer_size_));
736 stats_.total_fragmentation_bytes = stats_.total_allocation_bytes - min_size;
737 }
738
739 return Status::OK();
740}

779BufferAssignmentProto BufferAssignment::ToProto() const {
780 BufferAssignmentProto proto;
781 // NOTE: TuplePointsToAnalysis state is serialized here in BufferAssigment,
782 // because we need to do the HasAllocation check for each buffer. Otherwise
783 // the buffer_size_ call might fail for some backends.
784 const TuplePointsToAnalysis& points_to_analysis =
785 liveness_->points_to_analysis();
786 for (LogicalBuffer::Id id = 0; id < points_to_analysis.num_logical_buffers();
787 id++) {
788 auto& buffer = points_to_analysis.logical_buffer(id);
789 if (HasAllocation(buffer)) {
790 LogicalBufferProto proto_buffer = buffer.ToProto(buffer_size_);
791 proto.add_logical_buffers()->Swap(&proto_buffer);
792
793 // Fill buffer aliases.
794 for (const BufferAlias& alias :
795 points_to_analysis.GetBufferAliases(buffer)) {
796 if (alias.instruction() == buffer.instruction() &&
797 alias.index() == buffer.index()) {
798 continue; // skip self-aliases
799 }
800 BufferAssignmentProto::BufferAlias* proto_alias =
801 proto.add_buffer_aliases();
802 LogicalBufferProto::Location proto_alias_location =
803 LogicalBuffer::ToLocationProto(*alias.instruction(), alias.index());
804 proto_alias->set_source_buffer_id(buffer.id());
805 proto_alias->mutable_location()->Swap(&proto_alias_location);
806 }
807 }
808 }
809 for (const BufferAllocation& allocation : Allocations()) {
810 BufferAllocationProto proto_allocation = allocation.ToProto();
811 proto.add_buffer_allocations()->Swap(&proto_allocation);
812 for (const HeapSimulatorTrace& heap_trace : allocation.HeapTraces()) {
813 *proto.add_heap_simulator_traces() = heap_trace;
814 }
815 }
816 return proto;

831bool BufferAssigner::MaybeAssignBuffer(BufferAllocation* allocation,
832 const LogicalBuffer& buffer,
833 BufferAssignment* assignment) {
834 const LogicalBuffer::SizeFunction& buffer_size = assignment->buffer_size_;
835
836 CHECK(!assignment->HasAllocation(buffer))
837 << "buffer " << buffer << " already has an allocation assigned.";
838
839 VLOG(4) << "Trying to assign " << buffer << " to allocation: " << *allocation;
840
841 if (buffer.color() != allocation->color()) {
842 VLOG(4) << "Can't assign: buffer has color" << buffer.color()
843 << " and allocation has color " << allocation->color() << ".";
844 return false;
845 }
846
847 if (buffer_size(buffer) > allocation->size()) {
848 VLOG(4) << "Can't assign: buffer is larger than allocation ("
849 << buffer_size(buffer) << " > " << allocation->size() << ")";
850 return false;
851 }
852
853 if (allocation->is_entry_computation_parameter()) {
854 VLOG(4) << "Can't assign: allocation holds parameter";
855 return false;
856 }
857
858 if (!allocation->is_reusable()) {
859 VLOG(4) << "Can't assign: allocation is not reusable";
860 return false;
861 }
862
863 for (const auto& buffer_offset_size : allocation->assigned_buffers()) {
864 const LogicalBuffer& assigned_buffer = *buffer_offset_size.first;
865 if (assignment->liveness().MayInterfere(assigned_buffer, buffer)) {
866 VLOG(4) << "Can't assign: assignee " << assigned_buffer
867 << " may interfere with " << buffer;
868 return false;
869 }
870 // Copy instruction don't share a buffer with their input operand.
871 if (buffer.instruction()->IsUserOf(assigned_buffer.instruction()) &&
872 buffer.instruction()->opcode() == HloOpcode::kCopy) {
873 VLOG(4) << "Can't assign: assignee " << assigned_buffer
874 << " is used at copy instruction " << buffer;
875 return false;
876 }
877 }
878
879 if (allow_input_output_aliasing_ && allocation->maybe_live_out()) {
880 const HloComputation* entry_computation =
881 assignment->module_->entry_computation();
882 for (auto param : entry_computation->parameter_instructions()) {
883 for (auto& param_buffer :
884 assignment->points_to_analysis().GetBuffersDefinedByInstruction(
885 param)) {
886 if (assignment->liveness().MayInterfere(*param_buffer, buffer)) {
887 VLOG(4) << "Can't assign: Parameter interference with result";
888 return false;
889 }
890 }
891 }
892 }
893
894 // If the buffer is live out of the computation then it should only be
895 // assigned a buffer which exactly fits the result to avoid wasting memory
896 // (result buffers can have arbitrary lifetimes).
897 if (assignment->liveness().MaybeLiveOut(buffer) &&
898 allocation->size() != buffer_size(buffer)) {
899 VLOG(4) << "Can't assign: buffer " << buffer
900 << "is live out and size not the same as allocation";
901 return false;
902 }
903
904 assignment->AddAssignment(allocation, buffer, /*offset=*/0,
905 buffer_size(buffer));
906 return true;
907}
908
909Status BufferAssigner::AssignBuffersForComputation(
910 const HloComputation* computation, const DebugOptions& debug_options,
911 bool is_thread_local,
912 const FlatSet<const LogicalBuffer*>& colocated_buffers,
913 const FlatSet<BufferAllocation::Index>& colocated_allocations,
914 FlatMap<const HloComputation*, FlatSet<const LogicalBuffer*>>*
915 buffers_to_assign_sequentially,
916 BufferAssignment* assignment) {
917 // Buffers are sorted and assigned to BufferAllocations in decreasing order of
918 // size.
919 std::vector<const LogicalBuffer*> sorted_buffers;
920 for (auto* instruction : computation->instructions()) {
921 // Add all buffers which this instruction defines. Instruction which don't
922 // define buffers (eg, bitcast which just forwards a pointer) don't need
923 // any allocations.
924 for (const LogicalBuffer* buffer :
925 assignment->points_to_analysis().GetBuffersDefinedByInstruction(
926 instruction)) {
927 sorted_buffers.push_back(buffer);
928 }
929 }
930
931 // Generate a post order sort of instructions for sorting of the
932 // LogicalBuffers.
933 FlatMap<const HloInstruction*, int> post_order_position;
934 int position = 0;
935 for (auto* instruction : computation->MakeInstructionPostOrder()) {
936 post_order_position.emplace(instruction, position);
937 position++;
938 }
939
940 // If there is a sequential instruction ordering, we'll delay assignment of
941 // temp buffers until after the main assignment loop.
942 const BufferLiveness& liveness = assignment->liveness();
943 const bool has_sequential_order =
944 liveness.hlo_ordering().SequentialOrder(*computation) != nullptr;
945 if (has_sequential_order && buffers_to_assign_sequentially != nullptr) {
946 // Every sequential computation must get an entry in the
947 // buffers_to_assign_sequentially map, even if we end up with an empty set
948 // of buffers. This ensures we can correctly determine whether to run
949 // whole-module heap simulation.
950 buffers_to_assign_sequentially->emplace(computation,
951 FlatSet<const LogicalBuffer*>());
952 }
953
954 // Sort the LogicalBuffers first by size. We assign the larger LogicalBuffers
955 // first for simplicity. This means any previously created BufferAllocation is
956 // necessarily large enough to hold the output of the current Buffer in
957 // consideration.
958 //
959 // As a secondary sorting criteria, if the instructions are sequentially
960 // ordered, we assign live-out buffers before others. Note that for sequential
961 // computations, we'll take temp buffers that can't re-use any allocations and
962 // assign them via a heap scheduler. By assigning live-out buffers first, we
963 // increase the odds that temp buffers can re-use an allocation.
964 //
965 // As a final tiebreaker use post order position of the HLO instruction which
966 // defines the buffer. This means an instruction will appear after its
967 // operands (assuming operands are the same/larger size) enabling the
968 // important reuse case where an elementwise instruction reuses one of its
969 // operand's buffer. This improves locality.
970 std::sort(sorted_buffers.begin(), sorted_buffers.end(),
971 [this, has_sequential_order, &liveness, &post_order_position,
972 assignment](const LogicalBuffer* a, const LogicalBuffer* b) {
973 // Primary sort is by decreasing buffer size.
974 const int64 a_size = assignment->buffer_size_(*a);
975 const int64 b_size = assignment->buffer_size_(*b);
976 if (a_size != b_size) {
977 return a_size > b_size; // use ">" for decreasing size.
978 }
979 // Otherwise live out buffers come before others, if the
980 // instructions are sequentially ordered.
981 if (has_sequential_order) {
982 const bool a_live_out = liveness.MaybeLiveOut(*a);
983 const bool b_live_out = liveness.MaybeLiveOut(*b);
984 if (a_live_out != b_live_out) {
985 return a_live_out;
986 }
987 }
988 // Final tiebreaker is in instruction post order.
989 return post_order_position.at(a->instruction()) <
990 post_order_position.at(b->instruction());
991 });
992
993 // BufferAllocations are necessarily created in decreasing size order. Keep
994 // indices of previously created BufferAllocations in allocation_indices.
995 std::vector<BufferAllocation::Index> allocation_indices;
996 for (const LogicalBuffer* buffer : sorted_buffers) {
997 VLOG(3) << "Assigning allocation to: " << *buffer;
998 if (colocated_buffers.count(buffer) > 0) {
999 // Colocated buffers are currently assigned in an earlier pass.
1000 VLOG(3) << "Skipping colocated buffer: " << *buffer;
1001 continue;
1002 }
1003
1004 TF_RET_CHECK(!assignment->HasAllocation(*buffer));
1005
1006 const HloInstruction* instruction = buffer->instruction();
1007 if (instruction->opcode() == HloOpcode::kConstant) {
1008 // No BufferAllocations for constants.
1009 // TODO(b/32248867): For consistency, constants should get allocations.
1010 VLOG(3) << "Skipping constant: " << *buffer;
1011 continue;
1012 }
1013
1014 const int64 buffer_size = assignment->buffer_size_(*buffer);
1015
1016 const bool is_entry_parameter =
1017 instruction->opcode() == HloOpcode::kParameter &&
1018 computation == computation->parent()->entry_computation();
1019 if (is_entry_parameter) {
1020 // If the LogicalBuffer is part of an external parameter, creates a new
1021 // allocation and sets its parameter number. Parameters of non-entry
1022 // computations do not need special allocations because they live inside
1023 // callers.
1024 BufferAllocation* allocation =
1025 assignment->NewAllocation(*buffer, buffer_size,
1026 /*is_thread_local=*/false,
1027 /*is_reusable=*/false);
1028 allocation->set_entry_computation_parameter(
1029 instruction->parameter_number(), buffer->index());
1030 VLOG(3) << "New allocation #" << allocation->index()
1031 << " for entry computation parameter: " << *buffer;
1032 continue;
1033 }
1034
1035 if (is_thread_local) {
1036 // We do not reuse thread-local buffers for now, because they are
1037 // dynamically allocated and their lifetimes are hard to compute.
1038 BufferAllocation* allocation = assignment->NewAllocation(
1039 *buffer, buffer_size, is_thread_local, /*is_reusable=*/false);
1040 VLOG(3) << "New allocation #" << allocation->index()
1041 << " for thread-local: " << *buffer;
1042 continue;
1043 }
1044
1045 if (ShapeUtil::IsTuple(buffer->shape())) {
1046 // TODO(b/34669761): Don't reuse tuple buffers because the GPU backend
1047 // assumes longer buffer liveness than indicated by the analysis.
1048 BufferAllocation* allocation = assignment->NewAllocation(
1049 *buffer, buffer_size, is_thread_local, /*is_reusable=*/false);
1050 VLOG(3) << "New allocation #" << allocation->index()
1051 << " for tuple-shaped buffer: " << *buffer;
1052 continue;
1053 }
1054
1055 // First try to assign a LogicalBuffer to one of its operand allocations to
1056 // improve locality. This is only possible with elementwise operations
1057 // (checked in liveness analysis) which are necessarily top-level
1058 // array-shaped buffers.
1059 if (buffer->IsTopLevel() && !buffer->IsTuple()) {
1060 for (auto* operand : instruction->operands()) {
1061 bool assigned_operand = false;
1062 for (const auto& operand_slice :
1063 assignment->GetAllSlices(operand, /*index=*/{})) {
1064 BufferAllocation* allocation =
1065 assignment->GetMutableAllocation(operand_slice.index());
1066 if (colocated_allocations.count(allocation->index()) == 0) {
1067 // TODO(b/32491382) Colocated buffers are currently assigned in an
1068 // earlier pass, and so can break the "increasing allocation size"
1069 // invariant in this function (causing this CHECK to fail). However,
1070 // the call to MaybeAssignBuffer is safe as it returns false if
1071 // allocation.size < buffer.size.
1072 CHECK_GE(allocation->size(), buffer_size);
1073 }
1074 if (MaybeAssignBuffer(allocation, *buffer, assignment)) {
1075 VLOG(3) << "Reusing (operand) allocation #" << allocation->index()
1076 << " for: " << *buffer;
1077 assigned_operand = true;
1078 break;
1079 }
1080 }
1081 if (assigned_operand) {
1082 break;
1083 }
1084 }
1085 }
1086
1087 if (!assignment->HasAllocation(*buffer)) {
1088 // Find the smallest buffer which can be reused iterating from end of
1089 // allocation_indices (smallest) to beginning (largest).
1090 for (int allocation_index = allocation_indices.size() - 1;
1091 allocation_index >= 0; allocation_index--) {
1092 BufferAllocation* allocation = assignment->GetMutableAllocation(
1093 allocation_indices[allocation_index]);
1094 // Instructions are iterated in increasing buffer size, so any
1095 // previously create allocation must be large enough to hold this
1096 // instruction's output (with the exception of colocated buffers).
1097 if (colocated_allocations.count(allocation->index()) == 0) {
1098 // TODO(b/32491382) Colocated buffers are currently assigned in an
1099 // earlier pass, and so can break the "increasing allocation size"
1100 // invariant in this function (causing this CHECK to fail). However,
1101 // the call to MaybeAssignBuffer is safe as it returns false if
1102 // allocation.size < buffer.size.
1103 CHECK_GE(allocation->size(), buffer_size);
1104 }
1105
1106 if (MaybeAssignBuffer(allocation, *buffer, assignment)) {
1107 VLOG(3) << "Reusing allocation #" << allocation->index()
1108 << " for: " << *buffer;
1109 break;
1110 }
1111 }
1112 }
1113
1114 if (!assignment->HasAllocation(*buffer) && has_sequential_order &&
1115 !liveness.MaybeLiveOut(*buffer)) {
1116 // There is a sequential instruction ordering, so we delay assignment of
1117 // temp buffers until after the loop. We do this right before we decide to
1118 // create a new allocation, to ensure we've exhausted all the buffer
1119 // re-use cases above.
1120 //
1121 // Entry parameters and thread local buffers were already handled earlier
1122 // in this loop iteration. See BufferAllocation::IsPreallocatedTempBuffer
1123 // for the definition of temp buffers.
1124 CHECK(!is_entry_parameter) << *buffer;
1125 CHECK(!is_thread_local) << *buffer;
1126 (*buffers_to_assign_sequentially)[computation].insert(buffer);
1127 VLOG(3) << "Delaying assignment of temp buffer: " << *buffer;
1128 continue;
1129 }
1130
1131 if (!assignment->HasAllocation(*buffer)) {
1132 BufferAllocation* allocation = assignment->NewAllocation(
1133 *buffer, buffer_size, is_thread_local, /*is_reusable=*/true);
1134 allocation_indices.push_back(allocation->index());
1135 VLOG(3) << "New allocation #" << allocation->index()
1136 << " for: " << *buffer;
1137 }
1138 }
1139
1140 return Status::OK();
1141}

1156Status BufferAssigner::AssignBuffersWithSequentialOrdering(
1157 const FlatMap<const HloComputation*, FlatSet<const LogicalBuffer*>>&
1158 buffers_to_assign_sequentially,
1159 bool run_whole_module_heap_simulation, BufferAssignment* assignment) {
1160 // Run the sequence of instructions through the heap simulator. The heuristic
1161 // that seems to give the best results is lazy-best-fit, with all runs of
1162 // alloc / free calls sorted in decreasing size order.
1163 const HloOrdering& hlo_ordering = assignment->liveness().hlo_ordering();
1164 if (run_whole_module_heap_simulation) {
1165 // Run the heap simulation over the whole module. This reduces memory usage,
1166 // since buffers for kCall, kWhile, and kConditional sub-computations are
1167 // only live for the duration of their calling instructions.
1168 VLOG(1) << "Running whole-module heap simulation";
1169 SequentialHloOrdering::HloModuleSequence module_sequence;
1170 FlatSet<const LogicalBuffer*> all_buffers_to_assign;
1171 for (const auto& pair : buffers_to_assign_sequentially) {
1172 const HloComputation* computation = pair.first;
1173 const FlatSet<const LogicalBuffer*>& buffers_to_assign = pair.second;
1174 const std::vector<const HloInstruction*>* instruction_sequence =
1175 hlo_ordering.SequentialOrder(*computation);
1176 CHECK(instruction_sequence != nullptr) << computation->name();
1177 module_sequence[computation] = *instruction_sequence;
1178 all_buffers_to_assign.insert(buffers_to_assign.begin(),
1179 buffers_to_assign.end());
1180 }
1181 auto color_map = SplitBuffersByColor(all_buffers_to_assign);
1182 for (auto& single_colored_set : color_map) {
1183 auto color = single_colored_set.first;
1184 VLOG(2) << "Simulating heap for color " << color;
1185 int64 alignment = assignment->color_alignment_(color);
1186 HeapSimulator::Options options;
1187 options.buffers_to_assign = &single_colored_set.second;
1188 TF_ASSIGN_OR_RETURN(
1189 const HeapSimulator::Result result,
1190 HeapSimulator::Run(MakeUnique<DecreasingSizeRunsHeap>(
1191 MakeUnique<LazyBestFitHeap>(alignment)),
1192 assignment->module(), module_sequence,
1193 assignment->points_to_analysis(),
1194 assignment->buffer_size_, options));
1195 AssignBuffersFromHeapSimulator(result, assignment,
1196 single_colored_set.first);
1197 }
1198 } else {
1199 // Run the heap-simulation on a per-computation basis. Buffers for
1200 // sub-computations are assigned disjoint BufferAllocations, assuming the
1201 // worst-case that they may all be live concurrently.
1202 VLOG(1) << "Running per-computation heap simulation";
1203 for (const auto& pair : buffers_to_assign_sequentially) {
1204 const HloComputation* computation = pair.first;
1205 const FlatSet<const LogicalBuffer*>& buffers_to_assign = pair.second;
1206 const std::vector<const HloInstruction*>* instruction_sequence =
1207 hlo_ordering.SequentialOrder(*computation);
1208 CHECK(instruction_sequence != nullptr) << computation->name();
1209 auto color_map = SplitBuffersByColor(buffers_to_assign);
1210 for (auto& single_colored_set : color_map) {
1211 auto color = single_colored_set.first;
1212 VLOG(2) << "Simulating heap for color " << color;
1213 int64 alignment = assignment->color_alignment_(color);
1214 HeapSimulator::Options options;
1215 options.buffers_to_assign = &single_colored_set.second;
1216 TF_ASSIGN_OR_RETURN(
1217 const HeapSimulator::Result result,
1218 HeapSimulator::Run(MakeUnique<DecreasingSizeRunsHeap>(
1219 MakeUnique<LazyBestFitHeap>(alignment)),
1220 *computation, *instruction_sequence,
1221 assignment->points_to_analysis(),
1222 assignment->buffer_size_, options));
1223 AssignBuffersFromHeapSimulator(result, assignment,
1224 single_colored_set.first);
1225 }
1226 }
1227 }
1228 return Status::OK();
1229}

1254// Adds the 'colocated_set' of buffers to 'colocated_buffer_sets', maintaining
1255// the invariant that all sets in 'colocated_buffer_sets' are disjoint.
1256//
1257// A practical example of when this is necessary is a chain of kCall ops:
1258// computation.entry
1259// %a = call() -> computation.1
1260// computation.1
1261// %b = call() -> computation.2
1262// computation.2
1263// %c = parameter()
1264// This yields the logical sets {%a,%b} {%b,%c} {%c}, which need to be merged
1265// into a single set {%a,%b,%c}
1266void BufferAssigner::AddSetToColocatedBufferSets(
1267 const std::vector<const LogicalBuffer*>& colocated_set,
1268 std::vector<ColocatedBufferSet>* colocated_buffer_sets) {
1269 if (colocated_set.empty()) {
1270 return;
1271 }
1272 VLOG(5) << ColocatedBufferSetsToString(colocated_set,
1273 "Adding colocated buffer set");
1274 // Find existing sets that overlap with at least one buffer from the
1275 // colocated_set. The resulting 'overlap_set_indices' will have at most
1276 // colocated_buffer_sets->size() entries, and will be in increasing order.
1277 std::vector<size_t> overlap_set_indices;
1278 for (size_t index = 0; index < colocated_buffer_sets->size(); ++index) {
1279 for (const LogicalBuffer* buffer : colocated_set) {
1280 if ((*colocated_buffer_sets)[index].count(buffer) > 0) {
1281 VLOG(5) << "Found overlap with existing set on buffer "
1282 << buffer->ToString() << "\n"
1283 << ColocatedBufferSetsToString((*colocated_buffer_sets)[index],
1284 "Overlapping set");
1285 overlap_set_indices.push_back(index);
1286 break;
1287 }
1288 }
1289 }
1290
1291 // If there is no overlap with existing sets, create a new set.
1292 if (overlap_set_indices.empty()) {
1293 colocated_buffer_sets->emplace_back();
1294 colocated_buffer_sets->back().insert(colocated_set.begin(),
1295 colocated_set.end());
1296 VLOG(5) << "No overlap found, new group created";
1297 return;
1298 }
1299
1300 // Merge all overlap sets and the colocated set into the first overlap set.
1301 ColocatedBufferSet* first = &(*colocated_buffer_sets)[overlap_set_indices[0]];
1302 for (size_t index = 1; index < overlap_set_indices.size(); ++index) {
1303 const ColocatedBufferSet& overlap_set =
1304 (*colocated_buffer_sets)[overlap_set_indices[index]];
1305 first->insert(overlap_set.begin(), overlap_set.end());
1306 }
1307 first->insert(colocated_set.begin(), colocated_set.end());
1308 VLOG(5) << ColocatedBufferSetsToString(
1309 *first, "Result of the colocated buffer set merging");
1310
1311 // Remove overlap sets that we just merged. The offset accounts for the fact
1312 // that as elements are erased, the indices need to be adjusted. Keep in mind
1313 // that overlap_set_indices is in increasing order.
1314 for (size_t index = 1; index < overlap_set_indices.size(); ++index) {
1315 const size_t offset = overlap_set_indices[index] - index + 1;
1316 colocated_buffer_sets->erase(colocated_buffer_sets->begin() + offset);
1317 }
1318}

1320std::vector<BufferAssigner::ColocatedBufferSet>
1321BufferAssigner::MergeColocatedBufferSets(
1322 const std::vector<ColocatedBufferSet>& colocated_buffer_sets,
1323 const BufferLiveness& buffer_liveness,
1324 const LogicalBuffer::SizeFunction& buffer_size) {
1325 VLOG(1) << "colocation sets count before coalescing:"
1326 << colocated_buffer_sets.size();
1327
1328 // Returns true if the given buffer is for the entry parameter.
1329 auto is_entry_parameter = [](const LogicalBuffer& buffer) {
1330 auto* instruction = buffer.instruction();
1331 auto* computation = instruction->parent();
1332 auto* module = computation->parent();
1333 return instruction->opcode() == HloOpcode::kParameter &&
1334 computation == module->entry_computation();
1335 };
1336
1337 // Returns true if the two colocated buffer sets (specified by their indices
1338 // into the colocated_buffer_sets) can be merged into a single set.
1339 auto cannot_merge_buffer_sets = [&colocated_buffer_sets, &buffer_liveness,
1340 &buffer_size,
1341 &is_entry_parameter](int64 i, int64 j) {
1342 // Do not merge if one of the sets includes live outs or entry parameters.
1343 for (int64 key : {i, j}) {
1344 for (auto& buffer : colocated_buffer_sets[key]) {
1345 if (buffer_liveness.MaybeLiveOut(*buffer) ||
1346 is_entry_parameter(*buffer)) {
1347 return true;
1348 }
1349 }
1350 }
1351
1352 // Colocated sets satisfy the invariant that all buffers within a set have
1353 // the same size. That means we need to check whether the size is the same
1354 // between the two sets, but also that it's enough to look at just one
1355 // buffer within each set.
1356 if (buffer_size(**colocated_buffer_sets[i].begin()) !=
1357 buffer_size(**colocated_buffer_sets[j].begin())) {
1358 return true;
1359 }
1360
1361 // Do not merge if some pair of buffers interferes with each other.
1362 for (auto& buffer_a : colocated_buffer_sets[i]) {
1363 for (auto& buffer_b : colocated_buffer_sets[j]) {
1364 if (buffer_a->id() != buffer_b->id() &&
1365 buffer_liveness.MayInterfere(*buffer_a, *buffer_b)) {
1366 return true;
1367 }
1368 }
1369 }
1370
1371 return false;
1372 };
1373
1374 // Build the interference map among the colocated buffer sets (nodes), by
1375 // adding an edge between any two nodes that cannot be merged into a single
1376 // colocated buffer set.
1377 std::vector<std::vector<int64>> interference_map(
1378 colocated_buffer_sets.size());
1379 for (int64 i = 0; i < colocated_buffer_sets.size(); ++i) {
1380 for (int64 j = i + 1; j < colocated_buffer_sets.size(); ++j) {
1381 if (cannot_merge_buffer_sets(i, j)) {
1382 interference_map[i].push_back(j);
1383 interference_map[j].push_back(i);
1384 }
1385 }
1386 }
1387
1388 // Assign a color to each colocation set in colocated_buffer_sets, such that
1389 // the sets that can be merged are assigned with the same color.
1390 auto assigned_colors = ColorInterferenceGraph(interference_map);
1391
1392 // Merge the buffer sets with the same color.
1393 CHECK(!assigned_colors.empty());
1394 int64 num_sets =
1395 *std::max_element(assigned_colors.begin(), assigned_colors.end()) + 1;
1396 std::vector<ColocatedBufferSet> new_colocated_buffer_sets(num_sets);
1397 for (int64 i = 0; i < colocated_buffer_sets.size(); ++i) {
1398 const auto& buffer_set = colocated_buffer_sets[i];
1399 new_colocated_buffer_sets[assigned_colors[i]].insert(buffer_set.begin(),
1400 buffer_set.end());
1401 }
1402
1403 VLOG(1) << "colocation sets count after coalescing:"
1404 << colocated_buffer_sets.size();
1405 return new_colocated_buffer_sets;
1406}

1408// Builds sets of buffers in 'colocated_buffer_sets' which should be colocated
1409// in the same allocation (currently just supports kWhile, kCall, and
1410// kConditional).
1411void BufferAssigner::BuildColocatedBufferSets(
1412 const HloModule* module, const BufferLiveness& buffer_liveness,
1413 const LogicalBuffer::SizeFunction& buffer_size,
1414 std::vector<ColocatedBufferSet>* colocated_buffer_sets) {
1415 const TuplePointsToAnalysis& points_to_analysis =
1416 buffer_liveness.points_to_analysis();
1417 for (const HloComputation* computation : module->MakeComputationPostOrder()) {
1418 if (computation->IsFusionComputation()) {
1419 continue;
1420 }
1421 for (const HloInstruction* instruction :
1422 computation->MakeInstructionPostOrder()) {
1423 const HloOpcode opcode = instruction->opcode();
1424 if (opcode == HloOpcode::kWhile) {
1425 const HloInstruction* while_hlo = instruction;
1426 ShapeUtil::ForEachSubshape(
1427 while_hlo->shape(),
1428 [this, while_hlo, &points_to_analysis, &buffer_liveness,
1429 buffer_size, computation, colocated_buffer_sets](
1430 const Shape& /*subshape*/, const ShapeIndex& index) {
1431 std::vector<const LogicalBuffer*> colocated_set;
1432 // Add while.init.
1433 AddBufferToColocatedSet(while_hlo->operand(0), index,
1434 points_to_analysis, &colocated_set);
1435 // Add while.result.
1436 AddBufferToColocatedSet(while_hlo, index, points_to_analysis,
1437 &colocated_set);
1438 // Add while.cond.parameter.
1439 AddBufferToColocatedSet(
1440 while_hlo->while_condition()->parameter_instruction(0), index,
1441 points_to_analysis, &colocated_set);
1442 // Add while.body.parameter.
1443 AddBufferToColocatedSet(
1444 while_hlo->while_body()->parameter_instruction(0), index,
1445 points_to_analysis, &colocated_set);
1446 // Add while.body.root.
1447 AddBufferToColocatedSet(
1448 while_hlo->while_body()->root_instruction(), index,
1449 points_to_analysis, &colocated_set);
1450 AddSetToColocatedBufferSets(colocated_set, colocated_buffer_sets);
1451 });
1452 } else if (opcode == HloOpcode::kCall) {
1453 const HloInstruction* call_hlo = instruction;
1454 const HloInstruction* root_hlo =
1455 call_hlo->to_apply()->root_instruction();
1456 ShapeUtil::ForEachSubshape(
1457 call_hlo->shape(),
1458 [this, call_hlo, root_hlo, &points_to_analysis,
1459 colocated_buffer_sets](const Shape& /*subshape*/,
1460 const ShapeIndex& index) {
1461 std::vector<const LogicalBuffer*> colocated_set;
1462 // Add call.result.
1463 AddBufferToColocatedSet(call_hlo, index, points_to_analysis,
1464 &colocated_set);
1465 // Add call.subcomputation.root.
1466 AddBufferToColocatedSet(root_hlo, index, points_to_analysis,
1467 &colocated_set);
1468 AddSetToColocatedBufferSets(colocated_set, colocated_buffer_sets);
1469 });
1470 } else if (opcode == HloOpcode::kConditional) {
1471 const HloInstruction* conditional_hlo = instruction;
1472 ShapeUtil::ForEachSubshape(
1473 conditional_hlo->shape(),
1474 [this, conditional_hlo, &points_to_analysis, colocated_buffer_sets](
1475 const Shape& /*subshape*/, const ShapeIndex& index) {
1476 std::vector<const LogicalBuffer*> colocated_set;
1477 // Add conditional.result.
1478 AddBufferToColocatedSet(conditional_hlo, index,
1479 points_to_analysis, &colocated_set);
1480 // Add conditional.true_computation.root.
1481 AddBufferToColocatedSet(
1482 conditional_hlo->true_computation()->root_instruction(),
1483 index, points_to_analysis, &colocated_set);
1484 // Add conditional.false_computation.root.
1485 AddBufferToColocatedSet(
1486 conditional_hlo->false_computation()->root_instruction(),
1487 index, points_to_analysis, &colocated_set);
1488 AddSetToColocatedBufferSets(colocated_set, colocated_buffer_sets);
1489 });
1490
1491 // Add true_operand and conditional.true_computation.parameter(0) as a
1492 // colocated buffer set. Note that this has to be done for each subshape
1493 // in the true_operand of the conditional.
1494 ShapeUtil::ForEachSubshape(
1495 conditional_hlo->operand(1)->shape(),
1496 [this, conditional_hlo, &points_to_analysis, colocated_buffer_sets](
1497 const Shape& /*subshape*/, const ShapeIndex& index) {
1498 std::vector<const LogicalBuffer*> true_set;
1499 // Add conditional.true_operand.
1500 AddBufferToColocatedSet(conditional_hlo->operand(1), index,
1501 points_to_analysis, &true_set);
1502 // Add conditional.true_computation.parameter_instruction(0).
1503 AddBufferToColocatedSet(
1504 conditional_hlo->true_computation()->parameter_instruction(0),
1505 index, points_to_analysis, &true_set);
1506 AddSetToColocatedBufferSets(true_set, colocated_buffer_sets);
1507 });
1508
1509 // Add false_operand and conditional.false_computation.parameter(0) as a
1510 // colocated buffer set. Note that this has to be done for each subshape
1511 // in the false_operand of the conditional.
1512 ShapeUtil::ForEachSubshape(
1513 conditional_hlo->operand(2)->shape(),
1514 [this, conditional_hlo, &points_to_analysis, colocated_buffer_sets](
1515 const Shape& /*subshape*/, const ShapeIndex& index) {
1516 std::vector<const LogicalBuffer*> false_set;
1517 // Add conditional.false_operand.
1518 AddBufferToColocatedSet(conditional_hlo->operand(2), index,
1519 points_to_analysis, &false_set);
1520 // Add conditional.false_computation.parameter_instruction(0).
1521 AddBufferToColocatedSet(
1522 conditional_hlo->false_computation()->parameter_instruction(
1523 0),
1524 index, points_to_analysis, &false_set);
1525 AddSetToColocatedBufferSets(false_set, colocated_buffer_sets);
1526 });
1527 }
1528 }
1529 }
1530
1531 if (colocated_buffer_sets->empty()) {
1532 return;
1533 }
1534
1535 // Try to find more coalescing opportunities among the colocated buffer sets.
1536 //
1537 // TODO(b/32491382): We should be able to remove this by using the
1538 // module-level liveness analysis, which would let us directly detect buffer
1539 // sharing opportunities between the while instruction buffer and the buffers
1540 // from the predicate and body computation, as well as sharing across
1541 // different while instructions.
1542 std::vector<ColocatedBufferSet> new_colocated_buffer_sets =
1543 MergeColocatedBufferSets(*colocated_buffer_sets, buffer_liveness,
1544 buffer_size);
1545 std::swap(*colocated_buffer_sets, new_colocated_buffer_sets);
1546}

1548// Assigns all colocated buffer sets in 'colocated_buffer_sets' to the same
1549// allocation in 'assignment'.
1550void BufferAssigner::AssignColocatedBufferSets(
1551 const std::vector<ColocatedBufferSet>& colocated_buffer_sets,
1552 BufferAssignment* assignment,
1553 FlatSet<const LogicalBuffer*>* colocated_buffers,
1554 FlatSet<BufferAllocation::Index>* colocated_allocations) {
1555 for (const ColocatedBufferSet& colocated_buffer_set : colocated_buffer_sets) {
1556 BufferAllocation* allocation = nullptr;
1557 // Set 'entry_parameter_number' and 'entry_parameter_shape_idx' if entry
1558 // param in 'colocated_buffer_set'.
1559 int64 entry_parameter_number = -1;
1560 const ShapeIndex* entry_parameter_shape_idx = nullptr;
1561 for (const LogicalBuffer* buffer : colocated_buffer_set) {
1562 const HloInstruction* instruction = buffer->instruction();
1563 const HloComputation* computation = instruction->parent();
1564 if (instruction->opcode() == HloOpcode::kParameter &&
1565 computation == computation->parent()->entry_computation()) {
1566 entry_parameter_number = instruction->parameter_number();
1567 entry_parameter_shape_idx = &buffer->index();
1568 break;
1569 }
1570 }
1571
1572 for (const LogicalBuffer* buffer : colocated_buffer_set) {
1573 const int64 buffer_size = assignment->buffer_size_(*buffer);
1574 if (allocation == nullptr) {
1575 // TODO(b/32491382) Avoid current trivial solution of using new
1576 // allocations for each colocated buffer set. When liveness has
1577 // module-level scope, we can allow buffers to be shared across
1578 // computations (in some cases).
1579 allocation = assignment->NewAllocation(*buffer, buffer_size,
1580 /*is_thread_local=*/false,
1581 /*is_reusable=*/true);
1582 if (entry_parameter_number >= 0) {
1583 // This colocated buffer set contains an entry parameter and other
1584 // logical buffers which use the parameter as read-only in a while
1585 // body computation (which updates in place).
1586 // Set 'entry_computation_parameter' to indicate that it contains
1587 // an entry parameter, and to prevent reuse in MaybeAssignBuffer.
1588 allocation->set_entry_computation_parameter(
1589 entry_parameter_number, *entry_parameter_shape_idx);
1590 }
1591 colocated_allocations->insert(allocation->index());
1592 } else {
1593 CHECK_EQ(buffer_size, allocation->size())
1594 << "Buffer: " << *buffer << " size mismatch in colocated buffer "
1595 << "allocation: " << *allocation;
1596 assignment->AddAssignment(allocation, *buffer, /*offset=*/0,
1597 buffer_size);
1598 }
1599 colocated_buffers->insert(buffer);
1600 }
1601 }

1604StatusOr<std::unique_ptr<BufferAssignment>> BufferAssigner::CreateAssignment(
1605 const HloModule* module, std::unique_ptr<HloOrdering> hlo_ordering,
1606 LogicalBuffer::SizeFunction buffer_size,
1607 LogicalBuffer::AlignmentFunction color_alignment) {
1608 TF_ASSIGN_OR_RETURN(std::unique_ptr<BufferLiveness> liveness,
1609 BufferLiveness::Run(module, std::move(hlo_ordering)));
1610
1611 VLOG(1) << "Assigning buffers to module " << module->name();
1612 XLA_VLOG_LINES(2, module->ToString());
1613 XLA_VLOG_LINES(3, liveness->ToString());
1614 XLA_VLOG_LINES(3, liveness->points_to_analysis().ToString());
1615
1616 // Can't use MakeUnique because BufferAssignment constructor is private.
1617 std::unique_ptr<BufferAssignment> assignment(
1618 new BufferAssignment(module, std::move(liveness), std::move(buffer_size),
1619 std::move(color_alignment)));
1620
1621 // Assign buffers with the tightest constraints first (colocated buffer sets).
1622 // Once b/32491382 enables module-level liveness analysis, we may be able
1623 // to assign colocated buffers (or at least reuse their allocation for
1624 // buffers outside of the set) in AssignBuffersForComputation.
1625 FlatSet<const LogicalBuffer*> colocated_buffers;
1626 FlatSet<BufferAllocation::Index> colocated_allocations;
1627 std::vector<ColocatedBufferSet> colocated_buffer_sets;
1628 BuildColocatedBufferSets(module, assignment->liveness(),
1629 assignment->buffer_size_, &colocated_buffer_sets);
1630 TF_RETURN_IF_ERROR(colorer_(assignment->liveness()));
1631 VLOG(3) << "After coloring:";
1632 XLA_VLOG_LINES(3, assignment->points_to_analysis().ToString());
1633
1634 AssignColocatedBufferSets(colocated_buffer_sets, assignment.get(),
1635 &colocated_buffers, &colocated_allocations);
1636
1637 std::vector<const HloComputation*> thread_local_computations;
1638 std::vector<const HloComputation*> global_computations;
1639 TF_RETURN_IF_ERROR(GatherComputationsByAllocationType(
1640 module, &thread_local_computations, &global_computations));
1641
1642 // First assign buffers for global computatations. Temporary buffers for
1643 // sequential computations are collected in 'buffers_to_assign_sequentially'.
1644 FlatMap<const HloComputation*, FlatSet<const LogicalBuffer*>>
1645 buffers_to_assign_sequentially;
1646 for (auto* computation : global_computations) {
1647 TF_RETURN_IF_ERROR(AssignBuffersForComputation(
1648 computation, module->config().debug_options(),
1649 /*is_thread_local=*/false, colocated_buffers, colocated_allocations,
1650 &buffers_to_assign_sequentially, assignment.get()));
1651 }
1652 // Assign buffers with sequential ordering, if any. If all global computations
1653 // are sequential, we can run heap simuation on the whole module, which
1654 // reduces memory usage.
1655 const bool run_whole_module_heap_simulation =
1656 buffers_to_assign_sequentially.size() == global_computations.size();
1657 TF_RETURN_IF_ERROR(AssignBuffersWithSequentialOrdering(
1658 buffers_to_assign_sequentially, run_whole_module_heap_simulation,
1659 assignment.get()));
1660
1661 // Now assign buffers for thread-local computations. All LogicalBuffers get
1662 // their own BufferAllocation.
1663 for (auto* computation : thread_local_computations) {
1664 TF_RET_CHECK(computation != module->entry_computation());
1665 if (computation->IsFusionComputation()) {
1666 continue;
1667 }
1668 TF_RETURN_IF_ERROR(AssignBuffersForComputation(
1669 computation, module->config().debug_options(),
1670 /*is_thread_local=*/true, colocated_buffers, colocated_allocations,
1671 /*buffers_to_assign_sequentially=*/nullptr, assignment.get()));
1672 }
1673
1674 // Mark all buffers which may be live out of the entry computation as
1675 // "liveout".
1676 for (const LogicalBuffer* buffer :
1677 assignment->liveness().maybe_live_out_buffers()) {
1678 VLOG(3) << "maybe_live_out LogicalBuffer: " << *buffer;
1679 if (assignment->HasAllocation(*buffer)) {
1680 BufferAllocation* alloc =
1681 assignment->GetMutableAssignedAllocation(*buffer);
1682 alloc->set_maybe_live_out(true);
1683 VLOG(3) << "maybe_live_out BufferAllocation: " << *alloc;
1684 }
1685 }
1686
1687 // Combines allocations of temporary buffers into one big BufferAllocation.
1688 // This can only be performed after all buffers have been assigned, and after
1689 // maybe_live_out is marked, since it is used to determine whether an
1690 // allocation contains temporary buffers or not.
1691 assignment->CombineTempAllocations();
1692
1693 XLA_VLOG_LINES(2, assignment->ToString());
1694 TF_RETURN_IF_ERROR(assignment->ComputeSummaryStats());
1695 XLA_VLOG_LINES(1, assignment->GetStats().ToString());
1696 return std::move(assignment);
1697}
