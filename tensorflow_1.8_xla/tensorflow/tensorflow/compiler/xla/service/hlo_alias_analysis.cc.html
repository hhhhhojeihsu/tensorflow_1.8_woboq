41// Data structure used to construct the alias analysis. Thrown away after alias
42// analysis is complete. This data structure keeps track of which sets of
43// HloValues must be in the same HloBuffer. This is maintained as a map from a
44// buffer identifier (BufferNumber) to set of HLoValues.
45//
46// Initially each value is its own buffer. In MergeAliasedBuffers, sets of
47// values which must share the same buffer are merged together. The end result
48// is a partitioning of all HloValues into sets where each set needs its own
49// HloBuffer. By performing this analysis without constructing HloBuffers on the
50// fly, we can after-the-fact construct a vector of contiguously numbered
51// HloBuffers after the buffer requirement has been determined.
52class BufferValueMap {
53 public:
54 // A unique identifier for a set of colocated values which must share the same
55 // buffer. This is not necessarily the same as the HloBuffer::Id which will
56 // ultimately contain the values. The reason is that HloBuffer::Id's are
57 // contiguous, while BufferNumbers may not be. BufferNumbers may not be
58 // dense because buffers may be created and destroyed during the analysis
59 // construction process.
60 using BufferNumber = int64;
61
62 explicit BufferValueMap(const HloDataflowAnalysis& dataflow)
63 : dataflow_(dataflow) {
64 buffers_.reserve(dataflow_.values().size());
65 value_to_buffer_number_.reserve(dataflow_.values().size());
66 for (const HloValue* value : dataflow_.values()) {
67 BufferNumber buffer_number = next_buffer_number_++;
68 buffers_[buffer_number].insert(value);
69 value_to_buffer_number_[value] = buffer_number;
70 }
71 }
72
73 // Merge together sets of HloValues which must be in the same HloBuffer
74 // because of aliasing rules (eg, in-place kWhile instruction).
75 void MergeAliasedBuffers() {
76 for (const HloValue* value : dataflow_.values()) {
77 VLOG(3) << "Merging colocated values, value: " << value->ToShortString();
78
79 // Gather the set of buffers with aliasing rules (eg, kWhile) which this
80 // value must be contained in.
81 std::vector<BufferNumber> aliased_buffers = ComputeAliasedBuffers(*value);
82
83 BufferNumber current_buffer = value_to_buffer_number_.at(value);
84 if (aliased_buffers.empty()) {
85 // The buffer containing 'value' aliases no other buffers. If the buffer
86 // containing 'value' already only contains 'value', then no change is
87 // necessary. If the buffer containing 'value' does contain other
88 // values, then remove 'value' from the buffer and create a new buffer
89 // containing only 'value'
90 if (buffers_.at(current_buffer).size() == 1) {
91 CHECK_EQ(*buffers_.at(current_buffer).begin(), value);
92 } else {
93 MoveValueToNewBuffer(*value);
94 }
95 } else {
96 // If multiple buffers are aliased merge these buffers together into a
97 // single buffer (arbitrarily chosen as the first buffer in the vector).
98 if (aliased_buffers.size() > 1) {
99 for (int64 i = 1; i < aliased_buffers.size(); ++i) {
100 MergeBuffers(/*from=*/aliased_buffers[i],
101 /*to=*/aliased_buffers[0]);
102 }
103 }
104 BufferNumber new_buffer = aliased_buffers[0];
105 if (current_buffer != new_buffer) {
106 MoveValueToBuffer(*value, new_buffer);
107 }
108 }
109 }
110 }
111
112 // Compute and return a sorted vector of all BufferNumbers. Can be used to
113 // iterate through all buffers stabily.
114 std::vector<BufferNumber> ComputeSortedBufferNumbers() const {
115 std::vector<BufferNumber> buffer_numbers;
116 for (const auto& pair : buffers_) {
117 buffer_numbers.push_back(pair.first);
118 }
119 std::sort(buffer_numbers.begin(), buffer_numbers.end());
120 return buffer_numbers;
121 }
122
123 // Return a set of all the values in the given buffer.
124 const tensorflow::gtl::FlatSet<const HloValue*>& GetValuesInBuffer(
125 BufferNumber buffer_number) const {
126 return buffers_.at(buffer_number);
127 }
128
129 private:
130 // Create a new buffer.
131 void NewBuffer(const HloValue& value) {
132 BufferNumber buffer_number = next_buffer_number_++;
133 buffers_[buffer_number].insert(&value);
134 value_to_buffer_number_[&value] = buffer_number;
135 }
136
137 // Move the given value into a new buffer containing only the value.
138 void MoveValueToNewBuffer(const HloValue& value) {
139 BufferNumber new_buffer_number = next_buffer_number_++;
140 buffers_[new_buffer_number];
141 MoveValueToBuffer(value, new_buffer_number);
142 }
143
144 // Move the given value into the given buffer.
145 void MoveValueToBuffer(const HloValue& value, BufferNumber buffer_number) {
146 BufferNumber old_buffer_number = value_to_buffer_number_.at(&value);
147 tensorflow::gtl::FlatSet<const HloValue*>& old_value_set =
148 buffers_.at(old_buffer_number);
149 old_value_set.erase(&value);
150 if (old_value_set.empty()) {
151 buffers_.erase(old_buffer_number);
152 }
153
154 buffers_.at(buffer_number).insert(&value);
155 value_to_buffer_number_.at(&value) = buffer_number;
156 }
157
158 // Merge the buffer 'from' into the buffer 'to'.
159 void MergeBuffers(BufferNumber from, BufferNumber to) {
160 auto& from_value_set = buffers_.at(from);
161 buffers_.at(to).insert(from_value_set.begin(), from_value_set.end());
162 // NOTE: using a union-find algorithm to hold the colocated values might be
163 // faster.
164 for (const HloValue* value : from_value_set) {
165 value_to_buffer_number_.at(value) = to;
166 }
167 buffers_.erase(from);
168 }
169
170 BufferNumber GetBufferForValue(const HloValue& value) {
171 return value_to_buffer_number_.at(&value);
172 }
173
174 void ComputeWhileAliasedBuffers(const HloValue& value,
175 std::vector<BufferNumber>* aliased_buffers) {
176 VLOG(3) << "Compute kWhile aliases";
177 // Value is init of a while (use is while).
178 for (const HloUse& use : value.uses()) {
179 if (use.instruction->opcode() == HloOpcode::kWhile) {
180 // Determine the while value that this shares a buffer with.
181 const HloValue& while_value =
182 dataflow_.GetUniqueValueAt(use.instruction, use.operand_index);
183 aliased_buffers->push_back(GetBufferForValue(while_value));
184 VLOG(3) << " value is init value to a while; must share buffer with "
185 "while value "
186 << while_value.ToShortString();
187 }
188 }
189 // Value is a parameter of a while body/condition.
190 if (value.defining_instruction()->opcode() == HloOpcode::kParameter) {
191 const HloComputation* computation =
192 value.defining_instruction()->parent();
193 const CallGraphNode& call_graph_node =
194 dataflow_.call_graph().GetNode(computation);
195 for (const CallSite& callsite : call_graph_node.caller_callsites()) {
196 if (callsite.instruction()->opcode() == HloOpcode::kWhile) {
197 // Call graph must have been flattened.
198 CHECK_EQ(call_graph_node.caller_callsites().size(), 1);
199
200 const HloValue& while_value = dataflow_.GetUniqueValueAt(
201 callsite.instruction(), value.defining_index());
202 VLOG(3) << " value is parameter value of the body or condition of a "
203 "while; must share buffer with while value "
204 << while_value.ToShortString();
205 aliased_buffers->push_back(GetBufferForValue(while_value));
206 }
207 }
208 }
209 // Value is the root of a while body.
210 for (const HloPosition& position : value.positions()) {
211 const HloComputation* computation = position.instruction->parent();
212 const CallGraphNode& call_graph_node =
213 dataflow_.call_graph().GetNode(computation);
214 if (position.instruction == computation->root_instruction()) {
215 for (const CallSite& callsite : call_graph_node.caller_callsites()) {
216 if (callsite.instruction()->opcode() == HloOpcode::kWhile &&
217 callsite.instruction()->while_body() == computation) {
218 // Call graph must have been flattened.
219 CHECK_EQ(call_graph_node.caller_callsites().size(), 1);
220
221 const HloValue& while_value = dataflow_.GetUniqueValueAt(
222 callsite.instruction(), position.index);
223 VLOG(3) << " value @ " << position << " is root of "
224 << callsite.instruction()->name()
225 << "; body root and while value root must share buffer "
226 "among them : "
227 << while_value.ToShortString();
228 aliased_buffers->push_back(GetBufferForValue(while_value));
229 }
230 }
231 }
232 }
233 // Value is the output of the while instruction itself.
234 if (value.defining_instruction()->opcode() == HloOpcode::kWhile) {
235 VLOG(3) << " value is output of a while instruction";
236 aliased_buffers->push_back(GetBufferForValue(value));
237 }
238 }
239
240 void ComputeConditionalAliasedBuffers(
241 const HloValue& value, std::vector<BufferNumber>* aliased_buffers) {
242 VLOG(3) << "Compute kConditional aliases";
243 // Aliases the buffers of the true/false computations roots, with the one of
244 // the conditional.
245 for (const HloPosition& position : value.positions()) {
246 const HloComputation* computation = position.instruction->parent();
247 const CallGraphNode& call_graph_node =
248 dataflow_.call_graph().GetNode(computation);
249 if (position.instruction == computation->root_instruction()) {
250 for (const CallSite& callsite : call_graph_node.caller_callsites()) {
251 if (callsite.instruction()->opcode() == HloOpcode::kConditional) {
252 // Call graph must have been flattened.
253 CHECK_EQ(call_graph_node.caller_callsites().size(), 1);
254
255 const HloValue& cond_value = dataflow_.GetUniqueValueAt(
256 callsite.instruction(), position.index);
257 VLOG(3)
258 << " value @ " << position << " is root of "
259 << callsite.instruction()->name()
260 << "; true/false branch roots must share buffer among them : "
261 << cond_value.ToShortString();
262 aliased_buffers->push_back(GetBufferForValue(cond_value));
263 }
264 }
265 }
266 }
267 // Value is the output of the conditional instruction itself.
268 if (value.defining_instruction()->opcode() == HloOpcode::kConditional) {
269 VLOG(3) << " value is output of a conditional instruction";
270 aliased_buffers->push_back(GetBufferForValue(value));
271 }
272 }
273
274 // Compute and return a vector of buffers that the given value must be
275 // contained in due to HLO aliasing rules.
276 std::vector<BufferNumber> ComputeAliasedBuffers(const HloValue& value) {
277 for (const HloUse& use : value.uses()) {
278 VLOG(2) << "Use of value " << value.ToShortString() << ": " << use;
279 }
280 std::vector<BufferNumber> aliased_buffers;
281 ComputeWhileAliasedBuffers(value, &aliased_buffers);
282 ComputeConditionalAliasedBuffers(value, &aliased_buffers);
283 // Uniquify aliased buffers.
284 std::sort(aliased_buffers.begin(), aliased_buffers.end());
285 aliased_buffers.erase(
286 std::unique(aliased_buffers.begin(), aliased_buffers.end()),
287 aliased_buffers.end());
288 return aliased_buffers;
289 }
290
291 // Dataflow analysis used to construct the buffer map.
292 const HloDataflowAnalysis& dataflow_;
293
294 // A map containing the set of values contained in each buffer.
295 tensorflow::gtl::FlatMap<BufferNumber,
296 tensorflow::gtl::FlatSet<const HloValue*>>
297 buffers_;
298
299 // A map indicating which buffer each value is contained in.
300 tensorflow::gtl::FlatMap<const HloValue*, BufferNumber>
301 value_to_buffer_number_;
302
303 // The buffer number of the next buffer to be created.
304 BufferNumber next_buffer_number_ = 0;
305};

309const HloBuffer& HloAliasAnalysis::GetUniqueBufferAt(
310 const HloInstruction* instruction, const ShapeIndex& index) const {
311 std::vector<const HloBuffer*> buffers = ComputeBuffersAt(instruction, index);
312 CHECK_EQ(buffers.size(), 1);
313 return *buffers[0];

322
323std::vector<const HloBuffer*> HloAliasAnalysis::ComputeBuffersAt(
324 const HloInstruction* instruction, const ShapeIndex& index) const {
325 std::vector<const HloBuffer*> buffers;
326 for (const HloValue* value :
327 dataflow_analysis_->GetValueSet(instruction, index).values()) {
328 buffers.push_back(&GetBufferContainingValue(*value));
329 }
330
331 // Sort and uniquify vector before returning.
332 std::sort(buffers.begin(), buffers.end(), HloBuffer::IdLessThan);
333 buffers.erase(std::unique(buffers.begin(), buffers.end()), buffers.end());
334
335 return buffers;
336}

338bool HloAliasAnalysis::InstructionBuffersAreAmbiguous(
339 const HloInstruction* instruction) const {
340 for (const auto& pair :
341 dataflow_analysis_->GetInstructionValueSet(instruction)) {
342 const HloValueSet& value_set = pair.second;
343 const HloBuffer* buffer = nullptr;
344 for (const HloValue* value : value_set.values()) {
345 if (buffer == nullptr) {
346 buffer = &GetBufferContainingValue(*value);
347 } else if (buffer != &GetBufferContainingValue(*value)) {
348 return true;
349 }
350 }
351 }
352 return false;
353}
354
355bool HloAliasAnalysis::InstructionBuffersAreDistinct(
356 const HloInstruction* instruction) const {
357 tensorflow::gtl::FlatSet<const HloBuffer*> buffers_seen;
358 for (const auto& pair :
359 dataflow_analysis_->GetInstructionValueSet(instruction)) {
360 const HloValueSet& value_set = pair.second;
361 if (value_set.values().size() == 1) {
362 if (!buffers_seen
363 .insert(&GetBufferContainingValue(value_set.GetUniqueValue()))
364 .second) {
365 return false;
366 }
367 } else {
368 // It's possible for multiple values at this index to have the same
369 // HloBuffer. This does not result in non-distictness. To account for
370 // this case, add all of the buffers at this index after checking
371 // whether each buffer exists at an earlier index. This is a corner
372 // case, however, as the number of values at an index is almost always
373 // one.
374 std::vector<const HloBuffer*> buffers_at_this_index;
375 for (const HloValue* value : value_set.values()) {
376 const HloBuffer* buffer = &GetBufferContainingValue(*value);
377 if (ContainsKey(buffers_seen, buffer)) {
378 return false;
379 }
380 buffers_at_this_index.push_back(buffer);
381 }
382 buffers_seen.insert(buffers_at_this_index.begin(),
383 buffers_at_this_index.end());
384 }
385 }
386 return true;
387}

389Status HloAliasAnalysis::Verify() const {
390 // Verify consistency between the value_to_buffer_ map and
391 // HloBuffer::values().
392 for (const auto& pair : value_to_buffer_) {
393 const HloValue* value = pair.first;
394 const HloBuffer& buffer = *pair.second;
395 TF_RET_CHECK(std::find(buffer.values().begin(), buffer.values().end(),
396 value) != buffer.values().end());
397 }
398
399 for (HloBuffer::Id id = 0; id < buffers_.size(); ++id) {
400 const HloBuffer& buffer = buffers_[id];
401 TF_RET_CHECK(buffer.id() == id);
402
403 HloValue::Id last_value_id = -1;
404 for (const HloValue* value : buffer.values()) {
405 TF_RET_CHECK(GetBufferContainingValue(*value) == buffer);
406
407 // Also verify the values in HloBuffer are unique and sorted by id.
408 TF_RET_CHECK(value->id() > last_value_id);
409 last_value_id = value->id();
410 }
411 }
412
413 return Status::OK();
414}

416string HloAliasAnalysis::ToString() const {
417 string out = StrCat("HloAliasAnalysis, module ", module_->name(), "\n");
418 StrAppend(&out, " Buffers at each position:\n");
419 for (const HloComputation* computation : module_->computations()) {
420 for (const HloInstruction* instruction : computation->instructions()) {
421 StrAppend(&out, " ", instruction->name(), ":\n");
422 if (ShapeUtil::IsTuple(instruction->shape())) {
423 ShapeUtil::ForEachSubshape(
424 instruction->shape(),
425 [&out, &instruction, this](const Shape&, const ShapeIndex& index) {
426 StrAppend(&out, " tuple index ", index.ToString(), ":\n");
427 for (const HloBuffer* buffer :
428 ComputeBuffersAt(instruction, index)) {
429 StrAppend(&out, " ", buffer->ToString(), "\n");
430 }
431 });
432 } else {
433 for (const HloBuffer* buffer :
434 ComputeBuffersAt(instruction, /*index=*/{})) {
435 StrAppend(&out, " ", buffer->ToString(), "\n");
436 }
437 }
438 }
439 }
440
441 StrAppend(&out, " Buffers:\n");
442 for (const HloBuffer& buffer : buffers()) {
443 StrAppend(&out, " ", buffer.ToString(), "\n");
444 StrAppend(&out, " positions:\n");
445 for (const HloPosition& position : buffer.ComputePositions()) {
446 StrAppend(&out, " ", position.ToString(), "\n");
447 }
448 }
449
450 return out;
451}

453/* static */
454StatusOr<std::unique_ptr<HloAliasAnalysis>> HloAliasAnalysis::Run(
455 HloModule* module) {
456 VLOG(2) << "HloAliasAnalysis::Run on module " << module->name();
457 XLA_VLOG_LINES(2, module->ToString());
458
459 auto alias_analysis = WrapUnique(new HloAliasAnalysis(module));
460 TF_ASSIGN_OR_RETURN(
461 alias_analysis->dataflow_analysis_,
462 HloDataflowAnalysis::Run(*module, /*ssa_form=*/true,
463 /*bitcast_defines_value=*/false));
464
465 BufferValueMap buffer_map(alias_analysis->dataflow_analysis());
466 buffer_map.MergeAliasedBuffers();
467
468 // Create a vector of HloBuffers, one for each set of values in the
469 // BufferValueMap. Create the HloBuffers as a vector of contiguously numbered
470 // buffers.
471 std::vector<BufferValueMap::BufferNumber> sorted_buffer_numbers =
472 buffer_map.ComputeSortedBufferNumbers();
473 alias_analysis->buffers_.reserve(sorted_buffer_numbers.size());
474 HloBuffer::Id next_id = 0;
475 for (BufferValueMap::BufferNumber buffer_number : sorted_buffer_numbers) {
476 auto& value_set = buffer_map.GetValuesInBuffer(buffer_number);
477 std::vector<const HloValue*> sorted_values(value_set.begin(),
478 value_set.end());
479 std::sort(sorted_values.begin(), sorted_values.end(), HloValue::IdLessThan);
480 alias_analysis->buffers_.emplace_back(next_id++, sorted_values);
481 for (const HloValue* value : sorted_values) {
482 alias_analysis->value_to_buffer_[value] =
483 &alias_analysis->buffers_.back();
484 }
485 }
486
487 TF_DCHECK_OK(alias_analysis->Verify());
488
489 XLA_VLOG_LINES(2, alias_analysis->ToString());
490 return std::move(alias_analysis);
491}

493bool HloAliasAnalysis::HasLiveRangeInterference(
494 const HloOrdering& ordering) const {
495 for (const HloBuffer& buffer : buffers()) {
496 // Check that the values in the buffer are totally ordered with respect to
497 // 'ordering'. Begin by sorting the values with respect to 'ordering' with a
498 // tie-break using value ID. The tie-break is necessary because we need a
499 // strict weak order for std::sort.
500 std::vector<const HloValue*> values = buffer.values();
501 std::sort(values.begin(), values.end(),
502 [&ordering](const HloValue* a, const HloValue* b) {
503 if (ordering.IsDefinedBefore(*a, *b)) {
504 return true;
505 } else if (ordering.IsDefinedBefore(*b, *a)) {
506 return false;
507 } else {
508 return a->id() < b->id();
509 }
510 });
511
512 // Walk through the ordered vector of values. First verify that the values
513 // are totally ordered with respect to 'ordering', then check that no
514 // adjacent values have overlapping live ranges. Only adjacent values must
515 // be checked because of the property of live range interference. For
516 // example, if you have values A, B, and C (in program order) contained in
517 // a buffer and A interferes with C, then necessarily A also interferes
518 // with B. So to check interference you only need to check interference
519 // between A and B, and between B and C.
520 CHECK(!values.empty());
521 for (int i = 1; i < values.size(); ++i) {
522 if (!ordering.IsDefinedBefore(*values[i - 1], *values[i])) {
523 VLOG(1) << values[i - 1]->ToShortString() << " and "
524 << values[i]->ToShortString() << " are not ordered";
525 return true;
526 }
527 if (ordering.MayInterfere(*values[i - 1], *values[i],
528 dataflow_analysis())) {
529 VLOG(1) << "In buffer " << buffer.id() << " containing values:\n "
530 << Join(values, ", ",
531 [](string* out, const HloValue* value) {
532 StrAppend(out, value->ToShortString());
533 })
534
535 << "\nValue " << values[i - 1]->ToShortString()
536 << " may interfere with value " << values[i]->ToShortString();
537 return true;
538 }
539 }
540 }
541
542 return false;
543}
