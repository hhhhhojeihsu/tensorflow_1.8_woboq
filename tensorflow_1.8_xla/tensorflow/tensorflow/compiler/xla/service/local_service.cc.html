50/* static */ StatusOr<std::unique_ptr<LocalService>> LocalService::NewService(
51 const ServiceOptions& options) {
52 perftools::gputools::Platform* platform = options.platform();
53 if (platform == nullptr) {
54 TF_ASSIGN_OR_RETURN(platform, PlatformUtil::GetDefaultPlatform());
55 }
56
57 BackendOptions backend_options;
58 backend_options.set_platform(platform).set_intra_op_parallelism_threads(
59 options.intra_op_parallelism_threads());
60 TF_ASSIGN_OR_RETURN(std::unique_ptr<Backend> backend,
61 Backend::CreateBackend(backend_options));
62
63 std::unique_ptr<LocalService> service(
64 new LocalService(options, std::move(backend)));
65 return std::move(service);
66}

74// Retrieves the parameter metadata for the given computation and parameter
75// number.
76//
77// If the parameter number is invalid for this computation, nullopt is
78// returned. When the return value has_value(), nullptr will never be
79// the held value.
80tensorflow::gtl::optional<const OpMetadata*> ParameterMetadata(
81 const XlaComputation& computation, int parameter_number) {
82 for (const HloComputationProto& comp : computation.proto().computations()) {
83 if (comp.id() == computation.proto().entry_computation_id()) {
84 for (const HloInstructionProto& instr : comp.instructions()) {
85 if (instr.opcode() == HloOpcodeString(HloOpcode::kParameter) &&
86 instr.parameter_number() == parameter_number) {
87 if (!instr.has_metadata()) {
88 return tensorflow::gtl::nullopt;
89 }
90 return &instr.metadata();
91 }
92 }
93 }
94 }
95 return tensorflow::gtl::nullopt;
96}

98ExecutionOptions CreateExecutionOptions(
99 const ExecutableBuildOptions& build_options,
100 const ProgramShape* program_shape) {
101 ExecutionOptions execution_options = CreateDefaultExecutionOptions();
102 if (build_options.hlo_profile().has_value()) {
103 execution_options.mutable_debug_options()->set_xla_hlo_profile(
104 *build_options.hlo_profile());
105 }
106 if (build_options.generate_hlo_graph().has_value()) {
107 execution_options.mutable_debug_options()->set_xla_generate_hlo_graph(
108 build_options.generate_hlo_graph().value());
109 }
110 if (build_options.dump_optimized_hlo_proto_to().has_value()) {
111 execution_options.mutable_debug_options()
112 ->set_xla_dump_optimized_hlo_proto_to(
113 build_options.dump_optimized_hlo_proto_to().value());
114 }
115 if (build_options.dump_per_pass_hlo_proto_to().has_value()) {
116 execution_options.mutable_debug_options()
117 ->set_xla_dump_per_pass_hlo_proto_to(
118 build_options.dump_per_pass_hlo_proto_to().value());
119 }
120 if (build_options.result_layout() != nullptr) {
121 *execution_options.mutable_shape_with_output_layout() =
122 *build_options.result_layout();
123 } else {
124 *execution_options.mutable_shape_with_output_layout() =
125 program_shape->result();
126 LayoutUtil::SetToDefaultLayout(
127 execution_options.mutable_shape_with_output_layout());
128 }
129 return execution_options;
130}

134StatusOr<std::unique_ptr<Executable>> LocalService::CompileExecutable(
135 const ComputationHandle& computation,
136 const tensorflow::gtl::ArraySlice<const Shape*> argument_layouts,
137 const ExecutableBuildOptions& build_options) {
138 TF_ASSIGN_OR_RETURN(UserComputation * user_computation,
139 computation_tracker_.Resolve(computation));
140 VersionedComputationHandle versioned_handle =
141 user_computation->GetVersionedHandle();
142
143 TF_ASSIGN_OR_RETURN(
144 std::shared_ptr<const ProgramShape> program_shape,
145 user_computation->ComputeProgramShape(versioned_handle.version));
146
147 // Validate incoming layouts.
148 if (argument_layouts.size() != program_shape->parameters_size()) {
149 return InvalidArgument(
150 "Invalid number of arguments for computation: expected %d, got %zu.",
151 program_shape->parameters_size(), argument_layouts.size());
152 }
153 for (int i = 0; i < argument_layouts.size(); ++i) {
154 const Shape& argument_shape = *argument_layouts[i];
155 TF_RETURN_IF_ERROR(ShapeUtil::ValidateShape(argument_shape));
156 if (!ShapeUtil::Compatible(argument_shape, program_shape->parameters(i))) {
157 tensorflow::gtl::optional<const OpMetadata*> metadata =
158 user_computation->ParameterMetadata(i);
159 auto metadata_string = [&metadata]() -> string {
160 if (!metadata.has_value()) {
161 return "";
162 }
163 CHECK(metadata.value() != nullptr);
164 const OpMetadata& m = *metadata.value();
165 if (!m.source_file().empty()) {
166 return tensorflow::strings::Printf(
167 " (%s:%d)", m.source_file().c_str(), m.source_line());
168 }
169 return "";
170 };
171 return InvalidArgument(
172 "Invalid argument shape for argument %d%s, expected %s, got %s.", i,
173 metadata_string().c_str(),
174 ShapeUtil::HumanString(program_shape->parameters(i)).c_str(),
175 ShapeUtil::HumanString(argument_shape).c_str());
176 }
177 }
178 if (build_options.result_layout() != nullptr) {
179 TF_RETURN_IF_ERROR(ValidateResultShapeWithLayout(
180 *build_options.result_layout(), program_shape->result()));
181 }
182
183 ExecutionOptions execution_options =
184 CreateExecutionOptions(build_options, program_shape.get());
185 TF_ASSIGN_OR_RETURN(std::unique_ptr<HloModuleConfig> module_config,
186 CreateModuleConfig(*program_shape, argument_layouts,
187 &execution_options, user_computation));
188
189 TF_ASSIGN_OR_RETURN(
190 se::StreamExecutor * executor,
191 execute_backend_->stream_executor(build_options.device_ordinal()));
192
193 return BuildExecutable(versioned_handle, std::move(module_config),
194 execute_backend_.get(), executor,
195 build_options.device_allocator());
196}

198StatusOr<std::unique_ptr<Executable>> LocalService::CompileExecutable(
199 const XlaComputation& computation,
200 const tensorflow::gtl::ArraySlice<const Shape*> argument_layouts,
201 const ExecutableBuildOptions& build_options) {
202 const HloModuleProto& proto = computation.proto();
203 TF_RET_CHECK(proto.has_program_shape());
204 const ProgramShape& program_shape = proto.program_shape();
205
206 // Validate incoming layouts.
207 if (argument_layouts.size() != program_shape.parameters_size()) {
208 return InvalidArgument(
209 "Invalid number of arguments for computation: expected %d, got %zu.",
210 program_shape.parameters_size(), argument_layouts.size());
211 }
212
213 for (int i = 0; i < argument_layouts.size(); ++i) {
214 const Shape& argument_shape = *argument_layouts[i];
215 TF_RETURN_IF_ERROR(ShapeUtil::ValidateShape(argument_shape));
216 if (!ShapeUtil::Compatible(argument_shape, program_shape.parameters(i))) {
217 tensorflow::gtl::optional<const OpMetadata*> metadata =
218 ParameterMetadata(computation, /*parameter_number=*/i);
219 auto metadata_string = [&metadata]() -> string {
220 if (!metadata.has_value()) {
221 return "";
222 }
223 CHECK(metadata.value() != nullptr);
224 const OpMetadata& m = *metadata.value();
225 if (!m.source_file().empty()) {
226 return tensorflow::strings::Printf(
227 " (%s:%d)", m.source_file().c_str(), m.source_line());
228 }
229 return "";
230 };
231 return InvalidArgument(
232 "Invalid argument shape for argument %d%s, expected %s, got %s.", i,
233 metadata_string().c_str(),
234 ShapeUtil::HumanString(program_shape.parameters(i)).c_str(),
235 ShapeUtil::HumanString(argument_shape).c_str());
236 }
237 }
238 if (build_options.result_layout() != nullptr) {
239 TF_RETURN_IF_ERROR(ValidateResultShapeWithLayout(
240 *build_options.result_layout(), program_shape.result()));
241 }
242
243 ExecutionOptions execution_options =
244 CreateExecutionOptions(build_options, &program_shape);
245
246 TF_ASSIGN_OR_RETURN(
247 std::unique_ptr<HloModuleConfig> module_config,
248 CreateModuleConfig(program_shape, argument_layouts, &execution_options));
249
250 TF_ASSIGN_OR_RETURN(
251 se::StreamExecutor * executor,
252 execute_backend_->stream_executor(build_options.device_ordinal()));
253
254 return BuildExecutable(proto, std::move(module_config),
255 execute_backend_.get(), executor,
256 build_options.device_allocator());
257}

259StatusOr<int> LocalService::ReplicaNumberToDeviceOrdinal(int replica_number) {
260 return backend().computation_placer()->DeviceId(
261 replica_number, /*computation=*/0, options_.number_of_replicas(),
262 /*computation_count=*/1);
