25namespace xla {
26
27// HLO pass which removes dead instructions from each computation in the module
28// and removes dead computations from the module.
29//
30// An instruction is dead if it is not reachable from the root. A computation is
31// dead if it is not the entry computation of the module and it is not reachable
32// from the entry computation.
33//
34// This pass does not remove dead parameter instructions, as parameter
35// instructions cannot be deleted.
36class HloDCE : public HloPassInterface {
37 public:
38 ~HloDCE() override {}
39 tensorflow::StringPiece name() const override { return "dce"; }
40
41 // Run the pass on the given module. Returns whether the module was changed
42 // (instructions were removed).
43 StatusOr<bool> Run(HloModule* module) override;
44};
45
46} // namespace xla
