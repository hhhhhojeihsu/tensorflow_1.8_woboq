32StatusOr<bool> ParallelizationPreparation::Run(HloModule* module) {
33 XLA_VLOG_LINES(2, "ParallelizationPreparation ENTRY");
34 XLA_VLOG_LINES(2, module->ToString());
35
36 bool changed = false;
37 TF_ASSIGN_OR_RETURN(changed, RunParallelTaskAssignment(module));
38
39 HloComputation* entry_computation = module->entry_computation();
40 std::unordered_set<HloInstruction*> outlined;
41 std::vector<HloInstruction*> instructions_to_outline;
42 for (HloInstruction* instruction :
43 entry_computation->MakeInstructionPostOrder()) {
44 // If the instruction has been outlined, it no longer exists and we must not
45 // dereference it.
46 if (outlined.count(instruction) > 0) {
47 continue;
48 }
49
50 // Skip parameters and constants, there is nothing to parallelize.
51 if (instruction->opcode() == HloOpcode::kParameter ||
52 instruction->opcode() == HloOpcode::kConstant) {
53 continue;
54 }
55
56 // Outline 'instruction' in isolation if it was assigned parallel tasks.
57 if (OutlineParallelizableInstruction(instruction)) {
58 outlined.insert(instruction);
59 changed = true;
60 continue;
61 }
62
63 instructions_to_outline.clear();
64 HloInstruction* outline_candidate = instruction;
65 instructions_to_outline.push_back(outline_candidate);
66
67 // Outline sole users with the current instruction.
68 while (CanOutlineWithUser(outline_candidate)) {
69 HloInstruction* prior_candidate = outline_candidate;
70 outline_candidate = *outline_candidate->users().begin();
71 if (std::any_of(outline_candidate->operands().begin(),
72 outline_candidate->operands().end(),
73 [&](const HloInstruction* operand) {
74 // Do not consider any candidates which have operands
75 // other than the prior candidate, constants or
76 // parameters. Otherwise, we'd increase the fan-in which
77 // would reduce parallelism.
78 return operand->opcode() != HloOpcode::kParameter &&
79 operand->opcode() != HloOpcode::kConstant &&
80 operand != prior_candidate;
81 })) {
82 break;
83 }
84 instructions_to_outline.push_back(outline_candidate);
85 }
86
87 outlined.insert(instructions_to_outline.begin(),
88 instructions_to_outline.end());
89
90 // Optimization to avoid replacing a single existing kCall with another
91 // kCall that just calls the first one.
92 if (instructions_to_outline.size() == 1 &&
93 instructions_to_outline[0]->opcode() == HloOpcode::kCall) {
94 continue;
95 }
96
97 module->OutlineExpressionFromComputation(
98 instructions_to_outline,
99 tensorflow::strings::StrCat("pp_", instruction->name()),
100 entry_computation);
101 changed = true;
102 }
103
104 XLA_VLOG_LINES(2, "ParallelizationPreparation EXIT");
105 XLA_VLOG_LINES(2, module->ToString());
106 return changed;
107}

109StatusOr<bool> ParallelizationPreparation::RunParallelTaskAssignment(
110 HloModule* module) {
111 VLOG(1) << "RunParallelTaskAssignment max_parallelism_: " << max_parallelism_;
112 bool changed = false;
113 // Initialize ParallelTaskAssignment.
114 ParallelTaskAssignment parallel_task_assignment(max_parallelism_, shape_size_,
115 module);
116 // Assign parallel tasks to HLOs in entry computation.
117 HloComputation* computation = module->entry_computation();
118 for (auto* instruction : computation->instructions()) {
119 // Calculate target parallel task count in [1, max_parallelism_].
120 const int64 target_parallel_task_count =
121 parallel_task_assignment.GetTargetParallelTaskCount(instruction);
122 if (target_parallel_task_count == 1) {
123 continue;
124 }
125
126 // Assign feasible dimension partitions (based on actual dimension sizes).
127 auto dim_partition_counts = ShapePartitionAssigner(instruction->shape())
128 .Run(target_parallel_task_count);
129 const int64 total_partition_count =
130 ShapePartitionAssigner::GetTotalPartitionCount(dim_partition_counts);
131 if (total_partition_count <= 1) {
132 // Feasible partition calculation resulting in no partitioning, so skip.
133 continue;
134 }
135 VLOG(2) << "Assigning parallel task count: " << total_partition_count
136 << " to instruction: " << instruction->name();
137 // Map 'instruction' to assigned dimension partitioning.
138 instruction->set_outer_dimension_partitions(dim_partition_counts);
139 }
140
141 return changed;
142}

144bool ParallelizationPreparation::OutlineParallelizableInstruction(
145 HloInstruction* instruction) {
146 if (instruction->outer_dimension_partitions().empty()) {
147 return false;
148 }
149 // Store dimension partition counts before outlining (which clones
150 // 'instruction').
151 std::vector<int64> dim_partition_counts =
152 instruction->outer_dimension_partitions();
153 // Outline 'instruction' in its own sub-computation.
154 HloModule* module = instruction->parent()->parent();
155 auto* call = module->OutlineExpressionFromComputation(
156 {instruction}, tensorflow::strings::StrCat("pp_", instruction->name()),
157 module->entry_computation());
158 // Map previously assigned 'dim_partition_counts' to cloned root instruction.
159 VLOG(1) << "Outlining parallelizable"
160 << " caller: " << call->name()
161 << " callee: " << call->to_apply()->root_instruction()->name();
162 call->to_apply()->root_instruction()->set_outer_dimension_partitions(
163 dim_partition_counts);
164 return true;
165}
166
167bool ParallelizationPreparation::CanOutlineWithUser(
168 HloInstruction* instruction) {
169 if (instruction->users().size() != 1) {
170 // Do not outline 'instruction' with multiple users.
171 return false;
172 }
173 if (AssignedParallelTasks(instruction) ||
174 AssignedParallelTasks(*instruction->users().begin())) {
175 // Do not outline if 'instruction' (or user) were assigned parallel tasks.
176 return false;
177 }
178 return true;
179}
180
181bool ParallelizationPreparation::AssignedParallelTasks(
182 HloInstruction* instruction) {
183 return !instruction->outer_dimension_partitions().empty() ||
184 (instruction->opcode() == HloOpcode::kCall &&
185 !instruction->to_apply()
186 ->root_instruction()
187 ->outer_dimension_partitions()
188 .empty());
189}
