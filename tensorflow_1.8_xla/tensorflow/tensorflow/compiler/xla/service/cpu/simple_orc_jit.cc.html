49llvm::SmallVector<std::string, 0> DetectMachineAttributes() {
50 llvm::SmallVector<std::string, 0> result;
51 llvm::StringMap<bool> host_features;
52 if (llvm::sys::getHostCPUFeatures(host_features)) {
53 for (auto& feature : host_features) {
54 if (feature.second) {
55 llvm::StringRef feature_name = feature.first();
56 // Skip avx512 for now, it isn't quite ready in LLVM.
57 if (feature_name.startswith("avx512")) {
58 continue;
59 }
60 result.push_back(feature_name);
61 }
62 }
63 }
64 return result;
65}
66
67llvm::StringRef GetHostCpuName() {
68 auto cpu_name = llvm::sys::getHostCPUName();
69 // Skip avx512 for now, it isn't quite ready in LLVM.
70 cpu_name.consume_back("-avx512");
71 return cpu_name;
72}

75SimpleOrcJIT::SimpleOrcJIT(const llvm::TargetOptions& target_options,
76 llvm::CodeGenOpt::Level opt_level,
77 bool optimize_for_size, bool enable_fast_math,
78 bool disable_expensive_passes,
79 LLVMCompiler::ModuleHook pre_optimization_hook,
80 LLVMCompiler::ModuleHook post_optimization_hook)
81 : target_machine_(
82 CHECK_NOTNULL(llvm::EngineBuilder()
83 .setTargetOptions(target_options)
84 .setOptLevel(opt_level)
85 .selectTarget(
86 /*TargetTriple=*/llvm::Triple(), /*MArch=*/"",
87 /*MCPU=*/GetHostCpuName(),
88 /*MAttrs=*/DetectMachineAttributes()))),
89 disassembler_(*target_machine_),
90 data_layout_(target_machine_->createDataLayout()),
91 symbol_resolver_(llvm::orc::createLegacyLookupResolver(
92 [this](const std::string& name) -> llvm::JITSymbol {
93 return this->ResolveRuntimeSymbol(name);
94 },
95 [](llvm::Error Err) {
96 cantFail(std::move(Err), "lookupFlags failed");
97 })),
98 object_layer_(execution_session_,
99 [this](llvm::orc::VModuleKey) {
100 llvm::orc::RTDyldObjectLinkingLayer::Resources result;
101 result.MemMgr =
102 std::make_shared<llvm::SectionMemoryManager>(
103 orc_jit_memory_mapper::GetInstance());
104 result.Resolver = symbol_resolver_;
105 return result;
106 }),
107 compile_layer_(object_layer_,
108 CompilerFunctor(target_machine_.get(), &disassembler_,
109 opt_level, optimize_for_size,
110 enable_fast_math, disable_expensive_passes,
111 std::move(pre_optimization_hook),
112 std::move(post_optimization_hook))) {
113 VLOG(1) << "CPU target: " << target_machine_->getTargetCPU().str()
114 << " features: " << target_machine_->getTargetFeatureString().str();
115}

117llvm::JITSymbol SimpleOrcJIT::ResolveRuntimeSymbol(const std::string& name) {
118 if (const uint8* from_constant_pool =
119 external_constant_pool_.Find(string(name))) {
120 return llvm::JITEvaluatedSymbol(
121 reinterpret_cast<uint64_t>(from_constant_pool),
122 llvm::JITSymbolFlags::None);
123 }
124
125 void* func_addr = CustomCallTargetRegistry::Global()->Lookup(name);
126 if (func_addr == nullptr) {
127 return nullptr;
128 }
129 llvm::JITEvaluatedSymbol symbol_info(reinterpret_cast<uint64_t>(func_addr),
130 llvm::JITSymbolFlags::None);
131 return symbol_info;
132}
133
134SimpleOrcJIT::VModuleKeyT SimpleOrcJIT::AddModule(
135 std::unique_ptr<llvm::Module> module) {
136 auto key = execution_session_.allocateVModule();
137 cantFail(compile_layer_.addModule(key, std::move(module)));
138 module_keys_.push_back(key);
139 return key;
140}
141
142void SimpleOrcJIT::RemoveModule(SimpleOrcJIT::VModuleKeyT key) {
143 module_keys_.erase(std::remove(module_keys_.begin(), module_keys_.end(), key),
144 module_keys_.end());
145 cantFail(compile_layer_.removeModule(key));
146}
147
148llvm::JITSymbol SimpleOrcJIT::FindCompiledSymbol(const std::string& name) {
149 // Resolve symbol from last module to first, allowing later redefinitions of
150 // symbols shadow earlier ones.
151 for (auto& key :
152 llvm::make_range(module_keys_.rbegin(), module_keys_.rend())) {
153 if (auto symbol =
154 compile_layer_.findSymbolIn(key, name,
155 /*ExportedSymbolsOnly=*/true)) {
156 return symbol;
157 }
158 }
159
160 return nullptr;
161}

164// Register some known symbols with the CustomCallTargetRegistry.
165bool RegisterKnownJITSymbols() {
166 CustomCallTargetRegistry* registry = CustomCallTargetRegistry::Global();
167
168#define REGISTER_CPU_RUNTIME_SYMBOL(base_name) \
169 do { \
170 auto* function_address = \
171 reinterpret_cast<void*>(__xla_cpu_runtime_##base_name); \
172 registry->Register(xla::cpu::runtime::k##base_name##SymbolName, \
173 function_address); \
174 CHECK_EQ( \
175 tensorflow::StringPiece(xla::cpu::runtime::k##base_name##SymbolName), \
176 "__xla_cpu_runtime_" #base_name); \
177 } while (false)
178
179 REGISTER_CPU_RUNTIME_SYMBOL(AcquireInfeedBufferForDequeue);
180 REGISTER_CPU_RUNTIME_SYMBOL(AcquireOutfeedBufferForPopulation);
181 REGISTER_CPU_RUNTIME_SYMBOL(EigenConvF16);
182 REGISTER_CPU_RUNTIME_SYMBOL(EigenConvF32);
183 REGISTER_CPU_RUNTIME_SYMBOL(EigenFft);
184 REGISTER_CPU_RUNTIME_SYMBOL(EigenMatMulF16);

186 REGISTER_CPU_RUNTIME_SYMBOL(EigenMatMulF64);
187 REGISTER_CPU_RUNTIME_SYMBOL(MKLMatMulF32);
188 REGISTER_CPU_RUNTIME_SYMBOL(MKLMatMulF64);
189 REGISTER_CPU_RUNTIME_SYMBOL(MKLSingleThreadedMatMulF32);
190 REGISTER_CPU_RUNTIME_SYMBOL(MKLSingleThreadedMatMulF64);
191 REGISTER_CPU_RUNTIME_SYMBOL(EigenSingleThreadedConvF16);
192 REGISTER_CPU_RUNTIME_SYMBOL(EigenSingleThreadedConvF32);

200 registry->Register("__gnu_f2h_ieee", reinterpret_cast<void*>(__gnu_f2h_ieee));

203#undef REGISTER_CPU_RUNTIME_SYMBOL

205// Register both the f32 (float) and f64 (double) versions of a libm symbol.
206// Unfortunately the double versions are overloaded on some systems, e.g.
207// Mac so we need an explicit cast. This requires passing the function signature
208// for that case.
209#define REGISTER_LIBM_SYMBOL(name, double_sig) \
210 do { \
211 registry->Register(#name "f", reinterpret_cast<void*>(name##f)); \
212 registry->Register( \
213 #name, reinterpret_cast<void*>(static_cast<double_sig>(name))); \
214 } while (false)
215

233 REGISTER_LIBM_SYMBOL(fabs, double (*)(double));

238 REGISTER_LIBM_SYMBOL(fmin, double (*)(double, double));

275 REGISTER_LIBM_SYMBOL(tan, double (*)(double));
276 REGISTER_LIBM_SYMBOL(tanh, double (*)(double));

282 registry->Register("memcpy", reinterpret_cast<void*>(memcpy));
283 registry->Register("memmove", reinterpret_cast<void*>(memmove));
284 registry->Register("memset", reinterpret_cast<void*>(memset));
285 return true;
