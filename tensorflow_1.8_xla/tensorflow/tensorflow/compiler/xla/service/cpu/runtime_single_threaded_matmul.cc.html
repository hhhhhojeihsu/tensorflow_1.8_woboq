27template <typename T>
28void MatMul(const void* run_options_ptr, T* out, T* lhs, T* rhs, int64 m,
29 int64 n, int64 k, int32 transpose_lhs, int32 transpose_rhs) {
30 int64 lhs_rows = m;
31 int64 lhs_cols = k;
32 if (transpose_lhs) {
33 std::swap(lhs_rows, lhs_cols);
34 }
35
36 int64 rhs_rows = k;
37 int64 rhs_cols = n;
38 if (transpose_rhs) {
39 std::swap(rhs_rows, rhs_cols);
40 }
41
42 const Eigen::TensorMap<Eigen::Tensor<const T, 2>, Eigen::Aligned> A(
43 lhs, lhs_rows, lhs_cols);
44 const Eigen::TensorMap<Eigen::Tensor<const T, 2>, Eigen::Aligned> B(
45 rhs, rhs_rows, rhs_cols);
46 Eigen::TensorMap<Eigen::Tensor<T, 2>, Eigen::Aligned> C(out, m, n);
47
48 typedef typename Eigen::Tensor<T, 2>::DimensionPair DimPair;
49 int lhs_contract_dim = transpose_lhs ? 0 : 1;
50 int rhs_contract_dim = transpose_rhs ? 1 : 0;
51 const Eigen::array<DimPair, 1> dims(
52 {DimPair(lhs_contract_dim, rhs_contract_dim)});
53
54 // Matrix multiply is a special case of the "contract" operation where
55 // the contraction is performed along dimension 1 of the lhs and dimension
56 // 0 of the rhs.
57 C = A.contract(B, dims);
58}
59
60template <typename T>
61void SingleThreadedMatMul(const void* run_options_ptr, T* out, T* lhs, T* rhs,
62 int64 m, int64 n, int64 k, int32 transpose_lhs,
63 int32 transpose_rhs) {
64 if (m == 1 || n == 1) {
65 xla::EigenMatVec<T>(out, lhs, rhs, m, n, k, transpose_lhs, transpose_rhs);
66 } else {
67 MatMul<T>(run_options_ptr, out, lhs, rhs, m, n, k, transpose_lhs,
68 transpose_rhs);
69 }
70}
