18namespace xla {
19namespace cpu {
20
21CustomCallTargetRegistry* CustomCallTargetRegistry::Global() {
22 static auto* registry = new CustomCallTargetRegistry;
23 return registry;
24}
25
26void CustomCallTargetRegistry::Register(const std::string& symbol,
27 void* address) {
28 std::lock_guard<std::mutex> lock(mu_);
29 registered_symbols_[symbol] = address;
30}
31
32void* CustomCallTargetRegistry::Lookup(const std::string& symbol) const {
33 std::lock_guard<std::mutex> lock(mu_);
34 auto it = registered_symbols_.find(symbol);
35 return it == registered_symbols_.end() ? nullptr : it->second;
36}
