172// This visitor records which HLO instructions should have profiling information
173// recorded.
174class CollectProfileCandidates : public DfsHloVisitorWithDefault {
175 public:
176 static StatusOr<std::unordered_map<const HloInstruction*, int64>>
177 GetCandidatesForComputation(
178 HloComputation* computation,
179 const std::unordered_map<const HloInstruction*, int64>&
180 assigned_indices) {
181 std::unordered_map<const HloInstruction*, int64> hlo_to_profile_idx;
182 CollectProfileCandidates profile_candidates_for_computation(
183 &hlo_to_profile_idx, assigned_indices);
184 TF_RETURN_IF_ERROR(
185 computation->Accept(&profile_candidates_for_computation));
186 return hlo_to_profile_idx;
187 }
188
189 private:
190 CollectProfileCandidates(
191 std::unordered_map<const HloInstruction*, int64>* hlo_to_profile_idx,
192 const std::unordered_map<const HloInstruction*, int64>& assigned_indices)
193 : hlo_to_profile_idx_(hlo_to_profile_idx),
194 assigned_indices_(assigned_indices) {}
195
196 Status DefaultAction(HloInstruction* hlo_instruction) override {
197 hlo_to_profile_idx_->insert(
198 {hlo_instruction, FindOrDie(assigned_indices_, hlo_instruction)});
199 return Status::OK();
200 }
201
202 Status HandleCall(HloInstruction* call) override {
203 TF_RETURN_IF_ERROR(DefaultAction(call));
204 CollectProfileCandidates candidates_for_call(hlo_to_profile_idx_,
205 assigned_indices_);
206 TF_RETURN_IF_ERROR(call->to_apply()->Accept(&candidates_for_call));
207 return Status::OK();
208 }
209
210 // Skip constants, there is nothing to profile.
211 Status HandleConstant(HloInstruction*) override { return Status::OK(); }
212 // Skip parameters, they are a simple load.
213 Status HandleParameter(HloInstruction*) override { return Status::OK(); }
214 // It is important to recurse for "while" or else we risk overly coarse
215 // profiling information.
216 Status HandleWhile(HloInstruction* xla_while) override {
217 TF_RETURN_IF_ERROR(DefaultAction(xla_while));
218
219 CollectProfileCandidates candidates_for_condition(hlo_to_profile_idx_,
220 assigned_indices_);
221 TF_RETURN_IF_ERROR(
222 xla_while->while_condition()->Accept(&candidates_for_condition));
223
224 CollectProfileCandidates candidates_for_body(hlo_to_profile_idx_,
225 assigned_indices_);
226 TF_RETURN_IF_ERROR(xla_while->while_body()->Accept(&candidates_for_body));
227
228 return Status::OK();
229 }
230
231 std::unordered_map<const HloInstruction*, int64>* hlo_to_profile_idx_;
232 const std::unordered_map<const HloInstruction*, int64>& assigned_indices_;
233};

236Status CpuCompiler::RunHloPasses(HloModule* module, bool is_aot_compile) {
237 // Optimization pipeline.
238 HloPassPipeline pipeline("CPU");
239 pipeline.AddInvariantChecker<HloVerifier>();
240 pipeline.AddPass<CpuHloSupportChecker>();
241
242 ReducePrecisionInsertion::AddPasses(
243 &pipeline, module->config().debug_options(),
244 ReducePrecisionInsertion::PassTiming::BEFORE_OPTIMIZATION);
245
246 // TODO(b/35786417): Re-enable inliner pass after fixing the bug and deciding
247 // where we will take this pass in future.
248 // pipeline.AddPass<Inliner>();
249
250 // TODO(b/65775800): Fix wrong output bug in Call and remove the CallInliner
251 // pass.
252 pipeline.AddPass<CallInliner>();
253 pipeline.AddPass<DotDecomposer>();
254 pipeline.AddPass<ConvCanonicalization>();
255 {
256 auto& pass =
257 pipeline.AddPass<HloPassFix<HloPassPipeline>>("simplification");
258 pass.AddInvariantChecker<HloVerifier>();
259
260 pass.AddPass<BatchNormExpander>(
261 /*rewrite_training_op=*/true,
262 /*rewrite_inference_op=*/true,
263 /*rewrite_grad_op=*/true,
264 /*use_fusion=*/false);
265 pipeline.AddPass<GatherExpander>();
266 pass.AddPass<AlgebraicSimplifier>(
267 /*is_layout_sensitive=*/false,
268 [](const Shape&, const Shape&) { return false; },
269 /*enable_dot_strength_reduction=*/false);
270
271 // BatchNormExpander can create zero-sized ops, so zero-sized HLO
272 // elimination has to come after that pass.
273 pipeline.AddPass<ZeroSizedHloElimination>();
274
275 pass.AddPass<WhileLoopInvariantCodeMotion>();
276 pass.AddPass<TupleSimplifier>();
277 pass.AddPass<WhileLoopSimplifier>();
278 pass.AddPass<HloDCE>();
279 pass.AddPass<ReshapeMover>();
280 pass.AddPass<HloConstantFolding>();
281 pass.AddPass<ConditionalSimplifier>();
282 }
283 pipeline.AddPass<TransposeFolding>(
284 [](const HloInstruction& dot,
285 const TransposeFolding::OperandIndices& candidate_operands) {
286 return PotentiallyImplementedAsEigenDot(dot)
287 ? candidate_operands
288 : TransposeFolding::OperandIndices{};
289 },
290 TransposeFolding::NeverFoldTranspose);
291 pipeline.AddPass<HloCSE>(/*is_layout_sensitive=*/false);
292 pipeline.AddPass<CpuInstructionFusion>();
293
294 ReducePrecisionInsertion::AddPasses(
295 &pipeline, module->config().debug_options(),
296 ReducePrecisionInsertion::PassTiming::AFTER_FUSION);
297
298 pipeline.AddPass<CpuLayoutAssignment>(
299 module->mutable_entry_computation_layout());
300 // The LayoutAssignment pass may leave behind kCopy instructions which are
301 // duplicate or NOPs, so remove them with algebraic simplification and CSE.
302 pipeline.AddPass<HloPassFix<AlgebraicSimplifier>>(
303 /*is_layout_sensitive=*/true,
304 [](const Shape&, const Shape&) { return true; },
305 /*enable_dot_strength_reduction=*/false);
306 pipeline.AddPass<HloCSE>(/*is_layout_sensitive=*/true);
307 pipeline.AddPass<HloElementTypeConverter>(BF16, F32);
308 // Outline ops in the entry computation into calls to subcomputations.
309 const int max_parallelism =
310 module->config().intra_op_parallelism_threads() > 0
311 ? module->config().intra_op_parallelism_threads()
312 : tensorflow::port::NumSchedulableCPUs();
313 if (options::CpuParallelBackendRequested(module->config())) {
314 pipeline.AddPass<ParallelizationPreparation>(max_parallelism,
315 ShapeSizeBytesFunction());
316 } else if (!is_aot_compile) {
317 // Run ParallelTaskAssigner to assign parallel tasks to HLOs in module.
318 // Note this is not run for AOT because it would bring in thread pool
319 // and thread synchronization dependencies which would likely increase
320 // binary size (and most AOT applications are single-threaded).
321 // TODO(b/29630486) Support multi-threaded AOT.
322 pipeline.AddPass<ParallelTaskAssigner>(max_parallelism,
323 ShapeSizeBytesFunction());
324 }
325 // Copy insertion should be performed immediately before IR emission to avoid
326 // inserting unnecessary copies (later pass adds an instruction which
327 // materializes the value) or missing a necessary copy (later pass removes an
328 // instruction which materializes a value). DCE must be run immediately before
329 // (and sometime after) copy insertion, to avoid dead code from interfering
330 // with the rewrites.
331 pipeline.AddPass<HloDCE>();
332 pipeline.AddPass<FlattenCallGraph>();
333 pipeline.AddPass<CpuCopyInsertion>();
334 if (options::CpuParallelBackendRequested(module->config())) {
335 // Re-run the outlining, in case any copies were inserted into the entry
336 // computation.
337 pipeline.AddPass<ParallelizationPreparation>(max_parallelism,
338 ShapeSizeBytesFunction());
339 pipeline.AddPass<CpuCopyInsertion>();
340 }
341 pipeline.AddPass<HloDCE>();
342 return pipeline.Run(module).status();
343}

361llvm::CodeGenOpt::Level CodeGenOptLevel(const HloModuleConfig& module_config) {
362 VLOG(2) << "backend_optimization_level: "
363 << module_config.debug_options().xla_backend_optimization_level();
364 switch (module_config.debug_options().xla_backend_optimization_level()) {
365 case 1:
366 return llvm::CodeGenOpt::Less;
367 case 2:
368 return llvm::CodeGenOpt::Default;
369 case 3:
370 return llvm::CodeGenOpt::Aggressive;
371 default:
372 return llvm::CodeGenOpt::None;
373 }

376Status InitializeModuleHooks(
377 const HloModule& hlo_module,
378 const LLVMCompiler::ModuleHook& user_pre_optimization_hook,
379 const LLVMCompiler::ModuleHook& user_post_optimization_hook,
380 LLVMCompiler::ModuleHook* pre_optimization_ir_hook,
381 LLVMCompiler::ModuleHook* post_optimization_ir_hook) {
382 const string& ir_dump_directory =
383 hlo_module.config().debug_options().xla_dump_ir_to();
384 if (ir_dump_directory.empty()) {
385 *pre_optimization_ir_hook = user_pre_optimization_hook;
386 *post_optimization_ir_hook = user_post_optimization_hook;
387 return Status::OK();
388 }
389
390 const string& hlo_module_name = hlo_module.name();
391
392 // Create the IR hooks. If applicable, each IR hook does the following:
393 //
394 // * Calls the user supplied module hook.
395 // * Writes out the IR to a file in the output directory designated by
396 // --xla_dump_ir_to
397
398 *pre_optimization_ir_hook =
399 [user_pre_optimization_hook, ir_dump_directory,
400 hlo_module_name](const llvm::Module& llvm_module) {
401 if (user_pre_optimization_hook) {
402 TF_RETURN_IF_ERROR(user_pre_optimization_hook(llvm_module));
403 }
404 return llvm_ir::DumpIRToDirectory(/*directory_name=*/ir_dump_directory,
405 /*hlo_module_name=*/hlo_module_name,
406 llvm_module,
407 /*optimized=*/false);
408 };
409
410 *post_optimization_ir_hook =
411 [user_post_optimization_hook, ir_dump_directory,
412 hlo_module_name](const llvm::Module& llvm_module) {
413 if (user_post_optimization_hook) {
414 TF_RETURN_IF_ERROR(user_post_optimization_hook(llvm_module));
415 }
416 return llvm_ir::DumpIRToDirectory(/*directory_name=*/ir_dump_directory,
417 /*hlo_module_name=*/hlo_module_name,
418 llvm_module,
419 /*optimized=*/true);
420 };
421
422 return Status::OK();
423}

425Status VerifyLlvmModule(const llvm::Module& llvm_module) {
426 XLA_SCOPED_LOGGING_TIMER("CpuCompiler - Running LLVM verifier");
427
428 std::string err;
429 llvm::raw_string_ostream err_stream(err);
430
431 // verifyModule() returns true if the module is broken.
432 TF_RET_CHECK(!llvm::verifyModule(llvm_module, &err_stream))
433 << "Invalid LLVM IR before optimizations:\n"
434 << err_stream.str()
435 << "\nThis probably indicates a bug in the HLO -> LLVM IR lowering. "
436 "Rerun with --xla_dump_ir_to to get the IR. ";
437 return Status::OK();
438}

443 std::unique_ptr<HloModule> module,

453 return std::move(module);

456StatusOr<std::unique_ptr<Executable>> CpuCompiler::RunBackend(
457 std::unique_ptr<HloModule> module,
458 perftools::gputools::StreamExecutor* stream_exec,
459 DeviceMemoryAllocator* /*device_allocator*/) {
460 const string timer_message =
461 "Compiling [" + module->name() + "] for CPU using JIT";
462 XLA_SCOPED_LOGGING_TIMER(timer_message);
463
464 VLOG(1) << "Compiling: " << module->name();
465 TF_RET_CHECK(stream_exec != nullptr);
466 std::call_once(llvm_command_line_options_initialized,
467 &llvm_ir::InitializeLLVMCommandLineOptions, module->config());
468
469 ModuleHook pre_optimization_ir_hook;
470 ModuleHook post_optimization_ir_hook;
471 TF_RETURN_IF_ERROR(InitializeModuleHooks(
472 *module, user_pre_optimization_hook_, user_post_optimization_hook_,
473 &pre_optimization_ir_hook, &post_optimization_ir_hook));
474
475 // Compile must be thread-safe so create a new LLVM context for the module.
476 auto llvm_context = xla::MakeUnique<llvm::LLVMContext>();
477 auto llvm_module =
478 xla::MakeUnique<llvm::Module>("__compute_module", *llvm_context);
479
480 auto jit = xla::MakeUnique<SimpleOrcJIT>(
481 CompilerTargetOptions(module->config()),
482 CodeGenOptLevel(module->config()),
483 options::OptimizeForSizeRequested(module->config()),
484 module->config().debug_options().xla_enable_fast_math(),
485 module->config().debug_options().xla_llvm_disable_expensive_passes(),
486 pre_optimization_ir_hook, post_optimization_ir_hook);
487 llvm_module->setDataLayout(jit->data_layout());
488 llvm_module->setTargetTriple(jit->target_triple().getTriple());
489
490 HloComputation* entry_computation = module->entry_computation();
491 std::unordered_map<const HloInstruction*, int64> instruction_to_profile_idx;
492 std::unordered_map<const HloComputation*, int64> computation_to_profile_idx;
493 std::unique_ptr<HloProfileIndexMap> hlo_profile_index_map;
494 std::unique_ptr<HloProfilePrinterData> hlo_profile_printer_data;
495 if (module->config().hlo_profiling_enabled()) {
496 hlo_profile_index_map = MakeUnique<HloProfileIndexMap>(*module);
497
498 TF_ASSIGN_OR_RETURN(
499 instruction_to_profile_idx,
500 CollectProfileCandidates::GetCandidatesForComputation(
501 entry_computation,
502 hlo_profile_index_map->instruction_to_profile_idx()));
503
504 auto shape_size_bytes = [](const Shape& shape) {
505 // On the cpu, opaques are pointers.
506 if (ShapeUtil::IsOpaque(shape)) {
507 return static_cast<int64>(sizeof(void*));
508 }
509 return ShapeUtil::ByteSizeOf(shape, sizeof(void*));
510 };
511
512 HloCostAnalysis cost_analysis(shape_size_bytes);
513 TF_RETURN_IF_ERROR(entry_computation->Accept(&cost_analysis));
514 hlo_profile_printer_data =
515 CreateHloProfilePrinterData(*hlo_profile_index_map, cost_analysis);
516 computation_to_profile_idx =
517 hlo_profile_index_map->computation_to_profile_idx();
518 }
519
520 std::unique_ptr<Executable> cpu_executable;
521
522 // Cache these flags here since we'll want to access them after the module's
523 // ownership is std::moved.
524 const bool embed_ir_in_executable =
525 module->config().debug_options().xla_embed_ir_in_executable();
526 const string xla_dump_optimized_hlo_proto_to =
527 module->config().debug_options().xla_dump_optimized_hlo_proto_to();
528
529 if (options::CpuParallelBackendRequested(module->config())) {
530 VLOG(1) << "Using parallel cpu backend";
531
532 // Run buffer analysis on the HLO graph. This analysis figures out which
533 // temporary buffers are required to run the computation.
534 // DependencyHloOrdering is used for the parallel emitter because the order
535 // of HLO instruction execution is not known ahead of time.
536 // DependencyHloOrdering is the most conservative partial order and only
537 // uses data dependencies for determining order.
538 TF_ASSIGN_OR_RETURN(
539 std::unique_ptr<BufferAssignment> assignment,
540 BufferAssigner::Run(
541 module.get(), xla::MakeUnique<DependencyHloOrdering>(module.get()),
542 BufferSizeBytesFunction(), memory_alignment));
543 // BufferAssignment::ToString() includes a header, so no need for us to
544 // print one ourselves.
545 XLA_VLOG_LINES(2, assignment->ToString());
546
547 if (!xla_dump_optimized_hlo_proto_to.empty()) {
548 HloProto proto = MakeHloProto(*module, *assignment);
549 TF_RETURN_IF_ERROR(protobuf_util::DumpProtoToDirectory(
550 proto, xla_dump_optimized_hlo_proto_to, module->name()));
551 }
552
553 // If we are using the parallel CPU backend, we need to create map from
554 // HloInstruction to the corresponding generated function name.
555 std::map<HloComputation*, HloInstruction*> parallel_computations;
556 std::unordered_map<const HloInstruction*, std::unique_ptr<unsigned char[]>>
557 aligned_constants;
558 for (auto instruction : entry_computation->MakeInstructionPostOrder()) {
559 // Parameters and constants don't get their own computation.
560 if (instruction->opcode() == HloOpcode::kParameter) {
561 continue;
562 }
563 if (instruction->opcode() == HloOpcode::kConstant) {
564 // Copy the constant out of the ProtocolBuffer so that we can give it a
565 // higher alignment.
566 const void* data = instruction->literal().untyped_data();
567 int64 size = CpuExecutable::ShapeSizeBytes(instruction->shape());
568 auto iter = aligned_constants.emplace(
569 instruction, xla::MakeUnique<unsigned char[]>(size));
570 CHECK_EQ(iter.second, true);
571 unsigned char* aligned_data = iter.first->second.get();
572 memcpy(aligned_data, data, size);
573 continue;
574 }
575 // The parallel preparation should have ensured that the top-level
576 // computation consists solely of Call instructions.
577 TF_RET_CHECK(instruction->opcode() == HloOpcode::kCall)
578 << module->ToString();
579 HloComputation* to_apply = instruction->to_apply();
580 parallel_computations.emplace(to_apply, instruction);
581 }
582
583 IrEmitter ir_emitter(*module, *assignment, llvm_module.get(),
584 std::move(instruction_to_profile_idx),
585 std::move(computation_to_profile_idx),
586 jit->target_machine(), jit->external_constant_pool());
587
588 std::unique_ptr<HloInstructionMap<string>> function_names(
589 new HloInstructionMap<string>());
590 for (auto embedded_computation :
591 entry_computation->MakeEmbeddedComputationsList()) {
592 if (embedded_computation->IsFusionComputation()) {
593 continue;
594 }
595 auto parallel_computation_iter =
596 parallel_computations.find(embedded_computation);
597 // All parallel computations are considered to be an entry computation for
598 // IR generation purposes.
599 bool computation_is_parallel =
600 parallel_computation_iter != parallel_computations.end();
601 TF_ASSIGN_OR_RETURN(
602 llvm::Function * ir_function,
603 ir_emitter.EmitComputation(
604 embedded_computation, embedded_computation->name(),
605 /*is_top_level_computation=*/computation_is_parallel,
606 /*instruction_order=*/nullptr));
607 // If this computation is parallel, remember it in the function name map.
608 // This way we know what function to execute when we try to run code for
609 // the Call instruction.
610 if (computation_is_parallel) {
611 HloInstruction* call_instruction = parallel_computation_iter->second;
612 InsertOrDie(function_names.get(), call_instruction,
613 llvm_ir::AsString(ir_function->getName()));
614 }
615 }
616
617 string ir_module_string;
618 if (embed_ir_in_executable) {
619 ir_module_string = llvm_ir::DumpModuleToString(*llvm_module);
620 }
621 TF_RETURN_IF_ERROR(VerifyLlvmModule(*llvm_module));
622
623 // JIT compile the LLVM IR module to in-memory machine code.
624 jit->AddModule(std::move(llvm_module));
625 cpu_executable.reset(new ParallelCpuExecutable(
626 std::move(jit), std::move(assignment), std::move(module),
627 std::move(function_names), std::move(aligned_constants),
628 std::move(hlo_profile_printer_data), std::move(hlo_profile_index_map)));
629
630 if (embed_ir_in_executable) {
631 static_cast<CpuExecutable&>(*cpu_executable)
632 .set_ir_module_string(ir_module_string);
633 }
634 } else {
635 VLOG(1) << "Using sequential cpu backend";
636
637 // Select an order for emitting the HLO instructions for each
638 // computation. Using this sequence enables tighter buffer liveness analysis
639 // and reduced memory usage (as compared to using DependencyHloOrdering).
640 TF_ASSIGN_OR_RETURN(
641 SequentialHloOrdering::HloModuleSequence module_sequence,
642 CreateMemoryMinimizingSequence(*module, BufferSizeBytesFunction()));
643
644 // Run buffer analysis on the HLO graph. This analysis figures out which
645 // temporary buffers are required to run the computation.
646 TF_ASSIGN_OR_RETURN(
647 std::unique_ptr<BufferAssignment> assignment,
648 BufferAssigner::Run(module.get(),
649 xla::MakeUnique<SequentialHloOrdering>(
650 module.get(), module_sequence),
651 BufferSizeBytesFunction(), memory_alignment));
652 // BufferAssignment::ToString() includes a header, so no need for us to
653 // print one ourselves.
654 XLA_VLOG_LINES(2, assignment->ToString());
655
656 if (!xla_dump_optimized_hlo_proto_to.empty()) {
657 HloProto proto = MakeHloProto(*module, *assignment);
658 TF_RETURN_IF_ERROR(protobuf_util::DumpProtoToDirectory(
659 proto, xla_dump_optimized_hlo_proto_to, module->name()));
660 }
661
662 // Each computation is a single function. Emit all embedded computations
663 // before the entry computation. The order of computations returned from
664 // GetEmbeddedComputations guarantees that a called computation occurs
665 // before a caller computation.
666
667 IrEmitter ir_emitter(*module, *assignment, llvm_module.get(),
668 std::move(instruction_to_profile_idx),
669 std::move(computation_to_profile_idx),
670 jit->target_machine(), jit->external_constant_pool());
671
672 for (auto embedded_computation :
673 entry_computation->MakeEmbeddedComputationsList()) {
674 if (embedded_computation->IsFusionComputation()) {
675 continue;
676 }
677 TF_RETURN_IF_ERROR(
678 ir_emitter
679 .EmitComputation(embedded_computation,
680 embedded_computation->name(),
681 /*is_top_level_computation=*/false,
682 &module_sequence.at(embedded_computation))
683 .status());
684 }
685 string function_name_prefix = entry_computation->name().empty()
686 ? "__compute"
687 : entry_computation->name();
688 TF_ASSIGN_OR_RETURN(
689 llvm::Function * entry_function,
690 ir_emitter.EmitComputation(entry_computation, function_name_prefix,
691 /*is_top_level_computation=*/true,
692 &module_sequence.at(entry_computation)));
693
694 string function_name = llvm_ir::AsString(entry_function->getName());
695 string ir_module_string;
696 if (embed_ir_in_executable) {
697 ir_module_string = llvm_ir::DumpModuleToString(*llvm_module);
698 }
699 TF_RETURN_IF_ERROR(VerifyLlvmModule(*llvm_module));
700
701 XLA_VLOG_LINES(2, "LLVM IR:\n" + llvm_ir::DumpModuleToString(*llvm_module));
702
703 // JIT compile the LLVM IR module to in-memory machine code.
704 jit->AddModule(std::move(llvm_module));
705 cpu_executable.reset(new CpuExecutable(
706 std::move(jit), std::move(assignment), std::move(module), function_name,
707 std::move(hlo_profile_printer_data), std::move(hlo_profile_index_map)));
708
709 if (embed_ir_in_executable) {
710 static_cast<CpuExecutable&>(*cpu_executable)
711 .set_ir_module_string(ir_module_string);
712 }
713 }
714
715 VLOG(1) << "Compilation finished";
716 return std::move(cpu_executable);
717}

719StatusOr<std::vector<std::unique_ptr<AotCompilationResult>>>
720CpuCompiler::CompileAheadOfTime(std::vector<std::unique_ptr<HloModule>> modules,
721 const AotCompilationOptions& aot_options) {
722 TF_RET_CHECK(!modules.empty());
723 std::call_once(llvm_command_line_options_initialized,
724 &llvm_ir::InitializeLLVMCommandLineOptions,
725 modules[0]->config());
726
727 // We can pass just one llvm::TargetOptions when we compile the LLVM module,
728 // so we bail if the configs have conflicting flags. At the moment, the only
729 // flag that needs to be consistent is fast-math.
730 const bool fast_math_enabled =
731 modules[0]->config().debug_options().xla_enable_fast_math();
732 for (const auto& module : modules) {
733 if (module->config().debug_options().xla_enable_fast_math() !=
734 fast_math_enabled) {
735 return InvalidArgument(
736 "All HLO module configs must have the same value for "
737 "xla_enable_fast_math.");
738 }
739 }
740
741 if (aot_options.PlatformId() != se::host::kHostPlatformId) {
742 return InvalidArgument("Incompatible AOT compilation platform");
743 }
744 const CpuAotCompilationOptions& options =
745 static_cast<const CpuAotCompilationOptions&>(aot_options);
746 llvm::StringRef target_triple = llvm_ir::AsStringRef(options.triple());
747 llvm::Triple triple(llvm::Triple::normalize(target_triple));
748 std::string error;
749 const llvm::Target* target =
750 llvm::TargetRegistry::lookupTarget(triple.getTriple(), error);
751 if (target == nullptr) {
752 return InternalError("TargetRegistry::lookupTarget failed: %s",
753 error.c_str());
754 }
755
756 llvm::Reloc::Model reloc_model = llvm::Reloc::Static;
757 llvm::PICLevel::Level pic_level = llvm::PICLevel::NotPIC;
758 llvm::PIELevel::Level pie_level = llvm::PIELevel::Default;
759 switch (options.relocation_model()) {
760 case CpuAotCompilationOptions::RelocationModel::Static:
761 reloc_model = llvm::Reloc::Static;
762 pic_level = llvm::PICLevel::NotPIC;
763 pie_level = llvm::PIELevel::Default;
764 break;
765 case CpuAotCompilationOptions::RelocationModel::SmallPic:
766 reloc_model = llvm::Reloc::PIC_;
767 pic_level = llvm::PICLevel::SmallPIC;
768 pie_level = llvm::PIELevel::Default;
769 break;
770 case CpuAotCompilationOptions::RelocationModel::BigPic:
771 reloc_model = llvm::Reloc::PIC_;
772 pic_level = llvm::PICLevel::BigPIC;
773 pie_level = llvm::PIELevel::Default;
774 break;
775 case CpuAotCompilationOptions::RelocationModel::SmallPie:
776 reloc_model = llvm::Reloc::PIC_;
777 pic_level = llvm::PICLevel::SmallPIC;
778 pie_level = llvm::PIELevel::Small;
779 break;
780 case CpuAotCompilationOptions::RelocationModel::BigPie:
781 reloc_model = llvm::Reloc::PIC_;
782 pic_level = llvm::PICLevel::BigPIC;
783 pie_level = llvm::PIELevel::Large;
784 break;
785 }
786 llvm::StringRef cpu_name = llvm_ir::AsStringRef(options.cpu_name());
787 llvm::StringRef features = llvm_ir::AsStringRef(options.features());
788 llvm::CodeGenOpt::Level opt_level = CodeGenOptLevel(modules[0]->config());
789 std::unique_ptr<llvm::TargetMachine> target_machine = WrapUnique(
790 target->createTargetMachine(triple.getTriple(), cpu_name, features,
791 CompilerTargetOptions(modules[0]->config()),
792 reloc_model, llvm::None, opt_level));
793
794 // Compile must be thread-safe so create a new LLVM context for the module.
795 llvm::LLVMContext llvm_context;
796 llvm::Module llvm_module("__compute_module", llvm_context);
797 llvm_module.setDataLayout(target_machine->createDataLayout());
798 llvm_module.setTargetTriple(triple.getTriple());
799 if (pic_level != llvm::PICLevel::NotPIC) {
800 llvm_module.setPICLevel(pic_level);
801 }
802 if (pie_level != llvm::PIELevel::Default) {
803 llvm_module.setPIELevel(pie_level);
804 }
805
806 std::vector<std::unique_ptr<AotCompilationResult>> results;
807 for (size_t i = 0; i < modules.size(); ++i) {
808 HloModule* module = modules[i].get();
809 VLOG(1) << "Compiling ahead-of-time: " << module->name();
810
811 VLOG(2) << "Before optimization:";
812 XLA_VLOG_LINES(2, module->ToString());
813
814 TF_RETURN_IF_ERROR(RunHloPasses(module, /*is_aot_compile=*/true));
815
816 VLOG(2) << "After optimization:";
817 XLA_VLOG_LINES(2, module->ToString());
818
819 TF_ASSIGN_OR_RETURN(
820 SequentialHloOrdering::HloModuleSequence module_sequence,
821 CreateMemoryMinimizingSequence(*module, BufferSizeBytesFunction()));
822
823 // Run buffer analysis on the HLO graph. This analysis figures out which
824 // temporary buffers are required to run the computation.
825 TF_ASSIGN_OR_RETURN(
826 std::unique_ptr<BufferAssignment> assignment,
827 BufferAssigner::Run(
828 module,
829 xla::MakeUnique<SequentialHloOrdering>(module, module_sequence),
830 BufferSizeBytesFunction(), memory_alignment));
831 // BufferAssignment::ToString() includes a header, so no need for us to
832 // print one ourselves.
833 XLA_VLOG_LINES(2, assignment->ToString());
834
835 const string xla_dump_optimized_hlo_proto_to =
836 module->config().debug_options().xla_dump_optimized_hlo_proto_to();
837 if (!xla_dump_optimized_hlo_proto_to.empty()) {
838 HloProto proto = MakeHloProto(*module, *assignment);
839 TF_RETURN_IF_ERROR(protobuf_util::DumpProtoToDirectory(
840 proto, xla_dump_optimized_hlo_proto_to, module->name()));
841 }
842
843 IrEmitter ir_emitter(*module, *assignment, &llvm_module,
844 /*instruction_to_profile_idx=*/
845 std::unordered_map<const HloInstruction*, int64>{},
846 /*computation_to_profile_idx=*/
847 std::unordered_map<const HloComputation*, int64>{},
848 target_machine.get(),
849 /*external_constant_pool=*/nullptr);
850 HloComputation* computation = module->entry_computation();
851 for (auto embedded_computation :
852 computation->MakeEmbeddedComputationsList()) {
853 if (embedded_computation->IsFusionComputation()) {
854 continue;
855 }
856 TF_RETURN_IF_ERROR(
857 ir_emitter
858 .EmitComputation(embedded_computation,
859 embedded_computation->name(),
860 /*is_top_level_computation=*/false,
861 &module_sequence.at(embedded_computation))
862 .status());
863 }
864 const string& entry_point_name = options.entry_point_name();
865 TF_ASSIGN_OR_RETURN(
866 llvm::Function * entry_function,
867 ir_emitter.EmitComputation(computation, entry_point_name,
868 /*is_top_level_computation=*/true,
869 &module_sequence.at(computation)));
870
871 CHECK(entry_function->getName() == llvm_ir::AsStringRef(entry_point_name));
872
873 ModuleHook pre_optimization_ir_dump_hook;
874 ModuleHook post_optimization_ir_dump_hook;
875 TF_RETURN_IF_ERROR(InitializeModuleHooks(
876 *module, user_pre_optimization_hook_, user_post_optimization_hook_,
877 &pre_optimization_ir_dump_hook, &post_optimization_ir_dump_hook));
878
879 // Run the LLVM verifier over the unoptimized LLVM IR. If it fails, run the
880 // pre-optimization IR dump hook before returning.
881 {
882 Status verify_status = VerifyLlvmModule(llvm_module);
883 if (!verify_status.ok() && pre_optimization_ir_dump_hook) {
884 pre_optimization_ir_dump_hook(llvm_module).IgnoreError();
885 }
886 TF_RETURN_IF_ERROR(verify_status);
887 }
888
889 Disassembler disassembler(*target_machine);
890 CompilerFunctor compiler_functor(
891 target_machine.get(), &disassembler, opt_level,
892 options::OptimizeForSizeRequested(module->config()),
893 module->config().debug_options().xla_enable_fast_math(),
894 module->config().debug_options().xla_llvm_disable_expensive_passes(),
895 pre_optimization_ir_dump_hook, post_optimization_ir_dump_hook);
896 std::unique_ptr<llvm::MemoryBuffer> object_file =
897 compiler_functor(llvm_module);
898 ObjectFileData object_file_data(object_file->getBufferStart(),
899 object_file->getBufferEnd());
900
901 BufferSizes buffer_sizes;
902 for (const BufferAllocation& allocation : assignment->Allocations()) {
903 // Callers don't need to allocate temporary buffers for parameters.
904 if (allocation.is_entry_computation_parameter()) {
905 buffer_sizes.push_back(-1);
906 continue;
907 }
908 // Callers don't need to allocate anything for thread-local temporary
909 // buffers. They are lowered to allocas.
910 if (allocation.is_thread_local()) {
911 buffer_sizes.push_back(-1);
912 continue;
913 }
914 buffer_sizes.push_back(allocation.size());
915 }
916
917 TF_ASSIGN_OR_RETURN(const BufferAllocation::Slice result_slice,
918 assignment->GetUniqueTopLevelOutputSlice());
919
920 results.emplace_back(MakeUnique<CpuAotCompilationResult>(
921 std::move(object_file_data), std::move(buffer_sizes),
922 result_slice.index()));
923 }
924
925 VLOG(1) << "Compilation finished";
926 return std::move(results);
927}
