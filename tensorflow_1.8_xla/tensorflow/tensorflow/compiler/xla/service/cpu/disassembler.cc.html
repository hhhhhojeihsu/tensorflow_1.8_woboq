37Disassembler::Disassembler(const llvm::TargetMachine& target_machine)
38 : subtarget_info_(*target_machine.getMCSubtargetInfo()) {
39 objfile_info_.reset(new llvm::MCObjectFileInfo());
40 mc_context_.reset(new llvm::MCContext(target_machine.getMCAsmInfo(),
41 target_machine.getMCRegisterInfo(),
42 objfile_info_.get()));
43 disassembler_.reset(target_machine.getTarget().createMCDisassembler(
44 subtarget_info_, *mc_context_));
45 inst_printer_.reset(target_machine.getTarget().createMCInstPrinter(
46 target_machine.getTargetTriple(),
47 /*SyntaxVariant=*/1, // Use Intel syntax.
48 *target_machine.getMCAsmInfo(), *target_machine.getMCInstrInfo(),
49 *target_machine.getMCRegisterInfo()));
50 inst_analysis_.reset(target_machine.getTarget().createMCInstrAnalysis(
51 target_machine.getMCInstrInfo()));
52}

54// This code is based on llvm-objdump in llvm/tools.
55StatusOr<DisassemblerResult> Disassembler::DisassembleObjectFile(
56 const llvm::object::ObjectFile& object_file) const {
57 if (disassembler_ == nullptr) {
58 return NotFound("could not find a disassembler for this platform");
59 }
60
61 std::string buffer_string;
62 llvm::raw_string_ostream ostream(buffer_string);
63 uint64_t code_size_bytes = 0;
64
65 // Iterate through sections. Disassemble symbols of the text section(s).
66 for (auto& section : object_file.sections()) {
67 if (!section.isText()) {
68 continue;
69 }
70
71 // Gather symbols from the section.
72 std::vector<llvm::object::SymbolRef> symbols;
73 for (auto& symbol : object_file.symbols()) {
74 if (section.containsSymbol(symbol)) {
75 symbols.push_back(symbol);
76 }
77 }
78
79 // Sort the symbols in increasing address order.
80 std::sort(
81 symbols.begin(), symbols.end(),
82 [](const llvm::object::SymbolRef& a, const llvm::object::SymbolRef& b) {
83 // getAddress returns a Expected object. Assert there is no error
84 // before extracting the address.
85 llvm::Expected<uint64_t> a_address_or_error = a.getAddress();
86 CHECK(a_address_or_error);
87 llvm::Expected<uint64_t> b_address_or_error = b.getAddress();
88 CHECK(b_address_or_error);
89 return a_address_or_error.get() < b_address_or_error.get();
90 });
91
92 // Construct ArrayRef pointing to section contents.
93 llvm::StringRef section_content_string;
94 if (section.getContents(section_content_string)) {
95 continue;
96 }
97 llvm::ArrayRef<uint8_t> section_content_bytes(
98 reinterpret_cast<const uint8*>(section_content_string.data()),
99 section_content_string.size());
100
101 // Use int types from LLVM (eg, uint64_t) for values passed to and returned
102 // from the LLVM API. These values map to different types in LLVM and
103 // XLA (unsigned long vs unsigned long long).
104 uint64_t section_address = section.getAddress();
105 uint64_t section_size = section.getSize();
106
107 // Iterate through symbols in increasing address order and disassemble each
108 // one.
109 for (int i = 0; i < symbols.size(); ++i) {
110 auto symbol = symbols[i];
111 llvm::Expected<uint64_t> address = symbol.getAddress();
112 CHECK(address);
113 uint64_t start_index = address.get() - section_address;
114
115 // End of symbol is either the end of the section or the start of the next
116 // symbol.
117 uint64_t end_index;
118 if (i < symbols.size() - 1) {
119 llvm::Expected<uint64_t> next_address = symbols[i + 1].getAddress();
120 CHECK(next_address);
121 end_index = std::min(section_size, next_address.get());
122 } else {
123 end_index = section_size;
124 }
125
126 // Skip zero-length symbols.
127 if (start_index == end_index) {
128 continue;
129 }
130
131 llvm::Expected<llvm::StringRef> name_or_error = symbol.getName();
132 TF_RET_CHECK(name_or_error);
133 ostream << name_or_error.get().str() << ":\n";
134
135 // Update the code size statistic.
136 code_size_bytes += end_index - start_index;
137
138 // Disassemble symbol instruction-by-instruction.
139 uint64_t index = start_index;
140 while (index < end_index) {
141 llvm::MCInst instruction;
142 uint64_t size;
143 llvm::MCDisassembler::DecodeStatus decode_status =
144 disassembler_->getInstruction(instruction, size,
145 section_content_bytes.slice(index),
146 /*Address=*/section_address + index,
147 /*VStream=*/llvm::nulls(),
148 /*CStream=*/llvm::nulls());
149 // If we fail to disassemble, then we must skip past this address.
150 if (size == 0) {
151 size = 1;
152 }
153
154 ostream << tensorflow::strings::Printf("0x%08lx", index) << " ";
155
156 if (decode_status == llvm::MCDisassembler::Success) {
157 // For branches, try to determine the actual address and emit it as an
158 // annotation.
159 string annotation;
160 if (inst_analysis_ &&
161 (inst_analysis_->isUnconditionalBranch(instruction) ||
162 inst_analysis_->isConditionalBranch(instruction))) {
163 uint64_t target;
164 if (inst_analysis_->evaluateBranch(
165 instruction, section_address + index, size, target)) {
166 annotation = tensorflow::strings::Printf("[0x%08lx]", target);
167 }
168 }
169 inst_printer_->printInst(&instruction, ostream, annotation.c_str(),
170 subtarget_info_);
171 } else {
172 ostream << " <unknown>";
173 }
174
175 ostream << "\n";
176 index += size;
177 }
178 }
179 }
180
181 ostream.flush();
182 return DisassemblerResult{
183 string(buffer_string.data(), buffer_string.length()), code_size_bytes};
184}
