24// Performs a single-threaded matrix multiplication using Eigen. 'lhs' and 'rhs'
25// are pointers to buffers containing input matrices in column-major order.
26// 'out' is a pointer to a buffer sufficiently large to hold the result of the
27// operation. Following standard nomenclature: lhs is m x k, rhs is k x n, and
28// out is m x n.
29extern void __xla_cpu_runtime_EigenSingleThreadedMatMulF16(
30 const void* /* xla::ExecutableRunOptions* */ run_options_ptr,
31 Eigen::half* out, Eigen::half* lhs, Eigen::half* rhs, tensorflow::int64 m,
32 tensorflow::int64 n, tensorflow::int64 k, tensorflow::int32 transpose_lhs,
33 tensorflow::int32 transpose_rhs);
34
35extern void __xla_cpu_runtime_EigenSingleThreadedMatMulF32(
36 const void* /* xla::ExecutableRunOptions* */ run_options_ptr, float* out,
37 float* lhs, float* rhs, tensorflow::int64 m, tensorflow::int64 n,
38 tensorflow::int64 k, tensorflow::int32 transpose_lhs,
39 tensorflow::int32 transpose_rhs);
40
41extern void __xla_cpu_runtime_EigenSingleThreadedMatMulF64(
42 const void* /* xla::ExecutableRunOptions* */ run_options_ptr, double* out,
43 double* lhs, double* rhs, tensorflow::int64 m, tensorflow::int64 n,
44 tensorflow::int64 k, tensorflow::int32 transpose_lhs,
45 tensorflow::int32 transpose_rhs);
46
47} // extern "C"
