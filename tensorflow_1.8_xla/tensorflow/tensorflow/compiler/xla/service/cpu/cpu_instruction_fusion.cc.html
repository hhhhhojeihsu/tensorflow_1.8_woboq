24int64 BytesInDimension(const Shape& shape, int64 dimension) {
25 return ShapeUtil::ByteSizeOfPrimitiveType(shape.element_type()) *
26 shape.dimensions(dimension);
27}
28
29bool CanBeLoopFused(const HloInstruction& hlo) {
30 // These are the only ones we fuse since we rely on effective elemental IR
31 // generation.
32 return hlo.IsElementwise() || //
33 hlo.opcode() == HloOpcode::kBroadcast ||
34 hlo.opcode() == HloOpcode::kConcatenate ||
35 hlo.opcode() == HloOpcode::kDynamicSlice ||
36 hlo.opcode() == HloOpcode::kDynamicUpdateSlice ||
37 hlo.opcode() == HloOpcode::kPad ||
38 hlo.opcode() == HloOpcode::kReshape ||
39 hlo.opcode() == HloOpcode::kReverse ||
40 hlo.opcode() == HloOpcode::kSlice ||
41 hlo.opcode() == HloOpcode::kTranspose;
42}
43
44bool IsMatrixVectorDot(const HloInstruction* hlo) {
45 const Shape& hlo_shape = hlo->shape();
46 return hlo->opcode() == HloOpcode::kDot && hlo_shape.dimensions_size() == 2 &&
47 (hlo_shape.dimensions(0) == 1 || hlo_shape.dimensions(1) == 1);
48}
49
50bool CanBeOutputFused(const HloInstruction* producer,
51 const HloInstruction* consumer) {
52 return consumer->opcode() == HloOpcode::kAdd && IsMatrixVectorDot(producer) &&
53 producer->user_count() == 1;
54}
55
56bool CanBeOutputFusedIntoSomeOperand(const HloInstruction* consumer) {
57 return consumer->opcode() == HloOpcode::kAdd &&
58 (CanBeOutputFused(consumer->operand(0), consumer) ||
59 CanBeOutputFused(consumer->operand(1), consumer));
60}

63bool CpuInstructionFusion::ShouldFuse(HloInstruction* consumer,
64 int64 operand_index) {
65 HloInstruction* producer = consumer->mutable_operand(operand_index);
66 VLOG(2) << "Considering for fusion: operand " << operand_index << " of "
67 << consumer->ToString();
68
69 constexpr int kFusionThresholdBytes = 16 * 1024;
70
71 if (CanBeOutputFused(producer, consumer)) {
72 return true;
73 }
74
75 if (CanBeOutputFusedIntoSomeOperand(producer)) {
76 return false;
77 }
78
79 if (!CanBeLoopFused(*producer)) {
80 VLOG(2) << "Producer is not fusile.";
81 return false;
82 }
83
84 // Cost condition: not fuse (simple, expensive producers) and (consumers who
85 // reuse operand elements).
86 if (producer->opcode() != HloOpcode::kFusion &&
87 consumer->ReusesOperandElements(operand_index) &&
88 is_expensive(*producer)) {
89 VLOG(2) << "Fusion is not profitable.";
90 return false;
91 }
92
93 // TODO(b/28644064): see if the "producer->operand_count() == 0" check is
94 // necessary.
95 if (producer->operand_count() == 0 ||
96 !InstructionFusion::ShouldFuse(consumer, operand_index)) {
97 VLOG(2)
98 << "Not fusing: producer has no operands, or !ShouldFuse(consumer).";
99 return false;
100 }
101
102 // Output fusion is not currently supported on CPUs.
103 if (producer->opcode() == HloOpcode::kFusion) {
104 VLOG(2) << "Not fusing: producer is itself a fusion node.";
105 return false;
106 }
107
108 if (consumer->opcode() == HloOpcode::kDot) {
109 // In the general case we call out to optimized "black box" GEMM routines
110 // for Dot, which precludes fusion. However, in very specific cases, we try
111 // to fuse Dot operations by generating an elemental dot implementation.
112 //
113 // We need to be careful and conservative here since any benefit we get from
114 // fusion can easily be overshadowed by the overhead of a naive GEMM
115 // algorithm in the IR.
116 const Shape& output_shape = consumer->shape();
117 if (output_shape.dimensions_size() == 2) {
118 // We fuse in cases where we have dot([A,B],[B,1]) or dot([1,A],[A,B]) and
119 // fusion can get rid of the larger tensor. We assume that a naive
120 // traversal of a small enough (to fit in L1) column or row tensor is
121 // "good enough" from the perspective of cache management; and calling out
122 // to an optimized GEMM kernel is not a huge win.
123 if (output_shape.dimensions(0) == 1 && operand_index == 1 &&
124 BytesInDimension(output_shape, 1) < kFusionThresholdBytes) {
125 VLOG(2) << "Fusing small matrix-vector product.";
126 return true;
127 } else if (output_shape.dimensions(1) == 1 && operand_index == 0 &&
128 BytesInDimension(output_shape, 0) < kFusionThresholdBytes) {
129 VLOG(2) << "Fusing small matrix-vector product.";
130 return true;
131 }
132 }
133 }
134
135 if (consumer->opcode() == HloOpcode::kFusion &&
136 consumer->fusion_kind() == HloInstruction::FusionKind::kLoop) {
137 VLOG(2) << "Fusing: consumer is a fusion node.";
138 return true;
139 }
140
141 if (CanBeLoopFused(*consumer)) {
142 VLOG(2) << "Fusing: consumer is elementwise or fusile.";
143 return true;
144 }
145
146 VLOG(2) << "Not fusing.";
147 return false;
148}
149
150HloInstruction::FusionKind CpuInstructionFusion::ChooseKind(
151 const HloInstruction* producer, const HloInstruction* consumer) {
152 return CanBeOutputFused(producer, consumer)
153 ? HloInstruction::FusionKind::kOutput
154 : HloInstruction::FusionKind::kLoop;
155}
