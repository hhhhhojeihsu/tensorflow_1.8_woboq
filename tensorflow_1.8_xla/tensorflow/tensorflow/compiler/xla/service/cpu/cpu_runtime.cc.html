81tensorflow::string ShapeString(const void* shape_ptr, xla::int32 shape_length) {
82 xla::StatusOr<xla::Shape> shape =
83 xla::llvm_ir::DecodeSelfDescribingShapeConstant(shape_ptr, shape_length);
84 if (shape.ok()) {
85 return xla::ShapeUtil::HumanStringWithLayout(shape.ValueOrDie());
86 }
87 return "<invalid shape>";
88}

92void* __xla_cpu_runtime_AcquireInfeedBufferForDequeue(xla::int32 buffer_length,
93 const void* shape,
94 xla::int32 shape_length) {
95 if (VLOG_IS_ON(2)) {
96 LOG(INFO) << "AcquireInfeedBufferForDequeue: "
97 << ShapeString(shape, shape_length);
98 }
99 xla::cpu::runtime::XfeedManager* xfeed = xla::cpu::runtime::GetXfeedManager();
100 // Wait until there's a buffer to dequeue.
101 xla::cpu::runtime::XfeedBuffer* buffer =
102 xfeed->infeed()->BlockingDequeueBuffer();
103 CHECK_EQ(buffer->length(), buffer_length)
104 << "XLA program infeed request buffer size " << buffer_length
105 << " did not match the runtime's infed buffer length " << buffer->length()
106 << "; program reports desired shape: "
107 << ShapeString(shape, shape_length);
108 return buffer->data();
109}

111void __xla_cpu_runtime_ReleaseInfeedBufferAfterDequeue(
112 xla::int32 buffer_length, void* buffer_ptr, const void* shape_ptr,
113 xla::int32 shape_length) {
114 if (VLOG_IS_ON(2)) {
115 LOG(INFO) << "ReleaseInfeedBufferAfterDeque: "
116 << ShapeString(shape_ptr, shape_length);
117 }
118 xla::cpu::runtime::XfeedManager* xfeed = xla::cpu::runtime::GetXfeedManager();
119 xla::StatusOr<xla::Shape> shape =
120 xla::llvm_ir::DecodeSelfDescribingShapeConstant(shape_ptr, shape_length);
121 xfeed->infeed()->ReleaseCurrentBuffer(buffer_length, buffer_ptr,
122 std::move(shape));
123}

125void* __xla_cpu_runtime_AcquireOutfeedBufferForPopulation(
126 xla::int32 buffer_length, const void* shape_ptr, xla::int32 shape_length) {
127 if (VLOG_IS_ON(2)) {
128 LOG(INFO) << "AcquireOutfeedBufferForPopulation: "
129 << ShapeString(shape_ptr, shape_length);
130 }
131 xla::cpu::runtime::XfeedManager* xfeed = xla::cpu::runtime::GetXfeedManager();
132 // Wait until there's a buffer to dequeue.
133 xla::cpu::runtime::XfeedBuffer* buffer =
134 xfeed->outfeed()->BlockingDequeueBuffer();
135 CHECK_EQ(buffer->length(), buffer_length)
136 << "XLA program outfeed request buffer size " << buffer_length
137 << " did not match the runtime's outfeed buffer length "
138 << buffer->length() << "; program reports outfed shape: "
139 << ShapeString(shape_ptr, shape_length);
140 return buffer->data();
141}
