21#include <string>
22
23#include "tensorflow/compiler/xla/service/buffer_assignment.h"

28namespace xla {
29
30// Returns a serialized representation of the HLO state.
31HloProto MakeHloProto(const HloModule& module,
32 const BufferAssignment& assignment);
33
34// Returns a serialized representation of the HLO state, but buffer assignment
35// will not be included in the output.
36HloProto MakeHloProto(const HloModule& module);
37
38// Returns the shapes of the parameters of the entry computation. Shape pointers
39// refer to shapes inside of the given HloProto.
40StatusOr<std::vector<const Shape*>> EntryComputationParameterShapes(
41 const HloProto& hlo_proto);
42
43// Returns the shape of the output of the entry computation. The shape pointer
44// refers to the output shape inside of the given HloProto.
45StatusOr<const Shape*> EntryComputationOutputShape(const HloProto& hlo_proto);

47} // namespace xla
