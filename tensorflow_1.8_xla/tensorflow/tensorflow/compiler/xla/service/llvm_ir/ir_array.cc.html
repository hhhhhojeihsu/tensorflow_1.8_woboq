32static void Delinearize(std::vector<llvm::Value*>* multidim,
33 llvm::Value* linear, const Shape& shape,
34 llvm::IRBuilder<>* ir_builder) {
35 int64 divisor = 1;
36 const Layout& layout = shape.layout();
37 for (int64 i = 0; i < layout.minor_to_major_size(); ++i) {
38 int64 dimension = layout.minor_to_major(i);
39 int64 size_of_current_dimension = shape.dimensions(dimension);
40
41 // If i is not the last dimension, compute
42 // (linear_index / divisor) % current_dimension.
43 // If i is the last dimension, we can skip the mod, because we assume that
44 // linear is in bounds.
45 //
46 // TODO(jlebar): We could add bounds checks here and elsewhere in this file,
47 // guarded under some sort of xla-memcheck flag. This might be particularly
48 // useful because cuda-memcheck can't help us much in XLA: Most of our
49 // memory lives in one big allocation, so cuda-memcheck can't detect
50 // out-of-bounds accesses.
51 auto* quot = ir_builder->CreateUDiv(linear, ir_builder->getInt64(divisor));
52 if (i < layout.minor_to_major_size() - 1) {
53 (*multidim)[dimension] = ir_builder->CreateURem(
54 quot, ir_builder->getInt64(size_of_current_dimension));
55 } else {
56 (*multidim)[dimension] = quot;
57 }
58 divisor *= size_of_current_dimension;
59 }
60}

62IrArray::Index::Index(llvm::Value* linear, const Shape& shape,
63 llvm::IRBuilder<>* ir_builder)
64 : multidim_(ShapeUtil::Rank(shape)),
65 linear_(linear),
66 layout_(shape.layout()),
67 dims_(shape.dimensions().begin(), shape.dimensions().end()) {
68 CHECK(LayoutUtil::HasLayout(shape))
69 << "Shape " << ShapeUtil::HumanStringWithLayout(shape)
70 << " should have a layout.";
71 Delinearize(&multidim_, linear, shape, ir_builder);
72}

95IrArray::IrArray(llvm::Value* base_ptr, const Shape& shape)
96 : base_ptr_(base_ptr), shape_(&shape) {
97 TF_CHECK_OK(ShapeUtil::ValidateShape(shape));
98 CHECK(base_ptr_->getType()->isPointerTy());
99 int depth = 0;
100 element_type_ =
101 llvm::cast<llvm::PointerType>(base_ptr_->getType())->getElementType();
102 while (llvm::ArrayType* array_type =
103 llvm::dyn_cast<llvm::ArrayType>(element_type_)) {
104 element_type_ = array_type->getElementType();
105 ++depth;
106 }
107
108 if (!ShapeUtil::IsArray(*shape_) || ShapeUtil::IsScalar(*shape_)) {
109 DCHECK(depth == 1 || depth == 0) << depth;
110 } else {
111 DCHECK_EQ(depth, ShapeUtil::Rank(*shape_)) << shape.ShortDebugString();
112 }
113}

115// Returns whether the given linear index is valid on the given shape.
116bool IrArray::Index::LinearValidOnShape(const Shape& a) const {
117 auto b = ShapeUtil::MakeShape(a.element_type(), dims_);
118 *b.mutable_layout() = layout_;
119 return linear_ != nullptr &&
120 ShapeUtil::ElementsIn(a) == ShapeUtil::ElementsIn(b) &&
121 ShapeUtil::ReshapeIsBitcast(a, b);
122}

124IrArray::Index IrArray::Index::SourceIndexOfReshape(
125 const Shape& output_shape, const Shape& input_shape,
126 llvm::IRBuilder<>* builder) const {
127 const auto& target_index = *this;
128 CHECK_EQ(target_index.size(), ShapeUtil::Rank(output_shape));
129 std::vector<std::pair<int64, int64>> common_factors =
130 CommonFactors(AsInt64Slice(input_shape.dimensions()),
131 AsInt64Slice(output_shape.dimensions()));
132 std::vector<llvm::Value*> source_multidim_index(
133 ShapeUtil::Rank(input_shape),
134 llvm::UndefValue::get(builder->getInt64Ty()));
135 // We compute the source indices in each common factor from only the target
136 // indices in the same common factor.
137 for (ssize_t k = common_factors.size() - 2; k >= 0; --k) {
138 llvm::Value* logical_linear_index =
139 Index(tensorflow::gtl::ArraySlice<llvm::Value*>(
140 multidim_, common_factors[k].second,
141 common_factors[k + 1].second - common_factors[k].second))
142 .Linearize(
143 tensorflow::gtl::ArraySlice<int64>(
144 AsInt64Slice(output_shape.dimensions()),
145 common_factors[k].second,
146 common_factors[k + 1].second - common_factors[k].second),
147 builder);
148 // Delinearizes logical_linear_index for the source array in row-major
149 // collapsed order. The first rank-1 indices are the remainder of the
150 // linear index by each dimension size.
151 for (int64 i = common_factors[k + 1].first - 1;
152 i >= common_factors[k].first; --i) {
153 llvm::Value* divisor = builder->getInt64(input_shape.dimensions(i));
154 if (input_shape.dimensions(i) == 1) {
155 source_multidim_index[i] = builder->getInt64(0);
156 } else if (i == common_factors[k].first) {
157 source_multidim_index[i] = logical_linear_index;
158 } else {
159 source_multidim_index[i] =
160 builder->CreateURem(logical_linear_index, divisor);
161 }
162 logical_linear_index = builder->CreateUDiv(logical_linear_index, divisor);
163 }
164 }
165
166 if (linear() != nullptr && LayoutUtil::HasLayout(input_shape) &&
167 LayoutUtil::HasLayout(output_shape) &&
168 ShapeUtil::ReshapeIsBitcast(input_shape, output_shape)) {
169 return Index(source_multidim_index, linear(), input_shape);
170 }
171 return Index(source_multidim_index);
172}

174IrArray::Index IrArray::Index::SourceIndexOfSlice(
175 const Shape& shape, tensorflow::gtl::ArraySlice<int64> starts,
176 tensorflow::gtl::ArraySlice<int64> strides,
177 llvm::IRBuilder<>* builder) const {
178 Index source_index(multidim_.size());
179 for (int i = 0; i < multidim_.size(); ++i) {
180 int64 stride = strides[i];
181 auto type = multidim_[i]->getType();
182
183 if (stride != 1) {
184 source_index[i] = builder->CreateAdd(
185 builder->CreateMul(multidim_[i],
186 llvm::ConstantInt::get(type, stride)),
187 llvm::ConstantInt::get(type, starts[i]));
188 } else {
189 source_index[i] = builder->CreateAdd(
190 multidim_[i], llvm::ConstantInt::get(type, starts[i]));
191 }
192 }
193 return source_index;
194}

196IrArray::Index IrArray::Index::SourceIndexOfTranspose(
197 const Shape& shape, const Shape& operand_shape,
198 tensorflow::gtl::ArraySlice<int64> dimension_mapping,
199 llvm::IRBuilder<>* builder) const {
200 std::vector<llvm::Value*> operand_multidim_index =
201 Permute(dimension_mapping, multidim());
202
203 if (linear() != nullptr && LayoutUtil::HasLayout(operand_shape) &&
204 LayoutUtil::HasLayout(shape) &&
205 ShapeUtil::TransposeIsBitcast(operand_shape, shape, dimension_mapping)) {
206 return Index(operand_multidim_index, linear(), operand_shape);
207 }
208
209 return Index(operand_multidim_index);
210}

212IrArray::Index IrArray::Index::SourceIndexOfBitcast(
213 const Shape& shape, const Shape& operand_shape,
214 llvm::IRBuilder<>* builder) const {
215 CHECK(LayoutUtil::HasLayout(shape) && LayoutUtil::HasLayout(operand_shape));
216 // In case the bitcast is just a reshape, we can use SourceIndexOfReshape()
217 // instead. This will reuse linear() if possible, so we don't have to build a
218 // new 'linear_index'.
219 if (ShapeUtil::ReshapeIsBitcast(operand_shape, shape)) {
220 return SourceIndexOfReshape(shape, operand_shape, builder);
221 }
222
223 // First linearize the index coming from the output of the bitcast. We want
224 // the physical index of the element in the buffer. This is like Linearize,
225 // but takes the layout into account.
226 int64 scale = 1;
227 llvm::Value* linear_index = builder->getInt64(0);
228 for (auto dimension : LayoutUtil::MinorToMajor(shape)) {
229 linear_index = builder->CreateAdd(
230 linear_index,
231 builder->CreateMul(multidim_[dimension], builder->getInt64(scale), "",
232 /*HasNUW=*/true, /*HasNSW=*/true),
233 "", /*HasNUW=*/true, /*HasNSW=*/true);
234 scale *= shape.dimensions(dimension);
235 }
236
237 // Now delinearize it for the input of the bitcast.
238 std::vector<llvm::Value*> multi_index(operand_shape.dimensions_size());
239 Delinearize(&multi_index, linear_index, operand_shape, builder);
240
241 return Index(multi_index, linear_index, operand_shape);
242}

244IrArray::Index IrArray::Index::SourceIndexOfBroadcast(
245 const Shape& shape, const Shape& operand_shape,
246 tensorflow::gtl::ArraySlice<int64> dimension_mapping,
247 llvm::IRBuilder<>* builder) const {
248 int64 rank = ShapeUtil::Rank(operand_shape);
249 std::vector<llvm::Value*> source_index(rank);
250 for (int64 i = 0; i < rank; ++i) {
251 source_index[i] = multidim_[dimension_mapping[i]];
252 }
253 if (linear_ == nullptr || !LayoutUtil::HasLayout(operand_shape) ||
254 !LayoutUtil::HasLayout(shape)) {
255 return Index(source_index);
256 }
257 // High-level idea: we can reuse the linear index if the broadcasted
258 // dimensions are contiguous, and this part of the operation is a bitcast.
259 // The other dimensions can be masked out with a div and a mod operation.
260 std::vector<int64> logical_to_physical =
261 LayoutUtil::MakeLogicalToPhysical(shape.layout());
262 int64 output_rank = ShapeUtil::Rank(shape);
263 // The minimum physical dimension that is broadcasted.
264 int64 min_broadcasted_dimension = output_rank;
265 // The maximum physical dimension that is broadcasted.
266 int64 max_broadcasted_dimension = -1;
267 for (int64 i = 0; i < rank; ++i) {
268 int64 physical_dim = logical_to_physical[dimension_mapping[i]];
269 min_broadcasted_dimension =
270 std::min(min_broadcasted_dimension, physical_dim);
271 max_broadcasted_dimension =
272 std::max(max_broadcasted_dimension, physical_dim);
273 }
274 bool contiguous_broadcast_dimensions =
275 max_broadcasted_dimension - min_broadcasted_dimension == rank - 1;
276 if (!contiguous_broadcast_dimensions) {
277 return Index(source_index);
278 }
279 // Check if the mapped dimensions are a bitcast.
280 std::vector<int64> operand_logical_to_physical =
281 LayoutUtil::MakeLogicalToPhysical(operand_shape.layout());
282 for (int64 i = 0; i < rank; ++i) {
283 if (operand_logical_to_physical[i] !=
284 logical_to_physical[dimension_mapping[i]] - min_broadcasted_dimension) {
285 return Index(source_index);
286 }
287 }
288 llvm::Value* linear = linear_;
289 int64 divisor = 1;
290 for (int64 i = max_broadcasted_dimension + 1; i < output_rank; ++i) {
291 divisor *= shape.dimensions(LayoutUtil::Major(shape.layout(), i));
292 }
293 if (divisor > 1) {
294 linear = builder->CreateUDiv(linear, builder->getInt64(divisor));
295 }
296 if (min_broadcasted_dimension > 0) {
297 int64 mod = 1;
298 for (int64 i = min_broadcasted_dimension; i <= max_broadcasted_dimension;
299 ++i) {
300 mod *= shape.dimensions(LayoutUtil::Major(shape.layout(), i));
301 }
302 linear = builder->CreateURem(linear, builder->getInt64(mod));
303 }
304 return Index(source_index, linear, operand_shape);
305}

307llvm::Value* IrArray::Index::Linearize(
308 tensorflow::gtl::ArraySlice<int64> dimensions,
309 llvm::IRBuilder<>* builder) const {
310 // Each dimension is multiplied by the product of the sizes of all
311 // earlier dimensions and added to the accumulator logical_linear_index.
312 llvm::Value* logical_linear_index = builder->getInt64(0);
313 int64 multiplier = 1;
314 for (ssize_t i = size() - 1; i >= 0; --i) {
315 llvm::Value* addend =
316 builder->CreateMul((*this)[i], builder->getInt64(multiplier), "",
317 /*HasNUW=*/true, /*HasNSW=*/true);
318 logical_linear_index = builder->CreateAdd(logical_linear_index, addend, "",
319 /*HasNUW=*/true, /*HasNSW=*/true);
320 multiplier *= dimensions[i];
321 }
322 return logical_linear_index;
323}

325llvm::Value* IrArray::EmitArrayElementAddress(
326 const IrArray::Index& index, llvm::IRBuilder<>* ir_builder,
327 tensorflow::StringPiece name) const {
328 if (ShapeUtil::IsScalar(*shape_)) {
329 // Special handling of scalars: a scalar pretends to have the same value for
330 // every index, thus effectively implementing broadcasting of its value
331 // over higher-rank arrays.
332 return base_ptr_;
333 }
334 CHECK_EQ(index.size(), ShapeUtil::Rank(*shape_));
335
336 std::vector<llvm::Value*> actual_index;
337 bool is_implicit_broadcast = false;
338 // We perform broadcasting when the operand shape has dimension(s) of size
339 // 1. In this case we fix the index value for that dimension to zero. This
340 // effectively broadcasts along this dimension.
341 for (int64 i = 0; i < index.size(); ++i) {
342 auto dim = shape_->dimensions(i);
343 actual_index.push_back(dim == 1 ? ir_builder->getInt64(0) : index[i]);
344 is_implicit_broadcast |= dim == 1;
345 }
346
347 if (!is_implicit_broadcast && index.LinearValidOnShape(*shape_)) {
348 llvm::Module* module =
349 ir_builder->GetInsertBlock()->getParent()->getParent();
350 return ir_builder->CreateInBoundsGEP(
351 ir_builder->CreateBitCast(
352 base_ptr_, PrimitiveTypeToIrType(shape_->element_type(), module)
353 ->getPointerTo()),
354 {index.linear()}, llvm_ir::AsStringRef(name));
355 }
356
357 // "base_ptr_" has the type of "<ir_type_for_its_shape>*"
358 // (e.g. [3 x [2 x float]]*). Therefore, the address of the indexed element
359 // should be computed by
360 //
361 // getelementptr base_ptr_, 0, most major index, ..., most minor index
362 std::vector<llvm::Value*> gep_indices(1, ir_builder->getInt64(0));
363 for (int64 i = 0; i < LayoutUtil::MinorToMajor(*shape_).size(); ++i) {
364 int64 dimension = LayoutUtil::Major(shape_->layout(), i);
365 gep_indices.push_back(actual_index[dimension]);
366 }
367 return ir_builder->CreateInBoundsGEP(base_ptr_, gep_indices,
368 llvm_ir::AsStringRef(name));
369}

371void IrArray::AnnotateLoadStoreInstructionWithMetadata(
372 llvm::Instruction* instruction) const {
373 CHECK(llvm::isa<llvm::LoadInst>(instruction) ||
374 llvm::isa<llvm::StoreInst>(instruction));
375 CHECK(!llvm::isa<llvm::StoreInst>(instruction) || !is_invariant_)
376 << "Trying to create a store to an invariant IRArray.";
377
378 for (const auto& kind_md_pair : metadata_) {
379 instruction->setMetadata(kind_md_pair.first, kind_md_pair.second);
380 }
381}

383llvm::Value* IrArray::EmitReadArrayElement(const Index& index,
384 llvm::IRBuilder<>* ir_builder,
385 tensorflow::StringPiece name) const {
386 llvm::Value* element_address =
387 EmitArrayElementAddress(index, ir_builder, name);
388 llvm::LoadInst* load = ir_builder->CreateLoad(element_address);
389 AnnotateLoadStoreInstructionWithMetadata(load);
390 return load;
391}

393void IrArray::EmitWriteArrayElement(const Index& index, llvm::Value* value,
394 llvm::IRBuilder<>* ir_builder) const {
395 llvm::Value* element_address = EmitArrayElementAddress(index, ir_builder);
396 llvm::StoreInst* store = ir_builder->CreateStore(value, element_address);
397 AnnotateLoadStoreInstructionWithMetadata(store);
398}

400IrArray IrArray::CastToShape(const Shape& new_shape,
401 llvm::IRBuilder<>* ir_builder) const {
402 llvm::Module* module = ir_builder->GetInsertBlock()->getParent()->getParent();
403 llvm::Type* new_ir_type = llvm_ir::ShapeToIrType(new_shape, module);
404 return IrArray(
405 ir_builder->CreatePointerCast(base_ptr_, new_ir_type->getPointerTo()),
406 new_shape);
407}

409/* static */ IrArray::Index IrArray::BumpIndex(const Index& index,
410 int64 which_dimension,
411 int64 addend,
412 llvm::IRBuilder<>* ir_builder) {
413 Index new_index = index;
414 new_index[which_dimension] = ir_builder->CreateAdd(
415 index[which_dimension], ir_builder->getInt64(addend), "", /*HasNUW=*/true,
416 /*HasNSW=*/true);
417 return new_index;
418}
