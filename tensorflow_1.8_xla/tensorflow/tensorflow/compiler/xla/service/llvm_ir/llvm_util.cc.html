48// Note, this function is only useful in an insertion context; in a global
49// (e.g. constants) context it will CHECK fail.
50llvm::Module* ModuleFromIRBuilder(llvm::IRBuilder<>* ir_builder) {
51 auto block = CHECK_NOTNULL(ir_builder->GetInsertBlock());
52 auto fn = CHECK_NOTNULL(block->getParent());
53 auto module = CHECK_NOTNULL(fn->getParent());
54 return module;
55}

67std::unique_ptr<llvm::Module> DropConstantInitializers(
68 const llvm::Module& module) {
69 std::unique_ptr<llvm::Module> cloned_module = CloneModule(module);
70 for (llvm::GlobalVariable& global_var : cloned_module->globals()) {
71 global_var.setInitializer(nullptr);
72 global_var.setLinkage(llvm::GlobalValue::LinkageTypes::ExternalLinkage);
73 }
74 return cloned_module;

85llvm::Value* EmitCallToIntrinsic(
86 llvm::Intrinsic::ID intrinsic_id,
87 tensorflow::gtl::ArraySlice<llvm::Value*> operands,
88 tensorflow::gtl::ArraySlice<llvm::Type*> overloaded_types,
89 llvm::IRBuilder<>* ir_builder) {
90 std::vector<llvm::Type*> types;
91 for (auto type : overloaded_types) {
92 types.push_back(type);
93 }
94 llvm::Module* module = ModuleFromIRBuilder(ir_builder);
95 llvm::Function* intrinsic =
96 llvm::Intrinsic::getDeclaration(module, intrinsic_id, types);
97 std::vector<llvm::Value*> operands_vec;
98 for (auto operand : operands) {
99 operands_vec.push_back(operand);
100 }
101 return ir_builder->CreateCall(intrinsic, operands_vec);
102}

104llvm::Value* EmitFloatMax(llvm::Value* lhs_value, llvm::Value* rhs_value,
105 llvm::IRBuilder<>* ir_builder) {
106 if (ir_builder->getFastMathFlags().noNaNs()) {
107 auto cmp = ir_builder->CreateFCmpUGE(lhs_value, rhs_value);
108 return ir_builder->CreateSelect(cmp, lhs_value, rhs_value);
109 } else {
110 auto cmp_ge = ir_builder->CreateFCmpOGE(lhs_value, rhs_value);
111 auto lhs_is_nan = ir_builder->CreateFCmpUNE(lhs_value, lhs_value);
112 auto sel_lhs = ir_builder->CreateOr(cmp_ge, lhs_is_nan);
113 return ir_builder->CreateSelect(sel_lhs, lhs_value, rhs_value);
114 }
115}

153llvm::Type* PrimitiveTypeToIrType(PrimitiveType element_type,
154 llvm::Module* module) {
155 switch (element_type) {
156 case PRED:
157 case S8:
158 case U8:
159 return llvm::Type::getInt8Ty(module->getContext());
160 case S16:
161 case U16:
162 case BF16:
163 // For BF16 we just need some type that is 16 bits wide so that it will
164 // take up the right amount of space in memory. LLVM does not have a BF16
165 // type (the LLVM half type is IEEE 16 bit floating point, not bfloat), so
166 // we can't map it directly to an LLVM type. We will not map a BF16
167 // addition to an addition on this type (int16) - this is just the type
168 // used for storage.
169 return llvm::Type::getInt16Ty(module->getContext());
170 case F16:
171 return llvm::Type::getHalfTy(module->getContext());
172 case S32:
173 case U32:
174 return llvm::Type::getInt32Ty(module->getContext());
175 case S64:
176 case U64:
177 return llvm::Type::getInt64Ty(module->getContext());
178 case F32:
179 return llvm::Type::getFloatTy(module->getContext());
180 case F64:
181 return llvm::Type::getDoubleTy(module->getContext());
182 case C64: {
183 auto cplx_t = module->getTypeByName("complex64");
184 if (cplx_t == nullptr) {
185 // C++ standard dictates the memory layout of std::complex is contiguous
186 // real followed by imaginary. C++11 section 26.4 [complex.numbers]:
187 // If z is an lvalue expression of type cv std::complex<T> then the
188 // expression reinterpret_cast<cv T(&)[2]>(z) shall be well-formed,
189 // reinterpret_cast<cv T(&)[2]>(z)[0] shall designate the real part of
190 // z, and reinterpret_cast<cv T(&)[2]>(z)[1] shall designate the
191 // imaginary part of z.
192 return llvm::StructType::create(
193 {llvm::Type::getFloatTy(module->getContext()),
194 llvm::Type::getFloatTy(module->getContext())},
195 "complex64", /*isPacked=*/true);
196 }
197 return cplx_t;
198 }
199 // A Tuple contains an array of pointers. Use i8*.
200 case TUPLE:
201 // An Opaque is like a void*, use i8*.
202 case OPAQUE:
203 return llvm::Type::getInt8PtrTy(module->getContext());
204 default:
205 LOG(FATAL) << "unsupported type " << element_type;
206 }
207}

209int GetSizeInBits(llvm::Type* type) {
210 const llvm::StructType* struct_ty = llvm::dyn_cast<llvm::StructType>(type);
211 if (struct_ty) {
212 CHECK(struct_ty->isPacked());
213 int bits = 0;
214 for (auto element_type : struct_ty->elements()) {
215 bits += GetSizeInBits(element_type);
216 }
217 return bits;
218 }
219 int bits = type->getPrimitiveSizeInBits();
220 CHECK_GT(bits, 0) << "type is not sized";
221 return bits;
222}

224llvm::Type* ShapeToIrType(const Shape& shape, llvm::Module* module) {
225 llvm::Type* result_type = PrimitiveTypeToIrType(shape.element_type(), module);
226 if (ShapeUtil::IsTuple(shape)) {
227 // A tuple buffer is an array of pointers.
228 result_type = llvm::ArrayType::get(result_type, shape.tuple_shapes_size());
229 } else if (ShapeUtil::IsArray(shape)) {
230 for (int64 dimension : LayoutUtil::MinorToMajor(shape)) {
231 result_type =
232 llvm::ArrayType::get(result_type, shape.dimensions(dimension));
233 }
234 }
235 return result_type;
236}

258// Recursively construct a multidimensional LLVM constant which represents the
259// given literal. The minor-to-major dimension ordering in the constant matches
260// that of the literal. For example, given a [2 x 3 x 4] Literal (dimension 0
261// has size 4, dimension 1 has size 3, etc) of primitive type F32 with a
262// minor_to_major value of [2, 1, 0] (column major), a LLVM constant of type
263// [4 x [3 x [2 x float]] will be returned.
264//
265// multi_index is a multidimensional index into the array. dimension_index is an
266// index into the minor_to_major field in the literal shape. This determines
267// which dimension is iterated over in this level of the recursion. Dimensions
268// are iterated from most major down to most minor (highest dimension_index
269// value down to zero).
270llvm::Constant* LiteralToConstant(const Literal& literal, int64 dimension_index,
271 std::vector<int64>* multi_index,
272 llvm::Module* module) {
273 const Shape& shape = literal.shape();
274 llvm::Type* ir_element_type =
275 llvm_ir::PrimitiveTypeToIrType(shape.element_type(), module);
276 if (dimension_index == -1) {
277 // Base case of the recursion. Index into the data field of the protobuf
278 // with the multi index.
279 llvm::Constant* value;
280 switch (shape.element_type()) {
281 case PRED:
282 value = llvm::ConstantInt::get(ir_element_type,
283 literal.Get<bool>(*multi_index));
284 break;
285 case U8:
286 value = llvm::ConstantInt::get(ir_element_type,
287 literal.Get<uint8>(*multi_index));
288 break;
289 case S32:
290 value = llvm::ConstantInt::get(ir_element_type,
291 literal.Get<int32>(*multi_index));
292 break;
293 case U32:
294 value = llvm::ConstantInt::get(ir_element_type,
295 literal.Get<uint32>(*multi_index));
296 break;
297 case S64:
298 value = llvm::ConstantInt::get(ir_element_type,
299 literal.Get<int64>(*multi_index));
300 break;
301 case U64:
302 value = llvm::ConstantInt::get(ir_element_type,
303 literal.Get<uint64>(*multi_index));
304 break;
305 case F32:
306 value = llvm::ConstantFP::get(ir_element_type,
307 literal.Get<float>(*multi_index));
308 break;
309 case BF16:
310 value = llvm::ConstantInt::get(
311 ir_element_type,
312 tensorflow::bit_cast<uint16>(literal.Get<bfloat16>(*multi_index)));
313 break;
314 case F16:
315 value = llvm::ConstantFP::get(
316 ir_element_type,
317 static_cast<float>(literal.Get<half>(*multi_index)));
318 break;
319 case F64:
320 value = llvm::ConstantFP::get(ir_element_type,
321 literal.Get<double>(*multi_index));
322 break;
323 case C64: {
324 complex64 x = literal.Get<complex64>(*multi_index);
325 value = llvm::ConstantStruct::get(
326 static_cast<llvm::StructType*>(ir_element_type),
327 llvm::ConstantFP::get(llvm_ir::PrimitiveTypeToIrType(F32, module),
328 x.real()),
329 llvm::ConstantFP::get(llvm_ir::PrimitiveTypeToIrType(F32, module),
330 x.imag()));
331 break;
332 }
333 default:
334 LOG(FATAL) << "unsupported type " << shape.element_type();
335 }
336 return value;
337 }
338
339 // The dimension index starts at the one less than the rank of the array and
340 // decrements with each recursive call. We want to iterate through the
341 // dimensions in major-to-minor order as we recurse so just index into
342 // minor_to_major to get the dimension number for this level of the recursion.
343 int64 dimension = LayoutUtil::Minor(shape.layout(), dimension_index);
344
345 // Recursively call LiteralToConstant to construct subarrays for the
346 // more-minor dimensions. Gather the subarrays into a vector for bundling into
347 // a new (higher-dimensional) ConstantArray.
348 std::vector<llvm::Constant*> elements;
349 for (int64 i = 0; i < shape.dimensions(dimension); ++i) {
350 (*multi_index)[dimension] = i;
351 elements.push_back(
352 LiteralToConstant(literal, dimension_index - 1, multi_index, module));
353 }
354
355 llvm::Type* element_type;
356 if (elements.empty()) {
357 element_type = ir_element_type;
358 for (int i = 0; i < dimension_index; ++i) {
359 int64 index = LayoutUtil::Minor(shape.layout(), i);
360 element_type =
361 llvm::ArrayType::get(element_type, shape.dimensions(index));
362 }
363 } else {
364 element_type = elements[0]->getType();
365 }
366 llvm::ArrayType* aggregate_type =
367 llvm::ArrayType::get(element_type, shape.dimensions(dimension));
368 return llvm::ConstantArray::get(aggregate_type, elements);
369}

373llvm::Constant* ConvertLiteralToIrConstant(const Literal& literal,

378 &multi_index, module);

390llvm::AllocaInst* EmitAllocaAtFunctionEntryWithCount(
391 llvm::Type* type, llvm::Value* element_count, tensorflow::StringPiece name,
392 llvm::IRBuilder<>* ir_builder, int alignment) {
393 llvm::IRBuilder<>::InsertPoint insert_point = ir_builder->saveIP();
394 llvm::Function* function = ir_builder->GetInsertBlock()->getParent();
395 ir_builder->SetInsertPoint(&function->getEntryBlock(),
396 function->getEntryBlock().getFirstInsertionPt());
397 llvm::AllocaInst* alloca =
398 ir_builder->CreateAlloca(type, element_count, AsStringRef(name));
399 if (alignment != 0) {
400 alloca->setAlignment(alignment);
401 }
402 ir_builder->restoreIP(insert_point);
403 return alloca;
404}

416LlvmIfData EmitIfThenElse(llvm::Value* condition, tensorflow::StringPiece name,
417 llvm::IRBuilder<>* ir_builder, bool emit_else) {
418 llvm_ir::LlvmIfData if_data;
419 if_data.if_block = ir_builder->GetInsertBlock();
420 if_data.true_block = CreateBasicBlock(
421 nullptr, tensorflow::strings::StrCat(name, "-true"), ir_builder);
422 if_data.false_block =
423 emit_else ? CreateBasicBlock(nullptr,
424 tensorflow::strings::StrCat(name, "-false"),
425 ir_builder)
426 : nullptr;
427
428 // Add a terminator to the if block, if necessary.
429 if (if_data.if_block->getTerminator() == nullptr) {
430 ir_builder->SetInsertPoint(if_data.if_block);
431 if_data.after_block = CreateBasicBlock(
432 nullptr, tensorflow::strings::StrCat(name, "-after"), ir_builder);
433 ir_builder->CreateBr(if_data.after_block);
434 } else {
435 if_data.after_block = if_data.if_block->splitBasicBlock(
436 ir_builder->GetInsertPoint(),
437 AsStringRef(tensorflow::strings::StrCat(name, "-after")));
438 }
439
440 // Our basic block should now end with an unconditional branch. Remove it;
441 // we're going to replace it with a conditional branch.
442 if_data.if_block->getTerminator()->eraseFromParent();
443
444 ir_builder->SetInsertPoint(if_data.if_block);
445 ir_builder->CreateCondBr(
446 condition, if_data.true_block,
447 emit_else ? if_data.false_block : if_data.after_block);
448
449 ir_builder->SetInsertPoint(if_data.true_block);
450 ir_builder->CreateBr(if_data.after_block);
451
452 if (emit_else) {
453 ir_builder->SetInsertPoint(if_data.false_block);
454 ir_builder->CreateBr(if_data.after_block);
455 }
456
457 ir_builder->SetInsertPoint(if_data.after_block,
458 if_data.after_block->getFirstInsertionPt());
459
460 return if_data;
461}

463llvm::Value* EmitComparison(llvm::CmpInst::Predicate predicate,
464 llvm::Value* lhs_value, llvm::Value* rhs_value,
465 llvm::IRBuilder<>* ir_builder) {
466 llvm::Value* comparison_result;
467 if (lhs_value->getType()->isIntegerTy()) {
468 comparison_result = ir_builder->CreateICmp(predicate, lhs_value, rhs_value);
469 } else {
470 comparison_result = ir_builder->CreateFCmp(predicate, lhs_value, rhs_value);
471 }
472 // comparison_result is i1, but the NVPTX codegen incorrectly lowers i1
473 // arrays. So we extend it to i8 so that it's addressable.
474 return ir_builder->CreateZExt(
475 comparison_result,
476 llvm_ir::PrimitiveTypeToIrType(PRED, ModuleFromIRBuilder(ir_builder)));
477}

552string SanitizeFunctionName(string function_name) {
553 // The backend with the strictest requirements on function names is NVPTX, so
554 // we sanitize to its requirements.
555 //
556 // A slightly stricter version of the NVPTX requirements is that names match
557 // /[a-zA-Z_$][a-zA-Z0-9_$]*/, with the exception that the names "_" and "$"
558 // are illegal.
559
560 // Sanitize chars in function_name.
561 std::transform(function_name.begin(), function_name.end(),
562 function_name.begin(), [](char c) {
563 if (('a' <= c && c <= 'z') || ('A' <= c && c <= 'Z') ||
564 ('0' <= c && c <= '9') || c == '_' || c == '$') {
565 return c;
566 }
567 return '_';
568 });
569
570 // Ensure the name isn't empty.
571 if (function_name.empty()) {
572 function_name = "__unnamed";
573 }
574
575 // Ensure the name doesn't start with a number.
576 if (!function_name.empty() && function_name[0] >= '0' &&
577 function_name[0] <= '9') {
578 function_name.insert(function_name.begin(), '_');
579 }
580
581 // Ensure the name isn't "_" or "$".
582 if (function_name == "_" || function_name == "$") {
583 function_name += '_';
584 }
585
586 return function_name;
587}

601llvm::Value* CreateRor(llvm::Value* rotand, llvm::Value* rotor,
602 llvm::IRBuilder<>* builder) {
603 auto size = rotand->getType()->getPrimitiveSizeInBits();
604 auto size_value = builder->getIntN(size, size);
605 auto mod = [=](llvm::Value* x) { return builder->CreateURem(x, size_value); };
606 return builder->CreateOr(
607 builder->CreateShl(rotand, mod(builder->CreateSub(size_value, rotor))),
608 builder->CreateLShr(rotand, mod(rotor)));
609}

636std::map<int, llvm::MDNode*> MergeMetadata(
637 llvm::LLVMContext* context, const std::map<int, llvm::MDNode*>& a,
638 const std::map<int, llvm::MDNode*>& b) {
639 // We should extend this as needed to deal with other kinds of metadata like
640 // !dereferenceable and !range.
641
642 std::map<int, llvm::MDNode*> result;
643 for (auto kind_md_pair : a) {
644 if (kind_md_pair.first == llvm::LLVMContext::MD_alias_scope) {
645 llvm::SmallVector<llvm::Metadata*, 8> union_of_scopes;
646 llvm::SmallPtrSet<llvm::Metadata*, 8> scope_set;
647 for (const auto& scope_a : kind_md_pair.second->operands()) {
648 scope_set.insert(llvm::cast<llvm::MDNode>(scope_a.get()));
649 union_of_scopes.push_back(llvm::cast<llvm::MDNode>(scope_a.get()));
650 }
651 auto it = b.find(kind_md_pair.first);
652 if (it != b.end()) {
653 for (const auto& scope_b : it->second->operands()) {
654 if (!scope_set.count(llvm::cast<llvm::MDNode>(scope_b.get()))) {
655 union_of_scopes.push_back(llvm::cast<llvm::MDNode>(scope_b.get()));
656 }
657 }
658 }
659 result[llvm::LLVMContext::MD_alias_scope] =
660 llvm::MDNode::get(*context, union_of_scopes);
661 } else if (kind_md_pair.first == llvm::LLVMContext::MD_noalias) {
662 llvm::SmallVector<llvm::Metadata*, 8> intersection_of_scopes;
663 llvm::SmallPtrSet<llvm::Metadata*, 8> scope_set;
664 for (const auto& scope_a : kind_md_pair.second->operands()) {
665 scope_set.insert(llvm::cast<llvm::MDNode>(scope_a.get()));
666 }
667 auto it = b.find(kind_md_pair.first);
668 if (it != b.end()) {
669 for (const auto& scope_b : it->second->operands()) {
670 if (scope_set.count(llvm::cast<llvm::MDNode>(scope_b))) {
671 intersection_of_scopes.push_back(llvm::cast<llvm::MDNode>(scope_b));
672 }
673 }
674 }
675 if (!intersection_of_scopes.empty()) {
676 result[llvm::LLVMContext::MD_noalias] =
677 llvm::MDNode::get(*context, intersection_of_scopes);
678 }
679 }
680 }
681 return result;
682}

705Status DumpIRToDirectory(const string& directory_name,
706 const string& hlo_module_name,
707 const llvm::Module& llvm_module, bool optimized) {
708 // We can end up compiling different modules with the same name when using
709 // XlaJitCompiledCpuFunction::Compile. Avoid overwriting IR files previously
710 // dumped from the same process in such cases.
711 string unique_and_safe_file_name = GetProcessUniqueIrFileName(
712 tensorflow::strings::StrCat("ir-", SanitizeFileName(hlo_module_name), "-",
713 optimized ? "with" : "no", "-opt"));
714
715 string ir_file_name = tensorflow::io::JoinPath(
716 directory_name,
717 tensorflow::strings::StrCat(unique_and_safe_file_name, ".ll"));
718
719 // For some models the embedded constants can be huge, so also dump the module
720 // with the constants stripped to get IR that is easier to manipulate.
721 string ir_no_constant_initializers_file_name = tensorflow::io::JoinPath(
722 directory_name,
723 tensorflow::strings::StrCat(unique_and_safe_file_name, "-noconst.ll"));
724
725 TF_RETURN_IF_ERROR(CreateAndWriteStringToFile(
726 directory_name, ir_file_name, DumpModuleToString(llvm_module)));
727 return CreateAndWriteStringToFile(
728 directory_name, ir_no_constant_initializers_file_name,
729 DumpModuleToString(*DropConstantInitializers(llvm_module)));
730}

732llvm::Function* CreateFunction(llvm::FunctionType* function_type,
733 llvm::GlobalValue::LinkageTypes linkage,
734 bool enable_fast_math, bool optimize_for_size,
735 tensorflow::StringPiece name,
736 llvm::Module* module) {
737 llvm::Function* function =
738 llvm::Function::Create(function_type, linkage, AsStringRef(name), module);
739 function->setCallingConv(llvm::CallingConv::C);
740 function->addFnAttr("no-frame-pointer-elim", "false");
741
742 if (enable_fast_math) {
743 function->addFnAttr("unsafe-fp-math", "true");
744 function->addFnAttr("no-infs-fp-math", "true");
745 function->addFnAttr("no-nans-fp-math", "true");
746 function->addFnAttr("no-signed-zeros-fp-math", "true");
747 }
748
749 // Add the optize attribute to the function if optimizing for size. This
750 // controls internal behavior of some optimization passes (e.g. loop
751 // unrolling).
752 if (optimize_for_size) {
753 function->addFnAttr(llvm::Attribute::OptimizeForSize);
754 }
755
756 return function;
757}

759void InitializeLLVMCommandLineOptions(const HloModuleConfig& config) {
760 auto options = config.debug_options().xla_backend_extra_options();
761 if (!options.empty()) {
762 std::vector<string> fake_argv_storage;
763 fake_argv_storage.push_back("");
764 for (const auto& it : options) {
765 // Skip options the XLA backend itself consumes.
766 if (!tensorflow::str_util::StartsWith(it.first, "xla_")) {
767 if (it.second.empty()) {
768 fake_argv_storage.push_back(it.first);
769 } else {
770 fake_argv_storage.push_back(it.first + "=" + it.second);
771 }
772 }
773 }
774
775 VLOG(2) << "Passing argv to LLVM:";
776 std::vector<const char*> fake_argv;
777 for (const auto& s : fake_argv_storage) {
778 fake_argv.push_back(s.c_str());
779 VLOG(2) << s;
780 }
781 llvm::cl::ParseCommandLineOptions(fake_argv.size(), &fake_argv[0]);
782 }
