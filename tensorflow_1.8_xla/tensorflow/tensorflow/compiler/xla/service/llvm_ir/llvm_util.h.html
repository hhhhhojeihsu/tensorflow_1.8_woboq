56template <typename T>
57llvm::ArrayRef<T> AsArrayRef(const std::vector<T>& vec) {
58 return llvm::ArrayRef<T>(vec.data(), vec.size());
59}

61template <typename T>
62llvm::ArrayRef<T> AsArrayRef(const tensorflow::gtl::ArraySlice<T>& slice) {

66// Dump the given LLVM entity to a string. This works for Types and Values.
67template <typename T>
68string DumpToString(const T& entity) {
69 std::string buffer_string;
70 llvm::raw_string_ostream ostream(buffer_string);
71 entity.print(ostream);
72 ostream.flush();
73 return AsString(buffer_string);
74}

183// Struct with data on a conditional branch in a diamond shape created
184// via EmitIfThenElse.
185struct LlvmIfData {
186 // The block that has the conditional branch.
187 llvm::BasicBlock* if_block;
188
189 // The block that is executed if the condition is true.
190 llvm::BasicBlock* true_block;
191
192 // The block that is executed if the condition is false.
193 llvm::BasicBlock* false_block;
194
195 // The block that follows after both the true_block and the
196 // false_block.
197 llvm::BasicBlock* after_block;
198};
199
200// Inserts a diamond-shaped if-then-else construct at the current
201// insertion point of the builder. This involves splitting the current
202// block into two blocks, at the insertion point, and introducing a
203// true-block and a false-block that connect the two split pieces. The
204// true-block is executed if the condition parameter evaluates to true
205// and otherwise the false-block is executed. If `emit_else` is false,
206// it jumps to the after-block rather than the false-block if the
207// condition is false, and the returned `false_block` is null.
208//
209// Currently the insertion point of the builder must be a well-formed
210// block with a terminator. If you need to use this for a
211// non-terminated block, just make the function able to do that too.
212LlvmIfData EmitIfThenElse(llvm::Value* condition, tensorflow::StringPiece name,
213 llvm::IRBuilder<>* ir_builder, bool emit_else = true);
