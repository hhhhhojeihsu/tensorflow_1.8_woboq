16// Implementation note:
17//
18// The general idea behind this pass is that we're converting from this:
19// %param.A = OldShape
20// %param.B = OldShape
21// %reshape.A = NewShape reshape(%param.A)
22// %reshape.B = NewShape reshape(%param.B)
23// %instruction = NewShape instruction(%reshape.A, %reshape.B)
24// To this:
25// %param.A = OldShape
26// %param.B = OldShape
27// %instruction = OldShape instruction(%param.A, %param.B)
28// %reshape = NewShape reshape(%instruction)
29//
30// Where the instruction must be elementwise, and both reshapes and transposes
31// are moved.
32//
33// Most elementwise instructions support implicit broadcast of scalar operands,
34// but select is a special-case. The signature is Select(Pred, A, B), and the
35// only implicit scalar broadcast is on Pred, not on A or B. Since reshapes or
36// transposes to a scalar should be cheap, we simply never move them.
37

51bool IsReshapeOrTranspose(const HloInstruction* instruction) {
52 return instruction->opcode() == HloOpcode::kReshape ||
53 instruction->opcode() == HloOpcode::kTranspose;
54}

56// Returns true if `instruction` can change its shape simply by adjusting
57// metadata or if `instruction` is a broadcast of a scalar value.
58bool CanTriviallyChangeShape(const HloInstruction* instruction) {
59 // NOTE: Technically a sequence of reshape(reshape(constant)) is also
60 // trivially reshapable, so we might be tempted to simply recurse if
61 // IsReshapeOrTranspose(instruction)==true.
62 //
63 // But it's not that simple. E.g. reshape(reshape(rng)) is only trivially
64 // reshapable if *all* instructions in the chain have user_count == 1. And
65 // reshape(scalar) isn't trivial at all if the reshape itself isn't scalar; we
66 // rely on implicit scalar broadcast for scalars to be trivial. In addition,
67 // these cases make it harder to maintain correctness of the UpdateOperand
68 // logic below.
69 //
70 // So don't handle these chains, unless you update the tests and code to deal
71 // with these properly. One idea is to add a pass immediately beforehand that
72 // collapses trivial runs of reshapes / transposes.
73
74 // Scalars can operate with any shape.
75 if (ShapeUtil::IsScalar(instruction->shape())) {
76 return true;
77 }
78
79 // A constant can trivially reshape the literal it holds.
80 if (instruction->opcode() == HloOpcode::kConstant) {
81 return true;
82 }
83
84 // An Rng instruction can be any shape as long as it has one user. Two copies
85 // of the same Rng would be problematic if an Rng of a different shape would
86 // produce random numbers in a different order.
87 if (instruction->opcode() == HloOpcode::kRng &&
88 instruction->user_count() == 1) {
89 return true;
90 }
91
92 // A broadcase of scalar can trivially change its shape.
93 if (instruction->opcode() == HloOpcode::kBroadcast &&
94 ShapeUtil::IsScalar(instruction->operand(0)->shape())) {
95 return true;
96 }
97
98 return false;
99}

101// Returns true iff `instruction` is a reshape/transpose instruction for which
102// a shape change is nontrivial.
103bool IsNontrivialReshape(const HloInstruction* instruction) {
104 return !ShapeUtil::IsScalar(instruction->shape()) &&
105 IsReshapeOrTranspose(instruction) &&
106 !CanTriviallyChangeShape(instruction->operand(0));
107}

109// Finds the first operand of an instruction that is a non-trivial reshape or
110// transpose. Returns such an operand or nullptr if not found.
111HloInstruction* FirstNonScalarAndNonTrivialReshapeOperand(
112 const HloInstruction* hlo) {
113 for (HloInstruction* operand : hlo->operands()) {
114 if (IsNontrivialReshape(operand)) {
115 VLOG(5) << "Found first non-trivial reshape operand of "
116 << hlo->ToString(HloPrintOptions().set_print_metadata(false))
117 << ":\n\t"
118 << operand->ToString(HloPrintOptions().set_print_metadata(false));
119 return operand;
120 }
121 }
122 return nullptr;
123}

125// Returns whether `a` and `b` are equivalent reshapes/transposes.
126bool AreEquivalentReshapes(const HloInstruction* a, const HloInstruction* b) {
127 if (a->opcode() != b->opcode() ||
128 !ShapeUtil::SameDimensions(a->shape(), b->shape())) {
129 return false;
130 }
131 switch (a->opcode()) {
132 case HloOpcode::kTranspose:
133 return a->dimensions() == b->dimensions();
134 case HloOpcode::kReshape:
135 return ShapeUtil::SameDimensions(a->operand(0)->shape(),
136 b->operand(0)->shape());
137 default:
138 return false;
139 }
140}

142// This function is called once we've decided to sink reshape/transpose operands
143// across an instruction. It returns an updated `operand` with a shape that
144// plays nicely with `new_operand_shape`; either it has the same shape (of the
145// correct type), or it is a scalar that may be implicitly broadcast.
146HloInstruction* UpdateOperand(const HloInstruction* first_reshape_operand,
147 const Shape& new_operand_shape,
148 HloInstruction* operand) {
149 HloComputation* computation = operand->parent();
150 const PrimitiveType element_type = operand->shape().element_type();
151 const Shape new_shape =
152 ShapeUtil::ChangeElementType(new_operand_shape, element_type);
153
154 switch (operand->opcode()) {
155 case HloOpcode::kConstant: {
156 if (first_reshape_operand->opcode() == HloOpcode::kReshape) {
157 VLOG(5) << "Adding reshape to kConstant operand";
158 return computation->AddInstruction(
159 HloInstruction::CreateReshape(new_shape, operand));
160 } else {
161 CHECK(first_reshape_operand->opcode() == HloOpcode::kTranspose);
162 VLOG(5) << "Adding transpose to kConstant operand";
163 std::vector<int64> inverse_permutation =
164 InversePermutation(first_reshape_operand->dimensions());
165 return computation->AddInstruction(HloInstruction::CreateTranspose(
166 new_shape, operand, inverse_permutation));
167 }
168 }
169 case HloOpcode::kRng: {
170 CHECK_EQ(operand->user_count(), 1);
171 VLOG(5) << "Cloning kRng operand with new shape";
172 return computation->AddInstruction(
173 operand->CloneWithNewOperands(new_shape, operand->operands()));
174 }
175 case HloOpcode::kReshape:
176 case HloOpcode::kTranspose: {
177 VLOG(5) << "Using existing operand of kReshape or kTranspose";
178 return operand->mutable_operand(0);
179 }
180 case HloOpcode::kBroadcast: {
181 CHECK(ShapeUtil::IsScalar(operand->operand(0)->shape()));
182 HloInstruction* inst = computation->AddInstruction(
183 operand->CloneWithNewOperands(new_shape, operand->operands()));
184 VLOG(5) << "Changing broadcast from " << operand->ToString() << " to "
185 << inst->ToString();
186 return inst;
187 }
188
189 default:
190 LOG(FATAL) << "Unexpected operand opcode during update: " << operand;
191 }
192}

194// Actually performs the reshape-move transformation -- that is, sinks the
195// reshape or transpose operands of `instruction` across it.
196StatusOr<bool> PerformSinkReshapeOrTranspose(
197 HloInstruction* instruction, const HloInstruction* first_reshape_operand) {
198 auto print_no_metadata = HloPrintOptions().set_print_metadata(false);
199 // At this point we've decided to sink reshape/transpose operands.
200 const Shape& new_operand_shape = first_reshape_operand->operand(0)->shape();
201 VLOG(3) << "** Sinking reshape or transpose: "
202 << instruction->ToString(print_no_metadata)
203 << "\n\tfirst reshape operand: "
204 << first_reshape_operand->ToString(print_no_metadata)
205 << "\n\tnew operand shape: "
206 << ShapeUtil::HumanString(new_operand_shape);
207
208 auto operands = instruction->operands();
209 for (size_t i = 0; i < operands.size(); ++i) {
210 // All scalar operands remain as-is, even if they're reshape or transpose,
211 // to simplify handling wrt special scalar broadcast rules for ops like
212 // Select. Scalar reshapes should be cheap anyways.
213 if (ShapeUtil::IsScalar(operands[i]->shape())) {
214 continue;
215 }
216 VLOG(3) << "Updating operand #" << i << ": "
217 << operands[i]->ToString(print_no_metadata);
218 operands[i] =
219 UpdateOperand(first_reshape_operand, new_operand_shape, operands[i]);
220 }
221 if (HloOpcode::kFusion == instruction->opcode()) {
222 // Here we already know `instruction` is elementwise, and no operand is
223 // implicit broadcast as if it were the operands would not have easy shape
224 // changes, so all the fused instructions have the same dimensions.
225 for (const auto& fused_instruction : instruction->fused_instructions()) {
226 Shape* shape = fused_instruction->mutable_shape();
227 *shape->mutable_dimensions() = new_operand_shape.dimensions();
228 *shape->mutable_layout() = new_operand_shape.layout();
229 }
230 }
231 HloComputation* computation = instruction->parent();
232 HloInstruction* new_elementwise =
233 computation->AddInstruction(instruction->CloneWithNewOperands(
234 // `instruction` may change the element type, e.g., from
235 // operands[0] -> reshape -> convert (`instruction`)
236 // to
237 // operands[0] -> convert' -> reshape'
238 //
239 // In this case, convert' should have the same element type as
240 // `convert` and the same dimensions as operands[0].
241 ShapeUtil::ChangeElementType(new_operand_shape,
242 instruction->shape().element_type()),
243 operands));
244
245 std::unique_ptr<HloInstruction> new_reshape;
246 switch (first_reshape_operand->opcode()) {
247 case HloOpcode::kReshape:
248 VLOG(3) << "Creating new reshape for new elementwise op: "
249 << new_elementwise->ToString(print_no_metadata);
250 new_reshape =
251 HloInstruction::CreateReshape(instruction->shape(), new_elementwise);
252 break;
253 case HloOpcode::kTranspose:
254 new_reshape =
255 HloInstruction::CreateTranspose(instruction->shape(), new_elementwise,
256 first_reshape_operand->dimensions());
257 break;
258 default:
259 LOG(FATAL) << "Bad opcode";
260 }
261 TF_RETURN_IF_ERROR(computation->ReplaceWithNewInstruction(
262 instruction, std::move(new_reshape)));
263 return true;
264}

266// Returns true if the instruction is a reshape-move candidate.
267//
268// An instruction is a reshape-move candidate if the instruction is elementwise,
269// has at least one nontrivial reshape/transpose operand, and its operands are
270// either trivially reshapable or are equivalent nontrivial reshapes/transposes.
271bool IsReshapeMoveCandidate(HloInstruction* instruction) {
272 auto print_no_metadata = HloPrintOptions().set_print_metadata(false);
273 VLOG(5) << "** Checking instruction: "
274 << instruction->ToString(print_no_metadata);
275
276 // Only perform reshape-move for live elementwise instructions with operands.
277 const bool is_dead = instruction->user_count() == 0 &&
278 instruction != instruction->parent()->root_instruction();
279 if (!instruction->IsElementwise() || instruction->operands().empty() ||
280 is_dead) {
281 return false;
282 }
283
284 // Check whether all operands:
285 // 0. Have the same dimensions as the output -- if not, they may be
286 // implicitly broadcast, which can confound the movement's
287 // correctness.
288 //
289 // And one of the following:
290 // 1. Are reshapes or transposes that have the same input and
291 // output shapes as all other reshaped or transposed operands.
292 // or
293 // 2. Are one of kConstant, kRng, broadcast of a scalar value, and scalars
294 // that can change shape trivially.
295 const HloInstruction* first_reshape_operand = nullptr;
296 for (const HloInstruction* operand : instruction->operands()) {
297 if (!ShapeUtil::SameDimensions(operand->shape(), instruction->shape())) {
298 VLOG(5) << "Operand shape differs from output shape; may be "
299 "implicitly broadcast, so preventing "
300 "movement\n\toperand: "
301 << operand->ToString(print_no_metadata) << "\n\tinstruction: "
302 << instruction->ToString(print_no_metadata);
303 return false;
304 }
305
306 if (CanTriviallyChangeShape(operand)) {
307 VLOG(5) << "Operand can trivially change shape: "
308 << operand->ToString(print_no_metadata);
309 continue;
310 }
311
312 if (!IsNontrivialReshape(operand)) {
313 VLOG(5) << "Operand can't trivially change shape: "
314 << operand->ToString(print_no_metadata);
315 return false;
316 }
317
318 if (first_reshape_operand == nullptr) {
319 first_reshape_operand = operand;
320 VLOG(5) << "First reshape operand "
321 << operand->ToString(print_no_metadata);
322 } else if (AreEquivalentReshapes(first_reshape_operand, operand)) {
323 VLOG(5)
324 << "Operand is an equivalent reshape of the first reshape operand "
325 << operand->ToString(print_no_metadata);
326 } else {
327 // TODO(someone): Look into supporting general ops for the operands as
328 // well.
329 VLOG(5) << "Operand is a reshape but is not equivalent to the first "
330 "Reshape operand"
331 << operand->ToString(print_no_metadata);
332 return false;
333 }
334 }
335
336 if (first_reshape_operand) {
337 VLOG(5) << "All operands have easy shape changes: "
338 << instruction->ToString(print_no_metadata);
339 }
340
341 return first_reshape_operand != nullptr;
342}

344// Reshape-moves all qualifying instructions in reshape_candidates. Returns
345// true if it makes changes.
346//
347// `reshape_candidates` is a set of HloInstructions with nontrivial reshape
348// operands, and a instruction in the set can be reshape-moved iff all the users
349// of its nontrivial reshape operands can also be reshaped-moved.
350//
351// The algorithm here iteratively finds the nontrivial operands with users that
352// are outside the set of `reshape_candidates`, and removes their users from
353// `reshape_candidates`, until either `reshape_candidates` becomes empty or none
354// of the remaining nontrivial operands have users outside `reshape_candidates`.
355// In the later case, all the remaining instructions in `reshape_candidates`
356// are reshape-moved and the routine returns true.
357StatusOr<bool> TryReshapeMoveOnCandidates(
358 HloInstructionSet* reshape_candidates) {
359 bool removed = true;
360 while (!reshape_candidates->empty() && removed) {
361 if (VLOG_IS_ON(5)) {
362 for (const HloInstruction* instruction : *reshape_candidates) {
363 VLOG(5) << "candidate " << instruction->ToString();
364 }
365 }
366 ConstHloInstructionSet nontrivial_operands;
367 for (const HloInstruction* instruction : *reshape_candidates) {
368 for (const auto* operand : instruction->operands()) {
369 if (IsNontrivialReshape(operand)) {
370 nontrivial_operands.insert(operand);
371 }
372 }
373 }
374
375 removed = false;
376 for (auto operand : nontrivial_operands) {
377 if (c_any_of(operand->users(), [&](HloInstruction* user) {
378 return !reshape_candidates->count(user);
379 })) {
380 for (auto* user : operand->users()) {
381 removed |= reshape_candidates->erase(user) > 0;
382 }
383 }
384 }
385 }
386
387 if (reshape_candidates->empty()) {
388 return false;
389 }
390 for (HloInstruction* instruction : *reshape_candidates) {
391 const HloInstruction* first_reshape_operand =
392 FirstNonScalarAndNonTrivialReshapeOperand(instruction);
393 TF_ASSIGN_OR_RETURN(
394 bool did_change,
395 PerformSinkReshapeOrTranspose(instruction, first_reshape_operand));
396 CHECK(did_change);
397 }
398 return true;
399}

403StatusOr<bool> ReshapeMover::Run(HloModule* module) {
404 bool changed = false;
405 VLOG(2) << "Pre ReshapeMover HLO:";
406 XLA_VLOG_LINES(2, module->ToString());
407 for (auto* comp : module->MakeNonfusionComputations()) {
408 HloInstructionSet reshape_candidates;
409 for (HloInstruction* instruction : comp->instructions()) {
410 if (IsReshapeMoveCandidate(instruction)) {
411 reshape_candidates.insert(instruction);
412 }
413 }
414 TF_ASSIGN_OR_RETURN(bool did_change,
415 TryReshapeMoveOnCandidates(&reshape_candidates));
416 changed |= did_change;
417 }
418 VLOG(2) << "Post ReshapeMover HLO:";
419 XLA_VLOG_LINES(2, module->ToString());
420 return changed;
421}
