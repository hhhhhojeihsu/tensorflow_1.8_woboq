39StatusOr<int64> MinimumMemoryForSequence(
40 const SequentialHloOrdering::HloModuleSequence& module_sequence,
41 const LogicalBuffer::SizeFunction& size_function) {
42 if (module_sequence.empty()) {
43 return 0;
44 }
45
46 const HloModule* module = module_sequence.begin()->first->parent();
47 TF_ASSIGN_OR_RETURN(std::unique_ptr<TuplePointsToAnalysis> points_to_analysis,
48 TuplePointsToAnalysis::Run(module));
49
50 // The absolute minimum memory required for a given sequence of instructions
51 // is determined by the sequence of Alloc and Free calls on a simulated heap,
52 // ignoring fragmentation. We run the heap simulation on the whole module,
53 // rather than summing each computation, since it gives us a better lower
54 // bound, by minimizing the liveness of sub-computations.
55 TF_ASSIGN_OR_RETURN(
56 HeapSimulator::Result result,
57 HeapSimulator::Run(MakeUnique<NoFragmentationStatsHeap>(), *module,
58 module_sequence, *points_to_analysis, size_function));
59 return result.heap_size;
60}

64// Class implementing a list scheduler of HLO instructions which produces a
65// sequence which minimizes memory usage.
66class ListScheduler {
67 public:
68 // Construct and return a memory-minimizing sequence of HLO instructions
69 // containing the given HLO computation.
70 static StatusOr<std::vector<const HloInstruction*>> Run(
71 const HloComputation& computation,
72 const TuplePointsToAnalysis& points_to_analysis,
73 const LogicalBuffer::SizeFunction& size_function) {
74 ListScheduler scheduler(computation, points_to_analysis, size_function);
75 return scheduler.CreateSchedule();
76 }
77
78 // Returns whether the memory used by the given HLO should be ignored by the
79 // scheduling heuristic.
80 static bool IgnoreInstruction(const HloInstruction& instruction) {
81 return instruction.opcode() == HloOpcode::kParameter ||
82 instruction.opcode() == HloOpcode::kConstant;
83 }
84
85 private:
86 // The scheduling priority of an instruction is first the number of bytes
87 // freed by scheduling the instruction, and second (tie-breaker) by the number
88 // of users. This is represented as a std::pair containing these two values
89 // (first element is the bytes freed). std::pair provides the necessary
90 // comparison operators.
91 using Priority = std::pair<int64, int64>;
92
93 ListScheduler(const HloComputation& computation,
94 const TuplePointsToAnalysis& points_to_analysis,
95 const LogicalBuffer::SizeFunction& size_function)
96 : computation_(computation),
97 points_to_analysis_(points_to_analysis),
98 size_function_(size_function) {
99 // Create a map containing the LogicalBuffer uses for each HLO
100 // instruction. An HLO instruction "uses" a LogicalBuffer if the
101 // LogicalBuffer is in an operand of the instruction as indicated by
102 // points-to analysis.
103 for (auto* instruction : computation.instructions()) {
104 tensorflow::gtl::FlatSet<const LogicalBuffer*> instr_uses;
105 for (auto* operand : instruction->operands()) {
106 points_to_analysis.GetPointsToSet(operand).ForEachElement(
107 [&](const ShapeIndex& /*index*/,
108 const PointsToSet::BufferList& buffers) {
109 instr_uses.insert(buffers.begin(), buffers.end());
110 });
111 }
112 buffer_uses_[instruction] = std::vector<const LogicalBuffer*>(
113 instr_uses.begin(), instr_uses.end());
114 }
115
116 // Create map containing the number of unscheduled uses (hlo instructions)
117 // of each logical buffer.
118 for (auto* instruction : computation.instructions()) {
119 for (auto* buffer :
120 points_to_analysis.GetBuffersDefinedByInstruction(instruction)) {
121 unscheduled_use_count_[buffer] = 0;
122 }
123 }
124 for (auto* instruction : computation.instructions()) {
125 for (const LogicalBuffer* buffer : buffer_uses_.at(instruction)) {
126 ++unscheduled_use_count_[buffer];
127 }
128 }
129
130 // Buffers live out of the computation have an implicit use at the end of
131 // the computation.
132 for (const LogicalBuffer* live_out_buffer :
133 points_to_analysis.GetPointsToSet(computation.root_instruction())
134 .CreateFlattenedSet()) {
135 ++unscheduled_use_count_[live_out_buffer];
136 }
137 }
138
139 // Returns whether the memory used by the given buffer should be ignored by
140 // the scheduling heuristic.
141 static bool IgnoreBuffer(const LogicalBuffer& buffer) {
142 return IgnoreInstruction(*buffer.instruction());
143 }
144
145 // An entry in the worklist used by CreateSchedule. Corresponds to one
146 // HloInstruction, plus some cached metadata, saved for the purposes of making
147 // BytesFreedIfScheduled fast.
148 struct ReadyListEntry {
149 const HloInstruction* instruction;
150
151 // The total size of all buffers defined by this instruction.
152 int64 bytes_defined;
153
154 // For each buffer B used by this instruction, we keep a pair (B, U), where
155 // U is the number of uses of B that have not yet been scheduled. This pair
156 // is a pointer into the unscheduled_use_count_ map, so it gets updated for
157 // free when we update counts in the map.
158 std::vector<const std::pair<const LogicalBuffer* const, int64>*>
159 used_buffer_unscheduled_use_counts;
160 };
161
162 // Creates a ReadyListEntry for the given instruction.
163 ReadyListEntry MakeReadyListEntry(const HloInstruction* instruction) {
164 ReadyListEntry entry;
165 entry.instruction = instruction;
166
167 entry.bytes_defined = 0;
168 for (auto* buffer :
169 points_to_analysis_.GetBuffersDefinedByInstruction(instruction)) {
170 if (!IgnoreBuffer(*buffer)) {
171 entry.bytes_defined += size_function_(*buffer);
172 }
173 }
174
175 for (auto* buffer : buffer_uses_.at(instruction)) {
176 if (IgnoreBuffer(*buffer)) {
177 continue;
178 }
179 auto unscheduled_use_count_it = unscheduled_use_count_.find(buffer);
180 CHECK(unscheduled_use_count_it != unscheduled_use_count_.end());
181 entry.used_buffer_unscheduled_use_counts.push_back(
182 &*unscheduled_use_count_it);
183 }
184 return entry;
185 }
186
187 // Returns the number of bytes freed if the HLO instruction is scheduled.
188 int64 BytesFreedIfScheduled(const ReadyListEntry& entry) {
189 int64 freed_bytes = 0;
190 for (const auto& kv : entry.used_buffer_unscheduled_use_counts) {
191 auto buffer = kv->first;
192 auto use_count = kv->second;
193 if (use_count == 1) {
194 freed_bytes += size_function_(*buffer);
195 }
196 }
197 return freed_bytes - entry.bytes_defined;
198 }
199
200 // Constructs the scheduling priority of the given instruction.
201 Priority GetPriority(const ReadyListEntry& entry) {
202 return {BytesFreedIfScheduled(entry), entry.instruction->user_count()};
203 }
204
205 std::vector<const HloInstruction*> CreateSchedule() {
206 std::vector<const HloInstruction*> schedule;
207
208 // Populate the ready list with instructions which have no operands or
209 // control predecessors.
210 tensorflow::gtl::FlatMap<const HloInstruction*, int64>
211 unscheduled_pred_count;
212 for (auto* instruction : computation_.instructions()) {
213 // TODO(b/34466113): Replace this and above with successors() or
214 // predecessors() when these methods are added to HloInstruction.
215 for (const HloInstruction* user : instruction->users()) {
216 unscheduled_pred_count[user]++;
217 }
218 for (const HloInstruction* succ : instruction->control_successors()) {
219 unscheduled_pred_count[succ]++;
220 }
221 }
222
223 // Use a multimap to sort ReadyListEntry according to their priority.
224 std::multimap<Priority, ReadyListEntry> ready_queue;
225
226 // Map of ready instructions to their iterators in ready_queue.
227 tensorflow::gtl::FlatMap<const HloInstruction*,
228 std::multimap<Priority, ReadyListEntry>::iterator>
229 ready_instructions;
230
231 auto add_to_ready_queue = [&](HloInstruction* inst) {
232 auto entry = MakeReadyListEntry(inst);
233 auto it = ready_queue.emplace(GetPriority(entry), std::move(entry));
234 ready_instructions[inst] = it;
235 };
236
237 for (auto* instruction : computation_.instructions()) {
238 // Instruction with no operands or control predecessors will
239 // not be in the map.
240 if (unscheduled_pred_count.count(instruction) == 0) {
241 add_to_ready_queue(instruction);
242 }
243 }
244
245 while (!ready_queue.empty()) {
246 // Remove the selected instruction from the ready list and add it to the
247 // schedule.
248 auto best_it = ready_queue.end();
249 --best_it;
250 const HloInstruction* best = best_it->second.instruction;
251 ready_queue.erase(best_it);
252 ready_instructions.erase(best);
253 schedule.push_back(best);
254 scheduled_instructions_.insert(best);
255
256 bool adjust_ready_queue = false;
257 // Update the unscheduled uses of the logical buffers.
258 for (const LogicalBuffer* buffer : buffer_uses_.at(best)) {
259 int64& count = unscheduled_use_count_[buffer];
260 CHECK_GT(count, 0);
261 --count;
262 if (count == 1) {
263 adjust_ready_queue = true;
264 }
265 }
266
267 // Add new instructions to ready list.
268 auto update_pred_count = [&](HloInstruction* inst) {
269 int64 pred_count = --unscheduled_pred_count.at(inst);
270 CHECK_GE(pred_count, 0);
271 if (pred_count == 0) {
272 add_to_ready_queue(inst);
273 }
274 };
275 // TODO(b/34466113): Replace this and above with successors() or
276 // predecessors() when these methods are added to HloInstruction.
277 for (HloInstruction* user : best->users()) {
278 update_pred_count(user);
279 }
280 for (HloInstruction* succ : best->control_successors()) {
281 update_pred_count(succ);
282 }
283 // The unscheduled use count for a buffer has changed to 1, so the
284 // priorities of some ready instructions may go up. We update them in the
285 // ready queue, so that they can appear earlier.
286 if (adjust_ready_queue) {
287 for (HloInstruction* operand : best->operands()) {
288 for (HloInstruction* operand_user : operand->users()) {
289 auto ready_instructions_it = ready_instructions.find(operand_user);
290 if (ready_instructions_it == ready_instructions.end()) {
291 continue;
292 }
293 auto ready_queue_it = ready_instructions_it->second;
294 auto& entry = ready_queue_it->second;
295 Priority new_priority = GetPriority(entry);
296 if (new_priority == ready_queue_it->first) {
297 continue;
298 }
299 // Create a new entry in ready_queue, then update
300 // ready_instructions[operand_user] to refer to the new entry.
301 ready_instructions_it->second =
302 ready_queue.emplace(new_priority, std::move(entry));
303 // Remove the old entry in ready_queue.
304 ready_queue.erase(ready_queue_it);
305 }
306 }
307 }
308 }
309 CHECK_EQ(schedule.size(), computation_.instruction_count());
310 CHECK_EQ(scheduled_instructions_.size(), computation_.instruction_count());
311
312 return schedule;
313 }
314
315 const HloComputation& computation_;
316 const TuplePointsToAnalysis& points_to_analysis_;
317 const LogicalBuffer::SizeFunction& size_function_;
318
319 // A map containing the LogicalBuffers that each instruction uses.
320 tensorflow::gtl::FlatMap<const HloInstruction*,
321 std::vector<const LogicalBuffer*>>
322 buffer_uses_;
323
324 // A map containing the count of unscheduled HLOs which using a particular
325 // LogicalBuffer. We rely on iterator stability in this map, and that the map
326 // entries are std::pair's.
327 std::unordered_map<const LogicalBuffer*, int64> unscheduled_use_count_;
328
329 // Set of instructions which have been scheduled.
330 tensorflow::gtl::FlatSet<const HloInstruction*> scheduled_instructions_;
331};

333int64 SumLogicalBufferSizes(
334 const TuplePointsToAnalysis::BufferDefinitionVector& buffers,
335 const LogicalBuffer::SizeFunction& size_function) {
336 int64 size = 0;
337 for (const LogicalBuffer* buffer : buffers) {
338 size += size_function(*buffer);
339 }
340 return size;
341}

355StatusOr<std::vector<const HloInstruction*>> CreateMemoryMinimizingSequence(
356 const HloComputation& computation,
357 const TuplePointsToAnalysis& points_to_analysis,
358 const LogicalBuffer::SizeFunction& size_function,
359 const MemorySchedulerAlgorithm& algorithm) {
360 VLOG(2) << "Computation: " << computation.name();
361 if (algorithm) {
362 return algorithm(computation, points_to_analysis, size_function);
363 }
364 return DefaultMemoryScheduler(computation, points_to_analysis, size_function);

369StatusOr<std::vector<const HloInstruction*>> DFSMemoryScheduler(
370 const HloComputation& computation,
371 const TuplePointsToAnalysis& points_to_analysis,
372 const LogicalBuffer::SizeFunction& size_function) {
373 // This ordering is based on DFS post-order, with a heuristic to decide which
374 // operand to visit first. The heuristic is based on 'extra_users', which is
375 // simply users-1 for each instruction. By subtracting 1, we're saying that
376 // instructions with no users or a single user don't count; instructions with
377 // lots of fan-out will be visited earlier.
378 int64 cumulative_total_size = 0;
379 tensorflow::gtl::FlatMap<const HloInstruction*, int64> extra_users;
380 tensorflow::gtl::FlatMap<const HloInstruction*, int64> total_sizes;
381 for (const HloInstruction* hlo : computation.MakeInstructionPostOrder()) {
382 if (ListScheduler::IgnoreInstruction(*hlo)) {
383 extra_users[hlo] = 0;
384 total_sizes[hlo] = 0;
385 continue;
386 }
387 extra_users[hlo] = hlo->users().empty() ? 0 : hlo->users().size() - 1;
388 int64 logical_buffer_size = SumLogicalBufferSizes(
389 points_to_analysis.GetBuffersDefinedByInstruction(hlo), size_function);
390 total_sizes[hlo] = logical_buffer_size;
391 cumulative_total_size += logical_buffer_size;
392 tensorflow::gtl::FlatSet<const HloInstruction*> unique_operands(
393 hlo->operands().begin(), hlo->operands().end());
394 for (const HloInstruction* operand : unique_operands) {
395 extra_users[hlo] += extra_users[operand];
396 total_sizes[hlo] += total_sizes[operand];
397 }
398 total_sizes[hlo] = std::min(total_sizes[hlo], cumulative_total_size);
399 }
400 CHECK_EQ(extra_users.size(), computation.instruction_count());
401 CHECK_EQ(total_sizes.size(), computation.instruction_count());
402
403 // Construct a total order based on DFS post-order, visiting operands in
404 // decreasing cumulative extra user order, and next by cumulative size, with a
405 // tiebreaker by name for determinism.
406 std::vector<const HloInstruction*> sequence;
407 FunctionVisitor visitor([&sequence](HloInstruction* hlo) {
408 sequence.push_back(hlo);
409 return Status::OK();
410 });
411 TF_RETURN_IF_ERROR(computation.AcceptWithOperandOrder(
412 &visitor, [&extra_users, &total_sizes](const HloInstruction* a,
413 const HloInstruction* b) {
414 if (extra_users[a] != extra_users[b]) {
415 return extra_users[a] > extra_users[b];
416 }
417 if (total_sizes[a] != total_sizes[b]) {
418 return total_sizes[a] > total_sizes[b];
419 }
420 return a->name() < b->name();
421 }));
422 CHECK_EQ(sequence.size(), computation.instruction_count());
423 return sequence;
424}

433StatusOr<std::vector<const HloInstruction*>> DefaultMemoryScheduler(
434 const HloComputation& computation,
435 const TuplePointsToAnalysis& points_to_analysis,
436 const LogicalBuffer::SizeFunction& size_function) {
437 // We try both a list-scheduler based ordering and a DFS based ordering, and
438 // choose whichever returns a lower min-memory, not accounting for
439 // fragmentation.
440 //
441 // Note that this is just a heuristic. One obvious inaccuracy is that the
442 // memory required for sub-computations might be different when considered
443 // within the caller's context. But it's good enough for now.
444 TF_ASSIGN_OR_RETURN(
445 std::vector<const HloInstruction*> list_sequence,
446 ListMemoryScheduler(computation, points_to_analysis, size_function));
447 TF_ASSIGN_OR_RETURN(
448 const int64 list_memory,
449 MinimumMemoryForComputation(computation, list_sequence,
450 points_to_analysis, size_function));
451 VLOG(2) << "Min-memory list sequence: " << HumanReadableNumBytes(list_memory);
452
453 TF_ASSIGN_OR_RETURN(
454 std::vector<const HloInstruction*> dfs_sequence,
455 DFSMemoryScheduler(computation, points_to_analysis, size_function));
456 TF_ASSIGN_OR_RETURN(
457 const int64 dfs_memory,
458 MinimumMemoryForComputation(computation, dfs_sequence, points_to_analysis,
459 size_function));
460 VLOG(2) << "Min-memory dfs sequence: " << HumanReadableNumBytes(dfs_memory);
461
462 if (list_memory <= dfs_memory) {
463 VLOG(2) << "Chose min-memory list sequence: "
464 << HumanReadableNumBytes(list_memory);
465 return list_sequence;
466 } else {
467 VLOG(2) << "Chose min-memory dfs sequence: "
468 << HumanReadableNumBytes(dfs_memory);
469 return dfs_sequence;
470 }
471}

473StatusOr<SequentialHloOrdering::HloModuleSequence>
474CreateMemoryMinimizingSequence(const HloModule& module,
475 const LogicalBuffer::SizeFunction& size_function,
476 const MemorySchedulerAlgorithm& algorithm) {
477 SequentialHloOrdering::HloModuleSequence sequence;
478 TF_ASSIGN_OR_RETURN(std::unique_ptr<TuplePointsToAnalysis> points_to_analysis,
479 TuplePointsToAnalysis::Run(&module));
480 for (const auto* computation : module.MakeNonfusionComputations()) {
481 TF_ASSIGN_OR_RETURN(
482 sequence[computation],
483 CreateMemoryMinimizingSequence(*computation, *points_to_analysis,
484 size_function, algorithm));
485 }
486 return sequence;
487}
488
489StatusOr<std::vector<const HloInstruction*>> CreateMemoryMinimizingSequence(
490 const HloComputation& computation,
491 const LogicalBuffer::SizeFunction& size_function,
492 const MemorySchedulerAlgorithm& algorithm) {
493 CHECK(!computation.IsFusionComputation());
494 TF_ASSIGN_OR_RETURN(std::unique_ptr<TuplePointsToAnalysis> points_to_analysis,
495 TuplePointsToAnalysis::Run(computation.parent()));
496 return CreateMemoryMinimizingSequence(computation, *points_to_analysis,
497 size_function, algorithm);
498}
