50// Describes a computation at the HLO level.
51//
52// An HloComputation contains a directed acyclic graph of HLO instructions. The
53// computation has a single root instruction which produces the output of the
54// computation.
55class HloComputation {
56 public:
57 // Builder class for HloComputation.
58 class Builder {
59 public:
60 explicit Builder(const string& name,
61 HloInstruction* fusion_instruction = nullptr)
62 : name_(name),
63 last_added_instruction_(nullptr),
64 fusion_instruction_(fusion_instruction) {}
65
66 // Build and return an HloComputation. The parameter root_instruction
67 // specifies the already-added instruction to use as the root. If
68 // root_instruction is nullptr then use the last added instruction as the
69 // root.
70 std::unique_ptr<HloComputation> Build(
71 HloInstruction* root_instruction = nullptr);
72
73 HloInstruction* AddInstruction(
74 std::unique_ptr<HloInstruction> instruction) {
75 instructions_.push_back(std::move(instruction));
76 last_added_instruction_ = instructions_.back().get();
77 return last_added_instruction_;
78 }
79
80 Status ForEachInstruction(
81 const std::function<Status(const HloInstruction*)>& func) const {
82 for (const auto& instruction : instructions_) {
83 TF_RETURN_IF_ERROR(func(instruction.get()));
84 }
85 return Status::OK();
86 }
87
88 private:
89 const string name_;
90 HloInstruction* last_added_instruction_;
91 HloInstruction* fusion_instruction_;
92 std::vector<std::unique_ptr<HloInstruction>> instructions_;
93 };
94
95 // Add an instruction to the computation. The computation takes ownership of
96 // the instruction.
97 HloInstruction* AddInstruction(std::unique_ptr<HloInstruction> instruction);
98
99 // Remove the param_no'th parameter from the computation.
100 // Note this is only applicatable to the computation for the fusion
101 // instruction.
102 Status RemoveParameter(int64 param_no);
103
104 // Add new parameter instruction to the computation.
105 // This should be a new parameter. Instruction will be appended to parameters
106 // and inserted to the instruction list.
107 HloInstruction* AddParameter(std::unique_ptr<HloInstruction> instruction);
108
109 // Remove an instruction from the computation. The instruction must have no
110 // users. Instruction is deallocated with this call.
111 Status RemoveInstruction(HloInstruction* instruction);
112
113 // Remove an instruction from the computation and also transitively any
114 // operand that has no users post removing an instruction. The instruction
115 // must have no users. Instruction is deallocated with this call.
116 Status RemoveInstructionAndUnusedOperands(HloInstruction* instruction);
117
118 // Set the root of the computation to the given instruction. The instruction
119 // must have already been added to the computation and have the same shape as
120 // the result of the computation for non fusion computations.
121 void set_root_instruction(HloInstruction* new_root_instruction);
122
123 // Return the root instruction of the computation. The root instruction is the
124 // instruction which produces the output of the computation.
125 HloInstruction* root_instruction() const { return root_instruction_; }
126
127 // Returns the number of parameters for this computation.
128 int64 num_parameters() const { return param_instructions_.size(); }
129
130 // Returns the parameter instruction for the given parameter number.
131 HloInstruction* parameter_instruction(int64 param_no) const {
132 CHECK_GE(param_no, 0);
133 CHECK_LT(param_no, static_cast<int64>(param_instructions_.size()))
134 << "Computation " << name() << " has no parameter number " << param_no;
135 return param_instructions_[param_no];
136 }
137
138 const std::vector<HloInstruction*>& parameter_instructions() const {
139 return param_instructions_;
140 }
141
142 const string& name() const { return name_; }
143
144 // Use the given NameUniquer to select a unique name for the computation based
145 // on the computation's existing name.
146 void UniquifyName(NameUniquer* name_uniquer);
147
148 // Return a string representation of the computation.
149 //
150 // (We express the default options using an overload rather than a default
151 // param because gdb ignores default params, but does resolve overloads.)
152 string ToString() const { return ToString(HloPrintOptions()); }
153 string ToString(const HloPrintOptions& options) const;
154
155 // Returns a serialized representation of this computation.
156 HloComputationProto ToProto() const;
157
158 // Creates a computation from the given proto. Arguments:
159 //
160 // module: the module which will contain the computation. The newly created
161 // computation is *not* added to the module, however.
162 // proto: the proto to convert from.
163 // computation_map: a map from computation id to HloComputation*. This map
164 // must contain all computations which the newly constructed computation
165 // calls.
166 static StatusOr<std::unique_ptr<HloComputation>> CreateFromProto(
167 HloModule* module, const HloComputationProto& proto,
168 const tensorflow::gtl::FlatMap<int64, HloComputation*>& computation_map);
169
170 // Gets the instructions in this computation.
171 //
172 // The returned type is a range of HloInstruction*s, so you can iterate over
173 // it using a range-based for loop in the natural way:
174 //
175 // for (HloInstruction* instr : computation->instructions()) { ... }
176 //
177 tensorflow::gtl::iterator_range<UnwrappingIterator<
178 std::list<std::unique_ptr<HloInstruction>>::const_iterator>>
179 instructions() const {
180 return {MakeUnwrappingIterator(instructions_.begin()),
181 MakeUnwrappingIterator(instructions_.end())};
182 }
183 tensorflow::gtl::iterator_range<
184 UnwrappingIterator<std::list<std::unique_ptr<HloInstruction>>::iterator>>
185 instructions() {
186 return {MakeUnwrappingIterator(instructions_.begin()),
187 MakeUnwrappingIterator(instructions_.end())};
188 }
189
190 // Compute and return a post-order of the instructions in the computation. In
191 // this order, definitions of values always appear before their uses.
192 std::list<HloInstruction*> MakeInstructionPostOrder() const;
193
194 // Computes and returns the reachability between HLO instructions in the
195 // computation. The returned HloReachabilityMap is constructed such that
196 // HloReachabilityMap::IsReachable(a, b) returns true iff there exists a
197 // directed path (from producer to consumer) from 'a' to 'b'. Both data
198 // dependencies (operands) and control dependencies are considered for
199 // reachability. Trivially an instruction is reachable from itself.
200 std::unique_ptr<HloReachabilityMap> ComputeReachability() const;
201
202 // Updates the given reachability map after the immediate predecessor set
203 // (operands and control predecessors) of 'instruction' has changed.
204 void UpdateReachabilityThroughInstruction(
205 const HloInstruction* instruction, HloReachabilityMap* reachability_map);
206
207 int64 instruction_count() const { return instructions_.size(); }
208
209 // Creates and returns a list of the embedded computations called by this
210 // computation. This includes all embedded computations called directly or
211 // transitively. The embedded computations are sorted such that if computation
212 // A calls computation B (eg, via a map instruction) then A will appear after
213 // B in the list.
214 std::list<HloComputation*> MakeEmbeddedComputationsList() const;
215
216 // Creates a fusion instruction containing the given instructions.
217 // `fusion_kind` indicates the type of the fusion, e.g., loop fusion or fusion
218 // into a library call. Instructions must be in reverse topological order
219 // (root of the fused expression first). Replaces all uses of the original
220 // root instruction with the fusion instruction. The original instructions are
221 // removed if they have no uses after fusion (this is necessarily true for at
222 // least the root).
223 HloInstruction* CreateFusionInstruction(
224 tensorflow::gtl::ArraySlice<HloInstruction*> instructions_to_fuse,
225 HloInstruction::FusionKind fusion_kind);
226
227 // Create a deep copy of the given instruction and return the instruction
228 // producing the copied result. All instructions performing the copy are added
229 // to the computation. For array-shaped values, this method trivially returns
230 // a kCopy instruction. For tuple-shaped instructions, the copy is performed
231 // with a series of kGetTupleElement and kTuple instructions. If
232 // indices_to_copy is non-null then this ShapeTree indicates which elements
233 // (arrays) of the shape to copy. Non-copied elements are passed through
234 // transparently. If copies_added is non-null, then the added kCopy
235 // instructions will be inserted in the respective index in the given
236 // ShapeTree.
237 StatusOr<HloInstruction*> DeepCopyInstruction(
238 HloInstruction* instruction,
239 const ShapeTree<bool>* indices_to_copy = nullptr,
240 ShapeTree<HloInstruction*>* copies_added = nullptr);
241
242 // Computes and returns the ProgramShape of this computation (shape of
243 // parameters and result with layout).
244 ProgramShape ComputeProgramShape() const;
245
246 // Return whether `*this` and `other` are functionally equivalent.
247 bool operator==(const HloComputation& other) const;
248
249 // Replaces old instruction with newly created instruction. Removes old
250 // instruction from computation. Updates uses and root instruction.
251 Status ReplaceWithNewInstruction(
252 HloInstruction* old_instruction,
253 std::unique_ptr<HloInstruction> new_instruction);
254
255 // Replace old instruction with new instruction. Updates uses and root
256 // instruction. Removes old instruction from computation. Precondition:
257 // old_instruction and new_instruction must have the compatible shapes.
258 Status ReplaceInstruction(HloInstruction* old_instruction,
259 HloInstruction* new_instruction);
260
261 // Set/get the module containing this computation.
262 void set_parent(HloModule* module) { parent_ = module; }
263 const HloModule* parent() const { return parent_; }
264 HloModule* parent() { return parent_; }
265
266 // Visit every node in the computation in DFS post-order with the given
267 // visitor. This is similar to calling HloInstruction::Accept on the root of
268 // the computation except this method also visits instructions not reachable
269 // via the root. The root instruction of the computation is visited last, and
270 // the visitor's FinishVisit method is called once upon completion (with the
271 // root instruction as the argument).
272 template <typename HloInstructionPtr>
273 Status Accept(DfsHloVisitorBase<HloInstructionPtr>* visitor) const;
274
275 // Same as Accept() above, but the order of operand and control predecessor
276 // visitation is determined by the given operand order; if compare(A, B) ==
277 // true, A is visited before B.
278 Status AcceptWithOperandOrder(
279 DfsHloVisitor* visitor,
280 const HloInstruction::CompareFunction& operand_order) const;
281
282 // Visit every node in the computation in the given order. 'order' must
283 // be a topological sort of all instructions in the computation.
284 template <typename HloInstructionPtr>
285 Status AcceptOrdered(DfsHloVisitorBase<HloInstructionPtr>* visitor,
286 const std::vector<const HloInstruction*>& order) const;
287
288 // Same as Accept() above, but the visitor is given as a function.
289 Status Accept(const std::function<Status(HloInstruction*)>& visitor_func);
290 Status Accept(
291 const std::function<Status(const HloInstruction*)>& visitor_func) const;
292
293 // Returns a deep copy of this computation including all instructions.
294 // If the module pointer is not nullptr, it will be the module where
295 // the cloned computations will be added to (in order to support deep
296 // cloning).
297 std::unique_ptr<HloComputation> Clone(const string& suffix = "clone",
298 HloModule* module = nullptr);
299
300 // Like Clone(), but if an instruction is present in replacement_map, we use
301 // the map's value to replace that instruction in the cloned computation.
302 //
303 // If replacements maps a key to nullptr, we remove that instruction from the
304 // new computation.
305 std::unique_ptr<HloComputation> CloneWithReplacements(
306 std::unordered_map<const HloInstruction*, std::unique_ptr<HloInstruction>>
307 replacements,
308 HloModule* module = nullptr, const string& suffix = "clone");
309
310 // Returns true if the given instruction can be removed from the computation.
311 // Parameter instructions cannot be removed without violating invariants of
312 // the HLO computation with the exception of fusion computation. A parameter
313 // instruction is removable for a fusion computation.
314 //
315 // Note that IsRemovable() is a necessariy condition to remove an instruction
316 // rather than a sufficient condition. For example, instructions with
317 // side-effect (e.g., Send, Infeed) may be removed from a computation, but the
318 // transformation must guarantee the invariants relevant to the instructions
319 // still hold (e.g., Send and Recv must be removed together to make each
320 // channel complete).
321 bool IsRemovable(const HloInstruction* instruction);
322
323 // Returns true if this computation has a side effect. A computation has a
324 // side effect if it contains one or more instructions with a side effect.
325 bool HasSideEffect() const;
326
327 // Returns if this computation is a fusion computation.
328 bool IsFusionComputation() const { return fusion_instruction_ != nullptr; }
329
330 // Returns the owning fusion instruction, or nullptr if this is not a fusion
331 // computation.
332 HloInstruction* FusionInstruction() const { return fusion_instruction_; }
333 void SetFusionInstruction(HloInstruction* fusion_instruction) {
334 fusion_instruction_ = fusion_instruction;
335 }
336
337 // The id of this computation should be unique within the module.
338 void SetUniqueId(int64 id) {
339 CHECK_EQ(unique_id_, -1);
340 CHECK_GE(id, 0);
341 unique_id_ = id;
342 }
343
344 int64 unique_id() const { return unique_id_; }
345
346 private:
347 explicit HloComputation(
348 const string& name, int parameter_count,
349 std::vector<std::unique_ptr<HloInstruction>>* instructions,
350 HloInstruction* root_instruction, HloInstruction* fusion_instruction);
351
352 // Internal helper for adding instructions.
353 HloInstruction* AddInstructionInternal(
354 std::unique_ptr<HloInstruction> instruction);
355
356 // Fuses HLOs in instructions_to_fuse into fusion_instruction.
357 //
358 // Pre-condition: fusion_instruction's opcode is kFusion.
359 void FuseInstructionsInto(
360 tensorflow::gtl::ArraySlice<HloInstruction*> instructions_to_fuse,
361 HloInstruction* fusion_instruction);
362
363 // Internal helper for recursive copying of an instruction. Creates and
364 // returns a deep copy of the given instruction.
365 StatusOr<HloInstruction*> DeepCopyHelper(
366 HloInstruction* instruction, const ShapeTree<bool>* indices_to_copy,
367 ShapeTree<HloInstruction*>* copies_added, ShapeIndex* index);
368
369 // Internal helper to collect unreachable roots.
370 std::vector<HloInstruction*> CollectUnreachableRoots() const;
371
372 string name_;
373 int64 unique_id_;
374 HloInstruction* root_instruction_;
375
376 // If this computation is a fusion computation, this field points to the
377 // corresponding fusion instruction. Otherwise, this is null.
378 HloInstruction* fusion_instruction_;
379
380 // Module containing this computation.
381 HloModule* parent_ = nullptr;
382
383 // Store instructions in std::list as they can be added and removed
384 // arbitrarily and we want a stable iteration order. Keep a map from
385 // instruction pointer to location in the list for fast lookup.
386 using InstructionList = std::list<std::unique_ptr<HloInstruction>>;
387 InstructionList instructions_;
388 std::unordered_map<const HloInstruction*, InstructionList::iterator>
389 instruction_iterators_;
390
391 std::vector<HloInstruction*> param_instructions_;
392
393 TF_DISALLOW_COPY_AND_ASSIGN(HloComputation);
394};
