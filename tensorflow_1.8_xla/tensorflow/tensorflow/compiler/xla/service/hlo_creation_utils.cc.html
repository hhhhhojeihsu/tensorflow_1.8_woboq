26StatusOr<HloInstruction*> MakeBinaryHlo(HloOpcode opcode, HloInstruction* lhs,
27 HloInstruction* rhs) {
28 HloComputation* computation = lhs->parent();
29 CHECK_EQ(computation, rhs->parent());
30 TF_ASSIGN_OR_RETURN(Shape binary_op_shape,
31 ShapeInference::InferBinaryOpShape(opcode, lhs, rhs));
32 return computation->AddInstruction(
33 HloInstruction::CreateBinary(binary_op_shape, opcode, lhs, rhs));
34}

146StatusOr<HloInstruction*> MakeConcatHlo(ArraySlice<HloInstruction*> operands,
147 int64 dimension) {
148 CHECK_GT(operands.size(), 0);
149
150 HloComputation* computation = operands[0]->parent();
151 CHECK(c_all_of(operands, [&](HloInstruction* instr) {
152 return instr->parent() == computation;
153 }));
154
155 std::vector<const Shape*> operand_shapes;
156 c_transform(operands, std::back_inserter(operand_shapes),
157 [](HloInstruction* instr) { return &instr->shape(); });
158
159 TF_ASSIGN_OR_RETURN(Shape concat_shape, ShapeInference::InferConcatOpShape(
160 operand_shapes, dimension));
161 return computation->AddInstruction(
162 HloInstruction::CreateConcatenate(concat_shape, operands, dimension));
163}

165StatusOr<HloInstruction*> CollapseFirstNDims(HloInstruction* operand, int64 n) {
166 const Shape& operand_shape = operand->shape();
167 CHECK_GE(operand_shape.dimensions_size(), n);
168 int64 new_shape_leading_bound = 1;
169 for (int64 i = 0; i < n; i++) {
170 new_shape_leading_bound *= operand_shape.dimensions(i);
171 }
172
173 std::vector<int64> new_shape_dims;
174 new_shape_dims.reserve(operand_shape.dimensions_size() - n + 1);
175 new_shape_dims.push_back(new_shape_leading_bound);
176
177 std::copy(operand_shape.dimensions().begin() + n,
178 operand_shape.dimensions().end(),
179 std::back_inserter(new_shape_dims));
180
181 Shape output_shape =
182 ShapeUtil::MakeShape(operand_shape.element_type(), new_shape_dims);
183
184 return MakeReshapeHlo(output_shape, operand);
185}

187StatusOr<HloInstruction*> ExpandFirstDimIntoNDims(
188 HloInstruction* operand, ArraySlice<int64> expanded_dims) {
189 CHECK_GT(operand->shape().dimensions_size(), 0);
190 CHECK_EQ(operand->shape().dimensions(0), Product(expanded_dims));
191
192 std::vector<int64> expanded_shape_dim_bounds;
193 expanded_shape_dim_bounds.reserve(expanded_dims.size() +
194 operand->shape().dimensions_size() - 1);
195 c_copy(expanded_dims, std::back_inserter(expanded_shape_dim_bounds));
196 std::copy(operand->shape().dimensions().begin() + 1,
197 operand->shape().dimensions().end(),
198 std::back_inserter(expanded_shape_dim_bounds));
199 Shape new_shape = ShapeUtil::MakeShape(operand->shape().element_type(),
200 expanded_shape_dim_bounds);
201 return MakeReshapeHlo(new_shape, operand);
202}

204StatusOr<HloInstruction*> ElideDegenerateDims(HloInstruction* operand,
205 ArraySlice<int64> dims_to_elide) {
206 CHECK(c_is_sorted(dims_to_elide));
207
208 const Shape& input_shape = operand->shape();
209 // First accumulate in reverse
210 std::vector<int64> new_shape_dim_bounds;
211 new_shape_dim_bounds.reserve(input_shape.dimensions_size() -
212 dims_to_elide.size());
213 int64 dims_to_elide_idx = dims_to_elide.size() - 1;
214 for (int64 i = input_shape.dimensions_size() - 1; i >= 0; i--) {
215 if (dims_to_elide_idx >= 0 && i == dims_to_elide[dims_to_elide_idx]) {
216 CHECK_EQ(input_shape.dimensions(i), 1);
217 dims_to_elide_idx--;
218 } else {
219 new_shape_dim_bounds.push_back(input_shape.dimensions(i));
220 }
221 }
222
223 c_reverse(new_shape_dim_bounds);
224 Shape output_shape =
225 ShapeUtil::MakeShape(input_shape.element_type(), new_shape_dim_bounds);
226 return MakeReshapeHlo(output_shape, operand);
227}

229StatusOr<HloInstruction*> PadVectorWithZeros(HloInstruction* operand,
230 int64 zeros_to_prepend,
231 int64 zeros_to_append) {
232 HloComputation* computation = operand->parent();
233 CHECK_EQ(operand->shape().dimensions_size(), 1);
234 PaddingConfig padding_config;
235 PaddingConfig::PaddingConfigDimension padding_config_dim;
236 padding_config_dim.set_edge_padding_low(zeros_to_prepend);
237 padding_config_dim.set_edge_padding_high(zeros_to_append);
238 *padding_config.add_dimensions() = padding_config_dim;
239
240 HloInstruction* zero =
241 computation->AddInstruction(HloInstruction::CreateConstant(
242 MakeUnique<Literal>(Literal::Zero(operand->shape().element_type()))));
243 return MakePadHlo(operand, zero, padding_config);
244}

256StatusOr<std::unique_ptr<HloComputation>> CreateComputationWithSignature(
257 ArraySlice<const Shape*> domain, const Shape& range,
258 tensorflow::StringPiece name) {
259 HloComputation::Builder b(name.ToString());
260 int64 param_idx = 0;
261 for (const Shape* param_shape : domain) {
262 b.AddInstruction(HloInstruction::CreateParameter(
263 param_idx, *param_shape, StrCat("param.", param_idx)));
264 param_idx++;
265 }
266
267 // We can't change the root type of a computation once it is created so create
268 // a dummy root instruction to give the computation the right root shape. In
269 // the future we may want to use a (recursive) broadcast here to avoid
270 // creating large constants.
271 b.AddInstruction(
272 HloInstruction::CreateConstant(Literal::CreateFromShape(range)));
273
274 return b.Build();
275}
