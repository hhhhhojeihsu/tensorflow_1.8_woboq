26static StatusOr<HloComputation*> WidenWhileCondition(
27 HloComputation* narrow_condition, const Shape& wide_shape) {
28 const Shape& narrow_shape =
29 narrow_condition->parameter_instruction(0)->shape();
30
31 HloComputation* wide_while_cond = [&]() {
32 HloComputation::Builder builder(StrCat("wide.", narrow_condition->name()));
33 builder.AddInstruction(
34 HloInstruction::CreateParameter(0, wide_shape, "wide_param"));
35
36 // This is needed so that the root instruction is shaped as a PRED[] -- we
37 // need to get this right to begin with since we can't mutate the type of
38 // the root instruction later. We later change the root instruction to
39 // something more appropriate.
40 builder.AddInstruction(
41 HloInstruction::CreateConstant(Literal::CreateR0<bool>(false)));
42 return narrow_condition->parent()->AddEmbeddedComputation(builder.Build());
43 }();
44
45 HloInstruction* truncated_parameter =
46 TupleUtil::ExtractPrefix(wide_while_cond->parameter_instruction(0),
47 narrow_shape.tuple_shapes_size());
48 HloInstruction* call_narrow_cond = wide_while_cond->AddInstruction(
49 HloInstruction::CreateCall(ShapeUtil::MakeShape(PRED, {}),
50 {truncated_parameter}, narrow_condition));
51
52 wide_while_cond->set_root_instruction(call_narrow_cond);
53
54 TF_RETURN_IF_ERROR(CallInliner::Inline(call_narrow_cond).status());
55 return wide_while_cond;
56}

58static StatusOr<std::pair<HloComputation*, CallInliner::InlinedInstructionMap>>
59WidenWhileBody(HloComputation* narrow_body, const Shape& wide_shape) {
60 const Shape& narrow_shape = narrow_body->parameter_instruction(0)->shape();
61
62 HloComputation* wide_while_body = [&]() {
63 HloComputation::Builder builder(StrCat("wide.", narrow_body->name()));
64 builder.AddInstruction(
65 HloInstruction::CreateParameter(0, wide_shape, "wide_param"));
66 return narrow_body->parent()->AddEmbeddedComputation(builder.Build());
67 }();
68
69 HloInstruction* wide_parameter = wide_while_body->parameter_instruction(0);
70 HloInstruction* truncated_parameter = TupleUtil::ExtractPrefix(
71 wide_parameter, narrow_shape.tuple_shapes_size());
72 HloInstruction* call_narrow_body =
73 wide_while_body->AddInstruction(HloInstruction::CreateCall(
74 narrow_shape, {truncated_parameter}, narrow_body));
75
76 std::vector<HloInstruction*> live_through_values;
77 for (int i = narrow_shape.tuple_shapes_size();
78 i < wide_shape.tuple_shapes_size(); i++) {
79 live_through_values.push_back(
80 wide_while_body->AddInstruction(HloInstruction::CreateGetTupleElement(
81 wide_shape.tuple_shapes(i), wide_parameter, i)));
82 }
83
84 wide_while_body->set_root_instruction(
85 TupleUtil::AppendSuffix(call_narrow_body, live_through_values));
86
87 TF_ASSIGN_OR_RETURN(auto inlined_instructions_map,
88 CallInliner::Inline(call_narrow_body));
89 return {{wide_while_body, std::move(inlined_instructions_map)}};
90}

92/*static*/ StatusOr<WhileUtil::MakeInstructionsLiveInResult>
93WhileUtil::MakeInstructionsLiveIn(
94 HloInstruction* while_instr,
95 tensorflow::gtl::ArraySlice<HloInstruction*> instructions) {
96 CHECK(ShapeUtil::IsTuple(while_instr->shape()));
97
98 int64 elements_in_old_while_shape = while_instr->shape().tuple_shapes_size();
99 Shape new_while_shape = while_instr->shape();
100 for (auto* instruction : instructions) {
101 *new_while_shape.add_tuple_shapes() = instruction->shape();
102 }
103
104 TF_ASSIGN_OR_RETURN(
105 HloComputation * new_while_condition,
106 WidenWhileCondition(while_instr->while_condition(), new_while_shape));
107
108 HloComputation* new_while_body;
109 CallInliner::InlinedInstructionMap inlined_instructions_map;
110 TF_ASSIGN_OR_RETURN(
111 std::tie(new_while_body, inlined_instructions_map),
112 WidenWhileBody(while_instr->while_body(), new_while_shape));
113
114 HloInstruction* new_while_init =
115 TupleUtil::AppendSuffix(while_instr->mutable_operand(0), instructions);
116 HloComputation* containing_computation = while_instr->parent();
117 HloInstruction* new_while = containing_computation->AddInstruction(
118 HloInstruction::CreateWhile(new_while_shape, new_while_condition,
119 new_while_body, new_while_init));
120 TF_RETURN_IF_ERROR(containing_computation->ReplaceInstruction(
121 while_instr, TupleUtil::ExtractPrefix(
122 new_while, while_instr->shape().tuple_shapes_size())));
123
124 HloInstruction* while_body_param = new_while_body->parameter_instruction(0);
125 std::vector<HloInstruction*> live_in_instructions;
126 for (int64 i = elements_in_old_while_shape;
127 i < new_while_shape.tuple_shapes_size(); i++) {
128 live_in_instructions.push_back(
129 new_while_body->AddInstruction(HloInstruction::CreateGetTupleElement(
130 instructions[i - elements_in_old_while_shape]->shape(),
131 while_body_param, i)));
132 }
133
134 WhileUtil::MakeInstructionsLiveInResult result;
135
136 result.new_while_instr = new_while;
137 result.while_body_live_in_values = std::move(live_in_instructions);
138 result.while_body_instruction_map = std::move(inlined_instructions_map);
139
140 return std::move(result);
141}

143static StatusOr<std::unique_ptr<HloComputation>>
144MakeCountedLoopConditionComputation(const Shape& loop_state_shape,
145 int32 trip_count) {
146 Shape scalar_pred = ShapeUtil::MakeShape(PRED, {});
147
148 TF_ASSIGN_OR_RETURN(std::unique_ptr<HloComputation> cond_computation,
149 CreateComputationWithSignature(
150 {&loop_state_shape}, scalar_pred, "while_cond"));
151
152 HloInstruction* trip_count_constant = cond_computation->AddInstruction(
153 HloInstruction::CreateConstant(Literal::CreateR0<int32>(trip_count)));
154
155 HloInstruction* param = cond_computation->parameter_instruction(0);
156 TF_ASSIGN_OR_RETURN(HloInstruction * indvar,
157 MakeGetTupleElementHlo(param, 0));
158
159 TF_ASSIGN_OR_RETURN(
160 HloInstruction * compare,
161 MakeBinaryHlo(HloOpcode::kLt, indvar, trip_count_constant));
162 cond_computation->set_root_instruction(compare);
163 return std::move(cond_computation);
164}

166static StatusOr<std::unique_ptr<HloComputation>> MakeCountedLoopBodyComputation(
167 const Shape& loop_state_shape,
168 const std::function<StatusOr<WhileUtil::LoopStateTy>(
169 HloInstruction*, const WhileUtil::LoopStateTy&)>& loop_body_generator) {
170 TF_ASSIGN_OR_RETURN(std::unique_ptr<HloComputation> body_computation,
171 CreateComputationWithSignature(
172 {&loop_state_shape}, loop_state_shape, "while_body"));
173 HloInstruction* one = body_computation->AddInstruction(
174 HloInstruction::CreateConstant(Literal::CreateR0<int32>(1)));
175 HloInstruction* param = body_computation->parameter_instruction(0);
176 TF_ASSIGN_OR_RETURN(HloInstruction * indvar,
177 MakeGetTupleElementHlo(param, 0));
178 TF_ASSIGN_OR_RETURN(HloInstruction * next_indvar,
179 MakeBinaryHlo(HloOpcode::kAdd, indvar, one));
180
181 std::vector<HloInstruction*> loop_body_generator_args;
182 for (int64 i = 1, e = loop_state_shape.tuple_shapes_size(); i < e; i++) {
183 TF_ASSIGN_OR_RETURN(HloInstruction * tuple_element,
184 MakeGetTupleElementHlo(param, i));
185 loop_body_generator_args.push_back(tuple_element);
186 }
187 TF_ASSIGN_OR_RETURN(std::vector<HloInstruction*> next_state,
188 loop_body_generator(indvar, loop_body_generator_args));
189 next_state.insert(next_state.begin(), next_indvar);
190 HloInstruction* next_state_tuple =
191 body_computation->AddInstruction(HloInstruction::CreateTuple(next_state));
192 body_computation->set_root_instruction(next_state_tuple);
193
194 return std::move(body_computation);
195}

197static StatusOr<HloInstruction*> MakeInitTupleFromInitValues(
198 HloComputation* computation, const WhileUtil::LoopStateTy& init_values) {
199 std::vector<HloInstruction*> init_values_with_indvar;
200 init_values_with_indvar.reserve(init_values.size() + 1);
201 HloInstruction* zero = computation->AddInstruction(
202 HloInstruction::CreateConstant(Literal::CreateR0<int32>(0)));
203 init_values_with_indvar.push_back(zero);
204 c_copy(init_values, std::back_inserter(init_values_with_indvar));
205 return computation->AddInstruction(
206 HloInstruction::CreateTuple(init_values_with_indvar));
207}

209static Shape MakeLoopStateShape(const WhileUtil::LoopStateTy& init_values) {
210 std::vector<Shape> loop_state_shape_components;
211 loop_state_shape_components.reserve(init_values.size() + 1);
212 loop_state_shape_components.push_back(ShapeUtil::MakeShape(S32, {}));
213 c_transform(init_values, std::back_inserter(loop_state_shape_components),
214 [](HloInstruction* instr) { return instr->shape(); });
215 return ShapeUtil::MakeTupleShape(loop_state_shape_components);
216}

218/*static*/ StatusOr<WhileUtil::LoopStateTy> WhileUtil::MakeCountedLoop(
219 HloComputation* computation, int32 trip_count,
220 const WhileUtil::LoopStateTy& init_values,
221 const WhileUtil::LoopBodyGeneratorTy& loop_body_generator) {
222 CHECK_GE(trip_count, 0);
223
224 Shape loop_state_shape = MakeLoopStateShape(init_values);
225 TF_ASSIGN_OR_RETURN(
226 std::unique_ptr<HloComputation> cond,
227 MakeCountedLoopConditionComputation(loop_state_shape, trip_count));
228 TF_ASSIGN_OR_RETURN(
229 std::unique_ptr<HloComputation> body,
230 MakeCountedLoopBodyComputation(loop_state_shape, loop_body_generator));
231 TF_ASSIGN_OR_RETURN(HloInstruction * init_tuple,
232 MakeInitTupleFromInitValues(computation, init_values));
233 HloModule* module = computation->parent();
234 HloInstruction* while_instr =
235 computation->AddInstruction(HloInstruction::CreateWhile(
236 loop_state_shape, module->AddEmbeddedComputation(std::move(cond)),
237 module->AddEmbeddedComputation(std::move(body)), init_tuple));
238
239 std::vector<HloInstruction*> result;
240 for (int64 i = 0, e = init_values.size(); i < e; i++) {
241 TF_ASSIGN_OR_RETURN(HloInstruction * user_state,
242 MakeGetTupleElementHlo(while_instr, i + 1));
243 result.push_back(user_state);
244 }
245 return result;
246}
