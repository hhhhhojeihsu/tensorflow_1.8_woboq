30/*static*/
31StatusOr<HeapSimulator::Result> HeapSimulator::Run(
32 std::unique_ptr<HeapAlgorithm> algorithm, const HloModule& module,
33 const SequentialHloOrdering::HloModuleSequence& module_sequence,
34 const TuplePointsToAnalysis& points_to_analysis,
35 const LogicalBuffer::SizeFunction& size_fn, const Options& options) {
36 HeapSimulator heap(std::move(algorithm), size_fn, options, &module_sequence);
37 const HloComputation* entry_computation = module.entry_computation();
38 const std::vector<const HloInstruction*>& instruction_sequence =
39 FindOrDie(module_sequence, entry_computation);
40 TF_RETURN_IF_ERROR(heap.RunComputation(
41 *entry_computation, instruction_sequence, points_to_analysis));
42 return heap.Finish();
43}

45/*static*/
46StatusOr<HeapSimulator::Result> HeapSimulator::Run(
47 std::unique_ptr<HeapAlgorithm> algorithm, const HloComputation& computation,
48 const std::vector<const HloInstruction*>& instruction_sequence,
49 const TuplePointsToAnalysis& points_to_analysis,
50 const LogicalBuffer::SizeFunction& size_fn, const Options& options) {
51 HeapSimulator heap(std::move(algorithm), size_fn, options,
52 /*module_sequence=*/nullptr);
53 TF_RETURN_IF_ERROR(heap.RunComputation(computation, instruction_sequence,
54 points_to_analysis));
55 return heap.Finish();

58// Runs a heap simulation for the given 'computation', assuming the given
59// 'instruction_sequence'.
60Status HeapSimulator::RunComputation(
61 const HloComputation& computation,
62 const std::vector<const HloInstruction*>& instruction_sequence,
63 const TuplePointsToAnalysis& points_to_analysis) {
64 VLOG(3) << "Computation:\n" << computation.ToString();
65 // The goal here is to minimize memory usage, assuming the given sequential
66 // ordering of instructions. The strategy is to walk through the instruction
67 // sequence, calling Alloc and Free on the underlying heap algorithm. The
68 // heap algorithm takes care of packing and reducing fragmentation.
69 //
70 // 'live_buffers' tracks the liveness of each buffer that we assign, by
71 // associating it with a set of HloInstructions that need to be visited. When
72 // the set becomes empty, the buffer is no longer used, and can be freed.
73 // 'used_buffers' is the reverse map - it tracks which buffers were used by an
74 // instruction, so that we can remove the instructions from a buffer's live
75 // set after they are visited.
76 FlatMap<const LogicalBuffer*, FlatSet<const HloInstruction*>> live_buffers;
77 FlatMap<const HloInstruction*, FlatSet<const LogicalBuffer*>> used_buffers;
78 auto add_user_to_buffer = [this, &live_buffers, &used_buffers](
79 const HloInstruction* user,
80 const LogicalBuffer* buffer) {
81 if (!IgnoreBuffer(buffer)) {
82 VLOG(4) << " Adding user " << user->name() << " to buffer "
83 << buffer->ToString();
84 live_buffers[buffer].insert(user);
85 used_buffers[user].insert(buffer);
86 }
87 };
88
89 // Initialize live_buffers for each buffer that we're going to assign. The
90 // set of instructions that need to be visited contains all users of all
91 // aliases, that is, all users of all instructions that have the buffer
92 // contained in their points-to set.
93 for (const HloInstruction* instruction : instruction_sequence) {
94 const PointsToSet& points_to =
95 points_to_analysis.GetPointsToSet(instruction);
96 const PointsToSet::BufferSet& buffer_set = points_to.CreateFlattenedSet();
97 for (const HloInstruction* user : instruction->users()) {
98 if (user->opcode() != HloOpcode::kGetTupleElement) {
99 for (const LogicalBuffer* buffer : buffer_set) {
100 add_user_to_buffer(user, buffer);
101 }
102 } else {
103 // A GetTupleElement doesn't need to keep all of its operand's buffers
104 // alive. It only needs the buffers that relate to the element its
105 // extracting, and the tuple it's extracting from, but not the buffers
106 // for the other elements.
107 for (const LogicalBuffer* buffer : points_to.element({})) {
108 add_user_to_buffer(user, buffer);
109 }
110 const PointsToSet& gte_points_to =
111 points_to_analysis.GetPointsToSet(user);
112 for (const LogicalBuffer* buffer : gte_points_to.CreateFlattenedSet()) {
113 add_user_to_buffer(user, buffer);
114 }
115 }
116 }
117 }
118
119 const HloInstruction* root = computation.root_instruction();
120 auto output_source_buffers =
121 points_to_analysis.GetPointsToSet(root).CreateFlattenedSet();
122
123 std::vector<const LogicalBuffer*> dead_buffers_to_free;
124 std::vector<const LogicalBuffer*> operand_buffers_to_free;
125 for (const HloInstruction* instruction : instruction_sequence) {
126 const TuplePointsToAnalysis::BufferDefinitionVector&
127 buffers_defined_by_instruction =
128 points_to_analysis.GetBuffersDefinedByInstruction(instruction);
129
130 VLOG(3) << "Instruction: " << instruction->ToString();
131 for (const LogicalBuffer* buffer : buffers_defined_by_instruction) {
132 VLOG(4) << " Defines: " << buffer->ToString()
133 << (IgnoreBuffer(buffer) ? " (Ignored)" : "");
134 }
135
136 dead_buffers_to_free.clear();
137 for (const LogicalBuffer* buffer : buffers_defined_by_instruction) {
138 if (IgnoreBuffer(buffer)) {
139 continue;
140 }
141 // Add a nullptr sentry to ensure entry parameters and output source
142 // buffers are not freed until the very end.
143 const bool entry_parameter =
144 &computation == computation.parent()->entry_computation() &&
145 buffer->instruction()->opcode() == HloOpcode::kParameter;
146 const bool output = output_source_buffers.count(buffer) > 0;
147 if (entry_parameter || output) {
148 live_buffers[buffer].insert(nullptr);
149 }
150
151 // If the buffer has no users and isn't an entry parameter or output, it
152 // must be a dead value.
153 if (live_buffers.count(buffer) == 0) {
154 dead_buffers_to_free.push_back(buffer);
155 }
156 }
157
158 // Update live_buffers to indicate we've visited this instruction; this is
159 // the inverse of the initialization logic. We erase this instruction from
160 // all source buffers of all operands of this instruction. Buffers that
161 // have no instructions left to visit are moved from live_buffers to
162 // operand_buffers_to_free.
163 operand_buffers_to_free.clear();
164 for (const LogicalBuffer* operand_buffer : used_buffers[instruction]) {
165 if (IgnoreBuffer(operand_buffer)) {
166 continue;
167 }
168 VLOG(4) << " Removing user " << instruction->name() << " from buffer "
169 << operand_buffer->ToString();
170 auto it = live_buffers.find(operand_buffer);
171 FlatSet<const HloInstruction*>* live_set = &it->second;
172 live_set->erase(instruction);
173 if (live_set->empty()) {
174 live_buffers.erase(it);
175 operand_buffers_to_free.push_back(operand_buffer);
176 }
177 }
178 // Sort to get a deterministic iteration order.
179 std::sort(operand_buffers_to_free.begin(), operand_buffers_to_free.end(),
180 [](const LogicalBuffer* x, const LogicalBuffer* y) {
181 return x->id() < y->id();
182 });
183
184 // Allocate buffers defined by this instruction. This is the latest point
185 // that we can allocate; right before the buffer is first used. This must
186 // happen before dead or operand buffers are freed; the instruction reads
187 // the operand buffers to produce its output.
188 //
189 // INVARIANT: Either Alloc or ShareBuffer will be called for each buffer
190 // that we should assign.
191 for (const LogicalBuffer* buffer : buffers_defined_by_instruction) {
192 if (IgnoreBuffer(buffer)) {
193 continue;
194 }
195
196 // Check whether the buffer can share with one of its operands; we can
197 // save memory by sharing the buffer, rather than allocating a new one.
198 // We can only share with the operand buffer if it is about to be freed;
199 // we must be the last user of the buffer.
200 bool shared = false;
201 if (options_.may_reuse_operand_buffers) {
202 for (const LogicalBuffer* operand_buffer : operand_buffers_to_free) {
203 if (buffer->instruction()->IsUserOf(operand_buffer->instruction()) &&
204 buffer->instruction()->opcode() != HloOpcode::kCopy &&
205 CanShareOperandBufferWithUser(
206 operand_buffer->instruction(), operand_buffer->index(),
207 buffer->instruction(), buffer->index(), points_to_analysis)) {
208 VLOG(3) << " Sharing: " << buffer->ToString() << " with "
209 << operand_buffer->ToString();
210 ShareBuffer(buffer, operand_buffer, instruction);
211 shared = true;
212 break;
213 }
214 }
215 }
216
217 if (!shared) {
218 VLOG(3) << " Allocating: " << buffer->ToString();
219 Alloc(buffer, instruction);
220 }
221 }
222
223 // If the whole module is sequential, we can save memory by running the
224 // heap-simulation for sub-computations inline. E.g. the buffers for the
225 // condition and body of a kWhile instruction are only live for the duration
226 // of the instruction itself.
227 //
228 // The order that the sub-computations are simulated does not affect
229 // correctness; since the whole module is sequential, we know that the
230 // sub-computations will never be run concurrently.
231 if (module_sequence_ != nullptr) {
232 if (instruction->opcode() == HloOpcode::kCall ||
233 instruction->opcode() == HloOpcode::kConditional ||
234 instruction->opcode() == HloOpcode::kWhile) {
235 for (const HloComputation* called_computation :
236 instruction->called_computations()) {
237 const std::vector<const HloInstruction*>& called_sequence =
238 FindOrDie(*module_sequence_, called_computation);
239 TF_RETURN_IF_ERROR(RunComputation(
240 *called_computation, called_sequence, points_to_analysis));
241 }
242 }
243
244 // Other sub-computations (e.g. Map, Reduce, ...) are skipped; they are
245 // assigned "thread-local" allocations, meaning their buffers are not
246 // allocated up-front at the beginning of the computation.
247 }
248
249 // Free buffers that are no longer live. This is the earliest point that we
250 // can de-allocate; right after the last use of the buffer.
251 for (const LogicalBuffer* buffer : dead_buffers_to_free) {
252 VLOG(3) << " Freeing dead: " << buffer->ToString();
253 Free(buffer, instruction);
254 }
255 for (const LogicalBuffer* buffer : operand_buffers_to_free) {
256 VLOG(3) << " Freeing operand: " << buffer->ToString();
257 Free(buffer, instruction);
258 }
259 }
260
261 // Any remaining live buffers must be entry parameters or output source
262 // buffers, which had a nullptr sentry added. Free them now, in a
263 // deterministic order.
264 std::vector<const LogicalBuffer*> to_free;
265 to_free.reserve(live_buffers.size());
266 for (const auto& buffer_pending : live_buffers) {
267 const LogicalBuffer* buffer = buffer_pending.first;
268 const FlatSet<const HloInstruction*>& pending = buffer_pending.second;
269 CHECK_EQ(pending.size(), 1) << *buffer;
270 CHECK(*pending.begin() == nullptr) << *buffer;
271 to_free.push_back(buffer);
272 }
273
274 std::sort(to_free.begin(), to_free.end(),
275 [](const LogicalBuffer* x, const LogicalBuffer* y) {
276 return x->id() < y->id();
277 });
278 for (const LogicalBuffer* buffer : to_free) {
279 VLOG(3) << "Freeing pending: " << buffer->ToString();
280 Free(buffer, root);
281 }
282
283 return Status::OK();
284}

300bool HeapSimulator::IgnoreBuffer(const LogicalBuffer* buffer) const {
301 // Buffers for constants are ignored unless the alloc_constants option is
302 // set. Also ignore buffers that we're not meant to assign.
303 //
304 // TODO(b/32248867): For consistency, constants should get allocations.
305 if (!options_.alloc_constants &&
306 buffer->instruction()->opcode() == HloOpcode::kConstant) {
307 return true;
308 }
309 return options_.buffers_to_assign != nullptr &&
310 options_.buffers_to_assign->count(buffer) == 0;
311}

313// Alloc always calls the underlying heap algorithm.
314void HeapSimulator::Alloc(const LogicalBuffer* buffer,
315 const HloInstruction* instruction) {
316 CHECK(allocated_buffers_.count(buffer) == 0)
317 << "Alloc called on allocated buffer: " << *buffer;
318 CHECK(freed_buffers_.count(buffer) == 0)
319 << "Alloc called on freed buffer: " << *buffer;
320
321 allocated_buffers_.insert(buffer);
322 const int64 size = size_fn_(*buffer);
323 algorithm_->Alloc(buffer, size);
324 no_fragmentation_stats_->Alloc(buffer, size);
325
326 FillDebugTrace(HeapSimulatorTrace::Event::ALLOC, buffer, instruction,
327 nullptr);
328}

330// Free calls the underlying algorithm for non-shared buffers, and for shared
331// buffers whose group liveness has expired. Shared group liveness is tracked
332// by maintaining a refcount; the Free call on the last buffer in the group
333// causes Free to be called on the underlying algorithm.
334void HeapSimulator::Free(const LogicalBuffer* buffer,
335 const HloInstruction* instruction) {
336 auto shared_it = shared_buffers_.find(buffer);
337 if (shared_it != shared_buffers_.end()) {
338 std::shared_ptr<SharedGroup> group = shared_it->second;
339 --group->refcount;
340 if (group->refcount > 0) {
341 return;
342 }
343 CHECK_EQ(group->refcount, 0)
344 << "Free caused negative refcount on shared buffer: " << *buffer;
345 buffer = group->canonical;
346 }
347
348 CHECK(allocated_buffers_.count(buffer) > 0)
349 << "Free called on non-allocated buffer: " << *buffer;
350 CHECK(freed_buffers_.count(buffer) == 0)
351 << "Free called on freed buffer: " << *buffer;
352
353 freed_buffers_.insert(buffer);
354 const int64 size = size_fn_(*buffer);
355 algorithm_->Free(buffer, size);
356 no_fragmentation_stats_->Free(buffer, size);
357
358 FillDebugTrace(HeapSimulatorTrace::Event::FREE, buffer, instruction, nullptr);
359}
360
361// ShareBuffer associates buffers with their SharedGroup in shared_buffers_.
362// The 'buffer' must be a non-allocated, non-freed buffer, just like in calls to
363// Alloc. The 'shared' buffer must be a previously allocated or shared buffer.
364// Both 'buffer' and 'shared' will be associated with the same SharedGroup.
365void HeapSimulator::ShareBuffer(const LogicalBuffer* buffer,
366 const LogicalBuffer* shared,
367 const HloInstruction* instruction) {
368 CHECK_LE(size_fn_(*buffer), size_fn_(*shared))
369 << "ShareBuffer oversized buffer" << *buffer << " shared: " << *shared;
370 CHECK(allocated_buffers_.count(buffer) == 0)
371 << "ShareBuffer called on allocated buffer: " << *buffer;
372 CHECK(freed_buffers_.count(buffer) == 0)
373 << "ShareBuffer called on freed buffer: " << *buffer;
374 CHECK(freed_buffers_.count(shared) == 0)
375 << "ShareBuffer called on freed shared buffer: " << *shared;
376
377 const LogicalBuffer* canonical = nullptr;
378 auto shared_it = shared_buffers_.find(shared);
379 if (shared_it != shared_buffers_.end()) {
380 // The 'shared' buffer already has a group; it might be the canonical, but
381 // also might not be. Just add 'buffer' to the existing group.
382 std::shared_ptr<SharedGroup> group = shared_it->second;
383 canonical = group->canonical;
384 ++group->refcount;
385 shared_buffers_.emplace(buffer, group);
386 } else {
387 // The 'shared' buffer doesn't have a group; it must be the canonical. Add
388 // both 'buffer' and 'shared' to a new group.
389 CHECK(allocated_buffers_.count(shared) > 0)
390 << "ShareBuffer called on non-allocated shared buffer: " << *shared;
391 auto group = std::make_shared<SharedGroup>();
392 canonical = shared;
393 group->canonical = canonical;
394 group->refcount = 2;
395 shared_buffers_.emplace(buffer, group);
396 shared_buffers_.emplace(shared, group);
397 }
398
399 FillDebugTrace(HeapSimulatorTrace::Event::SHARE_WITH, buffer, instruction,
400 canonical);
401}

403HeapSimulator::Result HeapSimulator::Finish() {
404 Result result = algorithm_->Finish();
405
406 // Post-process the result to add chunks for shared buffers. An empty chunk
407 // map means that either no buffers were allocated, or the heap was only
408 // collecting statistics, e.g. NoFragmentationStatsHeap.
409 if (!result.chunk_map.empty()) {
410 for (const auto& share_pair : shared_buffers_) {
411 const LogicalBuffer* buffer = share_pair.first;
412 std::shared_ptr<SharedGroup> group = share_pair.second;
413 if (buffer != group->canonical) {
414 // The canonical must already exist in the chunk_map, since we called
415 // Alloc(canonical) on the underlying algorithm. Add non-canonical
416 // chunks with the same offset as the canonical.
417 Chunk chunk = FindOrDie(result.chunk_map, group->canonical);
418 chunk.size = size_fn_(*buffer);
419 result.chunk_map.emplace(buffer, chunk);
420 }
421 }
422 // If we were told to assign specific buffers, make sure we've assigned
423 // exactly that many buffers.
424 if (options_.buffers_to_assign != nullptr) {
425 CHECK_EQ(options_.buffers_to_assign->size(), result.chunk_map.size());
426 }
427 }
428
429 // Fragmentation is the difference between the actual and ideal sizes.
430 const Result no_frag_result = no_fragmentation_stats_->Finish();
431 result.fragmentation_size = result.heap_size - no_frag_result.heap_size;
432
433 // Copy the debug trace we collected to the final result.
434 result.debug_trace.Swap(&debug_trace_);
435
436 return result;
437}

497
498void DecreasingSizeRunsHeap::CallAndDrainRun() {
499 if (mode_ == kInit) {
500 CHECK(run_.empty());
501 return;
502 }
503
504 // Call ops in the run sorted by decreasing size, breaking ties by buffer id.
505 std::sort(run_.begin(), run_.end(), [](const Op& a, const Op& b) {
506 if (a.size != b.size) {
507 return a.size > b.size;
508 }
509 return a.buffer->id() < b.buffer->id();
510 });
511 for (const Op& op : run_) {
512 if (mode_ == kAlloc) {
513 algorithm_->Alloc(op.buffer, op.size);
514 } else {
515 algorithm_->Free(op.buffer, op.size);
516 }
517 }
518 run_.clear();
519}

521void LazyBestFitHeap::Alloc(const LogicalBuffer* buffer, int64 size) {
522 // Degenerate case: 0-sized buffers are always allocated at offset 0.
523 if (size == 0) {
524 result_.chunk_map.emplace(buffer, Chunk{0, 0});
525 }
526
527 // First try to allocate from the best-fitting free chunk.
528 auto best_fit_it = free_.lower_bound(Chunk{0, size});
529 while (best_fit_it != free_.end()) {
530 // Account for alignment.
531 const Chunk best = *best_fit_it;
532 const int64 new_offset = RoundUpToNearest(best.offset, alignment_);
533 const int64 new_end = new_offset + size;
534 if (new_end > best.chunk_end()) {
535 // We don't fit after accounting for alignment.
536 ++best_fit_it;
537 continue;
538 }
539 // The buffer is allocated a chunk out of the best-fitting free chunk.
540 free_.erase(best_fit_it);
541 result_.chunk_map.emplace(buffer, Chunk{new_offset, size});
542 // Add remaining portions of the best-fitting free chunk back into free_.
543 AddFreeChunk(best.offset, new_offset - best.offset);
544 AddFreeChunk(new_end, best.chunk_end() - new_end);
545 return;
546 }
547
548 // The buffer doesn't completely fit into any existing free chunk. If the
549 // last free chunk is adjacent to the end of the heap, allocate the buffer
550 // re-using that space, increasing the heap size.
551 //
552 // Allocating the buffer now causes the heap to grow by less than the buffer
553 // size, whereas if we allocated lazily in Free, the heap would grow by
554 // exactly the buffer size. However it's still a greedy heuristical approach;
555 // we might have ended up with a tighter packing by being lazy here.
556 //
557 // In theory we could also check if we could re-use space from the first free
558 // chunk and grow the heap at the front, and choose whether to grow from the
559 // front or back based on the amount of re-use. But that's more complicated,
560 // and these are all heuristics anyways, so it isn't implemented.
561 for (auto it = free_.begin(); it != free_.end(); ++it) {
562 if (it->chunk_end() == result_.heap_size) {
563 // Account for alignment in the last free chunk.
564 const Chunk last = *it;
565 const int64 new_offset = RoundUpToNearest(last.offset, alignment_);
566 if (new_offset >= last.chunk_end()) {
567 // There's no point in using the last free chunk if alignment causes us
568 // to skip over it anyways.
569 break;
570 }
571 // The buffer is allocated a chunk that includes the last free chunk.
572 free_.erase(it);
573 result_.chunk_map.emplace(buffer, Chunk{new_offset, size});
574 // Add remaining portion of the last free chunk back into free_.
575 AddFreeChunk(last.offset, new_offset - last.offset);
576 // Grow the heap.
577 const int64 new_end = new_offset + size;
578 CHECK_GT(new_end, result_.heap_size);
579 CHECK_LT(new_end, result_.heap_size + size);
580 result_.heap_size = new_end;
581 return;
582 }
583 }
584
585 // Otherwise lazily allocate the buffer in Free.
586 result_.chunk_map.emplace(buffer, Chunk{kLazyAllocOffset, size});
587}

589void LazyBestFitHeap::Free(const LogicalBuffer* buffer, int64 size) {
590 auto alloc_it = result_.chunk_map.find(buffer);
591 CHECK(alloc_it != result_.chunk_map.end())
592 << "Free called on non-allocated buffer: " << *buffer;
593 Chunk* alloc = &alloc_it->second;
594 CHECK_EQ(alloc->size, size) << "Free with mismatched sizes: " << *buffer;
595 if (alloc->offset != kLazyAllocOffset) {
596 // The buffer was already allocated in Alloc, do a normal free.
597 AddFreeChunk(alloc->offset, alloc->size);
598 } else {
599 // This buffer is lazily allocated, so we *can not* allocate out of existing
600 // free chunks, since that might cause interference between buffers. The
601 // buffer is allocated by growing the heap, accounting for alignment.
602 alloc->offset = RoundUpToNearest(result_.heap_size, alignment_);
603 const int64 new_end = alloc->chunk_end();
604 AddFreeChunk(result_.heap_size, new_end - result_.heap_size);
605 CHECK_GT(new_end, result_.heap_size);
606 CHECK_GE(new_end, result_.heap_size + alloc->size);
607 result_.heap_size = new_end;
608 }
609}
610
611void LazyBestFitHeap::AddFreeChunk(int64 offset, int64 size) {
612 if (size <= 0) {
613 return;
614 }
615
616 // Coalesce the chunk with adjacent free chunks on either side. We must
617 // remove the free chunks from free_, since it's ordered by size.
618 Chunk chunk{offset, size};
619 for (auto it = free_.begin(); it != free_.end();) {
620 if (it->chunk_end() == chunk.offset || it->offset == chunk.chunk_end()) {
621 chunk.offset = std::min(chunk.offset, it->offset);
622 chunk.size += it->size;
623 it = free_.erase(it);
624 } else {
625 ++it;
626 }
627 }
628
629 // This is the only place we add free chunks to free_. It maintains the
630 // invariant that all free chunks are disjoint and non-adjacent.
631 free_.emplace(chunk);
632}
633
634HeapSimulator::Result LazyBestFitHeap::Finish() {
635 if (!free_.empty()) {
636 // When Finish is called, all calls to Alloc must have had corresponding
637 // calls to Free, which will result in a single free chunk [0, heap_size).
638 CHECK_EQ(free_.size(), 1);
639 CHECK_EQ(free_.begin()->offset, 0);
640 CHECK_EQ(free_.begin()->size, result_.heap_size);
641 }
642 return result_;
643}
