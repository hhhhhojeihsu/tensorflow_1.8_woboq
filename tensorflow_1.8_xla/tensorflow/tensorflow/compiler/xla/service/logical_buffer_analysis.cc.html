28// Gather fusion instructions from 'instruction' into 'fusion_instructions'.
29void GatherFusionInstructions(
30 HloInstruction* instruction,
31 std::vector<HloInstruction*>* fusion_instructions) {
32 CHECK_EQ(HloOpcode::kFusion, instruction->opcode());
33 for (auto* fused : instruction->fused_instructions()) {
34 if (fused->opcode() == HloOpcode::kFusion) {
35 GatherFusionInstructions(fused, fusion_instructions);
36 }
37 }
38 fusion_instructions->push_back(instruction);
39}

43/* static */ StatusOr<std::unique_ptr<LogicalBufferAnalysis>>
44LogicalBufferAnalysis::Run(const HloModule* module) {
45 std::unique_ptr<LogicalBufferAnalysis> analysis(
46 new LogicalBufferAnalysis(module));
47 TF_RETURN_IF_ERROR(analysis->Analyze());
48 return std::move(analysis);
49}
50
51Status LogicalBufferAnalysis::Analyze() {
52 // Empirically we usually have a few more logical buffers than instructions,
53 // so reserve 10% more than the number of instructions to avoid frequent
54 // resizes.
55 logical_buffers_.clear();
56 logical_buffers_.reserve((module_->NumUniqueInstructionIds() * 11) / 10);
57
58 // We filter out fusion computations, and get to them through fusion
59 // instructions. This is because it's possible to have orphaned (unreachable)
60 // fusion computations, and we don't want to try to assign buffers to those.
61 std::vector<HloInstruction*> fusion_instructions;
62 for (auto* computation : module_->MakeNonfusionComputations()) {
63 TF_RETURN_IF_ERROR(computation->Accept(this));
64 for (auto* instruction : computation->instructions()) {
65 if (instruction->opcode() != HloOpcode::kFusion) {
66 continue;
67 }
68 GatherFusionInstructions(instruction, &fusion_instructions);
69 }
70 }
71 for (auto* instruction : fusion_instructions) {
72 TF_RETURN_IF_ERROR(instruction->fused_expression_root()->Accept(this));
73 }
74 return Status::OK();
75}

77LogicalBuffer& LogicalBufferAnalysis::GetBuffer(LogicalBuffer::Id id) const {
78 CHECK_GE(id, 0);
79 CHECK_LT(id, logical_buffers_.size());
80 return *logical_buffers_[id];
81}
82
83LogicalBuffer& LogicalBufferAnalysis::GetBuffer(HloInstruction* instruction,
84 const ShapeIndex& index) const {
85 return *output_buffers_.at(std::make_pair(instruction, index));
86}
87
88void LogicalBufferAnalysis::NewLogicalBuffer(HloInstruction* instruction,
89 const ShapeIndex& index) {
90 CHECK_EQ(logical_buffers_.size(), next_buffer_id_);
91 logical_buffers_.emplace_back(
92 MakeUnique<LogicalBuffer>(instruction, index, next_buffer_id_));
93 output_buffers_[std::make_pair(instruction, index)] =
94 logical_buffers_.back().get();
95
96 ++next_buffer_id_;
97}

99Status LogicalBufferAnalysis::DefaultAction(HloInstruction* hlo_instruction) {
100 // Create a logical buffer for each output of the instruction.
101 ShapeUtil::ForEachSubshape(
102 hlo_instruction->shape(),
103 [this, hlo_instruction](const Shape& shape, const ShapeIndex& index) {
104 NewLogicalBuffer(hlo_instruction, index);
105 });
106
107 return Status::OK();
108}

110Status LogicalBufferAnalysis::HandleGetTupleElement(HloInstruction*) {
111 // GetTupleElement does not create buffers.
112 return Status::OK();
113}

115Status LogicalBufferAnalysis::HandleCopy(HloInstruction* copy) {
116 // The top-level buffer (index={}) for kCopy is newly created, but all other
117 // buffers (in the case of a tuple shape) come from the operand
118 NewLogicalBuffer(copy, /*index=*/{});
119 return Status::OK();
120}
121
122Status LogicalBufferAnalysis::HandleBitcast(HloInstruction*) {
123 // A kBitcast instruction aliases its operand. That is, the buffer of its
124 // result *is* the buffer of its operand.
125 return Status::OK();
126}
127
128Status LogicalBufferAnalysis::HandleRecvDone(HloInstruction*) {
129 // RecvDone doesn't create a new buffer but rather aliases its input (Recv)
130 // tuple element at {0} to its output.
131 return Status::OK();
132}
133
134Status LogicalBufferAnalysis::HandleSend(HloInstruction* send) {
135 // Send creates new buffers for the top-level tuple and the context (tuple
136 // element at {1}). Tuple element at {0} is an alias of the Send operand, so
137 // we don't need to create a new Logical Buffer for that.
138 NewLogicalBuffer(send, /*index=*/{});
139 NewLogicalBuffer(send, /*index=*/{1});
140 return Status::OK();
141}
142
143Status LogicalBufferAnalysis::HandleTuple(HloInstruction* tuple) {
144 // A Tuple instruction only creates the top-level buffer.
145 NewLogicalBuffer(tuple, /*index=*/{});
146 return Status::OK();
147}
148
149Status LogicalBufferAnalysis::HandleSelect(HloInstruction* select) {
150 // Select allocates a new buffer and then shallow copies the on_true or
151 // on_false buffer into this new buffer.
152 NewLogicalBuffer(select, /*index=*/{});
153 return Status::OK();
154}
