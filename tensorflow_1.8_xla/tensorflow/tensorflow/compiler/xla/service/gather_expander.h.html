21namespace xla {
22
23// This pass rewrites gather operations into (roughly) while loops of dynamic
24// slices. This lets backends that don't support gather directly to
25// nevertheless have a minimum level of support.
26class GatherExpander : public HloPassInterface {
27 public:
28 tensorflow::StringPiece name() const override { return "gather_expander"; }
29 StatusOr<bool> Run(HloModule* module) override;
30
31 private:
32 StatusOr<HloInstruction*> ExpandGather(HloInstruction* gather_instr);
33};

35} // namespace xla
