25// Some lightweight utilities intended to make HLO instruction creation more
26// ergonomic. We don't have a complete set of helpers yet -- I expect we'll
27// expand this interface as needed on an ad-hoc basis.
28
29// Creates a binary HLO instruction and adds it to the computation containing
30// `lhs` and `rhs` (`lhs` and `rhs` must be in the same computation).
31StatusOr<HloInstruction*> MakeBinaryHlo(HloOpcode opcode, HloInstruction* lhs,
32 HloInstruction* rhs);
33
34// Creates a pad HLO instruction and adds it to the computation containing
35// `operand` and `padding_value` (`operand` and `padding_value` must be in the
36// same computation).
37StatusOr<HloInstruction*> MakePadHlo(HloInstruction* operand,
38 HloInstruction* padding_value,
39 const PaddingConfig& padding_config);
40
41// Creates a slice HLO instruction and adds it to the computation containing
42// `operand`.
43StatusOr<HloInstruction*> MakeSliceHlo(
44 HloInstruction* operand, tensorflow::gtl::ArraySlice<int64> start_indices,
45 tensorflow::gtl::ArraySlice<int64> limit_indices,
46 tensorflow::gtl::ArraySlice<int64> strides);
47
48// Creates a convolution HLO instruction and adds it to the computation
49// containing `lhs` and `rhs` (`lhs` and `rhs` must be in the same computation).
50StatusOr<HloInstruction*> MakeConvolveHlo(
51 HloInstruction* lhs, HloInstruction* rhs, const Window& window,
52 const ConvolutionDimensionNumbers& dimension_numbers);
53
54// Creates a transpose HLO instruction and adds it to the computation containing
55// `operand`.
56StatusOr<HloInstruction*> MakeTransposeHlo(
57 HloInstruction* operand, tensorflow::gtl::ArraySlice<int64> dimensions);
58
59// Creates a reshape HLO instruction and adds it to the computation containing
60// `operand`.
61StatusOr<HloInstruction*> MakeReshapeHlo(const Shape& result_shape,
62 HloInstruction* operand);
63
64StatusOr<HloInstruction*> MakeReshapeHlo(
65 tensorflow::gtl::ArraySlice<int64> result_shape_dim_bounds,
66 HloInstruction* operand);
67
68// Creates a dynamic-slice HLO instruction and adds it to the computation
69// containing `operand` and `start_indices` (`operand` and `start_indices` must
70// be in the same computation).
71StatusOr<HloInstruction*> MakeDynamicSliceHlo(
72 HloInstruction* operand, HloInstruction* start_indices,
73 tensorflow::gtl::ArraySlice<int64> slice_sizes);
74
75// Creates a dynamic-update-slice HLO instruction and adds it to the computation
76// containing `operand`, `update` and `start_indices` (`operand`, `update` and
77// `start_indices` must be in the same computation).
78StatusOr<HloInstruction*> MakeDynamicUpdateSliceHlo(
79 HloInstruction* operand, HloInstruction* update,
80 HloInstruction* start_indices);
81
82// Creates a broadcast HLO instruction and adds it to the computation containing
83// `operand`.
84StatusOr<HloInstruction*> MakeBroadcastHlo(
85 HloInstruction* operand,
86 tensorflow::gtl::ArraySlice<int64> broadcast_dimensions,
87 tensorflow::gtl::ArraySlice<int64> result_shape_bounds);
88
89// Creates a GetTupleElement HLO instruction and adds it to the computation
90// containing `operand`.
91StatusOr<HloInstruction*> MakeGetTupleElementHlo(HloInstruction* operand,
92 int64 index);
93
94// Creates a Concatenate HLO instruction and adds it to the computation
95// containing `operands` (`operands` must be non-empty and every element must be
96// contained in the same computation).
97StatusOr<HloInstruction*> MakeConcatHlo(
98 tensorflow::gtl::ArraySlice<HloInstruction*> operands, int64 dimension);

100// -----------------------------------------------------------------------------
101// Some other miscellaneous helpers to generate common HLO patterns. All of
102// these add all the instructions they generate into the computation containing
103// their operand(s).
104
105// Collapses (via reshape) the first N (logical) dimensions of `operand` into a
106// single leading dimension. `operand` must have rank > n.
107//
108// For instance if `operand` has shape f32[7,8,9] and n is 2 then the output is
109// the `operand` reshaped to [56,9].
110StatusOr<HloInstruction*> CollapseFirstNDims(HloInstruction* operand, int64 n);
111
112// Expands (via reshape) the first (logical) dimension of `operand` into a
113// sequence of `expanded_dims` dimensions. `operand` must at least be of rank 1
114// and the number of elements in its first dimension must be equal to the
115// product of `expanded_dims`.
116//
117// For instance if `operand` has shape f32[200,9,7] and expanded_dims is
118// {2,5,20} the result is `operand` reshaped to [2,5,20,9,7].
119StatusOr<HloInstruction*> ExpandFirstDimIntoNDims(
120 HloInstruction* operand, tensorflow::gtl::ArraySlice<int64> expanded_dims);
121
122// Elides (via reshape) a set of degenerate dimensions (dimensions containing
123// exactly one element), `dims_to_elide` from `operand`. Every dimension in
124// `dims_to_elide` must be a degenerate dimension. `dims_to_elide` must be
125// sorted and not contain duplicates.
126//
127// For example if `operand` is of shape f32[19,1,20,1,7,1,9] and dims_to_elide
128// is {1,5} then the result is `operand` reshaped to [19,20,1,7,9].
129StatusOr<HloInstruction*> ElideDegenerateDims(
130 HloInstruction* operand, tensorflow::gtl::ArraySlice<int64> dims_to_elide);
131
132// Pads `operand` (which must have rank 1) with `zeros_to_prepend` zeros in the
133// front and `zeros_to_append` zeros in the back.
134StatusOr<HloInstruction*> PadVectorWithZeros(HloInstruction* operand,
135 int64 zeros_to_prepend,
136 int64 zeros_to_append);
137
138// Broadcasts a zero value of type `element_type` into a tensor with element
139// type `element_type` and dimension bounds `broadcast_dimensions`. The
140// broadcast instruction is emitted into `computation`.
141StatusOr<HloInstruction*> BroadcastZeros(
142 HloComputation* computation, PrimitiveType element_type,
143 tensorflow::gtl::ArraySlice<int64> broadcast_dimensions);
144
145// Creates a HLO computation that takes arguments of type `domain` and produces
146// a value of type `range`.
147StatusOr<std::unique_ptr<HloComputation>> CreateComputationWithSignature(
148 tensorflow::gtl::ArraySlice<const Shape*> domain, const Shape& range,
149 tensorflow::StringPiece name);
