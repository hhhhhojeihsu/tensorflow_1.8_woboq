16// Shape inference is used by the XLA service as the user builds up
17// computation requests.
18

22#include <vector>
23
24#include "tensorflow/compiler/xla/service/hlo_instruction.h"
25#include "tensorflow/compiler/xla/service/hlo_opcode.h"

33namespace xla {

35// For a given operation and input shapes, infers what the resulting shape is
36// for the operation. With this functionality, the user does not need to specify
37// the expected result type for computations that are built up via the API --
38// the shape that results from an operation is inferred. Some methods have
39// overloads for inferring shape at the HLO level.
40//
41// TODO(b/73352135): Shape inference does not issue very good error messages, in
42// part because HloInstruction::ToString() is not available since shape
43// inference runs before the HloInstruction object is created. We need a
44// solution for this.
45class ShapeInference {
46 public:
47 // Infers the shape produced by applying the given unary operation to the
48 // given input shape.
49 static StatusOr<Shape> InferUnaryOpShape(UnaryOperation operation,
50 const Shape& arg);
51 static StatusOr<Shape> InferUnaryOpShape(HloOpcode opcode,
52 const Shape& shape);
53 static StatusOr<Shape> InferUnaryOpShape(HloOpcode opcode,
54 const HloInstruction* operand);
55
56 // Infers the shape produced by applying the given binary operation to the
57 // given input shapes.
58 static StatusOr<Shape> InferBinaryOpShape(
59 BinaryOperation operation, const Shape& lhs, const Shape& rhs,
60 tensorflow::gtl::ArraySlice<int64> broadcast_dimensions);
61 static StatusOr<Shape> InferBinaryOpShape(
62 HloOpcode opcode, const Shape& lhs, const Shape& rhs,
63 tensorflow::gtl::ArraySlice<int64> broadcast_dimensions);
64 static StatusOr<Shape> InferBinaryOpShape(HloOpcode opcode,
65 const HloInstruction* lhs,
66 const HloInstruction* rhs);
67
68 // Infers the shape produced by applying the given ternary operation to the
69 // given input shapes.
70 static StatusOr<Shape> InferTernaryOpShape(TernaryOperation operation,
71 const Shape& lhs, const Shape& rhs,
72 const Shape& ehs);
73 static StatusOr<Shape> InferTernaryOpShape(HloOpcode opcode, const Shape& lhs,
74 const Shape& rhs,
75 const Shape& ehs);
76 static StatusOr<Shape> InferTernaryOpShape(HloOpcode opcode,
77 const HloInstruction* lhs,
78 const HloInstruction* rhs,
79 const HloInstruction* ehs);
80
81 // Infers the shape produced by applying the given variadic operation to the
82 // given input operand shapes.
83 static StatusOr<Shape> InferVariadicOpShape(
84 VariadicOperation operation,
85 tensorflow::gtl::ArraySlice<const Shape*> operand_shapes);
86 static StatusOr<Shape> InferVariadicOpShape(
87 HloOpcode opcode,
88 tensorflow::gtl::ArraySlice<const Shape*> operand_shapes);
89 static StatusOr<Shape> InferVariadicOpShape(
90 HloOpcode opcode,
91 tensorflow::gtl::ArraySlice<const HloInstruction*> operands);
92
93 // Infers the shape produced by applying the given mapping computation shape
94 // to the given operand shapes.
95 static StatusOr<Shape> InferMapShape(
96 tensorflow::gtl::ArraySlice<const Shape*> arg_shapes,
97 const ProgramShape& to_apply,
98 tensorflow::gtl::ArraySlice<int64> dimensions);
99
100 // Infers the shape produced by InferBatchNormTraining with the given
101 // operands.
102 static StatusOr<Shape> InferBatchNormTrainingShape(const Shape& operand_shape,
103 const Shape& scale_shape,
104 const Shape& offset_shape,
105 int64 feature_index);
106
107 // Infers the shape produced by InferBatchNormInference with the given
108 // operands.
109 static StatusOr<Shape> InferBatchNormInferenceShape(
110 const Shape& operand_shape, const Shape& scale_shape,
111 const Shape& offset_shape, const Shape& mean_shape,
112 const Shape& variance_shape, int64 feature_index);
113
114 // Infers the shape produced by InferBatchNormGrad with the given operands.
115 static StatusOr<Shape> InferBatchNormGradShape(const Shape& operand_shape,
116 const Shape& scale_shape,
117 const Shape& mean_shape,
118 const Shape& var_shape,
119 const Shape& output_grad_shape,
120 int64 feature_index);
121
122 // Infers the shape produced by applying the given convolutional
123 // filter (rhs) to lhs in the way specified by the fields on window.
124 static StatusOr<Shape> InferConvolveShape(
125 const Shape& lhs, const Shape& rhs, const Window& window,
126 const ConvolutionDimensionNumbers& dimension_numbers);
127
128 // Infers the shape produced by the given FFT type on the given operand.
129 static StatusOr<Shape> InferFftShape(
130 const Shape& in, FftType fft_type,
131 tensorflow::gtl::ArraySlice<int64> fft_length);
132
133 // Infers the shape produced a cross replica sum with the given operand
134 // shapes.
135 static StatusOr<Shape> InferCrossReplicaSumShape(
136 tensorflow::gtl::ArraySlice<const Shape*> operand_shapes);
137
138 // Infers the shape produced by applying the given reduction computation
139 // shape to the given input operand shape.
140 //
141 // If pass_index is true, the reduce function is invoked with the element
142 // index as the leading parameter, and the program shape should match
143 // accordingly (or an error will result).
144 static StatusOr<Shape> InferReduceShape(
145 const Shape& arg, const Shape& init_value,
146 tensorflow::gtl::ArraySlice<int64> dimensions_to_reduce,
147 const ProgramShape& to_apply);
148
149 // Infers the shape produced by applying the given computation to the operand
150 // shape with the given window and stride dimensions.
151 static StatusOr<Shape> InferReduceWindowShape(
152 const Shape& operand_shape, const Shape& init_value, const Window& window,
153 const ProgramShape& to_apply_shape);
154
155 // Infers the shape produced by scattering the given source shape to the
156 // selected indices of each window on the operand shape.
157 static StatusOr<Shape> InferSelectAndScatterShape(
158 const Shape& operand_shape, const ProgramShape& select_shape,
159 const Window& window, const Shape& source_shape,
160 const Shape& init_value_shape, const ProgramShape& scatter_shape);
161
162 // Infers the shape produced by a reverse operation that reverses the order
163 // of the elements in the given dimensions.
164 static StatusOr<Shape> InferReverseShape(
165 const Shape& operand_shape,
166 tensorflow::gtl::ArraySlice<int64> dimensions);
167
168 // Infers the shape produced by a slice operation spanning from the starts to
169 // the limits in the original shape's dimensions.
170 //
171 // e.g. slice f32[32x32] 0:16 0:16 -> f32[16x16]
172 static StatusOr<Shape> InferSliceShape(
173 const Shape& arg, tensorflow::gtl::ArraySlice<int64> starts,
174 tensorflow::gtl::ArraySlice<int64> limits,
175 tensorflow::gtl::ArraySlice<int64> strides);
176
177 // Infers the shape produced by a dynamic slice operation of size specified
178 // in 'slice_sizes', with dynamic start indices shape 'start_indices_shape'.
179 static StatusOr<Shape> InferDynamicSliceShape(
180 const Shape& operand_shape, const Shape& start_indices_shape,
181 tensorflow::gtl::ArraySlice<int64> slice_sizes);
182
183 // Infers the shape produced by a dynamic update slice operation based
184 // on the shape of operand and update.
185 static StatusOr<Shape> InferDynamicUpdateSliceShape(
186 const Shape& operand_shape, const Shape& update_shape,
187 const Shape& start_indices_shape);
188
189 // Infers the shape produced by doing a compile-time-constant indexing into
190 // the given input shape. This is essential for operations on tuples, because
191 // it is impossible to infer the type that comes out of the tuple indexing if
192 // it is not a compile time constant.
193 static StatusOr<Shape> InferGetTupleElementShape(const Shape& arg,
194 int64 index);
195
196 // Infers the shape produced from a while node. condition and body are the
197 // shapes of computations for the condition and the body of a while node, and
198 // init is the shape of data initially passed in to the body as an argument.
199 // The shapes must match; condition: T -> PRED, body: T -> T, init: T
200 static StatusOr<Shape> InferWhileShape(const ProgramShape& condition,
201 const ProgramShape& body,
202 const Shape& init);
203
204 // Infers the shape produced by a conditional operation.
205 static StatusOr<Shape> InferConditionalShape(
206 const Shape& predicate, const Shape& true_operand,
207 const Shape& false_operand, const ProgramShape& true_computation,
208 const ProgramShape& false_computation);
209
210 // Infers the shape produced by a broadcast operation.
211 static StatusOr<Shape> InferBroadcastShape(
212 const Shape& operand, tensorflow::gtl::ArraySlice<int64> broadcast_sizes);
213
214 // Infers the shape produced by a reshape operation from the element type of
215 // its operand and the new dimension sizes specified.
216 static StatusOr<Shape> InferReshapeShape(
217 const Shape& operand, tensorflow::gtl::ArraySlice<int64> dimensions,
218 tensorflow::gtl::ArraySlice<int64> new_sizes);
219
220 // Infers the shape produced by a transpose operation from the element type of
221 // its operand and its dimensions field.
222 static StatusOr<Shape> InferTransposeShape(
223 const Shape& operand, tensorflow::gtl::ArraySlice<int64> dimensions);
224
225 // Helper that infers the shape produced by performing a concatenate operation
226 // with the given operand shapes.
227 static StatusOr<Shape> InferConcatOpShape(
228 tensorflow::gtl::ArraySlice<const Shape*> arg_shapes, int64 dimension);
229
230 // Helper that validates the given operand shape can be converted to the
231 // target output_shape via a convert instruction -- the requirement is that
232 // the shape is identical except for the element type.
233 static StatusOr<Shape> InferConvertShape(const Shape& operand_shape,
234 PrimitiveType new_element_type);
235
236 // Helper that validates the given operand shape can be bitcast converted to
237 // the target output_shape via a bitcast convert instruction -- the
238 // requirement is that the shape is identical except for the element type and
239 // the element types have identical bit-widths.
240 static StatusOr<Shape> InferBitcastConvertShape(
241 const Shape& operand_shape, PrimitiveType new_element_type);
242
243 // Helper that validates the input data type for a reduce-precision operation,
244 // and returns the result shape.
245 static StatusOr<Shape> InferReducePrecisionShape(const Shape& operand_shape,
246 const int exponent_bits,
247 const int mantissa_bits);
248
249 // Helper that infers the shape produced by a pad operation based on the
250 // padding configuration.
251 static StatusOr<Shape> InferPadShape(const Shape& operand_shape,
252 const Shape& padding_value_shape,
253 const PaddingConfig& padding_config);
254
255 // Helper that validates the given arg_shapes are compatible with the shape of
256 // the to_apply parameters, and returns the to_apply result shape.
257 static StatusOr<Shape> InferCallShape(
258 tensorflow::gtl::ArraySlice<const Shape*> arg_shapes,
259 const ProgramShape& to_apply);
260
261 // Helper that infers the shape produced by performing a dot operation with
262 // the given LHS and RHS shapes.
263 static StatusOr<Shape> InferDotOpShape(
264 const Shape& lhs, const Shape& rhs,
265 const DotDimensionNumbers& dimension_numbers);
266
267 // Helper that infers the shape of the tensor produced by a gather operation
268 // with the given input shape, gather indices shape and gather dimension
269 // numbers.
270 static StatusOr<Shape> InferGatherShape(
271 const Shape& input_shape, const Shape& gather_indices_shape,
272 const GatherDimensionNumbers& gather_dim_numbers,
273 tensorflow::gtl::ArraySlice<int64> window_bounds);
274
275 private:
276 // Helper that infers the shape produced by performing an element-wise binary
277 // operation with the given LHS and RHS shapes.
278 // Note: By "element-wise" we mean operations that look at a single element in
279 // the LHS and a single element in the RHS to produce a single output element,
280 // even in the presence of broadcasting of one of the operands over the other.
281 static StatusOr<Shape> InferElementwiseBinaryOpShape(
282 BinaryOperation operation, const Shape& lhs, const Shape& rhs,
283 tensorflow::gtl::ArraySlice<int64> broadcast_dimensions);
284
285 // Helper for inferring the shape of Clamp ops.
286 static StatusOr<Shape> InferClampShape(const Shape& min, const Shape& operand,
287 const Shape& max);
288
289 // Helper for inferring the shape of Select ops.
290 static StatusOr<Shape> InferSelectShape(const Shape& pred,
291 const Shape& on_true,
292 const Shape& on_false);
293
294 // Helper for inferring shapes of binary operations which use degenerate
295 // dimension broadcasting (a dimension of size 1 in one operand is broadcast
296 // up to match the size of the dimension in the other operand).
297 static StatusOr<Shape> InferDegenerateDimensionBroadcastShape(
298 BinaryOperation operation, const Shape& lhs, const Shape& rhs);
299
300 // Helper for inferring shapes of binary operations using "InDim"
301 // broadcasting. This is the broadcasting used in the *InDim binary operations
302 // (for example ComputationBuilder::AddInDim). smaller_shape must be a
303 // lower-rank shape than larger_shape. Returns the shape that the
304 // smaller_shape is broadcast to.
305 static StatusOr<Shape> InferInDimBroadcastShape(
306 BinaryOperation operation, const Shape& smaller_shape,
307 const Shape& larger_shape,
308 tensorflow::gtl::ArraySlice<int64> broadcast_dimensions);
309
310 TF_DISALLOW_COPY_AND_ASSIGN(ShapeInference);
311};
312
313} // namespace xla
