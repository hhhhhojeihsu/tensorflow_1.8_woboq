55llvm::Value* EmitReducePrecisionFloat(llvm::Value* x, int64 exponent_bits,
56 int64 mantissa_bits,
57 llvm::IRBuilder<>* ir_builder) {
58 // Integer and float types for casting and constant generation.
59 llvm::Type* float_type = x->getType();
60 llvm::IntegerType* int_type = ir_builder->getInt32Ty();
61
62 // Cast the input value to an integer for bitwise manipulation.
63 llvm::Value* x_as_int = ir_builder->CreateBitCast(x, int_type);
64
65 if (mantissa_bits < 23) {
66 // Last remaining mantissa bit.
67 const uint32_t last_mantissa_bit_mask = 1u << (23 - mantissa_bits);
68
69 // Compute rounding bias for round-to-nearest with ties to even. This is
70 // equal to a base value of 0111... plus one bit if the last remaining
71 // mantissa bit is 1.
72 const uint32_t base_rounding_bias = (last_mantissa_bit_mask >> 1) - 1;
73 llvm::Value* x_last_mantissa_bit = ir_builder->CreateLShr(
74 ir_builder->CreateAnd(
75 x_as_int, llvm::ConstantInt::get(int_type, last_mantissa_bit_mask)),
76 (23 - mantissa_bits));
77 llvm::Value* x_rounding_bias = ir_builder->CreateAdd(
78 x_last_mantissa_bit,
79 llvm::ConstantInt::get(int_type, base_rounding_bias));
80
81 // Add rounding bias, and mask out truncated bits. Note that the case
82 // where adding the rounding bias overflows into the exponent bits is
83 // correct; the non-masked mantissa bits will all be zero, and the
84 // exponent will be incremented by one.
85 const uint32_t truncation_mask = ~(last_mantissa_bit_mask - 1);
86 x_as_int = ir_builder->CreateAdd(x_as_int, x_rounding_bias);
87 x_as_int = ir_builder->CreateAnd(
88 x_as_int, llvm::ConstantInt::get(int_type, truncation_mask));
89 }
90
91 if (exponent_bits < 8) {
92 // Masks for f32 values.
93 const uint32_t f32_sign_bit_mask = 1u << 31;
94 const uint32_t f32_exp_bits_mask = 0xffu << 23;
95
96 // An exponent of 2^(n-1)-1 -- that is, 0111... with the zero in the most-
97 // significant bit -- is equal to 1.0f for all exponent sizes. Adding
98 // 2^(n-1)-1 to this gives us the highest non-infinite exponent for a bit-
99 // size of n, and subtracting 2^(n-1)-1 from this gives us the lowest'
100 // exponent (corresponding to 0.0f).
101 //
102 // Thus, the f32 exponent corresponding to the highest non-infinite
103 // exponent for a bit size of n is (2^7-1) + 2^(n-1)-1, and the f32
104 // exponent corresponding to the lowest exponent for a bit size of n is
105 // (2^7-1) - 2^(n-1)-1.
106 //
107 // Note that we have already checked that exponents_bits >= 1.
108 const uint32_t f32_exponent_bias = (1 << 7) - 1;
109 const uint32_t reduced_exponent_bias = (1 << (exponent_bits - 1)) - 1;
110 const uint32_t reduced_max_exponent =
111 f32_exponent_bias + reduced_exponent_bias;
112 const uint32_t reduced_min_exponent =
113 f32_exponent_bias - reduced_exponent_bias;
114
115 // Do we overflow or underflow?
116 llvm::Value* x_exponent = ir_builder->CreateAnd(
117 x_as_int, llvm::ConstantInt::get(int_type, f32_exp_bits_mask));
118 llvm::Value* x_overflows = ir_builder->CreateICmpUGT(
119 x_exponent,
120 llvm::ConstantInt::get(int_type, reduced_max_exponent << 23));
121 llvm::Value* x_underflows = ir_builder->CreateICmpULE(
122 x_exponent,
123 llvm::ConstantInt::get(int_type, reduced_min_exponent << 23));
124
125 // Compute appropriately-signed values of zero and infinity.
126 llvm::Value* x_signed_zero = ir_builder->CreateAnd(
127 x_as_int, llvm::ConstantInt::get(int_type, f32_sign_bit_mask));
128 llvm::Value* x_signed_inf = ir_builder->CreateOr(
129 x_signed_zero, llvm::ConstantInt::get(int_type, f32_exp_bits_mask));
130
131 // Force to zero or infinity if overflow or underflow. (Note that this
132 // truncates all denormal values to zero, rather than rounding them.)
133 x_as_int = ir_builder->CreateSelect(x_overflows, x_signed_inf, x_as_int);
134 x_as_int = ir_builder->CreateSelect(x_underflows, x_signed_zero, x_as_int);
135 }
136
137 // Cast the result back to a floating-point type.
138 llvm::Value* result = ir_builder->CreateBitCast(x_as_int, float_type);
139
140 // Correct result for NaN inputs.
141 //
142 // The exponent handling will "normalize" NaN values to infinities, which is
143 // undesirable (except in the case with no mantissa bits, in which case it
144 // is mandatory). This logic also handles cases where mantissa-rounding
145 // causes a NaN's mantissa to overflow into the exponent bits, which would
146 // otherwise create an erroneous zero value.
147 //
148 // If the fast-math flags are set to assume no NaNs, the comparison is likely
149 // to be optimized away, so there's no point in even emitting it.
150 if (!ir_builder->getFastMathFlags().noNaNs()) {
151 llvm::Value* x_is_nan = ir_builder->CreateFCmpUNO(x, x);
152
153 if (mantissa_bits > 0) {
154 result = ir_builder->CreateSelect(x_is_nan, x, result);
155 } else {
156 result = ir_builder->CreateSelect(
157 x_is_nan, llvm::ConstantFP::getInfinity(float_type), result);
158 }
159 }
160 return result;
161}

163llvm::Value* EmitF32ToBF16(llvm::Value* f32_value,
164 llvm::IRBuilder<>* ir_builder) {
165 auto reduced_precision = EmitReducePrecisionFloat(
166 f32_value,
167 /*exponent_bits=*/primitive_util::kBFloat16ExponentBits,
168 /*mantissa_bits=*/primitive_util::kBFloat16MantissaBits, ir_builder);
169 auto as_int32 =
170 ir_builder->CreateBitCast(reduced_precision, ir_builder->getInt32Ty());
171 auto shifted = ir_builder->CreateLShr(as_int32, 16);
172 auto truncated = ir_builder->CreateTrunc(shifted, ir_builder->getInt16Ty());
173 return ir_builder->CreateBitCast(truncated, ir_builder->getInt16Ty());
174}
175
176llvm::Value* EmitBF16ToF32(llvm::Value* bf16_value,
177 llvm::IRBuilder<>* ir_builder) {
178 auto as_int16 =
179 ir_builder->CreateBitCast(bf16_value, ir_builder->getInt16Ty());
180 auto as_int32 = ir_builder->CreateZExt(as_int16, ir_builder->getInt32Ty());
181 auto shifted = ir_builder->CreateShl(as_int32, 16);
182 return ir_builder->CreateBitCast(shifted, ir_builder->getFloatTy());
183}

185llvm::Value* EmitIntegralToFloating(llvm::Value* integer_value,
186 PrimitiveType from_type,
187 PrimitiveType to_type, llvm::Module* module,
188 llvm::IRBuilder<>* ir_builder) {
189 if (primitive_util::IsSignedIntegralType(from_type)) {
190 return ir_builder->CreateSIToFP(
191 integer_value, llvm_ir::PrimitiveTypeToIrType(to_type, module));
192 } else {
193 CHECK(primitive_util::IsUnsignedIntegralType(from_type) ||
194 from_type == PRED);
195 return ir_builder->CreateUIToFP(
196 integer_value, llvm_ir::PrimitiveTypeToIrType(to_type, module));
197 }
198}

202StatusOr<llvm::Value*> ElementalIrEmitter::EmitUnaryOp(
203 const HloInstruction* op, llvm::Value* operand_value) const {
204 if (op->opcode() == HloOpcode::kCopy) {
205 return operand_value;
206 } else if (ShapeUtil::ElementIsIntegral(op->operand(0)->shape()) ||
207 op->operand(0)->shape().element_type() == PRED) {
208 return EmitIntegerUnaryOp(op, operand_value);
209 } else if (ShapeUtil::ElementIsComplex(op->operand(0)->shape())) {
210 return EmitComplexUnaryOp(op, operand_value);
211 } else {
212 return EmitFloatUnaryOp(op, operand_value);
213 }
214}

216StatusOr<llvm::Value*> ElementalIrEmitter::EmitIntegerUnaryOp(
217 const HloInstruction* op, llvm::Value* operand_value) const {
218 switch (op->opcode()) {
219 case HloOpcode::kConvert: {
220 PrimitiveType from_type = op->operand(0)->shape().element_type();
221 PrimitiveType to_type = op->shape().element_type();
222 CHECK(primitive_util::IsIntegralType(from_type) || from_type == PRED);
223 if (from_type == to_type) {
224 return operand_value;
225 }
226 if (primitive_util::IsIntegralType(to_type)) {
227 return ir_builder_->CreateIntCast(
228 operand_value, llvm_ir::PrimitiveTypeToIrType(to_type, module_),
229 primitive_util::IsSignedIntegralType(from_type));
230 }
231 if (primitive_util::IsFloatingPointType(to_type)) {
232 if (to_type == BF16) {
233 return EmitF32ToBF16(
234 EmitIntegralToFloating(operand_value, from_type, F32, module_,
235 ir_builder_),
236 ir_builder_);
237 }
238 return EmitIntegralToFloating(operand_value, from_type, to_type,
239 module_, ir_builder_);
240 }
241 if (primitive_util::IsComplexType(to_type)) {
242 auto to_ir_component_type = llvm_ir::PrimitiveTypeToIrType(
243 primitive_util::ComplexComponentType(to_type), module_);
244 if (primitive_util::IsSignedIntegralType(from_type)) {
245 return EmitComposeComplex(
246 op,
247 ir_builder_->CreateSIToFP(operand_value, to_ir_component_type),
248 nullptr);
249 }
250 if (primitive_util::IsUnsignedIntegralType(from_type) ||
251 from_type == PRED) {
252 return EmitComposeComplex(
253 op,
254 ir_builder_->CreateUIToFP(operand_value, to_ir_component_type),
255 nullptr);
256 }
257 }
258 return Unimplemented("conversion from primitive type %s to %s",
259 PrimitiveType_Name(from_type).c_str(),
260 PrimitiveType_Name(to_type).c_str());
261 }
262 case HloOpcode::kBitcastConvert: {
263 PrimitiveType from_type = op->operand(0)->shape().element_type();
264 PrimitiveType to_type = op->shape().element_type();
265 CHECK(primitive_util::IsIntegralType(from_type));
266 if (from_type == to_type) {
267 return operand_value;
268 }
269 if (primitive_util::BitWidth(from_type) ==
270 primitive_util::BitWidth(to_type)) {
271 return ir_builder_->CreateBitCast(
272 operand_value, llvm_ir::PrimitiveTypeToIrType(to_type, module_));
273 }
274 return InvalidArgument(
275 "bitcast conversion from primitive type %s to %s with unequal "
276 "bit-widths (%u versus %u) ",
277 PrimitiveType_Name(from_type).c_str(),
278 PrimitiveType_Name(to_type).c_str(),
279 primitive_util::BitWidth(from_type),
280 primitive_util::BitWidth(to_type));
281 }
282 case HloOpcode::kAbs: {
283 bool is_signed =
284 primitive_util::IsSignedIntegralType(op->shape().element_type());
285 if (is_signed) {
286 auto type =
287 llvm_ir::PrimitiveTypeToIrType(op->shape().element_type(), module_);
288 auto zero = llvm::ConstantInt::get(type, 0);
289 auto cmp = ir_builder_->CreateICmpSGE(operand_value, zero);
290 return ir_builder_->CreateSelect(cmp, operand_value,
291 ir_builder_->CreateNeg(operand_value));
292 } else {
293 return operand_value;
294 }
295 }
296 case HloOpcode::kSign: {
297 bool is_signed =
298 primitive_util::IsSignedIntegralType(op->shape().element_type());
299 auto type =
300 llvm_ir::PrimitiveTypeToIrType(op->shape().element_type(), module_);
301 auto zero = llvm::ConstantInt::get(type, 0);
302 auto cmp = ir_builder_->CreateICmpEQ(operand_value, zero);
303 if (is_signed) {
304 auto ashr = ir_builder_->CreateAShr(operand_value,
305 type->getIntegerBitWidth() - 1);
306 return ir_builder_->CreateSelect(cmp, zero,
307 ir_builder_->CreateOr(ashr, 1));
308 } else {
309 return ir_builder_->CreateSelect(cmp, zero,
310 llvm::ConstantInt::get(type, 1));
311 }
312 }
313 case HloOpcode::kNegate:
314 return ir_builder_->CreateNeg(operand_value);
315 case HloOpcode::kNot: {
316 auto type = op->shape().element_type();
317 if (type == PRED) {
318 // It is not sufficient to just call CreateNot() here because a PRED
319 // is represented as an i8 and the truth value is stored only in the
320 // bottom bit.
321 return ir_builder_->CreateZExt(
322 ir_builder_->CreateNot(ir_builder_->CreateTrunc(
323 operand_value, ir_builder_->getInt1Ty())),
324 llvm_ir::PrimitiveTypeToIrType(PRED, module_));
325 } else if (primitive_util::IsIntegralType(type)) {
326 return ir_builder_->CreateNot(operand_value);
327 }
328 return Unimplemented("unary op Not is not defined for type '%d'", type);
329 }
330 default:
331 return Unimplemented("unary integer op '%s'",
332 HloOpcodeString(op->opcode()).c_str());
333 }

336StatusOr<llvm::Value*> ElementalIrEmitter::EmitFloatUnaryOp(
337 const HloInstruction* op, llvm::Value* operand_value) const {
338 switch (op->opcode()) {
339 case HloOpcode::kConvert: {
340 PrimitiveType from_type = op->operand(0)->shape().element_type();
341 PrimitiveType to_type = op->shape().element_type();
342 CHECK(primitive_util::IsFloatingPointType(from_type));
343 if (from_type == to_type) {
344 return operand_value;
345 }
346 if (primitive_util::IsComplexType(to_type)) {
347 PrimitiveType to_component_type =
348 primitive_util::ComplexComponentType(to_type);
349 if (from_type == to_component_type) {
350 return EmitComposeComplex(op, operand_value, nullptr);
351 }
352 return EmitComposeComplex(
353 op,
354 ir_builder_->CreateFPCast(
355 operand_value,
356 llvm_ir::PrimitiveTypeToIrType(to_component_type, module_)),
357 nullptr);
358 }
359 if (from_type == BF16) {
360 TF_RET_CHECK(to_type != BF16);
361 operand_value = EmitBF16ToF32(operand_value, ir_builder_);
362 from_type = F32;
363 if (from_type == to_type) {
364 return operand_value;
365 }
366 }
367 if (from_type == F32 && to_type == BF16) {
368 return EmitF32ToBF16(operand_value, ir_builder_);
369 }
370 if (primitive_util::IsFloatingPointType(to_type)) {
371 return ir_builder_->CreateFPCast(
372 operand_value, llvm_ir::PrimitiveTypeToIrType(to_type, module_));
373 }
374 if (primitive_util::IsSignedIntegralType(to_type)) {
375 return ir_builder_->CreateFPToSI(
376 operand_value, llvm_ir::PrimitiveTypeToIrType(to_type, module_));
377 }
378 if (primitive_util::IsUnsignedIntegralType(to_type)) {
379 return ir_builder_->CreateFPToUI(
380 operand_value, llvm_ir::PrimitiveTypeToIrType(to_type, module_));
381 }
382 return Unimplemented("unhandled conversion operation: %s => %s",
383 PrimitiveType_Name(from_type).c_str(),
384 PrimitiveType_Name(to_type).c_str());
385 }
386 case HloOpcode::kBitcastConvert: {
387 PrimitiveType from_type = op->operand(0)->shape().element_type();
388 PrimitiveType to_type = op->shape().element_type();
389 CHECK(primitive_util::IsFloatingPointType(from_type));
390 if (from_type == to_type) {
391 return operand_value;
392 }
393 if (primitive_util::BitWidth(from_type) ==
394 primitive_util::BitWidth(to_type)) {
395 return ir_builder_->CreateBitCast(
396 operand_value, llvm_ir::PrimitiveTypeToIrType(to_type, module_));
397 }
398 return InvalidArgument(
399 "bitcast conversion from primitive type %s to %s with unequal "
400 "bit-widths (%u versus %u) ",
401 PrimitiveType_Name(from_type).c_str(),
402 PrimitiveType_Name(to_type).c_str(),
403 primitive_util::BitWidth(from_type),
404 primitive_util::BitWidth(to_type));
405 }
406 case HloOpcode::kExp:
407 return EmitExp(op->shape().element_type(), operand_value);
408 case HloOpcode::kLog:
409 return EmitLog(op->shape().element_type(), operand_value);
410 case HloOpcode::kCos:
411 return EmitCos(op->shape().element_type(), operand_value);
412 case HloOpcode::kSin:
413 return EmitSin(op->shape().element_type(), operand_value);
414 case HloOpcode::kFloor:
415 return llvm_ir::EmitCallToIntrinsic(
416 llvm::Intrinsic::floor, {operand_value}, {operand_value->getType()},
417 ir_builder_);
418 case HloOpcode::kCeil:
419 return llvm_ir::EmitCallToIntrinsic(
420 llvm::Intrinsic::ceil, {operand_value}, {operand_value->getType()},
421 ir_builder_);
422 case HloOpcode::kAbs:
423 return llvm_ir::EmitCallToIntrinsic(
424 llvm::Intrinsic::fabs, {operand_value}, {operand_value->getType()},
425 ir_builder_);
426 case HloOpcode::kRoundNearestAfz:
427 return llvm_ir::EmitCallToIntrinsic(
428 llvm::Intrinsic::round, {operand_value}, {operand_value->getType()},
429 ir_builder_);
430 case HloOpcode::kSign: {
431 // TODO(b/32151903): Ensure consistent sign behavior for -0.0.
432 auto type = operand_value->getType();
433 auto zero = llvm::ConstantFP::get(type, 0.0);
434 auto oeq = ir_builder_->CreateFCmpOEQ(operand_value, zero);
435 auto olt = ir_builder_->CreateFCmpOLT(operand_value, zero);
436 return ir_builder_->CreateSelect(
437 oeq, zero,
438 ir_builder_->CreateSelect(olt, llvm::ConstantFP::get(type, -1.0),
439 llvm::ConstantFP::get(type, 1.0)));
440 }
441 case HloOpcode::kIsFinite: {
442 // (x == x) && abs(x) != inf
443 auto type = operand_value->getType();
444 auto equal_self =
445 ir_builder_->CreateFCmpOEQ(operand_value, operand_value);
446 auto abs_value = llvm_ir::EmitCallToIntrinsic(
447 llvm::Intrinsic::fabs, {operand_value}, {type}, ir_builder_);
448 auto infinity = llvm::ConstantFP::getInfinity(type);
449 auto not_infinite = ir_builder_->CreateFCmpONE(abs_value, infinity);
450 auto result_i1 = ir_builder_->CreateAnd(equal_self, not_infinite);
451 return ir_builder_->CreateZExt(
452 result_i1, llvm_ir::PrimitiveTypeToIrType(PRED, module_));
453 }
454 case HloOpcode::kNegate:
455 return ir_builder_->CreateFNeg(operand_value);
456 default:
457 return Unimplemented("unary floating-point op '%s'",
458 HloOpcodeString(op->opcode()).c_str());
459 }

462StatusOr<llvm::Value*> ElementalIrEmitter::EmitComplexUnaryOp(
463 const HloInstruction* op, llvm::Value* operand_value) const {
464 PrimitiveType input_type = op->operand(0)->shape().element_type();
465 PrimitiveType component_type =
466 primitive_util::IsComplexType(input_type)
467 ? primitive_util::ComplexComponentType(input_type)
468 : input_type;
469 switch (op->opcode()) {
470 case HloOpcode::kLog: {
471 // log(a+bi) = .5*log(a^2+b^2) + i*atan2(b, a)
472 auto a = EmitExtractReal(operand_value);
473 auto b = EmitExtractImag(operand_value);
474 llvm::Type* llvm_ty = a->getType();
475 auto sum_sq = ir_builder_->CreateFAdd(ir_builder_->CreateFMul(a, a),
476 ir_builder_->CreateFMul(b, b));
477 TF_ASSIGN_OR_RETURN(auto log_sum_sq, EmitLog(component_type, sum_sq));
478 TF_ASSIGN_OR_RETURN(auto angle, EmitAtan2(component_type, b, a));
479 auto one_half = llvm::ConstantFP::get(llvm_ty, 0.5);
480 return EmitComposeComplex(
481 op, ir_builder_->CreateFMul(one_half, log_sum_sq), angle);
482 }
483 case HloOpcode::kConvert: {
484 PrimitiveType from_type = op->operand(0)->shape().element_type();
485 TF_RET_CHECK(primitive_util::IsComplexType(from_type));
486 PrimitiveType to_type = op->shape().element_type();
487 TF_RET_CHECK(primitive_util::IsComplexType(to_type));
488 if (from_type == to_type) {
489 return operand_value;
490 }
491 PrimitiveType to_component_type =
492 primitive_util::ComplexComponentType(to_type);
493 auto to_ir_component_type =
494 llvm_ir::PrimitiveTypeToIrType(to_component_type, module_);
495 return EmitComposeComplex(
496 op,
497 ir_builder_->CreateFPCast(EmitExtractReal(operand_value),
498 to_ir_component_type),
499 ir_builder_->CreateFPCast(EmitExtractImag(operand_value),
500 to_ir_component_type));
501 }
502 case HloOpcode::kExp: {
503 // e^(a+bi) = e^a*(cos(b)+sin(b)i)
504 TF_ASSIGN_OR_RETURN(
505 auto exp_a, EmitExp(component_type, EmitExtractReal(operand_value)));
506 TF_ASSIGN_OR_RETURN(
507 auto cos_b, EmitCos(component_type, EmitExtractImag(operand_value)));
508 TF_ASSIGN_OR_RETURN(
509 auto sin_b, EmitSin(component_type, EmitExtractImag(operand_value)));
510 return EmitComposeComplex(op, ir_builder_->CreateFMul(exp_a, cos_b),
511 ir_builder_->CreateFMul(exp_a, sin_b));
512 }
513 case HloOpcode::kCos: {
514 // cos(z) = .5(e^(iz) + e^(-iz))
515 // cos(a+bi) = .5(e^(-b+ai) + e^(b-ai))
516 // now, e^(x+yi) = e^x*(cos(y)+sin(y)i), so we have
517 // cos(a+bi) = .5(e^-b*(cos(a)+sin(a)i) + e^b*(cos(-a)+sin(-a)i))
518 // cos(-x) = cos(x) and sin(-x) = -sin(x), so
519 // cos(a+bi) = .5(e^-b*(cos(a)+sin(a)i) + e^b*(cos(a)-sin(a)i))
520 // = .5(cos(a)*(e^-b+e^b) + i*sin(a)*(e^-b-e^b))
521 auto a = EmitExtractReal(operand_value);
522 auto b = EmitExtractImag(operand_value);
523 auto type = a->getType();
524 TF_ASSIGN_OR_RETURN(auto exp_b, EmitExp(component_type, b));
525 auto half_exp_b =
526 ir_builder_->CreateFMul(llvm::ConstantFP::get(type, 0.5), exp_b);
527 auto half_exp_neg_b =
528 ir_builder_->CreateFDiv(llvm::ConstantFP::get(type, 0.5), exp_b);
529 TF_ASSIGN_OR_RETURN(auto cos_a, EmitCos(component_type, a));
530 TF_ASSIGN_OR_RETURN(auto sin_a, EmitSin(component_type, a));
531 return EmitComposeComplex(
532 op,
533 ir_builder_->CreateFMul(
534 cos_a, ir_builder_->CreateFAdd(half_exp_neg_b, half_exp_b)),
535 ir_builder_->CreateFMul(
536 sin_a, ir_builder_->CreateFSub(half_exp_neg_b, half_exp_b)));
537 }
538 case HloOpcode::kSin: {
539 // sin(z) = .5i(e^(-iz) - e^(iz))
540 // sin(a+bi) = .5i(e^(-i(a+bi)) - e^(i(a+bi)))
541 // = .5i(e^(b-ai) - e^(-b+ai))
542 // now, e^(x+yi) = e^x*(cos(y)+sin(y)i), so we have
543 // sin(a+bi) = 0.5i(e^b*(cos(-a)+sin(-a)i) - e^-b*(cos(a)+sin(a)i))
544 // = 0.5(e^b*(cos(-a)i-sin(-a)) - e^-b*(cos(a)i-sin(a)))
545 // cos(-x) = cos(x) and sin(-x) = -sin(x), so
546 // = 0.5(e^b*(cos(a)i+sin(a)) - e^-b*(cos(a)i-sin(a)))
547 // = 0.5(sin(a)*(e^b+e^-b) + i*cos(a)*(e^b-e^-b)
548 auto a = EmitExtractReal(operand_value);
549 auto b = EmitExtractImag(operand_value);
550 auto type = a->getType();
551 TF_ASSIGN_OR_RETURN(auto exp_b, EmitExp(component_type, b));
552 auto half_exp_b =
553 ir_builder_->CreateFMul(llvm::ConstantFP::get(type, 0.5), exp_b);
554 auto half_exp_neg_b =
555 ir_builder_->CreateFDiv(llvm::ConstantFP::get(type, 0.5), exp_b);
556 TF_ASSIGN_OR_RETURN(auto cos_a, EmitCos(component_type, a));
557 TF_ASSIGN_OR_RETURN(auto sin_a, EmitSin(component_type, a));
558 return EmitComposeComplex(
559 op,
560 ir_builder_->CreateFMul(
561 sin_a, ir_builder_->CreateFAdd(half_exp_b, half_exp_neg_b)),
562 ir_builder_->CreateFMul(
563 cos_a, ir_builder_->CreateFSub(half_exp_b, half_exp_neg_b)));
564 }
565 case HloOpcode::kTanh: {
566 /*
567 tanh=(exp(x)-exp(-x)) / (exp(x)+exp(-x))
568 e^(a+bi) = e^a*(cos(b)+sin(b)i)
569 so tanh=(((cos(b)+sin(b)i)e^a - (cos(-b)+sin(-b)i)e^-a)) /
570 (((cos(b)+sin(b)i)e^a + (cos(-b)+sin(-b)i)e^-a))
571 cos(b)=cos(-b), sin(-b)=-sin(b)
572 so tanh=(((cos(b)+sin(b)i)e^a - (cos(b)-sin(b)i)e^-a)) /
573 (((cos(b)+sin(b)i)e^a + (cos(b)-sin(b)i)e^-a))
574 =(cos(b)e^a+i*sin(b)e^a + cos(b)(-e^-a)+i*sin(b)e^-a) /
575 (cos(b)e^a+i*sin(b)e^a + cos(b)e^-a+i*sin(b)(-e^-a))
576 =(cos(b)(e^a-e^-a) + i*sin(b)(e^a+e^-a)) /
577 (cos(b)(e^a+e^-a) + i*sin(b)(e^a-e^-a))
578 This is a complex division, so we can multiply by denom_conj/denom_conj
579 =(cos(b)(e^a-e^-a) + i*sin(b)(e^a+e^-a)) *
580 (cos(b)(e^a+e^-a) - i*sin(b)(e^a-e^-a)) /
581 ((cos(b)(e^a+e^-a))^2 + (sin(b)(e^a-e^-a))^2)
582 =(cos(b)^2(e^(2a)-e^(-2a)) + sin(b)^2(e^(2a)-e^(-2a)) +
583 i*(cos(b)sin(b)(e^a+e^-a)^2 - cos(b)sin(b)(e^a-e^-a)^2)) /
584 ((cos(b)(e^a+e^-a))^2 + (sin(b)(e^a-e^-a))^2)
585 */
586 auto a = EmitExtractReal(operand_value);
587 auto b = EmitExtractImag(operand_value);
588 TF_ASSIGN_OR_RETURN(auto exp_a, EmitExp(component_type, a));
589 TF_ASSIGN_OR_RETURN(auto cos_b, EmitCos(component_type, b));
590 TF_ASSIGN_OR_RETURN(auto sin_b, EmitSin(component_type, b));
591 auto exp_neg_a = ir_builder_->CreateFDiv(
592 llvm::ConstantFP::get(exp_a->getType(), 1), exp_a);
593 auto exp_2a_minus_exp_neg_2a = ir_builder_->CreateFSub(
594 ir_builder_->CreateFMul(exp_a, exp_a),
595 ir_builder_->CreateFMul(exp_neg_a, exp_neg_a));
596 auto cos_b_sq = ir_builder_->CreateFMul(cos_b, cos_b);
597 auto sin_b_sq = ir_builder_->CreateFMul(sin_b, sin_b);
598 auto real_num = ir_builder_->CreateFAdd(
599 ir_builder_->CreateFMul(cos_b_sq, exp_2a_minus_exp_neg_2a),
600 ir_builder_->CreateFMul(sin_b_sq, exp_2a_minus_exp_neg_2a));
601 auto cos_b_sin_b = ir_builder_->CreateFMul(cos_b, sin_b);
602 auto exp_a_plus_exp_neg_a = ir_builder_->CreateFAdd(exp_a, exp_neg_a);
603 auto exp_a_plus_exp_neg_a_sq =
604 ir_builder_->CreateFMul(exp_a_plus_exp_neg_a, exp_a_plus_exp_neg_a);
605 auto exp_a_minus_exp_neg_a = ir_builder_->CreateFSub(exp_a, exp_neg_a);
606 auto exp_a_minus_exp_neg_a_sq =
607 ir_builder_->CreateFMul(exp_a_minus_exp_neg_a, exp_a_minus_exp_neg_a);
608 auto imag_num = ir_builder_->CreateFMul(
609 cos_b_sin_b, ir_builder_->CreateFSub(exp_a_plus_exp_neg_a_sq,
610 exp_a_minus_exp_neg_a_sq));
611 auto denom = ir_builder_->CreateFAdd(
612 ir_builder_->CreateFMul(cos_b_sq, exp_a_plus_exp_neg_a_sq),
613 ir_builder_->CreateFMul(sin_b_sq, exp_a_minus_exp_neg_a_sq));
614 return EmitComposeComplex(op, ir_builder_->CreateFDiv(real_num, denom),
615 ir_builder_->CreateFDiv(imag_num, denom));
616 }
617 case HloOpcode::kAbs: {
618 auto sum_sq = ir_builder_->CreateFAdd(
619 ir_builder_->CreateFMul(EmitExtractReal(operand_value),
620 EmitExtractReal(operand_value)),
621 ir_builder_->CreateFMul(EmitExtractImag(operand_value),
622 EmitExtractImag(operand_value)));
623 return llvm_ir::EmitCallToIntrinsic(llvm::Intrinsic::sqrt, {sum_sq},
624 {sum_sq->getType()}, ir_builder_);
625 }
626 case HloOpcode::kSign: { // Sign(c) = c / |c|
627 auto sum_sq = ir_builder_->CreateFAdd(
628 ir_builder_->CreateFMul(EmitExtractReal(operand_value),
629 EmitExtractReal(operand_value)),
630 ir_builder_->CreateFMul(EmitExtractImag(operand_value),
631 EmitExtractImag(operand_value)));
632 auto cplx_abs = llvm_ir::EmitCallToIntrinsic(
633 llvm::Intrinsic::sqrt, {sum_sq}, {sum_sq->getType()}, ir_builder_);
634 auto type = cplx_abs->getType();
635 auto zero = llvm::ConstantFP::get(type, 0.0);
636 auto oeq = ir_builder_->CreateFCmpOEQ(cplx_abs, zero);
637 return ir_builder_->CreateSelect(
638 oeq, EmitComposeComplex(op, zero, zero),
639 EmitComposeComplex(
640 op,
641 ir_builder_->CreateFDiv(EmitExtractReal(operand_value), cplx_abs),
642 ir_builder_->CreateFDiv(EmitExtractImag(operand_value),
643 cplx_abs)));
644 }
645 case HloOpcode::kNegate:
646 return EmitComposeComplex(
647 op, ir_builder_->CreateFNeg(EmitExtractReal(operand_value)),
648 ir_builder_->CreateFNeg(EmitExtractImag(operand_value)));
649 case HloOpcode::kReal:
650 return EmitExtractReal(operand_value);
651 case HloOpcode::kImag:
652 return EmitExtractImag(operand_value);
653 default:
654 return Unimplemented("unary complex op '%s'",
655 HloOpcodeString(op->opcode()).c_str());
656 }
657}

660 const HloInstruction* op, llvm::Value* lhs_value,
661 llvm::Value* rhs_value) const {
662 PrimitiveType operand_type = op->operand(0)->shape().element_type();

664 operand_type == PRED) {

668 } else if (primitive_util::IsComplexType(operand_type)) {

674
675StatusOr<llvm::Value*> ElementalIrEmitter::EmitFloatBinaryOp(
676 const HloInstruction* op, llvm::Value* lhs_value,
677 llvm::Value* rhs_value) const {
678 switch (op->opcode()) {
679 case HloOpcode::kComplex:
680 return EmitComposeComplex(op, lhs_value, rhs_value);
681 case HloOpcode::kAdd:
682 return ir_builder_->CreateFAdd(lhs_value, rhs_value);
683 case HloOpcode::kSubtract:
684 return ir_builder_->CreateFSub(lhs_value, rhs_value);
685 case HloOpcode::kMultiply:
686 return ir_builder_->CreateFMul(lhs_value, rhs_value);
687 case HloOpcode::kDivide:
688 return ir_builder_->CreateFDiv(lhs_value, rhs_value);
689 case HloOpcode::kRemainder:
690 return ir_builder_->CreateFRem(lhs_value, rhs_value);
691 // LLVM comparisons can be "unordered" (U) or "ordered" (O) -- ordered
692 // comparisons always return false when one of the operands is NaN, whereas
693 // unordered comparisons return true.
694 //
695 // We use ordered comparisons for everything except kNe, where we use an
696 // unordered comparison. This makes x != y equivalent to !(x == y), and
697 // matches C++'s semantics.
698 case HloOpcode::kEq:
699 return llvm_ir::EmitComparison(llvm::CmpInst::FCMP_OEQ, lhs_value,
700 rhs_value, ir_builder_);
701 case HloOpcode::kNe:
702 return llvm_ir::EmitComparison(llvm::CmpInst::FCMP_UNE, lhs_value,
703 rhs_value, ir_builder_);
704 case HloOpcode::kLt:
705 return llvm_ir::EmitComparison(llvm::CmpInst::FCMP_OLT, lhs_value,
706 rhs_value, ir_builder_);
707 case HloOpcode::kGt:
708 return llvm_ir::EmitComparison(llvm::CmpInst::FCMP_OGT, lhs_value,
709 rhs_value, ir_builder_);
710 case HloOpcode::kLe:
711 return llvm_ir::EmitComparison(llvm::CmpInst::FCMP_OLE, lhs_value,
712 rhs_value, ir_builder_);
713 case HloOpcode::kGe:
714 return llvm_ir::EmitComparison(llvm::CmpInst::FCMP_OGE, lhs_value,
715 rhs_value, ir_builder_);
716
717 case HloOpcode::kMaximum:
718 return EmitFloatMax(lhs_value, rhs_value);
719 case HloOpcode::kMinimum:
720 return EmitFloatMin(lhs_value, rhs_value);
721 case HloOpcode::kPower:
722 return EmitPow(op->shape().element_type(), lhs_value, rhs_value);
723 case HloOpcode::kAtan2:
724 return EmitAtan2(op->shape().element_type(), lhs_value, rhs_value);
725 default:
726 return Unimplemented("binary floating point op '%s'",
727 HloOpcodeString(op->opcode()).c_str());
728 }
729}

731StatusOr<llvm::Value*> ElementalIrEmitter::EmitComplexBinaryOp(
732 const HloInstruction* op, llvm::Value* lhs_value,
733 llvm::Value* rhs_value) const {
734 switch (op->opcode()) {
735 case HloOpcode::kAdd:
736 return EmitComposeComplex(
737 op,
738 ir_builder_->CreateFAdd(EmitExtractReal(lhs_value),
739 EmitExtractReal(rhs_value)),
740 ir_builder_->CreateFAdd(EmitExtractImag(lhs_value),
741 EmitExtractImag(rhs_value)));
742 case HloOpcode::kSubtract:
743 return EmitComposeComplex(
744 op,
745 ir_builder_->CreateFSub(EmitExtractReal(lhs_value),
746 EmitExtractReal(rhs_value)),
747 ir_builder_->CreateFSub(EmitExtractImag(lhs_value),
748 EmitExtractImag(rhs_value)));
749 case HloOpcode::kMultiply:
750 return EmitComposeComplex(
751 op,
752 ir_builder_->CreateFSub(
753 ir_builder_->CreateFMul(EmitExtractReal(lhs_value),
754 EmitExtractReal(rhs_value)),
755 ir_builder_->CreateFMul(EmitExtractImag(lhs_value),
756 EmitExtractImag(rhs_value))),
757 ir_builder_->CreateFAdd(
758 ir_builder_->CreateFMul(EmitExtractReal(lhs_value),
759 EmitExtractImag(rhs_value)),
760 ir_builder_->CreateFMul(EmitExtractImag(lhs_value),
761 EmitExtractReal(rhs_value))));
762 case HloOpcode::kDivide: {
763 // (a+bi) / (c+di) = ((a+bi)(c-di)) / ((c+di)(c-di))
764 // = ((ac + bd) + (bc - ad)i) / (c^2 + d^2)
765 auto rhs_sum_sq = ir_builder_->CreateFAdd(
766 ir_builder_->CreateFMul(EmitExtractReal(rhs_value),
767 EmitExtractReal(rhs_value)),
768 ir_builder_->CreateFMul(EmitExtractImag(rhs_value),
769 EmitExtractImag(rhs_value)));
770 auto type = rhs_sum_sq->getType();
771 auto zero = llvm::ConstantFP::get(type, 0.0);
772 auto oeq = ir_builder_->CreateFCmpOEQ(rhs_sum_sq, zero);
773 auto real_inf_or_nan =
774 ir_builder_->CreateFDiv(EmitExtractReal(lhs_value), zero);
775 auto imag_inf_or_nan =
776 ir_builder_->CreateFDiv(EmitExtractImag(lhs_value), zero);
777 return ir_builder_->CreateSelect(
778 oeq, EmitComposeComplex(op, real_inf_or_nan, imag_inf_or_nan),
779 EmitComposeComplex(
780 op,
781 ir_builder_->CreateFDiv(
782 ir_builder_->CreateFAdd(
783 ir_builder_->CreateFMul(EmitExtractReal(lhs_value),
784 EmitExtractReal(rhs_value)),
785 ir_builder_->CreateFMul(EmitExtractImag(lhs_value),
786 EmitExtractImag(rhs_value))),
787 rhs_sum_sq),
788 ir_builder_->CreateFDiv(
789 ir_builder_->CreateFSub(
790 ir_builder_->CreateFMul(EmitExtractImag(lhs_value),
791 EmitExtractReal(rhs_value)),
792 ir_builder_->CreateFMul(EmitExtractReal(lhs_value),
793 EmitExtractImag(rhs_value))),
794 rhs_sum_sq)));
795 }
796 // LLVM comparisons can be "unordered" (U) or "ordered" (O) -- ordered
797 // comparisons always return false when one of the operands is NaN, whereas
798 // unordered comparisons return true.
799 //
800 // We use ordered comparisons for everything except kNe, where we use an
801 // unordered comparison. This makes x != y equivalent to !(x == y), and
802 // matches C++'s semantics.
803 case HloOpcode::kEq:
804 return ir_builder_->CreateAnd(
805 llvm_ir::EmitComparison(llvm::CmpInst::FCMP_OEQ,
806 EmitExtractReal(lhs_value),
807 EmitExtractReal(rhs_value), ir_builder_),
808 llvm_ir::EmitComparison(llvm::CmpInst::FCMP_OEQ,
809 EmitExtractImag(lhs_value),
810 EmitExtractImag(rhs_value), ir_builder_));
811 case HloOpcode::kNe:
812 return ir_builder_->CreateOr(
813 llvm_ir::EmitComparison(llvm::CmpInst::FCMP_UNE,
814 EmitExtractReal(lhs_value),
815 EmitExtractReal(rhs_value), ir_builder_),
816 llvm_ir::EmitComparison(llvm::CmpInst::FCMP_UNE,
817 EmitExtractImag(lhs_value),
818 EmitExtractImag(rhs_value), ir_builder_));
819
820 case HloOpcode::kPower: {
821 // (a+bi)^(c+di) =
822 // (a*a+b*b)^(0.5c) * exp(-d*atan2(b,a)) * (cos(q) + i*sin(q)),
823 // where q = c*atan2(b,a)+0.5d*ln(a*a+b*b)
824 PrimitiveType component_type =
825 primitive_util::ComplexComponentType(op->shape().element_type());
826 auto a = EmitExtractReal(lhs_value);
827 auto b = EmitExtractImag(lhs_value);
828 auto c = EmitExtractReal(rhs_value);
829 auto d = EmitExtractImag(rhs_value);
830 auto aa_p_bb = ir_builder_->CreateFAdd(ir_builder_->CreateFMul(a, a),
831 ir_builder_->CreateFMul(b, b));
832 auto one_half = llvm::ConstantFP::get(a->getType(), 0.5);
833 auto half_c = ir_builder_->CreateFMul(one_half, c);
834
835 TF_ASSIGN_OR_RETURN(auto aa_p_bb_to_half_c,
836 EmitPow(component_type, aa_p_bb, half_c));
837 auto neg_d = ir_builder_->CreateFNeg(d);
838 TF_ASSIGN_OR_RETURN(auto arg_lhs, EmitAtan2(component_type, b, a));
839 auto neg_d_arg_lhs = ir_builder_->CreateFMul(neg_d, arg_lhs);
840 TF_ASSIGN_OR_RETURN(auto e_to_neg_d_arg_lhs,
841 EmitExp(component_type, neg_d_arg_lhs));
842 auto coeff =
843 ir_builder_->CreateFMul(aa_p_bb_to_half_c, e_to_neg_d_arg_lhs);
844 TF_ASSIGN_OR_RETURN(auto ln_aa_p_bb, EmitLog(component_type, aa_p_bb));
845 auto half_d = ir_builder_->CreateFMul(one_half, d);
846 auto q =
847 ir_builder_->CreateFAdd(ir_builder_->CreateFMul(c, arg_lhs),
848 ir_builder_->CreateFMul(half_d, ln_aa_p_bb));
849 TF_ASSIGN_OR_RETURN(auto cos_q, EmitCos(component_type, q));
850 TF_ASSIGN_OR_RETURN(auto sin_q, EmitSin(component_type, q));
851 return EmitComposeComplex(op, ir_builder_->CreateFMul(coeff, cos_q),
852 ir_builder_->CreateFMul(coeff, sin_q));
853 }
854 default:
855 return Unimplemented("binary complex op '%s'",
856 HloOpcodeString(op->opcode()).c_str());
857 }
858}

870StatusOr<llvm::Value*> ElementalIrEmitter::EmitErfInv(PrimitiveType prim_type,
871 llvm::Value* x) const {
872 if (prim_type != F32) {
873 // TODO(b/34339814): Implement inverse erf for F64.
874 return Unimplemented(
875 "Inverse erf is only implemented for element "
876 "type F32.");
877 }
878 auto getFloat = [&](const float f) {
879 return llvm::ConstantFP::get(ir_builder_->getFloatTy(), f);
880 };
881 auto multiply_add = [&](tensorflow::gtl::ArraySlice<float> coefficients,
882 llvm::Value* w) {
883 llvm::Value* p = getFloat(coefficients.front());
884 coefficients.pop_front();
885 for (float coefficient : coefficients) {
886 p = ir_builder_->CreateFAdd(ir_builder_->CreateFMul(p, w),
887 getFloat(coefficient));
888 }
889 return p;
890 };
891
892 // Approximation for inverse error function from
893 // Giles, M., "Approximating the erfinv function".
894 // The approximation has the form:
895 // w = log((1-x)*(1+x))
896 // if ( w < 5 ) {
897 // w = w - 2.5
898 // p = sum_{i=1}^n lq[i]*w^i
899 // } else {
900 // w = sqrt(w) - 3
901 // p = sum_{i=1}^n gq[i]*w^i
902 // }
903 // return p*x
904 llvm::Function* logf_fn = llvm::Intrinsic::getDeclaration(
905 module_, llvm::Intrinsic::log, {ir_builder_->getFloatTy()});
906
907 llvm::Value* w = ir_builder_->CreateFNeg(ir_builder_->CreateCall(
908 logf_fn,
909 {ir_builder_->CreateFMul(ir_builder_->CreateFSub(getFloat(1.0f), x),
910 ir_builder_->CreateFAdd(getFloat(1.0f), x))}));
911
912 llvm::Value* p_addr = llvm_ir::EmitAllocaAtFunctionEntry(
913 ir_builder_->getFloatTy(), "p.addr", ir_builder_);
914
915 llvm_ir::LlvmIfData if_data =
916 llvm_ir::EmitIfThenElse(ir_builder_->CreateFCmpOLT(w, getFloat(5.0f)),
917 "w_less_than_five", ir_builder_);
918 // Handle true BB.
919 SetToFirstInsertPoint(if_data.true_block, ir_builder_);
920 {
921 llvm::Value* lw = ir_builder_->CreateFSub(w, getFloat(2.5f));
922 tensorflow::gtl::ArraySlice<float> lq{
923 2.81022636e-08f, 3.43273939e-07f, -3.5233877e-06f,
924 -4.39150654e-06f, 0.00021858087f, -0.00125372503f,
925 -0.00417768164f, 0.246640727f, 1.50140941f};
926 llvm::Value* p = multiply_add(lq, lw);
927 ir_builder_->CreateStore(p, p_addr);
928 }
929
930 // Handle false BB.
931 SetToFirstInsertPoint(if_data.false_block, ir_builder_);
932 {
933 llvm::Function* sqrtf_fn = llvm::Intrinsic::getDeclaration(
934 module_, llvm::Intrinsic::sqrt, {ir_builder_->getFloatTy()});
935
936 llvm::Value* gw = ir_builder_->CreateFSub(
937 ir_builder_->CreateCall(sqrtf_fn, {w}), getFloat(3.0f));
938 tensorflow::gtl::ArraySlice<float> gq{
939 -0.000200214257f, 0.000100950558f, 0.00134934322f,
940 -0.00367342844f, 0.00573950773f, -0.0076224613f,
941 0.00943887047f, 1.00167406f, 2.83297682f};
942 llvm::Value* p = multiply_add(gq, gw);
943 ir_builder_->CreateStore(p, p_addr);
944 }
945
946 SetToFirstInsertPoint(if_data.after_block, ir_builder_);
947 llvm::Value* p = ir_builder_->CreateLoad(p_addr);
948 return ir_builder_->CreateFMul(p, x);
949}

1006static llvm::Value* SaturateShiftIfNecessary(llvm::IRBuilder<>* ir_builder,
1007 llvm::Value* lhs, llvm::Value* rhs,
1008 llvm::Value* shift_result,
1009 bool saturate_to_sign_bit) {
1010 llvm::IntegerType* integer_type =
1011 llvm::cast<llvm::IntegerType>(lhs->getType());
1012 unsigned integer_bitsize = integer_type->getBitWidth();
1013 llvm::ConstantInt* integer_bitsize_constant =
1014 llvm::ConstantInt::get(integer_type, integer_bitsize);
1015 llvm::ConstantInt* zero = llvm::ConstantInt::get(integer_type, 0);
1016 llvm::ConstantInt* minus_one = llvm::ConstantInt::get(integer_type, -1);
1017 llvm::Value* saturated_value;
1018 if (saturate_to_sign_bit) {
1019 saturated_value = ir_builder->CreateSelect(
1020 ir_builder->CreateICmpSLT(lhs, zero), minus_one, zero);
1021 } else {
1022 saturated_value = zero;
1023 }
1024 llvm::Value* shift_amt_in_range =
1025 ir_builder->CreateICmpULT(rhs, integer_bitsize_constant, "shft.chk");
1026 return ir_builder->CreateSelect(shift_amt_in_range, shift_result,
1027 saturated_value);
1028}

1030StatusOr<llvm::Value*> ElementalIrEmitter::EmitIntegerBinaryOp(
1031 const HloInstruction* op, llvm::Value* lhs_value, llvm::Value* rhs_value,
1032 bool is_signed) const {
1033 switch (op->opcode()) {
1034 // TODO(jingyue): add the "nsw" attribute for signed types.
1035 case HloOpcode::kAdd:
1036 return ir_builder_->CreateAdd(lhs_value, rhs_value);
1037 case HloOpcode::kSubtract:
1038 return ir_builder_->CreateSub(lhs_value, rhs_value);
1039 case HloOpcode::kMultiply:
1040 return ir_builder_->CreateMul(lhs_value, rhs_value);
1041 case HloOpcode::kDivide:
1042 return is_signed ? ir_builder_->CreateSDiv(lhs_value, rhs_value)
1043 : ir_builder_->CreateUDiv(lhs_value, rhs_value);
1044 case HloOpcode::kRemainder:
1045 return is_signed ? ir_builder_->CreateSRem(lhs_value, rhs_value)
1046 : ir_builder_->CreateURem(lhs_value, rhs_value);
1047 case HloOpcode::kEq:
1048 return llvm_ir::EmitComparison(llvm::CmpInst::ICMP_EQ, lhs_value,
1049 rhs_value, ir_builder_);
1050 case HloOpcode::kNe:
1051 return llvm_ir::EmitComparison(llvm::CmpInst::ICMP_NE, lhs_value,
1052 rhs_value, ir_builder_);
1053 case HloOpcode::kLt:
1054 return llvm_ir::EmitComparison(
1055 is_signed ? llvm::CmpInst::ICMP_SLT : llvm::CmpInst::ICMP_ULT,
1056 lhs_value, rhs_value, ir_builder_);
1057 case HloOpcode::kGt:
1058 return llvm_ir::EmitComparison(
1059 is_signed ? llvm::CmpInst::ICMP_SGT : llvm::CmpInst::ICMP_UGT,
1060 lhs_value, rhs_value, ir_builder_);
1061 case HloOpcode::kLe:
1062 return llvm_ir::EmitComparison(
1063 is_signed ? llvm::CmpInst::ICMP_SLE : llvm::CmpInst::ICMP_ULE,
1064 lhs_value, rhs_value, ir_builder_);
1065 case HloOpcode::kGe:
1066 return llvm_ir::EmitComparison(
1067 is_signed ? llvm::CmpInst::ICMP_SGE : llvm::CmpInst::ICMP_UGE,
1068 lhs_value, rhs_value, ir_builder_);
1069 case HloOpcode::kMinimum:
1070 return EmitIntegralMin(lhs_value, rhs_value, is_signed);
1071 case HloOpcode::kMaximum:
1072 return EmitIntegralMax(lhs_value, rhs_value, is_signed);
1073 case HloOpcode::kAnd:
1074 return ir_builder_->CreateAnd(lhs_value, rhs_value);
1075 case HloOpcode::kOr:
1076 return ir_builder_->CreateOr(lhs_value, rhs_value);
1077
1078 // Shifting out bits >= the number of bits in the type being shifted
1079 // produces a poison value in LLVM which is basically "deferred undefined
1080 // behavior" -- doing something observable with such a value precipitates
1081 // UB. We replace the poison value with a constant to avoid this deferred
1082 // UB.
1083 case HloOpcode::kShiftRightArithmetic:
1084 return SaturateShiftIfNecessary(
1085 ir_builder_, lhs_value, rhs_value,
1086 ir_builder_->CreateAShr(lhs_value, rhs_value),
1087 /*saturate_to_sign_bit=*/true);
1088 case HloOpcode::kShiftLeft:
1089 return SaturateShiftIfNecessary(
1090 ir_builder_, lhs_value, rhs_value,
1091 ir_builder_->CreateShl(lhs_value, rhs_value),
1092 /*saturate_to_sign_bit=*/false);
1093 case HloOpcode::kShiftRightLogical:
1094 return SaturateShiftIfNecessary(
1095 ir_builder_, lhs_value, rhs_value,
1096 ir_builder_->CreateLShr(lhs_value, rhs_value),
1097 /*saturate_to_sign_bit=*/false);
1098 default:
1099 return Unimplemented("binary integer op '%s'",
1100 HloOpcodeString(op->opcode()).c_str());
1101 }
1102}

1124llvm_ir::IrArray::Index ElementalIrEmitter::ElementwiseSourceIndex(
1125 const llvm_ir::IrArray::Index& target_index, const HloInstruction& hlo,
1126 int64 operand_no) const {
1127 CHECK(hlo.IsElementwise())
1128 << "HLO " << hlo.ToString() << " is not elementwise.";
1129
1130 const Shape& operand_shape = hlo.operand(operand_no)->shape();
1131 // If the operand is scalar, the source index is always {}.
1132 if (ShapeUtil::IsScalar(operand_shape)) {
1133 return llvm_ir::IrArray::Index();
1134 }
1135
1136 // If no implicit broadcast is needed for this operand, returns the target
1137 // index as the source index.
1138 if (ShapeUtil::CompatibleIgnoringElementType(operand_shape, hlo.shape())) {
1139 return target_index;
1140 }
1141
1142 // If implicit broadcast is needed, the source dimensions that are broadcast
1143 // have index 0.
1144 CHECK_EQ(ShapeUtil::Rank(operand_shape), ShapeUtil::Rank(hlo.shape()));
1145 llvm_ir::IrArray::Index source_index;
1146 for (int64 i = 0; i < ShapeUtil::Rank(hlo.shape()); ++i) {
1147 if (hlo.shape().dimensions(i) == operand_shape.dimensions(i)) {
1148 source_index.push_back(target_index[i]);
1149 } else {
1150 CHECK_EQ(1, operand_shape.dimensions(i));
1151 source_index.push_back(ir_builder_->getInt64(0));
1152 }
1153 }
1154 return source_index;
1155}
1156
1157llvm_ir::ElementGenerator ElementalIrEmitter::MakeRngElementGenerator(
1158 const HloInstruction* hlo,
1159 const ElementalIrEmitter::HloToElementGeneratorMap& operand_to_generator)
1160 const {
1161 PrimitiveType param_prim_type = hlo->operand(0)->shape().element_type();
1162 llvm::Type* param_ir_type =
1163 llvm_ir::PrimitiveTypeToIrType(param_prim_type, module_);
1164
1165 // Same values as PCG library
1166 // https://github.com/imneme/pcg-c/blob/master/include/pcg_variants.h
1167 llvm::Value* multiplier = ir_builder_->getInt(
1168 llvm::APInt(128, {0x4385DF649FCCF645, 0x2360ED051FC65DA4}));
1169 llvm::Value* increment = ir_builder_->getInt(
1170 llvm::APInt(128, {0x14057B7EF767814F, 0x5851F42D4C957F2D}));
1171
1172 auto random_value = [hlo]() {
1173 const HloModule* module =
1174 hlo->IsFused() ? hlo->parent()->FusionInstruction()->parent()->parent()
1175 : hlo->parent()->parent();
1176 return module->RandomNew64();
1177 };
1178
1179 // Seed each RNG emitter with a new 64-bit seed from the HloModule. If the
1180 // compilation order is deterministic (i.e., RandomNew64 invocation order is
1181 // deterministic), then the order of RNG is deterministic for a given seed and
1182 // hence tests will be deterministic.
1183 // If the user provides a global seed instruction then we only use 64-bits of
1184 // the host's random number generator to seed the 128 bit value with the other
1185 // 64-bits is due to a user specified global seed instruction.
1186 // Create a GlobalVariable to maintain state between invocations. There is a
1187 // bug in NVPTX with GlobalVariable and 128 bit values, so using 2 64-bit
1188 // values.
1189 llvm::GlobalVariable* state_ptr0 = new llvm::GlobalVariable(
1190 /*M=*/*module_,
1191 /*Ty=*/ir_builder_->getInt64Ty(),
1192 /*isConstant=*/false,
1193 /*Linkage=*/llvm::GlobalValue::PrivateLinkage,
1194 /*Initializer=*/ir_builder_->getInt64(random_value()),
1195 /*Name=*/"state_ptr0");
1196 uint64 graph_seed = hlo_module_config_.seed() != 0 ? hlo_module_config_.seed()
1197 : random_value();
1198 llvm::GlobalVariable* state_ptr1 = new llvm::GlobalVariable(
1199 /*M=*/*module_,
1200 /*Ty=*/ir_builder_->getInt64Ty(),
1201 /*isConstant=*/false,
1202 /*Linkage=*/llvm::GlobalValue::PrivateLinkage,
1203 /*Initializer=*/ir_builder_->getInt64(graph_seed),
1204 /*Name=*/"state_ptr1");
1205
1206 // We want each thread to use its own stream, so we modify the increment per
1207 // thread. We want the increment to remain odd, so we shift the thread id left
1208 // 1 and add it to the increment.
1209 increment = ir_builder_->CreateAdd(increment,
1210 ir_builder_->CreateShl(EmitThreadId(), 1));
1211
1212 // PCG-XSL-RR algorithm
1213 // http://www.pcg-random.org/pdf/toms-oneill-pcg-family-v1.02.pdf
1214 // state = multiplier * state + increment
1215 // return uint64_t(state ^ (state >> 64))) >>> (state >> 122)
1216 // where ">>>" is bitwise rotation
1217 auto get_next_i64 = [=]() {
1218 llvm::Value* state0 = ir_builder_->CreateZExtOrTrunc(
1219 ir_builder_->CreateLoad(state_ptr0, "state0"),
1220 ir_builder_->getInt128Ty());
1221 llvm::Value* state1 = ir_builder_->CreateShl(
1222 ir_builder_->CreateZExtOrTrunc(
1223 ir_builder_->CreateLoad(state_ptr1, "state1"),
1224 ir_builder_->getInt128Ty()),
1225 64);
1226 llvm::Value* state = ir_builder_->CreateOr(state0, state1);
1227 llvm::Value* updated = ir_builder_->CreateAdd(
1228 ir_builder_->CreateMul(state, multiplier), increment);
1229 ir_builder_->CreateStore(
1230 ir_builder_->CreateTrunc(updated, ir_builder_->getInt64Ty()),
1231 state_ptr0);
1232 ir_builder_->CreateStore(
1233 ir_builder_->CreateTrunc(ir_builder_->CreateLShr(updated, 64),
1234 ir_builder_->getInt64Ty()),
1235 state_ptr1);
1236
1237 return llvm_ir::CreateRor(
1238 ir_builder_->CreateTrunc(
1239 ir_builder_->CreateXor(state, ir_builder_->CreateLShr(state, 64)),
1240 ir_builder_->getInt64Ty()),
1241 ir_builder_->CreateTrunc(ir_builder_->CreateLShr(state, 122),
1242 ir_builder_->getInt64Ty()),
1243 ir_builder_);
1244 };
1245
1246 auto get_next_uniform_float = [=]() {
1247 return ir_builder_->CreateFDiv(
1248 ir_builder_->CreateUIToFP(get_next_i64(), param_ir_type),
1249 llvm::ConstantFP::get(param_ir_type, 0x1p64));
1250 };
1251
1252 return [=](const llvm_ir::IrArray::Index& index) -> StatusOr<llvm::Value*> {
1253 switch (hlo->random_distribution()) {
1254 case RNG_UNIFORM: {
1255 TF_ASSIGN_OR_RETURN(llvm::Value * p,
1256 operand_to_generator.at(hlo->operand(0))(index));
1257 TF_ASSIGN_OR_RETURN(llvm::Value * q,
1258 operand_to_generator.at(hlo->operand(1))(index));
1259 if (primitive_util::IsFloatingPointType(param_prim_type)) {
1260 return ir_builder_->CreateFAdd(
1261 ir_builder_->CreateFMul(ir_builder_->CreateFSub(q, p),
1262 get_next_uniform_float()),
1263 p);
1264 } else {
1265 auto r = ir_builder_->CreateSub(q, p);
1266 auto leading_zeros = llvm_ir::EmitCallToIntrinsic(
1267 llvm::Intrinsic::ctlz, {r, ir_builder_->getInt1(true)},
1268 {param_ir_type}, ir_builder_);
1269 auto in_block = ir_builder_->GetInsertBlock();
1270
1271 // A terminator should be present iff we're emitting code
1272 // into the middle (as opposed to the end) of a basic block.
1273 CHECK_EQ(ir_builder_->GetInsertPoint() == in_block->end(),
1274 in_block->getTerminator() == nullptr);
1275
1276 llvm::BasicBlock* body_block;
1277 llvm::BasicBlock* out_block;
1278
1279 if (ir_builder_->GetInsertPoint() == in_block->end()) {
1280 body_block = llvm_ir::CreateBasicBlock(
1281 nullptr, IrName(hlo, "rng_body"), ir_builder_);
1282 out_block = llvm_ir::CreateBasicBlock(
1283 nullptr, IrName(hlo, "rng_out"), ir_builder_);
1284 llvm::BranchInst::Create(body_block, in_block);
1285 } else {
1286 body_block = in_block->splitBasicBlock(
1287 ir_builder_->GetInsertPoint(), "rng_body");
1288 out_block = body_block->splitBasicBlock(
1289 ir_builder_->GetInsertPoint(), "rng_out");
1290 body_block->getTerminator()->eraseFromParent();
1291 }
1292
1293 SetToFirstInsertPoint(body_block, ir_builder_);
1294 auto random = ir_builder_->CreateAnd(
1295 ir_builder_->CreateZExtOrTrunc(get_next_i64(), param_ir_type),
1296 ir_builder_->CreateLShr(llvm::ConstantInt::get(param_ir_type, ~0),
1297 leading_zeros));
1298 llvm::BranchInst::Create(out_block, body_block,
1299 ir_builder_->CreateICmpULT(random, r),
1300 body_block);
1301 SetToFirstInsertPoint(out_block, ir_builder_);
1302 return ir_builder_->CreateAdd(
1303 p, ir_builder_->CreateSelect(
1304 ir_builder_->CreateICmpEQ(p, q),
1305 llvm::ConstantInt::get(param_ir_type, 0), random));
1306 }
1307 }
1308 case RNG_NORMAL: {
1309 TF_ASSIGN_OR_RETURN(llvm::Value * m,
1310 operand_to_generator.at(hlo->operand(0))(index));
1311 TF_ASSIGN_OR_RETURN(llvm::Value * s,
1312 operand_to_generator.at(hlo->operand(1))(index));
1313 TF_ASSIGN_OR_RETURN(
1314 llvm::Value * r,
1315 EmitErfcInv(param_prim_type,
1316 ir_builder_->CreateFMul(
1317 llvm::ConstantFP::get(param_ir_type, 2.0),
1318 get_next_uniform_float())));
1319 return ir_builder_->CreateFAdd(ir_builder_->CreateFMul(r, s), m);
1320 }
1321 default:
1322 return InvalidArgument(
1323 "unhandled distribution %s",
1324 RandomDistribution_Name(hlo->random_distribution()).c_str());
1325 }
1326 };
1327}

1329llvm_ir::ElementGenerator ElementalIrEmitter::MakeElementGenerator(
1330 const HloInstruction* hlo,
1331 const ElementalIrEmitter::HloToElementGeneratorMap& operand_to_generator)
1332 const {
1333 switch (hlo->opcode()) {
1334 case HloOpcode::kAbs:
1335 case HloOpcode::kRoundNearestAfz:
1336 case HloOpcode::kCeil:
1337 case HloOpcode::kConvert:
1338 case HloOpcode::kBitcastConvert:
1339 case HloOpcode::kCopy:
1340 case HloOpcode::kCos:
1341 case HloOpcode::kExp:
1342 case HloOpcode::kFloor:
1343 case HloOpcode::kImag:
1344 case HloOpcode::kIsFinite:
1345 case HloOpcode::kLog:
1346 case HloOpcode::kNegate:
1347 case HloOpcode::kNot:
1348 case HloOpcode::kReal:
1349 case HloOpcode::kSign:
1350 case HloOpcode::kSin:
1351 case HloOpcode::kTanh:
1352 return [this, hlo, &operand_to_generator](
1353 const IrArray::Index& index) -> StatusOr<llvm::Value*> {
1354 TF_ASSIGN_OR_RETURN(llvm::Value * operand_value,
1355 operand_to_generator.at(hlo->operand(0))(
1356 ElementwiseSourceIndex(index, *hlo, 0)));
1357 return EmitUnaryOp(hlo, operand_value);
1358 };
1359 case HloOpcode::kAdd:
1360 case HloOpcode::kAnd:
1361 case HloOpcode::kAtan2:
1362 case HloOpcode::kComplex:
1363 case HloOpcode::kDivide:
1364 case HloOpcode::kEq:
1365 case HloOpcode::kGe:
1366 case HloOpcode::kGt:
1367 case HloOpcode::kLe:
1368 case HloOpcode::kLt:
1369 case HloOpcode::kMaximum:
1370 case HloOpcode::kMinimum:
1371 case HloOpcode::kMultiply:
1372 case HloOpcode::kNe:
1373 case HloOpcode::kOr:
1374 case HloOpcode::kPower:
1375 case HloOpcode::kRemainder:
1376 case HloOpcode::kShiftLeft:
1377 case HloOpcode::kShiftRightArithmetic:
1378 case HloOpcode::kShiftRightLogical:
1379 case HloOpcode::kSubtract:
1380 return [this, hlo, &operand_to_generator](
1381 const IrArray::Index& index) -> StatusOr<llvm::Value*> {
1382 const HloInstruction* lhs = hlo->operand(0);
1383 const HloInstruction* rhs = hlo->operand(1);
1384 TF_ASSIGN_OR_RETURN(llvm::Value * lhs_value,
1385 operand_to_generator.at(lhs)(
1386 ElementwiseSourceIndex(index, *hlo, 0)));
1387 TF_ASSIGN_OR_RETURN(llvm::Value * rhs_value,
1388 operand_to_generator.at(rhs)(
1389 ElementwiseSourceIndex(index, *hlo, 1)));
1390 return EmitBinaryOp(hlo, lhs_value, rhs_value);
1391 };
1392 case HloOpcode::kSelect:
1393 return [this, hlo, &operand_to_generator](
1394 const IrArray::Index& index) -> StatusOr<llvm::Value*> {
1395 TF_ASSIGN_OR_RETURN(llvm::Value * pred_value,
1396 operand_to_generator.at(hlo->operand(0))(
1397 ElementwiseSourceIndex(index, *hlo, 0)));
1398 TF_ASSIGN_OR_RETURN(llvm::Value * on_true_value,
1399 operand_to_generator.at(hlo->operand(1))(
1400 ElementwiseSourceIndex(index, *hlo, 1)));
1401 TF_ASSIGN_OR_RETURN(llvm::Value * on_false_value,
1402 operand_to_generator.at(hlo->operand(2))(
1403 ElementwiseSourceIndex(index, *hlo, 2)));
1404 return ir_builder_->CreateSelect(
1405 ir_builder_->CreateTrunc(pred_value, ir_builder_->getInt1Ty()),
1406 on_true_value, on_false_value);
1407 };
1408 case HloOpcode::kClamp:
1409 return [this, hlo, &operand_to_generator](
1410 const IrArray::Index& index) -> StatusOr<llvm::Value*> {
1411 TF_ASSIGN_OR_RETURN(llvm::Value * min_value,
1412 operand_to_generator.at(hlo->operand(0))(
1413 ElementwiseSourceIndex(index, *hlo, 0)));
1414 TF_ASSIGN_OR_RETURN(llvm::Value * arg_value,
1415 operand_to_generator.at(hlo->operand(1))(
1416 ElementwiseSourceIndex(index, *hlo, 1)));
1417 TF_ASSIGN_OR_RETURN(llvm::Value * max_value,
1418 operand_to_generator.at(hlo->operand(2))(
1419 ElementwiseSourceIndex(index, *hlo, 2)));
1420 PrimitiveType prim_type = hlo->shape().element_type();
1421 if (primitive_util::IsFloatingPointType(prim_type)) {
1422 return EmitFloatMin(max_value, EmitFloatMax(min_value, arg_value));
1423 } else if (primitive_util::IsIntegralType(prim_type)) {
1424 bool is_signed = primitive_util::IsSignedIntegralType(prim_type);
1425 return EmitIntegralMin(
1426 max_value, EmitIntegralMax(min_value, arg_value, is_signed),
1427 is_signed);
1428 } else {
1429 return Unimplemented("Clamp unimplemented for %s",
1430 PrimitiveType_Name(prim_type).c_str());
1431 }
1432 };
1433 case HloOpcode::kReducePrecision:
1434 return [this, hlo, &operand_to_generator](
1435 const IrArray::Index& index) -> StatusOr<llvm::Value*> {
1436 TF_ASSIGN_OR_RETURN(llvm::Value * operand_value,
1437 operand_to_generator.at(hlo->operand(0))(
1438 ElementwiseSourceIndex(index, *hlo, 0)));
1439 return EmitReducePrecision(hlo, operand_value);
1440 };
1441 case HloOpcode::kConcatenate:
1442 return [this, hlo, &operand_to_generator](
1443 const IrArray::Index target_index) -> StatusOr<llvm::Value*> {
1444 const int64 concat_dim = hlo->dimensions(0);
1445 auto source_index = target_index;
1446
1447 llvm::BasicBlock* init_block = ir_builder_->GetInsertBlock();
1448
1449 // A terminator should be present iff we're emitting code
1450 // into the middle (as opposed to the end) of a basic block.
1451 CHECK_EQ(ir_builder_->GetInsertPoint() == init_block->end(),
1452 init_block->getTerminator() == nullptr);
1453
1454 llvm::BasicBlock* exit_block;
1455 if (ir_builder_->GetInsertPoint() == init_block->end()) {
1456 exit_block = llvm_ir::CreateBasicBlock(
1457 /*insert_before=*/nullptr, IrName(hlo, "merge"), ir_builder_);
1458 } else {
1459 exit_block = init_block->splitBasicBlock(
1460 ir_builder_->GetInsertPoint(), AsStringRef(IrName(hlo, "merge")));
1461 init_block->getTerminator()->eraseFromParent();
1462 }
1463
1464 llvm_ir::SetToFirstInsertPoint(exit_block, ir_builder_);
1465 llvm::PHINode* output =
1466 ir_builder_->CreatePHI(llvm_ir::PrimitiveTypeToIrType(
1467 hlo->shape().element_type(), module_),
1468 hlo->operands().size());
1469 auto prior_insert_point = ir_builder_->GetInsertPoint();
1470
1471 ir_builder_->SetInsertPoint(init_block);
1472
1473 for (int64 operand_idx = 0; operand_idx < hlo->operand_count();
1474 ++operand_idx) {
1475 const HloInstruction* operand = hlo->operand(operand_idx);
1476 auto true_block = llvm_ir::CreateBasicBlock(
1477 exit_block, StrCat("concat_index_from_operand", operand_idx),
1478 ir_builder_);
1479 auto false_block = llvm_ir::CreateBasicBlock(
1480 exit_block, StrCat("concat_index_not_from_operand", operand_idx),
1481 ir_builder_);
1482 auto concat_dim_size =
1483 llvm::ConstantInt::get(source_index[concat_dim]->getType(),
1484 operand->shape().dimensions(concat_dim));
1485 ir_builder_->CreateCondBr(
1486 ir_builder_->CreateICmpULT(source_index[concat_dim],
1487 concat_dim_size),
1488 true_block, false_block);
1489
1490 // Create the terminator of the true block before calling operand
1491 // generators, because they require non-degenerate basic blocks.
1492 ir_builder_->SetInsertPoint(
1493 llvm::BranchInst::Create(exit_block, /*InsertAtEnd=*/true_block));
1494 TF_ASSIGN_OR_RETURN(llvm::Value * value,
1495 operand_to_generator.at(operand)(source_index));
1496 output->addIncoming(value, ir_builder_->GetInsertBlock());
1497
1498 // Subtract the size of the concat dimension of the current operand
1499 // from the source index.
1500 ir_builder_->SetInsertPoint(false_block);
1501 source_index[concat_dim] =
1502 ir_builder_->CreateSub(source_index[concat_dim], concat_dim_size);
1503 }
1504
1505 ir_builder_->CreateUnreachable();
1506 ir_builder_->SetInsertPoint(exit_block, prior_insert_point);
1507 return output;
1508 };
1509 case HloOpcode::kReverse:
1510 return [this, hlo, &operand_to_generator](
1511 const IrArray::Index& target_index) -> StatusOr<llvm::Value*> {
1512 const HloInstruction* operand = hlo->operand(0);
1513 auto source_index = target_index;
1514 for (int64 dim : hlo->dimensions()) {
1515 source_index[dim] = ir_builder_->CreateSub(
1516 llvm::ConstantInt::get(target_index[dim]->getType(),
1517 hlo->shape().dimensions(dim) - 1),
1518 target_index[dim]);
1519 }
1520 return operand_to_generator.at(operand)(source_index);
1521 };
1522 case HloOpcode::kBroadcast:
1523 return [this, hlo, &operand_to_generator](
1524 const IrArray::Index& target_index) -> StatusOr<llvm::Value*> {
1525 const HloInstruction* operand = hlo->operand(0);
1526 // The `dimensions` member of the broadcast instruction maps from
1527 // input dimensions to output dimensions.
1528 return operand_to_generator.at(
1529 operand)(target_index.SourceIndexOfBroadcast(
1530 hlo->shape(), operand->shape(), hlo->dimensions(), ir_builder_));
1531 };
1532 case HloOpcode::kSlice:
1533 return [this, hlo, &operand_to_generator](
1534 const IrArray::Index& index) -> StatusOr<llvm::Value*> {
1535 IrArray::Index sliced_index = index.SourceIndexOfSlice(
1536 /*shape=*/hlo->shape(), /*starts=*/hlo->slice_starts(),
1537 /*strides=*/hlo->slice_strides(), /*builder=*/ir_builder_);
1538 return operand_to_generator.at(hlo->operand(0))(sliced_index);
1539 };
1540 case HloOpcode::kDynamicSlice:
1541 return [this, hlo, &operand_to_generator](
1542 const IrArray::Index& index) -> StatusOr<llvm::Value*> {
1543 // Emit IR to read dynamic start indices from hlo->operand(1).
1544 const HloInstruction* input_hlo = hlo->operand(0);
1545 const int64 rank = ShapeUtil::Rank(input_hlo->shape());
1546 llvm_ir::IrArray::Index slice_start_index(rank);
1547 for (int64 i = 0; i < rank; ++i) {
1548 llvm_ir::IrArray::Index dim_index(1, ir_builder_->getInt64(i));
1549 TF_ASSIGN_OR_RETURN(
1550 llvm::Value * start_index_value,
1551 operand_to_generator.at(hlo->operand(1))(dim_index));
1552 start_index_value->setName(
1553 AsStringRef(IrName(hlo, StrCat("start_idx", i))));
1554 slice_start_index[i] = start_index_value;
1555 }
1556
1557 llvm_ir::IrArray::Index input_index(rank);
1558 for (int64 i = 0; i < rank; ++i) {
1559 // Emit IR which computes:
1560 // input_index = (start_index + offset_index) % dim_size
1561 // Security note: this is the code that keeps the indices in-bounds.
1562 llvm::Value* dim_size = llvm::ConstantInt::get(
1563 index[i]->getType(), input_hlo->shape().dimensions(i));
1564 llvm::Value* start_index = ir_builder_->CreateZExtOrBitCast(
1565 slice_start_index[i], index[i]->getType());
1566 input_index[i] = ir_builder_->CreateURem(
1567 ir_builder_->CreateAdd(start_index, index[i]), dim_size);
1568 }
1569 return operand_to_generator.at(input_hlo)(input_index);
1570 };
1571 case HloOpcode::kDynamicUpdateSlice:
1572 return [this, hlo, &operand_to_generator](
1573 const IrArray::Index& index) -> StatusOr<llvm::Value*> {
1574 const HloInstruction* input_hlo = hlo->operand(0);
1575 const HloInstruction* update_hlo = hlo->operand(1);
1576 const HloInstruction* start_hlo = hlo->operand(2);
1577 // Calculate slice start/end indices.
1578 const int64 rank = ShapeUtil::Rank(input_hlo->shape());
1579 llvm_ir::IrArray::Index slice_start_index(rank);
1580 llvm_ir::IrArray::Index slice_limit_index(rank);
1581 // Slice starts at update[index - slice_start_index_adjusted],
1582 // where adjusted value = slice_start_index when in bounds, and
1583 // adjusted value = slice_start_index - input_dim, when wrapping.
1584 llvm_ir::IrArray::Index slice_start_index_adjusted(rank);
1585
1586 // Slice intersection gathers (ANDs) conditions on all ranks for which
1587 // 'input' is set to 'update'
1588 llvm::Value* slice_intersection = ir_builder_->getTrue();
1589
1590 for (int64 i = 0; i < rank; ++i) {
1591 // Emit IR to read dynamic start indices from 'start_hlo'.
1592 llvm_ir::IrArray::Index dim_index(1, ir_builder_->getInt64(i));
1593 TF_ASSIGN_OR_RETURN(llvm::Value * start_index_value,
1594 operand_to_generator.at(start_hlo)(dim_index));
1595 start_index_value->setName(
1596 AsStringRef(IrName(hlo, StrCat("start_idx", i))));
1597 slice_start_index[i] = ir_builder_->CreateZExtOrBitCast(
1598 start_index_value, index[i]->getType());
1599
1600 llvm::Value* input_dim_size = llvm::ConstantInt::get(
1601 index[i]->getType(), input_hlo->shape().dimensions(i));
1602 llvm::Value* update_dim_size = llvm::ConstantInt::get(
1603 index[i]->getType(), update_hlo->shape().dimensions(i));
1604
1605 // Generate code to handle wrapping semantics:
1606 // slice_start_index[i] = slice_start_index[i] % input_dim_size;
1607 // slice_limit_index[i] = slice_start_index[i] + update_dim_size.
1608 // slice_start_index[i] is updated in place and it will now be in
1609 // range. slice_limit_index[i] may be out of range, and it's being
1610 // URem-ed below if so.
1611 slice_start_index[i] =
1612 ir_builder_->CreateURem(slice_start_index[i], input_dim_size);
1613 slice_limit_index[i] =
1614 ir_builder_->CreateAdd(slice_start_index[i], update_dim_size);
1615
1616 // Test if slice_limit_index[i] is in bounds
1617 llvm::Value* in_bounds =
1618 ir_builder_->CreateICmpULE(slice_limit_index[i], input_dim_size);
1619 llvm_ir::LlvmIfData if_in_bounds =
1620 llvm_ir::EmitIfThenElse(in_bounds, "in_bounds", ir_builder_);
1621
1622 // Handle true BB (slice_limit_index[i] <= input_dim_size).
1623 SetToFirstInsertPoint(if_in_bounds.true_block, ir_builder_);
1624 // Check that index[i] >= slice_start_index[i] &&
1625 // index[i] < slice_limit_index[i]
1626 llvm::Value* slice_intersection_in_bounds = ir_builder_->CreateAnd(
1627 slice_intersection,
1628 ir_builder_->CreateICmpSGE(index[i], slice_start_index[i]),
1629 "slice_intersection_in");
1630 slice_intersection_in_bounds = ir_builder_->CreateAnd(
1631 slice_intersection_in_bounds,
1632 ir_builder_->CreateICmpSLT(index[i], slice_limit_index[i]),
1633 "slice_intersection_in");
1634
1635 // Handle false BB (slice_limit_index[i] > input_dim_size).
1636 SetToFirstInsertPoint(if_in_bounds.false_block, ir_builder_);
1637 // Check that index[i] >= slice_start_index[i] ||
1638 // index[i] < slice_limit_index[i]%input_dim_size.
1639 llvm::Value* index_wraps = ir_builder_->CreateICmpSLT(
1640 index[i],
1641 ir_builder_->CreateURem(slice_limit_index[i], input_dim_size));
1642 llvm::Value* slice_intersection_or = ir_builder_->CreateOr(
1643 ir_builder_->CreateICmpSGE(index[i], slice_start_index[i]),
1644 index_wraps, "slice_intersection_out");
1645 llvm::Value* slice_intersection_out_of_bounds =
1646 ir_builder_->CreateAnd(slice_intersection, slice_intersection_or,
1647 "slice_intersection_out");
1648 // Create value for slice_start_index_adjusted[i] when out of bounds.
1649 // If within out-of-bounds if.
1650 llvm_ir::LlvmIfData if_start_needs_adjustment =
1651 llvm_ir::EmitIfThenElse(index_wraps, "adjust_start", ir_builder_);
1652 SetToFirstInsertPoint(if_start_needs_adjustment.true_block,
1653 ir_builder_);
1654 llvm::Value* slice_start_index_adjusted_oob =
1655 ir_builder_->CreateSub(slice_start_index[i], input_dim_size);
1656 SetToFirstInsertPoint(if_start_needs_adjustment.after_block,
1657 ir_builder_);
1658 llvm::PHINode* slice_start_index_adjusted_phi =
1659 ir_builder_->CreatePHI(slice_start_index_adjusted_oob->getType(),
1660 2);
1661 slice_start_index_adjusted_phi->addIncoming(
1662 slice_start_index_adjusted_oob,
1663 if_start_needs_adjustment.true_block);
1664 slice_start_index_adjusted_phi->addIncoming(
1665 slice_start_index[i], if_start_needs_adjustment.false_block);
1666 // End of if within if.
1667
1668 // After checking in/out of bounds.
1669 SetToFirstInsertPoint(if_in_bounds.after_block, ir_builder_);
1670 llvm::PHINode* phi_slice_intersection =
1671 ir_builder_->CreatePHI(slice_intersection->getType(), 2);
1672 phi_slice_intersection->addIncoming(slice_intersection_in_bounds,
1673 if_in_bounds.true_block);
1674 phi_slice_intersection->addIncoming(
1675 slice_intersection_out_of_bounds,
1676 if_start_needs_adjustment.after_block);
1677 slice_intersection = phi_slice_intersection;
1678
1679 llvm::PHINode* phi_index =
1680 ir_builder_->CreatePHI(slice_start_index[i]->getType(), 2);
1681 phi_index->addIncoming(slice_start_index[i], if_in_bounds.true_block);
1682 phi_index->addIncoming(slice_start_index_adjusted_phi,
1683 if_start_needs_adjustment.after_block);
1684 slice_start_index_adjusted[i] = phi_index;
1685 }
1686
1687 // Emit:
1688 // if (slice_intersection) -> return data from 'update'.
1689 // else -> return data from 'input'.
1690 llvm::Value* ret_value_addr = llvm_ir::EmitAllocaAtFunctionEntry(
1691 llvm_ir::PrimitiveTypeToIrType(hlo->shape().element_type(),
1692 module_),
1693 "ret_value_addr", ir_builder_);
1694 llvm_ir::LlvmIfData if_data = llvm_ir::EmitIfThenElse(
1695 slice_intersection, "slice_intersection", ir_builder_);
1696
1697 // Handle true BB (return data from 'update')
1698 SetToFirstInsertPoint(if_data.true_block, ir_builder_);
1699 // Compute update index for intersection case.
1700 llvm_ir::IrArray::Index update_index(rank);
1701 for (int64 i = 0; i < rank; ++i) {
1702 llvm::Value* update_dim_size = llvm::ConstantInt::get(
1703 index[i]->getType(), update_hlo->shape().dimensions(i));
1704 // NOTE: Subtraction will be positive due to bounds checking above.
1705 update_index[i] = ir_builder_->CreateURem(
1706 ir_builder_->CreateSub(index[i], slice_start_index_adjusted[i]),
1707 update_dim_size);
1708 }
1709 TF_ASSIGN_OR_RETURN(llvm::Value * true_value,
1710 operand_to_generator.at(update_hlo)(update_index));
1711 ir_builder_->CreateStore(true_value, ret_value_addr);
1712
1713 // Handle false BB (return data from 'input')
1714 SetToFirstInsertPoint(if_data.false_block, ir_builder_);
1715 TF_ASSIGN_OR_RETURN(llvm::Value * false_value,
1716 operand_to_generator.at(input_hlo)(index));
1717 ir_builder_->CreateStore(false_value, ret_value_addr);
1718
1719 SetToFirstInsertPoint(if_data.after_block, ir_builder_);
1720 return ir_builder_->CreateLoad(ret_value_addr);
1721 };
1722 case HloOpcode::kBitcast:
1723 CHECK_EQ(ShapeUtil::ElementsIn(hlo->shape()),
1724 ShapeUtil::ElementsIn(hlo->operand(0)->shape()));
1725 return [this, hlo, &operand_to_generator](const IrArray::Index& index) {
1726 const HloInstruction* operand = hlo->operand(0);
1727 return operand_to_generator.at(operand)(index.SourceIndexOfBitcast(
1728 hlo->shape(), operand->shape(), ir_builder_));
1729 };
1730 case HloOpcode::kReshape:
1731 CHECK_EQ(ShapeUtil::ElementsIn(hlo->shape()),
1732 ShapeUtil::ElementsIn(hlo->operand(0)->shape()));
1733 return [this, hlo, &operand_to_generator](const IrArray::Index& index) {
1734 const HloInstruction* operand = hlo->operand(0);
1735 return operand_to_generator.at(operand)(index.SourceIndexOfReshape(
1736 hlo->shape(), operand->shape(), ir_builder_));
1737 };
1738 case HloOpcode::kTranspose:
1739 return [this, hlo,
1740 &operand_to_generator](const IrArray::Index& target_index) {
1741 return operand_to_generator.at(hlo->operand(0))(
1742 target_index.SourceIndexOfTranspose(
1743 hlo->shape(), hlo->operand(0)->shape(), hlo->dimensions(),
1744 ir_builder_));
1745 };
1746 case HloOpcode::kRng:
1747 return MakeRngElementGenerator(hlo, operand_to_generator);
1748 case HloOpcode::kPad:
1749 return [=, &operand_to_generator](
1750 const IrArray::Index& padded_index) -> StatusOr<llvm::Value*> {
1751 auto index = padded_index;
1752 llvm::Value* in_bounds = ir_builder_->getTrue();
1753 for (size_t i = 0; i < index.size(); ++i) {
1754 auto index_typed_const = [=](int64 n) {
1755 return llvm::ConstantInt::get(index[i]->getType(), n);
1756 };
1757 const auto& pad_dim = hlo->padding_config().dimensions(i);
1758 index[i] = ir_builder_->CreateSub(
1759 index[i], index_typed_const(pad_dim.edge_padding_low()));
1760 in_bounds = ir_builder_->CreateAnd(
1761 in_bounds,
1762 ir_builder_->CreateICmpSGE(index[i], index_typed_const(0)),
1763 "in_bounds");
1764 in_bounds = ir_builder_->CreateAnd(
1765 in_bounds,
1766 ir_builder_->CreateICmpEQ(
1767 index_typed_const(0),
1768 ir_builder_->CreateURem(
1769 index[i],
1770 index_typed_const(pad_dim.interior_padding() + 1))),
1771 "in_bounds");
1772 index[i] = ir_builder_->CreateSDiv(
1773 index[i], index_typed_const(pad_dim.interior_padding() + 1));
1774 in_bounds = ir_builder_->CreateAnd(
1775 in_bounds,
1776 ir_builder_->CreateICmpSLT(
1777 index[i],
1778 index_typed_const(hlo->operand(0)->shape().dimensions(i))),
1779 "in_bounds");
1780 }
1781
1782 // if (in_bounds) {
1783 // ret_value = operand0[index]; // source
1784 // } else {
1785 // ret_value = *operand1; // padding
1786 // }
1787 llvm::Value* ret_value_addr = llvm_ir::EmitAllocaAtFunctionEntry(
1788 llvm_ir::PrimitiveTypeToIrType(hlo->shape().element_type(),
1789 module_),
1790 "pad_result_addr", ir_builder_);
1791 llvm_ir::LlvmIfData if_data =
1792 llvm_ir::EmitIfThenElse(in_bounds, "in_bounds", ir_builder_);
1793 SetToFirstInsertPoint(if_data.true_block, ir_builder_);
1794 TF_ASSIGN_OR_RETURN(llvm::Value * operand_value,
1795 operand_to_generator.at(hlo->operand(0))(index));
1796 ir_builder_->CreateStore(operand_value, ret_value_addr);
1797
1798 SetToFirstInsertPoint(if_data.false_block, ir_builder_);
1799 TF_ASSIGN_OR_RETURN(llvm::Value * padding_value,
1800 operand_to_generator.at(hlo->operand(1))({}));
1801 ir_builder_->CreateStore(padding_value, ret_value_addr);
1802
1803 SetToFirstInsertPoint(if_data.after_block, ir_builder_);
1804 // Don't create phi(operand_value, padding_value) here, because invoking
1805 // operand_to_generator may create new basic blocks, making the parent
1806 // of operand_value or padding_value no longer a predecessor of
1807 // if_data.after_block.
1808 return ir_builder_->CreateLoad(ret_value_addr);
1809 };
1810
1811 case HloOpcode::kDot:
1812 return [=, &operand_to_generator](const IrArray::Index& dot_result_index)
1813 -> StatusOr<llvm::Value*> {
1814 auto lhs_generator = operand_to_generator.at(hlo->operand(0));
1815 auto rhs_generator = operand_to_generator.at(hlo->operand(1));
1816 int64 contracted_dim_size = hlo->operand(0)->shape().dimensions(
1817 hlo->operand(0)->shape().dimensions_size() - 1);
1818 int64 lhs_dims = hlo->operand(0)->shape().dimensions_size();
1819 int64 rhs_dims = hlo->operand(1)->shape().dimensions_size();
1820
1821 std::unique_ptr<llvm_ir::ForLoop> inner_loop =
1822 llvm_ir::ForLoop::EmitForLoop(
1823 IrName(hlo, "inner"), ir_builder_->getInt64(0),
1824 ir_builder_->getInt64(contracted_dim_size),
1825 ir_builder_->getInt64(1), ir_builder_);
1826
1827 SetToFirstInsertPoint(inner_loop->GetPreheaderBasicBlock(),
1828 ir_builder_);
1829 PrimitiveType primitive_type = hlo->shape().element_type();
1830 llvm::Type* primitive_type_llvm =
1831 llvm_ir::PrimitiveTypeToIrType(primitive_type, module_);
1832 llvm::Value* accumulator_alloca = llvm_ir::EmitAllocaAtFunctionEntry(
1833 primitive_type_llvm, "dot_acc", ir_builder_);
1834 ir_builder_->CreateStore(
1835 llvm::Constant::getNullValue(primitive_type_llvm),
1836 accumulator_alloca);
1837
1838 SetToFirstInsertPoint(inner_loop->GetBodyBasicBlock(), ir_builder_);
1839
1840 // This is the inner reduction loop for a dot operation that produces
1841 // one element in the output. If the operands to the dot operation have
1842 // shapes [A,B,C,T] and [D,T,E], the result has a shape [A,B,C,D,E].
1843 // Given an output index [a,b,c,d,e] in the result, we compute:
1844 // sum(lhs[a,b,c,t]*rhs[d,t,e] for t in [0, T))
1845
1846 IrArray::Index lhs_index, rhs_index;
1847
1848 for (int64 i = 0; i < lhs_dims - 1; i++) {
1849 lhs_index.push_back(dot_result_index[i]);
1850 }
1851 lhs_index.push_back(inner_loop->GetIndVarValue());
1852
1853 for (int64 i = 0; i < rhs_dims - 2; i++) {
1854 rhs_index.push_back(dot_result_index[lhs_dims - 1 + i]);
1855 }
1856 rhs_index.push_back(inner_loop->GetIndVarValue());
1857 rhs_index.push_back(dot_result_index.back());
1858
1859 llvm::Value* current_accumulator =
1860 ir_builder_->CreateLoad(accumulator_alloca);
1861 TF_ASSIGN_OR_RETURN(llvm::Value * lhs_value, lhs_generator(lhs_index));
1862 TF_ASSIGN_OR_RETURN(llvm::Value * rhs_value, rhs_generator(rhs_index));
1863 llvm::Value* next_accumulator;
1864 if (primitive_util::IsComplexType(primitive_type)) {
1865 llvm::Value* product_real = ir_builder_->CreateFSub(
1866 ir_builder_->CreateFMul(EmitExtractReal(lhs_value),
1867 EmitExtractReal(rhs_value)),
1868 ir_builder_->CreateFMul(EmitExtractImag(lhs_value),
1869 EmitExtractImag(rhs_value)));
1870 llvm::Value* product_imag = ir_builder_->CreateFAdd(
1871 ir_builder_->CreateFMul(EmitExtractReal(lhs_value),
1872 EmitExtractImag(rhs_value)),
1873 ir_builder_->CreateFMul(EmitExtractImag(lhs_value),
1874 EmitExtractReal(rhs_value)));
1875 next_accumulator = ir_builder_->CreateInsertValue(
1876 current_accumulator,
1877 ir_builder_->CreateFAdd(EmitExtractReal(current_accumulator),
1878 product_real),
1879 {0});
1880 next_accumulator = ir_builder_->CreateInsertValue(
1881 next_accumulator,
1882 ir_builder_->CreateFAdd(EmitExtractImag(current_accumulator),
1883 product_imag),
1884 {1});
1885 } else if (primitive_util::IsFloatingPointType(primitive_type)) {
1886 next_accumulator = ir_builder_->CreateFAdd(
1887 current_accumulator,
1888 ir_builder_->CreateFMul(lhs_value, rhs_value));
1889 } else {
1890 next_accumulator = ir_builder_->CreateAdd(
1891 current_accumulator,
1892 ir_builder_->CreateMul(lhs_value, rhs_value));
1893 }
1894 ir_builder_->CreateStore(next_accumulator, accumulator_alloca);
1895
1896 SetToFirstInsertPoint(inner_loop->GetExitBasicBlock(), ir_builder_);
1897 return ir_builder_->CreateLoad(accumulator_alloca);
1898 };
1899 default:
1900 return [this, hlo, &operand_to_generator](const IrArray::Index& index) {
1901 return Unimplemented("Unhandled opcode for elemental IR emission: %s",
1902 HloOpcodeString(hlo->opcode()).c_str());
1903 };
1904 }
1905}
