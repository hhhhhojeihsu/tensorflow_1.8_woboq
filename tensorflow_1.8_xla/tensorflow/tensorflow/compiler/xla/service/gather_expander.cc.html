28static StatusOr<HloInstruction*> TransposeIndexVectorDimToLast(
29 HloInstruction* gather_indices, int64 index_vector_dim) {
30 const Shape& gather_indices_shape = gather_indices->shape();
31 if (index_vector_dim == (gather_indices_shape.dimensions_size() - 1)) {
32 return gather_indices;
33 }
34 std::vector<int64> permutation;
35 permutation.reserve(gather_indices_shape.dimensions_size());
36 for (int64 i = 0, e = gather_indices_shape.dimensions_size(); i < e; i++) {
37 if (i != index_vector_dim) {
38 permutation.push_back(i);
39 }
40 }
41 permutation.push_back(index_vector_dim);
42 return MakeTransposeHlo(gather_indices, permutation);
43}

45// If the gather_indices holds scalar indices (i.e. gather_indices has rank N
46// and index_vector_dim is N) then reshape it to have a trailing degenerate
47// dimension. This makes the code for slicing out the index vector more
48// uniform.
49static StatusOr<HloInstruction*> DeScalarizeGatherIndices(
50 HloInstruction* gather_indices, int64 index_vector_dim) {
51 const Shape& gather_indices_shape = gather_indices->shape();
52 if (index_vector_dim != gather_indices_shape.dimensions_size()) {
53 return gather_indices;
54 }
55
56 DCHECK_EQ(index_vector_dim, gather_indices_shape.dimensions_size());
57
58 std::vector<int64> result_shape_dims;
59 c_copy(gather_indices_shape.dimensions(),
60 std::back_inserter(result_shape_dims));
61 result_shape_dims.push_back(1);
62
63 return MakeReshapeHlo(result_shape_dims, gather_indices);
64}

66// Canonicalizes the gather_indices tensors so that we only have deal with some
67// specific cases in the while loop that does the heavy lifting.
68//
69// See the "High Level Algorithm" section for a broader picture.
70static StatusOr<HloInstruction*> CanonicalizeGatherIndices(
71 HloInstruction* gather_indices, int64 index_vector_dim) {
72 // If gather_indices holds scalar indices, normalize it to hold index vectors
73 // of size 1.
74 TF_ASSIGN_OR_RETURN(
75 HloInstruction * descalarized_gather_indices,
76 DeScalarizeGatherIndices(gather_indices, index_vector_dim));
77
78 // Transpose the non-index-vector dimensions to the front.
79 TF_ASSIGN_OR_RETURN(HloInstruction * transposed_gather_indices,
80 TransposeIndexVectorDimToLast(descalarized_gather_indices,
81 index_vector_dim));
82
83 // If there is only one index (i.e. gather_indices has rank 1 and this gather
84 // is really just a dynamic slice) add a leading degenerate dimension for
85 // uniformity. Otherwise create a "collapsed" leading dimension that subsumes
86 // all of the non-index-vector dimensions.
87 const Shape& shape = transposed_gather_indices->shape();
88 if (shape.dimensions_size() == 1) {
89 return ExpandFirstDimIntoNDims(transposed_gather_indices,
90 {1, shape.dimensions(0)});
91 } else {
92 return CollapseFirstNDims(transposed_gather_indices,
93 shape.dimensions_size() - 1);
94 }
95}

97// Expands out or contracts away the gather dimensions in the accumulator
98// produced by the while loop.
99static StatusOr<HloInstruction*> AdjustGatherDimsInAccumulator(
100 const Shape& gather_indices_shape, HloInstruction* accumulator,
101 int64 index_vector_dim) {
102 std::vector<int64> output_gather_dim_bounds;
103 output_gather_dim_bounds.reserve(gather_indices_shape.dimensions_size());
104 for (int64 i = 0, e = gather_indices_shape.dimensions_size(); i < e; i++) {
105 if (i != index_vector_dim) {
106 output_gather_dim_bounds.push_back(gather_indices_shape.dimensions(i));
107 }
108 }
109
110 if (output_gather_dim_bounds.empty()) {
111 // If output_gather_dim_bounds is empty we must be lowering a (effectively)
112 // dynamic-slice. In that case, there is a leading degenerate gather
113 // dimension that we added to make this special case play well with the
114 // general while loop which we need to remove now.
115 CHECK_EQ(accumulator->shape().dimensions(0), 1);
116 ArraySlice<int64> reshaped_dim_sizes =
117 AsInt64Slice(accumulator->shape().dimensions());
118 reshaped_dim_sizes.remove_prefix(1);
119 return MakeReshapeHlo(reshaped_dim_sizes, accumulator);
120 }
121
122 return ExpandFirstDimIntoNDims(accumulator, output_gather_dim_bounds);
123}

125// Expand an index vector from the gather_indices tensor into a vector that can
126// be used to dynamic-slice out of the gather operand.
127static StatusOr<HloInstruction*> ExpandIndexVectorIntoOperandSpace(
128 HloInstruction* index_vector, const GatherDimensionNumbers& dim_numbers,
129 int64 operand_rank) {
130 HloComputation* computation = index_vector->parent();
131 const Shape& index_shape = index_vector->shape();
132 HloInstruction* zero =
133 computation->AddInstruction(HloInstruction::CreateConstant(
134 Literal::CreateFromDimensions(index_shape.element_type(), {1})));
135
136 // We extract out individual components from the smaller index and concatenate
137 // them (interspersing zeros as needed) into the larger index.
138 std::vector<HloInstruction*> expanded_index_components;
139
140 for (int i = 0; i < operand_rank; i++) {
141 int64 index_vector_dim_index =
142 FindIndex(dim_numbers.gather_dims_to_operand_dims(), i);
143 if (index_vector_dim_index !=
144 dim_numbers.gather_dims_to_operand_dims_size()) {
145 TF_ASSIGN_OR_RETURN(
146 HloInstruction * component_to_concat,
147 MakeSliceHlo(index_vector, /*start_indices=*/{index_vector_dim_index},
148 /*limit_indices=*/{index_vector_dim_index + 1},
149 /*strides=*/{1}));
150 expanded_index_components.push_back(component_to_concat);
151 } else {
152 expanded_index_components.push_back(zero);
153 }
154 }
155
156 return MakeConcatHlo(expanded_index_components, /*dimension=*/0);
157}

159// This generates the body of the while that implements the main data movement
160// behavior of gather using dynamic-slice and dynamic-update-slice.
161static StatusOr<std::vector<HloInstruction*>> GatherLoopBody(
162 const HloInstruction& gather, HloInstruction* induction_var,
163 const std::vector<HloInstruction*>& incoming_loop_state) {
164 CHECK_EQ(incoming_loop_state.size(), 3);
165 HloInstruction* const operand = incoming_loop_state[0];
166 HloInstruction* const gather_indices = incoming_loop_state[1];
167 HloInstruction* const output_accumulator = incoming_loop_state[2];
168
169 int64 index_vector_size = gather_indices->shape().dimensions(1);
170
171 TF_ASSIGN_OR_RETURN(
172 HloInstruction * induction_var_as_vector,
173 MakeBroadcastHlo(induction_var, /*broadcast_dimensions=*/{},
174 /*result_shape_bounds=*/{1}));
175
176 TF_ASSIGN_OR_RETURN(
177 HloInstruction * index_into_gather_indices,
178 PadVectorWithZeros(induction_var_as_vector,
179 /*zeros_to_prepend=*/0, /*zeros_to_append=*/1));
180
181 TF_ASSIGN_OR_RETURN(
182 HloInstruction * index_vector_2d,
183 MakeDynamicSliceHlo(gather_indices, index_into_gather_indices,
184 {1, index_vector_size}));
185
186 TF_ASSIGN_OR_RETURN(HloInstruction * index_vector,
187 ElideDegenerateDims(index_vector_2d, {0}));
188
189 TF_ASSIGN_OR_RETURN(HloInstruction * gathered_slice_start,
190 ExpandIndexVectorIntoOperandSpace(
191 index_vector, gather.gather_dimension_numbers(),
192 operand->shape().dimensions_size()));
193
194 TF_ASSIGN_OR_RETURN(HloInstruction * gathered_slice,
195 MakeDynamicSliceHlo(operand, gathered_slice_start,
196 gather.gather_window_bounds()));
197
198 TF_ASSIGN_OR_RETURN(
199 HloInstruction * gathered_slice_for_update,
200 ExpandFirstDimIntoNDims(gathered_slice,
201 {1, gathered_slice->shape().dimensions(0)}));
202
203 TF_ASSIGN_OR_RETURN(
204 HloInstruction * index_vector_into_accumulator,
205 PadVectorWithZeros(
206 induction_var_as_vector, /*zeros_to_prepend=*/0,
207 /*zeros_to_append=*/gathered_slice->shape().dimensions_size()));
208
209 TF_ASSIGN_OR_RETURN(
210 HloInstruction * updated_accumulator,
211 MakeDynamicUpdateSliceHlo(output_accumulator, gathered_slice_for_update,
212 index_vector_into_accumulator));
213
214 // New loop state -- only the accumulator has changed. The
215 // WhileUtil::MakeCountedLoop functions takes care of the induction variable
216 // and the while loop exit condition.
217 return StatusOr<std::vector<HloInstruction*>>{
218 {operand, gather_indices, updated_accumulator}};
219}

221static StatusOr<HloInstruction*> CreateGatherLoopAccumulatorInitValue(
222 HloComputation* computation, PrimitiveType element_type,
223 ArraySlice<int64> window_bounds, int64 gather_loop_trip_count) {
224 std::vector<int64> accumulator_state_shape_dims;
225 accumulator_state_shape_dims.reserve(1 + window_bounds.size());
226 accumulator_state_shape_dims.push_back(gather_loop_trip_count);
227 c_copy(window_bounds, std::back_inserter(accumulator_state_shape_dims));
228 return BroadcastZeros(computation, element_type,
229 accumulator_state_shape_dims);
230}

232static StatusOr<HloInstruction*> ElideWindowDimsFromAccumulator(
233 HloInstruction* accumulator, const GatherDimensionNumbers& dim_numbers) {
234 std::vector<int64> dims_to_elide;
235 dims_to_elide.reserve(dim_numbers.elided_window_dims_size());
236 for (int64 elided_window_dim : dim_numbers.elided_window_dims()) {
237 dims_to_elide.push_back(elided_window_dim + 1);
238 }
239
240 return ElideDegenerateDims(accumulator, dims_to_elide);
241}

243// `accumulator` is almost the tensor the gather operation would have produced,
244// except that it has the dimensions in the wrong order -- the gather dimensions
245// are the major dimensions and the window dimensions are the minor dimensions.
246// Fix this up with a transpose.
247static StatusOr<HloInstruction*> PermuteGatherAndWindowDims(
248 HloInstruction* accumulator, ArraySlice<int64> output_window_dims,
249 int64 output_rank) {
250 std::vector<int64> permutation;
251 permutation.reserve(output_rank);
252
253 int64 gather_idx_counter = 0;
254 int64 window_idx_counter = output_rank - output_window_dims.size();
255 for (int64 i = 0; i < output_rank; i++) {
256 bool is_window_dim = c_binary_search(output_window_dims, i);
257 if (is_window_dim) {
258 permutation.push_back(window_idx_counter++);
259 } else {
260 permutation.push_back(gather_idx_counter++);
261 }
262 }
263
264 return MakeTransposeHlo(accumulator, permutation);
265}

267// High Level Algorithm
268//
269// We follow the following steps in sequence:
270//
271// 1. We canonicalize the gather_indices tensor such that it has rank
272// 2 (i.e. is a matrix) where each row is an index vector into the
273// operand.
274// 2. We iterate over the set of indices in the canonicalized
275// gather_indices tensor using a while loop, accumulating slices
276// of the operand tensor into an accumulator using
277// DynamicUpdateSlice.
278// 3. The accumulator result from the while loop from (2) is then
279// reshaped to split out all the individual gather dimensions and
280// then transposed to give the final result.
281//
282// As an example, if we started with the following operation:
283//
284// HloModule TensorFlowGatherMultipleBatchDims
285//
286// ENTRY main {
287// operand = s32[3,3] parameter(0)
288// indices = s32[2,2] parameter(1)
289// ROOT gather = s32[2,3,2] gather(operand, indices),
290// output_window_dims={1},
291// elided_window_dims={1},
292// gather_dims_to_operand_dims={1},
293// index_vector_dim=2,
294// window_bounds={3, 1}
295// }
296//
297// We'd first reshape indices to s32[4,1], where each row is an index
298// into operand. We'd then run a loop to slice out 4 tensors of shape
299// [3,1] out of operand into an accumulator of shape [4,3,1]. We then
300// reshape this result to [2,2,3] and finally transpose it to [2,3,2].
301
302StatusOr<HloInstruction*> GatherExpander::ExpandGather(
303 HloInstruction* gather_instr) {
304 CHECK(!ShapeUtil::HasZeroElements(gather_instr->shape()));
305
306 HloComputation* computation = gather_instr->parent();
307 HloInstruction* operand = gather_instr->mutable_operand(0);
308 HloInstruction* gather_indices = gather_instr->mutable_operand(1);
309 const Shape& gather_indices_shape = gather_indices->shape();
310 const Shape& output_shape = gather_instr->shape();
311 int64 output_rank = output_shape.dimensions_size();
312
313 const GatherDimensionNumbers& dim_numbers =
314 gather_instr->gather_dimension_numbers();
315
316 int64 gather_loop_trip_count = 1;
317 for (int64 i = 0, e = gather_indices_shape.dimensions_size(); i < e; i++) {
318 if (i != dim_numbers.index_vector_dim()) {
319 gather_loop_trip_count *= gather_indices_shape.dimensions(i);
320 }
321 }
322
323 if (!IsInt32(gather_loop_trip_count)) {
324 return Unimplemented(
325 "Gather operations with more than 2147483647 gather indices are not "
326 "supported. This error occurred for %s.",
327 gather_instr->ToString().c_str());
328 }
329
330 TF_ASSIGN_OR_RETURN(HloInstruction * canonical_gather_indices,
331 CanonicalizeGatherIndices(
332 gather_indices, dim_numbers.index_vector_dim()));
333
334 CHECK_EQ(gather_loop_trip_count,
335 canonical_gather_indices->shape().dimensions(0));
336
337 TF_ASSIGN_OR_RETURN(
338 HloInstruction * accumulator_init,
339 CreateGatherLoopAccumulatorInitValue(
340 computation, output_shape.element_type(),
341 gather_instr->gather_window_bounds(), gather_loop_trip_count));
342
343 StatusOr<std::vector<HloInstruction*>> gather_loop_result_or_error =
344 WhileUtil::MakeCountedLoop(
345 computation, gather_loop_trip_count,
346 {operand, canonical_gather_indices, accumulator_init},
347 [&](HloInstruction* indvar,
348 const std::vector<HloInstruction*>& loop_state) {
349 return GatherLoopBody(*gather_instr, indvar, loop_state);
350 });
351
352 TF_ASSIGN_OR_RETURN(std::vector<HloInstruction*> gather_loop_result,
353 gather_loop_result_or_error);
354
355 HloInstruction* accumulator_result = gather_loop_result.back();
356 TF_ASSIGN_OR_RETURN(
357 HloInstruction * accumulator_with_window_dims_elided,
358 ElideWindowDimsFromAccumulator(accumulator_result, dim_numbers));
359
360 TF_ASSIGN_OR_RETURN(
361 HloInstruction * accumulator_with_output_gather_dims_decanonicalized,
362 AdjustGatherDimsInAccumulator(gather_indices->shape(),
363 accumulator_with_window_dims_elided,
364 dim_numbers.index_vector_dim()));
365
366 return PermuteGatherAndWindowDims(
367 accumulator_with_output_gather_dims_decanonicalized,
368 AsInt64Slice(dim_numbers.output_window_dims()), output_rank);
369}
370
371StatusOr<bool> GatherExpander::Run(HloModule* module) {
372 auto is_nontrivial_gather = [](HloInstruction* inst) {
373 return inst->opcode() == HloOpcode::kGather &&
374 // Avoid expanding gather ops that produce zero sized tensors,
375 // instead punt these to ZeroSizedHloElimination.
376 !ShapeUtil::HasZeroElements(inst->shape());
377 };
378
379 std::vector<HloInstruction*> gather_instrs;
380 for (HloComputation* computation : module->MakeNonfusionComputations()) {
381 c_copy_if(computation->instructions(), std::back_inserter(gather_instrs),
382 is_nontrivial_gather);
383 }
384
385 for (HloInstruction* inst : gather_instrs) {
386 TF_ASSIGN_OR_RETURN(HloInstruction * expanded_root, ExpandGather(inst));
387 TF_RETURN_IF_ERROR(inst->parent()->ReplaceInstruction(inst, expanded_root));
388 }
389
390 return !gather_instrs.empty();
391}
