29// Finds and returns the non-constant operand in instr.
30//
31// CHECK-fails if instr doesn't have exactly one unique non-constant operand.
32static const HloInstruction* NonConstantOperand(const HloInstruction* instr) {
33 const HloInstruction* result = nullptr;
34 for (const HloInstruction* operand : instr->operands()) {
35 if (!operand->IsConstant()) {
36 if (result != nullptr) {
37 CHECK_EQ(result, operand);
38 }
39 result = operand;
40 }
41 }
42 CHECK_NE(result, nullptr);
43 return result;
44}

46// Determines whether the given instruction is a send/recv node, or has a
47// subcomputation which contains a send/recv node.
48static bool IsOrContainsSendOrRecv(const HloInstruction* instr);
49
50// Determines whether the given computation contains a send or recv node.
51static bool ContainsSendOrRecv(const HloComputation* comp) {
52 for (const auto* instr : comp->instructions()) {
53 if (IsOrContainsSendOrRecv(instr)) {
54 return true;
55 }
56 }
57 return false;
58}
59
60static bool IsOrContainsSendOrRecv(const HloInstruction* instr) {
61 if (instr->opcode() == HloOpcode::kSend ||
62 instr->opcode() == HloOpcode::kSendDone ||
63 instr->opcode() == HloOpcode::kRecv ||
64 instr->opcode() == HloOpcode::kRecvDone) {
65 return true;
66 }
67 for (const auto& subcomp : instr->called_computations()) {
68 if (ContainsSendOrRecv(subcomp)) {
69 return true;
70 }
71 }
72 return false;
73}

75// If all of instr's operands are either constants or have the form
76// get-tuple-element(gte_operand, N)
77// for the same value N, returns N. Otherwise, returns nullopt.
78static optional<int64> GetGTEOperandIndex(const HloInstruction* instr,
79 const HloInstruction* gte_operand) {
80 VLOG(2) << "GetGTEOperandIndex(" << instr->ToString() << ", "
81 << gte_operand->ToString() << ")";
82 optional<int64> tuple_idx;
83 for (const HloInstruction* operand : instr->operands()) {
84 if (operand->IsConstant()) {
85 continue;
86 }
87 if (operand->opcode() != HloOpcode::kGetTupleElement) {
88 VLOG(2) << "instr uses something other than gte(gte_operand): "
89 << operand->ToString();
90 return nullopt;
91 }
92 if (operand->operand(0) != gte_operand) {
93 VLOG(2) << "instr has gte whose operand is not gte_operand: "
94 << operand->ToString();
95 return nullopt;
96 }
97 if (tuple_idx && tuple_idx != operand->tuple_index()) {
98 VLOG(2) << "instr has operands with conflicting gte indices, "
99 << *tuple_idx << " vs " << operand->tuple_index();
100 return nullopt;
101 }
102
103 tuple_idx = operand->tuple_index();
104 }
105 return tuple_idx;
106}

108// Tries to get the tuple index of the induction variable of a while loop.
109//
110// Checks that the loop condition and root both plumb the induction variable
111// through the same tuple index, and that they both apply exactly one op to the
112// induction variable before deciding whether to do another loop iteration (in
113// the loop condition's case) or packing the induction variable into the result
114// tuple (in the loop body's case).
115//
116// Specifically, checks that the loop condition has structure
117//
118// root = op(constants, get-tuple-elem(param0, N), constants)
119//
120// and the loop body has the structure
121//
122// inc = op(constants, get-tuple-elem(param0, N), constants)
123// root = tuple(..., inc, ...) // inc is N'th operand of tuple().
124//
125// If so, returns N. Otherwise, returns nullopt.
126static optional<int64> GetLoopInductionVarTupleIdx(
127 const HloInstruction* while_op) {
128 CHECK_EQ(while_op->opcode(), HloOpcode::kWhile);
129 VLOG(2) << "Finding induction variable for loop "
130 << while_op->ToShortString();
131
132 // The while_cond computation should have the form
133 //
134 // while_cond_root =
135 // op(constants, get-tuple-elem(while_cond_param, N), constants).
136 //
137 // If it does, set indvar_tuple_idx to N.
138 auto* while_cond = while_op->while_condition();
139 auto* while_cond_root = while_cond->root_instruction();
140 auto* while_cond_param = while_cond->parameter_instruction(0);
141 optional<int64> indvar_tuple_idx =
142 GetGTEOperandIndex(while_cond_root, while_cond_param);
143 if (!indvar_tuple_idx) {
144 VLOG(2) << "Induction variable not found in loop condition: "
145 << while_cond->root_instruction()->ToString();
146 return nullopt;
147 }
148
149 // The while_body computation should have the form
150 //
151 // while_body_inc =
152 // op(constants, get-tuple-elem(while_body_param, N), constants)
153 // while_body_root = tuple(..., while_body_inc, ...)
154 //
155 // where while_body_inc is operand N of while_body_root.
156 auto* while_body = while_op->while_body();
157 auto* while_body_root = while_body->root_instruction();
158 if (while_body_root->opcode() != HloOpcode::kTuple) {
159 VLOG(2) << "While body's root is not a tuple instruction: "
160 << while_body_root->ToString();
161 return nullopt;
162 }
163
164 auto* while_body_inc = while_body_root->operand(*indvar_tuple_idx);
165 auto* while_body_param = while_body->parameter_instruction(0);
166 optional<int64> while_body_indvar_tuple_idx =
167 GetGTEOperandIndex(while_body_inc, while_body_param);
168 if (!while_body_indvar_tuple_idx) {
169 VLOG(2)
170 << "Induction variable not found in while body increment instruction: "
171 << while_body_inc->ToString();
172 return nullopt;
173 }
174 if (while_body_indvar_tuple_idx != indvar_tuple_idx) {
175 VLOG(2) << "Tuple index of induction variable does not match between loop "
176 "condition ("
177 << *indvar_tuple_idx << ") and while body ("
178 << *while_body_indvar_tuple_idx << ")";
179 return nullopt;
180 }
181
182 // Finally, check that the while loop's initial value is a tuple with enough
183 // elements.
184 auto* while_init = while_op->operand(0);
185 if (while_init->opcode() != HloOpcode::kTuple) {
186 VLOG(2) << "While init expected to be a tuple: " << while_init->ToString();
187 return nullopt;
188 }
189
190 VLOG(2) << "Induction variable's tuple index: " << *indvar_tuple_idx;
191 return indvar_tuple_idx;
192}

194// Tries to determine the number of times the given loop executes. Currently
195// simply returns 0, 1, or "can't tell" (nullopt).
196static optional<int64> GetLoopTripCount(HloInstruction* while_op) {
197 CHECK_EQ(while_op->opcode(), HloOpcode::kWhile);
198 VLOG(2) << "Getting trip count for loop " << while_op->ToString();
199
200 // The loop's induction variable is found at
201 //
202 // get-tuple-elem(comp->parameter_instruction(0), *indvar_tuple_idx),
203 //
204 // where comp is while_op->while_body() or while_op->while_condition().
205 optional<int64> indvar_tuple_idx = GetLoopInductionVarTupleIdx(while_op);
206 if (!indvar_tuple_idx) {
207 return nullopt;
208 }
209
210 VLOG(2) << "Induction variable is at index " << *indvar_tuple_idx
211 << " in input tuple.";
212
213 // Now that we know the index of the induction variable, we can we can try to
214 // compute how many times the loop executes. Start by computing the induction
215 // variable's initial value.
216 HloEvaluator evaluator(/*max_loop_iterations=*/0);
217 auto* while_init = while_op->mutable_operand(0);
218 auto* indvar_init = while_init->mutable_operand(*indvar_tuple_idx);
219 StatusOr<std::unique_ptr<Literal>> indvar_init_result =
220 evaluator.Evaluate(indvar_init);
221 if (!indvar_init_result.ok()) {
222 VLOG(2) << "Couldn't evaluate induction variable init: "
223 << indvar_init_result.status();
224 return nullopt;
225 }
226
227 // Evaluates the while loop's condition, returning either "true" (continue
228 // looping), "false" (stop looping), or nullopt (can't evaluate).
229 auto evaluate_while_cond = [&](const Literal& indvar) -> optional<bool> {
230 auto* while_cond = while_op->while_condition();
231 auto* while_cond_root = while_cond->root_instruction();
232 auto* while_cond_indvar = NonConstantOperand(while_cond_root);
233 StatusOr<std::unique_ptr<Literal>> result =
234 evaluator.EvaluateWithSubstitutions(while_cond_root,
235 {{while_cond_indvar, &indvar}});
236 if (!result.ok()) {
237 VLOG(2) << "Couldn't evaluate while cond: " << result.status();
238 return nullopt;
239 }
240 return result.ValueOrDie()->data<bool>() ==
241 tensorflow::gtl::ArraySlice<bool>{true};
242 };
243
244 // The initial value of the induction variable.
245 const Literal& indvar_iter0_val = *indvar_init_result.ValueOrDie();
246
247 // Evaluate whether the while condition is true when seeded with
248 // indvar_iter0_val.
249 optional<bool> while_cond_iter0_val = evaluate_while_cond(indvar_iter0_val);
250 if (while_cond_iter0_val == false) {
251 VLOG(2) << "Loop has static trip count of 0.";
252 return 0;
253 }
254
255 // Calculate the value of the induction variable after one iteration of the
256 // loop, and check whether the while condition is true with this new value.
257 auto* while_body = while_op->while_body();
258 auto* while_body_indvar_update =
259 while_body->root_instruction()->operand(*indvar_tuple_idx);
260 auto* while_body_indvar = NonConstantOperand(while_body_indvar_update);
261 StatusOr<std::unique_ptr<Literal>> indvar_iter1_result =
262 evaluator.EvaluateWithSubstitutions(
263 while_body_indvar_update, {{while_body_indvar, &indvar_iter0_val}});
264 if (!indvar_iter1_result.ok()) {
265 VLOG(2) << "Couldn't evaluate induction variable update: "
266 << indvar_iter1_result.status();
267 return nullopt;
268 }
269 const Literal& indvar_iter1_val = *indvar_iter1_result.ValueOrDie();
270 optional<bool> while_cond_iter1_val = evaluate_while_cond(indvar_iter1_val);
271 if (while_cond_iter1_val == false) {
272 VLOG(2) << "Determined that loop has static trip count of 1.";
273 return 1;
274 }
275
276 VLOG(2) << "Loop has unknown trip count >= 1.";
277 return nullopt;
278}

280// Tries to remove elements in a while loop's tuple that aren't used within the
281// loop.
282//
283// Specifically, if a loop is tuple-shaped, and there exists some element of
284// that tuple that is not used by the loop condition and is not used by the loop
285// body except to pass it to the next iteration of the loop, then we can remove
286// that element from the loop's tuples.
287static StatusOr<bool> TryRemoveDeadWhileParams(HloInstruction* while_op) {
288 CHECK_EQ(while_op->opcode(), HloOpcode::kWhile);
289
290 // Don't try this transformation if the while loop isn't removable, since if
291 // it succeeds ultimately we're going to have to replace the old while loop
292 // with a new one.
293 if (!while_op->parent()->IsRemovable(while_op) || while_op->HasSideEffect()) {
294 VLOG(2) << "Can't remove dead parameters from non-removable while op.";
295 return false;
296 }
297
298 HloModule* module = while_op->GetModule();
299 HloComputation* computation = while_op->parent();
300 HloInstruction* while_init = while_op->mutable_operand(0);
301 HloComputation* while_cond = while_op->while_condition();
302 HloComputation* while_body = while_op->while_body();
303 HloInstruction* while_body_root = while_body->root_instruction();
304
305 if (!ShapeUtil::IsTuple(while_init->shape())) {
306 VLOG(2) << "While op's carried value isn't tuple shaped.";
307 return false;
308 }
309
310 if (while_body_root->opcode() != HloOpcode::kTuple) {
311 VLOG(2) << "While body's root is not a tuple(...) instruction.";
312 return false;
313 }
314
315 auto print_no_metadata = HloPrintOptions().set_print_metadata(false);
316
317 // Bail if param0 of while_cond or while_body has users which aren't of type
318 // get-tuple-element.
319 for (const HloInstruction* instr : {while_body->parameter_instruction(0),
320 while_cond->parameter_instruction(0)}) {
321 for (const HloInstruction* user : instr->users()) {
322 if (user->opcode() != HloOpcode::kGetTupleElement) {
323 VLOG(2) << "Cowardly refusing to analyze while loop with "
324 << instr->ToString(print_no_metadata)
325 << " used by non-GTE instruction "
326 << user->ToString(print_no_metadata) << " in computation "
327 << instr->parent()->name();
328 return false;
329 }
330 }
331 }
332
333 const int64 tuple_size = ShapeUtil::TupleElementCount(while_init->shape());
334 if (tuple_size == 0) {
335 VLOG(2) << "Can't remove elements from while loop's tuple -- it's already "
336 "empty.";
337 return false;
338 }
339
340 tensorflow::gtl::FlatSet<int64> used_tuple_indices;
341 for (HloComputation* comp : {while_body, while_cond}) {
342 // The HLO verifier ensures that while_input's shape matches while_init's
343 // shape, which we verified above is a tuple.
344 HloInstruction* while_input = comp->parameter_instruction(0);
345
346 for (const HloInstruction* user : while_input->users()) {
347 // This user doesn't count if it's only used by the while body's root, and
348 // the root places the tuple element into the same index of the tuple as
349 // it came from. That just amounts to us carrying the variable through
350 // the loop.
351 //
352 // Careful: HloInstruction::operand_index returns the first index the
353 // operand appears in, but it may appear more than once!
354 if (user->user_count() == 1 && user->users().front() == while_body_root &&
355 while_body_root->operand_index(user) == user->tuple_index() &&
356 std::count(while_body_root->operands().begin(),
357 while_body_root->operands().end(), user) == 1) {
358 continue;
359 }
360
361 used_tuple_indices.insert(user->tuple_index());
362 if (used_tuple_indices.size() == tuple_size) {
363 VLOG(2) << "Loop " << while_op->ToString(print_no_metadata)
364 << " uses all of its inputs; no simplification possible.";
365 return false;
366 }
367 }
368 }
369
370 // If a tuple element is not passed unmodified from the while body's param0
371 // through to the while body's root, count that element as "used", since
372 // removing that element would be observable.
373 for (int64 i = 0; i < while_body_root->operand_count(); ++i) {
374 if (used_tuple_indices.count(i)) {
375 continue;
376 }
377
378 auto* operand = while_body_root->operand(i);
379 if (operand->opcode() != HloOpcode::kGetTupleElement ||
380 operand->operand(0) != while_body->parameter_instruction(0) ||
381 operand->tuple_index() != i) {
382 VLOG(2) << "Tuple index " << i
383 << " is not passed through loop body unmodified.";
384 used_tuple_indices.insert(i);
385
386 if (used_tuple_indices.size() == tuple_size) {
387 VLOG(2) << "Loop " << while_op->ToString(print_no_metadata)
388 << " uses all of its inputs; no simplification possible.";
389 return false;
390 }
391 }
392 }
393
394 // If we got here, used_tuple_indices.size() < tuple_size, meaning some
395 // elements of the loop's tuple aren't used by while_body or while_cond.
396 CHECK_LT(used_tuple_indices.size(), tuple_size);
397
398 VLOG(1) << "Eliminating " << tuple_size - used_tuple_indices.size()
399 << " elements from tuple of "
400 << while_op->ToString(print_no_metadata);
401
402 // Build up maps from the old/new to the new/old tuple indices.
403 std::vector<int64> new_to_old_tuple_idx(used_tuple_indices.begin(),
404 used_tuple_indices.end());
405 std::sort(new_to_old_tuple_idx.begin(), new_to_old_tuple_idx.end());
406
407 tensorflow::gtl::FlatMap<int64, int64> old_to_new_tuple_idx;
408 for (int64 new_idx = 0; new_idx < new_to_old_tuple_idx.size(); ++new_idx) {
409 int64 old_idx = new_to_old_tuple_idx[new_idx];
410 old_to_new_tuple_idx[old_idx] = new_idx;
411 VLOG(2) << "Remapping tuple index " << old_idx << " to " << new_idx;
412 }
413
414 // Compute the shape of the while op after we remove the dead indices.
415 std::vector<Shape> new_while_tuple_elem_shapes;
416 new_while_tuple_elem_shapes.reserve(new_to_old_tuple_idx.size());
417 for (int64 old_idx : new_to_old_tuple_idx) {
418 new_while_tuple_elem_shapes.push_back(
419 while_init->shape().tuple_shapes(old_idx));
420 }
421 Shape new_while_shape =
422 ShapeUtil::MakeTupleShape(new_while_tuple_elem_shapes);
423
424 // Returns a map from elements in the computation to new instructions which
425 // replace the old instructions after we remove unused elements from the while
426 // tuple.
427 auto make_while_computation_replacements = [&](const HloComputation* comp) {
428 std::unordered_map<const HloInstruction*, std::unique_ptr<HloInstruction>>
429 replacements;
430
431 auto* param = comp->parameter_instruction(0);
432 replacements.emplace(param, HloInstruction::CreateParameter(
433 0, new_while_shape, param->name()));
434
435 // Materialize param's users, since we're about to add new ones below.
436 std::vector<HloInstruction*> materialized_users(param->users().begin(),
437 param->users().end());
438 for (const auto* user : materialized_users) {
439 // The while body root is handled separately.
440 if (user == while_body_root) {
441 continue;
442 }
443 CHECK_EQ(user->opcode(), HloOpcode::kGetTupleElement)
444 << user->ToString(print_no_metadata);
445
446 int64 old_idx = user->tuple_index();
447 auto new_idx_iter = old_to_new_tuple_idx.find(old_idx);
448 if (new_idx_iter != old_to_new_tuple_idx.end()) {
449 // This is a GTE of an index that survives. Replace it.
450 replacements.emplace(
451 user, HloInstruction::CreateGetTupleElement(user->shape(), param,
452 new_idx_iter->second));
453 } else {
454 // This is a GTE of an index that we've removed. Remove it from the
455 // cloned computation.
456 CHECK(user->user_count() == 0 ||
457 user->user_count() == 1 &&
458 user->users().front() == while_body_root)
459 << "Instruction " << user->ToString(print_no_metadata)
460 << " should be unused (except by root of while body), but has "
461 "users: {"
462 << tensorflow::str_util::Join(
463 user->users(), ", ",
464 [&](string* out, const HloInstruction* instr) {
465 tensorflow::strings::StrAppend(
466 out, instr->ToString(print_no_metadata));
467 })
468 << "}";
469
470 replacements.emplace(user, nullptr);
471 }
472 }
473 return replacements;
474 };
475
476 // Create the new while condition, body, and init value.
477 std::unique_ptr<HloComputation> new_while_cond =
478 while_cond->CloneWithReplacements(
479 make_while_computation_replacements(while_cond));
480
481 std::unordered_map<const HloInstruction*, std::unique_ptr<HloInstruction>>
482 while_body_replacements = make_while_computation_replacements(while_body);
483 std::vector<HloInstruction*> new_while_body_root_elems;
484 new_while_body_root_elems.reserve(new_to_old_tuple_idx.size());
485 for (int64 old_idx : new_to_old_tuple_idx) {
486 new_while_body_root_elems.push_back(
487 while_body_root->mutable_operand(old_idx));
488 }
489 while_body_replacements.emplace(
490 while_body_root, HloInstruction::CreateTuple(new_while_body_root_elems));
491 std::unique_ptr<HloComputation> new_while_body =
492 while_body->CloneWithReplacements(std::move(while_body_replacements));
493
494 // Add a new while_init instruction that repackages the old while_init
495 // instruction's elements. We rely on the AlgebraicSimplifier and DCE to
496 // clean this up in the common case where while_init is a tuple op. (It's
497 // definitely tuple-shaped, but it's not necessarily a tuple op.)
498 std::vector<HloInstruction*> new_while_init_elems;
499 new_while_init_elems.reserve(new_to_old_tuple_idx.size());
500 for (int64 old_idx : new_to_old_tuple_idx) {
501 new_while_init_elems.push_back(
502 computation->AddInstruction(HloInstruction::CreateGetTupleElement(
503 while_init->shape().tuple_shapes(old_idx), while_init, old_idx)));
504 }
505 auto* new_while_init = computation->AddInstruction(
506 HloInstruction::CreateTuple(new_while_init_elems));
507
508 // Create the new while op.
509 auto* new_while_op = computation->AddInstruction(HloInstruction::CreateWhile(
510 new_while_shape,
511 module->AddEmbeddedComputation(std::move(new_while_cond)),
512 module->AddEmbeddedComputation(std::move(new_while_body)),
513 new_while_init));
514
515 // Create a tuple op that recreates the output of the old while op. That is,
516 // we transform to
517 //
518 // new_while_init while_init
519 // | |
520 // V |
521 // new_while |
522 // | |
523 // -------| |----
524 // V V
525 // new_tuple
526 // |
527 // V
528 // (orig. users of while op)
529 //
530 // The tuple simplifier will then simplify this if possible, removing
531 // new_tuple and while_init.
532 std::vector<HloInstruction*> new_tuple_elems;
533 for (int64 old_idx = 0; old_idx < tuple_size; ++old_idx) {
534 auto new_tuple_idx_it = old_to_new_tuple_idx.find(old_idx);
535 if (new_tuple_idx_it != old_to_new_tuple_idx.end()) {
536 int64 gte_idx = new_tuple_idx_it->second;
537 new_tuple_elems.push_back(
538 computation->AddInstruction(HloInstruction::CreateGetTupleElement(
539 new_while_op->shape().tuple_shapes(gte_idx), new_while_op,
540 gte_idx)));
541 } else {
542 new_tuple_elems.push_back(
543 computation->AddInstruction(HloInstruction::CreateGetTupleElement(
544 while_init->shape().tuple_shapes(old_idx), while_init, old_idx)));
545 }
546 }
547 HloInstruction* new_tuple =
548 computation->AddInstruction(HloInstruction::CreateTuple(new_tuple_elems));
549 TF_RETURN_IF_ERROR(while_op->ReplaceAllUsesWith(new_tuple));
550
551 return true;
552}

554// Tries to remove a while loop from the graph.
555//
556// - Loops with trip count of 0 can be replaced by the loop's "init" value.
557// - Loops with trip count of 1 can be replaced by the loop's body, with the
558// loop itself removed.
559//
560// Returns true if it made a change to the graph.
561static StatusOr<bool> TryRemoveWhileLoop(HloInstruction* while_op) {
562 // Cowardly refuse to remove loops that are not removable. In practice,
563 // this means that we can't remove loops that contain side-effecting
564 // instructions or have control predecessors/successors.
565 //
566 // This is not a fundamental limitation. The control operands can be moved
567 // onto the new HLOs after simplification, and any side-effecting ops inside
568 // the loop aren't removed, just cloned and added back to the loop. But
569 // moving an op out of the loop also removes implicit control dependencies
570 // between the op and the ops outside the loop, so we'd have to add those back
571 // for things like infeed/outfeed. It gets complicated. So for now we just
572 // avoid it.
573 if (!while_op->parent()->IsRemovable(while_op) || while_op->HasSideEffect()) {
574 VLOG(2) << "Not attempting to remove while loop it is not removable: "
575 << while_op->ToShortString();
576 return false;
577 }
578
579 // Remove while loops with static trip count of 0.
580 optional<int64> trip_count = GetLoopTripCount(while_op);
581 if (trip_count && *trip_count == 0) {
582 // The loop never executes, so the value of the loop is the value of its
583 // "init" operand.
584 auto computation = while_op->parent();
585
586 // Remove while_op (i.e., call ReplaceInstruction rather than
587 // ReplaceUsesWithInstruction) so that if the algebraic simplifier is run in
588 // a loop without an intervening DCE, we don't try to re-remove the loop.
589 TF_RETURN_IF_ERROR(computation->ReplaceInstruction(
590 while_op, while_op->mutable_operand(0)));
591 return true;
592 }
593
594 // Transform while loops with static trip count of 1 into a call op, then
595 // inline the call.
596 if (trip_count && *trip_count == 1) {
597 auto computation = while_op->parent();
598 auto call_op = computation->AddInstruction(HloInstruction::CreateCall(
599 while_op->shape(), while_op->operands(), while_op->while_body()));
600 TF_RETURN_IF_ERROR(computation->ReplaceInstruction(while_op, call_op));
601 TF_ASSIGN_OR_RETURN(auto inlined_instructions_map,
602 CallInliner::Inline(call_op));
603 (void)inlined_instructions_map;
604 return true;
605 }
606 return false;
607}
608
609static StatusOr<bool> TryPropagateConstant(HloInstruction* while_op) {
610 auto while_init = while_op->operand(0);
611 if (while_init->opcode() != HloOpcode::kTuple) {
612 return false;
613 }
614
615 auto while_body = while_op->while_body();
616 auto while_body_root = while_body->root_instruction();
617 if (while_body_root->opcode() != HloOpcode::kTuple) {
618 return false;
619 }
620
621 auto while_body_param = while_body->parameter_instruction(0);
622 const HloInstruction::InstructionVector& root_operands =
623 while_body_root->operands();
624
625 // Find the loop invariant tuple elements with scalar constant init value and
626 // build a map from the tuple element index to the constant value. Limit this
627 // to scalar constant values because propagating array constants can regress
628 // performance by forcing us to copy constants.
629 tensorflow::gtl::FlatMap<int, const HloInstruction*> index_to_constant;
630 for (int i = 0; i < root_operands.size(); i++) {
631 HloInstruction* instr = root_operands[i];
632 if (instr->opcode() == HloOpcode::kGetTupleElement &&
633 instr->tuple_index() == i && instr->operand(0) == while_body_param &&
634 ShapeUtil::IsScalar(instr->shape())) {
635 auto tuple_element = while_init->operand(i);
636 if (tuple_element->IsConstant()) {
637 VLOG(3) << "Found loop invariant tuple element " << i << " "
638 << tuple_element->ToString();
639 index_to_constant[i] = tuple_element;
640 }
641 }
642 }
643
644 if (index_to_constant.empty()) {
645 return false;
646 }
647
648 // Replace the use of each constant tuple element in the loop_condition and
649 // loop_body with the corresponding constant value.
650 auto propagate_constant = [&](HloComputation* computation) -> StatusOr<bool> {
651 HloInstruction* param = computation->parameter_instruction(0);
652 bool changed = false;
653 for (auto instr : param->users()) {
654 // Since only a while-loop with a tuple result reaches here, we can safely
655 // assume that `param` is a tuple and the first operand of the
656 // GetTupleElement instruction is a use of `param`.
657 if (instr->opcode() == HloOpcode::kGetTupleElement) {
658 VLOG(3) << "tuple index " << instr->tuple_index() << " "
659 << instr->ToString();
660 auto iter = index_to_constant.find(instr->tuple_index());
661 if (iter != index_to_constant.end()) {
662 const HloInstruction* hlo_constant = (*iter).second;
663 VLOG(3) << "Replace use of " << instr->ToString() << " with "
664 << hlo_constant->ToString();
665 TF_RETURN_IF_ERROR(instr->ReplaceAllUsesWith(
666 computation->AddInstruction(hlo_constant->Clone())));
667 changed = true;
668 }
669 }
670 }
671 return changed;
672 };
673
674 TF_ASSIGN_OR_RETURN(bool changed_cond,
675 propagate_constant(while_op->while_condition()));
676 TF_ASSIGN_OR_RETURN(bool changed_body, propagate_constant(while_body));
677
678 return changed_cond || changed_body;
679}
680
681StatusOr<bool> WhileLoopSimplifier::Run(HloModule* module) {
682 XLA_VLOG_LINES(3,
683 "WhileLoopSimplifier::Run(), before:\n" + module->ToString());
684 bool changed = false;
685
686 // Gather all the while ops in our module. We do this ahead of time so we
687 // don't have to worry about mutating the lists of computations or
688 // instructions while we iterate.
689 std::vector<HloInstruction*> while_ops;
690 for (auto* comp : module->computations()) {
691 for (auto* instr : comp->instructions()) {
692 if (instr->opcode() == HloOpcode::kWhile) {
693 while_ops.push_back(instr);
694 }
695 }
696 }
697
698 for (HloInstruction* while_op : while_ops) {
699 // We can't remove while loops that contain send/recv nodes, because we rely
700 // on the particular loop structure around the node matching on the send and
701 // recv sides. Removing dead while params requires us to remove the loop
702 // and replace it with a new one, so we can't do that either.
703 if (ContainsSendOrRecv(while_op->while_body()) ||
704 ContainsSendOrRecv(while_op->while_condition())) {
705 VLOG(2) << "Not attempting to simplify while loop because it contains a "
706 "send/recv node: "
707 << while_op->ToShortString();
708 continue;
709 }
710
711 StatusOr<bool> result = TryPropagateConstant(while_op);
712 TF_RETURN_IF_ERROR(result.status());
713 changed |= result.ValueOrDie();
714
715 result = TryRemoveWhileLoop(while_op);
716 TF_RETURN_IF_ERROR(result.status());
717 if (result.ValueOrDie()) {
718 changed = true;
719 // Don't try to remove dead while params after successfully removing the
720 // while loop -- that would result in use-after-free nastiness.
721 continue;
722 }
723
724 result = TryRemoveDeadWhileParams(while_op);
725 TF_RETURN_IF_ERROR(result.status());
726 changed |= result.ValueOrDie();
727 }
728
729 XLA_VLOG_LINES(3,
730 "WhileLoopSimplifier::Run(), after:\n" + module->ToString());
731 return changed;
732}
