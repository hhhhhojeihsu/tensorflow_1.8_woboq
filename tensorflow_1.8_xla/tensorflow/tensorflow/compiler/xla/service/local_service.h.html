34namespace xla {
35
36// Service implementation that extends the XLA Service to leverage running
37// in the same process as the client.
38class LocalService : public Service {
39 public:
40 // Factory for creating a LocalService.
41 static StatusOr<std::unique_ptr<LocalService>> NewService(
42 const ServiceOptions& options);
43
44 // Builds an Executable with the given argument layouts and options. If
45 // result_layout is non-null, then the executable is compiled to produce a
46 // result of the given layout. If device_allocator is non-null, then the
47 // compiler may use it to allocate temp space on the device. The compiler is
48 // responsible for freeing any memory it allocates this way.
49 StatusOr<std::unique_ptr<Executable>> CompileExecutable(
50 const ComputationHandle& computation,
51 const tensorflow::gtl::ArraySlice<const Shape*> argument_layouts,
52 const ExecutableBuildOptions& options);
53
54 // Builds an Executable with the given XlaComputation, argument layouts and
55 // options. If result_layout is non-null, then the executable is compiled to
56 // produce a result of the given layout. If device_allocator is non-null,
57 // then the compiler may use it to allocate temp space on the device. The
58 // compiler is responsible for freeing any memory it allocates this way.
59 //
60 // TODO(b/74197823): This is a part of a NOT YET ready refactor.
61 StatusOr<std::unique_ptr<Executable>> CompileExecutable(
62 const XlaComputation& computation,
63 const tensorflow::gtl::ArraySlice<const Shape*> argument_layouts,
64 const ExecutableBuildOptions& build_options);
65
66 // Returns the device ordinal that corresponds to the given replica number.
67 //
68 // This returns an error if there is not a one-to-one correspondence of
69 // replicas to device ordinals, but is useful as a short term mechanism for
70 // the "easy" case where a single replica is a single device.
71 StatusOr<int> ReplicaNumberToDeviceOrdinal(int replica_number);
72
73 private:
74 explicit LocalService(const ServiceOptions& options,
75 std::unique_ptr<Backend> backend);
76 LocalService(const LocalService&) = delete;
77 void operator=(const LocalService&) = delete;
78};
79
