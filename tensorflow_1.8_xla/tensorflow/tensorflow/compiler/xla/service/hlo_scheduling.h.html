29namespace xla {
30
31// Returns the minimum memory required to compute the given module sequence,
32// assuming no fragmentation.
33StatusOr<int64> MinimumMemoryForSequence(
34 const SequentialHloOrdering::HloModuleSequence& module_sequence,
35 const LogicalBuffer::SizeFunction& size_function);
36
37// A memory scheduler computes an execution sequence for the HLO instructions in
38// 'computation' that minimizes peak memory, given a points-to analysis result
39// that describes buffer aliasing, together with a target-specific size function
40// that maps a tensor's logical size to its padded size.
41typedef std::function<StatusOr<std::vector<const HloInstruction*>>(
42 const HloComputation&, const TuplePointsToAnalysis&,
43 const LogicalBuffer::SizeFunction&)>
44 MemorySchedulerAlgorithm;
45
46// List scheduler
47StatusOr<std::vector<const HloInstruction*>> ListMemoryScheduler(
48 const HloComputation& computation,
49 const TuplePointsToAnalysis& points_to_analysis,
50 const LogicalBuffer::SizeFunction& size_function);
51
52// DFS-order scheduler
53StatusOr<std::vector<const HloInstruction*>> DFSMemoryScheduler(
54 const HloComputation& computation,
55 const TuplePointsToAnalysis& points_to_analysis,
56 const LogicalBuffer::SizeFunction& size_function);
57
58// The default scheduling algorithm. Runs both the list scheduler
59// and the DFS scheduler, and chooses whichever returns a lower min-memory,
60// not accounting for fragmentation.
61StatusOr<std::vector<const HloInstruction*>> DefaultMemoryScheduler(
62 const HloComputation& computation,
63 const TuplePointsToAnalysis& points_to_analysis,
64 const LogicalBuffer::SizeFunction& size_function);
65
66// Returns an HloModuleSequence which seeks to minimize the memory required for
67// the computation. size_function is the function returning the number of bytes
68// required for a LogicalBuffer.
69StatusOr<SequentialHloOrdering::HloModuleSequence>
70CreateMemoryMinimizingSequence(const HloModule& module,
71 const LogicalBuffer::SizeFunction& size_function,
72 const MemorySchedulerAlgorithm& algorithm = {});
73
74// Overload of above that computes the sequence for a single computation.
75StatusOr<std::vector<const HloInstruction*>> CreateMemoryMinimizingSequence(
76 const HloComputation& computation,
77 const LogicalBuffer::SizeFunction& size_function,
78 const MemorySchedulerAlgorithm& algorithm = {});
79
80} // namespace xla
