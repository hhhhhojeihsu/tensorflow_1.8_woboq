47// Converts between little and big endian, assuming elements in the array are 16
48// bits long.
49void ConvertEndianShort(char* bytes, int64 size) {
50 CHECK_EQ(size / 2, 0);
51 for (int64 i = 0; i < size; i += 2) {
52 std::swap(bytes[i], bytes[i + 1]);
53 }

63Literal::StrideConfig::StrideConfig(
64 const Shape& source_shape, const Shape& dest_shape,
65 tensorflow::gtl::ArraySlice<int64> dimensions)
66 : dimensions(dimensions),
67 base(dimensions.size(), 0),
68 step(dimensions.size(), 1) {
69 if (!dimensions.empty()) {
70 // Selects the shape with the largest minor dimension as the one upon
71 // which to run the tight stride loop.
72 if (dimensions[LayoutUtil::Minor(source_shape.layout(), 0)] >=
73 dimensions[LayoutUtil::Minor(dest_shape.layout(), 0)]) {
74 minor_dimension = LayoutUtil::Minor(source_shape.layout(), 0);
75 dest_stride = IndexUtil::GetDimensionStride(dest_shape, minor_dimension);
76 } else {
77 minor_dimension = LayoutUtil::Minor(dest_shape.layout(), 0);
78 source_stride =
79 IndexUtil::GetDimensionStride(source_shape, minor_dimension);
80 }
81 minor_loop_size = dimensions[minor_dimension];
82 step[minor_dimension] = minor_loop_size;
83 }
84}

89Literal::Literal(const Shape& shape, bool allocate_arrays)
90 : shape_(shape), pieces_(shape), owns_buffers_(true) {
91 CHECK(LayoutUtil::HasLayout(shape));
92 for (auto& pair : pieces_) {
93 const ShapeIndex& index = pair.first;
94 Piece& piece = pair.second;
95
96 piece.set_subshape(&ShapeUtil::GetSubshape(shape_, index));
97 const Shape& subshape = piece.subshape();
98 if (ShapeUtil::IsArray(subshape)) {
99 if (allocate_arrays) {
100 piece.set_buffer(new char[piece.size_bytes()]);
101 if (LayoutUtil::IsSparseArray(subshape)) {
102 piece.set_sparse_indices(new SparseIndexArray(
103 LayoutUtil::MaxSparseElements(subshape.layout()),
104 ShapeUtil::Rank(subshape)));
105 }
106 } else {
107 piece.set_buffer(nullptr);
108 }
109 }
110 }
111}

115void Literal::DeallocateBuffers() {
116 if (owns_buffers_) {
117 for (auto& pair : pieces_) {
118 Piece& piece = pair.second;
119 if (piece.buffer() != nullptr) {
120 delete[] piece.buffer();
121 delete piece.sparse_indices();
122 }
123 }
124 }

127Literal::Literal(Literal&& other) {
128 shape_ = std::move(other.shape_);
129 pieces_ = std::move(other.pieces_);
130 // We need to iterate through the pieces to set the subshape pointer
131 // properly. It must refer to subshapes within shape_.
132 for (auto& pair : pieces_) {
133 const ShapeIndex& index = pair.first;
134 Piece& piece = pair.second;
135 piece.set_subshape(&ShapeUtil::GetSubshape(shape_, index));
136 }
137 owns_buffers_ = other.owns_buffers_;
138
139 other.shape_ = ShapeUtil::MakeNil();
140 other.pieces_ = ShapeTree<Piece>(other.shape_);
141 other.piece({}).set_subshape(&other.shape_);

189template <typename NativeT>
190Status Literal::CopySliceFromInternal(
191 const Literal& src_literal, tensorflow::gtl::ArraySlice<int64> src_base,
192 tensorflow::gtl::ArraySlice<int64> dest_base,
193 tensorflow::gtl::ArraySlice<int64> copy_size) {
194 TF_RET_CHECK(ShapeUtil::Rank(src_literal.shape()) == src_base.size());
195 TF_RET_CHECK(ShapeUtil::Rank(shape()) == dest_base.size());
196
197 auto linear_index = [](const Shape& shape,
198 tensorflow::gtl::ArraySlice<int64> multi_index) {
199 return IndexUtil::MultidimensionalIndexToLinearIndex(shape, multi_index);
200 };
201
202 if (ShapeUtil::Rank(src_literal.shape()) == 0 ||
203 ShapeUtil::Rank(shape()) == 0) {
204 // If any of the two shapes are scalars, we can just call the StridedCopy()
205 // directly, and we know we will be copying only one value.
206 TF_RET_CHECK(copy_size.empty());
207 StridedCopy(data<NativeT>(), linear_index(shape(), dest_base), 0,
208 src_literal.data<NativeT>(),
209 linear_index(src_literal.shape(), src_base), 0, 1);
210 } else if (!ShapeUtil::HasZeroElements(shape()) &&
211 !ShapeUtil::HasZeroElements(src_literal.shape())) {
212 // Perform copy if neither src nor dest has dimensions with zero element,
213 // otherwise it's a no-op.
214 TF_RET_CHECK(src_base.size() == dest_base.size());
215 TF_RET_CHECK(src_base.size() == copy_size.size());
216
217 // Scan the source from minor, stepping in copy size blocks, then within
218 // the index enumaration functor, do a strided copy advancing source index
219 // by one (walking through the minor dimension), and destination index by
220 // proper stride size at the matching dimension.
221 DimensionVector src_indexes(src_base.size(), 0);
222 DimensionVector dest_indexes(dest_base.size(), 0);
223 Literal::StrideConfig stride_config(src_literal.shape(), shape(),
224 copy_size);
225
226 auto copy_proc = [&](tensorflow::gtl::ArraySlice<int64> indexes) {
227 // Map from multi-dimensional index, to source index.
228 std::transform(indexes.begin(), indexes.end(), src_base.begin(),
229 src_indexes.begin(), std::plus<int64>());
230 // Map from multi-dimensional index, to destination index.
231 std::transform(indexes.begin(), indexes.end(), dest_base.begin(),
232 dest_indexes.begin(), std::plus<int64>());
233
234 int64 src_index = linear_index(src_literal.shape(), src_indexes);
235 int64 dest_index = linear_index(shape(), dest_indexes);
236
237 // `this->` is needed to workaround MSVC bug: #16882
238 StridedCopy(this->data<NativeT>(), dest_index, stride_config.dest_stride,
239 src_literal.data<NativeT>(), src_index,
240 stride_config.source_stride, stride_config.minor_loop_size);
241 return true;
242 };
243
244 ShapeUtil::ForEachIndex(src_literal.shape(), stride_config.base,
245 stride_config.dimensions, stride_config.step,
246 copy_proc);
247 }
248 return Status::OK();
249}

251Status Literal::CopyElementFrom(const Literal& src_literal,
252 tensorflow::gtl::ArraySlice<int64> src_index,
253 tensorflow::gtl::ArraySlice<int64> dest_index) {
254 DCHECK_EQ(shape().element_type(), src_literal.shape().element_type());
255 const int64 src_linear_index = IndexUtil::MultidimensionalIndexToLinearIndex(
256 src_literal.shape(), src_index);
257 const int64 dest_linear_index =
258 IndexUtil::MultidimensionalIndexToLinearIndex(shape(), dest_index);
259 const int64 primitive_size =
260 ShapeUtil::ByteSizeOfPrimitiveType(shape().element_type());
261
262 char* dest_address =
263 static_cast<char*>(untyped_data()) + dest_linear_index * primitive_size;
264 const char* source_address =
265 static_cast<const char*>(src_literal.untyped_data()) +
266 src_linear_index * primitive_size;
267 if (dest_address != source_address) {
268 memcpy(dest_address, source_address, primitive_size);
269 }
270 return Status::OK();
271}

273std::vector<Literal> Literal::DecomposeTuple() {
274 CHECK(ShapeUtil::IsTuple(shape()));
275 std::vector<Literal> elements;
276 for (int i = 0; i < ShapeUtil::TupleElementCount(shape()); ++i) {
277 elements.push_back(Literal(ShapeUtil::GetSubshape(shape(), {i}),
278 /*allocate_arrays=*/false));
279 Literal& element = elements.back();
280 for (auto& pair : element.pieces_) {
281 const ShapeIndex& index = pair.first;
282 Piece& dest_piece = pair.second;
283 ShapeIndex src_index = {i};
284 for (int64 j : index) {
285 src_index.push_back(j);
286 }
287 Piece& src_piece = piece(src_index);
288
289 // Move the respective buffer and sparse indices over to the element
290 // Literal.
291 dest_piece.set_buffer(src_piece.buffer());
292 src_piece.set_buffer(nullptr);
293 dest_piece.set_sparse_indices(src_piece.sparse_indices());
294 src_piece.set_sparse_indices(nullptr);
295 }
296 }
297 // Set this literal to be nil-shaped.
298 *this = Literal();
299 return elements;
300}

302/* static */ Literal Literal::MoveIntoTuple(
303 tensorflow::gtl::MutableArraySlice<Literal> elements) {
304 std::vector<Shape> element_shapes;
305 for (const Literal& element : elements) {
306 element_shapes.push_back(element.shape());
307 }
308 Literal literal(ShapeUtil::MakeTupleShape(element_shapes),
309 /*allocate_arrays=*/false);
310 for (int i = 0; i < elements.size(); ++i) {
311 TF_CHECK_OK(
312 literal.MoveFrom(std::move(elements[i]), /*dest_shape_index=*/{i}));
313 }
314 return literal;

319// Copies the elements in 'src' to 'dest'. The shape and layout of the data in
320// the array slices are indicated by dest_shape and src_shape respectively.
321template <typename NativeT>
322void CopyElementsBetween(tensorflow::gtl::MutableArraySlice<NativeT> dest,
323 tensorflow::gtl::ArraySlice<NativeT> src,
324 const Shape& dest_shape, const Shape& src_shape) {
325 CHECK(ShapeUtil::Compatible(dest_shape, src_shape));
326 if (ShapeUtil::HasZeroElements(dest_shape)) {
327 return;
328 }
329 std::vector<int64> index(ShapeUtil::Rank(dest_shape));
330 do {
331 dest[IndexUtil::MultidimensionalIndexToLinearIndex(dest_shape, index)] =
332 src[IndexUtil::MultidimensionalIndexToLinearIndex(src_shape, index)];
333 } while (IndexUtil::BumpIndices(dest_shape, &index));
334}

338Status Literal::Piece::CopyFrom(const Literal::Piece& src) {
339 if (ShapeUtil::Equal(subshape(), src.subshape())) {
340 // If the layouts are equal it's faster just to memcpy.
341 memcpy(buffer(), src.buffer(), src.size_bytes());
342 } else {
343 TF_RET_CHECK(ShapeUtil::Compatible(src.subshape(), subshape()));
344 std::vector<int64> origin(ShapeUtil::Rank(subshape()), 0);
345 switch (subshape().element_type()) {
346#define COPY_ELEMENTS(XLA_T, NATIVE_T) \
347 case (XLA_T): \
348 CopyElementsBetween<NATIVE_T>(data<NATIVE_T>(), src.data<NATIVE_T>(), \
349 subshape(), src.subshape()); \
350 break;
351 COPY_ELEMENTS(U8, uint8);
352 COPY_ELEMENTS(U16, uint16);
353 COPY_ELEMENTS(U32, uint32);
354 COPY_ELEMENTS(U64, uint64);
355 COPY_ELEMENTS(S8, int8);
356 COPY_ELEMENTS(S16, int16);
357 COPY_ELEMENTS(S32, int32);
358 COPY_ELEMENTS(S64, int64);
359 COPY_ELEMENTS(F16, half);
360 COPY_ELEMENTS(BF16, bfloat16);
361 COPY_ELEMENTS(F32, float);
362 COPY_ELEMENTS(F64, double);
363 COPY_ELEMENTS(C64, complex64);
364 COPY_ELEMENTS(PRED, bool);
365#undef COPY_ELEMENTS
366 default:
367 return Unimplemented(
368 "Copying a Literal object with element type %s is not implemented.",
369 PrimitiveType_Name(subshape().element_type()).c_str());
370 }
371 }
372 return Status::OK();
373}

375Status Literal::CopyFrom(const Literal& src_literal,
376 const ShapeIndex& dest_shape_index,
377 const ShapeIndex& src_shape_index) {
378 const Shape& dest_subshape =
379 ShapeUtil::GetSubshape(shape(), dest_shape_index);
380 const Shape& src_subshape =
381 ShapeUtil::GetSubshape(src_literal.shape(), src_shape_index);
382 if (!ShapeUtil::Compatible(dest_subshape, src_subshape)) {
383 return InvalidArgument(
384 "Destination subshape incompatible with source subshape: %s vs %s",
385 ShapeUtil::HumanString(dest_subshape).c_str(),
386 ShapeUtil::HumanString(src_subshape).c_str());
387 }
388
389 for (auto& pair : pieces_) {
390 const ShapeIndex& index = pair.first;
391 Piece& piece = pair.second;
392 if (!ShapeUtil::IsArray(piece.subshape())) {
393 continue;
394 }
395
396 // Determine if this index is in the part of this literal that we want to
397 // copy over from src_literal.
398 bool in_subtree_to_copy = true;
399 for (int i = 0; i < dest_shape_index.size(); ++i) {
400 if (index[i] != dest_shape_index[i]) {
401 in_subtree_to_copy = false;
402 break;
403 }
404 }
405 if (!in_subtree_to_copy) {
406 continue;
407 }
408
409 // Construct the index of the corresponding piece in the source literal.
410 ShapeIndex src_piece_index = src_shape_index;
411 for (int64 i = dest_shape_index.size(); i < index.size(); ++i) {
412 src_piece_index.push_back(index[i]);
413 }
414
415 TF_RETURN_IF_ERROR(piece.CopyFrom(src_literal.piece(src_piece_index)));
416 }
417 return Status::OK();
418}

420Status Literal::MoveFrom(Literal&& src_literal,
421 const ShapeIndex& dest_shape_index) {
422 const Shape& dest_subshape =
423 ShapeUtil::GetSubshape(shape(), dest_shape_index);
424 if (!ShapeUtil::Equal(dest_subshape, src_literal.shape())) {
425 return InvalidArgument(
426 "Destination subshape not equal to source shape: %s vs %s",
427 ShapeUtil::HumanString(dest_subshape).c_str(),
428 ShapeUtil::HumanString(src_literal.shape()).c_str());
429 }
430
431 if (!(owns_buffers_ && src_literal.owns_buffers_)) {
432 return InvalidArgument(
433 "Source and destination literals must both own their buffers (ie, not "
434 "be views)");
435 }
436
437 for (auto& pair : src_literal.pieces_) {
438 const ShapeIndex& src_index = pair.first;
439 Piece& src_piece = pair.second;
440 if (!ShapeUtil::IsArray(src_piece.subshape())) {
441 continue;
442 }
443
444 ShapeIndex dest_index = dest_shape_index;
445 for (int64 i : src_index) {
446 dest_index.push_back(i);
447 }
448 Piece& dest_piece = piece(dest_index);
449 delete[] dest_piece.buffer();
450 dest_piece.set_buffer(src_piece.buffer());
451 delete dest_piece.sparse_indices();
452 dest_piece.set_sparse_indices(src_piece.sparse_indices());
453 }
454
455 src_literal.shape_ = ShapeUtil::MakeNil();
456 src_literal.pieces_ = ShapeTree<Piece>(src_literal.shape_);
457 src_literal.piece({}).set_subshape(&src_literal.shape_);
458 return Status::OK();
459}

462 tensorflow::gtl::ArraySlice<int64> src_base,

466 TF_RET_CHECK(ShapeUtil::IsArray(src_literal.shape()))

474 case U16:

761StatusOr<std::unique_ptr<Literal>> Literal::Reshape(
762 tensorflow::gtl::ArraySlice<int64> dimensions) const {
763 if (!ShapeUtil::IsArray(shape())) {
764 return InvalidArgument("Reshape does not support tuples.");
765 }
766 std::unique_ptr<Literal> output;
767 if (!LayoutUtil::IsMonotonicWithDim0Major(shape().layout())) {
768 output =
769 Relayout(LayoutUtil::GetDefaultLayoutForRank(ShapeUtil::Rank(shape())));
770 } else {
771 output = CloneToUnique();
772 }
773 // Because the layout is monotonic, we can simply reuse the same sequence of
774 // values without changing their order.
775 output->shape_ = ShapeUtil::MakeShape(shape().element_type(), dimensions);

777 int64 elements_before = ShapeUtil::ElementsIn(shape());
778 int64 elements_after = ShapeUtil::ElementsIn(output->shape());
779 if (elements_before != elements_after) {

781 "Shapes before and after Literal::Reshape have different numbers "
782 "of elements: %s vs %s.",

784 ShapeUtil::HumanString(output->shape()).c_str());
785 }

789std::unique_ptr<Literal> Literal::Transpose(
790 tensorflow::gtl::ArraySlice<int64> permutation) const {
791 CHECK(ShapeUtil::IsArray(shape())) << "Tuple is not supported for transpose";
792 CHECK(IsPermutation(permutation, ShapeUtil::Rank(shape())))
793 << "Given permutation is not a permutation of dimension numbers";
794 // To transpose the array, we just permute the dimensions and layout, and
795 // do a straight memory copy of the raw data set.
796 // This is considerably faster than iterating over every array element using
797 // the EachCell<>() and Set<>() APIs.
798 std::vector<int64> inverse_permutation = InversePermutation(permutation);
799 Shape permuted_shape =
800 ShapeUtil::PermuteDimensions(inverse_permutation, shape());
801 // Replace the layout with one affine to this shape, such that a
802 // transpose operation can be performed by leaving the flat values
803 // representation intact.
804 // For example, consider the shape F32[11,8]{1,0} under a {1,0} permutation.
805 // The shape with affine layout resulting from that operation will be
806 // F32[8,11]{0,1}, since it leaves the original most minor (the 8 sized), the
807 // most minor.
808 //
809 // Essentially, given MinMaj(Di) the position of the Di dimension within the
810 // minor to major vector, and given T(Di) the index that the original Di
811 // dimension has within the transposed array, a layout is affine if
812 // MinMaj(Di) == TMinMaj(T(Di)), with TMinMaj() being the minor to major
813 // vector of the affine layout.
814 CHECK(LayoutUtil::IsDenseArray(permuted_shape));
815 Layout* layout = permuted_shape.mutable_layout();
816 layout->clear_minor_to_major();
817 for (auto index : LayoutUtil::MinorToMajor(shape())) {
818 layout->add_minor_to_major(inverse_permutation[index]);
819 }
820 std::unique_ptr<Literal> new_literal = CreateFromShape(permuted_shape);
821 DCHECK_GE(ShapeUtil::ByteSizeOf(new_literal->shape()),
822 ShapeUtil::ByteSizeOf(shape()));
823 std::memcpy(new_literal->root_piece().buffer(), root_piece().buffer(),
824 root_piece().size_bytes());
825 return new_literal;
826}

828std::unique_ptr<Literal> Literal::Slice(
829 tensorflow::gtl::ArraySlice<int64> start_indices,
830 tensorflow::gtl::ArraySlice<int64> limit_indices) const {
831 CHECK(ShapeUtil::IsArray(shape())) << "tuple is not supported for slice";
832
833 DimensionVector result_dimensions;
834 for (int64 dnum = 0; dnum < ShapeUtil::Rank(shape()); ++dnum) {
835 CHECK_GE(start_indices[dnum], 0);
836 CHECK_LE(limit_indices[dnum], shape().dimensions(dnum))
837 << "dnum = " << dnum;
838 int64 dimension = limit_indices[dnum] - start_indices[dnum];
839 CHECK_GE(dimension, 0) << "dnum = " << dnum;
840 result_dimensions.push_back(dimension);
841 }
842 const auto result_shape =
843 ShapeUtil::MakeShapeWithLayout(shape().element_type(), result_dimensions,
844 LayoutUtil::MinorToMajor(shape()));
845
846 auto result_literal = MakeUnique<Literal>(result_shape);
847
848 DimensionVector new_indices(ShapeUtil::Rank(result_shape));
849 switch (result_shape.element_type()) {
850 case F32:
851 result_literal->EachCell<float>(
852 [&](tensorflow::gtl::ArraySlice<int64> indices, float /*value*/) {
853 for (int64 i = 0; i < ShapeUtil::Rank(result_shape); ++i) {
854 new_indices[i] = indices[i] + start_indices[i];
855 }
856 float value = Get<float>(new_indices);
857 result_literal->Set<float>(indices, value);
858 });
859 return result_literal;
860 case C64:
861 result_literal->EachCell<complex64>(
862 [&](tensorflow::gtl::ArraySlice<int64> indices, complex64 /*value*/) {
863 for (int64 i = 0; i < ShapeUtil::Rank(result_shape); ++i) {
864 new_indices[i] = indices[i] + start_indices[i];
865 }
866 complex64 value = Get<complex64>(new_indices);
867 result_literal->Set<complex64>(indices, value);
868 });
869 return result_literal;
870 case S32:
871 result_literal->EachCell<int32>(
872 [&](tensorflow::gtl::ArraySlice<int64> indices, int32 /*value*/) {
873 for (int64 i = 0; i < ShapeUtil::Rank(result_shape); ++i) {
874 new_indices[i] = indices[i] + start_indices[i];
875 }
876 int32 value = Get<int32>(new_indices);
877 result_literal->Set<int32>(indices, value);
878 });
879 return result_literal;
880 case U32:
881 result_literal->EachCell<uint32>(
882 [&](tensorflow::gtl::ArraySlice<int64> indices, uint32 /*value*/) {
883 for (int64 i = 0; i < ShapeUtil::Rank(result_shape); ++i) {
884 new_indices[i] = indices[i] + start_indices[i];
885 }
886 uint32 value = Get<uint32>(new_indices);
887 result_literal->Set<uint32>(indices, value);
888 });
889 return result_literal;
890 default:
891 LOG(FATAL) << "not yet implemented: "
892 << PrimitiveType_Name(result_shape.element_type());
893 }
894}

1175 bool print_layout, std::vector<string>* pieces) {

1177
1178 auto shape_to_string = [print_layout](const Shape& shape) {
1179 if (print_layout) {
1180 return ShapeUtil::HumanStringWithLayout(shape);
1181 } else {
1182 return ShapeUtil::HumanString(shape);
1183 }
1184 };
1185
1186 // TODO(b/32894291): refactor this code to reduce code duplication.
1187 if (ShapeUtil::IsTuple(subshape)) {
1188 pieces->push_back(shape_to_string(subshape));
1189 pieces->push_back(" (\n");
1190 std::vector<string> tuple_pieces;
1191 for (int i = 0; i < ShapeUtil::TupleElementCount(subshape); ++i) {
1192 ShapeIndex element_index = shape_index;
1193 element_index.push_back(i);
1194 std::vector<string> element_pieces;
1195 ToStringHelper(literal, element_index, print_layout, &element_pieces);
1196 tuple_pieces.push_back(tensorflow::str_util::Join(element_pieces, ""));
1197 }
1198 pieces->push_back(tensorflow::str_util::Join(tuple_pieces, ",\n"));
1199 pieces->push_back("\n)");
1200 return;
1201 }
1202
1203 if (LayoutUtil::IsSparseArray(subshape)) {
1204 pieces->push_back(shape_to_string(subshape));
1205 pieces->push_back("{");
1206 int64 rank = ShapeUtil::Rank(subshape);
1207 int64 num_elements = literal.sparse_element_count();
1208 for (int64 i = 0; i < num_elements; ++i) {
1209 if (i > 0) {
1210 pieces->push_back(", ");
1211 }
1212 if (rank == 1) {
1213 pieces->push_back(StrCat(literal.GetSparseIndex(i)[0]));
1214 pieces->push_back(": ");
1215 } else {
1216 pieces->push_back("[");
1217 pieces->push_back(
1218 tensorflow::str_util::Join(literal.GetSparseIndex(i), ", "));
1219 pieces->push_back("]: ");
1220 }
1221 pieces->push_back(literal.GetSparseElementAsString(i));
1222 }
1223 pieces->push_back("}");
1224 return;
1225 }
1226
1227 CHECK(LayoutUtil::IsDenseArray(subshape));
1228
1229 auto element_to_string =
1230 [&](tensorflow::gtl::ArraySlice<int64> indices) -> string {
1231 PrimitiveType element_type = subshape.element_type();
1232 if (element_type == PRED) {
1233 // We display predicates in a densely packed form.
1234 return literal.Get<bool>(indices, shape_index) ? "1" : "0";
1235 }
1236 return ((!indices.empty() && indices.back() > 0) ? ", " : "") +
1237 literal.GetAsString(indices, shape_index);
1238 };
1239
1240 if (ShapeUtil::Rank(subshape) == 0) {
1241 pieces->push_back(literal.GetAsString({}, shape_index));
1242 } else if (ShapeUtil::Rank(subshape) == 1) {
1243 pieces->push_back("{");
1244 for (int64 i0 = 0; i0 < subshape.dimensions(0); ++i0) {
1245 pieces->push_back(element_to_string({i0}));
1246 }
1247 pieces->push_back("}");
1248 } else if (ShapeUtil::Rank(subshape) == 2) {
1249 pieces->push_back(shape_to_string(subshape));
1250 pieces->push_back(" {\n");
1251 for (int64 i0 = 0; i0 < subshape.dimensions(0); ++i0) {
1252 pieces->push_back(" { ");
1253 for (int64 i1 = 0; i1 < subshape.dimensions(1); ++i1) {
1254 pieces->push_back(element_to_string({i0, i1}));
1255 }
1256 pieces->push_back(" ");
1257 pieces->push_back(i0 == subshape.dimensions(0) - 1 ? "}\n" : "},\n");
1258 }
1259 pieces->push_back("}");

1388template <typename NativeSrcT, typename NativeDestT, typename ConverterType>

1393 src_literal.shape(),
1394 primitive_util::NativeToPrimitiveType<NativeDestT>()));

1398

1400 dest_data[i] = converter(src_data[i]);

1508 const Literal& literal, PrimitiveType primitive_dest_type, bool bitcast) {

1519 CONVERT_IF_DEST_TYPE_MATCHES(S8)
1520 CONVERT_IF_DEST_TYPE_MATCHES(S32)
1521 CONVERT_IF_DEST_TYPE_MATCHES(S64)
1522 CONVERT_IF_DEST_TYPE_MATCHES(U8)

1527 CONVERT_IF_DEST_TYPE_MATCHES(F64)
1528 CONVERT_IF_DEST_TYPE_MATCHES(BF16)

1531 default:
1532 return Unimplemented(

1534 (bitcast ? "Bitcast converting" : "Converting"),
1535 PrimitiveType_Name(literal.shape().element_type()).c_str(),

1876 return Get<double>(indices) == 0.0;
1877 case C64:
1878 return Get<complex64>(indices) == complex64(0.0f, 0.0f);
1879 case F16:
1880 return Get<half>(indices) == static_cast<half>(0.0f);
1881 case BF16:
1882 return Get<bfloat16>(indices) == static_cast<bfloat16>(0.0f);
1883 case PRED:
1884 return Get<bool>(indices) == false;
1885 default:
1886 LOG(FATAL) << "Input literal must be an array.";
1887 }

1900void Literal::Piece::WriteToProto(LiteralProto* proto) const {
1901 *proto->mutable_shape() = subshape();
1902 switch (subshape().element_type()) {
1903 case PRED:
1904 CopyToRepeatedField(proto->mutable_preds(), data<bool>());
1905 break;
1906 case U8:
1907 proto->set_u8s(static_cast<const unsigned char*>(data<uint8>().data()),
1908 element_count());
1909 break;
1910 case U32:
1911 CopyToRepeatedField(proto->mutable_u32s(), data<uint32>());
1912 break;
1913 case U64:
1914 CopyToRepeatedField(proto->mutable_u64s(), data<uint64>());
1915 break;
1916 case S32:
1917 CopyToRepeatedField(proto->mutable_s32s(), data<int32>());
1918 break;
1919 case S64:
1920 CopyToRepeatedField(proto->mutable_s64s(), data<int64>());
1921 break;
1922 case F16:
1923 *proto->mutable_f16s() = string(
1924 reinterpret_cast<const char*>(data<half>().data()), size_bytes());
1925 if (!kLittleEndian) {
1926 ConvertEndianShort(const_cast<char*>(proto->mutable_f16s()->data()),
1927 proto->f16s().size());
1928 }
1929 break;
1930 case BF16:
1931 *proto->mutable_bf16s() = string(
1932 reinterpret_cast<const char*>(data<bfloat16>().data()), size_bytes());
1933 if (!kLittleEndian) {
1934 ConvertEndianShort(const_cast<char*>(proto->mutable_bf16s()->data()),
1935 proto->bf16s().size());
1936 }
1937 break;
1938 case F32:
1939 CopyToRepeatedField(proto->mutable_f32s(), data<float>());
1940 break;
1941 case F64:
1942 CopyToRepeatedField(proto->mutable_f64s(), data<double>());
1943 break;
1944 case C64:
1945 for (complex64 value : data<complex64>()) {
1946 proto->add_c64s(value.real());
1947 proto->add_c64s(value.imag());
1948 }
1949 break;
1950 case TUPLE:
1951 // Nothing to do but assign the shape which is done above.
1952 return;
1953 default:
1954 LOG(FATAL) << "Unhandled primitive type " << subshape().element_type();
1955 }

1984Status Literal::Piece::CopyFromProto(const LiteralProto& proto) {
1985 // These conditions should have been checked in Literal::CreateFromProto.
1986 TF_RET_CHECK(proto.has_shape());
1987 TF_RET_CHECK(LayoutUtil::HasLayout(proto.shape()));
1988 TF_RET_CHECK(ShapeUtil::Equal(proto.shape(), subshape()));
1989
1990 switch (subshape().element_type()) {
1991 case PRED:
1992 TF_RETURN_IF_ERROR(CopyFromRepeatedField(data<bool>(), proto.preds()));
1993 break;
1994 case U8: {
1995 auto u8_data = data<uint8>();
1996 TF_RET_CHECK(proto.u8s().size() == u8_data.size());
1997 std::copy(proto.u8s().begin(), proto.u8s().end(), u8_data.begin());
1998 } break;
1999 case S32:
2000 TF_RETURN_IF_ERROR(CopyFromRepeatedField(data<int32>(), proto.s32s()));
2001 break;
2002 case S64:
2003 TF_RETURN_IF_ERROR(CopyFromRepeatedField(data<int64>(), proto.s64s()));
2004 break;
2005 case U32:
2006 TF_RETURN_IF_ERROR(CopyFromRepeatedField(data<uint32>(), proto.u32s()));
2007 break;
2008 case U64:
2009 TF_RETURN_IF_ERROR(CopyFromRepeatedField(data<uint64>(), proto.u64s()));
2010 break;
2011 case F16: {
2012 const string& s(proto.f16s());
2013 TF_RET_CHECK(data<half>().size() * sizeof(half) == s.size());
2014 memcpy(untyped_data(), s.data(), s.size());
2015 if (!kLittleEndian) {
2016 ConvertEndianShort(reinterpret_cast<char*>(untyped_data()), s.size());
2017 }
2018 } break;
2019
2020 case BF16: {
2021 const string& s(proto.bf16s());
2022 TF_RET_CHECK(data<bfloat16>().size() * sizeof(bfloat16) == s.size());
2023 memcpy(untyped_data(), s.data(), s.size());
2024 if (!kLittleEndian) {
2025 ConvertEndianShort(reinterpret_cast<char*>(untyped_data()), s.size());
2026 }
2027 } break;
2028 case F32:
2029 TF_RETURN_IF_ERROR(CopyFromRepeatedField(data<float>(), proto.f32s()));
2030 break;
2031 case F64:
2032 TF_RETURN_IF_ERROR(CopyFromRepeatedField(data<double>(), proto.f64s()));
2033 break;
2034 case C64: {
2035 auto complex_data = data<complex64>();
2036 TF_RET_CHECK(proto.c64s_size() == complex_data.size() * 2);
2037 for (int64 i = 0; i < complex_data.size(); ++i) {
2038 complex_data[i] = complex64{proto.c64s(i * 2), proto.c64s(i * 2 + 1)};
2039 }
2040 } break;
2041 case TUPLE:
2042 LOG(FATAL) << "Should not be called on tuple shapes: "
2043 << ShapeUtil::HumanString(subshape());
2044 break;
2045 default:
2046 LOG(FATAL) << "Unhandled primitive type " << subshape().element_type();
2047 }
2048 return Status::OK();
2049}

2051LiteralProto Literal::ToProto() const {
2052 LiteralProto proto;
2053 for (const auto& pair : pieces_) {
2054 const ShapeIndex& index = pair.first;
2055 const Piece& piece = pair.second;
2056
2057 LiteralProto* proto_piece = &proto;
2058 for (int64 i : index) {
2059 while (proto_piece->tuple_literals_size() <= i) {
2060 proto_piece->add_tuple_literals();
2061 }
2062 proto_piece = proto_piece->mutable_tuple_literals(i);
2063 }
2064 piece.WriteToProto(proto_piece);
2065 }
2066
2067 if (LayoutUtil::IsSparseArray(shape())) {
2068 CopyToRepeatedField(proto.mutable_sparse_indices(),
2069 sparse_indices()->data());
2070 }
2071
2072 return proto;
2073}

2075/* static */
2076StatusOr<std::unique_ptr<Literal>> Literal::CreateFromProto(
2077 const LiteralProto& proto) {
2078 if (!proto.has_shape()) {
2079 return InvalidArgument("LiteralProto has no shape");
2080 }
2081 if (!LayoutUtil::HasLayout(proto.shape())) {
2082 return InvalidArgument("LiteralProto has no layout");
2083 }
2084
2085 auto literal = MakeUnique<Literal>(proto.shape());
2086
2087 for (auto& pair : literal->pieces_) {
2088 const ShapeIndex& index = pair.first;
2089 Piece& piece = pair.second;
2090 const LiteralProto* proto_element = &proto;
2091 for (int64 i : index) {
2092 TF_RET_CHECK(i < proto_element->tuple_literals_size());
2093 proto_element = &proto_element->tuple_literals(i);
2094 }
2095
2096 if (ShapeUtil::IsTuple(piece.subshape())) {
2097 if (proto_element->tuple_literals_size() !=
2098 ShapeUtil::TupleElementCount(piece.subshape())) {
2099 return InvalidArgument(
2100 "Expected %lld tuple elements in LiteralProto, has %d",
2101 ShapeUtil::TupleElementCount(piece.subshape()),
2102 proto_element->tuple_literals_size());
2103 }
2104 continue;
2105 }
2106
2107 TF_RET_CHECK(ShapeUtil::IsArray(piece.subshape()));
2108 TF_RETURN_IF_ERROR(piece.CopyFromProto(*proto_element));
2109 }
2110 return std::move(literal);
