69// Recursive helper for comparing the equality of two shapes. Returns true if
70// the shapes are the same. If compare_layouts is true, then layouts must also
71// match.
72bool CompareShapes(const Shape& lhs, const Shape& rhs, bool compare_layouts) {
73 if (ShapeUtil::IsTuple(lhs) || ShapeUtil::IsTuple(rhs)) {
74 return ShapeUtil::IsTuple(lhs) && ShapeUtil::IsTuple(rhs) &&
75 ContainersEqual(lhs.tuple_shapes(), rhs.tuple_shapes(),
76 [=](const Shape& l, const Shape& r) {
77 return CompareShapes(l, r, compare_layouts);
78 });
79 } else if (ShapeUtil::IsOpaque(lhs) || ShapeUtil::IsOpaque(rhs)) {
80 return ShapeUtil::IsOpaque(lhs) && ShapeUtil::IsOpaque(rhs);
81 }
82
83 if (compare_layouts) {
84 if (lhs.layout().format() != rhs.layout().format()) {
85 return false;
86 }
87 if (LayoutUtil::IsDenseArray(lhs)) {
88 if (!ContainersEqual(LayoutUtil::MinorToMajor(lhs),
89 LayoutUtil::MinorToMajor(rhs))) {
90 VLOG(3) << "CompareShapes: lhs layout != rhs layout";
91 return false;
92 }
93 if (!ContainersEqual(lhs.layout().padded_dimensions(),
94 rhs.layout().padded_dimensions())) {
95 VLOG(3)
96 << "CompareShapes: lhs padded_dimensions != rhs padded_dimensions";
97 return false;
98 }
99 if (lhs.layout().padding_value() != rhs.layout().padding_value()) {
100 VLOG(3) << "CompareShapes: lhs padding value != rhs padding_value";
101 return false;
102 }
103 }
104 }
105
106 if (!ShapeUtil::SameDimensions(lhs, rhs)) {
107 VLOG(3) << "CompareShapes: lhs dimensions != rhs dimensions";
108 return false;
109 }
110 if (!ShapeUtil::SameElementType(lhs, rhs)) {
111 VLOG(3) << "CompareShapes: lhs element type != rhs element type";
112 return false;
113 }
114 return true;
115}

117// Constructs and returns the new shape with the given minor_to_major order in
118// its Layout.
119StatusOr<Shape> MakeShapeWithLayoutInternal(
120 PrimitiveType element_type, tensorflow::gtl::ArraySlice<int64> dimensions,
121 tensorflow::gtl::ArraySlice<int64> minor_to_major) {
122 if (dimensions.size() != minor_to_major.size()) {
123 return InvalidArgument("Dimensions size is %ld, but layout size is %ld.",
124 dimensions.size(), minor_to_major.size());
125 }
126 if (element_type == OPAQUE || element_type == TUPLE) {
127 return InvalidArgument("Unsupported element type: %s",
128 PrimitiveType_Name(element_type).c_str());
129 }
130 Shape shape = ShapeUtil::MakeShape(element_type, dimensions);
131 auto min2maj = shape.mutable_layout()->mutable_minor_to_major();
132 min2maj->Clear();
133 for (int64 value : minor_to_major) {
134 min2maj->Add(value);
135 }
136 if (!shape.has_layout()) {
137 return InvalidArgument("Shape has no layout.");
138 }
139 TF_RETURN_IF_ERROR(ShapeUtil::ValidateShape(shape));
140 return shape;
141}

145/* static */ bool ShapeUtil::Equal(const Shape& lhs, const Shape& rhs) {
146 bool equal = CompareShapes(lhs, rhs, /*compare_layouts=*/true);
147 if (!equal && VLOG_IS_ON(3)) {
148 VLOG(3) << "ShapeUtil::Equal differ: lhs = " << lhs.ShortDebugString()
149 << ", rhs = " << rhs.ShortDebugString();
150 }
151
152 return equal;
153}

156 CHECK(!ShapeUtil::IsTuple(shape))

161/* static */ int64 ShapeUtil::TrueRank(const Shape& shape) {
162 int64 accum = 0;
163 for (int64 dimension : shape.dimensions()) {
164 // We do not count zero dimensions.
165 if (dimension != 1) {
166 accum += 1;
167 }
168 }
169 return accum;

408// Class to memoize the computation of
409// tensorflow::str_util::Lowercase(PrimitiveType_Name(p))
410// for all PrimitiveType values "p"
411class PrimitiveTypeNameGenerator {
412 public:
413 PrimitiveTypeNameGenerator() {
414 for (int i = 0; i < PrimitiveType_ARRAYSIZE; i++) {
415 if (PrimitiveType_IsValid(i)) {
416 lowercase_name_[i] = tensorflow::str_util::Lowercase(
417 PrimitiveType_Name(static_cast<PrimitiveType>(i)));
418 }
419 }
420 }
421 const string& LowercaseName(PrimitiveType t) {
422 return lowercase_name_[static_cast<int>(t)];
423 }
424
425 private:
426 string lowercase_name_[PrimitiveType_ARRAYSIZE];
427};

429const string& LowercasePrimitiveTypeName(PrimitiveType s) {
430 static PrimitiveTypeNameGenerator* gen = new PrimitiveTypeNameGenerator();
431 return gen->LowercaseName(s);
432}

434StatusOr<PrimitiveType> StringToPrimitiveType(const string& name) {
435 static std::unordered_map<string, PrimitiveType>* name_to_type = [] {
436 static auto* map = new std::unordered_map<string, PrimitiveType>;
437 for (int i = 0; i < PrimitiveType_ARRAYSIZE; i++) {
438 if (PrimitiveType_IsValid(i)) {
439 auto value = static_cast<PrimitiveType>(i);
440 (*map)[LowercasePrimitiveTypeName(value)] = value;
441 }
442 }
443 return map;
444 }();
445 auto found = name_to_type->find(name);
446 if (found == name_to_type->end()) {
447 return InvalidArgument("Invalid element type string: \"%s\".",
448 name.c_str());
449 }
450 return found->second;
451}

455/* static */ string ShapeUtil::HumanStringWithLayout(const Shape& shape) {
456 if (IsTuple(shape)) {
457 string text = "(";
458 const char* prefix = "";
459 for (const Shape& elem_shape : shape.tuple_shapes()) {
460 tensorflow::strings::StrAppend(&text, prefix,
461 HumanStringWithLayout(elem_shape));
462 prefix = ", ";
463 }
464 text += ")";
465 return text;
466 } else {
467 string result = tensorflow::strings::StrCat(
468 LowercasePrimitiveTypeName(shape.element_type()), "[");
469 for (int i = 0; i < shape.dimensions().size(); i++) {
470 tensorflow::strings::StrAppend(&result, (i > 0) ? "," : "",
471 shape.dimensions(i));
472 }
473 result += "]";
474 if (!IsScalar(shape) && !IsOpaque(shape)) {
475 if (LayoutUtil::HasLayout(shape)) {
476 tensorflow::strings::StrAppend(&result,
477 LayoutUtil::HumanString(shape.layout()));
478 }
479 }
480 return result;
481 }

500// Parses shapes with simple recursive descent structure -- consumes from the
501// front of s and passes that view recursively as required.
502StatusOr<Shape> ParseShapeStringInternal(tensorflow::StringPiece* s) {
503 tensorflow::str_util::RemoveLeadingWhitespace(s);
504
505 if (tensorflow::str_util::ConsumePrefix(s, "(")) { // Tuple.
506 std::vector<Shape> shapes;
507 bool must_end = false;
508 while (true) {
509 if (tensorflow::str_util::ConsumePrefix(s, ")")) {
510 break;
511 } else if (must_end) {
512 return InvalidArgument("Expected end of tuple; got: \"%s\"",
513 s->ToString().c_str());
514 }
515 shapes.emplace_back();
516 TF_ASSIGN_OR_RETURN(shapes.back(), ParseShapeStringInternal(s));
517 tensorflow::str_util::RemoveLeadingWhitespace(s);
518 must_end = !tensorflow::str_util::ConsumePrefix(s, ",");
519 }
520 return ShapeUtil::MakeTupleShape(shapes);
521 }
522
523 string element_type_string;
524 string dimensions_string;
525 string format_string;
526 string layout_string;
527 // tensorflow::StringPiece is not compatible with internal RE2 StringPiece, so
528 // we convert in to the RE2-consumable type and then consume the corresponding
529 // amount from our StringPiece type.
530 tensorflow::RegexpStringPiece s_consumable(s->data(), s->size());
531 if (RE2::Consume(
532 &s_consumable,
533 "^(\\w*\\d*)\\[([\\d,]*)\\](?:\\s*(dense|sparse)?\\s*{([\\d,]+)})?",
534 &element_type_string, &dimensions_string, &format_string,
535 &layout_string)) {
536 size_t consumed = s->size() - s_consumable.size();
537 s->remove_prefix(consumed);
538 auto string_to_int64 = [&s](const string& input) -> StatusOr<int64> {
539 int64 element;
540 if (!tensorflow::strings::safe_strto64(input.c_str(), &element)) {
541 return InvalidArgument(
542 "Invalid s64 value in parsed shape string: \"%s\" in \"%s\"",
543 input.c_str(), s->ToString().c_str());
544 }
545 return element;
546 };
547
548 auto comma_list_to_int64s =
549 [&s,
550 string_to_int64](const string& input) -> StatusOr<std::vector<int64>> {
551 std::vector<int64> results;
552 for (const string& piece : tensorflow::str_util::Split(input, ',')) {
553 TF_ASSIGN_OR_RETURN(int64 element, string_to_int64(piece));
554 results.push_back(element);
555 }
556 return results;
557 };
558
559 // Extract the dimensions.
560 TF_ASSIGN_OR_RETURN(std::vector<int64> dimensions,
561 comma_list_to_int64s(dimensions_string));
562
563 // Extract the primitive element type.
564 TF_ASSIGN_OR_RETURN(const PrimitiveType primitive_type,
565 StringToPrimitiveType(element_type_string));
566 if (primitive_type == PRIMITIVE_TYPE_INVALID || primitive_type == TUPLE ||
567 primitive_type == OPAQUE) {
568 return InvalidArgument("Invalid element type string: \"%s\".",
569 element_type_string.c_str());
570 }
571
572 Shape result;
573 if (format_string.empty() && layout_string.empty()) {
574 // Create a shape without a layout set.
575 result = ShapeUtil::MakeShape(primitive_type, dimensions);
576 } else if (format_string == "sparse") {
577 TF_ASSIGN_OR_RETURN(int64 max_elements, string_to_int64(layout_string));
578 result = ShapeUtil::MakeShapeWithSparseLayout(primitive_type, dimensions,
579 max_elements);
580 } else if (format_string.empty() || format_string == "dense") {
581 // Extract the layout minor-to-major and set it.
582 TF_ASSIGN_OR_RETURN(std::vector<int64> min2maj,
583 comma_list_to_int64s(layout_string));
584 TF_ASSIGN_OR_RETURN(result, MakeShapeWithLayoutInternal(
585 primitive_type, dimensions, min2maj));
586 } else {
587 // This should not be reached.
588 LOG(FATAL) << "Unhandled condition when parsing shape; format: \""
589 << format_string << "\", layout: \"" << layout_string << "\"";
590 }
591 TF_RETURN_IF_ERROR(ShapeUtil::ValidateShape(result));
592 return std::move(result);
593 }
594
595 return InvalidArgument("Invalid shape string to parse: \"%s\"",
596 s->ToString().c_str());
597}
598} // namespace

617/* static */ bool ShapeUtil::Compatible(const Shape& lhs, const Shape& rhs) {
618 if (lhs.element_type() == TUPLE) {
619 return rhs.element_type() == TUPLE &&
620 ContainersEqual(lhs.tuple_shapes(), rhs.tuple_shapes(), Compatible);
621 }
622 if (lhs.element_type() == OPAQUE) {
623 return rhs.element_type() == OPAQUE;
624 }
625 return SameElementType(lhs, rhs) && SameDimensions(lhs, rhs);
626}
627
628/* static */ bool ShapeUtil::CompatibleIgnoringElementType(const Shape& lhs,
629 const Shape& rhs) {
630 if (lhs.element_type() == TUPLE) {
631 return rhs.element_type() == TUPLE &&
632 ContainersEqual(lhs.tuple_shapes(), rhs.tuple_shapes(),
633 CompatibleIgnoringElementType);
634 }
635 if (lhs.element_type() == OPAQUE) {
636 return rhs.element_type() == OPAQUE;
637 }
638 return ShapeUtil::IsArray(rhs) && SameDimensions(lhs, rhs);
639}
640
641/* static */ bool ShapeUtil::CompatibleIgnoringFpPrecision(const Shape& lhs,
642 const Shape& rhs) {
643 if (lhs.element_type() == TUPLE) {
644 return rhs.element_type() == TUPLE &&
645 ContainersEqual(lhs.tuple_shapes(), rhs.tuple_shapes(),
646 CompatibleIgnoringFpPrecision);
647 }
648 if (lhs.element_type() == OPAQUE) {
649 return rhs.element_type() == OPAQUE;
650 }
651 if (SameElementTypeIgnoringFpPrecision(lhs, rhs)) {
652 return CompatibleIgnoringElementType(lhs, rhs);
653 }
654 return false;
655}

671/* static */ int64 ShapeUtil::ByteSizeOfPrimitiveType(
672 PrimitiveType primitive_type) {
673 switch (primitive_type) {
674 case PRED:
675 return sizeof(int8);
676 case TUPLE:
677 LOG(FATAL) << "tuples have no definitive size";
678 case OPAQUE:
679 LOG(FATAL) << "opaque have no definitive size";
680 case S8:
681 return sizeof(int8);
682 case S16:
683 return sizeof(int16);
684 case S32:
685 return sizeof(int32);
686 case S64:
687 return sizeof(int64);
688 case U8:
689 return sizeof(uint8);
690 case U16:
691 return sizeof(uint16);
692 case U32:
693 return sizeof(uint32);
694 case U64:
695 return sizeof(uint64);
696 case BF16:
697 return sizeof(float) / 2;
698 case F16:
699 return sizeof(float) / 2;
700 case F32:
701 return sizeof(float);
702 case F64:
703 return sizeof(double);
704 case C64:
705 return sizeof(complex64);
706 default:
707 LOG(FATAL) << "Unhandled primitive type " << primitive_type;
708 }

853/* static */ Shape ShapeUtil::StripDegenerateDimensions(const Shape& shape) {
854 std::vector<int64> dimension_sizes;
855 std::vector<int64> degenerate_dimensions;
856 for (int64 i = 0; i < shape.dimensions_size(); ++i) {
857 if (shape.dimensions(i) == 1) {
858 degenerate_dimensions.push_back(i);
859 } else {
860 dimension_sizes.push_back(shape.dimensions(i));
861 }
862 }
863
864 // Construct minor_to_major of stripped shape. The order of the non-degenerate
865 // dimensions should be preserved from the original shape. First, create
866 // vector of the non-degenerate dimensions from the original minor_to_major
867 // array.
868 std::vector<int64> minor_to_major;
869 for (int64 i : shape.layout().minor_to_major()) {
870 if (std::find(degenerate_dimensions.begin(), degenerate_dimensions.end(),
871 i) == degenerate_dimensions.end()) {
872 minor_to_major.push_back(i);
873 }
874 }
875
876 // The dimensions in minor_to_major need to be renumbered to account for the
877 // degenerate dimensions which have removed. Decrement each dimension number
878 // once for each degenerate dimension which has a smaller number.
879 for (int i = 0; i < minor_to_major.size(); ++i) {
880 int adjustment = 0;
881 for (int64 dim : degenerate_dimensions) {
882 if (minor_to_major[i] > dim) {
883 adjustment++;
884 }
885 }
886 minor_to_major[i] -= adjustment;
887 }
888
889 {
890 std::vector<int64> dims(minor_to_major.size());
891 std::iota(dims.begin(), dims.end(), 0);
892 DCHECK(minor_to_major.size() == dims.size() &&
893 std::is_permutation(minor_to_major.begin(), minor_to_major.end(),
894 dims.begin()));
895 }
896 Shape stripped_shape =
897 shape.has_layout() ? MakeShapeWithLayout(shape.element_type(),
898 dimension_sizes, minor_to_major)
899 : MakeShape(shape.element_type(), dimension_sizes);
900
901 VLOG(10) << "Original_shape: " << HumanStringWithLayout(shape);
902 VLOG(10) << "Stripped_shape: " << HumanStringWithLayout(stripped_shape);
903 return stripped_shape;
904}

1008/* static */ std::tuple<bool, std::vector<int64>, std::vector<int64>>
1009ShapeUtil::InsertedOrDeleted1SizedDimensions(const Shape& shape_pre,
1010 const Shape& shape_post) {
1011 auto nil = std::make_tuple(false, std::vector<int64>(), std::vector<int64>());
1012
1013 std::vector<int64> deleted_indices;
1014 std::vector<int64> inserted_indices;
1015 // Returns false if any input/output index between prior_unmodified_dim_pair
1016 // and unmodified_dim_pair have size >1. Otherwise, returns true and appends
1017 // the degerenate input/output dimensions in the gap to
1018 // deleted_indices/inserted_indices respectively.
1019 auto check_modified_dims =
1020 [&shape_pre, &shape_post, &deleted_indices, &inserted_indices](
1021 std::pair<int64, int64> prior_unmodified_dim_pair,
1022 std::pair<int64, int64> unmodified_dim_pair) {
1023 for (int64 modified_input_dim = prior_unmodified_dim_pair.first + 1;
1024 modified_input_dim < unmodified_dim_pair.first;
1025 ++modified_input_dim) {
1026 if (shape_pre.dimensions(modified_input_dim) > 1) {
1027 return false;
1028 }
1029 deleted_indices.push_back(modified_input_dim);
1030 }
1031 for (int64 modified_output_dim = prior_unmodified_dim_pair.second + 1;
1032 modified_output_dim < unmodified_dim_pair.second;
1033 ++modified_output_dim) {
1034 if (shape_post.dimensions(modified_output_dim) > 1) {
1035 return false;
1036 }
1037 inserted_indices.push_back(modified_output_dim);
1038 }
1039 return true;
1040 };
1041
1042 std::vector<std::pair<int64, int64>> unmodified_dims =
1043 DimensionsUnmodifiedByReshape(shape_pre, shape_post);
1044 // Returns nil if the reshape modifies any non-degenerate input/output
1045 // dimension. DimensionsUnmodifiedByReshape gives us all unmodified
1046 // dimensions, so we only need to check whether dimensions in the gaps (thus
1047 // modified) have size >1.
1048 for (size_t i = 0; i <= unmodified_dims.size(); ++i) {
1049 // Check (modified) dimensions between unmodified_dims[i-1] and
1050 // unmodified_dims[i].
1051 auto prior_unmodified_dim_pair =
1052 i > 0 ? unmodified_dims[i - 1] : std::make_pair(-1LL, -1LL);
1053 auto unmodified_dim_pair =
1054 i < unmodified_dims.size()
1055 ? unmodified_dims[i]
1056 : std::make_pair(Rank(shape_pre), Rank(shape_post));
1057 if (!check_modified_dims(prior_unmodified_dim_pair, unmodified_dim_pair)) {
1058 return nil;
1059 }
1060 }
1061
1062 return std::make_tuple(true, deleted_indices, inserted_indices);
1063}

1065/* static */ std::vector<std::pair<int64, int64>>
1066ShapeUtil::DimensionsUnmodifiedByReshape(const Shape& input_shape,
1067 const Shape& output_shape) {
1068 // Unmodified dimensions are merely common factors of rank 1.
1069 auto common_factors = CommonFactors(AsInt64Slice(input_shape.dimensions()),
1070 AsInt64Slice(output_shape.dimensions()));
1071 for (size_t i = 0; i < common_factors.size() - 1;) {
1072 if (1 != common_factors[i + 1].first - common_factors[i].first ||
1073 1 != common_factors[i + 1].second - common_factors[i].second) {
1074 common_factors.erase(common_factors.begin() + i);
1075 } else {
1076 ++i;
1077 }
1078 }
1079 // `CommonFactors(a, b).back() == (a.rank, b.rank)` so we must pop it.
1080 common_factors.pop_back();
1081 return common_factors;
1082}

1084/* static */ bool ShapeUtil::TransposeIsBitcast(
1085 const Shape& input_shape, const Shape& output_shape,
1086 tensorflow::gtl::ArraySlice<int64> dimension_mapping) {
1087 CHECK(LayoutUtil::HasLayout(input_shape) &&
1088 LayoutUtil::HasLayout(output_shape));
1089
1090 if (!SameElementType(input_shape, output_shape)) {
1091 return false;
1092 }
1093
1094 // Padding is not handled.
1095 if (LayoutUtil::IsPadded(input_shape) && LayoutUtil::IsPadded(output_shape)) {
1096 return false;
1097 }
1098
1099 // Check the reshape permutes the positions of each dimension in the
1100 // minor-to-major order. positions[i]=k means dimension `i` is k-th minor.
1101 // input_positions = apply(dimension_mapping, output_positions)
1102 //
1103 // Because the positions of each dimension are the inverse permutation of the
1104 // minor-to-major order, the above check is equivalent to
1105 // inverse(input_dimensions) =
1106 // apply(dimension_mapping, inverse(output_dimensions))
1107 // # `I` indicates identity permutation.
1108 // apply(input_dimensions, I) =
1109 // apply(dimension_mapping, apply(output_dimensions, I))
1110 // apply(input_dimensions, I) =
1111 // apply((dimension_mapping * output_dimensions), I)
1112 // input_dimensions = dimension_mapping * output_dimensions
1113 return ContainersEqual(
1114 ComposePermutations(dimension_mapping,
1115 AsInt64Slice(output_shape.layout().minor_to_major())),
1116 input_shape.layout().minor_to_major());
1117}
1118
1119/* static */ bool ShapeUtil::ReshapeIsBitcast(const Shape& input_shape,
1120 const Shape& output_shape) {
1121 CHECK(LayoutUtil::HasLayout(input_shape) &&
1122 LayoutUtil::HasLayout(output_shape));
1123
1124 if (!SameElementType(input_shape, output_shape)) {
1125 return false;
1126 }
1127
1128 // Padding is not handled.
1129 if (LayoutUtil::IsPadded(input_shape) || LayoutUtil::IsPadded(output_shape)) {
1130 return false;
1131 }
1132
1133 CHECK_EQ(ElementsIn(input_shape), ElementsIn(output_shape));
1134 if (ElementsIn(input_shape) == 0) {
1135 return true;
1136 }
1137
1138 // TL;DR: The rest of the method checks that the reshape does not change the
1139 // physical location of any unit input or output index. Unit indices have
1140 // exactly one dimension that equals 1 and other dimensions 0. This condition
1141 // is necessary for the reshape to be a bitcast, because a bitcast-equivalent
1142 // reshape shouldn't change the physical location of any element. It is also a
1143 // sufficient condition as is proved below (note: many details are omitted for
1144 // space).
1145 //
1146 // Definitions:
1147 //
1148 // * Denote the input shape by IS and output shape by OS. IS[i] or OS[i] means
1149 // the size of i-th least significant dimension of IS or OS (this is opposite
1150 // to how we define the index of Shape::dimensions()).
1151 //
1152 // * Given an input or output index I, denote by p(I) I's physical linear
1153 // index (or physical index for short) and l(I) I's logical linear index (or
1154 // logical index for short).
1155 //
1156 // * Given a logical index k, denote by II(k) the input index whose linear
1157 // index is k, and OI(k) the corresponding output index.
1158 //
1159 // * Denote by IT[i] the increment of physical index if i-th dimension of the
1160 // input index is increased by 1. Similarly, OT[i] means the increment if i-th
1161 // dimension of the output index is increased by 1. Note that IT[i] or OT[i]
1162 // is a function of IS or OS and the layout, and not dependent on the specific
1163 // input or output index.
1164 //
1165 // To prove the reshape from IS to OS is a bitcast, it is sufficient to prove
1166 // that, for any linear index k, p(II(k))=p(OI(k)). We prove this by
1167 // induction. We know p(II(0))=p(OI(0)) is trivially true, so what's left is
1168 // to prove, with every increment on k, the above formula still holds.
1169 //
1170 // First, suppose reshaping from IS to OS is non-factorizable (we discuss
1171 // refactorizable reshapes later). A reshape from IS to OS is factorizable, if
1172 // there exists (i,j) such that
1173 //
1174 // 0<=i<=|IS|
1175 // 0<=j<=|OS|
1176 // |IS|-i+|OS|-j > 0 (i.e., i,j mustn't both point to the end)
1177 // product(IS[i], IS[i+1], ..., IS[|IS|-1])
1178 // = product(OS[j], OS[j+1], ..., OS[|OS|-1])
1179 //
1180 // p(II(k))=p(OI(k)) is trivially true for k=0 because p(II(0)) and p(OI(0))
1181 // are both 0. It's also trivially true for k=1, because II(1) and OI(1) are
1182 // unit indices which are already tested. This also means IT[0]=OT[0]
1183 // because p(II(1))=IT[0] and p(OI(1))=OT[0].
1184 //
1185 // Furthermore, p(II(k))=p(OI(k)) for k<min(IS[0],OS[0]), because each
1186 // increment of k adds IT[0] to the input physical and OT[0] (same as IT[0])
1187 // to the output physical.
1188 //
1189 // When k=min(IS[0],OS[0]), the first wrap happens. Without losing generality,
1190 // suppose IS[0]<OS[0] and thus k=IS[0]. Similar proof applies to IS[0]>OS[0].
1191 // Note that IS[0]!=OS[0] because the reshape is non-factorizable. From
1192 // logical index k-1 to logical index k, dimension 1 of the input index
1193 // is increased by 1 and dimension 0 is reset to 0 thus decreased by
1194 // IS[0]-1. Therefore, the physical input index is increased by
1195 //
1196 // p(II(k)) - p(II(k-1)) = IT[1] - (IS[0]-1) * IT[0]
1197 //
1198 // Because IS[0]<OS[0], the only change to the output index is that its
1199 // dimension 0 is increased by one. Therefore,
1200 //
1201 // p(OI(k)) - p(OI(k-1)) = OT[0] = IT[0]
1202 //
1203 // Because II(k) is an unit index -- (0,..,0,1,0), we already tested that
1204 // p(II(k))=p(OI(k)). Therefore,
1205 // IT[1] - (IS[0]-1) * IT[0] = IT[0]
1206 // IT[1] = IS[0] * IT[0]
1207 // In other words, input dimension 1 is immediately more major than input
1208 // dimension 0. We can now conceptually collapse these two dimensions because
1209 // an increment in the logical index affecting only these two dimensions maps
1210 // to IT[0] in the physical index.
1211 //
1212 // By induction (omitted here), we can prove IT[i]=IS[i-1]*IT[i-1] and
1213 // OT[i]=OS[i-1]*OT[i-1]. Therefore, both IS and OS are row-major and bitwise
1214 // identical.
1215 //
1216 // A factorizable reshape can be factorized into a list of non-factorizable
1217 // sub-reshapes, each of which can be handled similarly to the proof above.
1218 // For example,
1219 //
1220 // [7x9x2x15] -> [63x6x5]
1221 //
1222 // can be factorized into
1223 //
1224 // [7x9] -> [63] and [2x15] -> [6x5].
1225 //
1226 // Suppose input index I=(x3,x2,x1,x0) and output index O=(y2,y1,y0) have the
1227 // same logical linear index. According to the factorization, we know
1228 // l(x3,x2,0,0)=l(y2,0,0) and l(0,0,x1,x0)=l(0,y1,y0). Using the proof for
1229 // non-factorizable reshapes, we can prove p(0,0,x1,x0)=p(0,y1,y0). Using a
1230 // similar proof, with the increment of the logical index set to
1231 // IS[1]*IS[0]=OS[1]*OS[0]=30 instead of 1, we can prove
1232 // p(x3,x2,0,0)=p(y2,0,0) too. Therefore,
1233 //
1234 // p(x3,x2,x1,x0) = p(x3,x2,0,0) + p(0,0,x1,x0)
1235 // = p(y2,0,0) + p(0,0,y1,y0)
1236 // = p(y2,y1,y0)
1237 //
1238 // check_input_unit_indices checks one way of the condition: each input unit
1239 // index is mapped to an output index with the same physical location. This
1240 // lambda will be called again with input_shape and output_shape reversed to
1241 // check the other way.
1242 auto check_input_unit_indices = [](const Shape& input_shape,
1243 const Shape& output_shape) {
1244 // input_shape_dim0_major/output_shape_dim0_major has the same "dimensions"
1245 // as input_shape/output_shape and the dimension-0-major layout. These two
1246 // shapes are used for conversion between logical linear indices and
1247 // multi-dimensional indices.
1248 Shape input_shape_dim0_major = MakeShapeWithDescendingLayout(
1249 input_shape.element_type(), AsInt64Slice(input_shape.dimensions()));
1250 Shape output_shape_dim0_major = MakeShapeWithDescendingLayout(
1251 output_shape.element_type(), AsInt64Slice(output_shape.dimensions()));
1252
1253 for (int64 input_dim = 0; input_dim < Rank(input_shape); ++input_dim) {
1254 if (input_shape.dimensions(input_dim) <= 1) {
1255 continue;
1256 }
1257
1258 std::vector<int64> input_unit_index(Rank(input_shape), 0);
1259 input_unit_index[input_dim] = 1;
1260 int64 logical_linear_index =
1261 IndexUtil::MultidimensionalIndexToLinearIndex(input_shape_dim0_major,
1262 input_unit_index);
1263 // output_index has the same logical linear index as input_unit_index.
1264 std::vector<int64> output_index =
1265 IndexUtil::LinearIndexToMultidimensionalIndex(output_shape_dim0_major,
1266 logical_linear_index);
1267 // Check input_unit_index and output_index have the same physical linear
1268 // index.
1269 if (IndexUtil::MultidimensionalIndexToLinearIndex(input_shape,
1270 input_unit_index) !=
1271 IndexUtil::MultidimensionalIndexToLinearIndex(output_shape,
1272 output_index)) {
1273 return false;
1274 }
1275 }
1276 return true;
1277 };
1278 return check_input_unit_indices(input_shape, output_shape) &&
1279 check_input_unit_indices(output_shape, input_shape);
1280}

1282/* static */ tensorflow::gtl::optional<Shape> ShapeUtil::AlignLayouts(
1283 const Shape& input_shape, const Shape& output_shape) {
1284 int64 input_rank = Rank(input_shape);
1285 int64 output_rank = Rank(output_shape);
1286
1287 // First, calculate an alignment of the dimensions. A consecutive sequence of
1288 // input dimensions and output dimensions belong to the same alignment part if
1289 // the products of their dimension bounds are the same. In the easiest case,
1290 // an alignment part consists of one input dimension and one output dimension
1291 // which both have the same dimension bound. An alignment part specifies which
1292 // dimensions need to be kept together in a physical layout if we want a
1293 // reshape to be a bitcast. The order of the alignment parts is defined by the
1294 // physical layout of the input shape, so when we construct the layout for the
1295 // output shape we just process the alignment parts in this order, and then
1296 // layout the dimensions belonging to each part in descending (major to minor)
1297 // order.
1298
1299 // Stores the input and output dimension numbers where each alignment part
1300 // starts.
1301 std::vector<std::pair<int64, int64>> alignment;
1302 alignment.push_back({0, 0});
1303
1304 // Stores a mapping from the input dimension to the alignment part it belongs
1305 // to.
1306 std::vector<int64> dimension_to_alignment_index(input_rank);
1307 int64 input_dimension_product = 1, output_dimension_product = 1;
1308 for (int64 i = 0, j = 0; i < input_rank || j < output_rank;) {
1309 // Check if we have reached the end of an alignment part.
1310 if (input_dimension_product == output_dimension_product &&
1311 input_dimension_product > 1) {
1312 alignment.push_back({i, j});
1313 input_dimension_product = output_dimension_product = 1;
1314 }
1315 if (input_dimension_product < output_dimension_product ||
1316 j == output_rank) {
1317 if (i == input_rank) {
1318 return tensorflow::gtl::nullopt;
1319 }
1320 dimension_to_alignment_index[i] = alignment.size() - 1;
1321 input_dimension_product *= input_shape.dimensions(i);
1322 ++i;
1323 } else {
1324 output_dimension_product *= output_shape.dimensions(j);
1325 ++j;
1326 }
1327 }
1328 if (input_dimension_product != output_dimension_product) {
1329 return tensorflow::gtl::nullopt;
1330 }
1331 // We also need to store an end element so that we know where the last
1332 // alignment part ends.
1333 alignment.push_back({input_rank, output_rank});
1334
1335 // Now check if the physical layout can potentially be aligned to the output
1336 // shape by changing the physical layout of the output shape. We need to check
1337 // that all dimension numbers that belong to the same alignment part appear
1338 // consecutively, and are in descending order. However we can ignore any
1339 // trivial dimension bounds of 1, because they can be placed anywhere.
1340 auto input_dimension_numbers = input_shape.layout().minor_to_major();
1341 std::vector<int64> output_layout;
1342 output_layout.reserve(output_rank);
1343 for (int64 i = 0; i < input_rank;) {
1344 int64 current_dimension_number = input_dimension_numbers[i];
1345
1346 // Skip trivial dimensions with a bound of 1.
1347 if (input_shape.dimensions(current_dimension_number) == 1) {
1348 ++i;
1349 continue;
1350 }
1351
1352 // Calculate the number of non-trivial dimension bounds in the input shape
1353 // belonging to the current alignment part.
1354 const int64 current_alignment_index =
1355 dimension_to_alignment_index[current_dimension_number];
1356 // Because of the special end element that we added, we can be sure that
1357 // 'current_alignment_index' is < alignment.size() - 1.
1358 CHECK_LT(current_alignment_index, alignment.size() - 1);
1359 int64 num_non_trivial_dimensions_in_alignment_part = 0;
1360 for (int64 j = alignment[current_alignment_index].first;
1361 j < alignment[current_alignment_index + 1].first; ++j) {
1362 if (input_shape.dimensions(j) != 1) {
1363 ++num_non_trivial_dimensions_in_alignment_part;
1364 }
1365 }
1366
1367 // Check that the following 'num_non_trivial_dimensions_in_alignment_part'
1368 // dimension numbers (ignoring dimension numbers with dimension bound 1) are
1369 // in descending order and belong to the current alignment part.
1370 for (int64 j = 0; j < num_non_trivial_dimensions_in_alignment_part;
1371 ++i, ++j) {
1372 if (i == input_rank) {
1373 return tensorflow::gtl::nullopt;
1374 }
1375 // Skip trivial dimensions with a bound of 1.
1376 if (input_shape.dimensions(input_dimension_numbers[i]) == 1) {
1377 --j;
1378 continue;
1379 }
1380 // If the current dimension number belongs to a different alignment part,
1381 // or the dimension numbers are not in descending order, we can return
1382 // early.
1383 if (dimension_to_alignment_index[input_dimension_numbers[i]] !=
1384 current_alignment_index ||
1385 input_dimension_numbers[i] > current_dimension_number) {
1386 return tensorflow::gtl::nullopt;
1387 }
1388 current_dimension_number = input_dimension_numbers[i];
1389 }
1390
1391 // The output dimension numbers that belong to the current alignment part
1392 // need to appear in the same descending order as in the input. Again, we
1393 // can skip dimensions with a bound of 1.
1394 for (int64 j = alignment[current_alignment_index + 1].second - 1;
1395 j >= alignment[current_alignment_index].second; --j) {
1396 if (output_shape.dimensions(j) != 1) {
1397 output_layout.push_back(j);
1398 }
1399 }
1400 }
1401 // Now add all the dimensions with dimension bound 1 at the end of
1402 // 'output_layout'.
1403 for (int64 i = 0; i < output_rank; ++i) {
1404 if (output_shape.dimensions(i) == 1) {
1405 output_layout.push_back(i);
1406 }
1407 }
1408 CHECK_EQ(output_layout.size(), output_rank);
1409 Shape output_shape_with_layout = MakeShapeWithLayout(
1410 output_shape.element_type(), AsInt64Slice(output_shape.dimensions()),
1411 output_layout);
1412 CHECK(ReshapeIsBitcast(input_shape, output_shape_with_layout));
1413 return output_shape_with_layout;
1414}

1416/* static */ Shape ShapeUtil::DeleteDimension(int64 dim_to_delete,
1417 Shape shape) {
1418 shape.mutable_dimensions()->erase(shape.dimensions().begin() + dim_to_delete);
1419 if (LayoutUtil::HasLayout(shape)) {
1420 Layout* layout = shape.mutable_layout();
1421 layout->set_format(DENSE);
1422 for (size_t i = 0; i < layout->minor_to_major().size();) {
1423 if (layout->minor_to_major(i) == dim_to_delete) {
1424 layout->mutable_minor_to_major()->erase(
1425 layout->minor_to_major().begin() + i);
1426 continue;
1427 }
1428 if (layout->minor_to_major(i) > dim_to_delete) {
1429 (*layout->mutable_minor_to_major())[i] -= 1;
1430 }
1431 ++i;
1432 }
1433 }
1434 return shape;
1435}

1437/* static */ Shape ShapeUtil::FilterDimensions(
1438 const std::function<bool(int64)>& p, Shape shape) {
1439 std::vector<int64> dims_to_delete;
1440 for (int64 i = shape.dimensions().size() - 1; i >= 0; --i) {
1441 if (!p(i)) {
1442 dims_to_delete.push_back(i);
1443 }
1444 }
1445 for (int64 dim : dims_to_delete) {
1446 shape = DeleteDimension(dim, shape);
1447 }
1448 return shape;
1449}
