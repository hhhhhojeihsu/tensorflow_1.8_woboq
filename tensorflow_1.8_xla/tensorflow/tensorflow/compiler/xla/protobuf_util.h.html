26// Returns true if m1 is equal to m2.
27//
28// WARNING: We use protocol buffer serialization and then check for
29// equality of the serialized representation, which may miss some
30// cases of equality. However, for the purposes of the XLA code
31// base, this form of equality checking is sufficient.
32extern bool ProtobufEquals(const tensorflow::protobuf::Message& m1,
33 const tensorflow::protobuf::Message& m2);
34
35// Writes the given message in binary proto to the path formed by joining
36// 'directory/file_name.pb'. The 'directory' is recursively created if it
37// doesn't already exist, and the 'file_name' is sanitized by replacing
38// illegal characters with underscore '_'.
39Status DumpProtoToDirectory(const tensorflow::protobuf::Message& message,
40 const string& directory, const string& file_name);
