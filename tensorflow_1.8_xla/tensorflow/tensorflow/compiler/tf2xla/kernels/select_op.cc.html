25namespace tensorflow {
26namespace {
27
28class SelectOp : public XlaOpKernel {
29 public:
30 explicit SelectOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {}
31
32 void Compile(XlaOpKernelContext* ctx) override {
33 const TensorShape cond_shape = ctx->InputShape(0);
34 const TensorShape then_shape = ctx->InputShape(1);
35 const TensorShape else_shape = ctx->InputShape(2);
36
37 OP_REQUIRES(
38 ctx, then_shape.IsSameSize(else_shape),
39 errors::InvalidArgument(
40 "'then' and 'else' must have the same size. but received: ",
41 then_shape.DebugString(), " vs. ", else_shape.DebugString()));
42
43 xla::ComputationBuilder* builder = ctx->builder();
44
45 auto cond_handle = ctx->Input(0);
46 auto then_handle = ctx->Input(1);
47 auto else_handle = ctx->Input(2);
48
49 bool broadcasting = !cond_shape.IsSameSize(then_shape);
50 bool cond_is_scalar = TensorShapeUtils::IsScalar(cond_shape);
51 if (broadcasting && !cond_is_scalar) {
52 OP_REQUIRES(ctx, TensorShapeUtils::IsVector(cond_shape),
53 errors::InvalidArgument(
54 "'cond' must be a scalar or a vector, but saw shape: ",
55 cond_shape.DebugString()));
56 OP_REQUIRES(ctx, TensorShapeUtils::IsVectorOrHigher(then_shape),
57 errors::InvalidArgument(
58 "'then' must be at least a vector, but saw shape: ",
59 then_shape.DebugString()));
60 OP_REQUIRES(ctx, then_shape.dim_size(0) == cond_shape.num_elements(),
61 errors::InvalidArgument("Number of batches of 'then' must "
62 "match size of 'cond', but saw: ",
63 then_shape.dim_size(0), " vs. ",
64 cond_shape.num_elements()));
65
66 // TODO(phawkins): broadcasting on the right seems pretty awkward in
67 // XLA. It seems we have to broadcast on the left and then Reshape
68 // to get the dimensions in the right order.
69 const auto dim_sizes = then_shape.dim_sizes();
70 gtl::ArraySlice<int64> bdims = dim_sizes;
71 bdims.pop_front();
72 cond_handle = builder->Broadcast(cond_handle, bdims);
73
74 std::vector<int64> dim_order(then_shape.dims());
75 dim_order[0] = then_shape.dims() - 1;
76 std::iota(dim_order.begin() + 1, dim_order.end(), 0);
77 cond_handle = builder->Transpose(cond_handle, dim_order);
78 }
79 ctx->SetOutput(0, builder->Select(cond_handle, then_handle, else_handle));
80 }
81
82 private:
83 TF_DISALLOW_COPY_AND_ASSIGN(SelectOp);
84};
85
86REGISTER_XLA_OP(Name("Select"), SelectOp);
