25XlaIfOp::XlaIfOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {
26 const NameAttrList* name_attr;
27 OP_REQUIRES_OK(ctx, ctx->GetAttr("then_branch", &name_attr));
28 then_branch_ = *name_attr;
29 OP_REQUIRES_OK(ctx, ctx->GetAttr("else_branch", &name_attr));
30 else_branch_ = *name_attr;
31
32 OP_REQUIRES_OK(ctx, ctx->GetAttr("Tcond", &cond_type_));
33 OP_REQUIRES_OK(ctx, ctx->GetAttr("Tin", &input_types_));
34 OP_REQUIRES_OK(ctx, ctx->GetAttr("Tout", &output_types_));
35}

37// TODO(b/35949885): There is duplication here with the handling of the
38// while_op. Refactor the common code out/rework.
39void XlaIfOp::Compile(XlaOpKernelContext* ctx) {
40 xla::ComputationBuilder* b = ctx->builder();
41
42 OP_REQUIRES(ctx, cond_type_ == DT_BOOL,
43 errors::InvalidArgument(
44 "Condition argument must be a boolean for XLA compilation"));
45 OP_REQUIRES(ctx, TensorShapeUtils::IsScalar(ctx->InputShape(0)),
46 errors::InvalidArgument(
47 "Condition argument must be a scalar for XLA compilation"));
48
49 VLOG(1) << "Building If: " << input_types_.size() << " inputs";
50
51 std::vector<xla::ComputationDataHandle> inputs(input_types_.size());
52 std::vector<XlaCompiler::Argument> arguments(input_types_.size());
53 for (int i = 0; i < input_types_.size(); ++i) {
54 XlaCompiler::Argument& arg = arguments[i];
55 DataType type = ctx->input_type(i + 1);
56 if (type == DT_RESOURCE) {
57 XlaResource* resource;
58 OP_REQUIRES_OK(ctx, ctx->GetResourceInput(i + 1, &resource));
59
60 arg.initialized = resource->initialized();
61 arg.kind = XlaCompiler::Argument::kResource;
62 arg.resource_kind = resource->kind();
63 OP_REQUIRES_OK(ctx, resource->Pack(&inputs[i], b));
64
65 arg.type = resource->type();
66 arg.shape = resource->shape();
67 OP_REQUIRES(ctx, arg.initialized,
68 errors::Unimplemented("Uninitialized arguments: ", arg.name));
69 arg.tensor_array_size = resource->tensor_array_size();
70 for (const auto& gradient : resource->tensor_array_gradients()) {
71 arg.tensor_array_gradients.insert(gradient.first);
72 }
73 arg.name = resource->name();
74 VLOG(2) << "Resource " << resource->name()
75 << " type: " << DataTypeString(arg.type)
76 << " shape: " << arg.shape.DebugString()
77 << " initialized: " << arg.initialized;
78 } else {
79 arg.kind = XlaCompiler::Argument::kParameter;
80 arg.type = input_types_[i];
81 arg.shape = ctx->InputShape(i + 1);
82 inputs[i] = ctx->Input(i + 1);
83 VLOG(2) << "Arg type: " << DataTypeString(arg.type)
84 << " shape: " << arg.shape.DebugString();
85 }
86 }
87
88 // Compile both branches of the conditional.
89 XlaCompiler::CompileOptions options;
90 options.use_tuple_arg = true;
91 options.resolve_compile_time_constants = false;
92 options.return_updated_values_for_all_resources = true;
93 options.is_entry_computation = false;
94 XlaCompiler* compiler = ctx->compiler();
95
96 XlaCompiler::CompilationResult then_result;
97 OP_REQUIRES_OK(ctx, compiler->CompileFunction(options, then_branch_,
98 arguments, &then_result));
99 XlaCompiler::CompilationResult else_result;
100 OP_REQUIRES_OK(ctx, compiler->CompileFunction(options, else_branch_,
101 arguments, &else_result));
102
103 for (XlaCompiler::CompilationResult* result : {&then_result, &else_result}) {
104 for (const XlaCompiler::ResourceUpdate& update : result->resource_updates) {
105 XlaResource* resource;
106 OP_REQUIRES_OK(ctx,
107 ctx->GetResourceInput(update.input_index + 1, &resource));
108 XlaCompiler::Argument& arg = arguments[update.input_index];
109
110 // Add any TensorArray gradients touched by the then/else computation to
111 // the enclosing graph.
112 for (const string& grad_source : update.tensor_array_gradients_accessed) {
113 VLOG(5) << "TensorArray " << resource->name() << " accessed gradient "
114 << grad_source;
115 XlaResource* gradient;
116 OP_REQUIRES_OK(ctx, resource->GetOrCreateTensorArrayGradient(
117 grad_source, b, &gradient));
118 }
119 // Add all of the TensorArray gradients to the argument. For simplicity,
120 // we always pass all known gradients.
121 for (const auto& gradient : resource->tensor_array_gradients()) {
122 arg.tensor_array_gradients.insert(gradient.first);
123 }
124 }
125 }
126
127 // Check that both branches have identical input shapes.
128 OP_REQUIRES(ctx, then_result.xla_input_shapes.size() == 1,
129 errors::FailedPrecondition("Expected one input shape"));
130 xla::Shape then_input_shape = then_result.xla_input_shapes[0];
131 OP_REQUIRES(ctx, xla::ShapeUtil::IsTuple(then_input_shape),
132 errors::FailedPrecondition("Expected tuple shape"));
133 OP_REQUIRES(ctx, else_result.xla_input_shapes.size() == 1,
134 errors::FailedPrecondition("Expected one input shape"));
135 xla::Shape else_input_shape = else_result.xla_input_shapes[0];
136 OP_REQUIRES(ctx, xla::ShapeUtil::IsTuple(else_input_shape),
137 errors::FailedPrecondition("Expected tuple shape"));
138 OP_REQUIRES(ctx,
139 xla::ShapeUtil::Compatible(then_input_shape, else_input_shape),
140 errors::InvalidArgument(
141 "Input shapes of then and else branches do not match: ",
142 xla::ShapeUtil::HumanString(then_input_shape), " vs. ",
143 xla::ShapeUtil::HumanString(else_input_shape)));
144
145 // Check that both branches have identical output shapes.
146 OP_REQUIRES(
147 ctx,
148 xla::ShapeUtil::Compatible(then_result.xla_output_shape,
149 else_result.xla_output_shape),
150 errors::InvalidArgument(
151 "Output shapes of then and else branches do not match: ",
152 xla::ShapeUtil::HumanString(then_result.xla_output_shape), " vs. ",
153 xla::ShapeUtil::HumanString(else_result.xla_output_shape)));
154
155 VLOG(2) << "Input shape: " << xla::ShapeUtil::HumanString(then_input_shape);
156 VLOG(2) << "Output shape: "
157 << xla::ShapeUtil::HumanString(then_result.xla_output_shape);
158
159 // We set return_updated_values_for_all_resources=true and we pass the same
160 // arguments to both computations, so the resource update count must match.
161 OP_REQUIRES(ctx,
162 then_result.resource_updates.size() ==
163 else_result.resource_updates.size(),
164 errors::FailedPrecondition(
165 "Different number of resources in then and else branch"));
166 for (int i = 0; i < then_result.resource_updates.size(); ++i) {
167 const auto& lhs = then_result.resource_updates[i];
168 const auto& rhs = else_result.resource_updates[i];
169 bool equal = lhs.input_index == rhs.input_index && lhs.shape == rhs.shape &&
170 lhs.tensor_array_gradients_accessed ==
171 rhs.tensor_array_gradients_accessed;
172 OP_REQUIRES(
173 ctx, equal,
174 errors::FailedPrecondition(
175 "Mismatch in resource of then and else branch for resource ", i));
176 }
177
178 xla::ComputationDataHandle outputs =
179 b->Conditional(ctx->Input(0), b->Tuple(inputs), *then_result.computation,
180 b->Tuple(inputs), *else_result.computation);
181 // Sets non-variable outputs.
182 for (int i = 0; i < output_types_.size(); ++i) {
183 if (ctx->input_type(i) != DT_RESOURCE) {
184 xla::ComputationDataHandle output_handle = b->GetTupleElement(outputs, i);
185 if (VLOG_IS_ON(2)) {
186 LOG(INFO) << "Setting output " << i;
187 auto shape_or = b->GetShape(output_handle);
188 if (shape_or.ok()) {
189 LOG(INFO) << "Shape for output " << i << ": "
190 << xla::ShapeUtil::HumanString(*shape_or.ValueOrDie());
191 } else {
192 LOG(INFO) << "Shape unknown for output " << i;
193 }
194 }
195 ctx->SetOutput(i, output_handle);
196 }
197 }
198
199 // Updates the values of any resource variables modified by the conditional
200 // bodies.
201 for (XlaCompiler::CompilationResult* result : {&then_result, &else_result}) {
202 for (int i = 0; i < result->resource_updates.size(); ++i) {
203 const XlaCompiler::ResourceUpdate& update = result->resource_updates[i];
204 XlaResource* resource;
205 OP_REQUIRES_OK(ctx,
206 ctx->GetResourceInput(update.input_index + 1, &resource));
207 if (update.modified) {
208 int pos = result->outputs.size() + i;
209 OP_REQUIRES_OK(ctx,
210 resource->SetFromPack(
211 arguments[update.input_index].tensor_array_gradients,
212 b->GetTupleElement(outputs, pos), b));
213 }
214 VLOG(2) << "If variable: pos: " << update.input_index
215 << " name: " << resource->name()
216 << " modified: " << update.modified
217 << " type: " << DataTypeString(update.type)
218 << " shape: " << update.shape.DebugString();
219 }
220 }
221 VLOG(1) << "Done building If";
222}

224REGISTER_XLA_OP(Name("XlaIf").AllowResourceTypes(), XlaIfOp);
