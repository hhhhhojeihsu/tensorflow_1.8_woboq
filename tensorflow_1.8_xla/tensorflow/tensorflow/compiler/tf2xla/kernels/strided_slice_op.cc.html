33class StridedSliceOp : public XlaOpKernel {
34 public:
35 explicit StridedSliceOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {
36 OP_REQUIRES_OK(ctx, ctx->GetAttr("begin_mask", &begin_mask_));
37 OP_REQUIRES_OK(ctx, ctx->GetAttr("end_mask", &end_mask_));
38 OP_REQUIRES_OK(ctx, ctx->GetAttr("ellipsis_mask", &ellipsis_mask_));
39 OP_REQUIRES_OK(ctx, ctx->GetAttr("new_axis_mask", &new_axis_mask_));
40 OP_REQUIRES_OK(ctx, ctx->GetAttr("shrink_axis_mask", &shrink_axis_mask_));
41 OP_REQUIRES_OK(ctx, ctx->GetAttr("Index", &index_type_));
42 }
43
44 void Compile(XlaOpKernelContext* ctx) override {
45 const TensorShape input_shape = ctx->InputShape(0);
46
47 TensorShape final_shape;
48 gtl::InlinedVector<int64, 4> begin;
49 gtl::InlinedVector<int64, 4> end;
50 gtl::InlinedVector<int64, 4> strides;
51
52 xla::Literal begin_literal, end_literal, strides_literal;
53 OP_REQUIRES_OK(ctx, ctx->ConstantInput(1, &begin_literal));
54 OP_REQUIRES_OK(ctx, ctx->ConstantInput(2, &end_literal));
55 OP_REQUIRES_OK(ctx, ctx->ConstantInput(3, &strides_literal));
56
57 Tensor begin_tensor, end_tensor, strides_tensor;
58 OP_REQUIRES_OK(
59 ctx, LiteralToHostTensor(begin_literal, index_type_, &begin_tensor));
60 OP_REQUIRES_OK(ctx,
61 LiteralToHostTensor(end_literal, index_type_, &end_tensor));
62 OP_REQUIRES_OK(ctx, LiteralToHostTensor(strides_literal, index_type_,
63 &strides_tensor));
64
65 TensorShape dummy_processing_shape;
66 bool dummy = false;
67 OP_REQUIRES_OK(ctx,
68 ValidateStridedSliceOp(
69 &begin_tensor, &end_tensor, strides_tensor, input_shape,
70 begin_mask_, end_mask_, ellipsis_mask_, new_axis_mask_,
71 shrink_axis_mask_, &dummy_processing_shape, &final_shape,
72 &dummy, &dummy, &dummy, &begin, &end, &strides));
73
74 gtl::InlinedVector<int64, 4> dimensions_to_reverse;
75 gtl::InlinedVector<int64, 4> slice_begin, slice_end, slice_strides;
76
77 for (int i = 0; i < begin.size(); ++i) {
78 if (strides[i] > 0) {
79 slice_begin.push_back(begin[i]);
80 slice_end.push_back(std::max(end[i], begin[i]));
81 slice_strides.push_back(strides[i]);
82 } else {
83 // Negative stride: swap begin and end, add 1 because the interval
84 // is semi-open, and mark the dimension to be reversed.
85 slice_begin.push_back(input_shape.dim_size(i) - begin[i] - 1);
86 slice_end.push_back(std::max(input_shape.dim_size(i) - end[i] - 1,
87 input_shape.dim_size(i) - begin[i] - 1));
88 slice_strides.push_back(-strides[i]);
89 dimensions_to_reverse.push_back(i);
90 }
91 }
92
93 xla::ComputationDataHandle slice = ctx->Input(0);
94 if (!dimensions_to_reverse.empty()) {
95 slice = ctx->builder()->Rev(slice, dimensions_to_reverse);
96 }
97
98 slice = ctx->builder()->Slice(slice, slice_begin, slice_end, slice_strides);
99
100 slice = ctx->builder()->Reshape(slice, final_shape.dim_sizes());
101 ctx->SetOutput(0, slice);
102 }
103
104 private:
105 int32 begin_mask_, end_mask_;
106 int32 ellipsis_mask_, new_axis_mask_, shrink_axis_mask_;
107 DataType index_type_;
108};
109
110REGISTER_XLA_OP(Name("StridedSlice")
111 .CompileTimeConstInput("begin")
112 .CompileTimeConstInput("end")
113 .CompileTimeConstInput("strides"),
114 StridedSliceOp);

116class StridedSliceGradOp : public XlaOpKernel {
117 public:
118 explicit StridedSliceGradOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {
119 OP_REQUIRES_OK(ctx, ctx->GetAttr("begin_mask", &begin_mask_));
120 OP_REQUIRES_OK(ctx, ctx->GetAttr("end_mask", &end_mask_));
121 OP_REQUIRES_OK(ctx, ctx->GetAttr("ellipsis_mask", &ellipsis_mask_));
122 OP_REQUIRES_OK(ctx, ctx->GetAttr("new_axis_mask", &new_axis_mask_));
123 OP_REQUIRES_OK(ctx, ctx->GetAttr("shrink_axis_mask", &shrink_axis_mask_));
124 OP_REQUIRES_OK(ctx, ctx->GetAttr("Index", &index_type_));
125 }
126
127 void Compile(XlaOpKernelContext* ctx) override {
128 TensorShape processing_shape, final_shape;
129 gtl::InlinedVector<int64, 4> begin;
130 gtl::InlinedVector<int64, 4> end;
131 gtl::InlinedVector<int64, 4> strides;
132
133 TensorShape input_shape;
134 OP_REQUIRES_OK(ctx, ctx->ConstantInputAsShape(0, &input_shape));
135
136 xla::Literal begin_literal, end_literal, strides_literal;
137 OP_REQUIRES_OK(ctx, ctx->ConstantInput(1, &begin_literal));
138 OP_REQUIRES_OK(ctx, ctx->ConstantInput(2, &end_literal));
139 OP_REQUIRES_OK(ctx, ctx->ConstantInput(3, &strides_literal));
140
141 Tensor begin_tensor, end_tensor, strides_tensor;
142 OP_REQUIRES_OK(
143 ctx, LiteralToHostTensor(begin_literal, index_type_, &begin_tensor));
144 OP_REQUIRES_OK(ctx,
145 LiteralToHostTensor(end_literal, index_type_, &end_tensor));
146 OP_REQUIRES_OK(ctx, LiteralToHostTensor(strides_literal, index_type_,
147 &strides_tensor));
148
149 bool dummy = false;
150 OP_REQUIRES_OK(
151 ctx, ValidateStridedSliceOp(
152 &begin_tensor, &end_tensor, strides_tensor, input_shape,
153 begin_mask_, end_mask_, ellipsis_mask_, new_axis_mask_,
154 shrink_axis_mask_, &processing_shape, &final_shape, &dummy,
155 &dummy, &dummy, &begin, &end, &strides));
156
157 // Check to make sure dy is consistent with the original slice
158 const TensorShape dy_shape = ctx->InputShape(4);
159 OP_REQUIRES(
160 ctx, final_shape == dy_shape,
161 errors::InvalidArgument("shape of dy was ", dy_shape.DebugString(),
162 " instead of ", final_shape.DebugString()));
163
164 OP_REQUIRES(
165 ctx, input_shape.dims() == processing_shape.dims(),
166 errors::Internal(
167 "input shape and processing shape must have same number of dims"));
168
169 auto zero = XlaHelpers::Zero(ctx->builder(), ctx->expected_output_dtype(0));
170
171 xla::ComputationDataHandle grad = ctx->Input(4);
172
173 // Undo any new/shrink axes.
174 grad = ctx->builder()->Reshape(grad, processing_shape.dim_sizes());
175
176 // Pad the input gradients.
177 gtl::InlinedVector<int64, 4> dimensions_to_reverse;
178 xla::PaddingConfig padding_config;
179
180 for (int i = 0; i < processing_shape.dims(); ++i) {
181 auto* dims = padding_config.add_dimensions();
182 if (strides[i] > 0) {
183 dims->set_edge_padding_low(begin[i]);
184 dims->set_interior_padding(strides[i] - 1);
185
186 // Pad the upper dimension up to the expected input shape. (It's
187 // not sufficient simply to use "end[i]" to compute the padding in
188 // cases where the stride does not divide evenly into the interval
189 // between begin[i] and end[i].)
190 int64 size =
191 dims->edge_padding_low() + processing_shape.dim_size(i) +
192 (processing_shape.dim_size(i) - 1) * dims->interior_padding();
193 dims->set_edge_padding_high(input_shape.dim_size(i) - size);
194 } else {
195 dimensions_to_reverse.push_back(i);
196 dims->set_edge_padding_high(input_shape.dim_size(i) - begin[i] - 1);
197 dims->set_interior_padding(-strides[i] - 1);
198
199 // Pad the lower dimension up to the expected input shape.
200 int64 size =
201 dims->edge_padding_high() + processing_shape.dim_size(i) +
202 (processing_shape.dim_size(i) - 1) * dims->interior_padding();
203 dims->set_edge_padding_low(input_shape.dim_size(i) - size);
204 }
205 }
206 if (!dimensions_to_reverse.empty()) {
207 grad = ctx->builder()->Rev(grad, dimensions_to_reverse);
208 }
209 grad = ctx->builder()->Pad(grad, zero, padding_config);
210 ctx->SetOutput(0, grad);
211 }
212
213 private:
214 int32 begin_mask_, end_mask_;
215 int32 ellipsis_mask_, new_axis_mask_, shrink_axis_mask_;
216 DataType index_type_;
217};
218
219REGISTER_XLA_OP(Name("StridedSliceGrad")
220 .CompileTimeConstInput("shape")
221 .CompileTimeConstInput("begin")
222 .CompileTimeConstInput("end")
223 .CompileTimeConstInput("strides"),
224 StridedSliceGradOp);

226class StridedSliceAssignOp : public XlaOpKernel {
227 public:
228 explicit StridedSliceAssignOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {
229 OP_REQUIRES_OK(ctx, ctx->GetAttr("begin_mask", &begin_mask_));
230 OP_REQUIRES_OK(ctx, ctx->GetAttr("end_mask", &end_mask_));
231 OP_REQUIRES_OK(ctx, ctx->GetAttr("ellipsis_mask", &ellipsis_mask_));
232 OP_REQUIRES_OK(ctx, ctx->GetAttr("new_axis_mask", &new_axis_mask_));
233 OP_REQUIRES_OK(ctx, ctx->GetAttr("shrink_axis_mask", &shrink_axis_mask_));
234 OP_REQUIRES_OK(ctx, ctx->GetAttr("Index", &index_type_));
235 OP_REQUIRES_OK(ctx, ctx->GetAttr("T", &dtype_));
236 }
237
238 void Compile(XlaOpKernelContext* ctx) override {
239 TensorShape final_shape;
240 gtl::InlinedVector<int64, 4> begin;
241 gtl::InlinedVector<int64, 4> end;
242 gtl::InlinedVector<int64, 4> strides;
243
244 xla::Literal begin_literal, end_literal, strides_literal;
245 OP_REQUIRES_OK(ctx, ctx->ConstantInput(1, &begin_literal));
246 OP_REQUIRES_OK(ctx, ctx->ConstantInput(2, &end_literal));
247 OP_REQUIRES_OK(ctx, ctx->ConstantInput(3, &strides_literal));
248
249 Tensor begin_tensor, end_tensor, strides_tensor;
250 OP_REQUIRES_OK(
251 ctx, LiteralToHostTensor(begin_literal, index_type_, &begin_tensor));
252 OP_REQUIRES_OK(ctx,
253 LiteralToHostTensor(end_literal, index_type_, &end_tensor));
254 OP_REQUIRES_OK(ctx, LiteralToHostTensor(strides_literal, index_type_,
255 &strides_tensor));
256
257 TensorShape lhs_shape;
258 xla::ComputationDataHandle lhs;
259 OP_REQUIRES_OK(ctx, ctx->ReadVariableInput(0, dtype_, &lhs_shape, &lhs));
260
261 const TensorShape rhs_shape = ctx->InputShape(4);
262
263 TensorShape dummy_processing_shape;
264 bool dummy = false;
265 OP_REQUIRES_OK(ctx,
266 ValidateStridedSliceOp(
267 &begin_tensor, &end_tensor, strides_tensor, lhs_shape,
268 begin_mask_, end_mask_, ellipsis_mask_, new_axis_mask_,
269 shrink_axis_mask_, &dummy_processing_shape, &final_shape,
270 &dummy, &dummy, &dummy, &begin, &end, &strides));
271
272 if (final_shape.num_elements() == 0 && rhs_shape.num_elements() == 0) {
273 // DynamicUpdateSlice does not allow 0-element updates. We should probably
274 // check that rhs_shape can be broadcast to final_shape, but that is
275 // probably better handled when implementing broadcasting more generally.
276 return;
277 }
278
279 // TODO(aselle): This check is too strong, we only should need
280 // input_shape to be broadcastable to final_shape
281 OP_REQUIRES(ctx, final_shape == rhs_shape,
282 errors::Unimplemented(
283 "sliced l-value shape ", final_shape.DebugString(),
284 " does not match r-value shape ", rhs_shape.DebugString(),
285 ". Automatic broadcasting not yet implemented."));
286
287 xla::ComputationDataHandle rhs = ctx->Input(4);
288
289 gtl::InlinedVector<int64, 4> dimensions_to_reverse;
290 gtl::InlinedVector<int64, 4> slice_begin, slice_dims;
291 for (int i = 0; i < begin.size(); ++i) {
292 // TODO(phawkins): implement strides != 1
293 OP_REQUIRES(
294 ctx, strides[i] == 1 || strides[i] == -1,
295 errors::Unimplemented("Strides != 1 or -1 are not yet implemented"));
296 if (strides[i] > 0) {
297 slice_begin.push_back(begin[i]);
298 slice_dims.push_back(end[i] - begin[i]);
299 } else {
300 // Negative stride: swap begin and end, add 1 because the interval
301 // is semi-open, and mark the dimension to be reversed.
302 slice_begin.push_back(end[i] + 1);
303 slice_dims.push_back(begin[i] - end[i]);
304 dimensions_to_reverse.push_back(i);
305 }
306 }
307
308 if (!dimensions_to_reverse.empty()) {
309 rhs = ctx->builder()->Rev(rhs, dimensions_to_reverse);
310 }
311 rhs = ctx->builder()->Reshape(rhs, slice_dims);
312
313 if (lhs_shape.dims() == 0) {
314 // TODO(b/38323843): DynamicUpdateSlice crashes on rank 0 inputs. Fix
315 // and remove this workaround.
316 lhs = rhs;
317 } else {
318 lhs = ctx->builder()->DynamicUpdateSlice(
319 lhs, rhs, ctx->builder()->ConstantR1<int64>(slice_begin));
320 }
321
322 OP_REQUIRES_OK(ctx, ctx->AssignVariable(0, dtype_, lhs));
323 }
324
325 private:
326 int32 begin_mask_, end_mask_;
327 int32 ellipsis_mask_, new_axis_mask_, shrink_axis_mask_;
328 DataType index_type_;
329 DataType dtype_;
330};
331
332REGISTER_XLA_OP(Name("ResourceStridedSliceAssign")
333 .CompileTimeConstInput("begin")
334 .CompileTimeConstInput("end")
335 .CompileTimeConstInput("strides"),
336 StridedSliceAssignOp);
