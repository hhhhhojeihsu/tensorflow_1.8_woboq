59class ResourceApplyMomentum : public XlaOpKernel {
60 public:
61 explicit ResourceApplyMomentum(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {
62 OP_REQUIRES_OK(ctx, ctx->GetAttr("use_nesterov", &use_nesterov_));
63 }
64
65 void Compile(XlaOpKernelContext* ctx) override {
66 xla::ComputationBuilder* b = ctx->builder();
67
68 DataType type = ctx->input_type(2);
69
70 TensorShape var_shape, accum_shape;
71 xla::ComputationDataHandle var, accum;
72 OP_REQUIRES_OK(ctx, ctx->ReadVariableInput(0, type, &var_shape, &var));
73 OP_REQUIRES_OK(ctx, ctx->ReadVariableInput(1, type, &accum_shape, &accum));
74
75 OP_REQUIRES(ctx, var_shape.IsSameSize(accum_shape),
76 errors::InvalidArgument(
77 "var and accum do not have the same shape",
78 var_shape.DebugString(), " ", accum_shape.DebugString()));
79
80 TensorShape lr_shape = ctx->InputShape(2);
81 OP_REQUIRES(ctx, TensorShapeUtils::IsScalar(lr_shape),
82 errors::InvalidArgument("lr is not a scalar: ",
83 lr_shape.DebugString()));
84
85 TensorShape grad_shape = ctx->InputShape(3);
86 OP_REQUIRES(ctx, var_shape.IsSameSize(grad_shape),
87 errors::InvalidArgument(
88 "var and grad do not have the same shape",
89 var_shape.DebugString(), " ", grad_shape.DebugString()));
90
91 TensorShape momentum_shape = ctx->InputShape(4);
92 OP_REQUIRES(ctx, TensorShapeUtils::IsScalar(momentum_shape),
93 errors::InvalidArgument("momentum is not a scalar: ",
94 momentum_shape.DebugString()));
95
96 xla::ComputationDataHandle lr = ctx->Input(2);
97 xla::ComputationDataHandle grad = ctx->Input(3);
98 xla::ComputationDataHandle momentum = ctx->Input(4);
99
100 accum = b->Add(b->Mul(accum, momentum), grad);
101 if (use_nesterov_) {
102 // See https://github.com/tensorflow/tensorflow/pull/2798 for an
103 // explanation of the reparameterization used here.
104 var = b->Sub(
105 var, b->Add(b->Mul(grad, lr), b->Mul(b->Mul(accum, momentum), lr)));
106 } else {
107 var = b->Sub(var, b->Mul(accum, lr));
108 }
109 OP_REQUIRES_OK(ctx, ctx->AssignVariable(0, type, var));
110 OP_REQUIRES_OK(ctx, ctx->AssignVariable(1, type, accum));
111 }
112
113 private:
114 bool use_nesterov_;
115};
116REGISTER_XLA_OP(Name("ResourceApplyMomentum").TypeConstraint("T", kFloatTypes),
117 ResourceApplyMomentum);

157 OP_REQUIRES_OK(ctx, ctx->AssignVariable(1, type, accum));

163class ResourceApplyAdam : public XlaOpKernel {
164 public:
165 explicit ResourceApplyAdam(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {
166 OP_REQUIRES_OK(ctx, ctx->GetAttr("T", &dtype_));
167 }
168
169 void Compile(XlaOpKernelContext* ctx) override {
170 TensorShape var_shape, m_shape, v_shape;
171 xla::ComputationDataHandle var, m, v;
172 OP_REQUIRES_OK(ctx, ctx->ReadVariableInput(0, dtype_, &var_shape, &var));
173 OP_REQUIRES_OK(ctx, ctx->ReadVariableInput(1, dtype_, &m_shape, &m));
174 OP_REQUIRES_OK(ctx, ctx->ReadVariableInput(2, dtype_, &v_shape, &v));
175
176 TensorShape beta1_power_shape = ctx->InputShape(3);
177 TensorShape beta2_power_shape = ctx->InputShape(4);
178 TensorShape lr_shape = ctx->InputShape(5);
179 TensorShape beta1_shape = ctx->InputShape(6);
180 TensorShape beta2_shape = ctx->InputShape(7);
181 TensorShape epsilon_shape = ctx->InputShape(8);
182 TensorShape grad_shape = ctx->InputShape(9);
183
184 OP_REQUIRES(ctx, TensorShapeUtils::IsScalar(beta1_power_shape),
185 errors::InvalidArgument("beta1_power is not a scalar: ",
186 beta1_power_shape.DebugString()));
187 OP_REQUIRES(ctx, TensorShapeUtils::IsScalar(beta2_power_shape),
188 errors::InvalidArgument("beta2_power is not a scalar: ",
189 beta2_power_shape.DebugString()));
190 OP_REQUIRES(ctx, TensorShapeUtils::IsScalar(lr_shape),
191 errors::InvalidArgument("lr is not a scalar : ",
192 lr_shape.DebugString()));
193 OP_REQUIRES(ctx, TensorShapeUtils::IsScalar(beta1_shape),
194 errors::InvalidArgument("beta1 is not a scalar: ",
195 beta1_shape.DebugString()));
196 OP_REQUIRES(ctx, TensorShapeUtils::IsScalar(beta2_shape),
197 errors::InvalidArgument("beta2 is not a scalar: ",
198 beta2_shape.DebugString()));
199 OP_REQUIRES(ctx, TensorShapeUtils::IsScalar(epsilon_shape),
200 errors::InvalidArgument("epsilon is not a scalar: ",
201 epsilon_shape.DebugString()));
202
203 OP_REQUIRES(ctx, var_shape.IsSameSize(m_shape),
204 errors::InvalidArgument("var and m do not have the same shape",
205 var_shape.DebugString(), " ",
206 m_shape.DebugString()));
207 OP_REQUIRES(ctx, var_shape.IsSameSize(v_shape),
208 errors::InvalidArgument("var and v do not have the same shape",
209 var_shape.DebugString(), " ",
210 v_shape.DebugString()));
211 OP_REQUIRES(ctx, var_shape.IsSameSize(grad_shape),
212 errors::InvalidArgument(
213 "var and grad do not have the same shape",
214 var_shape.DebugString(), " ", grad_shape.DebugString()));
215
216 xla::ComputationDataHandle beta1_power = ctx->Input(3);
217 xla::ComputationDataHandle beta2_power = ctx->Input(4);
218 xla::ComputationDataHandle lr = ctx->Input(5);
219 xla::ComputationDataHandle beta1 = ctx->Input(6);
220 xla::ComputationDataHandle beta2 = ctx->Input(7);
221 xla::ComputationDataHandle epsilon = ctx->Input(8);
222 xla::ComputationDataHandle grad = ctx->Input(9);
223
224 // alpha <- learning_rate * sqrt(1 - beta2^t) / (1 - beta1^t)
225 // m_t <- beta1 * m_{t-1} + (1 - beta1) * g_t
226 // v_t <- beta2 * v_{t-1} + (1 - beta2) * g_t * g_t
227 // variable <- variable - alpha * m_t / (sqrt(v_t) + epsilon)
228
229 xla::ComputationBuilder* b = ctx->builder();
230 xla::ComputationDataHandle half = XlaHelpers::FloatLiteral(b, dtype_, 0.5);
231 xla::ComputationDataHandle one = XlaHelpers::FloatLiteral(b, dtype_, 1.0);
232 xla::ComputationDataHandle two = XlaHelpers::FloatLiteral(b, dtype_, 2.0);
233
234 xla::ComputationDataHandle alpha =
235 b->Div(b->Mul(lr, b->Pow(b->Sub(one, beta2_power), half)),
236 b->Sub(one, beta1_power));
237 m = b->Add(m, b->Mul(b->Sub(grad, m), b->Sub(one, beta1)));
238 v = b->Add(v, b->Mul(b->Sub(b->Pow(grad, two), v), b->Sub(one, beta2)));
239 var =
240 b->Sub(var, b->Div(b->Mul(m, alpha), b->Add(b->Pow(v, half), epsilon)));
241
242 OP_REQUIRES_OK(ctx, ctx->AssignVariable(0, dtype_, var));
243 OP_REQUIRES_OK(ctx, ctx->AssignVariable(1, dtype_, m));
244 OP_REQUIRES_OK(ctx, ctx->AssignVariable(2, dtype_, v));
245 }
246
247 private:
248 DataType dtype_;
249};
250REGISTER_XLA_OP(Name("ResourceApplyAdam").TypeConstraint("T", kFloatTypes),
251 ResourceApplyAdam);

253class ResourceApplyRMSProp : public XlaOpKernel {
254 public:
255 explicit ResourceApplyRMSProp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {}
256
257 void Compile(XlaOpKernelContext* ctx) override {
258 xla::ComputationBuilder* b = ctx->builder();
259
260 DataType type = ctx->input_type(3);
261
262 TensorShape var_shape, ms_shape, mom_shape;
263 xla::ComputationDataHandle var, ms, mom;
264 OP_REQUIRES_OK(ctx, ctx->ReadVariableInput(0, type, &var_shape, &var));
265 OP_REQUIRES_OK(ctx, ctx->ReadVariableInput(1, type, &ms_shape, &ms));
266 OP_REQUIRES_OK(ctx, ctx->ReadVariableInput(2, type, &mom_shape, &mom));
267
268 TensorShape lr_shape = ctx->InputShape(3);
269 OP_REQUIRES(ctx, TensorShapeUtils::IsScalar(lr_shape),
270 errors::InvalidArgument("lr is not a scalar: ",
271 lr_shape.DebugString()));
272 TensorShape rho_shape = ctx->InputShape(4);
273 OP_REQUIRES(ctx, TensorShapeUtils::IsScalar(rho_shape),
274 errors::InvalidArgument("rho is not a scalar: ",
275 rho_shape.DebugString()));
276 TensorShape momentum_shape = ctx->InputShape(5);
277 OP_REQUIRES(ctx, TensorShapeUtils::IsScalar(momentum_shape),
278 errors::InvalidArgument("momentum is not a scalar: ",
279 momentum_shape.DebugString()));
280 TensorShape epsilon_shape = ctx->InputShape(6);
281 OP_REQUIRES(ctx, TensorShapeUtils::IsScalar(epsilon_shape),
282 errors::InvalidArgument("epsilon is not a scalar: ",
283 epsilon_shape.DebugString()));
284 TensorShape grad_shape = ctx->InputShape(7);
285
286 // var should be the same shape as mom and ms.
287 OP_REQUIRES(ctx, var_shape.IsSameSize(ms_shape),
288 errors::InvalidArgument("var and ms do not have the same shape",
289 var_shape.DebugString(), " ",
290 ms_shape.DebugString()));
291 OP_REQUIRES(ctx, var_shape.IsSameSize(mom_shape),
292 errors::InvalidArgument(
293 "var and mom do not have the same shape",
294 var_shape.DebugString(), " ", mom_shape.DebugString()));
295 OP_REQUIRES(ctx, var_shape.IsSameSize(grad_shape),
296 errors::InvalidArgument(
297 "var and grad do not have the same shape",
298 var_shape.DebugString(), " ", grad_shape.DebugString()));
299
300 xla::ComputationDataHandle lr = ctx->Input(3);
301 xla::ComputationDataHandle rho = ctx->Input(4);
302 xla::ComputationDataHandle momentum = ctx->Input(5);
303 xla::ComputationDataHandle epsilon = ctx->Input(6);
304 xla::ComputationDataHandle grad = ctx->Input(7);
305
306 // ms <- rho * ms_{t-1} + (1-rho) * grad * grad
307 // mom <- momentum * mom_{t-1} + lr * grad / sqrt(ms + epsilon)
308 // var <- var - mom
309 //
310 // We use an alternate formulation of the ms equation:
311 //
312 // ms <- ms + (grad**2 - ms) * (1 - rho)
313 //
314 // Which expands to:
315 //
316 // ms <- ms + grad**2 - rho * grad ** 2 - ms + ms * rho
317 //
318 // Which simplifies to:
319 //
320 // ms <- grad**2 (1 - rho) + ms * rho
321 //
322 // Which is the equation listed above.
323 xla::ComputationDataHandle new_ms = b->Add(
324 ms,
325 b->Mul(b->Sub(b->Pow(grad, XlaHelpers::FloatLiteral(b, type, 2.0)), ms),
326 b->Sub(XlaHelpers::FloatLiteral(b, type, 1.0), rho)));
327 xla::ComputationDataHandle new_mom =
328 b->Add(b->Mul(mom, momentum),
329 b->Mul(b->Mul(grad, lr),
330 b->Pow(b->Add(new_ms, epsilon),
331 XlaHelpers::FloatLiteral(b, type, -0.5))));
332 xla::ComputationDataHandle new_var = b->Sub(var, new_mom);
333
334 OP_REQUIRES_OK(ctx, ctx->AssignVariable(0, type, new_var));
335 OP_REQUIRES_OK(ctx, ctx->AssignVariable(1, type, new_ms));
336 OP_REQUIRES_OK(ctx, ctx->AssignVariable(2, type, new_mom));
337 }
338};
339REGISTER_XLA_OP(Name("ResourceApplyRMSProp").TypeConstraint("T", kFloatTypes),
340 ResourceApplyRMSProp);

342void CompileFtrl(XlaOpKernelContext* ctx, DataType dtype,
343 bool has_l2_shrinkage) {
344 xla::ComputationBuilder* b = ctx->builder();
345
346 TensorShape var_shape, accum_shape, linear_shape;
347 xla::ComputationDataHandle var, accum, linear;
348 OP_REQUIRES_OK(ctx, ctx->ReadVariableInput(0, dtype, &var_shape, &var));
349 OP_REQUIRES_OK(ctx, ctx->ReadVariableInput(1, dtype, &accum_shape, &accum));
350 OP_REQUIRES_OK(ctx, ctx->ReadVariableInput(2, dtype, &linear_shape, &linear));
351
352 OP_REQUIRES(ctx, var_shape.IsSameSize(accum_shape),
353 errors::InvalidArgument(
354 "var and accum do not have the same shape",
355 var_shape.DebugString(), " ", accum_shape.DebugString()));
356
357 OP_REQUIRES(ctx, var_shape.IsSameSize(linear_shape),
358 errors::InvalidArgument(
359 "var and linear do not have the same shape",
360 var_shape.DebugString(), " ", linear_shape.DebugString()));
361
362 TensorShape grad_shape = ctx->InputShape(3);
363 TensorShape lr_shape = ctx->InputShape(4);
364 TensorShape l1_shape = ctx->InputShape(5);
365 TensorShape l2_shape = ctx->InputShape(6);
366 TensorShape l2_shrinkage_shape;
367 TensorShape lr_power_shape;
368 if (has_l2_shrinkage) {
369 l2_shrinkage_shape = ctx->InputShape(7);
370 lr_power_shape = ctx->InputShape(8);
371 } else {
372 lr_power_shape = ctx->InputShape(7);
373 }
374
375 OP_REQUIRES(ctx, var_shape.IsSameSize(grad_shape),
376 errors::InvalidArgument("var and grad do not have the same shape",
377 var_shape.DebugString(), " ",
378 grad_shape.DebugString()));
379
380 OP_REQUIRES(
381 ctx, TensorShapeUtils::IsScalar(lr_shape),
382 errors::InvalidArgument("lr is not a scalar: ", lr_shape.DebugString()));
383
384 OP_REQUIRES(
385 ctx, TensorShapeUtils::IsScalar(l1_shape),
386 errors::InvalidArgument("l1 is not a scalar: ", l1_shape.DebugString()));
387
388 OP_REQUIRES(
389 ctx, TensorShapeUtils::IsScalar(l2_shape),
390 errors::InvalidArgument("l2 is not a scalar: ", l2_shape.DebugString()));
391
392 if (has_l2_shrinkage) {
393 OP_REQUIRES(ctx, TensorShapeUtils::IsScalar(l2_shrinkage_shape),
394 errors::InvalidArgument("l2_shrinkage is not a scalar: ",
395 l2_shrinkage_shape.DebugString()));
396 }
397
398 OP_REQUIRES(ctx, TensorShapeUtils::IsScalar(lr_power_shape),
399 errors::InvalidArgument("lr_power is not a scalar: ",
400 lr_power_shape.DebugString()));
401
402 xla::ComputationDataHandle grad = ctx->Input(3);
403 xla::ComputationDataHandle lr = ctx->Input(4);
404 xla::ComputationDataHandle l1 = ctx->Input(5);
405 xla::ComputationDataHandle l2 = ctx->Input(6);
406 xla::ComputationDataHandle l2_shrinkage;
407 xla::ComputationDataHandle lr_power;
408 if (has_l2_shrinkage) {
409 l2_shrinkage = ctx->Input(7);
410 lr_power = ctx->Input(8);
411 } else {
412 lr_power = ctx->Input(7);
413 }
414
415 // grad_to_use = grad + 2 * l2_shrinkage * var
416 // new_accum = accum + grad_to_use * grad_to_use
417 // linear += grad_to_use -
418 // (new_accum^(-lr_power) - accum^(-lr_power)) / lr * var
419 // quadratic = (new_accum^(-lr_power) / lr) + 2 * l2
420 // linear_clipped = clamp linear in [-l1, l1]
421 // var = (linear_clipped - linear) / quadratic
422 // accum = new_accum
423
424 xla::ComputationDataHandle two = XlaHelpers::FloatLiteral(b, dtype, 2.0);
425 xla::ComputationDataHandle grad_to_use;
426 if (has_l2_shrinkage) {
427 grad_to_use = b->Add(grad, b->Mul(two, b->Mul(l2_shrinkage, var)));
428 } else {
429 grad_to_use = grad;
430 }
431
432 xla::ComputationDataHandle new_accum =
433 b->Add(accum, b->Pow(grad_to_use, two));
434 xla::ComputationDataHandle new_accum_lr_pow =
435 b->Pow(new_accum, b->Neg(lr_power));
436 xla::ComputationDataHandle accum_lr_pow = b->Pow(accum, b->Neg(lr_power));
437 linear = b->Add(
438 linear,
439 b->Sub(grad_to_use,
440 b->Mul(b->Div(b->Sub(new_accum_lr_pow, accum_lr_pow), lr), var)));
441 xla::ComputationDataHandle linear_clipped = b->Clamp(b->Neg(l1), linear, l1);
442 xla::ComputationDataHandle quadratic =
443 b->Add(b->Div(new_accum_lr_pow, lr), b->Mul(two, l2));
444 var = b->Div(b->Sub(linear_clipped, linear), quadratic);
445 accum = new_accum;
446
447 OP_REQUIRES_OK(ctx, ctx->AssignVariable(0, dtype, var));
448 OP_REQUIRES_OK(ctx, ctx->AssignVariable(1, dtype, accum));
449 OP_REQUIRES_OK(ctx, ctx->AssignVariable(2, dtype, linear));
450}
