28// A subclass of a TlaUnaryOp must build the lambda computation that
29// describes the scalar->scalar function to apply to each element of
30// the input.
31#define XLAJIT_MAKE_UNARY(NAME, COMPUTATION) \
32 class NAME##Op : public XlaOpKernel { \
33 public: \
34 explicit NAME##Op(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {} \
35 void Compile(XlaOpKernelContext* ctx) { \
36 xla::ComputationBuilder* b = ctx->builder(); \
37 xla::ComputationDataHandle x = ctx->Input(0); \
38 xla::ComputationDataHandle y = COMPUTATION; \
39 ctx->SetOutput(0, y); \
40 } \
41 }; \
42 REGISTER_XLA_OP(Name(#NAME), NAME##Op);

50// Return x if x>0, otherwise -x.
51XLAJIT_MAKE_UNARY(Abs, b->Abs(x));

53// acos(x) = 2 * atan(sqrt(1 - x^2) / (1 + x))
54XLAJIT_MAKE_UNARY(
55 Acos,
56 b->Mul(XlaHelpers::FloatLiteral(b, input_type(0), 2.0),
57 b->Atan2(b->Pow(b->Sub(XlaHelpers::One(b, input_type(0)),
58 b->Mul(x, x)),
59 XlaHelpers::FloatLiteral(b, input_type(0), 0.5)),
60 b->Add(XlaHelpers::One(b, input_type(0)), x))));
61
62// acosh(x) = log(x + sqrt(x^2 - 1))
63// = log(x + sqrt((x+1)*(x-1)))
64XLAJIT_MAKE_UNARY(
65 Acosh,
66 b->Log(b->Add(x,
67 b->Pow(b->Mul(b->Add(x, XlaHelpers::One(b, input_type(0))),
68 b->Sub(x, XlaHelpers::One(b, input_type(0)))),
69 XlaHelpers::FloatLiteral(b, input_type(0), 0.5)))));
70
71// asin(x) = 2 * atan(x / (1 + sqrt(1 - x^2)))
72XLAJIT_MAKE_UNARY(
73 Asin,
74 b->Mul(XlaHelpers::FloatLiteral(b, input_type(0), 2.0),
75 b->Atan2(x, b->Add(XlaHelpers::One(b, input_type(0)),
76 b->Pow(b->Sub(XlaHelpers::One(b, input_type(0)),
77 b->Mul(x, x)),
78 XlaHelpers::FloatLiteral(b, input_type(0),
79 0.5))))));
80
81// asinh(x) = log(x + sqrt(x^2 + 1))
82XLAJIT_MAKE_UNARY(
83 Asinh,
84 b->Log(b->Add(x, b->Pow(b->Add(b->Mul(x, x),
85 XlaHelpers::One(b, input_type(0))),
86 XlaHelpers::FloatLiteral(b, input_type(0), 0.5)))));
87
88XLAJIT_MAKE_UNARY(Atan, b->Atan2(x, XlaHelpers::One(b, input_type(0))));
89
90// atanh(x) = 0.5 * log((1 + x) / (1 - x))
91XLAJIT_MAKE_UNARY(
92 Atanh, b->Mul(b->Log(b->Div(b->Add(XlaHelpers::One(b, input_type(0)), x),
93 b->Sub(XlaHelpers::One(b, input_type(0)), x))),
94 XlaHelpers::FloatLiteral(b, input_type(0), 0.5)));
95XLAJIT_MAKE_UNARY(Ceil, b->Ceil(x));
96XLAJIT_MAKE_UNARY(Cos, b->Cos(x));
97XLAJIT_MAKE_UNARY(Cosh,
98 b->Mul(b->Add(b->Exp(x), b->Exp(b->Neg(x))),
99 XlaHelpers::FloatLiteral(b, input_type(0), 0.5)));
100XLAJIT_MAKE_UNARY(Sin, b->Sin(x));
101XLAJIT_MAKE_UNARY(Exp, b->Exp(x));

104XLAJIT_MAKE_UNARY(Expm1, b->Sub(b->Exp(x), XlaHelpers::One(b, input_type(0))));

107XLAJIT_MAKE_UNARY(IsFinite, b->IsFinite(x));
108XLAJIT_MAKE_UNARY(IsInf, b->Eq(b->Abs(x),
109 XlaHelpers::FloatLiteral(
110 b, input_type(0),
111 std::numeric_limits<double>::infinity())));
112XLAJIT_MAKE_UNARY(IsNan, b->Ne(x, x));
113// Return 1/x
114XLAJIT_MAKE_UNARY(Inv, b->Div(XlaHelpers::One(b, input_type(0)), x));

125// Implements Banker's rounding: numbers that are equidistant between two
126// integers are rounded towards even.
127static xla::ComputationDataHandle Round(xla::ComputationBuilder* b,
128 DataType dtype,
129 const xla::ComputationDataHandle& x) {
130 auto half = XlaHelpers::FloatLiteral(b, dtype, 0.5);
131 auto one = XlaHelpers::FloatLiteral(b, dtype, 1.0);
132 auto two = XlaHelpers::FloatLiteral(b, dtype, 2.0);
133
134 auto round_val = b->Floor(x);
135 auto fraction = b->Sub(x, round_val);
136 auto nearest_even_int =
137 b->Sub(round_val, b->Mul(two, b->Floor(b->Mul(half, x))));
138 auto is_odd = b->Eq(nearest_even_int, one);
139 return b->Select(
140 b->Or(b->Gt(fraction, half), b->And(b->Eq(fraction, half), is_odd)),
141 b->Add(round_val, one), round_val);
142}
143
144XLAJIT_MAKE_UNARY(Rint, Round(b, input_type(0), x));
145XLAJIT_MAKE_UNARY(Round, Round(b, input_type(0), x));
146
147XLAJIT_MAKE_UNARY(Rsqrt,
148 b->Pow(x, XlaHelpers::FloatLiteral(b, input_type(0), -0.5)));

150// Expresses sigmoid as a rescaled tanh: sigmoid(x) == (tanh(x/2) + 1) / 2.
151static xla::ComputationDataHandle Sigmoid(xla::ComputationBuilder* b,
152 DataType dtype,
153 const xla::ComputationDataHandle& x) {
154 auto half = XlaHelpers::FloatLiteral(b, dtype, 0.5);
155 return b->Add(half, b->Mul(half, b->Tanh(b->Mul(half, x))));
156}
157XLAJIT_MAKE_UNARY(Sigmoid, Sigmoid(b, input_type(0), x));

162 b->Mul(b->Sub(b->Exp(x), b->Exp(b->Neg(x))),
163 XlaHelpers::FloatLiteral(b, input_type(0), 0.5)));

165static xla::ComputationDataHandle Softplus(
166 xla::ComputationBuilder* b, DataType dtype,
167 const xla::ComputationDataHandle& features) {
168 xla::ComputationDataHandle threshold =
169 b->Add(b->Log(XlaHelpers::Epsilon(b, dtype)),
170 XlaHelpers::FloatLiteral(b, dtype, 2.0));
171 // Value above which exp(x) may overflow, but softplus(x) == x
172 // is within machine epsilon.
173 xla::ComputationDataHandle too_large = b->Gt(features, b->Neg(threshold));
174 // Value below which exp(x) may underflow, but softplus(x) == exp(x)
175 // is within machine epsilon.
176 xla::ComputationDataHandle too_small = b->Lt(features, threshold);
177 xla::ComputationDataHandle features_exp = b->Exp(features);
178 xla::ComputationDataHandle output = b->Select(
179 too_large, features,
180 b->Select(too_small, features_exp,
181 b->Log(b->Add(features_exp, XlaHelpers::One(b, dtype)))));
182 return output;
183}
184XLAJIT_MAKE_UNARY(Softplus, Softplus(b, input_type(0), x));
