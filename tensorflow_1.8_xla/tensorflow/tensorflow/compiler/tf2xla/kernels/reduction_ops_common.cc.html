27XlaReductionOp::XlaReductionOp(OpKernelConstruction* ctx,
28 DataType reduction_type)
29 : XlaOpKernel(ctx), reduction_type_(reduction_type) {
30 const DataType dt = BaseType(input_type(0));
31 OP_REQUIRES_OK(ctx, ctx->MatchSignature({dt, DT_INT32}, {dt}));
32
33 OP_REQUIRES_OK(ctx, ctx->GetAttr("keep_dims", &keep_dims_));
34}
35
36// Unless BuildFinalizer is overridden the reduction has no
37// finalizer.
38xla::ComputationDataHandle XlaReductionOp::BuildFinalizer(
39 xla::ComputationBuilder* builder,
40 const xla::ComputationDataHandle& reduce_output,
41 int64 num_elements_reduced) {
42 return reduce_output;
43}

45void XlaReductionOp::Compile(XlaOpKernelContext* ctx) {
46 const TensorShape data_shape = ctx->InputShape(0);
47 const TensorShape axes_tensor_shape = ctx->InputShape(1);
48 VLOG(1) << "ReductionOp: " << ctx->op_kernel().name();
49
50 if (axes_tensor_shape.num_elements() == 0) {
51 // The reduction axes is an empty vector, which means there are no
52 // axes to reduce so just pass the input directly through to the
53 // output.
54 ctx->SetOutput(0, ctx->Input(0));
55 return;
56 }
57
58 // Evaluate the constant, reshaping to a 1-vector if it is a scalar.
59 xla::Literal axes_literal;
60 OP_REQUIRES_OK(ctx,
61 ctx->ConstantInputReshaped(
62 1, {axes_tensor_shape.num_elements()}, &axes_literal));
63
64 VLOG(1) << "data shape: " << data_shape.DebugString();
65 VLOG(1) << "axes : " << axes_literal.ToString();
66
67 gtl::InlinedVector<bool, 4> bitmap(data_shape.dims(), false);
68 std::vector<int64> xla_axes;
69 int64 num_elements_reduced = 1LL;
70 for (int64 i = 0; i < axes_tensor_shape.num_elements(); ++i) {
71 int32 index = axes_literal.Get<int>({i});
72 OP_REQUIRES(ctx,
73 !(index < -data_shape.dims() || index >= data_shape.dims()),
74 errors::InvalidArgument("Invalid reduction dimension (", index,
75 " for input with ", data_shape.dims(),
76 " dimension(s)"));
77 index = (index + data_shape.dims()) % data_shape.dims();
78 bitmap[index] = true;
79 xla_axes.push_back(index);
80 num_elements_reduced *= data_shape.dim_size(index);
81 }
82
83 std::vector<int64> final_shape;
84 for (int i = 0; i < data_shape.dims(); ++i) {
85 if (!bitmap[i]) {
86 // If we are not reducing along dimension i.
87 int64 dim = data_shape.dim_size(i);
88 final_shape.push_back(dim);
89 } else if (keep_dims_) {
90 // We are reducing along dimension i, but we want to keep the
91 // same number of dimensions, so we set the dimension of i to
92 // '1'.
93 final_shape.push_back(1);
94 }
95 }
96
97 string desc = ctx->op_kernel().name();
98
99 xla::ComputationBuilder* const b = ctx->builder();
100 // Construct the builder for the reduction lambda.
101 xla::ComputationBuilder r(b->client(), strings::StrCat(desc, "-reduction"));
102 xla::PrimitiveType type;
103 TF_CHECK_OK(DataTypeToPrimitiveType(reduction_type_, &type));
104
105 auto data = b->ConvertElementType(ctx->Input(0), type);
106 // Call virtual method to get the initial value.
107 auto initial = b->ConvertElementType(InitialValue(b), type);
108 // Make two scalar parameters of the desired type for the lambda.
109 auto rx = r.Parameter(0, xla::ShapeUtil::MakeShape(type, {}), "x");
110 auto ry = r.Parameter(1, xla::ShapeUtil::MakeShape(type, {}), "y");
111 // Call virtual method to build the reduction lambda.
112 BuildReducer(&r, rx, ry);
113 xla::Computation reduction_computation = r.Build().ConsumeValueOrDie();
114
115 auto reduce = b->Reduce(data, initial, reduction_computation, xla_axes);
116 auto deconverted = XlaHelpers::ConvertElementType(b, reduce, input_type(0));
117 auto finalized = BuildFinalizer(b, deconverted, num_elements_reduced);
118 auto result = keep_dims_ ? b->Reshape(finalized, final_shape) : finalized;
119 ctx->SetOutput(0, result);
120}
