38// TODO(phawkins): implement double-sized windowed reductions in XLA and remove
39// the type constraint.
40constexpr std::array<DataType, 3> kScanOpTypes = {
41 {DT_HALF, DT_BFLOAT16, DT_FLOAT}};
42
43class ScanOp : public XlaOpKernel {
44 public:
45 ScanOp(OpKernelConstruction* ctx, bool sum) : XlaOpKernel(ctx), sum_(sum) {
46 OP_REQUIRES_OK(ctx, ctx->GetAttr("reverse", &reverse_));
47 OP_REQUIRES_OK(ctx, ctx->GetAttr("exclusive", &exclusive_));
48 }
49
50 void Compile(XlaOpKernelContext* ctx) override {
51 const TensorShape input_shape = ctx->InputShape(0);
52 const TensorShape tensor_axis_shape = ctx->InputShape(1);
53
54 OP_REQUIRES(ctx, TensorShapeUtils::IsScalar(tensor_axis_shape),
55 errors::InvalidArgument("ScanOp: axis must be a scalar, not ",
56 tensor_axis_shape.DebugString()));
57
58 int64 axis;
59 OP_REQUIRES_OK(ctx, ctx->ConstantInputAsIntScalar(1, &axis));
60 if (axis < 0) {
61 axis += input_shape.dims();
62 }
63 OP_REQUIRES(
64 ctx, FastBoundsCheck(axis, input_shape.dims()),
65 errors::InvalidArgument("ScanOp: Expected scan axis in the range [",
66 -input_shape.dims(), ", ", input_shape.dims(),
67 "), but got ", axis));
68
69 DataType dtype = XlaHelpers::SumAccumulationType(ctx->input_type(0));
70
71 if (input_shape.num_elements() == 0) {
72 // Exit early if there is nothing to compute.
73 ctx->SetOutput(0, ctx->Input(0));
74 return;
75 }
76
77 xla::ComputationBuilder* builder = ctx->builder();
78
79 std::vector<int64> window_strides(input_shape.dims(), 1);
80 std::vector<int64> window_dims(input_shape.dims(), 1);
81 window_dims[axis] = input_shape.dim_size(axis);
82
83 std::vector<std::pair<int64, int64>> padding(input_shape.dims(), {0, 0});
84 padding[axis].first = input_shape.dim_size(axis) - 1;
85 // In exclusive mode, add an extra padding element so there is a complete
86 // window of padding before the data starts.
87 if (exclusive_) {
88 ++padding[axis].first;
89 }
90 if (reverse_) {
91 std::swap(padding[axis].first, padding[axis].second);
92 }
93
94 xla::ComputationDataHandle init;
95 const xla::Computation* reducer;
96 if (sum_) {
97 init = XlaHelpers::Zero(builder, dtype);
98 reducer = ctx->GetOrCreateAdd(dtype);
99 } else {
100 init = XlaHelpers::One(builder, dtype);
101 reducer = ctx->GetOrCreateMul(dtype);
102 }
103 auto output = builder->ReduceWindowWithGeneralPadding(
104 XlaHelpers::ConvertElementType(builder, ctx->Input(0), dtype), init,
105 *reducer, window_dims, window_strides, padding);
106 output =
107 XlaHelpers::ConvertElementType(builder, output, ctx->input_type(0));
108
109 // In exclusive mode, we have computed an extra element containing the sum
110 // of all the input elements. Slice off this extra "last" element.
111 if (exclusive_) {
112 if (reverse_) {
113 output = builder->SliceInDim(output, 1, input_shape.dim_size(axis) + 1,
114 1, axis);
115
116 } else {
117 output =
118 builder->SliceInDim(output, 0, input_shape.dim_size(axis), 1, axis);
119 }
120 }
121 ctx->SetOutput(0, output);
122 }
123
124 private:
125 const bool sum_; // True=cumulative sum. False=cumulative product.
126 bool reverse_;
127 bool exclusive_;
128};
129
130class CumsumOp : public ScanOp {
131 public:
132 explicit CumsumOp(OpKernelConstruction* ctx) : ScanOp(ctx, /*sum=*/true) {}
133};
134REGISTER_XLA_OP(Name("Cumsum")
135 .TypeConstraint("T", kScanOpTypes)
136 .CompileTimeConstInput("axis"),
137 CumsumOp);
138
139class CumprodOp : public ScanOp {
140 public:
141 explicit CumprodOp(OpKernelConstruction* ctx) : ScanOp(ctx, /*sum=*/false) {}
142};
143REGISTER_XLA_OP(Name("Cumprod")
144 .TypeConstraint("T", kScanOpTypes)
145 .CompileTimeConstInput("axis"),
146 CumprodOp);
