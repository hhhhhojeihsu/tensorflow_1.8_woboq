24// Local response normalization
25class LRNOp : public XlaOpKernel {
26 public:
27 explicit LRNOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {
28 OP_REQUIRES_OK(ctx, ctx->GetAttr("depth_radius", &depth_radius_));
29
30 // TODO(phawkins): handle non-float types for attributes.
31 OP_REQUIRES_OK(ctx, ctx->GetAttr("bias", &bias_));
32 OP_REQUIRES_OK(ctx, ctx->GetAttr("alpha", &alpha_));
33 OP_REQUIRES_OK(ctx, ctx->GetAttr("beta", &beta_));
34 }
35
36 void Compile(XlaOpKernelContext* ctx) override {
37 const TensorShape in_shape = ctx->InputShape(0);
38 OP_REQUIRES(ctx, in_shape.dims() == 4,
39 errors::InvalidArgument("in must be 4-dimensional"));
40
41 xla::ComputationBuilder* builder = ctx->builder();
42 xla::ComputationDataHandle input = ctx->Input(0);
43
44 // sqr_sum[a, b, c, d] =
45 // sum(input[a, b, c, d - depth_radius : d + depth_radius + 1] ** 2)
46 // output = input / (bias + alpha * sqr_sum) ** beta
47
48 // We use a window of depth_radius_ * 2 + 1, to account for the current
49 // element and a depth_radius_ on either side.
50 auto accumulation_type = XlaHelpers::SumAccumulationType(input_type(0));
51 auto converted =
52 XlaHelpers::ConvertElementType(builder, input, accumulation_type);
53 auto squared = builder->Mul(converted, converted);
54 auto reduce = builder->ReduceWindow(
55 squared, XlaHelpers::Zero(builder, accumulation_type),
56 *ctx->GetOrCreateAdd(accumulation_type),
57 /* window_dimensions = */ {1, 1, 1, depth_radius_ * 2 + 1},
58 /* window_strides = */ {1, 1, 1, 1}, xla::Padding::kSame);
59 auto sqr_sum =
60 XlaHelpers::ConvertElementType(builder, reduce, input_type(0));
61
62 auto scale = builder->Pow(
63 builder->Add(builder->ConstantR0<float>(bias_),
64 builder->Mul(builder->ConstantR0<float>(alpha_), sqr_sum)),
65 builder->ConstantR0<float>(-beta_));
66
67 ctx->SetOutput(0, builder->Mul(input, scale));
68 }
69
70 private:
71 int64 depth_radius_;
72 float bias_;
73 float alpha_;
74 float beta_;
75};
76
77REGISTER_XLA_OP(Name("LRN"), LRNOp);

79class LRNGradOp : public XlaOpKernel {
80 public:
81 explicit LRNGradOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {
82 OP_REQUIRES_OK(ctx, ctx->GetAttr("depth_radius", &depth_radius_));
83
84 // TODO(phawkins): handle non-float types for attributes.
85 OP_REQUIRES_OK(ctx, ctx->GetAttr("bias", &bias_));
86 OP_REQUIRES_OK(ctx, ctx->GetAttr("alpha", &alpha_));
87 OP_REQUIRES_OK(ctx, ctx->GetAttr("beta", &beta_));
88 }
89
90 void Compile(XlaOpKernelContext* ctx) override {
91 const TensorShape in_grads_shape = ctx->InputShape(0);
92 const TensorShape in_image_shape = ctx->InputShape(1);
93 const TensorShape out_image_shape = ctx->InputShape(2);
94
95 OP_REQUIRES(ctx, in_grads_shape.dims() == 4 && in_image_shape.dims() == 4,
96 errors::InvalidArgument("inputs must be 4-dimensional"));
97 const int64 batch = in_grads_shape.dim_size(0);
98 const int64 rows = in_grads_shape.dim_size(1);
99 const int64 cols = in_grads_shape.dim_size(2);
100 const int64 depth = in_grads_shape.dim_size(3);
101 OP_REQUIRES(
102 ctx, in_image_shape.dim_size(0) == batch &&
103 in_image_shape.dim_size(1) == rows &&
104 in_image_shape.dim_size(2) == cols &&
105 in_image_shape.dim_size(3) == depth &&
106 out_image_shape.dim_size(0) == batch &&
107 out_image_shape.dim_size(1) == rows &&
108 out_image_shape.dim_size(2) == cols &&
109 out_image_shape.dim_size(3) == depth,
110 errors::InvalidArgument(
111 "input_grads, input_image, and out_image should have the same "
112 "shape"));
113
114 xla::ComputationBuilder* builder = ctx->builder();
115 xla::ComputationDataHandle in_grads = ctx->Input(0);
116 xla::ComputationDataHandle in_image = ctx->Input(1);
117 xla::ComputationDataHandle out_image = ctx->Input(2);
118
119 // This code is ported from tensorflow/core/kernels/lrn_op.cc. In Python
120 // pseudo-code, the Eigen code does this for each spatial position:
121 // grads = [0.0] * depth
122 // for j in range(depth):
123 // depth_begin = max(0, j - depth_radius)
124 // depth_end = min(depth, j + depth_radius + 1)
125 //
126 // norm = 0
127 // for k in range(depth_begin, depth_end):
128 // norm += in_image[k] * in_image[k]
129 // norm = alpha * norm + bias
130 //
131 // for k in range(depth_begin, depth_end):
132 // dyi = -2.0 * alpha * beta * in_image[k] * out_image[j] / norm
133 // if k == j:
134 // dyi += norm ** (-beta)
135 // dyi *= out_grads[j]
136 // grads[k] += dyi
137
138 auto accumulation_type = XlaHelpers::SumAccumulationType(input_type(0));
139 auto converted =
140 XlaHelpers::ConvertElementType(builder, in_image, accumulation_type);
141 auto squared = builder->Mul(converted, converted);
142 auto reduce = builder->ReduceWindow(
143 squared, XlaHelpers::Zero(builder, accumulation_type),
144 *ctx->GetOrCreateAdd(accumulation_type),
145 /* window_dimensions = */ {1, 1, 1, depth_radius_ * 2 + 1},
146 /* window_strides = */ {1, 1, 1, 1}, xla::Padding::kSame);
147 auto sqr_sum =
148 XlaHelpers::ConvertElementType(builder, reduce, input_type(0));
149
150 auto norm =
151 builder->Add(builder->ConstantR0<float>(bias_),
152 builder->Mul(builder->ConstantR0<float>(alpha_), sqr_sum));
153
154 auto dy = builder->Mul(
155 builder->Mul(builder->ConstantR0<float>(-2.0f * alpha_ * beta_),
156 builder->Div(out_image, norm)),
157 in_grads);
158
159 auto converted_dy =
160 XlaHelpers::ConvertElementType(builder, dy, accumulation_type);
161 auto dy_reduce = builder->ReduceWindow(
162 converted_dy, XlaHelpers::Zero(builder, accumulation_type),
163 *ctx->GetOrCreateAdd(accumulation_type),
164 /* window_dimensions = */ {1, 1, 1, depth_radius_ * 2 + 1},
165 /* window_strides = */ {1, 1, 1, 1}, xla::Padding::kSame);
166 auto dy_reduced =
167 XlaHelpers::ConvertElementType(builder, dy_reduce, input_type(0));
168
169 xla::ComputationDataHandle gradients = builder->Add(
170 builder->Mul(in_image, dy_reduced),
171 builder->Mul(in_grads,
172 builder->Pow(norm, builder->ConstantR0<float>(-beta_))));
173
174 ctx->SetOutput(0, gradients);
175 }
176
177 private:
178 int64 depth_radius_;
179 float bias_;
180 float alpha_;
181 float beta_;
182};
183
184REGISTER_XLA_OP(Name("LRNGrad"), LRNGradOp);
