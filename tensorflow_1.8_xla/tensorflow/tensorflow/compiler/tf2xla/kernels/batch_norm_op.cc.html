26class FusedBatchNormOp : public XlaOpKernel {
27 public:
28 explicit FusedBatchNormOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {
29 OP_REQUIRES_OK(ctx, ctx->GetAttr("epsilon", &epsilon_));
30 OP_REQUIRES_OK(ctx, ctx->GetAttr("is_training", &is_training_));
31 string data_format_str;
32 OP_REQUIRES_OK(ctx, ctx->GetAttr("data_format", &data_format_str));
33 OP_REQUIRES(
34 ctx, FormatFromString(data_format_str, &data_format_),
35 errors::InvalidArgument("Invalid data format: ", data_format_str));
36 OP_REQUIRES(ctx,
37 (data_format_ == FORMAT_NHWC || data_format_ == FORMAT_NCHW),
38 errors::InvalidArgument(
39 "Unsupported data format ", ToString(data_format_),
40 "; supported formats are NHWC and NCHW"));
41 }
42
43 void Compile(XlaOpKernelContext* ctx) override {
44 xla::PrimitiveType input_type;
45 OP_REQUIRES_OK(ctx,
46 DataTypeToPrimitiveType(ctx->input_type(0), &input_type));
47 xla::PrimitiveType scale_type;
48 OP_REQUIRES_OK(ctx,
49 DataTypeToPrimitiveType(ctx->input_type(1), &scale_type));
50
51 xla::ComputationBuilder* builder = ctx->builder();
52
53 xla::ComputationDataHandle input = ctx->Input(0);
54 TensorShape input_shape = ctx->InputShape(0);
55
56 int feature_index =
57 GetTensorFeatureDimIndex(input_shape.dims(), data_format_);
58
59 // TODO(b/69928690): support mixed precision in the XLA batch normalization
60 // operators. As a workaround, cast everything to the statistics type (which
61 // may be more precise than the input type).
62 input = builder->ConvertElementType(input, scale_type);
63
64 if (is_training_) {
65 xla::ComputationDataHandle output = builder->BatchNormTraining(
66 input, ctx->Input(1), ctx->Input(2), epsilon_, feature_index);
67
68 // In training mode, outputs the normalized value as well as the
69 // calculated mean and variance.
70 ctx->SetOutput(0, builder->ConvertElementType(
71 builder->GetTupleElement(output, 0), input_type));
72 ctx->SetOutput(1, builder->GetTupleElement(output, 1));
73 ctx->SetOutput(2, builder->GetTupleElement(output, 2));
74
75 // Output 3 and 4 for "FusedBatchNorm" are currently marked as "reserved
76 // space 1 & 2". They are used to pass the per-batch mean and
77 // variance to the gradient. Here we maintain the same behavior by setting
78 // them to the mean and variance calculated by BatchNormTraining.
79 ctx->SetOutput(3, builder->GetTupleElement(output, 1));
80 ctx->SetOutput(4, builder->GetTupleElement(output, 2));
81 } else {
82 xla::ComputationDataHandle output = builder->BatchNormInference(
83 input, ctx->Input(1), ctx->Input(2), ctx->Input(3), ctx->Input(4),
84 epsilon_, feature_index);
85 ctx->SetOutput(0, builder->ConvertElementType(output, input_type));
86 // Directly send input to output as mean and variance in inference mode.
87 ctx->SetOutput(1, ctx->Input(3));
88 ctx->SetOutput(2, ctx->Input(4));
89 ctx->SetOutput(3, ctx->Input(3));
90 ctx->SetOutput(4, ctx->Input(4));
91 }
92 }
93
94 private:
95 float epsilon_;
96 TensorFormat data_format_;
97 bool is_training_;
98};
99
100REGISTER_XLA_OP(Name("FusedBatchNorm"), FusedBatchNormOp);
101REGISTER_XLA_OP(Name("FusedBatchNormV2"), FusedBatchNormOp);

103class FusedBatchNormGradOp : public XlaOpKernel {
104 public:
105 explicit FusedBatchNormGradOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {
106 OP_REQUIRES_OK(ctx, ctx->GetAttr("epsilon", &epsilon_));
107 OP_REQUIRES_OK(ctx, ctx->GetAttr("is_training", &is_training_));
108 string data_format_str;
109 OP_REQUIRES_OK(ctx, ctx->GetAttr("data_format", &data_format_str));
110 OP_REQUIRES(
111 ctx, FormatFromString(data_format_str, &data_format_),
112 errors::InvalidArgument("Invalid data format: ", data_format_str));
113 OP_REQUIRES(ctx,
114 (data_format_ == FORMAT_NHWC || data_format_ == FORMAT_NCHW),
115 errors::InvalidArgument(
116 "Unsupported data format ", ToString(data_format_),
117 "; supported formats are NHWC and NCHW"));
118 }
119
120 void Compile(XlaOpKernelContext* ctx) override {
121 xla::ComputationBuilder* const b = ctx->builder();
122 DataType input_dtype = ctx->input_type(0);
123 DataType scale_dtype = ctx->input_type(2);
124
125 // TODO(b/69928690): support mixed precision in the XLA batch normalization
126 // operators. For now, cast everything to the statistics type (which
127 // may be more precise than the input type).
128 auto grad_backprop =
129 XlaHelpers::ConvertElementType(b, ctx->Input(0), scale_dtype);
130 auto activations =
131 XlaHelpers::ConvertElementType(b, ctx->Input(1), scale_dtype);
132 auto scale = ctx->Input(2);
133 auto mean = ctx->Input(3);
134 auto var = ctx->Input(4);
135
136 const int input_dims = ctx->InputShape(0).dims();
137 const int feature_index =
138 GetTensorFeatureDimIndex(input_dims, data_format_);
139
140 xla::ComputationDataHandle x_backprop;
141 xla::ComputationDataHandle scale_backprop;
142 xla::ComputationDataHandle offset_backprop;
143 if (is_training_) {
144 xla::ComputationDataHandle output =
145 b->BatchNormGrad(activations, scale, mean, var, grad_backprop,
146 epsilon_, feature_index);
147
148 x_backprop = b->GetTupleElement(output, 0);
149 scale_backprop = b->GetTupleElement(output, 1);
150 offset_backprop = b->GetTupleElement(output, 2);
151 } else {
152 // Reduce over all dimensions except the feature dim.
153 std::vector<int64> reduction_dims(input_dims - 1);
154 std::iota(reduction_dims.begin(), reduction_dims.begin() + feature_index,
155 0);
156 std::iota(reduction_dims.begin() + feature_index, reduction_dims.end(),
157 feature_index + 1);
158 // offset_backprop = sum(y_backprop)
159 // scale_backprop = y_backprop * ((x - pop_mean) * rsqrt(pop_var +
160 // epsilon))
161 // x_backprop = y_backprop * (scale * rsqrt(pop_var + epsilon))
162 const DataType accumulation_type =
163 XlaHelpers::SumAccumulationType(scale_dtype);
164 auto converted =
165 XlaHelpers::ConvertElementType(b, grad_backprop, accumulation_type);
166 auto reduce =
167 b->Reduce(converted, XlaHelpers::Zero(b, accumulation_type),
168 *ctx->GetOrCreateAdd(accumulation_type), reduction_dims);
169 offset_backprop = XlaHelpers::ConvertElementType(b, reduce, scale_dtype);
170
171 // scratch1 = rsqrt(pop_var + epsilon)
172 auto neg_half = XlaHelpers::FloatLiteral(b, scale_dtype, -0.5);
173 auto scratch1 =
174 b->Pow(b->Add(var, b->ConstantR0<float>(epsilon_)), neg_half);
175
176 // scratch2 = sum(y_backprop * (x - mean))
177 auto mul =
178 b->Mul(grad_backprop, b->Sub(activations, mean, {feature_index}));
179 converted = XlaHelpers::ConvertElementType(b, mul, accumulation_type);
180 reduce =
181 b->Reduce(converted, XlaHelpers::Zero(b, accumulation_type),
182 *ctx->GetOrCreateAdd(accumulation_type), reduction_dims);
183 auto scratch2 = XlaHelpers::ConvertElementType(b, reduce, scale_dtype);
184
185 x_backprop =
186 b->Mul(grad_backprop, b->Mul(scratch1, scale), {feature_index});
187 scale_backprop = b->Mul(scratch1, scratch2);
188 }
189
190 ctx->SetOutput(0,
191 XlaHelpers::ConvertElementType(b, x_backprop, input_dtype));
192 ctx->SetOutput(1, scale_backprop);
193 ctx->SetOutput(2, offset_backprop);
194 ctx->SetConstantOutput(3, Tensor(scale_dtype, {}));
195 ctx->SetConstantOutput(4, Tensor(scale_dtype, {}));
196 }
197
198 private:
199 TensorFormat data_format_;
200 float epsilon_;
201 bool is_training_;
202};

204REGISTER_XLA_OP(Name("FusedBatchNormGrad"), FusedBatchNormGradOp);
205REGISTER_XLA_OP(Name("FusedBatchNormGradV2"), FusedBatchNormGradOp);
