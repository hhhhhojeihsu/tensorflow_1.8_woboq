34XlaArgMinMaxOp::XlaArgMinMaxOp(OpKernelConstruction* ctx, bool is_min)
35 : XlaOpKernel(ctx), is_min_(is_min) {}
36
37void XlaArgMinMaxOp::Compile(XlaOpKernelContext* ctx) {
38 const TensorShape input_shape = ctx->InputShape(0);
39 const TensorShape dimension_shape = ctx->InputShape(1);
40
41 OP_REQUIRES(ctx, TensorShapeUtils::IsScalar(dimension_shape),
42 errors::InvalidArgument(
43 "dim must be a scalar, but received tensor of shape: ",
44 dimension_shape.DebugString()));
45
46 int64 dim;
47 OP_REQUIRES_OK(ctx, ctx->ConstantInputAsIntScalar(1, &dim));
48
49 const int input_dims = input_shape.dims();
50 const int axis = dim < 0 ? dim + input_dims : dim;
51
52 OP_REQUIRES(
53 ctx, axis >= 0 && axis < input_dims,
54 errors::InvalidArgument("Expected dimension in the range [", -input_dims,
55 ", ", input_dims, "), but got ", dim));
56 const int64 axis_size = input_shape.dim_size(axis);
57 OP_REQUIRES(
58 ctx, axis_size > 0,
59 errors::InvalidArgument("Reduction axis ", dim, " is empty in shape ",
60 input_shape.DebugString()));
61
62 DataType index_type = output_type(0);
63
64 xla::ComputationBuilder* b = ctx->builder();
65 xla::ComputationDataHandle input = ctx->Input(0);
66
67 xla::ComputationDataHandle output;
68 if (is_min_) {
69 OP_REQUIRES_OK(ctx,
70 XlaHelpers::ArgMin(b, ctx, input, input_shape, input_type(0),
71 index_type, axis, &output));
72 } else {
73 OP_REQUIRES_OK(ctx,
74 XlaHelpers::ArgMax(b, ctx, input, input_shape, input_type(0),
75 index_type, axis, &output));
76 }
77
78 ctx->SetOutput(0, output);
79}
