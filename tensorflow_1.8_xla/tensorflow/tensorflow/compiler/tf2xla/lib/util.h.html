16#ifndef TENSORFLOW_COMPILER_TF2XLA_LIB_UTIL_H_
17#define TENSORFLOW_COMPILER_TF2XLA_LIB_UTIL_H_
18
19#include "tensorflow/compiler/xla/client/computation.h"
20#include "tensorflow/compiler/xla/client/computation_builder.h"
21#include "tensorflow/compiler/xla/statusor.h"
22#include "tensorflow/core/lib/gtl/array_slice.h"
23
24namespace tensorflow {
25
26// Returns a zero-filled tensor with shape `shape`.
27xla::ComputationDataHandle Zeros(xla::ComputationBuilder* builder,
28 const xla::Shape& shape);
29
30// Returns a floating point scalar constant of 'type' with 'value'.
31// If 'type' is complex, returns a real value with zero imaginary component.
32xla::ComputationDataHandle FloatLiteral(xla::ComputationBuilder* builder,
33 xla::PrimitiveType type, double value);
34
35// Returns a integer scalar constant of 'type' with 'value'.
36// If 'type' is complex, returns a real value with zero imaginary component.
37xla::ComputationDataHandle IntegerLiteral(xla::ComputationBuilder* builder,
38 xla::PrimitiveType type, int64 value);
39
40// Performs a slice in the minor dimensions of a Tensor.
41xla::StatusOr<xla::ComputationDataHandle> SliceInMinorDims(
42 xla::ComputationBuilder* builder, const xla::ComputationDataHandle& x,
43 gtl::ArraySlice<int64> start, gtl::ArraySlice<int64> end);
44
45// Updates a slice of 'x', i.e.,
46// x[start[0], ..., start[n]] = update
47xla::StatusOr<xla::ComputationDataHandle> UpdateSlice(
48 xla::ComputationBuilder* builder, const xla::ComputationDataHandle& x,
49 const xla::ComputationDataHandle& update, gtl::ArraySlice<int64> start);
50
51// Updates a slice of 'x', where 'start' contains a list of minor dimensions:
52// x[..., start[0], ..., start[n]] = update
53xla::StatusOr<xla::ComputationDataHandle> UpdateSliceInMinorDims(
54 xla::ComputationBuilder* builder, const xla::ComputationDataHandle& x,
55 const xla::ComputationDataHandle& update, gtl::ArraySlice<int64> start);
56
57// Transposes a stack of matrices `x` by swapping the last two dimensions.
58xla::StatusOr<xla::ComputationDataHandle> TransposeInMinorDims(
59 xla::ComputationBuilder* builder, const xla::ComputationDataHandle& x);
60
61} // namespace tensorflow
