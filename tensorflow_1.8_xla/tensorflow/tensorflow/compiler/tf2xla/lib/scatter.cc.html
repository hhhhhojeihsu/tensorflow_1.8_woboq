33xla::StatusOr<xla::ComputationDataHandle> XlaScatter(
34 const xla::ComputationDataHandle& buffer,
35 const xla::ComputationDataHandle& updates,
36 const xla::ComputationDataHandle& indices, bool indices_are_vectors,
37 const std::function<xla::ComputationDataHandle(
38 xla::ComputationDataHandle, xla::ComputationDataHandle,
39 xla::ComputationBuilder*)>& combiner,
40 xla::ComputationBuilder* builder) {
41 TF_ASSIGN_OR_RETURN(std::unique_ptr<xla::Shape> buffer_shape,
42 builder->GetShape(buffer));
43 TF_ASSIGN_OR_RETURN(std::unique_ptr<xla::Shape> updates_shape,
44 builder->GetShape(updates));
45 TF_ASSIGN_OR_RETURN(std::unique_ptr<xla::Shape> indices_shape,
46 builder->GetShape(indices));
47 gtl::ArraySlice<int64> indices_dims =
48 xla::AsInt64Slice(indices_shape->dimensions());
49 gtl::ArraySlice<int64> buffer_dims =
50 xla::AsInt64Slice(buffer_shape->dimensions());
51
52 // If the indices are N-dimensional, the minor dimension of indices contains
53 // the indices to update. Otherwise the indices are all scalars.
54 int64 num_index_dims = 1;
55 if (indices_are_vectors) {
56 TF_RET_CHECK(!indices_dims.empty());
57 num_index_dims = indices_dims.back();
58 if (num_index_dims > xla::ShapeUtil::Rank(*buffer_shape)) {
59 return errors::InvalidArgument(
60 "The size of the minor dimension of the indices (shape: ",
61 xla::ShapeUtil::HumanString(*indices_shape),
62 ") must be <= the rank of the buffer (shape: ",
63 xla::ShapeUtil::HumanString(*buffer_shape), ")");
64 }
65 indices_dims.pop_back();
66 }
67
68 int64 num_indices = 1;
69 for (int64 dim : indices_dims) {
70 num_indices *= dim;
71 }
72
73 // Degenerate case: nothing to update. Return the buffer unchanged.
74 if (num_indices == 0) {
75 return buffer;
76 }
77
78 // If any of the indexed dimensions are zero in the buffer, the update cannot
79 // succeed since it updates a slice of size 1.
80 for (int64 i = 0; i < num_index_dims; ++i) {
81 if (xla::ShapeUtil::GetDimension(*buffer_shape, i) == 0) {
82 return errors::InvalidArgument(
83 "Scatter dimension ", i, " is of size zero in tensor with shape ",
84 xla::ShapeUtil::HumanString(*buffer_shape));
85 }
86 }
87
88 // Shape of the non-indexed dimensions of the buffer.
89 std::vector<int64> buffer_shape_post_axes(
90 buffer_dims.begin() + num_index_dims, buffer_dims.end());
91
92 // Flatten the major dimensions of indices and updates into a single dimension
93 // for ease of iteration.
94 std::vector<int64> flat_indices_shape({num_indices});
95 if (indices_are_vectors) {
96 flat_indices_shape.push_back(num_index_dims);
97 }
98
99 std::vector<int64> flat_updates_shape({num_indices});
100 flat_updates_shape.insert(flat_updates_shape.end(),
101 buffer_shape_post_axes.begin(),
102 buffer_shape_post_axes.end());
103
104 // Construct the initial values of the loop-carried Tensors.
105 auto flat_indices = builder->Reshape(indices, flat_indices_shape);
106 auto flat_updates = builder->Reshape(updates, flat_updates_shape);
107 auto init = {flat_indices, flat_updates, buffer};
108
109 // Constructs the loop body. The implementation of scatter is essentially:
110 // for i in range(num_indices):
111 // index = dynamic-slice(indices, i)
112 // update = dynamic-slice(updates, i)
113 // buffer = dynamic-update-slice(buffer, update, index)
114 auto body_fn = [&](xla::ComputationDataHandle i,
115 gtl::ArraySlice<xla::ComputationDataHandle> loop_vars,
116 xla::ComputationBuilder* body_builder) {
117 auto indices = loop_vars[0];
118 auto updates = loop_vars[1];
119 auto buffer = loop_vars[2];
120
121 auto zero_index = body_builder->ConstantLiteral(
122 xla::Literal::Zero(indices_shape->element_type()));
123
124 // Slice the i-th index from the indices array.
125 xla::ComputationDataHandle index;
126 auto indices_offset = body_builder->Reshape(i, {1});
127 if (indices_are_vectors) {
128 indices_offset = body_builder->Pad(indices_offset, zero_index,
129 xla::MakeEdgePaddingConfig({{0, 1}}));
130
131 index = body_builder->DynamicSlice(indices, indices_offset,
132 {1, num_index_dims});
133 index = body_builder->Collapse(index, {0, 1});
134 } else {
135 index = body_builder->DynamicSlice(indices, indices_offset, {1});
136 }
137
138 // Discard updates with negative indices, since some users expect this.
139 auto index_in_range =
140 body_builder->ReduceAll(body_builder->Le(zero_index, index),
141 body_builder->ConstantR0<bool>(true),
142 xla::CreateScalarAndComputation(body_builder));
143
144 // Make the index in bounds to prevent implementation defined behavior.
145 index = body_builder->Max(index, zero_index);
146 index = body_builder->Pad(
147 index, zero_index,
148 xla::MakeEdgePaddingConfig({{0, buffer_shape_post_axes.size()}}));
149
150 // Slice the i-th index from the updates array.
151 auto updates_offset = body_builder->Reshape(i, {1});
152 updates_offset = body_builder->Pad(
153 updates_offset, zero_index,
154 xla::MakeEdgePaddingConfig({{0, buffer_shape_post_axes.size()}}));
155 std::vector<int64> flat_updates_slice_shape({1});
156 flat_updates_slice_shape.insert(flat_updates_slice_shape.end(),
157 buffer_shape_post_axes.begin(),
158 buffer_shape_post_axes.end());
159 auto update = body_builder->DynamicSlice(updates, updates_offset,
160 flat_updates_slice_shape);
161
162 // Unflatten the major (iteration) dimensions of the slice to their
163 // original shape.
164 std::vector<int64> updates_slice_shape(num_index_dims, 1);
165 updates_slice_shape.insert(updates_slice_shape.end(),
166 buffer_shape_post_axes.begin(),
167 buffer_shape_post_axes.end());
168 update = body_builder->Reshape(update, updates_slice_shape);
169
170 // Apply the update to the buffer. If there is a combiner, use it to merge
171 // the current values with the update.
172 auto current_value =
173 body_builder->DynamicSlice(buffer, index, updates_slice_shape);
174 if (combiner) {
175 update = combiner(current_value, update, body_builder);
176 }
177 // Use the current value instead of the update if the index is out of
178 // bounds.
179 update = body_builder->Select(index_in_range, update, current_value);
180 // Apply the update.
181 buffer = body_builder->DynamicUpdateSlice(buffer, update, index);
182
183 return std::vector<xla::ComputationDataHandle>{indices, updates, buffer};
184 };
185
186 TF_ASSIGN_OR_RETURN(
187 auto outputs, XlaForEachIndex(num_indices, indices_shape->element_type(),
188 body_fn, init, "scatter", builder));
189 return outputs[2];
190}
