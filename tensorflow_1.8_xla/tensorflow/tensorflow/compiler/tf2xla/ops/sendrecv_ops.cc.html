21REGISTER_OP("_XLASend")
22 .Input("tensor: T")
23 .Attr("T: type")
24 .Attr("tensor_name: string")
25 .SetIsStateful()
26 .SetShapeFn(shape_inference::UnknownShape)
27 .Doc(R"doc(
28Sends the named tensor to another XLA computation.
29
30tensor: The tensor to send.
31tensor_name: The name of the tensor to send.
32)doc");
33
34REGISTER_OP("_XLARecv")
35 .Output("tensor: T")
36 .Attr("T: type")
37 .Attr("tensor_name: string")
38 .Attr("shape: shape")
39 .SetIsStateful()
40 .SetShapeFn([](shape_inference::InferenceContext* c) {
41 TensorShape shape_attr;
42 TF_RETURN_IF_ERROR(c->GetAttr("shape", &shape_attr));
43 shape_inference::ShapeHandle s;
44 TF_RETURN_IF_ERROR(c->MakeShapeFromTensorShape(shape_attr, &s));
45 c->set_output(0, s);
46 return Status::OK();
47 })
48 .Doc(R"doc(
49Receives the named tensor from another XLA computation.
50
51tensor: The tensor to receive.
52tensor_name: The name of the tensor to receive.
53shape: The shape of the input tensor.
54)doc");
55
