31struct NameCounts {
32 mutex counts_mutex;
33 std::unordered_map<string, int> counts;
34};
35
36string MakeUniquePath(string name) {
37 static NameCounts& instance = *new NameCounts;
38
39 // Remove illegal characters from `name`.
40 for (int i = 0; i < name.size(); ++i) {
41 char ch = name[i];
42 if (ch == '/' || ch == '[' || ch == ']' || ch == '*' || ch == '?') {
43 name[i] = '_';
44 }
45 }
46
47 int count;
48 {
49 mutex_lock lock(instance.counts_mutex);
50 count = instance.counts[name]++;
51 }
52
53 legacy_flags::DumpGraphFlags* flags = legacy_flags::GetDumpGraphFlags();
54 string path = strings::StrCat(flags->tf_dump_graph_prefix, "/", name);
55 if (count > 0) {
56 strings::StrAppend(&path, "_", count);
57 }
58 strings::StrAppend(&path, ".pbtxt");
59 return path;
60}
61
62} // anonymous namespace
63
64string DumpGraphDefToFile(const string& name, GraphDef const& graph_def) {
65 string path = MakeUniquePath(name);
66 Status status = WriteTextProto(Env::Default(), path, graph_def);
67 if (!status.ok()) {
68 VLOG(1) << "Failed to dump GraphDef to file: " << path << " : " << status;
69 path.clear();
70 path = "(unavailable)";
71 }
72 return path;
73}
74
75string DumpGraphToFile(const string& name, Graph const& graph,
76 const FunctionLibraryDefinition* flib_def) {
77 GraphDef graph_def;
78 graph.ToGraphDef(&graph_def);
79 if (flib_def) {
80 *graph_def.mutable_library() = flib_def->ToProto();
81 }
82 return DumpGraphDefToFile(name, graph_def);
83}
84
85string DumpFunctionDefToFile(const string& name, FunctionDef const& fdef) {
86 string path = MakeUniquePath(name);
87 Status status = WriteTextProto(Env::Default(), path, fdef);
88 if (!status.ok()) {
89 VLOG(1) << "Failed to dump FunctionDef to file: " << path << " : "
90 << status;
91 path.clear();
92 path = "(unavailable)";
93 }
94 return path;
95}
