27// Backwards dataflow analysis that finds arguments to a graph that must be
28// compile-time constants.
29Status BackwardsConstAnalysis(const Graph& g,
30 std::vector<bool>* compile_time_const_args) {
31 // Operators that don't look at the data of their inputs, just the shapes.
32 const std::unordered_set<string> metadata_ops = {
33 "Rank",
34 "Shape",
35 "ShapeN",
36 "Size",
37 };
38
39 Status status;
40 std::unordered_set<const Node*> must_be_const;
41 auto visit = [&status, &metadata_ops, &must_be_const,
42 compile_time_const_args](Node* node) {
43 if (!status.ok()) return;
44
45 // If this is a metadata-only op, don't propagate the const requirement.
46 if (metadata_ops.find(node->type_string()) != metadata_ops.end()) return;
47
48 // If this node must be const, and it isn't a metadata op, then all of its
49 // parents must be const.
50 if (must_be_const.find(node) != must_be_const.end()) {
51 if (node->type_string() == "_Arg") {
52 int index;
53 status = GetNodeAttr(node->attrs(), "index", &index);
54 if (!status.ok()) return;
55 compile_time_const_args->at(index) = true;
56 return;
57 }
58 for (const Edge* pred : node->in_edges()) {
59 if (!pred->IsControlEdge()) {
60 must_be_const.insert(pred->src());
61 }
62 }
63 return;
64 }
65
66 // Mark any compile-time constant operator arguments as const.
67 const std::unordered_set<string>* const_inputs =
68 XlaOpRegistry::CompileTimeConstantInputs(node->type_string());
69 if (!const_inputs || const_inputs->empty()) return;
70
71 NameRangeMap input_name_ranges;
72 status =
73 NameRangesForNode(*node, node->op_def(), &input_name_ranges, nullptr);
74 if (!status.ok()) return;
75
76 for (const string& input : *const_inputs) {
77 auto name_range = input_name_ranges.find(input);
78 if (name_range == input_name_ranges.end()) continue;
79
80 for (Edge const* edge : node->in_edges()) {
81 if (edge->dst_input() >= name_range->second.first &&
82 edge->dst_input() < name_range->second.second) {
83 must_be_const.insert(edge->src());
84 }
85 }
86 }
87 };
88
89 // Post-order traversal visits nodes in reverse topological order for an
90 // acyclic graph.
91 DFS(g, {}, visit);
92 return status;
93}
