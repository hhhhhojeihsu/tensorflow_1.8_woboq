26class XlaOpKernelContext;
27
28// Implementations of operators that generate XLA code should usually subclass
29// XlaOpKernel and implement the Compile() method. Unlike a regular OpKernel,
30// an XlaOpKernel produces and consumes symbolic values during compilation.
31//
32// See the comments in xla_context.h for more details.
33class XlaOpKernel : public OpKernel {
34 public:
35 explicit XlaOpKernel(OpKernelConstruction* construction);
36
37 // Subclasses should implement Compile(), much as standard OpKernels implement
38 // Compute().
39 virtual void Compile(XlaOpKernelContext* context) = 0;
40
41 private:
42 void Compute(OpKernelContext* context) final;
43};
44
45// The context passed to the Compile() method of XlaOpKernel. An
46// XlaOpKernelContext is a variant of the standard OpKernel class, tailored for
47// implementing operators that perform symbolic execution as part of the XLA
48// compiler. The key difference is that XlaOpKernelContext produces and consumes
49// data as XLA computations, rather than as standard Tensors.
50//
51// Under the hood, symbolic execution communicates using special Tensors that
52// wrap XlaExpression objects, however this is an implementation detail that
53// this class hides. The *only* correct way to allocate a Tensor during
54// compilation is using the XlaOpKernelContext methods, since they ensure there
55// is a valid XlaExpression backing the tensor. No Op should ever call
56// allocate_output or allocate_temp directly on the underlying OpKernelContext.
57class XlaOpKernelContext {
58 public:
59 explicit XlaOpKernelContext(OpKernelContext* context);
60
61 // Returns the XLA ComputationBuilder containing the output of compilation.
62 xla::ComputationBuilder* builder() const;
63
64 // Inputs
65
66 // Returns the number of inputs to the operator.
67 int num_inputs() const { return context_->num_inputs(); }
68
69 // Returns the type of input 'index'.
70 DataType input_type(int index) { return context_->input(index).dtype(); }
71
72 // Returns the shape of input 'index'.
73 TensorShape InputShape(int index);
74
75 // Returns input 'index' as a ComputationDataHandle. Unlike
76 // OpKernelContext::Input returns a symbolic value rather than a concrete
77 // Tensor.
78 const xla::ComputationDataHandle& Input(int index);
79
80 // Returns true if all inputs are the same shape, otherwise sets the
81 // status to a non-OK value and returns false.
82 // Usage: if (!context->ValidateInputsAreSameShape(this)) return;
83 bool ValidateInputsAreSameShape(OpKernel* op) TF_MUST_USE_RESULT;
84
85 // Returns the named list-valued immutable input in "list", as
86 // defined in the OpDef. If the named output is not list-valued,
87 // returns a one-element list.
88 Status InputList(StringPiece name,
89 std::vector<xla::ComputationDataHandle>* handles,
90 std::vector<TensorShape>* shapes);
91
92 // Helper methods for constant inputs.
93
94 // Evaluates input 'index' and stores it in '*constant_literal'. If the
95 // expression cannot be evaluated, e.g., because it depends on unbound
96 // parameters, returns a non-OK status.
97 Status ConstantInput(int index, xla::Literal* constant_literal);
98
99 // Evaluates input 'index', reshapes it to 'new_shape' if new_shape !=
100 // InputShape(index), and stores it in '*constant_literal'. If the input
101 // cannot be evaluated, e.g., because it depends on unbound parameters,
102 // returns a non-Ok status. If InputShape(index).num_elements() !=
103 // new_shape.num_elements(), returns an error status.
104 Status ConstantInputReshaped(int index, gtl::ArraySlice<int64> new_shape,
105 xla::Literal* constant_literal);
106
107 // Converts a constant scalar int32 or int64 tensor into an int64.
108 Status ConstantInputAsIntScalar(int index, int64* out);
109
110 // Converts a constant scalar float32 or float64 tensor into a float64.
111 Status ConstantInputAsFloatScalar(int index, double* out);
112
113 // Converts a constant 1D int32 or int64 tensor into a vector of int64s.
114 Status ConstantInputAsIntVector(int index, std::vector<int64>* out);
115
116 // Converts a constant int32 or int64 Tensor into an xla int64 Literal.
117 Status ConstantInputAsInt64Literal(int index, xla::Literal* out);
118
119 // Converts a constant 1D int32 or int64 tensor into a TensorShape.
120 Status ConstantInputAsShape(int index, TensorShape* shape);
121
122 // Returns the named list-valued immutable input in "list", as
123 // defined in the OpDef. If the named output is not list-valued,
124 // returns a one-element list.
125 Status ConstantInputList(StringPiece name,
126 std::vector<xla::Literal>* literals);
127
128 // Outputs
129
130 int num_outputs() const { return context_->num_outputs(); }
131 DataType expected_output_dtype(int index) const {
132 return context_->expected_output_dtype(index);
133 }
134
135 // Sets output 'index' to the ComputationDataHandle 'handle'.
136 // All outputs should be set using SetOutput and SetConstantOutput, not
137 // via the underlying OpKernelContext.
138 void SetOutput(int index, const xla::ComputationDataHandle& handle);
139
140 // Sets output 'index' to compile-time constant 'host_tensor', where
141 // 'host_tensor' is a tensor in host memory. It is preferable to use
142 // SetConstantOutput where possible.
143 void SetConstantOutput(int index, const Tensor& host_tensor);
144
145 // Sets output 'index' to an invalid value.
146 // Any subsequent attempt to consume this output will cause an error.
147 void SetInvalidOutput(int index);
148
149 // Status handling.
150 void SetStatus(const Status& status) { context_->SetStatus(status); }
151 Status status() { return context_->status(); }
152
153 // Variables
154
155 // Sets '*resource' to the resource associated with input `index`.
156 Status GetResourceInput(int index, XlaResource** resource);
157
158 // Sets output 'index' to be a reference to resource 'resource'.
159 void SetResourceOutput(int index, XlaResource* resource);
160
161 // Sets `*type` and `*shape` to the current type and shape of a variable's
162 // value.
163 Status GetVariableTypeAndShape(int index, DataType* type,
164 TensorShape* shape) const;
165
166 // Reads the current value of the resouce variable referred to by input
167 // 'index'. If `shape` is not nullptr, sets `*shape` to the shape of the
168 // variable. Returns an error if the variable has not been initialized, or if
169 // its type does not match `type`.
170 Status ReadVariableInput(int index, DataType type, TensorShape* shape,
171 xla::ComputationDataHandle* value);
172
173 // Assigns the value `handle` to the variable referenced by input
174 // `input_index`. The variable must be of `type`. Returns an error if the
175 // variable has been initialized with a different type or with a
176 // different shape.
177 Status AssignVariable(int input_index, DataType type,
178 xla::ComputationDataHandle handle);
179
180 // Helper routines for the OP_REQUIRES macros
181 void CtxFailure(const Status& s);
182 void CtxFailureWithWarning(const Status& s);
183 void CtxFailure(const char* file, int line, const Status& s);
184 void CtxFailureWithWarning(const char* file, int line, const Status& s);
185
186 // If this kernel invocation is within a function execution,
187 // call_frame() returns the call frame for the function call.
188 CallFrameInterface* call_frame() const { return context_->call_frame(); }
189
190 FunctionLibraryRuntime* function_library() const {
191 return context_->function_library();
192 }
193
194 const OpKernel& op_kernel() const { return context_->op_kernel(); }
195
196 // Returns the underlying OpKernelContext. Use rarely.
197 OpKernelContext* op_kernel_context() const { return context_; }
198
199 // Returns the XlaCompiler that is performing the compilation. Used for, e.g.,
200 // While to compile nested computations.
201 XlaCompiler* compiler() const;
202
203 // TODO(phawkins): find a better home for these helpers.
204
205 // Gets an XLA lambda to compute Max. This is cached in the
206 // XlaContext since it may be used by multiple Ops. There is a
207 // separate specialization of the computation for each DataType.
208 const xla::Computation* GetOrCreateMax(const DataType type);
209
210 // Gets an XLA lambda to compute Min. This is cached in the
211 // XlaContext since it may be used by multiple Ops. There is a
212 // separate specialization of the computation for each DataType.
213 const xla::Computation* GetOrCreateMin(const DataType type);
214
215 // Gets an XLA lambda to compute Add. This is cached in the
216 // XlaContext since it may be used by multiple Ops. There is a
217 // separate specialization of the computation for each DataType.
218 const xla::Computation* GetOrCreateAdd(const DataType type);
219
220 // Gets an XLA lambda to compute Mul. This is cached in the
221 // XlaContext since it may be used by multiple Ops. There is a
222 // separate specialization of the computation for each DataType.
223 const xla::Computation* GetOrCreateMul(const DataType type);
224
225 private:
226 OpKernelContext* const context_;
227};
228
229} // namespace tensorflow
