28// Inputs 2*N tensors, outputs the first N inputs.
29// Logs errors if input tensor i and i + N are not (near) identical
30// in any position.
31class ParallelCheckOp : public OpKernel {
32 public:
33 explicit ParallelCheckOp(OpKernelConstruction* ctx) : OpKernel(ctx) {}
34
35 template <typename T>
36 int CompareTensors(DataType dtype, const char* v0, const char* v1,
37 int64 num_elts, int input_idx) {
38 int failed = 0;
39 const T* p0 = reinterpret_cast<const T*>(v0);
40 const T* p1 = reinterpret_cast<const T*>(v1);
41 double rtol;
42 legacy_flags::ParallelCheckOpFlags* flags =
43 legacy_flags::GetParallelCheckOpFlags();
44 if (!tensorflow::strings::safe_strtod(flags->parallel_check_rtol.c_str(),
45 &rtol)) {
46 LOG(ERROR) << "can't convert parallel_check_rtol "
47 << flags->parallel_check_rtol << " to double";
48 }
49 double atol;
50 if (!tensorflow::strings::safe_strtod(flags->parallel_check_atol.c_str(),
51 &atol)) {
52 LOG(ERROR) << "can't convert parallel_check_atol "
53 << flags->parallel_check_atol << " to double";
54 }
55 for (int i = 0; i < num_elts; ++i) {
56 bool ok = (p0[i] == p1[i]);
57 VLOG(2) << "output " << input_idx << " element " << i << ": " << p0[i];
58 if (!ok) {
59 if (std::is_same<T, float>::value || std::is_same<T, double>::value) {
60 float tolerance =
61 std::max(atol, std::max(fabs(rtol * p0[i]), fabs(rtol * p1[i])));
62 T diff = p0[i] - p1[i];
63 if (diff < 0) diff = 0 - diff;
64 ok = (diff <= tolerance);
65 }
66 if (ok) continue;
67 LOG(ERROR) << "Op " << name() << " fails equality at output "
68 << input_idx << " type " << DataTypeString(dtype)
69 << " element " << i << ": std_val=" << p0[i]
70 << " test_val=" << p1[i] << " diff=" << (p0[i] - p1[i]);
71 if (++failed > 10) break;
72 }
73 }
74 return failed;
75 }
76
77 void Compute(OpKernelContext* ctx) override {
78 VLOG(1) << "Compute " << name();
79 const int num_pairs = ctx->num_inputs() / 2;
80 for (int i = 0; i < num_pairs; ++i) {
81 CHECK_EQ(ctx->input_dtype(i), ctx->input_dtype(i + num_pairs));
82 Tensor t0 = ctx->input(i);
83 Tensor t1 = ctx->input(i + num_pairs);
84 int64 num_elts = t0.NumElements();
85 CHECK_EQ(num_elts, t1.NumElements());
86
87 // Compare inputs elementwise for near-exact equality.
88 const char* v0 = t0.tensor_data().data();
89 const char* v1 = t1.tensor_data().data();
90 int failed = 0;
91 switch (ctx->input_dtype(i)) {
92 case DT_INT32:
93 failed =
94 CompareTensors<int32>(ctx->input_dtype(i), v0, v1, num_elts, i);
95 break;
96 case DT_INT64:
97 failed =
98 CompareTensors<int64>(ctx->input_dtype(i), v0, v1, num_elts, i);
99 break;
100 case DT_FLOAT:
101 failed =
102 CompareTensors<float>(ctx->input_dtype(i), v0, v1, num_elts, i);
103 break;
104 case DT_DOUBLE:
105 failed =
106 CompareTensors<double>(ctx->input_dtype(i), v0, v1, num_elts, i);
107 break;
108 case DT_BOOL:
109 failed =
110 CompareTensors<bool>(ctx->input_dtype(i), v0, v1, num_elts, i);
111 break;
112 default:
113 LOG(FATAL) << "unimpl: " << ctx->input_dtype(i);
114 }
115 if (failed > 0) {
116 LOG(ERROR) << "check failed for " << name() << " output " << i
117 << " num_elts: " << num_elts;
118 legacy_flags::ParallelCheckOpFlags* flags =
119 legacy_flags::GetParallelCheckOpFlags();
120 if (flags->parallel_check_failfast) {
121 LOG(QFATAL) << "failfast on first parallel-check failure";
122 }
123 } else {
124 VLOG(1) << "check passed for " << name() << " output " << i
125 << " num_elts: " << num_elts;
126 }
127
128 // Propagate the std value.
129 if (IsRefType(ctx->input_dtype(i))) {
130 ctx->forward_ref_input_to_ref_output(i, i);
131 } else {
132 ctx->set_output(i, ctx->input(i));
133 }
134 }
135 }
136
137 TF_DISALLOW_COPY_AND_ASSIGN(ParallelCheckOp);
138};
139
140REGISTER_KERNEL_BUILDER(Name("ParallelCheck").Device(DEVICE_CPU),
141 ParallelCheckOp);
