27std::map<int, OptionalTensor> GetVariables(OpKernelContext* ctx) {
28 std::map<int, OptionalTensor> variables;
29 for (int64 i = 0; i < ctx->num_inputs(); ++i) {
30 if (ctx->input(i).dtype() == DT_RESOURCE) {
31 Var* variable = nullptr;
32 ResourceHandle handle = HandleFromInput(ctx, i);
33 OptionalTensor& optional = variables[i];
34 optional.name = handle.name();
35 if (LookupResource(ctx, handle, &variable).ok()) {
36 tf_shared_lock lock(*variable->mu());
37 optional.present = true;
38 optional.value = *variable->tensor();
39 }
40 }
41 }
42 return variables;
43}
44} // namespace

46Status XlaCompileOnDemandOp::Run(OpKernelContext* ctx,
47 const XlaDevice::Metadata& metadata,
48 const XlaCompiler::CompilationResult* result,
49 xla::LocalExecutable* executable) {
50 std::map<int, OptionalTensor> variables = GetVariables(ctx);
51 int64 num_resource_args = variables.size();
52
53 xla::LocalClient* client = metadata.client();
54
55 // Builds an XLA allocator for the device.
56 XlaComputationLaunchContext launch_context(
57 num_resource_args, client, client->backend().memory_allocator(), true);
58
59 launch_context.PopulateInputs(ctx, result, variables);
60
61 perftools::gputools::Stream* stream =
62 ctx->op_device_context() ? ctx->op_device_context()->stream() : nullptr;
63 TF_RET_CHECK(stream);
64
65 VLOG(2) << "Executing computation.";
66 xla::ExecutableRunOptions run_options;
67 run_options.set_stream(stream);
68 run_options.set_allocator(client->backend().memory_allocator());
69 run_options.set_intra_op_thread_pool(&ctx->eigen_cpu_device());
70
71 auto run_result = executable->Run(launch_context.arguments(), run_options);
72 TF_RETURN_IF_ERROR(run_result.status());
73
74 launch_context.PopulateOutputs(ctx, result, run_result.ConsumeValueOrDie());
75 return Status::OK();
76}

78bool XlaCompileOnDemandOp::MustArgumentBeConstant(const OpKernel* op_kernel,
79 int64 argument_idx) {
80 // TODO(jmolloy): This could be expensive, so memoize.
81 auto* constant_inputs = tensorflow::XlaOpRegistry::CompileTimeConstantInputs(
82 op_kernel->def().op());
83 CHECK(constant_inputs);
84 std::set<int64> constant_input_indices;
85 for (const auto& name : *constant_inputs) {
86 int start, stop;
87 TF_CHECK_OK(op_kernel->InputRange(name, &start, &stop));
88 for (int i = start; i < stop; ++i) {
89 constant_input_indices.insert(i);
90 }
91 }
92 return constant_input_indices.count(argument_idx) > 0;
93}
94
95bool XlaCompileOnDemandOp::ShouldArgumentBeConstant(const OpKernel* op_kernel,
96 int64 argument_idx) {
97 // Right now we only create kConstant arguments when absolutely required, but
98 // there may be benefit in eagerly constant-folding a larger subset of
99 // arguments in the future.
100 return MustArgumentBeConstant(op_kernel, argument_idx);
101}

103Status XlaCompileOnDemandOp::Compile(
104 OpKernelContext* ctx, const XlaDevice::Metadata& metadata,
105 const XlaCompiler::CompilationResult** result,
106 xla::LocalExecutable** executable) {
107 std::map<int, Tensor> constant_arguments;
108 for (int64 i = 0; i < ctx->num_inputs(); ++i) {
109 const Tensor& device_tensor = ctx->input(i);
110 if (const XlaTensor* xla_tensor = XlaTensor::FromTensor(&device_tensor)) {
111 if (xla_tensor->has_host_tensor() &&
112 ShouldArgumentBeConstant(&ctx->op_kernel(), i)) {
113 constant_arguments[i] = xla_tensor->host_tensor();
114 }
115 }
116 if (constant_arguments.count(i) == 0 &&
117 MustArgumentBeConstant(&ctx->op_kernel(), i)) {
118 // Slow path; the argument is not available as a host constant so we must
119 // fetch it synchronously.
120 Tensor host_tensor;
121 AllocatorAttributes attrs;
122 attrs.set_on_host(true);
123 TF_RETURN_IF_ERROR(ctx->allocate_temp(
124 device_tensor.dtype(), device_tensor.shape(), &host_tensor, attrs));
125 Notification n;
126 ctx->op_device_context()->CopyDeviceTensorToCPU(
127 &device_tensor, "ConstantArgument",
128 reinterpret_cast<Device*>(ctx->device()), &host_tensor,
129 [&](Status status) { n.Notify(); });
130 n.WaitForNotification();
131 constant_arguments[i] = host_tensor;
132 }
133 }
134
135 // We store information about the JIT-compiled XLA computation
136 // in the ResourceMgr.
137 ResourceMgr* rm = ctx->resource_manager();
138 CHECK(rm);
139
140 XlaCompilationCache* cache;
141 TF_RETURN_IF_ERROR(rm->LookupOrCreate<XlaCompilationCache>(
142 rm->default_container(), "xla_cache", &cache,
143 [&](XlaCompilationCache** cache) {
144 *cache = new XlaCompilationCache(metadata.client(),
145 metadata.jit_device_type());
146 return Status::OK();
147 }));
148 // Hold the reference to the JIT during evaluation. (We could probably
149 // free it sooner because the ResourceMgr will retain a reference, but
150 // this is more obviously correct.)
151 core::ScopedUnref cache_ref(cache);
152
153 XlaCompiler::Options options;
154 DeviceType device_type = metadata.jit_device_type();
155 options.device_type = &device_type;
156 options.client = metadata.client();
157 options.flib_def =
158 new FunctionLibraryDefinition(OpRegistry::Global(), FunctionDefLibrary{});
159
160 std::map<int, OptionalTensor> variable_args = GetVariables(ctx);
161 return cache->CompileSingleOp(options, constant_arguments, variable_args, ctx,
162 result, executable,
163 /*compile_options=*/nullptr);
164}
165
166void XlaCompileOnDemandOp::Compute(OpKernelContext* ctx) {
167 const XlaCompiler::CompilationResult* result;
168 xla::LocalExecutable* executable;
169 const XlaDevice::Metadata* metadata;
170 OP_REQUIRES_OK(ctx, XlaDevice::GetMetadata(ctx, &metadata));
171 OP_REQUIRES_OK(ctx, Compile(ctx, *metadata, &result, &executable));
172 OP_REQUIRES_OK(ctx, Run(ctx, *metadata, result, executable));
173}
