21/*static*/ XlaTensor* XlaTensor::FromTensor(Tensor* tensor) {
22 if (tensor->NumElements() == 0) {
23 return nullptr;
24 }
25 XlaTensor* xla_tensor =
26 FromOpaquePointer(const_cast<char*>(tensor->tensor_data().data()));
27 return xla_tensor;
28}
29
30/*static*/ const XlaTensor* XlaTensor::FromTensor(const Tensor* tensor) {
31 return FromTensor(const_cast<Tensor*>(tensor));
32}
33
34/*static*/ perftools::gputools::DeviceMemoryBase
35XlaTensor::DeviceMemoryFromTensor(const Tensor& tensor) {
36 const XlaTensor* xla_tensor = FromTensor(&tensor);
37 if (xla_tensor) {
38 CHECK(xla_tensor->has_shaped_buffer());
39 return xla_tensor->shaped_buffer().root_buffer();
40 } else {
41 return perftools::gputools::DeviceMemoryBase(
42 const_cast<char*>(tensor.tensor_data().data()),
43 tensor.tensor_data().size());
44 }
45}

47Status XlaTensor::AllocateShapedBuffer(DataType dtype, const TensorShape& shape,
48 xla::LocalClient* client,
49 int device_ordinal) {
50 xla::Shape on_host_shape;
51 TF_RETURN_IF_ERROR(TensorShapeToXLAShape(dtype, shape, &on_host_shape));
52 xla::Shape on_device_shape =
53 client->backend().transfer_manager()->HostShapeToDeviceShape(
54 on_host_shape);
55
56 xla::ShapedBuffer buffer(on_host_shape, on_device_shape, client->platform(),
57 device_ordinal);
58 for (auto& index_to_buffer : buffer.buffers()) {
59 xla::Shape subshape =
60 xla::ShapeUtil::GetSubshape(on_device_shape, index_to_buffer.first);
61 uint64 size =
62 client->backend().transfer_manager()->GetByteSizeRequirement(subshape);
63 TF_ASSIGN_OR_RETURN(index_to_buffer.second,
64 client->backend().memory_allocator()->Allocate(
65 device_ordinal, size, /*retry_on_failure=*/false));
66 }
67
68 TF_ASSIGN_OR_RETURN(auto scoped_buffer,
69 xla::ScopedShapedBuffer::MakeScoped(
70 &buffer, client->backend().memory_allocator()));
71 set_shaped_buffer(std::move(scoped_buffer));
72 return Status::OK();
73}

75// The pointer tag, OR-ed into the XlaTensor's address to distinguish it from
76// device-side tensors, which are either CPU or GPU memory pointers. This works
77// because we're guaranteed that CPU and GPU pointers are aligned to > 1 bits.
78namespace {
79constexpr uintptr_t kTag = 0x1ULL;
80}
81
82/*static*/ XlaTensor* XlaTensor::FromOpaquePointer(void* ptr) {
83 uintptr_t value = reinterpret_cast<uintptr_t>(ptr);
84 if (value & kTag) {
85 return reinterpret_cast<XlaTensor*>(value & ~kTag);
86 } else {
87 return nullptr;
88 }
89}
90
91/*static*/ void* XlaTensor::ToOpaquePointer(XlaTensor* tensor) {
92 uintptr_t value = reinterpret_cast<uintptr_t>(tensor);
93 CHECK_EQ(value & kTag, 0);
94 value |= kTag;
95 return reinterpret_cast<XlaTensor*>(value);
96}
