16// Registers the XLA_CPU device, which is an XlaDevice instantiation that runs
17// operators using XLA via the XLA "Host" (CPU) backend.
18
19#include "tensorflow/compiler/jit/kernels/xla_launch_op.h"
20#include "tensorflow/compiler/jit/legacy_flags/xla_device_flags.h"
21#include "tensorflow/compiler/jit/xla_compile_on_demand_op.h"
22#include "tensorflow/compiler/jit/xla_device.h"
23#include "tensorflow/compiler/jit/xla_device_ops.h"

25#include "tensorflow/core/common_runtime/device_factory.h"
26#include "tensorflow/core/lib/core/status.h"

28namespace tensorflow {
29
30class XlaCpuDeviceFactory : public DeviceFactory {
31 public:
32 Status CreateDevices(const SessionOptions& options, const string& name_prefix,
33 std::vector<Device*>* devices) override;
34};
35
36Status XlaCpuDeviceFactory::CreateDevices(const SessionOptions& options,
37 const string& name_prefix,
38 std::vector<Device*>* devices) {
39 legacy_flags::XlaDeviceFlags* flags = legacy_flags::GetXlaDeviceFlags();
40 bool compile_on_demand = flags->tf_xla_compile_on_demand;
41
42 XlaOpRegistry::DeviceRegistration registration;
43 registration.compilation_device_name = DEVICE_CPU_XLA_JIT;
44 registration.requires_compilation = !compile_on_demand;
45 registration.enable_jit_by_default = false;
46 registration.compile_resource_ops = true;
47
48 static XlaDeviceOpRegistrations* registrations =
49 RegisterXlaDeviceKernels(DEVICE_XLA_CPU, DEVICE_CPU_XLA_JIT);
50 (void)registrations;
51
52 std::unique_ptr<XlaDevice> device;
53 TF_RETURN_IF_ERROR(XlaDevice::Create("Host", DEVICE_XLA_CPU, 0,
54 DEVICE_CPU_XLA_JIT, options, name_prefix,
55 registration,
56 /*transfer_as_literal=*/false, &device));
57 devices->push_back(device.release());
58 return Status::OK();
59}
60
61REGISTER_LOCAL_DEVICE_FACTORY(DEVICE_XLA_CPU, XlaCpuDeviceFactory);
62
63// Kernel registrations

65constexpr std::array<DataType, 7> kAllXlaCpuTypes = {

68REGISTER_XLA_LAUNCH_KERNEL(DEVICE_XLA_CPU, XlaLocalLaunchOp, kAllXlaCpuTypes);
