16// The XlaCompileOnDemandOp is an OpKernel that, when its Compute method is
17// called, will generate an xla::Computation and run it asynchronously.
18
19#ifndef TENSORFLOW_COMPILER_JIT_XLA_COMPILE_ON_DEMAND_OP_H_

29namespace tensorflow {
30
31// An OpKernel that compiles an op to an XLA computation and runs it. Unlike
32// _XlaLaunch this doesn't rely on any rewrites of the graphdef - it will run a
33// vanilla TensorFlow op as long as the bridge supports it.
34//
35// Importantly _XlaLaunch assumes all input and output tensors are on the host,
36// whereas XlacompileOnDemandOp works with tensors in device memory.
37class XlaCompileOnDemandOp : public OpKernel {
38 public:
39 explicit XlaCompileOnDemandOp(OpKernelConstruction* ctx) : OpKernel(ctx) {}
40 void Compute(OpKernelContext* ctx) override;
41
42 private:
43 XlaCompiler::Argument CreateCompilerArgument(OpKernelContext* ctx, int64 i);
44 bool ShouldArgumentBeConstant(const OpKernel* op_kernel, int64 argument_idx);
45 bool MustArgumentBeConstant(const OpKernel* op_kernel, int64 argument_idx);
46 Status Compile(OpKernelContext* ctx, const XlaDevice::Metadata& metadata,
47 const XlaCompiler::CompilationResult** result,
48 xla::LocalExecutable** executable);
49 Status Run(OpKernelContext* ctx, const XlaDevice::Metadata& metadata,
50 const XlaCompiler::CompilationResult* result,
51 xla::LocalExecutable* executable);
52};
53
54} // namespace tensorflow

56#endif // TENSORFLOW_COMPILER_JIT_XLA_COMPILE_ON_DEMAND_OP_H_
