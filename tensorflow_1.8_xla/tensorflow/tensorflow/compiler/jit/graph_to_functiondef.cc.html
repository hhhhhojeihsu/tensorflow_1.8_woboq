33// TODO(phawkins) add a canonical copy of these operator names and refactor
34// everything to use it.
35const char* const kArgOp = "_Arg";
36const char* const kRetValOp = "_Retval";

38// Class that maintains a one-to-one original node name -> new name mapping.
39// We have to normalize the names used as input and output arguments to
40// match regexp "[a-z][a-z0-9_]*". Once we rename them, we risk creating
41// a name collision with the other node names, so if necessary we add
42// a suffix to make names unique. So if we have an input named "A" and a
43// node in the function body named "a", they will be renamed to "a" and "a_0".
44class NodeNameMapping {
45 public:
46 NodeNameMapping() = default;
47
48 // Normalize the input/output name and then make it unique.
49 string Normalize(const string& name);
50
51 // Make the node name unique.
52 string Uniquify(const string& name);
53
54 // Look up how a node name was previously normalized/uniquified.
55 // Returns empty if name was never seen.
56 string Renormalize(const string& name) const;
57
58 private:
59 string NormalizeHelper(string name) const;
60 string UniquifyHelper(string name);
61
62 std::unordered_set<string> used_names_;
63 std::unordered_map<string, string> name_mapping_;
64};
65
66string NodeNameMapping::NormalizeHelper(string name) const {
67 // Convert letters to lowercase and non-alphanumeric characters to '_'.
68 if (name.empty()) name = "unknown";
69 const int n = name.size();
70 for (int i = 0; i < n; i++) {
71 char c = name[i];
72 if (isalnum(c)) {
73 if (isupper(c)) {
74 name[i] = tolower(c);
75 }
76 } else {
77 name[i] = '_';
78 }
79 }
80 return name;
81}
82
83string NodeNameMapping::UniquifyHelper(string name) {
84 // If the name hasn't been used yet, use it as-is.
85 if (used_names_.insert(name).second) return name;
86 // Add a suffix to name to make it unique.
87 for (int i = 0;; ++i) {
88 const string candidate = strings::StrCat(name, "_", i);
89 if (used_names_.insert(candidate).second) return candidate;
90 }
91}
92
93string NodeNameMapping::Normalize(const string& name) {
94 const string normalized = UniquifyHelper(NormalizeHelper(name));
95 name_mapping_[name] = normalized;
96 return normalized;
97}
98
99string NodeNameMapping::Uniquify(const string& name) {
100 const string uniqued = UniquifyHelper(name);
101 name_mapping_[name] = uniqued;
102 return uniqued;
103}
104
105string NodeNameMapping::Renormalize(const string& name) const {
106 const auto iter = name_mapping_.find(name);
107 if (iter == name_mapping_.end()) return string();
108 return iter->second;
109}

113// Graph to FunctionDef conversion. This code is closely modeled on the Python
114// code in third_party/tensorflow/python/framework/function.py.
115
116Status GraphToFunctionDef(const Graph& graph, const string& name,
117 FunctionDef* fdef) {
118 fdef->mutable_signature()->set_name(name);
119
120 std::unordered_map<string, string> tensor_renaming;
121 std::unordered_map<string, string> return_values;
122 NodeNameMapping node_names;
123
124 for (Node const* node : graph.op_nodes()) {
125 if (node->type_string() == kArgOp) {
126 int index;
127 DataType type;
128 TF_RETURN_IF_ERROR(GetNodeAttr(node->attrs(), "T", &type));
129 TF_RETURN_IF_ERROR(GetNodeAttr(node->attrs(), "index", &index));
130 while (fdef->signature().input_arg_size() <= index) {
131 fdef->mutable_signature()->add_input_arg();
132 }
133 OpDef::ArgDef* argdef =
134 fdef->mutable_signature()->mutable_input_arg(index);
135 argdef->set_type(type);
136 const string normalized = node_names.Normalize(node->name());
137 argdef->set_name(normalized);
138 tensor_renaming[strings::StrCat(node->name(), ":0")] = normalized;
139 continue;
140 }
141
142 if (node->type_string() == kRetValOp) {
143 int index;
144 DataType type;
145 TF_RETURN_IF_ERROR(GetNodeAttr(node->attrs(), "T", &type));
146 TF_RETURN_IF_ERROR(GetNodeAttr(node->attrs(), "index", &index));
147 while (fdef->signature().output_arg_size() <= index) {
148 fdef->mutable_signature()->add_output_arg();
149 }
150 OpDef::ArgDef* argdef =
151 fdef->mutable_signature()->mutable_output_arg(index);
152 argdef->set_type(type);
153 const string normalized = node_names.Normalize(node->name());
154 argdef->set_name(normalized);
155 Edge const* edge;
156 TF_CHECK_OK(node->input_edge(0, &edge));
157 return_values[normalized] =
158 strings::StrCat(edge->src()->name(), ":", edge->src_output());
159 continue;
160 }
161
162 NodeDef* node_def = fdef->add_node_def();
163 *node_def = node->def();
164 if (!node->assigned_device_name().empty()) {
165 node_def->set_device(node->assigned_device_name());
166 }
167 node_def->set_name(node_names.Uniquify(node->name()));
168
169 // Reset input names based on graph rather than the NodeDef.
170 node_def->clear_input();
171
172 // Edges, indexed by dst_input.
173 std::vector<const Edge*> in_edges;
174 std::vector<const Edge*> control_edges;
175 for (Edge const* edge : node->in_edges()) {
176 if (edge->src()->IsSource()) continue;
177
178 if (edge->IsControlEdge()) {
179 control_edges.push_back(edge);
180 } else {
181 if (in_edges.size() <= edge->dst_input()) {
182 in_edges.resize(edge->dst_input() + 1);
183 }
184 in_edges[edge->dst_input()] = edge;
185 }
186 }
187
188 // Add regular inputs
189 for (std::vector<const Edge*>::size_type i = 0; i < in_edges.size(); ++i) {
190 const Edge* edge = in_edges[i];
191 if (edge == nullptr) {
192 return errors::InvalidArgument(
193 "Nonconsecutive input edges; missing "
194 "input edge ",
195 i, " for node ", node->name());
196 }
197 node_def->add_input(
198 strings::StrCat(edge->src()->name(), ":", edge->src_output()));
199 }
200
201 // Add control inputs
202 for (const Edge* edge : control_edges) {
203 node_def->add_input(strings::StrCat("^", edge->src()->name()));
204 }
205
206 // Populate tensor_renaming.
207 NameRangeMap output_ranges;
208 TF_RETURN_IF_ERROR(
209 NameRangesForNode(*node, node->op_def(), nullptr, &output_ranges));
210 for (const auto& output : output_ranges) {
211 for (int i = output.second.first; i < output.second.second; ++i) {
212 const string tensor_name = strings::StrCat(
213 node_def->name(), ":", output.first, ":", i - output.second.first);
214 tensor_renaming[strings::StrCat(node->name(), ":", i)] = tensor_name;
215 }
216 }
217 }
218
219 // Detect missing function inputs.
220 for (int i = 0; i < fdef->signature().input_arg_size(); ++i) {
221 const string& input_name = fdef->signature().input_arg(i).name();
222 if (input_name.empty()) {
223 return errors::InvalidArgument("Missing input ", i, " to function ",
224 name);
225 }
226 }
227
228 // Remap input names. We do this as a second pass to allow the nodes to be in
229 // any order.
230 for (int n_index = 0; n_index < fdef->node_def_size(); ++n_index) {
231 NodeDef* node_def = fdef->mutable_node_def(n_index);
232 for (int i = 0; i < node_def->input_size(); ++i) {
233 if (str_util::StartsWith(node_def->input(i), "^")) {
234 // Control input
235 const string normalized =
236 node_names.Renormalize(node_def->input(i).substr(1));
237 if (normalized.empty()) {
238 return errors::InvalidArgument(
239 "Could not remap control input ", i, ", '", node_def->input(i),
240 "', of node '", node_def->name(), "' in function ", name);
241 }
242 *node_def->mutable_input(i) = strings::StrCat("^", normalized);
243 } else {
244 const auto iter = tensor_renaming.find(node_def->input(i));
245 if (iter == tensor_renaming.end()) {
246 return errors::InvalidArgument(
247 "Could not remap input ", i, ", '", node_def->input(i),
248 "', of node '", node_def->name(), "' in function ", name);
249 }
250 *node_def->mutable_input(i) = iter->second;
251 }
252 }
253 }
254
255 // Remap return values.
256 for (int r = 0; r < fdef->signature().output_arg_size(); ++r) {
257 const string& ret_name = fdef->signature().output_arg(r).name();
258 if (ret_name.empty()) {
259 return errors::InvalidArgument("Missing output ", r, " to function ",
260 name);
261 }
262 const string& return_value = return_values[ret_name];
263 const auto iter = tensor_renaming.find(return_value);
264 if (iter == tensor_renaming.end()) {
265 return errors::InvalidArgument("Could not remap return value ", r, ", '",
266 ret_name, "', of '", return_value,
267 "' in function ", name);
268 }
269 (*fdef->mutable_ret())[ret_name] = iter->second;
270 }
271
272 return Status::OK();
273}
