39std::map<int, OptionalTensor> SnapshotResourceVariables(OpKernelContext* ctx,
40 int num_variables) {
41 std::map<int, OptionalTensor> snapshot;
42 int first_variable = ctx->num_inputs() - num_variables;
43 for (int i = 0; i < num_variables; ++i) {
44 Var* variable = nullptr;
45 ResourceHandle handle = HandleFromInput(ctx, first_variable + i);
46 OptionalTensor& tensor = snapshot[first_variable + i];
47 if (LookupResource(ctx, handle, &variable).ok()) {
48 tf_shared_lock lock(*variable->mu());
49 tensor.name = handle.name();
50 tensor.present = true;
51 tensor.value = *variable->tensor();
52 }
53 }
54 return snapshot;
55}

62xla::StatusOr<gpu::DeviceMemoryBase> XlaAllocator::Allocate(
63 int device_ordinal, uint64 size, bool retry_on_failure) {
64 void* data = wrapped_->AllocateRaw(Allocator::kAllocatorAlignment, size);
65 if (data == nullptr) {
66 return errors::ResourceExhausted("Out of memory while trying to allocate ",
67 size, " bytes.");
68 } else {
69 return gpu::DeviceMemoryBase(data, size);
70 }
71}
72
73Status XlaAllocator::Deallocate(int device_ordinal,
74 gpu::DeviceMemoryBase* mem) {
75 wrapped_->DeallocateRaw(mem->opaque());
76 return Status::OK();
77}

80// Return the 'index''th subtree of the given ShapedBuffer as a
81// ScopedShapedBuffer. The returned ScopedShapedBuffer takes ownership of the
82// subtree, and sets the input's buffer pointers to nullptr for the subtree.
83std::unique_ptr<xla::ScopedShapedBuffer> ExtractSubShapedBuffer(
84 xla::ShapedBuffer* shaped_buffer, int index,
85 xla::DeviceMemoryAllocator* allocator) {
86 xla::Shape on_host_shape = xla::ShapeUtil::GetTupleElementShape(
87 shaped_buffer->on_host_shape(), index);
88 xla::Shape on_device_shape = xla::ShapeUtil::GetTupleElementShape(
89 shaped_buffer->on_device_shape(), index);
90
91 xla::ShapedBuffer sub_shaped_buffer(on_host_shape, on_device_shape,
92 shaped_buffer->platform(),
93 shaped_buffer->device_ordinal());
94
95 auto& shape_tree = shaped_buffer->buffers();
96 auto& sub_shape_tree = sub_shaped_buffer.buffers();
97 sub_shape_tree.CopySubtreeFrom(shape_tree,
98 /*source_base_index=*/{index},
99 /*target_base_index=*/{});
100 for (auto& index_to_buffer : shape_tree) {
101 if (!index_to_buffer.first.empty() && index_to_buffer.first[0] == index) {
102 index_to_buffer.second = gpu::DeviceMemoryBase(nullptr, 0);
103 }
104 }
105 return xla::ScopedShapedBuffer::MakeScoped(&sub_shaped_buffer, allocator)
106 .ValueOrDie();
107}

110XlaComputationLaunchContext::XlaComputationLaunchContext(
111 int64 num_resource_args, xla::LocalClient* client,
112 xla::DeviceMemoryAllocator* xla_allocator, bool allocate_xla_tensors)
113 : num_resource_args_(num_resource_args),
114 client_(client),
115 xla_allocator_(xla_allocator),
116 allocate_xla_tensors_(allocate_xla_tensors) {}

118void XlaComputationLaunchContext::PopulateInputs(
119 OpKernelContext* ctx, const XlaCompiler::CompilationResult* kernel,
120 const std::map<int, OptionalTensor>& variables) {
121 // Build xla::ShapedBuffers that point directly to the Tensor buffers.
122 arg_buffers_.reserve(kernel->xla_input_shapes.size() + 1);
123 arg_buffers_.resize(kernel->xla_input_shapes.size());
124 arg_ptrs_ = std::vector<xla::ShapedBuffer*>(arg_buffers_.size());
125
126 // Pass remaining parameters.
127 const Tensor* t;
128 for (int i = 0; i < kernel->xla_input_shapes.size(); ++i) {
129 int arg_num = kernel->input_mapping[i];
130 const xla::Shape& shape = kernel->xla_input_shapes[i];
131 if (variables.count(arg_num)) {
132 t = &(variables.at(arg_num).value);
133 CHECK(t);
134 } else {
135 t = &(ctx->input(arg_num));
136 }
137
138 const xla::Shape on_device_shape =
139 client_->backend().transfer_manager()->HostShapeToDeviceShape(shape);
140 if (xla::ShapeUtil::IsTuple(on_device_shape)) {
141 const XlaTensor* xla_tensor = XlaTensor::FromTensor(t);
142 CHECK(xla_tensor && xla_tensor->has_shaped_buffer());
143 arg_ptrs_[i] =
144 const_cast<xla::ShapedBuffer*>(&xla_tensor->shaped_buffer());
145 } else {
146 CHECK(xla::ShapeUtil::Equal(shape, on_device_shape))
147 << "On-device shape "
148 << xla::ShapeUtil::HumanStringWithLayout(on_device_shape)
149 << " not the same as on-host shape "
150 << xla::ShapeUtil::HumanStringWithLayout(shape);
151 gpu::DeviceMemoryBase dmem = XlaTensor::DeviceMemoryFromTensor(*t);
152 arg_buffers_[i] = xla::MakeUnique<xla::ShapedBuffer>(
153 /*on_host_shape=*/shape, /*on_device_shape=*/shape,
154 client_->platform(), client_->default_device_ordinal());
155 arg_buffers_[i]->set_buffer(dmem, /*index=*/{});
156 arg_ptrs_[i] = arg_buffers_[i].get();
157 }
158 }

161void XlaComputationLaunchContext::PopulateOutputs(
162 OpKernelContext* ctx, const XlaCompiler::CompilationResult* kernel,
163 std::unique_ptr<xla::ScopedShapedBuffer> output) {
164 gpu::Stream* stream =
165 ctx->op_device_context() ? ctx->op_device_context()->stream() : nullptr;
166
167 // Computation output should always be a tuple.
168 if (VLOG_IS_ON(2)) {
169 VLOG(2) << "Result tuple shape: " << output->on_host_shape().DebugString();
170 VLOG(2) << "Result tuple shape (on device): "
171 << output->on_device_shape().DebugString();
172 }
173 CHECK_EQ(ctx->num_outputs(), kernel->outputs.size());
174
175 // Copy XLA results to the OpOutputList.
176 int output_num = 0;
177 for (int i = 0; i < ctx->num_outputs(); ++i) {
178 Allocator* allocator = ctx->device()->GetAllocator({});
179 if (kernel->outputs[i].is_constant) {
180 // Output is a constant.
181 const Tensor& const_tensor = kernel->outputs[i].constant_value;
182 Tensor* output_tensor;
183 const size_t total_bytes = const_tensor.TotalBytes();
184 if (stream && total_bytes > 0) {
185 // Copy host -> device. (Empty tensors don't have backing buffers.)
186 // Manually allocate memory using an XlaTensorBuffer so we can allocate
187 // as much memory as the device requires (as given by
188 // GetByteSizeRequirement). This avoids XlaTransferManager having to
189 // reallocate the device buffer later.
190 VLOG(1) << "Constant output tensor on device";
191
192 OP_REQUIRES_OK(
193 ctx, ctx->allocate_output(i, const_tensor.shape(), &output_tensor));
194 if (XlaTensor* xla_tensor = XlaTensor::FromTensor(output_tensor)) {
195 OP_REQUIRES_OK(ctx, xla_tensor->AllocateShapedBuffer(
196 const_tensor.dtype(), const_tensor.shape(),
197 client_, stream->parent()->device_ordinal()));
198 }
199
200 Device* device = dynamic_cast<Device*>(ctx->device());
201 OP_REQUIRES(ctx, device != nullptr,
202 errors::Internal("DeviceBase was not a Device."));
203 ctx->op_device_context()->CopyCPUTensorToDevice(
204 &const_tensor, device, output_tensor,
205 [&](Status status) { TF_CHECK_OK(status); });
206
207 if (device->device_type() == DEVICE_GPU) {
208 // The GPUDeviceContext enqueues the host->device transfer in a
209 // separate stream from the main compute stream. We must ensure the
210 // compute stream is synchronized with the host->device transfer
211 // stream now otherwise we will create a race condition.
212 auto* gpu_device_context =
213 static_cast<GPUDeviceContext*>(ctx->op_device_context());
214 gpu_device_context->stream()->ThenWaitFor(
215 gpu_device_context->host_to_device_stream());
216 }
217 } else {
218 // No copy required.
219 ctx->set_output(i, const_tensor);
220 output_tensor = ctx->mutable_output(i);
221 }
222 if (XlaTensor* xla_tensor = XlaTensor::FromTensor(output_tensor)) {
223 xla_tensor->set_host_tensor(const_tensor);
224 }
225 } else {
226 const TensorShape& shape = kernel->outputs[i].shape;
227 VLOG(2) << "Retval " << i << " shape " << shape.DebugString();
228
229 gpu::DeviceMemoryBase buffer = output->buffer({output_num});
230 if (allocate_xla_tensors_) {
231 Tensor* output_tensor;
232 OP_REQUIRES_OK(ctx, ctx->allocate_output(i, shape, &output_tensor));
233 XlaTensor* xla_tensor = XlaTensor::FromTensor(output_tensor);
234 CHECK(xla_tensor);
235 xla_tensor->set_shaped_buffer(
236 ExtractSubShapedBuffer(output.get(), output_num, xla_allocator_));
237 } else {
238 Tensor output_tensor = XlaTensorBuffer::MakeTensor(
239 ctx->expected_output_dtype(i), shape, buffer, allocator);
240 output->set_buffer(gpu::DeviceMemoryBase(nullptr, 0), {output_num});
241 ctx->set_output(i, output_tensor);
242 }
243 ++output_num;
244 }
245
246 if (VLOG_IS_ON(3)) {
247 VLOG(3) << ctx->mutable_output(i)->DebugString();
248 }
249 }
250
251 // Apply variable updates, if any.
252 VLOG(2) << "Applying variable updates";
253 for (int i = 0; i < kernel->resource_updates.size(); ++i) {
254 Allocator* allocator = ctx->device()->GetAllocator({});
255 const XlaCompiler::ResourceUpdate& write = kernel->resource_updates[i];
256 OP_REQUIRES(ctx,
257 write.input_index >= 0 && write.input_index < ctx->num_inputs(),
258 errors::Internal("Invalid input index for variable write."));
259
260 gpu::DeviceMemoryBase buffer = output->buffer({output_num});
261
262 Var* variable = nullptr;
263 // TODO(b/35625933): tensorflow::Var should contain a PersistentTensor,
264 // not a Tensor.
265 OP_REQUIRES_OK(ctx, LookupOrCreateResource<Var>(
266 ctx, HandleFromInput(ctx, write.input_index),
267 &variable, [this, ctx, &write](Var** ptr) {
268 *ptr = new Var(write.type);
269 return Status::OK();
270 }));
271
272 core::ScopedUnref s(variable);
273
274 mutex_lock ml(*variable->mu());
275 OP_REQUIRES(ctx, variable->tensor()->dtype() == write.type,
276 errors::Internal("Mismatched type in variable write"));
277
278 if (allocate_xla_tensors_) {
279 Tensor output_tensor;
280 OP_REQUIRES_OK(
281 ctx, ctx->allocate_temp(write.type, write.shape, &output_tensor));
282 XlaTensor* xla_tensor = XlaTensor::FromTensor(&output_tensor);
283 CHECK(xla_tensor);
284 xla_tensor->set_shaped_buffer(
285 ExtractSubShapedBuffer(output.get(), output_num, xla_allocator_));
286 *variable->tensor() = output_tensor;
287 } else {
288 Tensor output_tensor = XlaTensorBuffer::MakeTensor(
289 write.type, write.shape, buffer, allocator);
290 output->set_buffer(gpu::DeviceMemoryBase(nullptr, 0), {output_num});
291 *variable->tensor() = output_tensor;
292 }
293 ++output_num;
294 }
295}
