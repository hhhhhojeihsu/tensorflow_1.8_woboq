36std::unordered_map<string, const AttrTypeMap*>* OpNameToAttrTypeMap() {
37 static auto* const m = new std::unordered_map<string, const AttrTypeMap*>;
38 return m;
39}

41const uint32 kIsList = 1U << 31;

45Status OpDefForOp(const char* op_name, const OpDef** op_def) {
46 const OpRegistrationData* op_reg_data = nullptr;
47 Status s = OpRegistry::Global()->LookUp(op_name, &op_reg_data);
48 if (s.ok()) {
49 *op_def = &op_reg_data->op_def;
50 }
51 return s;
52}

54Status AttrTypeMapForOp(const char* op_name, const AttrTypeMap** out) {
55 mutex_lock l(g_op_name_to_attr_type_map_lock);
56 *out = gtl::FindPtrOrNull(*OpNameToAttrTypeMap(), op_name);
57 if (*out != nullptr) return Status::OK();
58 const OpDef* op_def = nullptr;
59 Status s = OpDefForOp(op_name, &op_def);
60 if (!s.ok()) return s;
61 std::unique_ptr<AttrTypeMap> m(new AttrTypeMap);
62 // TODO(agarwal): Avoid having to create this "registry" at runtime,
63 // perhaps can be done at op registration time?
64 for (const auto& attr : op_def->attr()) {
65 string type = attr.type();
66 const bool is_list = (type.length() > 6 && type.compare(0, 4, "list") == 0);
67 if (is_list) {
68 type = type.substr(5, type.length() - 6);
69 }
70 uint32 t = is_list ? kIsList : 0;
71 if (type == "string") {
72 t |= TF_ATTR_STRING;
73 } else if (type == "int") {
74 t |= TF_ATTR_INT;
75 } else if (type == "float") {
76 t |= TF_ATTR_FLOAT;
77 } else if (type == "bool") {
78 t |= TF_ATTR_BOOL;
79 } else if (type == "type") {
80 t |= TF_ATTR_TYPE;
81 } else if (type == "shape") {
82 t |= TF_ATTR_SHAPE;
83 } else if (type == "tensor") {
84 t |= TF_ATTR_TENSOR;
85 } else if (type == "func") {
86 t |= TF_ATTR_FUNC;
87 } else {
88 return errors::Unimplemented(
89 "TODO(agarwal): Enable support for ops with attributes of type '",
90 type, "'");
91 }
92 gtl::InsertIfNotPresent(m.get(), attr.name(), t);
93 }
94 *out = m.get();
95 (*OpNameToAttrTypeMap())[op_name] = m.release();
96 return Status::OK();
97}

120void AttrBuilder::FillAttrValueMap(AttrValueMap* m,
121 bool include_those_in_node_def) const {
122 for (const auto& p : string_attrs_) {
123 SetInAttrValueMap(m, p.first, p.second);
124 }
125 for (const auto& p : int_attrs_) {
126 SetInAttrValueMap(m, p.first, p.second);
127 }
128 for (const auto& p : float_attrs_) {
129 SetInAttrValueMap(m, p.first, p.second);
130 }
131 for (const auto& p : bool_attrs_) {
132 SetInAttrValueMap(m, p.first, p.second);
133 }
134 for (const auto& p : type_attrs_) {
135 SetInAttrValueMap(m, p.first, p.second);
136 }
137 if (include_those_in_node_def && node_def_ != nullptr) {
138 for (AttrValueMap::const_iterator it = node_def_->attr().begin();
139 it != node_def_->attr().end(); ++it) {
140 m->insert(*it);
141 }
142 }
143}

145const NodeDef& AttrBuilder::BuildNodeDef() {
146 if (node_def_finalized_) return *node_def_;
147 MayBeInitializeNodeDef();
148 for (int i = 0; i < num_inputs_; ++i) {
149 node_def_->add_input("dummy_input");
150 }
151 FillAttrValueMap(node_def_->mutable_attr(), false);
152 node_def_finalized_ = true;
153 return *node_def_;
154}

156Status AttrTypeByName(const AttrTypeMap& m, const string& attr_name,
157 TF_AttrType* out, unsigned char* is_list) {
158 auto* t = gtl::FindOrNull(m, attr_name);
159 if (t == nullptr) {
160 return errors::InvalidArgument("Attribute '", attr_name,
161 "' does not exist for this operation");
162 }
163 *out = static_cast<TF_AttrType>(*t & ~kIsList);
164 if (*t & kIsList) {
165 *is_list = 1;
166 } else {
167 *is_list = 0;
168 }
169 return Status::OK();
170}

173inline tensorflow::Fprint128 FingerprintCat128(const tensorflow::Fprint128& a,
174 const tensorflow::Fprint128& b) {
175 return {tensorflow::FingerprintCat64(a.low64, b.low64),
176 tensorflow::FingerprintCat64(a.low64, b.low64)};
177}
178
179void CombineUnordered(const tensorflow::Fprint128& a,
180 tensorflow::Fprint128* b) {
181 b->low64 += a.low64;
182 b->high64 += a.high64;
183}

185inline tensorflow::Fprint128 CacheKeyHelper(StringPiece s,
186 const tensorflow::Fprint128& b) {
187 // TODO(agarwal): avoid ToString().
188 tensorflow::Fprint128 a = tensorflow::Fingerprint128(s.ToString());
189 return FingerprintCat128(a, b);
190}
191
192inline tensorflow::Fprint128 CacheKeyHelper(StringPiece s, uint64 b) {
193 return CacheKeyHelper(s, {b, b});
194}

198tensorflow::Fprint128 AttrBuilder::CacheKey(const string& device) const {
199 tensorflow::Fprint128 f = tensorflow::Fingerprint128(op_name_);
200 f = tensorflow::FingerprintCat128(f, tensorflow::Fingerprint128(device));
201 if (node_def_ != nullptr) {
202 // Some attributes are directly written to node_def_ instead of being
203 // stored explicitly.
204 string value;
205 for (const auto& attr : node_def_->attr()) {
206 attr.second.SerializeToString(&value);
207 CombineUnordered(
208 CacheKeyHelper(attr.first, tensorflow::Fingerprint128(value)), &f);
209 }
210 // Note that node_def_ may be created but not finalized. This can happen
211 // when the creation was triggered by a call to Set, but BuildNodeDef has
212 // not been called.
213 if (node_def_finalized_) return f;
214 }
215 for (const auto& p : string_attrs_) {
216 // TODO(agarwal): avoid ToString().
217 CombineUnordered(CacheKeyHelper(p.first, tensorflow::Fingerprint128(
218 p.second.ToString())),
219 &f);
220 }
221 for (const auto& p : int_attrs_) {
222 CombineUnordered(CacheKeyHelper(p.first, static_cast<uint64>(p.second)),
223 &f);
224 }
225 static std::hash<float> float_hasher;
226 for (const auto& p : float_attrs_) {
227 CombineUnordered(
228 CacheKeyHelper(p.first, static_cast<uint64>(float_hasher(p.second))),
229 &f);
230 }
231 for (const auto& p : bool_attrs_) {
232 CombineUnordered(CacheKeyHelper(p.first, p.second ? 1u : 0u), &f);
233 }
234 for (const auto& p : type_attrs_) {
235 CombineUnordered(CacheKeyHelper(p.first, static_cast<uint64>(p.second)),
236 &f);
237 }
238 return f;
239}

241void AttrBuilder::MayBeInitializeNodeDef() {
242 if (node_def_ == nullptr) {
243 node_def_.reset(new NodeDef());
244 node_def_->set_name(op_name_);
245 node_def_->set_op(op_name_);
246 }
