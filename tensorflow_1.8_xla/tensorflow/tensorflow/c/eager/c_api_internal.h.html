49struct TFE_ContextOptions {
50 TF_SessionOptions session_options;
51 // true if async execution is enabled.
52 bool async = false;
53 TFE_ContextDevicePlacementPolicy policy{TFE_DEVICE_PLACEMENT_SILENT};
54};
55
56struct TFE_Context {
57 explicit TFE_Context(const tensorflow::SessionOptions& opts,
58 TFE_ContextDevicePlacementPolicy default_policy,
59 bool async,
60 std::unique_ptr<tensorflow::DeviceMgr> device_mgr,
61 tensorflow::Rendezvous* rendezvous)
62 : context(opts,
63 static_cast<tensorflow::ContextDevicePlacementPolicy>(
64 default_policy),
65 async, std::move(device_mgr), rendezvous) {}
66
67 tensorflow::EagerContext context;
68};
69
70struct TFE_TensorHandle {
71 TFE_TensorHandle(const tensorflow::Tensor& t, tensorflow::Device* d,
72 tensorflow::Device* op_device)
73 : handle(new tensorflow::TensorHandle(t, d, op_device, nullptr)) {}
74
75 TFE_TensorHandle(tensorflow::uint64 node_id, tensorflow::DataType dtype,
76 tensorflow::EagerContext* ctx)
77 : handle(new tensorflow::TensorHandle(node_id, dtype, ctx)) {}
78
79 TFE_TensorHandle(tensorflow::TensorHandle* handle) : handle(handle) {}
80
81 tensorflow::TensorHandle* handle;
82};
83
84struct TFE_Op {
85 // t is NULL iff the TFE_Op corresponds to a TensorFlow function instead of a
86 // primitive operation.
87 TFE_Op(TFE_Context* ctx, const char* op, const tensorflow::AttrTypeMap* t)
88 : ctx(ctx), name(op), attrs(op), attr_types(t), device(nullptr) {}
89
90 ~TFE_Op();
91
92 bool const is_function() const { return attr_types == nullptr; }
93
94 TFE_Context* ctx; // Must outlive the TFE_Op.
95 const tensorflow::string name;
96 tensorflow::AttrBuilder attrs;
97 const tensorflow::AttrTypeMap* attr_types;
98 tensorflow::gtl::InlinedVector<tensorflow::TensorHandle*, 4> inputs;
99 tensorflow::Device* device;
100 bool use_xla = false;
101};
