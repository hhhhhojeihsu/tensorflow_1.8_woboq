74struct TF_Graph {
75 TF_Graph();
76
77 tensorflow::mutex mu;
78 tensorflow::Graph graph GUARDED_BY(mu);
79
80 // Runs shape inference.
81 tensorflow::ShapeRefiner refiner GUARDED_BY(mu);
82
83 // Maps from name of an operation to the Node* in 'graph'.
84 std::unordered_map<tensorflow::string, tensorflow::Node*> name_map
85 GUARDED_BY(mu);
86
87 // The keys of this map are all the active sessions using this graph. Each
88 // value records whether the graph has been mutated since the corresponding
89 // session has been run (this is detected in RecordMutation function). If the
90 // string is empty, no mutation has occurred. Otherwise the string is a
91 // description of the mutation suitable for returning to the user.
92 //
93 // Sessions are added to this map in TF_NewSession, and removed in
94 // TF_DeleteSession.
95 // TF_Graph may only / must be deleted when
96 // sessions.size() == 0 && delete_requested == true
97 //
98 // TODO(b/74949947): mutations currently trigger a warning instead of a bad
99 // status, this should be reverted when possible.
100 tensorflow::gtl::FlatMap<TF_Session*, tensorflow::string> sessions
101 GUARDED_BY(mu);
102 bool delete_requested GUARDED_BY(mu); // set true by TF_DeleteGraph
103
104 // Used to link graphs contained in TF_WhileParams to the parent graph that
105 // will eventually contain the full while loop.
106 TF_Graph* parent;
107 TF_Output* parent_inputs;
108};

199// Set the shapes and types of the output's handle.
200//
201// The lengths of the arrays pointed to by `shapes`, `ranks`, and `types` must
202// all be equal to `num_shapes_and_types`. If `ranks[i] != -1`, (i.e., if the
203// rank is known), then it must be equal to the length of `shapes[i]`; if
204// `ranks[i] == 1`, then `shapes[i]` may be nullptr.
205//
206// TODO(akshayka): Implement a corresponding getter method.
207void TF_GraphSetOutputHandleShapesAndTypes(TF_Graph* graph, TF_Output output,
208 int num_shapes_and_types,

211 const TF_DataType* types,
212 TF_Status* status);
