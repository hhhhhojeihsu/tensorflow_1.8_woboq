16#include "tensorflow/core/debug/debug_node_key.h"

18#include "tensorflow/core/lib/strings/str_util.h"
19
20namespace tensorflow {
21
22const char* const DebugNodeKey::kMetadataFilePrefix = "_tfdbg_";
23
24const char* const DebugNodeKey::kDeviceTag = "device_";
25
26DebugNodeKey::DebugNodeKey(const string& device_name, const string& node_name,
27 const int32 output_slot, const string& debug_op)
28 : device_name(device_name),
29 node_name(node_name),
30 output_slot(output_slot),
31 debug_op(debug_op),
32 debug_node_name(
33 strings::StrCat(node_name, ":", output_slot, ":", debug_op)),
34 device_path(DeviceNameToDevicePath(device_name)) {}
35
36bool DebugNodeKey::operator==(const DebugNodeKey& other) const {
37 return (device_name == other.device_name && node_name == other.node_name &&
38 output_slot == other.output_slot && debug_op == other.debug_op);
39}
40
41bool DebugNodeKey::operator!=(const DebugNodeKey& other) const {
42 return !((*this) == other);
43}
44
45const string DebugNodeKey::DeviceNameToDevicePath(const string& device_name) {
46 return strings::StrCat(
47 kMetadataFilePrefix, kDeviceTag,
48 str_util::StringReplace(
49 str_util::StringReplace(device_name, ":", "_", true), "/", ",",
50 true));
51}
