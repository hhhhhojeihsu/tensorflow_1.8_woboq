16#include "tensorflow/core/debug/debug_callback_registry.h"

18namespace tensorflow {
19
20DebugCallbackRegistry::DebugCallbackRegistry() {}
21
22/*static */ DebugCallbackRegistry* DebugCallbackRegistry::instance_ = nullptr;
23
24DebugCallbackRegistry* DebugCallbackRegistry::singleton() {
25 if (instance_ == nullptr) {
26 instance_ = new DebugCallbackRegistry();
27 }
28 return instance_;
29}
30
31void DebugCallbackRegistry::RegisterCallback(const string& key,
32 EventCallback callback) {
33 mutex_lock lock(mu_);
34 keyed_callback_[key] = std::move(callback);
35}
36
37DebugCallbackRegistry::EventCallback* DebugCallbackRegistry::GetCallback(
38 const string& key) {
39 mutex_lock lock(mu_);
40 auto iter = keyed_callback_.find(key);
41 return iter == keyed_callback_.end() ? nullptr : &iter->second;
42}
43
44void DebugCallbackRegistry::UnregisterCallback(const string& key) {
45 mutex_lock lock(mu_);
46 keyed_callback_.erase(key);
47}
