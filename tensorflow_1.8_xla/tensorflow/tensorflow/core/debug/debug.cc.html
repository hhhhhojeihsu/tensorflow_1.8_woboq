21namespace tensorflow {
22namespace {
23
24// Registers a concrete implementation of DebuggerState for use by
25// DirectSession.
26class DebuggerStateRegistration {
27 public:
28 static std::unique_ptr<DebuggerStateInterface> CreateDebuggerState(
29 const DebugOptions& options) {
30 return std::unique_ptr<DebuggerStateInterface>(new DebuggerState(options));
31 }
32
33 static std::unique_ptr<DebugGraphDecoratorInterface>
34 CreateDebugGraphDecorator(const DebugOptions& options) {
35 return std::unique_ptr<DebugGraphDecoratorInterface>(
36 new DebugGraphDecorator(options));
37 }
38
39 DebuggerStateRegistration() {
40 DebuggerStateRegistry::RegisterFactory(CreateDebuggerState);
41 DebugGraphDecoratorRegistry::RegisterFactory(CreateDebugGraphDecorator);
42 }
43};
44
45static DebuggerStateRegistration register_debugger_state_implementation;
