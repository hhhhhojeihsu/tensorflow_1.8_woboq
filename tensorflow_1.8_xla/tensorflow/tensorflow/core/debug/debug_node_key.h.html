16#ifndef TENSORFLOW_DEBUG_NODE_KEY_H_
17#define TENSORFLOW_DEBUG_NODE_KEY_H_
18
19#include <string>
20
21#include "tensorflow/core/platform/types.h"
22
23namespace tensorflow {
24
25// Encapsulates debug information for a node that was observed.
26struct DebugNodeKey {
27 static const char* const kMetadataFilePrefix;
28 static const char* const kDeviceTag;
29
30 DebugNodeKey(const string& device_name, const string& node_name,
31 const int32 output_slot, const string& debug_op);
32
33 // Converts a device name string to a device path string.
34 // E.g., /job:localhost/replica:0/task:0/cpu:0 will be converted to
35 // ,job_localhost,replica_0,task_0,cpu_0.
36 static const string DeviceNameToDevicePath(const string& device_name);
37
38 bool operator==(const DebugNodeKey& other) const;
39 bool operator!=(const DebugNodeKey& other) const;
40
41 const string device_name;
42 const string node_name;
43 const int32 output_slot;
44 const string debug_op;
45 const string debug_node_name;
46 const string device_path;
47};
48
49} // namespace tensorflow
50
51#endif // TENSORFLOW_DEBUG_NODE_KEY_H_
