24// Cwise binary ops
25Status GradForUnaryCwise(FunctionDef* g, std::vector<FDH::Node> nodes) {
26 for (auto& n : nodes) {
27 if (n.attr.empty()) {
28 n.attr = {{"T", "$T"}};
29 }
30 }
31 *g = FDH::Define(
32 // Arg defs
33 {"x: T", "dy: T"},
34 // Ret val defs
35 {"dx: T"},
36 // Attr defs
37 {{"T: {half, float, double}"}},
38 // Nodes
39 nodes);
40 return Status::OK();
41}

375// Cwise binary ops
376//
377// TODO(zhifengc): This can be arrange as a function in the standard
378// library.
379Status GradForBinaryCwise(FunctionDef* g, std::vector<FDH::Node> body) {
380 // clang-format off
381 std::vector<FDH::Node> nodes = {
382 {{"sx"}, "Shape", {"x"}},
383 {{"sy"}, "Shape", {"y"}},
384 };
385 nodes.insert(nodes.end(), body.begin(), body.end());
386 std::vector<FDH::Node> reshapes = {
387 {{"rx", "ry"}, "BroadcastGradientArgs", {"sx", "sy"}},
388 {{"sum_gx"}, "Sum", {"gx", "rx"}},
389 {{"dx"}, "Reshape", {"sum_gx", "sx"}},
390 {{"sum_gy"}, "Sum", {"gy", "ry"}},
391 {{"dy"}, "Reshape", {"sum_gy", "sy"}},
392 };
393 nodes.insert(nodes.end(), reshapes.begin(), reshapes.end());
394
395 // clang-format on
396 for (auto& n : nodes) {
397 // "BroadcastGradientArgs" doesn't need any attrs.
398 if (n.attr.empty() && n.op != "BroadcastGradientArgs") {
399 n.attr = {{"T", "$T"}};
400 }
401 }
402 *g = FDH::Define(
403 // Arg defs
404 {"x: T", "y: T", "dz: T"},
405 // Ret val defs
406 {"dx: T", "dy: T"},
407 // Attr defs
408 {{"T: {half, float, double}"}},
409 // Nodes
410 nodes);
411 return Status::OK();
412}

434Status MulGrad(const AttrSlice& attrs, FunctionDef* g) {
435 DataType T;
436 TF_RETURN_IF_ERROR(GetNodeAttr(attrs, "T", &T));
437 if (T == DT_COMPLEX64 || T == DT_COMPLEX128) {
438 return GradForBinaryCwise(
439 g, {
440 {{"cy"}, "Conj", {"y"}, {}, {"dz"}},
441 {{"gx"}, "Mul", {"dz", "cy"}}, // dz * Conj(y)
442 {{"cx"}, "Conj", {"x"}, {}, {"dz"}},
443 {{"gy"}, "Mul", {"cx", "dz"}}, // Conj(x) * dz
444 });
445 } else {
446 // clang-format off
447 return GradForBinaryCwise(g, {
448 {{"gx"}, "Mul", {"dz", "y"}}, // dz * y
449 {{"gy"}, "Mul", {"x", "dz"}}, // x * dz
450 });
451 // clang-format on
452 }
453}

482Status PowGrad(const AttrSlice& attrs, FunctionDef* g) {
483 // clang-format off
484 std::vector<FDH::Node> nodes = {
485 {{"z"}, "Pow", {"x", "y"}},
486 // dz * y * Pow(x, y - 1)
487 FDH::Const("const_zero", 0.0f),
488 FDH::Const("const_one", 1.0f),
489 {{"zero"}, "Cast", {"const_zero"}, {{"SrcT", DT_FLOAT}, {"DstT", "$T"}}},
490 {{"one"}, "Cast", {"const_one"}, {{"SrcT", DT_FLOAT}, {"DstT", "$T"}}},
491 {{"t0"}, "Sub", {"y", "one"}, {}, {"dz"}},
492 {{"t1"}, "Pow", {"x", "t0"}},
493 {{"t2"}, "Mul", {"dz", "y"}},
494 {{"gx"}, "Mul", {"t1", "t2"}},
495 {{"unsafe_log"}, "Log", {"x"}, {}, {"dz"}},
496 {{"zeros"}, "ZerosLike", {"x"}}};
497 // clang-format on
498 std::vector<FDH::Node> log_x_handling;
499 DataType T;
500 TF_RETURN_IF_ERROR(GetNodeAttr(attrs, "T", &T));
501 if (T == DT_COMPLEX64 || T == DT_COMPLEX128) {
502 // dz * z * (x != 0 ? Log(x) : 0)
503 // clang-format off
504 log_x_handling = {
505 {{"nz_x"}, "NotEqual", {"x", "zero"}},
506 {{"safe_log"}, "Select", {"nz_x", "unsafe_log", "zeros"}}};
507 // clang-format on
508 } else {
509 // dz * z * (x > 0 ? Log(x) : 0)
510 // clang-format off
511 log_x_handling = {
512 {{"pos_x"}, "Greater", {"x", "zero"}},
513 {{"safe_log"}, "Select", {"pos_x", "unsafe_log", "zeros"}}};
514 // clang-format on
515 }
516 nodes.insert(nodes.end(), log_x_handling.begin(), log_x_handling.end());
517 nodes.push_back({{"t4"}, "Mul", {"dz", "z"}});
518 nodes.push_back({{"gy"}, "Mul", {"safe_log", "t4"}});
519 return GradForBinaryCwise(g, nodes);
520}

576// Reduction ops
577//
578// TODO(zhifengc): This helper is pretty ugly. Do something better.
579// TODO(zhifengc): This can be arrange as a function in the standard library.
580Status GradForReductionOp(FunctionDef* g, std::vector<FDH::Node> body) {
581 // Shape manipulation nodes.
582
583 // clang-format off
584 std::vector<FDH::Node> nodes = {
585 {{"x_shape"}, "Shape", {"x"}},
586 {{"x_rank"}, "Rank", {"x"}},
587 {{"i_shape"}, "Shape", {"i"}, {{"T", DT_INT32}}},
588 FDH::Const("zero", 0),
589 FDH::Const("one", 1),
590 // stitch_idx0 = Range(0, x_rank, 1)
591 {{"stitch_val1"}, "Fill", {"i_shape:output:0", "one:output:0"},
592 {{"T", DT_INT32}}},
593 {{"y_shape"}, "DynamicStitch",
594 {"stitch_idx0:output:0", "i",
595 "x_shape:output:0", "stitch_val1:output:0"},
596 {{"N", 2}, {"T", DT_INT32}}},
597 {{"tile_scaling"}, "Div", {"x_shape:output:0", "y_shape:merged:0"},
598 {{"T", DT_INT32}}},
599 {{"di"}, "ZerosLike", {"i"}, {{"T", DT_INT32}}}
600 };
601 // clang-format on
602 nodes.insert(nodes.end(), body.begin(), body.end());
603 for (auto& n : nodes) {
604 if (n.attr.empty()) {
605 n.attr = {{"T", "$T"}};
606 }
607 }
608 // "Range" doesn't need any attr.
609 nodes.push_back({{"stitch_idx0"},
610 "Range",
611 {"zero:output:0", "x_rank:output:0", "one:output:0"},
612 {}});
613 *g = FDH::Create("_",
614 // Input defs
615 {"x:T", "i:int32", "dy:T"},
616 // Ret val defs
617 {"dx:T", "di:int32"},
618 // Attr defs
619 {{"T: {half, float, double}"}},
620 // Nodes
621 nodes,
622 // Return values
623 {{"dx", "dx:output:0"}, {"di", "di:y:0"}});
624 return Status::OK();
625}

700static Status MatMulGradHelper(FunctionDef* g, const string& opname,
701 const string& attr_adj_x,
702 const string& attr_adj_y, const string& x0,
703 bool ax0, const string& x1, bool ax1,
704 const string& y0, bool ay0, const string& y1,
705 bool ay1) {
706 *g = FDH::Define(
707 // Arg defs
708 {"x: T", "y: T", "dz: T"},
709 // Ret val defs
710 {"dx: T", "dy: T"},
711 // Attr defs
712 {{"T: {half, float, double}"}},
713 // Nodes
714 {
715 {{"dx"},
716 opname,
717 {x0, x1},
718 {{"T", "$T"}, {attr_adj_x, ax0}, {attr_adj_y, ax1}}},
719 {{"dy"},
720 opname,
721 {y0, y1},
722 {{"T", "$T"}, {attr_adj_x, ay0}, {attr_adj_y, ay1}}},
723 });
724 return Status::OK();
725}

727Status MatMulGradCommon(const string& opname, const string& attr_adj_x,
728 const string& attr_adj_y, const AttrSlice& attrs,
729 FunctionDef* g) {
730 DataType T;
731 TF_RETURN_IF_ERROR(GetNodeAttr(attrs, "T", &T));
732 if (T == DT_COMPLEX64 || T == DT_COMPLEX128) {
733 return errors::Unimplemented(
734 "MatMul gradient for complex is not supported yet.");
735 }
736 bool ta;
737 bool tb;
738 TF_RETURN_IF_ERROR(GetNodeAttr(attrs, attr_adj_x, &ta));
739 TF_RETURN_IF_ERROR(GetNodeAttr(attrs, attr_adj_y, &tb));
740 if (!ta && !tb) {
741 return MatMulGradHelper(g, opname, attr_adj_x, attr_adj_y, "dz", false, "y",
742 true, "x", true, "dz", false);
743 }
744 if (!ta && tb) {
745 return MatMulGradHelper(g, opname, attr_adj_x, attr_adj_y, "dz", false, "y",
746 false, "dz", true, "x", false);
747 }
748 if (ta && !tb) {
749 return MatMulGradHelper(g, opname, attr_adj_x, attr_adj_y, "y", false, "dz",
750 true, "x", false, "dz", false);
751 }
752 CHECK(ta && tb);
753 return MatMulGradHelper(g, opname, attr_adj_x, attr_adj_y, "y", true, "dz",
754 true, "dz", true, "x", true);
755}
