28// Return in <out> the result of making the end of <s> a square matrix.
29Status MakeBatchSquareMatrix(InferenceContext* c, ShapeHandle input,
30 ShapeHandle* out) {
31 ShapeHandle s;
32 TF_RETURN_IF_ERROR(c->WithRankAtLeast(input, 2, &s));
33
34 DimensionHandle d;
35 TF_RETURN_IF_ERROR(c->Merge(c->Dim(s, -2), c->Dim(s, -1), &d));
36
37 ShapeHandle batch_shape;
38 TF_RETURN_IF_ERROR(c->Subshape(s, 0, -2, &batch_shape));
39 TF_RETURN_IF_ERROR(c->Concatenate(batch_shape, c->Matrix(d, d), out));
40 return Status::OK();
41}
42
43Status BatchUnchangedSquareShapeFn(InferenceContext* c) {
44 ShapeHandle out;
45 TF_RETURN_IF_ERROR(MakeBatchSquareMatrix(c, c->input(0), &out));
46 c->set_output(0, out);
47 return Status::OK();
48}

50// The first input is [...,M,N] and second input is either [...,M,K] or [...,M].
51// Output is [...,N,K] or [...,N]. If <square>, then input is [...,M,M].
52Status MatrixSolveShapeFn(InferenceContext* c, bool square) {
53 ShapeHandle lhs;
54 ShapeHandle rhs;
55 if (square) {
56 TF_RETURN_IF_ERROR(MakeBatchSquareMatrix(c, c->input(0), &lhs));
57 } else {
58 TF_RETURN_IF_ERROR(c->WithRankAtLeast(c->input(0), 2, &lhs));
59 }
60 TF_RETURN_IF_ERROR(c->WithRankAtLeast(c->input(1), 2, &rhs));
61
62 ShapeHandle lhs_batch_shape;
63 ShapeHandle rhs_batch_shape;
64 // Make the common batch subshape.
65 TF_RETURN_IF_ERROR(c->Subshape(lhs, 0, -2, &lhs_batch_shape));
66 TF_RETURN_IF_ERROR(c->Subshape(rhs, 0, -2, &rhs_batch_shape));
67 // Make sure the batch dimensions match between lhs and rhs.
68 TF_RETURN_IF_ERROR(
69 c->Merge(lhs_batch_shape, rhs_batch_shape, &lhs_batch_shape));
70
71 DimensionHandle m;
72 // lhs and rhs have the same value for m to be compatible.
73 TF_RETURN_IF_ERROR(c->Merge(c->Dim(lhs, -2), c->Dim(rhs, -2), &m));
74 DimensionHandle n = c->Dim(lhs, -1);
75 if (square) {
76 TF_RETURN_IF_ERROR(c->Merge(m, n, &n));
77 }
78
79 ShapeHandle out;
80 // Build final shape (batch_shape + n + k) in <out>.
81 TF_RETURN_IF_ERROR(c->Concatenate(lhs_batch_shape, c->Vector(n), &out));
82 TF_RETURN_IF_ERROR(c->Concatenate(out, c->Vector(c->Dim(rhs, -1)), &out));
83 c->set_output(0, out);
84 return Status::OK();
85}

87// Input is [...,N,N]. Outputs are:
88// [...,N];[0], if compute_v is false,
89// [...,N];[...,N,N], if compute_v is true.
90Status SelfAdjointEigV2ShapeFn(InferenceContext* c) {
91 ShapeHandle input;
92 TF_RETURN_IF_ERROR(MakeBatchSquareMatrix(c, c->input(0), &input));
93 DimensionHandle n;
94 TF_RETURN_IF_ERROR(c->Merge(c->Dim(input, -2), c->Dim(input, -1), &n));
95 ShapeHandle batch_shape;
96 TF_RETURN_IF_ERROR(c->Subshape(input, 0, -2, &batch_shape));
97 ShapeHandle e_shape;
98 TF_RETURN_IF_ERROR(c->Concatenate(batch_shape, c->Vector(n), &e_shape));
99 c->set_output(0, e_shape);
100 bool compute_v;
101 TF_RETURN_IF_ERROR(c->GetAttr("compute_v", &compute_v));
102 if (compute_v) {
103 ShapeHandle v_shape;
104 TF_RETURN_IF_ERROR(c->Concatenate(batch_shape, c->Matrix(n, n), &v_shape));
105 c->set_output(1, v_shape);
106 } else {
107 c->set_output(1, c->Vector(0ll));
108 }
109 return Status::OK();
110}

112// Input is [...,M,N].
113// First and second outputs are:
114// [...,M,M]; [...,M,N], if full_matrices is true,
115// [...,M,P]; [...,P,N], if full_matrices is false,
116// where P = min(M,N).
117Status QrShapeFn(InferenceContext* c) {
118 ShapeHandle input;
119 TF_RETURN_IF_ERROR(c->WithRankAtLeast(c->input(0), 2, &input));
120 DimensionHandle m = c->Dim(input, -2);
121 DimensionHandle n = c->Dim(input, -1);
122 DimensionHandle p;
123 TF_RETURN_IF_ERROR(c->Min(m, n, &p));
124 ShapeHandle batch_shape;
125 TF_RETURN_IF_ERROR(c->Subshape(input, 0, -2, &batch_shape));
126 ShapeHandle q_shape;
127 ShapeHandle r_shape;
128 bool full_matrices;
129 TF_RETURN_IF_ERROR(c->GetAttr("full_matrices", &full_matrices));
130 if (full_matrices) {
131 TF_RETURN_IF_ERROR(c->Concatenate(batch_shape, c->Matrix(m, m), &q_shape));
132 TF_RETURN_IF_ERROR(c->Concatenate(batch_shape, c->Matrix(m, n), &r_shape));
133 } else {
134 TF_RETURN_IF_ERROR(c->Concatenate(batch_shape, c->Matrix(m, p), &q_shape));
135 TF_RETURN_IF_ERROR(c->Concatenate(batch_shape, c->Matrix(p, n), &r_shape));
136 }
137 c->set_output(0, q_shape);
138 c->set_output(1, r_shape);
139 return Status::OK();
140}

142// Input is [...,M,N]. First output is [...,min(M,N)].
143// Second and third outputs are:
144// [0]; [0], if compute_uv is false.
145// [...,M,M]; [...,N,N], if compute_uv is true and full_matrices is true,
146// [...,M,P]; [...,N,P], if compute_uv is true and full_matrices is false,
147// where P = min(M,N).
148Status SvdShapeFn(InferenceContext* c) {
149 ShapeHandle input;
150 TF_RETURN_IF_ERROR(c->WithRankAtLeast(c->input(0), 2, &input));
151 DimensionHandle m = c->Dim(input, -2);
152 DimensionHandle n = c->Dim(input, -1);
153 DimensionHandle p;
154 TF_RETURN_IF_ERROR(c->Min(m, n, &p));
155 ShapeHandle batch_shape;
156 TF_RETURN_IF_ERROR(c->Subshape(input, 0, -2, &batch_shape));
157 ShapeHandle e_shape;
158 TF_RETURN_IF_ERROR(c->Concatenate(batch_shape, c->Vector(p), &e_shape));
159 c->set_output(0, e_shape);
160 bool compute_uv;
161 TF_RETURN_IF_ERROR(c->GetAttr("compute_uv", &compute_uv));
162 if (compute_uv) {
163 ShapeHandle u_shape;
164 ShapeHandle v_shape;
165 bool full_matrices;
166 TF_RETURN_IF_ERROR(c->GetAttr("full_matrices", &full_matrices));
167 if (full_matrices) {
168 TF_RETURN_IF_ERROR(
169 c->Concatenate(batch_shape, c->Matrix(m, m), &u_shape));
170 TF_RETURN_IF_ERROR(
171 c->Concatenate(batch_shape, c->Matrix(n, n), &v_shape));
172 } else {
173 TF_RETURN_IF_ERROR(
174 c->Concatenate(batch_shape, c->Matrix(m, p), &u_shape));
175 TF_RETURN_IF_ERROR(
176 c->Concatenate(batch_shape, c->Matrix(n, p), &v_shape));
177 }
178 c->set_output(1, u_shape);
179 c->set_output(2, v_shape);
180 } else {
181 c->set_output(1, c->Vector(0ll));
182 c->set_output(2, c->Vector(0ll));
183 }
184 return Status::OK();
185}

189REGISTER_OP("MatrixDeterminant")
190 .Input("input: T")
191 .Output("output: T")
192 .Attr("T: {float, double, complex64, complex128}")
193 .SetShapeFn([](InferenceContext* c) {
194 ShapeHandle input;
195 TF_RETURN_IF_ERROR(c->WithRankAtLeast(c->input(0), 2, &input));
196
197 DimensionHandle unused;
198 TF_RETURN_IF_ERROR(
199 c->Merge(c->Dim(input, -1), c->Dim(input, -2), &unused));
200
201 ShapeHandle out;
202 TF_RETURN_IF_ERROR(c->Subshape(input, 0, -2, &out));
203 c->set_output(0, out);
204 return Status::OK();
205 });

262REGISTER_OP("SelfAdjointEig")
263 .Input("input: T")
264 .Output("output: T")
265 .Attr("T: {double, float}")
266 .Deprecated(11, "Use SelfAdjointEigV2 instead.")
267 .SetShapeFn([](InferenceContext* c) {
268 ShapeHandle input;
269 TF_RETURN_IF_ERROR(MakeBatchSquareMatrix(c, c->input(0), &input));
270
271 DimensionHandle d = c->Dim(input, -1);
272 DimensionHandle d_plus_1;
273 TF_RETURN_IF_ERROR(c->Add(d, 1, &d_plus_1));
274
275 ShapeHandle s;
276 TF_RETURN_IF_ERROR(c->Subshape(input, 0, -2, &s));
277 TF_RETURN_IF_ERROR(c->Concatenate(s, c->Matrix(d_plus_1, d), &s));
278 c->set_output(0, s);
279 return Status::OK();
280 });
