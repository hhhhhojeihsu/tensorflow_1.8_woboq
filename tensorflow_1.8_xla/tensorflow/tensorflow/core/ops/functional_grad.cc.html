16#include <vector>
17#include "tensorflow/core/framework/function.h"
18#include "tensorflow/core/lib/core/errors.h"

20namespace tensorflow {
21
22typedef FunctionDefHelper FDH;
23
24Status MapAccumulateGrad(const AttrSlice& attrs, FunctionDef* ret) {
25 const NameAttrList* func;
26 TF_RETURN_IF_ERROR(GetNodeAttr(attrs, "f", &func));
27 DataType T;
28 TF_RETURN_IF_ERROR(GetNodeAttr(attrs, "T", &T));
29 int k;
30 TF_RETURN_IF_ERROR(GetNodeAttr(attrs, "K", &k));
31 // The gradient function of f.
32 // f : (K*T, T, T) -> T
33 // g : (K*T, T, T, T) -> (K*T, T, T)
34 auto grad = FDH::FunctionRef("SymbolicGradient",
35 {{"f", *func},
36 {"Tin", std::vector<DataType>(k + 3, T)},
37 {"Tout", std::vector<DataType>(k + 2, T)}});
38 *ret = FDH::Define(
39 // Arg defs
40 {"theta: K*T", "x: T", "u: T", "dy: T"},
41 // Ret val defs
42 {"dtheta: K*T", "dx: T", "du: T"},
43 // Attr defs
44 {{"T: {float, double}"}},
45 // nodes.
46 {{{"y"},
47 "MapAccumulate",
48 {"theta", "x", "u"},
49 {{"f", *func}, {"T", "$T"}, {"K", k}}},
50 {{"dtheta", "dx", "du"},
51 "MapAccumulateGrad",
52 {"theta", "x", "u", "y", "dy"},
53 {{"g", grad}, {"T", "$T"}, {"K", k}}}});
54 return Status::OK();
55}
56REGISTER_OP_GRADIENT("MapAccumulate", MapAccumulateGrad);
