28Status SparseSparseMinOrMaxShapeFn(InferenceContext* c) {
29 ShapeHandle unused;
30 TF_RETURN_IF_ERROR(c->WithRank(c->input(0), 2, &unused)); // a_indices
31 TF_RETURN_IF_ERROR(c->WithRank(c->input(1), 1, &unused)); // a_values
32 TF_RETURN_IF_ERROR(c->WithRank(c->input(2), 1, &unused)); // a_shape
33 TF_RETURN_IF_ERROR(c->WithRank(c->input(3), 2, &unused)); // b_indices
34 TF_RETURN_IF_ERROR(c->WithRank(c->input(4), 1, &unused)); // b_values
35 TF_RETURN_IF_ERROR(c->WithRank(c->input(5), 1, &unused)); // b_shape
36 c->set_output(0, c->Matrix(InferenceContext::kUnknownDim,
37 InferenceContext::kUnknownDim));
38 c->set_output(1, c->Vector(InferenceContext::kUnknownDim));
39 return Status::OK();
40}

85REGISTER_OP("SparseTensorDenseMatMul")
86 .Input("a_indices: Tindices")
87 .Input("a_values: T")
88 .Input("a_shape: int64")
89 .Input("b: T")
90 .Output("product: T")
91 .Attr("T: type")
92 .Attr("Tindices: {int32,int64} = DT_INT64")
93 .Attr("adjoint_a: bool = false")
94 .Attr("adjoint_b: bool = false")
95 .SetShapeFn([](InferenceContext* c) {
96 DimensionHandle unused_dim;
97 ShapeHandle unused;
98 ShapeHandle b;
99 ShapeHandle a_shape;
100 ShapeHandle a_shape_shape;
101 TF_RETURN_IF_ERROR(c->WithRank(c->input(0), 2, &unused)); // a_indices
102 TF_RETURN_IF_ERROR(c->WithRank(c->input(1), 1, &unused)); // a_values
103 TF_RETURN_IF_ERROR(c->MakeShapeFromShapeTensor(2, &a_shape));
104 TF_RETURN_IF_ERROR(c->WithRank(a_shape, 2, &a_shape));
105 TF_RETURN_IF_ERROR(c->WithRank(c->input(3), 2, &b));
106
107 bool adjoint_a;
108 bool adjoint_b;
109 TF_RETURN_IF_ERROR(c->GetAttr("adjoint_a", &adjoint_a));
110 TF_RETURN_IF_ERROR(c->GetAttr("adjoint_b", &adjoint_b));
111
112 DimensionHandle output_right = c->Dim(b, adjoint_b ? 0 : 1);
113 DimensionHandle output_left = c->Dim(a_shape, adjoint_a ? 1 : 0);
114 DimensionHandle inner_left = c->Dim(a_shape, adjoint_a ? 0 : 1);
115 DimensionHandle inner_right = c->Dim(b, adjoint_b ? 1 : 0);
116 TF_RETURN_IF_ERROR(c->Merge(inner_left, inner_right, &unused_dim));
117 c->set_output(0, c->Matrix(output_left, output_right));
118 return Status::OK();
119 });

208REGISTER_OP("SparseConcat")
209 .Input("indices: N * int64")
210 .Input("values: N * T")
211 .Input("shapes: N * int64")
212 .Output("output_indices: int64")
213 .Output("output_values: T")
214 .Output("output_shape: int64")
215 .Attr("concat_dim: int")
216 .Attr("N: int >= 2")
217 .Attr("T: type")
218 .SetShapeFn([](InferenceContext* c) {
219 // These accumulates the sum.
220 DimensionHandle output_row_count = c->MakeDim(0ll);
221
222 // These are only merged.
223 DimensionHandle output_ind_cols = c->UnknownDim();
224 ShapeHandle output_shape = c->UnknownShape();
225
226 const int n = c->num_inputs() / 3;
227 for (int i = 0; i < n; i++) {
228 ShapeHandle ind;
229 TF_RETURN_IF_ERROR(c->WithRank(c->input(i), 2, &ind));
230 ShapeHandle val;
231 TF_RETURN_IF_ERROR(c->WithRank(c->input(i + n), 1, &val));
232 ShapeHandle shape;
233 TF_RETURN_IF_ERROR(c->WithRank(c->input(i + 2 * n), 1, &shape));
234
235 // Add to output_ind_rows.
236 DimensionHandle num_dim;
237 TF_RETURN_IF_ERROR(c->Merge(c->Dim(ind, 0), c->Dim(val, 0), &num_dim));
238 TF_RETURN_IF_ERROR(
239 c->Add(output_row_count, num_dim, &output_row_count));
240
241 // Merge into output_ind_cols and output_shape.
242 TF_RETURN_IF_ERROR(
243 c->Merge(output_ind_cols, c->Dim(ind, 1), &output_ind_cols));
244 TF_RETURN_IF_ERROR(c->Merge(output_shape, shape, &output_shape));
245 }
246
247 c->set_output(0, c->Matrix(output_row_count, output_ind_cols));
248 c->set_output(1, c->Vector(output_row_count));
249 c->set_output(2, output_shape);
250 return Status::OK();
251 });

276REGISTER_OP("SparseSplit")
277 .Input("split_dim: int64")
278 .Input("indices: int64")
279 .Input("values: T")
280 .Input("shape: int64")
281 .Output("output_indices: num_split * int64")
282 .Output("output_values: num_split * T")
283 .Output("output_shape: num_split * int64")
284 .Attr("num_split: int >= 1")
285 .Attr("T: type")
286 .SetShapeFn([](InferenceContext* c) {
287 ShapeHandle input_shape = c->input(3);
288 ShapeHandle output_indices =
289 c->Matrix(InferenceContext::kUnknownDim, c->NumElements(input_shape));
290 ShapeHandle output_values = c->Vector(InferenceContext::kUnknownDim);
291 ShapeHandle output_shape = input_shape;
292
293 // Copy the outputs into the output ranges.
294 int num_splits = c->num_outputs() / 3;
295 int out_idx = 0;
296 for (int i = 0; i < num_splits; ++i)
297 c->set_output(out_idx++, output_indices);
298 for (int i = 0; i < num_splits; ++i)
299 c->set_output(out_idx++, output_values);
300 for (int i = 0; i < num_splits; ++i)
301 c->set_output(out_idx++, output_shape);
302 return Status::OK();
303 });

545REGISTER_OP("SparseFillEmptyRows")
546 .Input("indices: int64")
547 .Input("values: T")
548 .Input("dense_shape: int64")
549 .Input("default_value: T")
550 .Output("output_indices: int64")
551 .Output("output_values: T")
552 .Output("empty_row_indicator: bool")
553 .Output("reverse_index_map: int64")
554 .Attr("T: type")
555 .SetShapeFn([](InferenceContext* c) {
556 ShapeHandle input_indices = c->input(0);
557 TF_RETURN_IF_ERROR(c->WithRank(input_indices, 2, &input_indices));
558 ShapeHandle input_values = c->input(1);
559 TF_RETURN_IF_ERROR(c->WithRank(input_values, 1, &input_values));
560 ShapeHandle input_shape = c->input(2);
561 TF_RETURN_IF_ERROR(c->WithRank(input_shape, 1, &input_shape));
562 ShapeHandle default_value = c->input(3);
563 TF_RETURN_IF_ERROR(c->WithRank(default_value, 0, &default_value));
564 DimensionHandle N = c->Dim(input_indices, 0);
565 TF_RETURN_IF_ERROR(c->Merge(N, c->Dim(input_values, 0), &N));
566 DimensionHandle unused_dim;
567 TF_RETURN_IF_ERROR(c->Merge(c->Dim(input_indices, 1),
568 c->Dim(input_shape, 0), &unused_dim));
569 ShapeHandle output_indices =
570 c->Matrix(InferenceContext::kUnknownDim, c->NumElements(input_shape));
571 ShapeHandle output_values = c->Vector(InferenceContext::kUnknownDim);
572 ShapeHandle constant_input_shape;
573 TF_RETURN_IF_ERROR(c->MakeShapeFromShapeTensor(2, &constant_input_shape));
574 ShapeHandle empty_row_indicator =
575 c->Vector(c->Dim(constant_input_shape, 0));
576 ShapeHandle reverse_index_map = c->Vector(N);
577 c->set_output(0, output_indices);
578 c->set_output(1, output_values);
579 c->set_output(2, empty_row_indicator);
580 c->set_output(3, reverse_index_map);
581 return Status::OK();
582 });
