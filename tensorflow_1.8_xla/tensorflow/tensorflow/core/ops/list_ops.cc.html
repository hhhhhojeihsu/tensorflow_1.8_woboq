23REGISTER_OP("EmptyTensorList")
24 .Input("element_shape: shape_type")
25 .Output("handle: variant")
26 .Attr("element_dtype: type")
27 .Attr("shape_type: {int32, int64}")
28 .SetShapeFn([](shape_inference::InferenceContext* c) {
29 c->set_output(0, c->Scalar());
30 DataType t;
31 TF_RETURN_IF_ERROR(c->GetAttr("element_dtype", &t));
32 shape_inference::ShapeHandle s;
33 TF_RETURN_IF_ERROR(
34 c->MakeShapeFromShapeTensorTreatScalarAsUnknownShape(0, &s));
35 c->set_output_handle_shapes_and_types(
36 0, std::vector<shape_inference::ShapeAndType>{{s, t}});
37 return Status::OK();
38 });

40REGISTER_OP("TensorListPushBack")
41 .Input("input_handle: variant")
42 .Input("tensor: element_dtype")
43 .Output("output_handle: variant")
44 .Attr("element_dtype: type")
45 .SetShapeFn([](shape_inference::InferenceContext* c) {
46 c->set_output(0, c->Scalar());
47 DataType t;
48 TF_RETURN_IF_ERROR(c->GetAttr("element_dtype", &t));
49 shape_inference::ShapeHandle s = c->UnknownShape();
50
51 auto* handle_data = c->input_handle_shapes_and_types(0);
52 if (handle_data != nullptr && handle_data->size() != 1) {
53 return errors::InvalidArgument(
54 "Trying to push to list with wrong variant data.");
55 }
56 if (handle_data != nullptr) {
57 const shape_inference::ShapeAndType& list_shape_type =
58 (*handle_data)[0];
59 if (list_shape_type.dtype != t) {
60 return errors::InvalidArgument(
61 "Trying to push to list with wrong element dtype. List has type ",
62 DataTypeString(list_shape_type.dtype),
63 " but trying to push element with type ", DataTypeString(t));
64 }
65 shape_inference::ShapeHandle ignored;
66 TF_RETURN_IF_ERROR(c->Merge(s, list_shape_type.shape, &ignored));
67 s = list_shape_type.shape;
68 }
69 c->set_output_handle_shapes_and_types(
70 0, std::vector<shape_inference::ShapeAndType>{{s, t}});
71 return Status::OK();
72 });

79REGISTER_OP("TensorListPopBack")
80 .Input("input_handle: variant")
81 .Output("output_handle: variant")
82 .Output("tensor: element_dtype")
83 .Attr("element_dtype: type")
84 .SetShapeFn([](shape_inference::InferenceContext* c) {
85 DataType t;
86 TF_RETURN_IF_ERROR(c->GetAttr("element_dtype", &t));
87 shape_inference::ShapeHandle s = c->UnknownShape();
88 auto* handle_data = c->input_handle_shapes_and_types(0);
89 if (handle_data != nullptr && handle_data->size() != 1) {
90 return errors::InvalidArgument(
91 "Trying to read from list with invalid variant data.");
92 }
93 if (handle_data != nullptr) {
94 const shape_inference::ShapeAndType& list_shape_type =
95 (*handle_data)[0];
96 if (list_shape_type.dtype != t) {
97 return errors::InvalidArgument(
98 "Trying to read from list with wrong element dtype. List has "
99 "type ",
100 DataTypeString(list_shape_type.dtype),
101 " but trying to push element with type ", DataTypeString(t));
102 }
103 shape_inference::ShapeHandle ignored;
104 TF_RETURN_IF_ERROR(c->Merge(s, list_shape_type.shape, &ignored));
105 c->set_output_handle_shapes_and_types(0, *handle_data);
106 s = list_shape_type.shape;
107 }
108 c->set_output(1, s);
109 c->set_output(0, c->Scalar());
110 return Status::OK();
111 });

113REGISTER_OP("TensorListStack")
114 .Input("input_handle: variant")
115 .Output("tensor: element_dtype")
116 .Attr("element_dtype: type")
117 .Attr("num_elements: int = -1")
118 .SetShapeFn([](shape_inference::InferenceContext* c) {
119 DataType t;
120 TF_RETURN_IF_ERROR(c->GetAttr("element_dtype", &t));
121 shape_inference::ShapeHandle s = c->UnknownShape();
122 auto* handle_data = c->input_handle_shapes_and_types(0);
123 if (handle_data != nullptr && handle_data->size() != 1) {
124 return errors::InvalidArgument(
125 "Trying to read from list with wrong variant data.");
126 }
127 if (handle_data != nullptr) {
128 const shape_inference::ShapeAndType& list_shape_type =
129 (*handle_data)[0];
130 if (list_shape_type.dtype != t) {
131 return errors::InvalidArgument(
132 "Trying to read from list with wrong element dtype. List has "
133 "type ",
134 DataTypeString(list_shape_type.dtype), " but expectec type ",
135 DataTypeString(t));
136 }
137 shape_inference::ShapeHandle ignored;
138 TF_RETURN_IF_ERROR(c->Merge(s, list_shape_type.shape, &ignored));
139 s = list_shape_type.shape;
140 }
141 int expected_num_elements = -1;
142 TF_RETURN_IF_ERROR(c->GetAttr("num_elements", &expected_num_elements));
143 shape_inference::ShapeHandle num_elements;
144 if (expected_num_elements == -1) {
145 num_elements = c->MakeShape({c->UnknownDim()});
146 } else {
147 num_elements = c->MakeShape({expected_num_elements});
148 }
149 shape_inference::ShapeHandle result;
150 TF_RETURN_IF_ERROR(c->Concatenate(num_elements, s, &result));
151 c->set_output(0, result);
152 return Status::OK();
153 });

155REGISTER_OP("TensorListFromTensor")
156 .Input("tensor: element_dtype")
157 .Input("element_shape: shape_type")
158 .Output("output_handle: variant")
159 .Attr("element_dtype: type")
160 .Attr("shape_type: {int32, int64}")
161 .SetShapeFn([](shape_inference::InferenceContext* c) {
162 c->set_output(0, c->Scalar());
163 DataType t;
164 TF_RETURN_IF_ERROR(c->GetAttr("element_dtype", &t));
165 shape_inference::ShapeHandle s = c->input(0);
166 shape_inference::ShapeHandle o;
167 TF_RETURN_IF_ERROR(c->Subshape(s, 1, &o));
168 shape_inference::ShapeHandle element_shape;
169 TF_RETURN_IF_ERROR(c->MakeShapeFromShapeTensor(1, &element_shape));
170 TF_RETURN_IF_ERROR(c->Merge(o, element_shape, &o));
171 c->set_output_handle_shapes_and_types(
172 0, std::vector<shape_inference::ShapeAndType>{{element_shape, t}});
173 return Status::OK();
174 });

176REGISTER_OP("TensorListElementShape")
177 .Input("input_handle: variant")
178 .Output("element_shape: shape_type")
179 .Attr("shape_type: {int32, int64}")
180 .SetShapeFn([](shape_inference::InferenceContext* c) {
181 auto* handle_data = c->input_handle_shapes_and_types(0);
182 if (handle_data == nullptr) {
183 c->set_output(0, c->Vector(c->UnknownDim()));
184 return Status::OK();
185 }
186 c->set_output(0, c->Vector(c->Rank((*handle_data)[0].shape)));
187 return Status::OK();
188 });

207REGISTER_OP("TensorListGetItem")
208 .Input("input_handle: variant")
209 .Input("index: int32")
210 .Output("item: element_dtype")
211 .Attr("element_dtype: type")
212 .SetShapeFn([](shape_inference::InferenceContext* c) {
213 DataType t;
214 TF_RETURN_IF_ERROR(c->GetAttr("element_dtype", &t));
215 auto* handle_data = c->input_handle_shapes_and_types(0);
216 shape_inference::ShapeHandle element_shape = c->UnknownShape();
217 if (handle_data != nullptr) {
218 const shape_inference::ShapeAndType& list_shape_type =
219 (*handle_data)[0];
220 element_shape = list_shape_type.shape;
221 if (list_shape_type.dtype != t) {
222 return errors::InvalidArgument("Expected list with element dtype ",
223 DataTypeString(t),
224 " but got list with element dtype ",
225 DataTypeString(list_shape_type.dtype));
226 }
227 }
228 c->set_output(0, element_shape);
229 return Status::OK();
230 });

232REGISTER_OP("TensorListSetItem")
233 .Input("input_handle: variant")
234 .Input("index: int32")
235 .Input("item: element_dtype")
236 .Output("output_handle: variant")
237 .Attr("element_dtype: type")
238 .SetShapeFn([](shape_inference::InferenceContext* c) {
239 DataType t;
240 TF_RETURN_IF_ERROR(c->GetAttr("element_dtype", &t));
241 auto* handle_data = c->input_handle_shapes_and_types(0);
242 c->set_output(0, c->Scalar());
243 if (handle_data == nullptr) {
244 c->set_output_handle_shapes_and_types(0, {{c->UnknownShape(), t}});
245 return Status::OK();
246 }
247 const shape_inference::ShapeAndType& list_shape_type = (*handle_data)[0];
248 shape_inference::ShapeHandle s = c->input(2);
249 TF_RETURN_IF_ERROR(c->Merge(s, list_shape_type.shape, &s));
250 c->set_output_handle_shapes_and_types(0, *handle_data);
251 return Status::OK();
252 });
