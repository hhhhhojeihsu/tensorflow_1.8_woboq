26REGISTER_OP("DecodeRaw")
27 .Input("bytes: string")
28 .Output("output: out_type")
29 .Attr("out_type: {half,float,double,int32,uint16,uint8,int16,int8,int64}")
30 .Attr("little_endian: bool = true")
31 .SetShapeFn([](InferenceContext* c) {
32 // Note: last dimension is data dependent.
33 ShapeHandle out;
34 TF_RETURN_IF_ERROR(c->Concatenate(
35 c->input(0), c->Vector(InferenceContext::kUnknownDim), &out));
36 c->set_output(0, out);
37 return Status::OK();
38 });

46REGISTER_OP("ParseExample")
47 .Input("serialized: string")
48 .Input("names: string")
49 .Input("sparse_keys: Nsparse * string")
50 .Input("dense_keys: Ndense * string")
51 .Input("dense_defaults: Tdense")
52 .Output("sparse_indices: Nsparse * int64")
53 .Output("sparse_values: sparse_types")
54 .Output("sparse_shapes: Nsparse * int64")
55 .Output("dense_values: Tdense")
56 .Attr("Nsparse: int >= 0") // Inferred from sparse_keys
57 .Attr("Ndense: int >= 0") // Inferred from dense_keys
58 .Attr("sparse_types: list({float,int64,string}) >= 0")
59 .Attr("Tdense: list({float,int64,string}) >= 0")
60 .Attr("dense_shapes: list(shape) >= 0")
61 .SetShapeFn([](InferenceContext* c) {
62 ParseExampleAttrs attrs;
63 TF_RETURN_IF_ERROR(attrs.Init(c));
64
65 ShapeHandle input;
66 TF_RETURN_IF_ERROR(c->WithRank(c->input(0), 1, &input));
67 ShapeHandle unused;
68 TF_RETURN_IF_ERROR(c->WithRank(c->input(1), 1, &unused)); // names
69
70 // Output sparse_indices, sparse_values, and sparse_shapes.
71 int output_idx = 0;
72 for (int i = 0; i < attrs.num_sparse; ++i) {
73 c->set_output(output_idx++, c->Matrix(c->UnknownDim(), 2));
74 }
75 for (int i = 0; i < attrs.num_sparse; ++i) {
76 c->set_output(output_idx++, c->Vector(c->UnknownDim()));
77 }
78 for (int i = 0; i < attrs.num_sparse; ++i) {
79 c->set_output(output_idx++, c->Vector(2));
80 }
81
82 // Output dense_shapes.
83 for (int i = 0; i < attrs.num_dense; ++i) {
84 ShapeHandle dense;
85 TF_RETURN_IF_ERROR(
86 c->MakeShapeFromPartialTensorShape(attrs.dense_shapes[i], &dense));
87 TF_RETURN_IF_ERROR(c->Concatenate(input, dense, &dense));
88 c->set_output(output_idx++, dense);
89 }
90 return Status::OK();
91 });

93REGISTER_OP("ParseSingleExample")
94 .Input("serialized: string")
95 .Input("dense_defaults: Tdense")
96 .Output("sparse_indices: num_sparse * int64")
97 .Output("sparse_values: sparse_types")
98 .Output("sparse_shapes: num_sparse * int64")
99 .Output("dense_values: Tdense")
100 .Attr("num_sparse: int >= 0")
101 .Attr("sparse_keys: list(string) >= 0")
102 .Attr("dense_keys: list(string) >= 0")
103 .Attr("sparse_types: list({float,int64,string}) >= 0")
104 .Attr("Tdense: list({float,int64,string}) >= 0")
105 .Attr("dense_shapes: list(shape) >= 0")
106 .SetShapeFn([](InferenceContext* c) {
107 ParseSingleExampleAttrs attrs;
108 TF_RETURN_IF_ERROR(attrs.Init(c));
109
110 ShapeHandle input;
111 TF_RETURN_IF_ERROR(c->WithRank(c->input(0), 0, &input));
112
113 // Output sparse_indices, sparse_values, and sparse_shapes.
114 int output_idx = 0;
115 for (int i = 0; i < attrs.sparse_keys.size(); ++i) {
116 c->set_output(output_idx++, c->Matrix(c->UnknownDim(), 1));
117 }
118 for (int i = 0; i < attrs.sparse_keys.size(); ++i) {
119 c->set_output(output_idx++, c->Vector(c->UnknownDim()));
120 }
121 for (int i = 0; i < attrs.sparse_keys.size(); ++i) {
122 c->set_output(output_idx++, c->Vector(1));
123 }
124
125 // Output dense_shapes.
126 for (int i = 0; i < attrs.dense_keys.size(); ++i) {
127 ShapeHandle dense;
128 TF_RETURN_IF_ERROR(
129 c->MakeShapeFromPartialTensorShape(attrs.dense_shapes[i], &dense));
130 c->set_output(output_idx++, dense);
131 }
132 return Status::OK();
133 });

135REGISTER_OP("ParseSingleSequenceExample")
136 .Input("serialized: string")
137 .Input("feature_list_dense_missing_assumed_empty: string")
138 .Input("context_sparse_keys: Ncontext_sparse * string")
139 .Input("context_dense_keys: Ncontext_dense * string")
140 .Input("feature_list_sparse_keys: Nfeature_list_sparse * string")
141 .Input("feature_list_dense_keys: Nfeature_list_dense * string")
142 .Input("context_dense_defaults: Tcontext_dense")
143 .Input("debug_name: string")
144 .Output("context_sparse_indices: Ncontext_sparse * int64")
145 .Output("context_sparse_values: context_sparse_types")
146 .Output("context_sparse_shapes: Ncontext_sparse * int64")
147 .Output("context_dense_values: Tcontext_dense")
148 .Output("feature_list_sparse_indices: Nfeature_list_sparse * int64")
149 .Output("feature_list_sparse_values: feature_list_sparse_types")
150 .Output("feature_list_sparse_shapes: Nfeature_list_sparse * int64")
151 .Output("feature_list_dense_values: feature_list_dense_types")
152 // Infer from context_sparse_keys
153 .Attr("Ncontext_sparse: int >= 0 = 0")
154 // Infer from context_dense_keys
155 .Attr("Ncontext_dense: int >= 0 = 0")
156 // Infer from feature_list_sparse_keys
157 .Attr("Nfeature_list_sparse: int >= 0 = 0")
158 // Infer from feature_list_dense_keys
159 .Attr("Nfeature_list_dense: int >= 0 = 0")
160 .Attr("context_sparse_types: list({float,int64,string}) >= 0 = []")
161 .Attr("Tcontext_dense: list({float,int64,string}) >= 0 = []")
162 .Attr("feature_list_dense_types: list({float,int64,string}) >= 0 = []")
163 .Attr("context_dense_shapes: list(shape) >= 0 = []")
164 .Attr("feature_list_sparse_types: list({float,int64,string}) >= 0 = []")
165 .Attr("feature_list_dense_shapes: list(shape) >= 0 = []")
166 .SetShapeFn([](InferenceContext* c) {
167 ShapeHandle unused;
168 ParseSingleSequenceExampleAttrs attrs;
169 TF_RETURN_IF_ERROR(attrs.Init(c));
170
171 ShapeHandle input;
172 TF_RETURN_IF_ERROR(c->WithRank(c->input(0), 0, &input));
173
174 // feature_list_dense_missing_assumed_empty
175 TF_RETURN_IF_ERROR(c->WithRank(c->input(1), 1, &unused));
176
177 int output_idx = 0;
178
179 // Output context_sparse_indices, context_sparse_values, and
180 // context_sparse_shapes.
181 for (int i = 0; i < attrs.num_context_sparse; ++i) {
182 c->set_output(output_idx++, c->Matrix(c->UnknownDim(), 1));
183 }
184 for (int i = 0; i < attrs.num_context_sparse; ++i) {
185 c->set_output(output_idx++, c->Vector(c->UnknownDim()));
186 }
187 for (int i = 0; i < attrs.num_context_sparse; ++i) {
188 c->set_output(output_idx++, c->Vector(1));
189 }
190
191 // Output context_dense_shapes.
192 for (int i = 0; i < attrs.num_context_dense; ++i) {
193 ShapeHandle s;
194 TF_RETURN_IF_ERROR(c->MakeShapeFromPartialTensorShape(
195 attrs.context_dense_shapes[i], &s));
196 c->set_output(output_idx++, s);
197 }
198
199 // Output feature_list_sparse_indices, feature_list_sparse_values,
200 // feature_list_sparse_shapes.
201 for (int i = 0; i < attrs.num_feature_list_sparse; ++i) {
202 c->set_output(output_idx++, c->Matrix(c->UnknownDim(), 2));
203 }
204 for (int i = 0; i < attrs.num_feature_list_sparse; ++i) {
205 c->set_output(output_idx++, c->Vector(c->UnknownDim()));
206 }
207 for (int i = 0; i < attrs.num_feature_list_sparse; ++i) {
208 c->set_output(output_idx++, c->Vector(2));
209 }
210
211 // Output feature_list_dense_shapes.
212 for (int i = 0; i < attrs.num_feature_list_dense; ++i) {
213 ShapeHandle s;
214 TF_RETURN_IF_ERROR(c->MakeShapeFromPartialTensorShape(
215 attrs.feature_list_dense_shapes[i], &s));
216 TF_RETURN_IF_ERROR(
217 c->Concatenate(c->Vector(InferenceContext::kUnknownDim), s, &s));
218 c->set_output(output_idx++, s);
219 }
220 return Status::OK();
221 });

240REGISTER_OP("DecodeCSV")
241 .Input("records: string")
242 .Input("record_defaults: OUT_TYPE")
243 .Output("output: OUT_TYPE")
244 .Attr("OUT_TYPE: list({float,double,int32,int64,string})")
245 .Attr("field_delim: string = ','")
246 .Attr("use_quote_delim: bool = true")
247 .Attr("na_value: string = ''")
248 .SetShapeFn([](InferenceContext* c) {
249 // Validate the record_defaults inputs.
250 for (int i = 1; i < c->num_inputs(); ++i) {
251 ShapeHandle v;
252 TF_RETURN_IF_ERROR(c->WithRank(c->input(i), 1, &v));
253 if (c->Value(c->Dim(v, 0)) > 1) {
254 return errors::InvalidArgument(
255 "Shape of a default must be a length-0 or length-1 vector");
256 }
257 }
258
259 // Propagate shape of the records input.
260 for (int i = 0; i < c->num_outputs(); ++i) c->set_output(i, c->input(0));
261 return Status::OK();
262 });
