41class StdThread : public Thread {
42 public:
43 // name and thread_options are both ignored.
44 StdThread(const ThreadOptions& thread_options, const string& name,
45 std::function<void()> fn)
46 : thread_(fn) {}
47 ~StdThread() override { thread_.join(); }
48
49 private:
50 std::thread thread_;
51};
52
53class PosixEnv : public Env {
54 public:
55 PosixEnv() {}
56
57 ~PosixEnv() override { LOG(FATAL) << "Env::Default() must not be destroyed"; }
58
59 bool MatchPath(const string& path, const string& pattern) override {
60 return fnmatch(pattern.c_str(), path.c_str(), FNM_PATHNAME) == 0;
61 }
62
63 void SleepForMicroseconds(int64 micros) override {
64 while (micros > 0) {
65 timespec sleep_time;
66 sleep_time.tv_sec = 0;
67 sleep_time.tv_nsec = 0;
68
69 if (micros >= 1e6) {
70 sleep_time.tv_sec =
71 std::min<int64>(micros / 1e6, std::numeric_limits<time_t>::max());
72 micros -= static_cast<int64>(sleep_time.tv_sec) * 1e6;
73 }
74 if (micros < 1e6) {
75 sleep_time.tv_nsec = 1000 * micros;
76 micros = 0;
77 }
78 while (nanosleep(&sleep_time, &sleep_time) != 0 && errno == EINTR) {
79 // Ignore signals and wait for the full interval to elapse.
80 }
81 }
82 }
83
84 Thread* StartThread(const ThreadOptions& thread_options, const string& name,
85 std::function<void()> fn) override {
86 return new StdThread(thread_options, name, fn);
87 }
88
89 void SchedClosure(std::function<void()> closure) override {
90 // TODO(b/27290852): Spawning a new thread here is wasteful, but
91 // needed to deal with the fact that many `closure` functions are
92 // blocking in the current codebase.
93 std::thread closure_thread(closure);
94 closure_thread.detach();
95 }
96
97 void SchedClosureAfter(int64 micros, std::function<void()> closure) override {
98 // TODO(b/27290852): Consuming a thread here is wasteful, but this
99 // code is (currently) only used in the case where a step fails
100 // (AbortStep). This could be replaced by a timer thread
101 SchedClosure([this, micros, closure]() {
102 SleepForMicroseconds(micros);
103 closure();
104 });
105 }
106
107 Status LoadLibrary(const char* library_filename, void** handle) override {
108 return tensorflow::internal::LoadLibrary(library_filename, handle);
109 }
110
111 Status GetSymbolFromLibrary(void* handle, const char* symbol_name,
112 void** symbol) override {
113 return tensorflow::internal::GetSymbolFromLibrary(handle, symbol_name,
114 symbol);
115 }
116
117 string FormatLibraryFileName(const string& name,
118 const string& version) override {
119 return tensorflow::internal::FormatLibraryFileName(name, version);
120 }
121
122 private:
123 void GetLocalTempDirectories(std::vector<string>* list) override;
124};

137void PosixEnv::GetLocalTempDirectories(std::vector<string>* list) {
138 list->clear();
139 // Directories, in order of preference. If we find a dir that
140 // exists, we stop adding other less-preferred dirs
141 const char* candidates[] = {
142 // Non-null only during unittest/regtest
143 getenv("TEST_TMPDIR"),
144
145 // Explicitly-supplied temp dirs
146 getenv("TMPDIR"),
147 getenv("TMP"),
148
149#if defined(__ANDROID__)
150 "/data/local/tmp",
151#endif
152
153 // If all else fails
154 "/tmp",
155 };
156
157 for (const char* d : candidates) {
158 if (!d || d[0] == '\0') continue; // Empty env var
159
160 // Make sure we don't surprise anyone who's expecting a '/'
161 string dstr = d;
162 if (dstr[dstr.size() - 1] != '/') {
163 dstr += "/";
164 }
165
166 struct stat statbuf;
167 if (!stat(d, &statbuf) && S_ISDIR(statbuf.st_mode) &&
168 !access(dstr.c_str(), 0)) {
169 // We found a dir that exists and is accessible - we're done.
170 list->push_back(dstr);
171 return;
172 }
173 }
