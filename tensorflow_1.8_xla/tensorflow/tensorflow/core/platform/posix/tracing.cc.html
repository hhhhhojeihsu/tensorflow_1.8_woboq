24static bool TryGetEnv(const char* name, const char** value) {
25 *value = getenv(name);
26 return *value != nullptr && (*value)[0] != '\0';
27}
28
29const char* Tracing::LogDir() {
30 const char* dir;
31 if (TryGetEnv("TEST_TMPDIR", &dir)) return dir;
32 if (TryGetEnv("TMP", &dir)) return dir;
33 if (TryGetEnv("TMPDIR", &dir)) return dir;
34 dir = "/tmp";
35 if (access(dir, R_OK | W_OK | X_OK) == 0) return dir;
36 return "."; // Default to current directory.
37}
