36class Tracing {
37 public:
38 // This enumeration contains the identifiers of all TensorFlow
39 // threadscape events and code regions. Threadscape assigns its
40 // own identifiers at runtime when we register our events and we
41 // cannot know in advance what IDs it will choose. The "RecordEvent"
42 // method and "ScopedActivity" use these event IDs for consistency
43 // and remap them to threadscape IDs at runtime. This enum is limited
44 // to 64 values since we use a bitmask to configure which events are
45 // enabled. It must also be kept in step with the code in
46 // "Tracing::EventCategoryString".
47 enum EventCategory {
48 kScheduleClosure = 0,
49 kRunClosure = 1,
50 kCompute = 2,
51 kEventCategoryMax = 3 // sentinel - keep last
52 };
53 // Note: We currently only support up to 64 categories.
54 static_assert(kEventCategoryMax <= 64, "only support up to 64 events");
55
56 // Called by main programs to initialize tracing facilities
57 static void Initialize();
58
59 // Return the pathname of the directory where we are writing log files.
60 static const char* LogDir();
61
62 // Returns a non-zero identifier which can be used to correlate
63 // related events.
64 static inline uint64 UniqueId();
65
66 // Returns true if a trace is in progress. Can be used to reduce tracing
67 // overheads in fast-path code.
68 static inline bool IsActive();
69
70 // Associate name with the current thread.
71 static void RegisterCurrentThread(const char* name);
72
73 // Posts an event with the supplied category and arg.
74 static void RecordEvent(EventCategory category, uint64 arg);
75
76 // Traces a region of code. Posts a tracing "EnterCodeRegion" event
77 // when created and an "ExitCodeRegion" event when destroyed.
78 class ScopedActivity {
79 public:
80 explicit ScopedActivity(EventCategory category, uint64 arg);
81 ~ScopedActivity();
82
83 private:
84#if defined(PLATFORM_GOOGLE)
85 const bool enabled_;
86 const int32 region_id_;
87#endif
88
89 TF_DISALLOW_COPY_AND_ASSIGN(ScopedActivity);
90 };
91
92 // Trace collection engine can be registered with this module.
93 // If no engine is registered, ScopedAnnotation and TraceMe are no-ops.
94 class Engine;
95 static void RegisterEngine(Engine*);
96
97 // Forward declaration of the GPU utility classes.
98 class ScopedAnnotation;
99 class TraceMe;
100
101 private:
102 friend class TracingTest;
103 friend class ScopedAnnotation;
104 friend class TraceMe;
105
106 // TODO: TF_EXPORT is for building //tensorflow/contrib/data:_dataset_ops.so
107 // on Windows. Figure out a way to remove TF_EXPORT here.
108 TF_EXPORT static std::atomic<Tracing::Engine*> tracing_engine_;
109 static Tracing::Engine* engine() {
110 return tracing_engine_.load(std::memory_order_acquire);
111 }
112
113 static void RegisterEvent(EventCategory id, const char* name);
114 static const char* EventCategoryString(EventCategory category);
115
116 //
117 // Parses event mask expressions in 'value' of the form:
118 // expr ::= <term> (,<term>)*
119 // term ::= <event> | "!" <event>
120 // event ::= "ALL" | <wait_event> | <other_event>
121 // wait_event ::= "ENewSession" | "ECloseSession" | ...
122 // other_event ::= "Send" | "Wait" | ...
123 // ALL denotes all events, <event> turns on tracing for this event, and
124 // !<event> turns off tracing for this event.
125 // If the expression can be parsed correctly it returns true and sets
126 // the event_mask_. Otherwise it returns false and the event_mask_ is left
127 // unchanged.
128 static bool ParseEventMask(const char* flagname, const string& value);
129
130 // Bit mask of enabled trace categories.
131 static uint64 event_mask_;
132
133 // Records the mappings between Threadscape IDs and the "EventCategory" enum.
134 static int32 category_id_[kEventCategoryMax];
135 static std::map<string, int32>* name_map_;
136};

138// Trace collection engine that actually implements collection.
139class Tracing::Engine {
140 public:
141 Engine() {}
142 virtual ~Engine();
143
144 // Returns true if Tracing is currently enabled.
145 virtual bool IsEnabled() const = 0;
146
147 // Represents an active annotation.
148 class Annotation {
149 public:
150 Annotation() {}
151 virtual ~Annotation();
152 };
153
154 // Represents an active trace.
155 class Tracer {
156 public:
157 Tracer() {}
158 virtual ~Tracer();
159 };
160
161 private:
162 friend class ScopedAnnotation;
163 friend class TraceMe;
164
165 // Register the specified name as an annotation on the current thread.
166 // Caller should delete the result to remove the annotation.
167 // Annotations from the same thread are destroyed in a LIFO manner.
168 // May return nullptr if annotations are not supported.
169 virtual Annotation* PushAnnotation(StringPiece name) = 0;
170
171 // Start tracing under the specified label. Caller should delete the result
172 // to stop tracing.
173 // May return nullptr if tracing is not supported.
174 virtual Tracer* StartTracing(StringPiece label, bool is_expensive) = 0;
175 // Same as above, but implementations can avoid copying the string.
176 virtual Tracer* StartTracing(string&& label, bool is_expensive) {
177 return StartTracing(StringPiece(label), is_expensive);
178 }
179
180 // Backwards compatibility one arg variants (assume is_expensive=true).
181 Tracer* StartTracing(StringPiece label) {
182 return StartTracing(label, /*is_expensive=*/true);
183 }
184 Tracer* StartTracing(string&& label) {
185 return StartTracing(StringPiece(label), /*is_expensive=*/true);
186 }
187};
188
189// This class permits a user to apply annotation on kernels and memcpys
190// when launching them. While an annotation is in scope, all activities
191// within that scope get their names replaced by the annotation. The kernel
192// name replacement is done when constructing the protobuf for sending out to
193// a client (e.g., the stubby requestor) for both API and Activity records.
194//
195// Ownership: The creator of ScopedAnnotation assumes ownership of the object.
196//
197// Usage: {
198// ScopedAnnotation annotation("first set of kernels");
199// Kernel1<<<x,y>>>;
200// LaunchKernel2(); // Which eventually launches a cuda kernel.
201// }
202// In the above scenario, the GPUProf UI would show 2 kernels with the name
203// "first set of kernels" executing -- they will appear as the same kernel.
204class Tracing::ScopedAnnotation {
205 public:
206 explicit ScopedAnnotation(StringPiece name);
207
208 // If tracing is enabled, set up an annotation with a label of
209 // "<name_part1>:<name_part2>". Can be cheaper than the
210 // single-argument constructor because the concatenation of the
211 // label string is only done if tracing is enabled.
212 ScopedAnnotation(StringPiece name_part1, StringPiece name_part2);
213
214 // Returns true iff scoped annotations are active.
215 static bool Enabled() {
216 auto e = Tracing::engine();
217 return e && e->IsEnabled();
218 }
219
220 private:
221 std::unique_ptr<Engine::Annotation> annotation_;
222};
223
224// TODO(opensource): clean up the scoped classes for GPU tracing.
225// This class permits user-specified (CPU) tracing activities. A trace
226// activity is started when an object of this class is created and stopped
227// when the object is destroyed.
228class Tracing::TraceMe {
229 public:
230 explicit TraceMe(StringPiece name);
231 TraceMe(StringPiece name, bool is_expensive);
232
233 // If tracing is enabled, set up a traceMe with a label of
234 // "<name_part1>:<name_part2>". This can be cheaper than the
235 // single-argument constructor because the concatenation of the
236 // label string is only done if tracing is enabled.
237 TraceMe(StringPiece name_part1, StringPiece name_part2);
238 TraceMe(StringPiece name_part1, StringPiece name_part2, bool is_expensive);
239
240 private:
241 std::unique_ptr<Engine::Tracer> tracer_;
242};
243
244inline Tracing::ScopedAnnotation::ScopedAnnotation(StringPiece name) {
245 auto e = Tracing::engine();
246 if (e && e->IsEnabled()) {
247 annotation_.reset(e->PushAnnotation(name));
248 }
249}
250
251inline Tracing::ScopedAnnotation::ScopedAnnotation(StringPiece name_part1,
252 StringPiece name_part2) {
253 auto e = Tracing::engine();
254 if (e && e->IsEnabled()) {
255 annotation_.reset(
256 e->PushAnnotation(strings::StrCat(name_part1, ":", name_part2)));
257 }
258}
259
260inline Tracing::TraceMe::TraceMe(StringPiece name) : TraceMe(name, true) {}
261
262inline Tracing::TraceMe::TraceMe(StringPiece name, bool is_expensive) {
263 auto e = Tracing::engine();
264 if (e && e->IsEnabled()) {
265 tracer_.reset(e->StartTracing(name, is_expensive));
266 }
267}
268
269inline Tracing::TraceMe::TraceMe(StringPiece name_part1, StringPiece name_part2)
270 : TraceMe(name_part1, name_part2, true) {}
271
272inline Tracing::TraceMe::TraceMe(StringPiece name_part1, StringPiece name_part2,
273 bool is_expensive) {
274 auto e = Tracing::engine();
275 if (e && e->IsEnabled()) {
276 tracer_.reset(e->StartTracing(strings::StrCat(name_part1, ":", name_part2),
277 is_expensive));
278 }
