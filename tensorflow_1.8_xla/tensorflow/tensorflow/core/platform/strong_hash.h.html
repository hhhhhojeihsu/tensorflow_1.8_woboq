24// This is a strong keyed hash function interface for strings.
25// The hash function is deterministic on the content of the string within the
26// process. The key of the hash is an array of 2 uint64 elements.
27// A strong hash make it dificult, if not infeasible, to compute inputs that
28// hash to the same bucket.
29//
30// Usage:
31// uint64 key[2] = {123, 456};
32// string input = "input string";
33// uint64 hash_value = StrongKeyedHash(key, input);
34//
35uint64 StrongKeyedHash(const uint64 (&)[2], const string&);
36
37} // namespace tensorflow
