41DeviceProperties GetLocalCPUInfo() {
42 DeviceProperties device;
43 device.set_type("CPU");
44
45 device.set_vendor(port::CPUVendorIDString());
46 // Combine cpu family and model into the model string.
47 device.set_model(
48 strings::StrCat((port::CPUFamily() << 4) + port::CPUModelNum()));
49 device.set_frequency(port::NominalCPUFrequency() * 1e-6);
50 device.set_num_cores(port::NumSchedulableCPUs());
51 device.set_l1_cache_size(Eigen::l1CacheSize());
52 device.set_l2_cache_size(Eigen::l2CacheSize());
53 device.set_l3_cache_size(Eigen::l3CacheSize());
54
55 int64 free_mem = port::AvailableRam();
56 if (free_mem < INT64_MAX) {
57 device.set_memory_size(free_mem);
58 }
59
60 (*device.mutable_environment())["cpu_instruction_set"] =
61 Eigen::SimdInstructionSetsInUse();
62
63 (*device.mutable_environment())["eigen"] = strings::StrCat(
64 EIGEN_WORLD_VERSION, ".", EIGEN_MAJOR_VERSION, ".", EIGEN_MINOR_VERSION);
65#ifdef EIGEN_USE_LIBXSMM
66 (*device.mutable_environment())["libxsmm"] = LIBXSMM_VERSION;
67#endif
68
69 return device;
70}

72DeviceProperties GetLocalGPUInfo(CudaGpuId cuda_gpu_id) {
73 DeviceProperties device;
74 device.set_type("GPU");
75
76#if GOOGLE_CUDA
77 cudaDeviceProp properties;
78 cudaError_t error = cudaGetDeviceProperties(&properties, cuda_gpu_id.value());
79 if (error != cudaSuccess) {
80 device.set_type("UNKNOWN");
81 LOG(ERROR) << "Failed to get device properties, error code: " << error;
82 return device;
83 }
84
85 device.set_vendor("NVIDIA");
86 device.set_model(properties.name);
87 device.set_frequency(properties.clockRate * 1e-3);
88 device.set_num_cores(properties.multiProcessorCount);
89 device.set_num_registers(properties.regsPerMultiprocessor);
90 // For compute capability less than 5, l1 cache size is configurable to
91 // either 16 KB or 48 KB. We use the initial configuration 16 KB here. For
92 // compute capability larger or equal to 5, l1 cache (unified with texture
93 // cache) size is 24 KB. This number may need to be updated for future
94 // compute capabilities.
95 device.set_l1_cache_size((properties.major < 5) ? 16 * 1024 : 24 * 1024);
96 device.set_l2_cache_size(properties.l2CacheSize);
97 device.set_l3_cache_size(0);
98 device.set_shared_memory_size_per_multiprocessor(
99 properties.sharedMemPerMultiprocessor);
100 device.set_memory_size(properties.totalGlobalMem);
101 // 8 is the number of bits per byte. 2 is accounted for
102 // double data rate (DDR).
103 device.set_bandwidth(properties.memoryBusWidth / 8 *
104 properties.memoryClockRate * 2);
105
106 (*device.mutable_environment())["architecture"] =
107 strings::StrCat(properties.major, ".", properties.minor);
108 (*device.mutable_environment())["cuda"] = strings::StrCat(CUDA_VERSION);
109 (*device.mutable_environment())["cudnn"] = strings::StrCat(CUDNN_VERSION);
110#endif
111
112 return device;
113}
114
115DeviceProperties GetDeviceInfo(const DeviceNameUtils::ParsedName& device) {
116 DeviceProperties unknown;
117 unknown.set_type("UNKNOWN");
118
119 if (device.type == "CPU") {
120 return GetLocalCPUInfo();
121 } else if (device.type == "GPU") {
122 if (device.has_id) {
123 TfGpuId tf_gpu_id(device.id);
124 CudaGpuId cuda_gpu_id;
125 Status s = GpuIdManager::TfToCudaGpuId(tf_gpu_id, &cuda_gpu_id);
126 if (!s.ok()) {
127 LOG(ERROR) << s;
128 return unknown;
129 }
130 return GetLocalGPUInfo(cuda_gpu_id);
131 } else {
132 return GetLocalGPUInfo(CudaGpuId(0));
133 }
134 }
135 return unknown;
136}
