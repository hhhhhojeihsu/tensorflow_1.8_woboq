33// A TensorFlow model to optimize.
34// Models are represented by the combination of a graph, one of more fetch
35// nodes, and potentially a set of nodes to feed.
36// TODO(volunteer_needed): turn this struct into a class.
37struct GrapplerItem {
38 GrapplerItem() {}
39 GrapplerItem(const GrapplerItem& other, GraphDef&& graphDef);
40
41 string id; // A unique id for this item
42
43 // Inputs
44 GraphDef graph;
45 std::vector<std::pair<string, Tensor>> feed;
46 std::vector<string> fetch;
47
48 // Initialization op(s).
49 std::vector<string> init_ops;
50 // Expected initialization time in seconds, or 0 if unknown
51 int64 expected_init_time = 0;
52
53 // Save/restore ops (if any)
54 string save_op;
55 string restore_op;
56 string save_restore_loc_tensor;
57
58 // Queue runner(s) required to run the queue(s) of this model.
59 std::vector<QueueRunnerDef> queue_runners;
60
61 // List of op names to keep in the graph. This includes nodes that are
62 // referenced in various collections, and therefore must be preserved to
63 // ensure that the optimized metagraph can still be loaded.
64 std::vector<string> keep_ops;
65
66 // Return the set of node evaluated during a regular train/inference step.
67 std::vector<const NodeDef*> MainOpsFanin() const;
68 // Return the set of node run to populate the queues (if any).
69 std::vector<const NodeDef*> EnqueueOpsFanin() const;
70 // Return the set nodes used by TensorFlow to initialize the graph.
71 std::vector<const NodeDef*> InitOpsFanin() const;
72 // Return the set of variables accessed during a regular train/inference step.
73 std::vector<const NodeDef*> MainVariables() const;
74 // Return a set of node names that must be preserved. This includes feed and
75 // fetch nodes, keep_ops, init_ops.
76 std::unordered_set<string> NodesToPreserve() const;
77};
78
79// Return the transitive fanin of a set of terminal nodes.
80std::vector<const NodeDef*> ComputeTransitiveFanin(
81 const GraphDef& graph, const std::vector<string>& terminal_nodes);
82
83// Return the transitive fanin of a set of terminal nodes. Sets 'ill_formed' to
84// true if one of the node is missing in the graph, or some node inputs don't
85// exist.
86std::vector<const NodeDef*> ComputeTransitiveFanin(
87 const GraphDef& graph, const std::vector<string>& terminal_nodes,
88 bool* ill_formed);
89
