31std::unique_ptr<GrapplerItem> GrapplerItemFromFunctionDef(
32 const FunctionDef& func,
33 const std::unordered_map<string, AttrValue>& func_attr,
34 const FunctionDefLibrary& library) {
35 if (func.signature().name().empty()) {
36 LOG(ERROR) << "function name must be specified.";
37 return nullptr;
38 }
39 std::unique_ptr<GrapplerItem> new_item(new GrapplerItem());
40 new_item->id = func.signature().name();
41
42 std::unordered_map<string, string> port_map;
43
44 // Add the function inputs as placeholder
45 for (const auto& inp : func.signature().input_arg()) {
46 NodeDef* ph = new_item->graph.add_node();
47 ph->set_name(inp.name());
48 ph->set_op("Placeholder");
49 if (inp.type() != DT_INVALID) {
50 (*ph->mutable_attr())["T"].set_type(inp.type());
51 } else {
52 auto it = func_attr.find(inp.type_attr());
53 if (it == func_attr.end()) {
54 LOG(ERROR) << "Unknown type attribute " << inp.type_attr()
55 << " for function input " << inp.name();
56 return nullptr;
57 } else {
58 (*ph->mutable_attr())["T"] = it->second;
59 }
60 }
61 port_map[inp.name()] = inp.name();
62 }
63
64 // Add the function body to the graph.
65 FunctionLibraryDefinition func_def(OpRegistry::Global(), library);
66
67 for (const NodeDef& node : func.node_def()) {
68 NodeDef* new_node = new_item->graph.add_node();
69 *new_node = node;
70 // Replace the placeholder attribute values with the specified value.
71 for (auto& attr : *new_node->mutable_attr()) {
72 const string& ph_name = attr.second.placeholder();
73 auto it = func_attr.find(ph_name);
74 if (it != func_attr.end()) {
75 attr.second = it->second;
76 }
77 }
78
79 // Functions use a custom format to encode connectivity. Map these custom
80 // strings to regular ones.
81 const OpRegistrationData* registration;
82 Status status = func_def.LookUp(node.op(), &registration);
83 if (!status.ok()) {
84 LOG(ERROR) << "Op " << node.op() << " not registered: " << status;
85 return nullptr;
86 }
87
88 tensorflow::NameRangeMap inputs;
89 tensorflow::NameRangeMap outputs;
90 status = tensorflow::NameRangesForNode(node, registration->op_def, &inputs,
91 &outputs);
92 if (!status.ok()) {
93 LOG(ERROR) << "Op " << node.op() << " invalid: " << status;
94 return nullptr;
95 }
96 for (const auto& name_range : outputs) {
97 string port_prefix =
98 strings::StrCat(node.name(), ":", name_range.first, ":");
99 int index_start = name_range.second.first;
100 int index_end = name_range.second.second;
101 for (int i = index_start; i < index_end; ++i) {
102 string port_id = strings::StrCat(port_prefix, i - index_start);
103 string port_name = strings::StrCat(node.name(), ":", i);
104 port_map[port_id] = port_name;
105 }
106 }
107 }
108
109 for (auto& node : *new_item->graph.mutable_node()) {
110 // Rewrite the inputs to use the normal naming convention.
111 for (int i = 0; i < node.input_size(); ++i) {
112 const string& input = node.input(i);
113 if (IsControlInput(input)) {
114 // No need to remap control dependencies.
115 continue;
116 } else {
117 auto it = port_map.find(input);
118 if (it == port_map.end()) {
119 LOG(ERROR) << "Unknown input: " << input;
120 return nullptr;
121 }
122 node.set_input(i, it->second);
123 }
124 }
125 }
126
127 // Add the function outputs to the list of fetch nodes, taking into account
128 // the output mapping if any.
129 for (const auto& out : func.signature().output_arg()) {
130 auto it = func.ret().find(out.name());
131 if (it != func.ret().end()) {
132 auto it2 = port_map.find(it->second);
133 if (it2 == port_map.end()) {
134 LOG(ERROR) << "Unknown output mapping: " << it->first << " to "
135 << it->second;
136 return nullptr;
137 } else {
138 new_item->fetch.emplace_back(it2->second);
139 }
140 } else {
141 new_item->fetch.emplace_back(out.name());
142 }
143 }
144 // Add the function inputs to the list of feeds.
145 for (const auto& inp : func.signature().input_arg()) {
146 new_item->feed.emplace_back(inp.name(), Tensor());
147 }
148
149 return new_item;
150}
