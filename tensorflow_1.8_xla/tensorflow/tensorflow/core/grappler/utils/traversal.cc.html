17#include "tensorflow/core/framework/node_def.pb.h"

19namespace tensorflow {
20namespace grappler {

22void ReverseDfs(const GraphView& graph_view, const std::vector<NodeDef*>& from,
23 const std::function<void(NodeDef*)>& pre_order,
24 const std::function<void(NodeDef*)>& post_order,
25 const std::function<void(NodeDef*, NodeDef*)>& on_back_edge) {
26 // Stack of work to do.
27 struct StackElem {
28 NodeDef* node;
29 bool children_visited;
30 NodeDef* src;
31 };
32 std::vector<StackElem> stack;
33
34 stack.reserve(from.size());
35 for (NodeDef* node : from) {
36 stack.push_back(StackElem{node, false});
37 }
38
39 enum NodeState { NOT_VISITED = 0, VISITING = 1, DONE = 2 };
40 std::unordered_map<NodeDef*, NodeState> node_state;
41 while (!stack.empty()) {
42 StackElem w = stack.back();
43 stack.pop_back();
44
45 if (w.children_visited) {
46 // We've processed all the children of this node
47 node_state[w.node] = DONE;
48 if (post_order) {
49 post_order(w.node);
50 }
51 continue;
52 }
53
54 auto& rslt = node_state[w.node];
55 if (rslt == DONE) {
56 continue;
57 } else if (rslt == VISITING) {
58 // Loop detected
59 if (on_back_edge) {
60 on_back_edge(w.src, w.node);
61 }
62 continue;
63 }
64 rslt = VISITING;
65 if (pre_order) {
66 pre_order(w.node);
67 }
68
69 // Enqueue the node again with the children_visited flag set to true.
70 stack.push_back(StackElem{w.node, true, w.src});
71
72 // Now enqueu the node children.
73 for (const auto fanin : graph_view.GetFanins(*w.node, true)) {
74 stack.push_back(StackElem{fanin.node, false, w.node});
75 }
76 }
77}
