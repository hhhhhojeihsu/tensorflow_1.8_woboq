36// Context owned by GraphOptimizer, and passed to every stage at construction
37// time. Each optimizer stage is responsible for updating it according to the
38// changes it made to the graph.
39//
40// If an optimizer needs access to some helper class that is not present in this
41// context, consider creating an extension context, specific to that
42// optimizer (see example of ArithmeticOptimizerContext). GraphOptimizerContext
43// should only have members that are useful to almost all optimizers.
44struct GraphOptimizerContext {
45 GraphOptimizerContext(const std::unordered_set<string>* nodes_to_preserve,
46 GraphDef* optimized_graph,
47 GraphProperties* graph_properties, NodeMap* node_map)
48 : nodes_to_preserve(nodes_to_preserve),
49 optimized_graph(optimized_graph),
50 graph_properties(graph_properties),
51 node_map(node_map) {}
52
53 const std::unordered_set<string>* nodes_to_preserve;
54 GraphDef* optimized_graph;
55 GraphProperties* graph_properties;
56 NodeMap* node_map;
57};

69// WARNING:
70// Optimizer stage must try to re-use original nodes of a graph and
71// make all updates in place. This helps to make robust node placement
72// decisions. Create new nodes only if there is a reason for that.
73
74// Make a name for a new node obtained by optimizing a single node of the
75// original graph. The optimized node is placed under the original node scope.
76//
77// Node name uniqueness is guaranteed by unique name of an original node in
78// a same scope.
79//
80// Empty sub_scope or prefix ignored. At least one of them must be non-empty.
81//
82// Example: a/b/c/Add -> a/b/c/${sub_scope}/${prefix}_Add.
83const string MakeOptimizedNodeName(const NodeScopeAndName& node,
84 const string& sub_scope,
85 const string& prefix);
86// Make a name for a new node obtained by optimizing multiple nodes of the
87// original graph, starting from "root". The optimized node is placed under
88// the original scope of a "root" node.
89//
90// Example: [a/b/c/Add, x/y/z/Mul] -> a/b/c/${sub_scope}/${prefix}_Add_Mul
91const string MakeOptimizedNodeName(const NodeScopeAndName& root,
92 const std::vector<string> node_names,
93 const string& sub_scope,
94 const string& prefix);

96// Base class for multi-stage GraphOptimizers (ArithmeticOptimizer, etc...).
97//
98// If a graph optimizer consists of large number of small independent
99// rewrites, each of them should be implemented as a separate stage.
100//
101// * Result:
102// Each graph optimizer choose what result is reported by each stage
103// (e.g. each stage can fill in the name of optimized nodes, or have more
104// complex result).
105template <typename Result>
106class GraphOptimizerStage {
107 public:
108 explicit GraphOptimizerStage(const string& optimizer_name,
109 const string& stage_name,
110 const GraphOptimizerContext& ctx)
111 : optimizer_name_(optimizer_name), stage_name_(stage_name), ctx_(ctx) {}
112 virtual ~GraphOptimizerStage() = default;
113
114 const string& stage_name() const { return stage_name_; }
115 const string& optimizer_name() const { return optimizer_name_; }
116
117 // Check if we should try to simplify node. Returning true doesn't
118 // guarantee that node will be simplified.
119 //
120 // Should implement just a basic sanity check, without any expensive graph
121 // traversals.
122 virtual bool IsSupported(const NodeDef* node) const = 0;
123
124 // Try to simplify the given node.
125 //
126 // Return error status only if some precondition is failed, or got an
127 // incorrect graph. In every other case return Status:OK(), even if didn't
128 // simplify anything.
129 //
130 // Report result using output argument. Each GraphOptimizer can choose it's
131 // own Result type.
132 // TODO(ezhulenev): if it will appear that Result output parameter is not
133 // sufficiently useful (used with a reason by most optimizers), get rid of it,
134 // and remove template parameter.
135 virtual Status TrySimplify(NodeDef* node, Result* result) = 0;
136
137 // Return InvalidArgumentError if node is not supported by the optimizer
138 // stage.
139 // TODO(ezhulenev): make this check part of non-virtual public API
140 // (TrySimplify), and make virtual implementation protected.
141 Status EnsureNodeIsSupported(const NodeDef* node) const {
142 return IsSupported(node)
143 ? Status::OK()
144 : errors::InvalidArgument(
145 "Node ", node->name(), " is not supported by optimizer ",
146 optimizer_name_, " and stage ", stage_name_);
147 }
148
149 // Get a name for a new node, created by this stage, based on one or multiple
150 // nodes of an original graph.
151 const string OptimizedNodeName(const NodeScopeAndName& node) const {
152 return MakeOptimizedNodeName(node, optimizer_name_, stage_name_);
153 }
154 const string OptimizedNodeName(const NodeScopeAndName& root,
155 const std::vector<string>& nodes) const {
156 return MakeOptimizedNodeName(root, nodes, optimizer_name_, stage_name_);
157 }
158 const string OptimizedNodeName(const NodeScopeAndName& node,
159 const string& rewrite_rule) const {
160 const string prefix = strings::StrCat(stage_name_, "_", rewrite_rule);
161 return MakeOptimizedNodeName(node, optimizer_name_, prefix);
162 }
163
164 // Get a node by input name from a node map. Return an error if node was not
165 // found.
166 Status GetInputNode(const string& input, NodeDef** node) const {
167 return ::tensorflow::grappler::GetInputNode(ctx_, input, node);
168 }
169 // Lookup tensor properties by name. Tensor name might have non-zero port
170 // number. Return an error if tensor node doesn't exists in a graph, or it
171 // doesn't have properties defined for requested port.
172 Status GetTensorProperties(const string& tensor,
173 OpInfo::TensorProperties* properties) const {
174 return ::tensorflow::grappler::GetTensorProperties(ctx_, tensor,
175 properties);
176 }
177
178 NodeDef* AddCopyNode(const string& name, const NodeDef* node_to_copy) {
179 return ::tensorflow::grappler::AddCopyNode(ctx_, name, node_to_copy);
180 }
181 NodeDef* AddEmptyNode(const string& name) {
182 return ::tensorflow::grappler::AddEmptyNode(ctx_, name);
183 }
184
185 protected: // Data members
186 const string optimizer_name_;
187 const string stage_name_;
188 const GraphOptimizerContext ctx_;
189};

191template <typename Result>
192class GraphOptimizerStagePipeline {
193 public:
194 // Break predicate specifies if a pipeline should stop early, and not pass
195 // a node to the next registered optimizer stage, typically that should be the
196 // case when a stage successfully optimized a node, and it wants to yield
197 // control to the optimizer.
198 explicit GraphOptimizerStagePipeline(
199 const std::function<bool(const Result&)> break_predicate)
200 : break_predicate_(break_predicate) {}
201
202 // Add a stage to the pipeline. It should be called with the arguments for the
203 // stage constructor:
204 //
205 // pipeline.AddStage<FooStage>(constructor_arg1, constructor_arg2);
206 //
207 // Returns a reference to the added stage.
208 template <typename T, typename... Args>
209 T& AddStage(Args&&... args) {
210 auto stage = new T(std::forward<Args>(args)...);
211 stages_.push_back(std::unique_ptr<T>(stage));
212 return *stage;
213 }
214
215 // Pass a node through all registered optimizer stages, until break predicate
216 // is true.
217 //
218 // Return true, if pipeline exited after a break predicate was evaluated as
219 // 'true', which typically means that a node was optimized by one of the
220 // registered stages.
221 //
222 // Return false, if node was not optimized by any of registered stages.
223 bool PassThroughAllStages(NodeDef* node, Result* result) {
224 for (auto& stage : stages_) {
225 if (stage->IsSupported(node)) {
226 const Status stage_status = stage->TrySimplify(node, result);
227 // Each stage must be "error safe" (just like exception safe). In
228 // case of any error it must leave optimized graph unmodified.
229 if (!stage_status.ok()) {
230 LOG(WARNING) << "Failed to run optimizer " << stage->optimizer_name()
231 << ", stage " << stage->stage_name()
232 << ". Error: " << stage_status.error_message();
233 }
234 if (break_predicate_(*result)) return true;
235 }
236 }
237 return false;
238 }
239
240 std::size_t NumStages() { return stages_.size(); }
241
242 private:
243 std::vector<std::unique_ptr<GraphOptimizerStage<Result>>> stages_;
244 std::function<bool(const Result&)> break_predicate_;
245
246 TF_DISALLOW_COPY_AND_ASSIGN(GraphOptimizerStagePipeline);
247};
