23BCast::Vec ShapeDims(const TensorShapeProto& shape) {
24 BCast::Vec dims;
25 dims.reserve(shape.dim_size());
26 for (int i = 0; i < shape.dim_size(); ++i)
27 dims.push_back(shape.dim(i).size());
28 return dims;
29}

33bool IsKnown(const TensorShapeProto::Dim& dim) { return dim.size() >= 0; }
34
35bool IsKnownSymbolically(const TensorShapeProto::Dim& dim) {
36 return dim.size() <= -2;
37}
38
39bool IsUnknown(const TensorShapeProto::Dim& dim) { return dim.size() == -1; }
40
41bool ShapeIsSymbolicallyDefined(const TensorShapeProto& shape) {
42 return !shape.unknown_rank() &&
43 std::all_of(
44 shape.dim().begin(), shape.dim().end(),
45 [](const TensorShapeProto::Dim& dim) { return !IsUnknown(dim); });
46}
47
48bool ShapeIsSymbolicallyDefined(const OpInfo::TensorProperties& properties) {
49 return ShapeIsSymbolicallyDefined(properties.shape());
50}
51
52bool ShapesSymbolicallyEqual(const TensorShapeProto& left,
53 const TensorShapeProto& right) {
54 if (left.unknown_rank() || right.unknown_rank() ||
55 left.dim_size() != right.dim_size()) {
56 return false;
57 }
58 for (int i = 0; i < left.dim_size(); ++i) {
59 const auto& ldim = left.dim(i);
60 const auto& rdim = right.dim(i);
61 if (IsUnknown(ldim) || IsUnknown(rdim) || ldim.size() != rdim.size()) {
62 return false;
63 }
64 }
65 return true;
66}
67
68bool ShapesSymbolicallyEqual(const OpInfo::TensorProperties& left,
69 const OpInfo::TensorProperties& right) {
70 return ShapesSymbolicallyEqual(left.shape(), right.shape());
71}
72
73bool ShapesBroadcastable(const TensorShapeProto& left,
74 const TensorShapeProto& right) {
75 if (!ShapeIsSymbolicallyDefined(left) || !ShapeIsSymbolicallyDefined(right)) {
76 return false;
77 }
78 BCast bcast(ShapeDims(left), ShapeDims(right),
79 /*fewer_dims_optimization*/ false);
80 return bcast.IsValid();
81}
82
83bool ShapesBroadcastable(const OpInfo::TensorProperties& left,
84 const OpInfo::TensorProperties& right) {
85 return ShapesBroadcastable(left.shape(), right.shape());
86}

88bool CompareSymbolicallyShapedTensorSizes(const TensorShapeProto& left,
89 const TensorShapeProto& right) {
90 // if one of the ranks is unknown, it's impossible to compare tensor sizes
91 if (left.unknown_rank() || right.unknown_rank()) {
92 return false;
93 }
94
95 // Tensor size, computed as a product of defined dimensions
96 int64 left_defined_size = 1;
97 int64 right_defined_size = 1;
98
99 // Keep how many times each unknown dimension appeared on the left and right
100 std::unordered_map<int64, int64> left_unknown_dims;
101 std::unordered_map<int64, int64> right_unknown_dims;
102
103 // Assign unique id to every unknown dimension (-1). We are going to
104 // assign positive ids, because negative values are already used by
105 // symbolic dimensions.
106 int64 unknown_dim_id = 1;
107
108 // For each shape dimension update "defined tensor size", if shape is defined,
109 // or increment a counter for unknown dim.
110 auto process_dimensions =
111 [&unknown_dim_id](const TensorShapeProto& shape, int64* defined_size,
112 std::unordered_map<int64, int64>* unknown_dims) {
113 for (int i = 0; i < shape.dim_size(); ++i) {
114 const auto& dim = shape.dim(i);
115 int64 dim_size = dim.size();
116 if (dim_size > 0) {
117 *defined_size *= dim_size;
118 } else if (IsUnknown(dim)) {
119 ++(*unknown_dims)[unknown_dim_id++];
120 } else if (IsKnownSymbolically(dim)) {
121 ++(*unknown_dims)[dim_size];
122 }
123 }
124 };
125
126 process_dimensions(left, &left_defined_size, &left_unknown_dims);
127 process_dimensions(right, &right_defined_size, &right_unknown_dims);
128
129 // Compute a union of unknown dimension ids appeared in both shapes
130 std::set<int64> unknown_dims;
131 for (const auto& el : left_unknown_dims) unknown_dims.insert(el.first);
132 for (const auto& el : right_unknown_dims) unknown_dims.insert(el.first);
133
134 // Cancel unknown dimensions that appeared in both shapes
135 for (int64 unknown_dim : unknown_dims) {
136 int64 co_occurrence = std::min(left_unknown_dims[unknown_dim],
137 right_unknown_dims[unknown_dim]);
138 left_unknown_dims[unknown_dim] -= co_occurrence;
139 right_unknown_dims[unknown_dim] -= co_occurrence;
140 }
141
142 // Count unbalanced unknown dimensions
143 int64 left_unbalanced_unknown_dims = 0;
144 int64 right_unbalanced_unknown_dims = 0;
145 for (const auto& el : left_unknown_dims)
146 left_unbalanced_unknown_dims += el.second;
147 for (const auto& el : right_unknown_dims)
148 right_unbalanced_unknown_dims += el.second;
149
150 if (left_unbalanced_unknown_dims == 0 && right_unbalanced_unknown_dims == 0) {
151 // If unknown dimensions cancelled each other, compare tensor sizes
152 // represented by defined dimensions
153 return left_defined_size < right_defined_size;
154 }
155
156 if (left_defined_size <= right_defined_size &&
157 left_unbalanced_unknown_dims == 0 && right_unbalanced_unknown_dims > 0) {
158 // If size of a 'left" tensor computed from defined dimensions less or
159 // equal, and shape on the right has unbalanced unknown dimensions, we can
160 // guarantee that shape on the left is strictly smaller (assuming that
161 // unknown dimension size is larger than 1)
162 return true;
163 }
164
165 // In every other case, assuming that unknown dimensions can be arbitrary
166 // large in size, we can't guarantee any ordering
167 return false;
168}
169
170bool CompareSymbolicallyShapedTensorSizes(
171 const OpInfo::TensorProperties& left,
172 const OpInfo::TensorProperties& right) {
173 return CompareSymbolicallyShapedTensorSizes(left.shape(), right.shape());
174}
