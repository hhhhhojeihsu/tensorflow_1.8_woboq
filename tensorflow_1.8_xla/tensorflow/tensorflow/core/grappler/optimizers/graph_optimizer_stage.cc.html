21const NodeScopeAndName ParseNodeScopeAndName(const string& node_name) {
22 auto pos = node_name.find_last_of("/");
23 if (pos == string::npos) {
24 return {"", node_name};
25 } else {
26 return {node_name.substr(0, pos), node_name.substr(pos + 1)};
27 }
28};

30Status GetInputNode(const GraphOptimizerContext& ctx, const string& input,
31 NodeDef** node) {
32 string node_name = NodeName(input);
33 NodeDef* node_by_name = ctx.node_map->GetNode(node_name);
34 if (node_by_name == nullptr) {
35 return errors::FailedPrecondition("Node ", node_name,
36 " doesn't exists in a node map");
37 }
38 *node = node_by_name;
39 return Status::OK();
40}

42Status GetTensorProperties(const GraphOptimizerContext& ctx,
43 const string& tensor,
44 OpInfo::TensorProperties* properties) {
45 if (ctx.graph_properties == nullptr) {
46 return errors::InvalidArgument("Graph properties are unknown.");
47 }
48
49 int port;
50 string tensor_node_name = ParseNodeName(tensor, &port);
51 if (port < 0) {
52 return errors::InvalidArgument(
53 "Can't get tensor properties of control dependency ", tensor);
54 }
55
56 const auto& output_properties =
57 ctx.graph_properties->GetOutputProperties(tensor_node_name);
58 auto num_outputs = output_properties.size();
59
60 if (num_outputs == 0 || port > num_outputs - 1) {
61 return errors::InvalidArgument(
62 "Node ", tensor_node_name,
63 " is missing output properties at position :", port,
64 " (num_outputs=", num_outputs, ")");
65 }
66
67 properties->CopyFrom(output_properties[port]);
68 return Status::OK();
69}

71NodeDef* AddCopyNode(const GraphOptimizerContext& ctx, const string& name,
72 const NodeDef* node_to_copy) {
73 CHECK(node_to_copy != nullptr);
74 CHECK(!ctx.node_map->NodeExists(name))
75 << "Node " << name << " already exists in a graph";
76 NodeDef* new_node = ctx.optimized_graph->add_node();
77 *new_node = *node_to_copy;
78 new_node->set_name(name);
79 ctx.node_map->AddNode(name, new_node);
80 return new_node;
81}
82
83NodeDef* AddEmptyNode(const GraphOptimizerContext& ctx, const string& name) {
84 CHECK(!ctx.node_map->NodeExists(name))
85 << "Node " << name << " already exists in a graph";
86 NodeDef* new_node = ctx.optimized_graph->add_node();
87 new_node->set_name(name);
88 ctx.node_map->AddNode(name, new_node);
89 return new_node;
90}

92const string MakeOptimizedNodeName(const NodeScopeAndName& node,
93 const string& sub_scope,
94 const string& prefix) {
95 CHECK(!sub_scope.empty() || !prefix.empty())
96 << "Either optimized node name prefix or sub-scope must be non-empty";
97 string optimized_node_name;
98 if (!node.scope.empty()) {
99 strings::StrAppend(&optimized_node_name, node.scope, "/");
100 }
101 if (!sub_scope.empty()) {
102 strings::StrAppend(&optimized_node_name, sub_scope, "/");
103 }
104 if (!prefix.empty()) {
105 strings::StrAppend(&optimized_node_name, prefix, "_");
106 }
107 strings::StrAppend(&optimized_node_name, node.name);
108 return optimized_node_name;
109}
110
111const string MakeOptimizedNodeName(const NodeScopeAndName& root,
112 const std::vector<string> node_names,
113 const string& sub_scope,
114 const string& prefix) {
115 string optimized_node_name = MakeOptimizedNodeName(root, sub_scope, prefix);
116 for (const string& node_name : node_names) {
117 auto name_and_scope = ParseNodeScopeAndName(node_name);
118 strings::StrAppend(&optimized_node_name, "_", name_and_scope.name);
119 }
120 return optimized_node_name;
121}
