53Status AnalyticalCostEstimator::PredictCosts(const GraphDef& optimized_graph,
54 CostGraphDef* cost_graph,
55 Costs* costs) const {
56 GrapplerItem item = item_;
57 item.graph = optimized_graph;
58
59 std::unordered_map<string, CostGraphDef::Node*> name_to_cost;
60 if (cost_graph) {
61 for (auto& node : *cost_graph->mutable_node()) {
62 name_to_cost[node.name()] = &node;
63 }
64 }
65 std::vector<string> inaccurate_nodes;
66 int nodes_executed = 0;
67 VirtualScheduler scheduler(&item, use_static_shapes_, cluster_,
68 node_manager_.get());
69 auto status = scheduler.Init();
70 if (!status.ok()) {
71 costs->execution_time = Costs::Duration::max();
72 return status;
73 }
74
75 Costs node_costs;
76 do {
77 ++nodes_executed;
78 OpContext op_context = scheduler.GetCurrNode();
79 const string& op_name = op_context.name;
80
81 node_costs = node_estimator_->PredictCosts(op_context);
82 if (node_costs.inaccurate) {
83 inaccurate_nodes.push_back(op_name);
84 }
85 if (cost_graph) {
86 auto it = name_to_cost.find(op_name);
87 CostGraphDef::Node* cost_node;
88 if (it != name_to_cost.end()) {
89 cost_node = it->second;
90 } else {
91 cost_node = cost_graph->add_node();
92 cost_node->set_name(op_name);
93 }
94 cost_node->set_device(op_context.device_name);
95 cost_node->set_compute_cost(
96 node_costs.execution_time.asMicroSeconds().count());
97 cost_node->set_compute_time(
98 node_costs.compute_time.asMicroSeconds().count());
99 cost_node->set_memory_time(
100 node_costs.memory_time.asMicroSeconds().count());
101 for (const auto& output : op_context.op_info.outputs()) {
102 auto output_info = cost_node->add_output_info();
103 output_info->set_dtype(output.dtype());
104 auto shape = output_info->mutable_shape();
105 *shape = output.shape();
106 }
107 }
108 } while (scheduler.MarkCurrNodeExecuted(node_costs));
109
110 RunMetadata run_metadata;
111 *costs = scheduler.Summary(&run_metadata);
112 VLOG(1) << inaccurate_nodes.size() << " out of " << nodes_executed
113 << " nodes have inaccurate time estimation";
114 if (VLOG_IS_ON(3)) {
115 for (const auto& node : inaccurate_nodes) {
116 VLOG(4) << "Node with inaccurate time estimation: " << node;
117 }
118 }
119
120 if (VLOG_IS_ON(1)) {
121 bool verbosity = VLOG_IS_ON(2);
122 VLOG(1) << GetStatsStringFromRunMetadata(run_metadata, verbosity);
123 }
124 return Status::OK();
125}
