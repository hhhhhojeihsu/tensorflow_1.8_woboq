25namespace tensorflow {
26class CostGraphDef;
27class GraphDef;
28} // namespace tensorflow
29
30namespace tensorflow {
31namespace grappler {
32
33class Cluster;
34struct GrapplerItem;
35
36// Estimate the cost of running a Grappler item based on the theoretical
37// performance of the hardware that will run the model.
38class AnalyticalCostEstimator : public CostEstimator {
39 public:
40 // Does not take ownership of cluster.
41 AnalyticalCostEstimator(Cluster* cluster, bool use_static_shapes);
42 // Does not take ownership of the cluster, but takes ownership of the
43 // node_estimator and the node_manager
44 AnalyticalCostEstimator(Cluster* cluster,
45 OpLevelCostEstimator* node_estimator,
46 ReadyNodeManager* node_manager,
47 bool use_static_shapes);
48 ~AnalyticalCostEstimator() override {}
49
50 // Initializes the estimator for the specified grappler item.
51 // This implementation always returns OK.
52 Status Initialize(const GrapplerItem& item) override;
53
54 // Predict the performance of each node of the optimized graph and annotate
55 // the CostGraphDef with the corresponding estimates. Also returns the
56 // expected latency for the whole graph.
57 Status PredictCosts(const GraphDef& optimized_graph, CostGraphDef* cost_graph,
58 Costs* overall_latency) const override;
59
60 private:
61 Cluster* cluster_; // Not owned.
62 GrapplerItem item_;
63 std::unique_ptr<OpLevelCostEstimator> node_estimator_;
64 std::unique_ptr<ReadyNodeManager> node_manager_;
65 bool use_static_shapes_;
66};
67
68} // end namespace grappler
69} // end namespace tensorflow
