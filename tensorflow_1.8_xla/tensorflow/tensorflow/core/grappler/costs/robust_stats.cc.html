23// Given a sorted vector of values, calculate the median.
24// Returns 0 for an empty vector. Does not verify sortedness.
25static double SortedMedian(const std::vector<double> &values) {
26 const int n = values.size();
27 if (n == 0) return 0.0;
28 if (n & 1) {
29 return values[n / 2];
30 } else {
31 return (values[n / 2] + values[n / 2 - 1]) / 2.0;
32 }
33}
34
35// Given a vector of values (sorted or not), calculate the median.
36static double Median(std::vector<double> &&values) {
37 const size_t n = values.size();
38 if (n == 0) return 0;
39 const auto middle = values.begin() + (n / 2);
40 // Put the middle value in its place.
41 std::nth_element(values.begin(), middle, values.end());
42 if (n & 1) {
43 return *middle;
44 }
45 // Return the average of the two elements, the max_element lower than
46 // *middle is found between begin and middle as a post-cond of
47 // nth_element.
48 const auto lower_middle = std::max_element(values.begin(), middle);
49 // Preventing overflow. We know that '*lower_middle <= *middle'.
50 // If both are on opposite sides of zero, the sum won't overflow, otherwise
51 // the difference won't overflow.
52 if (*lower_middle <= 0 && *middle >= 0) {
53 return (*lower_middle + *middle) / 2;
54 }
55 return *lower_middle + (*middle - *lower_middle) / 2;
56}

58// Given a set of values, calculates the scaled Median Absolute Deviation (a
59// robust approximation to the standard deviation). This is calculated as the
60// median of the absolute deviations from the median, scaled by 1.4826. Its
61// advantage over the standard deviation is that it is not (as) affected by
62// outlier values. Returns a pair<median, mad>.
63static std::pair<double, double> ScaledMedianAbsoluteDeviation(
64 const std::vector<double> &sorted_values) {
65 double median = SortedMedian(sorted_values);
66
67 // Next, we calculate the absolute deviations from the median,
68 // find the median of the resulting data, and scale by 1.4826.
69 std::vector<double> deviations;
70 deviations.reserve(sorted_values.size());
71 for (double d : sorted_values) {
72 deviations.push_back(std::abs(d - median));
73 }
74 double mad = Median(std::move(deviations)) * 1.4826;
75 return std::pair<double, double>(median, mad);
76}

83 lo_ = values[0];

88// Computes an updated mean using Huber's weighting function (values beyond
89// the margin are weighted by margin / abs(value - mean).
90double UpdateHuberMean(const std::vector<double> &sorted_values, double mean,
91 double margin) {
92 int num_within = 0;
93 double sum = 0.0;
94
95 for (double d : sorted_values) {
96 if (d < mean - margin) {
97 sum -= margin;
98 } else if (d > mean + margin) {
99 sum += margin;
100 } else {
101 sum += d;
102 ++num_within;
103 }
104 }
105
106 // It is possible, for a set with an interquartile distance of 0, i.e., with
107 // more than half of the values at the median, to encounter the case where
108 // the Huber mean drifts slightly off the median and there are no values
109 // within the margin. In that case, just return the old mean, and the caller
110 // will quit.
111 if (num_within > 0) {
112 return sum / num_within;
113 } else {
114 return mean;
115 }
116}

118// Given a list of values, this approximates the stddev using the MAD and then
119// uses it to compute a Huber robust mean (sandwich mean). A margin of
120// c*stddev is defined around the current mean, and values are weighted by
121// margin / abs(value - mean) if outside the margin, or 1 if inside. This
122// computes the mean iteratively, because each time it changes the margin
123// shifts a bit. It typically settles very quickly, but it's possible for it
124// to be unstable. We limit it to 10 iterations.
125//
126void RobustStats::HuberMAD(const std::vector<double> &sorted_values) {
127 const std::pair<double, double> median_mad =
128 ScaledMedianAbsoluteDeviation(sorted_values);
129 mean_ = median_mad.first;
130 stddev_ = median_mad.second;
131
132 // c = 1.345 is the commonly used cutoff with 95% efficiency at the normal.
133 // We're using c = 1.5 to be a little more conservative, and because that's
134 // the default in S-plus.
135 // TODO(dehnert): Specialize Stats for integral types so we don't implement
136 // methods that don't make sense.
137 const double c = 1.5;
138 const double margin = c * stddev_;
139
140 // Iterate 10 times, or until the Huber mean stabilizes.
141 // If the margin is zero, we don't want mean to drift from the median.
142 if (margin > 0.0) {
143 for (int k = 0; k < 10; ++k) {
144 double old_mean = mean_;
145 mean_ = UpdateHuberMean(sorted_values, mean_, margin);
146 if (mean_ == old_mean) break;
147 }
148 }
