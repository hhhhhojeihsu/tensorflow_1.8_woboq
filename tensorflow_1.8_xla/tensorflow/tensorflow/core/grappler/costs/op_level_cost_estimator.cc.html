73Padding GetPadding(const OpInfo& op_features) {

89// TODO(dyoon): support non-4D tensors in the c ost functions of convolution
90// related ops (Conv, Pool, BatchNorm, and their backprops) and the related

92std::vector<int64> GetStrides(const OpInfo& op_features) {
93 if (op_features.attr().find("strides") != op_features.attr().end()) {
94 const auto strides = op_features.attr().at("strides").list().i();
95 CHECK(strides.size() == 4) << "Attr strides is not a length-4 vector: "
96 << op_features.DebugString();
97 return {strides[0], strides[1], strides[2], strides[3]};
98 }
99 return {1, 1, 1, 1};
100}

102std::vector<int64> GetKernelSize(const OpInfo& op_info) {
103 if (op_info.attr().find("ksize") != op_info.attr().end()) {
104 const auto ksize = op_info.attr().at("ksize").list().i();
105 CHECK(ksize.size() == 4)
106 << "Attr ksize is not a length-4 vector: " << op_info.DebugString();
107 return {ksize[0], ksize[1], ksize[2], ksize[3]};
108 }
109 // Note that FusedBatchNorm doesn't have ksize attr, but GetKernelSize returns
110 // {1, 1, 1, 1} in that case.
111 return {1, 1, 1, 1};
112}

114int64 GetOutputSize(const int64 input, const int64 filter, const int64 stride,
115 const Padding& padding) {
116 // Logic for calculating output shape is from GetWindowedOutputSizeVerbose()
117 // function in third_party/tensorflow/core/framework/common_shape_fns.cc.
118 if (padding == Padding::VALID) {
119 return (input - filter + stride) / stride;
120 } else { // SAME.
121 return (input + stride - 1) / stride;
122 }
123}

125// Return a minimum shape if the shape is unknown. If known, return the original
126// shape.
127TensorShapeProto MaybeGetMinimumShape(const TensorShapeProto& original_shape,
128 int rank, bool* found_unknown_shapes) {
129 auto shape = original_shape;
130 if (shape.unknown_rank() || shape.dim_size() < rank) {
131 *found_unknown_shapes = true;
132 TensorShapeProto::Dim dim;
133 VLOG(2) << "Use minimum shape because the rank is unknown.";
134 // The size of each dimension is at least 1, if unknown.
135 dim.set_size(1);
136 for (int i = 0; i < rank; i++) {
137 *shape.add_dim() = dim;
138 }
139 } else {
140 for (int i = 0; i < shape.dim_size(); i++) {
141 if (shape.dim(i).size() < 0) {
142 *found_unknown_shapes = true;
143 VLOG(2) << "Use minimum dim size 1 because the shape is unknown.";
144 // The size of each dimension is at least 1, if unknown.
145 shape.mutable_dim(i)->set_size(1);
146 }
147 }
148 }
149 return shape;
150}

152// Return the output element count of a binary element-wise op considering
153// broadcasting.
154int64 CwiseOutputElementCount(const TensorShapeProto& input_shape_1,
155 const TensorShapeProto& input_shape_2) {
156 bool found_unknown_shapes;
157 int rank = std::max(1, input_shape_1.dim_size());
158 TensorShapeProto output_shape =
159 MaybeGetMinimumShape(input_shape_1, rank, &found_unknown_shapes);
160
161 if (input_shape_1.dim_size() == input_shape_2.dim_size()) {
162 auto shape_1 =
163 MaybeGetMinimumShape(input_shape_1, rank, &found_unknown_shapes);
164 auto shape_2 =
165 MaybeGetMinimumShape(input_shape_2, rank, &found_unknown_shapes);
166 if (shape_1.dim_size() == shape_2.dim_size()) {
167 for (int i = 0; i < shape_1.dim_size(); i++) {
168 output_shape.mutable_dim(i)->set_size(
169 std::max(shape_1.dim(i).size(), shape_2.dim(i).size()));
170 }
171 }
172 }
173
174 int64 count = 1;
175 for (int i = 0; i < output_shape.dim_size(); i++) {
176 count *= output_shape.dim(i).size();
177 }
178 return count;
179}

312Costs OpLevelCostEstimator::PredictCosts(const OpContext& op_context) const {
313 const auto& op_features = op_context.op_info;
314 auto it = device_cost_impl_.find(op_features.op());
315 if (it == device_cost_impl_.end()) {
316 if (elementwise_ops_.find(op_features.op()) != elementwise_ops_.end()) {
317 return PredictCwiseOp(op_context);
318 }
319
320 VLOG(1) << "Missing accurate estimator for op: " << op_features.op();
321
322 return PredictCostOfAnUnknownOp(op_context);
323 }
324
325 std::function<Costs(const OpContext&)> estimator = it->second;
326 Costs costs = estimator(op_context);
327 VLOG(1) << "Operation " << op_features.op() << " takes "
328 << costs.execution_time.count() << " ns.";
329 return costs;
330}

332OpLevelCostEstimator::DeviceInfo OpLevelCostEstimator::GetDeviceInfo(
333 const DeviceProperties& device) const {
334 double gflops = -1;
335 double gb_per_sec = -1;
336
337 if (device.type() == "CPU") {
338 // Check if vector instructions are available, and refine performance
339 // prediction based on this.
340 // Frequencies are stored in MHz in the DeviceProperties.
341 gflops = device.num_cores() * device.frequency() * 1e-3;
342 if (gb_per_sec < 0) {
343 if (device.bandwidth() > 0) {
344 gb_per_sec = device.bandwidth() / 1e6;
345 } else {
346 gb_per_sec = 32;
347 }
348 }
349 } else if (device.type() == "GPU") {
350 const string architecture = device.environment().at("architecture");
351 int cores_per_multiprocessor;
352 if (architecture < "3") {
353 // Fermi
354 cores_per_multiprocessor = 32;
355 } else if (architecture < "4") {
356 // Kepler
357 cores_per_multiprocessor = 192;
358 } else if (architecture < "6") {
359 // Maxwell
360 cores_per_multiprocessor = 128;
361 } else {
362 // Pascal (compute capability version 6) and Volta (compute capability
363 // version 7)
364 cores_per_multiprocessor = 64;
365 }
366 gflops = device.num_cores() * device.frequency() * 1e-3 *
367 cores_per_multiprocessor * kOpsPerMac;
368 if (device.bandwidth() > 0) {
369 gb_per_sec = device.bandwidth() / 1e6;
370 } else {
371 gb_per_sec = 100;
372 }
373 }
374 VLOG(1) << "Device: " << device.type() << " gflops: " << gflops
375 << " gb_per_sec: " << gb_per_sec;
376
377 DCHECK_LT(0, gflops) << device.DebugString();
378 DCHECK_LT(0, gb_per_sec) << device.DebugString();
379
380 return {gflops, gb_per_sec};
381}

383Costs OpLevelCostEstimator::PredictCwiseOp(const OpContext& op_context) const {
384 const auto& op_features = op_context.op_info;
385 bool found_unknown_shapes = false;
386 // For unary or binary element-wise operations, op count is the element count
387 // of any input. We use the count for the largest input here to be more robust
388 // in case that the shape is unknown or partially known for other input.
389 int64 op_count =
390 CalculateLargestInputCount(op_features, &found_unknown_shapes);
391 // If output shape is available, try use the element count calcuated from
392 // that.
393 if (op_features.outputs_size() > 0) {
394 op_count =
395 std::max(op_count, CalculateTensorElementCount(op_features.outputs(0),
396 &found_unknown_shapes));
397 }
398 // For binary ops, calculate the output shape possibly resulting from
399 // broadcasting.
400 if (op_features.inputs_size() >= 2) {
401 op_count = std::max(op_count,
402 CwiseOutputElementCount(op_features.inputs(0).shape(),
403 op_features.inputs(1).shape()));
404 }
405
406 int op_cost = 1;
407 bool is_known_elementwise_op = false;
408 auto it = elementwise_ops_.find(op_features.op());
409 if (it != elementwise_ops_.end()) {
410 op_cost = it->second;
411 is_known_elementwise_op = true;
412 } else {
413 LOG(WARNING) << "Not a cwise op: " << op_features.op();
414 }
415
416 Costs costs = PredictOpCountBasedCost(op_count * op_cost, op_features);
417 if (found_unknown_shapes || !is_known_elementwise_op) {
418 costs.inaccurate = true;
419 }
420 return costs;
421}

423Costs OpLevelCostEstimator::PredictCostOfAnUnknownOp(
424 const OpContext& op_context) const {
425 // Don't assume the operation is cwise, return cost based on input/output size
426 // and admit that it is inaccurate...
427 auto costs = PredictOpCountBasedCost(0, op_context.op_info);
428 costs.inaccurate = true;
429 return costs;
430}

432Costs OpLevelCostEstimator::PredictOpCountBasedCost(
433 double operations, const OpInfo& op_info) const {
434 bool unknown_shapes = false;
435 const double input_size = CalculateInputSize(op_info, &unknown_shapes);
436 const double output_size = CalculateOutputSize(op_info, &unknown_shapes);
437 const double total_io_bytes = input_size + output_size;
438 Costs costs = PredictOpCountBasedCost(operations, total_io_bytes, op_info);
439 costs.inaccurate = unknown_shapes;
440 costs.max_memory = output_size;
441 return costs;
442}

444Costs OpLevelCostEstimator::PredictOpCountBasedCost(
445 double operations, double total_io_bytes, const OpInfo& op_info) const {
446 const DeviceInfo device_info = GetDeviceInfo(op_info.device());
447 if (device_info.gigaops <= 0 || device_info.gb_per_sec <= 0) {
448 VLOG(1) << "BAD DEVICE. Op:" << op_info.op()
449 << " device type:" << op_info.device().type()
450 << " device model:" << op_info.device().model();
451 }
452
453 Costs::NanoSeconds compute_cost(std::ceil(operations / device_info.gigaops));
454 VLOG(1) << "Op:" << op_info.op() << " GOps:" << operations / 1e9
455 << " Compute Time (ns):" << compute_cost.count();
456
457 Costs::NanoSeconds memory_cost(
458 std::ceil(total_io_bytes / device_info.gb_per_sec));
459 VLOG(1) << "Op:" << op_info.op() << " Size (KB):" << (total_io_bytes) / 1e3
460 << " Memory Time (ns):" << memory_cost.count();
461
462 Costs costs;
463 costs.compute_time = compute_cost;
464 costs.memory_time = memory_cost;
465 CombineCostsAndUpdateExecutionTime(&costs);
466 return costs;
467}

474// Helper to translate the positional arguments into named fields.
475OpLevelCostEstimator::ConvolutionDimensions
476OpLevelCostEstimator::ConvolutionDimensionsFromInputs(
477 const TensorShapeProto& original_image_shape,
478 const TensorShapeProto& original_filter_shape, const OpInfo& op_features,
479 bool* found_unknown_shapes) {
480 VLOG(2) << "op features: " << op_features.DebugString();
481 VLOG(2) << "Original image shape: " << original_image_shape.DebugString();
482 VLOG(2) << "Original filter shape: " << original_filter_shape.DebugString();
483 auto image_shape =
484 MaybeGetMinimumShape(original_image_shape, 4, found_unknown_shapes);
485 auto filter_shape =
486 MaybeGetMinimumShape(original_filter_shape, 4, found_unknown_shapes);
487 VLOG(2) << "Image shape: " << image_shape.DebugString();
488 VLOG(2) << "Filter shape: " << filter_shape.DebugString();
489
490 int x_index, y_index, channel_index;
491 const string& data_format = GetDataFormat(op_features);
492 if (data_format == "NCHW") {
493 x_index = 2;
494 y_index = 3;
495 channel_index = 1;
496 } else {
497 x_index = 1;
498 y_index = 2;
499 channel_index = 3;
500 }
501 int64 batch = image_shape.dim(0).size();
502 int64 ix = image_shape.dim(x_index).size();
503 int64 iy = image_shape.dim(y_index).size();
504 int64 iz = image_shape.dim(channel_index).size();
505 int64 kx = filter_shape.dim(0).size();
506 int64 ky = filter_shape.dim(1).size();
507 std::vector<int64> strides = GetStrides(op_features);
508 const auto padding = GetPadding(op_features);
509 int64 sx = strides[x_index];
510 int64 sy = strides[y_index];
511 int64 ox = GetOutputSize(ix, kx, sx, padding);
512 int64 oy = GetOutputSize(iy, ky, sy, padding);
513 int64 oz = filter_shape.dim(3).size();
514 // Only check equality when both sizes are known (in other words, when
515 // neither is set to a minimum dimension size of 1).
516 if (iz != 1 && filter_shape.dim(2).size() != 1) {
517 CHECK_EQ(iz, filter_shape.dim(2).size());
518 } else {
519 iz = std::max<int64>(iz, filter_shape.dim(2).size());
520 }
521 OpLevelCostEstimator::ConvolutionDimensions conv_dims = {
522 batch, ix, iy, iz, kx, ky, oz, ox, oy, sx, sy, padding};
523
524 VLOG(1) << "Batch Size:" << batch;
525 VLOG(1) << "Image Dims:" << ix << "," << iy;
526 VLOG(1) << "Input Features:" << iz;
527 VLOG(1) << "Kernel Dims:" << kx << "," << ky;
528 VLOG(1) << "Output Features:" << oz;
529 VLOG(1) << "Output Dims:" << ox << "," << oy;
530 VLOG(1) << "Strides:" << sx << "," << sy;
531 VLOG(1) << "Padding:" << (padding == Padding::VALID ? "VALID" : "SAME");
532 return conv_dims;
533}
534
535int64 OpLevelCostEstimator::CountConv2DOperations(
536 const OpInfo& op_features, ConvolutionDimensions* conv_info,
537 bool* found_unknown_shapes) const {
538 if (op_features.op() != kConv2d) {
539 LOG(ERROR) << "Invalid Operation";
540 return 0;
541 }
542 ConvolutionDimensions conv_dims = ConvolutionDimensionsFromInputs(
543 op_features.inputs(0).shape(), op_features.inputs(1).shape(), op_features,
544 found_unknown_shapes);
545
546 int64 ops = conv_dims.batch;
547 ops *= conv_dims.ox * conv_dims.oy;
548 ops *= conv_dims.kx * conv_dims.ky;
549 ops *= conv_dims.iz * conv_dims.oz;
550 ops *= kOpsPerMac;
551
552 if (conv_info != nullptr) {
553 *conv_info = conv_dims;
554 }
555 return ops;
556}

565 const OpInfo& op_features, MatMulDimensions* mat_mul,
566 bool* found_unknown_shapes) const {
567 double ops = 0;
568
569 if (op_features.inputs_size() < 2) {
570 LOG(ERROR) << "Need 2 inputs but got " << op_features.inputs_size();
571 *found_unknown_shapes = true;
572 return 0;
573 }
574
575 auto& a_matrix = op_features.inputs(0);
576 auto& b_matrix = op_features.inputs(1);
577
578 bool transpose_a = false;
579 bool transpose_b = false;
580
581 double m_dim, n_dim, k_dim, k_dim_b = 0;
582
583 for (const auto& item : op_features.attr()) {
584 VLOG(1) << "Key:" << item.first
585 << " Value:" << SummarizeAttrValue(item.second);
586 if (item.first == "transpose_a" && item.second.b() == true)
587 transpose_a = true;
588 if (item.first == "transpose_b" && item.second.b() == true)
589 transpose_b = true;
590 }
591 VLOG(1) << "transpose_a:" << transpose_a;
592 VLOG(1) << "transpose_b:" << transpose_b;
593 auto a_matrix_shape =
594 MaybeGetMinimumShape(a_matrix.shape(), 2, found_unknown_shapes);
595 auto b_matrix_shape =
596 MaybeGetMinimumShape(b_matrix.shape(), 2, found_unknown_shapes);
597 if (transpose_a) {
598 m_dim = a_matrix_shape.dim(1).size();
599 k_dim = a_matrix_shape.dim(0).size();
600 } else {
601 m_dim = a_matrix_shape.dim(0).size();
602 k_dim = a_matrix_shape.dim(1).size();
603 }
604 if (transpose_b) {
605 k_dim_b = b_matrix_shape.dim(1).size();
606 n_dim = b_matrix_shape.dim(0).size();
607 } else {
608 k_dim_b = b_matrix_shape.dim(0).size();
609 n_dim = b_matrix_shape.dim(1).size();
610 }
611
612 VLOG(1) << "M, N, K: " << m_dim << "," << n_dim << "," << k_dim;
613 // Only check equality when both sizes are known (in other words, when
614 // neither is set to a minimum dimension size of 1).
615 if (k_dim_b != 1 && k_dim != 1 && k_dim_b != k_dim) {
616 LOG(ERROR) << "Incompatible Matrix dimensions";
617 return ops;
618 } else {
619 // One of k_dim and k_dim_b might be 1 (mininum dimension size).
620 k_dim = std::max(k_dim, k_dim_b);
621 }
622
623 ops = m_dim * n_dim * k_dim * 2;
624 VLOG(1) << "Operations for Matmul" << ops;
625
626 if (mat_mul != nullptr) {
627 mat_mul->m = m_dim;
628 mat_mul->n = n_dim;
629 mat_mul->k = k_dim;
630 }
631 return ops;
632}

634int64 OpLevelCostEstimator::CountBatchMatMulOperations(
635 const OpInfo& op_features, bool* found_unknown_shapes) const {
636 if (op_features.op() != kBatchMatMul) {
637 LOG(ERROR) << "Invalid Operation: " << op_features.op();
638 *found_unknown_shapes = true;
639 return 0;
640 }
641 if (op_features.inputs_size() != 2) {
642 LOG(ERROR) << "Expected 2 inputs but got " << op_features.inputs_size();
643 *found_unknown_shapes = true;
644 return 0;
645 }
646
647 double ops = 0;
648 const auto& a_input = op_features.inputs(0);
649 const auto& b_input = op_features.inputs(1);
650
651 // BatchMatMul requires inputs of at least matrix shape (rank 2).
652 // The two most minor dimensions of each input are matrices that
653 // need to be multiplied together. The other dimensions determine
654 // the number of such MatMuls. For example, if the BatchMatMul has
655 // inputs of shape:
656 // a_input_shape = [2, 3, 4, 5]
657 // b_input_shape = [2, 3, 5, 6]
658 // then there are 2*3 = 6 MatMuls of dimensions m = 4, k = 5, n = 6
659 // in this BatchMatMul.
660 const int matrix_rank = 2;
661
662 bool a_input_shape_unknown = false;
663 bool b_input_shape_unknown = false;
664
665 TensorShapeProto a_input_shape = MaybeGetMinimumShape(
666 a_input.shape(), std::max(matrix_rank, a_input.shape().dim_size()),
667 &a_input_shape_unknown);
668 TensorShapeProto b_input_shape = MaybeGetMinimumShape(
669 b_input.shape(), std::max(matrix_rank, b_input.shape().dim_size()),
670 &b_input_shape_unknown);
671
672 *found_unknown_shapes = a_input_shape_unknown || b_input_shape_unknown ||
673 (a_input.shape().dim_size() < matrix_rank) ||
674 (b_input.shape().dim_size() < matrix_rank);
675
676 // Compute the number of matmuls as the max indicated at each dimension
677 // by either input. Note that the shapes do not have to have
678 // the same rank due to incompleteness.
679 TensorShapeProto* bigger_rank_shape = &a_input_shape;
680 TensorShapeProto* smaller_rank_shape = &b_input_shape;
681 if (b_input_shape.dim_size() > a_input_shape.dim_size()) {
682 bigger_rank_shape = &b_input_shape;
683 smaller_rank_shape = &a_input_shape;
684 }
685 int num_matmuls = 1;
686 for (int b_i = 0,
687 s_i = smaller_rank_shape->dim_size() - bigger_rank_shape->dim_size();
688 b_i < bigger_rank_shape->dim_size() - matrix_rank; ++b_i, ++s_i) {
689 int b_dim = bigger_rank_shape->dim(b_i).size();
690 int s_dim = 1;
691 if (s_i >= 0) {
692 s_dim = smaller_rank_shape->dim(s_i).size();
693 }
694 num_matmuls *= std::max(b_dim, s_dim);
695 }
696
697 // Build the MatMul. Note that values are ignored here since we are just
698 // counting ops (e.g. only shapes matter).
699 OpInfo matmul_op_features;
700 matmul_op_features.set_op("MatMul");
701
702 AttrValue transpose_a;
703 transpose_a.set_b(false);
704 if (op_features.attr().find("adj_x") != op_features.attr().end()) {
705 transpose_a.set_b(op_features.attr().at("adj_x").b());
706 }
707 (*matmul_op_features.mutable_attr())["transpose_a"] = transpose_a;
708
709 AttrValue transpose_b;
710 transpose_b.set_b(false);
711 if (op_features.attr().find("adj_y") != op_features.attr().end()) {
712 transpose_b.set_b(op_features.attr().at("adj_y").b());
713 }
714 (*matmul_op_features.mutable_attr())["transpose_b"] = transpose_b;
715
716 OpInfo::TensorProperties* a_matrix = matmul_op_features.add_inputs();
717 a_matrix->set_dtype(a_input.dtype());
718 TensorShapeProto* a_matrix_shape = a_matrix->mutable_shape();
719 for (int i = std::max(0, a_input_shape.dim_size() - matrix_rank);
720 i < a_input_shape.dim_size(); ++i) {
721 *(a_matrix_shape->add_dim()) = a_input_shape.dim(i);
722 }
723
724 OpInfo::TensorProperties* b_matrix = matmul_op_features.add_inputs();
725 b_matrix->set_dtype(b_input.dtype());
726 TensorShapeProto* b_matrix_shape = b_matrix->mutable_shape();
727 for (int i = std::max(0, b_input_shape.dim_size() - matrix_rank);
728 i < b_input_shape.dim_size(); ++i) {
729 *(b_matrix_shape->add_dim()) = b_input_shape.dim(i);
730 }
731
732 for (int i = 0; i < num_matmuls; ++i) {
733 bool matmul_unknown_shapes = false;
734 ops += CountMatMulOperations(matmul_op_features, &matmul_unknown_shapes);
735 *found_unknown_shapes |= matmul_unknown_shapes;
736 }
737 return ops;
738}

740bool GetTensorShapeProtoFromTensorProto(const TensorProto& tensor_proto,
741 TensorShapeProto* tensor_shape_proto) {
742 tensor_shape_proto->Clear();
743 // First convert TensorProto into Tensor class so that it correctly parses
744 // data values within TensorProto (whether it's in int_val, int64_val,
745 // tensor_content, or anything.
746 Tensor tensor(tensor_proto.dtype());
747 if (!tensor.FromProto(tensor_proto)) {
748 LOG(WARNING) << "GetTensorShapeProtoFromTensorProto() -- "
749 << "failed to parse TensorProto: "
750 << tensor_proto.DebugString();
751 return false;
752 }
753 if (tensor.dims() != 1) {
754 LOG(WARNING) << "GetTensorShapeProtoFromTensorProto() -- "
755 << "tensor is not 1D: " << tensor.dims();
756 return false;
757 }
758 // Then, convert it back to TensorProto using AsProtoField, which makes sure
759 // the data is in int_val, int64_val, or such repeated data fields, not in
760 // tensor_content.
761 TensorProto temp_tensor;
762 tensor.AsProtoField(&temp_tensor);
763
764#define TENSOR_VALUES_TO_TENSOR_SHAPE_PROTO(type) \
765 do { \
766 for (const auto& value : temp_tensor.type##_val()) { \
767 tensor_shape_proto->add_dim()->set_size(value); \
768 } \
769 } while (0)
770
771 if (tensor.dtype() == DT_INT32 || tensor.dtype() == DT_INT16 ||
772 tensor.dtype() == DT_INT8 || tensor.dtype() == DT_UINT8) {
773 TENSOR_VALUES_TO_TENSOR_SHAPE_PROTO(int);
774 } else if (tensor.dtype() == DT_INT64) {
775 TENSOR_VALUES_TO_TENSOR_SHAPE_PROTO(int64);
776 } else if (tensor.dtype() == DT_UINT32) {
777 TENSOR_VALUES_TO_TENSOR_SHAPE_PROTO(uint32);
778 } else if (tensor.dtype() == DT_UINT64) {
779 TENSOR_VALUES_TO_TENSOR_SHAPE_PROTO(uint64);
780 } else {
781 LOG(WARNING) << "GetTensorShapeProtoFromTensorProto() -- "
782 << "Unsupported dtype: " << tensor.dtype();
783 return false;
784 }
785#undef TENSOR_VALUES_TO_TENSOR_SHAPE_PROTO
786
787 return true;
788}

823 input_shape, op_features.inputs(1).shape(), op_features,

827 ops *= conv_dims.ox * conv_dims.oy;
828 ops *= conv_dims.kx * conv_dims.ky;
829 ops *= conv_dims.iz * conv_dims.oz;
830 ops *= kOpsPerMac;

988Costs OpLevelCostEstimator::PredictFusedConv2DBiasActivation(
989 const OpContext& op_context) const {
990 // FusedConv2DBiasActivation computes a fused kernel which implements:
991 // 2D convolution, adds side input with separate scaling on convolution and
992 // side inputs, then adds bias, and finally applies the ReLU activation
993 // function to the result:
994 //
995 // Input -> Conv2D -> Add -> BiasAdd -> ReLU
996 // ^ ^ ^
997 // Filter Side Input Bias
998 //
999 // Note that when adding the side input, the operation multiplies the output
1000 // of Conv2D by conv_input_scale, confusingly, and the side_input by
1001 // side_input_scale.
1002 //
1003 // Note that in the special case that side_input_scale is 0, which we infer
1004 // from side_input having dimensions [], we skip that addition operation.
1005 //
1006 // For more information, see
1007 // contrib/fused_conv/kernels/fused_conv2d_bias_activation_op.cc
1008 auto& conv_input = op_context.op_info.inputs(0);
1009 auto& filter = op_context.op_info.inputs(1);
1010 auto& bias = op_context.op_info.inputs(2);
1011 auto& side_input = op_context.op_info.inputs(3);
1012 auto& conv_input_scale = op_context.op_info.inputs(4);
1013 auto& side_input_scale = op_context.op_info.inputs(5);
1014
1015 // Manually compute our convolution dimensions.
1016 bool found_unknown_shapes = false;
1017 auto dims = ConvolutionDimensionsFromInputs(
1018 conv_input.shape(), filter.shape(), op_context.op_info,
1019 &found_unknown_shapes);
1020
1021 // Construct the shape of our output tensor from our convolution dimensions
1022 // and format, as it may not be available yet.
1023 //
1024 // TODO(varomodt): should we centralize the Conv2D input/output shapes?
1025 bool unknown_conv_format = false;
1026 OpInfo::TensorProperties output;
1027 switch (GetConvolutionFormat(op_context)) {
1028 case NCHW:
1029 output =
1030 DescribeTensor(DT_FLOAT, {dims.batch, dims.oz, dims.ox, dims.oy});
1031 break;
1032 case NHWC:
1033 output =
1034 DescribeTensor(DT_FLOAT, {dims.batch, dims.ox, dims.oy, dims.oz});
1035 break;
1036 default:
1037 // TODO(b/77722245): support cost estimation for NCHW_VECT_C.
1038 LOG(WARNING) << "unsupported data format: "
1039 << GetDataFormat(op_context.op_info)
1040 << " Defaulting to NHWC.";
1041 output =
1042 DescribeTensor(DT_FLOAT, {dims.batch, dims.ox, dims.oy, dims.oz});
1043 unknown_conv_format = true;
1044 break;
1045 }
1046
1047 // Add the operations the fused op always computes.
1048 std::vector<OpContext> component_ops = {
1049 FusedChildContext(op_context, "Conv2D", output, {conv_input, filter}),
1050 FusedChildContext(op_context, "Mul", output, {output, conv_input_scale}),
1051 FusedChildContext(op_context, "BiasAdd", output, {output, bias}),
1052 FusedChildContext(op_context, "Relu", output, {output})};
1053
1054 // Add our side_input iff it's non-empty.
1055 if (side_input.shape().dim_size() > 0) {
1056 component_ops.push_back(FusedChildContext(op_context, "Mul", side_input,
1057 {side_input, side_input_scale}));
1058 component_ops.push_back(
1059 FusedChildContext(op_context, "Add", output, {side_input, output}));
1060 }
1061
1062 // Construct an op_context which definitely has our output shape.
1063 auto op_context_with_output = op_context;
1064 op_context_with_output.op_info.mutable_outputs()->Clear();
1065 *op_context_with_output.op_info.mutable_outputs()->Add() = output;
1066
1067 // Construct component operations and run the cost computation.
1068 auto costs = PredictFusedOp(op_context_with_output, component_ops);
1069 costs.inaccurate |= found_unknown_shapes || unknown_conv_format;
1070 return costs;
1071}

1135Costs OpLevelCostEstimator::PredictGatherOrSlice(
1136 const OpContext& op_context) const {
1137 // Gather & Slice ops can have a very large input, but only access a small
1138 // part of it. For these op the size of the output determines the memory cost.
1139 const auto& op_info = op_context.op_info;
1140
1141 const int inputs_needed = op_info.op() == "Slice" ? 3 : 2;
1142 if (op_info.outputs_size() == 0 || op_info.inputs_size() < inputs_needed) {
1143 Costs costs = Costs::ZeroCosts();
1144 costs.inaccurate = true;
1145 return costs;
1146 }
1147
1148 bool unknown_shapes = false;
1149
1150 // Each output element is a copy of some element from input.
1151 // For roofline estimate we assume each copy has a unit cost.
1152 const int64 op_count =
1153 CalculateTensorElementCount(op_info.outputs(0), &unknown_shapes);
1154
1155 const double output_size = CalculateOutputSize(op_info, &unknown_shapes);
1156 double input_size = output_size;
1157 if (op_info.op() == "Slice") {
1158 // Add 'begin' & 'size' tensors sizes.
1159 input_size +=
1160 CalculateTensorElementCount(op_info.inputs(1), &unknown_shapes) +
1161 CalculateTensorElementCount(op_info.inputs(2), &unknown_shapes);
1162 } else {
1163 // Assuming this is "Gather" or "GatherV2" op, add 'indices' size.
1164 input_size +=
1165 CalculateTensorElementCount(op_info.inputs(1), &unknown_shapes);
1166 }
1167
1168 const double total_io = input_size + output_size;
1169 Costs costs = PredictOpCountBasedCost(op_count, total_io, op_info);
1170 costs.inaccurate = unknown_shapes;
1171 costs.max_memory = output_size;
1172
1173 return costs;
1174}

1176Costs OpLevelCostEstimator::PredictFusedOp(
1177 const OpContext& op_context,
1178 const std::vector<OpContext>& fused_op_contexts) const {
1179 // Note that PredictOpCountBasedCost will get the correct memory_time from
1180 // the node's inputs and outputs; but we don't want to have to re-implement
1181 // the logic for computing the operation count of each of our component
1182 // operations here; so we simply add the compute times of each component
1183 // operation, then update the execution time.
1184 Costs fused_cost = PredictOpCountBasedCost(0, op_context.op_info);
1185 fused_cost.compute_time = 0;
1186 fused_cost.inaccurate = false;
1187 for (auto& fused_op : fused_op_contexts) {
1188 auto op_cost = PredictCosts(fused_op);
1189 fused_cost.compute_time += op_cost.compute_time;
1190 fused_cost.inaccurate |= op_cost.inaccurate;
1191 }
1192
1193 CombineCostsAndUpdateExecutionTime(&fused_cost);
1194 return fused_cost;
1195}

1283Costs OpLevelCostEstimator::PredictMaxPool(const OpContext& op_context) const {
1284 bool found_unknown_shapes = false;
1285 const auto& op_info = op_context.op_info;
1286 // x: op_info.inputs(0)
1287 ConvolutionDimensions dims = OpDimensionsFromInputs(
1288 op_info.inputs(0).shape(), op_info, &found_unknown_shapes);
1289 // kx * ky - 1 comparisons per output (kx * xy > 1)
1290 // or 1 copy per output (kx * k1 = 1).
1291 int per_output_ops = dims.kx * dims.ky == 1 ? 1 : dims.kx * dims.ky - 1;
1292 int64 ops = dims.batch * dims.ox * dims.oy * dims.oz * per_output_ops;
1293
1294 double total_input_size = 0;
1295 if (dims.ky >= dims.sy) {
1296 total_input_size =
1297 CalculateTensorSize(op_info.inputs(0), &found_unknown_shapes);
1298 } else { // dims.ky < dims.sy
1299 // Vertical stride is larger than vertical kernel; assuming row-major
1300 // format, skip unnecessary rows (or read every kx rows per sy rows, as the
1301 // others are not used for output).
1302 const auto data_size = DataTypeSize(BaseType(op_info.inputs(0).dtype()));
1303 total_input_size =
1304 data_size * dims.batch * dims.ix * dims.ky * dims.oy * dims.iz;
1305 }
1306 const double total_output_size =
1307 CalculateOutputSize(op_info, &found_unknown_shapes);
1308
1309 Costs costs = PredictOpCountBasedCost(
1310 ops, total_input_size + total_output_size, op_info);
1311 costs.inaccurate = found_unknown_shapes;
1312 costs.max_memory = total_output_size;
1313 return costs;
1314}

1316Costs OpLevelCostEstimator::PredictMaxPoolGrad(
1317 const OpContext& op_context) const {
1318 bool found_unknown_shapes = false;
1319 const auto& op_info = op_context.op_info;
1320 // x: op_info.inputs(0)
1321 // y: op_info.inputs(1)
1322 // y_grad: op_info.inputs(2)
1323 ConvolutionDimensions dims = OpDimensionsFromInputs(
1324 op_info.inputs(0).shape(), op_info, &found_unknown_shapes);
1325
1326 int64 ops = 0;
1327 if (dims.kx == 1 && dims.ky == 1) {
1328 // 1x1 window. No need to know which input was max.
1329 ops = dims.batch * dims.ix * dims.iy * dims.iz;
1330 } else if (dims.kx <= dims.sx && dims.ky <= dims.sy) {
1331 // Non-overlapping window: re-run maxpool, then assign zero or y_grad.
1332 ops = dims.batch * dims.iz *
1333 (dims.ox * dims.oy * (dims.kx * dims.ky - 1) + dims.ix * dims.iy);
1334 } else {
1335 // Overlapping window: initialize with zeros, re-run maxpool, then
1336 // accumulate y_gad to proper x_grad locations.
1337 ops = dims.batch * dims.iz *
1338 (dims.ox * dims.oy * (dims.kx * dims.ky - 1) + dims.ix * dims.iy * 2);
1339 }
1340
1341 // Just read x and y_grad; no need to read y as we assume MaxPoolGrad re-run
1342 // MaxPool internally.
1343 double total_input_size =
1344 CalculateTensorSize(op_info.inputs(0), &found_unknown_shapes);
1345 total_input_size +=
1346 CalculateTensorSize(op_info.inputs(2), &found_unknown_shapes);
1347 // Write x_grad; size equal to x.
1348 const double total_output_size =
1349 CalculateTensorSize(op_info.inputs(0), &found_unknown_shapes);
1350
1351 Costs costs = PredictOpCountBasedCost(
1352 ops, total_input_size + total_output_size, op_info);
1353 costs.inaccurate = found_unknown_shapes;
1354 costs.max_memory = total_output_size;
1355 return costs;
1356}

1390Costs OpLevelCostEstimator::PredictAvgPoolGrad(
1391 const OpContext& op_context) const {
1392 bool found_unknown_shapes = false;
1393 const auto& op_info = op_context.op_info;
1394 // x's shape: op_info.inputs(0)
1395 // y_grad: op_info.inputs(1)
1396
1397 // Extract x_shape from op_info.inputs(0).value() or op_info.outputs(0).
1398 bool shape_found = false;
1399 TensorShapeProto x_shape;
1400 if (op_info.inputs_size() >= 1 && op_info.inputs(0).has_value()) {
1401 const TensorProto& value = op_info.inputs(0).value();
1402 shape_found = GetTensorShapeProtoFromTensorProto(value, &x_shape);
1403 }
1404 if (!shape_found && op_info.outputs_size() > 0) {
1405 x_shape = op_info.outputs(0).shape();
1406 shape_found = true;
1407 }
1408 if (!shape_found) {
1409 // Set the minimum shape that's feasible.
1410 x_shape.Clear();
1411 for (int i = 0; i < 4; ++i) {
1412 x_shape.add_dim()->set_size(1);
1413 }
1414 found_unknown_shapes = true;
1415 }
1416
1417 ConvolutionDimensions dims =
1418 OpDimensionsFromInputs(x_shape, op_info, &found_unknown_shapes);
1419
1420 int64 ops = 0;
1421 if (dims.kx <= dims.sx && dims.ky <= dims.sy) {
1422 // Non-overlapping window.
1423 ops = dims.batch * dims.iz * (dims.ix * dims.iy + dims.ox * dims.oy);
1424 } else {
1425 // Overlapping window.
1426 ops = dims.batch * dims.iz *
1427 (dims.ix * dims.iy + dims.ox * dims.oy * (dims.kx * dims.ky + 1));
1428 }
1429
1430 const double total_input_size =
1431 CalculateInputSize(op_info, &found_unknown_shapes);
1432 const double total_output_size =
1433 CalculateOutputSize(op_info, &found_unknown_shapes);
1434
1435 Costs costs = PredictOpCountBasedCost(
1436 ops, total_input_size + total_output_size, op_info);
1437 costs.inaccurate = found_unknown_shapes;
1438 costs.max_memory = total_output_size;
1439 return costs;
1440}

1443 const OpContext& op_context) const {
1444 bool found_unknown_shapes = false;
1445 const auto& op_info = op_context.op_info;
1446 // x: op_info.inputs(0)
1447 // scale: op_info.inputs(1)
1448 // offset: op_info.inputs(2)
1449 // mean: op_info.inputs(3) --> only for inference
1450 // variance: op_info.inputs(4) --> only for inference
1451 ConvolutionDimensions dims = OpDimensionsFromInputs(
1452 op_info.inputs(0).shape(), op_info, &found_unknown_shapes);
1453 const bool is_training = IsTraining(op_info);
1454
1455 int64 ops = 0;
1456 const auto rsqrt_cost = Eigen::internal::functor_traits<
1457 Eigen::internal::scalar_rsqrt_op<float>>::Cost;
1458 if (is_training) {
1459 ops = dims.iz * (dims.batch * dims.ix * dims.iy * 4 + 6 + rsqrt_cost);
1460 } else {
1461 ops = dims.batch * dims.ix * dims.iy * dims.iz * 2;
1462 }
1463
1464 const double size_nhwc =
1465 CalculateTensorSize(op_info.inputs(0), &found_unknown_shapes);
1466 const double size_c =
1467 CalculateTensorSize(op_info.inputs(1), &found_unknown_shapes);
1468 double total_input_size = 0.0;
1469 double total_internal_read_size = 0.0;
1470 double total_output_size = 0.0;
1471 if (is_training) {
1472 total_input_size = size_nhwc + size_c * 2;
1473 total_output_size = size_nhwc + size_c * 4;
1474 total_internal_read_size = size_nhwc;
1475 } else {
1476 total_input_size = size_nhwc + size_c * 4;
1477 total_output_size = size_nhwc;
1478 }
1479
1480 Costs costs = PredictOpCountBasedCost(
1481 ops, total_input_size + total_output_size + total_internal_read_size,
1482 op_info);
1483 costs.inaccurate = found_unknown_shapes;
1484 costs.max_memory = total_output_size;
1485 return costs;
