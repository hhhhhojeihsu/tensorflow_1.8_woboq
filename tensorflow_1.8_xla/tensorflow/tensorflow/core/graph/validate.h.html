19#include "tensorflow/core/framework/graph.pb.h"

23namespace tensorflow {
24namespace graph {
25
26// Returns OK if every NodeDef in `graph_def` is valid with respect to
27// its corresponding OpDef (as defined by ValidateNodeDef()) as
28// registered in `op_registry`. Also checks for deprecated ops.
29//
30// REQUIRES:
31// * `op_registry` is not nullptr.
32// * `graph_def` has default attrs filled in (see AddDefaultAttrsToGraphDef()).
33Status ValidateGraphDef(const GraphDef& graph_def,
34 const OpRegistryInterface& op_registry);
35
36// Like ValidateGraphDef() except it makes a copy of `graph_def` and calls
37// AddDefaultAttrsToGraphDef() on the copy, removing that requirement from the
38// caller.
39Status ValidateGraphDefAgainstOpRegistry(
40 const GraphDef& graph_def, const OpRegistryInterface& op_registry);
41
42// Like ValidateGraphDefAgainstOpRegistry() except it takes an OpList
43// instead of an OpRegistryInterface. Note that the OpList need not
44// have descriptions, which can be a big space savings, see
45// GetOpListForValidation() below.
46Status ValidateGraphDefAgainstOpList(const GraphDef& graph_def,
47 const OpList& op_list);
48
49// Get an OpList from `*op_registry` with all the descriptions removed.
50void GetOpListForValidation(
51 OpList* op_list, const OpRegistry& op_registry = *OpRegistry::Global());
52
53} // namespace graph
54} // namespace tensorflow

56#endif // TENSORFLOW_CORE_GRAPH_VALIDATE_H_
