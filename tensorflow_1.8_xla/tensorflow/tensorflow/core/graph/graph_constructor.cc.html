47inline bool IsMerge(const NodeDef& node_def) {
48 return node_def.op() == "Merge" || node_def.op() == "RefMerge";

51inline bool IsNextIteration(const NodeDef& node_def) {
52 return node_def.op() == "NextIteration" ||

56bool IsValidNodeName(StringPiece s, bool allow_internal_ops) {
57 using ::tensorflow::strings::Scanner;
58 return Scanner(s)
59 .One(allow_internal_ops ? Scanner::LETTER_DIGIT_DOT_UNDERSCORE
60 : Scanner::LETTER_DIGIT_DOT)
61 .Any(Scanner::LETTER_DIGIT_DASH_DOT_SLASH_UNDERSCORE)
62 .Eos()
63 .GetResult();
64}

317// This could be expensive but we don't expect to call it often, if at all (only
318// if there are multiple nodes in g_ with the same name)
319bool NodeNameInValues(const std::map<TensorId, TensorId>& input_map,
320 const StringPiece& node_name) {
321 for (auto iter = input_map.begin(); iter != input_map.end(); ++iter) {
322 if (iter->second.first == node_name) return true;
323 }
324 return false;
325}
326
327bool NodeNameInValues(const std::vector<string>& control_dependencies,
328 const StringPiece& node_name) {
329 return std::find(control_dependencies.begin(), control_dependencies.end(),
330 node_name) != control_dependencies.end();
331}

333// Adds any prefixes of `node_name` (not including the full name itself) to
334// `prefixes`.
335void AddPrefixes(StringPiece node_name,
336 std::unordered_set<StringPiece, StringPieceHasher>* prefixes) {
337 size_t idx = -1;
338 while ((idx = node_name.find('/', idx + 1)) != StringPiece::npos) {
339 prefixes->insert(node_name.substr(0, idx));
340 }

343Status GraphConstructor::EnsureNoNameCollisions() {
344 existing_nodes_.reserve(g_->num_nodes());
345 // Populate existing_nodes_ and existing_prefixes_.
346 for (Node* n : g_->nodes()) {
347 bool already_exists = !existing_nodes_.insert({n->name(), n}).second;
348 if (already_exists) {
349 if (NodeNameInValues(opts_.input_map, n->name())) {
350 return errors::InvalidArgument(
351 "cannot resolve input_map because multiple nodes exist with name '",
352 n->name(), "'");
353 }
354 if (NodeNameInValues(opts_.control_dependencies, n->name())) {
355 return errors::InvalidArgument(
356 "cannot resolve control_dependencies because multiple nodes exist "
357 "with name '",
358 n->name(), "'");
359 }
360 }
361 AddPrefixes(n->name(), &existing_prefixes_);
362 }
363 if (prefix_.empty() && opts_.importing && !opts_.uniquify_names) {
364 for (const NodeDef* n : node_defs_) {
365 const string& name = n->name();
366 if (NameExistsInGraph(name)) {
367 return errors::InvalidArgument("Node name '", name,
368 "' already exists in the Graph");
369 }
370 }
371 } else if (!prefix_.empty()) {
372 StringPiece prefix_no_slash(prefix_);
373 prefix_no_slash.remove_suffix(1);
374 if (!IsValidNodeName(prefix_no_slash, false)) {
375 return errors::InvalidArgument("Imported node name prefix '", prefix_,
376 "' would lead to invalid node names");
377 }
378 if (NameExistsInGraph(prefix_no_slash) && opts_.uniquify_prefix) {
379 prefix_ = strings::StrCat(FindUniqueName(prefix_no_slash), "/");
380 }
381 }
382 return Status::OK();
383}

385Status GraphConstructor::ValidateInputMapAndControlDependencies() {
386 for (const auto& mapping : opts_.input_map) {
387 TensorId src = mapping.first;
388 TensorId dst = mapping.second;
389 if (existing_nodes_.count(dst.first) == 0) {
390 return errors::InvalidArgument(
391 "node '", dst.first, "' in input_map does not exist in graph ",
392 "(input_map entry: ", src.ToString(), "->", dst.ToString(), ")");
393 }
394 if ((src.second == Graph::kControlSlot) !=
395 (dst.second == Graph::kControlSlot)) {
396 return errors::InvalidArgument("input_map entry ", src.ToString(), "->",
397 dst.ToString(), " between ",
398 "control edge and non-control edge");
399 }
400 }
401 for (const string& node : opts_.control_dependencies) {
402 if (existing_nodes_.count(node) == 0) {
403 return errors::InvalidArgument(
404 "node '", node,
405 "' in control_dependencies does not exist in "
406 "graph");
407 }
408 }
409 return Status::OK();
410}

412Status GraphConstructor::BuildNodeIndex() {
413 // Validate the node names and add them to gdef_nodes_ and gdef_prefixes_.
414 for (int n = 0; n < node_defs_.size(); ++n) {
415 const NodeDef& node_def = *node_defs_[n];
416 if (!IsValidNodeName(node_def.name(), opts_.allow_internal_ops)) {
417 return errors::InvalidArgument(
418 "Node '", node_def.name(),
419 "': Node name contains invalid characters");
420 }
421 if (!gdef_nodes_
422 .insert(std::make_pair(StringPiece(node_def.name()), NodeInfo(n)))
423 .second) {
424 return errors::InvalidArgument("Node '", node_def.name(),
425 "' is not unique");
426 }
427 // Validate the operation's type.
428 if (node_def.op().empty()) {
429 return errors::InvalidArgument("Node '", node_def.name(),
430 "' does not specify an operation");
431 }
432 if (opts_.expect_device_spec && node_def.device().empty()) {
433 return errors::InvalidArgument("Node '", node_def.name(),
434 "' is missing a device specification");
435 }
436 // Validate control edges at end
437 bool in_control_dependence = false;
438 for (int i = 0; i < node_def.input_size(); ++i) {
439 StringPiece input_name = node_def.input(i);
440 if (!input_name.empty() && str_util::StartsWith(input_name, "^")) {
441 in_control_dependence = true;
442 } else if (in_control_dependence) {
443 return errors::InvalidArgument(
444 "Node '", node_def.name(),
445 "': Control dependencies must come after regular dependencies");
446 }
447 }
448 // Update gdef_prefixes_.
449 AddPrefixes(node_def.name(), &gdef_prefixes_);
450 }
451 return Status::OK();
452}

454std::unordered_set<string> GetNextIterationNodes(
455 const GraphConstructor::NodeDefSlice& node_defs) {
456 std::unordered_set<string> next_iteration_nodes;
457
458 for (int n = 0; n < node_defs.size(); ++n) {
459 const NodeDef& node_def = *node_defs[n];
460 if (IsNextIteration(node_def)) {
461 next_iteration_nodes.insert(node_def.name());
462 }
463 }
464
465 return next_iteration_nodes;
466}

468Status GraphConstructor::InitFromEdges() {
469 const int num_nodes = node_defs_.size();
470 pending_count_.reserve(num_nodes);
471 outputs_.resize(num_nodes);
472 std::unordered_set<string> next_iteration_nodes_ =
473 GetNextIterationNodes(node_defs_);
474
475 // Parse the inputs for each node.
476 for (int n = 0; n < num_nodes; ++n) {
477 const NodeDef& node_def = *node_defs_[n];
478 int pending_count = node_def.input_size();
479 if (IsMerge(node_def)) {
480 // Cycles in the graph are only allowed for while loops. A while loop is
481 // identified by an edge from a NextIteration node to a Merge node. For
482 // such Merge nodes, only wait for one non-control input before
483 // considering the node ready to process in Convert().
484 int32 num_control_edges = 0;
485 bool has_loop_back_edge = false;
486 for (int i = 0; i < node_def.input_size(); ++i) {
487 StringPiece input_name(node_def.input(i));
488 if (str_util::StartsWith(input_name, "^")) {
489 num_control_edges++;
490 } else {
491 TensorId id(ParseTensorName(input_name));
492 if (next_iteration_nodes_.find(id.first.ToString()) !=
493 next_iteration_nodes_.end()) {
494 has_loop_back_edge = true;
495 }
496 }
497 }
498 if (has_loop_back_edge) {
499 pending_count = num_control_edges + 1;
500 }
501 }
502 for (int i = 0; i < node_def.input_size(); ++i) {
503 StringPiece input_name = node_def.input(i);
504 TensorId id(ParseTensorName(input_name));
505 if (opts_.input_map.count(id) == 0) {
506 // If an input is not mapped, then the input should appear in the graph
507 // being imported.
508 auto iter = gdef_nodes_.find(id.first);
509 if (iter == gdef_nodes_.end()) {
510 return errors::InvalidArgument("Node '", node_def.name(),
511 "': Unknown input node '",
512 node_def.input(i), "'");
513 }
514 outputs_[iter->second.gdef_index].push_back(n);
515 } else {
516 // This input is mapped to an existing edge. Therefore this input is
517 // as good as being already processed.
518 --pending_count;
519 DCHECK_GE(pending_count, 0);
520 }
521 }
522 if (pending_count == 0) {
523 ready_.push_back(n);
524 }
525 pending_count_.push_back(pending_count);
526 }
527 return Status::OK();
528}

530Status GraphConstructor::ValidateColocationConstraints(
531 const NodeDef& node_def) {
532 if (!opts_.validate_colocation_constraints || !opts_.importing)
533 return Status::OK();
534 const auto iter = node_def.attr().find(kColocationAttrName);
535 if (iter == node_def.attr().end()) return Status::OK();
536 for (const string& c : iter->second.list().s()) {
537 StringPiece s(c);
538 if (str_util::ConsumePrefix(&s, kColocationGroupPrefix) &&
539 gdef_nodes_.find(s) == gdef_nodes_.end()) {
540 return errors::InvalidArgument(
541 "Node '", node_def.name(),
542 "' expects to be colocated with unknown node '", s, "'");
543 }
544 }
545 return Status::OK();
546}

548Status GraphConstructor::MakeNode(const NodeDef& node_def, Node** node) {
549 // Add the node to the graph.
550 Status status;
551 *node = g_->AddNode(node_def, &status);
552 if (!status.ok()) return status;
553 if (opts_.expect_device_spec) {
554 (*node)->set_assigned_device_name(node_def.device());
555 }
556 return Status::OK();

559Status GraphConstructor::ValidateShape(Node* node) {
560 if (!opts_.importing || !opts_.validate_shape) return Status::OK();
561 TF_RETURN_IF_ERROR(refiner_->AddNode(node));
562 // For nodes with the _output_shapes attribute, override the shape.
563 std::vector<TensorShapeProto> shape_attrs;
564 const char* kAttrName = "_output_shapes";
565 if (!GetNodeAttr(node->attrs(), kAttrName, &shape_attrs).ok()) {
566 // No _output_shapes attribute, the AddNode call above was sufficient.
567 return Status::OK();
568 }
569 auto* ic = refiner_->GetContext(node);
570 DCHECK(ic != nullptr)
571 << "ShapeRefiner::AddNode() should have created the InferenceContext";
572 if (shape_attrs.size() < node->num_outputs()) {
573 return errors::InvalidArgument(
574 "Node '", node->name(), "' has ", node->num_outputs(),
575 " outputs but the ", kAttrName, " attribute specifies shapes for ",
576 shape_attrs.size(), " outputs");
577 }
578 // NOTE(skyewm): we don't raise an error here because some users depend on
579 // this behavior, even though it's unsafe.
580 // TODO(b/74619486): raise an error.
581 if (shape_attrs.size() > node->num_outputs()) {
582 LOG(WARNING) << "Node '" << node->name() << "' has " << node->num_outputs()
583 << " outputs but the " << kAttrName
584 << " attribute specifies shapes for " << shape_attrs.size()
585 << " outputs. Output shapes may be inaccurate.";
586 }
587 for (int i = 0; i < node->num_outputs(); ++i) {
588 const TensorShapeProto& p = shape_attrs[i];
589 shape_inference::ShapeHandle h;
590 Status s = ic->MakeShapeFromShapeProto(p, &h);
591 if (!s.ok()) {
592 return errors::InvalidArgument("Node '", node->name(), " has an invalid ",
593 kAttrName, " attribute (shape #", i,
594 " error:'", s.error_message(), "'");
595 }
596 s = refiner_->SetShape(node, i, h);
597 if (!s.ok()) {
598 // If the output shape is incompatible with what is inferred
599 // by the graph for a very specific whitelist of ops, then we
600 // ignore this output shape. This can happen if there is a
601 // bug in the shape function for some operation, and the
602 // serialized graph def has the incorrect shape set when
603 // running on a newer binary with the fixed shape function.
604 // This is an escape hatch that allows us to correct shape
605 // functions that are not critical to correct execution but
606 // would cause graphs to fail if imported after correcting.
607 //
608 const string& op = node->type_string();
609 const std::vector<string> whitelist = {
610 // To be removed after 2017/03/08.
611 "RandomShuffleQueue",
612 "PaddingFIFOQueue",
613 "FIFOQueue",
614 "PriorityQueue",
615 "QueueSize",
616 "Stack",
617 "Barrier",
618 "BarrierReadySize",
619 "BarrierIncompleteSize",
620 "HashTable",
621 "MutableHashTable",
622 "MutableHashTableOfTensors",
623 "Mutex",
624 "CuckooTable",
625 "IndexTable",
626 "WholeFileReader",
627 "TextLineReader",
628 "FixedLengthRecordReader",
629 "TFRecordReader",
630 "IdentityReader",
631 "RefSwitch",
632 "RefEnter",
633 "RefNextIteration",
634 "RefMerge",
635 "RefIdentity",
636 "LMDBReader",
637 // To be removed after 2017/04/24.
638 "ConditionalAccumulator",
639 "SparseConditionalAccumulator",
640 "Table",
641 };
642 if (std::find(whitelist.begin(), whitelist.end(), op) ==
643 whitelist.end()) {
644 return errors::InvalidArgument(
645 "Node '", node->name(), "' has an ", kAttrName,
646 " attribute inconsistent with the GraphDef for output #", i, ": ",
647 s.error_message());
648 }
649 }
650 }
651 node->ClearAttr(kAttrName);
652 return Status::OK();
653}

656 const OpDef* op_def;
657 TF_RETURN_IF_ERROR(g_->op_registry()->LookUpOpDef(node_def->op(), &op_def));

666void RemoveInputs(const std::vector<int>& inputs_to_remove, NodeDef* node_def,
667 std::vector<bool>* input_already_exists) {
668 // Remove 'inputs_to_remove' from 'node_def'
669 // TODO(skyewm): is there a better way to do this?
670 std::vector<string> inputs;
671 inputs.reserve(node_def->input_size());
672 for (int i = 0; i < node_def->input_size(); ++i) {
673 inputs.push_back(node_def->input(i));
674 }
675 node_def->clear_input();
676 for (int i = 0, j = 0; i < inputs.size(); ++i) {
677 if (j < inputs_to_remove.size() && i == inputs_to_remove[j]) {
678 ++j;
679 } else {
680 node_def->add_input(inputs[i]);
681 }
682 }
683 // Remove 'inputs_to_remove' from 'input_already_exists'
684 for (int idx : inputs_to_remove) {
685 input_already_exists->erase(input_already_exists->begin() + idx);
686 }
687 DCHECK_EQ(input_already_exists->size(), node_def->input_size());
688}
689
690void GraphConstructor::RemapNodeDefInputs(
691 NodeDef* node_def, std::vector<bool>* input_already_exists) {
692 DCHECK_EQ(input_already_exists->size(), node_def->input_size());
693 std::set<TensorId> control_inputs;
694 std::vector<int> inputs_to_remove;
695
696 for (int i = 0; i < node_def->input_size(); ++i) {
697 auto iter = opts_.input_map.find(ParseTensorName(node_def->input(i)));
698 if (iter == opts_.input_map.end()) continue;
699 used_input_map_keys_.insert(iter->first);
700
701 TensorId new_input = iter->second;
702 if (new_input.second == Graph::kControlSlot) {
703 // Check if we've already remapped a different input to new_input, and if
704 // so remove this input.
705 if (control_inputs.count(new_input) > 0) {
706 inputs_to_remove.push_back(i);
707 continue;
708 }
709 control_inputs.insert(new_input);
710 }
711 node_def->set_input(i, new_input.ToString());
712 (*input_already_exists)[i] = true;
713 }
714 if (!inputs_to_remove.empty()) {
715 RemoveInputs(inputs_to_remove, node_def, input_already_exists);
716 }
717}
718
719void GraphConstructor::AddControlDependencies(
720 NodeDef* node_def, std::vector<bool>* input_already_exists) {
721 // To avoid adding redundant control dependencies to every imported node, skip
722 // nodes that will inherit the dependencies from another imported node.
723 bool inherits_deps = false;
724 for (int i = 0; i < node_def->input_size(); ++i) {
725 // Assume we won't inherit dependencies from remapped inputs that already
726 // exist in the graph. Even if we're wrong, we'll only add redundant
727 // dependencies.
728 if ((*input_already_exists)[i]) continue;
729
730 // If this input is a backedge, assume we won't inherit the dependencies.
731 // TODO(skyewm): we have many redundant ParseTensorName calls. It could be
732 // worth optimizing these.
733 TensorId id(ParseTensorName(node_def->input(i)));
734 auto iter = gdef_nodes_.find(id.first);
735 DCHECK(iter != gdef_nodes_.end()) << id.first;
736 if (iter->second.node == nullptr) {
737 // Input hasn't been created yet, indicating it's a backedge.
738 continue;
739 }
740 inherits_deps = true;
741 }
742 if (inherits_deps) return;
743
744 // node_def either has no inputs or all remapped inputs, add the control
745 // dependencies
746 for (const string& control_dep : opts_.control_dependencies) {
747 string input = TensorId(control_dep, Graph::kControlSlot).ToString();
748 const protobuf::RepeatedPtrField<string>& inputs = node_def->input();
749 if (std::find(inputs.begin(), inputs.end(), input) != inputs.end()) {
750 // Control dependency already exists
751 continue;
752 }
753 node_def->add_input(input);
754 input_already_exists->push_back(true);
755 }
756}
757
758void GraphConstructor::AddPrefixToNodeDef(
759 const std::vector<bool>& input_already_exists, NodeDef* node_def) {
760 if (prefix_.empty()) return;
761 node_def->set_name(strings::StrCat(prefix_, node_def->name()));
762 // Update names of input nodes
763 for (int i = 0; i < node_def->input_size(); ++i) {
764 StringPiece input(node_def->input(i));
765 // Skip remapped inputs (which already exist in g_ and are not being
766 // imported).
767 if (input_already_exists[i]) continue;
768 if (str_util::ConsumePrefix(&input, "^")) {
769 node_def->set_input(i, strings::StrCat("^", prefix_, input));
770 } else {
771 node_def->set_input(i, strings::StrCat(prefix_, input));
772 }
773 }
774 // Update names of colocation groups
775 if (node_def->attr().find(kColocationAttrName) != node_def->attr().end()) {
776 auto* list =
777 node_def->mutable_attr()->at(kColocationAttrName).mutable_list();
778 for (int i = 0; i < list->s_size(); ++i) {
779 StringPiece v(list->s(i));
780 if (str_util::ConsumePrefix(&v, kColocationGroupPrefix)) {
781 list->set_s(i, strings::StrCat(kColocationGroupPrefix, prefix_, v));
782 }
783 }
784 }
785}
786
787void GraphConstructor::UniquifyNames(
788 const std::vector<bool>& input_already_exists, NodeDef* node_def) {
789 if (NameExistsInGraph(node_def->name())) {
790 string old_name = node_def->name();
791 node_def->set_name(FindUniqueName(node_def->name()));
792 uniquified_names_[old_name] = node_def->name();
793 // Note that we don't have to update gdef_nodes_ or gdef_prefixes_ with
794 // `name` because we guarantee the original NodeDef names are unique,
795 // meaning we won't generate this name again.
796 }
797 for (int i = 0; i < node_def->input_size(); ++i) {
798 // Skip remapped inputs (which already exist in g_ and are not being
799 // imported).
800 if (input_already_exists[i]) continue;
801 TensorId id = ParseTensorName(node_def->input(i));
802 // We require that UniquifyNames() is called on all NodeDefs in topological
803 // order. This guarantees that node_def's inputs will already be uniquified
804 // if necessary.
805 auto iter = uniquified_names_.find(id.first.ToString());
806 if (iter == uniquified_names_.end()) continue;
807 id.first = iter->second;
808 node_def->set_input(i, id.ToString());
809 }
810}
811
812void GraphConstructor::UpdateUniquifiedColocationNames() {
813 for (const auto& pair : gdef_nodes_) {
814 Node* node = pair.second.node;
815 if (node == nullptr) continue;
816 std::vector<string> coloc_values;
817 Status status =
818 GetNodeAttr(node->attrs(), kColocationAttrName, &coloc_values);
819 if (!status.ok()) continue;
820 bool updated = false;
821 for (int i = 0; i < coloc_values.size(); ++i) {
822 StringPiece val(coloc_values[i]);
823 if (str_util::ConsumePrefix(&val, kColocationGroupPrefix)) {
824 const auto& name_pair = uniquified_names_.find(val.ToString());
825 if (name_pair == uniquified_names_.end()) continue;
826 updated = true;
827 coloc_values[i] =
828 strings::StrCat(kColocationGroupPrefix, name_pair->second);
829 }
830 }
831 if (updated) {
832 node->AddAttr(kColocationAttrName, coloc_values);
833 }
834 }
835}

849string GraphConstructor::FindUniqueName(StringPiece original_name) {
850 string name = original_name.ToString();
851 int count = 0;
852 // Check that any generated names don't collide with imported NodeDefs (as
853 // well as nodes in g_).
854 while (NameExistsInGraph(name) || (count > 0 && NameExistsInGraphDef(name))) {
855 name = strings::StrCat(original_name, "_", ++count);
856 }
857 return name;

876void UpdatePendingCountAndReady(
877 const std::vector<gtl::InlinedVector<int, 4>>& outputs, int o,
878 std::vector<int>* pending_count, std::vector<int>* ready) {
879 for (size_t i = 0; i < outputs[o].size(); ++i) {

883 ready->push_back(output);
884 }

890Status GraphConstructor::Convert() {
891 // Import functions before adding nodes, since imported nodes may refer to
892 // functions
893 if (library_) {
894 TF_RETURN_IF_ERROR(g_->AddFunctionLibrary(*library_));
895 }
896
897 std::vector<InputInfo> inputs;
898 int processed = 0;
899
900 std::vector<bool> input_already_exists;
901
902 // Process the NodeDefs in topological order.
903 // (InitFromEdges() sets this up by filling in ready_ with nodes that have no
904 // inputs, pending_counts_ with the number of inputs for each node and
905 // outputs_ with the outputs of each node).
906 while (!ready_.empty()) {
907 int o = ready_.back();
908 ready_.pop_back();
909 ++processed;
910 inputs.clear();
911 bool has_data_back_edge = false;
912
913 const NodeDef& original_node_def = *node_defs_[o];
914 NodeDef imported_node_def;
915 const NodeDef* node_def;
916
917 // input_already_exists[i] is true iff the i-th input of the node we're
918 // importing refers to a preexisting node in g_ (i.e. input[i] existed prior
919 // to importing node_defs_). Conversely, input_already_exists[i] is false
920 // iff the input refers to a node in node_defs_.
921 input_already_exists.clear();
922 input_already_exists.resize(original_node_def.input_size(), false);
923
924 if (opts_.importing) {
925 if (opts_.skip_mapped_nodes) {
926 bool is_node_mapped = false;
927 TF_RETURN_IF_ERROR(
928 IsNodeFullyMapped(original_node_def, &is_node_mapped));
929 if (is_node_mapped) {
930 // Skip this node after updating pending_count_ for outputs
931 UpdatePendingCountAndReady(outputs_, o, &pending_count_, &ready_);
932 continue;
933 }
934 }
935
936 // TODO(ashankar): The line below means an additional copy of the NodeDef,
937 // which can be expensive if the NodeDef contains large tensors in it.
938 // Might make sense to change the API for ImportGraphDef to take a mutable
939 // GraphDef* and avoid the copying.
940 imported_node_def = original_node_def;
941 if (!opts_.input_map.empty()) {
942 // Note that input_already_exists can shrink here
943 RemapNodeDefInputs(&imported_node_def, &input_already_exists);
944 }
945 if (!opts_.control_dependencies.empty()) {
946 // Note that input_already_exists can grow here
947 AddControlDependencies(&imported_node_def, &input_already_exists);
948 }
949 node_def = &imported_node_def;
950 } else {
951 node_def = &original_node_def;
952 }
953
954 DCHECK_EQ(node_def->input_size(), input_already_exists.size());
955 TF_RETURN_IF_ERROR(ValidateColocationConstraints(*node_def));
956 for (int i = 0; i < node_def->input_size(); ++i) {
957 TensorId id(ParseTensorName(node_def->input(i)));
958 Node* src_node;
959 int src_index;
960
961 if (!input_already_exists[i]) {
962 // Locate input in newly-imported nodes
963 auto iter = gdef_nodes_.find(id.first);
964 DCHECK(iter != gdef_nodes_.end()) << id.first;
965 src_node = iter->second.node;
966 src_index = id.second;
967 if (src_node == nullptr) has_data_back_edge = true;
968 } else {
969 // Input refers to preexistng node in graph
970 auto iter = existing_nodes_.find(id.first);
971 DCHECK(iter != existing_nodes_.end()) << id.first;
972 src_node = iter->second;
973 src_index = id.second;
974 }
975
976 if (src_node != nullptr && src_index >= src_node->num_outputs()) {
977 return errors::InvalidArgument(
978 "Node '", node_def->name(), "': Connecting to invalid output ",
979 id.second, " of source node ", id.first, " which has ",
980 src_node->num_outputs(), " outputs");
981 }
982
983 inputs.push_back(InputInfo(id.first.ToString(), src_node, src_index));
984 }
985
986 if (has_data_back_edge && !IsMerge(*node_def)) {
987 return errors::InvalidArgument(
988 "Node '", node_def->name(),
989 "' had a back edge, but only Merge nodes can have back edges.");
990 }
991
992 Node* node;
993 if (opts_.importing) {
994 if (!prefix_.empty()) {
995 AddPrefixToNodeDef(input_already_exists, &imported_node_def);
996 }
997 // Note: no need to uniquify names if the prefix already guarantees
998 // uniqueness
999 if (opts_.uniquify_names && (prefix_.empty() || !opts_.uniquify_prefix)) {
1000 UniquifyNames(input_already_exists, &imported_node_def);
1001 }
1002 TF_RETURN_IF_ERROR(ModifyNodeDefForImport(&imported_node_def));
1003 }
1004 TF_RETURN_IF_ERROR(MakeNode(*node_def, &node));
1005 // Use original_node_def so name StringPiece remains valid
1006 gdef_nodes_[original_node_def.name()].node = node;
1007
1008 // Add edges from inputs to *node to the graph.
1009 for (size_t i = 0; i < inputs.size(); ++i) {
1010 if (inputs[i].node == nullptr) {
1011 // Record this back edge, which will be added after all nodes
1012 // are created.
1013 back_edges_.push_back(
1014 EdgeInfo(inputs[i].name, inputs[i].index, node, i));
1015 } else if (inputs[i].index == Graph::kControlSlot) {
1016 g_->AddControlEdge(inputs[i].node, node);
1017 } else {
1018 TF_RETURN_IF_ERROR(MakeEdge(inputs[i].node, inputs[i].index, node, i));
1019 }
1020 }
1021
1022 // Function shape inference is supported on an opt-in basis per
1023 // ShapeRefiner.
1024 if (refiner_->function_shape_inference_supported() ||
1025 g_->flib_def().Find(node_def->name()) == nullptr) {
1026 TF_RETURN_IF_ERROR(ValidateShape(node));
1027 }
1028
1029 // Update pending_count_ for outputs.
1030 UpdatePendingCountAndReady(outputs_, o, &pending_count_, &ready_);
1031 }
1032
1033 if (processed < node_defs_.size()) {
1034 return errors::InvalidArgument(node_defs_.size() - processed,
1035 " nodes in a cycle");
1036 }
1037
1038 return Status::OK();
1039}

1041Status GraphConstructor::AddBackEdges() {
1042 // Add the back edges after all nodes are created.
1043 for (auto e : back_edges_) {
1044 Node* src_node = gdef_nodes_[e.src_name].node;
1045 if (e.src_index == Graph::kControlSlot) {
1046 g_->AddControlEdge(src_node, e.dst_node);
1047 } else {
1048 TF_RETURN_IF_ERROR(
1049 MakeEdge(src_node, e.src_index, e.dst_node, e.dst_index));
1050 }
1051
1052 VLOG(2) << "Add back edge: " << src_node->name() << " -> "
1053 << e.dst_node->name();
1054 }
1055 return Status::OK();
1056}

1058Status GraphConstructor::UpdateVersionDef() {
1059 if (versions_ == nullptr) return Status::OK();
1060
1061 if (!opts_.importing) {
1062 g_->set_versions(*versions_);
1063 return Status::OK();
1064 }
1065 VersionDef versions = g_->versions();
1066 versions.set_producer(std::min(versions.producer(), versions_->producer()));
1067 versions.set_min_consumer(
1068 std::max(versions.min_consumer(), versions_->min_consumer()));
1069 if (versions_->bad_consumers_size() > 0) {
1070 std::set<int> bad(versions.bad_consumers().begin(),
1071 versions.bad_consumers().end());
1072 bad.insert(versions_->bad_consumers().begin(),
1073 versions_->bad_consumers().end());
1074 versions.clear_bad_consumers();
1075 for (int v : bad) {
1076 versions.add_bad_consumers(v);
1077 }
1078 }
1079 g_->set_versions(versions);
1080 return Status::OK();
1081}

1083Status GraphConstructor::PopulateReturnTensors() {
1084 if (opts_.return_tensors.empty()) return Status::OK();
1085 for (const TensorId& id : opts_.return_tensors) {
1086 auto iter = opts_.input_map.find(id);
1087 if (iter == opts_.input_map.end()) {
1088 // Locate id in imported nodes
1089 auto iter = gdef_nodes_.find(id.first);
1090 if (iter == gdef_nodes_.end()) {
1091 return errors::InvalidArgument("Requested return tensor '",
1092 id.ToString(),
1093 "' not found in graph def");
1094 }
1095 int num_outputs = iter->second.node->num_outputs();
1096 if ((id.second < 0 || id.second >= num_outputs) &&
1097 id.second != Graph::kControlSlot) {
1098 return errors::InvalidArgument("Invalid return output ", id.second,
1099 " of node '", id.first, "', which has ",
1100 num_outputs, " output(s)");
1101 }
1102 return_tensors_->push_back({iter->second.node, id.second});
1103 } else {
1104 // id was remapped to existing node
1105 TensorId remapped_id = iter->second;
1106 DCHECK_GT(existing_nodes_.count(remapped_id.first), 0);
1107 Node* node = existing_nodes_[remapped_id.first];
1108 return_tensors_->push_back({node, remapped_id.second});
1109 }
1110 }
1111 return Status::OK();
1112}

1114Status GraphConstructor::PopulateReturnNodes() {
1115 if (opts_.return_nodes.empty()) return Status::OK();
1116 for (StringPiece name : opts_.return_nodes) {
1117 auto iter = gdef_nodes_.find(name);
1118 if (iter == gdef_nodes_.end()) {
1119 return errors::InvalidArgument("Requested return node '", name,
1120 "' not found in graph def");
1121 }
1122 return_nodes_->push_back(iter->second.node);
1123 }
1124 return Status::OK();
1125}

1127Status GraphConstructor::PopulateMissingUnusedInputMapKeys() {
1128 if (missing_unused_input_map_keys_ == nullptr) return Status::OK();
1129 for (const auto& input_map_pair : opts_.input_map) {
1130 TensorId key = input_map_pair.first;
1131 if (used_input_map_keys_.count(key) > 0) continue;
1132
1133 auto pair = gdef_nodes_.find(key.first);
1134 if (pair == gdef_nodes_.end()) {
1135 // key's node doesn't exist in GraphDef
1136 missing_unused_input_map_keys_->push_back(key);
1137 continue;
1138 }
1139
1140 // Check that key's index is in bounds. Get the number of outputs from the
1141 // NodeDef, rather than the imported Node, since the Node may not exist if
1142 // opts_.skip_mapped_nodes is true.
1143 const NodeDef* node_def = node_defs_[pair->second.gdef_index];
1144 const OpDef* op_def;
1145 TF_RETURN_IF_ERROR(g_->op_registry()->LookUpOpDef(node_def->op(), &op_def));
1146 if (key.second >= op_def->output_arg_size()) {
1147 // key's index out of bounds
1148 missing_unused_input_map_keys_->push_back(key);
1149 }
1150 }
1151 return Status::OK();
1152}

1163Status GraphConstructor::MakeEdge(Node* src, int output_index, Node* dst,
1164 int input_index) {
1165 DataType src_out = src->output_type(output_index);

1172 }
1173 g_->AddEdge(src, output_index, dst, input_index);
1174 return Status::OK();

1201
1202Status ImportGraphDef(const ImportGraphDefOptions& opts, const GraphDef& gdef,
1203 Graph* g, ShapeRefiner* refiner,
1204 ImportGraphDefResults* results) {
1205 if (!opts.return_tensors.empty()) {
1206 if (results == nullptr) {
1207 return errors::InvalidArgument(
1208 "results argument to ImportGraphDef() must be non-null if "
1209 "opts.return_tensors is non-empty");
1210 }
1211 }
1212
1213 if (!opts.return_nodes.empty()) {
1214 if (opts.skip_mapped_nodes) {
1215 return errors::InvalidArgument(
1216 "Requesting return_nodes with skip_mapped_nodes set is not currently "
1217 "supported");
1218 }
1219 if (results == nullptr) {
1220 return errors::InvalidArgument(
1221 "results argument to ImportGraphDef() must be non-null if "
1222 "opts.return_nodes is non-empty");
1223 }
1224 }
1225
1226 if (results != nullptr) {
1227 if (!results->return_tensors.empty() || !results->return_nodes.empty() ||
1228 !results->missing_unused_input_map_keys.empty()) {
1229 return errors::InvalidArgument(
1230 "All fields in results argument to ImportGraphDef() must be empty.");
1231 }
1232 }
1233
1234 ShapeRefiner default_refiner(gdef.versions().producer(), g->op_registry());
1235 if (refiner == nullptr) {
1236 refiner = &default_refiner;
1237 } else {
1238 // Log a warning if we are importing a GraphDef at an older
1239 // producer version after already having added non-source/sink
1240 // nodes to the graph in the past.
1241 if (gdef.versions().producer() > 0 &&
1242 gdef.versions().producer() < refiner->graph_def_version() &&
1243 g->num_nodes() > 2) {
1244 LOG(WARNING) << "Importing a graph with a lower producer version "
1245 << gdef.versions().producer()
1246 << " into an existing graph with producer version "
1247 << refiner->graph_def_version() << ". Shape inference will "
1248 << "have run different parts of the graph with different "
1249 << "producer versions.";
1250 }
1251 }
1252
1253 // Set the graph def version of the refiner as the min of the
1254 // current value and the version from the graph we are about to
1255 // import.
1256 //
1257 // Note: to match Run() semantics, we should re-run shape inference
1258 // on the entire graph if the producer version has changed. For now
1259 // we log the warning above.
1260 refiner->set_graph_def_version(
1261 std::min(refiner->graph_def_version(), gdef.versions().producer()));
1262
1263 if (results == nullptr) {
1264 return GraphConstructor::Construct(opts, gdef.node(), &gdef.versions(),
1265 &gdef.library(), g, refiner, nullptr,
1266 nullptr, nullptr);
1267 } else {
1268 return GraphConstructor::Construct(
1269 opts, gdef.node(), &gdef.versions(), &gdef.library(), g, refiner,
1270 &results->return_tensors, &results->return_nodes,
1271 &results->missing_unused_input_map_keys);
1272 }
1273}

1275void CopyGraph(const Graph& src, Graph* dest) {
1276 for (Node* n : dest->nodes()) {
1277 CHECK(n->IsSource() || n->IsSink()) << "*dest must be empty";
1278 }
1279
1280 // Copy GraphDef versions
1281 dest->set_versions(src.versions());
1282
1283 // Copy the nodes
1284 std::unordered_map<const Node*, Node*>
1285 node_map; // "Node in src" -> "Node in *dest"
1286 node_map[src.source_node()] = dest->source_node();
1287 node_map[src.sink_node()] = dest->sink_node();
1288 for (Node* n : src.op_nodes()) {
1289 node_map[n] = dest->CopyNode(n);
1290 }
1291
1292 // Copy the edges
1293 for (const Edge* e : src.edges()) {
1294 Node* src_copy = node_map[e->src()];
1295 Node* dst_copy = node_map[e->dst()];
1296 dest->AddEdge(src_copy, e->src_output(), dst_copy, e->dst_input());
1297 }
1298}
