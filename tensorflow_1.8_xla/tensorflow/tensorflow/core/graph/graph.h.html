16// A Graph describes a set of computations that are to be
17// performed, as well as the dependencies between those
18// computations. The basic model is a DAG (directed acyclic graph) with
19// * internal nodes representing computational operations to be performed;
20// * edges represent dependencies, indicating the target may only be
21// executed once the source has completed; and
22// * predefined "source" (start) and "sink" (finish) nodes -- the source
23// should be the only node that doesn't depend on anything, and the sink
24// should be the only node that nothing depends on.
25//
26// Note: Node ids are intended to be relatively dense in the
27// 0..max_id range, but there may be gaps since ids won't be reused.
28//
29// Note: Some dependencies between operations are due to one operation
30// consuming the output of another. In fact operations can produce
31// multiple outputs and consume multiple inputs, and some
32// optimizations will care about which specific outputs are connected
33// to which specific inputs. We therefore represent data dependency
34// between output O of layer A and input I of layer B using
35// "input index" and "output index" labels per edge.
36

330// Allows for iteration of the edges of a Graph, by iterating the underlying
331// Graph.edges_ vector while skipping over null entries.
332class GraphEdgesIterable {
333 private:
334 const std::vector<Edge*>& edges_;
335
336 public:
337 explicit GraphEdgesIterable(const std::vector<Edge*>& edges)
338 : edges_(edges) {}
339
340 typedef Edge* value_type;
341
342 class const_iterator {
343 private:
344 // The underlying iterator.
345 std::vector<value_type>::const_iterator iter_;
346
347 // The end of the underlying iterator.
348 std::vector<value_type>::const_iterator end_;
349
350 // Advances iter_ until it reaches a non-null item, or reaches the end.
351 void apply_filter() {
352 while (iter_ != end_ && *iter_ == nullptr) {
353 ++iter_;
354 }
355 }
356
357 public:
358 const_iterator(std::vector<value_type>::const_iterator iter,
359 std::vector<value_type>::const_iterator end)
360 : iter_(iter), end_(end) {
361 apply_filter();
362 }
363
364 bool operator==(const const_iterator& other) const {
365 return iter_ == other.iter_;
366 }
367
368 bool operator!=(const const_iterator& other) const {
369 return iter_ != other.iter_;
370 }
371
372 // This is the prefix increment operator (++x), which is the operator
373 // used by C++ range iteration (for (x : y) ...). We intentionally do not
374 // provide a postfix increment operator.
375 const_iterator& operator++() {
376 ++iter_;
377 apply_filter();
378 return *this;
379 }
380
381 value_type operator*() { return *iter_; }
382 };
383
384 const_iterator begin() {
385 return const_iterator(edges_.begin(), edges_.end());
386 }
387 const_iterator end() { return const_iterator(edges_.end(), edges_.end()); }
388};

391class Graph {
392 public:
393 // Constructs a graph with a single SOURCE (always id kSourceId) and a
394 // single SINK (always id kSinkId) node, and an edge from SOURCE->SINK.
395 //
396 // The graph can hold ops found in registry. `registry`s lifetime must be at
397 // least that of the constructed graph's.

399
400 // Constructs a graph with a single SOURCE (always id kSourceId) and a
401 // single SINK (always id kSinkId) node, and an edge from SOURCE->SINK.
402 //
403 // The graph can hold ops found in `flib_def`. Unlike the constructor taking
404 // an OpRegistryInterface, this constructor copies the function definitions in

406 // OpRegistryInterface backing `flib_def` must still have the lifetime of the
407 // graph though.
408 explicit Graph(const FunctionLibraryDefinition& flib_def);

418 // Adds a new node to this graph, and returns it. Infers the Op and
419 // input/output types for the node. *this owns the returned instance.
420 // Returns nullptr and sets *status on error.
421 Node* AddNode(const NodeDef& node_def, Status* status);
422
423 // Copies *node, which may belong to another graph, to a new node,
424 // which is returned. Does not copy any edges. *this owns the
425 // returned instance.

427

429 // *node should not be accessed after calling this function.

433 // Adds an edge that connects the xth output of `source` to the yth input of
434 // `dest` and returns it. Does not update dest's NodeDef.

436
437 // Adds a control edge (no data flows along this edge) that connects `source`
438 // to `dest`. If `dest`s NodeDef is missing the corresponding control input,
439 // adds the control input.
440 //
441 // If such a control edge already exists and `allow_duplicates` is false, no
442 // edge is added and the function returns nullptr. Otherwise the edge is
443 // unconditionally created and returned. The NodeDef is not updated if

449
450 // Removes edge from the graph. Does not update the destination node's
451 // NodeDef.
452 // REQUIRES: The edge must exist.
453 void RemoveEdge(const Edge* edge);
454
455 // Removes control edge `edge` from the graph. Note that this also updates
456 // the corresponding NodeDef to reflect the change.

460 // edge from `new_src` to `dst` is created. The NodeDef associated with `dst`

462 Status UpdateEdge(Node* new_src, int new_src_index, Node* dst, int dst_index);
463
464 // Adds the function and gradient definitions in `fdef_lib` to this graph's op
465 // registry. Ignores duplicate functions, and returns a bad status if an
466 // imported function differs from an existing function or op with the same
467 // name.
468 Status AddFunctionLibrary(const FunctionDefLibrary& fdef_lib);
469
470 // The number of live nodes in the graph.
471 //
472 // Because nodes can be removed from the graph, num_nodes() is often
473 // smaller than num_node_ids(). If one needs to create an array of
474 // nodes indexed by node ids, num_node_ids() should be used as the

484 // The number of live edges in the graph.

492 // Serialize the nodes starting at `from_node_id` to a GraphDef.

633 // table, which consists of a vector of unique strings (device_names_), as

638 // or locating the index of an existing entry.
639 //

641 // interning table is intentionally public. This allows algorithms that
642 // frequently access this field to do so efficiently, especially for the case
643 // where the assigned_device_name of one Node is copied directly from that
644 // of another Node.

797inline gtl::iterator_range<NodeIter> Graph::op_nodes() const {
798 // Note that NodeId 0 is always valid since we don't let the source
799 // node be removed from the graph.
800 //
801 // The current implementation of Graph maintains the invariant that the
802 // first two nodes are the source and sink nodes, and all other nodes are op
803 // nodes. This method (op_nodes()) relies on this invariant.
804 NodeIter begin(this, 0);
805 NodeIter end(this, num_node_ids());
806 if (begin != end) {
807 ++begin;
808 }
809 if (begin != end) {
810 ++begin;
811 }
812 return gtl::make_range(begin, end);
813}
