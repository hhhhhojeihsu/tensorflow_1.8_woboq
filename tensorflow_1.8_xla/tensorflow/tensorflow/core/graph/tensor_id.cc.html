25TensorId ParseTensorName(const string& name) {
26 return ParseTensorName(StringPiece(name.data(), name.size()));
27}
28
29TensorId ParseTensorName(StringPiece name) {
30 // Parse either a name, ^name, or name:digits. To do so, we go backwards from
31 // the end of the string, skipping over a run of digits. If we hit a ':'
32 // character, then we know we are in the 'name:digits' regime. Otherwise, we
33 // see if the name starts with '^', indicating a control edge. If we find
34 // neither ':' nor '^' characters, the output index is implicitly 0, and the
35 // whole name string forms the first part of the tensor name.
36 const char* base = name.data();
37 const char* p = base + name.size() - 1;
38 unsigned int index = 0;
39 unsigned int mul = 1;
40 while (p > base && (*p >= '0' && *p <= '9')) {
41 index += ((*p - '0') * mul);
42 mul *= 10;
43 p--;
44 }
45 TensorId id;
46 if (p > base && *p == ':' && mul > 1) {
47 id.first = StringPiece(base, p - base);
48 id.second = index;
49 } else if (str_util::StartsWith(name, "^")) {
50 // Control edge
51 id.first = StringPiece(base + 1);
52 id.second = Graph::kControlSlot;
53 } else {
54 id.first = name;
55 id.second = 0;
56 }
57 return id;
58}
