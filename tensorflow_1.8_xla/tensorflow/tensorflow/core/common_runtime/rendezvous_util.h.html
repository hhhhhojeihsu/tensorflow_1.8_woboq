15#ifndef TENSORFLOW_CORE_COMMON_RUNTIME_RENDEZVOUS_UTIL_H_
16#define TENSORFLOW_CORE_COMMON_RUNTIME_RENDEZVOUS_UTIL_H_
17
18#include <map>
19
20#include "tensorflow/core/framework/rendezvous.h"
21
22namespace tensorflow {
23
24typedef std::map<string, Tensor> NamedTensors;
25typedef std::function<void(const Status&)> StatusCallback;
26
27// Uses `rendezvous` to send tensors in `tensors_to_send`. `device_context`
28// should be the DeviceContext associated with the source of the tensors.
29// `alloc_attrs` contains information about how the `tensors_to_send` are
30// allocated. `alloc_attrs` should either be {} or should match the length of
31// `keys`.
32Status SendTensorsToRendezvous(
33 Rendezvous* rendezvous, DeviceContext* device_context,
34 const std::vector<AllocatorAttributes>& alloc_attrs,
35 const std::vector<string>& keys, gtl::ArraySlice<Tensor> tensors_to_send);
36
37// Uses `rendezvous` to obtain tensors. `device_context` should be the
38// DeviceContext associated with the receiving device. `alloc_attrs` contains
39// information as how to store the received tensors. Should be {} or match the
40// length of `keys`.
41void RecvOutputsFromRendezvousAsync(
42 Rendezvous* rendezvous, DeviceContext* device_context,
43 const std::vector<AllocatorAttributes>& alloc_attrs,
44 const std::vector<string>& keys, std::vector<Tensor>* received_tensors,
45 const StatusCallback& done);
46
47Status RecvOutputsFromRendezvous(Rendezvous* rendezvous, NamedTensors* out,
48 const Rendezvous::Args& args);
49
50} // namespace tensorflow
