16#ifndef TENSORFLOW_COMMON_RUNTIME_GPU_DEVICE_CONTEXT_H_
17#define TENSORFLOW_COMMON_RUNTIME_GPU_DEVICE_CONTEXT_H_
18
19#include "tensorflow/core/common_runtime/device.h"
20#include "tensorflow/core/framework/device_base.h"
21
22namespace perftools {
23namespace gputools {
24class Stream;
25} // namespace gputools
26} // namespace perftools
27
28namespace tensorflow {
29
30namespace gpu = ::perftools::gputools;
31
32class GPUDeviceContext : public DeviceContext {
33 public:
34 // Does not take ownership of streams.
35 GPUDeviceContext(int stream_id, gpu::Stream* stream,
36 gpu::Stream* host_to_device_stream,
37 gpu::Stream* device_to_host_stream,
38 gpu::Stream* device_to_device_stream)
39 : stream_id_(stream_id),
40 stream_(stream),
41 host_to_device_stream_(host_to_device_stream),
42 device_to_host_stream_(device_to_host_stream),
43 device_to_device_stream_(device_to_device_stream) {}
44
45 ~GPUDeviceContext() override {}
46
47 gpu::Stream* stream() const override { return stream_; }
48 gpu::Stream* host_to_device_stream() const { return host_to_device_stream_; }
49 gpu::Stream* device_to_host_stream() const { return device_to_host_stream_; }
50 gpu::Stream* device_to_device_stream() const {
51 return device_to_device_stream_;
52 }
53 int stream_id() const { return stream_id_; }
54
55 void CopyCPUTensorToDevice(const Tensor* cpu_tensor, Device* device,
56 Tensor* device_tensor,
57 StatusCallback done) const override;
58
59 void CopyDeviceTensorToCPU(const Tensor* device_tensor, StringPiece edge_name,
60 Device* device, Tensor* cpu_tensor,
61 StatusCallback done) override;
62
63 void MaintainLifetimeOnStream(
64 const Tensor* t, perftools::gputools::Stream* stream) const override {}
65
66 private:
67 int stream_id_;
68 // The default primary stream to use for this context.
69 // All the memory belongs to this stream.
70 gpu::Stream* stream_;
71 // The stream to use for copy data from host into GPU.
72 gpu::Stream* host_to_device_stream_;
73 // The stream to use for copy data from GPU to host.
74 gpu::Stream* device_to_host_stream_;
75 // The stream to use for copy data between GPU.
76 gpu::Stream* device_to_device_stream_;
77};
78
79} // namespace tensorflow
80
81#endif // TENSORFLOW_COMMON_RUNTIME_GPU_DEVICE_CONTEXT_H_
