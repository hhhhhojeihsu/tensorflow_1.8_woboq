66CollectiveExecutorMgr::~CollectiveExecutorMgr() {
67 for (auto iter : executor_table_) {
68 iter.second->Unref();
69 }
70}
71
72CollectiveExecutor* CollectiveExecutorMgr::FindOrCreate(int64 step_id) {
73 CollectiveExecutor* ce = nullptr;
74 {
75 mutex_lock l(exec_mu_);
76 auto it = executor_table_.find(step_id);
77 if (it != executor_table_.end()) {
78 ce = it->second;
79 } else {
80 ce = new DummyCollectiveExecutor(this);
81 executor_table_[step_id] = ce;
82 }
83 ce->Ref();
84 }
85 return ce;
86}
87
88void CollectiveExecutorMgr::Cleanup(int64 step_id) {
89 CollectiveExecutor* ce = nullptr;
90 {
91 mutex_lock l(exec_mu_);
92 auto it = executor_table_.find(step_id);
93 if (it != executor_table_.end()) {
94 ce = it->second;
95 executor_table_.erase(it);
96 }
97 }
98 if (ce) ce->Unref();
99}
