15#ifndef TENSORFLOW_CORE_COMMON_RUNTIME_PROCESS_FUNCTION_LIBRARY_RUNTIME_H_

18#include <unordered_map>
19
20#include "tensorflow/core/common_runtime/device_mgr.h"
21#include "tensorflow/core/framework/function.h"
22#include "tensorflow/core/protobuf/config.pb.h"

24namespace tensorflow {
25
26// A class that stores all the FunctionLibraryRuntime objects, one per device.
27class ProcessFunctionLibraryRuntime {
28 public:
29 // Creates FunctionLibraryRuntime objects for each device in the provided
30 // DeviceMgr. Caller needs to make sure that device_mgr, lib_def and parent
31 // (if provided) outlive this object.
32 ProcessFunctionLibraryRuntime(
33 const DeviceMgr* device_mgr, Env* env, int graph_def_version,
34 const FunctionLibraryDefinition* lib_def,
35 const OptimizerOptions& optimizer_options,
36 thread::ThreadPool* thread_pool = nullptr,
37 DistributedFunctionLibraryRuntime* parent = nullptr);
38
39 // With `custom_kernel_creator`.
40 ProcessFunctionLibraryRuntime(const DeviceMgr* device_mgr, Env* env,
41 int graph_def_version,
42 const FunctionLibraryDefinition* lib_def,
43 const OptimizerOptions& optimizer_options,
44 CustomKernelCreator custom_kernel_creator,
45 thread::ThreadPool* thread_pool,
46 DistributedFunctionLibraryRuntime* parent);
47
48 // Sends `tensors_to_send` from `source_device` to `target_device` using
49 // `rendezvous`. `key_prefix` is used as a prefix for the keys sent to the
50 // Rendezvous. `device_context` should be the DeviceContext of the device
51 // doing the sending. `alloc_attrs` should either be empty or be the size of
52 // `tensors_to_send` and indicates how the input tensors are allocated. Method
53 // takes references on each of the `tensors_to_send`. Method doesn't block.
54 static Status SendTensors(const string& source_device,
55 const string& target_device,
56 const string& key_prefix, int64 src_incarnation,
57 gtl::ArraySlice<Tensor> tensors_to_send,
58 DeviceContext* device_context,
59 const std::vector<AllocatorAttributes>& alloc_attrs,
60 Rendezvous* rendezvous);
61
62 typedef std::function<void(const Status&)> StatusCallback;
63
64 // Receives `received_tensors` from `target_device` (originally sent from
65 // `source_device`) using `rendezvous`. Uses `key_prefix` to construct the
66 // keys to be retrieved. `device_context` should be for the device receiving
67 // the tensors. `alloc_attrs` indicates how to allocate the received
68 // tensors and should either be empty or `num_tensors` in size. Method doesn't
69 // block and calls `done` when `num_tensors` are fetched.
70 static void ReceiveTensorsAsync(
71 const string& source_device, const string& target_device,
72 const string& key_prefix, int64 src_incarnation, int64 num_tensors,
73 DeviceContext* device_context,
74 const std::vector<AllocatorAttributes>& alloc_attrs,
75 Rendezvous* rendezvous, std::vector<Tensor>* received_tensors,
76 const StatusCallback& done);
77
78 static const char kDefaultFLRDevice[];
79 // Returns the FunctionLibraryRuntime for the corresponding device_name.
80 FunctionLibraryRuntime* GetFLR(const string& device_name) const;
81
82 // Returns the device incarnation for the given device_name.
83 Status GetDeviceIncarnation(const string& device_name, int64* incarnation);
84
85 // For a given canonicalized key signature of the function instantiated
86 // on device `device_name` and a `local_handle`, creates a handle and returns
87 // that value. Uses core/common_runtime/framework/function.h::Canonicalize
88 // to canonicalize the function signature.
89 FunctionLibraryRuntime::Handle AddHandle(
90 const string& function_key, const string& device_name,
91 FunctionLibraryRuntime::LocalHandle local_handle);
92
93 // Returns a handle if found for the given key, else returns kInvalidHandle.
94 FunctionLibraryRuntime::Handle GetHandle(const string& function_key) const;
95
96 // For the given handle instantiated on device `device_name` returns the local
97 // index of instantiation of that function. If the function was not
98 // instantiated on `device_name` returns kInvalidLocalHandle.
99 FunctionLibraryRuntime::LocalHandle GetHandleOnDevice(
100 const string& device_name, FunctionLibraryRuntime::Handle handle);
101
102 // Returns true if function with handle `handle` was instantiated on device
103 // `device_name`.
104 bool IsInstantiatedOnDevice(const string& device_name,
105 FunctionLibraryRuntime::Handle handle);
106
107 // Instantiates the function. See framework/function.h for more details.
108 // Allows for function_name to be instantiated on different devices
109 // as specified in attrs.
110 Status Instantiate(const string& function_name, AttrSlice attrs,
111 const FunctionLibraryRuntime::InstantiateOptions& options,
112 FunctionLibraryRuntime::Handle* handle);
113
114 // Delegates to the local FLR that owns state corresponding to `handle` and
115 // tells it to release it. If the `handle` isnt' needed at all, the local FLR
116 // might call RemoveHandle on this to get rid of the state owned by the Proc
117 // FLR.
118 Status ReleaseHandle(FunctionLibraryRuntime::Handle handle);
119
120 // Runs the function with given `handle`. Function could have been
121 // instantiated on any device. More details in framework/function.h
122 void Run(const FunctionLibraryRuntime::Options& opts,
123 FunctionLibraryRuntime::Handle handle, gtl::ArraySlice<Tensor> args,
124 std::vector<Tensor>* rets,
125 FunctionLibraryRuntime::DoneCallback done);
126
127 private:
128 // For a given device_name, returns a DeviceContext for copying
129 // tensors to/from the device.
130 Status GetDeviceContext(const string& device_name,
131 DeviceContext** device_context);
132
133 // Looks up the information for the given `handle` and returns the name
134 // of the device where the function is registered.
135 string GetDeviceName(FunctionLibraryRuntime::Handle handle);
136
137 // Removes handle from the state owned by this object.
138 Status RemoveHandle(FunctionLibraryRuntime::Handle handle);
139
140 Status Clone(Env* env, int graph_def_version,
141 const OptimizerOptions& optimizer_options,
142 CustomKernelCreator custom_kernel_creator,
143 std::unique_ptr<FunctionLibraryDefinition>* out_lib_def,
144 std::unique_ptr<ProcessFunctionLibraryRuntime>* out_pflr);
145
146 friend class FunctionLibraryRuntimeImpl;
147
148 mutable mutex mu_;
149
150 class FunctionData {
151 public:
152 FunctionData(const string& target_device,
153 FunctionLibraryRuntime::LocalHandle local_handle)
154 : target_device_(target_device), local_handle_(local_handle) {}
155
156 string target_device() { return target_device_; }
157
158 FunctionLibraryRuntime::LocalHandle local_handle() {
159 mutex_lock l(mu_);
160 return local_handle_;
161 }
162
163 // Initializes the FunctionData object by potentially making an Initialize
164 // call to the DistributedFunctionLibraryRuntime.
165 Status DistributedInit(
166 DistributedFunctionLibraryRuntime* parent, const string& function_name,
167 const FunctionLibraryDefinition& lib_def, AttrSlice attrs,
168 const FunctionLibraryRuntime::InstantiateOptions& options);
169
170 private:
171 mutex mu_;
172
173 const string target_device_;
174 FunctionLibraryRuntime::LocalHandle local_handle_ GUARDED_BY(mu_);
175 bool init_started_ GUARDED_BY(mu_) = false;
176 Status init_result_ GUARDED_BY(mu_);
177 Notification init_done_;
178 };
179
180 const DeviceMgr* const device_mgr_;
181 const FunctionLibraryDefinition* lib_def_;
182 thread::ThreadPool* default_thread_pool_;
183 // Holds all the function invocations here.
184 std::unordered_map<string, FunctionLibraryRuntime::Handle> table_
185 GUARDED_BY(mu_);
186 std::unordered_map<FunctionLibraryRuntime::Handle,
187 std::unique_ptr<FunctionData>>
188 function_data_ GUARDED_BY(mu_);
189 std::unordered_map<Device*, std::unique_ptr<FunctionLibraryRuntime>> flr_map_;
190 int next_handle_ GUARDED_BY(mu_);
191 DistributedFunctionLibraryRuntime* const parent_;
192};
193
194} // namespace tensorflow
195
