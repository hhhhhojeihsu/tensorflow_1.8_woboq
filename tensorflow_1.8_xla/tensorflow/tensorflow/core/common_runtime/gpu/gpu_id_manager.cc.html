27namespace tensorflow {
28namespace {
29// Manages the map between TfGpuId and CUDA GPU id.
30class TfToCudaGpuIdMap {
31 public:
32 static TfToCudaGpuIdMap* singleton() {
33 static auto* id_map = new TfToCudaGpuIdMap;
34 return id_map;
35 }
36
37 void InsertOrDie(TfGpuId tf_gpu_id, CudaGpuId cuda_gpu_id)
38 LOCKS_EXCLUDED(mu_) {
39 std::pair<IdMapType::iterator, bool> result;
40 {
41 mutex_lock lock(mu_);
42 result = id_map_.insert({tf_gpu_id.value(), cuda_gpu_id.value()});
43 }
44 if (!result.second) {
45 CHECK_EQ(cuda_gpu_id.value(), result.first->second)
46 << "Mapping the same TfGpuId to a different CUDA GPU id."
47 << " TfGpuId: " << tf_gpu_id
48 << " Existing mapped CUDA GPU id: " << result.first->second
49 << " CUDA GPU id being tried to map to: " << cuda_gpu_id;
50 }
51 }
52
53 CudaGpuId FindOrDie(TfGpuId tf_gpu_id) const LOCKS_EXCLUDED(mu_) {
54 mutex_lock lock(mu_);
55 return FindOrDieLocked(tf_gpu_id);
56 }
57
58 bool Find(TfGpuId tf_gpu_id, CudaGpuId* cuda_gpu_id) const
59 LOCKS_EXCLUDED(mu_) {
60 mutex_lock lock(mu_);
61 if (id_map_.count(tf_gpu_id.value()) == 0) return false;
62 *cuda_gpu_id = FindOrDieLocked(tf_gpu_id);
63 return true;
64 }
65
66 private:
67 TfToCudaGpuIdMap() = default;
68
69 CudaGpuId FindOrDieLocked(TfGpuId tf_gpu_id) const
70 EXCLUSIVE_LOCKS_REQUIRED(mu_) {
71 auto result = id_map_.find(tf_gpu_id.value());
72 CHECK(result != id_map_.end())
73 << "Could not find the mapping for TfGpuId: " << tf_gpu_id;
74 return CudaGpuId(result->second);
75 }
76
77 void TestOnlyReset() LOCKS_EXCLUDED(mu_) {
78 mutex_lock lock(mu_);
79 id_map_.clear();
80 }
81
82 using IdMapType = std::unordered_map<int32, int32>;
83 mutable mutex mu_;
84 IdMapType id_map_ GUARDED_BY(mu_);
85
86 friend class ::tensorflow::GpuIdManager;
87 TF_DISALLOW_COPY_AND_ASSIGN(TfToCudaGpuIdMap);
88};
89} // namespace
90
91void GpuIdManager::InsertTfCudaGpuIdPair(TfGpuId tf_gpu_id,
92 CudaGpuId cuda_gpu_id) {
93 TfToCudaGpuIdMap::singleton()->InsertOrDie(tf_gpu_id, cuda_gpu_id);
94}
95
96Status GpuIdManager::TfToCudaGpuId(TfGpuId tf_gpu_id, CudaGpuId* cuda_gpu_id) {
97 if (TfToCudaGpuIdMap::singleton()->Find(tf_gpu_id, cuda_gpu_id)) {
98 return Status::OK();
99 }
100 return errors::NotFound("TF GPU device with id ", tf_gpu_id.value(),
101 " was not registered");
102}
103
104CudaGpuId GpuIdManager::TfToCudaGpuId(TfGpuId tf_gpu_id) {
105 return TfToCudaGpuIdMap::singleton()->FindOrDie(tf_gpu_id);
106}
107
108void GpuIdManager::TestOnlyReset() {
109 TfToCudaGpuIdMap::singleton()->TestOnlyReset();
110}
