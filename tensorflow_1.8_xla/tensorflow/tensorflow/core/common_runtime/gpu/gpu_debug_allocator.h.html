31// An allocator that wraps a GPU allocator and adds debugging
32// functionality that verifies that users do not write outside their
33// allocated memory.
34class GPUDebugAllocator : public VisitableAllocator {
35 public:
36 explicit GPUDebugAllocator(VisitableAllocator* allocator,
37 CudaGpuId cuda_gpu_id);
38 ~GPUDebugAllocator() override;
39 string Name() override { return "gpu_debug"; }
40 void* AllocateRaw(size_t alignment, size_t num_bytes) override;
41 void DeallocateRaw(void* ptr) override;
42 void AddAllocVisitor(Visitor visitor) override;
43 void AddFreeVisitor(Visitor visitor) override;
44 bool TracksAllocationSizes() override;
45 size_t RequestedSize(const void* ptr) override;
46 size_t AllocatedSize(const void* ptr) override;
47 int64 AllocationId(const void* ptr) override;
48 void GetStats(AllocatorStats* stats) override;
49 void ClearStats() override;
50
51 // For testing.
52 bool CheckHeader(void* ptr);
53 bool CheckFooter(void* ptr);
54
55 private:
56 VisitableAllocator* base_allocator_ = nullptr; // owned
57
58 perftools::gputools::StreamExecutor* stream_exec_; // Not owned.
59
60 TF_DISALLOW_COPY_AND_ASSIGN(GPUDebugAllocator);
61};
62
63// An allocator that wraps a GPU allocator and resets the memory on
64// allocation and free to 'NaN', helping to identify cases where the
65// user forgets to initialize the memory.
66class GPUNanResetAllocator : public VisitableAllocator {
67 public:
68 explicit GPUNanResetAllocator(VisitableAllocator* allocator,
69 CudaGpuId cuda_gpu_id);
70 ~GPUNanResetAllocator() override;
71 string Name() override { return "gpu_nan_reset"; }
72 void* AllocateRaw(size_t alignment, size_t num_bytes) override;
73 void DeallocateRaw(void* ptr) override;
74 void AddAllocVisitor(Visitor visitor) override;
75 void AddFreeVisitor(Visitor visitor) override;
76 size_t RequestedSize(const void* ptr) override;
77 size_t AllocatedSize(const void* ptr) override;
78 void GetStats(AllocatorStats* stats) override;
79 void ClearStats() override;
80
81 private:
82 VisitableAllocator* base_allocator_ = nullptr; // owned
83
84 perftools::gputools::StreamExecutor* stream_exec_; // Not owned.
85
86 TF_DISALLOW_COPY_AND_ASSIGN(GPUNanResetAllocator);
87};
