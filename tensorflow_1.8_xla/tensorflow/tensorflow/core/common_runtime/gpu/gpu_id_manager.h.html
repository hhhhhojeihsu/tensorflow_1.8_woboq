16#ifndef TENSORFLOW_CORE_COMMON_RUNTIME_GPU_GPU_ID_MANAGER_H_
17#define TENSORFLOW_CORE_COMMON_RUNTIME_GPU_GPU_ID_MANAGER_H_
18
19#include "tensorflow/core/common_runtime/gpu/gpu_id.h"
20#include "tensorflow/core/lib/core/status.h"
21
22namespace tensorflow {
23
24// Class that maintains a map from TfGpuId to CudaGpuId, and manages the
25// translation between them.
26class GpuIdManager {
27 public:
28 // Adds a mapping from tf_gpu_id to cuda_gpu_id.
29 static void InsertTfCudaGpuIdPair(TfGpuId tf_gpu_id, CudaGpuId cuda_gpu_id);
30
31 // Gets the cuda_gpu_id associated with tf_gpu_id. Returns OK if found.
32 static Status TfToCudaGpuId(TfGpuId tf_gpu_id, CudaGpuId* cuda_gpu_id);
33 // Similar to the above version, but returns the result, and checks fail if
34 // no result is found.
35 static CudaGpuId TfToCudaGpuId(TfGpuId tf_gpu_id);
36
37 // Clears the map. Used in unit tests only.
38 static void TestOnlyReset();
39};
40
41} // namespace tensorflow

43#endif // TENSORFLOW_CORE_COMMON_RUNTIME_GPU_GPU_ID_MANAGER_H_
