30// Implements ParamResolverInterface for a single-task context.
31// It also implements the functionality necessary to serve as the
32// group leader for param resolution in a multi-task context.
33class CollectiveParamResolverLocal : public ParamResolverInterface {
34 public:
35 CollectiveParamResolverLocal(const DeviceMgr* dev_mgr,
36 DeviceResolverInterface* dev_resolver,
37 const string& task_name);
38
39 ~CollectiveParamResolverLocal() override {}
40
41 void CompleteParamsAsync(const string& device, CollectiveParams* cp,
42 CancellationManager* cancel_mgr,
43 const StatusCallback& done) override;
44
45 void CompleteGroupAsync(const CompleteGroupRequest* request,
46 CompleteGroupResponse* response,
47 CancellationManager* cancel_mgr,
48 const StatusCallback& done) override;
49
50 void CompleteInstanceAsync(const CompleteInstanceRequest* request,
51 CompleteInstanceResponse* response,
52 CancellationManager* cancel_mgr,
53 const StatusCallback& done) override;
54
55 protected:
56 // Used to complete/verify CollGroup.
57 struct GroupRec {
58 CollGroupParams group;
59 mutex mu;
60 Status status GUARDED_BY(mu);
61 std::set<string> device_set GUARDED_BY(mu);
62 std::vector<string> device_list GUARDED_BY(mu);
63 std::set<string> task_set GUARDED_BY(mu);
64 std::vector<string> task_list GUARDED_BY(mu);
65 std::vector<StatusCallback> waiting GUARDED_BY(mu);
66 };
67
68 // Finds the GroupRec that corresponds to cp->group_key.
69 // Also populates cp->group from that group_rec.
70 // Will wait until GroupRec is fully populated or an error arises before
71 // calling done. Callback GroupRec* arg is only valid if status is ok.
72 // Ownership of GroupRec stays with this object and does not pass to the
73 // callback.
74 typedef std::function<void(const Status& s, GroupRec* gr)> GroupRecCallback;
75 void CompleteGroupLocal(const string& device, CollectiveParams* cp,
76 const GroupRecCallback& done)
77 LOCKS_EXCLUDED(group_mu_);
78
79 // Used to complete/verify CollInstance.
80 struct InstanceRec;
81 typedef std::function<void(InstanceRec*)> IRConsumer;
82 struct InstanceRec {
83 // This structure has two mutexes so that a possibly long
84 // initialization can be done without holding the instance_mu_
85 // table lock the whole time (which can cause an excessive number
86 // of threads to block on it), and because the compiler may not
87 // permit mutex locks to be taken in more than one order.
88 //
89 // out_mu guards access to most of the fields.
90 // in_mu guards access to a queue of comsumer callbacks wanting to
91 // read the fields guarded by out_mu.
92 //
93 // The in_mu should be locked only while holding instance_mu_; the
94 // out_mu should be locked only while not holding
95 // instance_mu_.
96 //
97 // When is_init is false (the initial value) any potential user
98 // other than the creator should queue a callback on init_waiters.
99 // As soon as the shared member of this structure is fully
100 // initialized is_init will be set true and those callbacks will
101 // be invoked.
102 //
103 // Once inserted in the table this structure will never be replaced
104 // so users can capture the pointer while holding instance_mu_,
105 // drop that lock, then take a lock on out_mu before
106 // reading/modifying its values.
107 mutex in_mu;
108 bool is_init GUARDED_BY(in_mu);
109 std::vector<IRConsumer> init_waiters GUARDED_BY(in_mu);
110
111 // Values to be shared by all instances, constant after initialization.
112 mutex out_mu;
113 CollectiveParams shared GUARDED_BY(out_mu);
114 // If an error occurs during initialization this structure stays in
115 // the table with a non-OK status. Purging the table and restarting
116 // needs to be done at a higher level.
117 Status status GUARDED_BY(out_mu);
118
119 // These fields are used to count the instances that have called
120 // in and become known while resolving broadcast source identity.
121 int source_rank GUARDED_BY(out_mu);
122 int known_count GUARDED_BY(out_mu);
123 std::vector<bool> known GUARDED_BY(out_mu);
124 std::vector<IRConsumer> known_waiters GUARDED_BY(out_mu);
125
126 InstanceRec() : is_init(false), source_rank(-1), known_count(0) {}
127 };
128
129 // Find the InstanceRec with the same instance_key as cp. If it doesn't
130 // already exist, create and initialize from gr and cp.
131 //
132 // Precondition: *gr must be a complete GroupRec, i.e. the value set
133 // by CompleteGroupLocal. *cp must be populated with all the fields
134 // required by InitInstanceSharedParams. Ownership of InstanceRec stays
135 // with this object and does not pass to the callback.
136 typedef std::function<void(const Status& s, InstanceRec* ir)>
137 InstanceRecCallback;
138 void FindInstanceRec(GroupRec* gr, CollectiveParams* cp,
139 const InstanceRecCallback& done)
140 LOCKS_EXCLUDED(instance_mu_, gr->mu, group_mu_);
141
142 // Populate *ir with device membership from gr, then initialize to be specific
143 // to cp->instance_key, i.e. order the devices and tasks.
144 //
145 // Preconditions:
146 // cp is populated with all DeviceLocalities
147 Status InitInstanceSharedParams(GroupRec* gr, const CollectiveParams* cp,
148 InstanceRec* ir)
149 EXCLUSIVE_LOCKS_REQUIRED(ir->out_mu) LOCKS_EXCLUDED(gr->mu);
150
151 // Establishes the final order of ir->shared.instance.device_names and
152 // ir->shared.instance.task_names by considering localities of all devices.
153 void CompleteDefaultRanking(GroupRec* gr, const CollectiveParams* cp,
154 InstanceRec* ir,
155 const std::vector<DeviceLocality>& localities)
156 EXCLUSIVE_LOCKS_REQUIRED(ir->out_mu);
157
158 // Finish populating *cp.
159 // Precondition: *gr has been fully populated by CompleteGroupLocal.
160 void CompleteInstanceLocal(const string& device, GroupRec* gr,
161 CollectiveParams* cp, bool is_source,
162 const StatusCallback& done)
163 LOCKS_EXCLUDED(instance_mu_, gr->mu, group_mu_);
164
165 // Finish populating *cp from fully initialized *ir.
166 // Precondition: *gr and *ir are fully populated.
167 void CompleteInstanceFromInitializedIRec(const string& device, GroupRec* gr,
168 CollectiveParams* cp,
169 InstanceRec* ir, bool is_source,
170 const StatusCallback& done)
171 LOCKS_EXCLUDED(ir->out_mu);
172
173 // Complete source data for a broadcast instance.
174 // Precondition: *cp has complete group data and default_rank.
175 void CompleteInstanceSource(InstanceRec* ir, CollectiveParams* cp,
176 bool is_source, const IRConsumer& f)
177 LOCKS_EXCLUDED(ir->out_mu);
178
179 // If cp.device_names contains only devices local to this process
180 // populates *localities, else returns an error.
181 Status GetLocalDeviceLocalities(const CollectiveParams& cp,
182 std::vector<DeviceLocality>* localities);
183
184 // Sets CollTaskParams.is_local and CollectiveParams.default_rank.
185 // Precondition: cp->device_names is fully populated and in final order.
186 void CompleteTaskIsLocal(const string& task_name, CollectiveParams* cp);
187
188 // Sets cp->instance_default_rank according to location of device in
189 // current ordering of cp->instance.device_names.
190 void SetDefaultRank(const string& device, CollectiveParams* cp);
191
192 // Helper to grab status under lock, invoke callback out of lock.
193 void CallbackWithStatus(const InstanceRecCallback& done, InstanceRec* irec)
194 LOCKS_EXCLUDED(irec->out_mu);
195
196 const DeviceMgr* dev_mgr_;
197 DeviceResolverInterface* dev_resolver_;
198 string task_name_;
199 mutex group_mu_;
200 gtl::FlatMap<int32, std::unique_ptr<GroupRec>> group_table_
201 GUARDED_BY(group_mu_);
202 mutex instance_mu_;
203 gtl::FlatMap<int32, std::unique_ptr<InstanceRec>> instance_table_
204 GUARDED_BY(instance_mu_);
205};
