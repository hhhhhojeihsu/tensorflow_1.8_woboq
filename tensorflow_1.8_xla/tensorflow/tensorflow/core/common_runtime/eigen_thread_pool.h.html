16#ifndef TENSORFLOW_COMMON_RUNTIME_EIGEN_THREAD_POOL_H_
17#define TENSORFLOW_COMMON_RUNTIME_EIGEN_THREAD_POOL_H_
18
19#define EIGEN_USE_THREADS
20
21#include "third_party/eigen3/unsupported/Eigen/CXX11/Tensor"
22#include "tensorflow/core/lib/core/threadpool.h"
23
24namespace tensorflow {
25
26class EigenThreadPoolWrapper : public Eigen::ThreadPoolInterface {
27 public:
28 explicit EigenThreadPoolWrapper(thread::ThreadPool* pool) : pool_(pool) {}
29 ~EigenThreadPoolWrapper() override {}
30
31 void Schedule(std::function<void()> fn) override {
32 pool_->Schedule(std::move(fn));
33 }
34 int NumThreads() const override { return pool_->NumThreads(); }
35 int CurrentThreadId() const override { return pool_->CurrentThreadId(); }
36
37 private:
38 thread::ThreadPool* pool_ = nullptr;
39};
40
41} // namespace tensorflow
