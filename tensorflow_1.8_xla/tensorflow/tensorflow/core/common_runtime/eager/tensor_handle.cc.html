47bool TensorHandle::IsReady() {
48 if (node_id == 0) return true;
49 mutex_lock l(ctx_mutex_);
50 return is_ready_;
51}
52
53Status TensorHandle::WaitReady() {
54 if (node_id == 0) return Status::OK();
55 EagerExecutor* executor = nullptr;
56 {
57 mutex_lock l(ctx_mutex_);
58 if (is_ready_) return Status::OK();
59 executor = ctx_->Executor();
60 }
61 return executor->WaitFor(node_id);
62}
63
64Status TensorHandle::Tensor(const tensorflow::Tensor** t) {
65 TF_RETURN_IF_ERROR(WaitReady());
66 DCHECK(IsReady());
67 *t = &tensor_;
68 return Status::OK();
69}

71Status TensorHandle::Device(tensorflow::Device** d) {

73 DCHECK(IsReady());
74 *d = device_;

78Status TensorHandle::OpDevice(tensorflow::Device** d) {
79 TF_RETURN_IF_ERROR(WaitReady());

85Status TensorHandle::TensorAndDevice(const tensorflow::Tensor** tensor,
86 tensorflow::Device** device,
87 tensorflow::Device** op_device) {
88 TF_RETURN_IF_ERROR(WaitReady());
89 DCHECK(IsReady());
90 *tensor = &tensor_;
91 *device = device_;
92 *op_device = op_device_;
93 return Status::OK();
94}

96void TensorHandle::SetTensorAndDevice(const tensorflow::Tensor& tensor,
97 tensorflow::Device* device,
98 tensorflow::Device* op_device) {
99 mutex_lock l(ctx_mutex_);
100 DCHECK(node_id > 0 && !is_ready_)
101 << "SetTensorAndDevice should be only called "
102 << "on non-ready handles.";
103 is_ready_ = true;
104 tensor_ = tensor;
105 device_ = device;
106 op_device_ = op_device;
107}

109Status TensorHandle::CopyToDevice(EagerContext* ctx, tensorflow::Device* dstd,
110 TensorHandle** output) {
111 const tensorflow::Tensor* src = nullptr;
112 tensorflow::Device* srcd = nullptr;
113 // TODO(agarwal): src_opd is unused. Perhaps allow TensorAndDevice to accept
114 // nullptr.
115 tensorflow::Device* src_opd = nullptr;
116 TF_RETURN_IF_ERROR(TensorAndDevice(&src, &srcd, &src_opd));
117 if (srcd == nullptr) srcd = ctx->HostCPU();
118 bool is_same_device = (srcd == dstd) || (srcd->name() == dstd->name());
119 const bool dst_cpu = dstd->tensorflow_gpu_device_info() == nullptr;
120 const bool src_cpu = srcd->tensorflow_gpu_device_info() == nullptr;
121 // both_on_cpu can be true and yet is_same_device is false, if one of src/dst
122 // has device type XLA_CPU, and the other CPU.
123 const bool both_on_cpu = src_cpu && dst_cpu;
124 if (is_same_device || both_on_cpu) {
125 dstd = dst_cpu ? nullptr : dstd;
126 *output = new tensorflow::TensorHandle(*src, dstd, dstd, ctx);
127 return tensorflow::Status::OK();
128 }
129 if (!dst_cpu && (src->dtype() != tensorflow::DT_VARIANT &&
130 !tensorflow::DataTypeCanUseMemcpy(src->dtype()))) {
131 return tensorflow::errors::InvalidArgument(
132 "Can't copy Tensor with type ",
133 tensorflow::DataTypeString(src->dtype()), " to device ", dstd->name(),
134 ".");
135 }
136 tensorflow::AllocatorAttributes attr;
137 if (src->dtype() == tensorflow::DT_VARIANT) {
138 attr.set_on_host(true);
139 }
140 tensorflow::Tensor dst(dstd->GetAllocator(attr), src->dtype(), src->shape());
141 if (src->shape().num_elements() == 0) {
142 dstd = dst_cpu ? nullptr : dstd;
143 *output = new tensorflow::TensorHandle(dst, dstd, dstd, ctx);
144 return tensorflow::Status::OK();
145 }
146 tensorflow::DeviceContext* src_device_context = nullptr;
147 if (!src_cpu) {
148 src_device_context = srcd->tensorflow_gpu_device_info()->default_context;
149 }
150 tensorflow::DeviceContext* dst_device_context = nullptr;
151 if (!dst_cpu) {
152 dst_device_context = dstd->tensorflow_gpu_device_info()->default_context;
153 }
154 // TODO(ashankar): The Sync() call below may be more aggressive than
155 // necessary. It is based on knowledge of implementation details - that
156 // GPU devices are implemented using 3 streams - one for host->device copies,
157 // one for device->host copies and one for sending operations to the GPU.
158 // With that setup, Sync()ing across all 3 streams should be sufficient
159 // but more than necessary (since it waits for operations that might have
160 // nothing to do with this tensor to complete).
161 TF_RETURN_IF_ERROR(srcd->Sync());
162 tensorflow::Notification n;
163 tensorflow::Status status;
164 tensorflow::CopyTensor::ViaDMA("copy", src_device_context, dst_device_context,
165 srcd, dstd, tensorflow::AllocatorAttributes(),
166 tensorflow::AllocatorAttributes(), src, &dst,
167 [&status, &n](const tensorflow::Status& s) {
168 status = s;
169 n.Notify();
170 });
171 n.WaitForNotification();
172 if (status.ok()) {
173 dstd = dst_cpu ? nullptr : dstd;
174 *output = new tensorflow::TensorHandle(dst, dstd, dstd, ctx);
175 }
176 return status;
177}
