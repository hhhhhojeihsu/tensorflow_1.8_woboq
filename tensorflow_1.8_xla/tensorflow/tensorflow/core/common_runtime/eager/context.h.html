45// Note: there's a copy enum in eager/c_api.h. It should be kept in sync.
46enum ContextDevicePlacementPolicy {
47 // Running operations with input tensors on the wrong device will fail.
48 DEVICE_PLACEMENT_EXPLICIT = 0,
49 // Copy the tensor to the right device but log a warning.
50 DEVICE_PLACEMENT_WARN = 1,
51 // Silently copy the tensor, which has a performance cost since the operation
52 // will be blocked till the copy completes. This is the default policy.
53 DEVICE_PLACEMENT_SILENT = 2,
54 // Default placement policy which silently copies int32 tensors but not other
55 // dtypes.
56 DEVICE_PLACEMENT_SILENT_FOR_INT32 = 3,
57};
58
59class EagerContext {
60 public:
61 explicit EagerContext(const SessionOptions& opts,
62 ContextDevicePlacementPolicy default_policy, bool async,
63 std::unique_ptr<DeviceMgr> device_mgr,
64 Rendezvous* rendezvous);
65
66 ~EagerContext();
67
68 // Returns the function library runtime for the given device.
69 FunctionLibraryRuntime* func_lib(Device* d) const {
70 return pflr_->GetFLR(d->name());
71 }
72
73 // True if running in asynchronous mode.
74 bool Async() const;
75
76 EagerExecutor* Executor() { return &executor_; }
77
78 // Sets whether this thread should run in synchronous or asynchronous mode.
79 Status SetAsyncForThread(bool async);
80
81 // TODO(apassos) make this return a constant reference
82 gtl::FlatMap<string, Device*, StringPieceHasher>* device_map() {
83 return &devices_map_;
84 }
85
86 // TODO(apassos) make this return a constant reference
87 std::vector<Device*>* devices() { return &devices_; }
88
89 // Clears the kernel caches.
90 void ClearCaches();
91
92 // Sets the device placement policy for the current thread.
93 void SetThreadLocalDevicePlacementPolicy(ContextDevicePlacementPolicy policy);
94
95 // Returns the device placement policy for the current thread.
96 ContextDevicePlacementPolicy GetDevicePlacementPolicy();
97
98 Status AsyncWait() { return executor_.WaitForAllPendingNodes(); }
99
100 Status GetStatus() { return executor_.status(); }
101
102 void ClearAsyncError() { executor_.ClearError(); }
103
104 bool FindFunctionByName(const string& name);
105
106 Status FindFunctionOpData(const string& name,
107 const tensorflow::OpRegistrationData** op_data);
108
109 const FunctionDef* FindFunctionDef(const string& name);
110
111 Status FindDeviceByName(const string& name, Device** result);
112
113 Device* HostCPU() { return devices_[0]; }
114
115 uint64 NextId() { return executor_.NextId(); }
116
117 void ExecutorAdd(EagerNode* node) { executor_.Add(node); }
118
119 Status AddFunctionDef(const FunctionDef& fdef);
120
121 KernelAndDevice* GetCachedKernel(Fprint128 cache_key);
122
123 void AddKernelToCache(Fprint128 cache_key, KernelAndDevice* kernel);
124
125 bool LogDevicePlacement() { return log_device_placement_; }
126
127 Rendezvous* GetRendezvous() { return rendezvous_; }
128
129 mutex* FunctionsMu() { return &functions_mu_; }
130
131 tensorflow::DeviceMgr* device_mgr() { return device_manager_.get(); }
132
133 // TODO(apassos) remove the need for this
134 void ReleaseDeviceMgr() { device_manager_.release(); }
135
136 // TODO(apassos) clean up RunMetadata storage.
137 mutex* MetadataMu() { return &metadata_mu_; }
138 bool ShouldStoreMetadata() { return should_store_metadata_.load(); }
139 void SetShouldStoreMetadata(bool value);
140 RunMetadata* RunMetadataProto() { return &run_metadata_; }
141
142 FunctionLibraryDefinition* FuncLibDef() { return &func_lib_def_; }
143
144 private:
145 const ContextDevicePlacementPolicy policy_;
146
147 // Note: we cannot use C++11 thread_local here as there is no concept of a
148 // thread-local-object-local variable in C++11.
149 mutex policy_map_mu_;
150 std::unordered_map<std::thread::id, ContextDevicePlacementPolicy>
151 thread_local_policies_ GUARDED_BY(policy_map_mu_);
152
153 std::unique_ptr<DeviceMgr> device_manager_;
154 // Devices owned by device_manager
155 std::vector<Device*> devices_;
156 // All devices are not owned.
157 gtl::FlatMap<string, Device*, StringPieceHasher> devices_map_;
158 Rendezvous* const rendezvous_;
159
160 mutex functions_mu_;
161 FunctionLibraryDefinition func_lib_def_ GUARDED_BY(functions_mu_){
162 OpRegistry::Global(), {}};
163
164 std::unique_ptr<thread::ThreadPool> thread_pool_;
165
166 // One FunctionLibraryRuntime per device.
167 // func_libs[i] is the FunctionLibraryRuntime corresponding to
168 // session->devices[i].
169 const std::unique_ptr<ProcessFunctionLibraryRuntime> pflr_;
170
171 mutex cache_mu_;
172 std::unordered_map<Fprint128, KernelAndDevice*, Fprint128Hasher> kernel_cache_
173 GUARDED_BY(cache_mu_);
174
175 // Whether we should compute RunMetadata.
176 std::atomic<bool> should_store_metadata_{false};
177 mutex metadata_mu_;
178 RunMetadata run_metadata_ GUARDED_BY(metadata_mu_);
179 const bool log_device_placement_;
180 // EagerExecutor for async execution.
181 EagerExecutor executor_;
182
183 // True if the default value for execution mode is async. Note that this value
184 // can be overridden per thread based on `thread_local_async` overrides.
185 const bool async_default_;
186 mutable mutex async_map_mu_;
187 std::unordered_map<std::thread::id, bool> thread_local_async_
188 GUARDED_BY(async_map_mu_);
189};
