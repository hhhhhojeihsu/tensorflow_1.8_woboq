35void CollectiveParamResolverLocal::CompleteGroupLocal(
36 const string& device, CollectiveParams* cp, const GroupRecCallback& done) {
37 VLOG(1) << "CompleteGroupLocal " << cp << ": " << cp->ToString();
38 std::vector<StatusCallback> to_be_called;
39 GroupRec* gr = nullptr;
40 {
41 mutex_lock l(group_mu_);
42 auto it = group_table_.find(cp->group.group_key);
43 if (it == group_table_.end()) {
44 gr = new GroupRec;
45 gr->group.group_key = cp->group.group_key;
46 gr->group.group_size = cp->group.group_size;
47 gr->group.device_type = cp->group.device_type;
48 group_table_[gr->group.group_key].reset(gr);
49 VLOG(2) << "New group_key=" << gr->group.group_key
50 << " group_size=" << gr->group.group_size;
51 } else {
52 gr = it->second.get();
53 }
54 }
55 Status status;
56 {
57 mutex_lock gr_lock(gr->mu);
58 if (!gr->device_set.empty()) {
59 // Check for consistency with existing GroupRec.
60 if (cp->group.device_type != gr->group.device_type) {
61 status = errors::Internal(
62 "Collective Op ", cp->name, " is assigned to device ", device,
63 " with type ", cp->group.device_type.type_string(),
64 " and group_key ", cp->group.group_key, " but that group has type ",
65 gr->group.device_type.type_string());
66 } else if (cp->group.group_size != gr->group.group_size) {
67 status = errors::Internal(
68 "Collective Op ", cp->name, " has group_size ",
69 cp->group.group_size, " and group_key", cp->group.group_key,
70 " but that group has size ", gr->group.group_size);
71 }
72 }
73 if (status.ok()) {
74 // Insert device if not already present.
75 auto it = gr->device_set.find(device);
76 if (it == gr->device_set.end()) {
77 if (gr->device_set.size() == gr->group.group_size) {
78 // The group is already full.
79 status = errors::Internal(
80 "Collective Op ", cp->name, " is assigned to device ", device,
81 " and group_key ", cp->group.group_key,
82 " but that group doesn't contain that device.");
83 } else {
84 // This is a new device that has not yet joined the group.
85 gr->device_set.insert(device);
86 gr->device_list.push_back(device);
87 DeviceNameUtils::ParsedName parsed_device;
88 DeviceNameUtils::ParseFullName(device, &parsed_device);
89 string task_name = strings::StrCat("/job:", parsed_device.job,
90 "/replica:", parsed_device.replica,
91 "/task:", parsed_device.task);
92 gr->task_set.insert(task_name);
93 gr->task_list.push_back(task_name);
94 gr->group.num_tasks = static_cast<int32>(gr->task_set.size());
95 VLOG(1) << "group_key=" << gr->group.group_key
96 << " group_size=" << gr->group.group_size
97 << " dev_set=" << gr->device_set.size();
98 }
99 }
100 }
101
102 if (status.ok()) {
103 // If the group is not yet complete, queue to wait for it.
104 VLOG(2) << "group_size " << gr->group.group_size << " set size "
105 << gr->device_set.size() << " gr " << gr;
106
107 if (gr->device_set.size() < gr->group.group_size) {
108 gr->waiting.push_back(std::bind(done, std::placeholders::_1, gr));
109 return;
110 }
111 CHECK_EQ(gr->device_set.size(), gr->group.group_size);
112 if (!gr->waiting.empty()) {
113 std::swap(to_be_called, gr->waiting);
114 }
115 }
116 }
117 done(status, gr);
118 for (int i = 0; i < to_be_called.size(); ++i) {
119 to_be_called[i](Status::OK());
120 }
121}

125struct DevRec {
126 string task;
127 string device;
128 int original_rank;
129 int local_rank;
130 int global_rank;
131 const DeviceLocality* locality;
132};
133typedef std::unordered_map<string, DevRec> TaskDeviceMap;
134typedef std::unordered_map<string, TaskDeviceMap> GlobalDeviceMap;
135
136// Create a populated GlobalDeviceMap from CollInstanceParams and localities.
137GlobalDeviceMap BuildDevRecs(const CollInstanceParams& ip,
138 const std::vector<DeviceLocality>& localities) {
139 GlobalDeviceMap gdm;
140 CHECK_EQ(ip.device_names.size(), ip.task_names.size());
141 CHECK_EQ(ip.device_names.size(), localities.size());
142 for (int i = 0; i < ip.device_names.size(); ++i) {
143 TaskDeviceMap& tdm = gdm[ip.task_names[i]];
144 DevRec* dr = &tdm[ip.device_names[i]];
145 dr->task = ip.task_names[i];
146 dr->device = ip.device_names[i];
147 dr->original_rank = i;
148 dr->local_rank = 0; // Will be populated later by OrderTaskDeviceMap.
149 dr->global_rank = 0; // Will be populated later by EstablishGlobalRank.
150 dr->locality = &localities[i];
151 }
152 return gdm;
153}
154
155void OrderTaskDeviceMap(TaskDeviceMap* tdm) {
156 CHECK_GT(tdm->size(), 0); // Should never be called with 0 devices
157 int least_rank = -1;
158 string next_device;
159 std::set<string> selected;
160 // Starting device is one with the least initial rank.
161 for (const auto& it : *tdm) {
162 if (least_rank < 0 || it.second.original_rank < least_rank) {
163 least_rank = it.second.original_rank;
164 next_device = it.second.device;
165 }
166 }
167 CHECK_GE(least_rank, 0);
168 DeviceNameUtils::ParsedName parsed_name;
169 CHECK(DeviceNameUtils::ParseFullName(next_device, &parsed_name));
170 // NOTE: InterconnectLink has only a device_id, nothing more, so for
171 // the time being if there's more than one device at a task we
172 // assume they're all GPUs.
173
174 int next_rank = 0;
175 while (true) {
176 selected.insert(next_device);
177 DevRec* dr = &(*tdm)[next_device];
178 dr->local_rank = next_rank;
179 ++next_rank;
180 if (selected.size() == tdm->size()) {
181 break;
182 }
183 // For the present time we assume Locality links only cover GPUs.
184 // For multiple CPUs, just take them in order.
185 const InterconnectLink* best_link = nullptr;
186 if (parsed_name.type == "GPU") {
187 for (const InterconnectLink& il : dr->locality->links().link()) {
188 parsed_name.id = il.device_id();
189 string endpoint_device =
190 DeviceNameUtils::ParsedNameToString(parsed_name);
191 if (selected.find(endpoint_device) != selected.end()) {
192 continue;
193 }
194 if (best_link == nullptr || il.strength() > best_link->strength()) {
195 best_link = &il;
196 }
197 }
198 }
199 if (best_link != nullptr) {
200 // Follow the best edge
201 parsed_name.id = best_link->device_id();
202 next_device = DeviceNameUtils::ParsedNameToString(parsed_name);
203 } else {
204 // No good edges, alas. Pick the lowest initial rank among remaining
205 // devices.
206 least_rank = -1;
207 for (const auto& it : *tdm) {
208 if (selected.find(it.second.device) != selected.end()) {
209 continue;
210 }
211 if (least_rank < 0 || it.second.original_rank < least_rank) {
212 least_rank = it.second.original_rank;
213 next_device = it.second.device;
214 }
215 }
216 CHECK_GE(least_rank, 0);
217 }
218 }
219}

221// The first time a shared CollectiveParams is established for a
222// shared set of instances we compute a good rank order for all the
223// devices in the group, that is appropriate for a ring algorithm.
224// This order need not be the same across different instance groups
225// sharing the same device group where there is more than one good
226// order.
227GlobalDeviceMap EstablishGlobalRank(
228 CollectiveParams* cp, const std::vector<DeviceLocality>& localities) {
229 VLOG(1) << "EstablishGlobalRank";
230 GlobalDeviceMap gdm = BuildDevRecs(cp->instance, localities);
231 for (auto& iter : gdm) {
232 TaskDeviceMap& tdm = iter.second;
233 OrderTaskDeviceMap(&tdm);
234 }
235 // Connect the global rank order by the order in which tasks first appear.
236 std::set<string> ordered_tasks;
237 int next_rank = 0;
238 for (int i = 0; i < cp->instance.task_names.size(); ++i) {
239 const string& task_name = cp->instance.task_names[i];
240 if (ordered_tasks.find(task_name) != ordered_tasks.end()) {
241 continue;
242 }
243 ordered_tasks.insert(task_name);
244 TaskDeviceMap* tdm = &gdm[task_name];
245 for (auto& it : *tdm) {
246 it.second.global_rank = it.second.local_rank + next_rank;
247 }
248 next_rank += tdm->size();
249 }
250 return gdm;
251}

253// Sort cp->instance.device_names lexicographically, but do by first
254// computing a reordering permutation so we can keep cp->instance.task_names
255// in corresponding order.
256void SortDevicesAndTasks(CollectiveParams* cp) {
257 VLOG(1) << "SortDevicesAndTasks " << cp << " instance " << &cp->instance;
258 CHECK(cp);
259 CHECK_EQ(cp->group.group_size, cp->instance.device_names.size());
260 CHECK_EQ(cp->group.group_size, cp->instance.task_names.size());
261 std::vector<int> perm(cp->group.group_size);
262 // TODO(tucker): substitute std::iota when the windows build supports it.
263 // std::iota(perm.begin(), perm.end(), 0);
264 for (int i = 0; i < perm.size(); ++i) {
265 perm[i] = i;
266 }
267 std::sort(perm.begin(), perm.end(), [cp](const int& a, const int& b) {
268 return cp->instance.device_names[a] < cp->instance.device_names[b];
269 });
270 std::vector<string> new_devs;
271 std::vector<string> new_tasks;
272 new_devs.reserve(cp->group.group_size);
273 new_tasks.reserve(cp->group.group_size);
274 for (int pi : perm) {
275 new_devs.push_back(cp->instance.device_names[pi]);
276 new_tasks.push_back(cp->instance.task_names[pi]);
277 }
278 cp->instance.device_names = std::move(new_devs);
279 cp->instance.task_names = std::move(new_tasks);
280 VLOG(1) << "Modified device_names on " << cp;
281}

283// Establish the requested number of subdivision permutations based on the
284// ring order implicit in the device order.
285void GenerateSubdivPerms(const string& device, int source_rank,
286 CollectiveParams* cp) {
287 CHECK_GT(cp->instance.impl_details.subdiv_offsets.size(), 0);
288 cp->instance.impl_details.subdiv_permutations.resize(
289 cp->instance.impl_details.subdiv_offsets.size());
290 // Each subdiv permutation is a ring formed by rotating each
291 // single-task subsequence of devices by an offset. This makes most
292 // sense when each task has the same number of devices but we can't
293 // depend on that being the case so we'll compute something that
294 // works in any case.
295
296 // Start by counting the devices in each task.
297 // Precondition: device_names must be sorted so that all devices in
298 // the same task are adjacent.
299 VLOG(2) << "Sorted task names: "
300 << str_util::Join(cp->instance.task_names, ", ");
301 std::vector<int> dev_per_task;
302 const string* prior_task_name = &cp->instance.task_names[0];
303 int dev_count = 1;
304 for (int di = 1; di < cp->group.group_size; ++di) {
305 if (cp->instance.task_names[di] != *prior_task_name) {
306 dev_per_task.push_back(dev_count);
307 dev_count = 1;
308 prior_task_name = &cp->instance.task_names[di];
309 } else {
310 ++dev_count;
311 }
312 }
313 dev_per_task.push_back(dev_count);
314 CHECK_EQ(cp->group.num_tasks, dev_per_task.size());
315
316 // Generate a ring permutation for each requested offset.
317 CHECK_GT(cp->instance.impl_details.subdiv_offsets.size(), 0);
318 VLOG(2) << "Setting up perms for cp " << cp << " subdiv_permutations "
319 << &cp->instance.impl_details.subdiv_permutations;
320 cp->instance.impl_details.subdiv_permutations.resize(
321 cp->instance.impl_details.subdiv_offsets.size());
322 cp->subdiv_rank.resize(cp->instance.impl_details.subdiv_offsets.size(), -1);
323 for (int sdi = 0; sdi < cp->instance.impl_details.subdiv_offsets.size();
324 ++sdi) {
325 std::vector<int>& perm = cp->instance.impl_details.subdiv_permutations[sdi];
326 CHECK_EQ(perm.size(), 0);
327 int offset = cp->instance.impl_details.subdiv_offsets[sdi];
328 int prior_dev_count = 0;
329 for (int ti = 0; ti < cp->group.num_tasks; ++ti) {
330 for (int di = 0; di < dev_per_task[ti]; ++di) {
331 int offset_di = (di + offset) % dev_per_task[ti];
332 int permuted_di = prior_dev_count + offset_di;
333 perm.push_back(permuted_di);
334 if (cp->instance.device_names[prior_dev_count + di] == device) {
335 CHECK_EQ(prior_dev_count + di, cp->default_rank);
336 cp->subdiv_rank[sdi] = permuted_di;
337 }
338 }
339 prior_dev_count += dev_per_task[ti];
340 }
341 CHECK_EQ(cp->group.group_size, perm.size());
342 }
343
344 if (cp->instance.type == BROADCAST_COLLECTIVE) {
345 CHECK_GE(source_rank, 0);
346 cp->subdiv_source_rank.resize(
347 cp->instance.impl_details.subdiv_offsets.size(), -1);
348 for (int sdi = 0; sdi < cp->subdiv_source_rank.size(); ++sdi) {
349 for (int j = 0; j < cp->group.group_size; ++j) {
350 if (cp->instance.impl_details.subdiv_permutations[sdi][j] ==
351 source_rank) {
352 cp->subdiv_source_rank[sdi] = j;
353 break;
354 }
355 }
356 CHECK_GE(cp->subdiv_source_rank[sdi], 0);
357 }
358 }
359
360 if (VLOG_IS_ON(1)) {
361 // Log the computed ring order for each subdiv.
362 string buf;
363 for (int sdi = 0;
364 sdi < cp->instance.impl_details.subdiv_permutations.size(); ++sdi) {
365 buf = strings::StrCat("Subdiv ", sdi, " device order:\n");
366 for (int di = 0;
367 di < cp->instance.impl_details.subdiv_permutations[sdi].size();
368 ++di) {
369 int idx = cp->instance.impl_details.subdiv_permutations[sdi][di];
370 strings::StrAppend(&buf, cp->instance.device_names[idx], "\n");
371 }
372 strings::StrAppend(&buf, " subdiv_offsets: ");
373 for (auto o : cp->instance.impl_details.subdiv_offsets)
374 strings::StrAppend(&buf, o, " ");
375 strings::StrAppend(&buf, " SubdivRank: ");
376 for (auto d : cp->subdiv_rank) strings::StrAppend(&buf, d, " ");
377 VLOG(1) << buf;
378 }
379 }
380}

403Status CollectiveParamResolverLocal::InitInstanceSharedParams(
404 GroupRec* gr, const CollectiveParams* cp, InstanceRec* ir) {
405 VLOG(1) << "InitInstanceSharedParams " << ir;
406 ir->shared.instance = cp->instance;
407 {
408 mutex_lock gl(gr->mu);
409 ir->shared.group = gr->group;
410 ir->shared.instance.device_names.assign(gr->device_list.begin(),
411 gr->device_list.end());
412 ir->shared.instance.task_names.assign(gr->task_list.begin(),
413 gr->task_list.end());
414 VLOG(2) << "Initialized names for instance: "
415 << ir->shared.instance.ToString();
416 }
417 ir->shared.default_rank = -1;
418
419 // Sort devce_names lexicographcally, keeping task_names in
420 // corresponding order.
421 SortDevicesAndTasks(&ir->shared);
422
423 // Get Locality data for all devices.
424
425 // Set is_local and task_names in *shared prior to invoking
426 // GetDeviceLocalitiesAsync. In a distributed context this function can be
427 // called by a derived class, some of the devices may be non-local and
428 // GetDeviceLocalitiesAsync will use those fields to launch RPCs.
429 CompleteTaskIsLocal(task_name_, &ir->shared);
430 std::vector<DeviceLocality> localities;
431 Notification note;
432 Status status;
433 dev_resolver_->GetDeviceLocalitiesAsync(ir->shared.instance, &localities,
434 [&note, &status](const Status& s) {
435 status = s;
436 note.Notify();
437 });
438 note.WaitForNotification();
439 if (status.ok()) {
440 CompleteDefaultRanking(gr, cp, ir, localities);
441 }
442 return status;
443}

445void CollectiveParamResolverLocal::CompleteDefaultRanking(
446 GroupRec* gr, const CollectiveParams* cp, InstanceRec* ir,
447 const std::vector<DeviceLocality>& localities) {
448 // Establish an instance-specific default rank order for devices
449 // based on localities. This rank order should be a good ring
450 // order, if possible.
451 GlobalDeviceMap gdm = EstablishGlobalRank(&ir->shared, localities);
452 // Reflect the new global ranking on shared
453 size_t num_devices = ir->shared.group.group_size;
454 std::vector<string> new_device_names(num_devices, "");
455 std::vector<string> new_task_names(num_devices, "");
456 for (const auto& git : gdm) {
457 const TaskDeviceMap& tdm = git.second;
458 for (const auto& tit : tdm) {
459 const DevRec& dr = tit.second;
460 new_device_names[dr.global_rank] =
461 ir->shared.instance.device_names[dr.original_rank];
462 new_task_names[dr.global_rank] =
463 ir->shared.instance.task_names[dr.original_rank];
464 }
465 }
466
467 ir->shared.instance.device_names = new_device_names;
468 ir->shared.instance.task_names = new_task_names;
469 if (VLOG_IS_ON(2)) {
470 string buf;
471 for (const auto& d : cp->instance.device_names)
472 strings::StrAppend(&buf, "\n", d);
473 VLOG(2) << "Optimized device order for " << ir->shared.name << ": " << buf;
474 }
475}

487void CollectiveParamResolverLocal::FindInstanceRec(
488 GroupRec* gr, CollectiveParams* cp, const InstanceRecCallback& done) {
489 InstanceRec* irec = nullptr;
490 bool exit_outside_locks = false;
491 {
492 mutex_lock l(instance_mu_);
493 auto it = instance_table_.find(cp->instance.instance_key);
494 if (it != instance_table_.end()) {
495 irec = it->second.get();
496 {
497 mutex_lock l(irec->in_mu);
498 if (irec->is_init) {
499 exit_outside_locks = true;
500 } else {
501 irec->init_waiters.push_back([this, gr, cp, done](InstanceRec* irec) {
502 CallbackWithStatus(done, irec);
503 });
504 return;
505 }
506 }
507 } else {
508 // Create new InstanceRec.
509 irec = new InstanceRec;
510 instance_table_[cp->instance.instance_key].reset(irec);
511 }
512 }
513 if (exit_outside_locks) {
514 CallbackWithStatus(done, irec);
515 return;
516 }
517 // Initialize the new InstanceRec while holding out_mu.
518 {
519 mutex_lock il(irec->out_mu);
520 irec->known.resize(cp->group.group_size, false);
521 irec->status = InitInstanceSharedParams(gr, cp, irec);
522 }
523 // Prepare to invoke any waiters that accumlated during initialization.
524 std::vector<IRConsumer> init_waiters;
525 {
526 mutex_lock tl(instance_mu_);
527 {
528 mutex_lock l(irec->in_mu);
529 irec->is_init = true;
530 if (!irec->init_waiters.empty()) {
531 std::swap(init_waiters, irec->init_waiters);
532 }
533 }
534 }
535 CallbackWithStatus(done, irec);
536 for (auto& f : init_waiters) {
537 f(irec);
538 }
539}

565void CollectiveParamResolverLocal::CompleteInstanceLocal(
566 const string& device, GroupRec* gr, CollectiveParams* cp, bool is_source,
567 const StatusCallback& done) {
568 VLOG(1) << "CompleteInstanceLocal " << device
569 << " instance_key: " << cp->instance.instance_key << " gr " << gr;
570
571 // Populate the group portion of *cp from *gr. Most of it should already
572 // match.
573 DCHECK_EQ(cp->group.group_key, gr->group.group_key);
574 DCHECK_EQ(cp->group.group_size, gr->group.group_size);
575 DCHECK_EQ(cp->group.device_type, gr->group.device_type);
576 cp->group = gr->group;
577
578 // Get the shared InstanceRec for this instance.
579 FindInstanceRec(gr, cp,
580 [this, device, gr, cp, is_source, done](const Status& s,
581 InstanceRec* ir) {
582 if (s.ok()) {
583 CompleteInstanceFromInitializedIRec(device, gr, cp, ir,
584 is_source, done);
585 } else {
586 done(s);
587 }
588 });
589}

591void CollectiveParamResolverLocal::CompleteInstanceFromInitializedIRec(
592 const string& device, GroupRec* gr, CollectiveParams* cp, InstanceRec* ir,
593 bool is_source, const StatusCallback& done) {
594 // Populate the fields common across instance.
595 {
596 mutex_lock l(ir->out_mu);
597 // custom operator= does a deep copy.
598 cp->instance = ir->shared.instance;
599 }
600 // Populate the fields common across task, also default_rank.
601 SetDefaultRank(device, cp);
602 CompleteTaskIsLocal(task_name_, cp);
603 // If broadcast, may need to wait for source discovery.
604 if (cp->instance.type == BROADCAST_COLLECTIVE) {
605 CompleteInstanceSource(ir, cp, is_source,
606 [this, ir, device, cp, done](InstanceRec* irec) {
607 CHECK_EQ(ir, irec);
608 Status s;
609 int source_rank;
610 {
611 mutex_lock l(irec->out_mu);
612 s = irec->status;
613 source_rank = ir->source_rank;
614 }
615 if (s.ok()) {
616 GenerateSubdivPerms(device, source_rank, cp);
617 }
618 done(s);
619 });
620 return;
621 } else {
622 GenerateSubdivPerms(device, 0, cp);
623 }
624 done(Status::OK());
625}

627void CollectiveParamResolverLocal::CompleteInstanceSource(InstanceRec* ir,
628 CollectiveParams* cp,
629 bool is_source,
630 const IRConsumer& f) {
631 std::vector<IRConsumer> ready_waiters;
632 {
633 mutex_lock l(ir->out_mu);
634 CHECK_EQ(cp->group.group_size, ir->known.size());
635 CHECK_GE(cp->default_rank, 0);
636 if (!ir->known[cp->default_rank]) {
637 ir->known[cp->default_rank] = true;
638 ++ir->known_count;
639 if (is_source) {
640 if (ir->source_rank >= 0) {
641 ir->status = errors::Internal("Instance ", cp->instance.instance_key,
642 " already has source ", ir->source_rank,
643 ", recevied second claim from ",
644 cp->default_rank);
645 } else {
646 ir->source_rank = cp->default_rank;
647 }
648 }
649 }
650 if (ir->known_count < ir->shared.group.group_size) {
651 ir->known_waiters.push_back(f);
652 return;
653 }
654 CHECK_EQ(ir->known_count, ir->shared.group.group_size);
655 CHECK_GE(ir->source_rank, 0);
656 if (!ir->known_waiters.empty()) {
657 ready_waiters = std::move(ir->known_waiters);
658 }
659 }
660 f(ir);
661 for (auto& f : ready_waiters) {
662 f(ir);
663 }
664}
