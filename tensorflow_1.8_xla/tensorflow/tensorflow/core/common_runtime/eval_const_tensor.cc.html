34// Tries to infer tensor output based on the input shapes of the node. In some
35// cases, the shapes of the inputs are sufficient for inferring the contents of
36// the output tensor. For example, a Shape op with fully defined input shapes
37// can have its output tensor inferred.
38Status TryToInferTensorOutputFromInputShapes(const Edge& edge,
39 const ShapeRefiner& refiner,
40 Tensor* output, bool* success) {
41 *success = false;
42 const Node* node = edge.src();
43 InferenceContext* c = refiner.GetContext(node);
44 if (c == nullptr) {
45 return errors::FailedPrecondition("Node does not have context.");
46 }
47
48 if (node->type_string() == "Shape") {
49 // If input shapes to the shape op are fully defined,
50 // we can infer the shape op's output tensor.
51 bool fully_defined_inputs = c->FullyDefined(c->input(0));
52 if (fully_defined_inputs) {
53 int input_rank = c->Rank(c->input(0));
54 Tensor t(node->output_type(0), TensorShape({input_rank}));
55 if (node->output_type(0) == DT_INT32) {
56 auto flat = t.flat<int>();
57 for (int i = 0; i < input_rank; i++) {
58 int64 dimension = c->Value(c->Dim(c->input(0), i));
59 if (!FastBoundsCheck(dimension, std::numeric_limits<int32>::max())) {
60 return errors::InvalidArgument(
61 "Shape has output type int32, but dimension exceeds maximum "
62 "int32 value");
63 }
64 flat(i) = static_cast<int32>(dimension);
65 }
66 } else if (node->output_type(0) == DT_INT64) {
67 auto flat = t.flat<int64>();
68 for (int i = 0; i < input_rank; i++) {
69 flat(i) = c->Value(c->Dim(c->input(0), i));
70 }
71 } else {
72 return errors::FailedPrecondition(
73 "Shape has output type that is not int32 or int64");
74 }
75 *output = t;
76 *success = true;
77 }
78 } else if (node->type_string() == "Rank") {
79 bool rank_known = c->RankKnown(c->input(0));
80 if (rank_known) {
81 int32 input_rank = c->Rank(c->input(0));
82 Tensor t(node->output_type(0), TensorShape({}));
83 t.flat<int32>()(0) = input_rank;
84 *output = t;
85 *success = true;
86 }
87 } else if (node->type_string() == "Size") {
88 bool fully_defined_inputs = c->FullyDefined(c->input(0));
89 if (fully_defined_inputs) {
90 int32 rank = c->Rank(c->input(0));
91 Tensor t(node->output_type(0), TensorShape({}));
92 int64 size = 1;
93 for (int i = 0; i < rank; i++) {
94 size *= c->Value(c->Dim(c->input(0), i));
95 }
96 if (node->output_type(0) == DT_INT32) {
97 if (!FastBoundsCheck(size, std::numeric_limits<int32>::max())) {
98 return errors::InvalidArgument(
99 "Size has output type int32, but size exceeds maximum int32 "
100 "value");
101 }
102 t.flat<int32>()(0) = static_cast<int32>(size);
103 } else if (node->output_type(0) == DT_INT64) {
104 t.flat<int64>()(0) = size;
105 } else {
106 return errors::FailedPrecondition(
107 "Size has output type that is not int32 or int64");
108 }
109 *output = t;
110 *success = true;
111 }
112 }
113 return Status::OK();
114}

116// Extracts the subgraph ending at 'target_node' that is statically computable
117// and inserts into 'out_graph'. If statically computable, 'is_constant_graph'
118// will be set to true.
119Status ExtractConstantSubgraph(
120 const Node& target_node, const ShapeRefiner& refiner,
121 const std::unordered_map<string, Tensor>* cached_values, Graph* out_graph,
122 bool* is_constant_graph,
123 std::vector<std::pair<string, Tensor>>* const_inputs) {
124 *is_constant_graph = false;
125 std::unordered_set<string> const_inputs_added;
126
127 if (target_node.op_def().is_stateful()) {
128 return Status::OK();
129 }
130
131 if (IsMerge(&target_node)) {
132 return Status::OK();
133 }
134
135 if (target_node.type_string() == "PlaceholderWithDefault") {
136 return Status::OK();
137 }
138
139 // TODO(skyewm): should more of the filtering applied in input nodes below be
140 // applied to target_node here?
141
142 // Identify the possibly constant subgraph by recursively iterating backwards
143 // through the inputs to 'target_node' until we either 1) find an already
144 // existing input to our subgraph 'const_inputs', 2) Discover our graph is not
145 // constant, or 3) Hit a root node.
146
147 struct NodeAndRecursed {
148 Node* new_node = nullptr;
149 bool recursed = false;
150 };
151
152 std::map<const Node*, NodeAndRecursed> old_to_new_and_recursed;
153 Node* target_node_copy = out_graph->CopyNode(&target_node);
154 old_to_new_and_recursed[&target_node].new_node = target_node_copy;
155 old_to_new_and_recursed[&target_node].recursed = true;
156
157 // Add the target node's inputs to seed the recursion.
158 std::deque<const Edge*> edges_to_visit;
159 for (const Edge* e : target_node.in_edges()) {
160 // TODO(skyewm): control edges will be meaningful if/when we handle control
161 // flow (e.g. constants in cond branches are triggered via control edges).
162 if (e->IsControlEdge()) continue;
163 edges_to_visit.push_back(e);
164 }
165
166 *is_constant_graph = true;
167
168 // Iterate over the set of edges to visit (backwards).
169 while (!edges_to_visit.empty()) {
170 const Edge* current_edge = edges_to_visit.front();
171 edges_to_visit.pop_front();
172 Node* current_node = current_edge->src();
173
174 // If the node is stateful, assume the graph is not constant.
175 if (current_node->op_def().is_stateful()) {
176 *is_constant_graph = false;
177 return Status::OK();
178 }
179
180 // During construction or import from GraphConstructor, back edges may not
181 // be filled in. In addition, control flow constructs may depend on control
182 // edges which aren't handled by this method. Don't constant fold through
183 // merges at all for now.
184 if (IsMerge(current_node)) {
185 *is_constant_graph = false;
186 return Status::OK();
187 }
188
189 // Don't constant fold enter/exit currently either, as it's easy to end
190 // up with a partial frame.
191 if (IsEnter(current_node) || IsExit(current_node)) {
192 *is_constant_graph = false;
193 return Status::OK();
194 }
195
196 // Placeholders should never be constant folded because their outputs are
197 // fed by the user. Note that "Placeholder" nodes have no inputs so are
198 // handled below.
199 if (current_node->type_string() == "PlaceholderWithDefault") {
200 *is_constant_graph = false;
201 return Status::OK();
202 }
203
204 // If there is nothing more to recurse down, see if
205 // the generator node is a constant.
206 if (current_node->num_inputs() == 0) {
207 if (!current_node->IsConstant()) {
208 // Generator node is not a constant, so subgraph is not
209 // constant.
210 *is_constant_graph = false;
211 return Status::OK();
212 }
213 }
214
215 // Either the node is a constant, or the node is a potential
216 // intermediate node on the path from a constant.
217 //
218 // Add a copy of its node and a new edge to the new subgraph.
219
220 // Get or create the version of 'current_node' in the new graph.
221 Node* current_node_copy;
222 // This gets or creates the NodeAndRecursed entry for current_node.
223 NodeAndRecursed* node_and_recursed = &old_to_new_and_recursed[current_node];
224 if (node_and_recursed->new_node == nullptr) {
225 // First time processing this node.
226 current_node_copy = out_graph->CopyNode(current_node);
227 // Track the mapping from the original node to the new one.
228 node_and_recursed->new_node = current_node_copy;
229 } else {
230 current_node_copy = node_and_recursed->new_node;
231 }
232
233 // Add the edge to the destination node.
234 {
235 auto it = old_to_new_and_recursed.find(current_edge->dst());
236 if (it == old_to_new_and_recursed.end()) {
237 return errors::Internal(
238 "Could not find mapping from old to new copy of destination node: ",
239 current_edge->dst()->name());
240 }
241 Node* dst_copy = it->second.new_node;
242
243 out_graph->AddEdge(current_node_copy, current_edge->src_output(),
244 dst_copy, current_edge->dst_input());
245 }
246
247 const string& output_tensor_name =
248 strings::StrCat(current_node->name(), ":", current_edge->src_output());
249
250 // Some tensor values can be inferred. For example, a shape op
251 // with input shapes fully defined can have its output tensor inferred.
252 Tensor tensor_inferred;
253 bool successfully_inferred_tensor = false;
254 TF_RETURN_IF_ERROR(TryToInferTensorOutputFromInputShapes(
255 *current_edge, refiner, &tensor_inferred,
256 &successfully_inferred_tensor));
257 if (successfully_inferred_tensor) {
258 const_inputs->emplace_back(output_tensor_name, tensor_inferred);
259 const_inputs_added.insert(output_tensor_name);
260 continue;
261 }
262
263 // If we have a copy of the input tensor materialized already,
264 // then add to the list of inputs to feed and do not recurse further.
265 if (cached_values != nullptr) {
266 auto it = cached_values->find(output_tensor_name);
267 if (it != cached_values->end() &&
268 const_inputs_added.count(output_tensor_name) == 0) {
269 const_inputs->emplace_back(output_tensor_name, it->second);
270 const_inputs_added.insert(output_tensor_name);
271 continue;
272 }
273 }
274
275 // If this node's inputs have not been processed already, do so now.
276 if (!node_and_recursed->recursed) {
277 node_and_recursed->recursed = true;
278 for (const Edge* e : current_node->in_edges()) {
279 if (e->IsControlEdge()) continue;
280 edges_to_visit.push_back(e);
281 }
282 }
283 }
284
285 return Status::OK();
286}

290Status EvaluateConstantTensor(OutputTensor tensor, const ShapeRefiner& refiner,
291 const OpRegistryInterface& ops,
292 int32 graph_def_version, bool* evaluated,
293 Tensor* result, GraphRunner* graph_runner,
294 std::unordered_map<string, Tensor>* cached_values,
295 int64 max_cached_value_size,
296 bool disable_constant_propagation) {
297 *evaluated = false;
298 const Node* src = tensor.node;
299
300 // Simple case: the source node is a constant
301 if (src->IsConstant()) {
302 if (result->FromProto(src->def().attr().at("value").tensor())) {
303 *evaluated = true;
304 return Status::OK();
305 }
306 }
307
308 if (disable_constant_propagation) {
309 return Status::OK();
310 }
311
312 bool is_constant_graph = false;
313 Graph subgraph(&ops);
314 auto versions = subgraph.versions();
315 versions.set_producer(graph_def_version);
316 subgraph.set_versions(versions);
317
318 std::vector<std::pair<string, Tensor>> const_inputs;
319 TF_RETURN_IF_ERROR(ExtractConstantSubgraph(*src, refiner, cached_values,
320 &subgraph, &is_constant_graph,
321 &const_inputs));
322 if (!is_constant_graph) {
323 return Status::OK();
324 }
325 const string output_tensor_name =
326 strings::StrCat(src->name(), ":", tensor.index);
327 std::vector<Tensor> outputs;
328
329 std::unique_ptr<GraphRunner> graph_runner_storage;
330 if (graph_runner == nullptr) {
331 // TODO(skyewm): Convert to std::make_unique when available.
332 graph_runner_storage.reset(new GraphRunner(Env::Default()));
333 graph_runner = graph_runner_storage.get();
334 }
335
336 // NOTE; we should pass in a function library runtime if we want
337 // to support constant-expression evaluation on functions.
338 Status s = graph_runner->Run(&subgraph, nullptr /* function_library */,
339 const_inputs, {output_tensor_name}, &outputs);
340
341 // If all kernels in the constant graph are not registered
342 // in the process, GraphRunner::Run may fail, in which case
343 // we cannot propagate constants, so this is best-effort.
344 if (s.ok()) {
345 *result = outputs[0];
346 *evaluated = true;
347
348 // We memoize (small) constants evaluated so far, so
349 // ExtractConstantSubgraph can avoid extracting the full
350 // subgraph. As we build up large graphs, this avoids
351 // repeated computation of the early parts of a constant
352 // graph.
353 if (cached_values != nullptr &&
354 outputs[0].TotalBytes() <= max_cached_value_size) {
355 (*cached_values)[output_tensor_name] = outputs[0];
356 }
357 }
358 return Status::OK();
359}
