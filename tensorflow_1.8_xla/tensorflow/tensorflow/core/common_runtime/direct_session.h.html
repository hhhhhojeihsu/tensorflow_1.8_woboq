119 // We create one executor and its dependent library runtime for
120 // every partition.
121 struct PerPartitionExecutorsAndLib {
122 Graph* graph = nullptr; // not owned.
123 Device* device = nullptr; // not owned.
124 FunctionLibraryRuntime* flib = nullptr; // not owned.
125 std::unique_ptr<Executor> executor;
126 };

128 // An ExecutorsAndKeys is created for a given set of feeds/fetches.
129 // 'step_count' is the number of times this graph is executed.
130 // 'graph' is the entire graph being executed. 'name_to_node'
131 // maps node name to node. We keep 'graph' and 'name_to_node' only in
132 // the case of partial runs. Each item in 'items' is the executor for
133 // a partition of the graph bundled with its dependent library runtime.
134 // 'input_keys' are the rendezvous keys for the feeds and 'output_keys'
135 // are rendezvous keys for the fetches.
136 struct ExecutorsAndKeys {
137 ExecutorsAndKeys() : step_count(0) {}
138
139 std::atomic_int_fast64_t step_count;
140 std::unique_ptr<Graph> graph;
141 NameNodeMap name_to_node;
142 std::vector<PerPartitionExecutorsAndLib> items;
143 std::unordered_map<string, size_t> input_name_to_index;
144 std::unordered_map<string, string> input_name_to_rendezvous_key;
145 std::unordered_map<string, size_t> output_name_to_index;
146 std::unordered_map<string, string> output_name_to_rendezvous_key;
147
148 DataTypeVector input_types;
149 DataTypeVector output_types;
150
151 CallableOptions callable_options;
152 };
153
154 // A FunctionInfo object is created for every unique set of feeds/fetches.
155 // This info could be folded into the ExecutorsAndKeys object but we would
156 // like to maintain a deletion order in which the OpKernels (owned by the
157 // executor) should be destroyed first, followed by the resources in the
158 // device and then followed by the function stuff.
159 // TODO(rohanj): Consolidate function library definitions so that we can
160 // instantiate only one ProcFLR and lib_def and make this just a member
161 // variable and not a vector.
162 // 'flib_def' is the function library used.
163 // 'proc_flr' is the collection of FunctionLibraryRuntime objects, one per
164 // device.
165 struct FunctionInfo {
166 std::unique_ptr<FunctionLibraryDefinition> flib_def;
167 std::unique_ptr<ProcessFunctionLibraryRuntime> proc_flr;
168 };
169
170 // For each live partial execution, the session maintains a RunState.
171 // 'status' is the current status of this partial execution. 'executor_done'
172 // is "notified" when all executors are done. 'pending_inputs' are the set
173 // of pending feeds and 'pending_outputs' are the set of pending fetches.
174 struct RunState {
175 mutex mu_;
176 Status status GUARDED_BY(mu_);
177 IntraProcessRendezvous* rendez = nullptr;
178 std::unique_ptr<StepStatsCollector> collector;
179 Notification executors_done;
180 std::unordered_map<string, bool> pending_inputs; // true if fed
181 std::unordered_map<string, bool> pending_outputs; // true if fetched
182 TensorStore tensor_store;
183 ScopedStepContainer step_container;
184
185 RunState(int64 step_id, const std::vector<Device*>* devices);
186
187 RunState(const std::vector<string>& pending_input_names,
188 const std::vector<string>& pending_output_names, int64 step_id,
189 const std::vector<Device*>* devices);
190
191 // Returns true if all pending inputs and outputs have been completed.
192 bool PendingDone() const;
193
194 ~RunState();
195 };
