20// TODO(saeta): Convert to returning a std::unique_ptr?
21/* static */
22Device* RenamedDevice::NewRenamedDevice(const string& new_base,
23 Device* underlying,
24 bool owns_underlying,
25 bool isolate_session_state) {
26 DeviceNameUtils::ParsedName parsed_name;
27 CHECK(DeviceNameUtils::ParseFullName(new_base, &parsed_name));
28 DeviceNameUtils::ParsedName underlying_parsed_name =
29 underlying->parsed_name();
30 CHECK(underlying_parsed_name.has_type);
31 CHECK(underlying_parsed_name.has_id);
32 parsed_name.type = underlying_parsed_name.type;
33 parsed_name.id = underlying_parsed_name.id;
34 string name = DeviceNameUtils::FullName(parsed_name.job, parsed_name.replica,
35 parsed_name.task, parsed_name.type,
36 parsed_name.id);
37 DeviceAttributes attributes(underlying->attributes());
38 attributes.set_name(name);
39 return new RenamedDevice(underlying, attributes, owns_underlying,
40 isolate_session_state);
41}
42
43RenamedDevice::RenamedDevice(Device* underlying,
44 const DeviceAttributes& attributes,
45 bool owns_underlying, bool isolate_session_state)
46 : Device(underlying->env(), attributes),
47 underlying_(underlying),
48 owns_underlying_(owns_underlying),
49 isolate_session_state_(isolate_session_state) {}
50
51RenamedDevice::~RenamedDevice() {
52 if (owns_underlying_) {
53 delete underlying_;
54 }
55}
