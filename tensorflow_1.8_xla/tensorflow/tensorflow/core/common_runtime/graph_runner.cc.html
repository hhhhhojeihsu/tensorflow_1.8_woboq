45// A simple rendezvous class.
46// Assumes a single sender and a single receiver, no duplicate sends, and no
47// sends of dead tensors.
48class SimpleRendezvous : public Rendezvous {
49 public:
50 explicit SimpleRendezvous() {}
51
52 Status Send(const ParsedKey& parsed, const Args& send_args, const Tensor& val,
53 const bool is_dead) override {
54 if (is_dead) {
55 return errors::Internal("Send of a dead tensor");
56 }
57
58 mutex_lock l(mu_);
59 string edge_name = parsed.edge_name.ToString();
60 if (table_.count(edge_name) > 0) {
61 return errors::Internal("Send of an already sent tensor");
62 }
63 table_[edge_name] = val;
64 return Status::OK();
65 }
66
67 void RecvAsync(const ParsedKey& parsed, const Args& recv_args,
68 DoneCallback done) override {
69 Tensor tensor;
70 Status status = Status::OK();
71 {
72 string key = parsed.edge_name.ToString();
73 mutex_lock l(mu_);
74 if (table_.count(key) <= 0) {
75 status = errors::Internal("Did not find key ", key);
76 } else {
77 tensor = table_[key];
78 }
79 }
80 done(status, Args{}, recv_args, tensor, false);
81 }
82
83 void StartAbort(const Status& status) override {}
84
85 private:
86 typedef std::unordered_map<string, Tensor> Table;
87
88 mutex mu_;
89 Table table_ GUARDED_BY(mu_);
90};

101Status GraphRunner::Run(Graph* graph, FunctionLibraryRuntime* function_library,
102 const NamedTensorList& inputs,
103 const std::vector<string>& output_names,
104 std::vector<Tensor>* outputs) {
105 if (device_ == nullptr) {
106 return errors::NotFound("Cannot find a device for GraphRunner.");
107 }
108
109 if (function_library && function_library->device() &&
110 function_library->device()->device_type() != device_->device_type()) {
111 // Mismatch between function_library's device_type and device_'s
112 // device_type.
113 // TODO(matthewmurray) Can we create a new FunctionLibraryRuntime that is
114 // identical to function_library except that it uses the given 'device_'?
115 VLOG(1) << "Cannot run on: " << device_->device_type()
116 << " with a function library for a "
117 << function_library->device()->device_type() << " device.";
118 function_library = nullptr;
119 }
120
121 // TODO(vrv): Instead of copying the entire graph, consider modifying
122 // the existing graph, and then removing those removed edges.
123 // prior to returning.
124 std::unique_ptr<Graph> graph_to_run(new Graph(graph->op_registry()));
125 CopyGraph(*graph, graph_to_run.get());
126
127 SimpleRendezvous* rendez = new SimpleRendezvous;
128 core::ScopedUnref rendez_unref(rendez);
129
130 // Extract the input names and keys, and feed in the inputs.
131 std::vector<string> input_names;
132 for (const auto& in : inputs) {
133 const string& tensor_name = in.first;
134 input_names.emplace_back(tensor_name);
135 string full_key = Rendezvous::CreateKey("/device:CPU:0", 1, "/device:CPU:1",
136 tensor_name, FrameAndIter(0, 0));
137 Rendezvous::ParsedKey parsed;
138 TF_RETURN_IF_ERROR(Rendezvous::ParseKey(full_key, &parsed));
139 TF_RETURN_IF_ERROR(rendez->Send(parsed, Rendezvous::Args(), in.second,
140 false /* is_dead */));
141 }
142
143 // Call RewriteGraphForExecution
144 subgraph::RewriteGraphMetadata metadata;
145 TF_RETURN_IF_ERROR(subgraph::RewriteGraphForExecution(
146 graph_to_run.get(), input_names, output_names, {} /* target nodes */,
147 device_->attributes(), false /* use_function_convention */, &metadata));
148
149 // Create the local executor and the Rendezvous for fetching back the
150 // constants.
151
152 // Run operators on the local thread. We should not need concurrency here; we
153 // should not be running expensive operators.
154 auto runner = [](Executor::Args::Closure c) { c(); };
155
156 LocalExecutorParams params;
157 // The ownership of the output tensors are bound to this device's lifetime.
158 params.device = device_;
159 params.function_library = function_library;
160 const int producer = graph_to_run->versions().producer();
161 params.create_kernel = [this, producer](const NodeDef& ndef,
162 OpKernel** kernel) {
163 return CreateNonCachedKernel(device_, nullptr, ndef, producer, kernel);
164 };
165 params.delete_kernel = [](OpKernel* kernel) { delete kernel; };
166
167 Executor* executor;
168 TF_RETURN_IF_ERROR(
169 NewLocalExecutor(params, std::move(graph_to_run), &executor));
170 std::unique_ptr<Executor> executor_unref(executor);
171
172 Executor::Args args;
173 // NOTE: we could take a step id as an argument, but currently
174 // there is no need since we never trace the running of a graph
175 // called via this method.
176 args.step_id = LogMemory::CONSTANT_FOLDING_STEP_ID;
177 args.runner = runner;
178 args.rendezvous = rendez;
179
180 // Run the graph.
181 TF_RETURN_IF_ERROR(executor->Run(args));
182
183 outputs->resize(output_names.size());
184 for (size_t i = 0; i < output_names.size(); ++i) {
185 const string& output_key =
186 Rendezvous::CreateKey("/device:CPU:0", 1, "/device:CPU:1",
187 output_names[i], FrameAndIter(0, 0));
188 Rendezvous::ParsedKey parsed;
189 TF_RETURN_IF_ERROR(Rendezvous::ParseKey(output_key, &parsed));
190 bool is_dead;
191 Tensor output_tensor;
192 TF_RETURN_IF_ERROR(
193 rendez->Recv(parsed, Rendezvous::Args(), &output_tensor, &is_dead));
194 // Does a deep copy so that ownership of the tensor isn't tied to the
195 // allocator of the cpu device we created above. The allocator could be
196 // deleted along with the device.
197 (*outputs)[i] = tensor::DeepCopy(output_tensor);
198 }
199
200 return Status::OK();
201}
