27namespace tensorflow {
28
29class Env;
30
31// A simple single-threaded CPU device. This can be used to run inexpensive
32// computations. In particular, using this avoids initializing the global thread
33// pools in LocalDevice.
34class SingleThreadedCpuDevice : public Device {
35 public:
36 SingleThreadedCpuDevice(Env* env)
37 : Device(env, Device::BuildDeviceAttributes("/device:CPU:0", DEVICE_CPU,
38 Bytes(256 << 20),
39 DeviceLocality())) {
40 eigen_worker_threads_.num_threads = 1;
41 eigen_worker_threads_.workers = new thread::ThreadPool(
42 env, "graph_runner", eigen_worker_threads_.num_threads);
43 eigen_threadpool_wrapper_.reset(
44 new EigenThreadPoolWrapper(eigen_worker_threads_.workers));
45 eigen_device_.reset(new Eigen::ThreadPoolDevice(
46 eigen_threadpool_wrapper_.get(), eigen_worker_threads_.num_threads));
47 set_tensorflow_cpu_worker_threads(&eigen_worker_threads_);
48 set_eigen_cpu_device(eigen_device_.get());
49 }
50
51 ~SingleThreadedCpuDevice() override {
52 eigen_threadpool_wrapper_.reset();
53 eigen_device_.reset();
54 delete eigen_worker_threads_.workers;
55 }
56
57 Status Sync() override { return Status::OK(); }
58
59 Status MakeTensorFromProto(const TensorProto& tensor_proto,
60 const AllocatorAttributes alloc_attrs,
61 Tensor* tensor) override {
62 Tensor parsed(tensor_proto.dtype());
63 if (!parsed.FromProto(cpu_allocator(), tensor_proto)) {
64 return errors::InvalidArgument("Cannot parse tensor from tensor_proto.");
65 }
66 *tensor = parsed;
67 return Status::OK();
68 }
69
70 Allocator* GetAllocator(AllocatorAttributes attr) override {
71 return cpu_allocator();
72 }
73
74 private:
75 DeviceBase::CpuWorkerThreads eigen_worker_threads_;
76 std::unique_ptr<Eigen::ThreadPoolInterface> eigen_threadpool_wrapper_;
77 std::unique_ptr<Eigen::ThreadPoolDevice> eigen_device_;
78};
