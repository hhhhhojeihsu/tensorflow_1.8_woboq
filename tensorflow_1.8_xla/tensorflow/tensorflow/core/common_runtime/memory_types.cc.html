29struct Endpoint {
30 int node_id;
31 int output_index;
32};
33
34struct EndpointHash {
35 uint32 operator()(const Endpoint& x) const {
36 return Hash32(reinterpret_cast<const char*>(&x.node_id), sizeof(int),
37 x.output_index);
38 }
39};
40
41struct EndpointEq {
42 uint32 operator()(const Endpoint& x, const Endpoint& y) const {
43 return (x.node_id == y.node_id) && (x.output_index == y.output_index);
44 }
45};
46
47static Status ProcessMemoryTypes(
48 const DeviceType& device_type, const Graph* g,
49 const std::function<Status(const Edge*, MemoryType, MemoryType)>& fn) {
50 if (device_type != DEVICE_GPU && device_type != DEVICE_SYCL) {
51 // On non-GPU and non-SYCL devices, HOST_MEMORY and DEVICE_MEMORY are always
52 // compatible.
53 return Status::OK();
54 }
55 // For GPU and SYCL device, HOST_MEMORY and DEVICE_MEMORY is not
56 // compatible. I.e., a conversion/transfer must be done.
57 //
58 // {node id, slot id} -> memory type.
59 typedef std::unordered_map<Endpoint, MemoryType, EndpointHash, EndpointEq>
60 MemTypeMap;
61 MemTypeMap inp;
62 MemTypeMap out;
63 MemoryTypeVector inp_mvec;
64 MemoryTypeVector out_mvec;
65 for (const Node* n : g->nodes()) {
66 TF_RETURN_IF_ERROR(MemoryTypesForNode(g->op_registry(), device_type,
67 n->def(), &inp_mvec, &out_mvec));
68 for (size_t i = 0; i < inp_mvec.size(); ++i) {
69 VLOG(2) << "inp mvec " << n->id() << " " << i << " " << inp_mvec[i];
70 inp[{n->id(), static_cast<int>(i)}] = inp_mvec[i];
71 }
72 for (size_t i = 0; i < out_mvec.size(); ++i) {
73 VLOG(2) << "out mvec " << n->id() << " " << i << " " << out_mvec[i];
74 out[{n->id(), static_cast<int>(i)}] = out_mvec[i];
75 }
76 }
77 for (const Edge* e : g->edges()) {
78 if (e->IsControlEdge()) {
79 continue;
80 }
81 MemoryType sm = gtl::FindWithDefault(out, {e->src()->id(), e->src_output()},
82 DEVICE_MEMORY);
83 MemoryType dm = gtl::FindWithDefault(inp, {e->dst()->id(), e->dst_input()},
84 DEVICE_MEMORY);
85 VLOG(1) << e->src()->id() << ":" << e->src_output() << " -> "
86 << e->dst()->id() << ":" << e->dst_input() << ": " << sm << " -> "
87 << dm;
88 TF_RETURN_IF_ERROR(fn(e, sm, dm));
89 }
90 return Status::OK();
91}

93Status ValidateMemoryTypes(const DeviceType& device_type, const Graph* g) {
94 return ProcessMemoryTypes(
95 device_type, g, [](const Edge* e, MemoryType sm, MemoryType dm) {
96 if (sm == dm) {
97 return Status::OK();
98 }
99 return errors::Internal(
100 "Memory type mismatch (", sm, " ", dm,
101 ") between :", e->src()->id(), ":", e->src_output(), " and ",
102 e->dst()->id(), ":", e->dst_input(), " : from ",
103 e->src()->DebugString(), " to ", e->dst()->DebugString());
104 });
105}
106
107// Given an Edge whose two endpoints have different memory types and
108// are gonna to insert a pair of HostSend/Recv or Send/HostRecv nodes,
109// GetTensorName() returns a unique string that we can use as part of
110// the rendezvous key. The return string is guaranteed to be unique
111// within this process. That is sufficient because EnsureMemoryTypes
112// is only used on a TensorFlow graph that is gonna to be executed in
113// a single tf device (hence within a single process).
114static string GetTensorName(const Edge* edge) {
115 static std::atomic<int64> counter(0);
116 return strings::StrCat("memtype_", counter.fetch_add(1), "_",
117 edge->src()->name());
118}
119
120static Node* Send(Graph* g, const string& tensor_name,
121 const string& device_name, bool host, const Edge* edge) {
122 Node* ret;
123 TF_CHECK_OK(NodeBuilder(g->NewName("n"), host ? "_HostSend" : "_Send")
124 .Input(edge->src(), edge->src_output())
125 .Attr("tensor_name", tensor_name)
126 .Attr("send_device", device_name)
127 .Attr("send_device_incarnation", 0) // Do not care.
128 .Attr("recv_device", device_name)
129 .Attr("_hostmem_sendrecv", true)
130 .Finalize(g, &ret));
131 return ret;
132}
133
134static Node* Recv(Graph* g, const string& tensor_name,
135 const string& device_name, bool host, const Edge* edge) {
136 Node* ret;
137 TF_CHECK_OK(
138 NodeBuilder(g->NewName("n"), host ? "_HostRecv" : "_Recv")
139 .Attr("tensor_type", edge->src()->output_type(edge->src_output()))
140 .Attr("tensor_name", tensor_name)
141 .Attr("send_device", device_name)
142 .Attr("send_device_incarnation", 0)
143 .Attr("recv_device", device_name)
144 .Attr("_hostmem_sendrecv", true)
145 .Finalize(g, &ret));
146 return ret;
147}

149Status EnsureMemoryTypes(const DeviceType& device_type,
150 const string& device_name, Graph* g) {
151 struct Item {
152 const Edge* edge;
153 MemoryType sm;
154 MemoryType dm;
155 };
156 std::vector<Item> edges;
157 TF_RETURN_IF_ERROR(ProcessMemoryTypes(
158 device_type, g, [&edges](const Edge* e, MemoryType sm, MemoryType dm) {
159 if (sm == dm) {
160 return Status::OK();
161 }
162 if (((sm == HOST_MEMORY) && (dm == DEVICE_MEMORY)) ||
163 ((sm == DEVICE_MEMORY) && (dm == HOST_MEMORY))) {
164 edges.push_back({e, sm, dm});
165 return Status::OK();
166 }
167 return errors::Internal("Unexpected memory type pair on an edge: ", sm,
168 " vs. ", dm);
169 }));
170
171 // edges contains edges in 'g' that memtype is not
172 // compatible. Therefore, if we found any, we need to insert
173 // HostSend/Recv and Send/HostRecv pairs. recv_nodes records all
174 // nodes we added so that we don't copy the same tensor more than
175 // once.
176 if (!edges.empty()) {
177 std::unordered_map<Endpoint, Node*, EndpointHash, EndpointEq> recv_nodes;
178 for (const auto& item : edges) {
179 const Edge* e = item.edge;
180 const bool has_ref = IsRefType(e->src()->output_type(e->src_output()));
181 Node* recv = nullptr;
182 Endpoint key{e->src()->id(), e->src_output()};
183 auto iter = recv_nodes.find(key);
184 if (iter == recv_nodes.end()) {
185 const string tensor_name = GetTensorName(e);
186 Node* send =
187 Send(g, tensor_name, device_name, (item.sm == HOST_MEMORY), e);
188 recv = Recv(g, tensor_name, device_name, (item.dm == HOST_MEMORY), e);
189 if (!has_ref) {
190 // We only cache if there is no ref is involved.
191 recv_nodes[key] = recv;
192 }
193 g->AddControlEdge(send, recv);
194 } else {
195 recv = iter->second;
196 }
197 g->AddEdge(recv, 0, e->dst(), e->dst_input());
198 g->RemoveEdge(e);
199 }
200 }
201 return ValidateMemoryTypes(device_type, g);
202}

204Status MemoryTypeForOutput(const DeviceType& device_type, const Graph* g,
205 const Node* n, int index, MemoryType* memory_type) {
206 MemoryTypeVector inp_mvec;
207 MemoryTypeVector out_mvec;
208 TF_RETURN_IF_ERROR(MemoryTypesForNode(g->op_registry(), device_type, n->def(),
209 &inp_mvec, &out_mvec));
210 if (out_mvec.size() <= index) {
211 return errors::Internal("Trying to get the memory type for ", index,
212 "'th output of node ", n->DebugString(),
213 " that has only ", out_mvec.size(), " outputs");
214 }
215 *memory_type = out_mvec[index];
216 return Status::OK();
217}
