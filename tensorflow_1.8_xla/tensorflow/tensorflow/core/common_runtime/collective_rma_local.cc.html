23 buf_rendezvous_.StartAbort(s);

26void CollectiveRemoteAccessLocal::RecvFromPeer(
27 const string& peer_device, const string& peer_task, bool peer_is_local,
28 const string& key, Device* to_device, DeviceContext* to_device_ctx,
29 const AllocatorAttributes& to_alloc_attr, Tensor* to_tensor,
30 const DeviceLocality& client_locality, const StatusCallback& done) {
31 VLOG(1) << "RecvFromPeer " << this << " from " << peer_device << " key "
32 << key;
33 if (!peer_is_local) {
34 done(
35 errors::Internal("CollectiveRemoteAccessLocal::RecvFromPeer "
36 "called with peer_is_local=false"));
37 return;
38 }
39 buf_rendezvous_.ConsumeBuf(
40 key, [this, to_tensor, to_device_ctx, to_device, to_alloc_attr, done](
41 const Status& s, BufRendezvous::Hook* hook) {
42 if (!s.ok()) {
43 done(s);
44 delete hook;
45 } else {
46 int64 recv_bytes = to_tensor->TotalBytes();
47 CHECK_EQ(recv_bytes, hook->prod_value->TotalBytes());
48 MemCpyAsync(hook->prod_ctx, // src DeviceContext
49 to_device_ctx, // dst DeviceContext
50 hook->prod_dev, // src Device
51 to_device, // dst Device
52 hook->prod_attr, // src AllocatorAttributes
53 to_alloc_attr, // dst AllocatorAttributes
54 hook->prod_value, // src Tensor*
55 to_tensor, // dst Tensor*
56 [hook, done](const Status& s) {
57 done(s);
58 hook->prod_cb(s);
59 delete hook;
60 });
61 }
62 });
63}

65void CollectiveRemoteAccessLocal::PostToPeer(
66 const string& peer_device, const string& peer_task, const string& key,
67 Device* from_device, DeviceContext* from_device_ctx,
68 const AllocatorAttributes& from_alloc_attr, const Tensor* from_tensor,
69 const DeviceLocality& client_locality, const StatusCallback& done) {
70 VLOG(1) << "PostToPeer " << this << " key " << key
71 << " step_id_=" << step_id_;
72 buf_rendezvous_.ProvideBuf(key, from_device, from_device_ctx, from_tensor,
73 from_alloc_attr, done);
74}

76/*static*/
77void CollectiveRemoteAccessLocal::MemCpyAsync(
78 DeviceContext* src_dev_ctx, DeviceContext* dst_dev_ctx, Device* src_dev,
79 Device* dst_dev, const AllocatorAttributes& src_attr,
80 const AllocatorAttributes& dst_attr, const Tensor* src, Tensor* dst,
81 const StatusCallback& done) {
82 // We want a real copy to happen, i.e. the bytes inside of src should be
83 // transferred to the buffer backing dst. If src and dst are on different
84 // devices then CopyTensor::ViaDMA will do just that. But if they're both
85 // the same CPU, then it will actually just reset dst to point to src.
86 // Since this routine is used for copying between devices and within a
87 // device, we need to detect and bypass the wrong-semantics case.
88 const DeviceType src_device_type(
89 src_attr.on_host() ? DEVICE_CPU : src_dev->attributes().device_type());
90 const DeviceType dst_device_type(
91 dst_attr.on_host() ? DEVICE_CPU : dst_dev->attributes().device_type());
92 const bool non_cpu_src = src_device_type != DeviceType(DEVICE_CPU);
93 const bool non_cpu_dst = dst_device_type != DeviceType(DEVICE_CPU);
94 if (non_cpu_src) CHECK(src_dev_ctx);
95 if (non_cpu_dst) CHECK(dst_dev_ctx);
96 if (non_cpu_src || non_cpu_dst) {
97 CopyTensor::ViaDMA("", // edge name (non-existent)
98 src_dev_ctx, dst_dev_ctx, src_dev, dst_dev, src_attr,
99 dst_attr, src, dst, done);
100 } else {
101 int64 bytes = src->TotalBytes();
102 DCHECK_EQ(dst->TotalBytes(), bytes);
103 memcpy(DMAHelper::base(dst), DMAHelper::base(src), bytes);
104 done(Status::OK());
105 }
106}
