15#include "tensorflow/core/kernels/data/window_dataset.h"

17namespace tensorflow {
18namespace {
19
20class WindowDataset : public DatasetBase {
21 public:
22 WindowDataset(std::vector<std::vector<Tensor>> elements,
23 DataTypeVector output_types,
24 std::vector<PartialTensorShape> output_shapes)
25 : elements_(std::move(elements)),
26 output_types_(std::move(output_types)),
27 output_shapes_(std::move(output_shapes)) {}
28
29 std::unique_ptr<IteratorBase> MakeIterator(
30 const string& prefix) const override {
31 return std::unique_ptr<IteratorBase>(
32 new Iterator({this, strings::StrCat(prefix, "::Window")}));
33 }
34
35 const DataTypeVector& output_dtypes() const override { return output_types_; }
36
37 const std::vector<PartialTensorShape>& output_shapes() const override {
38 return output_shapes_;
39 }
40
41 string DebugString() override { return "WindowDataset"; }
42
43 private:
44 class Iterator : public DatasetIterator<WindowDataset> {
45 public:
46 explicit Iterator(const Params& params)
47 : DatasetIterator<WindowDataset>(params) {}
48
49 Status GetNextInternal(IteratorContext* ctx,
50 std::vector<Tensor>* out_tensors,
51 bool* end_of_sequence) override {
52 mutex_lock l(mu_);
53 if (i_ == dataset()->elements_.size()) {
54 *end_of_sequence = true;
55 } else {
56 *end_of_sequence = false;
57 *out_tensors = dataset()->elements_[i_++];
58 }
59 return Status::OK();
60 }
61
62 Status SaveInternal(IteratorStateWriter* writer) override {
63 mutex_lock l(mu_);
64 TF_RETURN_IF_ERROR(writer->WriteScalar(full_name("i"), i_));
65 return Status::OK();
66 }
67
68 Status RestoreInternal(IteratorContext* ctx,
69 IteratorStateReader* reader) override {
70 mutex_lock l(mu_);
71 int64 i;
72 TF_RETURN_IF_ERROR(reader->ReadScalar(full_name("i"), &i));
73 i_ = size_t(i);
74 return Status::OK();
75 }
76
77 mutex mu_;
78 size_t i_ GUARDED_BY(mu_) = 0;
79 };
80
81 const std::vector<std::vector<Tensor>> elements_;
82 const DataTypeVector output_types_;
83 const std::vector<PartialTensorShape> output_shapes_;
84};
85
86} // namespace
87
88Status NewWindowDataset(std::vector<std::vector<Tensor>> elements,
89 DataTypeVector output_types,
90 std::vector<PartialTensorShape> output_shapes,
91 DatasetBase** out_dataset) {
92 // TODO(mrry): If this becomes more public, we must validate that
93 // the elements match the output_types and output_shapes.
94 *out_dataset = new WindowDataset(std::move(elements), std::move(output_types),
95 std::move(output_shapes));
96 return Status::OK();
97}
