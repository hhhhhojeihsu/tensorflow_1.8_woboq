23// See documentation in ../ops/dataset_ops.cc for a high-level
24// description of the following op.
25
26class TakeDatasetOp : public UnaryDatasetOpKernel {
27 public:
28 explicit TakeDatasetOp(OpKernelConstruction* ctx)
29 : UnaryDatasetOpKernel(ctx) {}
30
31 protected:
32 void MakeDataset(OpKernelContext* ctx, DatasetBase* input,
33 DatasetBase** output) override {
34 // Create a new TakeDatasetOp::Dataset, and return it as the output.
35 int64 count;
36 OP_REQUIRES_OK(ctx, ParseScalarArgument<int64>(ctx, "count", &count));
37 *output = new Dataset(ctx, count, input);
38 }
39
40 private:
41 class Dataset : public GraphDatasetBase {
42 public:
43 Dataset(OpKernelContext* ctx, int64 count, const DatasetBase* input)
44 : GraphDatasetBase(ctx), count_(count), input_(input) {
45 input_->Ref();
46 }
47
48 ~Dataset() override { input_->Unref(); }
49
50 std::unique_ptr<IteratorBase> MakeIterator(
51 const string& prefix) const override {
52 if (count_ < 0) {
53 // Pass through
54 return input_->MakeIterator(prefix);
55 } else if (count_ == 0) {
56 return std::unique_ptr<IteratorBase>(
57 new EmptyIterator({this, strings::StrCat(prefix, "::EmptyTake")}));
58 } else {
59 return std::unique_ptr<IteratorBase>(new FiniteIterator(
60 {this, strings::StrCat(prefix, "::FiniteTake")}));
61 }
62 }
63
64 const DataTypeVector& output_dtypes() const override {
65 return input_->output_dtypes();
66 }
67
68 const std::vector<PartialTensorShape>& output_shapes() const override {
69 return input_->output_shapes();
70 }
71
72 string DebugString() override { return "TakeDatasetOp::Dataset"; }
73
74 protected:
75 Status AsGraphDefInternal(OpKernelContext* ctx, DatasetGraphDefBuilder* b,
76 Node** output) const override {
77 Node* input_graph_node = nullptr;
78 TF_RETURN_IF_ERROR(b->AddParentDataset(ctx, input_, &input_graph_node));
79 Node* count = nullptr;
80 TF_RETURN_IF_ERROR(b->AddScalar(count_, &count));
81 TF_RETURN_IF_ERROR(
82 b->AddDataset(this, {input_graph_node, count}, output));
83 return Status::OK();
84 }
85
86 private:
87 class EmptyIterator : public DatasetIterator<Dataset> {
88 public:
89 explicit EmptyIterator(const Params& params)
90 : DatasetIterator<Dataset>(params) {}
91 Status GetNextInternal(IteratorContext* ctx,
92 std::vector<Tensor>* out_tensors,
93 bool* end_of_sequence) override {
94 *end_of_sequence = true;
95 return Status::OK();
96 }
97
98 protected:
99 Status SaveInternal(IteratorStateWriter* writer) override {
100 return Status::OK();
101 }
102
103 Status RestoreInternal(IteratorContext* ctx,
104 IteratorStateReader* reader) override {
105 return Status::OK();
106 }
107 };
108
109 class FiniteIterator : public DatasetIterator<Dataset> {
110 public:
111 explicit FiniteIterator(const Params& params)
112 : DatasetIterator<Dataset>(params),
113 i_(0),
114 input_impl_(params.dataset->input_->MakeIterator(params.prefix)) {}
115
116 Status GetNextInternal(IteratorContext* ctx,
117 std::vector<Tensor>* out_tensors,
118 bool* end_of_sequence) override {
119 mutex_lock l(mu_); // TODO(mrry): Make locking less conservative.
120 if (!input_impl_) {
121 *end_of_sequence = true;
122 return Status::OK();
123 }
124 while (i_ < dataset()->count_) {
125 TF_RETURN_IF_ERROR(
126 input_impl_->GetNext(ctx, out_tensors, end_of_sequence));
127 if (!*end_of_sequence) {
128 ++i_;
129 return Status::OK();
130 }
131 break;
132 }
133 *end_of_sequence = true;
134 input_impl_.reset();
135 return Status::OK();
136 }
137
138 protected:
139 Status SaveInternal(IteratorStateWriter* writer) override {
140 mutex_lock l(mu_);
141 TF_RETURN_IF_ERROR(writer->WriteScalar(full_name("i"), i_));
142 if (input_impl_) {
143 TF_RETURN_IF_ERROR(SaveParent(writer, input_impl_));
144 } else {
145 TF_RETURN_IF_ERROR(
146 writer->WriteScalar(full_name("input_impl_empty"), ""));
147 }
148 return Status::OK();
149 }
150
151 Status RestoreInternal(IteratorContext* ctx,
152 IteratorStateReader* reader) override {
153 mutex_lock l(mu_);
154 TF_RETURN_IF_ERROR(reader->ReadScalar(full_name("i"), &i_));
155 if (!reader->Contains(full_name("input_impl_empty"))) {
156 TF_RETURN_IF_ERROR(RestoreParent(ctx, reader, input_impl_));
157 } else {
158 input_impl_.reset();
159 }
160 return Status::OK();
161 }
162
163 private:
164 mutex mu_;
165 int64 i_ GUARDED_BY(mu_);
166 std::unique_ptr<IteratorBase> input_impl_ GUARDED_BY(mu_);
167 };
168
169 const int64 count_;
170 const DatasetBase* const input_;
171 };
172};

174REGISTER_KERNEL_BUILDER(Name("TakeDataset").Device(DEVICE_CPU), TakeDatasetOp);
