24// See documentation in ../ops/dataset_ops.cc for a high-level
25// description of the following op.
26
27class SlideDatasetOp : public UnaryDatasetOpKernel {
28 public:
29 explicit SlideDatasetOp(OpKernelConstruction* ctx)
30 : UnaryDatasetOpKernel(ctx) {}
31
32 void MakeDataset(OpKernelContext* ctx, DatasetBase* input,
33 DatasetBase** output) override {
34 int64 window_size = 0;
35 int64 stride = 1;
36 OP_REQUIRES_OK(ctx,
37 ParseScalarArgument<int64>(ctx, "window_size", &window_size));
38 OP_REQUIRES_OK(ctx,
39 ParseScalarArgument<int64>(ctx, "stride", &stride));
40 OP_REQUIRES(
41 ctx, window_size > 0,
42 errors::InvalidArgument("Window size must be greater than zero."));
43 OP_REQUIRES(
44 ctx, stride > 0 && stride < window_size,
45 errors::InvalidArgument("Stride must be in [1, window_size)."));
46
47 *output = new Dataset(ctx, window_size, stride, input);
48 }
49
50 private:
51 class Dataset : public GraphDatasetBase {
52 public:
53 Dataset(OpKernelContext* ctx, int64 window_size, int64 stride, const DatasetBase* input)
54 : GraphDatasetBase(ctx), window_size_(window_size), stride_(stride), input_(input) {
55 input_->Ref();
56
57 const auto& input_shapes = input_->output_shapes();
58 output_shapes_.reserve(input_shapes.size());
59 for (const auto& input_shape : input_shapes) {
60 output_shapes_.emplace_back(
61 PartialTensorShape({-1}).Concatenate(input_shape));
62 }
63 }
64
65 ~Dataset() override { input_->Unref(); }
66
67 std::unique_ptr<IteratorBase> MakeIterator(
68 const string& prefix) const override {
69 return std::unique_ptr<IteratorBase>(new Iterator(
70 Iterator::Params{this, strings::StrCat(prefix, "::Slide")}));
71 }
72
73 const DataTypeVector& output_dtypes() const override {
74 return input_->output_dtypes();
75 }
76
77 const std::vector<PartialTensorShape>& output_shapes() const override {
78 return output_shapes_;
79 }
80
81 string DebugString() override {
82 return strings::StrCat("SlideDatasetOp(", window_size_, ", ", stride_, ")::Dataset");
83 }
84
85 protected:
86 Status AsGraphDefInternal(OpKernelContext* ctx, DatasetGraphDefBuilder* b,
87 Node** output) const override {
88 Node* input_graph_node = nullptr;
89 TF_RETURN_IF_ERROR(b->AddParentDataset(ctx, input_, &input_graph_node));
90 Node* window_size = nullptr;
91 Node* stride = nullptr;
92 TF_RETURN_IF_ERROR(b->AddScalar(window_size_, &window_size));
93 TF_RETURN_IF_ERROR(b->AddScalar(stride_, &stride));
94 TF_RETURN_IF_ERROR(
95 b->AddDataset(this, {input_graph_node, window_size, stride}, output));
96 return Status::OK();
97 }
98
99 private:
100
101 class Iterator : public DatasetIterator<Dataset> {
102 public:
103 explicit Iterator(const Params& params)
104 : DatasetIterator<Dataset>(params),
105 input_impl_(params.dataset->input_->MakeIterator(params.prefix)) {}
106
107 Status GetNextInternal(IteratorContext* ctx,
108 std::vector<Tensor>* out_tensors,
109 bool* end_of_sequence) override {
110 const int64 window_size = dataset()->window_size_;
111 const int64 stride = dataset()->stride_;
112 std::vector<std::vector<Tensor>> batch_elements;
113 {
114 mutex_lock l(mu_);
115 if (!input_impl_) {
116 *end_of_sequence = true;
117 return Status::OK();
118 }
119 batch_elements.reserve(window_size);
120 const bool first_call = cache_.empty();
121 if (first_call) {
122 cache_.reserve(window_size);
123 } else {
124 // Reuse cache in the previous iteration.
125 cache_.swap(batch_elements);
126 }
127 // Fill up with new elements.
128 *end_of_sequence = false;
129 for (size_t i = batch_elements.size(); i < window_size && !*end_of_sequence;
130 ++i) {
131 std::vector<Tensor> batch_element_tuple;
132 TF_RETURN_IF_ERROR(input_impl_->GetNext(ctx, &batch_element_tuple,
133 end_of_sequence));
134 if (!*end_of_sequence) {
135 batch_elements.push_back(std::move(batch_element_tuple));
136 } else {
137 input_impl_.reset();
138 }
139 }
140 // Drop the final smaller blocks.
141 if (batch_elements.size() < window_size) {
142 DCHECK(*end_of_sequence);
143 return Status::OK();
144 }
145 // Cache the data used for the next iteration.
146 for (size_t i = stride; i < window_size; ++i) {
147 cache_.emplace_back(batch_elements[i]);
148 }
149 }
150
151 // Construct output tensors.
152 // Those codes below are copied from batch_dataset_op.cc.
153 const size_t num_tuple_components = batch_elements[0].size();
154 const int64 num_batch_elements = batch_elements.size();
155 for (size_t component_index = 0; component_index < num_tuple_components;
156 ++component_index) {
157 const Tensor& first_element = batch_elements[0][component_index];
158 TensorShape batch_component_shape({num_batch_elements});
159 batch_component_shape.AppendShape(first_element.shape());
160 Tensor batch_component(cpu_allocator(), first_element.dtype(),
161 batch_component_shape);
162 // Build the output tuple component by copying one slice
163 // from each input element in the batch.
164 for (size_t i = 0; i < num_batch_elements; ++i) {
165 if (batch_elements[i][component_index].shape() !=
166 first_element.shape()) {
167 return errors::InvalidArgument(
168 "Cannot batch tensors with different shapes in component ",
169 component_index, ". First element had shape ",
170 first_element.shape().DebugString(), " and element ", i,
171 " had shape ",
172 batch_elements[i][component_index].shape().DebugString(),
173 ".");
174 }
175 TF_RETURN_IF_ERROR(batch_util::CopyElementToSlice(
176 std::move(batch_elements[i][component_index]), &batch_component,
177 i));
178 }
179 out_tensors->emplace_back(std::move(batch_component));
180 }
181 *end_of_sequence = false;
182 return Status::OK();
183 }
184
185 protected:
186 Status SaveInternal(IteratorStateWriter* writer) override {
187 mutex_lock l(mu_);
188 if (!input_impl_) {
189 TF_RETURN_IF_ERROR(
190 writer->WriteScalar(full_name("input_impl_empty"), ""));
191 } else {
192 TF_RETURN_IF_ERROR(SaveParent(writer, input_impl_));
193 }
194 // Save cache.
195 TF_RETURN_IF_ERROR(
196 writer->WriteScalar(strings::StrCat("cache_size"), cache_.size()));
197 for (int64 i = 0; i < cache_.size(); i++) {
198 TF_RETURN_IF_ERROR(writer->WriteScalar(
199 strings::StrCat("cache[", i, "]_size"), cache_[i].size()));
200 for (int64 j = 0; j < cache_[i].size(); j++) {
201 TF_RETURN_IF_ERROR(writer->WriteTensor(
202 strings::StrCat("cache[", i, "][", j, "]"), cache_[i][j]));
203 }
204 }
205 return Status::OK();
206 }
207
208 Status RestoreInternal(IteratorContext* ctx,
209 IteratorStateReader* reader) override {
210 mutex_lock l(mu_);
211 if (!reader->Contains(full_name("input_impl_empty"))) {
212 TF_RETURN_IF_ERROR(RestoreParent(ctx, reader, input_impl_));
213 } else {
214 input_impl_.reset();
215 }
216 // Restore cache.
217 int64 cache_size;
218 TF_RETURN_IF_ERROR(
219 reader->ReadScalar(strings::StrCat("cache_size"), &cache_size));
220 cache_.resize(cache_size);
221 for (int64 i = 0; i < cache_size; i++) {
222 int64 vector_size;
223 TF_RETURN_IF_ERROR(reader->ReadScalar(
224 strings::StrCat("cache[", i, "]_size"), &vector_size));
225 cache_[i].resize(vector_size);
226 for (int64 j = 0; j < vector_size; j++) {
227 TF_RETURN_IF_ERROR(reader->ReadTensor(
228 strings::StrCat("cache[", i, "][", j, "]"), &cache_[i][j]));
229 }
230 }
231 return Status::OK();
232 }
233
234 private:
235 mutex mu_;
236 std::vector<std::vector<Tensor>> cache_ GUARDED_BY(mu_);
237 std::unique_ptr<IteratorBase> input_impl_ GUARDED_BY(mu_);
238 };
239
240 const int64 window_size_;
241 const int64 stride_;
242 const DatasetBase* const input_;
243 std::vector<PartialTensorShape> output_shapes_;
244 };
245};

248 SlideDatasetOp);
