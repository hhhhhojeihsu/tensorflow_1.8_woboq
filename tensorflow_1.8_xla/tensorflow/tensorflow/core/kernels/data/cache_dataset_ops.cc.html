26// See documentation in ../ops/dataset_ops.cc for a high-level description of
27// the following op.
28
29class CacheDatasetOp : public UnaryDatasetOpKernel {
30 public:
31 explicit CacheDatasetOp(OpKernelConstruction* ctx)
32 : UnaryDatasetOpKernel(ctx) {}
33
34 void MakeDataset(OpKernelContext* ctx, DatasetBase* input,
35 DatasetBase** output) override {
36 // Parse out the filenames tensor.
37 string filename;
38 OP_REQUIRES_OK(ctx,
39 ParseScalarArgument<string>(ctx, "filename", &filename));
40
41 if (filename.empty()) {
42 *output = new MemoryDataset(input);
43 } else {
44 *output = new FileDataset(input, filename, ctx->env());
45 }
46 }
47
48 private:
49 class FileDataset : public DatasetBase {
50 public:
51 explicit FileDataset(const DatasetBase* input, string filename, Env* env)
52 : input_(input),
53 filename_(std::move(filename)),
54 env_(env),
55 num_tensors_(input->output_dtypes().size()),
56 tensor_index_padding_size_(StringPaddingSize(num_tensors_)),
57 item_index_padding_size_(StringPaddingSize(kMaxItems)),
58 tensor_format_string_(strings::Printf("%%%zuzu_%%%zuzu",
59 item_index_padding_size_,
60 tensor_index_padding_size_)) {
61 input_->Ref();
62 DCHECK_EQ(item_index_padding_size_, 7);
63 }
64
65 ~FileDataset() override { input_->Unref(); }
66
67 std::unique_ptr<IteratorBase> MakeIterator(
68 const string& prefix) const override {
69 if (env_->FileExists(strings::StrCat(filename_, ".index")).ok()) {
70 return std::unique_ptr<IteratorBase>(new FileReaderIterator(
71 {this, strings::StrCat(prefix, "::FileReader")}));
72 } else {
73 return std::unique_ptr<IteratorBase>(new FileWriterIterator(
74 {this, strings::StrCat(prefix, "::FileWriter")}));
75 }
76 }
77
78 const DataTypeVector& output_dtypes() const override {
79 return input_->output_dtypes();
80 }
81
82 const std::vector<PartialTensorShape>& output_shapes() const override {
83 return input_->output_shapes();
84 }
85
86 string DebugString() override { return "CacheDatasetOp::FileDataset"; }
87
88 private:
89 static size_t StringPaddingSize(size_t num_tensors) {
90 return strings::Printf("%zu", num_tensors - 1).size();
91 }
92
93 string FormatName(size_t item_index, size_t tensor_index) const {
94 return strings::Printf(tensor_format_string_.c_str(), item_index,
95 tensor_index);
96 }
97
98 // FileWriterIterator passes through and caches items from the input
99 // FileDataset.
100 //
101 // This iterator is used when the cache directory is not found on disk. It
102 // creates the cache directory, and passes on the underlying iterator's
103 // elements.
104 class FileWriterIterator : public DatasetIterator<FileDataset> {
105 public:
106 explicit FileWriterIterator(const Params& params)
107 : DatasetIterator<FileDataset>(params),
108 cur_index_(0),
109 input_impl_(params.dataset->input_->MakeIterator(params.prefix)),
110 writer_(params.dataset->env_, params.dataset->filename_),
111 lockfile_(strings::StrCat(params.dataset->filename_, ".lockfile")),
112 lockfile_created_(false),
113 iteration_completed_(false) {}
114
115 Status GetNextInternal(IteratorContext* ctx,
116 std::vector<Tensor>* out_tensors,
117 bool* end_of_sequence) override {
118 mutex_lock l(mu_);
119 TF_RETURN_IF_ERROR(EnsureLockFileExists());
120 TF_RETURN_IF_ERROR(writer_.status());
121 if (cur_index_ >= kMaxItems) {
122 // As a courtesy, close the [truncated] cache file.
123 Status s = Finish();
124 if (!s.ok()) {
125 LOG(ERROR) << s;
126 }
127 return errors::InvalidArgument(
128 "Upstream iterator is producing more than ", kMaxItems,
129 " items, which is more than the cache limit.");
130 }
131
132 TF_RETURN_IF_ERROR(
133 input_impl_->GetNext(ctx, out_tensors, end_of_sequence));
134 if (*end_of_sequence && out_tensors->empty()) {
135 TF_RETURN_IF_ERROR(Finish());
136 cur_index_++;
137 return Status::OK();
138 }
139 if (out_tensors->size() != dataset()->num_tensors_) {
140 return errors::Internal(
141 "Upstream iterator returned invalid number of tensors. Expected ",
142 dataset()->num_tensors_, " got: ", out_tensors->size());
143 }
144 size_t tensor_index = 0;
145 for (const Tensor& t : *out_tensors) {
146 DCHECK_LT(tensor_index, dataset()->num_tensors_);
147 string key = dataset()->FormatName(cur_index_, tensor_index++);
148 TF_RETURN_IF_ERROR(writer_.Add(key, t));
149 }
150 if (*end_of_sequence) {
151 TF_RETURN_IF_ERROR(Finish());
152 }
153 cur_index_++;
154 return Status::OK();
155 }
156
157 private:
158 Status EnsureLockFileExists() EXCLUSIVE_LOCKS_REQUIRED(mu_) {
159 if (iteration_completed_)
160 return errors::OutOfRange(
161 "Attempting to call get_next after iteration should have "
162 "finished.");
163 if (lockfile_created_ && !iteration_completed_) return Status::OK();
164 // Perform rudimentary locking to help catch concurrent writes to the
165 // same cache files.
166 if (dataset()->env_->FileExists(lockfile_).ok()) {
167 // Attempt to read the contents of the lockfile.
168 char contents_scratch[151] = {0}; // Initialize all to 0.
169 StringPiece contents;
170 std::unique_ptr<RandomAccessFile> file;
171 if (dataset()->env_->NewRandomAccessFile(lockfile_, &file).ok()) {
172 file->Read(0, 150, &contents, contents_scratch).IgnoreError();
173 }
174 return errors::AlreadyExists(
175 "There appears to be a concurrent caching iterator running - "
176 "cache lockfile already exists ('",
177 lockfile_,
178 "'). If you are sure no other running TF computations are using "
179 "this cache prefix, delete the lockfile and re-initialize the "
180 "iterator. Lockfile contents: ",
181 contents);
182 } else {
183 // Create the file, and write some basic contents.
184 std::unique_ptr<WritableFile> lockfile;
185 TF_RETURN_IF_ERROR(
186 dataset()->env_->NewWritableFile(lockfile_, &lockfile));
187 TF_RETURN_IF_ERROR(lockfile->Append(
188 strings::StrCat("Created at: ", dataset()->env_->NowSeconds())));
189 lockfile_created_ = true;
190 return Status::OK();
191 }
192 }
193
194 Status Finish() EXCLUSIVE_LOCKS_REQUIRED(mu_) {
195 iteration_completed_ = true;
196 TF_RETURN_IF_ERROR(writer_.Finish());
197 TF_RETURN_IF_ERROR(dataset()->env_->DeleteFile(lockfile_));
198 return Status::OK();
199 }
200
201 mutex mu_;
202 size_t cur_index_ GUARDED_BY(mu_);
203 std::unique_ptr<IteratorBase> input_impl_ GUARDED_BY(mu_);
204 BundleWriter writer_ GUARDED_BY(mu_);
205 const string lockfile_;
206 bool lockfile_created_ GUARDED_BY(mu_);
207 bool iteration_completed_ GUARDED_BY(mu_);
208 }; // FileWriterIterator

210 class FileReaderIterator : public DatasetIterator<FileDataset> {
211 public:
212 explicit FileReaderIterator(const Params& params)
213 : DatasetIterator<FileDataset>(params),
214 cur_index_(0),
215 reader_(dataset()->env_, dataset()->filename_) {}
216
217 Status GetNextInternal(IteratorContext* ctx,
218 std::vector<Tensor>* out_tensors,
219 bool* end_of_sequence) override {
220 mutex_lock l(mu_);
221 *end_of_sequence = false;
222 TF_RETURN_IF_ERROR(reader_.status());
223 if (!reader_.Valid()) {
224 return errors::Internal(
225 "Cache iterator is in an invalid state. (Perhaps GetNext called "
226 "after end_of_sequence?)");
227 }
228 out_tensors->clear();
229 out_tensors->resize(dataset()->num_tensors_);
230
231 for (size_t i = 0; i < dataset()->num_tensors_; ++i) {
232 reader_.Next(); // The first entry in the table is a header entry.
233 if (!reader_.Valid()) {
234 out_tensors->clear();
235 *end_of_sequence = true;
236 return Status::OK();
237 }
238 StringPiece key = reader_.key();
239 DCHECK_EQ(key, dataset()->FormatName(cur_index_, i));
240 TF_RETURN_IF_ERROR(reader_.ReadCurrent(&(*out_tensors)[i]));
241 TF_RETURN_IF_ERROR(reader_.status());
242 }
243 cur_index_++;
244 return Status::OK();
245 }
246
247 private:
248 mutex mu_;
249 size_t cur_index_ GUARDED_BY(mu_);
250 BundleReader reader_ GUARDED_BY(mu_);
251 }; // FileReaderIterator
252
253 const DatasetBase* const input_;
254 const string filename_;
255 Env* const env_;
256 const size_t num_tensors_;
257 const size_t tensor_index_padding_size_;
258 static const size_t kMaxItems = 10000000; // 10 million
259 const size_t item_index_padding_size_;
260 const string tensor_format_string_;
261 }; // FileDataset

263 class MemoryDataset : public DatasetBase {
264 public:
265 explicit MemoryDataset(const DatasetBase* input) : input_(input) {
266 input->Ref();
267 }
268
269 ~MemoryDataset() override { input_->Unref(); }
270
271 std::unique_ptr<IteratorBase> MakeIterator(
272 const string& prefix) const override {
273 mutex_lock l(mu_);
274 if (cache_) {
275 return std::unique_ptr<IteratorBase>(new MemoryReaderIterator(
276 {this, strings::StrCat(prefix, "::MemoryReader")}, cache_.get()));
277 }
278 if (!writer_iterator_created_) {
279 writer_iterator_created_ = true;
280 return std::unique_ptr<IteratorBase>(new MemoryWriterIterator(
281 {this, strings::StrCat(prefix, "::MemoryWriter")}));
282 }
283 return std::unique_ptr<IteratorBase>(new DuplicateWriterIterator(
284 {this, strings::StrCat(prefix, "::DuplicateWriter")}));
285 }
286
287 const DataTypeVector& output_dtypes() const override {
288 return input_->output_dtypes();
289 }
290
291 const std::vector<PartialTensorShape>& output_shapes() const override {
292 return input_->output_shapes();
293 }
294
295 string DebugString() override { return "CacheDatasetOp::MemoryDataset"; }
296
297 private:
298 // MemoryWriterIterator passes through and appends items from the input
299 // dataset to its vector.
300 //
301 // This iterator is used when dataset->cache_ is null. After buffering
302 // the tensors in memory, upon exhausing the underlying iterator, they are
303 // updated into the parent dataset's cache_ pointer.
304 class MemoryWriterIterator : public DatasetIterator<MemoryDataset> {
305 public:
306 explicit MemoryWriterIterator(const Params& params)
307 : DatasetIterator<MemoryDataset>(params),
308 input_impl_(params.dataset->input_->MakeIterator(params.prefix)),
309 cache_(new std::vector<std::vector<Tensor>>) {}
310
311 ~MemoryWriterIterator() override {
312 mutex_lock l(mu_);
313 if (cache_) {
314 LOG(ERROR)
315 << "The calling iterator did not fully read the dataset we were "
316 "attempting to cache. In order to avoid unexpected truncation "
317 "of the sequence, the current [partially cached] sequence "
318 "will be dropped. This can occur if you have a sequence "
319 "similar to `dataset.cache().take(k).repeat()`. Instead, swap "
320 "the order (i.e. `dataset.take(k).cache().repeat()`)";
321 mutex_lock l2(dataset()->mu_);
322 dataset()->writer_iterator_created_ = false;
323 }
324 }
325
326 Status GetNextInternal(IteratorContext* ctx,
327 std::vector<Tensor>* out_tensors,
328 bool* end_of_sequence) override {
329 mutex_lock l(mu_);
330 TF_RETURN_IF_ERROR(
331 input_impl_->GetNext(ctx, out_tensors, end_of_sequence));
332 if (*end_of_sequence) {
333 // Guard on cache_ to not crash if GetNext is called a second time
334 // after *end_of_sequence == true
335 if (cache_) {
336 mutex_lock l(dataset()->mu_);
337 DCHECK(dataset()->writer_iterator_created_);
338 DCHECK(!dataset()->cache_);
339 cache_.swap(dataset()->cache_);
340 }
341 return Status::OK();
342 }
343 cache_->emplace_back(*out_tensors);
344 return Status::OK();
345 }
346
347 private:
348 mutex mu_;
349 std::unique_ptr<IteratorBase> input_impl_ GUARDED_BY(mu_);
350 std::unique_ptr<std::vector<std::vector<Tensor>>> cache_ GUARDED_BY(mu_);
351 }; // MemoryWriterIterator
352
353 class MemoryReaderIterator : public DatasetIterator<MemoryDataset> {
354 public:
355 explicit MemoryReaderIterator(
356 const Params& params, const std::vector<std::vector<Tensor>>* cache)
357 : DatasetIterator<MemoryDataset>(params), cache_(cache), index_(0) {
358 CHECK(cache);
359 }
360
361 Status GetNextInternal(IteratorContext* ctx,
362 std::vector<Tensor>* out_tensors,
363 bool* end_of_sequence) override {
364 mutex_lock l(mu_);
365 if (index_ < cache_->size()) {
366 const std::vector<Tensor>& cache_tensors = (*cache_)[index_];
367 out_tensors->insert(out_tensors->begin(), cache_tensors.begin(),
368 cache_tensors.end());
369 index_++;
370 *end_of_sequence = false;
371 return Status::OK();
372 } else {
373 *end_of_sequence = true;
374 return Status::OK();
375 }
376 }
377
378 private:
379 mutex mu_;
380 const std::vector<std::vector<Tensor>>* const cache_;
381 size_t index_ GUARDED_BY(mu_);
382 }; // MemoryReaderIterator
383
384 class DuplicateWriterIterator : public DatasetIterator<MemoryDataset> {
385 public:
386 explicit DuplicateWriterIterator(const Params& params)
387 : DatasetIterator<MemoryDataset>(params) {}
388
389 Status GetNextInternal(IteratorContext* ctx,
390 std::vector<Tensor>* out_tensors,
391 bool* end_of_sequence) override {
392 return errors::AlreadyExists(
393 "There appears to be a concurrent caching iterator running.");
394 }
395 }; // DuplicateWriterIterator
396
397 const DatasetBase* const input_;
398 mutable mutex mu_;
399 mutable std::unique_ptr<std::vector<std::vector<Tensor>>> cache_
400 GUARDED_BY(mu_);
401 mutable bool writer_iterator_created_ GUARDED_BY(mu_) = false;
402 }; // MemoryDataset
403}; // CacheDatasetOp
