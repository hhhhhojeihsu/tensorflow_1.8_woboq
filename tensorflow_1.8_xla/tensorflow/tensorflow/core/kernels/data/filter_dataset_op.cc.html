27// See documentation in ../ops/dataset_ops.cc for a high-level
28// description of the following op.
29
30class FilterDatasetOp : public UnaryDatasetOpKernel {
31 public:
32 explicit FilterDatasetOp(OpKernelConstruction* ctx)
33 : UnaryDatasetOpKernel(ctx),
34 graph_def_version_(ctx->graph_def_version()) {
35 OP_REQUIRES_OK(ctx, ctx->GetAttr("predicate", &func_));
36 }
37
38 void MakeDataset(OpKernelContext* ctx, DatasetBase* input,
39 DatasetBase** output) override {
40 OpInputList inputs;
41 OP_REQUIRES_OK(ctx, ctx->input_list("other_arguments", &inputs));
42 std::vector<Tensor> other_arguments;
43 other_arguments.reserve(inputs.size());
44 for (const Tensor& t : inputs) {
45 other_arguments.push_back(t);
46 }
47
48 FunctionLibraryRuntime::Handle pred_handle;
49 OP_REQUIRES_OK(ctx,
50 ctx->function_library()->Instantiate(
51 func_.name(), AttrSlice(&func_.attr()), &pred_handle));
52 auto cleanup = gtl::MakeCleanup([ctx, pred_handle]() {
53 OP_REQUIRES_OK(ctx, ctx->function_library()->ReleaseHandle(pred_handle));
54 });
55
56 const FunctionBody* pred_body =
57 ctx->function_library()->GetFunctionBody(pred_handle);
58 OP_REQUIRES(ctx, pred_body->ret_nodes.size() == 1,
59 errors::InvalidArgument(
60 "predicate function must have a single return value."));
61 Node* ret_node = pred_body->ret_nodes[0];
62 Node* ret_input_node;
63 OP_REQUIRES_OK(ctx, ret_node->input_node(0, &ret_input_node));
64 std::unique_ptr<CapturedFunction> captured_func;
65 OP_REQUIRES_OK(ctx, CapturedFunction::Create(
66 func_, std::move(other_arguments), &captured_func));
67
68 if (ret_input_node->def().op() == "_Arg") {
69 int32 index = -1;
70 OP_REQUIRES_OK(ctx, GetNodeAttr(ret_input_node->def(), "index", &index));
71 *output = new FilterTensorDataset(ctx, input, func_,
72 std::move(captured_func), index);
73 } else {
74 *output = new FilterFunctionDataset(ctx, input, func_,
75 std::move(captured_func));
76 }
77 }
78
79 private:
80 const int graph_def_version_;
81
82 class FilterDatasetBase : public GraphDatasetBase {
83 public:
84 FilterDatasetBase(OpKernelContext* ctx, const DatasetBase* input,
85 const NameAttrList& func,
86 std::unique_ptr<CapturedFunction> captured_func)
87 : GraphDatasetBase(ctx),
88 input_(input),
89 func_(func),
90 captured_func_(std::move(captured_func)) {
91 input_->Ref();
92 }
93
94 ~FilterDatasetBase() override { input_->Unref(); }
95
96 std::unique_ptr<IteratorBase> MakeIterator(
97 const string& prefix) const override {
98 return std::unique_ptr<IteratorBase>(
99 new Iterator({this, strings::StrCat(prefix, "::Filter")}));
100 }
101
102 const DataTypeVector& output_dtypes() const override {
103 return input_->output_dtypes();
104 }
105 const std::vector<PartialTensorShape>& output_shapes() const override {
106 return input_->output_shapes();
107 }
108
109 string DebugString() override { return "FilterDatasetOp::Dataset"; }
110
111 protected:
112 Status AsGraphDefInternal(OpKernelContext* ctx, DatasetGraphDefBuilder* b,
113 Node** output) const override {
114 TF_RETURN_IF_ERROR(b->AddFunction(ctx, func_.name()));
115 Node* input_graph_node;
116 TF_RETURN_IF_ERROR(b->AddParentDataset(ctx, input_, &input_graph_node));
117
118 DataTypeVector other_arguments_types;
119 other_arguments_types.reserve(captured_func_->captured_inputs().size());
120 std::vector<Node*> other_arguments;
121 other_arguments.reserve(captured_func_->captured_inputs().size());
122 for (const Tensor& t : captured_func_->captured_inputs()) {
123 Node* node;
124 TF_RETURN_IF_ERROR(b->AddTensor(t, &node));
125 other_arguments.emplace_back(node);
126 other_arguments_types.emplace_back(t.dtype());
127 }
128 AttrValue f;
129 b->BuildAttrValue(func_, &f);
130 AttrValue other_arguments_types_attr;
131 b->BuildAttrValue(other_arguments_types, &other_arguments_types_attr);
132
133 TF_RETURN_IF_ERROR(b->AddDataset(
134 this, {{0, input_graph_node}}, {{1, other_arguments}},
135 {{"predicate", f}, {"Targuments", other_arguments_types_attr}},
136 output));
137 return Status::OK();
138 }
139
140 virtual Status EvaluatePredicate(IteratorContext* ctx,
141 const std::vector<Tensor>& element,
142 bool* out_matched) const = 0;
143
144 private:
145 class Iterator : public DatasetIterator<FilterDatasetBase> {
146 public:
147 explicit Iterator(const Params& params)
148 : DatasetIterator<FilterDatasetBase>(params),
149 input_impl_(params.dataset->input_->MakeIterator(params.prefix)) {}
150
151 Status GetNextInternal(IteratorContext* ctx,
152 std::vector<Tensor>* out_tensors,
153 bool* end_of_sequence) override {
154 // NOTE(mrry): This method is thread-safe as long as
155 // `input_impl_` and `f` are thread-safe. However, if multiple
156 // threads enter this method, outputs may be observed in a
157 // non-deterministic order.
158 bool matched;
159 do {
160 {
161 tf_shared_lock l(mu_);
162 if (!input_impl_) {
163 *end_of_sequence = true;
164 return Status::OK();
165 }
166 TF_RETURN_IF_ERROR(
167 input_impl_->GetNext(ctx, out_tensors, end_of_sequence));
168 }
169 if (*end_of_sequence) {
170 mutex_lock l(mu_);
171 input_impl_.reset();
172 return Status::OK();
173 }
174
175 TF_RETURN_IF_ERROR(
176 dataset()->EvaluatePredicate(ctx, *out_tensors, &matched));
177 if (!matched) {
178 // Clear the output tensor list since it didn't match.
179 out_tensors->clear();
180 }
181 } while (!matched);
182 *end_of_sequence = false;
183 return Status::OK();
184 }
185
186 protected:
187 Status SaveInternal(IteratorStateWriter* writer) override {
188 mutex_lock l(mu_);
189 if (input_impl_)
190 TF_RETURN_IF_ERROR(SaveParent(writer, input_impl_));
191 else
192 TF_RETURN_IF_ERROR(
193 writer->WriteScalar(full_name("input_impls_empty"), ""));
194 return Status::OK();
195 }
196
197 Status RestoreInternal(IteratorContext* ctx,
198 IteratorStateReader* reader) override {
199 mutex_lock l(mu_);
200 if (reader->Contains(full_name("input_impls_empty")))
201 input_impl_.reset();
202 else
203 TF_RETURN_IF_ERROR(RestoreParent(ctx, reader, input_impl_));
204 return Status::OK();
205 }
206
207 private:
208 mutex mu_;
209 std::unique_ptr<IteratorBase> input_impl_ GUARDED_BY(mu_);
210 };
211
212 const DatasetBase* const input_;
213 const NameAttrList func_;
214
215 protected:
216 const std::unique_ptr<CapturedFunction> captured_func_;
217 };
218
219 class FilterFunctionDataset : public FilterDatasetBase {
220 public:
221 using FilterDatasetBase::FilterDatasetBase;
222
223 protected:
224 Status EvaluatePredicate(IteratorContext* ctx,
225 const std::vector<Tensor>& element,
226 bool* out_matched) const override {
227 // TODO(mrry): Avoid blocking a threadpool thread. We will need to
228 // stack-rip the iterators and use async kernels.
229 std::vector<Tensor> result;
230 TF_RETURN_IF_ERROR(
231 captured_func_->RunWithBorrowedArgs(ctx, element, &result));
232
233 if (result.size() != 1 || result[0].dtype() != DT_BOOL ||
234 result[0].NumElements() != 1) {
235 return errors::InvalidArgument(
236 "Filter predicate `f` must return a scalar bool.");
237 }
238 *out_matched = result[0].scalar<bool>()();
239 return Status::OK();
240 }
241 };
242
243 class FilterTensorDataset : public FilterDatasetBase {
244 public:
245 FilterTensorDataset(OpKernelContext* ctx, const DatasetBase* input,
246 const NameAttrList& func,
247 std::unique_ptr<CapturedFunction> captured_func,
248 int32 index)
249 : FilterDatasetBase(ctx, input, func, std::move(captured_func)),
250 index_(index) {}
251
252 protected:
253 Status EvaluatePredicate(IteratorContext* ctx,
254 const std::vector<Tensor>& element,
255 bool* out_matched) const override {
256 const Tensor& predicate = element[index_];
257 if (predicate.dtype() != DT_BOOL || predicate.NumElements() != 1) {
258 return errors::InvalidArgument(
259 "Filter predicate `f` must return a scalar bool.");
260 }
261 *out_matched = predicate.scalar<bool>()();
262 return Status::OK();
263 }
264
265 private:
266 const int32 index_;
267 };
268
269 private:
270 NameAttrList func_;
271};
