29// See documentation in ../ops/dataset_ops.cc for a high-level
30// description of the following op.
31
32class ParallelMapDatasetOp : public UnaryDatasetOpKernel {
33 public:
34 explicit ParallelMapDatasetOp(OpKernelConstruction* ctx)
35 : UnaryDatasetOpKernel(ctx),
36 graph_def_version_(ctx->graph_def_version()) {
37 OP_REQUIRES_OK(ctx, ctx->GetAttr("f", &func_));
38 OP_REQUIRES_OK(ctx, ctx->GetAttr("output_types", &output_types_));
39 OP_REQUIRES_OK(ctx, ctx->GetAttr("output_shapes", &output_shapes_));
40 }
41
42 protected:
43 void MakeDataset(OpKernelContext* ctx, DatasetBase* input,
44 DatasetBase** output) override {
45 OpInputList inputs;
46 OP_REQUIRES_OK(ctx, ctx->input_list("other_arguments", &inputs));
47 std::vector<Tensor> other_arguments;
48 other_arguments.reserve(inputs.size());
49 for (const Tensor& t : inputs) {
50 other_arguments.push_back(t);
51 }
52
53 int32 num_parallel_calls;
54 OP_REQUIRES_OK(ctx, ParseScalarArgument(ctx, "num_parallel_calls",
55 &num_parallel_calls));
56 OP_REQUIRES(ctx, num_parallel_calls > 0,
57 errors::InvalidArgument(
58 "num_parallel_calls must be greater than zero."));
59
60 std::unique_ptr<CapturedFunction> captured_func;
61 OP_REQUIRES_OK(ctx, CapturedFunction::Create(
62 func_, std::move(other_arguments), &captured_func));
63
64 *output = new Dataset(ctx, input, func_, num_parallel_calls, output_types_,
65 output_shapes_, std::move(captured_func));
66 }
67
68 private:
69 class Dataset : public GraphDatasetBase {
70 public:
71 Dataset(OpKernelContext* ctx, const DatasetBase* input,
72 const NameAttrList& func, int32 num_parallel_calls,
73 const DataTypeVector& output_types,
74 const std::vector<PartialTensorShape>& output_shapes,
75 std::unique_ptr<CapturedFunction> captured_func)
76 : GraphDatasetBase(ctx),
77 input_(input),
78 func_(func),
79 num_parallel_calls_(num_parallel_calls),
80 output_types_(output_types),
81 output_shapes_(output_shapes),
82 captured_func_(std::move(captured_func)) {
83 input_->Ref();
84 }
85
86 ~Dataset() override { input_->Unref(); }
87
88 std::unique_ptr<IteratorBase> MakeIterator(
89 const string& prefix) const override {
90 return std::unique_ptr<IteratorBase>(
91 new Iterator({this, strings::StrCat(prefix, "::ParallelMap")}));
92 }
93
94 const DataTypeVector& output_dtypes() const override {
95 return output_types_;
96 }
97
98 const std::vector<PartialTensorShape>& output_shapes() const override {
99 return output_shapes_;
100 }
101
102 string DebugString() override { return "ParallelMapDatasetOp::Dataset"; }
103
104 protected:
105 Status AsGraphDefInternal(OpKernelContext* ctx, DatasetGraphDefBuilder* b,
106 Node** output) const override {
107 // Input: input_dataset
108 Node* input_graph_node = nullptr;
109 TF_RETURN_IF_ERROR(b->AddParentDataset(ctx, input_, &input_graph_node));
110
111 // Input: other_arguments
112 DataTypeVector other_arguments_types;
113 other_arguments_types.reserve(captured_func_->captured_inputs().size());
114 std::vector<Node*> other_arguments;
115 other_arguments.reserve(captured_func_->captured_inputs().size());
116 for (const Tensor& t : captured_func_->captured_inputs()) {
117 Node* node;
118 TF_RETURN_IF_ERROR(b->AddTensor(t, &node));
119 other_arguments.emplace_back(node);
120 other_arguments_types.emplace_back(t.dtype());
121 }
122
123 // Input: num_parallel_calls
124 Node* num_parallel_calls = nullptr;
125 TF_RETURN_IF_ERROR(
126 b->AddScalar(num_parallel_calls_, &num_parallel_calls));
127
128 // Attr: f
129 TF_RETURN_IF_ERROR(b->AddFunction(ctx, func_.name()));
130 AttrValue f;
131 b->BuildAttrValue(func_, &f);
132
133 // Attr: Targuments
134 AttrValue other_arguments_types_attr;
135 b->BuildAttrValue(other_arguments_types, &other_arguments_types_attr);
136
137 TF_RETURN_IF_ERROR(b->AddDataset(
138 this,
139 {std::make_pair(0, input_graph_node),
140 std::make_pair(2, num_parallel_calls)}, // Single tensor inputs.
141 {std::make_pair(1, other_arguments)}, // Tensor list inputs.
142 {std::make_pair("f", f),
143 std::make_pair("Targuments", other_arguments_types_attr)}, // Attrs
144 output));
145 return Status::OK();
146 }
147
148 private:
149 class Iterator : public DatasetIterator<Dataset> {
150 public:
151 explicit Iterator(const Params& params)
152 : DatasetIterator<Dataset>(params),
153 input_impl_(params.dataset->input_->MakeIterator(params.prefix)),
154 invocation_results_(params.dataset->num_parallel_calls_) {}
155
156 ~Iterator() override {
157 // TODO(mrry): Replace this cancellation logic with a
158 // CancellationManager. The syntax would be more heavyweight,
159 // but it would be possible to thread a cancellation manager
160 // through the IteratorContext to upstream,
161 // potentially-blocking iterators, when we add these.
162 {
163 mutex_lock l(mu_);
164 for (size_t i = 0; i < dataset()->num_parallel_calls_; ++i) {
165 if (invocation_results_[i].notification) {
166 invocation_results_[i].notification->WaitForNotification();
167 }
168 }
169 }
170 }
171
172 Status GetNextInternal(IteratorContext* ctx,
173 std::vector<Tensor>* out_tensors,
174 bool* end_of_sequence) override {
175 mutex_lock l(mu_);
176
177 // Ensure that there are `dataset()->num_parallel_calls_`
178 // invocations of `func_` outstanding at once.
179 while (input_impl_ && (num_inputs_consumed_ - num_outputs_consumed_ <
180 dataset()->num_parallel_calls_)) {
181 InvokeFunctionLocked(ctx);
182 }
183
184 if (!input_impl_ && num_inputs_consumed_ == num_outputs_consumed_) {
185 *end_of_sequence = true;
186 return Status::OK();
187 }
188
189 // Read the next result out of `invocation_results_`, which
190 // acts as a circular buffer.
191 const size_t result_index =
192 num_outputs_consumed_ % dataset()->num_parallel_calls_;
193 InvocationResult* result = &invocation_results_[result_index];
194 *end_of_sequence = false;
195 if (result->notification) {
196 result->notification->WaitForNotification();
197 if (result->status.ok()) {
198 std::swap(*out_tensors, result->return_values);
199 }
200 }
201 ++num_outputs_consumed_;
202 if (errors::IsOutOfRange(result->status)) {
203 // `f` may deliberately raise `errors::OutOfRange` to indicate
204 // that we should terminate the iteration early.
205 *end_of_sequence = true;
206 return Status::OK();
207 } else {
208 return result->status;
209 }
210 }
211
212 protected:
213 Status SaveInternal(IteratorStateWriter* writer) override {
214 mutex_lock l(mu_);
215 if (input_impl_) {
216 TF_RETURN_IF_ERROR(SaveParent(writer, input_impl_));
217 } else {
218 TF_RETURN_IF_ERROR(
219 writer->WriteScalar(full_name("end_of_input"), ""));
220 }
221 TF_RETURN_IF_ERROR(writer->WriteScalar(full_name("num_inputs_consumed"),
222 num_inputs_consumed_));
223 TF_RETURN_IF_ERROR(writer->WriteScalar(
224 full_name("num_outputs_consumed"), num_outputs_consumed_));
225
226 for (size_t i = 0; i < dataset()->num_parallel_calls_; i++) {
227 if (invocation_results_[i].notification) {
228 invocation_results_[i].notification->WaitForNotification();
229 TF_RETURN_IF_ERROR(
230 WriteStatusLocked(writer, i, invocation_results_[i].status));
231 TF_RETURN_IF_ERROR(writer->WriteScalar(
232 full_name(strings::StrCat("invocation_results[", i, "].size")),
233 invocation_results_[i].return_values.size()));
234 for (size_t j = 0; j < invocation_results_[i].return_values.size();
235 j++) {
236 TF_RETURN_IF_ERROR(writer->WriteTensor(
237 full_name(
238 strings::StrCat("invocation_results[", i, "][", j, "]")),
239 invocation_results_[i].return_values[j]));
240 }
241 } else {
242 TF_RETURN_IF_ERROR(writer->WriteScalar(
243 full_name(strings::StrCat("invocation_results[", i, "]_empty")),
244 ""));
245 }
246 }
247
248 return Status::OK();
249 }
250
251 Status RestoreInternal(IteratorContext* ctx,
252 IteratorStateReader* reader) override {
253 mutex_lock l(mu_);
254 if (reader->Contains(full_name("end_of_input"))) {
255 input_impl_.reset();
256 } else {
257 TF_RETURN_IF_ERROR(RestoreParent(ctx, reader, input_impl_));
258 }
259 TF_RETURN_IF_ERROR(reader->ReadScalar(full_name("num_inputs_consumed"),
260 &num_inputs_consumed_));
261 TF_RETURN_IF_ERROR(reader->ReadScalar(full_name("num_outputs_consumed"),
262 &num_outputs_consumed_));
263 for (size_t i = 0; i < dataset()->num_parallel_calls_; i++) {
264 InvocationResult* result = &invocation_results_[i];
265 *result = InvocationResult();
266 if (!reader->Contains(full_name(
267 strings::StrCat("invocation_results[", i, "]_empty")))) {
268 result->notification.reset(new Notification);
269 result->notification->Notify();
270 TF_RETURN_IF_ERROR(ReadStatusLocked(reader, i, &result->status));
271 size_t num_return_values;
272 {
273 int64 size;
274 TF_RETURN_IF_ERROR(
275 reader->ReadScalar(full_name(strings::StrCat(
276 "invocation_results[", i, "].size")),
277 &size));
278 num_return_values = static_cast<size_t>(size);
279 if (num_return_values != size) {
280 return errors::InvalidArgument(strings::StrCat(
281 full_name(
282 strings::StrCat("invocation_results[", i, "].size")),
283 ": ", size, " is not a valid value of type size_t."));
284 }
285 }
286 result->return_values.reserve(num_return_values);
287 for (size_t j = 0; j < num_return_values; j++) {
288 result->return_values.emplace_back();
289 TF_RETURN_IF_ERROR(reader->ReadTensor(
290 full_name(
291 strings::StrCat("invocation_results[", i, "][", j, "]")),
292 &result->return_values.back()));
293 }
294 }
295 }
296 return Status::OK();
297 }
298
299 private:
300 struct InvocationResult {
301 Status status;
302 std::unique_ptr<Notification> notification;
303 std::vector<Tensor> return_values;
304 };
305
306 void InvokeFunctionLocked(IteratorContext* ctx)
307 EXCLUSIVE_LOCKS_REQUIRED(mu_) {
308 DCHECK(input_impl_);
309 DCHECK(num_inputs_consumed_ - num_outputs_consumed_ <
310 dataset()->num_parallel_calls_);
311
312 // The result of invoking the function will be written into the next
313 // slot in `invocation_results_`, which acts as a circular buffer.
314 const size_t result_index =
315 num_inputs_consumed_ % dataset()->num_parallel_calls_;
316 InvocationResult* result = &invocation_results_[result_index];
317 *result = InvocationResult();
318
319 // Get the next input element.
320 std::vector<Tensor> input_element;
321 bool end_of_input = false;
322 result->status =
323 input_impl_->GetNext(ctx, &input_element, &end_of_input);
324 if (end_of_input) {
325 input_impl_.reset();
326 result->status = errors::OutOfRange("");
327 } else {
328 ++num_inputs_consumed_;
329 }
330
331 if (result->status.ok()) {
332 // Call `func_(input_element)`, store the result in
333 // `result->return_values`, and notify `result->notification`
334 // to unblock a consumer.
335 result->notification.reset(new Notification);
336 dataset()->captured_func_->RunAsync(
337 ctx, std::move(input_element), &result->return_values,
338 [result, result_index](Status ret_status) {
339 result->status.Update(ret_status);
340 result->notification->Notify();
341 });
342 }
343 }
344
345 Status WriteStatusLocked(IteratorStateWriter* writer, size_t index,
346 const Status& status)
347 EXCLUSIVE_LOCKS_REQUIRED(mu_) {
348 TF_RETURN_IF_ERROR(writer->WriteScalar(
349 CodeKey(index), static_cast<int64>(status.code())));
350 if (!status.ok()) {
351 TF_RETURN_IF_ERROR(writer->WriteScalar(ErrorMessageKey(index),
352 status.error_message()));
353 }
354 return Status::OK();
355 }
356
357 Status ReadStatusLocked(IteratorStateReader* reader, size_t index,
358 Status* status) EXCLUSIVE_LOCKS_REQUIRED(mu_) {
359 int64 code_int;
360 TF_RETURN_IF_ERROR(reader->ReadScalar(CodeKey(index), &code_int));
361 error::Code code = static_cast<error::Code>(code_int);
362
363 if (code != error::Code::OK) {
364 string error_message;
365 TF_RETURN_IF_ERROR(
366 reader->ReadScalar(ErrorMessageKey(index), &error_message));
367 *status = Status(code, error_message);
368 } else {
369 *status = Status::OK();
370 }
371 return Status::OK();
372 }
373
374 string CodeKey(size_t index) {
375 return full_name(
376 strings::StrCat("invocation_results[", index, "].code"));
377 }
378
379 string ErrorMessageKey(size_t index) {
380 return full_name(
381 strings::StrCat("invocation_results[", index, "].error_message"));
382 }
383
384 mutex mu_;
385 std::unique_ptr<IteratorBase> input_impl_ GUARDED_BY(mu_);
386 std::vector<InvocationResult> invocation_results_ GUARDED_BY(mu_);
387 int64 num_inputs_consumed_ GUARDED_BY(mu_) = 0;
388 int64 num_outputs_consumed_ GUARDED_BY(mu_) = 0;
389 };
390
391 const DatasetBase* const input_;
392 const NameAttrList func_;
393 const int32 num_parallel_calls_;
394 const DataTypeVector output_types_;
395 const std::vector<PartialTensorShape> output_shapes_;
396 const std::unique_ptr<CapturedFunction> captured_func_;
397 };
398
399 const int graph_def_version_;
400 DataTypeVector output_types_;
401 std::vector<PartialTensorShape> output_shapes_;
402 NameAttrList func_;
403};
