25// See documentation in ../ops/dataset_ops.cc for a high-level
26// description of the following op.
27
28class PaddedBatchDatasetOp : public UnaryDatasetOpKernel {
29 public:
30 explicit PaddedBatchDatasetOp(OpKernelConstruction* ctx)
31 : UnaryDatasetOpKernel(ctx) {}
32
33 void MakeDataset(OpKernelContext* ctx, DatasetBase* input,
34 DatasetBase** output) override {
35 int64 batch_size;
36 OP_REQUIRES_OK(ctx,
37 ParseScalarArgument<int64>(ctx, "batch_size", &batch_size));
38 OP_REQUIRES(
39 ctx, batch_size > 0,
40 errors::InvalidArgument("Batch size must be greater than zero."));
41
42 OpInputList padded_shape_tensors;
43 OP_REQUIRES_OK(ctx,
44 ctx->input_list("padded_shapes", &padded_shape_tensors));
45 std::vector<PartialTensorShape> padded_shapes;
46 padded_shapes.reserve(padded_shape_tensors.size());
47 OP_REQUIRES(ctx,
48 padded_shape_tensors.size() == input->output_shapes().size(),
49 errors::InvalidArgument("Number of padded shapes (",
50 padded_shape_tensors.size(),
51 ") must match the number of components "
52 "in the input dataset's elements (",
53 input->output_shapes().size(), ")"));
54 for (const Tensor& padded_shape_t : padded_shape_tensors) {
55 OP_REQUIRES(ctx, TensorShapeUtils::IsVector(padded_shape_t.shape()),
56 errors::InvalidArgument("All padded shapes must be vectors"));
57 PartialTensorShape padded_shape;
58 OP_REQUIRES_OK(ctx, PartialTensorShape::MakePartialShape(
59 padded_shape_t.vec<int64>().data(),
60 padded_shape_t.NumElements(), &padded_shape));
61 padded_shapes.push_back(std::move(padded_shape));
62 }
63 OpInputList padding_values_list;
64 OP_REQUIRES_OK(ctx,
65 ctx->input_list("padding_values", &padding_values_list));
66 std::vector<Tensor> padding_values;
67 OP_REQUIRES(ctx,
68 padding_values_list.size() == input->output_shapes().size(),
69 errors::InvalidArgument(
70 "Number of padding values (", padding_values_list.size(),
71 ") must match the number of components in the input "
72 "dataset's elements (",
73 input->output_shapes().size(), ")"));
74 for (int i = 0; i < padding_values_list.size(); ++i) {
75 const Tensor& padding_value_t = padding_values_list[i];
76 OP_REQUIRES(
77 ctx, TensorShapeUtils::IsScalar(padding_value_t.shape()),
78 errors::InvalidArgument("All padding values must be scalars"));
79 OP_REQUIRES(ctx, padding_value_t.dtype() == input->output_dtypes()[i],
80 errors::InvalidArgument(
81 "Mismatched type between padding value ", i,
82 " and input dataset's component ", i, ": ",
83 DataTypeString(padding_value_t.dtype()), " vs. ",
84 DataTypeString(input->output_dtypes()[i])));
85 padding_values.push_back(tensor::DeepCopy(padding_value_t));
86 }
87
88 *output = new Dataset(ctx, batch_size, std::move(padded_shapes),
89 std::move(padding_values), input);
90 }
91
92 private:
93 class Dataset : public GraphDatasetBase {
94 public:
95 Dataset(OpKernelContext* ctx, int64 batch_size,
96 std::vector<PartialTensorShape> padded_shapes,
97 std::vector<Tensor> padding_values, const DatasetBase* input)
98 : GraphDatasetBase(ctx),
99 batch_size_(batch_size),
100 padded_shapes_(std::move(padded_shapes)),
101 padding_values_(std::move(padding_values)),
102 input_(input) {
103 input_->Ref();
104
105 // NOTE(mrry): Currently we implement "batch up to"
106 // semantics. If we could tell statically that the input dataset
107 // is infinite, then we could always report `batch_size` as the
108 // 0th dimension.
109 // TODO(mrry): Need to validate that the input shape and the
110 // padded shape are "compatible" (i.e. that padded shape is >=
111 // input shape, with both static and dynamic checks as appropriate).
112 const auto& input_shapes = input_->output_shapes();
113 output_shapes_.reserve(input_shapes.size());
114 for (size_t i = 0; i < input_shapes.size(); ++i) {
115 output_shapes_.push_back(
116 PartialTensorShape({-1}).Concatenate(padded_shapes_[i]));
117 }
118 }
119
120 ~Dataset() override { input_->Unref(); }
121
122 std::unique_ptr<IteratorBase> MakeIterator(
123 const string& prefix) const override {
124 return std::unique_ptr<IteratorBase>(
125 new Iterator({this, strings::StrCat(prefix, "::PaddedBatch")}));
126 }
127
128 const DataTypeVector& output_dtypes() const override {
129 return input_->output_dtypes();
130 }
131
132 const std::vector<PartialTensorShape>& output_shapes() const override {
133 return output_shapes_;
134 }
135
136 string DebugString() override {
137 return strings::StrCat("PaddedBatchDatasetOp(", batch_size_,
138 ")::Dataset");
139 }
140
141 protected:
142 Status AsGraphDefInternal(OpKernelContext* ctx, DatasetGraphDefBuilder* b,
143 Node** output) const override {
144 Node* input_graph_node = nullptr;
145 TF_RETURN_IF_ERROR(b->AddParentDataset(ctx, input_, &input_graph_node));
146 Node* batch_size = nullptr;
147 TF_RETURN_IF_ERROR(b->AddScalar(batch_size_, &batch_size));
148
149 std::vector<Node*> padded_shapes;
150 padded_shapes.reserve(padded_shapes_.size());
151 for (int i = 0; i < padded_shapes_.size(); i++) {
152 Node* node;
153 Tensor t(DT_INT64, TensorShape({padded_shapes_[i].dims()}));
154 for (int j = 0; j < padded_shapes_[i].dims(); j++) {
155 t.vec<int64>()(j) = padded_shapes_[i].dim_size(j);
156 }
157 TF_RETURN_IF_ERROR(b->AddTensor(t, &node));
158 padded_shapes.emplace_back(node);
159 }
160
161 std::vector<Node*> padding_values;
162 padding_values.reserve(padding_values_.size());
163 for (const Tensor& t : padding_values_) {
164 Node* node;
165 TF_RETURN_IF_ERROR(b->AddTensor(t, &node));
166 padding_values.emplace_back(node);
167 }
168
169 AttrValue output_types;
170 b->BuildAttrValue(output_dtypes(), &output_types);
171
172 AttrValue N;
173 b->BuildAttrValue<int64>(padded_shapes_.size(), &N);
174
175 TF_RETURN_IF_ERROR(
176 b->AddDataset(this, {{0, input_graph_node}, {1, batch_size}},
177 {{2, padded_shapes}, {3, padding_values}},
178 {{"Toutput_types", output_types}, {"N", N}}, output));
179 return Status::OK();
180 }

182 private:
183 // Copies element into the index^th slice of parent (in the 0th dimension).
184 //
185
186 class Iterator : public DatasetIterator<Dataset> {
187 public:
188 explicit Iterator(const Params& params)
189 : DatasetIterator<Dataset>(params),
190 input_impl_(params.dataset->input_->MakeIterator(params.prefix)) {}
191
192 Status GetNextInternal(IteratorContext* ctx,
193 std::vector<Tensor>* out_tensors,
194 bool* end_of_sequence) override {
195 // Each row of `batch_elements` is a tuple of tensors from the
196 // input iterator.
197 std::vector<std::vector<Tensor>> batch_elements;
198 {
199 mutex_lock l(mu_);
200 if (!input_impl_) {
201 *end_of_sequence = true;
202 return Status::OK();
203 } else {
204 *end_of_sequence = false;
205 batch_elements.reserve(dataset()->batch_size_);
206 for (int i = 0; i < dataset()->batch_size_ && !*end_of_sequence;
207 ++i) {
208 std::vector<Tensor> batch_element_tuple;
209 TF_RETURN_IF_ERROR(input_impl_->GetNext(ctx, &batch_element_tuple,
210 end_of_sequence));
211 if (!*end_of_sequence) {
212 batch_elements.push_back(std::move(batch_element_tuple));
213 }
214 }
215 if (*end_of_sequence) {
216 input_impl_.reset();
217 }
218 }
219 }
220
221 if (batch_elements.empty()) {
222 DCHECK(*end_of_sequence);
223 return Status::OK();
224 }
225
226 // Copy the retrieved batch elements into one output tensor
227 // per tuple component.
228 // NOTE(mrry): If the input or output sizes are statically
229 // known, we could potentially read the input values in-place
230 // into their respective slice locations. This would require a
231 // different GetNext() overload that supports zero-copy, and might
232 // make sense in an optimization pass.
233 const size_t num_tuple_components = batch_elements[0].size();
234 const int64 num_batch_elements = batch_elements.size();
235 for (size_t component_index = 0; component_index < num_tuple_components;
236 ++component_index) {
237 // 1. Determine the shape of the padded tensor.
238 TensorShape batch_component_shape({num_batch_elements});
239 const PartialTensorShape& padded_shape =
240 dataset()->padded_shapes_[component_index];
241
242 for (int dim = 0; dim < padded_shape.dims(); ++dim) {
243 if (padded_shape.dim_size(dim) == -1) {
244 batch_component_shape.AddDim(0);
245 } else {
246 batch_component_shape.AddDim(padded_shape.dim_size(dim));
247 }
248 }
249
250 for (int64 i = 0; i < num_batch_elements; ++i) {
251 const TensorShape& element_shape =
252 batch_elements[i][component_index].shape();
253 // TODO(mrry): Perform this check in the shape function if
254 // enough static information is available to do so.
255 if (element_shape.dims() != padded_shape.dims()) {
256 return errors::InvalidArgument(
257 "All elements in a batch must have the same rank as the "
258 "padded shape for component",
259 component_index, ": expected rank ", padded_shape.dims(),
260 " but got element with rank ", element_shape.dims());
261 }
262 for (int dim = 0; dim < padded_shape.dims(); ++dim) {
263 if (padded_shape.dim_size(dim) == -1) {
264 // Take the max of all batch elements in this dimension.
265 if (batch_elements[i][component_index].shape().dim_size(dim) >
266 batch_component_shape.dim_size(dim + 1)) {
267 batch_component_shape.set_dim(
268 dim + 1,
269 batch_elements[i][component_index].shape().dim_size(dim));
270 }
271 } else {
272 if (batch_elements[i][component_index].shape().dim_size(dim) >
273 batch_component_shape.dim_size(dim + 1)) {
274 return errors::DataLoss(
275 "Attempted to pad to a smaller size than the input "
276 "element.");
277 }
278 }
279 }
280 }
281
282 // 2. Copy each batch element to the appropriate location in
283 // the output component tensor.
284 Tensor batch_component(ctx->allocator({}),
285 output_dtypes()[component_index],
286 batch_component_shape);
287 TF_RETURN_IF_ERROR(batch_util::SetElementZero(
288 &batch_component, dataset()->padding_values_[component_index]));
289
290 // Build the output tuple component by copying one slice
291 // from each input element in the batch.
292 TensorShape component_shape({});
293 for (int i = 1; i < batch_component_shape.dims(); ++i) {
294 component_shape.AddDim(batch_component_shape.dim_size(i));
295 }
296 for (int64 i = 0; i < num_batch_elements; ++i) {
297 // Take the fast path if possible.
298 if (batch_elements[i][component_index].shape() == component_shape) {
299 TF_RETURN_IF_ERROR(batch_util::CopyElementToSlice(
300 batch_elements[i][component_index], &batch_component, i));
301 } else {
302 TF_RETURN_IF_ERROR(batch_util::CopyElementToLargerSlice(
303 batch_elements[i][component_index], &batch_component, i));
304 }
305 }
306 out_tensors->push_back(std::move(batch_component));
307 }
308 *end_of_sequence = false;
309 return Status::OK();
310 }
311
312 protected:
313 Status SaveInternal(IteratorStateWriter* writer) override {
314 mutex_lock l(mu_);
315 if (input_impl_)
316 TF_RETURN_IF_ERROR(SaveParent(writer, input_impl_));
317 else
318 TF_RETURN_IF_ERROR(writer->WriteScalar(full_name("exhausted"), ""));
319 return Status::OK();
320 }
321
322 Status RestoreInternal(IteratorContext* ctx,
323 IteratorStateReader* reader) override {
324 mutex_lock l(mu_);
325 if (reader->Contains(full_name("exhausted"))) {
326 input_impl_.reset();
327 } else {
328 input_impl_ = dataset()->input_->MakeIterator(prefix());
329 TF_RETURN_IF_ERROR(RestoreParent(ctx, reader, input_impl_));
330 }
331 return Status::OK();
332 }
333
334 private:
335 mutex mu_;
336 std::unique_ptr<IteratorBase> input_impl_ GUARDED_BY(mu_);
337 };
338
339 const int64 batch_size_;
340 const std::vector<PartialTensorShape> padded_shapes_;
341 const std::vector<Tensor> padding_values_;
342 const DatasetBase* const input_;
343 std::vector<PartialTensorShape> output_shapes_;
344 };
345};
