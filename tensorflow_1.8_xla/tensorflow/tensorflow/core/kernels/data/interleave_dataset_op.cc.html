28// See documentation in ../ops/dataset_ops.cc for a high-level
29// description of the following op.
30
31class InterleaveDatasetOp : public UnaryDatasetOpKernel {
32 public:
33 explicit InterleaveDatasetOp(OpKernelConstruction* ctx)
34 : UnaryDatasetOpKernel(ctx),
35 graph_def_version_(ctx->graph_def_version()) {
36 OP_REQUIRES_OK(ctx, ctx->GetAttr("f", &func_));
37 OP_REQUIRES_OK(ctx, ctx->GetAttr("output_types", &output_types_));
38 OP_REQUIRES_OK(ctx, ctx->GetAttr("output_shapes", &output_shapes_));
39 }
40
41 void MakeDataset(OpKernelContext* ctx, DatasetBase* input,
42 DatasetBase** output) override {
43 OpInputList inputs;
44 OP_REQUIRES_OK(ctx, ctx->input_list("other_arguments", &inputs));
45 std::vector<Tensor> other_arguments;
46 other_arguments.reserve(inputs.size());
47 for (const Tensor& t : inputs) {
48 other_arguments.push_back(t);
49 }
50
51 const Tensor* cycle_length_t;
52 OP_REQUIRES_OK(ctx, ctx->input("cycle_length", &cycle_length_t));
53 OP_REQUIRES(ctx, TensorShapeUtils::IsScalar(cycle_length_t->shape()),
54 errors::InvalidArgument("cycle_length must be a scalar."));
55 const int64 cycle_length = cycle_length_t->flat<int64>()(0);
56 OP_REQUIRES(
57 ctx, cycle_length > 0,
58 errors::InvalidArgument("cycle_length must be greater than zero."));
59
60 const Tensor* block_length_t;
61 OP_REQUIRES_OK(ctx, ctx->input("block_length", &block_length_t));
62 OP_REQUIRES(ctx, TensorShapeUtils::IsScalar(block_length_t->shape()),
63 errors::InvalidArgument("block_length must be a scalar."));
64 const int64 block_length = block_length_t->flat<int64>()(0);
65 OP_REQUIRES(
66 ctx, block_length > 0,
67 errors::InvalidArgument("block_length must be greater than zero."));
68
69 std::unique_ptr<CapturedFunction> captured_func;
70 OP_REQUIRES_OK(ctx, CapturedFunction::Create(
71 func_, std::move(other_arguments), &captured_func));
72
73 *output =
74 new Dataset(ctx, input, func_, std::move(captured_func), cycle_length,
75 block_length, output_types_, output_shapes_);
76 }
77
78 private:
79 class Dataset : public GraphDatasetBase {
80 public:
81 Dataset(OpKernelContext* ctx, const DatasetBase* input,
82 const NameAttrList& func,
83 std::unique_ptr<CapturedFunction> captured_func, int64 cycle_length,
84 int64 block_length, const DataTypeVector& output_types,
85 const std::vector<PartialTensorShape>& output_shapes)
86 : GraphDatasetBase(ctx),
87 input_(input),
88 func_(func),
89 captured_func_(std::move(captured_func)),
90 cycle_length_(cycle_length),
91 block_length_(block_length),
92 output_types_(output_types),
93 output_shapes_(output_shapes) {
94 input_->Ref();
95 }
96
97 ~Dataset() override { input_->Unref(); }
98
99 std::unique_ptr<IteratorBase> MakeIterator(
100 const string& prefix) const override {
101 return std::unique_ptr<IteratorBase>(
102 new Iterator({this, strings::StrCat(prefix, "::Interleave")}));
103 }
104
105 const DataTypeVector& output_dtypes() const override {
106 return output_types_;
107 }
108 const std::vector<PartialTensorShape>& output_shapes() const override {
109 return output_shapes_;
110 }
111
112 string DebugString() override { return "InterleaveDatasetOp::Dataset"; }
113
114 protected:
115 Status AsGraphDefInternal(OpKernelContext* ctx, DatasetGraphDefBuilder* b,
116 Node** output) const override {
117 TF_RETURN_IF_ERROR(b->AddFunction(ctx, func_.name()));
118 Node* input_node;
119 TF_RETURN_IF_ERROR(b->AddParentDataset(ctx, input_, &input_node));
120 Node* cycle_length_node;
121 TF_RETURN_IF_ERROR(b->AddScalar(cycle_length_, &cycle_length_node));
122 Node* block_length_node;
123 TF_RETURN_IF_ERROR(b->AddScalar(block_length_, &block_length_node));
124 DataTypeVector other_arguments_types;
125 other_arguments_types.reserve(captured_func_->captured_inputs().size());
126 std::vector<Node*> other_arguments;
127 other_arguments.reserve(captured_func_->captured_inputs().size());
128 for (const Tensor& t : captured_func_->captured_inputs()) {
129 Node* node;
130 TF_RETURN_IF_ERROR(b->AddTensor(t, &node));
131 other_arguments.emplace_back(node);
132 other_arguments_types.emplace_back(t.dtype());
133 }
134 AttrValue f;
135 b->BuildAttrValue(func_, &f);
136 AttrValue other_arguments_types_attr;
137 b->BuildAttrValue(other_arguments_types, &other_arguments_types_attr);
138
139 TF_RETURN_IF_ERROR(b->AddDataset(
140 this,
141 {{0, input_node}, {2, cycle_length_node}, {3, block_length_node}},
142 {{1, other_arguments}},
143 {{"f", f}, {"Targuments", other_arguments_types_attr}}, output));
144 return Status::OK();
145 }
146
147 private:
148 class Iterator : public DatasetIterator<Dataset> {
149 public:
150 explicit Iterator(const Params& params)
151 : DatasetIterator<Dataset>(params),
152 input_impl_(params.dataset->input_->MakeIterator(params.prefix)),
153 current_elements_(params.dataset->cycle_length_),
154 args_list_(params.dataset->cycle_length_) {}
155
156 void AdvanceToNextInCycle() EXCLUSIVE_LOCKS_REQUIRED(mu_) {
157 block_index_ = 0;
158 cycle_index_ = (cycle_index_ + 1) % dataset()->cycle_length_;
159 }
160
161 void AdvancePosition() EXCLUSIVE_LOCKS_REQUIRED(mu_) {
162 ++block_index_;
163 if (block_index_ == dataset()->block_length_) {
164 AdvanceToNextInCycle();
165 }
166 }
167
168 Status GetNextInternal(IteratorContext* ctx,
169 std::vector<Tensor>* out_tensors,
170 bool* end_of_sequence) override {
171 mutex_lock l(mu_);
172 while (!end_of_input_ || num_open_ > 0) {
173 if (current_elements_[cycle_index_]) {
174 // We are currently processing a mapped element, so try to get the
175 // next subelement.
176 bool end_of_element;
177 TF_RETURN_IF_ERROR(current_elements_[cycle_index_]->GetNext(
178 ctx, out_tensors, &end_of_element));
179 if (!end_of_element) {
180 // Produce the subelement as output.
181 AdvancePosition();
182 *end_of_sequence = false;
183 return Status::OK();
184 }
185 // We have reached the end of the current element, so move
186 // on to the next element in the cycle.
187 current_elements_[cycle_index_].reset();
188 args_list_[cycle_index_].clear();
189 --num_open_;
190 AdvanceToNextInCycle();
191 } else if (!end_of_input_) {
192 // Get the next element from the input dataset, and create
193 // an iterator from it.
194 TF_RETURN_IF_ERROR(input_impl_->GetNext(
195 ctx, &args_list_[cycle_index_], &end_of_input_));
196 if (!end_of_input_) {
197 TF_RETURN_IF_ERROR(dataset::MakeIteratorFromInputElement(
198 ctx, args_list_[cycle_index_], cycle_index_,
199 dataset()->captured_func_.get(), prefix(),
200 &current_elements_[cycle_index_]));
201 ++num_open_;
202 }
203 } else {
204 AdvanceToNextInCycle();
205 }
206 }
207
208 *end_of_sequence = true;
209 return Status::OK();
210 }
211
212 protected:
213 Status SaveInternal(IteratorStateWriter* writer) override {
214 mutex_lock l(mu_);
215 TF_RETURN_IF_ERROR(SaveParent(writer, input_impl_));
216 TF_RETURN_IF_ERROR(
217 writer->WriteScalar(full_name("cycle_index"), cycle_index_));
218 TF_RETURN_IF_ERROR(
219 writer->WriteScalar(full_name("block_index"), block_index_));
220 if (end_of_input_) {
221 TF_RETURN_IF_ERROR(
222 writer->WriteScalar(full_name("end_of_input"), ""));
223 }
224 TF_RETURN_IF_ERROR(
225 writer->WriteScalar(full_name("num_open"), num_open_));
226 TF_RETURN_IF_ERROR(SaveCurrentElements(writer));
227 return Status::OK();
228 }
229
230 Status RestoreInternal(IteratorContext* ctx,
231 IteratorStateReader* reader) override {
232 mutex_lock l(mu_);
233 TF_RETURN_IF_ERROR(RestoreParent(ctx, reader, input_impl_));
234 int64 cycle_index;
235 TF_RETURN_IF_ERROR(
236 reader->ReadScalar(full_name("cycle_index"), &cycle_index));
237 cycle_index_ = size_t(cycle_index);
238 TF_RETURN_IF_ERROR(
239 reader->ReadScalar(full_name("block_index"), &block_index_));
240 if (reader->Contains(full_name("end_of_input"))) end_of_input_ = true;
241 int64 num_open;
242 TF_RETURN_IF_ERROR(
243 reader->ReadScalar(full_name("num_open"), &num_open));
244 num_open_ = size_t(num_open);
245 TF_RETURN_IF_ERROR(RestoreCurrentElements(ctx, reader));
246 return Status::OK();
247 }
248
249 private:
250 Status SaveCurrentElements(IteratorStateWriter* writer)
251 EXCLUSIVE_LOCKS_REQUIRED(mu_) {
252 for (int idx = 0; idx < current_elements_.size(); idx++) {
253 if (current_elements_[idx]) {
254 TF_RETURN_IF_ERROR(SaveParent(writer, current_elements_[idx]));
255 TF_RETURN_IF_ERROR(writer->WriteScalar(
256 full_name(strings::StrCat("args_size[", idx, "]")),
257 args_list_[idx].size()));
258 for (int i = 0; i < args_list_[idx].size(); i++) {
259 TF_RETURN_IF_ERROR(writer->WriteTensor(
260 full_name(strings::StrCat("args_list_[", idx, "][", i, "]")),
261 args_list_[idx][i]));
262 }
263 }
264 }
265 return Status::OK();
266 }
267
268 Status RestoreCurrentElements(IteratorContext* ctx,
269 IteratorStateReader* reader)
270 EXCLUSIVE_LOCKS_REQUIRED(mu_) {
271 for (int idx = 0; idx < current_elements_.size(); idx++) {
272 if (reader->Contains(
273 full_name(strings::StrCat("args_size[", idx, "]")))) {
274 int64 args_size;
275 TF_RETURN_IF_ERROR(reader->ReadScalar(
276 full_name(strings::StrCat("args_size[", idx, "]")),
277 &args_size));
278 args_list_[idx].resize(args_size);
279 for (int i = 0; i < args_size; i++) {
280 TF_RETURN_IF_ERROR(reader->ReadTensor(
281 full_name(strings::StrCat("args_list_[", idx, "][", i, "]")),
282 &args_list_[idx][i]));
283 }
284 TF_RETURN_IF_ERROR(dataset::MakeIteratorFromInputElement(
285 ctx, args_list_[idx], idx, dataset()->captured_func_.get(),
286 prefix(), &current_elements_[idx]));
287 TF_RETURN_IF_ERROR(
288 RestoreParent(ctx, reader, current_elements_[idx]));
289 } else {
290 current_elements_[idx].reset();
291 }
292 }
293 return Status::OK();
294 }
295
296 mutex mu_;
297 const std::unique_ptr<IteratorBase> input_impl_ GUARDED_BY(mu_);
298 std::vector<std::unique_ptr<IteratorBase>> current_elements_
299 GUARDED_BY(mu_);
300 std::vector<std::vector<Tensor>> args_list_ GUARDED_BY(mu_);
301 size_t cycle_index_ GUARDED_BY(mu_) = 0;
302 int64 block_index_ GUARDED_BY(mu_) = 0;
303 bool end_of_input_ GUARDED_BY(mu_) = false;
304 size_t num_open_ GUARDED_BY(mu_) = 0;
305 };
306
307 const DatasetBase* const input_;
308 const NameAttrList func_;
309 const std::unique_ptr<CapturedFunction> captured_func_;
310 const int64 cycle_length_;
311 const int64 block_length_;
312 const DataTypeVector output_types_;
313 const std::vector<PartialTensorShape> output_shapes_;
314 };
315
316 const int graph_def_version_;
317 DataTypeVector output_types_;
318 std::vector<PartialTensorShape> output_shapes_;
319 NameAttrList func_;
320};
