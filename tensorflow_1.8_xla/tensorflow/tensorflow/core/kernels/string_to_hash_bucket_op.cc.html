16#include "tensorflow/core/kernels/string_to_hash_bucket_op.h"

18#include "tensorflow/core/lib/hash/hash.h"
19#include "tensorflow/core/platform/fingerprint.h"
20#include "tensorflow/core/platform/strong_hash.h"
21
22namespace tensorflow {
23
24// Deprecated class. It also uses `string_tensor` as Op argument instead of
25// `input`.
26class LegacyStringToHashBucketOp : public OpKernel {
27 public:
28 explicit LegacyStringToHashBucketOp(OpKernelConstruction* ctx)
29 : OpKernel(ctx) {
30 OP_REQUIRES_OK(ctx, ctx->GetAttr("num_buckets", &num_buckets_));
31 }
32
33 void Compute(OpKernelContext* context) override {
34 const Tensor* input_tensor;
35 OP_REQUIRES_OK(context, context->input("string_tensor", &input_tensor));
36 const auto& input_flat = input_tensor->flat<string>();
37
38 Tensor* output_tensor = nullptr;
39 OP_REQUIRES_OK(context,
40 context->allocate_output("output", input_tensor->shape(),
41 &output_tensor));
42 auto output_flat = output_tensor->flat<int64>();
43
44 typedef decltype(input_flat.size()) Index;
45 for (Index i = 0; i < input_flat.size(); ++i) {
46 const uint64 input_hash = Hash64(input_flat(i));
47 const uint64 bucket_id = input_hash % num_buckets_;
48 // The number of buckets is always in the positive range of int64 so is
49 // the resulting bucket_id. Casting the bucket_id from uint64 to int64 is
50 // safe.
51 output_flat(i) = static_cast<int64>(bucket_id);
52 }
53 }
54
55 private:
56 int64 num_buckets_;
57
58 TF_DISALLOW_COPY_AND_ASSIGN(LegacyStringToHashBucketOp);
59};
60
61// StringToHashBucket is deprecated in favor of StringToHashBucketFast/Strong.
62REGISTER_KERNEL_BUILDER(Name("StringToHashBucket").Device(DEVICE_CPU),
63 LegacyStringToHashBucketOp);
64
65REGISTER_KERNEL_BUILDER(Name("StringToHashBucketFast").Device(DEVICE_CPU),
66 StringToHashBucketOp<Fingerprint64>);
67
68REGISTER_KERNEL_BUILDER(Name("StringToHashBucketStrong").Device(DEVICE_CPU),
69 StringToKeyedHashBucketOp<StrongKeyedHash>);
70
71} // namespace tensorflow
