41template <typename T>
42class SerializeSparseOp : public OpKernel {
43 public:
44 explicit SerializeSparseOp(OpKernelConstruction* context)
45 : OpKernel(context) {}
46
47 bool IsExpensive() override;
48
49 Status Initialize(Tensor* result);
50 Status Serialize(const Tensor& input, T* result);
51
52 void Compute(OpKernelContext* context) override {
53 const Tensor* input_indices;
54 const Tensor* input_values;
55 const Tensor* input_shape;
56
57 OP_REQUIRES_OK(context, context->input("sparse_indices", &input_indices));
58 OP_REQUIRES_OK(context, context->input("sparse_values", &input_values));
59 OP_REQUIRES_OK(context, context->input("sparse_shape", &input_shape));
60 OP_REQUIRES(context, TensorShapeUtils::IsMatrix(input_indices->shape()),
61 errors::InvalidArgument(
62 "Input indices should be a matrix but received shape ",
63 input_indices->shape().DebugString()));
64
65 OP_REQUIRES(context, TensorShapeUtils::IsVector(input_values->shape()),
66 errors::InvalidArgument(
67 "Input values should be a vector but received shape ",
68 input_values->shape().DebugString()));
69
70 OP_REQUIRES(context, TensorShapeUtils::IsVector(input_shape->shape()),
71 errors::InvalidArgument(
72 "Input shape should be a vector but received shape ",
73 input_shape->shape().DebugString()));
74
75 Tensor serialized_sparse;
76 OP_REQUIRES_OK(context, Initialize(&serialized_sparse));
77
78 auto serialized_sparse_t = serialized_sparse.vec<T>();
79 OP_REQUIRES_OK(context, Serialize(*input_indices, &serialized_sparse_t(0)));
80 OP_REQUIRES_OK(context, Serialize(*input_values, &serialized_sparse_t(1)));
81 OP_REQUIRES_OK(context, Serialize(*input_shape, &serialized_sparse_t(2)));
82
83 context->set_output(0, serialized_sparse);
84 }
85};
86
87// NOTE(mrry): We specialize the IsExpensive() method differently for
88// the string and variant cases, because (i) the string version
89// actually performs memory copies as part of its serialization (and
90// is hence potentially expensive), and (ii) the variant version
91// performs O(1) shallow copies (and hence is much cheaper than
92// dispatching to another thread would be).
93template <>
94bool SerializeSparseOp<string>::IsExpensive() {
95 return true;
96}
97template <>
98bool SerializeSparseOp<Variant>::IsExpensive() {
99 return false;
100}

103Status SerializeSparseOp<string>::Initialize(Tensor* result) {
104 *result = Tensor(DT_STRING, TensorShape({3}));
105 return Status::OK();
106}

153template <typename T, typename U>
154class SerializeManySparseOp : public SerializeManySparseOpBase<U> {
155 public:
156 explicit SerializeManySparseOp(OpKernelConstruction* context)
157 : SerializeManySparseOpBase<U>(context) {}
158
159 void Compute(OpKernelContext* context) override {
160 const Tensor* input_indices;
161 const Tensor* input_values;
162 const Tensor* input_shape;
163 OP_REQUIRES_OK(context, context->input("sparse_indices", &input_indices));
164 OP_REQUIRES_OK(context, context->input("sparse_values", &input_values));
165 OP_REQUIRES_OK(context, context->input("sparse_shape", &input_shape));
166 OP_REQUIRES(context, TensorShapeUtils::IsMatrix(input_indices->shape()),
167 errors::InvalidArgument(
168 "Input indices should be a matrix but received shape ",
169 input_indices->shape().DebugString()));
170
171 OP_REQUIRES(context, TensorShapeUtils::IsVector(input_values->shape()),
172 errors::InvalidArgument(
173 "Input values should be a vector but received shape ",
174 input_values->shape().DebugString()));
175
176 OP_REQUIRES(context, TensorShapeUtils::IsVector(input_shape->shape()),
177 errors::InvalidArgument(
178 "Input shape should be a vector but received shape ",
179 input_shape->shape().DebugString()));
180
181 int rank = input_shape->NumElements();
182
183 OP_REQUIRES(
184 context, rank > 1,
185 errors::InvalidArgument(
186 "Rank of input SparseTensor should be > 1, but saw rank: ", rank));
187
188 TensorShape tensor_input_shape(input_shape->vec<int64>());
189 gtl::InlinedVector<int64, 8> std_order(rank);
190 std::iota(std_order.begin(), std_order.end(), 0);
191 SparseTensor input_st(*input_indices, *input_values, tensor_input_shape,
192 std_order);
193
194 auto input_shape_t = input_shape->vec<int64>();
195 const int64 N = input_shape_t(0);
196 Tensor serialized_sparse;
197 OP_REQUIRES_OK(context, this->Initialize(N, &serialized_sparse));
198 auto serialized_sparse_t = serialized_sparse.matrix<U>();
199
200 OP_REQUIRES_OK(context, input_st.IndicesValid());
201
202 // Initialize output with empty values and the proper shapes.
203 Tensor output_blank_indices(DT_INT64, {0, rank - 1});
204 U serialized_indices;
205 OP_REQUIRES_OK(context,
206 this->Serialize(output_blank_indices, &serialized_indices));
207 serialized_sparse_t.template chip<1>(0).setConstant(serialized_indices);
208
209 Tensor output_blank_values(DataTypeToEnum<T>::value, {0});
210 U serialized_values;
211 OP_REQUIRES_OK(context,
212 this->Serialize(output_blank_values, &serialized_values));
213 serialized_sparse_t.template chip<1>(1).setConstant(serialized_values);
214
215 Tensor output_shape(DT_INT64, {rank - 1});
216 auto output_shape_t = output_shape.vec<int64>();
217 for (int d = 1; d < rank; d++) output_shape_t(d - 1) = input_shape_t(d);
218 U serialized_shape;
219 OP_REQUIRES_OK(context, this->Serialize(output_shape, &serialized_shape));
220 serialized_sparse_t.template chip<1>(2).setConstant(serialized_shape);
221
222 // Get groups by minibatch dimension
223 sparse::GroupIterable minibatch = input_st.group({0});
224 for (const auto& subset : minibatch) {
225 const int64 b = subset.group()[0];
226 OP_REQUIRES(
227 context, b > -1 && b < N,
228 errors::InvalidArgument(
229 "Received unexpected column 0 value in input SparseTensor: ", b,
230 " < 0 or >= N (= ", N, ")"));
231
232 const auto indices = subset.indices();
233 const auto values = subset.values<T>();
234 const int64 num_entries = values.size();
235
236 Tensor output_indices = Tensor(DT_INT64, {num_entries, rank - 1});
237 Tensor output_values = Tensor(DataTypeToEnum<T>::value, {num_entries});
238
239 auto output_indices_t = output_indices.matrix<int64>();
240 auto output_values_t = output_values.vec<T>();
241
242 for (int i = 0; i < num_entries; ++i) {
243 for (int d = 1; d < rank; ++d) {
244 output_indices_t(i, d - 1) = indices(i, d);
245 }
246 output_values_t(i) = values(i);
247 }
248
249 OP_REQUIRES_OK(
250 context, this->Serialize(output_indices, &serialized_sparse_t(b, 0)));
251 OP_REQUIRES_OK(
252 context, this->Serialize(output_values, &serialized_sparse_t(b, 1)));
253 }
254
255 context->set_output(0, serialized_sparse);
256 }
257};

309template <typename T>
310class DeserializeSparseOp : public OpKernel {
311 public:
312 explicit DeserializeSparseOp(OpKernelConstruction* context)
313 : OpKernel(context) {
314 OP_REQUIRES_OK(context, context->GetAttr("dtype", &dtype_));
315 }
316
317 void Compute(OpKernelContext* context) override {
318 const Tensor& serialized_sparse = context->input(0);
319 const int ndims = serialized_sparse.shape().dims();
320
321 OP_REQUIRES(
322 context, ndims > 0,
323 errors::InvalidArgument("Serialized sparse should have non-zero rank ",
324 serialized_sparse.shape().DebugString()));
325
326 OP_REQUIRES(context, serialized_sparse.shape().dim_size(ndims - 1) == 3,
327 errors::InvalidArgument(
328 "Serialized sparse should have 3 as the last dimension ",
329 serialized_sparse.shape().DebugString()));
330
331 int num_sparse_tensors = 1;
332 for (int i = 0; i < ndims - 1; ++i) {
333 num_sparse_tensors *= serialized_sparse.shape().dim_size(i);
334 }
335
336 OP_REQUIRES(
337 context, num_sparse_tensors > 0,
338 errors::InvalidArgument(
339 "Serialized sparse should have at least 1 serialized tensor, "
340 "but has a zero dimension ",
341 serialized_sparse.shape().DebugString()));
342
343 if (num_sparse_tensors == 0 && serialized_sparse.shape().dims() == 1) {
344 // Special case with a single sparse tensor. We can avoid data
345 // motion in the Concat and Reshape.
346 const auto& serialized_sparse_t = serialized_sparse.vec<T>();
347
348 Tensor output_indices;
349 Tensor output_values;
350 Tensor output_shape;
351 OP_REQUIRES_OK(context,
352 this->GetAndValidateSparseTensor(
353 serialized_sparse_t(0), serialized_sparse_t(1),
354 serialized_sparse_t(2), dtype_, 0 /* index */,
355 &output_indices, &output_values, &output_shape));
356 context->set_output(0, output_indices);
357 context->set_output(1, output_values);
358 context->set_output(2, output_shape);
359 return;
360 }
361
362 std::vector<Tensor> indices;
363 std::vector<Tensor> values;
364 TensorShape shape;
365 indices.reserve(num_sparse_tensors);
366 values.reserve(num_sparse_tensors);
367
368 const auto& serialized_sparse_t = serialized_sparse.flat_inner_dims<T, 2>();
369 for (int i = 0; i < num_sparse_tensors; ++i) {
370 Tensor output_indices;
371 Tensor output_values;
372 Tensor output_shape;
373 OP_REQUIRES_OK(context,
374 this->GetAndValidateSparseTensor(
375 serialized_sparse_t(i, 0), serialized_sparse_t(i, 1),
376 serialized_sparse_t(i, 2), dtype_, i, &output_indices,
377 &output_values, &output_shape));
378 int64 num_entries = output_indices.dim_size(0);
379 int rank = output_indices.dim_size(1);
380
381 // Now we expand each SparseTensors' indices and shape by
382 // prefixing a dimension
383 Tensor expanded_indices(DT_INT64, TensorShape({num_entries, 1 + rank}));
384 const auto& output_indices_t = output_indices.matrix<int64>();
385 auto expanded_indices_t = expanded_indices.matrix<int64>();
386 expanded_indices_t.chip<1>(0).setZero();
387 Eigen::DSizes<Eigen::DenseIndex, 2> indices_start(0, 1);
388 Eigen::DSizes<Eigen::DenseIndex, 2> indices_sizes(num_entries, rank);
389 expanded_indices_t.slice(indices_start, indices_sizes) = output_indices_t;
390
391 Tensor expanded_shape(DT_INT64, TensorShape({1 + rank}));
392 const auto& output_shape_t = output_shape.vec<int64>();
393 auto expanded_shape_t = expanded_shape.vec<int64>();
394 expanded_shape_t(0) = 1;
395 std::copy_n(&output_shape_t(0), rank, &expanded_shape_t(1));
396
397 TensorShape expanded_tensor_shape(expanded_shape.vec<int64>());
398
399 indices.push_back(expanded_indices);
400 values.push_back(output_values);
401 if (i == 0) {
402 shape = expanded_tensor_shape;
403 } else {
404 OP_REQUIRES(
405 context, shape.dims() == expanded_tensor_shape.dims(),
406 errors::InvalidArgument(
407 "Inconsistent shape across SparseTensors: rank prior to "
408 "SparseTensor[",
409 i, "] was: ", shape.dims() - 1, " but rank of SparseTensor[", i,
410 "] is: ", expanded_tensor_shape.dims() - 1));
411 for (int j = 1; j < shape.dims(); ++j) {
412 // NOTE(mrry): For compatibility with the implementations of
413 // DeserializeManySparse, and many ops that generate
414 // SparseTensors to batch that do not have a fixed
415 // dense_shape (e.g. `tf.parse_single_example()`), we
416 // compute the maximum in each dimension to find the
417 // smallest dense_shape that bounds all of the input
418 // SparseTensors.
419 shape.set_dim(j, std::max(shape.dim_size(j),
420 expanded_tensor_shape.dim_size(j)));
421 }
422 }
423 }
424
425 // Dimension 0 is the primary dimension.
426 int rank = shape.dims();
427 gtl::InlinedVector<int64, 8> std_order(rank);
428 std::iota(std_order.begin(), std_order.end(), 0);
429
430 std::vector<SparseTensor> tensors;
431 tensors.reserve(num_sparse_tensors);
432 for (int i = 0; i < num_sparse_tensors; ++i) {
433 tensors.emplace_back(indices[i], values[i], shape, std_order);
434 }
435
436 gtl::optional<SparseTensor> maybe_output;
437#define HANDLE_TYPE(T) \
438 case DataTypeToEnum<T>::value: { \
439 maybe_output = SparseTensor::Concat<T>(tensors); \
440 break; \
441 }
442
443 switch (dtype_) {
444 TF_CALL_ALL_TYPES(HANDLE_TYPE);
445 TF_CALL_QUANTIZED_TYPES(HANDLE_TYPE);
446#undef HANDLE_TYPE
447 default:
448 OP_REQUIRES(context, false,
449 errors::Unimplemented(
450 "DeserializeSparse Unhandled data type: ", dtype_));
451 }
452 DCHECK(maybe_output);
453 SparseTensor& output = maybe_output.value();
454
455 // Compute the input shape for the reshape operation.
456 Tensor input_shape(DT_INT64, TensorShape({output.dims()}));
457 std::copy_n(output.shape().data(), output.dims(),
458 input_shape.vec<int64>().data());
459
460 // Compute the target shape for the reshape operation.
461 Tensor target_shape(DT_INT64, TensorShape({ndims + output.dims() - 2}));
462 for (int i = 0; i < ndims - 1; ++i) {
463 target_shape.vec<int64>()(i) = serialized_sparse.shape().dim_size(i);
464 }
465 for (int i = 0; i < output.dims() - 1; ++i) {
466 target_shape.vec<int64>()(i + ndims - 1) = output.shape().data()[i + 1];
467 }
468
469 Tensor output_indices;
470 Tensor output_shape;
471 Reshape(context, output.indices(), input_shape, target_shape,
472 0 /* output indices index */, 2 /* output shape index */);
473 context->set_output(1, output.values());
474 }
475
476 protected:
477 Status Deserialize(const T& serialized, Tensor* result);
478
479 Status GetAndValidateSparseTensor(
480 const T& serialized_indices, const T& serialized_values,
481 const T& serialized_shape, DataType values_dtype, int index,
482 Tensor* output_indices, Tensor* output_values, Tensor* output_shape) {
483 // Deserialize and validate the indices.
484 TF_RETURN_IF_ERROR(this->Deserialize(serialized_indices, output_indices));
485 if (!TensorShapeUtils::IsMatrix(output_indices->shape())) {
486 return errors::InvalidArgument(
487 "Expected serialized_sparse[", index,
488 ", 0] to represent an index matrix but received shape ",
489 output_indices->shape().DebugString());
490 }
491 int64 num_entries = output_indices->dim_size(0);
492 int rank = output_indices->dim_size(1);
493
494 // Deserialize and validate the values.
495 TF_RETURN_IF_ERROR(this->Deserialize(serialized_values, output_values));
496 if (!TensorShapeUtils::IsVector(output_values->shape())) {
497 return errors::InvalidArgument(
498 "Expected serialized_sparse[", index,
499 ", 1] to represent a values vector but received shape ",
500 output_values->shape().DebugString());
501 }
502 if (values_dtype != output_values->dtype()) {
503 return errors::InvalidArgument(
504 "Requested SparseTensor of type ", DataTypeString(values_dtype),
505 " but SparseTensor[", index,
506 "].values.dtype() == ", DataTypeString(output_values->dtype()));
507 }
508 if (num_entries != output_values->dim_size(0)) {
509 return errors::InvalidArgument(
510 "Expected row counts of SparseTensor[", index,
511 "].indices and SparseTensor[", index,
512 "].values to match but they do not: ", num_entries, " vs. ",
513 output_values->dim_size(0));
514 }
515
516 // Deserialize and validate the shape.
517 TF_RETURN_IF_ERROR(this->Deserialize(serialized_shape, output_shape));
518 if (!TensorShapeUtils::IsVector(output_shape->shape())) {
519 return errors::InvalidArgument(
520 "Expected serialized_sparse[", index,
521 ", 1] to be a shape vector but its shape is ",
522 output_shape->shape().DebugString());
523 }
524 if (rank != output_shape->dim_size(0)) {
525 return errors::InvalidArgument("Expected column counts of SparseTensor[",
526 index,
527 "].indices to match size of SparseTensor[",
528 index, "].shape but they do not: ", rank,
529 " vs. ", output_shape->dim_size(0));
530 }
531 return Status::OK();
532 }
533
534 DataType dtype_;
535};
536
537template <>
538Status DeserializeSparseOp<string>::Deserialize(const string& serialized,
539 Tensor* result) {
540 TensorProto proto;
541 if (!ParseProtoUnlimited(&proto, serialized)) {
542 return errors::InvalidArgument("Could not parse serialized proto");
543 }
544 Tensor tensor;
545 if (!tensor.FromProto(proto)) {
546 return errors::InvalidArgument("Could not construct tensor from proto");
547 }
548 *result = tensor;
549 return Status::OK();
550}
