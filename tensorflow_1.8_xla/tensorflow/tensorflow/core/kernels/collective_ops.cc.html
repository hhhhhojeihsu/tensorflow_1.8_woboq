24class CollectiveOpKernel : public AsyncOpKernel {
25 public:
26 explicit CollectiveOpKernel(OpKernelConstruction* c) : AsyncOpKernel(c) {}
27
28 // A string encoding instance, frame and iter to be handed off to
29 // the implementation for use in generating RecvBuf keys.
30 string GetCollectiveKey(OpKernelContext* c) {
31 return strings::StrCat(col_params_.instance.instance_key, ":",
32 c->frame_iter().frame_id, ":",
33 c->frame_iter().iter_id);
34 }
35
36 // Returns false if calling invocation of ComputeAsync should return
37 // immediately.
38 bool CanProceedWithCompute(OpKernelContext* c, CollectiveExecutor* col_exec,
39 const DoneCallback& done) {
40 if (col_params_.group.group_size >
41 col_params_.instance.device_names.size()) {
42 // This is the first invocation: Finish initializing col_params_.
43 // Call in a blockable thread because it's not guaranteed that
44 // this call cannot block.
45 c->env()->SchedClosure([this, c, done, col_exec]() {
46 col_exec->CompleteParamsAsync(c->device()->name(), &col_params_,
47 c->cancellation_manager(),
48 [this, c, done](const Status& s) {
49 if (s.ok()) {
50 ComputeAsync(c, done);
51 } else {
52 c->SetStatus(s);
53 done();
54 }
55 });
56 });
57 return false;
58 }
59 return true;
60 }
61
62 CollectiveParams col_params_;
63};

65class CollectiveReduceOpKernel : public CollectiveOpKernel {
66 public:
67 explicit CollectiveReduceOpKernel(OpKernelConstruction* c)
68 : CollectiveOpKernel(c) {
69 col_params_.instance.type = REDUCTION_COLLECTIVE;
70 OP_REQUIRES_OK(c, c->GetAttr("group_size", &col_params_.group.group_size));
71 OP_REQUIRES_OK(c, c->GetAttr("group_key", &col_params_.group.group_key));
72 OP_REQUIRES_OK(
73 c, c->GetAttr("instance_key", &col_params_.instance.instance_key));
74 OP_REQUIRES_OK(
75 c, c->GetAttr("subdiv_offsets",
76 &col_params_.instance.impl_details.subdiv_offsets));
77 string merge_op_name;
78 OP_REQUIRES_OK(c, c->GetAttr("merge_op", &merge_op_name));
79 OP_REQUIRES(c, merge_op_name == "Add" || merge_op_name == "Mul",
80 errors::InvalidArgument(
81 "merge_op must be one of {\"Add\", \"Mul\"} but got ",
82 merge_op_name));
83 string final_op_name;
84 OP_REQUIRES_OK(c, c->GetAttr("final_op", &final_op_name));
85 OP_REQUIRES(c, final_op_name == "Id" || final_op_name == "Div",
86 errors::InvalidArgument(
87 "final_op must be one of {\"Id\", \"Div\"} but got ",
88 final_op_name));
89 OP_REQUIRES_OK(c, c->GetAttr("T", &col_params_.instance.data_type));
90
91 const NodeDef& real_node = c->def();
92 col_params_.name = strings::StrCat(real_node.name(), ": Reduce(",
93 merge_op_name, ",", final_op_name, ")");
94 col_params_.group.device_type = c->device_type();
95
96 // Find the OpKernels by name, type and device type.
97 NodeDef sub_node;
98 // The merge_op takes two inputs
99 sub_node.add_input(real_node.input(0));
100 sub_node.add_input(real_node.input(0));
101 sub_node.set_device(real_node.device());
102 SetAttrValue(col_params_.instance.data_type,
103 &(*sub_node.mutable_attr())["T"]);
104 col_params_.merge_op = BuildOpKernel(c, merge_op_name, &sub_node);
105 col_params_.final_op = BuildOpKernel(c, final_op_name, &sub_node);
106 }
107
108 std::unique_ptr<OpKernel> BuildOpKernel(OpKernelConstruction* c,
109 const string& name,
110 NodeDef* sub_node) {
111 std::unique_ptr<OpKernel> k;
112 if (name.empty() || name == "Id") return k;
113 sub_node->set_name(name);
114 sub_node->set_op(name);
115 Status status;
116 k = CreateOpKernel(c->device_type(), c->device(),
117 c->device()->GetAllocator(AllocatorAttributes()),
118 *sub_node, c->graph_def_version(), &status);
119 if (!status.ok()) {
120 c->CtxFailureWithWarning(errors::Internal("Failed to build OpKernel for ",
121 name, " : ",
122 status.error_message()));
123 }
124 return k;
125 }
126
127 void ComputeAsync(OpKernelContext* c, DoneCallback done) override {
128 CollectiveExecutor* col_exec = c->collective_executor();
129 OP_REQUIRES_ASYNC(
130 c, col_exec,
131 errors::Internal(
132 "Failed to get CollectiveExecutor from OpKernelContext for Op ",
133 col_params_.name),
134 done);
135 if (!CanProceedWithCompute(c, col_exec, done)) return;
136 // Allocate the output tensor, trying to reuse the input.
137 Tensor* output = nullptr;
138 OP_REQUIRES_OK_ASYNC(c,
139 c->forward_input_or_allocate_output(
140 {0}, 0, c->input(0).shape(), &output),
141 done);
142
143 auto actual_done = [c, col_exec, done](const Status& s) {
144 OP_REQUIRES_OK_ASYNC(c, s, done);
145 done();
146 };
147 col_exec->ExecuteAsync(c, col_params_, GetCollectiveKey(c), actual_done);
148 }
149
150 private:
151 TF_DISALLOW_COPY_AND_ASSIGN(CollectiveReduceOpKernel);
152};
153
154REGISTER_KERNEL_BUILDER(Name("CollectiveReduce").Device(DEVICE_CPU),
155 CollectiveReduceOpKernel);
156REGISTER_KERNEL_BUILDER(Name("CollectiveReduce").Device(DEVICE_GPU),
157 CollectiveReduceOpKernel);

159class CollectiveBcastSendOpKernel : public CollectiveOpKernel {
160 public:
161 explicit CollectiveBcastSendOpKernel(OpKernelConstruction* c)
162 : CollectiveOpKernel(c) {
163 col_params_.instance.type = BROADCAST_COLLECTIVE;
164 OP_REQUIRES_OK(c, c->GetAttr("group_size", &col_params_.group.group_size));
165 OP_REQUIRES_OK(c, c->GetAttr("group_key", &col_params_.group.group_key));
166 OP_REQUIRES_OK(
167 c, c->GetAttr("instance_key", &col_params_.instance.instance_key));
168 OP_REQUIRES_OK(c, c->GetAttr("T", &col_params_.instance.data_type));
169 OP_REQUIRES_OK(c, c->GetAttr("shape", &shape_));
170 col_params_.is_source = true;
171 col_params_.instance.impl_details.subdiv_offsets = {0};
172
173 col_params_.name =
174 strings::StrCat(name(), ": Broadcast(", col_params_.is_source, ")");
175 col_params_.group.device_type = c->device_type();
176 }
177
178 void ComputeAsync(OpKernelContext* c, DoneCallback done) override {
179 CollectiveExecutor* col_exec = c->collective_executor();
180 OP_REQUIRES_ASYNC(
181 c, col_exec,
182 errors::Internal(
183 "Failed to get CollectiveExecutor from OpKernelContext for Op ",
184 col_params_.name),
185 done);
186 if (!CanProceedWithCompute(c, col_exec, done)) return;
187 OP_REQUIRES_ASYNC(
188 c, shape_.IsSameSize(c->input(0).shape()),
189 errors::Internal("Declared shape of op ", col_params_.name,
190 " does not match shape of input"),
191 done);
192 // Allocate the output Tensor, trying to reuse the input.
193 Tensor* output = nullptr;
194 OP_REQUIRES_OK_ASYNC(
195 c, c->forward_input_or_allocate_output({0}, 0, shape_, &output), done);
196
197 auto actual_done = [c, col_exec, done](const Status& s) {
198 OP_REQUIRES_OK_ASYNC(c, s, done);
199 done();
200 };
201 col_exec->ExecuteAsync(c, col_params_, GetCollectiveKey(c), actual_done);
202 }
203
204 private:
205 TensorShape shape_;
206
207 TF_DISALLOW_COPY_AND_ASSIGN(CollectiveBcastSendOpKernel);
208};

215class CollectiveBcastRecvOpKernel : public CollectiveOpKernel {
216 public:
217 explicit CollectiveBcastRecvOpKernel(OpKernelConstruction* c)
218 : CollectiveOpKernel(c) {
219 col_params_.instance.type = BROADCAST_COLLECTIVE;
220 OP_REQUIRES_OK(c, c->GetAttr("group_size", &col_params_.group.group_size));
221 OP_REQUIRES_OK(c, c->GetAttr("group_key", &col_params_.group.group_key));
222 OP_REQUIRES_OK(
223 c, c->GetAttr("instance_key", &col_params_.instance.instance_key));
224 OP_REQUIRES_OK(c, c->GetAttr("T", &col_params_.instance.data_type));
225 OP_REQUIRES_OK(c, c->GetAttr("shape", &shape_));
226 col_params_.is_source = false;
227 col_params_.instance.impl_details.subdiv_offsets = {0};
228
229 col_params_.name =
230 strings::StrCat(name(), ": Broadcast(", col_params_.is_source, ")");
231 col_params_.group.device_type = c->device_type();
232 }
233
234 void ComputeAsync(OpKernelContext* c, DoneCallback done) override {
235 CollectiveExecutor* col_exec = c->collective_executor();
236 OP_REQUIRES_ASYNC(
237 c, col_exec,
238 errors::Internal(
239 "Failed to get CollectiveExecutor from OpKernelContext for Op ",
240 col_params_.name),
241 done);
242 if (!CanProceedWithCompute(c, col_exec, done)) return;
243 // No input, so must allocate output.
244 Tensor* output = nullptr;
245 OP_REQUIRES_OK_ASYNC(c, c->allocate_output(0, shape_, &output), done);
246
247 auto actual_done = [c, col_exec, done](const Status& s) {
248 OP_REQUIRES_OK_ASYNC(c, s, done);
249 done();
250 };
251 col_exec->ExecuteAsync(c, col_params_, GetCollectiveKey(c), actual_done);
252 }
253
254 private:
255 TensorShape shape_;
256
257 TF_DISALLOW_COPY_AND_ASSIGN(CollectiveBcastRecvOpKernel);
258};
