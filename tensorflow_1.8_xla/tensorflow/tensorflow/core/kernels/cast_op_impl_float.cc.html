20namespace tensorflow {
21
22typedef Eigen::ThreadPoolDevice CPUDevice;
23typedef Eigen::GpuDevice GPUDevice;
24
25std::function<void(OpKernelContext*, const Tensor&, Tensor*)>
26GetCpuCastFromFloat(DataType dst_dtype) {
27 CURRY_TYPES3(CAST_CASE, CPUDevice, float);
28 return nullptr;
29}
30
31#if GOOGLE_CUDA
32std::function<void(OpKernelContext*, const Tensor&, Tensor*)>
33GetGpuCastFromFloat(DataType dst_dtype) {
34 CURRY_TYPES3(CAST_CASE, GPUDevice, float);
35 return nullptr;
36}
37#endif // GOOGLE_CUDA
38
39#ifdef TENSORFLOW_USE_SYCL
40typedef Eigen::SyclDevice SYCLDevice;
41std::function<void(OpKernelContext*, const Tensor&, Tensor*)>
42GetSyclCastFromFloat(DataType dst_dtype) {
43 CURRY_TYPES3_NO_HALF(CAST_CASE, SYCLDevice, float);
44 return nullptr;
45}
46#endif // TENSORFLOW_USE_SYCL
47
48} // namespace tensorflow
