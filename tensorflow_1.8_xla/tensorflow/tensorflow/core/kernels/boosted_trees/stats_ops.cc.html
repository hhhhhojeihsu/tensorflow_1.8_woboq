27class BoostedTreesCalculateBestGainsPerFeatureOp : public OpKernel {
28 public:
29 explicit BoostedTreesCalculateBestGainsPerFeatureOp(
30 OpKernelConstruction* const context)
31 : OpKernel(context) {
32 OP_REQUIRES_OK(context, context->GetAttr("max_splits", &max_splits_));
33 OP_REQUIRES_OK(context, context->GetAttr("num_features", &num_features_));
34 }
35
36 void Compute(OpKernelContext* const context) override {
37 // node_id_range
38 const Tensor* node_id_range_t;
39 OP_REQUIRES_OK(context, context->input("node_id_range", &node_id_range_t));
40 const auto node_id_range = node_id_range_t->vec<int32>();
41 const int32 node_id_first = node_id_range(0); // inclusive
42 const int32 node_id_last = node_id_range(1); // exclusive
43 // stats_summary_list
44 OpInputList stats_summary_list;
45 OP_REQUIRES_OK(context, context->input_list("stats_summary_list",
46 &stats_summary_list));
47 const int64 num_buckets = stats_summary_list[0].dim_size(1);
48 std::vector<TTypes<float, 3>::ConstTensor> stats_summary;
49 stats_summary.reserve(stats_summary_list.size());
50 for (const auto& tensor : stats_summary_list) {
51 stats_summary.emplace_back(tensor.tensor<float, 3>());
52 }
53 const Tensor* l1_t;
54 OP_REQUIRES_OK(context, context->input("l1", &l1_t));
55 const auto l1 = l1_t->scalar<float>()();
56 const Tensor* l2_t;
57 OP_REQUIRES_OK(context, context->input("l2", &l2_t));
58 const auto l2 = l2_t->scalar<float>()();
59 const Tensor* tree_complexity_t;
60 OP_REQUIRES_OK(context,
61 context->input("tree_complexity", &tree_complexity_t));
62 const auto tree_complexity = tree_complexity_t->scalar<float>()();
63 const Tensor* min_node_weight_t;
64 OP_REQUIRES_OK(context,
65 context->input("min_node_weight", &min_node_weight_t));
66 const auto min_node_weight = min_node_weight_t->scalar<float>()();
67
68 // Allocate output lists of tensors:
69 OpOutputList output_node_ids_list;
70 OP_REQUIRES_OK(
71 context, context->output_list("node_ids_list", &output_node_ids_list));
72 OpOutputList output_gains_list;
73 OP_REQUIRES_OK(context,
74 context->output_list("gains_list", &output_gains_list));
75 OpOutputList output_thresholds_list;
76 OP_REQUIRES_OK(context, context->output_list("thresholds_list",
77 &output_thresholds_list));
78 OpOutputList output_left_node_contribs_list;
79 OP_REQUIRES_OK(context,
80 context->output_list("left_node_contribs_list",
81 &output_left_node_contribs_list));
82 OpOutputList output_right_node_contribs_list;
83 OP_REQUIRES_OK(context,
84 context->output_list("right_node_contribs_list",
85 &output_right_node_contribs_list));
86
87 // Get the best split info per node for each feature.
88 for (int feature_idx = 0; feature_idx < num_features_; ++feature_idx) {
89 std::vector<float> cum_grad;
90 std::vector<float> cum_hess;
91 cum_grad.reserve(num_buckets);
92 cum_hess.reserve(num_buckets);
93
94 std::vector<int32> output_node_ids;
95 std::vector<float> output_gains;
96 std::vector<int32> output_thresholds;
97 std::vector<float> output_left_node_contribs;
98 std::vector<float> output_right_node_contribs;
99 for (int node_id = node_id_first; node_id < node_id_last; ++node_id) {
100 // Calculate gains.
101 cum_grad.clear();
102 cum_hess.clear();
103 float total_grad = 0.0;
104 float total_hess = 0.0;
105 for (int bucket = 0; bucket < num_buckets; ++bucket) {
106 // TODO(nponomareva): Consider multi-dimensional gradients/hessians.
107 total_grad += stats_summary[feature_idx](node_id, bucket, 0);
108 total_hess += stats_summary[feature_idx](node_id, bucket, 1);
109 cum_grad.push_back(total_grad);
110 cum_hess.push_back(total_hess);
111 }
112 // Check if node has enough of average hessian.
113 if (total_hess < min_node_weight) {
114 // Do not split the node because not enough avg hessian.
115 continue;
116 }
117 float best_gain = std::numeric_limits<float>::lowest();
118 float best_bucket = 0;
119 float best_contrib_for_left = 0.0;
120 float best_contrib_for_right = 0.0;
121 // Parent gain.
122 float parent_gain;
123 float unused;
124 CalculateWeightsAndGains(total_grad, total_hess, l1, l2, &unused,
125 &parent_gain);
126
127 for (int bucket = 0; bucket < num_buckets; ++bucket) {
128 const float cum_grad_bucket = cum_grad[bucket];
129 const float cum_hess_bucket = cum_hess[bucket];
130 // Left child.
131 float contrib_for_left;
132 float gain_for_left;
133 CalculateWeightsAndGains(cum_grad_bucket, cum_hess_bucket, l1, l2,
134 &contrib_for_left, &gain_for_left);
135 // Right child.
136 float contrib_for_right;
137 float gain_for_right;
138 CalculateWeightsAndGains(total_grad - cum_grad_bucket,
139 total_hess - cum_hess_bucket, l1, l2,
140 &contrib_for_right, &gain_for_right);
141
142 if (gain_for_left + gain_for_right > best_gain) {
143 best_gain = gain_for_left + gain_for_right;
144 best_bucket = bucket;
145 best_contrib_for_left = contrib_for_left;
146 best_contrib_for_right = contrib_for_right;
147 }
148 } // for bucket
149 output_node_ids.push_back(node_id);
150 // Remove the parent gain for the parent node.
151 output_gains.push_back(best_gain - parent_gain);
152 output_thresholds.push_back(best_bucket);
153 output_left_node_contribs.push_back(best_contrib_for_left);
154 output_right_node_contribs.push_back(best_contrib_for_right);
155 } // for node_id
156 const int num_nodes = output_node_ids.size();
157 // output_node_ids
158 Tensor* output_node_ids_t;
159 OP_REQUIRES_OK(context,
160 output_node_ids_list.allocate(feature_idx, {num_nodes},
161 &output_node_ids_t));
162 auto output_node_ids_vec = output_node_ids_t->vec<int32>();
163 // output_gains
164 Tensor* output_gains_t;
165 OP_REQUIRES_OK(context, output_gains_list.allocate(
166 feature_idx, {num_nodes}, &output_gains_t));
167 auto output_gains_vec = output_gains_t->vec<float>();
168 // output_thresholds
169 Tensor* output_thresholds_t;
170 OP_REQUIRES_OK(context,
171 output_thresholds_list.allocate(feature_idx, {num_nodes},
172 &output_thresholds_t));
173 auto output_thresholds_vec = output_thresholds_t->vec<int32>();
174 // output_left_node_contribs
175 Tensor* output_left_node_contribs_t;
176 OP_REQUIRES_OK(context, output_left_node_contribs_list.allocate(
177 feature_idx, {num_nodes, 1},
178 &output_left_node_contribs_t));
179 auto output_left_node_contribs_matrix =
180 output_left_node_contribs_t->matrix<float>();
181 // output_right_node_contribs
182 Tensor* output_right_node_contribs_t;
183 OP_REQUIRES_OK(context, output_right_node_contribs_list.allocate(
184 feature_idx, {num_nodes, 1},
185 &output_right_node_contribs_t));
186 auto output_right_node_contribs_matrix =
187 output_right_node_contribs_t->matrix<float>();
188 // Sets output tensors from vectors.
189 for (int i = 0; i < num_nodes; ++i) {
190 output_node_ids_vec(i) = output_node_ids[i];
191 // Adjust the gains to penalize by tree complexity.
192 output_gains_vec(i) = output_gains[i] - tree_complexity;
193 output_thresholds_vec(i) = output_thresholds[i];
194 // Logits are 1-dimensional for now.
195 // TODO(nponomareva): Consider multi-dimensional logits.
196 output_left_node_contribs_matrix(i, 0) = output_left_node_contribs[i];
197 output_right_node_contribs_matrix(i, 0) = output_right_node_contribs[i];
198 }
199 } // for f
200 }
201
202 private:
203 void CalculateWeightsAndGains(const float g, const float h, const float l1,
204 const float l2, float* weight, float* gain) {
205 //
206 // The formula for weight is -(g+l1*sgn(w))/(H+l2), for gain it is
207 // (g+l1*sgn(w))^2/(h+l2).
208 // This is because for each leaf we optimize
209 // 1/2(h+l2)*w^2+g*w+l1*abs(w)
210 float g_with_l1 = g;
211 // Apply L1 regularization.
212 // 1) Assume w>0 => w=-(g+l1)/(h+l2)=> g+l1 < 0 => g < -l1
213 // 2) Assume w<0 => w=-(g-l1)/(h+l2)=> g-l1 > 0 => g > l1
214 // For g from (-l1, l1), thus there is no solution => set to 0.
215 if (l1 > 0) {
216 if (g > l1) {
217 g_with_l1 -= l1;
218 } else if (g < -l1) {
219 g_with_l1 += l1;
220 } else {
221 *weight = 0.0;
222 *gain = 0.0;
223 return;
224 }
225 }
226 // Apply L2 regularization.
227 if (h + l2 <= kEps) {
228 // Avoid division by 0 or infinitesimal.
229 *weight = 0;
230 *gain = 0;
231 } else {
232 *weight = -g_with_l1 / (h + l2);
233 *gain = -g_with_l1 * (*weight);
234 }
235 }
236
237 int max_splits_;
238 int num_features_;
239};
240
241REGISTER_KERNEL_BUILDER(
242 Name("BoostedTreesCalculateBestGainsPerFeature").Device(DEVICE_CPU),
243 BoostedTreesCalculateBestGainsPerFeatureOp);

245class BoostedTreesMakeStatsSummaryOp : public OpKernel {
246 public:
247 explicit BoostedTreesMakeStatsSummaryOp(OpKernelConstruction* const context)
248 : OpKernel(context) {
249 OP_REQUIRES_OK(context, context->GetAttr("max_splits", &max_splits_));
250 OP_REQUIRES_OK(context, context->GetAttr("num_buckets", &num_buckets_));
251 OP_REQUIRES_OK(context, context->GetAttr("num_features", &num_features_));
252 }
253
254 void Compute(OpKernelContext* const context) override {
255 // node_ids
256 const Tensor* node_ids_t;
257 OP_REQUIRES_OK(context, context->input("node_ids", &node_ids_t));
258 const auto node_ids = node_ids_t->vec<int32>();
259 // gradients
260 const Tensor* gradients_t;
261 OP_REQUIRES_OK(context, context->input("gradients", &gradients_t));
262 const auto gradients = gradients_t->matrix<float>();
263 // hessians
264 const Tensor* hessians_t;
265 OP_REQUIRES_OK(context, context->input("hessians", &hessians_t));
266 const auto hessians = hessians_t->matrix<float>();
267 // bucketized_features
268 OpInputList bucketized_features_list;
269 OP_REQUIRES_OK(context, context->input_list("bucketized_features_list",
270 &bucketized_features_list));
271 std::vector<tensorflow::TTypes<int32>::ConstVec> bucketized_features;
272 bucketized_features.reserve(num_features_);
273 for (const Tensor& tensor : bucketized_features_list) {
274 bucketized_features.emplace_back(tensor.vec<int32>());
275 }
276
277 // Infer batch size.
278 const int64 batch_size = node_ids_t->dim_size(0);
279 // Allocate output stats tensor (Rank 4).
280 Tensor* output_stats_summary_t = nullptr;
281 OP_REQUIRES_OK(context, context->allocate_output(
282 "stats_summary",
283 {num_features_, max_splits_, num_buckets_, 2},
284 &output_stats_summary_t));
285 auto output_stats_summary = output_stats_summary_t->tensor<float, 4>();
286 output_stats_summary.setZero();
287
288 // Partition by node, and then bucketize.
289 for (int feature_idx = 0; feature_idx < num_features_; ++feature_idx) {
290 const auto& features = bucketized_features[feature_idx];
291 for (int i = 0; i < batch_size; ++i) {
292 const int32 node = node_ids(i);
293 const int32 bucket = features(i);
294 output_stats_summary(feature_idx, node, bucket, 0) += gradients(i, 0);
295 output_stats_summary(feature_idx, node, bucket, 1) += hessians(i, 0);
296 }
297 }
298 }
299
300 private:
301 int max_splits_;
302 int num_buckets_;
303 int num_features_;
304};
305
306REGISTER_KERNEL_BUILDER(Name("BoostedTreesMakeStatsSummary").Device(DEVICE_CPU),
307 BoostedTreesMakeStatsSummaryOp);
