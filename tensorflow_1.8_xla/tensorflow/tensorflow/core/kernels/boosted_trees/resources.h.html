26// A StampedResource is a resource that has a stamp token associated with it.
27// Before reading from or applying updates to the resource, the stamp should
28// be checked to verify that the update is not stale.
29class StampedResource : public ResourceBase {
30 public:
31 StampedResource() : stamp_(-1) {}
32
33 bool is_stamp_valid(int64 stamp) const { return stamp_ == stamp; }
34
35 int64 stamp() const { return stamp_; }
36 void set_stamp(int64 stamp) { stamp_ = stamp; }
37
38 private:
39 int64 stamp_;
40};
41
42// Keep a tree ensemble in memory for efficient evaluation and mutation.
43class BoostedTreesEnsembleResource : public StampedResource {
44 public:
45 // Constructor.
46 BoostedTreesEnsembleResource()
47 : tree_ensemble_(
48 protobuf::Arena::CreateMessage<boosted_trees::TreeEnsemble>(
49 &arena_)) {}
50
51 string DebugString() override {
52 return strings::StrCat("TreeEnsemble[size=", tree_ensemble_->trees_size(),
53 "]");
54 }
55
56 bool InitFromSerialized(const string& serialized, const int64 stamp_token) {
57 CHECK_EQ(stamp(), -1) << "Must Reset before Init.";
58 if (ParseProtoUnlimited(tree_ensemble_, serialized)) {
59 set_stamp(stamp_token);
60 return true;
61 }
62 return false;
63 }
64
65 string SerializeAsString() const {
66 return tree_ensemble_->SerializeAsString();
67 }
68
69 int32 num_trees() const { return tree_ensemble_->trees_size(); }
70
71 // Find the next node to which the example (specified by index_in_batch)
72 // traverses down from the current node indicated by tree_id and node_id.
73 // Args:
74 // tree_id: the index of the tree in the ensemble.
75 // node_id: the index of the node within the tree.
76 // index_in_batch: the index of the example within the batch (relevant to
77 // the index of the row to read in each bucketized_features).
78 // bucketized_features: vector of feature Vectors.
79 int32 next_node(
80 const int32 tree_id, const int32 node_id, const int32 index_in_batch,
81 const std::vector<TTypes<int32>::ConstVec>& bucketized_features) const;
82
83 float node_value(const int32 tree_id, const int32 node_id) const;
84
85 int32 GetNumLayersGrown(const int32 tree_id) const {
86 DCHECK_LT(tree_id, tree_ensemble_->trees_size());
87 return tree_ensemble_->tree_metadata(tree_id).num_layers_grown();
88 }
89
90 void SetNumLayersGrown(const int32 tree_id, int32 new_num_layers) const {
91 DCHECK_LT(tree_id, tree_ensemble_->trees_size());
92 tree_ensemble_->mutable_tree_metadata(tree_id)->set_num_layers_grown(
93 new_num_layers);
94 }
95
96 void UpdateLastLayerNodesRange(const int32 node_range_start,
97 int32 node_range_end) const {
98 tree_ensemble_->mutable_growing_metadata()->set_last_layer_node_start(
99 node_range_start);
100 tree_ensemble_->mutable_growing_metadata()->set_last_layer_node_end(
101 node_range_end);
102 }
103
104 void GetLastLayerNodesRange(int32* node_range_start,
105 int32* node_range_end) const {
106 *node_range_start =
107 tree_ensemble_->growing_metadata().last_layer_node_start();
108 *node_range_end = tree_ensemble_->growing_metadata().last_layer_node_end();
109 }
110
111 int64 GetNumNodes(const int32 tree_id) {
112 DCHECK_LT(tree_id, tree_ensemble_->trees_size());
113 return tree_ensemble_->trees(tree_id).nodes_size();
114 }
115
116 void UpdateGrowingMetadata() const;
117
118 int32 GetNumLayersAttempted() {
119 return tree_ensemble_->growing_metadata().num_layers_attempted();
120 }
121
122 bool is_leaf(const int32 tree_id, const int32 node_id) const {
123 DCHECK_LT(tree_id, tree_ensemble_->trees_size());
124 DCHECK_LT(node_id, tree_ensemble_->trees(tree_id).nodes_size());
125 const auto& node = tree_ensemble_->trees(tree_id).nodes(node_id);
126 return node.node_case() == boosted_trees::Node::kLeaf;
127 }
128
129 int32 feature_id(const int32 tree_id, const int32 node_id) const {
130 const auto node = tree_ensemble_->trees(tree_id).nodes(node_id);
131 DCHECK_EQ(node.node_case(), boosted_trees::Node::kBucketizedSplit);
132 return node.bucketized_split().feature_id();
133 }
134
135 int32 bucket_threshold(const int32 tree_id, const int32 node_id) const {
136 const auto node = tree_ensemble_->trees(tree_id).nodes(node_id);
137 DCHECK_EQ(node.node_case(), boosted_trees::Node::kBucketizedSplit);
138 return node.bucketized_split().threshold();
139 }
140
141 int32 left_id(const int32 tree_id, const int32 node_id) const {
142 const auto node = tree_ensemble_->trees(tree_id).nodes(node_id);
143 DCHECK_EQ(node.node_case(), boosted_trees::Node::kBucketizedSplit);
144 return node.bucketized_split().left_id();
145 }
146
147 int32 right_id(const int32 tree_id, const int32 node_id) const {
148 const auto node = tree_ensemble_->trees(tree_id).nodes(node_id);
149 DCHECK_EQ(node.node_case(), boosted_trees::Node::kBucketizedSplit);
150 return node.bucketized_split().right_id();
151 }
152
153 // Add a tree to the ensemble and returns a new tree_id.
154 int32 AddNewTree(const float weight);
155
156 // Grows the tree by adding a split and leaves.
157 void AddBucketizedSplitNode(const int32 tree_id, const int32 node_id,
158 const int32 feature_id, const int32 threshold,
159 const float gain, const float left_contrib,
160 const float right_contrib, int32* left_node_id,
161 int32* right_node_id);
162
163 // Retrieves tree weights and returns as a vector.
164 // It involves a copy, so should be called only sparingly (like once per
165 // iteration, not per example).
166 std::vector<float> GetTreeWeights() const {
167 return {tree_ensemble_->tree_weights().begin(),
168 tree_ensemble_->tree_weights().end()};
169 }
170
171 float GetTreeWeight(const int32 tree_id) const {
172 return tree_ensemble_->tree_weights(tree_id);
173 }
174
175 float IsTreeFinalized(const int32 tree_id) const {
176 DCHECK_LT(tree_id, tree_ensemble_->trees_size());
177 return tree_ensemble_->tree_metadata(tree_id).is_finalized();
178 }
179
180 float IsTreePostPruned(const int32 tree_id) const {
181 DCHECK_LT(tree_id, tree_ensemble_->trees_size());
182 return tree_ensemble_->tree_metadata(tree_id)
183 .post_pruned_nodes_meta_size() > 0;
184 }
185
186 void SetIsFinalized(const int32 tree_id, const bool is_finalized) {
187 DCHECK_LT(tree_id, tree_ensemble_->trees_size());
188 return tree_ensemble_->mutable_tree_metadata(tree_id)->set_is_finalized(
189 is_finalized);
190 }
191
192 // Sets the weight of i'th tree.
193 void SetTreeWeight(const int32 tree_id, const float weight) {
194 DCHECK_GE(tree_id, 0);
195 DCHECK_LT(tree_id, num_trees());
196 tree_ensemble_->set_tree_weights(tree_id, weight);
197 }
198
199 // Resets the resource and frees the protos in arena.
200 // Caller needs to hold the mutex lock while calling this.
201 virtual void Reset();
202
203 void PostPruneTree(const int32 current_tree);
204
205 // For a given node, returns the id in a pruned tree, as well as correction
206 // to the cached prediction that should be applied. If tree was not
207 // post-pruned, current_node_id will be equal to initial_node_id and logit
208 // update will be equal to zero.
209 void GetPostPruneCorrection(const int32 tree_id, const int32 initial_node_id,
210 int32* current_node_id,
211 float* logit_update) const;
212 mutex* get_mutex() { return &mu_; }
213
214 private:
215 // Helper method to check whether a node is a terminal node in that it
216 // only has leaf nodes as children.
217 bool IsTerminalSplitNode(const int32 tree_id, const int32 node_id) const;
218
219 // For each pruned node, finds the leaf where it finally ended up and
220 // calculates the total update from that pruned node prediction.
221 void CalculateParentAndLogitUpdate(
222 const int32 start_node_id,
223 const std::vector<std::pair<int32, float>>& nodes_change,
224 int32* parent_id, float* change) const;
225
226 // Helper method to collect the information to be used to prune some nodes in
227 // the tree.
228 void RecursivelyDoPostPrunePreparation(
229 const int32 tree_id, const int32 node_id,
230 std::vector<int32>* nodes_to_delete,
231 std::vector<std::pair<int32, float>>* nodes_meta);
232
233 protected:
234 protobuf::Arena arena_;
235 mutex mu_;
236 boosted_trees::TreeEnsemble* tree_ensemble_;
237};
