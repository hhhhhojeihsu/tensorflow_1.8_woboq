38// The Op used during training time to get the predictions so far with the
39// current ensemble being built.
40// Expect some logits are cached from the previous step and passed through
41// to be reused.
42class BoostedTreesTrainingPredictOp : public OpKernel {
43 public:
44 explicit BoostedTreesTrainingPredictOp(OpKernelConstruction* const context)
45 : OpKernel(context) {
46 OP_REQUIRES_OK(context, context->GetAttr("num_bucketized_features",
47 &num_bucketized_features_));
48 OP_REQUIRES_OK(context,
49 context->GetAttr("logits_dimension", &logits_dimension_));
50 OP_REQUIRES(context, logits_dimension_ == 1,
51 errors::InvalidArgument(
52 "Currently only one dimensional outputs are supported."));
53 }
54
55 void Compute(OpKernelContext* const context) override {
56 BoostedTreesEnsembleResource* resource;
57 // Get the resource.
58 OP_REQUIRES_OK(context, LookupResource(context, HandleFromInput(context, 0),
59 &resource));
60 // Release the reference to the resource once we're done using it.
61 core::ScopedUnref unref_me(resource);
62
63 // Get the inputs.
64 OpInputList bucketized_features_list;
65 OP_REQUIRES_OK(context, context->input_list("bucketized_features",
66 &bucketized_features_list));
67 std::vector<tensorflow::TTypes<int32>::ConstVec> batch_bucketized_features;
68 batch_bucketized_features.reserve(bucketized_features_list.size());
69 for (const Tensor& tensor : bucketized_features_list) {
70 batch_bucketized_features.emplace_back(tensor.vec<int32>());
71 }
72 const int batch_size = batch_bucketized_features[0].size();
73
74 const Tensor* cached_tree_ids_t;
75 OP_REQUIRES_OK(context,
76 context->input("cached_tree_ids", &cached_tree_ids_t));
77 const auto cached_tree_ids = cached_tree_ids_t->vec<int32>();
78
79 const Tensor* cached_node_ids_t;
80 OP_REQUIRES_OK(context,
81 context->input("cached_node_ids", &cached_node_ids_t));
82 const auto cached_node_ids = cached_node_ids_t->vec<int32>();
83
84 // Allocate outputs.
85 Tensor* output_partial_logits_t = nullptr;
86 OP_REQUIRES_OK(context,
87 context->allocate_output("partial_logits",
88 {batch_size, logits_dimension_},
89 &output_partial_logits_t));
90 auto output_partial_logits = output_partial_logits_t->matrix<float>();
91
92 Tensor* output_tree_ids_t = nullptr;
93 OP_REQUIRES_OK(context, context->allocate_output("tree_ids", {batch_size},
94 &output_tree_ids_t));
95 auto output_tree_ids = output_tree_ids_t->vec<int32>();
96
97 Tensor* output_node_ids_t = nullptr;
98 OP_REQUIRES_OK(context, context->allocate_output("node_ids", {batch_size},
99 &output_node_ids_t));
100 auto output_node_ids = output_node_ids_t->vec<int32>();
101
102 // Indicate that the latest tree was used.
103 const int32 latest_tree = resource->num_trees() - 1;
104
105 if (latest_tree < 0) {
106 // Ensemble was empty. Nothing changes.
107 output_node_ids = cached_node_ids;
108 output_tree_ids = cached_tree_ids;
109 // All the predictions are zeros.
110 output_partial_logits.setZero();
111 } else {
112 output_tree_ids.setConstant(latest_tree);
113 auto do_work = [&resource, &batch_bucketized_features, &cached_tree_ids,
114 &cached_node_ids, &output_partial_logits,
115 &output_node_ids, batch_size,
116 latest_tree](int32 start, int32 end) {
117 for (int32 i = start; i < end; ++i) {
118 int32 tree_id = cached_tree_ids(i);
119 int32 node_id = cached_node_ids(i);
120 float partial_tree_logit = 0.0;
121
122 // If the tree was pruned, returns the node id into which the
123 // current_node_id was pruned, as well the correction of the cached
124 // logit prediction.
125 resource->GetPostPruneCorrection(tree_id, node_id, &node_id,
126 &partial_tree_logit);
127
128 // Logic in the loop adds the cached node value again if it is a leaf.
129 // If it is not a leaf anymore we need to subtract the old node's
130 // value. The following logic handles both of these cases.
131 partial_tree_logit -= resource->node_value(tree_id, node_id);
132 float partial_all_logit = 0.0;
133 while (true) {
134 if (resource->is_leaf(tree_id, node_id)) {
135 partial_tree_logit += resource->node_value(tree_id, node_id);
136
137 // Tree is done
138 partial_all_logit +=
139 resource->GetTreeWeight(tree_id) * partial_tree_logit;
140 partial_tree_logit = 0.0;
141 // Stop if it was the latest tree.
142 if (tree_id == latest_tree) {
143 break;
144 }
145 // Move onto other trees.
146 ++tree_id;
147 node_id = 0;
148 } else {
149 node_id = resource->next_node(tree_id, node_id, i,
150 batch_bucketized_features);
151 }
152 }
153 output_node_ids(i) = node_id;
154 output_partial_logits(i, 0) = partial_all_logit;
155 }
156 };
157 // 30 is the magic number. The actual value might be a function of (the
158 // number of layers) * (cpu cycles spent on each layer), but this value
159 // would work for many cases. May be tuned later.
160 const int64 cost = 30;
161 thread::ThreadPool* const worker_threads =
162 context->device()->tensorflow_cpu_worker_threads()->workers;
163 Shard(worker_threads->NumThreads(), worker_threads, batch_size,
164 /*cost_per_unit=*/cost, do_work);
165 }
166 }
167
168 private:
169 int32 logits_dimension_; // the size of the output prediction vector.
170 int32 num_bucketized_features_; // Indicates the number of features.
171};
172
173REGISTER_KERNEL_BUILDER(Name("BoostedTreesTrainingPredict").Device(DEVICE_CPU),
174 BoostedTreesTrainingPredictOp);

176// The Op to get the predictions at the evaluation/inference time.
177class BoostedTreesPredictOp : public OpKernel {
178 public:
179 explicit BoostedTreesPredictOp(OpKernelConstruction* const context)
180 : OpKernel(context) {
181 OP_REQUIRES_OK(context, context->GetAttr("num_bucketized_features",
182 &num_bucketized_features_));
183 OP_REQUIRES_OK(context,
184 context->GetAttr("logits_dimension", &logits_dimension_));
185 OP_REQUIRES(context, logits_dimension_ == 1,
186 errors::InvalidArgument(
187 "Currently only one dimensional outputs are supported."));
188 }
189
190 void Compute(OpKernelContext* const context) override {
191 BoostedTreesEnsembleResource* resource;
192 // Get the resource.
193 OP_REQUIRES_OK(context, LookupResource(context, HandleFromInput(context, 0),
194 &resource));
195 // Release the reference to the resource once we're done using it.
196 core::ScopedUnref unref_me(resource);
197
198 // Get the inputs.
199 OpInputList bucketized_features_list;
200 OP_REQUIRES_OK(context, context->input_list("bucketized_features",
201 &bucketized_features_list));
202 std::vector<tensorflow::TTypes<int32>::ConstVec> batch_bucketized_features;
203 batch_bucketized_features.reserve(bucketized_features_list.size());
204 for (const Tensor& tensor : bucketized_features_list) {
205 batch_bucketized_features.emplace_back(tensor.vec<int32>());
206 }
207 const int batch_size = batch_bucketized_features[0].size();
208
209 // Allocate outputs.
210 Tensor* output_logits_t = nullptr;
211 OP_REQUIRES_OK(context, context->allocate_output(
212 "logits", {batch_size, logits_dimension_},
213 &output_logits_t));
214 auto output_logits = output_logits_t->matrix<float>();
215
216 const int32 latest_tree = resource->num_trees() - 1;
217
218 auto do_work = [&resource, &batch_bucketized_features, &output_logits,
219 batch_size, latest_tree](int32 start, int32 end) {
220 for (int32 i = start; i < end; ++i) {
221 float tree_logit = 0.0;
222 int32 tree_id = 0;
223 int32 node_id = 0;
224 while (true) {
225 if (resource->is_leaf(tree_id, node_id)) {
226 tree_logit += resource->GetTreeWeight(tree_id) *
227 resource->node_value(tree_id, node_id);
228
229 // Stop if it was the latest tree.
230 if (tree_id == latest_tree) {
231 break;
232 }
233 // Move onto other trees.
234 ++tree_id;
235 node_id = 0;
236 } else {
237 node_id = resource->next_node(tree_id, node_id, i,
238 batch_bucketized_features);
239 }
240 }
241 output_logits(i, 0) = tree_logit;
242 }
243 };
244 // 10 is the magic number. The actual number might depend on (the number of
245 // layers in the trees) and (cpu cycles spent on each layer), but this
246 // value would work for many cases. May be tuned later.
247 const int64 cost = (latest_tree + 1) * 10;
248 thread::ThreadPool* const worker_threads =
249 context->device()->tensorflow_cpu_worker_threads()->workers;
250 Shard(worker_threads->NumThreads(), worker_threads, batch_size,
251 /*cost_per_unit=*/cost, do_work);
252 }
253
254 private:
255 int32
256 logits_dimension_; // Indicates the size of the output prediction vector.
257 int32 num_bucketized_features_; // Indicates the number of features.
258};
259
260REGISTER_KERNEL_BUILDER(Name("BoostedTreesPredict").Device(DEVICE_CPU),
261 BoostedTreesPredictOp);
