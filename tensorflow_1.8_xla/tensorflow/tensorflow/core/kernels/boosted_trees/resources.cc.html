24int32 BoostedTreesEnsembleResource::next_node(
25 const int32 tree_id, const int32 node_id, const int32 index_in_batch,
26 const std::vector<TTypes<int32>::ConstVec>& bucketized_features) const {
27 DCHECK_LT(tree_id, tree_ensemble_->trees_size());
28 DCHECK_LT(node_id, tree_ensemble_->trees(tree_id).nodes_size());
29 const auto& node = tree_ensemble_->trees(tree_id).nodes(node_id);
30 DCHECK_EQ(node.node_case(), boosted_trees::Node::kBucketizedSplit);
31 const auto& split = node.bucketized_split();
32 if (bucketized_features[split.feature_id()](index_in_batch) <=
33 split.threshold()) {
34 return split.left_id();
35 } else {
36 return split.right_id();
37 }
38}

40float BoostedTreesEnsembleResource::node_value(const int32 tree_id,
41 const int32 node_id) const {
42 DCHECK_LT(tree_id, tree_ensemble_->trees_size());
43 DCHECK_LT(node_id, tree_ensemble_->trees(tree_id).nodes_size());
44 const auto& node = tree_ensemble_->trees(tree_id).nodes(node_id);
45 if (node.node_case() == boosted_trees::Node::kLeaf) {
46 return node.leaf().scalar();
47 } else {
48 return node.metadata().original_leaf().scalar();
49 }
50}

52void BoostedTreesEnsembleResource::UpdateGrowingMetadata() const {
53 tree_ensemble_->mutable_growing_metadata()->set_num_layers_attempted(
54 tree_ensemble_->growing_metadata().num_layers_attempted() + 1);
55
56 const int n_trees = num_trees();
57
58 if (n_trees <= 0 ||
59 // Checks if we are building the first layer of the dummy empty tree
60 ((n_trees == 1 || IsTreeFinalized(n_trees - 2)) &&
61 (tree_ensemble_->trees(n_trees - 1).nodes_size() == 1))) {
62 tree_ensemble_->mutable_growing_metadata()->set_num_trees_attempted(
63 tree_ensemble_->growing_metadata().num_trees_attempted() + 1);
64 }
65}

67// Add a tree to the ensemble and returns a new tree_id.
68int32 BoostedTreesEnsembleResource::AddNewTree(const float weight) {
69 const int32 new_tree_id = tree_ensemble_->trees_size();
70 auto* node = tree_ensemble_->add_trees()->add_nodes();
71 node->mutable_leaf()->set_scalar(0.0);
72 tree_ensemble_->add_tree_weights(weight);
73 tree_ensemble_->add_tree_metadata();
74
75 return new_tree_id;
76}

78void BoostedTreesEnsembleResource::AddBucketizedSplitNode(
79 const int32 tree_id, const int32 node_id, const int32 feature_id,
80 const int32 threshold, const float gain, const float left_contrib,
81 const float right_contrib, int32* left_node_id, int32* right_node_id) {
82 auto* tree = tree_ensemble_->mutable_trees(tree_id);
83 auto* node = tree->mutable_nodes(node_id);
84 DCHECK_EQ(node->node_case(), boosted_trees::Node::kLeaf);
85 float prev_node_value = node->leaf().scalar();
86 *left_node_id = tree->nodes_size();
87 *right_node_id = *left_node_id + 1;
88 auto* left_node = tree->add_nodes();
89 auto* right_node = tree->add_nodes();
90 if (node_id != 0) {
91 // Save previous leaf value if it is not the first leaf in the tree.
92 node->mutable_metadata()->mutable_original_leaf()->Swap(
93 node->mutable_leaf());
94 }
95 node->mutable_metadata()->set_gain(gain);
96 auto* new_split = node->mutable_bucketized_split();
97 new_split->set_feature_id(feature_id);
98 new_split->set_threshold(threshold);
99 new_split->set_left_id(*left_node_id);
100 new_split->set_right_id(*right_node_id);
101 // TODO(npononareva): this is LAYER-BY-LAYER boosting; add WHOLE-TREE.
102 left_node->mutable_leaf()->set_scalar(prev_node_value + left_contrib);
103 right_node->mutable_leaf()->set_scalar(prev_node_value + right_contrib);
104}

106void BoostedTreesEnsembleResource::Reset() {
107 // Reset stamp.
108 set_stamp(-1);
109
110 // Clear tree ensemle.
111 arena_.Reset();
112 CHECK_EQ(0, arena_.SpaceAllocated());
113 tree_ensemble_ =
114 protobuf::Arena::CreateMessage<boosted_trees::TreeEnsemble>(&arena_);
115}

117void BoostedTreesEnsembleResource::PostPruneTree(const int32 current_tree) {
118 // No-op if tree is empty.
119 auto* tree = tree_ensemble_->mutable_trees(current_tree);
120 int32 num_nodes = tree->nodes_size();
121 if (num_nodes == 0) {
122 return;
123 }
124
125 std::vector<int32> nodes_to_delete;
126 // If a node was pruned, we need to save the change of the prediction from
127 // this node to its parent, as well as the parent id.
128 std::vector<std::pair<int32, float>> nodes_changes;
129 nodes_changes.reserve(num_nodes);
130 for (int32 i = 0; i < num_nodes; ++i) {
131 nodes_changes.emplace_back(i, 0.0);
132 }
133 // Prune the tree recursively starting from the root. Each node that has
134 // negative gain and only leaf children will be pruned recursively up from
135 // the bottom of the tree. This method returns the list of nodes pruned, and
136 // updates the nodes in the tree not to refer to those pruned nodes.
137 RecursivelyDoPostPrunePreparation(current_tree, 0, &nodes_to_delete,
138 &nodes_changes);
139
140 if (nodes_to_delete.empty()) {
141 // No pruning happened, and no post-processing needed.
142 return;
143 }
144
145 // Sort node ids so they are in asc order.
146 std::sort(nodes_to_delete.begin(), nodes_to_delete.end());
147
148 // We need to
149 // - update split left and right children ids with new indices
150 // - actually remove the nodes that need to be removed
151 // - save the information about pruned node so we could recover the
152 // predictions from cache. Build a map for old node index=>new node index.
153 // nodes_to_delete contains nodes who's indices should be skipped, in
154 // ascending order. Save the information about new indices into meta.
155 std::map<int32, int32> old_to_new_ids;
156 int32 new_index = 0;
157 int32 index_for_deleted = 0;
158 auto* post_prune_meta = tree_ensemble_->mutable_tree_metadata(current_tree)
159 ->mutable_post_pruned_nodes_meta();
160
161 for (int32 i = 0; i < num_nodes; ++i) {
162 if (index_for_deleted < nodes_to_delete.size() &&
163 i == nodes_to_delete[index_for_deleted]) {
164 // Node i will get removed,
165 ++index_for_deleted;
166 // Update meta info that will allow us to use cached predictions from
167 // those nodes.
168 int32 new_id;
169 float logit_change;
170 CalculateParentAndLogitUpdate(i, nodes_changes, &new_id, &logit_change);
171 auto* meta = post_prune_meta->Add();
172 meta->set_new_node_id(old_to_new_ids[new_id]);
173 meta->set_logit_change(logit_change);
174 } else {
175 old_to_new_ids[i] = new_index++;
176 auto* meta = post_prune_meta->Add();
177 // Update meta info that will allow us to use cached predictions from
178 // those nodes.
179 meta->set_new_node_id(old_to_new_ids[i]);
180 meta->set_logit_change(0.0);
181 }
182 }
183 index_for_deleted = 0;
184 int32 i = 0;
185 protobuf::RepeatedPtrField<boosted_trees::Node> new_nodes;
186 new_nodes.Reserve(old_to_new_ids.size());
187 for (auto node : *(tree->mutable_nodes())) {
188 if (index_for_deleted < nodes_to_delete.size() &&
189 i == nodes_to_delete[index_for_deleted]) {
190 ++index_for_deleted;
191 ++i;
192 continue;
193 } else {
194 if (node.node_case() == boosted_trees::Node::kBucketizedSplit) {
195 node.mutable_bucketized_split()->set_left_id(
196 old_to_new_ids[node.bucketized_split().left_id()]);
197 node.mutable_bucketized_split()->set_right_id(
198 old_to_new_ids[node.bucketized_split().right_id()]);
199 }
200 *new_nodes.Add() = std::move(node);
201 }
202 ++i;
203 }
204 // Replace all the nodes in a tree with the ones we keep.
205 *tree->mutable_nodes() = std::move(new_nodes);
206
207 // Note that if the whole tree got pruned, we will end up with one node.
208 // We can't remove that tree because it will cause problems with cache.
209}

211void BoostedTreesEnsembleResource::GetPostPruneCorrection(
212 const int32 tree_id, const int32 initial_node_id, int32* current_node_id,
213 float* logit_update) const {
214 DCHECK_LT(tree_id, tree_ensemble_->trees_size());
215 if (IsTreeFinalized(tree_id) && IsTreePostPruned(tree_id)) {
216 DCHECK_LT(
217 initial_node_id,
218 tree_ensemble_->tree_metadata(tree_id).post_pruned_nodes_meta_size());
219 const auto& meta =
220 tree_ensemble_->tree_metadata(tree_id).post_pruned_nodes_meta(
221 initial_node_id);
222 *current_node_id = meta.new_node_id();
223 *logit_update += meta.logit_change();
224 }
225}

227bool BoostedTreesEnsembleResource::IsTerminalSplitNode(
228 const int32 tree_id, const int32 node_id) const {
229 const auto& node = tree_ensemble_->trees(tree_id).nodes(node_id);
230 DCHECK_EQ(node.node_case(), boosted_trees::Node::kBucketizedSplit);
231 const int32 left_id = node.bucketized_split().left_id();
232 const int32 right_id = node.bucketized_split().right_id();
233 return is_leaf(tree_id, left_id) && is_leaf(tree_id, right_id);
234}

236// For each pruned node, finds the leaf where it finally ended up and
237// calculates the total update from that pruned node prediction.
238void BoostedTreesEnsembleResource::CalculateParentAndLogitUpdate(
239 const int32 start_node_id,
240 const std::vector<std::pair<int32, float>>& nodes_change, int32* parent_id,
241 float* change) const {
242 *change = 0.0;
243 int32 node_id = start_node_id;
244 int32 parent = nodes_change[node_id].first;
245
246 while (parent != node_id) {
247 (*change) += nodes_change[node_id].second;
248 node_id = parent;
249 parent = nodes_change[node_id].first;
250 }
251 *parent_id = parent;
252}

254void BoostedTreesEnsembleResource::RecursivelyDoPostPrunePreparation(
255 const int32 tree_id, const int32 node_id,
256 std::vector<int32>* nodes_to_delete,
257 std::vector<std::pair<int32, float>>* nodes_meta) {
258 auto* node = tree_ensemble_->mutable_trees(tree_id)->mutable_nodes(node_id);
259 DCHECK_NE(node->node_case(), boosted_trees::Node::NODE_NOT_SET);
260 // Base case when we reach a leaf.
261 if (node->node_case() == boosted_trees::Node::kLeaf) {
262 return;
263 }
264
265 // Traverse node children first and recursively prune their sub-trees.
266 RecursivelyDoPostPrunePreparation(tree_id, node->bucketized_split().left_id(),
267 nodes_to_delete, nodes_meta);
268 RecursivelyDoPostPrunePreparation(tree_id,
269 node->bucketized_split().right_id(),
270 nodes_to_delete, nodes_meta);
271
272 // Two conditions must be satisfied to prune the node:
273 // 1- The split gain is negative.
274 // 2- After depth-first pruning, the node only has leaf children.
275 const auto& node_metadata = node->metadata();
276 if (node_metadata.gain() < 0 && IsTerminalSplitNode(tree_id, node_id)) {
277 const int32 left_id = node->bucketized_split().left_id();
278 const int32 right_id = node->bucketized_split().right_id();
279
280 // Save children that need to be deleted.
281 nodes_to_delete->push_back(left_id);
282 nodes_to_delete->push_back(right_id);
283
284 // Change node back into leaf.
285 *node->mutable_leaf() = node_metadata.original_leaf();
286 const float parent_value = node_value(tree_id, node_id);
287
288 // Save the old values of weights of children.
289 (*nodes_meta)[left_id].first = node_id;
290 (*nodes_meta)[left_id].second = parent_value - node_value(tree_id, left_id);
291
292 (*nodes_meta)[right_id].first = node_id;
293 (*nodes_meta)[right_id].second =
294 parent_value - node_value(tree_id, right_id);
295
296 // Clear gain for leaf node.
297 node->clear_metadata();
298 }
