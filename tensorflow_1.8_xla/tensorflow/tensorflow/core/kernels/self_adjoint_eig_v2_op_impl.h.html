29namespace tensorflow {
30
31template <class Scalar>
32class SelfAdjointEigV2Op : public LinearAlgebraOp<Scalar> {
33 public:
34 typedef LinearAlgebraOp<Scalar> Base;
35
36 explicit SelfAdjointEigV2Op(OpKernelConstruction* context) : Base(context) {
37 OP_REQUIRES_OK(context, context->GetAttr("compute_v", &compute_v_));
38 }
39
40 using TensorShapes = typename Base::TensorShapes;
41 using Matrix = typename Base::Matrix;
42 using MatrixMaps = typename Base::MatrixMaps;
43 using ConstMatrixMap = typename Base::ConstMatrixMap;
44 using ConstMatrixMaps = typename Base::ConstMatrixMaps;
45
46 TensorShapes GetOutputMatrixShapes(
47 const TensorShapes& input_matrix_shapes) const final {
48 int64 n = input_matrix_shapes[0].dim_size(0);
49 if (compute_v_) {
50 return TensorShapes({TensorShape({n}), TensorShape({n, n})});
51 } else {
52 return TensorShapes({TensorShape({n})});
53 }
54 }
55
56 void ComputeMatrix(OpKernelContext* context, const ConstMatrixMaps& inputs,
57 MatrixMaps* outputs) final {
58 const int64 rows = inputs[0].rows();
59 if (rows == 0) {
60 // If X is an empty matrix (0 rows, 0 col), X * X' == X.
61 // Therefore, we return X.
62 return;
63 }
64
65 // This algorithm relies on denormals, so switch them back on locally.
66 port::ScopedDontFlushDenormal dont_flush_denormals;
67
68 Eigen::SelfAdjointEigenSolver<Matrix> eig(
69 inputs[0],
70 compute_v_ ? Eigen::ComputeEigenvectors : Eigen::EigenvaluesOnly);
71 // TODO(rmlarsen): Output more detailed error info on failure.
72 OP_REQUIRES(
73 context, eig.info() == Eigen::Success,
74 errors::InvalidArgument("Self-adjoint eigen decomposition was not "
75 "successful. The input might not be valid."));
76
77 outputs->at(0) = eig.eigenvalues().template cast<Scalar>();
78 if (compute_v_) {
79 outputs->at(1) = eig.eigenvectors();
80 }
81 }
82
83 private:
84 bool compute_v_;
85};
