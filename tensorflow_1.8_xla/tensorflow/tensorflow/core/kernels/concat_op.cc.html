42enum AxisArgumentName { NAME_IS_AXIS, NAME_IS_CONCAT_DIM };
43
44// --------------------------------------------------------------------------
45template <typename Device, typename T, AxisArgumentName AxisArgName>
46class ConcatBaseOp : public OpKernel {
47 public:
48 typedef std::vector<std::unique_ptr<typename TTypes<T, 2>::ConstMatrix>>
49 ConstMatrixVector;
50
51 explicit ConcatBaseOp(OpKernelConstruction* c) : OpKernel(c) {}
52
53 void Compute(OpKernelContext* c) override {
54 const Tensor* concat_dim_tensor;
55 const char* axis_attribute_name =
56 AxisArgName == NAME_IS_AXIS ? "axis" : AxisArgName == NAME_IS_CONCAT_DIM
57 ? "concat_dim"
58 : "<invalid>";
59 OP_REQUIRES_OK(c, c->input(axis_attribute_name, &concat_dim_tensor));
60 OP_REQUIRES(c, IsLegacyScalar(concat_dim_tensor->shape()),
61 errors::InvalidArgument(
62 axis_attribute_name,
63 " tensor should be a scalar integer, but got shape ",
64 concat_dim_tensor->shape().DebugString()));
65 int64 concat_dim;
66 // In case of ConcatV2, "axis" could be int32 or int64
67 if (AxisArgName == NAME_IS_AXIS) {
68 OP_REQUIRES(
69 c, (concat_dim_tensor->dtype() == DT_INT32 ||
70 concat_dim_tensor->dtype() == DT_INT64),
71 errors::InvalidArgument(axis_attribute_name,
72 " tensor should be int32 or int64, but got ",
73 concat_dim_tensor->dtype()));
74 } else {
75 OP_REQUIRES(c, (concat_dim_tensor->dtype() == DT_INT32),
76 errors::InvalidArgument(axis_attribute_name,
77 " tensor should be int32, but got ",
78 concat_dim_tensor->dtype()));
79 }
80 if (concat_dim_tensor->dtype() == DT_INT32) {
81 concat_dim =
82 internal::SubtleMustCopy(concat_dim_tensor->scalar<int32>()());
83 } else {
84 concat_dim =
85 internal::SubtleMustCopy(concat_dim_tensor->scalar<int64>()());
86 }
87
88 OpInputList values;
89 OP_REQUIRES_OK(c, c->input_list("values", &values));
90 const int N = values.size();
91 const int input_dims = values[0].dims();
92 const TensorShape& input_shape = values[0].shape();
93
94 int32 axis = concat_dim < 0 ? concat_dim + input_dims : concat_dim;
95 OP_REQUIRES(c,
96 (0 <= axis && axis < input_dims) ||
97 (allow_legacy_scalars() && concat_dim == 0),
98 errors::InvalidArgument(
99 "ConcatOp : Expected concatenating dimensions in the range "
100 "[",
101 -input_dims, ", ", input_dims, "), but got ", concat_dim));
102 // Note that we reduce the concat of n-dimensional tensors into a two
103 // dimensional concat. Assuming the dimensions of any input/output
104 // tensor are {x0, x1,...,xn-1, y0, y1,...,ym-1}, where the concat is along
105 // the dimension indicated with size y0, we flatten it to {x, y}, where y =
106 // Prod_i(yi) and x = ((n > 0) ? Prod_i(xi) : 1).
107 ConstMatrixVector inputs_flat;
108 inputs_flat.reserve(N);
109 int64 inputs_flat_dim0 = 1;
110 for (int d = 0; d < axis; ++d) {
111 inputs_flat_dim0 *= input_shape.dim_size(d);
112 }
113 int64 output_concat_dim = 0;
114 const bool input_is_scalar = IsLegacyScalar(input_shape);
115 for (int i = 0; i < N; ++i) {
116 const auto in = values[i];
117 const bool in_is_scalar = IsLegacyScalar(in.shape());
118 OP_REQUIRES(
119 c, in.dims() == input_dims || (input_is_scalar && in_is_scalar),
120 errors::InvalidArgument(
121 "ConcatOp : Ranks of all input tensors should match: shape[0] = ",
122 input_shape.DebugString(), " vs. shape[", i,
123 "] = ", in.shape().DebugString()));
124 for (int j = 0; j < input_dims; ++j) {
125 if (j == axis) {
126 continue;
127 }
128 OP_REQUIRES(
129 c, in.dim_size(j) == input_shape.dim_size(j),
130 errors::InvalidArgument(
131 "ConcatOp : Dimensions of inputs should match: shape[0] = ",
132 input_shape.DebugString(), " vs. shape[", i,
133 "] = ", in.shape().DebugString()));
134 }
135 if (in.NumElements() > 0) {
136 int64 inputs_flat_dim1 = in.NumElements() / inputs_flat_dim0;
137 inputs_flat.emplace_back(new typename TTypes<T, 2>::ConstMatrix(
138 in.shaped<T, 2>({inputs_flat_dim0, inputs_flat_dim1})));
139 }
140 // TODO(irving): Remove check once !allow_legacy_scalars().
141 output_concat_dim += in.dims() > 0 ? in.dim_size(axis) : 1;
142 }
143
144 TensorShape output_shape(input_shape);
145 // TODO(irving): Remove rank 0 case once !allow_legacy_scalars().
146 if (output_shape.dims() == 0) {
147 output_shape.AddDim(output_concat_dim);
148 } else {
149 output_shape.set_dim(axis, output_concat_dim);
150 }
151 Tensor* output = nullptr;
152 OP_REQUIRES_OK(c, c->allocate_output(0, output_shape, &output));
153 if (output->NumElements() > 0) {
154 int64 output_dim1 = output->NumElements() / inputs_flat_dim0;
155 auto output_flat = output->shaped<T, 2>({inputs_flat_dim0, output_dim1});
156#if GOOGLE_CUDA
157 if (std::is_same<Device, GPUDevice>::value) {
158 ConcatGPU<T>(c, inputs_flat, output, &output_flat);
159 return;
160 }
161#endif // GOOGLE_CUDA
162#ifdef TENSORFLOW_USE_SYCL
163 if (std::is_same<Device, SYCLDevice>::value) {
164 ConcatSYCL<T>(c->eigen_sycl_device(), inputs_flat, &output_flat);
165 return;
166 }
167#endif // TENSORFLOW_USE_SYCL
168 ConcatCPU<T>(c->device(), inputs_flat, &output_flat);
169 }
170 }
171};

275class ConcatOffsetOp : public OpKernel {
276 public:
277 explicit ConcatOffsetOp(OpKernelConstruction* ctx) : OpKernel(ctx) {}
278
279 void Compute(OpKernelContext* ctx) override {
280 const Tensor& concat_dim = ctx->input(0);
281 OP_REQUIRES(
282 ctx, IsLegacyScalar(concat_dim.shape()),
283 errors::InvalidArgument(
284 "Concat dim tensor should be a scalar integer, but got shape ",
285 concat_dim.shape().DebugString()));
286 for (int i = 1; i < ctx->num_inputs(); ++i) {
287 const Tensor& inp = ctx->input(i);
288 OP_REQUIRES(ctx, TensorShapeUtils::IsVector(inp.shape()),
289 errors::InvalidArgument("input ", i,
290 " should be a vector, but got shape ",
291 inp.shape().DebugString()));
292 }
293 // Suppose a Concat() op needs to Concatenate N tensors, each of
294 // which has the same number of dimensions. Their shapes match
295 // except the concat dimension.
296 //
297 // E.g., say, we want to concatenate 3 tensors in the 2nd
298 // dimension, and their shapes are:
299 //
300 // [2, 2, 5, 7]
301 // [2, 3, 5, 7]
302 // [2, 4, 5, 7]
303 //
304 // Here, N=3, cdim=1, dims=4. The concatenated tensor has shape
305 // [2,9,5,7]. We will compute the cumulative sum along the 2nd
306 // dimension to figure out each input's offset in the concatenated
307 // output:
308 // [0, 0, 0, 0]
309 // [0, 2, 0, 0]
310 // [0, 5, 0, 0]
311 const int32 N = ctx->num_inputs() - 1;
312 const Tensor& inp0 = ctx->input(1);
313 auto inp0_vec = inp0.vec<int32>();
314 const int64 cdim = internal::SubtleMustCopy(concat_dim.scalar<int32>()());
315 const int64 dims = inp0.NumElements();
316 int32 axis = cdim < 0 ? cdim + dims : cdim;
317 OP_REQUIRES(ctx, FastBoundsCheck(axis, dims),
318 errors::InvalidArgument("Concat dim is out of range: ", cdim,
319 " vs. ", dims));
320 int32 offset = 0;
321 for (int i = 0; i < N; ++i) {
322 const Tensor& inp = ctx->input(1 + i);
323 OP_REQUIRES(
324 ctx, dims == inp.NumElements(),
325 errors::InvalidArgument("input ", i, " should contain ", dims,
326 " elements, but got ", inp.NumElements()));
327 auto inp_vec = inp.vec<int32>();
328 Tensor* out = nullptr;
329 OP_REQUIRES_OK(ctx, ctx->allocate_output(i, {dims}, &out));
330 auto out_vec = out->vec<int32>();
331 for (int64 j = 0; j < dims; ++j) {
332 if (j == axis) {
333 out_vec(j) = offset;
334 offset += inp_vec(j);
335 } else {
336 OP_REQUIRES(ctx, (inp0_vec(j) == inp_vec(j)),
337 errors::InvalidArgument(
338 "All dimensions except ", axis, " must match. Input ",
339 i, " has shape [", inp.SummarizeValue(10),
340 "] and doesn't match input 0 with shape [",
341 inp0.SummarizeValue(10), "]."));
342 out_vec(j) = 0;
343 }
344 }
345 }
346 }
347
348 bool IsExpensive() override { return false; }
349};
