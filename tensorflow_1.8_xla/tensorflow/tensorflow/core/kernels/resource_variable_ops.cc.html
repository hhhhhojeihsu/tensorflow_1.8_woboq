16// Our general strategy for preventing conflicts between concurrent
17// reads and writes of resource variables is to:
18// * For read operations, we:
19// - acquire the variable's mutex (in "shared" mode);
20// - make a (shallow) copy of the Tensor object, which increments
21// the reference count on the variable's TensorBuffer;
22// - release the variable's mutex;
23// - use the copy of the Tensor object to do the read.
24// * For write operations, we:
25// - acquire the variable's mutex (in "exclusive" mode);
26// - check the reference count of variable's TensorBuffer and
27// if it is >1, make a deep copy of the variable's Tensor;
28// - mutate the variable's Tensor;
29// - and release the variable's mutex.
30// This allows several read operations to all use the same
31// TensorBuffer without needing to copy. When it comes time to write
32// it will only make a copy if there is an outstanding read using the
33// buffer. Write operations are serialized by the variable's mutex.
34//
35// For sparse operations (scatter, gather, sparse optimizer updates),
36// we need to avoid copies, since there may not be enough memory for
37// to copies of the whole tensor. To support this, we make two
38// modifications to the above strategy:
39// * For sparse reads (gather), we hold the variable's mutex (still in
40// "shared" mode) for the duration of the whole read. This means
41// that as long as you only do sparse read operations no write will
42// see the reference count >1.
43// * For sparse write operations where the user explicitly specifies
44// that they want to perform the write without locks held
45// (use_locking=false), we never copy even if the variable's
46// reference count is >1.

75class ReadVariableOp : public OpKernel {
76 public:
77 explicit ReadVariableOp(OpKernelConstruction* c) : OpKernel(c) {
78 OP_REQUIRES_OK(c, c->GetAttr("dtype", &dtype_));
79 }
80
81 void Compute(OpKernelContext* ctx) override {
82 Var* variable = nullptr;
83 ResourceHandle handle = HandleFromInput(ctx, 0);
84 const auto status = LookupResource(ctx, handle, &variable);
85 OP_REQUIRES(ctx, status.ok(),
86 errors::FailedPrecondition(
87 "Error while reading resource variable ", handle.name(),
88 " from Container: ", handle.container(),
89 ". This could mean that the variable was uninitialized. ",
90 status.ToString()));
91
92 core::ScopedUnref s(variable);
93 // We're acquiring a reference to the underlying buffer while
94 // holding a shared lock to guarantee ordering of reads and
95 // writes.
96 tf_shared_lock ml(*variable->mu());
97 const Tensor& t = *variable->tensor();
98 OP_REQUIRES(
99 ctx, dtype_ == t.dtype(),
100 errors::InvalidArgument(
101 "Trying to read variable with wrong dtype. Expected ",
102 DataTypeString(dtype_), " got ", DataTypeString(t.dtype())));
103 ctx->set_output(0, t);
104 }
105
106 private:
107 DataType dtype_;
108};

138template <typename T>
139class VariableShapeOp : public OpKernel {
140 public:
141 explicit VariableShapeOp(OpKernelConstruction* c) : OpKernel(c) {}
142
143 void Compute(OpKernelContext* ctx) override {
144 Var* variable = nullptr;
145 OP_REQUIRES_OK(ctx,
146 LookupResource(ctx, HandleFromInput(ctx, 0), &variable));
147 core::ScopedUnref s(variable);
148 variable->mu()->lock_shared();
149 TensorShape shape = variable->tensor()->shape();
150 variable->mu()->unlock_shared();
151 Tensor* output;
152 OP_REQUIRES_OK(ctx, ctx->allocate_output(0, {shape.dims()}, &output));
153 for (int i = 0; i < shape.dims(); ++i) {
154 output->flat<T>()(i) = shape.dim_size(i);
155 }
156 }
157};

209template <typename Device, typename T>
210class AssignVariableOp : public OpKernel {
211 public:
212 explicit AssignVariableOp(OpKernelConstruction* c) : OpKernel(c) {
213 OP_REQUIRES_OK(c, c->GetAttr("dtype", &dtype_));
214 }
215
216 void Compute(OpKernelContext* context) override {
217 OP_REQUIRES(context, dtype_ == context->input(1).dtype(),
218 errors::InvalidArgument(
219 "Variable and value dtypes don't match; respectively, ",
220 dtype_, " and ", context->input(1).dtype()));
221 Var* variable = nullptr;
222 OP_REQUIRES_OK(
223 context,
224 LookupOrCreateResource<Var>(
225 context, HandleFromInput(context, 0), &variable,
226 [this, context](Var** ptr) {
227 *ptr = new Var(dtype_);
228 PersistentTensor unused;
229 Tensor* tmp;
230 AllocatorAttributes attr;
231 attr.set_gpu_compatible(true);
232 attr.set_nic_compatible(true);
233 TF_RETURN_IF_ERROR(context->allocate_persistent(
234 dtype_, context->input(1).shape(), &unused, &tmp, attr));
235 *(*ptr)->tensor() = *tmp;
236 return Status::OK();
237 }));
238 core::ScopedUnref s(variable);
239
240 OP_REQUIRES(context, variable->tensor()->dtype() == dtype_,
241 errors::InvalidArgument(
242 "Trying to assign variable with wrong dtype. Expected ",
243 DataTypeString(variable->tensor()->dtype()), " got ",
244 DataTypeString(dtype_)));
245
246 const Tensor& value = context->input(1);
247 AllocatorAttributes attr;
248 attr.set_gpu_compatible(true);
249 attr.set_nic_compatible(true);
250
251 // Copying is unnecessary if we are the last user of the value
252 // tensor, we can just adopt the input tensor's buffer instead.
253 std::unique_ptr<Tensor> input_alias = context->forward_input(
254 1, OpKernelContext::Params::kNoReservation /*output_index*/, dtype_,
255 value.shape(), DEVICE_MEMORY, attr);
256 mutex_lock ml(*variable->mu());
257 variable->is_initialized = true;
258 if (input_alias) {
259 *variable->tensor() = *input_alias;
260 return;
261 }
262
263 // Need to copy, but maybe we can re-use variable's buffer?
264 if (!variable->tensor()->RefCountIsOne() ||
265 !variable->tensor()->shape().IsSameSize(value.shape())) {
266 // Copy to new buffer
267 PersistentTensor unused;
268 Tensor* tmp;
269 OP_REQUIRES_OK(context, context->allocate_persistent(
270 dtype_, value.shape(), &unused, &tmp, attr));
271 *variable->tensor() = *tmp;
272 }
273 functor::DenseUpdate<Device, T, ASSIGN> copy_functor;
274 copy_functor(context->eigen_device<Device>(), variable->tensor()->flat<T>(),
275 value.flat<T>());
276 }
277
278 private:
279 DataType dtype_;
280};

282template <typename Device>
283class AssignVariableOp<Device, Variant> : public OpKernel {
284 public:
285 explicit AssignVariableOp(OpKernelConstruction* c) : OpKernel(c) {
286 OP_REQUIRES_OK(c, c->GetAttr("dtype", &dtype_));
287 OP_REQUIRES(c, dtype_ == DT_VARIANT,
288 errors::Internal("Variant kernel called with dtype: ",
289 DataTypeString(dtype_)));
290 }
291
292 void Compute(OpKernelContext* context) override {
293 const Tensor& value = context->input(1);
294 Var* variable = nullptr;
295 OP_REQUIRES_OK(context, LookupOrCreateResource<Var>(
296 context, HandleFromInput(context, 0), &variable,
297 [](Var** ptr) {
298 // Created on host.
299 *ptr = new Var(DT_VARIANT);
300 return Status::OK();
301 }));
302 core::ScopedUnref s(variable);
303 OP_REQUIRES(context, variable->tensor()->dtype() == DT_VARIANT,
304 errors::InvalidArgument(
305 "Trying to assign variable with wrong dtype. Expected ",
306 DataTypeString(variable->tensor()->dtype()), " got ",
307 DataTypeString(DT_VARIANT)));
308
309 AllocatorAttributes attr;
310 attr.set_on_host(true);
311
312 // Copying is unnecessary if we are the last user of the value
313 // tensor, we can just adopt the input tensor's buffer instead.
314 // Note that Variant objects themselves always reside on host.
315 //
316 // We nevertheless want to signal to the runtime that the tensor
317 // should reside in memory of the associated device, as Variant
318 // tensors may be marked as sitting on either CPU or GPU. This
319 // helps to elide one or more copies.
320 std::unique_ptr<Tensor> input_alias = context->forward_input(
321 1, OpKernelContext::Params::kNoReservation /*output_index*/, DT_VARIANT,
322 value.shape(),
323 std::is_same<Device, CPUDevice>::value ? HOST_MEMORY : DEVICE_MEMORY,
324 attr);
325
326 mutex_lock ml(*variable->mu());
327 variable->is_initialized = true;
328 *variable->tensor() = Tensor(DT_VARIANT, value.shape());
329
330 if (input_alias) {
331 *variable->tensor() = *input_alias;
332 return;
333 }
334
335 // Need to copy, but maybe we can re-use variable's buffer?
336 if (!variable->tensor()->RefCountIsOne() ||
337 !variable->tensor()->shape().IsSameSize(value.shape())) {
338 PersistentTensor unused;
339 Tensor* tmp;
340 OP_REQUIRES_OK(context,
341 context->allocate_persistent(DT_VARIANT, value.shape(),
342 &unused, &tmp, attr));
343 *variable->tensor() = *tmp;
344 }
345
346 const auto elements_in = value.flat<Variant>();
347 auto elements_out = variable->tensor()->flat<Variant>();
348 for (int64 i = 0; i < elements_in.size(); ++i) {
349 elements_out(i) = elements_in(i);
350 }
351 }
352
353 private:
354 DataType dtype_;
355};

358 REGISTER_KERNEL_BUILDER(Name("AssignVariableOp") \
359 .Device(DEVICE_CPU) \
360 .TypeConstraint<type>("dtype"), \
361 AssignVariableOp<Eigen::ThreadPoolDevice, type>);
362
363TF_CALL_ALL_TYPES(REGISTER_KERNELS);
364TF_CALL_QUANTIZED_TYPES(REGISTER_KERNELS);
365#undef REGISTER_KERNELS

471template <typename Device, typename T, typename Index>
472class ResourceGatherOp : public OpKernel {
473 public:
474 explicit ResourceGatherOp(OpKernelConstruction* c) : OpKernel(c) {}
475
476 void Compute(OpKernelContext* c) override {
477 Var* v = nullptr;
478 OP_REQUIRES_OK(c, LookupResource(c, HandleFromInput(c, 0), &v));
479 core::ScopedUnref su(v);
480 // NOTE: We hold the lock for the whole gather operation instead
481 // of increasing the reference count of v->tensor() to avoid a
482 // situation where a write to the same variable will see a
483 // reference count greater than one and make a copy of the
484 // (potentially very large) tensor buffer.
485 tf_shared_lock ml(*v->mu());
486 const Tensor& params = *v->tensor();
487 const Tensor& indices = c->input(1);
488 OP_REQUIRES(
489 c, TensorShapeUtils::IsVectorOrHigher(params.shape()),
490 errors::InvalidArgument("params must be at least 1 dimensional"));
491
492 // Check that we have enough index space
493 const int64 N = indices.NumElements();
494 OP_REQUIRES(
495 c, params.dim_size(0) <= std::numeric_limits<Index>::max(),
496 errors::InvalidArgument("params.shape[0] too large for ",
497 DataTypeString(DataTypeToEnum<Index>::v()),
498 " indexing: ", params.dim_size(0), " > ",
499 std::numeric_limits<Index>::max()));
500
501 // The result shape is indices.shape + params.shape[1:].
502 TensorShape result_shape = indices.shape();
503 for (int i = 1; i < params.dims(); i++) {
504 result_shape.AddDim(params.dim_size(i));
505 }
506
507 Tensor* out = nullptr;
508 Tensor tmp;
509 if (params.dtype() == DT_VARIANT) {
510 tmp = Tensor(DT_VARIANT, result_shape);
511 c->set_output(0, tmp);
512 out = &tmp;
513 } else {
514 OP_REQUIRES_OK(c, c->allocate_output(0, result_shape, &out));
515 }
516 if (N > 0) {
517 const int64 gather_dim_size = params.dim_size(0);
518 int64 inner_size = 1;
519 for (int i = 1; i < params.dims(); i++) {
520 inner_size *= params.dim_size(i);
521 }
522 auto params_flat = params.shaped<T, 3>({1, gather_dim_size, inner_size});
523 auto indices_flat = indices.flat<Index>();
524 auto out_flat = out->shaped<T, 3>({1, N, out->NumElements() / N});
525
526 functor::GatherFunctor<Device, T, Index> functor;
527 int64 bad_i = functor(c, params_flat, indices_flat, out_flat);
528
529 OP_REQUIRES(
530 c, bad_i < 0,
531 errors::InvalidArgument(
532 "indices", SliceDebugString(indices.shape(), bad_i), " = ",
533 indices_flat(bad_i), " is not in [0, ", params.dim_size(0), ")"));
534 }
535 }
536};

586template <typename Device, typename T, typename Index, scatter_op::UpdateOp op>
587class ResourceScatterUpdateOp : public OpKernel {
588 public:
589 explicit ResourceScatterUpdateOp(OpKernelConstruction* c) : OpKernel(c) {}
590
591 void Compute(OpKernelContext* c) override {
592 Var* v = nullptr;
593 OP_REQUIRES_OK(c, LookupResource(c, HandleFromInput(c, 0), &v));
594 core::ScopedUnref unref_v(v);
595 mutex_lock ml(*v->mu());
596 Tensor* params = v->tensor();
597 OP_REQUIRES_OK(c, PrepareToUpdateVariable<Device, T>(c, params));
598 const Tensor& indices = c->input(1);
599 const Tensor& updates = c->input(2);
600
601 // Check that we have enough index space
602 const int64 N_big = indices.NumElements();
603 OP_REQUIRES(
604 c, N_big <= std::numeric_limits<Index>::max(),
605 errors::InvalidArgument("indices has too many elements for ",
606 DataTypeString(DataTypeToEnum<Index>::v()),
607 " indexing: ", N_big, " > ",
608 std::numeric_limits<Index>::max()));
609 const Index N = static_cast<Index>(N_big);
610 OP_REQUIRES(
611 c, params->dim_size(0) <= std::numeric_limits<Index>::max(),
612 errors::InvalidArgument("params.shape[0] too large for ",
613 DataTypeString(DataTypeToEnum<Index>::v()),
614 " indexing: ", params->dim_size(0), " > ",
615 std::numeric_limits<Index>::max()));
616
617 if (N > 0) {
618 auto indices_flat = indices.flat<Index>();
619 auto params_flat = params->flat_outer_dims<T>();
620 if (TensorShapeUtils::IsScalar(updates.shape())) {
621 const auto update = updates.scalar<T>();
622
623 functor::ScatterScalarFunctor<Device, T, Index, op> functor;
624 const Index bad_i = functor(c, c->template eigen_device<Device>(),
625 params_flat, update, indices_flat);
626 OP_REQUIRES(c, bad_i < 0,
627 errors::InvalidArgument(
628 "indices", SliceDebugString(indices.shape(), bad_i),
629 " = ", indices_flat(bad_i), " is not in [0, ",
630 params->dim_size(0), ")"));
631 } else {
632 int64 num_updates = updates.NumElements();
633 OP_REQUIRES(c, num_updates % N == 0,
634 errors::InvalidArgument(
635 "shape of indices (", indices.shape().DebugString(),
636 ") is not compatible with the shape of updates (",
637 updates.shape().DebugString(), ")"));
638 auto updates_flat = updates.shaped<T, 2>({N, num_updates / N});
639
640 functor::ScatterFunctor<Device, T, Index, op> functor;
641 const Index bad_i = functor(c, c->template eigen_device<Device>(),
642 params_flat, updates_flat, indices_flat);

644 errors::InvalidArgument(

646 " = ", indices_flat(bad_i), " is not in [0, ",
647 params->dim_size(0), ")"));
648 }
649 }
