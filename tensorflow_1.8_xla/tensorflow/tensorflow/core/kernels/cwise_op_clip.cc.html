21typedef Eigen::GpuDevice GPUDevice;

23// Basic coefficient-wise tenary operations.
24// This is the case for example of the clip_by_value.
25// Device: E.g., CPUDevice, GPUDevice.
26// Functor: defined above. E.g., functor::clip.
27template <typename Device, typename T>
28class ClipOp : public OpKernel {
29 public:
30 explicit ClipOp(OpKernelConstruction* ctx) : OpKernel(ctx) {}
31
32 void Compute(OpKernelContext* ctx) override {
33 const Tensor& in0 = ctx->input(0);
34 const Tensor& in1 = ctx->input(1);
35 const Tensor& in2 = ctx->input(2);
36
37 auto in0_flat = in0.flat<T>();
38 auto in1_flat = in1.flat<T>();
39 auto in2_flat = in2.flat<T>();
40 const Device& d = ctx->eigen_device<Device>();
41
42 Tensor* out = nullptr;
43 OP_REQUIRES_OK(
44 ctx, ctx->forward_input_or_allocate_output({0}, 0, in0.shape(), &out));
45 auto out_flat = out->flat<T>();
46 if (in1.shape() == in2.shape()) {
47 if (in0.shape() == in1.shape()) {
48 functor::TernaryClipOp<Device, T>()(d, in0_flat, in1_flat, in2_flat,
49 out_flat);
50 } else {
51 OP_REQUIRES(ctx, TensorShapeUtils::IsScalar(in1.shape()),
52 errors::InvalidArgument(
53 "clip_value_min and clip_value_max must be either of "
54 "the same shape as input, or a scalar. ",
55 "input shape: ", in0.shape().DebugString(),
56 "clip_value_min shape: ", in1.shape().DebugString(),
57 "clip_value_max shape: ", in2.shape().DebugString()));
58 functor::UnaryClipOp<Device, T>()(d, in0_flat, in1_flat, in2_flat,
59 out_flat);
60 }
61 } else {
62 if (in0.shape() == in1.shape()) {
63 OP_REQUIRES(ctx, TensorShapeUtils::IsScalar(in2.shape()),
64 errors::InvalidArgument(
65 "clip_value_min and clip_value_max must be either of "
66 "the same shape as input, or a scalar. ",
67 "input shape: ", in0.shape().DebugString(),
68 "clip_value_min shape: ", in1.shape().DebugString(),
69 "clip_value_max shape: ", in2.shape().DebugString()));
70 functor::BinaryLeftClipOp<Device, T>()(d, in0_flat, in1_flat, in2_flat,
71 out_flat);
72 } else {
73 OP_REQUIRES(ctx, (in0.shape() == in2.shape() &&
74 TensorShapeUtils::IsScalar(in1.shape())),
75 errors::InvalidArgument(
76 "clip_value_min and clip_value_max must be either of "
77 "the same shape as input, or a scalar. ",
78 "input shape: ", in0.shape().DebugString(),
79 "clip_value_min shape: ", in1.shape().DebugString(),
80 "clip_value_max shape: ", in2.shape().DebugString()));
81 functor::BinaryRightClipOp<Device, T>()(d, in0_flat, in1_flat, in2_flat,
82 out_flat);
83 }
84 }
85 }
86};

88namespace functor {
89// Unary functor for clip [Tensor, Scalar, Scalar]
90template <typename T>
91struct UnaryClipFunc {
92 UnaryClipFunc(const T& value_min, const T& value_max)
93 : value_min_(value_min), value_max_(value_max) {}
94 const T operator()(const T& value) const {
95 return std::max(std::min(value, value_max_), value_min_);
96 }
97 T value_min_;
98 T value_max_;
99};
100template <typename T>
101struct UnaryClipOp<CPUDevice, T> {
102 void operator()(const CPUDevice& d, typename TTypes<T>::ConstFlat& in0_flat,
103 typename TTypes<T>::ConstFlat& in1_flat,
104 typename TTypes<T>::ConstFlat& in2_flat,
105 typename TTypes<T>::Flat& out_flat) const {
106 out_flat = in0_flat.unaryExpr(UnaryClipFunc<T>(in1_flat(0), in2_flat(0)));
107 }
108};
109
110// Binary functor for clip [Tensor, Scalar, Tensor]
111template <typename T>
112struct BinaryRightClipFunc {
113 BinaryRightClipFunc(const T& value_min) : value_min_(value_min) {}
114 const T operator()(const T& value, const T& value_max) const {
115 return std::max(std::min(value, value_max), value_min_);
116 }
117 T value_min_;
118};
119template <typename T>
120struct BinaryRightClipOp<CPUDevice, T> {
121 void operator()(const CPUDevice& d, typename TTypes<T>::ConstFlat& in0_flat,
122 typename TTypes<T>::ConstFlat& in1_flat,
123 typename TTypes<T>::ConstFlat& in2_flat,
124 typename TTypes<T>::Flat& out_flat) const {
125 out_flat =
126 in0_flat.binaryExpr(in2_flat, BinaryRightClipFunc<T>(in1_flat(0)));
127 }
128};
129
130// Binary functor for clip [Tensor, Tensor, Scalar]
131template <typename T>
132struct BinaryLeftClipFunc {
133 BinaryLeftClipFunc(const T& value_max) : value_max_(value_max) {}
134 const T operator()(const T& value, const T& value_min) const {
135 return std::max(std::min(value, value_max_), value_min);
136 }
137 T value_max_;
138};
139template <typename T>
140struct BinaryLeftClipOp<CPUDevice, T> {
141 void operator()(const CPUDevice& d, typename TTypes<T>::ConstFlat& in0_flat,
142 typename TTypes<T>::ConstFlat& in1_flat,
143 typename TTypes<T>::ConstFlat& in2_flat,
144 typename TTypes<T>::Flat& out_flat) const {
145 out_flat =
146 in0_flat.binaryExpr(in1_flat, BinaryLeftClipFunc<T>(in2_flat(0)));
147 }
148};
149
150// Ternary functor for clip [Tensor, Tensor, Tensor]
151template <typename T>
152struct TernaryClipOp<CPUDevice, T> {
153 void operator()(const CPUDevice& d, typename TTypes<T>::ConstFlat& in0_flat,
154 typename TTypes<T>::ConstFlat& in1_flat,
155 typename TTypes<T>::ConstFlat& in2_flat,
156 typename TTypes<T>::Flat& out_flat) const {
157 out_flat.device(d) = in0_flat.cwiseMin(in2_flat).cwiseMax(in1_flat);
158 }
159};
