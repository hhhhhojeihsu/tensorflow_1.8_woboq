98template <typename Scalar, typename Exponent>
99struct safe_scalar_binary_pow_op {
100 static_assert(std::is_integral<Scalar>::value, "Integer type expected");
101 static_assert(std::is_integral<Exponent>::value &&
102 std::is_signed<Exponent>::value,
103 "Signed integer type expected");
104
105 bool* const error;
106
107 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE safe_scalar_binary_pow_op(bool* error)
108 : error(error) {}
109
110 EIGEN_DEVICE_FUNC inline Scalar operator()(const Scalar& a,
111 const Exponent& b) const {
112 const Exponent safe_b = tensorflow::internal::SubtleMustCopy(b);
113 if (TF_PREDICT_TRUE(safe_b >= 0)) {
114 return numext::pow(a, safe_b);
115 } else {
116 *error = true;
117 return 0;
118 }
119 }
120};

156// scalar_left and scalar_right are template helpers to partially
157// apply a binary function.
158//
159// Suppose Binary is a binary functor f(x, y), scalar_left<> is a
160// unary functor g_x(y) = f(x, y), where x is provided via the
161// constructor. Similarly, scalar_right<> is a unary functor g_y(x) =
162// f(x, y).
163
164template <typename Tout, typename Tin, typename Binary>
165struct scalar_left : private Binary {
166 typedef Tout result_type;
167 const Tin* left;
168
169 EIGEN_DEVICE_FUNC inline scalar_left(const scalar_left& other) = default;
170
171 template <typename... Args>
172 EIGEN_DEVICE_FUNC inline explicit scalar_left(const Tin* c, Args... args)
173 : Binary(args...), left(c) {}
174
175 EIGEN_DEVICE_FUNC inline Tout operator()(const Tin& right) const {
176 return Binary::operator()(*left, right);
177 }
178
179 template <typename Packet>
180 EIGEN_DEVICE_FUNC inline Packet packetOp(const Packet& right_packet) const {
181 const Packet left_packet = Eigen::internal::pset1<Packet>(*left);
182 return Binary::packetOp(left_packet, right_packet);
183 }
184};
185
186template <typename Tout, typename Tin, typename Binary>
187struct functor_traits<scalar_left<Tout, Tin, Binary>> {
188 enum {
189 Cost = functor_traits<Binary>::Cost,
190 PacketAccess = functor_traits<Binary>::PacketAccess,
191 };
192};
193
194template <typename Tout, typename Tin, typename Binary>
195struct scalar_right : private Binary {
196 typedef Tout result_type;
197 const Tin* right;
198
199 EIGEN_DEVICE_FUNC inline scalar_right(const scalar_right& other) = default;
200
201 template <typename... Args>
202 EIGEN_DEVICE_FUNC inline explicit scalar_right(const Tin* c, Args... args)
203 : Binary(args...), right(c) {}
204
205 EIGEN_DEVICE_FUNC inline Tout operator()(const Tin& left) const {
206 return Binary::operator()(left, *right);
207 }
208
209 template <typename Packet>
210 EIGEN_DEVICE_FUNC inline Packet packetOp(const Packet& left_packet) const {
211 const Packet right_packet = Eigen::internal::pset1<Packet>(*right);
212 return Binary::packetOp(left_packet, right_packet);
213 }
214};
215
216template <typename Tout, typename Tin, typename Binary>
217struct functor_traits<scalar_right<Tout, Tin, Binary>> {
218 enum {
219 Cost = functor_traits<Binary>::Cost,
220 PacketAccess = functor_traits<Binary>::PacketAccess,
221 };
222};

303template <typename T, typename Enable = void>
304struct google_floor_div {
305 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE const T operator()(const T& x,
306 const T& y) const {
307 if ((x < T(0)) != (y < T(0))) {
308 T abs_x = std::abs(x);
309 T abs_y = std::abs(y);
310 return -(abs_x + abs_y - 1) / abs_y;
311 } else {
312 return x / y;
313 }
314 }
315};

403template <typename Scalar>
404struct scalar_round_op_google {
405 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE const Scalar
406 operator()(const Scalar& x) const {
407 EIGEN_STATIC_ASSERT((!NumTraits<Scalar>::IsComplex),
408 NUMERIC_TYPE_MUST_BE_REAL)
409
410 Scalar round_val = Eigen::numext::floor(x);
411 const Scalar fraction = x - round_val;
412 if (fraction > Scalar(.5)) {
413 round_val += Scalar(1.0);
414 } else if (fraction == Scalar(.5)) {
415 const Scalar nearest_even_int =
416 round_val - Scalar(2) * Eigen::numext::floor(Scalar(.5) * x);
417 bool is_odd = (nearest_even_int == Scalar(1));
418 if (is_odd) {
419 round_val += Scalar(1);
420 }
421 }
422 return round_val;
423 }
424};
425

427struct functor_traits<scalar_round_op_google<Scalar>> {
428 enum { Cost = 4 * NumTraits<Scalar>::AddCost, PacketAccess = false };
429};

463// Base template for functors whose input scalar type is T and
464// output scalar type is R.
465template <typename T, typename F, typename R = T>
466struct base {
467 // func defines operator() and its vectorized version packetOp().
468 typedef F func;
469
470 // If true, the functor's corresponding binary op will instantiate
471 // specialized kernels to perform an optimized broadcast
472 // operation. Each functor for which this is enabled increases the
473 // code size, so by default this is disabled for binary functors and
474 // is enabled on a per-op basis as needed.
475 static const bool use_bcast_optimization = false;
476
477 // operator() has the signature:
478 // out_type operator()(in_type in0, in_type in1 ...)
479 typedef R out_type;
480 typedef T in_type;
481
482 // TensorFlow provides tensor-ized version of "func". Roughly
483 // speaking, the tensorflow operation has the signature:
484 // tout_type op(tin_type in0)
485 // tout_type op(tin_type in0, tin_type in1)
486 // tout_type op(tin_type in0, in_type scalar)
487 typedef typename TTypes<out_type>::Flat tout_type;
488 typedef typename TTypes<in_type>::ConstFlat tin_type;
489 typedef typename TTypes<in_type>::ConstScalar tscalar_type;
490
491 // Whether the functor can error out. Currently applies only to integer
492 // div and mod.
493 static const bool has_errors = false;
494};

847template <typename T>
848struct left_shift_op {
849 EIGEN_EMPTY_STRUCT_CTOR(left_shift_op)
850 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE const T operator()(const T& x,
851 const T& y) const {
852 // Avoids UB: don't shift by larger than the bitwidth of T, and
853 // performs left shifts as unsigned shifts.
854 T y_clamped = y;
855 if (y_clamped < 0) {
856 y_clamped = 0;
857 } else if (y_clamped > sizeof(T) * CHAR_BIT - 1) {
858 y_clamped = sizeof(T) * CHAR_BIT - 1;
859 }
860 using U = typename std::make_unsigned<T>::type;
861 return static_cast<T>(static_cast<U>(x) << static_cast<U>(y_clamped));
862 }
863};
864
865template <typename T>
866struct right_shift_op {
867 EIGEN_EMPTY_STRUCT_CTOR(right_shift_op)
868 EIGEN_DEVICE_FUNC EIGEN_STRONG_INLINE const T operator()(const T& x,
869 const T& y) const {
870 // Avoids UB: don't shift by larger than the bitwidth of T.
871 T y_clamped = y;
872 if (y_clamped < 0) {
873 y_clamped = 0;
874 } else if (y_clamped > sizeof(T) * CHAR_BIT - 1) {
875 y_clamped = sizeof(T) * CHAR_BIT - 1;
876 }
877 // Technically right shifts of signed integers are not necessarily
878 // arithmetic shifts according to the C++ standard. However in practice most
879 // implementations are arithmetic shifts. If this proves to be a problem in
880 // practice, we may need to use an alternative implementation.
881 return x >> y_clamped;
882 }
883};
