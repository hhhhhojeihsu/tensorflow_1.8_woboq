18namespace tensorflow {
19#define REGISTER_COMPLEX(D, R, C) \
20 REGISTER_KERNEL_BUILDER(Name("Imag") \
21 .Device(DEVICE_##D) \
22 .TypeConstraint<C>("T") \
23 .TypeConstraint<R>("Tout"), \
24 UnaryOp<D##Device, functor::get_imag<C>>);
25
26REGISTER_COMPLEX(CPU, float, complex64);
27REGISTER_COMPLEX(CPU, double, complex128);
28
29#if GOOGLE_CUDA
30REGISTER_COMPLEX(GPU, float, complex64);
31REGISTER_COMPLEX(GPU, double, complex128);
32#endif
33
34#undef REGISTER_COMPLEX
35} // namespace tensorflow
