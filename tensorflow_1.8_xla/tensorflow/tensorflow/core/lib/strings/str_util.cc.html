28static char hex_char[] = "0123456789abcdef";

30string CEscape(StringPiece src) {
31 string dest;
32
33 for (unsigned char c : src) {
34 switch (c) {
35 case '\n':
36 dest.append("\\n");
37 break;
38 case '\r':
39 dest.append("\\r");
40 break;
41 case '\t':
42 dest.append("\\t");
43 break;
44 case '\"':
45 dest.append("\\\"");
46 break;
47 case '\'':
48 dest.append("\\'");
49 break;
50 case '\\':
51 dest.append("\\\\");
52 break;
53 default:
54 // Note that if we emit \xNN and the src character after that is a hex
55 // digit then that digit must be escaped too to prevent it being
56 // interpreted as part of the character code by C.
57 if ((c >= 0x80) || !isprint(c)) {
58 dest.append("\\");
59 dest.push_back(hex_char[c / 64]);
60 dest.push_back(hex_char[(c % 64) / 8]);
61 dest.push_back(hex_char[c % 8]);
62 } else {
63 dest.push_back(c);
64 break;
65 }
66 }
67 }
68
69 return dest;
70}

74inline bool is_octal_digit(unsigned char c) { return c >= '0' && c <= '7'; }
75
76inline bool ascii_isxdigit(unsigned char c) {
77 return (c >= '0' && c <= '9') || (c >= 'a' && c <= 'f') ||
78 (c >= 'A' && c <= 'F');
79}
80
81inline int hex_digit_to_int(char c) {
82 int x = static_cast<unsigned char>(c);
83 if (x > '9') {
84 x += 9;
85 }
86 return x & 0xf;
87}

89bool CUnescapeInternal(StringPiece source, string* dest,
90 string::size_type* dest_len, string* error) {
91 const char* p = source.data();
92 const char* end = source.end();
93 const char* last_byte = end - 1;
94
95 // We are going to write the result to dest with its iterator. If our string
96 // implementation uses copy-on-write, this will trigger a copy-on-write of
97 // dest's buffer; that is, dest will be assigned a new buffer.
98 //
99 // Note that the following way is NOT a legal way to modify a string's
100 // content:
101 //
102 // char* d = const_cast<char*>(dest->data());
103 //
104 // This won't trigger copy-on-write of the string, and so is dangerous when
105 // the buffer is shared.
106 auto d = dest->begin();
107
108 // Small optimization for case where source = dest and there's no escaping
109 if (source.data() == dest->data()) {
110 while (p < end && *p != '\\') {
111 p++;
112 d++;
113 }
114 }
115
116 while (p < end) {
117 if (*p != '\\') {
118 *d++ = *p++;
119 } else {
120 if (++p > last_byte) { // skip past the '\\'
121 if (error) *error = "String cannot end with \\";
122 return false;
123 }
124 switch (*p) {
125 case 'a':
126 *d++ = '\a';
127 break;
128 case 'b':
129 *d++ = '\b';
130 break;
131 case 'f':
132 *d++ = '\f';
133 break;
134 case 'n':
135 *d++ = '\n';
136 break;
137 case 'r':
138 *d++ = '\r';
139 break;
140 case 't':
141 *d++ = '\t';
142 break;
143 case 'v':
144 *d++ = '\v';
145 break;
146 case '\\':
147 *d++ = '\\';
148 break;
149 case '?':
150 *d++ = '\?';
151 break; // \? Who knew?
152 case '\'':
153 *d++ = '\'';
154 break;
155 case '"':
156 *d++ = '\"';
157 break;
158 case '0':
159 case '1':
160 case '2':
161 case '3': // octal digit: 1 to 3 digits
162 case '4':
163 case '5':
164 case '6':
165 case '7': {
166 const char* octal_start = p;
167 unsigned int ch = *p - '0';
168 if (p < last_byte && is_octal_digit(p[1])) ch = ch * 8 + *++p - '0';
169 if (p < last_byte && is_octal_digit(p[1]))
170 ch = ch * 8 + *++p - '0'; // now points at last digit
171 if (ch > 0xff) {
172 if (error) {
173 *error = "Value of \\" +
174 string(octal_start, p + 1 - octal_start) +
175 " exceeds 0xff";
176 }
177 return false;
178 }
179 *d++ = ch;
180 break;
181 }
182 case 'x':
183 case 'X': {
184 if (p >= last_byte) {
185 if (error) *error = "String cannot end with \\x";
186 return false;
187 } else if (!ascii_isxdigit(p[1])) {
188 if (error) *error = "\\x cannot be followed by a non-hex digit";
189 return false;
190 }
191 unsigned int ch = 0;
192 const char* hex_start = p;
193 while (p < last_byte && ascii_isxdigit(p[1]))
194 // Arbitrarily many hex digits
195 ch = (ch << 4) + hex_digit_to_int(*++p);
196 if (ch > 0xFF) {
197 if (error) {
198 *error = "Value of \\" + string(hex_start, p + 1 - hex_start) +
199 " exceeds 0xff";
200 }
201 return false;
202 }
203 *d++ = ch;
204 break;
205 }
206 default: {
207 if (error) *error = string("Unknown escape sequence: \\") + *p;
208 return false;
209 }
210 }
211 p++; // read past letter we escaped
212 }
213 }
214 *dest_len = d - dest->begin();
215 return true;
216}

218template <typename T>
219bool SplitAndParseAsInts(StringPiece text, char delim,
220 std::function<bool(StringPiece, T*)> converter,
221 std::vector<T>* result) {
222 result->clear();
223 std::vector<string> num_strings = Split(text, delim);
224 for (const auto& s : num_strings) {
225 T num;
226 if (!converter(s, &num)) return false;
227 result->push_back(num);
228 }
229 return true;
230}

234bool CUnescape(StringPiece source, string* dest, string* error) {
235 dest->resize(source.size());
236 string::size_type dest_size;
237 if (!CUnescapeInternal(source, dest, &dest_size, error)) {
238 return false;
239 }
240 dest->erase(dest_size);
241 return true;
242}

244void StripTrailingWhitespace(string* s) {
245 string::size_type i;
246 for (i = s->size(); i > 0 && isspace((*s)[i - 1]); --i) {
247 }
248 s->resize(i);
249}

251// Return lower-cased version of s.
252string Lowercase(StringPiece s) {
253 string result(s.data(), s.size());
254 for (char& c : result) {
255 c = tolower(c);
256 }
257 return result;

261string Uppercase(StringPiece s) {
262 string result(s.data(), s.size());
263 for (char& c : result) {
264 c = toupper(c);
265 }
266 return result;

269string ArgDefCase(StringPiece s) {
270 const size_t n = s.size();
271
272 // Compute the size of resulting string.
273 // Number of extra underscores we will need to add.
274 size_t extra_us = 0;
275 // Number of non-alpha chars in the beginning to skip.
276 size_t to_skip = 0;
277 for (size_t i = 0; i < n; ++i) {
278 // If we are skipping and current letter is non-alpha, skip it as well
279 if (i == to_skip && !isalpha(s[i])) {
280 ++to_skip;
281 continue;
282 }
283
284 // If we are here, we are not skipping any more.
285 // If this letter is upper case, not the very first char in the
286 // resulting string, and previous letter isn't replaced with an underscore,
287 // we will need to insert an underscore.
288 if (isupper(s[i]) && i != to_skip && i > 0 && isalnum(s[i - 1])) {
289 ++extra_us;
290 }
291 }
292
293 // Initialize result with all '_'s. There is no string
294 // constructor that does not initialize memory.
295 string result(n + extra_us - to_skip, '_');
296 // i - index into s
297 // j - index into result
298 for (size_t i = to_skip, j = 0; i < n; ++i, ++j) {
299 DCHECK_LT(j, result.size());
300 char c = s[i];
301 // If c is not alphanumeric, we don't need to do anything
302 // since there is already an underscore in its place.
303 if (isalnum(c)) {
304 if (isupper(c)) {
305 // If current char is upper case, we might need to insert an
306 // underscore.
307 if (i != to_skip) {
308 DCHECK_GT(j, 0);
309 if (result[j - 1] != '_') ++j;
310 }
311 result[j] = tolower(c);
312 } else {
313 result[j] = c;
314 }
315 }
316 }
317
318 return result;
319}

321void TitlecaseString(string* s, StringPiece delimiters) {
322 bool upper = true;
323 for (string::iterator ss = s->begin(); ss != s->end(); ++ss) {
324 if (upper) {
325 *ss = toupper(*ss);
326 }
327 upper = (delimiters.find(*ss) != StringPiece::npos);
328 }
329}

331string StringReplace(StringPiece s, StringPiece oldsub, StringPiece newsub,
332 bool replace_all) {
333 // TODO(jlebar): We could avoid having to shift data around in the string if
334 // we had a StringPiece::find() overload that searched for a StringPiece.
335 string res = s.ToString();
336 size_t pos = 0;
337 while ((pos = res.find(oldsub.data(), pos, oldsub.size())) != string::npos) {
338 res.replace(pos, oldsub.size(), newsub.data(), newsub.size());
339 pos += newsub.size();
340 if (oldsub.empty()) {
341 pos++; // Match at the beginning of the text and after every byte
342 }
343 if (!replace_all) {
344 break;
345 }
346 }
347 return res;
348}

350size_t RemoveLeadingWhitespace(StringPiece* text) {
351 size_t count = 0;
352 const char* ptr = text->data();
353 while (count < text->size() && isspace(*ptr)) {
354 count++;
355 ptr++;
356 }
357 text->remove_prefix(count);
358 return count;
359}
360
361size_t RemoveTrailingWhitespace(StringPiece* text) {
362 size_t count = 0;
363 const char* ptr = text->data() + text->size() - 1;
364 while (count < text->size() && isspace(*ptr)) {
365 ++count;
366 --ptr;
367 }
368 text->remove_suffix(count);
369 return count;
370}

372size_t RemoveWhitespaceContext(StringPiece* text) {
373 // use RemoveLeadingWhitespace() and RemoveTrailingWhitespace() to do the job
374 return (RemoveLeadingWhitespace(text) + RemoveTrailingWhitespace(text));
375}

377bool ConsumePrefix(StringPiece* s, StringPiece expected) {
378 if (StartsWith(*s, expected)) {
379 s->remove_prefix(expected.size());
380 return true;
381 }
382 return false;
383}
384
385bool ConsumeSuffix(StringPiece* s, StringPiece expected) {
386 if (EndsWith(*s, expected)) {
387 s->remove_suffix(expected.size());
388 return true;
389 }
390 return false;
391}

393bool ConsumeLeadingDigits(StringPiece* s, uint64* val) {
394 const char* p = s->data();
395 const char* limit = p + s->size();
396 uint64 v = 0;
397 while (p < limit) {
398 const char c = *p;
399 if (c < '0' || c > '9') break;
400 uint64 new_v = (v * 10) + (c - '0');
401 if (new_v / 8 < v) {
402 // Overflow occurred
403 return false;
404 }
405 v = new_v;
406 p++;
407 }
408 if (p > s->data()) {
409 // Consume some digits
410 s->remove_prefix(p - s->data());
411 *val = v;
412 return true;
413 } else {
414 return false;
415 }
416}
417
418bool ConsumeNonWhitespace(StringPiece* s, StringPiece* val) {
419 const char* p = s->data();
420 const char* limit = p + s->size();
421 while (p < limit) {
422 const char c = *p;
423 if (isspace(c)) break;
424 p++;
425 }
426 const size_t n = p - s->data();
427 if (n > 0) {
428 *val = StringPiece(s->data(), n);
429 s->remove_prefix(n);
430 return true;
431 } else {
432 *val = StringPiece();
433 return false;
434 }
435}

457bool StrContains(StringPiece haystack, StringPiece needle) {
458 return std::search(haystack.begin(), haystack.end(), needle.begin(),
459 needle.end()) != haystack.end();
460}
461
462bool StartsWith(StringPiece text, StringPiece prefix) {
463 return prefix.empty() ||
464 (text.size() >= prefix.size() &&
465 memcmp(text.data(), prefix.data(), prefix.size()) == 0);
466}
467
468bool EndsWith(StringPiece text, StringPiece suffix) {
469 return suffix.empty() || (text.size() >= suffix.size() &&
470 memcmp(text.data() + (text.size() - suffix.size()),
471 suffix.data(), suffix.size()) == 0);
472}
