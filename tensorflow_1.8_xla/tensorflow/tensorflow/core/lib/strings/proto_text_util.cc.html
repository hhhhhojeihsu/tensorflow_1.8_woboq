21bool ProtoParseBoolFromScanner(Scanner* scanner, bool* value) {
22 StringPiece bool_str;
23 if (!scanner->RestartCapture()
24 .Many(Scanner::LETTER_DIGIT)
25 .GetResult(nullptr, &bool_str)) {
26 return false;
27 }
28 ProtoSpaceAndComments(scanner);
29 if (bool_str == "false" || bool_str == "False" || bool_str == "0") {
30 *value = false;
31 return true;
32 } else if (bool_str == "true" || bool_str == "True" || bool_str == "1") {
33 *value = true;
34 return true;
35 } else {
36 return false;
37 }
38}
39
40bool ProtoParseStringLiteralFromScanner(Scanner* scanner, string* value) {
41 const char quote = scanner->Peek();
42 if (quote != '\'' && quote != '"') return false;
43
44 StringPiece value_sp;
45 if (!scanner->One(Scanner::ALL)
46 .RestartCapture()
47 .ScanEscapedUntil(quote)
48 .StopCapture()
49 .One(Scanner::ALL)
50 .GetResult(nullptr, &value_sp)) {
51 return false;
52 }
53 ProtoSpaceAndComments(scanner);
54 return str_util::CUnescape(value_sp, value, nullptr /* error */);
55}
