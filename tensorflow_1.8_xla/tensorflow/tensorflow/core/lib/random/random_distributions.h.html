43// A class that generates uniform distribution random numbers from the
44// underlying random integer generator.
45// Arguments:
46// Generator: a generator type that returns a number of uint32 upon each
47// invocation. It needs to define kResultElementCount for the
48// sample count for each invocation, and ResultType for the
49// actual returned sample type.
50// RealType: the data type of the real numbers that will be returned by the
51// distribution. This could be either float or double for now.
52// This class is meant to be implemented through specialization. The default
53// is not defined by design.
54template <class Generator, typename RealType>
55class UniformDistribution;
56
57template <class Generator>
58class UniformDistribution<Generator, Eigen::half> {
59 public:
60 // The number of elements that will be returned.
61 static const int kResultElementCount = Generator::kResultElementCount;
62 // Cost of generation of a single element (in cycles).
63 static const int kElementCost = 3;
64 // Indicate that this distribution may take variable number of samples
65 // during the runtime.
66 static const bool kVariableSamplesPerOutput = false;
67 typedef Array<Eigen::half, kResultElementCount> ResultType;
68 typedef Eigen::half ResultElementType;
69
70 PHILOX_DEVICE_INLINE
71 ResultType operator()(Generator* gen) {
72 typename Generator::ResultType sample = (*gen)();
73 ResultType result;
74 for (int i = 0; i < kResultElementCount; ++i) {
75 result[i] = Uint16ToHalf(sample[i]); // Truncate the upper 16 bits.
76 }
77 return result;
78 }
79};

153template <class Generator>
154class UniformDistribution<Generator, int32> {
155 public:
156 // The number of elements that will be returned.
157 static const int kResultElementCount = Generator::kResultElementCount;
158 // Cost of generation of a single element (in cycles).
159 static const int kElementCost = 3;
160 // Indicate that this distribution may take variable number of samples
161 // during the runtime.
162 static const bool kVariableSamplesPerOutput = false;
163 typedef Array<int32, kResultElementCount> ResultType;
164 typedef int32 ResultElementType;
165
166 // Must have lo < hi
167 UniformDistribution(int32 lo, int32 hi) : lo_(lo), range_(hi - lo) {}
168
169 PHILOX_DEVICE_INLINE
170 ResultType operator()(Generator* gen) {
171 typename Generator::ResultType sample = (*gen)();
172 ResultType result;
173 for (int i = 0; i < kResultElementCount; ++i) {
174 result[i] = lo_ + static_cast<int32>(sample[i] % range_);
175 }
176 return result;
177 }
178
179 private:
180 // Note that lo_ is intentionally signed while range_ is intentionally
181 // unsigned. This is because hi - lo can overflow signed integers if
182 // lo < 0 < hi, but always fits in unsigned.
183 int32 lo_;
184 uint32 range_;
185};

187template <class Generator>
188class UniformDistribution<Generator, int64> {
189 public:
190 // The number of elements that will be returned.
191 static const int kResultElementCount = Generator::kResultElementCount / 2;
192 // Cost of generation of a single element (in cycles).
193 static const int kElementCost = 3;
194 // Indicate that this distribution may take variable number of samples
195 // during the runtime.
196 static const bool kVariableSamplesPerOutput = false;
197 typedef Array<int64, kResultElementCount> ResultType;
198 typedef int64 ResultElementType;
199
200 // Must have lo < hi
201 UniformDistribution(int64 lo, int64 hi) : lo_(lo), range_(hi - lo) {}
202
203 PHILOX_DEVICE_INLINE
204 ResultType operator()(Generator* gen) {
205 typename Generator::ResultType sample = (*gen)();
206 ResultType result;
207 for (int i = 0; i < kResultElementCount; ++i) {
208 auto bits = sample[2 * i] | static_cast<uint64>(sample[2 * i + 1]) << 32;
209 result[i] = lo_ + static_cast<int64>(bits % range_);
210 }
211 return result;
212 }
213
214 private:
215 // Note that lo_ is intentionally signed while range_ is intentionally
216 // unsigned. This is because hi - lo can overflow signed integers if
217 // lo < 0 < hi, but always fits in unsigned.
218 int64 lo_;
219 uint64 range_;
220};

222// A class that adapts the underlying native multiple samples to return a single
223// sample at a time.
224template <class Generator>
225class SingleSampleAdapter {
226 public:
227 // The number of elements that will be returned.
228 static const int kResultElementCount = 1;
229 // The number of elements that will be returned by the underlying generator.
230 static const int kNativeElementCount = Generator::kResultElementCount;
231 typedef typename Generator::ResultElementType ResultType;
232 typedef typename Generator::ResultElementType ResultElementType;
233
234 PHILOX_DEVICE_INLINE
235 explicit SingleSampleAdapter(Generator* gen)
236 : generator_(gen), used_result_index_(Generator::kResultElementCount) {}
237
238 PHILOX_DEVICE_INLINE
239 ResultType operator()() {
240 if (used_result_index_ == Generator::kResultElementCount) {
241 unused_results_ = (*generator_)();
242 used_result_index_ = 0;
243 }
244
245 return unused_results_[used_result_index_++];
246 }
247
248 PHILOX_DEVICE_INLINE
249 void Skip(uint64 num_skips) {
250 if (!num_skips) {
251 return;
252 }
253 int num_unused_results = kNativeElementCount - used_result_index_;
254 if (num_skips <= num_unused_results) {
255 used_result_index_ += num_skips;
256 return;
257 }
258 num_skips -= num_unused_results;
259 used_result_index_ = kNativeElementCount;
260 SkipFromGenerator(num_skips / kNativeElementCount);
261 num_skips = num_skips % kNativeElementCount;
262 if (num_skips) {
263 unused_results_ = (*generator_)();
264 used_result_index_ = num_skips;
265 }
266 }
267
268 private:
269 // This implementation iteratively skips over `num_skips` samples
270 // from `generator_`. There is an O(1) implementation for PhiloxRandom
271 // in random_distributions.cc.
272 PHILOX_DEVICE_INLINE
273 void SkipFromGenerator(uint64 num_skips) {
274 while (num_skips--) {
275 (*generator_)();
276 }
277 }
278
279 Generator* generator_;
280 typename Generator::ResultType unused_results_;
281 int used_result_index_;
282};

284// A class that generates unit normal distribution random numbers from the
285// underlying random integer generator.
286// Arguments:
287// Generator: a generator type that returns a number of uint32 upon each
288// each invocation. It needs to define kResultElementCount for the
289// sample count for each invocation, and ResultType for actual
290// returned sample type.
291// RealType: the data type of the real numbers that will be returned by the
292// distribution. This could be either float or double for now.
293// This class is meant to be implemented through specialization. The default
294// is not defined by design.
295template <class Generator, typename RealType>
296class NormalDistribution;

305// Exactly like the float version, except that we convert to half afterwards;
306// since we don't have half-precision sin/cos even on GPUs, there's nothing to
307// gain from working in half internally.
308template <class Generator>
309class NormalDistribution<Generator, Eigen::half> {
310 public:
311 // The number of elements that will be returned.
312 static const int kResultElementCount = Generator::kResultElementCount;
313 // Cost of generation of a single element (in cycles).
314 static const int kElementCost = 70;
315 // Indicate that this distribution may take variable number of samples
316 // during the runtime.
317 static const bool kVariableSamplesPerOutput = false;
318 typedef Array<Eigen::half, kResultElementCount> ResultType;
319 typedef Eigen::half ResultElementType;
320
321 PHILOX_DEVICE_INLINE
322 ResultType operator()(Generator* gen) {
323 typename Generator::ResultType sample = (*gen)();
324 ResultType result;
325 for (int i = 0; i < kResultElementCount; i += 2) {
326 float f[2];
327 BoxMullerFloat(sample[i], sample[i + 1], &f[0], &f[1]);
328 result[i] = Eigen::half(f[0]);
329 result[i + 1] = Eigen::half(f[1]);
330 }
331 return result;
332 }
333};

415// A class that returns standard normal distribution between
416// [-kTruncateValue, kTruncateValue].
417// Arguments:
418// Generator: a generator type that returns a number of uint32 upon each
419// each invocation. It needs to define kResultElementCount for the
420// sample count for each invocation, and ResultType for actual
421// returned sample type.
422// RealType: the data type of the real numbers that will be returned by the
423// distribution. This could be either float or double for now.
424// This class is meant to be implemented through specialization. The default
425// is not defined by design.
426template <class SingleSampleGenerator, typename RealType>
427class TruncatedNormalDistribution;
428
429// Exactly like the float version, except that we convert to half afterwards;
430// since we don't have half-precision sin/cos even on GPUs, there's nothing to
431// gain from working in half internally.
432template <class SingleSampleGenerator>
433class TruncatedNormalDistribution<SingleSampleGenerator, Eigen::half> {
434 public:
435 // The number of elements that will be returned.
436 static const int kResultElementCount =
437 SingleSampleGenerator::kNativeElementCount;
438 // Cost of generation of a single element (in cycles).
439 static const int kElementCost = 90;
440 // Indicate that this distribution may take variable number of samples
441 // during the runtime.
442 static const bool kVariableSamplesPerOutput = true;
443 // The threshold where the normal distribution is truncated.
444 const float kTruncateValue = 2.0f;
445
446 typedef Array<Eigen::half, kResultElementCount> ResultType;
447 typedef Eigen::half ResultElementType;
448
449 PHILOX_DEVICE_INLINE
450 ResultType operator()(SingleSampleGenerator* gen) {
451 ResultType results;
452 int index = 0;
453 while (true) {
454 // Repeatedly take samples from the normal distribution, until we have
455 // the desired number of elements that fall within the pre-defined cutoff
456 // threshold.
457 const uint32 x0 = (*gen)();
458 const uint32 x1 = (*gen)();
459 float f[2];
460 BoxMullerFloat(x0, x1, &f[0], &f[1]);
461
462 for (int i = 0; i < 2; ++i) {
463 if (Eigen::numext::abs(f[i]) < kTruncateValue) {
464 results[index++] = Eigen::half(f[i]);
465 if (index >= kResultElementCount) {
466 return results;
467 }
468 }
469 }
470 }
471 }
472};

601// Helper function to convert two 32-bit uniform integers to two floats
602// under the unit normal distribution.
603PHILOX_DEVICE_INLINE
604void BoxMullerFloat(uint32 x0, uint32 x1, float* f0, float* f1) {
605 // This function implements the Box-Muller transform:
606 // http://en.wikipedia.org/wiki/Box%E2%80%93Muller_transform#Basic_form
607 // Do not send a really small number to log().
608 // We cannot mark "epsilon" as "static const" because NVCC would complain
609 const float epsilon = 1.0e-7f;
610 float u1 = Uint32ToFloat(x0);
611 if (u1 < epsilon) {
612 u1 = epsilon;
613 }
614 const float v1 = 2.0f * M_PI * Uint32ToFloat(x1);
615 const float u2 = Eigen::numext::sqrt(-2.0f * Eigen::numext::log(u1));
616#if defined(TENSORFLOW_USE_SYCL) || !defined(__linux__)
617 *f0 = Eigen::numext::sin(v1);
618 *f1 = Eigen::numext::cos(v1);
619#else
620 sincosf(v1, f0, f1);
621#endif
622 *f0 *= u2;
623 *f1 *= u2;
624}

626// Helper function to convert four 32-bit uniform integers to two doubles
627// under the unit normal distribution.
628PHILOX_DEVICE_INLINE
629void BoxMullerDouble(uint32 x0, uint32 x1, uint32 x2, uint32 x3, double* d0,
630 double* d1) {
631 // This function implements the Box-Muller transform:
632 // http://en.wikipedia.org/wiki/Box%E2%80%93Muller_transform#Basic_form
633 // Do not send a really small number to log().
634 // We cannot mark "epsilon" as "static const" because NVCC would complain
635 const double epsilon = 1.0e-7;
636 double u1 = Uint64ToDouble(x0, x1);
637 if (u1 < epsilon) {
638 u1 = epsilon;
639 }
640 const double v1 = 2 * M_PI * Uint64ToDouble(x2, x3);
641 const double u2 = Eigen::numext::sqrt(-2.0 * Eigen::numext::log(u1));
642#if defined(TENSORFLOW_USE_SYCL) || !defined(__linux__)
643 *d0 = Eigen::numext::sin(v1);
644 *d1 = Eigen::numext::cos(v1);
645#else
646 sincos(v1, d0, d1);
647#endif
648 *d0 *= u2;
649 *d1 *= u2;
650}

652// Helper function to convert an 16-bit integer to a half between [0..1).
653PHILOX_DEVICE_INLINE Eigen::half Uint16ToHalf(uint16 x) {
654 // IEEE754 halfs are formatted as follows (MSB first):
655 // sign(1) exponent(5) mantissa(10)
656 // Conceptually construct the following:
657 // sign == 0
658 // exponent == 15 -- an excess 15 representation of a zero exponent
659 // mantissa == 10 random bits
660 const uint16 man = x & 0x3ffu; // 10 bit mantissa
661 const uint16 exp = static_cast<uint16>(15);
662 const uint16 val = (exp << 10) | man;
663
664 Eigen::half result;
665 result.x = val;
666 return result - Eigen::half(1.0);
667}
668
669// Helper function to convert an 16-bit integer to a bfloat16 between [0..1).
670// This can create a uniform distribution of values between [0..1).
671PHILOX_DEVICE_INLINE bfloat16 Uint16ToGfloat16(uint16 x) {
672 // bfloat are formatted as follows (MSB first):
673 // sign(1) exponent(8) mantissa(7)
674 // Conceptually construct the following:
675 // sign == 0
676 // exponent == 127 -- an excess 127 representation of a zero exponent
677 // mantissa == 7 random bits
678 const uint16 man = x & 0x7fu; // 7 bit mantissa
679 const uint16 exp = static_cast<uint16>(127);
680 const uint16 val = (exp << 7) | man;
681
682 bfloat16 result;
683 memcpy(&result, &val, sizeof(val));
684 // The mantissa has an implicit leading 1, so the above code creates a value
685 // in [1, 2). The minus will not cause a rounding that makes the result 1.
686 // Instead it will just be close to 1.
687 return result - bfloat16(1.0);
688}

690// Helper function to convert an 32-bit integer to a float between [0..1).
691PHILOX_DEVICE_INLINE float Uint32ToFloat(uint32 x) {
692 // IEEE754 floats are formatted as follows (MSB first):
693 // sign(1) exponent(8) mantissa(23)
694 // Conceptually construct the following:
695 // sign == 0
696 // exponent == 127 -- an excess 127 representation of a zero exponent
697 // mantissa == 23 random bits
698 const uint32 man = x & 0x7fffffu; // 23 bit mantissa
699 const uint32 exp = static_cast<uint32>(127);
700 const uint32 val = (exp << 23) | man;
701
702 // Assumes that endian-ness is same for float and uint32.
703 float result;
704 memcpy(&result, &val, sizeof(val));
705 return result - 1.0f;
706}

708// Helper function to convert two 32-bit integers to a double between [0..1).
709PHILOX_DEVICE_INLINE double Uint64ToDouble(uint32 x0, uint32 x1) {
710 // IEEE754 doubles are formatted as follows (MSB first):
711 // sign(1) exponent(11) mantissa(52)
712 // Conceptually construct the following:
713 // sign == 0
714 // exponent == 1023 -- an excess 1023 representation of a zero exponent
715 // mantissa == 52 random bits
716 const uint32 mhi = x0 & 0xfffffu; // upper 20 bits of mantissa
717 const uint32 mlo = x1; // lower 32 bits of mantissa
718 const uint64 man = (static_cast<uint64>(mhi) << 32) | mlo; // mantissa
719 const uint64 exp = static_cast<uint64>(1023);
720 const uint64 val = (exp << 52) | man;
721 // Assumes that endian-ness is same for double and uint64.
722 double result;
723 memcpy(&result, &val, sizeof(val));
724 return result - 1.0;
725}
