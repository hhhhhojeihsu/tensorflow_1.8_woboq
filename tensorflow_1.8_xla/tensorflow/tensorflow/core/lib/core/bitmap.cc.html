23void Bitmap::Reset(size_t n) {
24 const size_t num_words = NumWords(n);
25 if (num_words != NumWords(nbits_)) {
26 // Reallocate.
27 Word* w = new Word[num_words];
28 delete[] word_;
29 word_ = w;
30 }
31 memset(word_, 0, sizeof(word_[0]) * num_words);
32 nbits_ = n;
33}
34
35// Return 1+index of the first set bit in w; return 0 if w == 0.
36static size_t FindFirstSet(uint32 w) {
37 // TODO(jeff,sanjay): If this becomes a performance issue, we could
38 // use the __builtin_ffs(w) routine on GCC, or the ffs(w) routine on
39 // some other platforms.
40
41 // clang-format off
42 static uint8 kLowestBitSet[256] = {
43 /* 0*/ 0, 1, 2, 1, 3, 1, 2, 1, 4, 1, 2, 1, 3, 1, 2, 1,
44 /* 16*/ 5, 1, 2, 1, 3, 1, 2, 1, 4, 1, 2, 1, 3, 1, 2, 1,
45 /* 32*/ 6, 1, 2, 1, 3, 1, 2, 1, 4, 1, 2, 1, 3, 1, 2, 1,
46 /* 48*/ 5, 1, 2, 1, 3, 1, 2, 1, 4, 1, 2, 1, 3, 1, 2, 1,
47 /* 64*/ 7, 1, 2, 1, 3, 1, 2, 1, 4, 1, 2, 1, 3, 1, 2, 1,
48 /* 80*/ 5, 1, 2, 1, 3, 1, 2, 1, 4, 1, 2, 1, 3, 1, 2, 1,
49 /* 96*/ 6, 1, 2, 1, 3, 1, 2, 1, 4, 1, 2, 1, 3, 1, 2, 1,
50 /*112*/ 5, 1, 2, 1, 3, 1, 2, 1, 4, 1, 2, 1, 3, 1, 2, 1,
51 /*128*/ 8, 1, 2, 1, 3, 1, 2, 1, 4, 1, 2, 1, 3, 1, 2, 1,
52 /*144*/ 5, 1, 2, 1, 3, 1, 2, 1, 4, 1, 2, 1, 3, 1, 2, 1,
53 /*160*/ 6, 1, 2, 1, 3, 1, 2, 1, 4, 1, 2, 1, 3, 1, 2, 1,
54 /*176*/ 5, 1, 2, 1, 3, 1, 2, 1, 4, 1, 2, 1, 3, 1, 2, 1,
55 /*192*/ 7, 1, 2, 1, 3, 1, 2, 1, 4, 1, 2, 1, 3, 1, 2, 1,
56 /*208*/ 5, 1, 2, 1, 3, 1, 2, 1, 4, 1, 2, 1, 3, 1, 2, 1,
57 /*224*/ 6, 1, 2, 1, 3, 1, 2, 1, 4, 1, 2, 1, 3, 1, 2, 1,
58 /*240*/ 5, 1, 2, 1, 3, 1, 2, 1, 4, 1, 2, 1, 3, 1, 2, 1,
59 };
60 // clang-format on
61 if (w & 0xff) {
62 return kLowestBitSet[w & 0xff];
63 } else if ((w >> 8) & 0xff) {
64 return kLowestBitSet[(w >> 8) & 0xff] + 8;
65 } else if ((w >> 16) & 0xff) {
66 return kLowestBitSet[(w >> 16) & 0xff] + 16;
67 } else if ((w >> 24) & 0xff) {
68 return kLowestBitSet[(w >> 24) & 0xff] + 24;
69 } else {
70 return 0;
71 }
72}

74size_t Bitmap::FirstUnset(size_t start) const {
75 if (start >= nbits_) {
76 return nbits_;
77 }
78
79 // Mask to or-into first word to account for bits to skip in that word.
80 size_t mask = (1ull << (start % kBits)) - 1;
81 const size_t nwords = NumWords(nbits_);
82 for (size_t i = start / kBits; i < nwords; i++) {
83 Word word = word_[i] | mask;
84 mask = 0; // Only ignore bits in the first word we process.
85 size_t r = FindFirstSet(~word);
86
87 if (r) {
88 size_t result = i * kBits + (r - 1);
89 if (result > nbits_) result = nbits_;
90 return result;
91 }
92 }
93
94 return nbits_;
95}

97string Bitmap::ToString() const {
98 string result;
99 result.resize(bits());
100 for (int i = 0; i < nbits_; i++) {
101 result[i] = get(i) ? '1' : '0';
102 }
103 return result;
104}
