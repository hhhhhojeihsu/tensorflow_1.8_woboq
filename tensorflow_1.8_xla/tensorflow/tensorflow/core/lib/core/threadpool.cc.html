31struct EigenEnvironment {
32 typedef Thread EnvThread;
33 struct TaskImpl {
34 std::function<void()> f;
35 Context context;
36 uint64 trace_id;
37 };
38 struct Task {
39 std::unique_ptr<TaskImpl> f;
40 };
41
42 Env* const env_;
43 const ThreadOptions thread_options_;
44 const string name_;
45
46 EigenEnvironment(Env* env, const ThreadOptions& thread_options,
47 const string& name)
48 : env_(env), thread_options_(thread_options), name_(name) {}
49
50 EnvThread* CreateThread(std::function<void()> f) {
51 return env_->StartThread(thread_options_, name_, [=]() {
52 // Set the processor flag to flush denormals to zero.
53 port::ScopedFlushDenormal flush;
54 // Set the processor rounding mode to ROUND TO NEAREST.
55 port::ScopedSetRound round(FE_TONEAREST);
56 f();
57 });
58 }
59
60 Task CreateTask(std::function<void()> f) {
61 uint64 id = 0;
62 if (port::Tracing::IsActive()) {
63 id = port::Tracing::UniqueId();
64 port::Tracing::RecordEvent(port::Tracing::EventCategory::kScheduleClosure,
65 id);
66 }
67 return Task{
68 std::unique_ptr<TaskImpl>(new TaskImpl{
69 std::move(f),
70 Context(ContextKind::kThread),
71 id,
72 }),
73 };
74 }
75
76 void ExecuteTask(const Task& t) {
77 WithContext wc(t.f->context);
78 if (t.f->trace_id != 0) {
79 port::Tracing::ScopedActivity region(
80 port::Tracing::EventCategory::kRunClosure, t.f->trace_id);
81 t.f->f();
82 } else {
83 t.f->f();
84 }
85 }
86};
87
88struct ThreadPool::Impl : Eigen::ThreadPoolTempl<EigenEnvironment> {
89 Impl(Env* env, const ThreadOptions& thread_options, const string& name,
90 int num_threads, bool low_latency_hint)
91 : Eigen::ThreadPoolTempl<EigenEnvironment>(
92 num_threads, low_latency_hint,
93 EigenEnvironment(env, thread_options, name)) {}
94
95 void ParallelFor(int64 total, int64 cost_per_unit,
96 std::function<void(int64, int64)> fn) {
97 CHECK_GE(total, 0);
98 CHECK_EQ(total, (int64)(Eigen::Index)total);
99 Eigen::ThreadPoolDevice device(this, this->NumThreads());
100 device.parallelFor(
101 total, Eigen::TensorOpCost(0, 0, cost_per_unit),
102 [&fn](Eigen::Index first, Eigen::Index last) { fn(first, last); });
103 }
104};
105
106ThreadPool::ThreadPool(Env* env, const string& name, int num_threads)
107 : ThreadPool(env, ThreadOptions(), name, num_threads, true) {}
108
109ThreadPool::ThreadPool(Env* env, const ThreadOptions& thread_options,
110 const string& name, int num_threads)
111 : ThreadPool(env, thread_options, name, num_threads, true) {}
112
113ThreadPool::ThreadPool(Env* env, const ThreadOptions& thread_options,
114 const string& name, int num_threads,
115 bool low_latency_hint) {
116 CHECK_GE(num_threads, 1);
117 impl_.reset(new ThreadPool::Impl(env, thread_options, "tf_" + name,
118 num_threads, low_latency_hint));
119}
120
121ThreadPool::~ThreadPool() {}
122
123void ThreadPool::Schedule(std::function<void()> fn) {
124 CHECK(fn != nullptr);
125 impl_->Schedule(std::move(fn));
126}
127
128void ThreadPool::ParallelFor(int64 total, int64 cost_per_unit,
129 std::function<void(int64, int64)> fn) {
130 impl_->ParallelFor(total, cost_per_unit, std::move(fn));
131}
132
133void ThreadPool::ParallelForWithWorkerId(
134 int64 total, int64 cost_per_unit,
135 const std::function<void(int64, int64, int)>& fn) {
136 impl_->ParallelFor(total, cost_per_unit,
137 [this, &fn](int64 start, int64 limit) {
138 // ParallelFor may use the current thread to do some
139 // work synchronously. When calling CurrentThreadId()
140 // from outside of the thread pool, we get -1, so we can
141 // shift every id up by 1.
142 int id = CurrentThreadId() + 1;
143 fn(start, limit, id);
144 });
145}
146
147int ThreadPool::NumThreads() const { return impl_->NumThreads(); }
148
149int ThreadPool::CurrentThreadId() const { return impl_->CurrentThreadId(); }
