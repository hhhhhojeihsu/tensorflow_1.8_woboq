28// The different metric kinds available.
29//
30// Gauge indicates that the metric's values are instantaneous measurements of a
31// (typically) continuously varying value. Examples: a process's current heap
32// size, a queue's current length, the name of the binary used by a process,
33// whether a task is complete.
34//
35// Cumulative indicates that the metric's values represent non-negative changes
36// over specified time periods. Example: the number of rpc calls to a service.
37enum class MetricKind : int { kGauge = 0, kCumulative };
38
39// The type of the metric values.
40enum class ValueType : int { kInt64 = 0, kHistogram, kString, kBool };
41
42// Everything in the internal namespace is implementation details. Do not depend
43// on this.
44namespace internal {
45
46template <typename Value>
47ValueType GetValueType();
48
49template <>
50inline ValueType GetValueType<int64>() {
51 return ValueType::kInt64;
52}
53
54template <>
55inline ValueType GetValueType<HistogramProto>() {
56 return ValueType::kHistogram;
57}
58
59template <>
60inline ValueType GetValueType<string>() {
61 return ValueType::kString;
62}
63
64template <>
65inline ValueType GetValueType<bool>() {
66 return ValueType::kBool;
67}

69} // namespace internal

71// Abstract base class for a metric definition.
72//
73// Unlike MetricDef, this class is non-templatized and allows storing and
74// accessing metric definitions without the full type information.
75//
76// Everything except the value type of a metric is stored here. Please read
77// MetricDef class comments for more details.
78class AbstractMetricDef {
79 public:
80 MetricKind kind() const { return kind_; }
81
82 ValueType value_type() const { return value_type_; }
83
84 StringPiece name() const { return name_; }
85
86 StringPiece description() const { return description_; }
87
88 const std::vector<string>& label_descriptions() const {
89 return label_descriptions_;
90 }
91
92 private:
93 template <MetricKind kind, typename Value, int NumLabels>
94 friend class MetricDef;
95
96 AbstractMetricDef(const MetricKind kind, const ValueType value_type,
97 const StringPiece name, const StringPiece description,
98 const std::vector<string>& label_descriptions)
99 : kind_(kind),
100 value_type_(value_type),
101 name_(name.ToString()),
102 description_(description.ToString()),
103 label_descriptions_(std::vector<string>(label_descriptions.begin(),
104 label_descriptions.end())) {}
105
106 const MetricKind kind_;
107 const ValueType value_type_;
108 const string name_;
109 const string description_;
110 const std::vector<string> label_descriptions_;
111};
112
113// Metric definition.
114//
115// A metric is defined by its kind, value-type, name, description and the
116// description of its labels.
117//
118// NOTE: Name, description, and label descriptions should be logically static,
119// but do not have to live for the lifetime of the MetricDef.
120//
121// By "logically static", we mean that they should never contain dynamic
122// information, but is static for the lifetime of the MetricDef, and
123// in-turn the metric; they do not need to be compile-time constants.
124// This allows for e.g. prefixed metrics in a CLIF wrapped environment.
125template <MetricKind metric_kind, typename Value, int NumLabels>
126class MetricDef : public AbstractMetricDef {
127 public:
128 template <typename... LabelDesc>
129 MetricDef(const StringPiece name, const StringPiece description,
130 const LabelDesc&... label_descriptions)
131 : AbstractMetricDef(metric_kind, internal::GetValueType<Value>(), name,
132 description, {label_descriptions...}) {
133 static_assert(sizeof...(LabelDesc) == NumLabels,
134 "Mismatch between Counter<NumLabels> and number of label "
135 "descriptions.");
136 }
137};
