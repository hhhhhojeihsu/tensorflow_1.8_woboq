22void GroupIterable::IteratorStep::UpdateEndOfGroup() {
23 ++next_loc_;
24 int64 N = iter_->ix_.dim_size(0);
25 auto ix_t = iter_->ix_.template matrix<int64>();
26 while (next_loc_ < N && iter_->GroupMatches(ix_t, loc_, next_loc_)) {
27 ++next_loc_;
28 }
29}
30
31bool GroupIterable::IteratorStep::operator!=(const IteratorStep& rhs) const {
32 CHECK_EQ(rhs.iter_, iter_) << "Can't compare steps from different iterators";
33 return (rhs.loc_ != loc_);
34}
35
36bool GroupIterable::IteratorStep::operator==(const IteratorStep& rhs) const {
37 CHECK_EQ(rhs.iter_, iter_) << "Can't compare steps from different iterators";
38 return (rhs.loc_ == loc_);
39}
40
41GroupIterable::IteratorStep& GroupIterable::IteratorStep::
42operator++() { // prefix ++
43 loc_ = next_loc_;
44 UpdateEndOfGroup();
45 return *this;
46}
47
48GroupIterable::IteratorStep GroupIterable::IteratorStep::operator++(
49 int) { // postfix ++
50 IteratorStep lhs(*this);
51 ++(*this);
52 return lhs;
53}
54
55std::vector<int64> Group::group() const {
56 std::vector<int64> g;
57 auto ix_t = iter_->ix_.template matrix<int64>();
58 for (const int d : iter_->group_dims_) {
59 g.push_back(ix_t(loc_, d));
60 }
61 return g;
62}
63
64TTypes<int64>::UnalignedConstMatrix Group::indices() const {
65 return TTypes<int64>::UnalignedConstMatrix(
66 &(iter_->ix_.matrix<int64>()(loc_, 0)), next_loc_ - loc_, iter_->dims_);
67}
