21namespace tensorflow {
22
23string MetaFilename(StringPiece prefix) {
24 return strings::Printf("%.*s.index", static_cast<int>(prefix.size()),
25 prefix.data());
26}
27
28string DataFilename(StringPiece prefix, int32 shard_id, int32 num_shards) {
29 DCHECK_GT(num_shards, 0);
30 DCHECK_LT(shard_id, num_shards);
31 return strings::Printf("%.*s.data-%05d-of-%05d",
32 static_cast<int>(prefix.size()), prefix.data(),
33 shard_id, num_shards);
34}
