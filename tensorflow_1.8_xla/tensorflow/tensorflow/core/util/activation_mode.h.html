29// ActivationMode: the activation function we apply to the input tensor:
30enum ActivationMode {
31 NONE = 0,
32 SIGMOID = 1,
33 RELU = 2,
34 RELU6 = 3,
35 RELUX = 4,
36 TANH = 5,
37 BANDPASS = 6,
38};

40// Specialization to parse an attribute directly into a ActivationMode enum.
41Status GetActivationModeFromString(const string& str_value,
42 ActivationMode* value);
