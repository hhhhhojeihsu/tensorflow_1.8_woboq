26uint64 DecodeUint64LittleEndian(const uint8* buffer) {
27 uint64 result = 0;
28 for (int i = 0; i < static_cast<int>(sizeof(uint64)); ++i) {
29 result |= static_cast<uint64>(buffer[i]) << (8 * i);
30 }
31 return result;
32}

38class ReadOnlyMemoryRegionFromMemmapped : public ReadOnlyMemoryRegion {
39 public:
40 ReadOnlyMemoryRegionFromMemmapped(const void* data, uint64 length)
41 : data_(data), length_(length) {}
42 ~ReadOnlyMemoryRegionFromMemmapped() override = default;
43 const void* data() override { return data_; }
44 uint64 length() override { return length_; }
45
46 private:
47 const void* const data_;
48 const uint64 length_;
49 // intentionally copyable
50};
51
52class RandomAccessFileFromMemmapped : public RandomAccessFile {
53 public:
54 RandomAccessFileFromMemmapped(const void* data, uint64 length)
55 : data_(data), length_(length) {}
56
57 ~RandomAccessFileFromMemmapped() override = default;
58
59 Status Read(uint64 offset, size_t to_read, StringPiece* result,
60 char* scratch) const override {
61 if (offset >= length_) {
62 *result = StringPiece(scratch, 0);
63 return Status(error::OUT_OF_RANGE, "Read after file end");
64 }
65 const uint64 region_left =
66 std::min(length_ - offset, static_cast<uint64>(to_read));
67 *result =
68 StringPiece(reinterpret_cast<const char*>(data_) + offset, region_left);
69 return (region_left == to_read)
70 ? Status::OK()
71 : Status(error::OUT_OF_RANGE, "Read less bytes than requested");
72 }
73
74 private:
75 const void* const data_;
76 const uint64 length_;
77 // intentionally copyable
78};

84Status MemmappedFileSystem::FileExists(const string& fname) {
85 if (!mapped_memory_) {
86 return errors::FailedPrecondition("MemmappedEnv is not initialized");
87 }
88 const auto dir_element = directory_.find(fname);
89 if (dir_element != directory_.end()) {
90 return Status::OK();
91 }
92 return errors::NotFound(fname, " not found");
93}

95Status MemmappedFileSystem::NewRandomAccessFile(
96 const string& filename, std::unique_ptr<RandomAccessFile>* result) {
97 if (!mapped_memory_) {
98 return errors::FailedPrecondition("MemmappedEnv is not initialized");
99 }
100 const auto dir_element = directory_.find(filename);
101 if (dir_element == directory_.end()) {
102 return errors::NotFound("Region ", filename, " is not found");
103 }
104 result->reset(new RandomAccessFileFromMemmapped(
105 GetMemoryWithOffset(dir_element->second.offset),
106 dir_element->second.length));
107 return Status::OK();
108}

196Status MemmappedFileSystem::InitializeFromFile(Env* env,
197 const string& filename) {
198 TF_RETURN_IF_ERROR(
199 env->NewReadOnlyMemoryRegionFromFile(filename, &mapped_memory_));
200 directory_.clear();
201 if (mapped_memory_->length() <= sizeof(uint64)) {
202 return errors::DataLoss("Corrupted memmapped model file: ", filename,
203 " Invalid package size");
204 }
205 const auto memory_start =
206 reinterpret_cast<const uint8*>(mapped_memory_->data());
207 const uint64 directory_offset = DecodeUint64LittleEndian(
208 memory_start + mapped_memory_->length() - sizeof(uint64));
209 if (directory_offset > mapped_memory_->length() - sizeof(uint64)) {
210 return errors::DataLoss("Corrupted memmapped model file: ", filename,
211 " Invalid directory offset");
212 }
213 MemmappedFileSystemDirectory proto_directory;
214 if (!ParseProtoUnlimited(
215 &proto_directory, memory_start + directory_offset,
216 mapped_memory_->length() - directory_offset - sizeof(uint64))) {
217 return errors::DataLoss("Corrupted memmapped model file: ", filename,
218 " Can't parse its internal directory");
219 }
220
221 // Iterating in reverse order to get lengths of elements;
222 uint64 prev_element_offset = directory_offset;
223 for (auto element_iter = proto_directory.element().rbegin();
224 element_iter != proto_directory.element().rend(); ++element_iter) {
225 // Check that the element offset is in the right range.
226 if (element_iter->offset() >= prev_element_offset) {
227 return errors::DataLoss("Corrupted memmapped model file: ", filename,
228 " Invalid offset of internal component");
229 }
230 if (!directory_
231 .insert(std::make_pair(
232 element_iter->name(),
233 FileRegion(element_iter->offset(),
234 prev_element_offset - element_iter->offset())))
235 .second) {
236 return errors::DataLoss("Corrupted memmapped model file: ", filename,
237 " Duplicate name of internal component ",
238 element_iter->name());
239 }
240 prev_element_offset = element_iter->offset();
241 }
242 return Status::OK();
243}

250bool IsValidRegionChar(char c) {
251 return (c >= 'A' && c <= 'Z') || (c >= 'a' && c <= 'z') ||
252 (c >= '0' && c <= '9') || c == '_' || c == '.';
253}

256bool MemmappedFileSystem::IsWellFormedMemmappedPackageFilename(
257 const string& filename) {
258 if (!IsMemmappedPackageFilename(filename)) {
259 return false;
260 }
261 for (char c :
262 filename.substr(strlen(kMemmappedPackagePrefix),
263 filename.length() - strlen(kMemmappedPackagePrefix))) {
264 if (!IsValidRegionChar(c)) {
265 return false;
266 }
267 }
268 return true;
269}

273Status MemmappedEnv::GetFileSystemForFile(const string& fname,
274 FileSystem** result) {
275 if (MemmappedFileSystem::IsMemmappedPackageFilename(fname)) {
276 if (!memmapped_file_system_) {
277 return errors::FailedPrecondition(
278 "MemmappedEnv is not initialized from a file.");
279 }
280 *result = memmapped_file_system_.get();
281 return Status::OK();
282 }
283 return EnvWrapper::GetFileSystemForFile(fname, result);
284}

295Status MemmappedEnv::InitializeFromFile(const string& package_filename) {
296 std::unique_ptr<MemmappedFileSystem> file_system_ptr(new MemmappedFileSystem);
297 const auto status =
298 file_system_ptr->InitializeFromFile(target(), package_filename);
299 if (status.ok()) {
300 memmapped_file_system_ = std::move(file_system_ptr);
301 }
302 return status;
303}
