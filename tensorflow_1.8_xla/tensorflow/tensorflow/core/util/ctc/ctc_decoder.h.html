29// The CTCDecoder is an abstract interface to be implemented when providing a
30// decoding method on the timestep output of a RNN trained with CTC loss.
31//
32// The two types of decoding available are:
33// - greedy path, through the CTCGreedyDecoder
34// - beam search, through the CTCBeamSearchDecoder
35class CTCDecoder {
36 public:
37 typedef Eigen::Map<const Eigen::ArrayXi> SequenceLength;
38 typedef Eigen::Map<const Eigen::MatrixXf> Input;
39 typedef std::vector<std::vector<int>> Output;
40 typedef Eigen::Map<Eigen::MatrixXf> ScoreOutput;
41
42 CTCDecoder(int num_classes, int batch_size, bool merge_repeated)
43 : num_classes_(num_classes),
44 blank_index_(num_classes - 1),
45 batch_size_(batch_size),
46 merge_repeated_(merge_repeated) {}
47
48 virtual ~CTCDecoder() {}
49
50 // Dimensionality of the input/output is expected to be:
51 // - seq_len[b] - b = 0 to batch_size_
52 // - input[t].rows(b) - t = 0 to timesteps; b = 0 t batch_size_
53 // - output.size() specifies the number of beams to be returned.
54 // - scores(b, i) - b = 0 to batch_size; i = 0 to output.size()
55 virtual Status Decode(const SequenceLength& seq_len,
56 const std::vector<Input>& input,
57 std::vector<Output>* output, ScoreOutput* scores) = 0;
58
59 int batch_size() { return batch_size_; }
60 int num_classes() { return num_classes_; }
61
62 protected:
63 int num_classes_;
64 int blank_index_;
65 int batch_size_;
66 bool merge_repeated_;
67};
68
69// CTCGreedyDecoder is an implementation of the simple best path decoding
70// algorithm, selecting at each timestep the most likely class at each timestep.
71class CTCGreedyDecoder : public CTCDecoder {
72 public:
73 CTCGreedyDecoder(int num_classes, int batch_size, bool merge_repeated)
74 : CTCDecoder(num_classes, batch_size, merge_repeated) {}
75
76 Status Decode(const CTCDecoder::SequenceLength& seq_len,
77 const std::vector<CTCDecoder::Input>& input,
78 std::vector<CTCDecoder::Output>* output,
79 CTCDecoder::ScoreOutput* scores) override {
80 if (output->empty() || (*output)[0].size() < batch_size_) {
81 return errors::InvalidArgument(
82 "output needs to be of size at least (1, batch_size).");
83 }
84 if (scores->rows() < batch_size_ || scores->cols() == 0) {
85 return errors::InvalidArgument(
86 "scores needs to be of size at least (batch_size, 1).");
87 }
88 // For each batch entry, identify the transitions
89 for (int b = 0; b < batch_size_; ++b) {
90 int seq_len_b = seq_len[b];
91 // Only writing to beam 0
92 std::vector<int>& output_b = (*output)[0][b];
93
94 int prev_class_ix = -1;
95 (*scores)(b, 0) = 0;
96 for (int t = 0; t < seq_len_b; ++t) {
97 auto row = input[t].row(b);
98 int max_class_ix;
99 (*scores)(b, 0) += -row.maxCoeff(&max_class_ix);
100 if (max_class_ix != blank_index_ &&
101 !(merge_repeated_ && max_class_ix == prev_class_ix)) {
102 output_b.push_back(max_class_ix);
103 }
104 prev_class_ix = max_class_ix;
105 }
106 }
107 return Status::OK();
108 }
109};
