15// This checker checks common wrong configurations of operations.
16//
17#ifndef TENSORFLOW_CORE_PROFILER_INTERNAL_ADVISOR_OPERATION_CHECKER_H_
18#define TENSORFLOW_CORE_PROFILER_INTERNAL_ADVISOR_OPERATION_CHECKER_H_
19
20#include "tensorflow/core/profiler/internal/advisor/checker.h"
21
22namespace tensorflow {
23namespace tfprof {
24
25class OperationChecker : public Checker {
26 public:
27 string name() const override { return kCheckers[1]; }
28
29 private:
30 AdviceProto::Checker Check(const AdvisorOptionsProto::CheckerOption& options,
31 const TFStats* stats) override {
32 if (!stats) {
33 fprintf(stderr, "Missing profiles (e.g. graph, run_meta). Skip %s\n",
34 name().c_str());
35 return reports_;
36 }
37 bool use_batch_norm = false;
38 bool use_fused_batch_norm = false;
39 bool recommend_nchw = false;
40 for (const auto& n : stats->nodes()) {
41 const TFGraphNode* node = n.second.get();
42 if (node->name().find("BatchNorm") != node->name().npos) {
43 use_batch_norm = true;
44 }
45 if (node->op_types().find("FusedBatchNorm") != node->op_types().end()) {
46 use_fused_batch_norm = true;
47 }
48
49 const AttrValue* attr = node->op_attrs("data_format");
50 if (attr) {
51 if (attr->s() == "NHWC" &&
52 IsPlacedOnAccelerator(node->canonical_device())) {
53 recommend_nchw = true;
54 }
55 }
56 }
57 if (use_batch_norm && !use_fused_batch_norm) {
58 reports_.add_reports(
59 "Maybe use faster FusedBatchNorm instead of BatchNorm");
60 }
61 if (recommend_nchw) {
62 // TODO(xpan): Maybe print which Op supports NCHW.
63 reports_.add_reports(
64 "Found operation using NHWC data_format on GPU. Maybe "
65 "NCHW is faster.");
66 }
67 return reports_;
68 }
69
70 private:
71 AdviceProto::Checker reports_;
72};
73
74} // namespace tfprof
75} // namespace tensorflow
76
77#endif // TENSORFLOW_CORE_PROFILER_INTERNAL_ADVISOR_OPERATION_CHECKER_H_
