16// Master implements the service MasterSerivce.
17//
18// A Master maintains the state of live graph computation
19// sessions, each session orchestrates both local and remote devices
20// to carry out the graph computation.
21//
22// A Master knows ahead of time local devices available as
23// client devices.
24//
25// A Master discovers remote devices on-demand and keeps track of
26// statistics of those remote devices.
27//
28// Each session analyzes the graph, places nodes across available
29// devices, and ultimately drives the graph computation by initiating
30// RunGraph on the workers.
31

63Master::Master(MasterEnv* env, double session_gc_seconds)
64 : env_(env),
65 last_1000_steps_(1000),
66 step_count_(0),
67 session_gc_seconds_(session_gc_seconds) {
68 // Right now, a master service must be co-located with a device.
69 // Otherwise, fetches do not work.
70 CHECK(!env->local_devices.empty());
71
72 if (session_gc_seconds_ > 0.0) {
73 gc_thread_ = env_->env->StartThread(ThreadOptions(), "TF_master_GC",
74 [this]() { GC(); });
75 } else {
76 gc_thread_ = nullptr;
77 }
78}
79
80Master::~Master() {
81 if (gc_thread_) {
82 mutex_lock l(mu_);
83 shutdown_ = true;
84 shutdown_cv_.notify_all();
85 delete gc_thread_;
86 }
87}

89void Master::GC() {
90 Env* env = Env::Default();
91 while (true) {
92 mutex_lock l(mu_);
93 const int kTimeoutMilliseconds = 10 * 1000; // 10 seconds.
94 WaitForMilliseconds(&l, &shutdown_cv_, kTimeoutMilliseconds);
95 if (shutdown_) {
96 break;
97 }
98 std::vector<string> handles;
99 const int64 num_micros = static_cast<int64>(session_gc_seconds_ * 1000000);
100 for (const auto& entry : sessions_) {
101 int64 lat = entry.second->last_access_time_usec();
102 if (static_cast<int64>(env->NowMicros()) - lat > num_micros) {
103 handles.push_back(entry.first);
104 auto* sess = entry.second;
105 SchedClosure([this, sess]() {
106 LOG(WARNING) << "GC session " << sess->handle() << " after "
107 << session_gc_seconds_ << " seconds. "
108 << "Note that if you are starting multiple replicas "
109 << "on a staggered delay, session_gc_seconds may need "
110 << "to be raised.";
111 sess->GarbageCollect();
112 });
113 }
114 }
115 for (const auto& handle : handles) sessions_.erase(handle);
116 }
117}

119MasterSession* Master::FindMasterSession(const string& handle) {
120 MasterSession* session = nullptr;
121 {
122 mutex_lock l(mu_);
123 session = gtl::FindPtrOrNull(sessions_, handle);
124 if (session != nullptr) {
125 session->Ref();
126 }
127 }
128 return session;
129}

131class DeviceFinder {
132 public:
133 static Status GetRemoteDevices(
134 const protobuf::RepeatedPtrField<string>& device_filters, MasterEnv* env,
135 WorkerCacheInterface* worker_cache,
136 std::vector<std::unique_ptr<Device>>* out_remote) {
137 DeviceFinder finder(device_filters, env, worker_cache);
138 finder.Start();
139 TF_RETURN_IF_ERROR(finder.Wait());
140 finder.GetRemoteDevices(env->local_devices, out_remote);
141 return Status::OK();
142 }
143
144 static void GetRemoteWorkers(
145 const protobuf::RepeatedPtrField<string>& device_filters, MasterEnv* env,
146 WorkerCacheInterface* worker_cache, std::vector<string>* workers) {
147 DeviceFinder finder(device_filters, env, worker_cache);
148 *workers = finder.targets_;
149 }
150
151 private:
152 explicit DeviceFinder(
153 const protobuf::RepeatedPtrField<string>& device_filters, MasterEnv* env,
154 WorkerCacheInterface* worker_cache)
155 : env_(env), worker_cache_(worker_cache) {
156 CHECK(worker_cache) << "Worker cache was null!";
157 auto process_filter = [this](const string& filter) {
158 DeviceNameUtils::ParsedName parsed;
159 if (DeviceNameUtils::ParseFullName(filter, &parsed)) {
160 filters_.push_back(parsed);
161 } else {
162 LOG(FATAL) << "Skipping invalid filter: " << filter;
163 }
164 };
165 for (const string& filter : device_filters) {
166 process_filter(filter);
167 }
168 // Enumerates all known workers' target. A target name is a
169 // prefix of a device name. E.g., /job:mnist/replica:0/task:10.
170 std::vector<string> workers;
171 worker_cache->ListWorkers(&workers);
172 if (filters_.empty()) {
173 std::swap(workers, targets_);
174 } else {
175 for (const string& name : workers) {
176 if (MatchFilters(name)) {
177 targets_.push_back(name);
178 }
179 }
180 }
181 seen_targets_.assign(targets_.size(), false);
182 }
183
184 ~DeviceFinder() {
185 for (Device* dev : found_) delete dev;
186 }
187
188 void Start() {
189 {
190 mutex_lock l(mu_);
191 num_pending_ = targets_.size();
192 if (num_pending_ == 0) {
193 pending_zero_.notify_all();
194 }
195 }
196 // Talk to all workers to get the list of available devices.
197 using std::placeholders::_1;
198 using std::placeholders::_2;
199 for (size_t i = 0; i < targets_.size(); ++i) {
200 // TODO(mrry): Propagate a timeout here, since `this->WhenFound()` may
201 // never be called.
202 NewRemoteDevices(env_->env, worker_cache_, targets_[i],
203 std::bind(&ME::WhenFound, this, i, _1, _2));
204 }
205 }
206
207 // Every `kLoggingPeriodMs`, while the DeviceFinder is still waiting
208 // to hear from workers, log a list of the workers who have not
209 // responded.
210 const int32 kLoggingPeriodMs = 10 * 1000;
211
212 Status Wait() {
213 mutex_lock l(mu_);
214 // TODO(mrry): Propagate a timeout here, since `num_pending_` may
215 // never become zero.
216 while (num_pending_ != 0) {
217 pending_zero_.wait_for(l, std::chrono::milliseconds(kLoggingPeriodMs));
218 if (num_pending_ != 0) {
219 for (size_t i = 0; i < targets_.size(); ++i) {
220 if (!seen_targets_[i]) {
221 LOG(INFO)
222 << "CreateSession still waiting for response from worker: "
223 << targets_[i];
224 }
225 }
226 }
227 }
228 return status_;
229 }

231 // The caller takes the ownership of returned remote devices.
232 void GetRemoteDevices(const std::vector<Device*>& local,
233 std::vector<std::unique_ptr<Device>>* remote) {
234 std::unordered_set<string> names(local.size());
235 for (Device* dev : local) names.insert(dev->name());
236 mutex_lock l(mu_);
237 for (Device* dev : found_) {
238 const string& name = dev->name();
239 if (names.insert(name).second && MatchFilters(name)) {
240 remote->push_back(std::unique_ptr<Device>(dev));
241 } else {
242 delete dev;
243 }
244 }
245 found_.clear();
246 }
247
248 typedef DeviceFinder ME;
249 const MasterEnv* env_;
250 WorkerCacheInterface* worker_cache_;
251 std::vector<DeviceNameUtils::ParsedName> filters_;
252
253 mutex mu_;
254 int num_pending_ GUARDED_BY(mu_);
255 condition_variable pending_zero_;
256 std::vector<Device*> found_ GUARDED_BY(mu_);
257 // List of targets to be contacted by this DeviceFinder. The
258 // respective `bool` in `seen_targets_` indicates whether we have
259 // heard from this target or not.
260 std::vector<string> targets_;
261 std::vector<bool> seen_targets_ GUARDED_BY(mu_);
262 Status status_;
263
264 void WhenFound(int target_index, const Status& s,
265 std::vector<Device*>* devices) {
266 mutex_lock l(mu_);
267 seen_targets_[target_index] = true;
268 if (!s.ok()) {
269 LOG(ERROR) << "Master init: " << s;
270 status_.Update(s);
271 } else {
272 found_.insert(found_.end(), devices->begin(), devices->end());
273 devices->clear();
274 }
275 --num_pending_;
276 if (num_pending_ == 0) {
277 pending_zero_.notify_all();
278 }
279 }
280
281 // Returns true iff the set of devices allowed by 'x' intersects
282 // with the set of devices allowed by 'y'.
283 bool Intersects(const DeviceNameUtils::ParsedName& x,
284 const DeviceNameUtils::ParsedName& y) {
285 return (!x.has_job || !y.has_job || x.job == y.job) &&
286 (!x.has_replica || !y.has_replica || x.replica == y.replica) &&
287 (!x.has_task || !y.has_task || x.task == y.task) &&
288 (!x.has_type || !y.has_type || x.type == y.type) &&
289 (!x.has_id || !y.has_id || x.id == y.id);
290 }
291
292 // Returns true iff 'name' matches one of the filters_.
293 bool MatchFilters(const string& name) {
294 if (filters_.empty()) return true;
295 DeviceNameUtils::ParsedName x;
296 if (DeviceNameUtils::ParseFullName(name, &x)) {
297 for (const auto& filter : filters_) {
298 if (Intersects(x, filter)) return true;
299 }
300 }
301 return false;
302 }
303
304 TF_DISALLOW_COPY_AND_ASSIGN(DeviceFinder);
305};

307void Master::CreateSession(const CreateSessionRequest* req,
308 CreateSessionResponse* resp, MyClosure done) {
309 SchedClosure([this, req, resp, done]() {
310 Status status;
311 WorkerCacheFactoryOptions worker_cache_factory_options;
312 string grpc_protocol("grpc");
313 worker_cache_factory_options.protocol = &grpc_protocol;
314 auto call_done = gtl::MakeCleanup([&status, &done] { done(status); });
315 status = ValidateExternalGraphDefSyntax(req->graph_def());
316 if (!status.ok()) return;
317
318 // The following 4 variables are set differently, depending on whether this
319 // session uses a client-provided clusterspec or not.
320 WorkerCacheInterface* worker_cache = nullptr;
321 // Note: worker_cache_ptr will be null except if this session is using a
322 // client-supplied ClusterDef (ClusterSpec propagation).
323 std::unique_ptr<WorkerCacheInterface> worker_cache_ptr;
324 std::unique_ptr<DeviceSet> device_set;
325 // TODO(saeta): Convert to std::make_unique when available.
326 std::unique_ptr<std::vector<std::unique_ptr<Device>>> remote_devices(
327 new std::vector<std::unique_ptr<Device>>());
328
329 if (req->config().has_cluster_def()) {
330 worker_cache_factory_options.cluster_def = &req->config().cluster_def();
331
332 // Set the server_def's job_name and task_index fields.
333 string normalized_string;
334 string grpc_protocol(kGrpcProtocol);
335 if (req->target().compare(0, grpc_protocol.length(), grpc_protocol) ==
336 0) {
337 normalized_string =
338 req->target().substr(grpc_protocol.length(), string::npos);
339 } else {
340 normalized_string = req->target();
341 }
342 for (auto&& job : req->config().cluster_def().job()) {
343 for (auto&& task : job.tasks()) {
344 if (task.second == normalized_string) {
345 if (worker_cache_factory_options.job_name != nullptr) {
346 status = errors::InvalidArgument(
347 "Found multiple matching tasks that correspond to "
348 "to the master. Master target: '",
349 req->target(), "'. ClusterDef: ",
350 req->config().cluster_def().ShortDebugString());
351 LOG(ERROR) << status;
352 return;
353 }
354 if (env_->local_devices[0]->parsed_name().job == job.name() &&
355 env_->local_devices[0]->parsed_name().task == task.first) {
356 // TODO(b/37868888): Remove this limitation when resolved
357 status = errors::InvalidArgument(
358 "The ClusterSpec names the job and task index to be the same "
359 "names that were provided when the server booted. This is "
360 "currently not allowed. Job: ",
361 job.name(), ", task index: ", task.first);
362 return;
363 }
364 worker_cache_factory_options.job_name = &job.name();
365 worker_cache_factory_options.task_index = task.first;
366 }
367 }
368 }
369
370 // Create the worker cache from the computed server_def.
371 status = env_->worker_cache_factory(worker_cache_factory_options,
372 &worker_cache);
373 if (!status.ok()) return;
374 worker_cache_ptr = std::unique_ptr<WorkerCacheInterface>(worker_cache);
375 // Ping all the workers and build the list of devices that the
376 // session will use.
377 status =
378 DeviceFinder::GetRemoteDevices(req->config().device_filters(), env_,
379 worker_cache, remote_devices.get());
380 if (!status.ok()) return;
381 device_set.reset(new DeviceSet);
382 for (auto&& d : *remote_devices) {
383 device_set->AddDevice(d.get());
384 DeviceNameUtils::ParsedName name = d->parsed_name();
385 if (name.job == *worker_cache_factory_options.job_name &&
386 name.task == worker_cache_factory_options.task_index &&
387 name.type == "CPU" && name.id == 0) {
388 device_set->set_client_device(d.get());
389 }
390 }
391 } else {
392 worker_cache = env_->worker_cache;
393 // Ping all the workers and build the list of devices that the
394 // session will use.
395 status =
396 DeviceFinder::GetRemoteDevices(req->config().device_filters(), env_,
397 worker_cache, remote_devices.get());
398 if (!status.ok()) return;
399 device_set.reset(new DeviceSet);
400 for (auto&& d : *remote_devices) {
401 device_set->AddDevice(d.get());
402 }
403 int num_local_devices = 0;
404 for (Device* d : env_->local_devices) {
405 device_set->AddDevice(d);
406 if (num_local_devices == 0) {
407 // Uses the first local device as the client device.
408 device_set->set_client_device(d);
409 }
410 num_local_devices++;
411 }
412 }
413
414 CHECK(device_set->client_device()) << "No client device found. Missing "
415 << "CPU:0 device?";
416
417 SessionOptions options;
418 options.config = req->config();
419
420 MasterSession* session = env_->master_session_factory(
421 options, env_, std::move(remote_devices), std::move(worker_cache_ptr),
422 std::move(device_set));
423
424 GraphDef* gdef =
425 const_cast<CreateSessionRequest*>(req)->mutable_graph_def();
426
427 status = session->Create(gdef, worker_cache_factory_options);
428 if (!status.ok()) {
429 session->Close().IgnoreError();
430 session->Unref();
431 return;
432 }
433 resp->set_session_handle(session->handle());
434 // Insert into the session map, which takes ownership of the session.
435 {
436 mutex_lock l(mu_);
437 CHECK(sessions_.insert({session->handle(), session}).second);
438 }
439 });
440}

475void Master::RunStep(CallOptions* opts, const RunStepRequestWrapper* req,
476 MutableRunStepResponseWrapper* resp, MyClosure done) {
477 auto start_time = env_->env->NowMicros();
478 auto session = FindMasterSession(req->session_handle());
479 if (session == nullptr) {
480 done(errors::Aborted("Session ", req->session_handle(), " is not found."));
481 return;
482 }
483
484 SchedClosure([this, start_time, session, opts, req, resp, done]() {
485 Status status = session->Run(opts, *req, resp);
486 session->Unref();
487 uint64 done_time = env_->env->NowMicros();
488 done(status);
489 mutex_lock l(mu_);
490 last_1000_steps_.AddValue((done_time - start_time) / 1e9);
491 ++step_count_;
492 });
493}

495void Master::CloseSession(const CloseSessionRequest* req,
496 CloseSessionResponse* resp, MyClosure done) {
497 MasterSession* session = nullptr;
498 {
499 mu_.lock();
500 auto iter = sessions_.find(req->session_handle());
501 if (iter == sessions_.end()) {
502 mu_.unlock();
503 done(errors::Aborted(
504 "Session ", req->session_handle(),
505 " is not found. Possibly, this master has restarted."));
506 return;
507 }
508 // NOTE(mrry): One reference to the session is transferred from
509 // `sessions_[req->session_handle()]` to `session`.
510 session = iter->second;
511 sessions_.erase(iter);
512 mu_.unlock();
513 }
514
515 // Session Close() blocks on thread shutdown. Therefore, we need to
516 // delete it in non-critical thread.
517 SchedClosure([session, done]() {
518 Status s = session->Close();
519 session->Unref();
520 done(s);
521 });
522}

524void Master::ListDevices(const ListDevicesRequest* req,
525 ListDevicesResponse* resp, MyClosure done) {
526 SchedClosure([this, req, resp, done]() {
527 if (!req->session_handle().empty()) {
528 auto session = FindMasterSession(req->session_handle());
529 if (session == nullptr) {
530 done(errors::InvalidArgument(
531 "Session ", req->session_handle(),
532 " is not found. Possibly, this master has restarted."));
533 return;
534 }
535 core::ScopedUnref ref(session);
536 Status s = session->ListDevices(resp);
537 done(s);
538 return;
539 }
540 std::vector<std::unique_ptr<Device>> remote_devices;
541 Status s = DeviceFinder::GetRemoteDevices({}, env_, env_->worker_cache,
542 &remote_devices);
543 if (s.ok()) {
544 for (Device* dev : env_->local_devices) {
545 *(resp->add_local_device()) = dev->attributes();
546 }
547 for (auto&& dev : remote_devices) {
548 *(resp->add_remote_device()) = dev->attributes();
549 }
550 }
551 done(s);
552 });
553}

555void Master::CleanupWorkers(const ResetRequest& reset) {
556 std::vector<string> worker_names;
557 DeviceFinder::GetRemoteWorkers(reset.device_filters(), env_,
558 env_->worker_cache, &worker_names);
559 if (!worker_names.empty()) {
560 const int num_workers = worker_names.size();
561 std::vector<Notification> n(num_workers);
562 CleanupAllRequest req;
563 (*req.mutable_container()) = reset.container();
564 std::vector<CleanupAllResponse> resp(num_workers);
565 int c = 0;
566 for (int i = 0; i < num_workers; ++i) {
567 const string& worker_name = worker_names[i];
568 auto worker = env_->worker_cache->CreateWorker(worker_name);
569 if (worker) {
570 worker->CleanupAllAsync(
571 &req, &resp[i], [this, &n, worker_name, worker, c](Status s) {
572 TF_CHECK_OK(s);
573 env_->worker_cache->ReleaseWorker(worker_name, worker);
574 n[c].Notify();
575 });
576 } else {
577 n[c].Notify();
578 }
579 ++c;
580 }
581 for (size_t i = 0; i < n.size(); ++i) {
582 n[i].WaitForNotification();
583 }
584 }
585}

587void Master::Reset(const ResetRequest* req, ResetResponse* resp,
588 MyClosure done) {
589 // Vector to hold the session pointers present in the sessions_
590 // (string->Session*) map.
591 std::vector<MasterSession*> sessions_to_close;
592 {
593 mutex_lock l(mu_);
594 // NOTE(mrry): Transfer one reference to each session from the
595 // `sessions_` map to the `sessions_to_close` vector.
596 for (const auto& entry : sessions_) {
597 sessions_to_close.push_back(entry.second);
598 }
599 sessions_.clear();
600 }
601
602 CleanupWorkers(*req);
603
604 SchedClosure([sessions_to_close, done]() {
605 Status s;
606 for (MasterSession* session : sessions_to_close) {
607 s.Update(session->Close());
608 session->Unref();
609 }
610 done(s);
611 });
612}
