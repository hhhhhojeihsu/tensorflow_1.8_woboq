26RecentRequestIds::RecentRequestIds(int num_tracked_request_ids)
27 : circular_buffer_(num_tracked_request_ids) {
28 set_.reserve(num_tracked_request_ids);
29}
30
31Status RecentRequestIds::TrackUnique(int64 request_id,
32 const string& method_name,
33 const protobuf::Message& request) {
34 if (request_id == 0) {
35 // For backwards compatibility, allow all requests with request_id 0.
36 return Status::OK();
37 }
38
39 mutex_lock l(mu_);
40 const bool inserted = set_.insert(request_id).second;
41 if (!inserted) {
42 // Note: RecentRequestIds is not strict LRU because we don't update
43 // request_id's age in the circular_buffer_ if it's tracked again. Strict
44 // LRU is not useful here because returning this error will close the
45 // current Session.
46 return errors::Aborted("The same ", method_name,
47 " request was received twice. ",
48 request.ShortDebugString());
49 }
50
51 // Remove the oldest request_id from the set_. circular_buffer_ is
52 // zero-initialized, and zero is never tracked, so it's safe to do this even
53 // when the buffer is not yet full.
54 set_.erase(circular_buffer_[next_index_]);
55 circular_buffer_[next_index_] = request_id;
56 next_index_ = (next_index_ + 1) % circular_buffer_.size();
57 return Status::OK();
58}
