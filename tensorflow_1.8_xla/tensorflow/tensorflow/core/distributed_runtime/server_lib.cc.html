20#include "tensorflow/core/lib/core/errors.h"

23namespace tensorflow {
24
25namespace {
26mutex* get_server_factory_lock() {
27 static mutex server_factory_lock(LINKER_INITIALIZED);
28 return &server_factory_lock;
29}
30
31typedef std::unordered_map<string, ServerFactory*> ServerFactories;
32ServerFactories* server_factories() {
33 static ServerFactories* factories = new ServerFactories;
34 return factories;
35}
36} // namespace
37
38/* static */
39void ServerFactory::Register(const string& server_type,
40 ServerFactory* factory) {
41 mutex_lock l(*get_server_factory_lock());
42 if (!server_factories()->insert({server_type, factory}).second) {
43 LOG(ERROR) << "Two server factories are being registered under "
44 << server_type;
45 }
46}
47
48/* static */
49Status ServerFactory::GetFactory(const ServerDef& server_def,
50 ServerFactory** out_factory) {
51 mutex_lock l(*get_server_factory_lock());
52 // TODO(mrry): Improve the error reporting here.
53 for (const auto& server_factory : *server_factories()) {
54 if (server_factory.second->AcceptsOptions(server_def)) {
55 *out_factory = server_factory.second;
56 return Status::OK();
57 }
58 }
59 return errors::NotFound(
60 "No server factory registered for the given ServerDef: ",
61 server_def.DebugString());
62}
63
64// Creates a server based on the given `server_def`, and stores it in
65// `*out_server`. Returns OK on success, otherwise returns an error.
66Status NewServer(const ServerDef& server_def,
67 std::unique_ptr<ServerInterface>* out_server) {
68 ServerFactory* factory;
69 TF_RETURN_IF_ERROR(ServerFactory::GetFactory(server_def, &factory));
70 return factory->NewServer(server_def, out_server);
71}
