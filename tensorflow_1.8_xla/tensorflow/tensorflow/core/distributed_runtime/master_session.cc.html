109 // Turn RPC logging on or off, both at the WorkerCache used by this
110 // master process, and at each remote worker in use for the current
111 // partitions.
112 void SetRPCLogging(bool active) {
113 worker_cache_->SetLogging(active);
114 // Logging is a best-effort activity, so we make async calls to turn
115 // it on/off and don't make use of the responses.
116 for (auto& p : partitions_) {
117 LoggingRequest* req = new LoggingRequest;
118 req->set_rpc_logging(active);
119 LoggingResponse* resp = new LoggingResponse;
120 Ref();
121 p.worker->LoggingAsync(req, resp, [this, req, resp](const Status& s) {
122 delete req;
123 delete resp;
124 // ReffedClientGraph owns p.worker so we need to hold a ref to
125 // ensure that the method doesn't attempt to access p.worker after
126 // ReffedClient graph has deleted it.
127 // TODO(suharshs): Simplify this ownership model.
128 Unref();
129 });
130 }
131 }

133 // Retrieve all RPC logs data accumulated for the current step, both
134 // from the local WorkerCache in use by this master process and from
135 // all the remote workers executing the remote partitions.
136 void RetrieveLogs(int64 step_id, StepStats* ss) {
137 // Get the local data first, because it sets *ss without merging.
138 worker_cache_->RetrieveLogs(step_id, ss);
139
140 // Then merge in data from all the remote workers.
141 LoggingRequest req;
142 req.add_fetch_step_id(step_id);
143 int waiting_for = partitions_.size();
144 if (waiting_for > 0) {
145 mutex scoped_mu;
146 BlockingCounter all_done(waiting_for);
147 for (auto& p : partitions_) {
148 LoggingResponse* resp = new LoggingResponse;
149 p.worker->LoggingAsync(
150 &req, resp,
151 [step_id, ss, resp, &scoped_mu, &waiting_for,
152 &all_done](const Status& s) {
153 {
154 mutex_lock l(scoped_mu);
155 if (s.ok()) {
156 for (auto& lss : resp->step()) {
157 if (step_id != lss.step_id()) {
158 LOG(ERROR) << "Wrong step_id in LoggingResponse";
159 continue;
160 }
161 ss->MergeFrom(lss.step_stats());
162 }
163 }
164 delete resp;
165 }
166 // Must not decrement all_done until out of critical section where
167 // *ss is updated.
168 all_done.DecrementCount();
169 });
170 }
171 all_done.Wait();
172 }

301Status MasterSession::ReffedClientGraph::RegisterPartitions(
302 const PartitionOptions& popts) {
303 { // Ensure register once.
304 mu_.lock();
305 if (!init_started_) {
306 init_started_ = true;
307 mu_.unlock();
308 std::unordered_map<string, GraphDef> graph_defs;
309 Status s = DoBuildPartitions(popts, &graph_defs);
310 if (s.ok()) {
311 // NOTE(mrry): The pointers in `graph_defs_for_publishing` do not remain
312 // valid after the call to DoRegisterPartitions begins, so
313 // `stats_publisher_` must make a copy if it wants to retain the
314 // GraphDef objects.
315 std::vector<const GraphDef*> graph_defs_for_publishing;
316 graph_defs_for_publishing.reserve(partitions_.size());
317 for (const auto& name_def : graph_defs) {
318 graph_defs_for_publishing.push_back(&name_def.second);
319 }
320 stats_publisher_->PublishGraphProto(graph_defs_for_publishing);
321 s = DoRegisterPartitions(popts, std::move(graph_defs));
322 }
323 mu_.lock();
324 init_result_ = s;
325 init_done_.Notify();
326 } else {
327 mu_.unlock();
328 init_done_.WaitForNotification();
329 mu_.lock();
330 }
331 const Status result = init_result_;
332 mu_.unlock();
333 return result;
334 }

346void MasterSession::ReffedClientGraph::TrackFeedsAndFetches(
347 Part* part, const GraphDef& graph_def, const PartitionOptions& popts) {
348 for (int i = 0; i < graph_def.node_size(); ++i) {
349 const NodeDef& ndef = graph_def.node(i);
350 const bool is_recv = ndef.op() == "_Recv";
351 const bool is_send = ndef.op() == "_Send";
352
353 if (is_recv || is_send) {
354 // Only send/recv nodes that were added as feeds and fetches
355 // (client-terminated) should be tracked. Other send/recv nodes
356 // are for transferring data between partitions / memory spaces.
357 bool client_terminated;
358 TF_CHECK_OK(GetNodeAttr(ndef, "client_terminated", &client_terminated));
359 if (client_terminated) {
360 string name;
361 TF_CHECK_OK(GetNodeAttr(ndef, "tensor_name", &name));
362 string send_device;
363 TF_CHECK_OK(GetNodeAttr(ndef, "send_device", &send_device));
364 string recv_device;
365 TF_CHECK_OK(GetNodeAttr(ndef, "recv_device", &recv_device));
366 uint64 send_device_incarnation;
367 TF_CHECK_OK(
368 GetNodeAttr(ndef, "send_device_incarnation",
369 reinterpret_cast<int64*>(&send_device_incarnation)));
370 const string& key =
371 Rendezvous::CreateKey(send_device, send_device_incarnation,
372 recv_device, name, FrameAndIter(0, 0));
373
374 if (is_recv) {
375 part->feed_key.insert({name, key});
376 } else {
377 part->key_fetch.insert({key, name});
378 }
379 }
380 }
381 }

400Status MasterSession::ReffedClientGraph::DoRegisterPartitions(
401 const PartitionOptions& popts,
402 std::unordered_map<string, GraphDef> graph_partitions) {
403 partitions_.reserve(graph_partitions.size());
404 Status s;
405 for (auto& name_def : graph_partitions) {
406 partitions_.resize(partitions_.size() + 1);
407 Part* part = &partitions_.back();
408 part->name = name_def.first;
409 TrackFeedsAndFetches(part, name_def.second, popts);
410 part->worker = worker_cache_->CreateWorker(part->name);
411 if (part->worker == nullptr) {
412 s = errors::NotFound("worker ", part->name);
413 break;
414 }
415 }
416 if (!s.ok()) {
417 for (Part& part : partitions_) {
418 worker_cache_->ReleaseWorker(part.name, part.worker);
419 }
420 return s;
421 }
422 struct Call {
423 RegisterGraphRequest req;
424 RegisterGraphResponse resp;
425 Status status;
426 };
427 const int num = partitions_.size();
428 gtl::InlinedVector<Call, 4> calls(num);
429 BlockingCounter done(num);
430 for (int i = 0; i < num; ++i) {
431 const Part& part = partitions_[i];
432 Call* c = &calls[i];
433 c->req.set_session_handle(session_handle_);
434 c->req.mutable_graph_def()->Swap(&graph_partitions[part.name]);
435 *c->req.mutable_graph_options() = session_opts_.config.graph_options();
436 *c->req.mutable_debug_options() =
437 callable_opts_.run_options().debug_options();
438 VLOG(2) << "Register " << c->req.graph_def().DebugString();
439 auto cb = [c, &done](const Status& s) {
440 c->status = s;
441 done.DecrementCount();
442 };
443 part.worker->RegisterGraphAsync(&c->req, &c->resp, cb);
444 }
445 done.Wait();
446 for (int i = 0; i < num; ++i) {
447 Call* c = &calls[i];
448 s.Update(c->status);
449 partitions_[i].graph_handle = c->resp.graph_handle();
450 }
451 return s;
452}

454// Helper class to manage "num" parallel RunGraph calls.
455class RunManyGraphs {
456 public:
457 explicit RunManyGraphs(int num) : calls_(num), pending_(num) {}
458
459 ~RunManyGraphs() {}
460
461 // Returns the index-th call.
462 struct Call {
463 CallOptions opts;
464 std::unique_ptr<MutableRunGraphRequestWrapper> req;
465 std::unique_ptr<MutableRunGraphResponseWrapper> resp;
466 };
467 Call* get(int index) { return &calls_[index]; }
468
469 // When the index-th call is done, updates the overall status.
470 void WhenDone(int index, const Status& s) {
471 TRACEPRINTF("Partition %d %s", index, s.ToString().c_str());
472 auto resp = get(index)->resp.get();
473 if (resp->status_code() != error::Code::OK) {
474 // resp->status_code will only be non-OK if s.ok().
475 mutex_lock l(mu_);
476 UpdateStatusLocked(
477 Status(resp->status_code(), resp->status_error_message()));
478 } else if (!s.ok()) {
479 mutex_lock l(mu_);
480 UpdateStatusLocked(s);
481 }
482 pending_.DecrementCount();
483 }
484
485 void StartCancel() {
486 mutex_lock l(mu_);
487 UpdateStatusLocked(errors::Cancelled("RunManyGraphs"));
488 }
489
490 void Wait() { pending_.Wait(); }
491
492 Status status() const {
493 mutex_lock l(mu_);
494 return status_;
495 }
496
497 private:
498 gtl::InlinedVector<Call, 4> calls_;
499
500 BlockingCounter pending_;
501 mutable mutex mu_;
502 Status status_ GUARDED_BY(mu_);
503
504 void UpdateStatusLocked(const Status& s) EXCLUSIVE_LOCKS_REQUIRED(mu_) {
505 if (status_.ok()) {
506 status_ = s;
507 for (Call& call : calls_) {
508 call.opts.StartCancel();
509 }
510 }
511 }
512
513 TF_DISALLOW_COPY_AND_ASSIGN(RunManyGraphs);
514};

548template <class FetchListType, class ClientRequestType,
549 class ClientResponseType>
550Status MasterSession::ReffedClientGraph::RunPartitionsHelper(
551 const std::unordered_map<StringPiece, size_t, StringPieceHasher>& feeds,
552 const FetchListType& fetches, const MasterEnv* env, int64 step_id,
553 int64 execution_count, PerStepState* pss, CallOptions* call_opts,
554 const ClientRequestType& req, ClientResponseType* resp,
555 CancellationManager* cm, bool is_last_partial_run) {
556 // Collect execution cost stats on a smoothly decreasing frequency.
557 ExecutorOpts exec_opts;
558 if (pss->report_tensor_allocations_upon_oom) {
559 exec_opts.set_report_tensor_allocations_upon_oom(true);
560 }
561 if (pss->collect_costs) {
562 exec_opts.set_record_costs(true);
563 }
564 if (pss->collect_timeline) {
565 exec_opts.set_record_timeline(true);
566 }
567 if (pss->collect_rpcs) {
568 SetRPCLogging(true);
569 }
570 if (pss->collect_partition_graphs) {
571 exec_opts.set_record_partition_graphs(true);
572 }
573 if (pss->collect_costs || pss->collect_timeline) {
574 pss->step_stats.resize(partitions_.size());
575 }
576
577 const int num = partitions_.size();
578 RunManyGraphs calls(num);
579
580 for (int i = 0; i < num; ++i) {
581 const Part& part = partitions_[i];
582 RunManyGraphs::Call* c = calls.get(i);
583 c->req.reset(part.worker->CreateRunGraphRequest());
584 c->resp.reset(part.worker->CreateRunGraphResponse());
585 if (is_partial_) {
586 c->req->set_is_partial(is_partial_);
587 c->req->set_is_last_partial_run(is_last_partial_run);
588 }
589 c->req->set_session_handle(session_handle_);
590 c->req->set_graph_handle(part.graph_handle);
591 c->req->set_step_id(step_id);
592 *c->req->mutable_exec_opts() = exec_opts;
593 c->req->set_store_errors_in_response_body(true);
594 // If any feeds are provided, send the feed values together
595 // in the RunGraph request.
596 // In the partial case, we only want to include feeds provided in the req.
597 // In the non-partial case, all feeds in the request are in the part.
598 // We keep these as separate paths for now, to ensure we aren't
599 // inadvertently slowing down the normal run path.
600 if (is_partial_) {
601 for (const auto& name_index : feeds) {
602 const auto iter = part.feed_key.find(name_index.first.ToString());
603 if (iter == part.feed_key.end()) {
604 // The provided feed must be for a different partition.
605 continue;
606 }
607 const string& key = iter->second;
608 TF_RETURN_IF_ERROR(AddSendFromClientRequest(req, c->req.get(),
609 name_index.second, key));
610 }
611 // TODO(suharshs): Make a map from feed to fetch_key to make this faster.
612 // For now, we just iterate through partitions to find the matching key.
613 for (const string& req_fetch : fetches) {
614 for (const auto& key_fetch : part.key_fetch) {
615 if (key_fetch.second == req_fetch) {
616 c->req->add_recv_key(key_fetch.first);
617 break;
618 }
619 }
620 }
621 } else {
622 for (const auto& feed_key : part.feed_key) {
623 const string& feed = feed_key.first;
624 const string& key = feed_key.second;
625 auto iter = feeds.find(feed);
626 if (iter == feeds.end()) {
627 return errors::Internal("No feed index found for feed: ", feed);
628 }
629 const int64 feed_index = iter->second;
630 TF_RETURN_IF_ERROR(
631 AddSendFromClientRequest(req, c->req.get(), feed_index, key));
632 }
633 for (const auto& key_fetch : part.key_fetch) {
634 const string& key = key_fetch.first;
635 c->req->add_recv_key(key);
636 }
637 }
638 }
639
640 // Issues RunGraph calls.
641 for (int i = 0; i < num; ++i) {
642 const Part& part = partitions_[i];
643 RunManyGraphs::Call* call = calls.get(i);
644 TRACEPRINTF("Partition %d %s", i, part.name.c_str());
645 part.worker->RunGraphAsync(
646 &call->opts, call->req.get(), call->resp.get(),
647 std::bind(&RunManyGraphs::WhenDone, &calls, i, std::placeholders::_1));
648 }
649
650 // Waits for the RunGraph calls.
651 call_opts->SetCancelCallback([&calls]() { calls.StartCancel(); });
652 auto token = cm->get_cancellation_token();
653 const bool success =
654 cm->RegisterCallback(token, [&calls]() { calls.StartCancel(); });
655 if (!success) {
656 calls.StartCancel();
657 }
658 calls.Wait();
659 call_opts->ClearCancelCallback();
660 if (success) {
661 cm->DeregisterCallback(token);
662 } else {
663 return errors::Cancelled("Step was cancelled");
664 }
665 TF_RETURN_IF_ERROR(calls.status());
666
667 // Collects fetches and metadata.
668 Status status;
669 for (int i = 0; i < num; ++i) {
670 const Part& part = partitions_[i];
671 MutableRunGraphResponseWrapper* run_graph_resp = calls.get(i)->resp.get();
672 for (size_t j = 0; j < run_graph_resp->num_recvs(); ++j) {
673 auto iter = part.key_fetch.find(run_graph_resp->recv_key(j));
674 if (iter == part.key_fetch.end()) {
675 status.Update(errors::Internal("Unexpected fetch key: ",
676 run_graph_resp->recv_key(j)));
677 break;
678 }
679 const string& fetch = iter->second;
680 status.Update(
681 resp->AddTensorFromRunGraphResponse(fetch, run_graph_resp, j));
682 if (!status.ok()) {
683 break;
684 }
685 }
686 if (pss->collect_timeline) {
687 pss->step_stats[i].Swap(run_graph_resp->mutable_step_stats());
688 }
689 if (pss->collect_costs) {
690 CostGraphDef* cost_graph = run_graph_resp->mutable_cost_graph();
691 for (int j = 0; j < cost_graph->node_size(); ++j) {
692 resp->mutable_metadata()->mutable_cost_graph()->add_node()->Swap(
693 cost_graph->mutable_node(j));
694 }
695 }
696 if (pss->collect_partition_graphs) {
697 protobuf::RepeatedPtrField<GraphDef>* partition_graph_defs =
698 resp->mutable_metadata()->mutable_partition_graphs();
699 for (size_t i = 0; i < run_graph_resp->num_partition_graphs(); i++) {
700 partition_graph_defs->Add()->Swap(
701 run_graph_resp->mutable_partition_graph(i));
702 }
703 }
704 }
705 return status;
706}

736 RunCallableResponse* resp, CancellationManager* cm) {

745 callable_opts_.feed(i));
746 }
747 }

750 // rearrange them for the RunCallableResponse.

755 feeds, callable_opts_.fetch(), env, step_id, execution_count, pss,
756 call_opts, req, &wrapped_resp, cm, false /* is_last_partial_run */));

758 // Collects fetches.
759 // TODO(b/74355905): Add a specialized implementation that avoids
760 // copying the tensor into the RunCallableResponse when at least
761 // two of the {client, master, worker} are in the same process.
762 for (const string& fetch : callable_opts_.fetch()) {

765 if (iter == wrapped_resp.fetch_key_to_protos.end()) {
766 return errors::Internal("Worker did not return a value for fetch: ",
767 fetch);
768 }
769 fetch_proto->Swap(&iter->second);
770 }
771 return Status::OK();
772}

776class CleanupBroadcastHelper {
777 public:
778 CleanupBroadcastHelper(int64 step_id, int num_calls, StatusCallback done)
779 : resps_(num_calls), num_pending_(num_calls), done_(std::move(done)) {
780 req_.set_step_id(step_id);
781 }
782
783 // Returns a non-owned pointer to a request buffer for all calls.
784 CleanupGraphRequest* request() { return &req_; }
785
786 // Returns a non-owned pointer to a response buffer for the ith call.
787 CleanupGraphResponse* response(int i) { return &resps_[i]; }
788
789 // Called when the ith response is received.
790 void call_done(int i, const Status& s) {
791 bool run_callback = false;
792 Status status_copy;
793 {
794 mutex_lock l(mu_);
795 status_.Update(s);
796 if (--num_pending_ == 0) {
797 run_callback = true;
798 status_copy = status_;
799 }
800 }
801 if (run_callback) {
802 done_(status_copy);
803 // This is the last call, so delete the helper object.
804 delete this;
805 }
806 }
807
808 private:
809 // A single request shared between all workers.
810 CleanupGraphRequest req_;
811 // One response buffer for each worker.
812 gtl::InlinedVector<CleanupGraphResponse, 4> resps_;
813
814 mutex mu_;
815 // Number of requests remaining to be collected.
816 int num_pending_ GUARDED_BY(mu_);
817 // Aggregate status of the operation.
818 Status status_ GUARDED_BY(mu_);
819 // Callback to be called when all operations complete.
820 StatusCallback done_;
821
822 TF_DISALLOW_COPY_AND_ASSIGN(CleanupBroadcastHelper);
823};

933// TODO(suharshs): Merge with CheckFetches in DirectSession.
934// TODO(suharsh,mrry): Build a map from fetch target to set of feeds it depends
935// on once at setup time to prevent us from computing the dependencies
936// everytime.
937Status MasterSession::ReffedClientGraph::CheckFetches(
938 const RunStepRequestWrapper& req, const RunState* run_state,
939 GraphExecutionState* execution_state) {
940 // Build the set of pending feeds that we haven't seen.
941 std::unordered_set<TensorId, TensorId::Hasher> pending_feeds;
942 for (const auto& input : run_state->pending_inputs) {
943 // Skip if already fed.
944 if (input.second) continue;
945 TensorId id(ParseTensorName(input.first));
946 const Node* n = execution_state->get_node_by_name(id.first.ToString());
947 if (n == nullptr) {
948 return errors::NotFound("Feed ", input.first, ": not found");
949 }
950 pending_feeds.insert(id);
951 }
952 for (size_t i = 0; i < req.num_feeds(); ++i) {
953 const TensorId id(ParseTensorName(req.feed_name(i)));
954 pending_feeds.erase(id);
955 }
956
957 // Initialize the stack with the fetch nodes.
958 std::vector<const Node*> stack;
959 for (size_t i = 0; i < req.num_fetches(); ++i) {
960 const string& fetch = req.fetch_name(i);
961 const TensorId id(ParseTensorName(fetch));
962 const Node* n = execution_state->get_node_by_name(id.first.ToString());
963 if (n == nullptr) {
964 return errors::NotFound("Fetch ", fetch, ": not found");
965 }
966 stack.push_back(n);
967 }
968
969 // Any tensor needed for fetches can't be in pending_feeds.
970 // We need to use the original full graph from execution state.
971 const Graph* graph = execution_state->full_graph();
972 std::vector<bool> visited(graph->num_node_ids(), false);
973 while (!stack.empty()) {
974 const Node* n = stack.back();
975 stack.pop_back();
976
977 for (const Edge* in_edge : n->in_edges()) {
978 const Node* in_node = in_edge->src();
979 if (pending_feeds.count({in_node->name(), in_edge->src_output()}) > 0) {
980 return errors::InvalidArgument("Fetch ", in_node->name(), ":",
981 in_edge->src_output(),
982 " can't be computed from the feeds"
983 " that have been fed so far.");
984 }
985 if (!visited[in_node->id()]) {
986 visited[in_node->id()] = true;
987 stack.push_back(in_node);
988 }
989 }
990 }
991 return Status::OK();
992}

1181Status MasterSession::CreateWorkerSessions(
1182 const WorkerCacheFactoryOptions& options) {
1183 std::vector<string> worker_names;
1184 WorkerCacheInterface* worker_cache = get_worker_cache();
1185 worker_cache->ListWorkers(&worker_names);
1186
1187 struct WorkerGroup {
1188 // The worker name. (Not owned.)
1189 const string* name;
1190
1191 // The worker referenced by name. (Not owned.)
1192 WorkerInterface* worker = nullptr;
1193
1194 // Request and responses used for a given worker.
1195 CreateWorkerSessionRequest request;
1196 CreateWorkerSessionResponse response;
1197 Status status = Status::OK();
1198 };
1199 BlockingCounter done(worker_names.size());
1200 std::vector<WorkerGroup> workers(worker_names.size());
1201
1202 // Release the workers.
1203 auto cleanup = gtl::MakeCleanup([this, &workers, worker_cache] {
1204 for (auto&& worker_group : workers) {
1205 if (worker_group.worker != nullptr) {
1206 worker_cache->ReleaseWorker(*worker_group.name, worker_group.worker);
1207 }
1208 }
1209 });
1210
1211 Status status = Status::OK();
1212 // Create all the workers & kick off the computations.
1213 for (size_t i = 0; i < worker_names.size(); ++i) {
1214 workers[i].name = &worker_names[i];
1215 workers[i].worker = worker_cache->CreateWorker(worker_names[i]);
1216 workers[i].request.set_session_handle(handle_);
1217 if (options.cluster_def) {
1218 *workers[i].request.mutable_server_def()->mutable_cluster() =
1219 *options.cluster_def;
1220 workers[i].request.mutable_server_def()->set_protocol(*options.protocol);
1221 // Session state is always isolated when ClusterSpec propagation
1222 // is in use.
1223 workers[i].request.set_isolate_session_state(true);
1224 } else {
1225 workers[i].request.set_isolate_session_state(
1226 session_opts_.config.isolate_session_state());
1227 }
1228
1229 DeviceNameUtils::ParsedName name;
1230 if (!DeviceNameUtils::ParseFullName(worker_names[i], &name)) {
1231 status = errors::Internal("Could not parse name ", worker_names[i]);
1232 LOG(WARNING) << status;
1233 return status;
1234 }
1235 if (!name.has_job || !name.has_task) {
1236 status = errors::Internal("Incomplete worker name ", worker_names[i]);
1237 LOG(WARNING) << status;
1238 return status;
1239 }
1240
1241 workers[i].request.mutable_server_def()->set_job_name(name.job);
1242 workers[i].request.mutable_server_def()->set_task_index(name.task);
1243 }
1244
1245 for (size_t i = 0; i < worker_names.size(); ++i) {
1246 auto cb = [i, &workers, &done](const Status& s) {
1247 workers[i].status = s;
1248 done.DecrementCount();
1249 };
1250 workers[i].worker->CreateWorkerSessionAsync(&workers[i].request,
1251 &workers[i].response, cb);
1252 }
1253
1254 done.Wait();
1255 for (size_t i = 0; i < workers.size(); ++i) {
1256 status.Update(workers[i].status);
1257 }
1258 return status;
1259}

1261Status MasterSession::DeleteWorkerSessions() {
1262 WorkerCacheInterface* worker_cache = get_worker_cache();
1263 std::vector<string> worker_names;
1264 worker_cache->ListWorkers(&worker_names);
1265
1266 struct WorkerGroup {
1267 // The worker name. (Not owned.)
1268 const string* name;
1269
1270 // The worker referenced by name. (Not owned.)
1271 WorkerInterface* worker = nullptr;
1272
1273 // Request and responses used for a given worker.
1274 DeleteWorkerSessionRequest request;
1275 DeleteWorkerSessionResponse response;
1276 Status status = Status::OK();
1277 };
1278 BlockingCounter done(worker_names.size());
1279 std::vector<WorkerGroup> workers(worker_names.size());
1280
1281 // Release the workers.
1282 auto cleanup = gtl::MakeCleanup([this, &workers, worker_cache] {
1283 for (auto&& worker_group : workers) {
1284 if (worker_group.worker != nullptr) {
1285 worker_cache->ReleaseWorker(*worker_group.name, worker_group.worker);
1286 }
1287 }
1288 });
1289
1290 Status status = Status::OK();
1291 // Create all the workers & kick off the computations.
1292 for (size_t i = 0; i < worker_names.size(); ++i) {
1293 workers[i].name = &worker_names[i];
1294 workers[i].worker = worker_cache->CreateWorker(worker_names[i]);
1295 workers[i].request.set_session_handle(handle_);
1296 }
1297
1298 for (size_t i = 0; i < worker_names.size(); ++i) {
1299 auto cb = [i, &workers, &done](const Status& s) {
1300 workers[i].status = s;
1301 done.DecrementCount();
1302 };
1303 workers[i].worker->DeleteWorkerSessionAsync(&workers[i].request,
1304 &workers[i].response, cb);
1305 }
1306
1307 done.Wait();
1308 for (size_t i = 0; i < workers.size(); ++i) {
1309 status.Update(workers[i].status);
1310 }
1311 return status;
1312}

1498Status MasterSession::BuildAndRegisterPartitions(ReffedClientGraph* rcg) {
1499 // Registers subgraphs if haven't done so.
1500 PartitionOptions popts;
1501 popts.node_to_loc = SplitByWorker;

1503 // RegisterPartitions() below, so do not need a Ref()/Unref() pair to keep
1504 // "this" alive during the closure.
1505 popts.new_name = [this](const string& prefix) {
1506 mutex_lock l(mu_);

1508 };
1509 popts.flib_def = rcg->client_graph()->flib_def.get();

1511 Device* d = devices_->FindDeviceByName(name);
1512 if (d == nullptr) {
1513 return PartitionOptions::kIllegalIncarnation;
1514 } else {
1515 return d->attributes().incarnation();
1516 }
1517 };
1518 popts.control_flow_added = false;
1519 const bool enable_bfloat16_sendrecv =
1520 session_opts_.config.graph_options().enable_bfloat16_sendrecv();
1521 popts.should_cast = [enable_bfloat16_sendrecv](const Edge* e) {
1522 if (e->IsControlEdge()) {
1523 return DT_FLOAT;
1524 }
1525 DataType dtype = BaseType(e->src()->output_type(e->src_output()));
1526 if (enable_bfloat16_sendrecv && dtype == DT_FLOAT) {
1527 return DT_BFLOAT16;
1528 } else {
1529 return dtype;
1530 }
1531 };
1532 if (session_opts_.config.graph_options().enable_recv_scheduling()) {
1533 popts.scheduling_for_recvs = true;
1534 popts.need_to_record_start_times = true;
1535 }
1536
1537 TF_RETURN_IF_ERROR(rcg->RegisterPartitions(popts));
1538
1539 return Status::OK();

1542Status MasterSession::DoPartialRun(CallOptions* opts,
1543 const RunStepRequestWrapper& req,
1544 MutableRunStepResponseWrapper* resp) {
1545 auto cleanup = gtl::MakeCleanup([this] { MarkRunCompletion(); });
1546 const string& prun_handle = req.partial_run_handle();
1547 RunState* run_state = nullptr;
1548 {
1549 mutex_lock l(mu_);
1550 auto it = partial_runs_.find(prun_handle);
1551 if (it == partial_runs_.end()) {
1552 return errors::InvalidArgument(
1553 "Must run PartialRunSetup before performing partial runs");
1554 }
1555 run_state = it->second.get();
1556 }
1557
1558 // If this is the first partial run, initialize the PerStepState.
1559 if (!run_state->step_started) {
1560 run_state->step_started = true;
1561 PerStepState pss;
1562
1563 const auto count = run_state->count;
1564 pss.collect_timeline =
1565 req.options().trace_level() == RunOptions::FULL_TRACE;
1566 pss.collect_rpcs = req.options().trace_level() == RunOptions::FULL_TRACE;
1567 pss.report_tensor_allocations_upon_oom =
1568 req.options().report_tensor_allocations_upon_oom();
1569
1570 // Build the cost model every 'build_cost_model_every' steps after skipping
1571 // an
1572 // initial 'build_cost_model_after' steps.
1573 const int64 build_cost_model_after =
1574 session_opts_.config.graph_options().build_cost_model_after();
1575 const int64 build_cost_model_every =
1576 session_opts_.config.graph_options().build_cost_model();
1577 pss.collect_costs =
1578 build_cost_model_every > 0 &&
1579 ((count + 1 - build_cost_model_after) % build_cost_model_every == 0);
1580 pss.collect_partition_graphs = req.options().output_partition_graphs();
1581
1582 std::unique_ptr<ProfileHandler> ph = run_state->rcg->GetProfileHandler(
1583 run_state->step_id, count, req.options());
1584 if (ph) {
1585 pss.collect_timeline = true;
1586 pss.collect_rpcs = ph->should_collect_rpcs();
1587 }
1588
1589 run_state->pss = std::move(pss);
1590 run_state->ph = std::move(ph);
1591 }
1592
1593 // Make sure that this is a new set of feeds that are still pending.
1594 for (size_t i = 0; i < req.num_feeds(); ++i) {
1595 const string& feed = req.feed_name(i);
1596 auto it = run_state->pending_inputs.find(feed);
1597 if (it == run_state->pending_inputs.end()) {
1598 return errors::InvalidArgument(
1599 "The feed ", feed, " was not specified in partial_run_setup.");
1600 } else if (it->second) {
1601 return errors::InvalidArgument("The feed ", feed,
1602 " has already been fed.");
1603 }
1604 }
1605 // Check that this is a new set of fetches that are still pending.
1606 for (size_t i = 0; i < req.num_fetches(); ++i) {
1607 const string& fetch = req.fetch_name(i);
1608 auto it = run_state->pending_outputs.find(fetch);
1609 if (it == run_state->pending_outputs.end()) {
1610 return errors::InvalidArgument(
1611 "The fetch ", fetch, " was not specified in partial_run_setup.");
1612 } else if (it->second) {
1613 return errors::InvalidArgument("The fetch ", fetch,
1614 " has already been fetched.");
1615 }
1616 }
1617
1618 // Ensure that the requested fetches can be computed from the provided feeds.
1619 {
1620 mutex_lock l(mu_);
1621 TF_RETURN_IF_ERROR(
1622 run_state->rcg->CheckFetches(req, run_state, execution_state_.get()));
1623 }
1624
1625 // Determine if this partial run satisfies all the pending inputs and outputs.
1626 for (size_t i = 0; i < req.num_feeds(); ++i) {
1627 auto it = run_state->pending_inputs.find(req.feed_name(i));
1628 it->second = true;
1629 }
1630 for (size_t i = 0; i < req.num_fetches(); ++i) {
1631 auto it = run_state->pending_outputs.find(req.fetch_name(i));
1632 it->second = true;
1633 }
1634 bool is_last_partial_run = run_state->PendingDone();
1635
1636 Status s = run_state->rcg->RunPartitions(
1637 env_, run_state->step_id, run_state->count, &run_state->pss, opts, req,
1638 resp, &cancellation_manager_, is_last_partial_run);
1639
1640 // Delete the run state if there is an error or all fetches are done.
1641 if (!s.ok() || is_last_partial_run) {
1642 ReffedClientGraph* rcg = run_state->rcg;
1643 run_state->pss.end_micros = Env::Default()->NowMicros();
1644 // Schedule post-processing and cleanup to be done asynchronously.
1645 Ref();
1646 rcg->Ref();
1647 rcg->ProcessStats(run_state->step_id, &run_state->pss, run_state->ph.get(),
1648 req.options(), resp->mutable_metadata());
1649 cleanup.release(); // MarkRunCompletion called in done closure.
1650 rcg->CleanupPartitionsAsync(
1651 run_state->step_id, [this, rcg, prun_handle](const Status& s) {
1652 if (!s.ok()) {
1653 LOG(ERROR) << "Cleanup partition error: " << s;
1654 }
1655 rcg->Unref();
1656 MarkRunCompletion();
1657 Unref();
1658 });
1659 mutex_lock l(mu_);
1660 partial_runs_.erase(prun_handle);
1661 }
1662 return s;
1663}

1885 }
1886 int64 handle = req.handle();
1887 if (handle >= next_callable_handle_) {
1888 return errors::InvalidArgument("No such callable handle: ", handle);
1889 }
1890 auto iter = callables_.find(req.handle());
1891 if (iter == callables_.end()) {
1892 return errors::InvalidArgument(
1893 "Attempted to run callable after handle was released: ", handle);
1894 }
1895 callable = iter->second;
1896 callable->Ref();
1897 ++num_running_;
1898 }
1899 core::ScopedUnref unref_callable(callable);
1900 return DoRunCallable(opts, callable, req, resp);
1901}

1903Status MasterSession::ReleaseCallable(const ReleaseCallableRequest& req,
1904 ReleaseCallableResponse* resp) {
1905 UpdateLastAccessTime();
1906 ReffedClientGraph* to_unref = nullptr;
1907 {
1908 mutex_lock l(mu_);
1909 auto iter = callables_.find(req.handle());
1910 if (iter != callables_.end()) {
1911 to_unref = iter->second;
1912 callables_.erase(iter);
1913 }
1914 }
1915 if (to_unref != nullptr) {
1916 to_unref->Unref();
1917 }
1918 return Status::OK();
1919}

1921Status MasterSession::Close() {
1922 {
1923 mutex_lock l(mu_);
1924 closed_ = true; // All subsequent calls to Run() or Extend() will fail.
1925 }
1926 cancellation_manager_.StartCancel();
1927 std::vector<ReffedClientGraph*> to_unref;
1928 {
1929 mutex_lock l(mu_);
1930 while (num_running_ != 0) {
1931 num_running_is_zero_.wait(l);
1932 }
1933 ClearRunsTable(&to_unref, &run_graphs_);
1934 ClearRunsTable(&to_unref, &partial_run_graphs_);
1935 ClearRunsTable(&to_unref, &callables_);
1936 }
1937 for (ReffedClientGraph* rcg : to_unref) rcg->Unref();
1938 if (should_delete_worker_sessions_) {
1939 Status s = DeleteWorkerSessions();
1940 if (!s.ok()) {
1941 LOG(WARNING) << s;
1942 }
1943 }
1944 return Status::OK();
1945}
