75GrpcServer::~GrpcServer() {
76 TF_CHECK_OK(Stop());
77 TF_CHECK_OK(Join());
78
79 delete master_service_;
80 delete worker_service_;
81
82 // TODO(mrry): Refactor the *Env classes so that it is less fiddly
83 // to destroy them.
84
85 // Shut down all outstanding rendezvous.
86 delete worker_env_.rendezvous_mgr;
87
88 // We must delete graph_mgr before device_mgr, due to shared
89 // ownership of OpKernels in the executors. (The graph_mgr will
90 // free all stateless OpKernels, and pass over borrowed stateful
91 // OpKernels, which are also held in their respective devices'
92 // OpSegments.)
93 if (worker_env_.session_mgr != nullptr) {
94 delete worker_env_.session_mgr; // Deletes graph_mgr's.
95 } else {
96 // Note: session_mgr's legacy_session_ deletes device_mgr now.
97 delete worker_env_.device_mgr;
98 }
99
100 // Do not delete (as these are not owned by the server):
101 // - master_env_.env
102 // - worker_env_.env
103 // - worker_env_.compute_pool
104}

106Status GrpcServer::Init(
107 ServiceInitFunction service_func,
108 const RendezvousMgrCreationFunction& rendezvous_mgr_func,
109 const WorkerCreationFunction& worker_func,
110 const StatsPublisherFactory& stats_factory) {
111 mutex_lock l(mu_);
112 CHECK_EQ(state_, NEW);
113 master_env_.env = env_;
114 worker_env_.env = env_;
115
116 SessionOptions sess_opts;
117 ConfigProto config = server_def_.default_session_config();
118 sess_opts.config = config;
119
120 // Configure shared devices between master and worker.
121 string name_prefix =
122 strings::StrCat("/job:", server_def_.job_name(), "/replica:0",
123 "/task:", server_def_.task_index());
124 TF_RETURN_IF_ERROR(DeviceFactory::AddDevices(sess_opts, name_prefix,
125 &master_env_.local_devices));
126 worker_env_.local_devices = master_env_.local_devices;
127 worker_env_.device_mgr = new DeviceMgr(worker_env_.local_devices);
128 worker_env_.rendezvous_mgr = rendezvous_mgr_func == nullptr
129 ? new RpcRendezvousMgr(&worker_env_)
130 : rendezvous_mgr_func(&worker_env_);
131 string unused;
132 string default_worker_name;
133 if (!DeviceNameUtils::SplitDeviceName(master_env_.local_devices[0]->name(),
134 &default_worker_name, &unused)) {
135 return errors::Internal("Could not parse worker name.");
136 }
137
138 // Look up the port that has been requested for this task in `server_def_`.
139 int requested_port = -1;
140 for (const auto& job : server_def_.cluster().job()) {
141 if (job.name() == server_def_.job_name()) {
142 auto iter = job.tasks().find(server_def_.task_index());
143 if (iter == job.tasks().end()) {
144 return errors::InvalidArgument("Task ", server_def_.task_index(),
145 " was not defined in job \"",
146 server_def_.job_name(), "\"");
147 }
148 const std::vector<string> hostname_port =
149 str_util::Split(iter->second, ':');
150 if (hostname_port.size() != 2 ||
151 !strings::safe_strto32(hostname_port[1], &requested_port)) {
152 return errors::InvalidArgument(
153 "Could not parse port for local server from \"", iter->second,
154 "\"");
155 } else {
156 break;
157 }
158 }
159 }
160 if (requested_port == -1) {
161 return errors::Internal("Job \"", server_def_.job_name(),
162 "\" was not defined in cluster");
163 }
164
165 // N.B. The order of initialization here is intricate, because we
166 // wish to allow `requested_port == 0` (for choosing any port,
167 // mostly for testing). Therefore, the construction of the channel
168 // and worker caches depends on `bound_port_`, which is not set
169 // until we call `builder.BuildAndStart()`. We must create the
170 // service objects before calling `builder.BuildAndStart()`, but
171 // `master_env_` and `worker_env_` are only partially
172 // configured. However, this is not dangerous, because we do not
173 // start serving requests until `this->Start()` is called, which
174 // happens after this method returns.
175 //
176 // TODO(mrry): Provide a general mechanism for dynamically setting
177 // the identities of tasks in the worker pool after the service is
178 // running.
179 ::grpc::ServerBuilder builder;
180 builder.AddListeningPort(strings::StrCat("0.0.0.0:", requested_port),
181 GetServerCredentials(server_def_), &bound_port_);
182 builder.SetMaxMessageSize(std::numeric_limits<int32>::max());
183 builder.SetOption(
184 std::unique_ptr<::grpc::ServerBuilderOption>(new NoReusePortOption));
185 master_impl_ = CreateMaster(&master_env_);
186 master_service_ = NewGrpcMasterService(master_impl_.get(), config, &builder);
187 worker_impl_ =
188 worker_func ? worker_func(&worker_env_) : NewGrpcWorker(&worker_env_);
189 worker_service_ =
190 NewGrpcWorkerService(worker_impl_.get(), &builder).release();
191 // extra service:
192 if (service_func != nullptr) {
193 service_func(&worker_env_, &builder);
194 }
195 server_ = builder.BuildAndStart();
196
197 if (!server_) {
198 return errors::Unknown("Could not start gRPC server");
199 }
200
201 WorkerCacheInterface* worker_cache;
202 WorkerCacheFactoryOptions worker_cache_factory_options(server_def_);
203 TF_RETURN_IF_ERROR(
204 WorkerCacheFactory(worker_cache_factory_options, &worker_cache));
205 CHECK_NE(nullptr, worker_cache);
206
207 // Set up worker environment.
208 worker_env_.session_mgr = new SessionMgr(
209 &worker_env_, SessionMgr::WorkerNameFromServerDef(server_def_),
210 std::unique_ptr<WorkerCacheInterface>(worker_cache),
211 [this](const ServerDef& server_def, WorkerCacheInterface** worker_cache) {
212 WorkerCacheFactoryOptions options(server_def);
213 return WorkerCacheFactory(options, worker_cache);
214 });
215 worker_env_.compute_pool = ComputePool(sess_opts);
216
217 // Finish setting up master environment.
218 master_env_.ops = OpRegistry::Global();
219 master_env_.worker_cache = worker_cache;
220 master_env_.master_session_factory =
221 [config, stats_factory](
222 SessionOptions options, const MasterEnv* env,
223 std::unique_ptr<std::vector<std::unique_ptr<Device>>> remote_devs,
224 std::unique_ptr<WorkerCacheInterface> worker_cache,
225 std::unique_ptr<DeviceSet> device_set) {
226 options.config.MergeFrom(config);
227 return new MasterSession(options, env, std::move(remote_devs),
228 std::move(worker_cache), std::move(device_set),
229 stats_factory);
230 };
231 master_env_.worker_cache_factory =
232 [this](const WorkerCacheFactoryOptions& options,
233 WorkerCacheInterface** worker_cache) {
234 return WorkerCacheFactory(options, worker_cache);
235 };
236
237 // Provide direct access to the master from in-process clients.
238 LocalMaster::Register(target(), master_impl_.get(),
239 config.operation_timeout_in_ms());
240
241 return Status::OK();
242}

260Status GrpcServer::ParseChannelSpec(const WorkerCacheFactoryOptions& options,
261 GrpcChannelSpec* channel_spec) {
262 for (const auto& job : options.cluster_def->job()) {
263 std::map<int, string> host_ports;
264 for (const auto& task : job.tasks()) {
265 string& host_port = host_ports[task.first];
266 if (!host_port.empty()) {
267 return errors::InvalidArgument("JobDef for job \"", job.name(),
268 "\" specified two addresses for task \"",
269 task.first, "\": ", host_port, " and ",
270 task.second);
271 }
272 if (job.name() == *options.job_name && task.first == options.task_index) {
273 host_port = strings::StrCat("localhost:", bound_port_);
274 } else {
275 host_port = task.second;
276 }
277 }
278 TF_RETURN_IF_ERROR(channel_spec->AddHostPortsJob(job.name(), host_ports));
279 }
280 return Status::OK();
281}

283Status GrpcServer::WorkerCacheFactory(const WorkerCacheFactoryOptions& options,
284 WorkerCacheInterface** worker_cache) {
285 if (options.job_name == nullptr || options.job_name->empty()) {
286 Status s = errors::InvalidArgument(
287 "The master (current machine) is not included in the provided "
288 "cluster_def. ",
289 options.cluster_def->DebugString());
290 LOG(WARNING) << s;
291 return s;
292 }
293
294 GrpcChannelSpec channel_spec;
295 TF_RETURN_IF_ERROR(ParseChannelSpec(options, &channel_spec));
296
297 std::unique_ptr<GrpcChannelCache> channel_cache(
298 NewGrpcChannelCache(channel_spec, GetChannelCreationFunction()));
299
300 string name_prefix = strings::StrCat("/job:", *options.job_name, "/replica:0",
301 "/task:", options.task_index);
302
303 const string host_port = channel_cache->TranslateTask(name_prefix);
304 int requested_port;
305
306 if (!strings::safe_strto32(str_util::Split(host_port, ':')[1],
307 &requested_port)) {
308 return errors::Internal("Could not parse port for local server from \"",
309 channel_cache->TranslateTask(name_prefix), "\".");
310 }
311 if (requested_port != bound_port_) {
312 return errors::InvalidArgument("Requested port ", requested_port,
313 " differs from expected port ", bound_port_);
314 }
315
316 *worker_cache = NewGrpcWorkerCacheWithLocalWorker(
317 channel_cache.release(), worker_impl_.get(), name_prefix);
318 return Status::OK();
319}

321Status GrpcServer::Start() {
322 mutex_lock l(mu_);
323 switch (state_) {
324 case NEW: {
325 master_thread_.reset(
326 env_->StartThread(ThreadOptions(), "TF_master_service",
327 [this] { master_service_->HandleRPCsLoop(); }));
328 worker_thread_.reset(
329 env_->StartThread(ThreadOptions(), "TF_worker_service",
330 [this] { worker_service_->HandleRPCsLoop(); }));
331 state_ = STARTED;
332 LOG(INFO) << "Started server with target: " << target();
333 return Status::OK();
334 }
335 case STARTED:
336 LOG(INFO) << "Server already started (target: " << target() << ")";
337 return Status::OK();
338 case STOPPED:
339 return errors::FailedPrecondition("Server has stopped.");
340 default:
341 LOG(FATAL);
342 }
343}
344
345Status GrpcServer::Stop() {
346 mutex_lock l(mu_);
347 switch (state_) {
348 case NEW:
349 state_ = STOPPED;
350 return Status::OK();
351 case STARTED:
352 return errors::Unimplemented(
353 "Clean shutdown is not currently implemented");
354 case STOPPED:
355 LOG(INFO) << "Server already stopped (target: " << target() << ")";
356 return Status::OK();
357 default:
358 LOG(FATAL);
359 }
360}
361
362Status GrpcServer::Join() {
363 mutex_lock l(mu_);
364 switch (state_) {
365 case NEW:
366 // Prevent the server from being started subsequently.
367 state_ = STOPPED;
368 return Status::OK();
369 case STARTED:
370 case STOPPED:
371 master_thread_.reset();
372 worker_thread_.reset();
373 return Status::OK();
374 default:
375 LOG(FATAL);
376 }
377}

398/* static */
399Status GrpcServer::Create(const ServerDef& server_def, Env* env,
400 std::unique_ptr<ServerInterface>* out_server) {
401 std::unique_ptr<GrpcServer> ret(
402 new GrpcServer(server_def, env == nullptr ? Env::Default() : env));
403 ServiceInitFunction service_func = nullptr;
404 TF_RETURN_IF_ERROR(ret->Init(service_func, NewRpcRendezvousMgr));
405 *out_server = std::move(ret);
406 return Status::OK();
407}

423// Registers a `ServerFactory` for `GrpcServer` instances.
424class GrpcServerRegistrar {
425 public:
426 GrpcServerRegistrar() {
427 gpr_allocation_functions alloc_fns;
428 memset(&alloc_fns, 0, sizeof(alloc_fns));
429 alloc_fns.malloc_fn = port::Malloc;
430 alloc_fns.realloc_fn = port::Realloc;
431 alloc_fns.free_fn = port::Free;
432 gpr_set_allocation_functions(alloc_fns);
433 ServerFactory::Register("GRPC_SERVER", new GrpcServerFactory());
434 }
435};
