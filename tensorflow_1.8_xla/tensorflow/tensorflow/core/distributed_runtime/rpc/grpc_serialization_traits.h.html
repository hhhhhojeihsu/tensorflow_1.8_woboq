27const int kGrpcBufferWriterMaxBufferLength = 8192;
28
29class GrpcBufferWriter final
30 : public ::grpc::protobuf::io::ZeroCopyOutputStream {
31 public:
32 explicit GrpcBufferWriter(grpc_byte_buffer** bp, int block_size)
33 : block_size_(block_size), byte_count_(0), have_backup_(false) {
34 *bp = grpc_raw_byte_buffer_create(NULL, 0);
35 slice_buffer_ = &(*bp)->data.raw.slice_buffer;
36 }
37
38 ~GrpcBufferWriter() override {
39 if (have_backup_) {
40 grpc_slice_unref(backup_slice_);
41 }
42 }
43
44 bool Next(void** data, int* size) override {
45 if (have_backup_) {
46 slice_ = backup_slice_;
47 have_backup_ = false;
48 } else {
49 slice_ = grpc_slice_malloc(block_size_);
50 }
51 *data = GRPC_SLICE_START_PTR(slice_);
52 // On win x64, int is only 32bit
53 GPR_CODEGEN_ASSERT(GRPC_SLICE_LENGTH(slice_) <= INT_MAX);
54 byte_count_ += * size = (int)GRPC_SLICE_LENGTH(slice_);
55 grpc_slice_buffer_add(slice_buffer_, slice_);
56 return true;
57 }
58
59 void BackUp(int count) override {
60 grpc_slice_buffer_pop(slice_buffer_);
61 if (count == block_size_) {
62 backup_slice_ = slice_;
63 } else {
64 backup_slice_ =
65 grpc_slice_split_tail(&slice_, GRPC_SLICE_LENGTH(slice_) - count);
66 grpc_slice_buffer_add(slice_buffer_, slice_);
67 }
68 // It's dangerous to keep an inlined grpc_slice as the backup slice, since
69 // on a following Next() call, a reference will be returned to this slice
70 // via GRPC_SLICE_START_PTR, which will not be an address held by
71 // slice_buffer_.
72 have_backup_ = backup_slice_.refcount != NULL;
73 byte_count_ -= count;
74 }
75
76 grpc::protobuf::int64 ByteCount() const override { return byte_count_; }
77
78 private:
79 const int block_size_;
80 int64_t byte_count_;
81 grpc_slice_buffer* slice_buffer_;
82 bool have_backup_;
83 grpc_slice backup_slice_;
84 grpc_slice slice_;
85};
86
87class GrpcBufferReader final
88 : public ::grpc::protobuf::io::ZeroCopyInputStream {
89 public:
90 explicit GrpcBufferReader(grpc_byte_buffer* buffer)
91 : byte_count_(0), backup_count_(0) {
92 (void)grpc_byte_buffer_reader_init(&reader_, buffer);
93 }
94 ~GrpcBufferReader() override { grpc_byte_buffer_reader_destroy(&reader_); }
95
96 bool Next(const void** data, int* size) override {
97 if (backup_count_ > 0) {
98 *data = GRPC_SLICE_START_PTR(slice_) + GRPC_SLICE_LENGTH(slice_) -
99 backup_count_;
100 GPR_CODEGEN_ASSERT(backup_count_ <= INT_MAX);
101 *size = (int)backup_count_;
102 backup_count_ = 0;
103 return true;
104 }
105 if (!grpc_byte_buffer_reader_next(&reader_, &slice_)) {
106 return false;
107 }
108 grpc_slice_unref(slice_);
109 *data = GRPC_SLICE_START_PTR(slice_);
110 // On win x64, int is only 32bit
111 GPR_CODEGEN_ASSERT(GRPC_SLICE_LENGTH(slice_) <= INT_MAX);
112 byte_count_ += * size = (int)GRPC_SLICE_LENGTH(slice_);
113 return true;
114 }
115
116 void BackUp(int count) override { backup_count_ = count; }
117
118 bool Skip(int count) override {
119 const void* data;
120 int size;
121 while (Next(&data, &size)) {
122 if (size >= count) {
123 BackUp(size - count);
124 return true;
125 }
126 // size < count;
127 count -= size;
128 }
129 // error or we have too large count;
130 return false;
131 }
132
133 grpc::protobuf::int64 ByteCount() const override {
134 return byte_count_ - backup_count_;
135 }
136
137 private:
138 int64_t byte_count_;
139 int64_t backup_count_;
140 grpc_byte_buffer_reader reader_;
141 grpc_slice slice_;
142};

146// Defines specialized serialization/deserialization routines that
147// default to allowing a 2GB max message size.
148//
149// To instantiate this template for a particular type `T`, use
150// `TF_GRPC_ALLOW_UNLIMITED_MESSAGE_SIZE(T)`, as defined below.
151template <typename T>
152class UnlimitedSizeProtoSerializationTraits {
153 public:
154 static Status Serialize(const T& msg, grpc_byte_buffer** bp,
155 bool* own_buffer) {
156 *own_buffer = true;
157 int byte_size = msg.ByteSize();
158 if (byte_size < 0) {
159 return Status(StatusCode::INTERNAL, "Message length was negative");
160 } else if (byte_size <=
161 tensorflow_helper::kGrpcBufferWriterMaxBufferLength) {
162 grpc_slice slice = grpc_slice_malloc(byte_size);
163 GPR_CODEGEN_ASSERT(
164 GRPC_SLICE_END_PTR(slice) ==
165 msg.SerializeWithCachedSizesToArray(GRPC_SLICE_START_PTR(slice)));
166 *bp = grpc_raw_byte_buffer_create(&slice, 1);
167 grpc_slice_unref(slice);
168 return Status::OK;
169 } else {
170 tensorflow_helper::GrpcBufferWriter writer(
171 bp, tensorflow_helper::kGrpcBufferWriterMaxBufferLength);
172 return msg.SerializeToZeroCopyStream(&writer)
173 ? Status::OK
174 : Status(StatusCode::INTERNAL, "Failed to serialize message");
175 }
176 }
177
178 static Status Deserialize(grpc_byte_buffer* buffer, T* msg,
179 int max_message_size = INT_MAX) {
180 if (buffer == nullptr) {
181 return Status(StatusCode::INTERNAL, "No payload");
182 }
183 Status result = Status::OK;
184 {
185 tensorflow_helper::GrpcBufferReader reader(buffer);
186 ::grpc::protobuf::io::CodedInputStream decoder(&reader);
187 if (max_message_size == 0) {
188 // NOTE(mrry): Override maximum message size to 2GB.
189 decoder.SetTotalBytesLimit(INT_MAX, INT_MAX);
190 } else {
191 decoder.SetTotalBytesLimit(max_message_size, max_message_size);
192 }
193 if (!msg->ParseFromCodedStream(&decoder)) {
194 result = Status(StatusCode::INTERNAL, msg->InitializationErrorString());
195 }
196 if (!decoder.ConsumedEntireMessage()) {
197 result = Status(StatusCode::INTERNAL, "Did not read entire message");
198 }
199 }
200 grpc_byte_buffer_destroy(buffer);
201 return result;
202 }
203};
204
205} // namespace grpc

207// For the given protobuf message type `MessageType`, specializes the
208// gRPC serialization and deserialization such that the default
209// maximum message size is 2GB.
210#define TF_GRPC_ALLOW_UNLIMITED_MESSAGE_SIZE(MessageType) \
211 namespace grpc { \
212 template <> \
213 class SerializationTraits<MessageType> \
214 : public UnlimitedSizeProtoSerializationTraits<MessageType> {}; \
215 } // namespace grpc
