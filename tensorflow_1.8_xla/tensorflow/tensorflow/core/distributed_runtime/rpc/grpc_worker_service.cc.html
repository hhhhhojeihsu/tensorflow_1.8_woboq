53class GrpcWorkerService : public AsyncServiceInterface {
54 // TODO(ncteisen): consider adding a config var or flag for this
55 static constexpr const size_t kGrpcWorkerServiceThreadCount = 8;
56
57 public:
58 GrpcWorkerService(GrpcWorker* worker, ::grpc::ServerBuilder* builder)
59 : is_shutdown_(false) {
60 builder->RegisterService(&worker_service_);
61 for (int i = 0; i < kGrpcWorkerServiceThreadCount; i++) {
62 threads_.emplace_back(
63 new GrpcWorkerServiceThread(worker, builder, &worker_service_));
64 }
65 }
66
67 void Shutdown() override {
68 bool did_shutdown = false;
69 {
70 mutex_lock l(service_shutdown_mu_);
71 if (!is_shutdown_) {
72 LOG(INFO) << "Shutting down GrpcWorkerService.";
73 is_shutdown_ = true;
74 did_shutdown = true;
75 }
76 }
77 if (did_shutdown) {
78 for (auto& worker_thread : threads_) {
79 worker_thread->Shutdown();
80 }
81 }
82 }
83
84// This macro creates a new request for the given RPC method name
85// (e.g., `ENQUEUE_REQUEST(GetStatus, false);`), and enqueues it on
86// `this->cq_`.
87//
88// This macro is invoked one or more times for each RPC method to
89// ensure that there are sufficient completion queue entries to
90// handle incoming requests without blocking.
91//
92// The implementation of the request handler for each RPC method
93// must ensure that it calls ENQUEUE_REQUEST() for that RPC method,
94// to keep accepting new requests.
95#define ENQUEUE_REQUEST(method, supports_cancel) \
96 do { \
97 mutex_lock l(shutdown_mu_); \
98 if (!is_shutdown_) { \
99 Call<GrpcWorkerServiceThread, grpc::WorkerService::AsyncService, \
100 method##Request, method##Response>:: \
101 EnqueueRequestForMethod( \
102 worker_service_, cq_.get(), \
103 static_cast<int>(GrpcWorkerMethod::k##method), \
104 &GrpcWorkerServiceThread::method##Handler, (supports_cancel)); \
105 } \
106 } while (0)
107
108 // This method blocks forever handling requests from the completion queue.
109 void HandleRPCsLoop() override {
110 for (auto& worker_thread : threads_) {
111 worker_thread->Start();
112 }
113 for (auto& worker_thread : threads_) {
114 worker_thread->Join();
115 }
116 }
117
118 private:
119 // Thread wrapping class that drives work over a single gRPC
120 // CompletionQueue.
121 class GrpcWorkerServiceThread {
122 public:
123 explicit GrpcWorkerServiceThread(
124 GrpcWorker* worker, ::grpc::ServerBuilder* builder,
125 grpc::WorkerService::AsyncService* worker_service)
126 : worker_(worker),
127 worker_service_(worker_service),
128 is_shutdown_(false) {
129 cq_ = builder->AddCompletionQueue();
130 }
131
132 void Start() {
133 thread_.reset(worker_->env()->env->StartThread(
134 ThreadOptions(), "grpc_worker_service",
135 [this]() { HandleRPCsLoop(); }));
136 }
137
138 void Join() { thread_.reset(); } // Blocks until thread exits
139
140 void Shutdown() {
141 {
142 mutex_lock lock(shutdown_mu_);
143 is_shutdown_ = true;
144 }
145 cq_->Shutdown();
146 }
147
148 private:
149 void HandleRPCsLoop() {
150 // TODO(ncteisen): This may require performance engineering. We can
151 // change the number of threads, the number of handlers per thread,
152 // or even decide to specialize certain threads to certain methods.
153 ENQUEUE_REQUEST(GetStatus, false);
154 ENQUEUE_REQUEST(CreateWorkerSession, false);
155 ENQUEUE_REQUEST(DeleteWorkerSession, false);
156 ENQUEUE_REQUEST(CleanupAll, false);
157 ENQUEUE_REQUEST(RegisterGraph, false);
158 ENQUEUE_REQUEST(DeregisterGraph, false);
159
160 // TODO(ncteisen): Determine a better policy for enqueuing the
161 // appropriate number of each request type.
162 for (int i = 0; i < 1000; ++i) {
163 EnqueueRecvTensorRequestRaw();
164 }
165 for (int i = 0; i < 100; ++i) {
166 ENQUEUE_REQUEST(RunGraph, true);
167 }
168 for (int i = 0; i < 100; ++i) {
169 ENQUEUE_REQUEST(CleanupGraph, false);
170 }
171
172 ENQUEUE_REQUEST(Logging, false);
173 ENQUEUE_REQUEST(Tracing, false);
174
175 void* tag;
176 bool ok;
177
178 while (cq_->Next(&tag, &ok)) {
179 UntypedCall<GrpcWorkerServiceThread>::Tag* callback_tag =
180 static_cast<UntypedCall<GrpcWorkerServiceThread>::Tag*>(tag);
181 CHECK(callback_tag);
182 callback_tag->OnCompleted(this, ok);
183 }
184 }
185
186 private:
187 void Schedule(std::function<void()> f) {
188 worker_->env()->compute_pool->Schedule(std::move(f));
189 }
190
191 // The following section contains one request handler method per
192 // RPC. The `FooHandler` method is called (indirectly) by
193 // `HandleRPCsLoop()` when the next Foo RPC is received. Each
194 // `FooHandler` call schedules a closure on `worker_->env()->compute_pool`,
195 // and is responsible for requesting the next Foo call by calling
196 // `ENQUEUE_REQUEST(Foo)`.
197
198 template <class RequestMessage, class ResponseMessage>
199 using WorkerCall =
200 Call<GrpcWorkerServiceThread, grpc::WorkerService::AsyncService,
201 RequestMessage, ResponseMessage>;
202
203 void GetStatusHandler(
204 WorkerCall<GetStatusRequest, GetStatusResponse>* call) {
205 Schedule([this, call]() {
206 Status s = worker_->GetStatus(&call->request, &call->response);
207 call->SendResponse(ToGrpcStatus(s));
208 });
209 ENQUEUE_REQUEST(GetStatus, false);
210 }
211
212 void CreateWorkerSessionHandler(
213 WorkerCall<CreateWorkerSessionRequest, CreateWorkerSessionResponse>*
214 call) {
215 Schedule([this, call]() {
216 Status s =
217 worker_->CreateWorkerSession(&call->request, &call->response);
218 call->SendResponse(ToGrpcStatus(s));
219 });
220 ENQUEUE_REQUEST(CreateWorkerSession, false);
221 }
222
223 void DeleteWorkerSessionHandler(
224 WorkerCall<DeleteWorkerSessionRequest, DeleteWorkerSessionResponse>*
225 call) {
226 Schedule([this, call]() {
227 Status s =
228 worker_->DeleteWorkerSession(&call->request, &call->response);
229 call->SendResponse(ToGrpcStatus(s));
230 });
231 ENQUEUE_REQUEST(DeleteWorkerSession, false);
232 }
233
234 void CleanupAllHandler(
235 WorkerCall<CleanupAllRequest, CleanupAllResponse>* call) {
236 Schedule([this, call]() {
237 Status s = worker_->CleanupAll(&call->request, &call->response);
238 call->SendResponse(ToGrpcStatus(s));
239 });
240 ENQUEUE_REQUEST(CleanupAll, false);
241 }
242
243 void RegisterGraphHandler(
244 WorkerCall<RegisterGraphRequest, RegisterGraphResponse>* call) {
245 Schedule([this, call]() {
246 Status s = worker_->RegisterGraph(&call->request, &call->response);
247 call->SendResponse(ToGrpcStatus(s));
248 });
249 ENQUEUE_REQUEST(RegisterGraph, false);
250 }
251
252 void DeregisterGraphHandler(
253 WorkerCall<DeregisterGraphRequest, DeregisterGraphResponse>* call) {
254 Schedule([this, call]() {
255 Status s = worker_->DeregisterGraph(&call->request, &call->response);
256 call->SendResponse(ToGrpcStatus(s));
257 });
258 ENQUEUE_REQUEST(DeregisterGraph, false);
259 }
260
261 void RunGraphHandler(WorkerCall<RunGraphRequest, RunGraphResponse>* call) {
262 Schedule([this, call]() {
263 CallOptions* call_opts = new CallOptions;
264 ProtoRunGraphRequest* wrapped_request =
265 new ProtoRunGraphRequest(&call->request);
266 NonOwnedProtoRunGraphResponse* wrapped_response =
267 new NonOwnedProtoRunGraphResponse(&call->response);
268 call->SetCancelCallback([call_opts]() { call_opts->StartCancel(); });
269 worker_->RunGraphAsync(call_opts, wrapped_request, wrapped_response,
270 [call, call_opts, wrapped_request,
271 wrapped_response](const Status& s) {
272 call->ClearCancelCallback();
273 delete call_opts;
274 delete wrapped_request;
275 delete wrapped_response;
276 call->SendResponse(ToGrpcStatus(s));
277 });
278 });
279 ENQUEUE_REQUEST(RunGraph, true);
280 }
281
282 void RecvTensorHandlerRaw(
283 WorkerCall<RecvTensorRequest, ::grpc::ByteBuffer>* call) {
284 Schedule([this, call]() {
285 CallOptions* call_opts = new CallOptions;
286 call->SetCancelCallback([call_opts]() { call_opts->StartCancel(); });
287 worker_->GrpcRecvTensorAsync(call_opts, &call->request, &call->response,
288 [call, call_opts](const Status& s) {
289 call->ClearCancelCallback();
290 delete call_opts;
291 call->SendResponse(ToGrpcStatus(s));
292 });
293 });
294 EnqueueRecvTensorRequestRaw();
295 }
296
297 void CleanupGraphHandler(
298 WorkerCall<CleanupGraphRequest, CleanupGraphResponse>* call) {
299 Schedule([this, call]() {
300 Status s = worker_->CleanupGraph(&call->request, &call->response);
301 call->SendResponse(ToGrpcStatus(s));
302 });
303 ENQUEUE_REQUEST(CleanupGraph, false);
304 }
305
306 void LoggingHandler(WorkerCall<LoggingRequest, LoggingResponse>* call) {
307 Schedule([this, call]() {
308 Status s = worker_->Logging(&call->request, &call->response);
309 call->SendResponse(ToGrpcStatus(s));
310 });
311 ENQUEUE_REQUEST(Logging, false);
312 }
313
314 void TracingHandler(WorkerCall<TracingRequest, TracingResponse>* call) {
315 Schedule([this, call]() {
316 Status s = worker_->Tracing(&call->request, &call->response);
317 call->SendResponse(ToGrpcStatus(s));
318 });
319 ENQUEUE_REQUEST(Tracing, false);
320 }
321#undef ENQUEUE_REQUEST
322
323 void EnqueueRecvTensorRequestRaw() {
324 mutex_lock l(shutdown_mu_);
325 if (!is_shutdown_) {
326 Call<GrpcWorkerServiceThread, grpc::WorkerService::AsyncService,
327 RecvTensorRequest, ::grpc::ByteBuffer>::
328 EnqueueRequestForMethod(
329 worker_service_, cq_.get(),
330 static_cast<int>(GrpcWorkerMethod::kRecvTensor),
331 &GrpcWorkerServiceThread::RecvTensorHandlerRaw,
332 true /* supports cancel*/);
333 }
334 }
335
336 GrpcWorker* const worker_ = nullptr; // Not owned.
337 std::unique_ptr<::grpc::ServerCompletionQueue> cq_;
338 std::unique_ptr<Thread> thread_;
339 grpc::WorkerService::AsyncService* const worker_service_;
340
341 mutex shutdown_mu_;
342 bool is_shutdown_ GUARDED_BY(shutdown_mu_);
343 TF_DISALLOW_COPY_AND_ASSIGN(GrpcWorkerServiceThread);
344 }; // GrpcWorkerServiceThread
345
346 grpc::WorkerService::AsyncService worker_service_;
347 std::vector<std::unique_ptr<GrpcWorkerServiceThread>> threads_;
348
349 mutex service_shutdown_mu_;
350 bool is_shutdown_ GUARDED_BY(service_shutdown_mu_);
351
352 TF_DISALLOW_COPY_AND_ASSIGN(GrpcWorkerService);
353};

360// GrpcRecvTensorAsync: unlike the other Worker methods, which use protocol
361// buffers for a response object, to avoid extra protocol buffer serialization
362// overhead we generate our response directly into a ::grpc::ByteBuffer object
363void GrpcWorker::GrpcRecvTensorAsync(CallOptions* opts,
364 const RecvTensorRequest* request,
365 ::grpc::ByteBuffer* response,
366 StatusCallback done) {
367 Status s = recv_tensor_recent_request_ids_.TrackUnique(
368 request->request_id(), "RecvTensor (GrpcWorker)", *request);
369 if (!s.ok()) {
370 done(s);
371 return;
372 }
373
374 const int64 step_id = request->step_id();
375 const string& key = request->rendezvous_key();
376 TRACEPRINTF("RecvTensor: %lld %s", step_id, key.c_str());
377 Rendezvous::ParsedKey parsed;
378 s = Rendezvous::ParseKey(key, &parsed);
379 Device* src_dev = nullptr;
380 if (s.ok()) {
381 s = PrepareRecvTensor(parsed, &src_dev);
382 }
383 if (!s.ok()) {
384 done(s);
385 return;
386 }
387
388 // Request the tensor associated with the rendezvous key. Any time
389 // while waiting for the tensor to be produced, up until the start
390 // of execution of the callback lambda body below, an RPC
391 // cancellation should abort the rendezvous.
392 opts->SetCancelCallback([this, step_id]() { AbortStep(step_id); });
393 env_->rendezvous_mgr->RecvLocalAsync(
394 step_id, parsed,
395 [opts, response, done, src_dev](const Status& status,
396 const Rendezvous::Args& send_args,
397 const Rendezvous::Args& recv_args,
398 const Tensor& val, const bool is_dead) {
399 opts->ClearCancelCallback();
400 if (status.ok()) {
401 // DMA can only be used for Tensors that do not fall into
402 // the following three odd edge cases: 1) a zero-size
403 // buffer, 2) a dead tensor which has an uninit value, and
404 // 3) the tensor has the on_host allocation attribute,
405 // i.e. it's in CPU RAM *independent of its assigned
406 // device type*.
407 const bool on_host = send_args.alloc_attrs.on_host();
408 {
409 // Non-DMA cases.
410 if (src_dev->tensorflow_gpu_device_info() && (!on_host)) {
411#if GOOGLE_CUDA
412 const DeviceContext* send_dev_context = send_args.device_context;
413 AllocatorAttributes alloc_attrs;
414 alloc_attrs.set_gpu_compatible(true);
415 alloc_attrs.set_on_host(true);
416 Allocator* alloc = src_dev->GetAllocator(alloc_attrs);
417 Tensor* copy = new Tensor(alloc, val.dtype(), val.shape());
418 CHECK(send_dev_context)
419 << "send dev name: " << src_dev->name()
420 << " gpu_info: " << src_dev->tensorflow_gpu_device_info();
421 // "val" is on a GPU. Uses GPUUtil to fill the copy on host.
422 StatusCallback copy_ready = [response, done, copy,
423 is_dead](const Status& s) {
424 // The value is now ready to be returned on the wire.
425 grpc::EncodeTensorToByteBuffer(is_dead, *copy, response);
426 done(s);
427 delete copy;
428 };
429
430 GPUUtil::CopyGPUTensorToCPU(src_dev, send_dev_context, &val, copy,
431 copy_ready);
432#else
433 done(errors::Internal("No GPU device in process"));
434#endif // GOOGLE_CUDA
435 } else {
436 grpc::EncodeTensorToByteBuffer(is_dead, val, response);
437 done(Status::OK());
438 }
439 }
440 } else {
441 // !s.ok()
442 done(status);
443 }
444 });
445}

447void GrpcWorker::LoggingAsync(const LoggingRequest* request,
448 LoggingResponse* response, StatusCallback done) {
449 auto env = this->env();
450 if (env) {
451 auto session_mgr = (SessionMgr*)env->session_mgr;
452 if (session_mgr) {
453 session_mgr->SetLogging(request->rpc_logging());
454 for (const auto& step_id : request->fetch_step_id()) {
455 session_mgr->RetrieveLogs(step_id, response);
456 }
457 if (request->clear()) {
458 session_mgr->ClearLogs();
459 }
460 }
461 }
462 done(Status::OK());
463}
