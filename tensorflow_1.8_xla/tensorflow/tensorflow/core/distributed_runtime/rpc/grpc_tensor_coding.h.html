16#ifndef TENSORFLOW_CORE_DISTRIBUTED_RUNTIME_RPC_GRPC_TENSOR_CODING_H_
17#define TENSORFLOW_CORE_DISTRIBUTED_RUNTIME_RPC_GRPC_TENSOR_CODING_H_

19namespace grpc {
20class ByteBuffer;
21} // namespace grpc
22
23namespace tensorflow {
24class Tensor;
25class RecvTensorResponse;
26
27// TODO(jeff,sanjay): this should not be grpc specific. Instead of
28// grpc::ByteBuffer*, it should accept an object of an interface type
29// to which owned byte-arrays can be added.
30namespace grpc {
31
32// Encode a RecvTensorResponse protocol buffer into a byte buffer in a
33// format that is parseable as a RecvTensorResponse protocol buffer
34// holding "proto".
35//
36// Discards original contents of *result.
37void EncodeRecvTensorResponseToByteBuffer(const RecvTensorResponse& proto,
38 ::grpc::ByteBuffer* result);
39
40// Encode a Tensor into a byte buffer in a format that is parseable
41// as a RecvTensorResponse protocol buffer holding "val".
42//
43// "is_dead" is the value to encode for "RecvTensorResponse::is_dead"
44// (tensor is the output of a dead node and content is invalid because
45// control flow operations elsewhere caused the path on which this
46// Tensor exists to not be taken).
47//
48// "val" holds the tensor value to be encoded.
49//
50// Discards original contents of *result.
51void EncodeTensorToByteBuffer(bool is_dead, const Tensor& val,
52 ::grpc::ByteBuffer* result);
53
54} // namespace grpc
55} // namespace tensorflow
