36const size_t kSchemePrefixLength = strlen(kSchemePrefix);

38GrpcSession::GrpcSession(const SessionOptions& options)
39 : options_(options), current_graph_version_(-1) {}

41GrpcSession::~GrpcSession() {}

43/* static */
44Status GrpcSession::Create(const SessionOptions& options,
45 std::unique_ptr<GrpcSession>* out_session) {
46 std::unique_ptr<GrpcSession> session(new GrpcSession(options));
47 std::unique_ptr<MasterInterface> master;
48 // For testing, we enable the client to disable the use of the local
49 // master registry, so that the RPC stack is exercised.
50 if (!options.config.rpc_options().use_rpc_for_inprocess_master()) {
51 master = LocalMaster::Lookup(options.target);
52 }
53 if (!master) {
54 SharedGrpcChannelPtr master_channel;
55 TF_RETURN_IF_ERROR(NewHostPortGrpcChannel(
56 options.target.substr(kSchemePrefixLength), &master_channel));
57 master.reset(NewGrpcMaster(master_channel));
58 }
59 session->SetRemoteMaster(std::move(master));
60 *out_session = std::move(session);
61 return Status::OK();
62}

65// Re-encodes constant represented in tensor proto into
66// tensor_content, which is slightly better (less copies and lower peak
67// memory usage) when used with rpc subsystems.
68void ReEncodeConsts(GraphDef* gdef) {
69 for (NodeDef& ndef : *(gdef->mutable_node())) {
70 if (ndef.op() == "Const") {
71 TensorProto* proto = nullptr;
72 for (auto& attr : *ndef.mutable_attr()) {
73 if (attr.first == "value") {
74 proto = attr.second.mutable_tensor();
75 }
76 }
77 if (proto != nullptr && proto->tensor_content().empty() &&
78 proto->ByteSizeLong() > 64) {
79 // If the constant is encoded with repeated proto fields and
80 // it is moderate large, we re-encode it in tensor_content as
81 // a Cord. This is mildly helpful for reducing the peak memory
82 // usage on the server side where GraphDef/NodeDef are copied
83 // quite often.
84 Tensor parsed(proto->dtype());
85 if (parsed.FromProto(*proto)) {
86 parsed.AsProtoTensorContent(proto);
87 }
88 }
89 }
90 }
91}

94Status GrpcSession::Handle(string* out_handle) {
95 mutex_lock l(mu_);
96 if (handle_.empty()) {
97 return errors::InvalidArgument("A session is not created yet....");
98 }
99 *out_handle = handle_;
100 return Status::OK();
101}

103Status GrpcSession::CreateImpl(CallOptions* call_options,
104 const GraphDef& graph) {
105 {
106 mutex_lock l(mu_);
107 if (!handle_.empty()) {
108 return errors::InvalidArgument("A session is alive.");
109 }
110 }
111 CreateSessionRequest req;
112 *req.mutable_config() = options_.config;
113 *req.mutable_graph_def() = graph;
114 req.set_target(options_.target);
115 ReEncodeConsts(req.mutable_graph_def());
116 CreateSessionResponse resp;
117 Status s = master_->CreateSession(call_options, &req, &resp);
118 if (s.ok()) {
119 mutex_lock l(mu_);
120 swap(handle_, *(resp.mutable_session_handle()));
121 current_graph_version_ = resp.graph_version();
122 }
123 return s;
124}

126Status GrpcSession::Create(const GraphDef& graph) {
127 CallOptions call_options;
128 call_options.SetTimeout(options_.config.operation_timeout_in_ms());
129 return CreateImpl(&call_options, graph);
130}

139Status GrpcSession::ExtendImpl(CallOptions* call_options,
140 const GraphDef& graph) {
141 bool handle_is_empty;
142 {
143 mutex_lock l(mu_);
144 handle_is_empty = handle_.empty();
145 }
146 if (handle_is_empty) {
147 // Session was unitialized, so simply initialize the session with 'graph'.
148 return Create(graph);
149 }
150 mutex_lock l(mu_);
151 ExtendSessionRequest req;
152 req.set_session_handle(handle_);
153 *req.mutable_graph_def() = graph;
154 req.set_current_graph_version(current_graph_version_);
155 ExtendSessionResponse resp;
156 Status s = master_->ExtendSession(call_options, &req, &resp);
157 if (s.ok()) {
158 current_graph_version_ = resp.new_graph_version();
159 }
160 return s;
161}

176Status GrpcSession::RunHelper(
177 const RunOptions& run_options,
178 const std::vector<std::pair<string, Tensor>>& inputs,
179 const std::vector<string>& output_tensor_names,
180 const std::vector<string>& target_node_names, std::vector<Tensor>* outputs,
181 RunMetadata* run_metadata, const string& prun_handle) {
182 // Convert to proto
183 std::unique_ptr<MutableRunStepRequestWrapper> req(
184 master_->CreateRunStepRequest());
185 std::unique_ptr<MutableRunStepResponseWrapper> resp(
186 master_->CreateRunStepResponse());
187
188 *req->mutable_options() = run_options;
189
190 if (run_options.timeout_in_ms() == 0) {
191 req->mutable_options()->set_timeout_in_ms(
192 options_.config.operation_timeout_in_ms());
193 }
194
195 if (!prun_handle.empty()) {
196 req->set_partial_run_handle(prun_handle);
197 }
198
199 for (const auto& it : inputs) {
200 req->add_feed(it.first, it.second);
201 }
202
203 // Support long error messages by storing the error code in the response body.
204 req->set_store_errors_in_response_body(true);
205
206 // Build an index from fetch tensor name to first index in
207 // output_tensor_names.
208 std::unordered_map<string, int> output_name_to_offset;
209 for (int i = 0; i < output_tensor_names.size(); ++i) {
210 const string& name = output_tensor_names[i];
211 if (output_name_to_offset.insert(std::make_pair(name, i)).second) {
212 req->add_fetch(name);
213 }
214 }
215 for (const string& target : target_node_names) {
216 req->add_target(target);
217 }
218
219 CallOptions call_options;
220 call_options.SetTimeout(req->options().timeout_in_ms());
221 TF_RETURN_IF_ERROR(RunProto(&call_options, req.get(), resp.get()));
222
223 // Look for an extended error returned in the response body.
224 if (resp->status_code() != error::Code::OK) {
225 return Status(resp->status_code(), resp->status_error_message());
226 }
227
228 if (!output_tensor_names.empty()) {
229 outputs->resize(output_tensor_names.size());
230 }
231
232 // Convert response back to Tensors in the correct order.
233 for (size_t i = 0; i < resp->num_tensors(); ++i) {
234 auto fetch_it = output_name_to_offset.find(resp->tensor_name(i));
235 if (fetch_it == output_name_to_offset.end()) {
236 return errors::Internal("Received response for unrequested fetch: ",
237 resp->tensor_name(i));
238 }
239
240 Tensor output;
241 TF_RETURN_IF_ERROR(resp->TensorValue(i, &output));
242 (*outputs)[fetch_it->second] = output;
243 }
244 // In the unlikely event that output_tensor_names contains duplicates, fill in
245 // the duplicate values.
246 if (output_name_to_offset.size() != output_tensor_names.size()) {
247 for (int i = 0; i < output_tensor_names.size(); ++i) {
248 const string& name = output_tensor_names[i];
249 int offset = output_name_to_offset[name];
250 if (offset != i) {
251 (*outputs)[i] = (*outputs)[offset];
252 }
253 }
254 }
255
256 if (run_metadata) {
257 run_metadata->Swap(resp->mutable_metadata());
258 }
259
260 return Status::OK();
261}

292Status GrpcSession::PRunSetup(const std::vector<string>& input_names,
293 const std::vector<string>& output_names,
294 const std::vector<string>& target_nodes,
295 string* handle) {
296 // Convert to proto
297 PartialRunSetupRequest req;
298 PartialRunSetupResponse resp;
299 CallOptions call_options;
300 TF_RETURN_IF_ERROR(Handle(req.mutable_session_handle()));
301 for (const string& feed : input_names) {
302 req.add_feed(feed);
303 }
304 for (const string& fetch : output_names) {
305 req.add_fetch(fetch);
306 }
307 for (const string& target : target_nodes) {
308 req.add_target(target);
309 }
310 call_options.SetTimeout(options_.config.operation_timeout_in_ms());
311 TF_RETURN_IF_ERROR(master_->PartialRunSetup(&call_options, &req, &resp));
312 *handle = resp.partial_run_handle();
313 return Status::OK();
314}

326Status GrpcSession::Close() {
327 CloseSessionRequest req;
328 {
329 mutex_lock l(mu_);
330 if (handle_.empty()) {
331 return Status::OK();
332 }
333 req.set_session_handle(handle_);
334 handle_.clear();
335 }
336 CloseSessionResponse resp;
337 CallOptions call_options;
338 call_options.SetTimeout(options_.config.operation_timeout_in_ms());
339 return master_->CloseSession(&call_options, &req, &resp);
340}

342Status GrpcSession::ListDevices(std::vector<DeviceAttributes>* response) {
343 ListDevicesRequest req;
344 {
345 mutex_lock l(mu_);
346 req.set_session_handle(handle_);
347 }
348 if (req.session_handle().empty()) {
349 LOG(WARNING) << "GrpcSession::ListDevices will initialize the session with "
350 "an empty graph and other defaults because the session has "
351 "not yet been created.";
352 GraphDef graph_def;
353 TF_RETURN_IF_ERROR(Create(graph_def));
354 {
355 mutex_lock l(mu_);
356 req.set_session_handle(handle_);
357 }
358 }
359 ListDevicesResponse resp;
360 CallOptions call_options;
361 call_options.SetTimeout(options_.config.operation_timeout_in_ms());
362 Status s = master_->ListDevices(&call_options, &req, &resp);
363 if (!s.ok()) {
364 LOG(ERROR) << "Could not list devices: " << s;
365 return s;
366 }
367
368 response->clear();
369 response->reserve(resp.local_device_size() + resp.remote_device_size());
370 for (const auto& device_attr : resp.local_device()) {
371 response->emplace_back(device_attr);
372 }
373 for (const auto& device_attr : resp.remote_device()) {
374 response->emplace_back(device_attr);
375 }
376 return Status::OK();
377}

383// Static method.
384Status GrpcSession::Reset(const SessionOptions& options,
385 const std::vector<string>& containers) {
386 SharedGrpcChannelPtr master_channel;
387 TF_RETURN_IF_ERROR(NewHostPortGrpcChannel(
388 options.target.substr(kSchemePrefixLength), &master_channel));
389 auto master = NewGrpcMaster(master_channel);
390 ResetRequest req;
391 for (const auto& c : containers) req.add_container(c);
392 ResetResponse resp;
393 CallOptions call_options;
394 call_options.SetTimeout(options.config.operation_timeout_in_ms());
395 Status ret = master->Reset(&call_options, &req, &resp);
396 delete master;
397 return ret;
398}

400Status GrpcSession::MakeCallable(const CallableOptions& callable_options,
401 CallableHandle* out_handle) {
402 MakeCallableRequest req;
403 TF_RETURN_IF_ERROR(Handle(req.mutable_session_handle()));
404 *req.mutable_options() = callable_options;
405 MakeCallableResponse resp;
406 CallOptions call_options;
407 call_options.SetTimeout(options_.config.operation_timeout_in_ms());
408 TF_RETURN_IF_ERROR(master_->MakeCallable(&call_options, &req, &resp));
409 *out_handle = resp.handle();
410 return Status::OK();
411}

413Status GrpcSession::RunCallable(CallableHandle handle,
414 const std::vector<Tensor>& feed_tensors,
415 std::vector<Tensor>* fetch_tensors,
416 RunMetadata* run_metadata) {
417 RunCallableRequest req;
418 TF_RETURN_IF_ERROR(Handle(req.mutable_session_handle()));
419 req.set_handle(handle);
420 for (const Tensor& feed : feed_tensors) {
421 feed.AsProtoTensorContent(req.mutable_feed()->Add());
422 }
423
424 RunCallableResponse resp;
425 CallOptions call_options;
426 call_options.SetTimeout(options_.config.operation_timeout_in_ms());
427 TF_RETURN_IF_ERROR(master_->RunCallable(&call_options, &req, &resp));
428 for (const TensorProto& fetch : resp.fetch()) {
429 Tensor fetch_tensor;
430 if (!fetch_tensor.FromProto(cpu_allocator(), fetch)) {
431 return errors::Internal(
432 "Could not parse fetched tensor data in response from master.");
433 }
434 fetch_tensors->push_back(std::move(fetch_tensor));
435 }
436 return Status::OK();
437}

449class GrpcSessionFactory : public SessionFactory {
450 public:
451 bool AcceptsOptions(const SessionOptions& options) override {
452 return str_util::StartsWith(options.target, kSchemePrefix);
453 }
454
455 Session* NewSession(const SessionOptions& options) override {
456 std::unique_ptr<GrpcSession> ret;
457 Status s = GrpcSession::Create(options, &ret);
458 if (s.ok()) {
459 return ret.release();
460 } else {
461 LOG(ERROR) << "Error during session construction: " << s.ToString();
462 return nullptr;
463 }
464 }
465
466 // Invokes the session specific static method to reset containers.
467 Status Reset(const SessionOptions& options,
468 const std::vector<string>& containers) override {
469 return GrpcSession::Reset(options, containers);
470 }
471};
