34class GrpcCall {
35 public:
36 explicit GrpcCall(CallContainer<GrpcCall>* container, int index, bool try_rpc,
37 const string* request_msg, string* response_msg,
38 int32* status_code, string* status_message)
39 : container_(container),
40 index_(index),
41 try_rpc_(try_rpc),
42 request_msg_(request_msg),
43 response_msg_(response_msg),
44 status_code_(status_code),
45 status_message_(status_message) {}
46
47 void StartCancel() { call_opts_.StartCancel(); }
48
49 void Done(const Status& s) {
50 DCHECK(container_ != nullptr);
51 if (!s.ok() && try_rpc_) {
52 DCHECK(status_code_ != nullptr);
53 DCHECK(status_message_ != nullptr);
54 *status_code_ = s.code();
55 *status_message_ = s.error_message();
56 }
57 container_->Done(s, index_);
58 }
59
60 const string& request() const { return *request_msg_; }
61 string* response() const { return response_msg_; }
62 CallOptions* call_opts() { return &call_opts_; }
63
64 private:
65 CallContainer<GrpcCall>* const container_;
66 const int index_;
67 bool try_rpc_;
68 CallOptions call_opts_;
69 const string* request_msg_;
70 string* response_msg_;
71 int* status_code_;
72 string* status_message_;
73};

77GrpcRPCFactory::GrpcRPCFactory(OpKernelConstruction* ctx, bool fail_fast,
78 int64 timeout_in_ms)
79 : RPCFactory(), fail_fast_(fail_fast), timeout_in_ms_(timeout_in_ms) {
80 // TODO(ebrevdo): Investigate possible performance improvements by
81 // replacing this thread with a threadpool.
82 polling_thread_ =
83 ctx->env()->StartThread(ThreadOptions(), "rpc_op_grpc_factory", [this]() {
84 void* tag;
85 bool ok;
86 while (completion_queue_.Next(&tag, &ok)) {
87 GrpcClientCQTag* callback_tag = static_cast<GrpcClientCQTag*>(tag);
88 callback_tag->OnCompleted(ok);
89 }
90 });
91}
92
93GrpcRPCFactory::~GrpcRPCFactory() {
94 // The amount of time we wait depends on several parameters, including:
95 // - the value of the fail_fast attribute.
96 // - the timeout option of the rpc call in the proto declaration.
97 // - the network roundtrip time and service's execution time.
98 //
99 // If a connection is made but the service doesn't ever respond, and
100 // there is no timeout option set for this rpc call, then it is
101 // possible the RPC request will wait forever.
102 //
103 completion_queue_.Shutdown();
104 delete polling_thread_;
105}

107void GrpcRPCFactory::Call(OpKernelContext* ctx, int64 num_elements,
108 const Tensor& address_t, const Tensor& method_t,
109 const Tensor& request_t, const bool try_rpc,
110 Tensor* response_t, Tensor* status_code_t,
111 Tensor* status_message_t,
112 AsyncOpKernel::DoneCallback done) {
113 auto address = address_t.flat<string>();
114 auto method = method_t.flat<string>();
115 auto request = request_t.flat<string>();
116
117 // Stubs are maintained by the GrpcRPCFactory class and will be
118 // deleted when the class is destroyed.
119 ::grpc::GenericStub* singleton_stub = nullptr;
120 if (address.size() == 1) {
121 singleton_stub = GetOrCreateStubForAddress(address(0));
122 }
123 auto get_stub = [&address, this,
124 singleton_stub](int64 ix) -> ::grpc::GenericStub* {
125 return (address.size() > 1) ? GetOrCreateStubForAddress(address(ix))
126 : singleton_stub;
127 };
128 auto get_method_ptr = [&method](int64 ix) -> const string* {
129 return (method.size() > 1) ? &(method(ix)) : &(method(0));
130 };
131 auto get_request_ptr = [&request](int64 ix) -> const string* {
132 return (request.size() > 1) ? &(request(ix)) : &(request(0));
133 };
134
135 if (try_rpc) {
136 // In this case status_code will never be set in the response,
137 // so we just set it to OK.
138 DCHECK(status_code_t != nullptr);
139 status_code_t->flat<int32>().setConstant(
140 static_cast<int>(errors::Code::OK));
141 }
142
143 CancellationManager* cm = ctx->cancellation_manager();
144 CancellationToken cancellation_token = cm->get_cancellation_token();
145
146 // This object will delete itself when done.
147 auto* container =
148 new CallContainer<GrpcCall>(ctx, num_elements, fail_fast_, try_rpc,
149 std::move(done), cancellation_token);
150
151 auto response = response_t->flat<string>();
152 int32* status_code_ptr = nullptr;
153 string* status_message_ptr = nullptr;
154 if (try_rpc) {
155 status_code_ptr = status_code_t->flat<int32>().data();
156 status_message_ptr = status_message_t->flat<string>().data();
157 }
158 for (int i = 0; i < num_elements; ++i) {
159 container->calls()->emplace_back(
160 container, i, try_rpc, get_request_ptr(i), &response(i),
161 (try_rpc) ? &status_code_ptr[i] : nullptr,
162 (try_rpc) ? &status_message_ptr[i] : nullptr);
163 }
164
165 int i = 0;
166 for (GrpcCall& call : *(container->calls())) {
167 // This object will delete itself when done.
168 new RPCState<string>(get_stub(i), &completion_queue_, *get_method_ptr(i),
169 call.request(), call.response(),
170 /*done=*/[&call](const Status& s) { call.Done(s); },
171 call.call_opts(), fail_fast_, timeout_in_ms_);
172 ++i;
173 }
174
175 // Need to register this callback after all the RPCs are in
176 // flight; otherwise we may try to cancel an RPC *before* it
177 // launches, which is a no-op, and then fall into a deadlock.
178 bool is_cancelled = !cm->RegisterCallback(
179 cancellation_token, [container]() { container->StartCancel(); });
180
181 if (is_cancelled) {
182 ctx->SetStatus(errors::Cancelled("Operation has been cancelled."));
183 // container's reference counter will take care of calling done().
184 container->StartCancel();
185 }
186}

188::grpc::GenericStub* GrpcRPCFactory::GetOrCreateStubForAddress(
189 const string& address) {
190 mutex_lock lock(mu_);
191
192 auto stub = stubs_.find(address);
193 if (stub != stubs_.end()) return stub->second.get();
194
195 ChannelPtr channel = CreateChannelForAddress(address);
196 auto* created = new ::grpc::GenericStub(channel);
197 stubs_[address].reset(created);
198 return created;
199}
200
201GrpcRPCFactory::ChannelPtr GrpcRPCFactory::CreateChannelForAddress(
202 const string& address) {
203 ::grpc::ChannelArguments args;
204 args.SetInt(GRPC_ARG_MAX_MESSAGE_LENGTH, std::numeric_limits<int32>::max());
205
206 // Set a standard backoff timeout of 1s instead of the
207 // (sometimes default) 20s.
208 args.SetInt("grpc.testing.fixed_reconnect_backoff_ms", 1000);
209 return ::grpc::CreateCustomChannel(
210 /*target=*/address, ::grpc::InsecureChannelCredentials(), args);
211}
