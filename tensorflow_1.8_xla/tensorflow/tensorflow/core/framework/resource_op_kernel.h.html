31// ResourceOpKernel<T> is a virtual base class for resource op implementing
32// interface type T. The inherited op looks up the resource name (determined by
33// ContainerInfo), and creates a new resource if necessary.
34//
35// Requirements:
36// - Op must be marked as stateful.
37// - Op must have `container` and `shared_name` attributes. Empty `container`
38// means using the default container. Empty `shared_name` means private
39// resource.
40// - Subclass must override CreateResource().
41// - Subclass is encouraged to override VerifyResource().
42template <typename T>
43class ResourceOpKernel : public OpKernel {
44 public:
45 explicit ResourceOpKernel(OpKernelConstruction* context) : OpKernel(context) {
46 OP_REQUIRES_OK(context,
47 context->allocate_persistent(DT_STRING, TensorShape({2}),
48 &handle_, nullptr));
49 }
50
51 // The resource is deleted from the resource manager only when it is private
52 // to kernel. Ideally the resource should be deleted when it is no longer held
53 // by anyone, but it would break backward compatibility.
54 ~ResourceOpKernel() override {
55 if (resource_ != nullptr) {
56 resource_->Unref();
57 if (cinfo_.resource_is_private_to_kernel()) {
58 if (!cinfo_.resource_manager()
59 ->template Delete<T>(cinfo_.container(), cinfo_.name())
60 .ok()) {
61 // Do nothing; the resource can have been deleted by session resets.
62 }
63 }
64 }
65 }
66
67 void Compute(OpKernelContext* context) override LOCKS_EXCLUDED(mu_) {
68 mutex_lock l(mu_);
69 if (resource_ == nullptr) {
70 ResourceMgr* mgr = context->resource_manager();
71 OP_REQUIRES_OK(context, cinfo_.Init(mgr, def()));
72
73 T* resource;
74 OP_REQUIRES_OK(
75 context,
76 mgr->LookupOrCreate<T>(cinfo_.container(), cinfo_.name(), &resource,
77 [this](T** ret) EXCLUSIVE_LOCKS_REQUIRED(mu_) {
78 Status s = CreateResource(ret);
79 if (!s.ok() && *ret != nullptr) {
80 CHECK((*ret)->Unref());
81 }
82 return s;
83 }));
84
85 Status s = VerifyResource(resource);
86 if (TF_PREDICT_FALSE(!s.ok())) {
87 resource->Unref();
88 context->SetStatus(s);
89 return;
90 }
91
92 auto h = handle_.AccessTensor(context)->template flat<string>();
93 h(0) = cinfo_.container();
94 h(1) = cinfo_.name();
95 resource_ = resource;
96 }
97 if (context->expected_output_dtype(0) == DT_RESOURCE) {
98 OP_REQUIRES_OK(context, MakeResourceHandleToOutput(
99 context, 0, cinfo_.container(), cinfo_.name(),
100 MakeTypeIndex<T>()));
101 } else {
102 context->set_output_ref(0, &mu_, handle_.AccessTensor(context));
103 }
104 }
105
106 protected:
107 // Variables accessible from subclasses.
108 mutex mu_;
109 ContainerInfo cinfo_ GUARDED_BY(mu_);
110 T* resource_ GUARDED_BY(mu_) = nullptr;
111
112 private:
113 // Must return a T descendant allocated with new that ResourceOpKernel will
114 // take ownership of.
115 virtual Status CreateResource(T** resource) EXCLUSIVE_LOCKS_REQUIRED(mu_) = 0;
116
117 // During the first Compute(), resource is either created or looked up using
118 // shared_name. In the latter case, the resource found should be verified if
119 // it is compatible with this op's configuration. The verification may fail in
120 // cases such as two graphs asking queues of the same shared name to have
121 // inconsistent capacities.
122 virtual Status VerifyResource(T* resource) { return Status::OK(); }
123
124 PersistentTensor handle_ GUARDED_BY(mu_);
125};
