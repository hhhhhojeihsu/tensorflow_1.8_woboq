38// Extracts the actual type from "attr_values" based on its definition
39// "arg_def".
40//
41// If "arg_def" is a N*T type, *is_type_list is set to false, and
42// *dtypes is set to be a vector of size N and each element is T.
43//
44// If "arg_def" is a list(type), *is_type_list is set to true, and
45// *dtypes is set to be a vector of types specified in attrs for
46// arg_def.
47//
48// Otherwise (arg_def is a simple type T), *is_type_list is set to
49// false, and *dtypes is set to a single element vector, whose only
50// element is T.
51Status ArgNumType(AttrSlice attrs, const OpDef::ArgDef& arg_def,
52 bool* is_type_list, DataTypeVector* dtypes) {
53 dtypes->clear();
54 if (!arg_def.type_list_attr().empty()) {
55 const AttrValue* v = attrs.Find(arg_def.type_list_attr());
56 if (v == nullptr) {
57 return errors::NotFound("type attr not found: ",
58 arg_def.type_list_attr());
59 }
60 *is_type_list = true;
61 for (int i = 0; i < v->list().type_size(); ++i) {
62 dtypes->push_back(v->list().type(i));
63 }
64 return Status::OK();
65 }
66
67 *is_type_list = false;
68 int num = 1;
69 if (!arg_def.number_attr().empty()) {
70 const AttrValue* v = attrs.Find(arg_def.number_attr());
71 if (v == nullptr) {
72 return errors::NotFound("type attr not found: ", arg_def.type_attr());
73 }
74 num = v->i();
75 }
76
77 DataType dtype;
78 if (arg_def.type() != DT_INVALID) {
79 dtype = arg_def.type();
80 } else if (arg_def.type_attr().empty()) {
81 dtype = DT_INVALID;
82 } else {
83 const AttrValue* v = attrs.Find(arg_def.type_attr());
84 if (v == nullptr) {
85 return errors::NotFound("type attr not found: ", arg_def.type_attr());
86 }
87 dtype = v->type();
88 }
89 dtypes->resize(num, dtype);
90 return Status::OK();
91}

100Status ValidateSignatureWithAttrs(const OpDef& sig, AttrSlice attr_values) {
101 // attr_values should specify all attrs defined in fdef.
102 for (const auto& a : sig.attr()) {
103 const AttrValue* v = attr_values.Find(a.name());
104 if (!v) {
105 return errors::NotFound("Attr ", a.name(), " is not found from ",
106 SummarizeOpDef(sig));
107 }
108 Status status = AttrValueHasType(*v, a.type());
109 if (!status.ok()) {
110 errors::AppendToMessage(&status, "for attr '", a.name(), "'");
111 return status;
112 }
113 }
114

116// Right now the C++ function gradient code assumes it can pass
117// all the attrs of the function to the gradient, and any attrs that
118// the gradient doesn't care about will be ignored.
119#if 0
120 if (attr_values.size() != sig.attr_size()) {
121 for (const auto& a : attr_values) {
122 // TODO(josh11b): Possibly should ignore attrs that start with "_" here?
123 bool found = false;
124 for (const auto& s : sig.attr()) {
125 if (a.first == s.name()) {
126 found = true;
127 break;
128 }
129 }
130 if (!found) {
131 return errors::NotFound("Attr ", a.first, " is not found in ",
132 SummarizeOpDef(sig));
133 }
134 }
135 }
136#endif
137
138 return Status::OK();
139}

148 result_.nodes.clear();

151 // Builds index for nodes that can be used as node's input arguments.
152 Status BuildInputArgIndex(const OpDef::ArgDef& arg_def,
153 AttrSlice attr_values) {
154 bool is_type_list;
155 DataTypeVector dtypes;
156 TF_RETURN_IF_ERROR(
157 ArgNumType(attr_values, arg_def, &is_type_list, &dtypes));
158 CHECK_GE(dtypes.size(), size_t{1});
159 int arg_index = result_.nodes.size();
160 TF_RETURN_IF_ERROR(
161 AddItem(arg_def.name(), {true, arg_index, 0, is_type_list, dtypes}));
162 // Creates dtypes.size() nodes in the graph.
163 for (size_t i = 0; i < dtypes.size(); ++i) {
164 TF_RETURN_IF_ERROR(AddItem(strings::StrCat(arg_def.name(), ":", i),
165 {true, arg_index, 0, false, {dtypes[i]}}));
166 DCHECK_EQ(arg_index, result_.nodes.size());
167 string name = arg_def.name();
168 if (dtypes.size() > 1) {
169 strings::StrAppend(&name, "_", i);
170 }
171 NodeDef* gnode = AddNode(name);
172 gnode->set_op(FunctionLibraryDefinition::kArgOp);
173 AddAttr("T", dtypes[i], gnode);
174 AddAttr("index", arg_index, gnode);
175 result_.arg_types.push_back(dtypes[i]);
176 ++arg_index;
177 }
178 return Status::OK();
179 }

181 Status BuildNodeOutputIndex(const NodeDef& node, AttrSlice attrs,
182 const int arg_index) {
183 const OpDef* node_sig = nullptr;
184 TF_RETURN_IF_ERROR(get_function_(node.op(), &node_sig));
185 if (node_sig->output_arg_size() == 0) {
186 return AddItem(node.name(), {false, arg_index, 0, false, {}});
187 }
188 const int num_retval = node_sig->output_arg_size();
189 int start = 0;
190 bool is_type_list;
191 DataTypeVector dtypes;
192 for (int i = 0; i < num_retval; ++i) {
193 TF_RETURN_IF_ERROR(
194 ArgNumType(attrs, node_sig->output_arg(i), &is_type_list, &dtypes));
195 // Note that we rely on the backwards-compatibility test enforcing
196 // that output_arg(*).name() doesn't change here.
197 const string base_name =
198 strings::StrCat(node.name(), ":", node_sig->output_arg(i).name());
199 TF_RETURN_IF_ERROR(
200 AddItem(base_name, {false, arg_index, start, is_type_list, dtypes}));
201 for (int j = 0; j < static_cast<int>(dtypes.size()); ++j) {
202 TF_RETURN_IF_ERROR(
203 AddItem(strings::StrCat(base_name, ":", j),
204 {false, arg_index, start + j, false, {dtypes[j]}}));
205 }
206 start += dtypes.size();
207 }
208 return Status::OK();
209 }

211 Status InstantiateNode(const NodeDef& fnode, AttrSlice attrs) {
212 const OpDef* fnode_sig = nullptr;
213 TF_CHECK_OK(get_function_(fnode.op(), &fnode_sig));
214 NodeDef* gnode = AddNode(fnode.name());
215 gnode->set_op(fnode.op());
216 gnode->set_device(fnode.device());
217 int gnode_idx = nodes_.size() - 1;
218
219 // Input
220 const int num_args = fnode_sig->input_arg_size();
221 bool is_type_list; // ignored
222 DataTypeVector dtypes;
223 int fnode_arg_index = 0;
224 for (int i = 0; i < num_args; ++i) {
225 TF_RETURN_IF_ERROR(
226 ArgNumType(attrs, fnode_sig->input_arg(i), &is_type_list, &dtypes));
227 // Consume inputs (indexed by fnode_arg_index) until we have
228 // matched each element of dtypes (indexed by j).
229 for (size_t j = 0; j < dtypes.size(); ++fnode_arg_index) {
230 if (fnode_arg_index >= fnode.input_size()) {
231 // Should never happen if we computed dtypes correctly.
232 return errors::InvalidArgument(
233 "Attempt to access beyond input size: ", fnode_arg_index,
234 " >= ", fnode.input_size());
235 }
236 // Look up the next input.
237 const string& input_name = fnode.input(fnode_arg_index);
238 const auto* item = GetItemOrNull(input_name);
239 if (item == nullptr) {
240 return errors::InvalidArgument(
241 "input ", input_name, " is not found: ", SummarizeNodeDef(fnode));
242 }
243 if (item->dtypes.size() > dtypes.size() - j) {
244 return errors::InvalidArgument("Input ", input_name, " too long for ",
245 fnode_sig->input_arg(i).name());
246 }
247 // Match up all the elements of this input (indexed by k) with
248 // elements of dtypes (advancing j).
249 for (int k = 0; k < item->dtypes.size(); ++k, ++j) {
250 if (item->dtypes[k] != dtypes[j]) {
251 return errors::InvalidArgument(
252 "input ", fnode_sig->input_arg(i).name(), "[", j,
253 "] expected type ", DataTypeString(dtypes[j]),
254 " != ", DataTypeString(item->dtypes[k]), ", the type of ",
255 input_name, "[", k, "]");
256 }
257 if (item->is_func_arg) {
258 AddInput(gnode_idx, item->nid + k, 0);
259 } else {
260 AddInput(gnode_idx, item->nid, item->idx + k);
261 }
262 }
263 }
264 }
265
266 // Control deps.
267 for (int i = fnode_arg_index; i < fnode.input_size(); ++i) {
268 const string& input = fnode.input(i);
269 if (input.empty() || input[0] != '^') {
270 return errors::InvalidArgument("Expected input[", i, "] == '", input,
271 "' to be a control input.");
272 }
273 int nid = -1;
274 const string node_name = input.substr(1);
275 const string node_colon = node_name + ":";
276 const string node_colon_bound = node_name + ";";
277 // index_ is a map sorted lexicographically, so the key we are looking for
278 // must lie in the range [node_name, node_colon_bound).
279 auto it = index_.lower_bound(node_name);
280 while (it != index_.end() && it->first <= node_colon_bound) {
281 if (it->first == node_name ||
282 tensorflow::str_util::StartsWith(it->first, node_colon)) {
283 nid = it->second.nid;
284 break;
285 }
286 ++it;
287 }
288 if (nid == -1) {
289 return errors::InvalidArgument("input[", i, "] == '", input,
290 "', is not found.");
291 }
292 AddDep(gnode_idx, nid);
293 }
294
295 // Attrs.
296 for (const auto& p : attrs) {
297 (*gnode->mutable_attr())[p.first] = p.second;
298 }
299
300 return Status::OK();
301 }

303 Status AddReturnNode(
304 const OpDef::ArgDef& ret_def, AttrSlice attrs,
305 const ::tensorflow::protobuf::Map<string, string>& ret_map,
306 int* ret_index) {
307 auto ret_iter = ret_map.find(ret_def.name());
308 if (ret_iter == ret_map.end()) {
309 return errors::InvalidArgument("Return ", ret_def.name(), " missing.");
310 }
311 bool is_type_list;
312 DataTypeVector dtypes;
313 TF_RETURN_IF_ERROR(ArgNumType(attrs, ret_def, &is_type_list, &dtypes));
314 CHECK_GE(dtypes.size(), size_t{1});
315 const auto* item = GetItemOrNull(ret_iter->second);
316 if (item == nullptr) {
317 return errors::InvalidArgument("Return ", ret_def.name(), " -> ",
318 ret_iter->second, " is not found.");
319 }
320 if (dtypes != item->dtypes) {
321 return errors::InvalidArgument("Invalid ret types ", ret_def.name(),
322 " : ", DataTypeVectorString(dtypes),
323 " vs. ",
324 DataTypeVectorString(item->dtypes));
325 }
326 for (size_t i = 0; i < dtypes.size(); ++i) {
327 string name = strings::StrCat(ret_def.name(), "_RetVal");
328 if (dtypes.size() > 1) {
329 strings::StrAppend(&name, "_", i);
330 }
331 NodeDef* gnode = AddNode(name);
332 gnode->set_op(FunctionLibraryDefinition::kRetOp);
333 AddInput(nodes_.size() - 1, item->nid, item->idx + i);
334 AddAttr("T", dtypes[i], gnode);
335 AddAttr("index", (*ret_index)++, gnode);
336 result_.ret_types.push_back(dtypes[i]);
337 }
338 return Status::OK();
339 }

341 // Adds the actual node inputs to the result graph by converting indexes to
342 // the node names.
343 void AddNodeInputs() {
344 for (int i = 0; i < result_.nodes.size(); i++) {
345 NodeInfo& node_info = nodes_[i];
346 for (const auto& p : node_info.data_inputs) {
347 result_.nodes[i].add_input(Name(p.first, p.second));
348 }
349 for (int index : node_info.control_inputs) {
350 result_.nodes[i].add_input(Dep(index));
351 }
352 }
353 }

355 private:
356 // This is used to build a small index for all names that can be used as a
357 // node's input arguments.
358 //
359 // If is_func_arg is true, the name is a function's argument. In
360 // this case, the produced graph def has node[nid:nid + dtype.size()].
361 //
362 // Otherwise, the name is a function body's node return value. In
363 // this case, the produced graph def has one node node[nid] and
364 // the node's output index [idx ... idx + num) corresponds to the
365 // named outputs.
366 //
367 // In all cases, "dtype" specifies the data type.
368 struct NameInfoItem {
369 bool is_func_arg;
370 int nid;
371 int idx;
372 bool is_type_list;
373 DataTypeVector dtypes;
374 };
375
376 // Adds an item into the input name index.
377 Status AddItem(const string& name, const NameInfoItem& item) {
378 if (!index_.insert({name, item}).second) {
379 return errors::InvalidArgument(
380 strings::StrCat("Duplicated ", item.is_func_arg ? "arg" : "ret",
381 " name: "),
382 name);
383 }
384 return Status::OK();
385 }
386
387 const NameInfoItem* GetItemOrNull(const string& name) const {
388 return gtl::FindOrNull(index_, name);
389 }
390
391 string Dep(int node_index) const {
392 return strings::StrCat("^", Name(node_index));
393 }
394
395 string Name(int node_index) const {
396 CHECK_LT(node_index, nodes_.size());
397 return nodes_[node_index].name;
398 }
399
400 string Name(int node_index, int output_index) const {
401 if (output_index == 0) {
402 return Name(node_index);
403 } else {
404 return strings::StrCat(Name(node_index), ":", output_index);
405 }
406 }

408 NodeDef* AddNode(const string& name) {
409 result_.nodes.emplace_back();
410 NodeDef* gnode = &result_.nodes.back();
411 gnode->set_name(name);
412 nodes_.push_back({name, {}, {}});
413 CHECK_EQ(result_.nodes.size(), nodes_.size());
414 return gnode;
415 }

417 void AddInput(int node_index, int output_node, int output_index) {
418 CHECK_LT(node_index, nodes_.size());
419 nodes_[node_index].data_inputs.push_back(
420 std::make_pair(output_node, output_index));
421 }
422
423 void AddDep(int node_index, int dep_index) {
424 CHECK_LT(node_index, nodes_.size());
425 nodes_[node_index].control_inputs.push_back(dep_index);
426 }

432 // This contains information about a node in the new graph including the node

434 struct NodeInfo {
435 string name;
436 // Data inputs where <n, k> means arg k of node n.

442 std::vector<NodeInfo> nodes_;
443};

445// Various helpers Print(proto) to print relevant protos to ascii.
446string Print(const OpDef::ArgDef& arg) {
447 string out;
448 strings::StrAppend(&out, arg.name(), ":");
449 if (arg.is_ref()) strings::StrAppend(&out, "Ref(");
450 if (!arg.number_attr().empty()) {
451 strings::StrAppend(&out, arg.number_attr(), "*");
452 }
453 if (arg.type() != DT_INVALID) {
454 strings::StrAppend(&out, DataTypeString(arg.type()));
455 } else {
456 strings::StrAppend(&out, arg.type_attr());
457 }
458 if (arg.is_ref()) strings::StrAppend(&out, ")");
459 return out;
460}

462// TODO(josh11b): Merge this with SummarizeAttrValue().
463string Print(const AttrValue& attr_value) {
464 if (attr_value.value_case() == AttrValue::kType) {
465 return DataTypeString(attr_value.type());
466 } else if ((attr_value.value_case() == AttrValue::kList) &&
467 (attr_value.list().type_size() > 0)) {
468 string ret = "{";
469 for (int i = 0; i < attr_value.list().type_size(); ++i) {
470 if (i > 0) strings::StrAppend(&ret, ", ");
471 strings::StrAppend(&ret, DataTypeString(attr_value.list().type(i)));
472 }
473 strings::StrAppend(&ret, "}");
474 return ret;
475 } else if (attr_value.value_case() == AttrValue::kFunc) {
476 if (attr_value.func().attr_size() == 0) {
477 return attr_value.func().name();
478 }
479 std::vector<string> entries;
480 for (auto p : attr_value.func().attr()) {
481 entries.push_back(strings::StrCat(p.first, "=", Print(p.second)));
482 }
483 std::sort(entries.begin(), entries.end());
484 return strings::StrCat(attr_value.func().name(), "[",
485 str_util::Join(entries, ", "), "]");
486 }
487 return SummarizeAttrValue(attr_value);
488}

490// TODO(josh11b): Merge this with SummarizeNodeDef().
491string Print(const NodeDef& n) {
492 string out;
493 strings::StrAppend(&out, n.name(), " = ", n.op());
494 if (n.attr_size() > 0) {
495 std::vector<string> entries;
496 for (auto& a : n.attr()) {
497 entries.push_back(strings::StrCat(a.first, "=", Print(a.second)));
498 }
499 std::sort(entries.begin(), entries.end());
500 strings::StrAppend(&out, "[", str_util::Join(entries, ", "), "]");
501 }
502 strings::StrAppend(&out, "(");
503 std::vector<StringPiece> dat;
504 std::vector<string> dep;
505 for (StringPiece s : n.input()) {
506 if (str_util::ConsumePrefix(&s, "^")) {
507 dep.push_back(s.ToString());
508 } else {
509 dat.push_back(s);
510 }
511 }
512 strings::StrAppend(&out, str_util::Join(dat, ", "), ")");
513 if (!dep.empty()) {
514 strings::StrAppend(&out, " @ ", str_util::Join(dep, ", "));
515 }
516 return out;
517}

519string Print(const FunctionDef& fdef) {
520 string out;
521 const OpDef& sig = fdef.signature();
522 strings::StrAppend(&out, "\n", sig.name());
523 if (sig.attr_size() > 0) {
524 strings::StrAppend(&out, "[");
525 for (int i = 0; i < sig.attr_size(); ++i) {
526 const auto& a = sig.attr(i);
527 if (i > 0) strings::StrAppend(&out, ", ");
528 if (a.type() == "type") {
529 strings::StrAppend(&out, a.name(), ":", Print(a.allowed_values()));
530 } else {
531 strings::StrAppend(&out, a.name(), ":", a.type());
532 }
533 }
534 strings::StrAppend(&out, "]");
535 }
536 strings::StrAppend(&out, "(");
537 for (int i = 0; i < sig.input_arg_size(); ++i) {
538 if (i > 0) strings::StrAppend(&out, ", ");
539 strings::StrAppend(&out, Print(sig.input_arg(i)));
540 }
541 strings::StrAppend(&out, ") -> (");
542 for (int i = 0; i < sig.output_arg_size(); ++i) {
543 if (i > 0) strings::StrAppend(&out, ", ");
544 strings::StrAppend(&out, Print(sig.output_arg(i)));
545 }
546 strings::StrAppend(&out, ") {\n");
547 for (const auto& n : fdef.node_def()) {
548 strings::StrAppend(&out, " ", Print(n), "\n");
549 }
550 for (const auto& r : fdef.ret()) {
551 strings::StrAppend(&out, " return ", r.first, " = ", r.second, "\n");
552 }
553 strings::StrAppend(&out, "}\n");
554 return out;
555}

557string Print(gtl::ArraySlice<const NodeDef*> nodes) {
558 std::vector<const NodeDef*> arg;
559 std::vector<const NodeDef*> ret;
560 std::vector<const NodeDef*> body;
561 for (const NodeDef* n : nodes) {
562 if (n->op() == FunctionLibraryDefinition::kArgOp) {
563 arg.push_back(n);
564 } else if (n->op() == FunctionLibraryDefinition::kRetOp) {
565 ret.push_back(n);
566 } else {
567 body.push_back(n);
568 }
569 }
570 auto comp = [](const NodeDef* x, const NodeDef* y) {
571 int xi;
572 TF_CHECK_OK(GetNodeAttr(*x, "index", &xi));
573 int yi;
574 TF_CHECK_OK(GetNodeAttr(*y, "index", &yi));
575 return xi < yi;
576 };
577 std::sort(arg.begin(), arg.end(), comp);
578 std::sort(ret.begin(), ret.end(), comp);
579 string out;
580 strings::StrAppend(&out, "\n(");
581 auto get_type = [](const NodeDef& n) {
582 DataType dt;
583 if (!GetNodeAttr(n, "T", &dt).ok()) {
584 dt = DT_INVALID;
585 }
586 return DataTypeString(dt);
587 };
588 for (size_t i = 0; i < arg.size(); ++i) {
589 const NodeDef* n = arg[i];
590 if (i > 0) strings::StrAppend(&out, ", ");
591 CHECK_GE(n->attr_size(), 2);
592 strings::StrAppend(&out, n->name(), ":", get_type(*n));
593 }
594 strings::StrAppend(&out, ") -> (");
595 for (size_t i = 0; i < ret.size(); ++i) {
596 const NodeDef* n = ret[i];
597 if (i > 0) strings::StrAppend(&out, ", ");
598 CHECK_LE(2, n->attr_size());
599 CHECK_EQ(1, n->input_size());
600 strings::StrAppend(&out, n->input(0), ":", get_type(*n));
601 }
602 strings::StrAppend(&out, ") {\n");
603 for (size_t i = 0; i < body.size(); ++i) {
604 strings::StrAppend(&out, " ", Print(*body[i]), "\n");
605 }
606 strings::StrAppend(&out, "}\n");
607 return out;
608}

610Status AddDefaultAttrs(const string& op,
611 const GetFunctionSignature& get_function,
612 AttrValueMap* attrs) {
613 const OpDef* op_def = nullptr;
614 TF_RETURN_IF_ERROR(get_function(op, &op_def));
615 AttrSlice attr_slice(attrs);
616 for (const auto& attr_def : op_def->attr()) {
617 if (attr_def.has_default_value() && !attr_slice.Find(attr_def.name())) {
618 if (!attrs->insert({attr_def.name(), attr_def.default_value()}).second) {
619 return errors::Internal("Somehow duplicated: ", attr_def.name());
620 }
621 }
622 }
623 return Status::OK();
624}

628Status InstantiateFunction(const FunctionDef& fdef, AttrSlice attr_values,
629 GetFunctionSignature get_function,
630 InstantiationResult* result) {
631 VLOG(3) << "Instantiation Function: " << Print(fdef);
632
633 const OpDef& sig = fdef.signature();
634 TF_RETURN_IF_ERROR(ValidateSignatureWithAttrs(sig, attr_values));
635
636 FunctionInstantiationHelper helper(get_function, result);
637 Status s;
638 for (const OpDef::ArgDef& arg_def : sig.input_arg()) {
639 s = helper.BuildInputArgIndex(arg_def, attr_values);
640 if (!s.ok()) {
641 errors::AppendToMessage(&s, "In ", Print(arg_def));
642 return s;
643 }
644 }
645
646 auto substitute = [attr_values](StringPiece name, AttrValue* val) {
647 if (const AttrValue* v = attr_values.Find(name)) {
648 *val = *v;
649 return true;
650 }
651 return false;
652 };
653
654 // Makes a copy of all attrs in fdef and substitutes placeholders.
655 // After this step, every attr is bound to a concrete value.
656 std::vector<AttrValueMap> node_attrs;
657 node_attrs.resize(fdef.node_def_size());
658 for (int i = 0; i < fdef.node_def_size(); ++i) {
659 for (auto attr : fdef.node_def(i).attr()) {
660 if (!SubstitutePlaceholders(substitute, &attr.second)) {
661 return errors::InvalidArgument("Failed to bind all placeholders in ",
662 SummarizeAttrValue(attr.second));
663 }
664 if (!node_attrs[i].insert(attr).second) {
665 return errors::Internal("Somehow duplicated: ", attr.first);
666 }
667 }
668 TF_RETURN_IF_ERROR(
669 AddDefaultAttrs(fdef.node_def(i).op(), get_function, &node_attrs[i]));
670 }
671
672 for (int i = 0; i < fdef.node_def_size(); ++i) {
673 s = helper.BuildNodeOutputIndex(fdef.node_def(i), AttrSlice(&node_attrs[i]),
674 result->nodes.size() + i);
675 if (!s.ok()) {
676 errors::AppendToMessage(&s, "In ", SummarizeNodeDef(fdef.node_def(i)));
677 return s;
678 }
679 }
680 // Emits one node for each fdef.node_def.
681 for (int i = 0; i < fdef.node_def_size(); ++i) {
682 s = helper.InstantiateNode(fdef.node_def(i), AttrSlice(&node_attrs[i]));
683 if (!s.ok()) {
684 errors::AppendToMessage(&s, "In ", SummarizeNodeDef(fdef.node_def(i)));
685 return s;
686 }
687 }
688
689 // Emits nodes for the function's return values.
690 int ret_index = 0;
691 for (const OpDef::ArgDef& ret_def : sig.output_arg()) {
692 s = helper.AddReturnNode(ret_def, attr_values, fdef.ret(), &ret_index);
693 if (!s.ok()) {
694 errors::AppendToMessage(&s, "In function output ", Print(ret_def));
695 return s;
696 }
697 }
698
699 // Adds the actual node inputs using the input indexes.
700 helper.AddNodeInputs();
701
702 return Status::OK();
703}

737// Returns the name -> attr mapping of fdef's attrs that have a value set. In
738// Python, it's possible to access unset attrs, which returns a default value
739// and adds an unset attr to the map.
740std::map<string, AttrValue> GetSetAttrs(const FunctionDef& fdef) {
741 std::map<string, AttrValue> set_attrs;
742 for (auto pair : fdef.attr()) {
743 if (pair.second.value_case() != AttrValue::VALUE_NOT_SET) {
744 set_attrs[pair.first] = pair.second;
745 }
746 }
747 return set_attrs;

752bool FunctionDefsEqual(const FunctionDef& f1, const FunctionDef& f2) {
753 if (!OpDefEqual(f1.signature(), f2.signature())) return false;
754
755 std::map<string, AttrValue> f1_attrs = GetSetAttrs(f1);
756 std::map<string, AttrValue> f2_attrs = GetSetAttrs(f2);
757 if (f1_attrs.size() != f2_attrs.size()) return false;
758 for (auto iter1 : f1_attrs) {
759 auto iter2 = f2_attrs.find(iter1.first);
760 if (iter2 == f2_attrs.end()) return false;
761 if (!AreAttrValuesEqual(iter1.second, iter2->second)) return false;
762 }
763
764 if (!EqualRepeatedNodeDef(f1.node_def(), f2.node_def(), nullptr)) {
765 return false;
766 }
767
768 std::map<string, string> ret1(f1.ret().begin(), f1.ret().end());
769 std::map<string, string> ret2(f2.ret().begin(), f2.ret().end());
770 if (ret1 != ret2) return false;
771
772 return true;
773}

775uint64 FunctionDefHash(const FunctionDef& fdef) {
776 // signature
777 uint64 h = OpDefHash(fdef.signature());
778
779 // attrs
780 std::map<string, AttrValue> attrs = GetSetAttrs(fdef);
781 for (const auto& p : attrs) {
782 h = Hash64(p.first.data(), p.first.size(), h);
783 h = Hash64Combine(AttrValueHash(p.second), h);
784 }
785
786 // node defs
787 h = Hash64Combine(RepeatedNodeDefHash(fdef.node_def()), h);
788
789 // output names
790 std::map<string, string> ret(fdef.ret().begin(), fdef.ret().end());
791 for (const auto& p : ret) {
792 h = Hash64(p.first.data(), p.first.size(), h);
793 h = Hash64(p.second.data(), p.second.size(), h);
794 }
795
796 return h;
797}

799string Canonicalize(const string& funcname, AttrSlice attrs,
800 const FunctionLibraryRuntime::InstantiateOptions& options) {
801 std::vector<string> entries;
802 entries.reserve(options.target.empty() ? attrs.size() : (attrs.size() + 1));
803 for (auto p : attrs) {
804 entries.push_back(strings::StrCat(p.first, "=", Print(p.second)));
805 }
806 if (!options.target.empty()) {
807 entries.push_back(
808 strings::StrCat("_target", "=", str_util::CEscape(options.target)));
809 }
810 if (options.overlay_lib) {
811 entries.push_back(strings::StrCat(

813 }
814 if (!options.state_handle.empty()) {
815 entries.push_back(
816 strings::StrCat("_state_handle", "=", options.state_handle));
817 }
818 std::sort(entries.begin(), entries.end());
819 return strings::StrCat(funcname, "[", str_util::Join(entries, ","), "]");
820}

953Status FunctionLibraryDefinition::AddFunctionDefHelper(const FunctionDef& fdef,
954 bool* added) {
955 *added = false;
956 std::unique_ptr<FunctionDefAndOpRegistration>* entry =
957 &function_defs_[fdef.signature().name()];
958 if (*entry != nullptr) {
959 if (!FunctionDefsEqual((*entry)->fdef, fdef)) {
960 return errors::InvalidArgument(
961 "Cannot add function '", fdef.signature().name(),
962 "' because a different function with the same name already "
963 "exists.");
964 }
965 // Ignore duplicate FunctionDefs
966 return Status::OK();
967 }
968 const OpDef* op_def;
969 if (default_registry_->LookUpOpDef(fdef.signature().name(), &op_def).ok()) {
970 return errors::InvalidArgument(
971 "Cannot add function '", fdef.signature().name(),
972 "' because an op with the same name already exists.");
973 }
974 entry->reset(new FunctionDefAndOpRegistration(fdef));
975 *added = true;
976 return Status::OK();
977}

1003Status FunctionLibraryDefinition::AddLibrary(
1004 const FunctionLibraryDefinition& other) {
1005 // Remember the funcs and grads that we added successfully so that
1006 // we can roll them back on error.
1007 std::vector<string> funcs;
1008 std::vector<string> funcs_with_grads;
1009 Status s;
1010 bool added;
1011 for (auto iter : other.function_defs_) {
1012 s = AddFunctionDefHelper(iter.second->fdef, &added);
1013 if (!s.ok()) {
1014 Remove(funcs, funcs_with_grads);
1015 return s;
1016 }
1017 if (added) {
1018 funcs.push_back(iter.second->fdef.signature().name());
1019 }
1020 }
1021 for (auto iter : other.func_grad_) {
1022 GradientDef grad;
1023 grad.set_function_name(iter.first);
1024 grad.set_gradient_func(iter.second);
1025 s = AddGradientDefHelper(grad, &added);
1026 if (!s.ok()) {
1027 Remove(funcs, funcs_with_grads);
1028 return s;
1029 }
1030 if (added) {
1031 funcs_with_grads.push_back(grad.function_name());
1032 }
1033 }
1034 return Status::OK();
1035}

1115const FunctionDef* FunctionLibraryDefinition::GetAttrImpl(
1116 const NodeDef& ndef) const {
1117 if (ndef.op() != kGradientOp) {
1118 // If 'ndef' calls a function and the function's def has the attr,
1119 // returns it.
1120 return Find(ndef.op());
1121 }
1122
1123 // If ndef is SymbolicGradient[f=Foo], we use Foo's gradient or
1124 // Foo's attributes.
1125 const NameAttrList* forward_func_attrs;
1126 if (!GetNodeAttr(ndef, kFuncAttr, &forward_func_attrs).ok()) {
1127 return nullptr;
1128 }
1129 const string& func_name = forward_func_attrs->name();
1130 const string& grad_name = FindGradient(func_name);
1131 // If 'func' has a user-defined gradient function, uses the grad
1132 // function's attrs to see if noinline is specified. Otherwise,
1133 // uses func's attrs.
1134 if (!grad_name.empty()) {
1135 return Find(grad_name);
1136 }
1137 return Find(func_name);
1138}

1213/* static */
1214FunctionDef FunctionDefHelper::Create(
1215 const string& function_name, gtl::ArraySlice<string> in_def,
1216 gtl::ArraySlice<string> out_def, gtl::ArraySlice<string> attr_def,
1217 gtl::ArraySlice<Node> node_def,
1218 gtl::ArraySlice<std::pair<string, string>> ret_def) {
1219 FunctionDef fdef;
1220
1221 // Signature
1222 OpDefBuilder b(function_name);
1223 for (const auto& i : in_def) b.Input(i);
1224 for (const auto& o : out_def) b.Output(o);
1225 for (const auto& a : attr_def) b.Attr(a);
1226
1227 OpRegistrationData op_reg_data;
1228 TF_CHECK_OK(b.Finalize(&op_reg_data));
1229 fdef.mutable_signature()->Swap(&op_reg_data.op_def);
1230
1231 // Function body
1232 for (const auto& n : node_def) {
1233 *(fdef.add_node_def()) = n.ToNodeDef();
1234 }
1235
1236 // Returns
1237 for (const auto& r : ret_def) {
1238 fdef.mutable_ret()->insert({r.first, r.second});
1239 }
1240 return fdef;
1241}

1243/* static */
1244FunctionDef FunctionDefHelper::Define(const string& name,
1245 gtl::ArraySlice<string> arg_def,
1246 gtl::ArraySlice<string> ret_def,
1247 gtl::ArraySlice<string> attr_def,
1248 gtl::ArraySlice<Node> node_def) {
1249 FunctionDef fdef;
1250 OpDefBuilder b(name);
1251 for (const auto& a : arg_def) b.Input(a);
1252 for (const auto& r : ret_def) b.Output(r);
1253 for (const auto& a : attr_def) b.Attr(a);
1254
1255 OpRegistrationData op_reg_data;
1256 TF_CHECK_OK(b.Finalize(&op_reg_data));
1257 fdef.mutable_signature()->Swap(&op_reg_data.op_def);
1258
1259 // Mapping from legacy output names to NodeDef outputs.
1260 std::unordered_map<string, string> ret_index;
1261 for (const auto& a : fdef.signature().input_arg()) {
1262 ret_index[a.name()] = a.name();
1263 }
1264
1265 // For looking up OpDefs
1266 auto* op_def_registry = OpRegistry::Global();
1267
1268 // Function body
1269 for (const auto& src : node_def) {
1270 NodeDef* n = fdef.add_node_def();
1271 n->set_op(src.op);
1272 n->set_name(src.ret[0]);
1273 for (const auto& a : src.attr) {
1274 n->mutable_attr()->insert({a.first, a.second.proto});
1275 }
1276 for (const string& a : src.arg) {
1277 const auto iter = ret_index.find(a);
1278 CHECK(iter != ret_index.end())
1279 << "Node input '" << a << "' in '" << src.ret[0] << "' of " << name;
1280 n->add_input(iter->second);
1281 }
1282 for (const string& d : src.dep) {
1283 n->add_input(strings::StrCat("^", d));
1284 }
1285
1286 // Add the outputs of this node to ret_index.
1287 const OpDef* op_def = nullptr;
1288 TF_CHECK_OK(op_def_registry->LookUpOpDef(n->op(), &op_def)) << n->op();
1289 CHECK(op_def != nullptr) << n->op();
1290 NameRangeMap output_names;
1291 TF_CHECK_OK(NameRangesForNode(*n, *op_def, nullptr, &output_names));
1292 for (const auto& o : output_names) {
1293 CHECK_LE(o.second.second, src.ret.size())
1294 << "Missing ret for output '" << o.first << "' in '" << src.ret[0]
1295 << "' of " << name;
1296 for (int i = o.second.first; i < o.second.second; ++i) {
1297 ret_index[src.ret[i]] =
1298 strings::StrCat(src.ret[0], ":", o.first, ":", i - o.second.first);
1299 }
1300 }
1301 }
1302
1303 // Returns
1304 for (const auto& r : fdef.signature().output_arg()) {
1305 const auto iter = ret_index.find(r.name());
1306 CHECK(iter != ret_index.end()) << "Return '" << r.name() << "' in " << name;
1307 fdef.mutable_ret()->insert({r.name(), iter->second});
1308 }
1309 return fdef;
1310}
