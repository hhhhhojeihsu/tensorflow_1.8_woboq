28void AllocatorStats::Clear() {
29 this->num_allocs = 0;
30 this->bytes_in_use = 0;
31 this->max_bytes_in_use = 0;
32 this->max_alloc_size = 0;
33 this->bytes_limit = 0;
34}

36string AllocatorStats::DebugString() const {
37 return strings::Printf(
38 "Limit: %20lld\n"
39 "InUse: %20lld\n"
40 "MaxInUse: %20lld\n"
41 "NumAllocs: %20lld\n"
42 "MaxAllocSize: %20lld\n",
43 this->bytes_limit, this->bytes_in_use, this->max_bytes_in_use,
44 this->num_allocs, this->max_alloc_size);
45}

51void RunResourceCtor(ResourceHandle* p, size_t n) {

59// If true, cpu allocator collects more stats.
60static bool cpu_allocator_collect_stats = false;
61// If true, cpu allocator collects full stats.
62static bool cpu_allocator_collect_full_stats = false;
63
64// Individual allocations large than this amount will trigger a warning.
65static const double kLargeAllocationWarningThreshold = 0.1;
66
67// If cpu_allocator_collect_stats is true, warn when the total allocated memory
68// exceeds this threshold.
69static const double kTotalAllocationWarningThreshold = 0.5;
70
71// Cache first invocation to port::AvailableRam, as it can be expensive.
72static int64_t LargeAllocationWarningBytes() {
73 static int64_t value = static_cast<int64>(port::AvailableRam() *
74 kLargeAllocationWarningThreshold);
75 return value;
76}
77
78static int64_t TotalAllocationWarningBytes() {
79 static int64_t value = static_cast<int64>(port::AvailableRam() *
80 kTotalAllocationWarningThreshold);
81 return value;
82}
83
84void EnableCPUAllocatorStats(bool enable) {
85 cpu_allocator_collect_stats = enable;
86}
87void EnableCPUAllocatorFullStats(bool enable) {
88 cpu_allocator_collect_full_stats = enable;
89}

91class CPUAllocator : public Allocator {
92 public:
93 CPUAllocator() : total_allocation_warning_triggered_(false) {}
94
95 ~CPUAllocator() override {}
96
97 string Name() override { return "cpu"; }
98
99 void* AllocateRaw(size_t alignment, size_t num_bytes) override {
100 if (num_bytes > LargeAllocationWarningBytes()) {
101 LOG(WARNING) << "Allocation of " << num_bytes << " exceeds "
102 << 100 * kLargeAllocationWarningThreshold
103 << "% of system memory.";
104 }
105
106 void* p = port::AlignedMalloc(num_bytes, alignment);
107 if (cpu_allocator_collect_stats) {
108 const std::size_t alloc_size = port::MallocExtension_GetAllocatedSize(p);
109 mutex_lock l(mu_);
110 ++stats_.num_allocs;
111 stats_.bytes_in_use += alloc_size;
112 stats_.max_bytes_in_use =
113 std::max<int64>(stats_.max_bytes_in_use, stats_.bytes_in_use);
114 stats_.max_alloc_size =
115 std::max<int64>(stats_.max_alloc_size, alloc_size);
116
117 if (stats_.bytes_in_use > TotalAllocationWarningBytes() &&
118 !total_allocation_warning_triggered_) {
119 LOG(WARNING) << "Total allocated memory " << stats_.bytes_in_use
120 << "exceeds " << 100 * kTotalAllocationWarningThreshold
121 << "% of system memory";
122 total_allocation_warning_triggered_ = true;
123 }
124 }
125 return p;
126 }
127
128 void DeallocateRaw(void* ptr) override {
129 if (cpu_allocator_collect_stats) {
130 const std::size_t alloc_size =
131 port::MallocExtension_GetAllocatedSize(ptr);
132 mutex_lock l(mu_);
133 stats_.bytes_in_use -= alloc_size;
134 }
135 port::AlignedFree(ptr);
136 }
137
138 void GetStats(AllocatorStats* stats) override {
139 mutex_lock l(mu_);
140 *stats = stats_;
141 }
142
143 void ClearStats() override {
144 mutex_lock l(mu_);
145 stats_.num_allocs = 0;
146 stats_.max_bytes_in_use = stats_.bytes_in_use;
147 stats_.max_alloc_size = 0;
148 }
149
150 size_t AllocatedSizeSlow(const void* ptr) override {
151 return port::MallocExtension_GetAllocatedSize(ptr);
152 }
153
154 private:
155 mutex mu_;
156 AllocatorStats stats_ GUARDED_BY(mu_);
157 bool total_allocation_warning_triggered_ GUARDED_BY(mu_);
158
159 TF_DISALLOW_COPY_AND_ASSIGN(CPUAllocator);
160};
161
162Allocator* cpu_allocator() {
163 static Allocator* cpu_alloc = AllocatorRegistry::Global()->GetAllocator();
164 if (cpu_allocator_collect_full_stats && !cpu_alloc->TracksAllocationSizes()) {
165 cpu_alloc = new TrackingAllocator(cpu_alloc, true);
166 }
167 return cpu_alloc;
168}

170REGISTER_MEM_ALLOCATOR("DefaultCPUAllocator", 100, CPUAllocator);
