37Status OpRegistryInterface::LookUpOpDef(const string& op_type_name,
38 const OpDef** op_def) const {
39 *op_def = nullptr;
40 const OpRegistrationData* op_reg_data = nullptr;
41 TF_RETURN_IF_ERROR(LookUp(op_type_name, &op_reg_data));
42 *op_def = &op_reg_data->op_def;
43 return Status::OK();
44}

52void OpRegistry::Register(const OpRegistrationDataFactory& op_data_factory) {
53 mutex_lock lock(mu_);
54 if (initialized_) {
55 TF_QCHECK_OK(RegisterAlreadyLocked(op_data_factory));
56 } else {
57 deferred_.push_back(op_data_factory);
58 }
59}
60
61Status OpRegistry::LookUp(const string& op_type_name,
62 const OpRegistrationData** op_reg_data) const {
63 *op_reg_data = nullptr;
64 const OpRegistrationData* res = nullptr;
65
66 bool first_call = false;
67 bool first_unregistered = false;
68 { // Scope for lock.
69 mutex_lock lock(mu_);
70 first_call = MustCallDeferred();
71 res = gtl::FindWithDefault(registry_, op_type_name, nullptr);
72
73 static bool unregistered_before = false;
74 first_unregistered = !unregistered_before && (res == nullptr);
75 if (first_unregistered) {
76 unregistered_before = true;
77 }
78 // Note: Can't hold mu_ while calling Export() below.
79 }
80 if (first_call) {
81 TF_QCHECK_OK(ValidateKernelRegistrations(*this));
82 }
83 if (res == nullptr) {
84 if (first_unregistered) {
85 OpList op_list;
86 Export(true, &op_list);
87 if (VLOG_IS_ON(3)) {
88 LOG(INFO) << "All registered Ops:";
89 for (const auto& op : op_list.op()) {
90 LOG(INFO) << SummarizeOpDef(op);
91 }
92 }
93 }
94 Status status =
95 errors::NotFound("Op type not registered '", op_type_name,
96 "' in binary running on ", port::Hostname(), ". ",
97 "Make sure the Op and Kernel are registered in the "
98 "binary running in this process.");
99 VLOG(1) << status.ToString();
100 return status;
101 }
102 *op_reg_data = res;
103 return Status::OK();
104}

106void OpRegistry::GetRegisteredOps(std::vector<OpDef>* op_defs) {
107 mutex_lock lock(mu_);
108 MustCallDeferred();
109 for (const auto& p : registry_) {
110 op_defs->push_back(p.second->op_def);
111 }
112}

114void OpRegistry::GetOpRegistrationData(
115 std::vector<OpRegistrationData>* op_data) {
116 mutex_lock lock(mu_);
117 MustCallDeferred();
118 for (const auto& p : registry_) {
119 op_data->push_back(*p.second);
120 }
121}

123Status OpRegistry::SetWatcher(const Watcher& watcher) {
124 mutex_lock lock(mu_);
125 if (watcher_ && watcher) {
126 return errors::AlreadyExists(
127 "Cannot over-write a valid watcher with another.");
128 }
129 watcher_ = watcher;
130 return Status::OK();
131}

133void OpRegistry::Export(bool include_internal, OpList* ops) const {
134 mutex_lock lock(mu_);
135 MustCallDeferred();
136
137 std::vector<std::pair<string, const OpRegistrationData*>> sorted(
138 registry_.begin(), registry_.end());
139 std::sort(sorted.begin(), sorted.end());
140
141 auto out = ops->mutable_op();
142 out->Clear();
143 out->Reserve(sorted.size());
144
145 for (const auto& item : sorted) {
146 if (include_internal || !str_util::StartsWith(item.first, "_")) {
147 *out->Add() = item.second->op_def;
148 }
149 }
150}

162Status OpRegistry::ProcessRegistrations() const {

167string OpRegistry::DebugString(bool include_internal) const {
168 OpList op_list;
169 Export(include_internal, &op_list);
170 string ret;
171 for (const auto& op : op_list.op()) {
172 strings::StrAppend(&ret, SummarizeOpDef(op), "\n");
173 }
174 return ret;
175}

177bool OpRegistry::MustCallDeferred() const {
178 if (initialized_) return false;
179 initialized_ = true;
180 for (size_t i = 0; i < deferred_.size(); ++i) {
181 TF_QCHECK_OK(RegisterAlreadyLocked(deferred_[i]));
182 }
183 deferred_.clear();
184 return true;
185}
186
187Status OpRegistry::CallDeferred() const {
188 if (initialized_) return Status::OK();
189 initialized_ = true;
190 for (size_t i = 0; i < deferred_.size(); ++i) {
191 Status s = RegisterAlreadyLocked(deferred_[i]);
192 if (!s.ok()) {
193 return s;
194 }
195 }
196 deferred_.clear();
197 return Status::OK();
198}
199
200Status OpRegistry::RegisterAlreadyLocked(
201 const OpRegistrationDataFactory& op_data_factory) const {
202 std::unique_ptr<OpRegistrationData> op_reg_data(new OpRegistrationData);
203 Status s = op_data_factory(op_reg_data.get());
204 if (s.ok()) {
205 s = ValidateOpDef(op_reg_data->op_def);
206 if (s.ok() &&
207 !gtl::InsertIfNotPresent(&registry_, op_reg_data->op_def.name(),
208 op_reg_data.get())) {
209 s = errors::AlreadyExists("Op with name ", op_reg_data->op_def.name());
210 }
211 }
212 Status watcher_status = s;
213 if (watcher_) {
214 watcher_status = watcher_(s, op_reg_data->op_def);
215 }
216 if (s.ok()) {
217 op_reg_data.release();
218 } else {
219 op_reg_data.reset();
220 }
221 return watcher_status;
222}

232OpListOpRegistry::OpListOpRegistry(const OpList* op_list) {
233 for (const OpDef& op_def : op_list->op()) {
234 auto* op_reg_data = new OpRegistrationData();
235 op_reg_data->op_def = op_def;
236 index_[op_def.name()] = op_reg_data;
237 }
238}
239
240OpListOpRegistry::~OpListOpRegistry() {
241 for (const auto& e : index_) delete e.second;
242}
243
244Status OpListOpRegistry::LookUp(const string& op_type_name,
245 const OpRegistrationData** op_reg_data) const {
246 auto iter = index_.find(op_type_name);
247 if (iter == index_.end()) {
248 *op_reg_data = nullptr;
249 return errors::NotFound("Op type not registered '", op_type_name,
250 "' in binary running on ", port::Hostname(), ". ",
251 "Make sure the Op and Kernel are registered in the "
252 "binary running in this process.");
253 }
254 *op_reg_data = iter->second;
255 return Status::OK();
256}
