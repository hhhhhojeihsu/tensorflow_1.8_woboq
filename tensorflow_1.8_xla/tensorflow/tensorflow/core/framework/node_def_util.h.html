113class AttrSlice {
114 public:
115 AttrSlice(const NodeDef& node_def); // NOLINT(runtime/explicit)
116
117 AttrSlice(); // Empty
118 explicit AttrSlice(const AttrValueMap* a);
119
120 int size() const { return attrs_->size(); }
121
122 // Returns the attr with attr_name if found. Otherwise, returns
123 // nullptr.
124 const AttrValue* Find(StringPiece attr_name) const;
125
126 // Returns the attr_value for attr_name if found. Otherwise, returns a
127 // NotFound status.
128 Status Find(StringPiece attr_name, const AttrValue** attr_value) const;
129
130 // Helper class to avoid allocations in EqualAttrs.
131 // TODO(irving): Will go away once NodeInfo is used.
132 struct Scratch {
133 string a;
134 string b;
135 };
136
137 // Check if all attrs and attr values match. Does not take defaults into
138 // account.
139 //
140 // TODO(irving): There is a bug in this routine inherited from its
141 // OptimizerCSE::EqualAttrs precedecessor. The same tensor attr can be
142 // represented in more than one way as an AttrValue, since TensorProto is
143 // not 1-1. This bug will go away once I replace everything with NodeInfo,
144 // which stores a Tensor object directly. The Scratch object will also go
145 // away.
146 bool EqualAttrs(AttrSlice other, Scratch* scratch) const;
147
148 // If this AttrSlice has an attached NodeDef, summarize it. This is for
149 // error messages only: we intentionally do not provide direct access to the
150 // NodeDef, since it is not always there.
151 string SummarizeNode() const;
152
153 // Iteration over all attrs
154 AttrValueMap::const_iterator begin() const { return attrs_->begin(); }
155 AttrValueMap::const_iterator end() const { return attrs_->end(); }
156
157 private:
158 const NodeDef* ndef_;
159 const AttrValueMap* attrs_;
160};

262// Computes the mapping from input/output argument name to the
263// corresponding input/output index range. For example,
264// input "foo" corresponds to input indices
265// [ (*inputs)["foo"].first, (*inputs)["foo"].second ).
266// NOTE(mrry): To reduce allocations when the map is used and save
267// space, the returned `NameRangeMap` objects borrow the input/output
268// argument names from `op_def`. The `op_def` must outlive the
269// returned `NameRangeMap` objects.
270typedef gtl::FlatMap<StringPiece, std::pair<int, int>, hash<StringPiece>>
271 NameRangeMap;
272Status NameRangesForNode(const NodeDef& node_def, const OpDef& op_def,
273 NameRangeMap* inputs, NameRangeMap* outputs);
274Status NameRangesForNode(const Node& node, const OpDef& op_def,
275 NameRangeMap* inputs, NameRangeMap* outputs);

280// Validates the syntax of a NodeDef provided externally.
281//
282// The following is an EBNF-style syntax for NodeDef objects. Note that
283// Node objects are actually specified as tensorflow::NodeDef protocol buffers,
284// which contain many other fields that are not (currently) validated.
285//
286// Node = NodeName, Inputs
287// Inputs = ( DataInput * ), ( ControlInput * )
288// DataInput = NodeName, ( ":", [1-9], [0-9] * ) ?
289// ControlInput = "^", NodeName
290// NodeName = [A-Za-z0-9.], [A-Za-z0-9_./] *
291Status ValidateExternalNodeDefSyntax(const NodeDef& node_def);
