60bool ConsumeQuotedString(char quote_ch, StringPiece* sp, StringPiece* out) {
61 const string quote_str(1, quote_ch);
62 return Scanner(*sp)
63 .OneLiteral(quote_str.c_str())
64 .RestartCapture()
65 .ScanEscapedUntil(quote_ch)
66 .StopCapture()
67 .OneLiteral(quote_str.c_str())
68 .AnySpace()
69 .GetResult(sp, out);

80bool ConsumeAttrNumber(StringPiece* sp, int64* out) {
81 Scanner scan(*sp);
82 StringPiece match;
83 StringPiece remaining;
84
85 scan.AnySpace().RestartCapture();
86 if (scan.Peek() == '-') {
87 scan.OneLiteral("-");
88 }
89 if (!scan.Many(Scanner::DIGIT)
90 .StopCapture()
91 .AnySpace()
92 .GetResult(&remaining, &match)) {
93 return false;
94 }
95 int64 value = 0;
96 if (!strings::safe_strto64(match, &value)) {
97 return false;
98 }
99 *out = value;
100 *sp = remaining;
101 return true;
102}

113bool ConsumeCompoundAttrType(StringPiece* sp, StringPiece* out) {
114 auto capture_begin = sp->begin();
115 if (str_util::ConsumePrefix(sp, "numbertype") ||
116 str_util::ConsumePrefix(sp, "numerictype") ||
117 str_util::ConsumePrefix(sp, "quantizedtype") ||
118 str_util::ConsumePrefix(sp, "realnumbertype") ||
119 str_util::ConsumePrefix(sp, "realnumberictype")) {
120 *out = StringPiece(capture_begin, sp->begin() - capture_begin);
121 return true;
122 }
123 return false;
124}

126bool ProcessCompoundType(const StringPiece type_string, AttrValue* allowed) {
127 if (type_string == "numbertype" || type_string == "numerictype") {
128 for (DataType dt : NumberTypes()) {
129 allowed->mutable_list()->add_type(dt);
130 }
131 } else if (type_string == "quantizedtype") {
132 for (DataType dt : QuantizedTypes()) {
133 allowed->mutable_list()->add_type(dt);
134 }
135 } else if (type_string == "realnumbertype" ||
136 type_string == "realnumerictype") {
137 for (DataType dt : RealNumberTypes()) {
138 allowed->mutable_list()->add_type(dt);
139 }
140 } else {
141 return false;
142 }
143 return true;
144}

146void FinalizeAttr(StringPiece spec, OpDef* op_def,
147 std::vector<string>* errors) {
148 OpDef::AttrDef* attr = op_def->add_attr();
149 StringPiece orig(spec);
150
151 // Parse "<name>:" at the beginning.
152 StringPiece tmp_name;
153 VERIFY(ConsumeAttrName(&spec, &tmp_name), "Trouble parsing '<name>:'");
154 attr->set_name(tmp_name.data(), tmp_name.size());
155
156 // Read "<type>" or "list(<type>)".
157 bool is_list = ConsumeListPrefix(&spec);
158 string type;
159 StringPiece type_string; // Used if type == "type"
160 if (str_util::ConsumePrefix(&spec, "string")) {
161 type = "string";
162 } else if (str_util::ConsumePrefix(&spec, "int")) {
163 type = "int";
164 } else if (str_util::ConsumePrefix(&spec, "float")) {
165 type = "float";
166 } else if (str_util::ConsumePrefix(&spec, "bool")) {
167 type = "bool";
168 } else if (str_util::ConsumePrefix(&spec, "type")) {
169 type = "type";
170 } else if (str_util::ConsumePrefix(&spec, "shape")) {
171 type = "shape";
172 } else if (str_util::ConsumePrefix(&spec, "tensor")) {
173 type = "tensor";
174 } else if (str_util::ConsumePrefix(&spec, "func")) {
175 type = "func";
176 } else if (ConsumeCompoundAttrType(&spec, &type_string)) {
177 type = "type";
178 AttrValue* allowed = attr->mutable_allowed_values();
179 VERIFY(ProcessCompoundType(type_string, allowed),
180 "Expected to see a compound type, saw: ", type_string);
181 } else if (str_util::ConsumePrefix(&spec, "{")) {
182 // e.g. "{ int32, float, bool }" or "{ \"foo\", \"bar\" }"
183 AttrValue* allowed = attr->mutable_allowed_values();
184 str_util::RemoveLeadingWhitespace(&spec);
185 if (str_util::StartsWith(spec, "\"") || str_util::StartsWith(spec, "'")) {
186 type = "string"; // "{ \"foo\", \"bar\" }" or "{ 'foo', 'bar' }"
187 while (true) {
188 StringPiece escaped_string;
189 VERIFY(ConsumeQuotedString('"', &spec, &escaped_string) ||
190 ConsumeQuotedString('\'', &spec, &escaped_string),
191 "Trouble parsing allowed string at '", spec, "'");
192 string unescaped;
193 string error;
194 VERIFY(str_util::CUnescape(escaped_string, &unescaped, &error),
195 "Trouble unescaping \"", escaped_string,
196 "\", got error: ", error);
197 allowed->mutable_list()->add_s(unescaped);
198 if (str_util::ConsumePrefix(&spec, ",")) {
199 str_util::RemoveLeadingWhitespace(&spec);
200 if (str_util::ConsumePrefix(&spec, "}"))
201 break; // Allow ending with ", }".
202 } else {
203 VERIFY(str_util::ConsumePrefix(&spec, "}"),
204 "Expected , or } after strings in list, not: '", spec, "'");
205 break;
206 }
207 }
208 } else { // "{ bool, numbertype, string }"
209 type = "type";
210 while (true) {
211 VERIFY(ConsumeAttrType(&spec, &type_string),
212 "Trouble parsing type string at '", spec, "'");
213 if (ProcessCompoundType(type_string, allowed)) {
214 // Processed a compound type.
215 } else {
216 DataType dt;
217 VERIFY(DataTypeFromString(type_string, &dt),
218 "Unrecognized type string '", type_string, "'");
219 allowed->mutable_list()->add_type(dt);
220 }
221 if (str_util::ConsumePrefix(&spec, ",")) {
222 str_util::RemoveLeadingWhitespace(&spec);
223 if (str_util::ConsumePrefix(&spec, "}"))
224 break; // Allow ending with ", }".
225 } else {
226 VERIFY(str_util::ConsumePrefix(&spec, "}"),
227 "Expected , or } after types in list, not: '", spec, "'");
228 break;
229 }
230 }
231 }
232 } else { // if spec.Consume("{")
233 VERIFY(false, "Trouble parsing type string at '", spec, "'");
234 }
235 str_util::RemoveLeadingWhitespace(&spec);
236
237 // Write the type into *attr.
238 if (is_list) {
239 VERIFY(str_util::ConsumePrefix(&spec, ")"),
240 "Expected ) to close 'list(', not: '", spec, "'");
241 str_util::RemoveLeadingWhitespace(&spec);
242 attr->set_type(strings::StrCat("list(", type, ")"));
243 } else {
244 attr->set_type(type);
245 }
246
247 // Read optional minimum constraint at the end.
248 if ((is_list || type == "int") && str_util::ConsumePrefix(&spec, ">=")) {
249 int64 min_limit = -999;
250 VERIFY(ConsumeAttrNumber(&spec, &min_limit),
251 "Could not parse integer lower limit after '>=', found '", spec,
252 "' instead");
253 attr->set_has_minimum(true);
254 attr->set_minimum(min_limit);
255 }
256
257 // Parse default value, if present.
258 if (str_util::ConsumePrefix(&spec, "=")) {
259 str_util::RemoveLeadingWhitespace(&spec);
260 VERIFY(ParseAttrValue(attr->type(), spec, attr->mutable_default_value()),
261 "Could not parse default value '", spec, "'");
262 } else {
263 VERIFY(spec.empty(), "Extra '", spec, "' unparsed at the end");
264 }
265}

328void FinalizeInputOrOutput(StringPiece spec, bool is_output, OpDef* op_def,
329 std::vector<string>* errors) {
330 OpDef::ArgDef* arg =
331 is_output ? op_def->add_output_arg() : op_def->add_input_arg();
332
333 StringPiece orig(spec);
334
335 // Parse "<name>:" at the beginning.
336 StringPiece tmp_name;
337 VERIFY(ConsumeInOutName(&spec, &tmp_name), "Trouble parsing 'name:'");
338 arg->set_name(tmp_name.data(), tmp_name.size());
339
340 // Detect "Ref(...)".
341 if (ConsumeInOutRefOpen(&spec)) {
342 arg->set_is_ref(true);
343 }
344
345 { // Parse "<name|type>" or "<name>*<name|type>".
346 StringPiece first, second, type_or_attr;
347 VERIFY(ConsumeInOutNameOrType(&spec, &first),
348 "Trouble parsing either a type or an attr name at '", spec, "'");
349 if (ConsumeInOutTimesType(&spec, &second)) {
350 arg->set_number_attr(first.data(), first.size());
351 type_or_attr = second;
352 } else {
353 type_or_attr = first;
354 }
355 DataType dt;
356 if (DataTypeFromString(type_or_attr, &dt)) {
357 arg->set_type(dt);
358 } else {
359 const OpDef::AttrDef* attr = FindAttr(type_or_attr, *op_def);
360 VERIFY(attr != nullptr, "Reference to unknown attr '", type_or_attr, "'");
361 if (attr->type() == "type") {
362 arg->set_type_attr(type_or_attr.data(), type_or_attr.size());
363 } else {
364 VERIFY(attr->type() == "list(type)", "Reference to attr '",
365 type_or_attr, "' with type ", attr->type(),
366 " that isn't type or list(type)");
367 arg->set_type_list_attr(type_or_attr.data(), type_or_attr.size());
368 }
369 }
370 }
371
372 // Closing ) for Ref(.
373 if (arg->is_ref()) {
374 VERIFY(ConsumeInOutRefClose(&spec),
375 "Did not find closing ')' for 'Ref(', instead found: '", spec, "'");
376 }
377
378 // Should not have anything else.
379 VERIFY(spec.empty(), "Extra '", spec, "' unparsed at the end");
380
381 // Int attrs that are the length of an input or output get a default
382 // minimum of 1.
383 if (!arg->number_attr().empty()) {
384 OpDef::AttrDef* attr = FindAttrMutable(arg->number_attr(), op_def);
385 if (attr != nullptr && !attr->has_minimum()) {
386 attr->set_has_minimum(true);
387 attr->set_minimum(1);
388 }
389 } else if (!arg->type_list_attr().empty()) {
390 // If an input or output has type specified by a list(type) attr,
391 // it gets a default minimum of 1 as well.
392 OpDef::AttrDef* attr = FindAttrMutable(arg->type_list_attr(), op_def);
393 if (attr != nullptr && attr->type() == "list(type)" &&
394 !attr->has_minimum()) {
395 attr->set_has_minimum(true);
396 attr->set_minimum(1);
397 }
398 }
399
400 // If the arg's dtype is resource we should mark the op as stateful as it
401 // likely touches a resource manager. This deliberately doesn't cover inputs /
402 // outputs which resolve to resource via Attrs as those mostly operate on
403 // resource handles as an opaque type (as opposed to ops which explicitly take
404 // / produce resources).
405 if (arg->type() == DT_RESOURCE) {
406 op_def->set_is_stateful(true);
407 }
408}

413 size_t i = 0;

435void FinalizeDoc(const string& text, OpDef* op_def,
436 std::vector<string>* errors) {
437 std::vector<string> lines = str_util::Split(text, '\n');
438
439 // Remove trailing spaces.
440 for (string& line : lines) {
441 str_util::StripTrailingWhitespace(&line);
442 }
443
444 // First non-blank line -> summary.
445 int l = 0;
446 while (static_cast<size_t>(l) < lines.size() && lines[l].empty()) ++l;
447 if (static_cast<size_t>(l) < lines.size()) {
448 op_def->set_summary(lines[l]);
449 ++l;
450 }
451 while (static_cast<size_t>(l) < lines.size() && lines[l].empty()) ++l;
452
453 // Lines until we see name: -> description.
454 int start_l = l;
455 while (static_cast<size_t>(l) < lines.size() && !IsDocNameColon(lines[l])) {
456 ++l;
457 }
458 int end_l = l;
459 // Trim trailing blank lines from the description.
460 while (start_l < end_l && lines[end_l - 1].empty()) --end_l;
461 string desc = str_util::Join(
462 gtl::ArraySlice<string>(lines.data() + start_l, end_l - start_l), "\n");
463 if (!desc.empty()) op_def->set_description(desc);
464
465 // name: description
466 // possibly continued on the next line
467 // if so, we remove the minimum indent
468 StringPiece name;
469 std::vector<StringPiece> description;
470 while (static_cast<size_t>(l) < lines.size()) {
471 description.clear();
472 description.push_back(lines[l]);
473 ConsumeDocNameColon(&description.back(), &name);
474 ++l;
475 while (static_cast<size_t>(l) < lines.size() && !IsDocNameColon(lines[l])) {
476 description.push_back(lines[l]);
477 ++l;
478 }
479 // Remove any trailing blank lines.
480 while (!description.empty() && description.back().empty()) {
481 description.pop_back();
482 }
483 // Compute the minimum indent of all lines after the first.
484 int min_indent = -1;
485 for (size_t i = 1; i < description.size(); ++i) {
486 if (!description[i].empty()) {
487 int indent = num_leading_spaces(description[i]);
488 if (min_indent < 0 || indent < min_indent) min_indent = indent;
489 }
490 }
491 // Remove min_indent spaces from all lines after the first.
492 for (size_t i = 1; i < description.size(); ++i) {
493 if (!description[i].empty()) description[i].remove_prefix(min_indent);
494 }
495 // Concatenate lines into a single string.
496 const string complete(str_util::Join(description, "\n"));
497
498 // Find name.
499 bool found = false;
500 for (int i = 0; !found && i < op_def->input_arg_size(); ++i) {
501 if (op_def->input_arg(i).name() == name) {
502 op_def->mutable_input_arg(i)->set_description(complete);
503 found = true;
504 }
505 }
506 for (int i = 0; !found && i < op_def->output_arg_size(); ++i) {
507 if (op_def->output_arg(i).name() == name) {
508 op_def->mutable_output_arg(i)->set_description(complete);
509 found = true;
510 }
511 }
512 for (int i = 0; !found && i < op_def->attr_size(); ++i) {
513 if (op_def->attr(i).name() == name) {
514 op_def->mutable_attr(i)->set_description(complete);
515 found = true;
516 }
517 }
518 if (!found) {
519 errors->push_back(
520 strings::StrCat("No matching input/output/attr for name '", name,
521 "' from Doc() for Op ", op_def->name()));
522 return;
523 }
524 }
525}

603Status OpDefBuilder::Finalize(OpRegistrationData* op_reg_data) const {
604 std::vector<string> errors = errors_;
605 *op_reg_data = op_reg_data_;
606
607 OpDef* op_def = &op_reg_data->op_def;
608 for (StringPiece attr : attrs_) {
609 FinalizeAttr(attr, op_def, &errors);
610 }
611 for (StringPiece input : inputs_) {
612 FinalizeInputOrOutput(input, false, op_def, &errors);
613 }
614 for (StringPiece output : outputs_) {
615 FinalizeInputOrOutput(output, true, op_def, &errors);
616 }
617 FinalizeDoc(doc_, op_def, &errors);
618
619 if (errors.empty()) return Status::OK();
620 return errors::InvalidArgument(str_util::Join(errors, "\n"));
621}
