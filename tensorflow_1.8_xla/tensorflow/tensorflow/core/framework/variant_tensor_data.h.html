27class VariantTensorDataProto;
28class Tensor;
29
30// The serialization format for Variant objects. Objects with references to
31// other Tensors can simply store those tensors in the `tensors` field, and
32// serialize other metadata content in to the `metadata` field. Objects can
33// optionally set the `type_name` for type-checking before deserializing an
34// object.
35//
36// This is the native C++ class equivalent of VariantTensorDataProto. They are
37// separate so that kernels do not need to depend on protos.
38class VariantTensorData {
39 public:
40 VariantTensorData();
41 VariantTensorData(const VariantTensorDataProto& proto);
42 ~VariantTensorData();
43
44 // Name of the type of objects being serialized.
45 const string& type_name() const { return type_name_; }
46 void set_type_name(const string& type_name) { type_name_ = type_name; }
47
48 template <typename T, bool = std::is_pod<typename std::decay<T>::type>::value>
49 struct PODResolver {};
50
51 // Portions of the object that are not Tensors.
52 // Directly supported types include string POD types.
53 template <typename T>
54 void set_metadata(const T& value) {
55 SetMetadata<T>(value, PODResolver<T>());
56 }
57
58 template <typename T>
59 bool get_metadata(T* value) const {
60 return GetMetadata<T>(value, PODResolver<T>());
61 }
62
63 // Tensors contained within objects being serialized.
64 int tensors_size() const;
65 const Tensor& tensors(int index) const;
66 const std::vector<Tensor>& tensors() const;
67 Tensor* add_tensors();
68
69 // Conversion to and from VariantTensorDataProto
70 void ToProto(VariantTensorDataProto* proto) const;
71 bool FromProto(const VariantTensorDataProto& proto);
72
73 // Serialization via VariantTensorDataProto
74 string SerializeAsString() const;
75 bool SerializeToString(string* buf);
76 bool ParseFromString(const string& s);
77
78 string DebugString() const;
79
80 public:
81 string type_name_;
82 string metadata_;
83 std::vector<Tensor> tensors_;
84
85 private:
86 template <typename T>
87 void SetMetadata(const string& value, PODResolver<T, false /* is_pod */>) {
88 metadata_ = value;
89 }
90
91 template <typename T>
92 bool GetMetadata(string* value, PODResolver<T, false /* is_pod */>) const {
93 *value = metadata_;
94 return true;
95 }
96
97 template <typename T>
98 void SetMetadata(const T& value, PODResolver<T, true /* is_pod */>) {
99 metadata_.assign(reinterpret_cast<const char*>(&value), sizeof(T));
100 }
101
102 template <typename T>
103 bool GetMetadata(T* value, PODResolver<T, true /* is_pod */>) const {
104 if (metadata_.size() != sizeof(T)) return false;
105 std::copy_n(metadata_.data(), sizeof(T), reinterpret_cast<char*>(value));
106 return true;
107 }
108};

111string ProtoDebugString(const VariantTensorData& object);
