16#include "tensorflow/core/framework/resource_handle.h"
17#include "tensorflow/core/framework/resource_handle.pb.h"
18#include "tensorflow/core/lib/strings/strcat.h"
19
20namespace tensorflow {
21
22ResourceHandle::ResourceHandle() {}
23
24ResourceHandle::ResourceHandle(const ResourceHandleProto& proto) {
25 FromProto(proto);
26}
27
28ResourceHandle::~ResourceHandle() {}
29
30void ResourceHandle::AsProto(ResourceHandleProto* proto) const {
31 proto->set_device(device());
32 proto->set_container(container());
33 proto->set_name(name());
34 proto->set_hash_code(hash_code());
35 proto->set_maybe_type_name(maybe_type_name());
36}
37
38void ResourceHandle::FromProto(const ResourceHandleProto& proto) {
39 set_device(proto.device());
40 set_container(proto.container());
41 set_name(proto.name());
42 set_hash_code(proto.hash_code());
43 set_maybe_type_name(proto.maybe_type_name());
44}
45
46string ResourceHandle::SerializeAsString() const {
47 ResourceHandleProto proto;
48 AsProto(&proto);
49 return proto.SerializeAsString();
50}
51
52bool ResourceHandle::ParseFromString(const string& s) {
53 ResourceHandleProto proto;
54 const bool status = proto.ParseFromString(s);
55 if (status) FromProto(proto);
56 return status;
57}
58
59string ResourceHandle::DebugString() const {
60 return strings::StrCat("device: ", device(), " container: ", container(),
61 " name: ", name(), " hash_code: ", hash_code(),
62 " maybe_type_name: ", maybe_type_name());
63}
64
65string ProtoDebugString(const ResourceHandle& handle) {
66 return handle.DebugString();
67}
