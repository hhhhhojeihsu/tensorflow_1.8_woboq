36string SummarizeString(const string& str) {
37 string escaped = str_util::CEscape(str);
38
39 // If the string is long, replace the middle with ellipses.
40 constexpr int kMaxStringSummarySize = 80;
41 if (escaped.size() >= kMaxStringSummarySize) {
42 StringPiece prefix(escaped);
43 StringPiece suffix = prefix;
44 prefix.remove_suffix(escaped.size() - 10);
45 suffix.remove_prefix(escaped.size() - 10);
46 return strings::StrCat("\"", prefix, "...", suffix, "\"");
47 } else {
48 return strings::StrCat("\"", escaped, "\"");
49 }
50}

52string SummarizeTensor(const TensorProto& tensor_proto) {
53 Tensor t;
54 if (!t.FromProto(tensor_proto)) {
55 return strings::StrCat(
56 "<Invalid TensorProto: ", ProtoShortDebugString(tensor_proto), ">");
57 }
58 return t.DebugString();
59}

61string SummarizeFunc(const NameAttrList& func) {
62 std::vector<string> entries;
63 for (auto p : func.attr()) {
64 entries.push_back(
65 strings::StrCat(p.first, "=", SummarizeAttrValue(p.second)));
66 }
67 std::sort(entries.begin(), entries.end());
68 return strings::StrCat(func.name(), "[", str_util::Join(entries, ", "), "]");
69}

73string SummarizeAttrValue(const AttrValue& attr_value) {
74 switch (attr_value.value_case()) {
75 case AttrValue::kS:
76 return SummarizeString(attr_value.s());
77 case AttrValue::kI:
78 return strings::StrCat(attr_value.i());
79 case AttrValue::kF:
80 return strings::StrCat(attr_value.f());
81 case AttrValue::kB:
82 return attr_value.b() ? "true" : "false";
83 case AttrValue::kType:
84 return EnumName_DataType(attr_value.type());
85 case AttrValue::kShape:
86 return PartialTensorShape::DebugString(attr_value.shape());
87 case AttrValue::kTensor:
88 return SummarizeTensor(attr_value.tensor());
89 case AttrValue::kList: {
90 std::vector<string> pieces;
91 if (attr_value.list().s_size() > 0) {
92 for (int i = 0; i < attr_value.list().s_size(); ++i) {
93 pieces.push_back(SummarizeString(attr_value.list().s(i)));
94 }
95 } else if (attr_value.list().i_size() > 0) {
96 for (int i = 0; i < attr_value.list().i_size(); ++i) {
97 pieces.push_back(strings::StrCat(attr_value.list().i(i)));
98 }
99 } else if (attr_value.list().f_size() > 0) {
100 for (int i = 0; i < attr_value.list().f_size(); ++i) {
101 pieces.push_back(strings::StrCat(attr_value.list().f(i)));
102 }
103 } else if (attr_value.list().b_size() > 0) {
104 for (int i = 0; i < attr_value.list().b_size(); ++i) {
105 pieces.push_back(attr_value.list().b(i) ? "true" : "false");
106 }
107 } else if (attr_value.list().type_size() > 0) {
108 for (int i = 0; i < attr_value.list().type_size(); ++i) {
109 pieces.push_back(EnumName_DataType(attr_value.list().type(i)));
110 }
111 } else if (attr_value.list().shape_size() > 0) {
112 for (int i = 0; i < attr_value.list().shape_size(); ++i) {
113 pieces.push_back(
114 TensorShape::DebugString(attr_value.list().shape(i)));
115 }
116 } else if (attr_value.list().tensor_size() > 0) {
117 for (int i = 0; i < attr_value.list().tensor_size(); ++i) {
118 pieces.push_back(SummarizeTensor(attr_value.list().tensor(i)));
119 }
120 } else if (attr_value.list().func_size() > 0) {
121 for (int i = 0; i < attr_value.list().func_size(); ++i) {
122 pieces.push_back(SummarizeFunc(attr_value.list().func(i)));
123 }
124 }
125 constexpr int kMaxListSummarySize = 15;
126 if (pieces.size() >= kMaxListSummarySize) {
127 pieces.erase(pieces.begin() + 5, pieces.begin() + (pieces.size() - 6));
128 pieces[5] = "...";
129 }
130 return strings::StrCat("[", str_util::Join(pieces, ", "), "]");
131 }
132 case AttrValue::kFunc: {
133 return SummarizeFunc(attr_value.func());
134 }
135 case AttrValue::kPlaceholder:
136 return strings::StrCat("$", attr_value.placeholder());
137 case AttrValue::VALUE_NOT_SET:
138 return "<Unknown AttrValue type>";
139 }
140 return "<Unknown AttrValue type>"; // Prevent missing return warning
141}

143Status AttrValueHasType(const AttrValue& attr_value, StringPiece type) {
144 int num_set = 0;
145
146#define VALIDATE_FIELD(name, type_string, oneof_case) \
147 do { \
148 if (attr_value.has_list()) { \
149 if (attr_value.list().name##_size() > 0) { \
150 if (type != "list(" type_string ")") { \
151 return errors::InvalidArgument( \
152 "AttrValue had value with type 'list(" type_string ")' when '", \
153 type, "' expected"); \
154 } \
155 ++num_set; \
156 } \
157 } else if (attr_value.value_case() == AttrValue::oneof_case) { \
158 if (type != type_string) { \
159 return errors::InvalidArgument( \
160 "AttrValue had value with type '" type_string "' when '", type, \
161 "' expected"); \
162 } \
163 ++num_set; \
164 } \
165 } while (false)
166
167 VALIDATE_FIELD(s, "string", kS);
168 VALIDATE_FIELD(i, "int", kI);
169 VALIDATE_FIELD(f, "float", kF);
170 VALIDATE_FIELD(b, "bool", kB);
171 VALIDATE_FIELD(type, "type", kType);
172 VALIDATE_FIELD(shape, "shape", kShape);
173 VALIDATE_FIELD(tensor, "tensor", kTensor);
174 VALIDATE_FIELD(func, "func", kFunc);
175
176#undef VALIDATE_FIELD
177
178 if (attr_value.value_case() == AttrValue::kPlaceholder) {
179 return errors::InvalidArgument(
180 "AttrValue had value with unexpected type 'placeholder'");
181 }
182
183 // If the attr type is 'list', we expect attr_value.has_list() to be
184 // true. However, proto3's attr_value.has_list() can be false when
185 // set to an empty list for GraphDef versions <= 4. So we simply
186 // check if has_list is false and some other field in attr_value is
187 // set to flag the error. This test can be made more strict once
188 // support for GraphDef versions <= 4 is dropped.
189 if (str_util::StartsWith(type, "list(") && !attr_value.has_list()) {
190 if (num_set) {
191 return errors::InvalidArgument(
192 "AttrValue missing value with expected type '", type, "'");
193 } else {
194 // Indicate that we have a list, but an empty one.
195 ++num_set;
196 }
197 }
198
199 // Okay to have an empty list, but not to be missing a non-list value.
200 if (num_set == 0 && !str_util::StartsWith(type, "list(")) {
201 return errors::InvalidArgument(
202 "AttrValue missing value with expected type '", type, "'");
203 }
204
205 // Ref types and DT_INVALID are illegal, and DataTypes must
206 // be a valid enum type.
207 if (type == "type") {
208 if (!DataType_IsValid(attr_value.type())) {
209 return errors::InvalidArgument("AttrValue has invalid DataType enum: ",
210 attr_value.type());
211 }
212 if (IsRefType(attr_value.type())) {
213 return errors::InvalidArgument(
214 "AttrValue must not have reference type value of ",
215 DataTypeString(attr_value.type()));
216 }
217 if (attr_value.type() == DT_INVALID) {
218 return errors::InvalidArgument("AttrValue has invalid DataType");
219 }
220 } else if (type == "list(type)") {
221 for (auto as_int : attr_value.list().type()) {
222 const DataType dtype = static_cast<DataType>(as_int);
223 if (!DataType_IsValid(dtype)) {
224 return errors::InvalidArgument("AttrValue has invalid DataType enum: ",
225 as_int);
226 }
227 if (IsRefType(dtype)) {
228 return errors::InvalidArgument(
229 "AttrValue must not have reference type value of ",
230 DataTypeString(dtype));
231 }
232 if (dtype == DT_INVALID) {
233 return errors::InvalidArgument("AttrValue contains invalid DataType");
234 }
235 }
236 }
237
238 return Status::OK();
239}

241bool ParseAttrValue(StringPiece type, StringPiece text, AttrValue* out) {
242 // Parse type.
243 string field_name;
244 bool is_list = str_util::ConsumePrefix(&type, "list(");
245 if (str_util::ConsumePrefix(&type, "string")) {
246 field_name = "s";
247 } else if (str_util::ConsumePrefix(&type, "int")) {
248 field_name = "i";
249 } else if (str_util::ConsumePrefix(&type, "float")) {
250 field_name = "f";
251 } else if (str_util::ConsumePrefix(&type, "bool")) {
252 field_name = "b";
253 } else if (str_util::ConsumePrefix(&type, "type")) {
254 field_name = "type";
255 } else if (str_util::ConsumePrefix(&type, "shape")) {
256 field_name = "shape";
257 } else if (str_util::ConsumePrefix(&type, "tensor")) {
258 field_name = "tensor";
259 } else if (str_util::ConsumePrefix(&type, "func")) {
260 field_name = "func";
261 } else if (str_util::ConsumePrefix(&type, "placeholder")) {
262 field_name = "placeholder";
263 } else {
264 return false;
265 }
266 if (is_list && !str_util::ConsumePrefix(&type, ")")) {
267 return false;
268 }
269
270 // Construct a valid text proto message to parse.
271 string to_parse;
272 if (is_list) {
273 // TextFormat parser considers "i: 7" to be the same as "i: [7]",
274 // but we only want to allow list values with [].
275 StringPiece cleaned = text;
276 str_util::RemoveLeadingWhitespace(&cleaned);
277 str_util::RemoveTrailingWhitespace(&cleaned);
278 if (cleaned.size() < 2 || cleaned[0] != '[' ||
279 cleaned[cleaned.size() - 1] != ']') {
280 return false;
281 }
282 cleaned.remove_prefix(1);
283 str_util::RemoveLeadingWhitespace(&cleaned);
284 if (cleaned.size() == 1) {
285 // User wrote "[]", so return empty list without invoking the TextFormat
286 // parse which returns an error for "i: []".
287 out->Clear();
288 out->mutable_list();
289 return true;
290 }
291 to_parse = strings::StrCat("list { ", field_name, ": ", text, " }");
292 } else {
293 to_parse = strings::StrCat(field_name, ": ", text);
294 }
295
296 return ProtoParseFromString(to_parse, out);
297}

414bool AreAttrValuesEqual(const AttrValue& a, const AttrValue& b) {
415 // There are multiple equivalent representations of attr values containing
416 // TensorProtos. Compare them by constructing Tensors and serializing them
417 // back. Comparing Tensor objects is pretty tricky.
418 if (a.has_tensor() != b.has_tensor()) {
419 return false;
420 } else if (a.has_tensor() && b.has_tensor()) {
421 Tensor at(a.tensor().dtype());
422 bool success = at.FromProto(a.tensor());
423 DCHECK(success);
424
425 Tensor bt(b.tensor().dtype());
426 success = bt.FromProto(b.tensor());
427 DCHECK(success);
428
429 TensorProto ap;
430 at.AsProtoTensorContent(&ap);
431
432 TensorProto bp;
433 bt.AsProtoTensorContent(&bp);
434
435 string a_str, b_str;
436 SerializeToStringDeterministic(ap, &a_str);
437 SerializeToStringDeterministic(bp, &b_str);
438 return a_str == b_str;
439 }
440
441 // `func` field contains a nested AttrValue. Compare such AttrValues
442 // recursively.
443 if (a.has_func() != b.has_func()) {
444 return false;
445 } else if (a.has_func() && b.has_func()) {
446 const NameAttrList& af = a.func();
447 const NameAttrList& bf = b.func();
448 if (af.name() != bf.name()) return false;
449 std::unordered_map<string, AttrValue> am(af.attr().begin(),
450 af.attr().end());
451 for (const auto& bm_pair : bf.attr()) {
452 const auto& iter = am.find(bm_pair.first);
453 if (iter == am.end()) return false;
454 if (!AreAttrValuesEqual(iter->second, bm_pair.second)) return false;
455 am.erase(iter);
456 }
457 if (!am.empty()) return false;
458 return true;
459 }
460
461 // All other fields in AttrValue have deterministic representations.
462 // It is safe to compare their serialized strings.
463 string a_str, b_str;
464 SerializeToStringDeterministic(a, &a_str);
465 SerializeToStringDeterministic(b, &b_str);
466 return a_str == b_str;
467}

469uint64 AttrValueHash(const AttrValue& a) {
470 if (a.has_tensor()) {
471 // Deal with multiple representations by parsing TensorProto to
472 // Tensor and serializing it back. This is slow, but current use case
473 // don't need high efficiency.
474 Tensor tensor(a.tensor().dtype());
475 bool success = tensor.FromProto(a.tensor());
476 DCHECK(success);
477 TensorProto p;
478 tensor.AsProtoTensorContent(&p);
479 string s;
480 SerializeToStringDeterministic(p, &s);
481 return Hash64(s);
482 }
483 if (a.has_func()) {
484 const NameAttrList& func = a.func();
485 uint64 h = Hash64(func.name());
486 std::map<string, AttrValue> map(func.attr().begin(), func.attr().end());
487 for (const auto& pair : map) {
488 h = Hash64(pair.first.data(), pair.first.size(), h);
489 h = Hash64Combine(AttrValueHash(pair.second), h);
490 }
491 return h;
492 }
493
494 // If `a` is not a tensor or func, get a hash of serialized string.
495 string s;
496 SerializeToStringDeterministic(a, &s);
497 return Hash64(s);
498}

500bool HasPlaceHolder(const AttrValue& val) {
501 switch (val.value_case()) {
502 case AttrValue::kList: {
503 for (const NameAttrList& func : val.list().func()) {
504 for (const auto& p : func.attr()) {
505 if (HasPlaceHolder(p.second)) {
506 return true;
507 }
508 }
509 }
510 break;
511 }
512 case AttrValue::kFunc:
513 for (const auto& p : val.func().attr()) {
514 if (HasPlaceHolder(p.second)) {
515 return true;
516 }
517 }
518 break;
519 case AttrValue::kPlaceholder:
520 return true;
521 default:
522 break;
523 }
524 return false;
525}
526
527bool SubstitutePlaceholders(const SubstituteFunc& substitute,
528 AttrValue* value) {
529 switch (value->value_case()) {
530 case AttrValue::kList: {
531 for (NameAttrList& func : *value->mutable_list()->mutable_func()) {
532 for (auto& p : *func.mutable_attr()) {
533 if (!SubstitutePlaceholders(substitute, &p.second)) {
534 return false;
535 }
536 }
537 }
538 break;
539 }
540 case AttrValue::kFunc:
541 for (auto& p : *(value->mutable_func()->mutable_attr())) {
542 if (!SubstitutePlaceholders(substitute, &p.second)) {
543 return false;
544 }
545 }
546 break;
547 case AttrValue::kPlaceholder:
548 return substitute(value->placeholder(), value);
549 case AttrValue::VALUE_NOT_SET:
550 return false;
551 default:
552 break;
553 }
554 return true;
555}
