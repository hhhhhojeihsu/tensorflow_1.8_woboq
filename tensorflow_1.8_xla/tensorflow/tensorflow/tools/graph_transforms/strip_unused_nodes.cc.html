32Status TypeForPlaceholder(const TransformFuncContext& context,
33 const string& node_name, DataType* result) {
34 // If we don't find anything else, return float.
35 *result = DT_FLOAT;
36
37 // Check to see if we have been given a default for all placeholders.
38 if (context.params.count("type")) {
39 if (context.params.at("type").size() != 1) {
40 return errors::InvalidArgument(
41 "You must pass no more than one default 'type' to "
42 "strip_unused_nodes");
43 }
44 const string& type_string = context.params.at("type")[0];
45 if (!DataTypeFromString(type_string, result)) {
46 return errors::InvalidArgument("Couldn't understand type argument '",
47 type_string, "'");
48 }
49 }
50
51 // See if there's a particular type specified for this placeholder.
52 if (context.params.count("name") || context.params.count("type_for_name")) {
53 if (!context.params.count("name") ||
54 !context.params.count("type_for_name") ||
55 (context.params.at("type_for_name").size() !=
56 context.params.at("name").size())) {
57 return errors::InvalidArgument(
58 "You must pass a 'type_for_name' arg for every 'name', e.g. "
59 "strip_unused_nodes(name=foo, type_for_name=float, name=bar, "
60 "type_for_name=quint8");
61 }
62 const int name_count = context.params.at("name").size();
63 for (int i = 0; i < name_count; ++i) {
64 if (context.params.at("name")[i] == node_name) {
65 const string& type_string = context.params.at("type_for_name")[i];
66 if (!DataTypeFromString(type_string, result)) {
67 return errors::InvalidArgument("Couldn't understand type argument '",
68 type_string, "'");
69 }
70 }
71 }
72 }
73
74 return Status::OK();
75}

77Status ShapeForPlaceholder(const TransformFuncContext& context,
78 const string& node_name, TensorShape* result) {
79 // If we don't find anything else, return scalar.
80 *result = {};
81
82 // Check to see if we have been given a default for all placeholders.
83 if (context.params.count("type")) {
84 if (context.params.at("shape").size() != 1) {
85 return errors::InvalidArgument(
86 "You must pass no more than one default 'shape' to "
87 "strip_unused_nodes");
88 }
89 const string& shape_string = context.params.at("shape")[0];
90 TF_RETURN_IF_ERROR(TensorShapeFromString(shape_string, result));
91 }
92
93 // See if there's a particular type specified for this placeholder.
94 if (context.params.count("name") || context.params.count("type_for_name")) {
95 if (!context.params.count("name") ||
96 !context.params.count("type_for_name") ||
97 (context.params.at("type_for_name").size() !=
98 context.params.at("name").size())) {
99 return errors::InvalidArgument(
100 "You must pass a 'shape_for_name' arg for every 'name', e.g. "
101 "strip_unused_nodes(name=foo, shape_for_name=\"2,2,1\", name=bar, "
102 "shape_for_name=\"1\"");
103 }
104 const int name_count = context.params.at("name").size();
105 for (int i = 0; i < name_count; ++i) {
106 if (context.params.at("name")[i] == node_name) {
107 const string& shape_string = context.params.at("shape_for_name")[i];
108 TF_RETURN_IF_ERROR(TensorShapeFromString(shape_string, result));
109 }
110 }
111 }
112
113 return Status::OK();
114}

117// Delete any nodes that don't contribute to the inference result.
118Status StripUnusedNodes(const GraphDef& input_graph_def,
119 const TransformFuncContext& context,
120 GraphDef* output_graph_def) {
121 std::set<string> required_nodes;
122 std::set<string> input_nodes;
123 for (const string& input : context.input_names) {
124 required_nodes.insert(NodeNameFromInput(input));
125 input_nodes.insert(NodeNameFromInput(input));
126 }
127 for (const string& output : context.output_names) {
128 required_nodes.insert(output);
129 }
130
131 std::map<string, const NodeDef*> node_lookup;
132 MapNamesToNodes(input_graph_def, &node_lookup);
133
134 std::vector<string> current_inputs;
135 for (const string& output_name : context.output_names) {
136 current_inputs.push_back(NodeNameFromInput(output_name));
137 }
138
139 while (!current_inputs.empty()) {
140 std::set<string> next_inputs;
141 for (const string& current_input : current_inputs) {
142 required_nodes.insert(current_input);
143 if (input_nodes.count(current_input)) {
144 continue;
145 }
146 if (!node_lookup.count(current_input)) {
147 return errors::InvalidArgument("Input node ", current_input,
148 " not found in graph");
149 }
150 const NodeDef* current_node = node_lookup[current_input];
151 for (const string& input_name : current_node->input()) {
152 string input_node_name = NodeNameFromInput(input_name);
153 if (!required_nodes.count(input_node_name)) {
154 next_inputs.insert(input_node_name);
155 }
156 }
157 }
158 current_inputs =
159 std::vector<string>(next_inputs.begin(), next_inputs.end());
160 }
161
162 GraphDef filtered_graph_def;
163 FilterGraphDef(input_graph_def,
164 [&](const NodeDef& node) {
165 return required_nodes.count(node.name()) > 0;
166 },
167 &filtered_graph_def);
168
169 output_graph_def->Clear();
170 for (const NodeDef& node : filtered_graph_def.node()) {
171 if (input_nodes.count(node.name())) {
172 NodeDef placeholder_node;
173 if (node.op() == "Placeholder") {
174 placeholder_node = node;
175 } else {
176 placeholder_node.set_op("Placeholder");
177 placeholder_node.set_name(node.name());
178 DataType type;
179 TF_RETURN_IF_ERROR(TypeForPlaceholder(context, node.name(), &type));
180 TensorShape shape;
181 TF_RETURN_IF_ERROR(ShapeForPlaceholder(context, node.name(), &shape));
182 SetNodeAttr("dtype", type, &placeholder_node);
183 SetNodeAttr("shape", shape, &placeholder_node);
184 }
185 *(output_graph_def->mutable_node()->Add()) = placeholder_node;
186 } else {
187 *(output_graph_def->mutable_node()->Add()) = node;
188 }
189 }
190 return Status::OK();
191}

193REGISTER_GRAPH_TRANSFORM("strip_unused_nodes", StripUnusedNodes);
