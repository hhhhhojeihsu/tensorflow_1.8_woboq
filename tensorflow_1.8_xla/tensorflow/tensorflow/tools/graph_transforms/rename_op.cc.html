30// Changes the op type of a specified op.
31Status RenameOp(const GraphDef& input_graph_def,
32 const TransformFuncContext& context,
33 GraphDef* output_graph_def) {
34 if (!context.params.count("old_op_name") ||
35 (context.params.at("old_op_name").size() != 1) ||
36 !context.params.count("new_op_name") ||
37 (context.params.at("new_op_name").size() != 1)) {
38 return errors::InvalidArgument(
39 "remove_nodes expects exactly one 'old_op_name' and 'new_op_name' "
40 "argument, e.g. rename_op(old_op_name=Mul, new_op_name=Multiply)");
41 }
42
43 const string old_op_name = context.params.at("old_op_name")[0];
44 const string new_op_name = context.params.at("new_op_name")[0];
45 output_graph_def->Clear();
46 for (const NodeDef& node : input_graph_def.node()) {
47 NodeDef* new_node = output_graph_def->mutable_node()->Add();
48 *new_node = node;
49 if (node.op() == old_op_name) {
50 new_node->set_op(new_op_name);
51 }
52 }
53
54 return Status::OK();
55}
56
57REGISTER_GRAPH_TRANSFORM("rename_op", RenameOp);
