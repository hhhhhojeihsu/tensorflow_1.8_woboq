27Status FlattenAtrousConv(const GraphDef& input_graph_def,
28 const TransformFuncContext& context,
29 GraphDef* output_graph_def) {
30 GraphDef replaced_graph_def;
31 TF_RETURN_IF_ERROR(ReplaceMatchingOpTypes(
32 input_graph_def, // clang-format off
33 {"BatchToSpaceND",
34 {
35 {"Conv2D|DepthwiseConv2dNative",
36 {
37 {"SpaceToBatchND",
38 {
39 {"*"}, // Input to the flattened op.
40 {"*"}, // block_shape
41 {"*"} // paddings
42 }
43 },
44 {"*"} // filter
45 }
46 },
47 {"*"}, // block_shape
48 {"*"} // crops
49 }
50 }, // clang-format on
51 [](const NodeMatch& match, const std::set<string>& input_nodes,
52 const std::set<string>& output_nodes,
53 std::vector<NodeDef>* new_nodes) {
54 // Find all the nodes we expect in the subgraph.
55 const NodeDef& batch_to_space_node = match.node;
56 const NodeDef& conv_node = match.inputs[0].node;
57 const NodeDef& filter_node = match.inputs[0].inputs[1].node;
58 const NodeDef& input_node = match.inputs[0].inputs[0].inputs[0].node;
59 const NodeDef& space_to_batch_block_shape_node =
60 match.inputs[0].inputs[0].inputs[1].node;
61
62 // The atrous rate value is inferred from the block shape.
63 Tensor block_shape =
64 GetNodeTensorAttr(space_to_batch_block_shape_node, "value");
65 const int32 block_height = block_shape.flat<int32>()(0);
66 const int32 block_width = block_shape.flat<int32>()(1);
67
68 // Compute the upsampled filter.
69 const Tensor& filter = GetNodeTensorAttr(filter_node, "value");
70 const int32 filter_height = filter.dim_size(0);
71 const int32 filter_width = filter.dim_size(1);
72 const int32 in_channels = filter.dim_size(2);
73 const int32 out_channels = filter.dim_size(3);
74
75 const int32 upsampled_filter_height =
76 (filter_height - 1) * block_height + 1;
77 const int32 upsampled_filter_width =
78 (filter_width - 1) * block_width + 1;
79 Tensor upsampled_filter(
80 DT_FLOAT,
81 TensorShape({upsampled_filter_height, upsampled_filter_width,
82 in_channels, out_channels}));
83
84 auto filter_eigen = filter.tensor<float, 4>();
85 auto upsampled_filter_eigen = upsampled_filter.tensor<float, 4>();
86
87 upsampled_filter_eigen.setZero();
88 for (int h = 0; h < filter_height; ++h) {
89 for (int w = 0; w < filter_width; ++w) {
90 for (int c_in = 0; c_in < in_channels; ++c_in) {
91 for (int c_out = 0; c_out < out_channels; ++c_out) {
92 upsampled_filter_eigen(block_height * h, block_width * w, c_in,
93 c_out) = filter_eigen(h, w, c_in, c_out);
94 }
95 }
96 }
97 }
98
99 NodeDef upsampled_filter_node;
100 upsampled_filter_node.set_op("Const");
101 upsampled_filter_node.set_name(filter_node.name());
102 SetNodeAttr("dtype", DT_FLOAT, &upsampled_filter_node);
103 SetNodeTensorAttr<float>("value", upsampled_filter,
104 &upsampled_filter_node);
105
106 // Set up the new flattened version of the convolution op.
107 NodeDef flattened_conv_node;
108
109 flattened_conv_node.set_name(batch_to_space_node.name());
110 flattened_conv_node.set_op(conv_node.op());
111 flattened_conv_node.set_device(conv_node.device());
112
113 AddNodeInput(input_node.name(), &flattened_conv_node);
114 AddNodeInput(upsampled_filter_node.name(), &flattened_conv_node);
115
116 CopyNodeAttr(conv_node, "T", "T", &flattened_conv_node);
117 CopyNodeAttr(conv_node, "strides", "strides", &flattened_conv_node);
118 SetNodeAttr("padding", "SAME", &flattened_conv_node);
119 CopyNodeAttr(conv_node, "data_format", "data_format",
120 &flattened_conv_node);
121
122 if (conv_node.op() == "Conv2D") {
123 CopyNodeAttr(conv_node, "use_cudnn_on_gpu", "use_cudnn_on_gpu",
124 &flattened_conv_node);
125 }
126
127 new_nodes->push_back(input_node);
128 new_nodes->push_back(upsampled_filter_node);
129 new_nodes->push_back(flattened_conv_node);
130
131 return Status::OK();
132 },
133 {}, &replaced_graph_def));
134 *output_graph_def = replaced_graph_def;
135 return Status::OK();
136}
137
138REGISTER_GRAPH_TRANSFORM("flatten_atrous_conv", FlattenAtrousConv);
