30// Ensures the tensor is the expected shape.
31Status ErrorIfNotVector(const Tensor& input, const string& input_name,
32 int expected_width) {
33 if ((input.shape().dims() != 1) ||
34 (input.shape().dim_size(0) != expected_width)) {
35 return errors::InvalidArgument(
36 input_name,
37 " input to batch norm has bad shape: ", input.shape().DebugString());
38 }
39 return Status::OK();
40}

42Status GetScaleAndOffsetValues(const NodeMatch& match,
43 std::vector<float>* scale_values,
44 std::vector<float>* offset_values) {
45 // Find all the nodes we expect in the subgraph.
46 const NodeDef& batch_norm_node = match.node;
47 // BatchNormWithGlobalNormalization and FusedBatchNorm ops only differ
48 // by input order and attribute names.
49 CHECK(batch_norm_node.op() == "BatchNormWithGlobalNormalization" ||
50 batch_norm_node.op() == "FusedBatchNorm");
51 const bool is_fused = batch_norm_node.op() == "FusedBatchNorm";
52 const int mean_idx = is_fused ? 3 : 1;
53 const int var_idx = is_fused ? 4 : 2;
54 const int beta_idx = is_fused ? 2 : 3;
55 const int gamma_idx = is_fused ? 1 : 4;
56 const string epsilon_attr = is_fused ? "epsilon" : "variance_epsilon";
57 // FusedBatchNorm always scales after normalization.
58 const bool scale_after_normalization =
59 is_fused || batch_norm_node.attr().at("scale_after_normalization").b();
60
61 const NodeDef& mean_node = match.inputs[mean_idx].node;
62 CHECK_EQ("Const", mean_node.op());
63 const NodeDef& variance_node = match.inputs[var_idx].node;
64 CHECK_EQ("Const", variance_node.op());
65 const NodeDef& beta_node = match.inputs[beta_idx].node;
66 CHECK_EQ("Const", beta_node.op());
67 const NodeDef& gamma_node = match.inputs[gamma_idx].node;
68 CHECK_EQ("Const", gamma_node.op());
69
70 // We have a set of vectors that we want to combine into a vector of
71 // scale values and offset values.
72 Tensor mean = GetNodeTensorAttr(mean_node, "value");
73 Tensor variance = GetNodeTensorAttr(variance_node, "value");
74 Tensor beta = GetNodeTensorAttr(beta_node, "value");
75 Tensor gamma = GetNodeTensorAttr(gamma_node, "value");
76 const float variance_epsilon = batch_norm_node.attr().at(epsilon_attr).f();
77
78 // Make sure all the inputs really are vectors with the same shape.
79 const int64 num_cols = mean.shape().dim_size(0);
80 TF_RETURN_IF_ERROR(ErrorIfNotVector(variance, "Variance", num_cols));
81 TF_RETURN_IF_ERROR(ErrorIfNotVector(beta, "Beta", num_cols));
82 TF_RETURN_IF_ERROR(ErrorIfNotVector(gamma, "gamma", num_cols));
83
84 scale_values->resize(num_cols);
85 offset_values->resize(num_cols);
86
87 // Calculate the scale and offset values to apply.
88 if (scale_after_normalization) {
89 for (int i = 0; i < num_cols; ++i) {
90 (*scale_values)[i] =
91 (1.0f / sqrtf(variance.flat<float>()(i) + variance_epsilon)) *
92 gamma.flat<float>()(i);
93 }
94 } else {
95 for (int i = 0; i < num_cols; ++i) {
96 (*scale_values)[i] =
97 (1.0f / sqrtf(variance.flat<float>()(i) + variance_epsilon));
98 }
99 }
100 for (int i = 0; i < num_cols; ++i) {
101 (*offset_values)[i] =
102 (-mean.flat<float>()(i) * (*scale_values)[i]) + beta.flat<float>()(i);
103 }
104 return Status::OK();
105}

107Status FuseScaleOffsetToConvWeights(const std::vector<float>& scale_values,
108 const std::vector<float>& offset_values,
109 const NodeMatch& conv_node_match,
110 const string& conv_output_name,
111 std::vector<NodeDef>* new_nodes) {
112 const NodeDef& conv_node = conv_node_match.node;
113 CHECK_EQ("Conv2D", conv_node.op());
114 const NodeDef& input_node = conv_node_match.inputs[0].node;
115 const NodeDef& weights_node = conv_node_match.inputs[1].node;
116 CHECK_EQ("Const", weights_node.op());
117
118 Tensor weights = GetNodeTensorAttr(weights_node, "value");
119 const int64 weights_cols = weights.shape().dim_size(3);
120 CHECK_EQ(weights_cols, scale_values.size());
121
122 // Multiply the original weights by the scale vector.
123 auto weights_matrix = weights.flat_inner_dims<float>();
124 Tensor scaled_weights(DT_FLOAT, weights.shape());
125 auto scaled_weights_matrix = scaled_weights.flat_inner_dims<float>();
126 for (int64 row = 0; row < weights_matrix.dimension(0); ++row) {
127 for (int64 col = 0; col < weights_cols; ++col) {
128 scaled_weights_matrix(row, col) =
129 weights_matrix(row, col) * scale_values[col];
130 }
131 }
132 // Figure out the remaining bias to add on.
133 Tensor bias_offset(DT_FLOAT, {weights_cols});
134 auto bias_offset_vector = bias_offset.flat<float>();
135 for (int64 col = 0; col < weights_cols; ++col) {
136 bias_offset_vector(col) = offset_values[col];
137 }
138
139 // Construct the new nodes.
140 NodeDef scaled_weights_node;
141 scaled_weights_node.set_op("Const");
142 scaled_weights_node.set_name(weights_node.name());
143 SetNodeAttr("dtype", DT_FLOAT, &scaled_weights_node);
144 SetNodeTensorAttr<float>("value", scaled_weights, &scaled_weights_node);
145 new_nodes->push_back(scaled_weights_node);
146
147 // The input and convolution can be copied straight over, since the
148 // name of the scaled weights constant is the same as the original.
149 new_nodes->push_back(input_node);
150 new_nodes->push_back(conv_node);
151
152 NodeDef bias_offset_node;
153 bias_offset_node.set_op("Const");
154 bias_offset_node.set_name(conv_node.name() + "_bn_offset");
155 SetNodeAttr("dtype", DT_FLOAT, &bias_offset_node);
156 SetNodeTensorAttr<float>("value", bias_offset, &bias_offset_node);
157 new_nodes->push_back(bias_offset_node);
158
159 NodeDef bias_add_node;
160 bias_add_node.set_op("BiasAdd");
161 bias_add_node.set_name(conv_output_name);
162 CopyNodeAttr(conv_node, "T", "T", &bias_add_node);
163 AddNodeInput(conv_node.name(), &bias_add_node);
164 AddNodeInput(bias_offset_node.name(), &bias_add_node);
165 new_nodes->push_back(bias_add_node);
166 return Status::OK();
167}

169Status FuseBatchNormWithConv(const NodeMatch& match,
170 std::vector<NodeDef>* new_nodes) {
171 // Calculate the scale and offset values to apply.
172 std::vector<float> scale_values;
173 std::vector<float> offset_values;
174 TF_RETURN_IF_ERROR(
175 GetScaleAndOffsetValues(match, &scale_values, &offset_values));
176
177 // Fuse conv weights, and set the final output node name as batch_norm_node.
178 const NodeDef& batch_norm_node = match.node;
179 TF_RETURN_IF_ERROR(
180 FuseScaleOffsetToConvWeights(scale_values, offset_values, match.inputs[0],
181 batch_norm_node.name(), new_nodes));
182 return Status::OK();
183}

185Status FuseBatchNormWithBatchToSpace(const NodeMatch& match,
186 std::vector<NodeDef>* new_nodes) {
187 // Calculate the scale and offset values to apply.
188 std::vector<float> scale_values;
189 std::vector<float> offset_values;
190 TF_RETURN_IF_ERROR(
191 GetScaleAndOffsetValues(match, &scale_values, &offset_values));
192
193 // Fuse conv weights, and set the final output node name as batch_norm_node.
194 const NodeDef& batch_norm_node = match.node;
195 const NodeMatch& batch_to_space_node_match = match.inputs[0];
196 const NodeMatch& conv_node_match = batch_to_space_node_match.inputs[0];
197 const NodeDef& batch_to_space_node = batch_to_space_node_match.node;
198 const NodeDef& conv_node = conv_node_match.node;
199
200 string biasadd_name = conv_node.name() + "/biasadd";
201 TF_RETURN_IF_ERROR(
202 FuseScaleOffsetToConvWeights(scale_values, offset_values, conv_node_match,
203 biasadd_name , new_nodes));
204
205 NodeDef new_batch_to_space_node = batch_to_space_node;
206 // reuse batch_norm node name
207 new_batch_to_space_node.set_name(batch_norm_node.name());
208 new_batch_to_space_node.set_input(0, biasadd_name);
209 new_nodes->push_back(batch_to_space_node_match.inputs[1].node);
210 new_nodes->push_back(batch_to_space_node_match.inputs[2].node);
211 new_nodes->push_back(new_batch_to_space_node);
212 return Status::OK();
213}

215Status FuseBatchNormWithConvConcat(const NodeMatch& match,
216 std::vector<NodeDef>* new_nodes) {
217 // Calculate the scale and offset values to apply.
218 std::vector<float> scale_values;
219 std::vector<float> offset_values;
220 TF_RETURN_IF_ERROR(
221 GetScaleAndOffsetValues(match, &scale_values, &offset_values));
222
223 // Find all the nodes we expect in the subgraph.
224 const NodeDef& batch_norm_node = match.node;
225 const NodeMatch& concat_node_match = match.inputs[0];
226 NodeDef concat_node = concat_node_match.node;
227 CHECK_EQ("ConcatV2", concat_node.op());
228
229 // First process the axis.
230 NodeDef axis_node = concat_node_match.inputs[2].node;
231 CHECK_EQ("Const", axis_node.op());
232 Tensor axis = GetNodeTensorAttr(axis_node, "value");
233 int32 axis_scalar = (axis.scalar<int32>())();
234
235 // Set both conv0 and conv1 have the same scale and offset in default.
236 std::vector<float> scale0(scale_values);
237 std::vector<float> offset0(offset_values);
238 std::vector<float> scale1(scale_values);
239 std::vector<float> offset1(offset_values);
240 if (axis_scalar == 3) {
241 // If axis is 3, then scale and offset will be split into two halfs.
242 const NodeDef& weights0_node = concat_node_match.inputs[0].inputs[1].node;
243 Tensor weights0 = GetNodeTensorAttr(weights0_node, "value");
244 const int64 split_cols = weights0.shape().dim_size(3);
245 // Only keep the first half for scale0/offset0.
246 scale0.erase(scale0.begin() + split_cols, scale0.end());
247 offset0.erase(offset0.begin() + split_cols, offset0.end());
248 // Only keep the second half for scale1/offset1.
249 scale1.erase(scale1.begin(), scale1.begin() + split_cols);
250 offset1.erase(offset1.begin(), offset1.begin() + split_cols);
251 }
252
253 // Fuse the weights for input0 of conv2d.
254 const string concat0_output_name = concat_node.name() + "_bn_in0";
255 TF_RETURN_IF_ERROR(
256 FuseScaleOffsetToConvWeights(scale0, offset0, concat_node_match.inputs[0],
257 concat0_output_name, new_nodes));
258
259 // Fuse the weights for input1 of conv2d.
260 const string concat1_output_name = concat_node.name() + "_bn_in1";
261 TF_RETURN_IF_ERROR(
262 FuseScaleOffsetToConvWeights(scale1, offset1, concat_node_match.inputs[1],
263 concat1_output_name, new_nodes));
264
265 // Push the shape node.
266 new_nodes->push_back(concat_node_match.inputs[2].node);
267
268 // Set the final output op name to batch_normal_node.
269 concat_node.set_name(batch_norm_node.name());
270 concat_node.set_input(0, concat0_output_name);
271 concat_node.set_input(1, concat1_output_name);
272 new_nodes->push_back(concat_node);
273 return Status::OK();
274}

277// Finds monolithic batch norm ops (as used in early versions of TensorFlow) and
278// converts them into premultiplied weight inputs to convolutions.
279Status FoldOldBatchNorms(const GraphDef& input_graph_def,
280 const TransformFuncContext& context,
281 GraphDef* output_graph_def) {
282 GraphDef current_graph_def = input_graph_def;
283 // We have to do several passes to catch all the old BN nodes, since many of
284 // them may share inputs and so be excluded from replacement in one pass.
285 bool did_graph_change;
286 do {
287 did_graph_change = false;
288 GraphDef replaced_graph_def;
289 TF_RETURN_IF_ERROR(ReplaceMatchingOpTypes(
290 current_graph_def, // clang-format off
291 {"BatchNormWithGlobalNormalization|FusedBatchNorm", // batch_norm_node
292 {
293 {"Conv2D", // conv_node
294 {
295 {"*"}, // input_node
296 {"Const"}, // weights_node
297 }
298 },
299 {"Const"}, // mean_node
300 {"Const"}, // variance_node
301 {"Const"}, // beta_node
302 {"Const"}, // gamma_node
303 }
304 }, // clang-format on
305 [&did_graph_change](const NodeMatch& match,
306 const std::set<string>& input_nodes,
307 const std::set<string>& output_nodes,
308 std::vector<NodeDef>* new_nodes) {
309 TF_RETURN_IF_ERROR(FuseBatchNormWithConv(match, new_nodes));
310 did_graph_change = true;
311 return Status::OK();
312 },
313 {}, &replaced_graph_def));
314 current_graph_def = replaced_graph_def;
315 } while (did_graph_change);
316
317 do {
318 did_graph_change = false;
319 GraphDef replaced_graph_def;
320 TF_RETURN_IF_ERROR(ReplaceMatchingOpTypes(
321 current_graph_def, // clang-format off
322 {"BatchNormWithGlobalNormalization|FusedBatchNorm", // batch_norm_node
323 {
324 {"BatchToSpaceND", // batch_to_space_node
325 {
326 {"Conv2D", // conv_node
327 {
328 {"*"}, // input_node
329 {"Const"}, // weights_node
330 }
331 },
332 {"Const"}, // block_shape
333 {"Const"}, // crops
334 }
335 },
336 {"Const"}, // mean_node
337 {"Const"}, // variance_node
338 {"Const"}, // beta_node
339 {"Const"}, // gamma_node
340 }
341 }, // clang-format on
342 [&did_graph_change](const NodeMatch& match,
343 const std::set<string>& input_nodes,
344 const std::set<string>& output_nodes,
345 std::vector<NodeDef>* new_nodes) {
346 TF_RETURN_IF_ERROR(FuseBatchNormWithBatchToSpace(match, new_nodes));
347 did_graph_change = true;
348 return Status::OK();
349 },
350 {}, &replaced_graph_def));
351 current_graph_def = replaced_graph_def;
352 } while (did_graph_change);
353
354 do {
355 did_graph_change = false;
356 GraphDef replaced_graph_def;
357 // Replace BatchNorm with concat as input.
358 TF_RETURN_IF_ERROR(ReplaceMatchingOpTypes(
359 current_graph_def, // clang-format off
360 {"BatchNormWithGlobalNormalization|FusedBatchNorm", // batch_norm_node
361 {
362 {"ConcatV2|Concat", // concat two conv2d.
363 {
364 {"Conv2D", // conv_node
365 {
366 {"*"}, // input_node
367 {"Const"}, // weights_node
368 }
369 },
370 {"Conv2D", // conv_node
371 {
372 {"*"}, // input_node
373 {"Const"}, // weights_node
374 }
375 },
376 {"Const"}, // axis
377 },
378 },
379 {"Const"}, // mean_node
380 {"Const"}, // variance_node
381 {"Const"}, // beta_node
382 {"Const"}, // gamma_node
383 }
384 }, // clang-format on
385 [&did_graph_change](const NodeMatch& match,
386 const std::set<string>& input_nodes,
387 const std::set<string>& output_nodes,
388 std::vector<NodeDef>* new_nodes) {
389 TF_RETURN_IF_ERROR(FuseBatchNormWithConvConcat(match, new_nodes));
390 did_graph_change = true;
391 return Status::OK();
392 },
393 {}, &replaced_graph_def));
394 current_graph_def = replaced_graph_def;
395 } while (did_graph_change);
396
397 *output_graph_def = current_graph_def;
398 return Status::OK();
399}
