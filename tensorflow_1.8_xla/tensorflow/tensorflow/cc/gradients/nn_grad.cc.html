27Status SoftmaxGrad(const Scope& scope, const Operation& op,
28 const std::vector<Output>& grad_inputs,
29 std::vector<Output>* grad_outputs) {
30 // Softmax gradient function.
31 // p = softmax(x) maps from [batch, n] to [batch, m]
32 // dp/dx = [dp0/dx0 ... dp0/dxn-1 ]
33 // [ ... ... ]
34 // [dpm-1/dx0 ... dpm-1/dxn-1]
35 // dL/dx = dp/dx * dL/dy
36 //
37 // Using alternative formula:
38 // dL/dx = dL/dy * y - sum(dL/dy * y) * y
39 // = (dL/dy - sum(dL/dy * y)) * y
40 auto y = op.output(0);
41 auto dyy = Mul(scope, grad_inputs[0], y);
42 auto sum = Reshape(scope, Sum(scope, dyy, {1}), {-1, 1});
43 auto sub = Sub(scope, grad_inputs[0], sum);
44 auto dx = Mul(scope, sub, y);
45 grad_outputs->push_back(dx);
46 return scope.status();
47}
48REGISTER_GRADIENT_OP("Softmax", SoftmaxGrad);

50Status LogSoftmaxGrad(const Scope& scope, const Operation& op,
51 const std::vector<Output>& grad_inputs,
52 std::vector<Output>* grad_outputs) {
53 auto softmax = Exp(scope, op.output(0));
54 auto sum = Sum(scope, grad_inputs[0], {1}, Sum::KeepDims(true));
55 auto mul = Mul(scope, sum, softmax);
56 auto dx = Sub(scope, grad_inputs[0], mul);
57 grad_outputs->push_back(dx);
58 return scope.status();
59}
60REGISTER_GRADIENT_OP("LogSoftmax", LogSoftmaxGrad);

106Status BiasAddGradHelper(const Scope& scope, const Operation& op,
107 const std::vector<Output>& grad_inputs,
108 std::vector<Output>* grad_outputs) {
109 string data_format;
110 TF_RETURN_IF_ERROR(
111 GetNodeAttr(op.output(0).node()->attrs(), "data_format", &data_format));
112 auto dx_1 =
113 BiasAddGrad(scope, grad_inputs[0], BiasAddGrad::DataFormat(data_format));
114 grad_outputs->push_back(Identity(scope, grad_inputs[0]));
115 grad_outputs->push_back(dx_1);
116 return scope.status();
117}

120Status Conv2DGrad(const Scope& scope, const Operation& op,
121 const std::vector<Output>& grad_inputs,
122 std::vector<Output>* grad_outputs) {
123 string data_format;
124 string padding;
125 std::vector<int32> strides;
126 bool use_cudnn_on_gpu;
127 auto attrs = op.output(0).node()->attrs();
128 TF_RETURN_IF_ERROR(GetNodeAttr(attrs, "data_format", &data_format));
129 TF_RETURN_IF_ERROR(GetNodeAttr(attrs, "padding", &padding));
130 TF_RETURN_IF_ERROR(GetNodeAttr(attrs, "strides", &strides));
131 TF_RETURN_IF_ERROR(GetNodeAttr(attrs, "use_cudnn_on_gpu", &use_cudnn_on_gpu));
132 auto dx_1 = Conv2DBackpropInput(scope, Shape(scope, op.input(0)), op.input(1),
133 grad_inputs[0], strides, padding,
134 Conv2DBackpropInput::DataFormat(data_format)
135 .UseCudnnOnGpu(use_cudnn_on_gpu));
136 grad_outputs->push_back(dx_1);
137 auto dx_2 =
138 Conv2DBackpropFilter(scope, op.input(0), Shape(scope, op.input(1)),
139 grad_inputs[0], strides, padding,
140 Conv2DBackpropFilter::DataFormat(data_format)
141 .UseCudnnOnGpu(use_cudnn_on_gpu));
142 grad_outputs->push_back(dx_2);
143 return scope.status();
144}
145REGISTER_GRADIENT_OP("Conv2D", Conv2DGrad);
