18namespace tensorflow {
19namespace ops {
20
21// static
22GradOpRegistry* GradOpRegistry::Global() {
23 static GradOpRegistry* grad_op_registry = new GradOpRegistry;
24 return grad_op_registry;
25}
26
27bool GradOpRegistry::Register(const string& op, GradFunc func) {
28 CHECK(registry_.insert({op, func}).second) << "Existing gradient for " << op;
29 return true;
30}
31
32Status GradOpRegistry::Lookup(const string& op, GradFunc* func) const {
33 auto iter = registry_.find(op);
34 if (iter == registry_.end()) {
35 const string error_msg =
36 "No gradient defined for op: " + op +
37 ". Please see "
38 "https://www.tensorflow.org/code/"
39 "tensorflow/cc/gradients/README.md"
40 " for instructions on how to add C++ gradients.";
41 return errors::NotFound(error_msg);
42 }
43 *func = iter->second;
44 return Status::OK();
45}
