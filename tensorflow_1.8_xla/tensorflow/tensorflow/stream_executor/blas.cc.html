24string TransposeString(Transpose t) {
25 switch (t) {
26 case Transpose::kNoTranspose:
27 return "NoTranspose";
28 case Transpose::kTranspose:
29 return "Transpose";
30 case Transpose::kConjugateTranspose:
31 return "ConjugateTranspose";
32 default:
33 LOG(FATAL) << "Unknown transpose " << static_cast<int32>(t);
34 }
35}
36
37string UpperLowerString(UpperLower ul) {
38 switch (ul) {
39 case UpperLower::kUpper:
40 return "Upper";
41 case UpperLower::kLower:
42 return "Lower";
43 default:
44 LOG(FATAL) << "Unknown upperlower " << static_cast<int32>(ul);
45 }
46}

48string DiagonalString(Diagonal d) {

74string ComputationTypeString(ComputationType ty) {
75 switch (ty) {
76 case ComputationType::kF16:
77 return "f16";
78 case ComputationType::kF32:
79 return "f32";
80 case ComputationType::kF64:
81 return "f64";
82 case ComputationType::kI32:
83 return "i32";
84 case ComputationType::kComplexF32:
85 return "complex f32";
86 case ComputationType::kComplexF64:
87 return "complex f64";
88 default:
89 LOG(FATAL) << "Unknown ComputationType " << static_cast<int32>(ty);
90 }
91}
92
93std::ostream& operator<<(std::ostream& os, ComputationType ty) {
94 return os << ComputationTypeString(ty);
95}
