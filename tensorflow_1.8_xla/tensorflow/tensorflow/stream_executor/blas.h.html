16// Exposes the family of BLAS routines as pre-canned high performance calls for
17// use in conjunction with the StreamExecutor abstraction.
18//
19// Note that this interface is optionally supported by platforms; see
20// StreamExecutor::SupportsBlas() for details.
21//
22// This abstraction makes it simple to entrain BLAS operations on GPU data into
23// a Stream -- users typically will not use this API directly, but will use the
24// Stream builder methods to entrain these operations "under the hood". For
25// example:
26//
27// DeviceMemory<float> x = stream_exec->AllocateArray<float>(1024);
28// DeviceMemory<float> y = stream_exec->AllocateArray<float>(1024);
29// // ... populate x and y ...
30// Stream stream{stream_exec};
31// stream
32// .Init()
33// .ThenBlasAxpy(1024, 5.5, x, 1, &y, 1);
34// SE_CHECK_OK(stream.BlockHostUntilDone());
35//
36// By using stream operations in this manner the user can easily intermix custom
37// kernel launches (via StreamExecutor::ThenLaunch()) with these pre-canned BLAS
38// routines.

109// Opaque identifier for an "algorithm" used by a blas routine. This functions
110// as a hint to the blas library.
111typedef int64 AlgorithmType;
112constexpr AlgorithmType kDefaultAlgorithm = -1;
113constexpr AlgorithmType kDefaultBlasGemm = -2;
114constexpr AlgorithmType kDefaultBlasGemv = -3;
115constexpr AlgorithmType kNoAlgorithm = -4;
116
117// blas uses -1 to represent the default algorithm. This happens to match up
118// with the CUBLAS_GEMM_DFALT constant, so cuda_blas.cc is using static_cast
119// to convert from AlgorithmType to cublasGemmAlgo_t, and uses a static_assert
120// to ensure that this assumption does not break.
121// If another blas implementation uses a different value for the default
122// algorithm, then it needs to convert kDefaultGemmAlgo to that value
123// (e.g. via a function called ToWhateverGemmAlgo).
124constexpr AlgorithmType kDefaultGemmAlgo = -1;
125
126// Describes the result of a performance experiment, usually timing the speed of
127// a particular AlgorithmType.
128//
129// If the call we were benchmarking failed (a common occurrence; not all
130// algorithms are valid for all calls), is_valid() will be false.
131class ProfileResult {
132 public:
133 bool is_valid() const { return is_valid_; }
134 void set_is_valid(bool val) { is_valid_ = val; }
135 AlgorithmType algorithm() const { return algorithm_; }
136 void set_algorithm(AlgorithmType val) { algorithm_ = val; }
137 float elapsed_time_in_ms() const { return elapsed_time_in_ms_; }
138 void set_elapsed_time_in_ms(float val) { elapsed_time_in_ms_ = val; }
139
140 private:
141 bool is_valid_ = false;
142 AlgorithmType algorithm_ = kDefaultAlgorithm;
143 float elapsed_time_in_ms_ = std::numeric_limits<float>::max();
144};

146class AlgorithmConfig {
147 public:
148 AlgorithmConfig() : algorithm_(kDefaultAlgorithm) {}
149 explicit AlgorithmConfig(AlgorithmType algorithm) : algorithm_(algorithm) {}
150 AlgorithmType algorithm() const { return algorithm_; }
151 void set_algorithm(AlgorithmType val) { algorithm_ = val; }
152 bool operator==(const AlgorithmConfig &other) const {
153 return this->algorithm_ == other.algorithm_;
154 }
155 bool operator!=(const AlgorithmConfig &other) const {
156 return !(*this == other);
157 }
158 string ToString() const;
159
160 private:
161 AlgorithmType algorithm_;
162};
