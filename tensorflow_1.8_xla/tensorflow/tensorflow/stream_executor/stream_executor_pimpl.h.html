59// A StreamExecutor manages a single device, in terms of executing work (kernel
60// launches) and memory management (allocation/deallocation, memory copies to
61// and from the device). It is conceptually the "handle" for a device -- Stream
62// objects, which are used to enqueue work to run on the
63// coprocessor have a StreamExecutor instance as their "parent" object.
64//
65// StreamExecutor objects have an underlying platform that is specified up
66// front;
67// e.g. either it is a CUDA or OpenCL executor.
68//
69// Thread-safe after initialization.
70// StreamExecutor interface should not be invoked from a signal handler.
71class StreamExecutor {
72 public:
73 explicit StreamExecutor(PlatformKind kind,
74 const PluginConfig &plugin_config = PluginConfig());

145 // Allocate a memory region inside another allocated memory region.
146 // Offset and size are specified in terms of T elements.
147 // Warning: Do not free a parent buffer before its sub-buffers; this may cause
148 // use-after-free issues (the specific behavior is not consistent across
149 // platforms).
150 // - Note: OpenCL uses refcounting to manage buffer lifetimes, so use of a
151 // sub-buffer after parent deallocation is expected to be safe. This will
152 // render your code non-platform-portable, however.
153 template <typename T>
154 DeviceMemory<T> AllocateSubBuffer(DeviceMemory<T> *parent,
155 uint64 element_offset,
156 uint64 element_count);
157
158 // As AllocateSubBuffer(), but returns a ScopedDeviceMemory<T>.
159 template <typename T>
160 ScopedDeviceMemory<T> AllocateOwnedSubBuffer(DeviceMemory<T> *parent,
161 uint64 element_offset,
162 uint64 element_count) {
163 return ScopedDeviceMemory<T>(
164 this, AllocateSubBuffer<T>(parent, element_offset, element_count));
165 }

244 // Alternative interface for memcpying from host to device that takes an
245 // array slice. Checks that the destination size can accommodate the host
246 // slice size.
247 template <class T>
248 port::Status SynchronousMemcpyH2D(port::ArraySlice<T> host_src,
249 DeviceMemoryBase *device_dst) {
250 auto host_size = host_src.size() * sizeof(T);
251 CHECK(device_dst->size() == 0 || device_dst->size() >= host_size);
252 return SynchronousMemcpyH2D(host_src.begin(), host_size, device_dst);
253 }
254
255 // Same as SynchronousMemcpy(void*, ...) above.
256 port::Status SynchronousMemcpyD2H(const DeviceMemoryBase &device_src,
257 int64 size, void *host_dst);
258
259 // Alternative interface for memcpying from device to host that takes an
260 // array slice. Checks that the destination size can accommodate the host
261 // slice size.
262 template <typename T>
263 port::Status SynchronousMemcpyD2H(const DeviceMemory<T> &device_src,
264 port::MutableArraySlice<T> host_dst) {
265 auto host_size = host_dst.size() * sizeof(T);
266 CHECK(device_src.size() == 0 || host_size >= device_src.size());
267 return SynchronousMemcpyD2H(device_src, host_size, host_dst.begin());
268 }

634 // The device ordinal that this object was initialized with.
635 //
636 // Immutable post-initialization.
637 int device_ordinal_;
638
639 // Executor for handling host callback work that cannot be performed
640 // by a host callback thread - for example, cleanup after a host BLAS routine
641 // (which may make device API calls). This work cannot block the host
642 // callback thread, will be completed asynchronously, and should be treated
643 // as fire-and-forget. Assume no ordering guarantees WRT the tasks enqueued
644 // here.
645 //
646 // Immutable post-initialization. Object is thread-safe.
647 std::unique_ptr<port::ThreadPool> background_threads_;
648
649 // Counter for the current number of live streams. This is used to check
650 // for accidentally-outstanding streams at StreamExecutor teardown time, as
651 // well
652 // as to indicate leaks (via a large outstanding count being logged) in the
653 // case we can't allocate more streams.
654 std::atomic_int_fast32_t live_stream_count_;
655
656 // Only one worker thread is needed; little work will be done by the
657 // executor.
658 static const int kNumBackgroundThreads = 1;
659
660 // Indicates if StreamExecutor operation tracing should be performed.
661 bool tracing_enabled_;
662
663 // The set of TraceListeners registered for this StreamExecutor.
664 std::set<TraceListener*> listeners_ GUARDED_BY(mu_);
665
666 SE_DISALLOW_COPY_AND_ASSIGN(StreamExecutor);

679template <typename T>
680inline port::StatusOr<DeviceMemory<T>> StreamExecutor::GetSymbol(
681 const string &symbol_name) {
682 // If failed to get the symbol, opaque/bytes are unchanged. Initialize them to
683 // be nullptr/0 for consistency with DeviceMemory semantics.
684 void *opaque = nullptr;
685 size_t bytes = 0;
686 if (GetSymbol(symbol_name, &opaque, &bytes)) {
687 CHECK_EQ(bytes % sizeof(T), 0);
688 return DeviceMemory<T>::MakeFromByteSize(opaque, bytes);
689 }
690 return port::Status(
691 port::error::NOT_FOUND,
692 port::StrCat("Check if kernel using the symbol is loaded: ",
693 symbol_name));
694}

743template <typename T>
744DeviceMemory<T> StreamExecutor::AllocateZeroed() {
745 void *opaque = Allocate(sizeof(T));
746 if (opaque == nullptr) {
747 return DeviceMemory<T>{};
748 }
749
750 DeviceMemory<T> result = DeviceMemory<T>::MakeFromByteSize(opaque, sizeof(T));
751 bool ok = SynchronousMemZero(&result, sizeof(T));
752 if (!ok) {
753 Deallocate(&result);
754 return DeviceMemory<T>{};
755 }
756
757 return result;
758}

760template <typename T>
761DeviceMemory<T> StreamExecutor::AllocateSubBuffer(DeviceMemory<T> *parent,
762 uint64 element_offset,
763 uint64 element_count) {
764 if (element_offset + element_count > parent->ElementCount()) {
765 LOG(ERROR) << "requested sub-buffer allocation (offset + size) is greater "
766 << "than parent allocation size: (" << element_offset << " + "
767 << element_count << ") vs. (" << parent->ElementCount() << ")";
768 return DeviceMemory<T>{};
769 }
770
771 void *opaque = implementation_->AllocateSubBuffer(
772 parent, sizeof(T) * element_offset, sizeof(T) * element_count);
773 if (opaque == nullptr) {
774 return DeviceMemory<T>{};
775 }
776 CreateAllocRecord(opaque, sizeof(T) * element_count);
777 return DeviceMemory<T>(DeviceMemoryBase(opaque, sizeof(T) * element_count,
778 true /* = is_sub_buffer */));
779}

781template <typename... Params, typename... Args>
782inline Stream &Stream::ThenLaunch(ThreadDim thread_dims, BlockDim block_dims,
783 const TypedKernel<Params...> &kernel,
784 Args... args) {
785 KernelInvocationChecker<std::tuple<Params...>,
786 std::tuple<Args...>>::CheckAllStaticAssert();
787 if (ok()) {
788 // This is the core that allows type-safe kernel launching.
789 // Since the platforms take kernel arguments as tuples of (void *, size),
790 // we pack the variadic parameters passed as ...args into the desired
791 // tuple form and pass that packed form to the StreamExecutor::Launch()
792 // implementation.
793 KernelArgsArray<sizeof...(args)> kernel_args;
794 kernel.PackParams(&kernel_args, args...);
795 DCHECK(parent_ != nullptr);
796 bool ok =
797 parent_->Launch(this, thread_dims, block_dims, kernel, kernel_args);
798 if (!ok) {
799 SetError();
800 LOG(WARNING) << "parent failed to launch kernel: " << &kernel;
801 }
802 }
803 return *this;
804}
