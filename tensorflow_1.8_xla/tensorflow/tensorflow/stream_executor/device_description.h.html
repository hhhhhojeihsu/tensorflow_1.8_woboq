36// Data that describes the execution target of the StreamExecutor, in terms of
37// important logical parameters. These include dimensionality limits and
38// physical parameters of interest, such as number of cores present on the
39// device.
40//
41// Thread-safe: immutable post-initialization.
42class DeviceDescription {
43 public:
44 // Returns the platform being run on; this value is primarily intended for
45 // printing, and comes out something like "OpenCL 1.2" or "Compute Capability
46 // 3.5".
47 const string &platform_version() const { return platform_version_; }
48
49 // Returns the driver version interfacing with the underlying platform. Vendor
50 // dependent format.
51 const string &driver_version() const { return driver_version_; }
52
53 // Return the runtime version, if one is provided by the underlying platform.
54 // Vendor dependent format / usefulness.
55 const string &runtime_version() const { return runtime_version_; }
56
57 // Returns the name that the device reports. Vendor dependent.
58 const string &name() const { return name_; }
59
60 // Returns the PCI bus identifier for this device, of the form
61 // [domain]:[bus]:[device].[function]
62 const string &pci_bus_id() const { return pci_bus_id_; }
63
64 // Returns the NUMA node associated with this device, for use in
65 // determining socket locality. If the NUMA node could not be determined, -1
66 // is returned.
67 int numa_node() const { return numa_node_; }
68
69 // Number of cores (traditional notion of core; i.e. an SM on an NVIDIA device
70 // or an AMD Compute Unit.
71 int core_count() const { return core_count_; }
72
73 // Returns the limit on the thread dimensionality values in each of the
74 // respective dimensions. These limits affect what constitutes a legitimate
75 // kernel launch request.
76 const ThreadDim &thread_dim_limit() const { return thread_dim_limit_; }
77
78 // Returns the limit on the block dimensionality values in each of the
79 // respective dimensions. These limits may affect what constitutes a
80 // legitimate kernel launch request.
81 const BlockDim &block_dim_limit() const { return block_dim_limit_; }
82
83 // Returns the limit on the number of simultaneously resident blocks
84 // on a multiprocessor.
85 uint64 blocks_per_core_limit() const { return blocks_per_core_limit_; }
86
87 // Returns the limit on the total number of threads that can be launched in a
88 // single block; i.e. the limit on x * y * z dimensions of a ThreadDim.
89 // This limit affects what constitutes a legitimate kernel launch request.
90 const uint64 &threads_per_block_limit() const {
91 return threads_per_block_limit_;
92 }
93
94 // Returns the limit on the total number of threads that can be simultaneously
95 // launched on a given multiprocessor.
96 const uint64 &threads_per_core_limit() const {
97 return threads_per_core_limit_;
98 }
99
100 // Returns the number of threads per warp/wavefront.
101 const uint64 &threads_per_warp() const { return threads_per_warp_; }
102
103 // Returns the limit on the total number of registers per core.
104 const uint64 &registers_per_core_limit() const {
105 return registers_per_core_limit_;
106 }
107
108 // Returns the limit on the total number of registers that can be
109 // simultaneously used by a block.
110 const uint64 &registers_per_block_limit() const {
111 return registers_per_block_limit_;
112 }
113
114 // Returns the limit on the total number of registers that can be
115 // allocated to a thread.
116 const uint64 &registers_per_thread_limit() const {
117 return registers_per_thread_limit_;
118 }
119
120 // Returns the granularity at which warps are allocated resources.
121 const uint64 &warp_alloc_granularity() const {
122 return warp_alloc_granularity_;
123 }
124
125 // Returns the granularity at which registers are allocated to warps.
126 const uint64 &register_alloc_granularity() const {
127 return register_alloc_granularity_;
128 }
129
130 // Returns the granularity at which shared memory is allocated to warps.
131 const uint64 &shared_memory_alloc_granularity() const {
132 return shared_memory_alloc_granularity_;
133 }
134
135 // Returns the number of address bits available to kernel code running on the
136 // platform. This affects things like the maximum allocation size and perhaps
137 // types used in kernel code such as size_t.
138 const uint64 &device_address_bits() const { return device_address_bits_; }
139
140 // Returns the device memory size in bytes.
141 uint64 device_memory_size() const { return device_memory_size_; }
142
143 // Returns the device's memory bandwidth in bytes/sec. (This is for
144 // reads/writes to/from the device's own memory, not for transfers between the
145 // host and device.)
146 uint64 memory_bandwidth() const { return memory_bandwidth_; }
147
148 // Returns the device's core clock rate in GHz.
149 float clock_rate_ghz() const { return clock_rate_ghz_; }
150
151 // Returns whether ECC is enabled.
152 bool ecc_enabled() const { return ecc_enabled_; }
153
154 // Returns the device vendor string, e.g., "NVIDIA Corporation", "Advanced
155 // Micro Devices, Inc.", or "GenuineIntel".
156 const string &device_vendor() const { return device_vendor_; }
157
158 // Returns the CUDA compute capability if we're running on the CUDA platform.
159 // If a CUDA compute capability is not available, the major version will be
160 // zero, and the return value will be false.
161 bool cuda_compute_capability(int *major, int *minor) const;
162
163 // Returns the maximum amount of shared memory present on a single core
164 // (i.e. Streaming Multiprocessor on NVIDIA GPUs; Compute Unit for OpenCL
165 // devices). Note that some devices, such as NVIDIA's have a configurable
166 // partitioning between shared memory and L1 cache.
167 uint64 shared_memory_per_core() const { return shared_memory_per_core_; }
168
169 // Returns the maximum amount of shared memory available for a single block.
170 uint64 shared_memory_per_block() const { return shared_memory_per_block_; }
171
172 // TODO(leary): resident blocks per core will be useful.
173
174 // Convenience typedef for the string-based DeviceDescription mapping.
175 typedef std::map<string, string> Map;
176
177 // Returns a mapping from readable names to readable values that describe the
178 // device. This is useful for things like printing.
179 std::unique_ptr<Map> ToMap() const;
180
181 // For string values that are not available via the underlying platform, this
182 // value will be provided.
183 static const char *kUndefinedString;
184
185 private:
186 friend class internal::DeviceDescriptionBuilder;
187
188 DeviceDescription();
189
190 // For description of the following members, see the corresponding accessor
191 // above.
192 //
193 // N.B. If another field is added, update ToMap() above.
194 string device_vendor_;
195 string platform_version_;
196 string driver_version_;
197 string runtime_version_;
198 string pci_bus_id_;
199 string name_;
200
201 ThreadDim thread_dim_limit_;
202 BlockDim block_dim_limit_;
203
204 uint64 blocks_per_core_limit_;
205
206 uint64 threads_per_core_limit_;
207 uint64 threads_per_block_limit_;
208 uint64 threads_per_warp_;
209
210 uint64 registers_per_core_limit_;
211 uint64 registers_per_block_limit_;
212 uint64 registers_per_thread_limit_;
213
214 uint64 warp_alloc_granularity_;
215 uint64 register_alloc_granularity_;
216 uint64 shared_memory_alloc_granularity_;
217
218 uint64 device_address_bits_;
219 uint64 device_memory_size_;
220 uint64 memory_bandwidth_;
221
222 // Shared memory limits on a given device.
223 uint64 shared_memory_per_core_;
224 uint64 shared_memory_per_block_;
225
226 float clock_rate_ghz_;
227
228 // CUDA "CC" major value, -1 if not available.
229 int cuda_compute_capability_major_;
230 int cuda_compute_capability_minor_;
231
232 int numa_node_;
233 int core_count_;
234 bool ecc_enabled_;
235
236 SE_DISALLOW_COPY_AND_ASSIGN(DeviceDescription);
237};

241// Helper class the builds a device description, given that it has a large
242// number of fields that would be easily confused in constructor form.
243class DeviceDescriptionBuilder {
244 public:
245 DeviceDescriptionBuilder();
246
247 // For descriptions of the following fields, see comments on the corresponding
248 // DeviceDescription::* accessors above.
249
250 void set_device_vendor(const string &value) {
251 device_description_->device_vendor_ = value;
252 }
253 void set_platform_version(const string &value) {
254 device_description_->platform_version_ = value;
255 }
256 void set_driver_version(const string &value) {
257 device_description_->driver_version_ = value;
258 }
259 void set_runtime_version(const string &value) {
260 device_description_->runtime_version_ = value;
261 }
262 void set_pci_bus_id(const string &value) {
263 device_description_->pci_bus_id_ = value;
264 }
265 void set_name(const string &value) { device_description_->name_ = value; }
266
267 void set_thread_dim_limit(const ThreadDim &value) {
268 device_description_->thread_dim_limit_ = value;
269 }
270 void set_block_dim_limit(const BlockDim &value) {
271 device_description_->block_dim_limit_ = value;
272 }
273
274 void set_blocks_per_core_limit(uint64 value) {
275 device_description_->blocks_per_core_limit_ = value;
276 }
277
278 void set_threads_per_core_limit(uint64 value) {
279 device_description_->threads_per_core_limit_ = value;
280 }
281 void set_threads_per_block_limit(uint64 value) {
282 device_description_->threads_per_block_limit_ = value;
283 }
284 void set_threads_per_warp(uint64 value) {
285 device_description_->threads_per_warp_ = value;
286 }
287
288 void set_registers_per_core_limit(uint64 value) {
289 device_description_->registers_per_core_limit_ = value;
290 }
291 void set_registers_per_block_limit(uint64 value) {
292 device_description_->registers_per_block_limit_ = value;
293 }
294 void set_registers_per_thread_limit(uint64 value) {
295 device_description_->registers_per_thread_limit_ = value;
296 }
297
298 void set_warp_alloc_granularity(uint64 value) {
299 device_description_->warp_alloc_granularity_ = value;
300 }
301 void set_register_alloc_granularity(uint64 value) {
302 device_description_->register_alloc_granularity_ = value;

304 void set_shared_memory_alloc_granularity(uint64 value) {
305 device_description_->shared_memory_alloc_granularity_ = value;

307
308 void set_device_address_bits(uint64 value) {
309 device_description_->device_address_bits_ = value;
310 }
311 void set_device_memory_size(uint64 value) {
312 device_description_->device_memory_size_ = value;
313 }
314 void set_memory_bandwidth(uint64 value) {
315 device_description_->memory_bandwidth_ = value;
316 }
317
318 void set_shared_memory_per_core(int64 value) {
319 device_description_->shared_memory_per_core_ = value;
320 }
321 void set_shared_memory_per_block(int64 value) {
322 device_description_->shared_memory_per_block_ = value;
323 }
324
325 void set_clock_rate_ghz(float value) {
326 device_description_->clock_rate_ghz_ = value;
327 }
328
329 void set_cuda_compute_capability(int major, int minor) {
330 device_description_->cuda_compute_capability_major_ = major;
331 device_description_->cuda_compute_capability_minor_ = minor;
332 }
333
334 void set_numa_node(int value) { device_description_->numa_node_ = value; }
335 void set_core_count(int value) { device_description_->core_count_ = value; }
336 void set_ecc_enabled(bool value) {
337 device_description_->ecc_enabled_ = value;
338 }
339
340 // Returns a built DeviceDescription with ownership transferred to the
341 // caller. There are currently no restrictions on which fields must be set in
342 // order to build the descriptor.
343 //
344 // Once the description is built, this builder object should be discarded.
345 std::unique_ptr<DeviceDescription> Build() {
346 return std::move(device_description_);
347 }
348
349 private:
350 std::unique_ptr<DeviceDescription> device_description_;
351
352 SE_DISALLOW_COPY_AND_ASSIGN(DeviceDescriptionBuilder);
353};

357// Returns whether the given thread_dim is acceptable given the limits described
358// in device_description. For detailed reasons for failing the predicate, enable
359// VLOG(2) for this module.
360bool ThreadDimOk(const DeviceDescription &device_description,
361 const ThreadDim &thread_dim);
362
363// [deprecated] Use MathUtil::CeilOfRatio directly instead.
364//
365// Equivalent to ceil(double(element_count) / threads_per_block).
366uint64 DivideCeil(uint64 x, uint64 y);
367
368// Calculate the number of threads/blocks required to process element_count
369// elements. Note that you can still end up with more threads than
370// element_count due to rounding, so kernels often start with an "is this
371// thread id in the element_count range?" test.
372void CalculateDimensionality(const DeviceDescription &device_description,
373 uint64 element_count, uint64 *threads_per_block,
374 uint64 *block_count);
375
376// Compute and return maximum blocks per core (occupancy) based on the
377// device description, some kernel characteristics and the number of threads per
378// block. If unable to compute occupancy, zero is returned.
379uint64 CalculateOccupancy(const DeviceDescription &device_description,
380 uint64 registers_per_thread,
381 uint64 shared_memory_per_block,
382 const ThreadDim &thread_dims);
383
384// Compute and return the maximum number of registers per thread which
385// achieves the target occupancy. If the target is not possible then
386// zero is returned.
387uint64 CalculateRegisterLimitForTargetOccupancy(
388 const DeviceDescription &device_description, uint64 shared_memory_per_block,
389 const ThreadDim &thread_dims, uint64 target_blocks_per_core);
