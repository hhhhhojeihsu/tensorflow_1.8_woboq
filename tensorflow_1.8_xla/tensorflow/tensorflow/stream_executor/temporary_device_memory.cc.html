18#include "tensorflow/stream_executor/stream.h"

20namespace perftools {
21namespace gputools {
22
23TemporaryDeviceMemoryBase::~TemporaryDeviceMemoryBase() {
24 parent_->temporary_memory_manager()->MarkFinalized(device_memory_,
25 allocation_generation_,
26 /*must_exist=*/false);
27}
28
29DeviceMemoryBase* TemporaryDeviceMemoryBase::mutable_device_memory() {
30 DCHECK(!IsFinalized())
31 << "should not access device memory after finalization";
32 return &device_memory_;
33}
34
35const DeviceMemoryBase& TemporaryDeviceMemoryBase::device_memory() const {
36 DCHECK(!IsFinalized())
37 << "should not access device memory after finalization";
38 return device_memory_;
39}
40
41void TemporaryDeviceMemoryBase::Finalize() {
42 DCHECK(!IsFinalized()) << "should not finalize more than once";
43 parent_->temporary_memory_manager()->MarkFinalized(device_memory_,
44 allocation_generation_,
45 /*must_exist=*/true);
46}
47
48bool TemporaryDeviceMemoryBase::IsFinalized() const {
49 return parent_->temporary_memory_manager()->IsFinalized(
50 device_memory_, allocation_generation_);
51}
52
53bool TemporaryDeviceMemoryBase::IsAllocated() const {
54 return parent_->temporary_memory_manager()->HasAllocated(
55 device_memory_, allocation_generation_);
56}
57
58TemporaryDeviceMemoryBase::TemporaryDeviceMemoryBase(
59 Stream* parent, DeviceMemoryBase device_memory,
60 uint64 allocation_generation)
61 : device_memory_(device_memory),
62 allocation_generation_(allocation_generation),
63 parent_(parent) {
64 DCHECK(IsAllocated());
65}
