28string PlatformKindString(PlatformKind kind) {
29 switch (kind) {
30 case PlatformKind::kCuda:
31 return "CUDA";
32 case PlatformKind::kOpenCL:
33 return "OpenCL";
34 case PlatformKind::kHost:
35 return "Host";
36 case PlatformKind::kMock:
37 return "Mock";
38 default:
39 return port::StrCat("InvalidPlatformKind(", static_cast<int>(kind), ")");
40 }
41}
42
43PlatformKind PlatformKindFromString(string kind) {
44 for (int i = 0; i < static_cast<int>(PlatformKind::kSize); ++i) {
45 if (kind == PlatformKindString(static_cast<PlatformKind>(i))) {
46 return static_cast<PlatformKind>(i);
47 }
48 }
49
50 return PlatformKind::kInvalid;
51}

53bool PlatformIsRunnable(PlatformKind kind) {
54 switch (kind) {
55 case PlatformKind::kCuda:
56 case PlatformKind::kOpenCL:
57 case PlatformKind::kHost:
58 return true;
59 default:
60 return false;
61 }
62}

64bool PlatformIsRunnableOnDevice(PlatformKind kind) {
65 switch (kind) {
66 case PlatformKind::kCuda:
67 case PlatformKind::kOpenCL:
68 return true;
69 default:
70 return false;
71 }
72}

74void CheckPlatformKindIsValid(PlatformKind kind) {
75 CHECK(static_cast<int>(PlatformKind::kCuda) <= static_cast<int>(kind) &&
76 static_cast<int>(kind) <= static_cast<int>(PlatformKind::kMock))
77 << "invalid GPU executor kind: " << PlatformKindString(kind);
78}

104std::unique_ptr<Platform::PeerAccessMap> Platform::GetPeerAccessMap() {
105 auto *map = new PeerAccessMap;
106
107 int device_count = VisibleDeviceCount();
108 for (int i = 0; i < device_count; ++i) {
109 for (int j = 0; j < device_count; ++j) {
110 StreamExecutor *from = ExecutorForDevice(i).ValueOrDie();
111 StreamExecutor *to = ExecutorForDevice(j).ValueOrDie();
112 (*map)[{i, j}] = from->CanEnablePeerAccessTo(to);
113 }
114 }
115
116 return std::unique_ptr<Platform::PeerAccessMap>{map};
117}
118
119port::Status Platform::EnablePeerAccess() {
120 auto peer_access_map = GetPeerAccessMap();
121 for (const auto &access : *peer_access_map) {
122 auto devices = access.first;
123 if (access.second) {
124 StreamExecutor *from = ExecutorForDevice(devices.first).ValueOrDie();
125 StreamExecutor *to = ExecutorForDevice(devices.second).ValueOrDie();
126 auto status = from->EnablePeerAccessTo(to);
127 if (!status.ok()) {
128 return status;
129 }
130 } else {
131 LOG(INFO) << "cannot enable peer access from device ordinal "
132 << devices.first << " to device ordinal " << devices.second;
133 }
134 }
135 return port::Status::OK();
136}
