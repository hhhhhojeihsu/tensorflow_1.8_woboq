194// Describes the dimensions that a layer consumes/produces.
195//
196// This is a matrix (height, width), its "depth" (feature_map_count),
197// how many of these matrices are present (count),
198// and the maximum and minimum values expected in the matrix (value_max,
199// value_min).
200// If input is quantized, all values greater
201// than value_max will be clipped to value_max and all values less than
202// value_min will be clipped to value_min.
203// When quantized output is dequantized no value will be greater than
204// value_max or less than value_min.
205//
206// Uses the named argument construction form:
207//
208// auto input_batch_dimensions =
209// BatchDescriptor().set_count(42).set_feature_map_count(7)...
210//
211// Details:
212//
213// For a convolutional layer, a single inference takes a 3-dimensional matrix
214// of input and produces a 3-dimensional matrix of output. We call the three
215// dimensions height, width and feature_map_count, where for an image, the
216// height and width correspond to the Y and X pixel indices, respectively, and
217// the feature_map_count corresponds to the RGB dimension of the input data.
218// Then the count indicates how many 3D matrices are being presented to be
219// processed at once; this corresponds to the neural network concept of
220// minibatch size.
221//
222// For a fully connected layer, it's better to put the nodes of the layer in
223// the feature_map_count, and leave the height and weight as degenerate (== 1).
224// Count indicates how many input vectors (degenerate 3D matrices) are to be
225// processed.
226//
227// If unspecified, value_max and value_min default to 0.0.
228// If value_max == value_min the Stream will attempt to derive valid values -
229// for example the output of Relu6 activation will always be in the range
230// [0.0, 6.0].
231//
232// If unspecified, layout defaults to kYXDepthBatch.
233class BatchDescriptor {
234 public:
235 // Creates a "blank" batch descriptor, which should be initialized via the
236 // named argument helpers.
237 BatchDescriptor();
238 explicit BatchDescriptor(int ndims);
239
240 // Clones values from 'other' for initialization.
241 void CloneFrom(const BatchDescriptor& other);
242
243 string ToString() const;
244 string ToShortString() const;
245

247 int64 count() const { return count_; }
248 int64 feature_map_count() const { return feature_map_count_; }
249 int64 height() const { return GetDim(spatial_size_, DimIndex::Y); }
250 int64 width() const { return GetDim(spatial_size_, DimIndex::X); }
251 int64 spatial_dim(DimIndex dim) const { return GetDim(spatial_size_, dim); }
252 int ndims() const { return ndims_; }
253 float value_max() const { return value_max_; }
254 float value_min() const { return value_min_; }
255 DataLayout layout() const { return layout_; }
256 QuantizedActivationMode quantized_activation_mode() const {
257 return quantized_activation_mode_;
258 }
259 // Full dimensions of the underlying data, ordered according to a specific
260 // layout.
261 std::vector<int64> full_dims(const DataLayout& layout) const;
262
263 // Full strides of the underlying data, ordered according to a specific
264 // layout.
265 std::vector<int64> full_strides(const DataLayout& layout) const;

267 // Named-argument helpers for avoiding user error during construction.
268 BatchDescriptor& set_count(int64 value) {
269 count_ = value;
270 return *this;
271 }
272 BatchDescriptor& set_feature_map_count(int64 value) {

279 }
280 BatchDescriptor& set_width(int64 value) {
281 SetDim(&spatial_size_, DimIndex::X, value);

283 }
284 BatchDescriptor& set_spatial_dim(DimIndex dim, int64 value) {

287 }
288 BatchDescriptor& set_value_max(float value) {
289 value_max_ = value;
290 return *this;
291 }
292 BatchDescriptor& set_value_min(float value) {
293 value_min_ = value;
294 return *this;

296 BatchDescriptor& set_layout(DataLayout layout) {
297 layout_ = layout;
298 return *this;
299 }
300 BatchDescriptor& set_quantized_activation_mode(
301 QuantizedActivationMode quantized_activation_mode) {

471// Describes a convolution.
472//
473// Uses the named argument construction form:
474//
475// ConvolutionDescriptor convolution_dimensions;
476// convolution_dimensions
477// .set_vertical_filter_stride(2)
478// .set_horizontal_filter_stride(2)
479// ...
480//
481// Arguments:
482// - zero_padding_height: padding of the "y dimension" of the input data. Note
483// that this is different from the height of the filter.
484// - zero_padding_width: analogous to the height above, but in the "x
485// dimension".
486// - vertical_filter_stride: the convolution slides a 2-dimensional window of
487// filter-height-by-filter-width over the input layer -- the center of that
488// window is moved in the "y dimension" according to this stride value.
489// - horizontal_filter_stride: analogous to the vertical stride above, but in
490// the "x dimension".
491// - vertical_dilation_rate: there will be (vertical_dilation_rate - 1) skipped
492// cells between each filter element in the "y dimension".
493// - horizontal_dilation_rate: there will be (horizontal_dilation_rate - 1)
494// skipped cells between each filter element in the "x dimension".

496 public:
497 // By default construction, there is no zero-padding and the filter stride is

776// Describes a local response normalization (LRN). LRN is used e.g. in
777// dist_belief.
778//
779// Let V be the vector of feature maps at some (batch, y, x)
780// coordinate. LRN applies independently to each vector V in the
781// input, across all coordinates (batch, y, x), by mapping each V to
782// another vector U of the same size using the formula
783//
784// U_i = V_i / ((bias + alpha * (sum_j V_j^2)) ^ beta)
785//
786// where the sum is taken over j in the closed range [i - range, i + range].
787//
788// When calculating U_i the j in the sum can extend beyond the bounds
789// of V. If wrap_around is true, then V_j = V_{j mod F} where F is the
790// size of V, which is the number of feature maps. If wrap_around is
791// false, then V_j = 0 for j outside [0, F-1].
792//
793// If segment_size <= F, where F is the number of feature_maps, then
794// segment_size has no effect. Otherwise, each consecutive segment of
795// segment_size entries in V are normalized separately.
796//
797// Not all StreamExecutors allow wrap_around == true or segment_size
798// != 64. Some do not implement normalization at all.
799class NormalizeDescriptor {
800 public:
801 NormalizeDescriptor();
802
803 NormalizeDescriptor& set_bias(float bias) {
804 bias_ = bias;
805 return *this;
806 }
807
808 NormalizeDescriptor& set_range(int32 range) {
809 range_ = range;
810 return *this;
811 }
812
813 NormalizeDescriptor& set_alpha(float alpha) {
814 alpha_ = alpha;
815 return *this;
816 }
817
818 NormalizeDescriptor& set_beta(float beta) {
819 beta_ = beta;
820 return *this;
821 }
822
823 NormalizeDescriptor& set_wrap_around(bool wrap_around) {
824 wrap_around_ = wrap_around;
825 return *this;
826 }
827
828 NormalizeDescriptor& set_segment_size(int32 segment_size) {
829 segment_size_ = segment_size;
830 return *this;
831 }
832
833 void CloneFrom(const NormalizeDescriptor& other);
834
835 string ToString() const;
836 string ToShortString() const;
837
838 float bias() const { return bias_; }
839 int32 range() const { return range_; }
840 float alpha() const { return alpha_; }
841 float beta() const { return beta_; }
842 bool wrap_around() const { return wrap_around_; }
843 int32 segment_size() const { return segment_size_; }

845 private:

850 bool wrap_around_;

1410 // Fully connects the "nodes" (float values) in input_data with
1411 // shape input_dimensions to output_data with output_dimensions
1412 // using provided weights. This is equivalent to computing a matrix
1413 // product, hence the name MatMul.
1414 //
1415 // A BatchDescriptor has four dimensions: batch, y, x, depth. Matrix products
1416 // happen in two dimensions. To get down to two dimensions, we consider the
1417 // input y, x and depth dimension as one combined dimension T. For now,
1418 // assume that the output height and width are 1 and let OD be the output
1419 // depth.
1420 //
1421 // There are three device memory buffers passed in to this
1422 // function. We can now view all three as matrices:
1423 //
1424 // input_data: A batch x T matrix
1425 // weights: A T x OD matrix
1426 // output_data: A batch x OD matrix
1427 //
1428 // This function then computes the matrix product of input_data and
1429 // weights and writes the result into output_data.
1430 //
1431 // Here the weights buffer is in row major order, i.e. the first OD
1432 // entries in weights are the first row, the second OD entries in
1433 // weights are the second row and so on.
1434 //
1435 // The case for output width*height > 1 is more complicated. Let K =
1436 // OY * OX where OY is the output height and OX is the output
1437 // width. Then weights is divided into K sub-arrays W_i, for
1438 // i=0,...,k-1, that each represent a T x OD matrix. This function
1439 // then computes the K matrix multiplications of input_data with
1440 // each W_i. This creates K matrices with dimensions batch x
1441 // OD. These K matrices are concatenated horizontally to form one
1442 // larger matrix with dimensions batch x (K*OD); note that this is
1443 // not the same as concatenating the bytes of the matrices. The
1444 // combined matrix can then be interpreted as a tensor with
1445 // dimensions (batch, OY, OX, OD). If the output tensor format is
1446 // not kBatchYXDepth, this function would then need to arrange for
1447 // the output to be in the requested layout, if that is
1448 // supported. Note that the case K=1 is equivalent to the
1449 // description above. It is recommended to prefer the case K=1.
1450 //
1451 // Arguments (all borrowed):
1452 // stream: borrowed pointer to the stream that the 'fully connect' operation
1453 // should be enqueued onto.
1454 // output_data: un-owned device memory region in which to place the
1455 // fully connected result.
