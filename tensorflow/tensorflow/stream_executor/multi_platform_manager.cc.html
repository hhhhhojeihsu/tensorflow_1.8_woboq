26/* static */ mutex MultiPlatformManager::platforms_mutex_{LINKER_INITIALIZED};
27
28/* static */ port::StatusOr<Platform*> MultiPlatformManager::LookupByNameLocked(
29 const string& target) {
30 PlatformMap* platform_map = GetPlatformMap();
31 auto it = platform_map->find(port::Lowercase(target));
32 if (it == platform_map->end()) {
33 return port::Status(
34 port::error::NOT_FOUND,
35 "could not find registered platform with name: \"" + target + "\"");
36 }
37 return it->second;
38}
39
40/* static */ port::StatusOr<Platform*> MultiPlatformManager::LookupByIdLocked(
41 const Platform::Id& id) {
42 PlatformIdMap* platform_map = GetPlatformByIdMap();
43 auto it = platform_map->find(id);
44 if (it == platform_map->end()) {
45 return port::Status(
46 port::error::NOT_FOUND,
47 port::Printf("could not find registered platform with id: 0x%p", id));
48 }
49 return it->second;
50}
51
52/* static */ port::Status MultiPlatformManager::RegisterPlatform(
53 std::unique_ptr<Platform> platform) {
54 CHECK(platform != nullptr);
55 string key = port::Lowercase(platform->Name());
56 mutex_lock lock(platforms_mutex_);
57 if (GetPlatformMap()->find(key) != GetPlatformMap()->end()) {
58 return port::Status(port::error::INTERNAL,
59 "platform is already registered with name: \"" +
60 platform->Name() + "\"");
61 }
62 GetPlatformByIdMap()->insert(std::make_pair(platform->id(), platform.get()));
63 // Release ownership/uniqueness to prevent destruction on program exit.
64 // This avoids Platforms "cleaning up" on program exit, because otherwise,
65 // there are _very_ tricky races between StreamExecutor and underlying
66 // platforms (CUDA, OpenCL) during exit. Since these are fixed-size and 1x per
67 // program, these are deemed acceptable.
68 (*GetPlatformMap())[key] = platform.release();
69 return port::Status::OK();
70}

72/* static */ port::StatusOr<Platform*> MultiPlatformManager::PlatformWithName(
73 const string& target) {
74 mutex_lock lock(platforms_mutex_);
75
76 SE_ASSIGN_OR_RETURN(Platform * platform, LookupByNameLocked(target));
77 if (!platform->Initialized()) {
78 SE_RETURN_IF_ERROR(platform->Initialize({}));
79 }
80
81 return platform;
82}

84/* static */ port::StatusOr<Platform*> MultiPlatformManager::PlatformWithId(
85 const Platform::Id& id) {
86 mutex_lock lock(platforms_mutex_);
87
88 SE_ASSIGN_OR_RETURN(Platform * platform, LookupByIdLocked(id));
89 if (!platform->Initialized()) {
90 SE_RETURN_IF_ERROR(platform->Initialize({}));
91 }
92
93 return platform;
94}

96/* static */ port::StatusOr<Platform*>
97MultiPlatformManager::InitializePlatformWithName(
98 const string& target, const std::map<string, string>& options) {
99 mutex_lock lock(platforms_mutex_);
100
101 SE_ASSIGN_OR_RETURN(Platform * platform, LookupByNameLocked(target));
102 if (platform->Initialized()) {
103 return port::Status(port::error::FAILED_PRECONDITION,
104 "platform \"" + target + "\" is already initialized");
105 }
106
107 SE_RETURN_IF_ERROR(platform->Initialize(options));
108
109 return platform;
110}
111
112/* static */ port::StatusOr<Platform*>
113MultiPlatformManager::InitializePlatformWithId(
114 const Platform::Id& id, const std::map<string, string>& options) {
115 mutex_lock lock(platforms_mutex_);
116
117 SE_ASSIGN_OR_RETURN(Platform * platform, LookupByIdLocked(id));
118 if (platform->Initialized()) {
119 return port::Status(
120 port::error::FAILED_PRECONDITION,
121 port::Printf("platform with id 0x%p is already initialized", id));
122 }
123
124 SE_RETURN_IF_ERROR(platform->Initialize(options));
125
126 return platform;
127}
128
129/* static */ void MultiPlatformManager::ClearPlatformRegistry() {
130 mutex_lock lock(platforms_mutex_);
131 GetPlatformMap()->clear();
132 GetPlatformByIdMap()->clear();
133}
