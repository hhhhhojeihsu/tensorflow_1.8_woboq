46REGISTER_OP("OutfeedDequeue")
47 .Output("output: dtype")
48 .Attr("dtype: type")
49 .Attr("shape: shape")
50 .Attr("device_ordinal: int = -1")
51 .SetIsStateful()
52 .SetShapeFn(shape_inference::ExplicitShape)
53 .Doc(R"doc(
54Retrieves a single tensor from the computation outfeed. This operation will
55block indefinitely until data is available.
56
57output: A tensor that will be read from the device outfeed.
58dtype: The type of elements in the tensor.
59shape: The shape of the tensor.
60device_ordinal: The TPU device to use. This should be -1 when the Op
61is running on a TPU device, and >= 0 when the Op is running on the CPU
62device.
63)doc");
64
65REGISTER_OP("OutfeedDequeueTuple")
66 .Output("outputs: dtypes")
67 .Attr("dtypes: list(type)")
68 .Attr("shapes: list(shape)")
69 .Attr("device_ordinal: int = -1")
70 .SetIsStateful()
71 .SetShapeFn([](InferenceContext* c) {
72 std::vector<PartialTensorShape> shapes;
73 std::vector<DataType> dtypes;
74 TF_RETURN_IF_ERROR(c->GetAttr("shapes", &shapes));
75 TF_RETURN_IF_ERROR(c->GetAttr("dtypes", &dtypes));
76 if (shapes.size() != dtypes.size()) {
77 return errors::InvalidArgument(
78 "Incorrect number of output shapes specified");
79 }
80 for (int i = 0; i < shapes.size(); ++i) {
81 ShapeHandle out;
82 TF_RETURN_IF_ERROR(c->MakeShapeFromPartialTensorShape(shapes[i], &out));
83 c->set_output(i, out);
84 }
85 return Status::OK();
86 })
87 .Doc(R"doc(
88Retrieve multiple values that will be emitted by the computation as an XLA
89tuple. This operations will block indefinitely until data is available.
90Output `i` corresponds to XLA tuple element `i`.
91
92outputs: A list of tensors that will be read from the outfeed.
93dtypes: The element types of each element in `outputs`.
94shapes: The shapes of each tensor in `outputs`.
95device_ordinal: The TPU device to use. This should be -1 when the Op
96is running on a TPU device, and >= 0 when the Op is running on the CPU
97device.
98)doc");
