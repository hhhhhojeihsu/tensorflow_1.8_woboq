33#define CPUReduceSliceFunctorReduceop(reduceop, beginning) \
34 template <typename T, typename Index> \
35 struct ReduceSliceFunctor##reduceop<CPUDevice, T, Index> { \
36 private: \
37 struct XYZ { \
38 Index x, y, z; \
39 XYZ() = default; \
40 XYZ(Index x, Index y, Index z) : x(x), y(y), z(z) {} \
41 }; \
42 inline static XYZ global_index_to_xyz(Index global, XYZ size) { \
43 XYZ ret; \
44 ret.x = global / (size.y * size.z); \
45 ret.y = global % (size.y * size.z) / size.z; \
46 ret.z = global % size.z; \
47 return ret; \
48 } \
49 \
50 public: \
51 virtual ~ReduceSliceFunctor##reduceop() {} \
52 virtual void operator()(OpKernelContext* ctx, const CPUDevice& d, \
53 Index indices_width, \
54 typename TTypes<Index, 1>::ConstTensor indices, \
55 typename TTypes<T, 3>::ConstTensor data, \
56 typename TTypes<T, 3>::Tensor output) { \
57 Index bound = data.dimension(1); \
58 Index dim1 = output.dimension(0); \
59 Index dim2 = output.dimension(1); \
60 Index dim3 = output.dimension(2); \
61 Index size = dim1 * dim2 * dim3; \
62 if (size == 0) { \
63 return; \
64 } \
65 T zero = beginning<T>(); \
66 ThreadPool* thread_pool = \
67 ctx->device()->tensorflow_cpu_worker_threads()->workers; \
68 /* shard the work */ \
69 auto work = [&](Index start, Index end) { \
70 for (Index global = start; global < end; ++global) { \
71 XYZ xyz = global_index_to_xyz(global, XYZ(dim1, dim2, dim3)); \
72 Index x = xyz.x; \
73 Index y = xyz.y; \
74 Index z = xyz.z; \
75 output(x, y, z) = zero; \
76 Index slice_head = indices(y * indices_width); \
77 Index slice_end = std::min(indices(y * indices_width + 1), bound); \
78 for (Index i = slice_head; i < slice_end; ++i) { \
79 output(x, y, z) = reduceop(output(x, y, z), data(x, i, z)); \
80 } \
81 } \
82 }; \
83 /* Here assumes the number of average CPU cycles for each slice equals \
84 * the average length of each slice */ \
85 thread_pool->ParallelFor(size, std::max(bound / dim2, (Index)1), work); \
86 } \
87 };
88
89CALL_ALL_REDUCEOPS(CPUReduceSliceFunctorReduceop)
90#undef CPUReduceSliceFunctorReduceop

118template <typename Device, typename T, typename Index,
119 template <typename Device2, typename T2, typename Index2>
120 class Functor>
121class ReduceSliceKernel : public OpKernel {
122 public:
123 explicit ReduceSliceKernel(OpKernelConstruction* context)
124 : OpKernel(context) {}
125
126 void Compute(OpKernelContext* context) override {
127 const Tensor& data = context->input(0);
128 const Tensor& indices = context->input(1);
129 const Tensor& _axis = context->input(2);
130 int64 axis = _axis.scalar<int64>()();
131
132 int indices_width = 2;
133 int out_axis_dim_size = indices.shape().dim_size(0);
134 if (indices.dims() == 1 || indices.shape().dim_size(1) == 1) {
135 indices_width = 1;
136 if (out_axis_dim_size > 0) {
137 out_axis_dim_size--;
138 }
139 }
140
141 TensorShape output_shape = data.shape();
142 output_shape.set_dim(axis, out_axis_dim_size);
143 Tensor* output = nullptr;
144 OP_REQUIRES_OK(context, context->allocate_output(0, output_shape, &output));
145 auto functor = Functor<Device, T, Index>();
146 functor(context, context->eigen_device<Device>(), indices_width,
147 indices.flat<Index>(), data.flat_inner_outer_dims<T, 3>(axis - 1),
148 output->flat_inner_outer_dims<T, 3>(axis - 1));
149 }
150};
