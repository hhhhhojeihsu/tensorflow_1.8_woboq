35// SplitCandidate holds the split candidate node along with the stats.
36struct SplitCandidate {
37 // Id of handler that generated the split candidate.
38 int64 handler_id;
39
40 // Split gain.
41 float gain;
42
43 // Split info.
44 learner::SplitInfo split_info;
45};
46
47// Checks that the leaf is not empty.
48bool IsLeafWellFormed(const Leaf& leaf) {
49 return leaf.has_sparse_vector() || leaf.has_vector();
50}
51
52// Helper method to update the best split per partition given
53// a current candidate.
54void UpdateBestSplit(
55 const boosted_trees::learner::LearnerConfig& learner_config,
56 int32 partition_id, SplitCandidate* split,
57 std::map<int32, SplitCandidate>* best_splits) {
58 // Don't consider nodeless splits.
59 if (TF_PREDICT_FALSE(split->split_info.split_node().node_case() ==
60 TreeNode::NODE_NOT_SET)) {
61 return;
62 }
63
64 // Don't consider negative splits if we're pre-pruning the tree.
65 // Note that zero-gain splits are acceptable as they're mostly doing as well
66 // as what bias centering in that partition would do.
67 if (learner_config.pruning_mode() ==
68 boosted_trees::learner::LearnerConfig::PRE_PRUNE &&
69 split->gain < 0) {
70 return;
71 }
72
73 // If the current node is pure, one of the leafs will be empty, so the split
74 // is meaningless and we should not split.
75 if (!(IsLeafWellFormed(split->split_info.right_child()) &&
76 IsLeafWellFormed(split->split_info.left_child()))) {
77 VLOG(1) << "Split does not actually split anything";
78 return;
79 }
80
81 // Take the split if we don't have a candidate yet.
82 auto best_split_it = best_splits->find(partition_id);
83 if (best_split_it == best_splits->end()) {
84 best_splits->insert(std::make_pair(partition_id, std::move(*split)));
85 return;
86 }
87
88 // Determine if best split so far needs to be replaced.
89 SplitCandidate& best_split = best_split_it->second;
90 if (TF_PREDICT_FALSE(split->gain == best_split.gain)) {
91 // Tie break on node case preferring simpler tree node types.
92 VLOG(2) << "Attempting to tie break with smaller node case. "
93 << "(current split: " << split->split_info.split_node().node_case()
94 << ", best split: "
95 << best_split.split_info.split_node().node_case() << ")";
96 if (split->split_info.split_node().node_case() <
97 best_split.split_info.split_node().node_case()) {
98 best_split = std::move(*split);
99 } else if (split->split_info.split_node().node_case() ==
100 best_split.split_info.split_node().node_case()) {
101 // Tie break on handler Id.
102 VLOG(2) << "Tie breaking with higher handler Id. "
103 << "(current split: " << split->handler_id
104 << ", best split: " << best_split.handler_id << ")";
105 if (split->handler_id > best_split.handler_id) {
106 best_split = std::move(*split);
107 }
108 }
109 } else if (split->gain > best_split.gain) {
110 best_split = std::move(*split);
111 }
112}

114// Helper method to check whether a node is a terminal node in that it
115// only has leaf nodes as children.
116bool IsTerminalSplitNode(const size_t node_id,
117 const std::vector<int32>& children,
118 const std::vector<TreeNode>& nodes) {
119 for (const int32 child_id : children) {
120 const auto& child_node = nodes[child_id];
121 CHECK(child_node.node_case() != TreeNode::NODE_NOT_SET);
122 if (child_node.node_case() != TreeNode::kLeaf) {
123 return false;
124 }
125 }
126 return true;
127}

129// Helper method to recursively prune the tree in a depth-first fashion.
130void RecursivePruneTree(const size_t node_id, std::vector<TreeNode>* nodes) {
131 // Base case when we reach a leaf.
132 TreeNode& tree_node = (*nodes)[node_id];
133 CHECK(tree_node.node_case() != TreeNode::NODE_NOT_SET);
134 if (tree_node.node_case() == TreeNode::kLeaf) {
135 return;
136 }
137
138 // Traverse node children first and recursively prune their sub-trees.
139 const std::vector<int32> children =
140 boosted_trees::trees::DecisionTree::GetChildren(tree_node);
141 for (const int32 child_id : children) {
142 RecursivePruneTree(child_id, nodes);
143 }
144
145 // Two conditions must be satisfied to prune the node:
146 // 1- The split gain is negative.
147 // 2- After depth-first pruning, the node only has leaf children.
148 TreeNodeMetadata* node_metadata = tree_node.mutable_node_metadata();
149 if (node_metadata->gain() < 0 &&
150 IsTerminalSplitNode(node_id, children, (*nodes))) {
151 // Clear node children.
152 for (const int32 child_id : children) {
153 auto& child_node = (*nodes)[child_id];
154 child_node.Clear();
155 }
156
157 // Change node back into leaf.
158 (*tree_node.mutable_leaf()) = *node_metadata->mutable_original_leaf();
159
160 // Clear gain for leaf node.
161 tree_node.clear_node_metadata();
162 } else {
163 // Clear original leaf as it's no longer needed for back-track pruning.
164 node_metadata->clear_original_leaf();
165 }
166}

170class CenterTreeEnsembleBiasOp : public OpKernel {
171 public:
172 explicit CenterTreeEnsembleBiasOp(OpKernelConstruction* const context)
173 : OpKernel(context) {
174 // Read learner config.
175 string serialized_learner_config;
176 OP_REQUIRES_OK(context, context->GetAttr("learner_config",
177 &serialized_learner_config));
178 OP_REQUIRES(context,
179 learner_config_.ParseFromString(serialized_learner_config),
180 errors::InvalidArgument("Unable to parse learner config."));
181
182 // Read centering epsilon.
183 OP_REQUIRES_OK(context,
184 context->GetAttr("centering_epsilon", &centering_epsilon_));
185 }
186
187 void Compute(OpKernelContext* const context) override {
188 // Get decision tree ensemble.
189 boosted_trees::models::DecisionTreeEnsembleResource* ensemble_resource;
190 OP_REQUIRES_OK(context, LookupResource(context, HandleFromInput(context, 0),
191 &ensemble_resource));
192 core::ScopedUnref unref_me(ensemble_resource);
193 mutex_lock l(*ensemble_resource->get_mutex());
194
195 // Get the stamp token.
196 const Tensor* stamp_token_t;
197 OP_REQUIRES_OK(context, context->input("stamp_token", &stamp_token_t));
198 int64 stamp_token = stamp_token_t->scalar<int64>()();
199
200 // Only the Chief should run this Op and it is guaranteed to be in
201 // a consistent state so the stamps must always match.
202 CHECK(ensemble_resource->is_stamp_valid(stamp_token));
203
204 // Get the next stamp token.
205 const Tensor* next_stamp_token_t;
206 OP_REQUIRES_OK(context,
207 context->input("next_stamp_token", &next_stamp_token_t));
208 int64 next_stamp_token = next_stamp_token_t->scalar<int64>()();
209 CHECK(stamp_token != next_stamp_token);
210
211 // Update the ensemble stamp.
212 ensemble_resource->set_stamp(next_stamp_token);
213
214 // Get the delta updates.
215 const Tensor* delta_updates_t;
216 OP_REQUIRES_OK(context, context->input("delta_updates", &delta_updates_t));
217 auto delta_updates = delta_updates_t->vec<float>();
218 const int64 logits_dimension = delta_updates_t->dim_size(0);
219
220 // Get the bias.
221 boosted_trees::trees::Leaf* const bias =
222 RetrieveBias(ensemble_resource, logits_dimension);
223 CHECK(bias->has_vector());
224
225 // Update the bias.
226 float total_delta = 0;
227 auto* bias_vec = bias->mutable_vector();
228 for (size_t idx = 0; idx < bias->vector().value_size(); ++idx) {
229 float delta = delta_updates(idx);
230 bias_vec->set_value(idx, bias_vec->value(idx) + delta);
231 total_delta += std::abs(delta);
232 }
233
234 // Make a centering continuation decision based on current update.
235 bool continue_centering = total_delta > centering_epsilon_;
236 if (continue_centering) {
237 VLOG(1) << "Continuing to center bias, delta=" << total_delta;
238 } else {
239 VLOG(1) << "Done centering bias, delta=" << total_delta;
240 ensemble_resource->LastTreeMetadata()->set_is_finalized(true);
241 }
242 Tensor* continue_centering_t = nullptr;
243 OP_REQUIRES_OK(
244 context, context->allocate_output("continue_centering", TensorShape({}),
245 &continue_centering_t));
246 continue_centering_t->scalar<bool>()() = continue_centering;
247 }
248
249 private:
250 // Helper method to retrieve the bias from the tree ensemble.
251 boosted_trees::trees::Leaf* RetrieveBias(
252 boosted_trees::models::DecisionTreeEnsembleResource* ensemble_resource,
253 int64 logits_dimension) {
254 const int32 num_trees = ensemble_resource->num_trees();
255 if (num_trees <= 0) {
256 // Add a new bias leaf.
257 ensemble_resource->IncrementAttempts();
258 boosted_trees::trees::DecisionTreeConfig* const tree_config =
259 ensemble_resource->AddNewTree(1.0);
260 auto* const leaf = tree_config->add_nodes()->mutable_leaf();
261 for (size_t idx = 0; idx < logits_dimension; ++idx) {
262 leaf->mutable_vector()->add_value(0.0);
263 }
264 return leaf;
265 } else if (num_trees == 1) {
266 // Confirms that the only tree is a bias and returns its leaf.
267 boosted_trees::trees::DecisionTreeConfig* const tree_config =
268 ensemble_resource->LastTree();
269 CHECK_EQ(tree_config->nodes_size(), 1);
270 CHECK_EQ(tree_config->nodes(0).node_case(), TreeNode::kLeaf);
271 return tree_config->mutable_nodes(0)->mutable_leaf();
272 } else {
273 LOG(FATAL) << "Unable to center bias on an already grown ensemble";
274 }
275 }
276
277 boosted_trees::learner::LearnerConfig learner_config_;
278 float centering_epsilon_;
279};

284class GrowTreeEnsembleOp : public OpKernel {
285 public:
286 explicit GrowTreeEnsembleOp(OpKernelConstruction* const context)
287 : OpKernel(context) {
288 // Read number of handlers, note that this is the static number of
289 // all handlers but any subset of these handlers may be active at a time.
290 OP_REQUIRES_OK(context, context->GetAttr("num_handlers", &num_handlers_));
291
292 OP_REQUIRES_OK(context, context->GetAttr("center_bias", &center_bias_));
293
294 // Read learner config.
295 string serialized_learner_config;
296 OP_REQUIRES_OK(context, context->GetAttr("learner_config",
297 &serialized_learner_config));
298 OP_REQUIRES(context,
299 learner_config_.ParseFromString(serialized_learner_config),
300 errors::InvalidArgument("Unable to parse learner config."));
301
302 // Determine whether dropout was used when building this tree.
303 if (learner_config_.has_learning_rate_tuner() &&
304 learner_config_.learning_rate_tuner().tuner_case() ==
305 LearningRateConfig::kDropout) {
306 dropout_config_ = learner_config_.learning_rate_tuner().dropout();
307 dropout_was_applied_ = true;
308 } else {
309 dropout_was_applied_ = false;
310 }
311 }
312
313 void Compute(OpKernelContext* const context) override {
314 // Get decision tree ensemble.
315 boosted_trees::models::DecisionTreeEnsembleResource* ensemble_resource;
316 OP_REQUIRES_OK(context, LookupResource(context, HandleFromInput(context, 0),
317 &ensemble_resource));
318 core::ScopedUnref unref_me(ensemble_resource);
319 mutex_lock l(*ensemble_resource->get_mutex());
320
321 // Get the stamp token.
322 const Tensor* stamp_token_t;
323 OP_REQUIRES_OK(context, context->input("stamp_token", &stamp_token_t));
324 int64 stamp_token = stamp_token_t->scalar<int64>()();
325
326 // Only the Chief should run this Op and it is guaranteed to be in
327 // a consistent state so the stamps must always match.
328 CHECK(ensemble_resource->is_stamp_valid(stamp_token));
329
330 // Get the next stamp token.
331 const Tensor* next_stamp_token_t;
332 OP_REQUIRES_OK(context,
333 context->input("next_stamp_token", &next_stamp_token_t));
334 int64 next_stamp_token = next_stamp_token_t->scalar<int64>()();
335 CHECK(stamp_token != next_stamp_token);
336
337 // Update the ensemble stamp regardless of whether a layer
338 // or tree is actually grown.
339 ensemble_resource->set_stamp(next_stamp_token);
340
341 // Read the learning_rate.
342 const Tensor* learning_rate_t;
343 OP_REQUIRES_OK(context, context->input("learning_rate", &learning_rate_t));
344 float learning_rate = learning_rate_t->scalar<float>()();
345
346 // Read seed that was used for dropout.
347 const Tensor* seed_t;
348 OP_REQUIRES_OK(context, context->input("dropout_seed", &seed_t));
349 // Cast seed to uint64.
350 const uint64 dropout_seed = seed_t->scalar<int64>()();
351
352 // Read partition Ids, gains and split candidates.
353 OpInputList partition_ids_list;
354 OpInputList gains_list;
355 OpInputList splits_list;
356 OP_REQUIRES_OK(context,
357 context->input_list("partition_ids", &partition_ids_list));
358 OP_REQUIRES_OK(context, context->input_list("gains", &gains_list));
359 OP_REQUIRES_OK(context, context->input_list("splits", &splits_list));
360
361 // Increment attempt stats.
362 ensemble_resource->IncrementAttempts();
363
364 // Find best splits for each active partition.
365 std::map<int32, SplitCandidate> best_splits;
366 FindBestSplitsPerPartition(context, partition_ids_list, gains_list,
367 splits_list, &best_splits);
368
369 // No-op if no new splits can be considered.
370 if (best_splits.empty()) {
371 LOG(WARNING) << "Not growing tree ensemble as no good splits were found.";
372 return;
373 }
374
375 // Update and retrieve the growable tree.
376 // If the tree is fully built and dropout was applied, it also adjusts the
377 // weights of dropped and the last tree.
378 boosted_trees::trees::DecisionTreeConfig* const tree_config =
379 UpdateAndRetrieveGrowableTree(ensemble_resource, learning_rate,
380 dropout_seed);
381
382 // Split tree nodes.
383 for (auto& split_entry : best_splits) {
384 SplitTreeNode(split_entry.first, &split_entry.second, tree_config,
385 ensemble_resource);
386 }
387
388 // Post-prune finalized tree if needed.
389 if (learner_config_.pruning_mode() ==
390 boosted_trees::learner::LearnerConfig::POST_PRUNE &&
391 ensemble_resource->LastTreeMetadata()->is_finalized()) {
392 VLOG(2) << "Post-pruning finalized tree.";
393 PruneTree(tree_config);
394
395 // If after post-pruning the whole tree has no gain, remove the tree
396 // altogether from the ensemble.
397 if (tree_config->nodes_size() <= 0) {
398 ensemble_resource->RemoveLastTree();
399 }
400 }
401 }
402
403 private:
404 // Helper method which effectively does a reduce over all split candidates
405 // and finds the best split for each partition.
406 void FindBestSplitsPerPartition(
407 OpKernelContext* const context,
408 const OpInputList& partition_ids_list, const OpInputList& gains_list,
409 const OpInputList& splits_list,
410 std::map<int32, SplitCandidate>* best_splits) {
411 // Find best split per partition going through every feature candidate.
412 // TODO(salehay): Is this worth parallelizing?
413 for (int64 handler_id = 0; handler_id < num_handlers_; ++handler_id) {
414 const auto& partition_ids = partition_ids_list[handler_id].vec<int32>();
415 const auto& gains = gains_list[handler_id].vec<float>();
416 const auto& splits = splits_list[handler_id].vec<string>();
417 OP_REQUIRES(context, partition_ids.size() == gains.size(),
418 errors::InvalidArgument(
419 "Inconsistent partition Ids and gains tensors: ",
420 partition_ids.size(), " != ", gains.size()));
421 OP_REQUIRES(context, partition_ids.size() == splits.size(),
422 errors::InvalidArgument(
423 "Inconsistent partition Ids and splits tensors: ",
424 partition_ids.size(), " != ", splits.size()));
425 for (size_t candidate_idx = 0; candidate_idx < splits.size();
426 ++candidate_idx) {
427 // Get current split candidate.
428 const auto& partition_id = partition_ids(candidate_idx);
429 const auto& gain = gains(candidate_idx);
430 const auto& serialized_split = splits(candidate_idx);
431 SplitCandidate split;
432 split.handler_id = handler_id;
433 split.gain = gain;
434 OP_REQUIRES(context, split.split_info.ParseFromString(serialized_split),
435 errors::InvalidArgument("Unable to parse split info."));
436
437 // Update best split for partition based on the current candidate.
438 UpdateBestSplit(learner_config_, partition_id, &split, best_splits);
439 }
440 }
441 }

443 void UpdateTreeWeightsIfDropout(
444 boosted_trees::models::DecisionTreeEnsembleResource* const
445 ensemble_resource,
446 const uint64 dropout_seed) {
447 // It is possible that the tree was built with dropout. If it is the case,
448 // we need to adjust the tree weight, or bail out.
449 if (!dropout_was_applied_ ||
450 !ensemble_resource->LastTreeMetadata()->is_finalized()) {
451 return;
452 }
453 const int32 num_trees = ensemble_resource->num_trees();
454
455 // Based on seed, figure out what trees were dropped before.
456 std::unordered_set<int32> trees_not_to_drop;
457 if (center_bias_) {
458 trees_not_to_drop.insert(0);
459 }
460 // Last tree is the current tree that is built.
461 const int32 current_tree = num_trees - 1;
462 trees_not_to_drop.insert(current_tree);
463
464 // Since only chief builds the trees, we are sure that the other tree
465 // weights didn't change.
466 std::vector<float> weights = ensemble_resource->GetTreeWeights();
467 std::vector<int32> dropped_trees;
468 std::vector<float> dropped_trees_weights;
469 const auto dropout_status = DropoutUtils::DropOutTrees(
470 dropout_seed, dropout_config_, trees_not_to_drop, weights,
471 &dropped_trees, &dropped_trees_weights);
472 CHECK(dropout_status.ok())
473 << "Can't figure out what trees were dropped out before, error is "
474 << dropout_status.error_message();
475
476 // Now we have dropped trees, update their weights and the current tree
477 // weight.
478 if (!dropped_trees.empty()) {
479 std::vector<int32> increment_num_updates(num_trees, 0);
480 DropoutUtils::GetTreesWeightsForAddingTrees(
481 dropped_trees, dropped_trees_weights, current_tree,
482 1 /* only 1 tree was added */, &weights, &increment_num_updates);
483
484 // Update the weights and num of updates for trees.
485 for (int i = 0; i < num_trees; ++i) {
486 ensemble_resource->SetTreeWeight(i, weights[i],
487 increment_num_updates[i]);
488 }
489 }
490 }

492 // Helper method to update the growable tree which is by definition the last
493 // tree in the ensemble.
494 boosted_trees::trees::DecisionTreeConfig* UpdateAndRetrieveGrowableTree(
495 boosted_trees::models::DecisionTreeEnsembleResource* const
496 ensemble_resource,
497 const float learning_rate, const uint64 dropout_seed) {
498 const auto num_trees = ensemble_resource->num_trees();
499 if (num_trees <= 0 ||
500 ensemble_resource->LastTreeMetadata()->is_finalized()) {
501 // Create a new tree with a no-op leaf.
502 boosted_trees::trees::DecisionTreeConfig* const tree_config =
503 ensemble_resource->AddNewTree(learning_rate);
504 VLOG(1) << "Adding layer #0 to tree #" << num_trees << " of ensemble of "
505 << num_trees + 1 << " trees.";
506 tree_config->add_nodes()->mutable_leaf();
507 boosted_trees::trees::DecisionTreeMetadata* const tree_metadata =
508 ensemble_resource->LastTreeMetadata();
509 tree_metadata->set_is_finalized(
510 learner_config_.constraints().max_tree_depth() <= 1);
511 tree_metadata->set_num_tree_weight_updates(1);
512 } else {
513 // The growable tree is by definition the last tree in the ensemble.
514 boosted_trees::trees::DecisionTreeMetadata* const tree_metadata =
515 ensemble_resource->LastTreeMetadata();
516 const auto new_num_layers = tree_metadata->num_layers_grown() + 1;
517 VLOG(1) << "Adding layer #" << new_num_layers - 1 << " to tree #"
518 << num_trees - 1 << " of ensemble of " << num_trees << " trees.";
519 // Update growable tree metadata.
520 tree_metadata->set_num_layers_grown(new_num_layers);
521 tree_metadata->set_is_finalized(
522 new_num_layers >= learner_config_.constraints().max_tree_depth());
523 }
524 UpdateTreeWeightsIfDropout(ensemble_resource, dropout_seed);
525 return ensemble_resource->LastTree();
526 }

528 // Helper method to merge leaf weights as the tree is being grown.
529 boosted_trees::trees::Leaf* MergeLeafWeights(
530 const boosted_trees::trees::Leaf& source,
531 boosted_trees::trees::Leaf* dest) {
532 // Resolve leaf merging method based on how the trees are being grown.
533 if (learner_config_.growing_mode() ==
534 boosted_trees::learner::LearnerConfig::WHOLE_TREE) {
535 // No merging occurs when building a whole tree at a time.
536 return dest;
537 }
538
539 // Handle leaf merging based on type.
540 switch (source.leaf_case()) {
541 case boosted_trees::trees::Leaf::kVector: {
542 // No-op if source is empty
543 const auto& src_vec = source.vector();
544 if (src_vec.value_size() == 0) {
545 break;
546 }
547 CHECK(source.leaf_case() == dest->leaf_case());
548
549 // Dense add leaf vectors.
550 auto* dst_vec = dest->mutable_vector();
551 CHECK(src_vec.value_size() == dst_vec->value_size());
552 for (size_t idx = 0; idx < source.vector().value_size(); ++idx) {
553 (*dst_vec->mutable_value()->Mutable(idx)) += src_vec.value(idx);
554 }
555 break;
556 }
557 case boosted_trees::trees::Leaf::kSparseVector: {
558 // No-op if source is empty
559 const auto& src_vec = source.sparse_vector();
560 CHECK(src_vec.value_size() == src_vec.index_size());
561 if (src_vec.value_size() == 0) {
562 break;
563 }
564 CHECK(source.leaf_case() == dest->leaf_case());
565
566 // Get mapping of dimension to value for destination.
567 std::unordered_map<int32, float> dst_map;
568 auto* dst_vec = dest->mutable_sparse_vector();
569 CHECK(dst_vec->value_size() == dst_vec->index_size());
570 dst_map.reserve(dst_vec->value_size());
571 for (size_t idx = 0; idx < dst_vec->value_size(); ++idx) {
572 dst_map[dst_vec->index(idx)] = dst_vec->value(idx);
573 }
574 // Sparse add source vector to destination vector.
575 for (size_t idx = 0; idx < src_vec.value_size(); ++idx) {
576 dst_map[src_vec.index(idx)] += src_vec.value(idx);
577 }
578 // Rebuild merged destination leaf.
579 dst_vec->clear_index();
580 dst_vec->clear_value();
581 for (const auto& entry : dst_map) {
582 dst_vec->add_index(entry.first);
583 dst_vec->add_value(entry.second);
584 }
585 break;
586 }
587 case boosted_trees::trees::Leaf::LEAF_NOT_SET: {
588 // No-op as there is nothing to merge.
589 break;
590 }
591 }
592 return dest;
593 }

595 // Helper method to split a tree node and append its respective
596 // leaf children given the split candidate.
597 void SplitTreeNode(
598 const int32 node_id, SplitCandidate* split,
599 boosted_trees::trees::DecisionTreeConfig* tree_config,
600 boosted_trees::models::DecisionTreeEnsembleResource* ensemble_resource) {
601 // No-op if we have no real node.
602 CHECK(node_id < tree_config->nodes_size())
603 << "Invalid node " << node_id << " to split.";
604 // Ensure new split node is valid.
605 CHECK(split->split_info.split_node().node_case() != TreeNode::NODE_NOT_SET);
606 CHECK(tree_config->nodes(node_id).node_case() == TreeNode::kLeaf)
607 << "Unexpected node type to split "
608 << tree_config->nodes(node_id).node_case() << " for node_id " << node_id
609 << ". Tree config: " << tree_config->DebugString();
610
611 // Add left leaf.
612 int32 left_id = tree_config->nodes_size();
613 (*tree_config->add_nodes()->mutable_leaf()) =
614 *MergeLeafWeights(tree_config->nodes(node_id).leaf(),
615 split->split_info.mutable_left_child());
616
617 // Add right leaf.
618 int32 right_id = tree_config->nodes_size();
619 (*tree_config->add_nodes()->mutable_leaf()) =
620 *MergeLeafWeights(tree_config->nodes(node_id).leaf(),
621 split->split_info.mutable_right_child());
622
623 // Link children and add them as new roots.
624 boosted_trees::trees::DecisionTree::LinkChildren(
625 {left_id, right_id}, split->split_info.mutable_split_node());
626
627 // Add split gain and, if needed, original leaf to node metadata.
628 TreeNodeMetadata* node_metadata =
629 split->split_info.mutable_split_node()->mutable_node_metadata();
630 node_metadata->set_gain(split->gain);
631 if (learner_config_.pruning_mode() ==
632 boosted_trees::learner::LearnerConfig::POST_PRUNE) {
633 (*node_metadata->mutable_original_leaf()) =
634 *tree_config->mutable_nodes(node_id)->mutable_leaf();
635 }
636
637 // Replace node in tree.
638 (*tree_config->mutable_nodes(node_id)) =
639 *split->split_info.mutable_split_node();
640 if (learner_config_.constraints().max_number_of_unique_feature_columns()) {
641 ensemble_resource->MaybeAddUsedHandler(split->handler_id);
642 }
643 }
644
645 void PruneTree(boosted_trees::trees::DecisionTreeConfig* tree_config) {
646 // No-op if tree is empty.
647 if (tree_config->nodes_size() <= 0) {
648 return;
649 }
650
651 // Copy nodes to temp vector and clear original tree.
652 std::vector<TreeNode> tree_nodes;
653 tree_nodes.reserve(tree_config->nodes_size());
654 for (auto& node : (*tree_config->mutable_nodes())) {
655 tree_nodes.push_back(node);
656 node.Clear();
657 }
658 tree_config->clear_nodes();
659
660 // Prune the tree recursively starting from the root.
661 RecursivePruneTree(0, &tree_nodes);
662
663 // Rebuild compacted tree.
664 (*tree_config->add_nodes()) = tree_nodes[0];
665 std::unordered_map<size_t, size_t> nodes_map;
666 nodes_map[0] = 0;
667 for (size_t node_idx = 0; node_idx < tree_nodes.size(); ++node_idx) {
668 // Skip pruned nodes.
669 auto& original_node = tree_nodes[node_idx];
670 if (original_node.node_case() == TreeNode::NODE_NOT_SET) {
671 continue;
672 }
673
674 // Find node mapped in tree ensemble.
675 auto mapped_node_it = nodes_map.find(node_idx);
676 CHECK(mapped_node_it != nodes_map.end());
677 auto& mapped_node = (*tree_config->mutable_nodes(mapped_node_it->second));
678
679 // Get node children
680 auto children =
681 boosted_trees::trees::DecisionTree::GetChildren(original_node);
682 for (int32& child_idx : children) {
683 auto new_idx = tree_config->nodes_size();
684 (*tree_config->add_nodes()) = tree_nodes[child_idx];
685 nodes_map[child_idx] = new_idx;
686 child_idx = new_idx;
687 }
688 boosted_trees::trees::DecisionTree::LinkChildren(children, &mapped_node);
689 }
690
691 // Check if there are any nodes with gain left.
692 if (tree_config->nodes_size() == 1 &&
693 tree_config->nodes(0).node_metadata().gain() <= 0) {
694 // The whole tree should be pruned.
695 VLOG(2) << "No useful nodes left after post-pruning tree.";
696 tree_config->clear_nodes();
697 }
698 }
699
700 private:
701 boosted_trees::learner::LearnerConfig learner_config_;
702 int64 num_handlers_;
703 LearningRateDropoutDrivenConfig dropout_config_;
704 bool dropout_was_applied_;
705 bool center_bias_;
706};

711class TreeEnsembleStatsOp : public OpKernel {
712 public:
713 explicit TreeEnsembleStatsOp(OpKernelConstruction* const context)
714 : OpKernel(context) {}
715
716 void Compute(OpKernelContext* const context) override {
717 // Get decision tree ensemble.
718 boosted_trees::models::DecisionTreeEnsembleResource* ensemble_resource;
719 OP_REQUIRES_OK(context, LookupResource(context, HandleFromInput(context, 0),
720 &ensemble_resource));
721 core::ScopedUnref unref_me(ensemble_resource);
722 tf_shared_lock l(*ensemble_resource->get_mutex());
723
724 // Get the stamp token.
725 const Tensor* stamp_token_t;
726 OP_REQUIRES_OK(context, context->input("stamp_token", &stamp_token_t));
727 int64 stamp_token = stamp_token_t->scalar<int64>()();
728
729 // Only the Chief should run this Op and it is guaranteed to be in
730 // a consistent state so the stamps must always match.
731 CHECK(ensemble_resource->is_stamp_valid(stamp_token));
732 const boosted_trees::trees::DecisionTreeEnsembleConfig& ensemble_config =
733 ensemble_resource->decision_tree_ensemble();
734
735 // Set tree stats.
736 Tensor* num_trees_t = nullptr;
737 OP_REQUIRES_OK(context, context->allocate_output(
738 "num_trees", TensorShape({}), &num_trees_t));
739 Tensor* active_tree_t = nullptr;
740 OP_REQUIRES_OK(context,
741 context->allocate_output("active_tree", TensorShape({}),
742 &active_tree_t));
743 Tensor* attempted_tree_t = nullptr;
744 OP_REQUIRES_OK(context,
745 context->allocate_output("attempted_trees", TensorShape({}),
746 &attempted_tree_t));
747
748 const int num_trees = ensemble_resource->num_trees();
749 active_tree_t->scalar<int64>()() = num_trees;
750 num_trees_t->scalar<int64>()() =
751 (num_trees <= 0 ||
752 ensemble_resource->LastTreeMetadata()->is_finalized())
753 ? num_trees
754 : num_trees - 1;
755 attempted_tree_t->scalar<int64>()() =
756 ensemble_config.growing_metadata().num_trees_attempted();
757
758 // Set layer stats.
759 Tensor* num_layers_t = nullptr;
760 OP_REQUIRES_OK(context, context->allocate_output(
761 "num_layers", TensorShape({}), &num_layers_t));
762 Tensor* active_layer_t = nullptr;
763 OP_REQUIRES_OK(context,
764 context->allocate_output("active_layer", TensorShape({}),
765 &active_layer_t));
766 Tensor* attempted_layers_t = nullptr;
767 OP_REQUIRES_OK(context,
768 context->allocate_output("attempted_layers", TensorShape({}),
769 &attempted_layers_t));
770
771 int64 num_layers = 0;
772 for (const auto& tree_metadata : ensemble_config.tree_metadata()) {
773 num_layers += tree_metadata.num_layers_grown();
774 }
775 num_layers_t->scalar<int64>()() = num_layers;
776 int tree_metadata_size = ensemble_config.tree_metadata_size();
777 active_layer_t->scalar<int64>()() =
778 tree_metadata_size > 0
779 ? ensemble_config.tree_metadata(tree_metadata_size - 1)
780 .num_layers_grown()
781 : 0;
782 attempted_layers_t->scalar<int64>()() =
783 ensemble_config.growing_metadata().num_layers_attempted();
784 }
785};
