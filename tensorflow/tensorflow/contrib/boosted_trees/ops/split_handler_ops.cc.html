25REGISTER_OP("BuildDenseInequalitySplits")
26 .Attr("feature_column_group_id: int")
27 .Attr("l1_regularization: float")
28 .Attr("l2_regularization: float")
29 .Attr("tree_complexity_regularization: float")
30 .Attr("min_node_weight: float")
31 .Attr("multiclass_strategy: int")
32 .Input("num_minibatches: int64")
33 .Input("partition_ids: int32")
34 .Input("bucket_ids: int64")
35 .Input("gradients: float32")
36 .Input("hessians: float32")
37 .Input("bucket_boundaries: float32")
38 .Input("class_id: int32")
39 .Output("output_partition_ids: int32")
40 .Output("gains: float32")
41 .Output("split_infos: string")
42 .SetShapeFn([](InferenceContext* c) {
43 DimensionHandle unused_dim;
44 ShapeHandle unused_shape;
45 TF_RETURN_IF_ERROR(c->WithRank(c->input(0), 0, &unused_shape));
46
47 ShapeHandle partition_ids_shape;
48 TF_RETURN_IF_ERROR(c->WithRank(c->input(1), 1, &partition_ids_shape));
49 ShapeHandle bucket_ids_shape;
50 TF_RETURN_IF_ERROR(c->WithRank(c->input(2), 2, &bucket_ids_shape));
51 ShapeHandle gradients_shape;
52 TF_RETURN_IF_ERROR(c->WithRankAtLeast(c->input(3), 1, &gradients_shape));
53 TF_RETURN_IF_ERROR(c->Merge(c->Dim(partition_ids_shape, 0),
54 c->Dim(gradients_shape, 0), &unused_dim));
55 ShapeHandle hessians_shape;
56 TF_RETURN_IF_ERROR(c->WithRankAtLeast(c->input(4), 1, &hessians_shape));
57 TF_RETURN_IF_ERROR(c->Merge(c->Dim(partition_ids_shape, 0),
58 c->Dim(hessians_shape, 0), &unused_dim));
59 ShapeHandle bucket_boundaries_shape;
60 TF_RETURN_IF_ERROR(c->WithRank(c->input(5), 1, &bucket_boundaries_shape));
61 c->set_output(0, c->Vector(c->UnknownDim()));
62 c->set_output(1, c->Vector(c->UnknownDim()));
63 c->set_output(2, c->Vector(c->UnknownDim()));
64 return Status::OK();
65 })
66 .Doc(R"doc(
67Find the split that has the best gain for the accumulated stats.
68
69num_minibatches: A scalar, the number of times per example gradients & hessians
70 were accumulated. The stats are divided by this to get per example stats.
71partition_ids: A rank 1 tensor of partition IDs.
72bucket_ids: A rank 2 tensor of buckets IDs and dimensions.
73gradients: A rank 1 tensor of gradients.
74hessians: A rank 1 tensor of hessians.
75bucket_boundaries: A rank 1 tensor, thresholds that were used for bucketization.
76output_partition_ids: A rank 1 tensor, the partition IDs that we created splits
77 for.
78gains: A rank 1 tensor, for the computed gain for the created splits.
79split_infos: A rank 1 tensor of serialized protos which contains the
80 `SplitInfo`s.
81)doc");
