22REGISTER_OP("CenterTreeEnsembleBias")
23 .Attr("learner_config: string")
24 .Attr("centering_epsilon: float = 0.01")
25 .Input("tree_ensemble_handle: resource")
26 .Input("stamp_token: int64")
27 .Input("next_stamp_token: int64")
28 .Input("delta_updates: float")
29 .Output("continue_centering: bool")
30 .SetShapeFn([](shape_inference::InferenceContext* c) {
31 shape_inference::ShapeHandle unused_input;
32 TF_RETURN_IF_ERROR(c->WithRank(c->input(0), 0, &unused_input));
33 TF_RETURN_IF_ERROR(c->WithRank(c->input(1), 0, &unused_input));
34 TF_RETURN_IF_ERROR(c->WithRank(c->input(2), 0, &unused_input));
35 TF_RETURN_IF_ERROR(c->WithRank(c->input(3), 1, &unused_input));
36 c->set_output(0, c->Scalar());
37 return Status::OK();
38 })
39 .Doc(R"doc(
40Centers the tree ensemble bias before adding trees based on feature splits.
41
42learner_config: Config for the learner of type LearnerConfig proto.
43tree_ensemble_handle: Handle to the ensemble variable.
44stamp_token: Stamp token for validating operation consistency.
45next_stamp_token: Stamp token to be used for the next iteration.
46delta_updates: Rank 1 Tensor containing delta updates per bias dimension.
47continue_centering: Scalar indicating whether more centering is needed.
48)doc");

50REGISTER_OP("GrowTreeEnsemble")
51 .Attr("learner_config: string")
52 .Attr("num_handlers: int >= 0")
53 .Attr("center_bias: bool")
54 .Input("tree_ensemble_handle: resource")
55 .Input("stamp_token: int64")
56 .Input("next_stamp_token: int64")
57 .Input("learning_rate: float")
58 .Input("dropout_seed: int64")
59 .Input("partition_ids: num_handlers * int32")
60 .Input("gains: num_handlers * float")
61 .Input("splits: num_handlers * string")
62 .SetShapeFn([](shape_inference::InferenceContext* c) {
63 shape_inference::ShapeHandle unused_input;
64 TF_RETURN_IF_ERROR(c->WithRank(c->input(0), 0, &unused_input));
65 TF_RETURN_IF_ERROR(c->WithRank(c->input(1), 0, &unused_input));
66 TF_RETURN_IF_ERROR(c->WithRank(c->input(2), 0, &unused_input));
67 TF_RETURN_IF_ERROR(c->WithRank(c->input(3), 0, &unused_input));
68 // Dropout seed.
69 TF_RETURN_IF_ERROR(c->WithRank(c->input(4), 0, &unused_input));
70 return Status::OK();
71 })
72 .Doc(R"doc(
73Grows the tree ensemble by either adding a layer to the last tree being grown
74or by starting a new tree.
75
76learner_config: Config for the learner of type LearnerConfig proto.
77num_handlers: Number of handlers generating candidates.
78tree_ensemble_handle: Handle to the ensemble variable.
79stamp_token: Stamp token for validating operation consistency.
80next_stamp_token: Stamp token to be used for the next iteration.
81learning_rate: Scalar learning rate.
82partition_ids: List of Rank 1 Tensors containing partition Id per candidate.
83gains: List of Rank 1 Tensors containing gains per candidate.
84splits: List of Rank 1 Tensors containing serialized SplitInfo protos per candidate.
85)doc");

87REGISTER_OP("TreeEnsembleStats")
88 .Input("tree_ensemble_handle: resource")
89 .Input("stamp_token: int64")
90 .Output("num_trees: int64")
91 .Output("num_layers: int64")
92 .Output("active_tree: int64")
93 .Output("active_layer: int64")
94 .Output("attempted_trees: int64")
95 .Output("attempted_layers: int64")
96 .SetShapeFn([](shape_inference::InferenceContext* c) {
97 shape_inference::ShapeHandle unused_input;
98 TF_RETURN_IF_ERROR(c->WithRank(c->input(0), 0, &unused_input));
99 TF_RETURN_IF_ERROR(c->WithRank(c->input(1), 0, &unused_input));
100 c->set_output(0, c->Scalar());
101 c->set_output(1, c->Scalar());
102 c->set_output(2, c->Scalar());
103 c->set_output(3, c->Scalar());
104 c->set_output(4, c->Scalar());
105 c->set_output(5, c->Scalar());
106 return Status::OK();
107 })
108 .Doc(R"doc(
109Retrieves stats related to the tree ensemble.
110
111tree_ensemble_handle: Handle to the ensemble variable.
112stamp_token: Stamp token for validating operation consistency.
113num_trees: Scalar indicating the number of finalized trees in the ensemble.
114num_layers: Scalar indicating the number of layers in the ensemble.
115active_tree: Scalar indicating the active tree being trained.
116active_layer: Scalar indicating the active layer being trained.
117)doc");
