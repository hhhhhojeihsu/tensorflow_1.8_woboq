21namespace tensorflow {
22namespace boosted_trees {
23namespace utils {
24
25// Utility class holding an optional value.
26template <typename T>
27class OptionalValue {
28 public:
29 OptionalValue() : value_(), has_value_(false) {}
30 explicit OptionalValue(T value) : value_(value), has_value_(true) {}
31
32 bool has_value() const { return has_value_; }
33 const T& get_value() const {
34 QCHECK(has_value());
35 return value_;
36 }
37
38 private:
39 T value_;
40 bool has_value_;
41};
