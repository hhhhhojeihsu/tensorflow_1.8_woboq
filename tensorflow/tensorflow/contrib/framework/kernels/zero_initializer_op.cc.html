27namespace tensorflow {
28
29using CPUDevice = Eigen::ThreadPoolDevice;
30using GPUDevice = Eigen::GpuDevice;
31
32template <typename Device, typename T>
33class ZeroInitializerOp : public OpKernel {
34 public:
35 explicit ZeroInitializerOp(OpKernelConstruction* ctx) : OpKernel(ctx) {
36 OP_REQUIRES(ctx, IsRefType(ctx->input_type(0)),
37 errors::InvalidArgument("input needs to be a ref type"));
38 }
39
40 void Compute(OpKernelContext* ctx) override {
41 mutex_lock l(*ctx->input_ref_mutex(0));
42 Tensor input = ctx->mutable_input(0, true);
43 OP_REQUIRES(ctx, !input.IsInitialized(),
44 errors::InvalidArgument("input is already initialized"));
45 AllocatorAttributes attr;
46 attr.set_gpu_compatible(true);
47 attr.set_nic_compatible(true);
48 PersistentTensor out_persistent;
49 Tensor* out_tensor = nullptr;
50 OP_REQUIRES_OK(
51 ctx, ctx->allocate_persistent(input.dtype(), input.shape(),
52 &out_persistent, &out_tensor, attr));
53 functor::TensorSetZero<Device, T>()(ctx->eigen_device<Device>(),
54 out_tensor->flat<T>());
55 ctx->replace_ref_input(0, *out_tensor, true);
56 // we always return the input ref.
57 ctx->forward_ref_input_to_ref_output(0, 0);
58 }
59};
60
61#define REGISTER_KERNELS(D, T) \
62 REGISTER_KERNEL_BUILDER( \
63 Name("ZeroInitializer").Device(DEVICE_##D).TypeConstraint<T>("T"), \
64 ZeroInitializerOp<D##Device, T>);
65#define REGISTER_CPU_KERNELS(T) REGISTER_KERNELS(CPU, T);
66TF_CALL_REAL_NUMBER_TYPES(REGISTER_CPU_KERNELS);
67#undef REGISTER_CPU_KERNELS
