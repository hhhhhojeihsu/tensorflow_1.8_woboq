48REGISTER_OP("FunctionBufferingResource")
49 .Input("string_arg: string")
50 .Input("target_device: string")
51 .Output("resource: resource")
52 .Attr("shared_name: string")
53 .Attr("container: string")
54 .Attr("f: func")
55 .Attr("buffer_size: int")
56 .SetShapeFn(shape_inference::UnknownShape)
57 .Doc(R"doc(
58Creates a resource that fills up a buffer by making function calls.
59
60string_arg: String argument to the function call.
61target_device: Target device to execute the function on.
62resource: Handle to the resource created.
63f: Function to be executed.
64buffer_size: Size of the buffer.
65container: If non-empty, this resource is placed in the given container.
66 Otherwise, a default container is used.
67shared_name: If non-empty, this resource will be shared under the given name
68 across multiple sessions.
69)doc");

106REGISTER_OP("ThreadPoolHandle")
107 .Output("handle: resource")
108 .SetShapeFn(shape_inference::ScalarShape)
109 .Attr("num_threads: int")
110 .Attr("display_name: string")
111 .Attr("container: string = ''")
112 .Attr("shared_name: string = ''")
113 .Doc(R"doc(
114Creates a custom thread pool with the given number of threads.
115
116handle: A resource that can be consumed by one or more ThreadPoolDataset ops.
117num_threads: The number of threads in the thread pool.
118display_name: A human-readable name for the threads that may be visible in
119 some visualizations.
120)doc");
