16#include "tensorflow/core/framework/op.h"
17#include "tensorflow/core/framework/shape_inference.h"
18
19namespace tensorflow {
20
21using ::tensorflow::shape_inference::InferenceContext;
22using ::tensorflow::shape_inference::ShapeHandle;
23
24REGISTER_OP("Resampler")
25 .Input("data: T")
26 .Input("warp: T")
27 .Output("output: T")
28 .Attr("T: {half, float, double}")
29 .SetShapeFn([](InferenceContext* c) {
30 ShapeHandle data;
31 ShapeHandle warp;
32 TF_RETURN_IF_ERROR(c->WithRankAtLeast(c->input(0), 1, &data));
33 TF_RETURN_IF_ERROR(c->WithRankAtLeast(c->input(1), 1, &warp));
34
35 ShapeHandle output; // will be warp[:-1] + [data[-1]]
36 TF_RETURN_IF_ERROR(c->Subshape(warp, 0, -1, &output));
37 TF_RETURN_IF_ERROR(
38 c->Concatenate(output, c->Vector(c->Dim(data, -1)), &output));
39
40 c->set_output(0, output);
41 return ::tensorflow::Status::OK();
42 })
43 .Doc(R"doc(Resampler op.)doc");
44
45REGISTER_OP("ResamplerGrad")
46 .Input("data: T")
47 .Input("warp: T")
48 .Input("grad_output: T")
49 .Output("grad_data: T")
50 .Output("grad_warp: T")
51 .Attr("T: {half, float, double}")
52 .SetShapeFn([](InferenceContext* c) {
53 c->set_output(0, c->input(0));
54 c->set_output(1, c->input(1));
55 return ::tensorflow::Status::OK();
56 })
57 .Doc(R"doc(Resampler Grad op.)doc");
58
59} // namespace tensorflow
