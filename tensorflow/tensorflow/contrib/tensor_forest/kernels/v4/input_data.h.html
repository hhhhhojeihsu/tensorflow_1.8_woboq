29namespace tensorflow {
30namespace tensorforest {
31
32typedef TTypes<const float, 2>::ConstTensor DenseStorageType;
33typedef TTypes<const int64, 2>::ConstTensor SparseIndicesStorageType;
34typedef TTypes<const float, 1>::ConstTensor SparseValuesStorageType;
35
36class TensorDataSet {
37 public:
38 TensorDataSet(const tensorforest::TensorForestDataSpec& input_spec,
39 int32 seed)
40 : dense_data_(nullptr),
41 sparse_indices_(nullptr),
42 sparse_values_(nullptr),
43 input_spec_(input_spec),
44 split_sampling_random_seed_(seed) {
45 int column_count = 0;
46 for (int i = 0; i < input_spec_.dense_size(); ++i) {
47 for (int j = 0; j < input_spec_.dense(i).size(); ++j) {
48 ++column_count;
49 }
50 }
51 available_features_.reserve(column_count);
52 decision_trees::FeatureId id;
53 for (int i = 0; i < column_count; i++) {
54 id.mutable_id()->set_value(strings::StrCat(i));
55 available_features_.emplace_back(id);
56 }
57
58 // Set up the random number generator.
59 if (split_sampling_random_seed_ == 0) {
60 single_rand_ = std::unique_ptr<random::PhiloxRandom>(
61 new random::PhiloxRandom(random::New64()));
62 } else {
63 single_rand_ = std::unique_ptr<random::PhiloxRandom>(
64 new random::PhiloxRandom(split_sampling_random_seed_));
65 }
66
67 rng_ = std::unique_ptr<random::SimplePhilox>(
68 new random::SimplePhilox(single_rand_.get()));
69 }
70 virtual ~TensorDataSet() {}
71
72 void set_input_tensors(const Tensor& dense, const Tensor& sparse_indices,
73 const Tensor& sparse_values,
74 const Tensor& sparse_shape);
75
76 float get_input_value(int offset, int col) {
77 return (*dense_data_)(offset, col);
78 }
79
80 int NumItems() const {
81 if (dense_data_ != nullptr) {
82 return dense_data_->dimensions()[0];
83 } else if (sparse_indices_ != nullptr) {
84 return sparse_batch_size_;
85 } else {
86 return 0;
87 }
88 }
89
90 // This looks up a value by example and int32_id, which is much faster than
91 // GetFeature.
92 float GetExampleValue(int example,
93 const decision_trees::FeatureId& feature_id) const;
94
95 // Same as overload with FeatureId, but if you already have the feature as
96 // an int32 you can avoid the atoi32.
97 virtual float GetExampleValue(int example, int32 feature_id) const;
98
99 int num_features() { return available_features_.size(); }
100
101 const Tensor& original_tensor() const { return original_dense_tensor_; }
102
103 bool Decide(const decision_trees::BinaryNode& node, int example) const;
104
105 // Randomly samples a feature from example, returns its id in feature_name,
106 // the value in bias, and it's type from input_spec in type.
107 void RandomSample(int example, decision_trees::FeatureId* feature_name,
108 float* bias, int* type) const;
109
110 private:
111 std::unique_ptr<DenseStorageType> dense_data_;
112 std::unique_ptr<SparseIndicesStorageType> sparse_indices_;
113 std::unique_ptr<SparseValuesStorageType> sparse_values_;
114 int sparse_batch_size_;
115
116 Tensor original_dense_tensor_;
117 const tensorforest::TensorForestDataSpec input_spec_;
118 std::vector<decision_trees::FeatureId> available_features_;
119
120 int32 split_sampling_random_seed_;
121 std::unique_ptr<random::PhiloxRandom> single_rand_;
122 std::unique_ptr<random::SimplePhilox> rng_;
123};
124} // namespace tensorforest
125} // namespace tensorflow
