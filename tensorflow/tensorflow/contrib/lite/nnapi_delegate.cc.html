72// Adds the tensors of the interpreter to the NN API model.
73// Returns the number of operands added.
74uint32_t addTensorOperands(tflite::Interpreter* interpreter,
75 ANeuralNetworksModel* nn_model) {
76 uint32_t next_id = 0;
77 for (size_t i = 0; i < interpreter->tensors_size(); i++) {
78 int32_t nn_type = 0;
79 float scale = 1.0f;
80 int32_t zeroPoint = 0;
81 TfLiteTensor* tensor = interpreter->tensor(i);
82 switch (tensor->type) {
83 case kTfLiteNoType:
84 // Tensors added during initialization of Ops don't have a type yet and
85 // should not be registered with the NNAPI.
86 continue;
87 case kTfLiteFloat32:
88 nn_type = ANEURALNETWORKS_TENSOR_FLOAT32;
89 break;
90 case kTfLiteUInt8:
91 nn_type = ANEURALNETWORKS_TENSOR_QUANT8_ASYMM;
92 scale = tensor->params.scale;
93 zeroPoint = tensor->params.zero_point;
94 break;
95 case kTfLiteInt32:
96 nn_type = ANEURALNETWORKS_TENSOR_INT32;
97 scale = tensor->params.scale;
98 zeroPoint = tensor->params.zero_point;
99 break;
100 default:
101 FATAL("Unsupported type.");
102 }
103 // TODO(aselle): Note, many of these are intermediate results. Do I need
104 // to ever specify these sizes. I am currently below doing setValue
105 // on all of them, but I shouldn't in the future.
106 // Answer(jeanluc): If all the operators can set the dimension correctly,
107 // you won't need to.
108 ANeuralNetworksOperandType operand_type{
109 nn_type, static_cast<uint32_t>(tensor->dims->size),
110 reinterpret_cast<uint32_t*>(tensor->dims->data), scale, zeroPoint};
111 CHECK_NN(ANeuralNetworksModel_addOperand(nn_model, &operand_type));
112
113 // TODO(aselle): Based on Michael's suggestion, limiting this to read
114 // only memory
115 if (tensor->allocation_type == kTfLiteMmapRo) {
116 if (const NNAPIAllocation* alloc = dynamic_cast<const NNAPIAllocation*>(
117 static_cast<const Allocation*>(tensor->allocation))) {
118 CHECK_NN(ANeuralNetworksModel_setOperandValueFromMemory(
119 nn_model, i, alloc->memory(), alloc->offset(tensor->data.raw),
120 tensor->bytes));
121 } else {
122 CHECK_NN(ANeuralNetworksModel_setOperandValue(
123 nn_model, i, tensor->data.raw, tensor->bytes));
124 }
125 }
126 ++next_id;
127 }
128 return next_id;
129}

131// Adds the operations and their parameters to the NN API model.
132// 'next-id' is the operand ID of the next operand of the model.
133void AddOpsAndParams(tflite::Interpreter* interpreter,
134 ANeuralNetworksModel* nn_model, uint32_t next_id) {
135 for (size_t i = 0; i < interpreter->nodes_size(); i++) {
136 const auto* node_and_registration = interpreter->node_and_registration(i);
137 const TfLiteNode& node = node_and_registration->first;
138 const TfLiteRegistration& registration = node_and_registration->second;
139 tflite::BuiltinOperator builtin =
140 static_cast<tflite::BuiltinOperator>(registration.builtin_code);
141
142 // Add the parameters.
143 std::vector<uint32_t> augmented_inputs(
144 node.inputs->data, node.inputs->data + node.inputs->size);
145
146 auto add_scalar_int32 = [&nn_model, &augmented_inputs,
147 &next_id](int value) {
148 ANeuralNetworksOperandType operand_type{.type = ANEURALNETWORKS_INT32};
149 CHECK_NN(ANeuralNetworksModel_addOperand(nn_model, &operand_type))
150 CHECK_NN(ANeuralNetworksModel_setOperandValue(nn_model, next_id, &value,
151 sizeof(int32_t)))
152 augmented_inputs.push_back(next_id++);
153 };
154
155 auto add_scalar_float32 = [&nn_model, &augmented_inputs,
156 &next_id](float value) {
157 ANeuralNetworksOperandType operand_type{.type = ANEURALNETWORKS_FLOAT32};
158 CHECK_NN(ANeuralNetworksModel_addOperand(nn_model, &operand_type))
159 CHECK_NN(ANeuralNetworksModel_setOperandValue(nn_model, next_id, &value,
160 sizeof(float)))
161 augmented_inputs.push_back(next_id++);
162 };
163
164 auto duplicate_state_tensor_float32 =
165 [interpreter, &nn_model, &augmented_inputs](int tensor_id) {
166 const TfLiteTensor* tensor = interpreter->tensor(tensor_id);
167 CHECK_NN(ANeuralNetworksModel_setOperandValue(
168 nn_model, tensor_id, tensor->data.raw, tensor->bytes));
169 augmented_inputs.push_back(tensor_id);
170 };
171
172 auto add_add_params = [&add_scalar_int32]() { add_scalar_int32(0); };
173
174 auto add_pooling_params = [&add_scalar_int32](void* data) {
175 auto builtin = reinterpret_cast<TfLitePoolParams*>(data);
176 add_scalar_int32(builtin->padding);
177 add_scalar_int32(builtin->stride_width);
178 add_scalar_int32(builtin->stride_height);
179 add_scalar_int32(builtin->filter_width);
180 add_scalar_int32(builtin->filter_height);
181 add_scalar_int32(builtin->activation);
182 };
183
184 auto add_convolution_params = [&add_scalar_int32](void* data) {
185 auto builtin = reinterpret_cast<TfLiteConvParams*>(data);
186 add_scalar_int32(builtin->padding);
187 add_scalar_int32(builtin->stride_width);
188 add_scalar_int32(builtin->stride_height);
189 add_scalar_int32(builtin->activation);
190 };
191
192 auto add_depthwise_conv_params = [&add_scalar_int32](void* data) {
193 auto builtin = reinterpret_cast<TfLiteDepthwiseConvParams*>(data);
194 add_scalar_int32(builtin->padding);
195 add_scalar_int32(builtin->stride_width);
196 add_scalar_int32(builtin->stride_height);
197 add_scalar_int32(builtin->depth_multiplier);
198 add_scalar_int32(builtin->activation);
199 };
200
201 auto add_fully_connected_params = [&add_scalar_int32](void* data) {
202 auto builtin = reinterpret_cast<TfLiteFullyConnectedParams*>(data);
203 add_scalar_int32(builtin->activation);
204 };
205
206 auto add_concatenation_params = [&add_scalar_int32](void* data) {
207 auto builtin = reinterpret_cast<TfLiteConcatenationParams*>(data);
208 add_scalar_int32(builtin->axis);
209 if (builtin->activation != kTfLiteActNone) {
210 FATAL("Concatenation does not support fused activation in NNAPI");
211 }
212 };
213
214 auto add_softmax_params = [&add_scalar_float32](void* data) {
215 auto builtin = reinterpret_cast<TfLiteSoftmaxParams*>(data);
216 add_scalar_float32(builtin->beta);
217 };
218
219 auto add_space_to_depth_params = [&add_scalar_int32](void* data) {
220 auto builtin = reinterpret_cast<TfLiteSpaceToDepthParams*>(data);
221 add_scalar_int32(builtin->block_size);
222 };
223
224 auto add_lstm_params = [&add_scalar_int32,
225 &add_scalar_float32](void* data) {
226 auto builtin = reinterpret_cast<TfLiteLSTMParams*>(data);
227 add_scalar_int32(builtin->activation);
228 add_scalar_float32(builtin->cell_clip);
229 add_scalar_float32(builtin->proj_clip);
230 };
231
232#if 0
233 auto add_reshape_params = [&](void* data) {
234 auto builtin = reinterpret_cast<TfLiteReshapeParams*>(data);
235 uint32_t tensor_size_shape = builtin->num_dimensions;
236 ANeuralNetworksOperandType operand_type{
237 ANEURALNETWORKS_TENSOR_INT32,
238 {static_cast<uint32_t>(1),
239 reinterpret_cast<uint32_t*>(&tensor_size_shape)},
240 0,
241 0};
242 CHECK_NN(ANeuralNetworksModel_addOperand(nn_model, &operand_type))
243 CHECK_NN(ANeuralNetworksModel_setOperandValue(
244 nn_model, next_id, builtin->shape,
245 sizeof(int) * builtin->num_dimensions));
246 augmented_inputs.push_back(next_id++);
247 };
248#endif
249
250 ANeuralNetworksOperationType nn_op_type;
251 switch (builtin) {
252 case tflite::BuiltinOperator_ADD:
253 nn_op_type = ANEURALNETWORKS_ADD;
254 add_add_params();
255 break;
256 case tflite::BuiltinOperator_AVERAGE_POOL_2D:
257 add_pooling_params(node.builtin_data);
258 nn_op_type = ANEURALNETWORKS_AVERAGE_POOL_2D;
259 break;
260 case tflite::BuiltinOperator_MAX_POOL_2D:
261 add_pooling_params(node.builtin_data);
262 nn_op_type = ANEURALNETWORKS_MAX_POOL_2D;
263 break;
264 case tflite::BuiltinOperator_L2_POOL_2D:
265 add_pooling_params(node.builtin_data);
266 nn_op_type = ANEURALNETWORKS_L2_POOL_2D;
267 break;
268 case tflite::BuiltinOperator_CONV_2D:
269 add_convolution_params(node.builtin_data);
270 nn_op_type = ANEURALNETWORKS_CONV_2D;
271 break;
272 case tflite::BuiltinOperator_RELU:
273 nn_op_type = ANEURALNETWORKS_RELU;
274 break;
275 case tflite::BuiltinOperator_RELU6:
276 nn_op_type = ANEURALNETWORKS_RELU6;
277 break;
278 case tflite::BuiltinOperator_TANH:
279 nn_op_type = ANEURALNETWORKS_TANH;
280 break;
281 case tflite::BuiltinOperator_LOGISTIC:
282 nn_op_type = ANEURALNETWORKS_LOGISTIC;
283 break;
284 case tflite::BuiltinOperator_DEPTHWISE_CONV_2D:
285 add_depthwise_conv_params(node.builtin_data);
286 nn_op_type = ANEURALNETWORKS_DEPTHWISE_CONV_2D;
287 break;
288 case tflite::BuiltinOperator_CONCATENATION:
289 add_concatenation_params(node.builtin_data);
290 nn_op_type = ANEURALNETWORKS_CONCATENATION;
291 break;
292 case tflite::BuiltinOperator_SOFTMAX:
293 add_softmax_params(node.builtin_data);
294 nn_op_type = ANEURALNETWORKS_SOFTMAX;
295 break;
296 case tflite::BuiltinOperator_FULLY_CONNECTED:
297 add_fully_connected_params(node.builtin_data);
298 nn_op_type = ANEURALNETWORKS_FULLY_CONNECTED;
299 break;
300 case tflite::BuiltinOperator_RESHAPE:
301 nn_op_type = ANEURALNETWORKS_RESHAPE;
302 // add_reshape_params(node.builtin_data);
303 break;
304 case tflite::BuiltinOperator_SPACE_TO_DEPTH:
305 add_space_to_depth_params(node.builtin_data);
306 nn_op_type = ANEURALNETWORKS_SPACE_TO_DEPTH;
307 break;
308 case tflite::BuiltinOperator_LSTM: {
309 duplicate_state_tensor_float32(
310 node.outputs->data[/*kOutputStateTensor*/ 1]);
311 duplicate_state_tensor_float32(
312 node.outputs->data[/*kCellStateTensor*/ 2]);
313 add_lstm_params(node.builtin_data);
314 nn_op_type = ANEURALNETWORKS_LSTM;
315 break;
316 }
317 case tflite::BuiltinOperator_CONCAT_EMBEDDINGS:
318 case tflite::BuiltinOperator_LSH_PROJECTION:
319 case tflite::BuiltinOperator_SVDF:
320 case tflite::BuiltinOperator_HASHTABLE_LOOKUP:
321 case tflite::BuiltinOperator_RNN:
322 case tflite::BuiltinOperator_BIDIRECTIONAL_SEQUENCE_RNN:
323 case tflite::BuiltinOperator_UNIDIRECTIONAL_SEQUENCE_RNN:
324 case tflite::BuiltinOperator_EMBEDDING_LOOKUP:
325 case tflite::BuiltinOperator_EMBEDDING_LOOKUP_SPARSE:
326 case tflite::BuiltinOperator_BIDIRECTIONAL_SEQUENCE_LSTM:
327 case tflite::BuiltinOperator_UNIDIRECTIONAL_SEQUENCE_LSTM:
328 case tflite::BuiltinOperator_L2_NORMALIZATION:
329 case tflite::BuiltinOperator_LOCAL_RESPONSE_NORMALIZATION:
330 case tflite::BuiltinOperator_MUL:
331 case tflite::BuiltinOperator_PAD:
332 case tflite::BuiltinOperator_RESIZE_BILINEAR:
333 case tflite::BuiltinOperator_CALL:
334 case tflite::BuiltinOperator_SKIP_GRAM:
335 case tflite::BuiltinOperator_RELU_N1_TO_1:
336 case tflite::BuiltinOperator_GATHER:
337 case tflite::BuiltinOperator_SPACE_TO_BATCH_ND:
338 case tflite::BuiltinOperator_BATCH_TO_SPACE_ND:
339 case tflite::BuiltinOperator_TOPK_V2:
340 case tflite::BuiltinOperator_TRANSPOSE:
341 case tflite::BuiltinOperator_MEAN:
342 case tflite::BuiltinOperator_DIV:
343 case tflite::BuiltinOperator_SUB:
344 case tflite::BuiltinOperator_SPLIT:
345 case tflite::BuiltinOperator_SQUEEZE:
346 case tflite::BuiltinOperator_STRIDED_SLICE:
347 case tflite::BuiltinOperator_EXP:
348 case tflite::BuiltinOperator_LOG_SOFTMAX:
349 case tflite::BuiltinOperator_DEQUANTIZE:
350 case tflite::BuiltinOperator_DELEGATE:
351 case tflite::BuiltinOperator_CAST:
352 case tflite::BuiltinOperator_PRELU:
353 case tflite::BuiltinOperator_MAXIMUM:
354 FATAL("Op code %d is currently not delegated to NNAPI", builtin);
355 nn_op_type = -1; // set to invalid
356 break;
357 case tflite::BuiltinOperator_CUSTOM:
358 FATAL("Custom operations are not supported when using NNAPI.");
359 nn_op_type = -1; // set to invalid
360 break;
361 }
362
363 // Add the operation.
364 CHECK_NN(ANeuralNetworksModel_addOperation(
365 nn_model, nn_op_type, static_cast<uint32_t>(augmented_inputs.size()),
366 augmented_inputs.data(), static_cast<uint32_t>(node.outputs->size),
367 reinterpret_cast<uint32_t*>(node.outputs->data)));
368 }
369}

371TfLiteStatus NNAPIDelegate::BuildGraph(Interpreter* interpreter) {
372 // TODO(aselle): This is not correct. need to handle resize invalidation.
373 if (nn_model_ && nn_compiled_model_) return kTfLiteOk;
374
375 if (!nn_model_) {
376 CHECK_NN(ANeuralNetworksModel_create(&nn_model_));
377
378 uint32_t next_id = addTensorOperands(interpreter, nn_model_);
379 AddOpsAndParams(interpreter, nn_model_, next_id);
380 CHECK_NN(ANeuralNetworksModel_identifyInputsAndOutputs(
381 nn_model_, static_cast<uint32_t>(interpreter->inputs().size()),
382 reinterpret_cast<const uint32_t*>(interpreter->inputs().data()),
383 static_cast<uint32_t>(interpreter->outputs().size()),
384 reinterpret_cast<const uint32_t*>(interpreter->outputs().data())));
385 CHECK_NN(ANeuralNetworksModel_finish(nn_model_));
386 }
387 if (!nn_compiled_model_) {
388 CHECK_NN(ANeuralNetworksCompilation_create(nn_model_, &nn_compiled_model_));
389 CHECK_NN(ANeuralNetworksCompilation_finish(nn_compiled_model_));
390 }
391 return kTfLiteOk;
392}

394TfLiteStatus NNAPIDelegate::Invoke(Interpreter* interpreter) {
395 if (!nn_model_) {
396 TF_LITE_ENSURE_STATUS(BuildGraph(interpreter));
397 }
398
399 ANeuralNetworksExecution* execution = nullptr;
400 CHECK_NN(ANeuralNetworksExecution_create(nn_compiled_model_, &execution));
401
402 // Currently perform deep copy of input buffer
403 for (size_t i = 0; i < interpreter->inputs().size(); i++) {
404 int input = interpreter->inputs()[i];
405 // TODO(aselle): Is this what we want or do we want input instead?
406 // TODO(aselle): This should be called setInputValue maybe to be cons.
407 TfLiteTensor* tensor = interpreter->tensor(input);
408 CHECK_NN(ANeuralNetworksExecution_setInput(
409 execution, i, nullptr, tensor->data.raw, tensor->bytes));
410 }
411 // Tell nn api where to place final data.
412 for (size_t i = 0; i < interpreter->outputs().size(); i++) {
413 int output = interpreter->outputs()[i];
414 TfLiteTensor* tensor = interpreter->tensor(output);
415 CHECK_NN(ANeuralNetworksExecution_setOutput(
416 execution, i, nullptr, tensor->data.raw, tensor->bytes));
417 }
418 // Currently use blocking compute.
419 ANeuralNetworksEvent* event = nullptr;
420 CHECK_NN(ANeuralNetworksExecution_startCompute(execution, &event));
421 CHECK_NN(ANeuralNetworksEvent_wait(event));
422 ANeuralNetworksEvent_free(event);
423 ANeuralNetworksExecution_free(execution);
424
425#if 0
426 printf("From the NN API:\n");
427 TfLiteTensor* tensor = interpreter->tensor(interpreter->outputs()[0]);
428 if (float* data =
429 interpreter->typed_tensor<float>(interpreter->outputs()[0])) {
430 size_t num = tensor->bytes / sizeof(float);
431 for (float* p = data; p < data + num; p++) {
432 printf(" %f", *p);
433 }
434 printf("\n");
435 }
436#endif
437
438 return kTfLiteOk;
439}
