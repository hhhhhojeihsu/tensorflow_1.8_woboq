17namespace tflite {
18
19TfLiteIntArray* ConvertVectorToTfLiteIntArray(const std::vector<int>& input) {
20 return ConvertArrayToTfLiteIntArray(input.size(), input.data());
21}
22
23TfLiteIntArray* ConvertArrayToTfLiteIntArray(const int rank, const int* dims) {
24 TfLiteIntArray* output = TfLiteIntArrayCreate(rank);
25 for (size_t i = 0; i < rank; i++) {
26 output->data[i] = dims[i];
27 }
28 return output;
29}
30
31bool EqualArrayAndTfLiteIntArray(const TfLiteIntArray* a, const int b_size,
32 const int* b) {
33 if (!a) return false;
34 if (a->size != b_size) return false;
35 for (int i = 0; i < a->size; ++i) {
36 if (a->data[i] != b[i]) return false;
37 }
38 return true;
39}
