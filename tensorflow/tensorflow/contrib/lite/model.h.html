15// Deserialization infrastructure for tflite. Provides functionality
16// to go from a serialized tflite model in flatbuffer format to an
17// interpreter.
18//
19// using namespace tflite;
20// StderrReporter error_reporter;
21// auto model = FlatBufferModel::BuildFromFile("interesting_model.tflite",
22// &error_reporter);
23// MyOpResolver resolver; // You need to subclass OpResolver to provide
24// // implementations.
25// InterpreterBuilder builder(*model, resolver);
26// std::unique_ptr<Interpreter> interpreter;
27// if(builder(&interpreter) == kTfLiteOk) {
28// .. run model inference with interpreter
29// }
30//
31// OpResolver must be defined to provide your kernel implementations to the
32// interpreter. This is environment specific and may consist of just the builtin
33// ops, or some custom operators you defined to extend tflite.
34#ifndef TENSORFLOW_CONTRIB_LITE_MODEL_H_
35#define TENSORFLOW_CONTRIB_LITE_MODEL_H_

40#include "tensorflow/contrib/lite/schema/schema_generated.h"

44// Abstract interface that verifies whether a given model is legit.
45// It facilitates the use-case to verify and build a model without loading it
46// twice.
47class TfLiteVerifier {
48 public:
49 // Returns true if the model is legit.
50 virtual bool Verify(const char* data, int length,
51 ErrorReporter* reporter) = 0;
52 virtual ~TfLiteVerifier() {}
53};
54
55// An RAII object that represents a read-only tflite model, copied from disk,
56// or mmapped. This uses flatbuffers as the serialization format.
57class FlatBufferModel {
58 public:
59 // Builds a model based on a file.
60 // Caller retains ownership of `error_reporter` and must ensure its lifetime
61 // is longer than the FlatBufferModel instance.
62 // Returns a nullptr in case of failure.
63 static std::unique_ptr<FlatBufferModel> BuildFromFile(
64 const char* filename,
65 ErrorReporter* error_reporter = DefaultErrorReporter());
66
67 // Verifies whether the content of the file is legit, then builds a model
68 // based on the file.
69 // Caller retains ownership of `error_reporter` and must ensure its lifetime
70 // is longer than the FlatBufferModel instance.
71 // Returns a nullptr in case of failure.
72 static std::unique_ptr<FlatBufferModel> VerifyAndBuildFromFile(
73 const char* filename, TfLiteVerifier* verifier = nullptr,
74 ErrorReporter* error_reporter = DefaultErrorReporter());
75
76 // Builds a model based on a pre-loaded flatbuffer. The caller retains
77 // ownership of the buffer and should keep it alive until the returned object
78 // is destroyed. Caller retains ownership of `error_reporter` and must ensure
79 // its lifetime is longer than the FlatBufferModel instance.
80 // Returns a nullptr in case of failure.
81 static std::unique_ptr<FlatBufferModel> BuildFromBuffer(
82 const char* buffer, size_t buffer_size,
83 ErrorReporter* error_reporter = DefaultErrorReporter());
84
85 // Builds a model directly from a flatbuffer pointer. The caller retains
86 // ownership of the buffer and should keep it alive until the returned object
87 // is destroyed. Caller retains ownership of `error_reporter` and must ensure
88 // its lifetime is longer than the FlatBufferModel instance.
89 // Returns a nullptr in case of failure.
90 static std::unique_ptr<FlatBufferModel> BuildFromModel(
91 const tflite::Model* model_spec,
92 ErrorReporter* error_reporter = DefaultErrorReporter());
93
94 // Releases memory or unmaps mmaped memory.
95 ~FlatBufferModel();
96
97 // Copying or assignment is disallowed to simplify ownership semantics.
98 FlatBufferModel(const FlatBufferModel&) = delete;
99 FlatBufferModel& operator=(const FlatBufferModel&) = delete;
100
101 bool initialized() const { return model_ != nullptr; }
102 const tflite::Model* operator->() const { return model_; }
103 const tflite::Model* GetModel() const { return model_; }
104 ErrorReporter* error_reporter() const { return error_reporter_; }
105 const Allocation* allocation() const { return allocation_; }
106
107 // Returns true if the model identifier is correct (otherwise false and
108 // reports an error).
109 bool CheckModelIdentifier() const;
110
111 private:
112 // Loads a model from a given allocation. FlatBufferModel will take over the
113 // ownership of `allocation`, and delete it in destructor. The ownership of
114 // `error_reporter`remains with the caller and must have lifetime at least
115 // as much as FlatBufferModel. This is to allow multiple models to use the
116 // same ErrorReporter instance.
117 FlatBufferModel(Allocation* allocation,
118 ErrorReporter* error_reporter = DefaultErrorReporter());
119
120 // Loads a model from Model flatbuffer. The `model` has to remain alive and
121 // unchanged until the end of this flatbuffermodel's lifetime.
122 FlatBufferModel(const Model* model, ErrorReporter* error_reporter);
123
124 // Flatbuffer traverser pointer. (Model* is a pointer that is within the
125 // allocated memory of the data allocated by allocation's internals.
126 const tflite::Model* model_ = nullptr;
127 // The error reporter to use for model errors and subsequent errors when
128 // the interpreter is created
129 ErrorReporter* error_reporter_;
130 // The allocator used for holding memory of the model.
131 Allocation* allocation_ = nullptr;
132};

134// Abstract interface that returns TfLiteRegistrations given op codes or custom
135// op names. This is the mechanism that ops being referenced in the flatbuffer
136// model are mapped to executable function pointers (TfLiteRegistrations).
137class OpResolver {
138 public:
139 // Finds the op registration for a builtin operator by enum code.
140 virtual TfLiteRegistration* FindOp(tflite::BuiltinOperator op) const = 0;
141 // Finds the op registration of a custom operator by op name.
142 virtual TfLiteRegistration* FindOp(const char* op) const = 0;
143 virtual ~OpResolver() {}
144};
145
146// Build an interpreter capable of interpreting `model`.
147//
148// model: a scoped model whose lifetime must be at least as long as
149// the interpreter. In principle multiple interpreters can be made from
150// a single model.
151// op_resolver: An instance that implements the Resolver interface which maps
152// custom op names and builtin op codes to op registrations.
153// reportError: a functor that is called to report errors that handles
154// printf var arg semantics. The lifetime of the reportError object must
155// be greater than or equal to the Interpreter created by operator().
156//
157// Returns a kTfLiteOk when successful and sets interpreter to a valid
158// Interpreter. Note: the user must ensure the model lifetime is at least as
159// long as interpreter's lifetime.
160class InterpreterBuilder {
161 public:
162 InterpreterBuilder(const FlatBufferModel& model,
163 const OpResolver& op_resolver);
164 // Builds an interpreter given only the raw flatbuffer Model object (instead
165 // of a FlatBufferModel). Mostly used for testing.
166 // If `error_reporter` is null, then DefaultErrorReporter() is used.
167 InterpreterBuilder(const ::tflite::Model* model,
168 const OpResolver& op_resolver,
169 ErrorReporter* error_reporter = DefaultErrorReporter());
170 InterpreterBuilder(const InterpreterBuilder&) = delete;
171 InterpreterBuilder& operator=(const InterpreterBuilder&) = delete;
172 TfLiteStatus operator()(std::unique_ptr<Interpreter>* interpreter);
173 TfLiteStatus operator()(std::unique_ptr<Interpreter>* interpreter,
174 int num_threads);
175
176 private:
177 TfLiteStatus BuildLocalIndexToRegistrationMapping();
178 TfLiteStatus ParseNodes(
179 const flatbuffers::Vector<flatbuffers::Offset<Operator>>* operators,
180 Interpreter* interpreter);
181 TfLiteStatus ParseTensors(
182 const flatbuffers::Vector<flatbuffers::Offset<Buffer>>* buffers,
183 const flatbuffers::Vector<flatbuffers::Offset<Tensor>>* tensors,
184 Interpreter* interpreter);
185
186 const ::tflite::Model* model_;
187 const OpResolver& op_resolver_;
188 ErrorReporter* error_reporter_;
189
190 std::vector<TfLiteRegistration*> flatbuffer_op_index_to_registration_;
191 std::vector<BuiltinOperator> flatbuffer_op_index_to_registration_types_;
192 const Allocation* allocation_ = nullptr;
193};
