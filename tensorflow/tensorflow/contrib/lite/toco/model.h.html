136// Helper to deal with TensorFlow arrays using a different ordering of
137// dimensions
138// ("axes") than our own.
139// TODO(benoitjacob): Ultimately, we shouldn't have any "ordering" of axes,
140// we should have associative arrays mapping symbolic axes identifiers (like
141// "output_depth") to dimensions. We would then not need this anymore.
142enum class AxesOrder {
143 kOneAxis, // one-dimensional array, one unique axis.
144 kCR, // column-major matrix storage order. Our standard.
145 kRC, // row-major matrix storage order. TensorFlow default.
146 kOHWI, // Our standard for conv weights
147 kHWIO, // TensorFlow conv weights
148 k1HWO, // Our standard for DepthwiseConv weights
149 kHWIM, // TensorFlow DepthwiseConv weights
150 kNHWC, // TensorFlow activations
151};

153// The type of the scalars in an array.
154// Note that that does not by itself tell whether the values in the array are
155// real (are literally interpreted as real numbers) or quantized (only acquire
156// a meaning as real numbers in conjunction with QuantizationParams).
157//
158// In practice though:
159// float values are always real
160// uint8 values are always quantized
161// int32 values are either real or quantized (depending on whether
162// QuantizationParams are present).
163// other types are unused at the moment.
164//
165// kNone means that we don't know the data type yet, or that we don't care
166// because we'll be dropping the array anyway (e.g. some exotic array types
167// may be involved only in debug-only subgraphs that we may not be interested
168// in actually supporting).
169enum class ArrayDataType {
170 kNone, // 0
171 kBool,
172 kFloat,
173 kInt8,
174 kUint8,
175 kInt16, // 5
176 kUint16,
177 kInt32,
178 kUint32,
179 kInt64,
180 kUint64, // 10
181 kString
182};
183
184// Compile-time logic to map ArrayDataType to the corresponding C++ scalar type
185template <ArrayDataType A>
186struct DataTypeImpl {};
187template <>
188struct DataTypeImpl<ArrayDataType::kNone> {
189 typedef int Type;
190};
191template <>
192struct DataTypeImpl<ArrayDataType::kBool> {
193 typedef bool Type;
194};
195template <>
196struct DataTypeImpl<ArrayDataType::kFloat> {
197 typedef float Type;
198};
199template <>
200struct DataTypeImpl<ArrayDataType::kInt8> {
201 typedef int8 Type;
202};
203template <>
204struct DataTypeImpl<ArrayDataType::kUint8> {
205 typedef uint8 Type;
206};
207template <>
208struct DataTypeImpl<ArrayDataType::kInt16> {
209 typedef int16 Type;
210};
211template <>
212struct DataTypeImpl<ArrayDataType::kUint16> {
213 typedef uint16 Type;
214};
215template <>
216struct DataTypeImpl<ArrayDataType::kInt32> {
217 typedef int32 Type;
218};
219template <>
220struct DataTypeImpl<ArrayDataType::kUint32> {
221 typedef uint32 Type;
222};
223template <>
224struct DataTypeImpl<ArrayDataType::kInt64> {
225 typedef int64 Type;
226};
227template <>
228struct DataTypeImpl<ArrayDataType::kUint64> {
229 typedef uint64 Type;
230};
231template <>

233 typedef string Type;
234};

239// Base class for type-specific buffer types.
240struct GenericBuffer {
241 // Non-default-constructible: only ArrayDataType-specific subclass
242 // objects may be constructed.
243 GenericBuffer() = delete;
244 // Non-copyable-or-movable: we should only store pointers-to-Buffer
245 // in containers, not Operators themselves, so there should be no
246 // copy or move.
247 GenericBuffer(const GenericBuffer&) = delete;
248 GenericBuffer(const GenericBuffer&&) = delete;
249
250 // We need a virtual destructor so we can store pointers-to-Buffer
251 // in containers and have the containers call the right subclass destructor.
252 virtual ~GenericBuffer() {}
253
254 virtual int Length() const = 0;
255
256 const ArrayDataType type;
257
258 protected:
259 // Constructor used by subclasses for specific ArrayDataType's.
260 explicit GenericBuffer(ArrayDataType t) : type(t) {}
261};
262
263// Type-specific buffer, containing type-specific storage.
264template <ArrayDataType A>
265struct Buffer : GenericBuffer {
266 Buffer() : GenericBuffer(A) {}
267
268 int Length() const override { return data.size(); }
269
270 std::vector<DataType<A>> data;
271};

273// Base class for all operator classes.
274struct Operator {
275 // Non-default-constructible: only OperatorType-specific subclass
276 // objects may be constructed.
277 Operator() = delete;
278 // Non-copyable-or-movable: we should only store pointers-to-Operator
279 // in containers, not Operators themselves, so there should be no
280 // copy or move.
281 Operator(const Operator&) = delete;
282 Operator(const Operator&&) = delete;
283
284 // We need a virtual destructor so we can store pointers-to-Operator
285 // in containers and have the containers call the right subclass destructor.
286 virtual ~Operator() {}
287
288 // The specific type of operator. Corresponds 1:1 to subclasses.
289 const OperatorType type;
290
291 // The activation function that may be fused into this operator,
292 // or None if no activation function is fused.
293 FusedActivationFunctionType fused_activation_function;
294
295 // Input arrays: either activation arrays or constant array parameters.
296 // We refer to them by their name, not by their address; the mapping of
297 // names to addresses is given by the Model, which owns both Operator's and
298 // Array's. Thus, an Operator on its own doesn't contain much information,
299 // it is meant to be used in conjunction with the Model that owns it.
300 std::vector<string> inputs;
301
302 // Output activation arrays. Same comments as for inputs apply here too.
303 std::vector<string> outputs;
304
305 // If true, the array has more outputs than are listed in the 'outputs'
306 // member. These need to be resolved by some graph transformation.
307 // This flag is only here to indicate that an operator should not be
308 // discarded as unused, even if from its 'outputs' member alone it
309 // looks unused.
310 bool unresolved_outputs = false;
311
312 protected:
313 // Constructor used by subclasses for specific OperatorType's.
314 explicit Operator(OperatorType t)
315 : type(t),
316 fused_activation_function(FusedActivationFunctionType::kNone) {}
317};

319// Padding types for Conv-like operators. This is how padding is typically
320// specified in model files. But for inference, we will need to resolve this
321// to a FixedPadding, see below.
322enum class PaddingType { kNone, kSame, kValid };
323
324// Padding as resolved for a specific layer shape, as needed for inference.
325// For a given layer shape, a given padding type will resolve to a choice of
326// a number of padding rows and columns, which we call the padding height and
327// width respectively.
328struct FixedPadding {
329 int width = 0;
330 int height = 0;
331};
332
333// "Universal" padding struct containing both a generic PaddingType (as
334// represented in a model file), and a FixedPadding (as needed for inference).
335// The latter is resolved during the PropagateFixedSizes pass.
336struct Padding {
337 FixedPadding& GetOrCreateFixedPadding() {
338 if (!fixed) {
339 FixedPadding* ptr = new FixedPadding;
340 fixed = std::unique_ptr<FixedPadding>(ptr);
341 }
342 return *fixed;
343 }
344
345 Padding() : type(PaddingType::kNone) {}
346 PaddingType type;
347 std::unique_ptr<FixedPadding> fixed;
348};

350// "Convolutional" layer, as represented in model files.
351//
352// Inputs:
353// inputs[0]: required: the input activations array
354// inputs[1]: required: the Conv weights
355// inputs[2]: optional: the bias vector, specifying the biases for each output
356// channel.
357//
358// Outputs:
359// outputs[0]: required: the output activations array
360// outputs[1]: optional: the intermediate array of im2col-replicated input
361// activations. Present when targeting implementations
362// of Conv layers as Im2col+GEMM.
363//
364// TensorFlow equivalent: Conv2D
365struct ConvOperator : Operator {
366 ConvOperator() : Operator(OperatorType::kConv) {}
367 Padding padding;
368 int stride_width = 0;
369 int stride_height = 0;
370 // A dilation_rate of 0 is invalid and this field is an optional attribute.
371 // Thus initializing it to 1 to allow default conv behavior when the
372 // attribute is not present.
373 int dilation_width_factor = 1;
374 int dilation_height_factor = 1;
375};

430// Dequantization operator, converting a quantized array of integers with
431// quantization parameters specifying how these integers correspond to real
432// numbers
433// (see QuantizationParams) to an output activations array of floating-point
434// values.
435//
436// In floating-point image models, there is typically a Dequantization operator
437// at the very beginning, converting the input image RGB data, consisting of
438// uint8 integer values, to floating-point input activations. That is where
439// image model parameters such as "mean_value" and "std_value" are typically
440// handled.
441//
442// This is the only operator type that converts from quantized to
443// floating-point,
444// and there is at the moment no operator type at all to convert from
445// floating-point
446// to quantized. Every other operator does either float->float or
447// quantized->quantized.
448//
449// Inputs:
450// inputs[0]: required: the input quantized activations array
451//
452// TensorFlow equivalent: Dequantize
453struct DequantizeOperator : Operator {
454 DequantizeOperator() : Operator(OperatorType::kDequantize) {}
455};

457// Batch-normalization operator.
458//
459// We only support batch-normalization using pre-learned moments, so this is
460// just
461// computing (input - mean) * multiplier + offset. As such, this can be
462// expressed as a combination of Add and Mul nodes, and indeed this is how
463// we break it down during tooling for the purpose of fusing it into
464// other operators.
465//
466// Inputs:
467// inputs[0]: required: the input activations array
468// inputs[1]: required: the learned mean array
469// inputs[2]: required: the learned multiplier array
470// inputs[3]: required: the learned offset array
471//
472// TensorFlow equivalent: a combination of Add and Mul nodes
473struct BatchNormalizationOperator : Operator {
474 BatchNormalizationOperator()
475 : Operator(OperatorType::kBatchNormalization),
476 global_normalization(false) {}
477 bool global_normalization;
478};

494// LSTM Cell operator.
495//
496// Inputs:
497// inputs[0]: required: the input data array
498// inputs[1]: required: the previous output activations array
499// inputs[2]: required: the learned weights array
500// inputs[3]: required: the learned biases array
501// inputs[4]: required: the previous output state
502// outputs[0]: required: the output activations array
503// outputs[1]: required: the new state array
504//
505// TensorFlow equivalent: none. In TensorFlow, an LSTM is implemented
506// with a sub-graph of lower-level arithmetic nodes; during tooling, we identify
507// such sub-graphs and replace them with LstmCells. See IdentifyLstmCell().
508struct LstmCellOperator : Operator {
509 enum Inputs {
510 DATA_INPUT = 0,
511 PREV_ACTIV_INPUT = 1,
512 WEIGHTS_INPUT = 2,
513 BIASES_INPUT = 3,
514 PREV_STATE_INPUT = 4,
515 NUM_INPUTS = 5
516 };
517 enum Outputs {
518 ACTIV_OUTPUT = 0,
519 STATE_OUTPUT = 1,
520 CONCAT_TEMP = 2,
521 ACTIV_TEMP = 3,
522 NUM_OUTPUTS = 4
523 };
524 LstmCellOperator() : Operator(OperatorType::kLstmCell) {}
525};

724// The expected [min, max] range of values in a given array.
725// Used for quantization only.
726// This information typically comes from special nodes found in quantized
727// models,
728// see FakeQuantOperator, and is used during quantization to resolve
729// actual quantization parameters (see QuantizationParams).
730struct MinMax {
731 double min = 0.;
732 double max = 0.;
733};
734
735inline bool operator==(const MinMax& m1, const MinMax& m2) {
736 return m1.min == m2.min && m1.max == m2.max;
737}

1541// Array represents an array (either a constant parameter array or an
1542// activations array) in a Model.
1543struct Array {
1544 template <ArrayDataType A>
1545 const Buffer<A>& GetBuffer() const {
1546 DCHECK(buffer);
1547 DCHECK(buffer->type == A);
1548 return *static_cast<const Buffer<A>*>(buffer.get());
1549 }
1550 template <ArrayDataType A>
1551 Buffer<A>& GetMutableBuffer() {
1552 if (!buffer) {
1553 Buffer<A>* ptr = new Buffer<A>;
1554 buffer = std::unique_ptr<GenericBuffer>(ptr);
1555 }
1556 DCHECK(buffer);
1557 DCHECK(buffer->type == A);
1558 return *static_cast<Buffer<A>*>(buffer.get());
1559 }
1560 Alloc& GetOrCreateAlloc() {
1561 if (!alloc) {
1562 alloc = std::unique_ptr<Alloc>(new Alloc);
1563 }
1564 return *alloc;
1565 }
1566 MinMax& GetOrCreateMinMax() {
1567 if (!minmax) {
1568 minmax = std::unique_ptr<MinMax>(new MinMax);
1569 }
1570 return *minmax;
1571 }
1572 MinMax& GetMinMax() const {
1573 DCHECK(minmax);
1574 return *minmax;
1575 }
1576 QuantizationParams& GetOrCreateQuantizationParams() {
1577 if (!quantization_params) {
1578 quantization_params =
1579 std::unique_ptr<QuantizationParams>(new QuantizationParams);
1580 }
1581 return *quantization_params;
1582 }
1583 QuantizationParams& GetQuantizationParams() const {
1584 DCHECK(quantization_params);
1585 return *quantization_params;
1586 }
1587
1588 // The data type of the actual elements of this array, that is:
1589 // - If there is a buffer (see 'buffer' member), it must be of the same
1590 // type.
1591 // - If there is no buffer, meaning that this is a runtime (i.e. activations)
1592 // array, then this specifies the type of elements that there will be
1593 // at runtime.
1594 //
1595 // Note that this only specifies the storage type of elements; this does
1596 // not specify whether these are to be treated as 'real' or 'quantized'
1597 // values.
1598 // That is decided by whether the 'quantization_params' member is null.
1599 ArrayDataType data_type = ArrayDataType::kNone;
1600 // The final value that data_type should have at the end of graph
1601 // transformations
1602 ArrayDataType final_data_type = ArrayDataType::kNone;
1603 // The dimensions of this array --- this specifies both sizes and strides
1604 // (the storage layout).
1605 //
1606 // Issues with shape handling that remain include:
1607 // - No way to distinguish between 0-dimensional dims and missing dims.
1608 // - No way to describe dims that may be runtime-variable.
1609 // - Addressing of dims by integer index differs in different graph formats
1610 // (TensorFlow vs. other frameworks vs. what we have informally grown
1611 // within toco).
1612 // This is currently quite messy; see ReorderAxesOperator which is how we
1613 // bridge some of these discrepancies at the moment. This is overdue for
1614 // a redesign; I'm thinking that it would be nice to have more flexible
1615 // dims that allow mapping 1:1, cleanly, dims as they are in various
1616 // formats,
1617 // then explicitly convert between different conventions.
1618
1619 // Proto-style accessors
1620 bool has_shape() const { return array_shape != nullptr; }
1621 const Shape& shape() const {
1622 CHECK(has_shape());
1623 return *array_shape;
1624 }
1625 Shape* mutable_shape() {
1626 if (!array_shape) {
1627 array_shape.reset(new Shape);
1628 }
1629 return array_shape.get();
1630 }
1631 void copy_shape(const Shape& src_shape) { *mutable_shape() = src_shape; }
1632 void clear_shape() { array_shape = nullptr; }
1633
1634 // The constant buffer backing this array. This is non-null if and only if
1635 // this is a constant parameter array. Conversely, this is null for
1636 // activations arrays.
1637 //
1638 // Note that this buffer is pure storage. In the case of quantized values,
1639 // it only stores the quantized values, it does not know by itself about the
1640 // quantization parameters necessary to interprete these values, that is
1641 // in the separate 'quantization_params' field. In fact, this 'buffer' field
1642 // does no even know whether values are quantized. It only has a data_type,
1643 // which must equal the 'data_type' member here, and which only describes
1644 // the storage type of element, does not tell whether they are quantized i.e.
1645 // whether they are to be interpreted with quantization_params.
1646 std::unique_ptr<GenericBuffer> buffer;
1647 // Only for activation arrays (i.e. when 'buffer' is null).
1648 // Only for code generation.
1649 //
1650 // Describes the allocation of this array within the workspace buffer
1651 // allocated
1652 // for all transient arrays.
1653 std::unique_ptr<Alloc> alloc;
1654 // Describes the [min, max] range of values
1655 // to be assumed when determining quantization_params.
1656 //
1657 // Only used for quantization. In fact, only used for determining
1658 // quantization_params.
1659 //
1660 // Used for both constant arrays (those having a 'buffer') and non-constant
1661 // arrays (activations). Indeed, it is important to use the same min-max range
1662 // as was used during training, even if that min-max range is slightly wrong
1663 // w.r.t. actual buffer elements. Doing otherwise would defeat the point of
1664 // re-training for quantization.
1665 std::unique_ptr<MinMax> minmax;
1666 // Quantization parameters. The non-null-ness of this pointer is what
1667 // defines whether this array is quantized or not.
1668 //
1669 // If this is non-null, then these quantization parameters are to be used
1670 // to assign a meaning as real numbers to the elements of this array.
1671 std::unique_ptr<QuantizationParams> quantization_params;
1672
1673 private:
1674 std::unique_ptr<Shape> array_shape;
1675};

1677// Our Model struct, represents an entire model (our "top-level" struct).
1678// Owns everything.
1679class Model {
1680 public:
1681 using ArrayMap = std::unordered_map<string, std::unique_ptr<Array>>;
1682
1683 bool HasArray(const string& name) const { return arrays.count(name) > 0; }
1684 Array& GetArray(const string& name) const {
1685 DCHECK(HasArray(name)) << "Array not found: " << name;
1686 return *arrays.at(name);
1687 }
1688 Array& GetOrCreateArray(const string& name) {
1689 // Make sure name is not used by an optional array
1690 DCHECK(!optional_arrays.count(name));
1691 if (!HasArray(name)) {
1692 Array* ptr = new Array;
1693 arrays[name] = std::unique_ptr<Array>(ptr);
1694 }
1695 Array& result = GetArray(name);
1696 return result;
1697 }
1698 void CreateOptionalArray(const string& name) {
1699 DCHECK(!arrays.count(name) && !optional_arrays.count(name));
1700 optional_arrays.insert(name);
1701 }
1702 bool IsOptionalArray(const string& name) const {
1703 return optional_arrays.count(name);
1704 }
1705
1706 // Note that this invalidates all array iterators.
1707 void EraseArray(const string& name) { arrays.erase(name); }
1708 void EraseArrays(std::function<bool(const string&)> discardable) {
1709 for (auto it = arrays.begin(); it != arrays.end();) {
1710 if (discardable(it->first)) {
1711 it = arrays.erase(it);
1712 } else {
1713 ++it;
1714 }
1715 }
1716 }
1717 const ArrayMap& GetArrayMap() const { return arrays; }
1718
1719 // Optional arrays are used for optional tensors,
1720 // these tensors do not have data, but with reserved names as op inputs.
1721 std::set<string> optional_arrays;
1722
1723 // The list of operators. Notice how it's a list of unique_ptr's, implying
1724 // that the Model is what owns Operator's and keeps them alive.
1725 std::vector<std::unique_ptr<Operator>> operators;
1726
1727 // Generic flags, a place where we combine information passed to us via
1728 // command-line parameters (e.g. --input_width=N) with information that
1729 // we may or may not find in the input model file.
1730 ModelFlags flags;
1731 // For code-generation only: required size of the transient_data buffer
1732 std::size_t transient_data_size = 0;
1733 // For code-generation only: required alignment of the transient_data buffer
1734 std::size_t transient_data_alignment = 0;
1735
1736 private:
1737 // The associative array mapping names to Array's.
1738 // Notice how it's a container of unique_ptr's, implying
1739 // that the Model is what owns Array's and keeps them alive.
1740 // The Operator's refer to these Array's by their name strings, not by their
1741 // addresses. See Operator::inputs, Operator::outputs.
1742 std::unordered_map<string, std::unique_ptr<Array>> arrays;
1743};
