23void Transpose2DTensor(const float* tensor, int row, int col,
24 float* transposed_tensor) {
25 float* result = transposed_tensor;
26 for (int r = 0; r < row; ++r) {
27 for (int c = 0; c < col; ++c) {
28 *(result + c * row) = *tensor++;
29 }
30 ++result;
31 }
32}
