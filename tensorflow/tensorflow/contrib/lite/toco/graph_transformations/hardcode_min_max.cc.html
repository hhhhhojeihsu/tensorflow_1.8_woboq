29bool HardcodeMinMaxForIm2colArray(Model* model, Operator* op) {
30 if (op->outputs.size() != 2) {
31 return false;
32 }
33 auto& im2col_array = model->GetArray(op->outputs[1]);
34 if (im2col_array.minmax) {
35 return false;
36 }
37 const auto& input_array = model->GetArray(op->inputs[0]);
38 if (!input_array.minmax) {
39 return false;
40 }
41 const auto& input_minmax = input_array.GetMinMax();
42 CHECK(!im2col_array.minmax);
43 auto& im2col_minmax = im2col_array.GetOrCreateMinMax();
44 im2col_minmax.min = input_minmax.min;
45 im2col_minmax.max = input_minmax.max;
46 return true;
47}

49bool HardcodeMinMaxForL2Normalization(Model* model, Operator* op) {
50 auto& output_array = model->GetArray(op->outputs[0]);
51 if (output_array.minmax) {
52 return false;
53 }
54 const auto& input_array = model->GetArray(op->inputs[0]);
55 if (!input_array.minmax) {
56 return false;
57 }
58 const auto& input_minmax = input_array.GetMinMax();
59 CHECK(!output_array.minmax);
60 auto& output_minmax = output_array.GetOrCreateMinMax();
61 output_minmax.min = input_minmax.min >= 0. ? 0. : -1.;
62 output_minmax.max = input_minmax.max <= 0. ? 0. : 1.;
63 return true;
64}

66bool HardcodeMinMaxForConcatenation(Model* model, Operator* op) {
67 // Do not early return if the output already has min/max:
68 // we may still need to adjust the inputs min/max.
69 bool has_minmax = false;
70 double overall_min = std::numeric_limits<double>::infinity();
71 double overall_max = -std::numeric_limits<double>::infinity();
72 for (const auto& input : op->inputs) {
73 if (model->GetArray(input).minmax) {
74 has_minmax = true;
75 const auto* minmax = model->GetArray(input).minmax.get();
76 if (minmax) {
77 overall_min = std::min(overall_min, minmax->min);
78 overall_max = std::max(overall_max, minmax->max);
79 }
80 }
81 }
82 auto& output = model->GetArray(op->outputs[0]);
83 if (output.minmax) {
84 has_minmax = true;
85 const auto* minmax = model->GetArray(op->outputs[0]).minmax.get();
86 if (minmax) {
87 overall_min = std::min(overall_min, minmax->min);
88 overall_max = std::max(overall_max, minmax->max);
89 }
90 }
91 if (!has_minmax) {
92 return false;
93 }
94 MinMax overall_minmax;
95 overall_minmax.min = overall_min;
96 overall_minmax.max = overall_max;
97 bool changed = false;
98 if (model->flags.change_concat_input_ranges()) {
99 for (const auto& input : op->inputs) {
100 auto& array = model->GetArray(input);
101 if (!array.minmax) {
102 changed = true;
103 } else if (!(overall_minmax == array.GetMinMax())) {
104 changed = true;
105 LOG(WARNING)
106 << "Tweaking the MinMax of array " << input << ", which is "
107 << "an input to " << LogName(*op) << ", because we want all inputs "
108 << "and outputs of a Concatenation operator to have the same "
109 << "MinMax so that it can be implemented as a pure byte-copy, no "
110 "arithmetic.";
111 }
112 array.GetOrCreateMinMax() = overall_minmax;
113 }
114 }
115 if (!output.minmax) {
116 changed = true;
117 } else if (!(overall_minmax == output.GetMinMax())) {
118 if (model->flags.change_concat_input_ranges()) {
119 changed = true;
120 LOG(WARNING)
121 << "Tweaking the MinMax of the output array of " << LogName(*op)
122 << ", because we want all inputs "
123 << "and outputs of a Concatenation operator to have the same MinMax "
124 << "so that it can be implemented as a pure byte-copy, no "
125 << "arithmetic.";
126 } else {
127 return false;
128 }
129 }
130 output.GetOrCreateMinMax() = overall_minmax;
131
132 return changed;
133}

135bool HardcodeMinMaxForSplit(Model* model, Operator* op) {
136 for (const auto& output : op->outputs) {
137 if (model->GetArray(output).minmax) {
138 LOG(WARNING) << "Skipping min-max setting for " << LogName(*op)
139 << " because output " << output << " already has min-max.";
140 return false;
141 }
142 }
143 // Data is in second input.
144 auto& input_array = model->GetArray(op->inputs[1]);
145 if (!input_array.minmax) {
146 return false;
147 } else {
148 for (const auto& output : op->outputs) {
149 auto& array = model->GetArray(output);
150 array.GetOrCreateMinMax() = *input_array.minmax;
151 }
152 return true;
153 }
154}

156// The output of average or max pooling is within the same range as its input.
157bool HardcodeMinMaxForAverageOrMaxPool(Model* model, Operator* op) {
158 auto& output_array = model->GetArray(op->outputs[0]);
159 if (output_array.minmax) {
160 return false;
161 }
162 const auto& input_array = model->GetArray(op->inputs[0]);
163 if (!input_array.minmax) {
164 return false;
165 }
166 const auto& input_minmax = input_array.GetMinMax();
167 CHECK(!output_array.minmax);
168 auto& output_minmax = output_array.GetOrCreateMinMax();
169 output_minmax.min = std::min(input_minmax.min, 0.);
170 output_minmax.max = std::max(input_minmax.max, 0.);
171 return true;
172}

174bool HardcodeMinMaxFromFirstInput(Model* model, Operator* op) {
175 auto& output_array = model->GetArray(op->outputs[0]);

177 return false;
178 }
179 const auto& input_array = model->GetArray(op->inputs[0]);

181 return false;
182 }
183 const auto& input_minmax = input_array.GetMinMax();
184 CHECK(!output_array.minmax);
185 auto& output_minmax = output_array.GetOrCreateMinMax();
186 output_minmax.min = input_minmax.min;
187 output_minmax.max = input_minmax.max;
188 return true;

191bool HardcodeMinMaxForOutput(Model* model, Operator* op, double min,
192 double max) {
193 CHECK_EQ(op->outputs.size(), 1);
194 auto& output_array = model->GetArray(op->outputs[0]);
195 if (output_array.minmax) {
196 return false;
197 }
198 const auto& input_array = model->GetArray(op->inputs[0]);
199 if (!input_array.minmax) {
200 return false;
201 }
202 CHECK(!output_array.minmax);
203 auto& output_minmax = output_array.GetOrCreateMinMax();
204 output_minmax.min = min;
205 output_minmax.max = max;
206 return true;
207}

209// Propagates MinMax from any of the listed arrays, to all others.
210// If multiple of these arrays have MinMax, then these are required
211// to agree with each other.
212bool PropagateMinMaxAmongArrays(Model* model,
213 const std::vector<string> array_names) {
214 string reference_array_name;
215 MinMax* reference_minmax = nullptr;
216 for (const string& array_name : array_names) {
217 if (model->GetArray(array_name).minmax) {
218 reference_array_name = array_name;
219 reference_minmax = model->GetArray(array_name).minmax.get();
220 break;
221 }
222 }
223 // No MinMax info is available to propagate.
224 if (!reference_minmax) {
225 return false;
226 }
227 bool changed = false;
228 for (const string& array_name : array_names) {
229 auto& array = model->GetArray(array_name);
230 if (array.minmax) {
231 CHECK(*array.minmax == *reference_minmax)
232 << "Both the following arrays have minmax, and they disagree: "
233 << reference_array_name << " (" << reference_minmax->min << ","
234 << reference_minmax->max << ") and " << array_name << " ("
235 << array.minmax->min << "," << array.minmax->max
236 << "). Expected that either only one of them would have minmax, or "
237 "at "
238 "least that they would agree.";
239 } else {
240 array.GetOrCreateMinMax() = *reference_minmax;
241 changed = true;
242 }
243 }
244 return changed;
245}

247bool HardcodeMinMaxForLstmCell(Model* model, Operator* op) {
248 CHECK_EQ(op->inputs.size(), LstmCellOperator::NUM_INPUTS);
249 CHECK_EQ(op->outputs.size(), LstmCellOperator::NUM_OUTPUTS);
250
251 bool changed = false;
252 changed |= PropagateMinMaxAmongArrays(
253 model, {op->inputs[LstmCellOperator::PREV_STATE_INPUT],
254 op->outputs[LstmCellOperator::STATE_OUTPUT]});
255
256 auto& input_activations =
257 model->GetArray(op->inputs[LstmCellOperator::DATA_INPUT]);
258 if (!input_activations.minmax) {
259 auto& minmax = input_activations.GetOrCreateMinMax();
260 minmax.min = -1;
261 minmax.max = 127. / 128.;
262 changed = true;
263 }
264
265 auto& prev_output_activations =
266 model->GetArray(op->inputs[LstmCellOperator::PREV_ACTIV_INPUT]);
267 if (!prev_output_activations.minmax) {
268 auto& minmax = prev_output_activations.GetOrCreateMinMax();
269 minmax.min = -1;
270 minmax.max = 127. / 128.;
271 changed = true;
272 }
273
274 auto& output_concat_temp =
275 model->GetArray(op->outputs[LstmCellOperator::CONCAT_TEMP]);
276 if (!output_concat_temp.minmax) {
277 auto& minmax = output_concat_temp.GetOrCreateMinMax();
278 minmax.min = -1;
279 minmax.max = 127. / 128.;
280 changed = true;
281 }
282
283 auto& output_activations =
284 model->GetArray(op->outputs[LstmCellOperator::ACTIV_OUTPUT]);
285 if (!output_activations.minmax) {
286 auto& minmax = output_activations.GetOrCreateMinMax();
287 minmax.min = -1;
288 minmax.max = 127. / 128.;
289 changed = true;
290 }
291
292 // (This comment should morph into proper documentation for
293 // quantization of LSTM models. It isn't just a local implementation detail,
294 // the training code for LSTM models needs to be adjusted to that.)
295 //
296 // Finally, output_activations_temp holds the output of the fully-connected
297 // node inside the LSTM cell. For it, we hardcode a minmax of [-8, 8].
298 // The rationale for that is given in a lengthy comment on the LstmCell
299 // quantized runtime implementation in reference_ops.h.
300 auto& output_activations_temp =
301 model->GetArray(op->outputs[LstmCellOperator::ACTIV_TEMP]);
302 if (!output_activations_temp.minmax) {
303 auto& minmax = output_activations_temp.GetOrCreateMinMax();
304 minmax.min = -8;
305 minmax.max = 8 * 32767. / 32768.;
306 changed = true;
307 }
308
309 return changed;
310}

313bool HardcodeMinMax::Run(Model* model, std::size_t op_index) {
314 auto it = model->operators.begin() + op_index;
315 auto* op = it->get();
316 bool changed = false;
317 switch (op->type) {
318 case OperatorType::kConv:
319 changed = HardcodeMinMaxForIm2colArray(model, op);
320 break;
321
322 case OperatorType::kL2Normalization:
323 changed = HardcodeMinMaxForL2Normalization(model, op);
324 break;
325
326 case OperatorType::kConcatenation:
327 changed = HardcodeMinMaxForConcatenation(model, op);
328 break;
329
330 case OperatorType::kTensorFlowSplit:
331 changed = HardcodeMinMaxForSplit(model, op);
332 break;
333
334 case OperatorType::kAveragePool:
335 case OperatorType::kMaxPool:
336 changed = HardcodeMinMaxForAverageOrMaxPool(model, op);
337 break;
338
339 case OperatorType::kStridedSlice:
340 case OperatorType::kSqueeze:
341 case OperatorType::kTensorFlowReshape:
342 case OperatorType::kPad:
343 case OperatorType::kGather:
344 case OperatorType::kTranspose:
345 case OperatorType::kMean:
346 changed = HardcodeMinMaxFromFirstInput(model, op);
347 break;
348
349 case OperatorType::kLogistic:
350 // We hardcode quantization_params to: zero_point=0, scale=1/256.
351 // This choice of minmax is the one that is equivalent to that.
352 changed = HardcodeMinMaxForOutput(model, op, 0, 255. / 256.);
353 break;
354
355 case OperatorType::kSoftmax:
356 // We hardcode quantization_params to: zero_point=0, scale=1/256.
357 // This choice of minmax is the one that is equivalent to that.
358 changed = HardcodeMinMaxForOutput(model, op, 0, 255. / 256.);
359 break;
360
361 case OperatorType::kTanh:
362 // We hardcode quantization_params to: zero_point=127, scale=1/128.
363 // This choice of minmax is the one that is equivalent to that.
364 changed = HardcodeMinMaxForOutput(model, op, -127. / 128., 1.0);
365 break;
366
367 case OperatorType::kLstmCell:
368 changed = HardcodeMinMaxForLstmCell(model, op);
369 break;
370
371 default:
372 break;
373 }
374 if (changed) {
375 AddMessageF("Hardcoded min-max through %s", LogName(*op));
376 }
377 return changed;
378}
