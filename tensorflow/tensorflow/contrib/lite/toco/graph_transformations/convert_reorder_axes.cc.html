28// Creates a Reshape operator from ReorderAxes operator.
29TensorFlowReshapeOperator* CreateReshapeFromReorderAxes(
30 Model* model, ReorderAxesOperator* reorder_op, const Shape& input_shape) {
31 auto* reshape_op = new TensorFlowReshapeOperator;
32
33 // Copy inputs and outputs to Reshape.
34 reshape_op->inputs.push_back(reorder_op->inputs[0]);
35 reshape_op->outputs = reorder_op->outputs;
36
37 // Create reshape dimensions based on input shape. Conversion from
38 // ReorderAxes to Reshape requires a 4D input shape.
39 CHECK_EQ(input_shape.dimensions_count(), 4);
40 std::vector<int> reshape_dims = {1, input_shape.dims(0), input_shape.dims(1),
41 input_shape.dims(3) * input_shape.dims(2)};
42
43 // Create a new input array for Reshape.
44 string reshape_array_name =
45 AvailableArrayName(*model, reshape_op->outputs[0]);
46 reshape_op->inputs.push_back(reshape_array_name);
47
48 Array& reshape_array = model->GetOrCreateArray(reshape_array_name);
49 *(reshape_array.mutable_shape()->mutable_dims()) = {
50 1, static_cast<int>(reshape_dims.size())};
51 reshape_array.data_type = ArrayDataType::kInt32;
52 auto& reshape_buffer =
53 reshape_array.GetMutableBuffer<ArrayDataType::kInt32>();
54 reshape_buffer.data = reshape_dims;
55
56 return reshape_op;
57}

59// Creates a Transpose operator from ReorderAxes operator.
60TransposeOperator* CreateTransposeFromReorderAxes(
61 Model* model, ReorderAxesOperator* reorder_op, const Shape& input_shape,
62 const AxesOrder& input_axes_order, const AxesOrder& output_axes_order) {
63 auto* transpose_op = new TransposeOperator;
64
65 // Copy inputs and outputs to Transpose.
66 transpose_op->inputs.push_back(reorder_op->inputs[0]);
67 transpose_op->outputs = reorder_op->outputs;
68
69 // Create permutations data based on input and output axes order.
70 std::vector<int> permutations_data;
71 GetShuffleShape(input_axes_order, output_axes_order, &permutations_data);
72
73 // Create a new input permutations array for Transpose.
74 string perm_array_name = AvailableArrayName(*model, transpose_op->outputs[0]);
75 transpose_op->inputs.push_back(perm_array_name);
76
77 Array& perm_array = model->GetOrCreateArray(perm_array_name);
78 *(perm_array.mutable_shape()->mutable_dims()) = {
79 static_cast<int>(permutations_data.size())};
80 perm_array.data_type = ArrayDataType::kInt32;
81 auto& perm_buffer = perm_array.GetMutableBuffer<ArrayDataType::kInt32>();
82 perm_buffer.data = permutations_data;
83
84 return transpose_op;
85}

87// Converts ReorderAxes into Transpose and Reshape which are compatible with the
88// TFLite interpreter.
89bool ConvertReorderAxes::Run(Model* model, std::size_t op_index) {
90 auto reorder_it = model->operators.begin() + op_index;
91 if (reorder_it->get()->type != OperatorType::kReorderAxes) return false;
92
93 auto* reorder_op = static_cast<ReorderAxesOperator*>(reorder_it->get());
94 CHECK_EQ(reorder_op->inputs.size(), 1);
95 CHECK_EQ(reorder_op->outputs.size(), 1);
96
97 const auto& input_array_name = reorder_op->inputs[0];
98 const auto& output_array_name = reorder_op->outputs[0];
99 auto& input_array = model->GetArray(input_array_name);
100 auto& output_array = model->GetArray(output_array_name);
101
102 // Get input array. If kFakeQuant is the input into ReorderAxes, get the input
103 // array passed into kFakeQuant. kFakeQuant op is dropped when possible.
104 string constant_input_array_name = input_array_name;
105 if (!input_array.buffer) {
106 const auto* op_producing_input = GetOpWithOutput(*model, input_array_name);
107 if (op_producing_input &&
108 op_producing_input->type == OperatorType::kFakeQuant) {
109 constant_input_array_name = op_producing_input->inputs[0];
110 }
111 }
112
113 // Yield if input array contains constants or if output array size has not
114 // been adjusted to reflect the permutations in ReorderAxes. ReorderAxes will
115 // be merged into a constant array when possible.
116 if (IsConstantParameterArray(*model, constant_input_array_name)) return false;
117 if (!output_array.has_shape()) return false;
118
119 const auto input_axes_order = reorder_op->input_axes_order;
120 const auto output_axes_order = reorder_op->output_axes_order;
121 const Shape input_shape = input_array.shape();
122
123 // Creates a Reshape or Transpose operator depending on the conversion.
124 if (input_axes_order == AxesOrder::kHWIM &&
125 output_axes_order == AxesOrder::k1HWO) {
126 // Add Reshape operator into the graph. This special case is not just a
127 // permutation. The input dimensions get merged into 3 dimensions while the
128 // order of the elements does not change.
129 auto* reshape_op =
130 CreateReshapeFromReorderAxes(model, reorder_op, input_shape);
131 const auto reshape_it = model->operators.emplace(reorder_it, reshape_op);
132 reorder_it = reshape_it + 1;
133 } else {
134 // Add Transpose operator into the graph.
135 auto* transpose_op = CreateTransposeFromReorderAxes(
136 model, reorder_op, input_shape, input_axes_order, output_axes_order);
137 const auto transpose_it =
138 model->operators.emplace(reorder_it, transpose_op);
139 reorder_it = transpose_it + 1;
140 }
141
142 // Remove ReorderAxes operator from the graph.
143 CHECK_EQ(reorder_it->get(), reorder_op);
144 model->operators.erase(reorder_it);
145
146 return true;
147}
