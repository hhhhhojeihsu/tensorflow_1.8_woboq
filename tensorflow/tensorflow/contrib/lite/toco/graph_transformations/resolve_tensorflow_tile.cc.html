29void RemoveTileOperator(Model* model, Operator* tile_op, Operator* binary_op,
30 int operand_index) {
31 CHECK(tile_op->type == OperatorType::kTensorFlowTile);
32 CHECK_EQ(binary_op->inputs.size(), 2);
33 CHECK_EQ(tile_op->inputs.size(), 2);
34 const string tile_multiplier_array = tile_op->inputs[1];
35 const string tile_output_array = tile_op->outputs[0];
36 binary_op->inputs[operand_index] = tile_op->inputs[0];
37 auto tile_it = model->operators.begin();
38 for (; tile_it != model->operators.end(); ++tile_it) {
39 if (tile_it->get() == tile_op) {
40 break;
41 }
42 }
43 CHECK(tile_it != model->operators.end());
44 CHECK(tile_it->get() == tile_op);
45 model->operators.erase(tile_it);
46 if (!CountOpsWithInput(*model, tile_multiplier_array) &&
47 !GetOpWithOutput(*model, tile_multiplier_array)) {
48 model->EraseArray(tile_multiplier_array);
49 }
50 if (!CountOpsWithInput(*model, tile_output_array)) {
51 model->EraseArray(tile_output_array);
52 }
53}

56bool ResolveTensorFlowTile::Run(Model* model, std::size_t op_index) {
57 const auto binary_it = model->operators.begin() + op_index;
58 auto* binary_op = binary_it->get();
59 // Test for binary ops of types that we know how to resolve
60 if (binary_op->inputs.size() != 2) {
61 return false;
62 }
63 if (binary_op->type != OperatorType::kAdd &&
64 binary_op->type != OperatorType::kMul &&
65 binary_op->type != OperatorType::kSub &&
66 binary_op->type != OperatorType::kDiv) {
67 return false;
68 }
69
70 Operator* const op[2] = {
71 GetOpWithOutput(*model, binary_op->inputs[0]),
72 GetOpWithOutput(*model, binary_op->inputs[1]),
73 };
74
75 // In the unlikely case where both operands are Tile, we can't infer the
76 // output
77 // size without the Tile nodes, so we have to bail out.
78 if (op[0] && op[0]->type == OperatorType::kTensorFlowTile && op[1] &&
79 op[1]->type == OperatorType::kTensorFlowTile) {
80 return false;
81 }
82
83 for (int i = 0; i < 2; i++) {
84 if (op[i] && op[i]->type == OperatorType::kTensorFlowTile) {
85 // We can only remove a Tile operator is no other op than the present
86 // binary op was consuming its tiled output.
87 if (CountOpsWithInput(*model, binary_op->inputs[i]) == 1) {
88 AddMessageF("Removing %s", LogName(*op[i]));
89 RemoveTileOperator(model, op[i], binary_op, i);
90 return true;
91 }
92 }
93 }
94 return false;
95}
