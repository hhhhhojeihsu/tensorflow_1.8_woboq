28template <ArrayDataType Type>
29bool ComputeRandomUniformArray(Model* model, RandomUniformOperator* op) {
30 typedef tensorflow::random::UniformDistribution<
31 tensorflow::random::PhiloxRandom, DataType<Type>>
32 Distribution;
33
34 // Allocate output
35 auto& output_array = model->GetArray(op->outputs[0]);
36 CHECK(output_array.data_type == Type);
37 std::vector<DataType<Type>>& data =
38 output_array.GetMutableBuffer<Type>().data;
39 data.resize(RequiredBufferSizeForShape(output_array.shape()));
40
41 // We use the same random number generator and distribution as TensorFlow to
42 // produce the exact same values given the same seeds. See
43 // tensorflow::functor::FillPhiloxRandomTask<Distribution, false> in
44 // //third_party/tensorflow/core/kernels/random_op.cc for the implementation.
45 tensorflow::random::PhiloxRandom generator(op->seed, op->seed2);
46 Distribution dist;
47
48 // The generator creates Distribution::kResultElementCount samples at a time.
49 size_t offset = 0;
50 size_t num_samples = Distribution::kResultElementCount;
51 while (offset < data.size()) {
52 const typename Distribution::ResultType samples = dist(&generator);
53 std::copy(&samples[0],
54 &samples[0] + std::min(num_samples, data.size() - offset),
55 &data[0] + offset);
56 offset += num_samples;
57 }
58
59 return true;
60}
61
62bool ResolveConstantRandomUniform::Run(Model* model, std::size_t op_index) {
63 const auto it = model->operators.begin() + op_index;
64 auto* base_op = it->get();
65 if (base_op->type != OperatorType::kRandomUniform) {
66 return false;
67 }
68 auto* op = static_cast<RandomUniformOperator*>(base_op);
69
70 CHECK_EQ(op->inputs.size(), 1);
71 CHECK_EQ(op->outputs.size(), 1);
72
73 auto& output_array = model->GetArray(op->outputs[0]);
74 if (output_array.data_type == ArrayDataType::kNone) {
75 // Yield until the output type has been set by PropagateArrayDataTypes
76 return false;
77 }
78
79 if (!output_array.has_shape()) {
80 // Yield until the output shape has been set by PropagateFixedShapes
81 return false;
82 }
83
84 if ((op->seed == 0) && (op->seed2 == 0)) {
85 LOG(WARNING) << "RandomUniform op outputting \"" << op->outputs[0]
86 << "\" is truly random (using /dev/random system entropy). "
87 "Therefore, cannot resolve as constant. Set \"seed\" or "
88 "\"seed2\" attr non-zero to fix this";
89 return false;
90 }
91
92 switch (output_array.data_type) {
93 case ArrayDataType::kFloat:
94 if (!ComputeRandomUniformArray<ArrayDataType::kFloat>(model, op)) {
95 return false;
96 }
97 break;
98 // For future support of double or half.
99 // case ArrayDataType::kDouble...
100 default:
101 LOG(FATAL)
102 << "Unsupported data type given to RandomUniform op with output \""
103 << op->outputs[0] << "\"";
104 break;
105 }
106
107 // Erase input arrays if no longer used
108 toco::DeleteArrayIfUsedOnce(op->inputs[0], model);
109
110 // Erase the operator
111 model->operators.erase(it);
112
113 return true;
114}
