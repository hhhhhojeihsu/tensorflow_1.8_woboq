27bool ResolveTensorFlowMerge::Run(Model* model, std::size_t op_index) {
28 const auto merge_it = model->operators.begin() + op_index;
29 const auto* merge_op = merge_it->get();
30 if (merge_op->type != OperatorType::kTensorFlowMerge) {
31 return false;
32 }
33
34 // We need to yield until this Merge node has only 1 input, which will mean
35 // that that is the selected input. Other graph transformations on other nodes
36 // such as ResolveTensorFlowSwitch, will take care of trimming the
37 // non-selected inputs, so that at some point there will be only 1 input left.
38 if (merge_op->inputs.size() > 1) {
39 AddMessageF("Waiting for %s to be resolved", LogName(*merge_op));
40 return false;
41 }
42
43 // Now that the merge node has 1 input exactly, it is the same as an Identity
44 // node and can be resolved trivially.
45 CHECK_EQ(merge_op->inputs.size(), 1);
46
47 // Update the edges of the graph ahead of removing the node.
48 for (const auto& other_op : model->operators) {
49 for (auto& input : other_op->inputs) {
50 if (input == merge_op->outputs[0]) {
51 input = merge_op->inputs[0];
52 }
53 }
54 }
55
56 // Remove the node and its output array.
57 AddMessageF("Removing already-resolved %s", LogName(*merge_op));
58 model->EraseArray(merge_op->outputs[0]);
59 model->operators.erase(merge_it);
60 return true;
61}
