27bool ResolveSpaceToBatchNDAttributes::Run(Model* model, std::size_t op_index) {
28 const auto op_it = model->operators.begin() + op_index;
29 if (op_it->get()->type != OperatorType::kSpaceToBatchND) return false;
30
31 auto* op = static_cast<SpaceToBatchNDOperator*>(op_it->get());
32
33 // The attributes are resolved only when the 3 attributes (block_shape,
34 // before_paddings, after_paddings) are all constant.
35 if (!op->block_shape.empty()) {
36 return false;
37 }
38
39 const int block_shape_index = 1;
40 const int paddings_index = 2;
41
42 CHECK_EQ(op->inputs.size(), 3);
43 if (!IsConstantParameterArray(*model, op->inputs[block_shape_index]) ||
44 !IsConstantParameterArray(*model, op->inputs[paddings_index]))
45 return false;
46
47 // Handle paddings.
48 const auto& paddings_array = model->GetArray(op->inputs[paddings_index]);
49 if (!paddings_array.has_shape()) return false;
50 const std::vector<int>& paddings_dims = paddings_array.shape().dims();
51 if (paddings_dims.size() != 2) {
52 // Code only handles padding of 2 dimensions. Perhaps another transformation
53 // will delete this op.
54 return false;
55 }
56 std::vector<int> paddings_buffer =
57 paddings_array.GetBuffer<ArrayDataType::kInt32>().data;
58 for (int i = 0; i < paddings_dims[0]; ++i) {
59 op->before_paddings.push_back(paddings_buffer[i * 2]);
60 op->after_paddings.push_back(paddings_buffer[i * 2 + 1]);
61 }
62
63 // Handle block_shape.
64 const auto& block_shape_array =
65 model->GetArray(op->inputs[block_shape_index]);
66 if (!block_shape_array.has_shape()) return false;
67 const std::vector<int>& block_shape_dims = block_shape_array.shape().dims();
68 CHECK_EQ(block_shape_dims.size(), 1);
69 std::vector<int> block_shape_buffer =
70 block_shape_array.GetBuffer<ArrayDataType::kInt32>().data;
71 for (int i = 0; i < block_shape_dims[0]; ++i) {
72 op->block_shape.push_back(block_shape_buffer[i]);
73 }
74
75 return true;
76}
