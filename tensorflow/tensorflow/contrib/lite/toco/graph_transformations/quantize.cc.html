33bool SupportsQuantization(const Operator& op) {
34 auto type = op.type;
35 if (type == OperatorType::kTensorFlowUnsupported) {
36 auto* unsupported = static_cast<const TensorFlowUnsupportedOperator*>(&op);
37 return unsupported->quantized;
38 }
39 return type == OperatorType::kConv || type == OperatorType::kDepthwiseConv ||
40 type == OperatorType::kFullyConnected ||
41 type == OperatorType::kConcatenation ||
42 type == OperatorType::kL2Normalization || type == OperatorType::kAdd ||
43 type == OperatorType::kAveragePool || type == OperatorType::kMaxPool ||
44 type == OperatorType::kTensorFlowMinimum ||
45 type == OperatorType::kTensorFlowMaximum ||
46 type == OperatorType::kLogistic || type == OperatorType::kSoftmax ||
47 type == OperatorType::kTensorFlowSplit || type == OperatorType::kSub ||
48 type == OperatorType::kSqueeze || type == OperatorType::kPad ||
49 type == OperatorType::kTensorFlowReshape ||
50 type == OperatorType::kTanh || type == OperatorType::kMul ||
51 type == OperatorType::kSpaceToDepth ||
52 type == OperatorType::kStridedSlice ||
53 type == OperatorType::kDepthToSpace ||
54 type == OperatorType::kLstmCell || type == OperatorType::kGather ||
55 type == OperatorType::kTranspose || type == OperatorType::kMean;
56}

58template <ArrayDataType A>
59std::unique_ptr<GenericBuffer> QuantizeBuffer(
60 const GenericBuffer& buffer,
61 const QuantizationParams& quantization_params) {
62 const auto inverse_scale = 1. / quantization_params.scale;
63 CHECK(buffer.type == ArrayDataType::kFloat);
64 const auto& float_buffer =
65 static_cast<const Buffer<ArrayDataType::kFloat>&>(buffer);
66 auto* quantized_buffer = new Buffer<A>;
67 quantized_buffer->data.resize(float_buffer.data.size());
68 for (std::size_t i = 0; i < float_buffer.data.size(); i++) {
69 const float src_val = float_buffer.data[i];
70 double scaled_val; // Astonishingly, using 'float' degrades accuracy just
71 // enough to make a few tests fail!
72 if (quantization_params.scale == 0) {
73 CHECK_EQ(src_val, 0) << "The quantization scale for this array is 0, "
74 << "so all its values should be 0.";
75 scaled_val = quantization_params.zero_point;
76 } else {
77 scaled_val = quantization_params.zero_point + inverse_scale * src_val;
78 }
79 quantized_buffer->data[i] =
80 tflite::SafeCast<DataType<A>>(std::round(scaled_val));
81 }
82 return std::unique_ptr<GenericBuffer>(quantized_buffer);
83}

85template <ArrayDataType A>
86void QuantizeArray(GraphTransformation* transformation, Model* model,
87 const string& name,
88 const QuantizationParams& quantization_params) {
89 auto& array = model->GetArray(name);
90 CHECK(array.data_type == ArrayDataType::kFloat);
91 CHECK(!array.quantization_params);
92 array.GetOrCreateQuantizationParams() = quantization_params;
93 if (array.buffer) {
94 array.buffer = QuantizeBuffer<A>(*array.buffer, quantization_params);
95 }
96 array.data_type = A;
97 transformation->AddMessageF("Quantized array %s", name);
98}

100void QuantizeArray(GraphTransformation* transformation, Model* model,
101 const string& name, ArrayDataType quantized_data_type,
102 const QuantizationParams& quantization_params) {
103 ArrayDataType adjusted_data_type = quantized_data_type;
104 auto& array = model->GetArray(name);
105 if (array.final_data_type == ArrayDataType::kInt16) {
106 adjusted_data_type = array.final_data_type;
107 }
108
109 switch (adjusted_data_type) {
110 case ArrayDataType::kUint8:
111 return QuantizeArray<ArrayDataType::kUint8>(transformation, model, name,
112 quantization_params);
113 case ArrayDataType::kInt16:
114 return QuantizeArray<ArrayDataType::kInt16>(transformation, model, name,
115 quantization_params);
116 case ArrayDataType::kInt32:
117 return QuantizeArray<ArrayDataType::kInt32>(transformation, model, name,
118 quantization_params);
119 default:
120 LOG(FATAL) << "Unhandled case.";
121 }
122}
123
124const MinMax& GetOrComputeMinMax(Model* model, const string& array_name) {
125 auto& array = model->GetArray(array_name);
126 // Normally we should have a MinMax recorded on this Array,
127 // so we just use it.
128 if (array.minmax != nullptr) {
129 return *array.minmax;
130 }
131
132 // We don't have a MinMax. That's bad news: we need
133 // the graph to provide MinMax info for all arrays in order
134 // for inference to reproduce faithfully the same quantization
135 // error as the training process had.
136 //
137 // But we still want to support a fallback for constant arrays,
138 // just using the plain min and max computed from array elements.
139 // We should hopefully never rely on that in production, as that
140 // will not give very good accuracy as that typically won't be
141 // exactly what the training process used. But it will be useful
142 // to allow easily trying out quantization even if the graph
143 // lacks some minmax information.
144 if (array.buffer != nullptr) {
145 LOG(WARNING)
146 << "Constant array " << array_name
147 << " lacks MinMax information. To make up for that, we will now compute"
148 << " the MinMax from actual array elements. That will result in"
149 << " quantization parameters that probably do not match whichever "
150 "arithmetic"
151 << " was used during training, and thus will probably be a cause of "
152 "poor"
153 << " inference accuracy.";
154 CHECK(array.buffer->type == ArrayDataType::kFloat);
155 const auto& data = array.GetBuffer<ArrayDataType::kFloat>().data;
156 // We always want [min, max] to contain 0.
157 float min = 0.f;
158 float max = 0.f;
159 for (auto val : data) {
160 min = std::min(min, val);
161 max = std::max(max, val);
162 }
163 auto& minmax = array.GetOrCreateMinMax();
164 minmax.min = min;
165 minmax.max = max;
166 return minmax;
167 }
168
169 LOG(FATAL) << "Array " << array_name
170 << " does not have MinMax information, "
171 "and is not a constant array. Cannot "
172 "proceed with quantization.";
173}

175struct QuantizationPoints {
176 int64 min_value;
177 int64 max_value;
178 int64 central_value;
179};
180
181template <ArrayDataType A>
182QuantizationPoints GetQuantizationPoints() {
183 QuantizationPoints qp;
184 using Integer = DataType<A>;
185 qp.min_value = std::numeric_limits<Integer>::min();
186 qp.max_value = std::numeric_limits<Integer>::max();
187 // eg [-128,127]...
188 qp.central_value = (qp.min_value / 2 + // -128 -> -64.
189 (qp.max_value - 1) / 2 + // 127 -> 63.
190 1);
191 return qp;
192}
193
194QuantizationPoints GetQuantizationPoints(ArrayDataType data_type) {
195 switch (data_type) {
196 case ArrayDataType::kUint8:
197 return GetQuantizationPoints<ArrayDataType::kUint8>();
198 case ArrayDataType::kInt16:
199 return GetQuantizationPoints<ArrayDataType::kInt16>();
200 case ArrayDataType::kInt32:
201 return GetQuantizationPoints<ArrayDataType::kInt32>();
202 default:
203 LOG(FATAL) << "Unhandled case.";
204 }
205}

207ArrayDataType GetQuantizedDataType(const Array& array,
208 ArrayDataType default_type) {
209 switch (array.final_data_type) {
210 case ArrayDataType::kInt8:
211 case ArrayDataType::kUint8:
212 case ArrayDataType::kInt16:
213 case ArrayDataType::kUint16:
214 case ArrayDataType::kInt32:
215 case ArrayDataType::kUint32:
216 case ArrayDataType::kInt64:
217 case ArrayDataType::kUint64:
218 return array.final_data_type;
219 case ArrayDataType::kFloat:
220 case ArrayDataType::kNone:
221 return default_type;
222 default:
223 LOG(FATAL) << "Unhandled final quantization type "
224 << static_cast<int>(array.final_data_type);
225 }
226}

229 QuantizationParams* quantization_params) {
230 switch (data_type) {
231 case ArrayDataType::kInt8:
232 GetQuantizationParamsFromMinMax<ArrayDataType::kInt8>(
233 minmax, quantization_params);
234 break;
235 case ArrayDataType::kUint8:
236 GetQuantizationParamsFromMinMax<ArrayDataType::kUint8>(
237 minmax, quantization_params);
238 break;
239 case ArrayDataType::kInt16:
240 GetQuantizationParamsFromMinMax<ArrayDataType::kInt16>(
241 minmax, quantization_params);
242 break;
243 case ArrayDataType::kUint16:
244 GetQuantizationParamsFromMinMax<ArrayDataType::kUint16>(
245 minmax, quantization_params);
246 break;
247 case ArrayDataType::kInt32:
248 GetQuantizationParamsFromMinMax<ArrayDataType::kInt32>(
249 minmax, quantization_params);
250 break;
251 case ArrayDataType::kUint32:
252 GetQuantizationParamsFromMinMax<ArrayDataType::kUint32>(
253 minmax, quantization_params);
254 break;
255 case ArrayDataType::kInt64:
256 GetQuantizationParamsFromMinMax<ArrayDataType::kInt64>(
257 minmax, quantization_params);
258 break;
259 case ArrayDataType::kUint64:
260 GetQuantizationParamsFromMinMax<ArrayDataType::kUint64>(

264 case ArrayDataType::kNone:
265 default:
266 LOG(FATAL) << "Unhandled final quantization type "
267 << static_cast<int>(data_type);

271bool ChooseQuantizationForOperatorInput(
272 GraphTransformation* transformation, Model* model, const Operator& op,
273 std::size_t input_index, ArrayDataType* quantized_data_type,
274 QuantizationParams* quantization_params) {
275 const auto& input = op.inputs[input_index];
276 auto& array = model->GetArray(input);
277 if (array.data_type != ArrayDataType::kFloat) {
278 return false;
279 }
280
281 // Quantization of bias vectors
282 bool is_bias_vector = false;
283 int activations_input_index;
284 int weights_input_index;
285 if (op.type == OperatorType::kConv ||
286 op.type == OperatorType::kDepthwiseConv ||
287 op.type == OperatorType::kFullyConnected) {
288 if (input_index == 2) {
289 is_bias_vector = true;
290 activations_input_index = 0;
291 weights_input_index = 1;
292 }
293 }
294 if (op.type == OperatorType::kLstmCell) {
295 if (input_index == LstmCellOperator::BIASES_INPUT) {
296 is_bias_vector = true;
297 activations_input_index = LstmCellOperator::DATA_INPUT;
298 weights_input_index = LstmCellOperator::WEIGHTS_INPUT;
299 }
300 }
301 if (is_bias_vector) {
302 // Quantization of bias vector.
303 // We need both of the mandatory inputs (input activations and weights) to
304 // have been already quantized.
305 const auto& input_activations =
306 model->GetArray(op.inputs[activations_input_index]);
307 const auto& input_weights = model->GetArray(op.inputs[weights_input_index]);
308 if (!input_activations.quantization_params ||
309 !input_weights.quantization_params) {
310 return false;
311 }
312 const auto input_activations_scale =
313 input_activations.quantization_params->scale;
314 const auto input_weights_scale = input_weights.quantization_params->scale;
315 quantization_params->scale = input_activations_scale * input_weights_scale;
316 quantization_params->zero_point = 0;
317 *quantized_data_type = GetQuantizedDataType(array, ArrayDataType::kInt32);
318 transformation->AddMessageF(
319 "Input array %s is a bias vector. Choosing quantization params "
320 "accordingly.",
321 input);
322 return true;
323 }
324
325 const MinMax& minmax = GetOrComputeMinMax(model, input);
326
327 if (op.type == OperatorType::kLstmCell) {
328 if (input_index == LstmCellOperator::PREV_STATE_INPUT) {
329 *quantized_data_type = ArrayDataType::kInt16;
330 GetQuantizationParams(*quantized_data_type, minmax, quantization_params);
331 return true;
332 }
333 }
334
335 *quantized_data_type = GetQuantizedDataType(array, ArrayDataType::kUint8);
336 GetQuantizationParams(*quantized_data_type, minmax, quantization_params);
337 transformation->AddMessageF(
338 "For input array %s with min=%g"
339 ", max=%g"
340 ", chose to quantize as %s with zero_point=%d"
341 ", scale=%g",
342 input, minmax.min, minmax.max, ArrayDataTypeName(*quantized_data_type),
343 quantization_params->zero_point, quantization_params->scale);
344 return true;
345}

347bool IsExactlyRepresentable(double real_value, ArrayDataType data_type,
348 const QuantizationParams& quantization_params) {
349 const double scaled_value =
350 quantization_params.zero_point + real_value / quantization_params.scale;
351 const double fractional_scaled_value =
352 scaled_value - std::round(scaled_value);
353 if (std::abs(fractional_scaled_value) > 1e-12) {
354 return false;
355 }
356 const double rounded_scaled_value = std::round(scaled_value);
357 if (data_type == ArrayDataType::kUint8) {
358 if (rounded_scaled_value < 0 || rounded_scaled_value > 255) {
359 return false;
360 }
361 }
362 return true;
363}

365// Quantized data type is preset to the type of the input before this function.
366bool ChooseHardcodedQuantizationForOperatorOutput(
367 const Operator& op, const Array& array, ArrayDataType* quantized_data_type,
368 QuantizationParams* quantization_params) {
369 if (op.type == OperatorType::kL2Normalization) {
370 // L2Normalization has range: [-1, 1].
371 // 0 should be exactly representable, as values will typically be centered
372 // around 0, with many values near 0.
373 *quantized_data_type = GetQuantizedDataType(array, *quantized_data_type);
374 const QuantizationPoints qp = GetQuantizationPoints(*quantized_data_type);
375 quantization_params->zero_point = qp.central_value;
376 quantization_params->scale = 1. / (qp.central_value - qp.min_value);
377 CHECK(
378 IsExactlyRepresentable(0., *quantized_data_type, *quantization_params));
379 return true;
380 }
381 if ((op.type == OperatorType::kLogistic) ||
382 (op.type == OperatorType::kSoftmax)) {
383 // Logistic and Softmax have range: [0, 1].
384 //
385 // For Logistic, 0.5 should be exactly representable, as implementations
386 // will typically exploit the symmetry logistic(-x) = 1 - logistic(x), and
387 // the glueing of the two halves of the graph will only be seamless if we
388 // are accurately representing logistic(0) == 0.5.
389 *quantized_data_type = GetQuantizedDataType(array, *quantized_data_type);
390 const QuantizationPoints qp = GetQuantizationPoints(*quantized_data_type);
391 quantization_params->zero_point = 0;
392 quantization_params->scale = 1. / (qp.max_value + 1);
393 CHECK(IsExactlyRepresentable(0.5, *quantized_data_type,
394 *quantization_params));
395 return true;
396 }
397 if (op.type == OperatorType::kTanh) {
398 // Tanh has the range: [-1, 1].
399 *quantized_data_type = GetQuantizedDataType(array, *quantized_data_type);
400 const QuantizationPoints qp = GetQuantizationPoints(*quantized_data_type);
401 quantization_params->zero_point = qp.central_value;
402 quantization_params->scale = 1. / (qp.central_value - qp.min_value);
403 // 0 should be exactly representable, as values will typically be centered
404 // around 0, with many values near 0.
405 CHECK(
406 IsExactlyRepresentable(0., *quantized_data_type, *quantization_params));
407 return true;
408 }
409 return false;
410}

412bool ChooseQuantizationForOperatorOutput(
413 GraphTransformation* transformation, Model* model, const Operator& op,
414 std::size_t output_index, ArrayDataType* quantized_data_type,
415 QuantizationParams* quantization_params) {
416 const auto& output = op.outputs[output_index];
417 auto& array = model->GetArray(output);
418 if (array.data_type != ArrayDataType::kFloat) {
419 return false;
420 }
421 *quantized_data_type = model->GetArray(op.inputs[0]).data_type;
422 if (ChooseHardcodedQuantizationForOperatorOutput(
423 op, array, quantized_data_type, quantization_params)) {
424 transformation->AddMessageF(
425 "Output array %s is produced by a %s operator. Choosing fixed "
426 "quantization params accordingly.",
427 output, OperatorTypeName(op.type));
428 return true;
429 }
430 if ((op.type == OperatorType::kDepthToSpace) ||
431 (op.type == OperatorType::kSpaceToDepth) ||
432 (op.type == OperatorType::kTensorFlowReshape) ||
433 (op.type == OperatorType::kTensorFlowSplit) ||
434 (op.type == OperatorType::kConcatenation &&
435 model->flags.change_concat_input_ranges())) {
436 int data_input_index = 0;
437 if (op.type == OperatorType::kTensorFlowSplit) {
438 data_input_index = 1;
439 }
440 // Copying and rearrangement ops should preserve the quantization parameters
441 // of the input array.
442 const auto& input_array = model->GetArray(op.inputs[data_input_index]);
443 const auto& input_quantization_params = input_array.GetQuantizationParams();
444 *quantized_data_type =
445 GetQuantizedDataType(input_array, ArrayDataType::kUint8);
446 *quantized_data_type = GetQuantizedDataType(array, *quantized_data_type);
447 quantization_params->zero_point = input_quantization_params.zero_point;
448 quantization_params->scale = input_quantization_params.scale;
449
450 transformation->AddMessageF(
451 "Output array %s is produced by a %s operator. Copying quantization "
452 "params from input array.",
453 output, OperatorTypeName(op.type));
454 return true;
455 }
456 const MinMax& minmax = GetOrComputeMinMax(model, output);
457 if (op.type == OperatorType::kLstmCell) {
458 if (output_index == LstmCellOperator::STATE_OUTPUT ||
459 output_index == LstmCellOperator::ACTIV_TEMP) {
460 *quantized_data_type = ArrayDataType::kInt16;
461 GetQuantizationParams(*quantized_data_type, minmax, quantization_params);
462 return true;
463 }
464 }
465 *quantized_data_type = GetQuantizedDataType(array, ArrayDataType::kUint8);
466 GetQuantizationParams(*quantized_data_type, minmax, quantization_params);
467 transformation->AddMessageF(
468 "For output array %s with min=%g, max=%g"
469 ", chose to quantize as %s with zero_point=%d"
470 ", scale=%g",
471 output, minmax.min, minmax.max, ArrayDataTypeName(*quantized_data_type),
472 quantization_params->zero_point, quantization_params->scale);
473
474 return true;
475}

477// Fixes array minmax info to match the quantization parameters.
478// This is required for when quantization parameters change for an array during
479// quantization (such as ChooseQuantizationForOperatorOutput).
480void FixMinMaxPostQuantization(ArrayDataType quantized_data_type,
481 const QuantizationParams& quantization_params,
482 MinMax* minmax) {
483 double qmin, qmax;
484 switch (quantized_data_type) {
485 case ArrayDataType::kUint8:
486 qmin = 0;
487 qmax = 255;
488 break;
489 case ArrayDataType::kInt16:
490 qmin = -32768;
491 qmax = 32767;
492 break;
493 default:
494 // No update required.
495 return;
496 }
497
498 // Compute new minmax values.
499 double min =
500 (qmin - quantization_params.zero_point) * quantization_params.scale;
501 double max =
502 (qmax - quantization_params.zero_point) * quantization_params.scale;
503
504 // If we are close to the existing minmax values don't bother changing them.
505 // This prevents propagating small floating point precision errors.
506 constexpr double kMinMaxThreshold = 1e-5;
507 const double width = max - min;
508 if (std::abs(min - minmax->min) > kMinMaxThreshold * width ||
509 std::abs(max - minmax->max) > kMinMaxThreshold * width) {
510 minmax->min = min;
511 minmax->max = max;
512 }
513}

516bool Quantize::Run(Model* model, std::size_t op_index) {
517 // Our general "quantization" graph transformation consists in replacing
518 // QuantizedInputArrays[] ->
519 // DequantizeOperators[] ->
520 // FloatInputArrays[] ->
521 // Operator ->
522 // FloatOutputArray
523 // by
524 // QuantizedInputArrays[] ->
525 // Operator ->
526 // QuantizedOutputArray ->
527 // DequantizeOperator ->
528 // FloatOutputArray
529 //
530 // In other words, this is pushing Dequantize operators to the right of
531 // other operators.
532 //
533
534 auto& op = *model->operators[op_index];
535 if (op.type == OperatorType::kDequantize ||
536 op.type == OperatorType::kFakeQuant) {
537 return false;
538 }
539
540 // Our assumption here is that the input arrays are already quantized -
541 // that is typically the case in models operating on an input bitmap
542 // image, and MakeInitialDequantizeOp should have already resolved
543 // the handling of the input image as an initial Dequantize op.
544 //
545 // Thus we are building around the assumption that the graph always starts
546 // with a quantized input array, and only after some Dequantize op do we have
547 // float arrays. The problem of quantizing the graph thus becomes a problem of
548 // pushing Dequantize ops to the right of other ops.
549 //
550 // Let us just guard this assumption by the following assertion:
551 for (const auto& input : op.inputs) {
552 const auto& input_array = model->GetArray(input);
553 if (IsInputArray(*model, input) &&
554 input_array.data_type == ArrayDataType::kFloat) {
555 CHECK(input_array.quantization_params)
556 << "Input array " << input << " is missing quantization_params";
557 }
558 }
559 if (!SupportsQuantization(op)) {
560 LOG(FATAL) << "Unimplemented: this graph contains an operator of type "
561 << HelpfulOperatorTypeName(op)
562 << " for which the quantized form is not yet implemented. "
563 "Sorry, and patches welcome (that's a relatively fun patch "
564 "to write, mostly providing the actual quantized arithmetic "
565 "code for this op).";
566 }
567
568 for (const auto& input : op.inputs) {
569 const auto& array = model->GetArray(input);
570 if (array.data_type == ArrayDataType::kFloat) {
571 if (!array.minmax && !array.buffer) {
572 LOG(ERROR) << "Can't quantize input array " << input
573 << " because it lacks min/max info";
574 return false;
575 }
576 const auto* other_op = GetOpWithOutput(*model, input);
577 if (other_op && other_op->type != OperatorType::kDequantize) {
578 AddMessageF(
579 "Not quantizing %s for now, because its input array %s is not "
580 "produced by a Dequantize op, "
581 "which means that we should yield and let other ops "
582 "get quantized first",
583 LogName(op), input);
584 return false;
585 }
586 }
587 }
588
589 bool changed = false;
590
591 // Quantize inputs, remove any Dequantize op on the inputs side
592 for (std::size_t input_index = 0; input_index < op.inputs.size();
593 input_index++) {
594 ArrayDataType quantized_data_type;
595 QuantizationParams quantization_params;
596 if (ChooseQuantizationForOperatorInput(this, model, op, input_index,
597 &quantized_data_type,
598 &quantization_params)) {
599 const auto& input = op.inputs[input_index];
600 if (IsConstantParameterArray(*model, input)) {
601 QuantizeArray(this, model, input, quantized_data_type,
602 quantization_params);
603 changed = true;
604 } else {
605 auto dequantize_it = FindOpWithOutput(*model, input);
606 if (dequantize_it != model->operators.end()) {
607 auto* dequantize_op = dequantize_it->get();
608 CHECK(dequantize_op->type == OperatorType::kDequantize);
609 op.inputs[input_index] = dequantize_op->inputs[0];
610 // Check if the output of that Dequantize op was not used by any
611 // other operator. We will then erase that Dequantize op.
612 if (!CountOpsWithInput(*model, dequantize_op->outputs[0])) {
613 // If any of the model's output_arrays was pointing to the
614 // Dequantize op's output, let it point to the Dequantize op's
615 // input instead.
616 for (int i = 0; i < model->flags.output_arrays_size(); i++) {
617 if (model->flags.output_arrays(i) == dequantize_op->outputs[0]) {
618 model->flags.set_output_arrays(i, dequantize_op->inputs[0]);
619 }
620 }
621 model->EraseArray(dequantize_op->outputs[0]);
622 model->operators.erase(dequantize_it);
623 }
624 changed = true;
625 } else {
626 // This input array is not produced by a Dequantize op.
627 // We have encountered this situation in RNN graphs, whose cyclic
628 // nature defeats the basic assumption underlying the quantization
629 // algorithm implemented here. For now, when we have seen this
630 // happening, the array in question was a RNN state array itself,
631 // so let us just implement this case here, and guard that assumption
632 // with a CHECK. A more general fix would involve revisiting the
633 // design of this whole Quantization transformation.
634 bool is_rnn_state_array = false;
635 for (const auto& rnn_state : model->flags.rnn_states()) {
636 if (rnn_state.state_array() == input) {
637 is_rnn_state_array = true;
638 break;
639 }
640 }
641 CHECK(is_rnn_state_array);
642 QuantizeArray(this, model, input, quantized_data_type,
643 quantization_params);
644 changed = true;
645 }
646 }
647 }
648 }
649
650 // Quantize outputs, add Dequantize ops as needed on the outputs side
651 for (std::size_t output_index = 0; output_index < op.outputs.size();
652 output_index++) {
653 ArrayDataType quantized_data_type;
654 QuantizationParams quantization_params;
655 if (ChooseQuantizationForOperatorOutput(this, model, op, output_index,
656 &quantized_data_type,
657 &quantization_params)) {
658 changed = true;
659 const auto& output = op.outputs[output_index];
660 auto& output_array = model->GetArray(output);
661
662 // Fix up the min/max information on the output array to match the chosen
663 // quantization parameters.
664 auto& output_minmax = output_array.GetMinMax();
665 FixMinMaxPostQuantization(quantized_data_type, quantization_params,
666 &output_minmax);
667
668 QuantizeArray(this, model, output, quantized_data_type,
669 quantization_params);
670
671 const auto& dequantized_output =
672 AvailableArrayName(*model, output + "_dequantized");
673 auto& dequantized_output_array =
674 model->GetOrCreateArray(dequantized_output);
675 dequantized_output_array.data_type = ArrayDataType::kFloat;
676 auto& dequantized_output_minmax =
677 dequantized_output_array.GetOrCreateMinMax();
678 dequantized_output_minmax.min = output_minmax.min;
679 dequantized_output_minmax.max = output_minmax.max;
680 for (const auto& other_op : model->operators) {
681 for (auto& other_op_input : other_op->inputs) {
682 if (other_op_input == output) {
683 other_op_input = dequantized_output;
684 }
685 }
686 }
687 auto* dequantize_op = new DequantizeOperator;
688 dequantize_op->inputs = {output};
689 dequantize_op->outputs = {dequantized_output};
690 for (int i = 0; i < model->flags.output_arrays_size(); i++) {
691 if (model->flags.output_arrays(i) == output) {
692 model->flags.set_output_arrays(i, dequantized_output);
693 }
694 }
695 const auto op_it = FindOp(*model, &op);
696 model->operators.emplace(op_it + 1, dequantize_op);
697 }
698 }
699
700 return changed;
701}
