26int StartForAxis(StridedSliceOperator const& op, Shape const& input_shape,
27 int axis) {
28 int start;
29 if (op.begin_mask & 1 << axis) {
30 // If begin mask bit is set, use the first element
31 start = 0;
32 } else {
33 // Otherwise, use the specified element
34 start = op.start_indices[axis];
35 if (start < 0) {
36 // Handle negative indices
37 start += input_shape.dims(axis);
38 }
39 }
40 return start;
41}
42
43int StopForAxis(StridedSliceOperator const& op, Shape const& input_shape,
44 int axis) {
45 int stop;
46 if (op.end_mask & (1 << axis)) {
47 // If end mask bit set, use the last element
48 stop = input_shape.dims(axis);
49 } else {
50 // Otherwise, use the specified element
51 stop = op.stop_indices[axis];
52 if (stop < 0) {
53 // Handle negative indices
54 stop += input_shape.dims(axis);
55 }
56 }
57 return stop;
58}

60template <ArrayDataType Type>
61void StridedSlice(StridedSliceOperator const& op, Array const& input_array,
62 Array* output_array) {
63 // The TensorFlow documentation for StridedSlice is a bit ambiguous in places
64 // (https://www.tensorflow.org/api_docs/cc/class/tensorflow/ops/strided-slice).
65 // Use the source code at /third_party/tensorflow/core/util/strided_op.cc as
66 // "master documentation".
67
68 CHECK(input_array.data_type == Type);
69 CHECK(output_array->data_type == Type);
70 CHECK_EQ(op.ellipsis_mask, 0);
71 CHECK_EQ(op.new_axis_mask, 0);
72
73 int num_input_axes = op.start_indices.size();
74 CHECK_EQ(num_input_axes, op.stop_indices.size());
75 CHECK_EQ(num_input_axes, op.strides.size());
76 for (int i = 0; i < op.strides.size(); i++) {
77 CHECK_GE(op.strides[i], 0) << "Negative strides usupported";
78 }
79
80 // Create a buffer for the output array
81 std::vector<DataType<Type>>& output_data =
82 output_array->GetMutableBuffer<Type>().data;
83 output_data.resize(RequiredBufferSizeForShape(output_array->shape()));
84
85 // Initialize source coordinate
86 Shape const& input_shape = input_array.shape();
87 Buffer<Type> const& input_buffer = input_array.GetBuffer<Type>();
88 std::vector<int> src_coord(op.start_indices.size());
89 for (int axis = 0; axis < num_input_axes; axis++) {
90 src_coord[axis] = StartForAxis(op, input_shape, axis);
91 }
92
93 // In order to handle any number (N) of dimensions, we copy elements one by
94 // one and treat the source coordinate as an N digit number (src_coord here).
95 // Each "digit" is incremented individually (by the stride). When it overflows
96 // (becomes greater than the stop), that digit is reset and a carry flag is
97 // used to increment the next digit.
98 int dst_offset = 0;
99 do {
100 // Copy element.
101 output_data[dst_offset] = input_buffer.data[Offset(input_shape, src_coord)];
102
103 // Compute next source input coordinates.
104 bool carry = true;
105 for (int axis = 0; axis < num_input_axes; axis++) {
106 // Increment this axis if we carried from the previous one
107 if (carry) {
108 src_coord[axis] += op.strides[axis];
109 }
110
111 // Check if we've overflowed.
112 if (src_coord[axis] >= StopForAxis(op, input_shape, axis)) {
113 // Reset axis and set carry
114 src_coord[axis] = StartForAxis(op, input_shape, axis);
115 carry = true;
116 } else {
117 carry = false;
118 }
119 }
120 // increment destination buffer offset
121 dst_offset++;
122 } while (dst_offset < output_data.size());
123}

127bool ResolveConstantStridedSlice::Run(Model* model, std::size_t op_index) {
128 const auto it = model->operators.begin() + op_index;
129 const auto* base_op = it->get();
130 if (base_op->type != OperatorType::kStridedSlice) {
131 return false;
132 }
133
134 const StridedSliceOperator* op =
135 static_cast<const StridedSliceOperator*>(base_op);
136
137 CHECK_EQ(op->outputs.size(), 1);
138 auto& output_array = model->GetArray(op->outputs[0]);
139 if (output_array.data_type == ArrayDataType::kNone) {
140 // Yield until the output type has been set by PropagateArrayDataTypes
141 return false;
142 }
143
144 if (!output_array.has_shape()) {
145 // Yield until the output shape has been set by PropagateFixedShapes
146 return false;
147 }
148
149 if (op->start_indices.empty() || op->stop_indices.empty() ||
150 op->strides.empty()) {
151 // Attributes have not resolved yet.
152 return false;
153 }
154
155 const auto& input_array = model->GetArray(op->inputs[0]);
156 if (!input_array.has_shape()) {
157 // Yield until the value shape has been resolved.
158 return false;
159 }
160 if (!IsConstantParameterArray(*model, op->inputs[0])) {
161 // Yield until the value is constant.
162 return false;
163 }
164
165 CHECK(!output_array.buffer);
166 switch (output_array.data_type) {
167 case ArrayDataType::kFloat:
168 StridedSlice<ArrayDataType::kFloat>(*op, input_array, &output_array);
169 break;
170 case ArrayDataType::kUint8:
171 StridedSlice<ArrayDataType::kUint8>(*op, input_array, &output_array);
172 break;
173 case ArrayDataType::kInt32:
174 StridedSlice<ArrayDataType::kInt32>(*op, input_array, &output_array);
175 break;
176 case ArrayDataType::kInt64:
177 StridedSlice<ArrayDataType::kInt64>(*op, input_array, &output_array);
178 break;
179 default:
180 LOG(FATAL)
181 << "Unsupported data type input to StridedSlice op with output \""
182 << op->outputs[0] << "\"";
183 break;
184 }
185
186 // Erase input array if no longer used
187 if (IsDiscardableArray(*model, op->inputs[0]) &&
188 CountOpsWithInput(*model, op->inputs[0]) == 1) {
189 model->EraseArray(op->inputs[0]);
190 }
191
192 // Erase the operator
193 model->operators.erase(it);
194
195 return true;
196}
