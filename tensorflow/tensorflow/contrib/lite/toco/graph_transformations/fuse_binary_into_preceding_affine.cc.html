30void FuseAddOrSubParamsIntoPrecedingAffine(Model* model, Operator* preceding_op,
31 const Operator* add_or_sub_op,
32 int index_of_constant_input) {
33 CHECK(add_or_sub_op->type == OperatorType::kAdd ||
34 add_or_sub_op->type == OperatorType::kSub);
35 CHECK(index_of_constant_input == 0 || index_of_constant_input == 1);
36 if (preceding_op->inputs.size() < 3) {
37 LOG(FATAL) << "Missing bias parameter";
38 }
39 auto& bias = model->GetArray(preceding_op->inputs[2]);
40 bias.minmax = nullptr;
41 const auto& operand =
42 model->GetArray(add_or_sub_op->inputs[index_of_constant_input]);
43
44 const Shape& bias_shape = bias.shape();
45 const Shape& operand_shape = operand.shape();
46 auto& bias_buffer = bias.GetMutableBuffer<ArrayDataType::kFloat>();
47 float* const bias_data = bias_buffer.data.data();
48 const auto& operand_buffer = operand.GetBuffer<ArrayDataType::kFloat>();
49 const float* const operand_data = operand_buffer.data.data();
50
51 // TODO(b/62904716): Bias array should become 1-D when padding removed.
52 const int depth = bias_shape.dims(bias_shape.dimensions_count() - 1);
53 int operand_channel_increment = 0;
54 if (operand_shape.dimensions_count() >= 1 &&
55 operand_shape.dims(operand_shape.dimensions_count() - 1) ==
56 bias_shape.dims(bias_shape.dimensions_count() - 1)) {
57 operand_channel_increment = 1;
58 } else if (operand_shape.dimensions_count() == 0 ||
59 operand_shape.dims(operand_shape.dimensions_count() - 1) == 1) {
60 operand_channel_increment = 0;
61 } else {
62 LOG(FATAL) << "Operand shape mismatch.";
63 }
64
65 enum class OpType { BiasPlusOperand, BiasMinusOperand, OperandMinusBias };
66
67 const OpType optype = (add_or_sub_op->type == OperatorType::kAdd)
68 ? OpType::BiasPlusOperand
69 : (index_of_constant_input == 1)
70 ? OpType::BiasMinusOperand
71 : OpType::OperandMinusBias;
72
73 int operand_channel = 0;
74 for (int i = 0; i < depth; i++) {
75 float& bias_val = bias_data[i];
76 const float operand_val = operand_data[operand_channel];
77 if (optype == OpType::BiasPlusOperand) {
78 bias_val += operand_val;
79 } else if (optype == OpType::BiasMinusOperand) {
80 bias_val -= operand_val;
81 } else if (optype == OpType::OperandMinusBias) {
82 bias_val = operand_val - bias_val;
83 } else {
84 LOG(FATAL) << "Should not get here.";
85 }
86 operand_channel += operand_channel_increment;
87 }
88}

90void FuseMulOrDivParamsIntoPrecedingAffine(Model* model, Operator* preceding_op,
91 const Operator* mul_or_div_op,
92 int index_of_constant_input) {
93 CHECK(mul_or_div_op->type == OperatorType::kMul ||
94 mul_or_div_op->type == OperatorType::kDiv);
95 CHECK(index_of_constant_input == 0 || index_of_constant_input == 1);
96 // If the op is a division, the constant input should be the right hand side.
97 // This should have been checked before this point.
98 CHECK(mul_or_div_op->type != OperatorType::kDiv ||
99 index_of_constant_input == 1);
100 if (preceding_op->inputs.size() < 3) {
101 LOG(FATAL) << "Missing bias parameter";
102 }
103 const auto& weights_name = preceding_op->inputs[1];
104 const auto& bias_name = preceding_op->inputs[2];
105 auto& weights = model->GetArray(weights_name);
106 DropMinMax(model, weights_name);
107 auto& bias = model->GetArray(bias_name);
108 DropMinMax(model, bias_name);
109 const auto& operand =
110 model->GetArray(mul_or_div_op->inputs[index_of_constant_input]);
111
112 const Shape& weights_shape = weights.shape();
113 const Shape& bias_shape = bias.shape();
114 const Shape& operand_shape = operand.shape();
115 auto& weights_buffer = weights.GetMutableBuffer<ArrayDataType::kFloat>();
116 float* const weights_data = weights_buffer.data.data();
117 auto& bias_buffer = bias.GetMutableBuffer<ArrayDataType::kFloat>();
118 float* const bias_data = bias_buffer.data.data();
119 const auto& operand_buffer = operand.GetBuffer<ArrayDataType::kFloat>();
120 const float* const operand_data = operand_buffer.data.data();
121
122 // We support broadcasting the operand along the depth dimension,
123 // when the operand's depth is 1.
124 int operand_channel_increment = 0;
125 if (operand_shape.dimensions_count() >= 1 &&
126 operand_shape.dims(operand_shape.dimensions_count() - 1) ==
127 bias_shape.dims(bias_shape.dimensions_count() - 1)) {
128 operand_channel_increment = 1;
129 } else if (operand_shape.dimensions_count() == 0 ||
130 operand_shape.dims(operand_shape.dimensions_count() - 1) == 1) {
131 operand_channel_increment = 0;
132 } else {
133 LOG(FATAL) << "Operand shape mismatch.";
134 }
135
136 int output_depth;
137
138 if (preceding_op->type == OperatorType::kConv ||
139 preceding_op->type == OperatorType::kFullyConnected) {
140 output_depth = weights_shape.dims(0);
141 } else if (preceding_op->type == OperatorType::kDepthwiseConv) {
142 output_depth = weights_shape.dims(weights_shape.dimensions_count() - 1);
143 } else {
144 LOG(FATAL) << "Should not get here";
145 }
146
147 const int weights_size = RequiredBufferSizeForShape(weights_shape);
148 const int weights_per_depth = weights_size / output_depth;
149 CHECK_EQ(weights_size, weights_per_depth * output_depth);
150
151 int operand_channel = 0;
152 for (int c = 0; c < output_depth; c++) {
153 if (mul_or_div_op->type == OperatorType::kMul) {
154 bias_data[c] *= operand_data[operand_channel];
155 } else if (mul_or_div_op->type == OperatorType::kDiv) {
156 bias_data[c] /= operand_data[operand_channel];
157 } else {
158 LOG(FATAL) << "Should not get here";
159 }
160 if (preceding_op->type == OperatorType::kConv ||
161 preceding_op->type == OperatorType::kFullyConnected) {
162 for (int i = 0; i < weights_per_depth; i++) {
163 if (mul_or_div_op->type == OperatorType::kMul) {
164 weights_data[c * weights_per_depth + i] *=
165 operand_data[operand_channel];
166 } else if (mul_or_div_op->type == OperatorType::kDiv) {
167 weights_data[c * weights_per_depth + i] /=
168 operand_data[operand_channel];
169 } else {
170 LOG(FATAL) << "Should not get here";
171 }
172 }
173 } else if (preceding_op->type == OperatorType::kDepthwiseConv) {
174 for (int k = 0; k < weights_per_depth; k++) {
175 if (mul_or_div_op->type == OperatorType::kMul) {
176 weights_data[k * output_depth + c] *= operand_data[operand_channel];
177 } else if (mul_or_div_op->type == OperatorType::kDiv) {
178 weights_data[k * output_depth + c] /= operand_data[operand_channel];
179 } else {
180 LOG(FATAL) << "Should not get here";
181 }
182 }
183 } else {
184 LOG(FATAL) << "Should not get here";
185 }
186 operand_channel += operand_channel_increment;
187 }
188}

191bool FuseBinaryIntoPrecedingAffine::Run(Model* model, std::size_t op_index) {
192 const auto binary_it = model->operators.begin() + op_index;
193 const auto* binary_op = binary_it->get();
194 if (binary_op->type != OperatorType::kAdd &&
195 binary_op->type != OperatorType::kMul &&
196 binary_op->type != OperatorType::kSub &&
197 binary_op->type != OperatorType::kDiv) {
198 return false;
199 }
200
201 CHECK_EQ(binary_op->inputs.size(), 2);
202
203 // We only can fuse an binary when the two operands break down as follows:
204 // 1. One operand is the (variable) output of a typical affine (linear plus
205 // bias)
206 // op of a finite list of possible types: at the moment Conv,
207 // DepthwiseConv and
208 // FullyConnected are supported.
209 // 2. The other operand is a constant param array.
210 const bool is_input_constant[2] = {
211 IsConstantParameterArray(*model, binary_op->inputs[0]),
212 IsConstantParameterArray(*model, binary_op->inputs[1]),
213 };
214 if (!is_input_constant[0] && !is_input_constant[1]) {
215 // Neither input is constant, so nothing we can fuse into a constant.
216 return false;
217 }
218 if (is_input_constant[0] && is_input_constant[1]) {
219 // Both inputs are constants. That's a job for constants
220 // propagation, not for us to handle here.
221 return false;
222 }
223 const int index_of_constant_input = is_input_constant[0] ? 0 : 1;
224 const int index_of_variable_input = is_input_constant[0] ? 1 : 0;
225 CHECK(is_input_constant[index_of_constant_input]);
226 CHECK(!is_input_constant[index_of_variable_input]);
227
228 // For division, we can only fuse if the denominator is constant.
229 if (binary_op->type == OperatorType::kDiv) {
230 if (index_of_constant_input != 1) {
231 AddMessageF("Not fusing %s because the denominator is not constant",
232 LogName(*binary_op));
233 return false;
234 }
235 }
236
237 Operator* preceding_op =
238 GetOpWithOutput(*model, binary_op->inputs[index_of_variable_input]);
239 if (!preceding_op) {
240 AddMessageF("Not fusing %s because it is not the output of another op",
241 LogName(*binary_op));
242 return false;
243 }
244
245 for (const string& output_array : model->flags.output_arrays()) {
246 if (preceding_op->outputs[0] == output_array) {
247 return false;
248 }
249 }
250
251 if (preceding_op->type != OperatorType::kConv &&
252 preceding_op->type != OperatorType::kFullyConnected &&
253 preceding_op->type != OperatorType::kDepthwiseConv) {
254 AddMessageF(
255 "Not fusing %s because the preceding %s is not of one of the supported "
256 "types",
257 LogName(*binary_op), LogName(*preceding_op));
258 return false;
259 }
260
261 if (preceding_op->fused_activation_function !=
262 FusedActivationFunctionType::kNone) {
263 AddMessageF(
264 "Not fusing %s because the preceding %s has a fused activation "
265 "function",
266 LogName(*binary_op), LogName(*preceding_op));
267 return false;
268 }
269
270 if (preceding_op->inputs.size() < 3) {
271 AddMessageF(
272 "Not fusing %s because the preceding %s does not have a bias vector",
273 LogName(*binary_op), LogName(*preceding_op));
274 return false;
275 }
276
277 const auto& weights = model->GetArray(preceding_op->inputs[1]);
278 const auto& bias = model->GetArray(preceding_op->inputs[2]);
279 if (binary_op->type == OperatorType::kAdd ||
280 binary_op->type == OperatorType::kSub) {
281 if (!bias.buffer) {
282 AddMessageF(
283 "Not fusing %s because the preceding %s has a non-constant bias "
284 "array",
285 LogName(*binary_op), LogName(*preceding_op));
286 return false;
287 }
288 } else {
289 if (!weights.buffer || !bias.buffer) {
290 AddMessageF(
291 "Not fusing %s because the preceding %s has non-constant weights or "
292 "bias arrays",
293 LogName(*binary_op), LogName(*preceding_op));
294 return false;
295 }
296 }
297
298 int count_ops_consuming_output =
299 CountOpsWithInput(*model, preceding_op->outputs[0]);
300 DCHECK_GE(count_ops_consuming_output, 1);
301 if (count_ops_consuming_output > 1) {
302 AddMessageF(
303 "Not fusing %s because the output of the preceding %s is consumed by "
304 "another op",
305 LogName(*binary_op), LogName(*preceding_op));
306 return false;
307 }
308
309 AddMessageF("Fusing %s into the preceding %s", LogName(*binary_op),
310 LogName(*preceding_op));
311
312 if (binary_op->type == OperatorType::kAdd ||
313 binary_op->type == OperatorType::kSub) {
314 FuseAddOrSubParamsIntoPrecedingAffine(model, preceding_op, binary_op,
315 index_of_constant_input);
316 } else if (binary_op->type == OperatorType::kMul ||
317 binary_op->type == OperatorType::kDiv) {
318 FuseMulOrDivParamsIntoPrecedingAffine(model, preceding_op, binary_op,
319 index_of_constant_input);
320 } else {
321 LOG(FATAL) << "should not get here";
322 }
323
324 model->EraseArray(preceding_op->outputs[0]);
325 preceding_op->outputs[0] = binary_op->outputs[0];
326 preceding_op->fused_activation_function =
327 binary_op->fused_activation_function;
328 const auto& old_constant_param_name =
329 binary_op->inputs[index_of_constant_input];
330 CHECK(IsConstantParameterArray(*model, old_constant_param_name));
331 if (CountOpsWithInput(*model, old_constant_param_name) == 1) {
332 model->EraseArray(old_constant_param_name);
333 }
334 model->operators.erase(binary_it);
335 return true;
336}
