28bool UnfuseActivationFunctions::Run(Model* model, std::size_t op_index) {
29 const auto it = model->operators.begin() + op_index;
30 auto* op = it->get();
31
32 // If a conv operation has an im2col array, yield: it should be dropped first.
33 if ((op->type == OperatorType::kConv) && (op->outputs.size() == 2)) {
34 return false;
35 }
36
37 Operator* ac_op = nullptr;
38 switch (op->fused_activation_function) {
39 case FusedActivationFunctionType::kRelu:
40 ac_op = new ReluOperator;
41 break;
42 case FusedActivationFunctionType::kRelu6:
43 ac_op = new Relu6Operator;
44 break;
45 case FusedActivationFunctionType::kRelu1:
46 ac_op = new Relu1Operator;
47 break;
48 default:
49 return false;
50 }
51
52 // At this point we know that the op has a fused activation function. At the
53 // moment that only happens with ops having a single output, may be
54 // relaxed in the future.
55 CHECK_EQ(op->outputs.size(), 1);
56
57 // Emplace unfused activation function, drop the fused one.
58 model->operators.emplace(it + 1, ac_op);
59 op->fused_activation_function = FusedActivationFunctionType::kNone;
60
61 // Wire up arrays, constructing a new intermediate array to connect the
62 // op to its new unfused activation function.
63 ac_op->outputs = op->outputs;
64 const string& tmp_array_name =
65 AvailableArrayName(*model, op->outputs[0] + "_unfused");
66 CHECK(!model->HasArray(tmp_array_name));
67 model->GetOrCreateArray(tmp_array_name);
68 ac_op->inputs = {tmp_array_name};
69 op->outputs = {tmp_array_name};
70 return true;
71}
