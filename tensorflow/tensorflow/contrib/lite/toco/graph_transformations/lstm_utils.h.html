27// For consistency with the parameters defined in extended LstmCell's kernel
28// (tensorflow/contrib/lite/kernels/lstm.cc),
29// use lowercase for these constants.
30
31enum ExtendedLstmCellInputs {
32 kInputTensor = 0,
33 kInputToInputWeightsTensor = 1, // Optional
34 kInputToForgetWeightsTensor = 2,
35 kInputToCellWeightsTensor = 3,
36 kInputToOutputWeightsTensor = 4,
37 kRecurrentToInputWeightsTensor = 5, // Optional
38 kRecurrentToForgetWeightsTensor = 6,
39 kRecurrentToCellWeightsTensor = 7,
40 kRecurrentToOutputWeightsTensor = 8,
41 kCellToInputWeightsTensor = 9, // Optional
42 kCellToForgetWeightsTensor = 10, // Optional
43 kCellToOutputWeightsTensor = 11, // Optional
44 kInputGateBiasTensor = 12, // Optional
45 kForgetGateBiasTensor = 13,
46 kCellGateBiasTensor = 14,
47 kOutputGateBiasTensor = 15,
48 kProjectionWeightsTensor = 16, // Optional
49 kProjectionBiasTensor = 17, // Optional
50 kExtendedLstmInputCount = 18
51};
52
53enum ExtendedLstmCellOutputs {
54 kScratchBufferTensor = 0,
55 kOutputStateTensor = 1,
56 kCellStateTensor = 2,
57 kOutputTensor = 3
58};

60// Create optional array used for optional tensor in ExtendedLstmCell inputs.
61void CreateOptionalArray(Model* model, string* input_array_buffer,
62 const string& array_name);
63
64// Create float array and get its buffer.
65Buffer<ArrayDataType::kFloat>* CreateFloatArrayBuffer(Model* model,
66 string* array_name,
67 const Shape& shape);
68
69// Copy data from one array to the other one (supports 1D and 2D array),
70// for 1D array, the 2nd dim's size is 1.
71// Arguments:
72// src_buffer: the source buffer
73// src_stride: the stride of source buffer, i.e., 2nd dim's size
74// src_start_idx1: the 1st dim index of start point in src matrix
75// src_start_idx2: the 2nd dim index of start point in src matrix
76// dst_buffer: the destination buffer
77// dst_stride: the stride of destination buffer, i.e., 2nd dim's size
78// dst_start_idx1: the 1st dim index of start point in dst matrix
79// dst_start_idx2: the 2nd dim index of start point in dst matrix
80// dim1_copy_size: 1st dim size of copy data
81// dim2_copy_size: 2nd dim size of copy data
82void CopyArrayData(const Buffer<ArrayDataType::kFloat>& src_buffer,
83 int src_stride, int src_start_idx1, int src_start_idx2,
84 Buffer<ArrayDataType::kFloat>* dst_buffer, int dst_stride,
85 int dst_start_idx1, int dst_start_idx2, int dim1_copy_size,
86 int dim2_copy_size);
87
88// Copy a subset of array data and create a smaller array,
89// mostly used for spliting weights and bias for Lstm cell.
90void CopySubArrayToArray(Model* model, string* array_name,
91 const string& tensor_name, int dim1_size,
92 int dim2_size, const Array& original_array,
93 int start_idx1, int start_idx2);
94
95// Copy array data to a large array's submatrix,
96// mostly used for merging weights and bias for Lstm cell.
97void CopyArrayToSubArray(Buffer<ArrayDataType::kFloat>& tensor_buffer,
98 int tensor_stride, const Array& sub_array,
99 int start_idx1, int start_idx2);
100
101// Get mating rnn array inputs using rnn_states flag.
102bool GetMatchingRnnArray(Model* model, const string& back_edge_source_array,
103 string* rnn_array);
