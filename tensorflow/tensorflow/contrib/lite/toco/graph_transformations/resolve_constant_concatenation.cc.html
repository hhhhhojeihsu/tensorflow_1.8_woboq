30// Copies data from multiple source arrays to a destination array based on a
31// concatenation dimension. From each array in input_arrays, it copies chunk
32// sizes provided in array_copy_size vector (per array). It uses the buffer
33// in concatenated_array as destination buffer.
34template <ArrayDataType A, typename T>
35void CopyTensorSegments(const std::vector<Array*>& input_arrays,
36 const std::vector<int>& array_copy_size,
37 const int num_elements_concatenated_array,
38 Array* concatenated_array) {
39 for (Array* input_array : input_arrays) {
40 if (!input_array->buffer) {
41 return;
42 }
43 }
44
45 auto& concatenated_array_buffer =
46 concatenated_array->GetMutableBuffer<A>().data;
47 concatenated_array_buffer.resize(num_elements_concatenated_array);
48
49 // It does not matter which array to use to find the value for the total
50 // number of copy steps.
51 CHECK(!input_arrays.empty());
52 CHECK_NE(array_copy_size[0], 0);
53 const int total_copy_steps =
54 input_arrays[0]->GetBuffer<A>().data.size() / array_copy_size[0];
55
56 // Initialize the source pointers to point to beginning of the array buffers.
57 std::vector<const T*> src_ptr;
58 src_ptr.reserve(input_arrays.size());
59 for (Array* input_array : input_arrays) {
60 src_ptr.push_back(input_array->GetBuffer<A>().data.data());
61 }
62
63 // Copy the data from input_arrays to concatenated_array_buffer.
64 T* dest_ptr = concatenated_array_buffer.data();
65 for (int s = 0; s < total_copy_steps; s++) {
66 for (int i = 0; i < input_arrays.size(); i++) {
67 std::copy(src_ptr[i], src_ptr[i] + array_copy_size[i], dest_ptr);
68 src_ptr[i] += array_copy_size[i];
69 dest_ptr += array_copy_size[i];
70 }
71 }
72}

74// Receives a series of input arrays of type Array and an integer showing the
75// axis on which those arrays will be concatenated. It returns the concatenated
76// array.
77template <ArrayDataType A>
78void ConcatenateTensorBuffers(const std::vector<Array*>& input_arrays,
79 int concatenation_axis,
80 Array* concatenated_array) {
81 int num_elements_concatenated_array = 1;
82 for (int i = 0; i < concatenated_array->shape().dimensions_count(); i++) {
83 num_elements_concatenated_array *= concatenated_array->shape().dims()[i];
84 }
85 // Prepare the data needed for segmented copy from multiple source arrays to
86 // a destination array based on a oncatenation dimension.
87 std::vector<int> array_copy_size(input_arrays.size());
88 int count = 0;
89 for (Array* input_array : input_arrays) {
90 const Shape array_shape = input_array->shape();
91 array_copy_size[count] = 1;
92 for (int i = concatenation_axis; i < array_shape.dimensions_count(); i++) {
93 array_copy_size[count] *= array_shape.dims()[i];
94 }
95 count++;
96 }
97
98 // Do the actual data copy.
99 CopyTensorSegments<A, DataType<A>>(input_arrays, array_copy_size,
100 num_elements_concatenated_array,
101 concatenated_array);
102}

104// Sets the minimum and maximum values for the concatenated array. If it's
105// already set (e.g. because of previous pass in TOCO), it doesn't change it and
106// returns. Otherwise it uses the input arrays min and max values to compute the
107// concatenated array min and max.
108void SetMinMaxForConcatenedArray(const std::vector<Array*>& input_arrays,
109 Array* concatenated_array) {
110 CHECK(concatenated_array->data_type == ArrayDataType::kFloat);
111 // If the minmax is already set, use it
112 if (concatenated_array->minmax) return;
113
114 double concat_min = std::numeric_limits<double>::infinity();
115 double concat_max = -std::numeric_limits<double>::infinity();
116
117 for (Array* input_array : input_arrays) {
118 // If any of the input arrays minmax is not set, return.
119 // TODO(ghodrat): shall we add the logic to compute the minmax?
120 if (!input_array->minmax) return;
121 const MinMax& input_minmax = input_array->GetMinMax();
122 concat_min = std::min(concat_min, input_minmax.min);
123 concat_max = std::max(concat_max, input_minmax.max);
124 }
125 MinMax& minmax = concatenated_array->GetOrCreateMinMax();
126 minmax.min = concat_min;
127 minmax.max = concat_max;
128}

132// Resolves the concatenation operator if all its inputs are constant arrays.
133bool ResolveConstantConcatenation::Run(Model* model, std::size_t op_index) {
134 const auto concat_it = model->operators.begin() + op_index;
135 const auto* concat_base_op = concat_it->get();
136 if (concat_base_op->type != OperatorType::kConcatenation) {
137 return false;
138 }
139 const auto* concat_op =
140 static_cast<const ConcatenationOperator*>(concat_base_op);
141
142 for (const string& input_name : concat_op->inputs) {
143 // We only expect constant unquantized arrays as input, otherwise we return.
144 // We also make sure the shapes of the input arrays are known and they are
145 // all discardable.
146 const Operator* input_op = GetOpWithOutput(*model, input_name);
147 if (input_op) return false;
148 if (!IsConstantParameterArray(*model, input_name)) return false;
149 if (!model->GetArray(input_name).has_shape()) return false;
150 if (model->GetArray(input_name).quantization_params) return false;
151 if (!IsDiscardableArray(*model, input_name)) return false;
152 }
153
154 const int concatenation_axis = concat_op->axis;
155
156 CHECK_EQ(concat_op->outputs.size(), 1);
157 string concatenated_array_name = concat_op->outputs[0];
158 Array& concatenated_array = model->GetOrCreateArray(concatenated_array_name);
159 std::vector<Array*> input_arrays;
160 for (const string& input_name : concat_op->inputs) {
161 input_arrays.push_back(&model->GetArray(input_name));
162 }
163
164 switch (concatenated_array.data_type) {
165 case ArrayDataType::kFloat:
166 ConcatenateTensorBuffers<ArrayDataType::kFloat>(
167 input_arrays, concatenation_axis, &concatenated_array);
168 SetMinMaxForConcatenedArray(input_arrays, &concatenated_array);
169 break;
170 case ArrayDataType::kUint8:
171 ConcatenateTensorBuffers<ArrayDataType::kUint8>(
172 input_arrays, concatenation_axis, &concatenated_array);
173 break;
174 case ArrayDataType::kInt32:
175 ConcatenateTensorBuffers<ArrayDataType::kInt32>(
176 input_arrays, concatenation_axis, &concatenated_array);
177 break;
178 case ArrayDataType::kInt64:
179 ConcatenateTensorBuffers<ArrayDataType::kInt64>(
180 input_arrays, concatenation_axis, &concatenated_array);
181 break;
182 case ArrayDataType::kString:
183 ConcatenateTensorBuffers<ArrayDataType::kString>(
184 input_arrays, concatenation_axis, &concatenated_array);
185 break;
186 default:
187 LOG(FATAL) << "ArrayDataType not supported";
188 }
189
190 // Remove all the resolved arrays.
191 for (const string& input_name : concat_op->inputs) {
192 // Check to prevent removal of shared tensors
193 if (CountOpsWithInput(*model, input_name) == 1) {
194 model->EraseArray(input_name);
195 }
196 }
197
198 // Remove concatenate operator
199 model->operators.erase(concat_it);
200 return true;
201}
