29// Loads a SavedModel from the directory specified in parsed_toco_flags.
30// Returns a SavedModelBundle with the requested MetaGraphDef.
31const tensorflow::SavedModelBundle* LoadSavedModel(
32 const ParsedTocoFlags& parsed_toco_flags) {
33 const string model_path = parsed_toco_flags.savedmodel_directory.value();
34 QCHECK(tensorflow::MaybeSavedModelDirectory(model_path))
35 << "Model is not saved in the supported SavedModel format.\n";
36
37 // Gets the tags identifying the MetaGraphDef from the command line arguments.
38 string tags_str;
39 if (parsed_toco_flags.savedmodel_tagset.specified()) {
40 tags_str = parsed_toco_flags.savedmodel_tagset.value();
41 } else {
42 tags_str = parsed_toco_flags.savedmodel_tagset.default_value();
43 }
44 auto tags = absl::StrSplit(tags_str, ',');
45
46 // Loads MetaGraphDef.
47 auto* bundle = new tensorflow::SavedModelBundle;
48 TF_CHECK_OK(tensorflow::LoadSavedModel(tensorflow::SessionOptions(),
49 tensorflow::RunOptions(), model_path,
50 tags, bundle))
51 << "Failed to load exported model from " << model_path
52 << ". Ensure the model contains the required tags '" << tags_str
53 << "'.\n";
54 return bundle;
55}

57// Returns the array name without the postfix.
58//
59// e.g. reduces "input:0" to "input".
60string GetArrayName(const string& name) {
61 const std::vector<string>& names = absl::StrSplit(name, ':');
62 return names[0];
63}
64
65// Returns the list of array names without the postfix sorted alphabetically.
66std::set<string> GetSortedNames(const std::unordered_set<string>& names) {
67 std::vector<string> final_names;
68 final_names.reserve(names.size());
69 for (const auto& name : names) {
70 final_names.push_back(GetArrayName(name));
71 }
72 return std::set<string>(final_names.begin(), final_names.end());
73}

75// Gets the final shape after replacing the first dimension with batch size, if
76// it is undefined (containing the value -1). Returns whether the shape is
77// valid.
78bool ReplaceShapeBatchSize(const tensorflow::TensorShapeProto& shape,
79 int batch_size,
80 tensorflow::TensorShapeProto* final_shape) {
81 for (int idx = 0; idx < shape.dim().size(); ++idx) {
82 int64 final_dim = shape.dim()[idx].size();
83 if (final_dim == -1) {
84 if (idx > 0) return false;
85 final_dim = batch_size;
86 }
87 final_shape->add_dim()->set_size(final_dim);
88 }
89 return true;
90}

92// Updates the input arrays in ModelFlags to contain the shape of the array.
93void ProcessInputShapes(const tensorflow::GraphDef& graph_def, int batch_size,
94 ModelFlags* model_flags) {
95 // Build map of input array names to input arrays.
96 std::unordered_map<string, InputArray*> input_data_map;
97 for (auto& input : *model_flags->mutable_input_arrays()) {
98 input_data_map[input.name()] = &input;
99 }
100
101 // Adds shapes to the input arrays if the shape is valid.
102 for (const tensorflow::NodeDef& node_def : graph_def.node()) {
103 if (input_data_map.find(node_def.name()) != input_data_map.end()) {
104 const auto shape_it = node_def.attr().find("shape");
105 if (shape_it != node_def.attr().end()) {
106 tensorflow::TensorShapeProto final_shape;
107 bool is_valid = ReplaceShapeBatchSize(shape_it->second.shape(),
108 batch_size, &final_shape);
109
110 if (is_valid) {
111 auto* shape = input_data_map.at(node_def.name())->mutable_shape();
112 QCHECK_EQ(shape->dims_size(), 0)
113 << "The shape for the input '" << node_def.name()
114 << "' was previously defined. For clarity please define inputs "
115 << "via --input_arrays and input_shapes flags.\n";
116 for (const auto& dim : final_shape.dim()) {
117 shape->add_dims(dim.size());
118 }
119 }
120 }
121 }
122 }
123
124 // Checks all input arrays have a shape.
125 for (auto const& input : model_flags->input_arrays()) {
126 QCHECK(input.shape().dims_size() > 0)
127 << "A valid input shape was not found for input '" << input.name()
128 << "'. Please define via --input_arrays and --input_shapes flags.\n";
129 }
130}

134void ParseMetaData(const tensorflow::GraphDef& graph_def,
135 const std::unordered_set<string>& inputs,
136 const std::unordered_set<string>& outputs,
137 const ParsedTocoFlags& parsed_toco_flags,
138 const ParsedModelFlags& parsed_model_flags,
139 TocoFlags* toco_flags, ModelFlags* model_flags) {
140 if (!parsed_model_flags.input_arrays.specified()) {
141 const std::set<string> sorted_inputs = GetSortedNames(inputs);
142 for (const auto& input_name : sorted_inputs) {
143 model_flags->add_input_arrays()->set_name(input_name);
144 }
145 }
146
147 if (!parsed_model_flags.output_arrays.specified()) {
148 const std::set<string> sorted_outputs = GetSortedNames(outputs);
149 for (const auto& output_name : sorted_outputs) {
150 model_flags->add_output_arrays(GetArrayName(output_name));
151 }
152 }
153
154 if (!parsed_model_flags.input_shapes.specified()) {
155 int batch_size = parsed_model_flags.batch_size.value();
156 ProcessInputShapes(graph_def, batch_size, model_flags);
157 }
158
159 if (!parsed_toco_flags.inference_type.specified()) {
160 toco_flags->set_inference_type(IODataType::FLOAT);
161 }
162}

164// TODO(nupurgarg): Add top level tests.
165void GetSavedModelContents(const ParsedTocoFlags& parsed_toco_flags,
166 const ParsedModelFlags& parsed_model_flags,
167 TocoFlags* toco_flags, ModelFlags* model_flags,
168 string* graph_def_contents) {
169 // Loads the MetaGraphDef within a SavedModelBundle.
170 auto bundle = LoadSavedModel(parsed_toco_flags);
171
172 // Converts the MetaGraphDef to frozen GraphDef.
173 tensorflow::GraphDef frozen_graph_def;
174 std::unordered_set<string> inputs;
175 std::unordered_set<string> outputs;
176 TF_CHECK_OK(tensorflow::FreezeSavedModel(*bundle, &frozen_graph_def, &inputs,
177 &outputs));
178
179 // Reads the frozen GraphDef into a string.
180 QCHECK(frozen_graph_def.SerializeToString(graph_def_contents))
181 << "Unable to generate serialized GraphDef.\n";
182
183 // Process inputs and outputs and metadata within GraphDef.
184 const tensorflow::GraphDef graph_def = bundle->meta_graph_def.graph_def();
185 ParseMetaData(graph_def, inputs, outputs, parsed_toco_flags,
186 parsed_model_flags, toco_flags, model_flags);
187}
