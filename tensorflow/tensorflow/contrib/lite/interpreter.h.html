30// Map statically from a c++ type to a TfLiteType (used below for safe casts).
31template <class T>
32constexpr TfLiteType typeToTfLiteType() {
33 return kTfLiteNoType;
34}
35template <>
36constexpr TfLiteType typeToTfLiteType<int>() {
37 return kTfLiteInt32;
38}
39template <>
40constexpr TfLiteType typeToTfLiteType<int64_t>() {
41 return kTfLiteInt64;
42}
43template <>
44constexpr TfLiteType typeToTfLiteType<float>() {
45 return kTfLiteFloat32;
46}
47template <>
48constexpr TfLiteType typeToTfLiteType<unsigned char>() {
49 return kTfLiteUInt8;
50}

55// An interpreter for a graph of nodes that input and output from tensors.
56// Each node of the graph processes a set of input tensors and produces a
57// set of output Tensors. All inputs/output tensors are referenced by index.
58//
59// Usage:
60//
61// -- Create basic model
62// Interpreter foo(2, 1);
63// foo.SetTensorParametersReadWrite(0, ...);
64// foo.SetTensorParametersReadOnly(1, ...);
65// foo.SetNodeParameters(0, ...)
66//
67// -- Resize input array to 1 length.
68// foo.ResizeInputTensor(0, 1);
69// foo.AllocateTensors();
70// -- Install array data
71// foo.typed_tensor<float>(0)[0] = 3;
72// foo.Invoke();
73// foo.typed_tensor<float>(0)[0] = 4;
74// foo.Invoke();
75// -- Resize input array and set data.
76// foo.ResizeInputTensor(0, 2);
77// foo.AllocateTensors();
78// foo.typed_tensor<float>(0)[0] = 4;
79// foo.typed_tensor<float>(0)[1] = 8;
80// foo.Invoke();
81//
82
83struct TfLiteIntArrayDeleter {
84 void operator()(TfLiteIntArray* a) {
85 if (a) TfLiteIntArrayFree(a);
86 }
87};

115 // Adds a node with the given parameters and returns the index of the new

198 // Get a tensor data structure.
199 // TODO(aselle): Create a safe ArrayHandle interface to avoid exposing this

201 TfLiteTensor* tensor(int tensor_index) {

207 // Get a pointer to an operation and registration data structure if in bounds.

217 // Perform a checked cast to the appropriate tensor type.
218 template <class T>
219 T* typed_tensor(int tensor_index) {
220 if (TfLiteTensor* tensor_ptr = tensor(tensor_index)) {
221 if (tensor_ptr->type == typeToTfLiteType<T>()) {
222 return reinterpret_cast<T*>(tensor_ptr->data.raw);
223 }
224 }
225 return nullptr;
226 }

228 // Return a pointer into the data of a given input tensor. The given index
229 // must be between 0 and inputs().size().
230 template <class T>
231 T* typed_input_tensor(int index) {
232 return typed_tensor<T>(inputs_[index]);
233 }
234
235 // Return a pointer into the data of a given output tensor. The given index
236 // must be between 0 and outputs().size().
237 template <class T>
238 T* typed_output_tensor(int index) {
239 return typed_tensor<T>(outputs_[index]);
240 }

242 // Change the dimensionality of a given tensor. Note, this is only acceptable
243 // for tensor indices that are inputs.
244 // Returns status of failure or success.
245 // TODO(aselle): Consider implementing ArraySlice equivalent to make this
246 // more adept at accepting data without an extra copy. Use absl::ArraySlice

251 // Update allocations for all tensors. This will redim dependent tensors using
252 // the input tensor dimensionality as given. This is relatively expensive.

256 TfLiteStatus AllocateTensors();

258 // Invoke the interpreter (run the whole graph in dependency order).
259 //
260 // NOTE: It is possible that the interpreter is not in a ready state

272 // Allow a delegate to look at the graph and modify the graph to handle
273 // parts of the graph themselves. After this is called, the graph may
274 // contain new nodes that replace 1 more nodes.

279 // Ensure the data in `tensor.data` is readable. In case delegate is used,
280 // it might require to copy the data from delegate buffer to raw memory.
281 TfLiteStatus EnsureTensorDataIsReadable(int tensor_index) {
282 TF_LITE_ENSURE(&context_, tensor_index < tensors_size());
283 TfLiteTensor* tensor = &tensors_[tensor_index];
284 if (tensor->data_is_stale) {
285 TF_LITE_ENSURE(&context_, tensor->delegate != nullptr);
286 TF_LITE_ENSURE(&context_,
287 tensor->buffer_handle != kTfLiteNullBufferHandle);
288 // This can be null if the delegate doesn't use its own buffer.
289 TF_LITE_ENSURE(&context_,
290 tensor->delegate->CopyFromBufferHandle != nullptr);
291 tensor->delegate->CopyFromBufferHandle(tensor->delegate,
292 tensor->buffer_handle,
293 tensor->data.raw, tensor->bytes);
294 tensor->data_is_stale = false;
295 }
296 return kTfLiteOk;
297 }

299 // Set the delegate buffer handle to a tensor. It can be called in the
300 // following cases:
301 // 1. Set the buffer handle to a tensor that's not being written by a
302 // delegate. For example, feeding an OpenGL texture as the input of the
303 // inference graph.
304 // 2. Set the buffer handle to a tensor that uses the same delegate.
305 // For example, set an OpenGL texture as the output of inference, while
306 // the node which produces output is an OpenGL delegate node.
307 // WARNING: This is an experimental API and subject to change.
308 TfLiteStatus SetBufferHandle(int tensor_index,
309 TfLiteBufferHandle buffer_handle,
310 TfLiteDelegate* delegate);

312 // Get the delegate buffer handle, and the delegate which can process the
313 // buffer handle.

315 TfLiteStatus GetBufferHandle(int tensor_index,
316 TfLiteBufferHandle* buffer_handle,
317 TfLiteDelegate** delegate);

319 // The default capacity of `tensors_` vector.
320 static constexpr int kTensorsReservedCapacity = 128;
321 // The capacity headroom of `tensors_` vector before calling ops'
322 // `prepare` and `invoke` function. In these functions, it's guaranteed
323 // allocating up to `kTensorsCapacityHeadroom` more tensors won't invalidate
324 // pointers to existing tensors.
325 static constexpr int kTensorsCapacityHeadroom = 16;
326
327 private:
328 // Give 'op_reg' a chance to initialize itself using the contents of
329 // 'buffer'.
330 void* OpInit(const TfLiteRegistration& op_reg, const char* buffer,
331 size_t length) {
332 if (op_reg.init == nullptr) return nullptr;
333 return op_reg.init(&context_, buffer, length);
334 }
335
336 // Let 'op_reg' release any memory it might have allocated via 'OpInit'.
337 void OpFree(const TfLiteRegistration& op_reg, void* buffer) {
338 if (op_reg.free == nullptr) return;
339 if (buffer) {
340 op_reg.free(&context_, buffer);
341 }
342 }
343
344 // Prepare the given 'node' for execution.
345 TfLiteStatus OpPrepare(const TfLiteRegistration& op_reg, TfLiteNode* node) {
346 if (op_reg.prepare == nullptr) return kTfLiteOk;
347 return op_reg.prepare(&context_, node);
348 }
349
350 // Invoke the operator represented by 'node'.
351 TfLiteStatus OpInvoke(const TfLiteRegistration& op_reg, TfLiteNode* node) {
352 if (op_reg.invoke == nullptr) return kTfLiteError;
353 return op_reg.invoke(&context_, node);
354 }

356 // Call OpPrepare() for as many ops as possible, allocating memory for their
357 // tensors. If an op containing dynamic tensors is found, preparation will be
358 // postponed until this function is called again. This allows the interpreter
359 // to wait until Invoke() to resolve the sizes of dynamic tensors.
360 TfLiteStatus PrepareOpsAndTensors();
361
362 // Call OpPrepare() for all ops starting at 'first_node'. Stop when a
363 // dynamic tensors is found or all ops have been prepared. Fill
364 // 'last_node_prepared' with the id of the op containing dynamic tensors, or
365 // the last in the graph.
366 TfLiteStatus PrepareOpsStartingAt(int first_execution_plan_index,
367 int* last_execution_plan_index_prepared);

440 // Ensures that `tensors_` has at least `kTensorsCapacityHeadroom` extra
441 // capacity. Calling this function may invalidate existing pointers to
442 // tensors. After calling this function, adding `kTensorsCapacityHeadroom`
443 // more tensors won't invalidate the pointer to existing tensors.
444 void EnsureTensorsVectorCapacity() {
445 const int required_capacity = tensors_size() + kTensorsCapacityHeadroom;
446 if (required_capacity > tensors_.capacity()) {
447 tensors_.reserve(required_capacity);
448 context_.tensors = tensors_.data();
449 }
450 }

452 // The state of the Interpreter.
453 enum State {
454 // The interpreter isn't ready to be invoked.
455 // `AllocateTensor` need to be called to enter an invokable state.
456 kStateUninvokable = 0,
457 // The interpreter is ready to be invoked.
458 kStateInvokable,
459 // The interpreter is ready to be invoked, and graph can't be further
460 // modified. The interpreter will enter this state when calling
461 // `ModifyGraphWithDelegate` with `allow_dynamic_tensors=false`.
462 kStateInvokableAndImmutable,
463 };
464 State state_ = kStateUninvokable;

466 // A pure C data structure used to communicate with the pure C plugin
467 // interface. To avoid copying tensor metadata, this is also the definitive
468 // structure to store tensors.
469 TfLiteContext context_;
470
471 // Node inputs/outputs are stored in TfLiteNode and TfLiteRegistration stores
472 // function pointers to actual implementation.
473 std::vector<std::pair<TfLiteNode, TfLiteRegistration>>
474 nodes_and_registration_;
475
476 // Whether the model is consistent. That is to say if the inputs and outputs
477 // of every node and the global inputs and outputs are valid indexes into
478 // the tensor array.
479 bool consistent_ = true;
480
481 // Array of indices representing the tensors that are inputs to the
482 // interpreter.
483 std::vector<int> inputs_;
484
485 // Array of indices representing the tensors that are outputs to the
486 // interpreter.
487 std::vector<int> outputs_;
488
489 // The error reporter delegate that tflite will forward queries errors to.
490 ErrorReporter* error_reporter_;
491
492 // Index of the next node to prepare.
493 // During Invoke(), Interpreter will allocate input tensors first, which are
494 // known to be fixed size. Then it will allocate outputs from nodes as many
495 // as possible. When there is a node that produces dynamic sized tensor.
496 // Interpreter will stop allocating tensors, set the value of next allocate
497 // node id, and execute the node to generate the output tensor before continue
498 // to allocate successors. This process repeats until all nodes are executed.
499 // NOTE: this relies on the order of nodes that is in topological order.
500 int next_execution_plan_index_to_prepare_;
501
502 // WARNING: This is an experimental interface that is subject to change.
503 // This is a list of node indices (to index into nodes_and_registration).
504 // This represents a valid topological sort (dependency ordered) execution
505 // plan. In particular, it is valid for this ordering to contain only a
506 // subset of the node indices.
507 std::vector<int> execution_plan_;
508
509 // In the future, we'd like a TfLiteIntArray compatible representation.
510 // TODO(aselle): replace execution_plan_ with this.
511 std::unique_ptr<TfLiteIntArray, TfLiteIntArrayDeleter> plan_cache_;
512
513 // Whether to delegate to NN API
514 std::unique_ptr<NNAPIDelegate> nnapi_delegate_;
515
516 std::unique_ptr<MemoryPlanner> memory_planner_;
