21REGISTER_OP("WALSComputePartialLhsAndRhs")
22 .Input("factors: float32")
23 .Input("factor_weights: float32")
24 .Input("unobserved_weights: float32")
25 .Input("input_weights: float32")
26 .Input("input_indices: int64")
27 .Input("input_values: float32")
28 .Input("input_block_size: int64")
29 .Input("input_is_transpose: bool")
30 .Output("partial_lhs: float32")
31 .Output("partial_rhs: float32")
32 .SetShapeFn(shape_inference::UnknownShape)
33 .Doc(R"(
34Computes the partial left-hand side and right-hand side of WALS update.
35
36factors: Matrix of size m * k.
37factor_weights: Vector of size m. Corresponds to column weights
38unobserved_weights: Scalar. Weight for unobserved input entries.
39input_weights: Vector of size n. Corresponds to row weights.
40input_indices: Indices for the input SparseTensor.
41input_values: Values for the input SparseTensor.
42input_block_size: Scalar. Number of rows spanned by input.
43input_is_transpose: If true, logically transposes the input for processing.
44partial_lhs: 3-D tensor with size input_block_size x k x k.
45partial_rhs: Matrix with size input_block_size x k.
46)");

48REGISTER_OP("MaskedMatmul")
49 .Input("a: float32")
50 .Input("b: float32")
51 .Input("mask_indices: int64")
52 .Input("transpose_a: bool")
53 .Input("transpose_b: bool")
54 .Output("prod_values: float32")
55 .SetShapeFn(shape_inference::UnknownShape)
56 .Doc(R"(
57Computes the product a * b, but only for indices (i, j) in mask_indices. The
58result is stored in prod_values, a rank 1 tensor, such that for all i,
59prod_values[i] = (a * b)[mask_indices[i, 0], mask_indices[i, 1]].
60Note that the shapes of the input matrices a, b should be compatible (after
61transposing as specified by the arguments transpose_a and transpose_b).
62
63Input arguments:
64a: A rank 2 tensor of shape [m, n].
65b: A rank 2 tensor of shape [s, t]. The inner dimensions of a and b should match
66 after transposition.
67mask_indices: A rank 2 tensor, of shape [nnz, 2] where nnz is the number of
68 non-zero elements in the output. The indices are not assumed to be in
69 lexicographic, or any particular order.
70 For all i, mask_indices[i, :] should represent a valid index of the product
71 matrix (a * b) (after transposition). That is:
72 mask_indices[i, 0] should be in [0, m) if !transpose_a, and in [0, n)
73 otherwise.
74 mask_indices[i, 1] should be in [0, t) if !transpose_b, and in [0, s)
75 otherwise.
76transpose_a: A boolean, specifies whether to transpose the matrix a.
77transpose_b: A boolean, specifies whether to transpose the matrix b.
78
79Output arguments:
80prod_values: A rank 1 tensor of shape [nnz], representing the values of the
81 non-zero elements in the product, such that for all i,
82 prod_values[i] = (a * b)[mask_indices[i, 0], mask_indices[i, 1]].
83)");
