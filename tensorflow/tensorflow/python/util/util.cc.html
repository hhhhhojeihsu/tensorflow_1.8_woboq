31bool IsString(PyObject* o) {
32 return PyBytes_Check(o) ||
33#if PY_MAJOR_VERSION < 3
34 PyString_Check(o) ||
35#endif
36 PyUnicode_Check(o);
37}

39// Equivalent to Python's 'o.__class__.__name__'
40// Note that '__class__' attribute is set only in new-style classes.
41// A lot of tensorflow code uses __class__ without checks, so it seems like
42// we only support new-style classes.
43StringPiece GetClassName(PyObject* o) {
44 // __class__ is equivalent to type() for new style classes.
45 // type() is equivalent to PyObject_Type()
46 // (https://docs.python.org/3.5/c-api/object.html#c.PyObject_Type)
47 // PyObject_Type() is equivalent to o->ob_type except for Py_INCREF, which
48 // we don't need here.
49 PyTypeObject* type = o->ob_type;
50
51 // __name__ is the value of `tp_name` after the last '.'
52 // (https://docs.python.org/2/c-api/typeobj.html#c.PyTypeObject.tp_name)
53 StringPiece name(type->tp_name);
54 size_t pos = name.rfind('.');
55 if (pos != StringPiece::npos) {
56 name.remove_prefix(pos + 1);
57 }
58 return name;
59}

61string PyObjectToString(PyObject* o) {
62 if (o == nullptr) {
63 return "<null object>";
64 }
65 PyObject* str = PyObject_Str(o);
66 if (str) {
67#if PY_MAJOR_VERSION < 3
68 string s(PyString_AS_STRING(str));
69#else
70 string s(PyUnicode_AsUTF8(str));
71#endif
72 Py_DECREF(str);
73 return tensorflow::strings::StrCat("type=", GetClassName(o), " str=", s);
74 } else {
75 return "<failed to execute str() on object>";
76 }
77}

79// Implements the same idea as tensorflow.util.nest._yield_value
80// During construction we check if the iterable is a dictionary.
81// If so, we construct a sequence from its sorted keys that will be used
82// for iteration.
83// If not, we construct a sequence directly from the iterable.
84// At each step, we get the next element from the sequence and use it
85// either as a key or return it directly.
86//
87// 'iterable' must not be modified while ValIterator is used.
88class ValIterator {
89 public:
90 explicit ValIterator(PyObject* iterable) : dict_(nullptr), index_(0) {
91 if (PyDict_Check(iterable)) {
92 dict_ = iterable;
93 // PyDict_Keys returns a list, which can be used with
94 // PySequence_Fast_GET_ITEM.
95 seq_ = PyDict_Keys(iterable);
96 // Iterate through dictionaries in a deterministic order by sorting the
97 // keys. Notice this means that we ignore the original order of
98 // `OrderedDict` instances. This is intentional, to avoid potential
99 // bugs caused by mixing ordered and plain dicts (e.g., flattening
100 // a dict but using a corresponding `OrderedDict` to pack it back).
101 PyList_Sort(seq_);
102 } else {
103 seq_ = PySequence_Fast(iterable, "");
104 }
105 size_ = PySequence_Fast_GET_SIZE(seq_);
106 }
107
108 ~ValIterator() { Py_DECREF(seq_); }
109
110 // Return a borrowed reference to the next element from iterable.
111 // Return nullptr when iteration is over.
112 PyObject* next() {
113 PyObject* element = nullptr;
114 if (index_ < size_) {
115 // Both PySequence_Fast_GET_ITEM and PyDict_GetItem return borrowed
116 // references.
117 element = PySequence_Fast_GET_ITEM(seq_, index_);
118 ++index_;
119 if (dict_ != nullptr) {
120 element = PyDict_GetItem(dict_, element);
121 if (element == nullptr) {
122 PyErr_SetString(PyExc_RuntimeError,
123 "Dictionary was modified during iteration over it");
124 return nullptr;
125 }
126 }
127 }
128 return element;
129 }
130
131 private:
132 PyObject* seq_;
133 PyObject* dict_;
134 Py_ssize_t size_;
135 Py_ssize_t index_;
136};

138// Returns 1 if `o` is considered a sequence for the purposes of Flatten().
139// Returns 0 otherwise.
140// Returns -1 if an error occurred.
141int IsSequenceHelper(PyObject* o) {
142 if (PyDict_Check(o)) return true;
143 if (PySet_Check(o) && !WarnedThatSetIsNotSequence) {
144 LOG(WARNING) << "Sets are not currently considered sequences, "
145 "but this may change in the future, "
146 "so consider avoiding using them.";
147 WarnedThatSetIsNotSequence = true;
148 }
149 if (TF_PREDICT_FALSE(CollectionsSequenceType == nullptr)) {
150 PyErr_SetString(
151 PyExc_RuntimeError,
152 tensorflow::strings::StrCat(
153 "collections.Sequence type has not been set. "
154 "Please call RegisterSequenceClass before using this module")
155 .c_str());
156 return -1;
157 }
158 int is_instance = PyObject_IsInstance(o, CollectionsSequenceType);
159 if (is_instance == -1) return -1;
160 return static_cast<int>(is_instance != 0 && !IsString(o));
161}

163bool FlattenHelper(PyObject* nested, PyObject* list) {
164 // if nested is not a sequence, append itself and exit
165 int is_seq = IsSequenceHelper(nested);
166 if (is_seq == -1) return false;
167 if (!is_seq) {
168 return PyList_Append(list, nested) != -1;
169 }
170
171 // if nested if dictionary, sort it by key and recurse on each value
172 if (PyDict_Check(nested)) {
173 PyObject* keys = PyDict_Keys(nested);
174 if (PyList_Sort(keys) == -1) return false;
175 Py_ssize_t size = PyList_Size(keys);
176 for (Py_ssize_t i = 0; i < size; ++i) {
177 // We know that key and val will not be deleted because nested owns
178 // a reference to them and callers of flatten must not modify nested
179 // while the method is running.
180 PyObject* key = PyList_GET_ITEM(keys, i);
181 PyObject* val = PyDict_GetItem(nested, key);
182 if (Py_EnterRecursiveCall(" in flatten")) {
183 Py_DECREF(keys);
184 return false;
185 }
186 const bool success = FlattenHelper(val, list);
187 Py_LeaveRecursiveCall();
188 if (!success) {
189 Py_DECREF(keys);
190 return false;
191 }
192 }
193 Py_DECREF(keys);
194 return true;
195 }
196
197 // iterate and recurse
198 PyObject* item;
199 PyObject* iterator = PyObject_GetIter(nested);
200 while ((item = PyIter_Next(iterator)) != nullptr) {
201 if (Py_EnterRecursiveCall(" in flatten")) {
202 Py_DECREF(iterator);
203 Py_DECREF(item);
204 return false;
205 }
206 bool success = FlattenHelper(item, list);
207 Py_LeaveRecursiveCall();
208 if (!success) {
209 Py_DECREF(iterator);
210 Py_DECREF(item);
211 return false;
212 }
213 Py_DECREF(item);
214 }
215 Py_DECREF(iterator);
216 return true;
217}

219// Sets error using keys of 'dict1' and 'dict2'.
220// 'dict1' and 'dict2' are assumed to be Python dictionaries.
221void SetDifferentKeysError(PyObject* dict1, PyObject* dict2, string* error_msg,
222 bool* is_type_error) {
223 PyObject* k1 = PyDict_Keys(dict1);
224 PyObject* k2 = PyDict_Keys(dict2);
225 *is_type_error = false;
226 *error_msg = tensorflow::strings::StrCat(
227 "The two dictionaries don't have the same set of keys. "
228 "First structure has keys ",
229 PyObjectToString(k1), ", while second structure has keys ",
230 PyObjectToString(k2));
231 Py_DECREF(k1);
232 Py_DECREF(k2);
233}

235// Returns true iff there were no "internal" errors. In other words,
236// errors that has nothing to do with structure checking.
237// If an "internal" error occured, the appropriate Python error will be
238// set and the caller can propage it directly to the user.
239//
240// Both `error_msg` and `is_type_error` must be non-null. `error_msg` must
241// be empty.
242// Leaves `error_msg` empty if structures matched. Else, fills `error_msg`
243// with appropriate error and sets `is_type_error` to true iff
244// the error to be raised should be TypeError.
245bool AssertSameStructureHelper(PyObject* o1, PyObject* o2, bool check_types,
246 string* error_msg, bool* is_type_error) {
247 DCHECK(error_msg);
248 DCHECK(is_type_error);
249 const bool is_seq1 = IsSequence(o1);
250 const bool is_seq2 = IsSequence(o2);
251 if (PyErr_Occurred()) return false;
252 if (is_seq1 != is_seq2) {
253 string seq_str = is_seq1 ? PyObjectToString(o1) : PyObjectToString(o2);
254 string non_seq_str = is_seq1 ? PyObjectToString(o2) : PyObjectToString(o1);
255 *is_type_error = false;
256 *error_msg = tensorflow::strings::StrCat(
257 "Substructure \"", seq_str, "\" is a sequence, while substructure \"",
258 non_seq_str, "\" is not");
259 return true;
260 }
261
262 // Got to scalars, so finished checking. Structures are the same.
263 if (!is_seq1) return true;
264
265 if (check_types) {
266 const PyTypeObject* type1 = o1->ob_type;
267 const PyTypeObject* type2 = o2->ob_type;
268
269 // We treat two different namedtuples with identical name and fields
270 // as having the same type.
271 const PyObject* o1_tuple = IsNamedtuple(o1, true);
272 if (o1_tuple == nullptr) return false;
273 const PyObject* o2_tuple = IsNamedtuple(o2, true);
274 if (o2_tuple == nullptr) {
275 Py_DECREF(o1_tuple);
276 return false;
277 }
278 bool both_tuples = o1_tuple == Py_True && o2_tuple == Py_True;
279 Py_DECREF(o1_tuple);
280 Py_DECREF(o2_tuple);
281
282 if (both_tuples) {
283 const PyObject* same_tuples = SameNamedtuples(o1, o2);
284 if (same_tuples == nullptr) return false;
285 bool not_same_tuples = same_tuples != Py_True;
286 Py_DECREF(same_tuples);
287 if (not_same_tuples) {
288 *is_type_error = true;
289 *error_msg = tensorflow::strings::StrCat(
290 "The two namedtuples don't have the same sequence type. "
291 "First structure ",
292 PyObjectToString(o1), " has type ", type1->tp_name,
293 ", while second structure ", PyObjectToString(o2), " has type ",
294 type2->tp_name);
295 return true;
296 }
297 } else if (type1 != type2) {
298 *is_type_error = true;
299 *error_msg = tensorflow::strings::StrCat(
300 "The two namedtuples don't have the same sequence type. "
301 "First structure ",
302 PyObjectToString(o1), " has type ", type1->tp_name,
303 ", while second structure ", PyObjectToString(o2), " has type ",
304 type2->tp_name);
305 return true;
306 }
307
308 if (PyDict_Check(o1)) {
309 if (PyDict_Size(o1) != PyDict_Size(o2)) {
310 SetDifferentKeysError(o1, o2, error_msg, is_type_error);
311 return true;
312 }
313
314 PyObject* key;
315 Py_ssize_t pos = 0;
316 while (PyDict_Next(o1, &pos, &key, nullptr)) {
317 if (PyDict_GetItem(o2, key) == nullptr) {
318 SetDifferentKeysError(o1, o2, error_msg, is_type_error);
319 return true;
320 }
321 }
322 }
323 }
324
325 ValIterator iter1(o1);
326 ValIterator iter2(o2);
327
328 while (true) {
329 PyObject* v1 = iter1.next();
330 PyObject* v2 = iter2.next();
331 if (v1 != nullptr && v2 != nullptr) {
332 if (Py_EnterRecursiveCall(" in assert_same_structure")) {
333 return false;
334 }
335 bool no_internal_errors = AssertSameStructureHelper(
336 v1, v2, check_types, error_msg, is_type_error);
337 Py_LeaveRecursiveCall();
338 if (!no_internal_errors) return false;
339 if (!error_msg->empty()) return true;
340 } else if (v1 == nullptr && v2 == nullptr) {
341 // Done with all recursive calls. Structure matched.
342 return true;
343 } else {
344 *is_type_error = false;
345 *error_msg = tensorflow::strings::StrCat(
346 "The two structures don't have the same number of elements. ",
347 "First structure: ", PyObjectToString(o1),
348 ". Second structure: ", PyObjectToString(o2));
349 return true;
350 }
351 }
352}

356void RegisterSequenceClass(PyObject* sequence_class) {
357 if (!PyType_Check(sequence_class)) {
358 PyErr_SetString(
359 PyExc_TypeError,
360 tensorflow::strings::StrCat(
361 "Expecting a class definition for `collections.Sequence`. Got ",
362 Py_TYPE(sequence_class)->tp_name)
363 .c_str());
364 return;
365 }
366 CollectionsSequenceType = sequence_class;
367}

369bool IsSequence(PyObject* o) { return IsSequenceHelper(o) == 1; }

371PyObject* Flatten(PyObject* nested) {
372 PyObject* list = PyList_New(0);
373 if (FlattenHelper(nested, list)) {
374 return list;
375 } else {
376 Py_DECREF(list);
377 return nullptr;
378 }
379}

381PyObject* IsNamedtuple(PyObject* o, bool strict) {
382 // Must be subclass of tuple
383 if (!PyTuple_Check(o)) {
384 Py_RETURN_FALSE;
385 }
386
387 // If strict, o.__class__.__base__ must be tuple
388 if (strict) {
389 PyObject* klass = PyObject_GetAttrString(o, "__class__");
390 if (klass == nullptr) return nullptr;
391 PyObject* base = PyObject_GetAttrString(klass, "__base__");
392 Py_DECREF(klass);
393 if (base == nullptr) return nullptr;
394
395 const PyTypeObject* base_type = reinterpret_cast<PyTypeObject*>(base);
396 // built-in object types are singletons
397 bool tuple_base = base_type == &PyTuple_Type;
398 Py_DECREF(base);
399 if (!tuple_base) {
400 Py_RETURN_FALSE;
401 }
402 }
403
404 if (TF_PREDICT_FALSE(CollectionsSequenceType == nullptr)) {
405 PyErr_SetString(
406 PyExc_RuntimeError,
407 tensorflow::strings::StrCat(
408 "collections.Sequence type has not been set. "
409 "Please call RegisterSequenceClass before using this module")
410 .c_str());
411 return nullptr;
412 }
413
414 // o must have attribute '_fields' and every element in
415 // '_fields' must be a string.
416 int has_fields = PyObject_HasAttrString(o, "_fields");
417 if (!has_fields) {
418 Py_RETURN_FALSE;
419 }
420
421 Safe_PyObjectPtr fields = make_safe(PyObject_GetAttrString(o, "_fields"));
422 int is_instance = PyObject_IsInstance(fields.get(), CollectionsSequenceType);
423 if (is_instance == 0) {
424 Py_RETURN_FALSE;
425 } else if (is_instance == -1) {
426 return nullptr;
427 }
428
429 Safe_PyObjectPtr seq = make_safe(PySequence_Fast(fields.get(), ""));
430 const Py_ssize_t s = PySequence_Fast_GET_SIZE(seq.get());
431 for (Py_ssize_t i = 0; i < s; ++i) {
432 // PySequence_Fast_GET_ITEM returns borrowed ref
433 PyObject* elem = PySequence_Fast_GET_ITEM(seq.get(), i);
434 if (!IsString(elem)) {
435 Py_RETURN_FALSE;
436 }
437 }
438
439 Py_RETURN_TRUE;
440}

442PyObject* SameNamedtuples(PyObject* o1, PyObject* o2) {
443 PyObject* f1 = PyObject_GetAttrString(o1, "_fields");
444 PyObject* f2 = PyObject_GetAttrString(o2, "_fields");
445 if (f1 == nullptr || f2 == nullptr) {
446 Py_XDECREF(f1);
447 Py_XDECREF(f2);
448 PyErr_SetString(
449 PyExc_RuntimeError,
450 "Expected namedtuple-like objects (that have _fields attr)");
451 return nullptr;
452 }
453
454 if (PyObject_RichCompareBool(f1, f2, Py_NE)) {
455 Py_RETURN_FALSE;
456 }
457
458 if (GetClassName(o1).compare(GetClassName(o2)) == 0) {
459 Py_RETURN_TRUE;
460 } else {
461 Py_RETURN_FALSE;
462 }
463}

465PyObject* AssertSameStructure(PyObject* o1, PyObject* o2, bool check_types) {
466 string error_msg;
467 bool is_type_error = false;
468 AssertSameStructureHelper(o1, o2, check_types, &error_msg, &is_type_error);
469 if (!error_msg.empty()) {
470 PyErr_SetString(
471 is_type_error ? PyExc_TypeError : PyExc_ValueError,
472 tensorflow::strings::StrCat(
473 "The two structures don't have the same nested structure.\n\n",
474 "First structure: ", PyObjectToString(o1), "\n\nSecond structure: ",
475 PyObjectToString(o2), "\n\nMore specifically: ", error_msg)
476 .c_str());
477 return nullptr;
478 }
479 Py_RETURN_NONE;
480}
