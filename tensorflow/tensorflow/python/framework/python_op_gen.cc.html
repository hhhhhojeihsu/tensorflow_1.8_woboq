48bool IsPythonReserved(const string& s) {
49 static const std::set<string>* const kPythonReserved = new std::set<string>(
50 {// Keywords in Python, from:
51 // import keyword
52 // print keyword.kwlist
53 "and", "as", "assert", "break", "class", "continue", "def", "del",
54 "elif", "else", "except", "exec", "finally", "for", "from", "global",
55 "if", "import", "in", "is", "lambda", "not", "or", "pass", "print",
56 "raise", "return", "try", "while", "with", "yield",
57 // Built-in functions and types in Python, from:
58 // [x for x in dir(__builtins__) if not x[0].islower()]
59 "ArithmeticError", "AssertionError", "AttributeError", "BaseException",
60 "BufferError", "BytesWarning", "DeprecationWarning", "EOFError",
61 "Ellipsis", "EnvironmentError", "Exception", "False",
62 "FloatingPointError", "FutureWarning", "GeneratorExit", "IOError",
63 "ImportError", "ImportWarning", "IndentationError", "IndexError",
64 "KeyError", "KeyboardInterrupt", "LookupError", "MemoryError",
65 "NameError", "None", "NotImplemented", "NotImplementedError", "OSError",
66 "OverflowError", "PendingDeprecationWarning", "ReferenceError",
67 "RuntimeError", "RuntimeWarning", "StandardError", "StopIteration",
68 "SyntaxError", "SyntaxWarning", "SystemError", "SystemExit", "TabError",
69 "True", "TypeError", "UnboundLocalError", "UnicodeDecodeError",
70 "UnicodeEncodeError", "UnicodeError", "UnicodeTranslateError",
71 "UnicodeWarning", "UserWarning", "ValueError", "Warning",
72 "ZeroDivisionError", "__debug__", "__doc__", "__import__", "__name__",
73 "__package__"});
74
75 return kPythonReserved->count(s) > 0;
76}

110// Indent the first line by "initial" spaces and all following lines
111// by "rest" spaces.
112string Indent(int initial, int rest, StringPiece in) {
113 // TODO(josh11b): Also word-wrapping?
114 string copy(in.data(), in.size());
115 str_util::StripTrailingWhitespace(&copy);
116 std::vector<string> v = str_util::Split(copy, '\n');
117
118 string result;
119 bool first = true;
120 for (const string& line : v) {
121 if (first) {
122 result = strings::StrCat(Spaces(initial), line, "\n");
123 first = false;
124 } else {
125 if (line.empty()) {
126 strings::StrAppend(&result, "\n");
127 } else {
128 strings::StrAppend(&result, Spaces(rest), line, "\n");
129 }
130 }
131 }
132 return result;
133}

135// Adds append to *dest, with a space if the first line will be <= width,
136// or a newline otherwise.
137void AppendWithinWidth(string* dest, StringPiece append, int width) {
138 auto first_line = append.find('\n');
139 if (first_line == string::npos) first_line = append.size();
140 if (dest->size() + first_line + 1 /* space */ > static_cast<size_t>(width)) {
141 strings::StrAppend(dest, "\n", append);
142 } else {
143 strings::StrAppend(dest, " ", append);
144 }
145}

168string TypeListString(const AttrValue& value) {
169 string ret;
170 for (int t : value.list().type()) {
171 if (!ret.empty()) strings::StrAppend(&ret, ", ");
172 DataType dtype = static_cast<DataType>(t);
173 if (IsRefType(dtype)) {
174 strings::StrAppend(&ret, PythonDataTypeString(RemoveRefType(dtype)),
175 " mutable");
176 } else {
177 strings::StrAppend(&ret, "`", PythonDataTypeString(dtype), "`");
178 }
179 }
180 return ret;
181}

190string ArgTypeName(const OpDef& op_def, const OpDef::ArgDef& arg,
191 const std::unordered_map<string, string>& inferred_attrs,
192 bool is_output) {
193 if (!arg.number_attr().empty()) {
194 // N Tensors with the same type
195 const string* original_arg =
196 gtl::FindOrNull(inferred_attrs, arg.number_attr());
197 string prefix;
198 if (original_arg == nullptr) {
199 prefix = strings::StrCat("A list of `", arg.number_attr(), "`");
200 } else if (*original_arg == arg.name()) {
201 const OpDef::AttrDef* attr = FindAttr(arg.number_attr(), op_def);
202 if (attr->has_minimum() && attr->minimum() > 0) {
203 prefix = strings::StrCat("A list of at least ", attr->minimum());
204 } else {
205 prefix = "A list of";
206 }
207 } else {
208 prefix = strings::StrCat("A list with the same length as `",
209 AvoidPythonReserved(*original_arg), "` of");
210 }
211
212 if (arg.type() != DT_INVALID) {
213 return strings::StrCat(prefix, " `Tensor` objects with type ",
214 TypeString(arg.type(), arg.is_ref()), ".");
215 } else {
216 original_arg = gtl::FindOrNull(inferred_attrs, arg.type_attr());
217 if (arg.is_ref()) {
218 strings::StrAppend(&prefix, " mutable");
219 }
220 if (original_arg == nullptr) {
221 return strings::StrCat(prefix, " `Tensor` objects with type `",
222 arg.type_attr(), "`.");
223 } else if (*original_arg == arg.name()) {
224 const OpDef::AttrDef* attr = FindAttr(arg.type_attr(), op_def);
225 if (attr->has_allowed_values()) {
226 return strings::StrCat(prefix,
227 " `Tensor` objects with the same type in: ",
228 TypeListString(attr->allowed_values()), ".");
229 } else {
230 return strings::StrCat(prefix,
231 " `Tensor` objects with the same type.");
232 }
233 } else {
234 return strings::StrCat(prefix,
235 " `Tensor` objects with the same type as `",
236 AvoidPythonReserved(*original_arg), "`.");
237 }
238 }
239 } else if (!arg.type_attr().empty() || !arg.type_list_attr().empty()) {
240 const bool is_list = !arg.type_list_attr().empty();
241 const string attr_name = is_list ? arg.type_list_attr() : arg.type_attr();
242 const OpDef::AttrDef* attr = FindAttr(attr_name, op_def);
243 const string mutable_str = arg.is_ref() ? "mutable " : "";
244 const string prefix =
245 is_list ? strings::StrCat("A list of ", mutable_str, "`Tensor` objects")
246 : strings::StrCat("A ", mutable_str, "`Tensor`");
247 const string* original_arg = gtl::FindOrNull(inferred_attrs, attr_name);
248 if (original_arg == nullptr) {
249 return strings::StrCat(prefix, " of type `", attr_name, "`.");
250 } else if (*original_arg == arg.name()) {
251 if (attr->has_allowed_values()) {
252 if (is_list) {
253 return strings::StrCat(prefix, " with types from: ",
254 TypeListString(attr->allowed_values()), ".");
255 } else {
256 return strings::StrCat(
257 prefix, is_output ? ". Has one of the following types: "
258 : ". Must be one of the following types: ",
259 TypeListString(attr->allowed_values()), ".");
260 }
261 } else {
262 return strings::StrCat(prefix, ".");
263 }
264 } else {
265 return strings::StrCat(prefix,
266 is_output ? ". Has the same type as `"
267 : ". Must have the same type as `",
268 AvoidPythonReserved(*original_arg), "`.");
269 }
270 } else {
271 return SingleTensorName(arg.type(), arg.is_ref());
272 }
273}

275string GetReturns(const OpDef& op_def,
276 const std::vector<string>& output_type_string) {
277 string result;
278 DCHECK_EQ(op_def.output_arg_size(), output_type_string.size());
279 const int num_outs = op_def.output_arg_size();
280 strings::StrAppend(&result, "\n Returns:\n");
281 if (num_outs == 0) {
282 strings::StrAppend(&result, " The created Operation.\n");
283 } else {
284 if (num_outs == 1) {
285 StringPiece description = op_def.output_arg(0).description();
286 if (ConsumeEquals(&description)) { // Skip the generated type info.
287 strings::StrAppend(&result, Indent(4, 4, description));
288 } else {
289 // Special case of one output, don't use the name of the output unless
290 // there is no description.
291 string desc = output_type_string.empty() ? kUnknownTensorType
292 : output_type_string[0];
293 if (desc == kUnknownTensorType) {
294 // Special case where we don't understand how the output tensor type
295 // depends on the input tensor types, just use the output arg
296 // description if we can.
297 if (!description.empty()) {
298 desc = op_def.output_arg(0).description();
299 } else if (!op_def.output_arg(0).name().empty()) {
300 desc = strings::StrCat(" The ", op_def.output_arg(0).name(),
301 " `Tensor`.");
302 }
303 } else if (!description.empty()) {
304 AppendWithinWidth(&desc, description, kRightMargin - 4 /* indent */);
305 }
306 strings::StrAppend(&result, Indent(4, 4, desc));
307 }
308 } else {
309 std::vector<string> out_names(num_outs);
310 for (int i = 0; i < num_outs; ++i) {
311 if (!op_def.output_arg(i).name().empty()) {
312 out_names[i] = op_def.output_arg(i).name();
313 } else {
314 out_names[i] = strings::StrCat("output", i);
315 }
316 }
317 strings::StrAppend(&result, " A tuple of `Tensor` objects (",
318 str_util::Join(out_names, ", "), ").\n\n");
319 for (int i = 0; i < num_outs; ++i) {
320 string desc = strings::StrCat(out_names[i], ": ");
321 StringPiece description = op_def.output_arg(i).description();
322 if (ConsumeEquals(&description)) { // Skip the generated type info.
323 strings::StrAppend(&desc, description);
324 } else {
325 const string type = static_cast<size_t>(i) < output_type_string.size()
326 ? output_type_string[i]
327 : kUnknownTensorType;
328 if (!description.empty()) {
329 if (type == kUnknownTensorType) {
330 // Special case where we don't understand how the output tensor
331 // type depends on the input tensor types, so we just use the
332 // output arg description.
333 strings::StrAppend(&desc, description);
334 } else {
335 strings::StrAppend(&desc, type, " ", description);
336 }
337 } else {
338 strings::StrAppend(&desc, type);
339 }
340 }
341 strings::StrAppend(&result, Indent(4, 6, desc));
342 }
343 }
344 }
345 return result;
346}

356string ShapeToPython(const TensorShapeProto& shape) {
357 if (shape.unknown_rank()) {
358 return "None";
359 }
360 string python = "[";
361 for (const auto& dim : shape.dim()) {
362 if (python.size() > 1) strings::StrAppend(&python, ", ");
363 if (!dim.name().empty()) {
364 strings::StrAppend(&python, "(", StringToPython(dim.name()), ", ",
365 dim.size(), ")");
366 } else {
367 strings::StrAppend(&python, dim.size());
368 }
369 }
370 strings::StrAppend(&python, "]");
371 return python;
372}

378string AttrListToPython(const AttrValue& value,
379 const string& dtype_module = "tf.") {
380 string ret;
381 if (value.list().s_size() > 0) {
382 for (int i = 0; i < value.list().s_size(); ++i) {
383 if (i > 0) strings::StrAppend(&ret, ", ");
384 strings::StrAppend(&ret, StringToPython(value.list().s(i)));
385 }
386 } else if (value.list().i_size() > 0) {
387 for (int i = 0; i < value.list().i_size(); ++i) {
388 if (i > 0) strings::StrAppend(&ret, ", ");
389 strings::StrAppend(&ret, value.list().i(i));
390 }
391 } else if (value.list().f_size() > 0) {
392 for (int i = 0; i < value.list().f_size(); ++i) {
393 if (i > 0) strings::StrAppend(&ret, ", ");
394 strings::StrAppend(&ret, value.list().f(i));
395 }
396 } else if (value.list().b_size() > 0) {
397 for (int i = 0; i < value.list().b_size(); ++i) {
398 if (i > 0) strings::StrAppend(&ret, ", ");
399 strings::StrAppend(&ret, value.list().b(i) ? "True" : "False");
400 }
401 } else if (value.list().type_size() > 0) {
402 for (int i = 0; i < value.list().type_size(); ++i) {
403 if (i > 0) strings::StrAppend(&ret, ", ");
404 strings::StrAppend(&ret,
405 DataTypeToPython(value.list().type(i), dtype_module));
406 }
407 } else if (value.list().shape_size() > 0) {
408 for (int i = 0; i < value.list().shape_size(); ++i) {
409 if (i > 0) strings::StrAppend(&ret, ", ");
410 strings::StrAppend(&ret, ShapeToPython(value.list().shape(i)));
411 }
412 } else if (value.list().tensor_size() > 0) {
413 for (int i = 0; i < value.list().tensor_size(); ++i) {
414 if (i > 0) strings::StrAppend(&ret, ", ");
415 strings::StrAppend(&ret, TensorToPython(value.list().tensor(i)));
416 }
417 } else if (value.list().func_size() > 0) {
418 for (int i = 0; i < value.list().func_size(); ++i) {
419 if (i > 0) strings::StrAppend(&ret, ", ");
420 strings::StrAppend(&ret, StringToPython(value.list().func(i).name()));
421 }
422 }
423 return ret;
424}

426// NOTE: The return value may contain spaces (for example, it could be
427// a string "foo bar" with an embedded space) and is not safe to pass
428// to WordWrap().
429string AttrValueToPython(const string& type, const AttrValue& value,
430 const string& dtype_module) {
431 if (type == "string") {
432 return StringToPython(value.s());
433 } else if (type == "int") {
434 return strings::StrCat(value.i());
435 } else if (type == "float") {
436 if (std::isnan(value.f()) || std::isinf(value.f())) {
437 return strings::StrCat("float('", value.f(), "')");
438 } else {
439 return strings::StrCat(value.f());
440 }
441 } else if (type == "bool") {
442 return value.b() ? "True" : "False";
443 } else if (type == "type") {
444 return DataTypeToPython(value.type(), dtype_module);
445 } else if (type == "shape") {
446 return ShapeToPython(value.shape());
447 } else if (type == "tensor") {
448 return TensorToPython(value.tensor());
449 } else if (type == "func") {
450 return StringToPython(value.func().name());
451 } else if (str_util::StartsWith(type, "list(")) {
452 return strings::StrCat("[", AttrListToPython(value, dtype_module), "]");
453 } else {
454 return "?";
455 }
456}

458void GenerateLowerCaseOpName(const string& str, string* result) {
459 const char joiner = '_';
460 const int last_index = str.size() - 1;
461 for (int i = 0; i <= last_index; ++i) {
462 const char c = str[i];
463 // Emit a joiner only if a previous-lower-to-now-upper or a
464 // now-upper-to-next-lower transition happens.
465 if (isupper(c) && (i > 0)) {
466 if (islower(str[i - 1]) || ((i < last_index) && islower(str[i + 1]))) {
467 result->push_back(joiner);
468 }
469 }
470 result->push_back(tolower(c));
471 }
472}

505string GenPythonOp::Code() {
506 // This has all the input args followed by those attrs that don't have
507 // defaults.
508 std::vector<ParamNames> params_no_default;
509 // The parameters with defaults (these have to be listed after those without).
510 // No input args are included, just attrs.
511 std::vector<ParamNames> params_with_default;
512
513 for (int i = 0; i < api_def_.arg_order_size(); ++i) {
514 const auto& arg = *FindInputArg(api_def_.arg_order(i), op_def_);
515 const auto& api_def_arg = *FindInputArg(api_def_.arg_order(i), api_def_);
516 params_no_default.emplace_back(api_def_arg.name(), api_def_arg.rename_to());
517 if (!arg.type_attr().empty()) {
518 gtl::InsertIfNotPresent(&inferred_attrs_, arg.type_attr(), arg.name());
519 } else if (!arg.type_list_attr().empty()) {
520 gtl::InsertIfNotPresent(&inferred_attrs_, arg.type_list_attr(),
521 arg.name());
522 }
523 if (!arg.number_attr().empty()) {
524 gtl::InsertIfNotPresent(&inferred_attrs_, arg.number_attr(), arg.name());
525 }
526 }
527 for (int i = 0; i < api_def_.attr_size(); ++i) {
528 const auto& attr(api_def_.attr(i));
529 // Do not add inferred attrs to the Python function signature.
530 if (inferred_attrs_.find(attr.name()) == inferred_attrs_.end()) {
531 if (attr.has_default_value()) {
532 params_with_default.emplace_back(attr.name(), attr.rename_to());
533 } else {
534 params_no_default.emplace_back(attr.name(), attr.rename_to());
535 }
536 }
537 }
538
539 // Save the list of attr parameters (attrs that won't be inferred),
540 // those with defaults go at the end.
541 // Get the attrs in the order we want by taking the attrs without defaults
542 // from the end of args_no_default, and adding args_no_default.
543 attrs_.reserve(params_no_default.size() - op_def_.input_arg_size() +
544 params_with_default.size());
545 for (int i = op_def_.input_arg_size(); i < params_no_default.size(); ++i) {
546 attrs_.push_back(params_no_default[i].GetName());
547 }
548 for (int i = 0; i < params_with_default.size(); ++i) {
549 attrs_.push_back(params_with_default[i].GetName());
550 }
551
552 param_names_.reserve(params_no_default.size() + params_with_default.size());
553 param_names_.insert(param_names_.begin(), params_no_default.begin(),
554 params_no_default.end());
555 for (const auto& param : params_with_default) {
556 param_names_.push_back(param);
557 }
558
559 string parameters;
560 for (const auto& param : params_no_default) {
561 AddDelimiter(&parameters, ", ");
562 strings::StrAppend(&parameters, param.GetRenameTo());
563 }
564 for (const auto& param_and_default : params_with_default) {
565 AddDelimiter(&parameters, ", ");
566 strings::StrAppend(&parameters, param_and_default.GetRenameTo(), "=None");
567 }
568 AddDelimiter(&parameters, ", ");
569 strings::StrAppend(&parameters, "name=None");
570
571 AddExport();
572 AddDefLine(parameters);
573 AddDocStringDescription();
574 AddDocStringArgs();
575 AddDocStringInputs();
576 AddDocStringAttrs();
577 AddDocStringNameArg();
578 AddOutputGlobals();
579 AddDocStringOutputs();
580 strings::StrAppend(&result_, " \"\"\"\n");
581 AddBody(" ");
582 strings::StrAppend(&result_, "\n\n");
583
584 return prelude_ + result_;
585}

587void GenPythonOp::AddExport() {
588 if (api_def_.visibility() != ApiDef::VISIBLE) {
589 return;
590 }
591
592 strings::StrAppend(&result_, "@tf_export(");
593
594 // Add all endpoint names to tf_export.
595 bool first_endpoint = true;
596 for (const auto& endpoint : api_def_.endpoint()) {
597 if (!first_endpoint) {
598 strings::StrAppend(&result_, ", ");
599 } else {
600 first_endpoint = false;
601 }

655void GenPythonOp::AddDocStringAttrs() {
656 for (const string& name : attrs_) {
657 const auto& attr = *FindAttr(name, op_def_);
658 const auto& api_def_attr = *FindAttr(name, api_def_);
659 string desc =
660 strings::StrCat(AvoidPythonReserved(api_def_attr.rename_to()), ": ");
661
662 static const char* const kAttrTypeName[][2] = {
663 {"string", "`string`"},
664 {"list(string)", "list of `strings`"},
665 {"int", "`int`"},
666 {"list(int)", "list of `ints`"},
667 {"float", "`float`"},
668 {"list(float)", "list of `floats`"},
669 {"bool", "`bool`"},
670 {"list(bool)", "list of `bools`"},
671 {"type", "`tf.DType`"},
672 {"list(type)", "list of `tf.DTypes`"},
673 {"shape", "`tf.TensorShape` or list of `ints`"},
674 {"list(shape)",
675 "list of shapes (each a `tf.TensorShape` or list of `ints`)"},
676 {"tensor", "`tf.TensorProto`"},
677 {"list(tensor)", "list of `tf.TensorProto` objects"},
678 {"func", "function decorated with @Defun"},
679 {"list(func)", "list of functions decorated with @Defun"},
680 };
681 for (size_t i = 0; i < TF_ARRAYSIZE(kAttrTypeName); ++i) {
682 if (attr.type() == kAttrTypeName[i][0]) {
683 string s;
684 if (api_def_attr.has_default_value()) {
685 s = strings::StrCat("optional ", kAttrTypeName[i][1]);
686 } else {
687 s = kAttrTypeName[i][1];
688 }
689 if (s[0] == 'o' || (s[0] == '`' && (s[1] == 'i' || s[1] == 'o'))) {
690 strings::StrAppend(&desc, "An ", s);
691 } else {
692 strings::StrAppend(&desc, "A ", s);
693 }
694 break;
695 }
696 }
697
698 if (attr.has_allowed_values()) {
699 strings::StrAppend(&desc, " from: `",
700 AttrListToPython(attr.allowed_values()), "`");
701 }
702
703 if (attr.has_minimum()) {
704 if (attr.type() == "int") {
705 strings::StrAppend(&desc, " that is `>= ", attr.minimum(), "`");
706 } else if (attr.minimum() > 0) {
707 strings::StrAppend(&desc, " that has length `>= ", attr.minimum(), "`");
708 }
709 }
710
711 strings::StrAppend(&desc, ".");
712
713 if (api_def_attr.has_default_value()) {
714 strings::StrAppend(
715 &desc, " Defaults to `",
716 AttrValueToPython(attr.type(), api_def_attr.default_value()), "`.");
717 }
718 if (!api_def_attr.description().empty()) {
719 AppendWithinWidth(&desc, api_def_attr.description(),
720 kRightMargin - 4 /* indent */);
721 }
722 strings::StrAppend(&result_, Indent(4, 6, desc));

808string GetPythonOps(const OpList& ops, const ApiDefMap& api_defs,
809 const std::vector<string>& hidden_ops,
810 bool require_shapes) {
811 string result;
812 // Header
813 // TODO(josh11b): Mention the library for which wrappers are being generated.
814 strings::StrAppend(&result, R"("""Python wrappers around TensorFlow ops.
815
816This file is MACHINE GENERATED! Do not edit.
817"""
818
819import collections as _collections
820
821from tensorflow.core.framework import op_def_pb2 as _op_def_pb2
822
823# Needed to trigger the call to _set_call_cpp_shape_fn.
824from tensorflow.python.framework import common_shapes as _common_shapes
825
826from tensorflow.python.framework import op_def_registry as _op_def_registry
827from tensorflow.python.framework import ops as _ops
828from tensorflow.python.framework import op_def_library as _op_def_library
829from tensorflow.python.util.tf_export import tf_export
830)");
831
832 // We'll make a copy of ops that filters out descriptions.
833 OpList cleaned_ops;
834 auto out = cleaned_ops.mutable_op();
835 out->Reserve(ops.op_size());
836 for (const auto& op_def : ops.op()) {
837 const auto* api_def = api_defs.GetApiDef(op_def.name());
838
839 if (api_def->visibility() == ApiDef::SKIP) {
840 continue;
841 }
842
843 // An op is hidden if either its ApiDef visibility is HIDDEN
844 // or it is in the hidden_ops list.
845 bool is_hidden = api_def->visibility() == ApiDef::HIDDEN;
846 bool hidden_by_api_def = is_hidden;
847 if (!is_hidden) {
848 for (const string& hidden : hidden_ops) {
849 if (op_def.name() == hidden) {
850 is_hidden = true;
851 break;
852 }
853 }
854 }
855
856 string function_name;
857 python_op_gen_internal::GenerateLowerCaseOpName(op_def.name(),
858 &function_name);
859 bool is_reserved = python_op_gen_internal::IsPythonReserved(function_name);
860
861 // Prefix an op with underscore if the op is listed in hidden_ops or
862 // name is reserved or it is of the exceptions in IsOpWithUnderscorePrefix.
863 // Do not add underscores to ops set to HIDDEN in ApiDef otherwise.
864 // TODO(annarev): don't prefix with underscores even if op is in hidden_ops.
865 if (is_hidden) {
866 if (!hidden_by_api_def || is_reserved ||
867 python_op_gen_internal::IsOpWithUnderscorePrefix(function_name)) {
868 function_name = strings::StrCat("_", function_name);
869 }
870 } else if (is_reserved) {
871 // When users create custom python wrappers, they may link in the
872 // default op registry by accident, and because they can't
873 // enumerate all 'hidden' symbols, this guard is to prevent
874 // instantiating a python reserved word in their wrapper.
875 continue;
876 }
877
878 strings::StrAppend(&result, GetPythonOp(op_def, *api_def, function_name));
879
880 if (!require_shapes) {
881 strings::StrAppend(&result, "_ops.RegisterShape(\"", op_def.name(),
882 "\")(None)\n");
883 }
884
885 auto added = out->Add();
886 *added = op_def;
887 RemoveNonDeprecationDescriptionsFromOpDef(added);
888 }
889
890 result.append(R"(def _InitOpDefLibrary(op_list_proto_bytes):
891 op_list = _op_def_pb2.OpList()
892 op_list.ParseFromString(op_list_proto_bytes)
893 _op_def_registry.register_op_list(op_list)
894 op_def_lib = _op_def_library.OpDefLibrary()
895 op_def_lib.add_op_list(op_list)
896 return op_def_lib
897
898
899)");
900
901 result.append("# ");
902 auto ops_text = ProtoDebugString(cleaned_ops);
903 str_util::StripTrailingWhitespace(&ops_text);
904 result.append(str_util::StringReplace(ops_text, "\n", "\n# ", true));
905 result.append("\n");
906 strings::Appendf(&result, "_op_def_lib = _InitOpDefLibrary(b\"%s\")\n",
907 str_util::CEscape(cleaned_ops.SerializeAsString()).c_str());
908 return result;
909}
