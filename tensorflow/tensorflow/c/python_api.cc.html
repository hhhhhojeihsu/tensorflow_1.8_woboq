23void AddControlInput(TF_Graph* graph, TF_Operation* op, TF_Operation* input) {
24 mutex_lock l(graph->mu);
25 graph->graph.AddControlEdge(&input->node, &op->node);
26 RecordMutation(graph, *op, "adding control input");
27}

29void SetAttr(TF_Graph* graph, TF_Operation* op, const char* attr_name,
30 TF_Buffer* attr_value_proto, TF_Status* status) {
31 AttrValue attr_val;
32 if (!attr_val.ParseFromArray(attr_value_proto->data,
33 attr_value_proto->length)) {
34 status->status =
35 tensorflow::errors::InvalidArgument("Invalid AttrValue proto");
36 return;
37 }
38
39 mutex_lock l(graph->mu);
40 op->node.AddAttr(attr_name, attr_val);
41 RecordMutation(graph, *op, "setting attribute");
42}

50void UpdateEdge(TF_Graph* graph, TF_Output new_src, TF_Input dst,
51 TF_Status* status) {
52 mutex_lock l(graph->mu);
53 tensorflow::shape_inference::InferenceContext* ic =
54 graph->refiner.GetContext(&new_src.oper->node);
55
56 if (ic->num_outputs() <= new_src.index) {
57 status->status = tensorflow::errors::OutOfRange(
58 "Cannot update edge. Output index [", new_src.index,
59 "] is greater than the number of total outputs [", ic->num_outputs(),
60 "].");
61 return;
62 }
63 tensorflow::shape_inference::ShapeHandle shape = ic->output(new_src.index);
64
65 tensorflow::shape_inference::InferenceContext* ic_dst =
66 graph->refiner.GetContext(&dst.oper->node);
67 if (ic_dst->num_inputs() <= dst.index) {
68 status->status = tensorflow::errors::OutOfRange(
69 "Cannot update edge. Input index [", dst.index,
70 "] is greater than the number of total inputs [", ic_dst->num_inputs(),
71 "].");
72 return;
73 }
74 if (!ic_dst->MergeInput(dst.index, shape)) {
75 status->status = tensorflow::errors::InvalidArgument(
76 "Cannot update edge, incompatible shapes: ", ic_dst->DebugString(shape),
77 " and ", ic_dst->DebugString(ic_dst->input(dst.index)), ".");
78 return;
79 }
80 status->status = graph->graph.UpdateEdge(&new_src.oper->node, new_src.index,
81 &dst.oper->node, dst.index);
82
83 if (status->status.ok()) {
84 // This modification only updates the destination node for
85 // the purposes of running this graph in a session. Thus, we don't
86 // record the source node as being modified.
87 RecordMutation(graph, *dst.oper, "updating input tensor");
88 }
89}

91void RemoveAllControlInputs(TF_Graph* graph, TF_Operation* op) {
92 mutex_lock l(graph->mu);
93 std::vector<const Edge*> control_edges;
94 for (const Edge* edge : op->node.in_edges()) {
95 if (!edge->IsControlEdge()) continue;
96 control_edges.push_back(edge);
97 }
98 for (const Edge* edge : control_edges) {
99 graph->graph.RemoveControlEdge(edge);
100 }
101}

113std::string ResourceHandleShapeAndType(TF_Graph* graph, TF_Output output) {
114 Node* node = &output.oper->node;
115 CppShapeInferenceResult::HandleData handle_data;
116 handle_data.set_is_set(true);
117 {
118 mutex_lock l(graph->mu);
119 tensorflow::shape_inference::InferenceContext* ic =
120 graph->refiner.GetContext(node);
121 CHECK(ic != nullptr);
122 CHECK_LT(output.index, ic->num_outputs());
123 const auto* shapes_and_types =
124 ic->output_handle_shapes_and_types(output.index);
125 if (shapes_and_types == nullptr) return "";
126
127 for (const auto& p : *shapes_and_types) {
128 auto* out_shape_and_type = handle_data.add_shape_and_type();
129 ic->ShapeHandleToProto(p.shape, out_shape_and_type->mutable_shape());
130 out_shape_and_type->set_dtype(p.dtype);
131 }
132 }
133 string result;
134 handle_data.SerializeToString(&result);
135 return result;
136}
