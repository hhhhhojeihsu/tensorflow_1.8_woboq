32// Information about a tensor.
33struct TapeTensor {
34 int64 id; // Expected to be unique in the lifetime of this process.
35 DataType dtype;
36 TensorShape shape;
37};
38
39// Represents an entry in the tape.
40template <typename BackwardFunction>
41struct OpTapeEntry {
42 string op_type;
43 std::vector<TapeTensor> output_tensor_info;
44 std::vector<int64> input_tensor_id;
45
46 // TODO(apassos) consider narrowing down this interface.
47 BackwardFunction* backward_function;
48
49 // Should be called before deleting the backward function. TODO(apassos) use
50 // unique_ptrs to ensure this happens.
51 std::function<void()> backward_function_deleter;
52};
53
54// Map from tensor_id to internally-defined operation-id of the operation which
55// produced this tensor. A value of -1 means that the tensor was directly
56// watched and not the result of any operation in the tape.
57using TensorTape = gtl::FlatMap<int64, int64>;
58
59// Map from operation-id to tape entry.
60template <typename BackwardFunction>
61using OpTape = gtl::FlatMap<int64, OpTapeEntry<BackwardFunction>>;
62
63// Operations the tape needs to perform on tensors to do backpropagation. Named
64// "vspace" because a subset of these are related to a vector space, such as
65// adding gradients, getting zeroes, etc. Currently cannot be implemented
66// without using tensorflow python code, hence left unspecified here.
67//
68// Gradient is the type returned by gradient functions. In Python TF it's either
69// Tensor or IndexedSlices or None, which here we map to nullptr. Gradients need
70// to allow their size to be computed and they need to be passable to a backward
71// function and deleted (as the backprop code creates lots of gradients the user
72// is not interested in).
73//
74// BackwardFunction needs to be a closure which stores intermediate activations
75// from the forward computation and calls a vector-jacobian product function
76// (also known as adjoint function) to compute, given downstream gradients,
77// upstream gradients.
78//
79// TODO(apassos) provide concrete template instantiations for TFE_TensorHandle
80// specialization, which is blocked by quite a few things needing to loop back
81// into python now.
82template <typename Gradient, typename BackwardFunction>
83class VSpace {
84 public:
85 virtual ~VSpace() {}
86
87 // Returns the number of elements in the gradient tensor.
88 virtual int64 NumElements(Gradient* tensor) const = 0;
89
90 // Consumes references to the tensors in the gradient_tensors list and returns
91 // a tensor with the result.
92 virtual Gradient* AggregateGradients(
93 gtl::ArraySlice<Gradient*> gradient_tensors) const = 0;
94
95 // Returns a tensor of the right shape and dtype filled with zeros.
96 virtual Gradient* Zeros(TensorShape shape, DataType dtype) const = 0;
97
98 // Returns a Tensor which is filled with ones and like the input.
99 virtual Gradient* Ones(TensorShape shape, DataType dtype) const = 0;
100
101 // Calls the passed-in backward function.
102 virtual Status CallBackwardFunction(
103 BackwardFunction* backward_function,
104 gtl::ArraySlice<Gradient*> output_gradients,
105 std::vector<Gradient*>* result) const = 0;
106
107 // Deletes the input tensor.
108 virtual void DeleteGradient(Gradient* gradient) const = 0;
109
110 // Lets this VSpace know that it can release resources held by the
111 // `backward_function`, It will not be called again.
112 // `backward_function` must not be null.
113 virtual void ReleaseBackwardFunction(
114 BackwardFunction* backward_function) const = 0;
115};

117// Traces the execution of operations, doing eager garbage collection, and
118// exporting a full trace so other code can do backpropagation. Not thread-safe.
119template <typename Gradient, typename BackwardFunction>
120class GradientTape {
121 public:
122 // If `persistent` is true, GradientTape will not eagerly delete backward
123 // functions (and hence the tensors they keep alive). Instead, everything
124 // is deleted in ~GradientTape. Persistent GradientTapes are useful when
125 // users want to compute multiple gradients over the same tape.
126 GradientTape(bool persistent) : persistent_(persistent) {}
127 ~GradientTape() {
128 for (const auto& pair : op_tape_) {
129 pair.second.backward_function_deleter();
130 }
131 }
132
133 bool ShouldRecord(gtl::ArraySlice<int64> tensor_ids);
134
135 void Watch(int64 tensor_id);
136
137 void RecordOperation(const string& op_type,
138 gtl::ArraySlice<TapeTensor> output_tensors,
139 gtl::ArraySlice<int64> input_tensor_id,
140 BackwardFunction* backward_function,
141 const std::function<void()>& backward_function_deleter);
142
143 void DeleteTrace(int64 tensor_id);
144
145 // Consumes the internal state of the tape (so cannot be called more than
146 // once) and produces the gradient of the target tensors with respect to the
147 // source tensors. The output gradients are used if not empty and not
148 // null. The result is populated with one tensor per target element.
149 Status ComputeGradient(const VSpace<Gradient, BackwardFunction>& vspace,
150 gtl::ArraySlice<int64> target_tensor_ids,
151 gtl::ArraySlice<int64> source_tensor_id,
152 gtl::ArraySlice<Gradient*> output_gradients,
153 std::vector<Gradient*>* result);
154
155 bool IsPersistent() const { return persistent_; }
156
157 private:
158 TensorTape tensor_tape_;
159 OpTape<BackwardFunction> op_tape_;
160 int64 next_op_id_{0};
161
162 // Map from tensor id to number of remaining usages (i.e. how many entries in
163 // the tape refer to it); to aid in tape garbage collection.
164 gtl::FlatMap<int64, int64> tensor_usage_;
165
166 // If false, all activations are deleted in the first call to ComputeGradient.
167 // Else, only when this is destructed.
168 bool persistent_;
169};

173template <typename Gradient, typename BackwardFunction>
174bool GradientTape<Gradient, BackwardFunction>::ShouldRecord(
175 gtl::ArraySlice<int64> tensor_ids) {
176 for (int64 i : tensor_ids) {
177 if (tensor_tape_.find(i) != tensor_tape_.end()) {
178 return true;
179 }
180 }
181 return false;
182}
183
184template <typename Gradient, typename BackwardFunction>
185void GradientTape<Gradient, BackwardFunction>::Watch(int64 tensor_id) {
186 tensor_tape_.emplace(tensor_id, -1);
187}
188
189template <typename Gradient, typename BackwardFunction>
190void GradientTape<Gradient, BackwardFunction>::RecordOperation(
191 const string& op_type, gtl::ArraySlice<TapeTensor> output_tensors,
192 gtl::ArraySlice<int64> input_tensor_id, BackwardFunction* backward_function,
193 const std::function<void()>& backward_function_deleter) {
194 if (!ShouldRecord(input_tensor_id)) {
195 backward_function_deleter();
196 return;
197 }
198 std::vector<int64> ids;
199 ids.reserve(input_tensor_id.size());
200 for (int64 i : input_tensor_id) {
201 tensor_usage_[i]++;
202 ids.push_back(i);
203 }
204 const int64 op_id = next_op_id_++;
205 std::vector<TapeTensor> tensors;
206 tensors.reserve(output_tensors.size());
207 for (const TapeTensor& o : output_tensors) {
208 // Note: the tensor can have already been watched and hence be in the tape,
209 // so we cannot check that we're inserting it here.
210 tensor_tape_[o.id] = op_id;
211 tensor_usage_[o.id] = 1;
212 tensors.push_back(o);
213 }
214 op_tape_[op_id] = OpTapeEntry<BackwardFunction>{
215 op_type, tensors, ids, backward_function, backward_function_deleter};
216}

218template <typename Gradient, typename BackwardFunction>
219void GradientTape<Gradient, BackwardFunction>::DeleteTrace(int64 tensor_id) {
220 auto it = tensor_usage_.find(tensor_id);
221 if (it == tensor_usage_.end()) {
222 return;
223 }
224 it->second--;
225 if (it->second != 0) {
226 return;
227 }
228 tensor_usage_.erase(it);
229 auto tensor_op_it = tensor_tape_.find(tensor_id);
230 if (tensor_op_it == tensor_tape_.end()) {
231 return;
232 }
233 const int64 op_id = tensor_op_it->second;
234 if (op_id == -1) {
235 // Do not delete watched tensors.
236 return;
237 }
238 tensor_tape_.erase(tensor_op_it);
239 auto op_it = op_tape_.find(op_id);
240 CHECK(op_it != op_tape_.end());
241 for (const auto& output : op_it->second.output_tensor_info) {
242 if (tensor_usage_.find(output.id) != tensor_usage_.end()) {
243 // Found a usage for an output, so cannot delete the op.
244 return;
245 }
246 }
247 for (int64 id : op_it->second.input_tensor_id) {
248 DeleteTrace(id);
249 }
250 op_it->second.backward_function_deleter();
251 op_tape_.erase(op_it);
252}

254// Terminology:
255//
256// - op: a possibly composite operation, which has an entry in the tape
257// - target: dy in dx/dy
258// - source: dx in dx/dy
259// - tensor: one of the many inputs or outputs of an operation
260//
261// Below here we do the gradient algorithm. It works as follows:
262//
263// First we filter the tape to just the subset of operations we want to
264// differentiate. In the process of doing so we count how many times each Tensor
265// is used as an input to an op (so we know when we're done computing gradients
266// for that Tensor). We also count, for each tape entry, how many of its output
267// Tensors need gradients to be computed (Tensors which are not used do not need
268// any gradients to be computed).
269//
270// Finally, we start a backprop stack with a set of tape entries for which we
271// have all gradients available. This set usually is a subset of the set of
272// targets (not all since targets which have outputs in the tape will not have
273// gradients available initially).
274//
275// Then we repeatedly pop an entry from the stack, run its backprop, and update
276// the gradients of its inputs. Once we have computed all gradients for a single
277// input we can mark this input as done, and this can trigger adding an entry to
278// the stack if all outputs of that entry are now done.
279//
280// When the stack is empty we have gradients for all tensors we're interested
281// in.
282
283namespace {
284
285template <typename BackwardFunction>
286struct BackpropInitialState {
287 OpTape<BackwardFunction> op_tape;
288
289 // Map from tensor ID to how many references still exist for this tensor in
290 // the tape.
291 gtl::FlatMap<int64, int64> tensor_usage_counts;
292
293 // Maps from op ID to how many output tensors of this op still need to have
294 // their gradients computed.
295 gtl::FlatMap<int64, int64> op_missing_tensor;
296};
297
298// If `persistent_tape` is true, op_tape is not changed and none of the
299// backwards functions are deleted.
300// If `persistent_tape` is false, op_tape is cleared and backwards functions
301// not needed for gradient computation are deleted. Backwards functions that
302// are needed, are copied and returned in BackpropInitialState.
303template <typename BackwardFunction>
304BackpropInitialState<BackwardFunction> PrepareBackprop(
305 gtl::ArraySlice<int64> target, const TensorTape& tensor_tape,
306 OpTape<BackwardFunction>* op_tape, const gtl::FlatSet<int64>& sources_set,
307 bool persistent_tape) {
308 std::vector<int64> tensor_stack;
309 tensor_stack.reserve(target.size());
310 for (auto t : target) {
311 tensor_stack.push_back(t);
312 }
313 BackpropInitialState<BackwardFunction> result;
314 while (!tensor_stack.empty()) {
315 int64 tensor_id = tensor_stack.back();
316 tensor_stack.pop_back();
317 auto op_id_it = tensor_tape.find(tensor_id);
318 if (op_id_it == tensor_tape.end()) {
319 continue;
320 }
321 int64 op_id = op_id_it->second;
322 auto op_it = op_tape->find(op_id);
323 auto result_op_it = result.op_tape.find(op_id);
324 if (op_id == -1 || op_it == op_tape->end() ||
325 result_op_it != result.op_tape.end()) {
326 continue;
327 }
328 CHECK(result.op_tape.emplace(op_id, op_it->second).second);
329 for (auto it : op_it->second.input_tensor_id) {
330 auto count_it = result.tensor_usage_counts.find(it);
331 if (count_it != result.tensor_usage_counts.end()) {
332 count_it->second++;
333 } else {
334 result.tensor_usage_counts[it] = 1;
335 if (sources_set.find(it) == sources_set.end() &&
336 tensor_tape.find(it) != tensor_tape.end()) {
337 tensor_stack.push_back(it);
338 }
339 }
340 }
341 if (!persistent_tape) {
342 op_tape->erase(op_it);
343 }
344 }
345 for (auto& pair : result.tensor_usage_counts) {
346 auto it = tensor_tape.find(pair.first);
347 if (it != tensor_tape.end() && it->second != -1) {
348 result.op_missing_tensor[it->second] += 1;
349 }
350 }
351 if (!persistent_tape) {
352 // Call destructors for all unneeded gradient functions and
353 // clear the op_tape. We can clear the tape because ownership of
354 // backward functions that will be used for gradient computation
355 // has been transferred to `result`.
356 for (const auto& op_pair : *op_tape) {
357 op_pair.second.backward_function_deleter();
358 }
359 op_tape->clear();
360 }
361 return result;
362}
363
364template <typename BackwardFunction>
365std::vector<int64> InitialStack(
366 const OpTape<BackwardFunction>& op_tape,
367 const gtl::FlatMap<int64, int64>& op_missing_tensor) {
368 std::vector<int64> result;
369 for (auto& op_entry : op_tape) {
370 if (op_missing_tensor.find(op_entry.first) == op_missing_tensor.end()) {
371 result.push_back(op_entry.first);
372 }
373 }
374 return result;
375}

377template <typename Gradient, typename BackwardFunction>
378Status InitialGradients(const VSpace<Gradient, BackwardFunction>& vspace,
379 gtl::ArraySlice<int64> target_tensor_ids,
380 gtl::ArraySlice<Gradient*> output_gradients,
381 const TensorTape& tensor_tape,
382 const OpTape<BackwardFunction>& op_tape,
383 const gtl::FlatMap<int64, int64>& tensor_usage_counts,
384 gtl::FlatMap<int64, std::vector<Gradient*>>* result) {
385 for (int i = 0; i < target_tensor_ids.size(); ++i) {
386 const int64 id = target_tensor_ids[i];
387 if (tensor_usage_counts.find(id) != tensor_usage_counts.end()) {
388 if (!output_gradients.empty() && output_gradients[i] != nullptr) {
389 // TODO(apassos) figure out how to print debugging information here.
390 return errors::InvalidArgument(
391 "A gradient was provided for a tensor which is used as part of the "
392 "computation.");
393 }
394 } else {
395 if (output_gradients.empty() || output_gradients[i] == nullptr) {
396 auto tensor_it = tensor_tape.find(id);
397 if (tensor_it != tensor_tape.end() && tensor_it->second != -1) {
398 auto op_it = op_tape.find(tensor_it->second);
399 if (op_it == op_tape.end()) {
400 return errors::Internal(
401 "Internal state of the gradient tape is invalid: "
402 "failed to find operation producing a tensor");
403 }
404 bool found = false;
405 for (int j = 0; j < op_it->second.output_tensor_info.size(); ++j) {
406 if (op_it->second.output_tensor_info[j].id == id) {
407 found = true;
408 (*result)[id].push_back(
409 vspace.Ones(op_it->second.output_tensor_info[j].shape,
410 op_it->second.output_tensor_info[j].dtype));
411 break;
412 }
413 }
414 if (!found) {
415 return errors::Internal(
416 "Internal state of the gradient tape is invalid: "
417 "none of operations outputs match expected tensor");
418 }
419 } else {
420 // No record of the target tensor found on the tape, so no gradient
421 // needs to be computed from it. Do nothing.
422 }
423 } else {
424 (*result)[id].push_back(output_gradients[i]);
425 }
426 }
427 }
428 return Status::OK();
429}

433// If over kMinAggregateCount gradients are accumulated and the total
434// memory consumption is over kMinAggregateBytes, do an early aggregation
435// so as to release the gradient tensor to save memory.
436constexpr int kMinAggregateCount = 4;
437constexpr int kMinAggregateBytes = 128 * 1024 * 1024;
438
439template <typename Gradient, typename BackwardFunction>
440Status GradientTape<Gradient, BackwardFunction>::ComputeGradient(
441 const VSpace<Gradient, BackwardFunction>& vspace,
442 gtl::ArraySlice<int64> target_tensor_ids,
443 gtl::ArraySlice<int64> source_tensor_ids,
444 gtl::ArraySlice<Gradient*> output_gradients,
445 std::vector<Gradient*>* result) {
446 gtl::FlatSet<int64> sources_set(source_tensor_ids.begin(),
447 source_tensor_ids.end());
448 BackpropInitialState<BackwardFunction> state = PrepareBackprop(
449 target_tensor_ids, tensor_tape_, &op_tape_, sources_set, persistent_);
450 std::vector<int64> op_stack =
451 InitialStack(state.op_tape, state.op_missing_tensor);
452 gtl::FlatMap<int64, std::vector<Gradient*>> gradients;
453 Status s = InitialGradients(vspace, target_tensor_ids, output_gradients,
454 tensor_tape_, state.op_tape,
455 state.tensor_usage_counts, &gradients);
456 auto cleanup = [this, &state]() {
457 if (!persistent_) {
458 // Release all backprop functions
459 for (const auto& pair : state.op_tape) {
460 pair.second.backward_function_deleter();
461 }
462 }
463 };
464 if (!s.ok()) {
465 cleanup();
466 return s;
467 }
468 gtl::FlatMap<int64, int64> gradients_size;
469 // TODO(apassos) multiple threads could be dequeuing from op_stack at the same
470 // time, for better CPU backprop performance.
471 VLOG(1) << "Initial stack:";
472 if (VLOG_IS_ON(1)) {
473 for (auto t : op_stack) {
474 VLOG(1) << " " << t;
475 }
476 }
477 gtl::FlatMap<string, gtl::FlatSet<int>> functions_accept_none_for_indices({
478 {"SoftmaxCrossEntropyWithLogits", {1}},
479 {"FusedBatchNorm", {1, 2, 3, 4}},
480 });
481 while (!op_stack.empty()) {
482 const int64 op = op_stack.back();
483 VLOG(1) << "Popped " << op;
484 op_stack.pop_back();
485 auto op_it = state.op_tape.find(op);
486 if (op_it == state.op_tape.end()) {
487 // It is possible for ops to end up on the stack if they are unrelated to
488 // the target; we should just skip them.
489 continue;
490 }
491 auto trace = std::move(op_it->second);
492 state.op_tape.erase(op_it);
493 std::vector<Gradient*> out_gradients;
494 out_gradients.reserve(trace.output_tensor_info.size());
495 bool any_gradient_nonzero = false;
496 for (int i = 0; i < trace.output_tensor_info.size(); ++i) {
497 const int64 id = trace.output_tensor_info[i].id;
498 auto grad_it = gradients.find(id);
499 if (grad_it == gradients.end()) {
500 auto func_name_it =
501 functions_accept_none_for_indices.find(trace.op_type);
502 if (func_name_it != functions_accept_none_for_indices.end() &&
503 func_name_it->second.find(i) != func_name_it->second.end()) {
504 out_gradients.push_back(nullptr);
505 } else {
506 out_gradients.push_back(
507 vspace.Zeros(trace.output_tensor_info[i].shape,
508 trace.output_tensor_info[i].dtype));
509 }
510 } else {
511 any_gradient_nonzero = true;
512 out_gradients.push_back(vspace.AggregateGradients(grad_it->second));
513 if (sources_set.find(grad_it->first) == sources_set.end()) {
514 gradients.erase(grad_it);
515 }
516 }
517 }
518 std::vector<Gradient*> in_gradients;
519 if (any_gradient_nonzero) {
520 Status s = vspace.CallBackwardFunction(trace.backward_function,
521 out_gradients, &in_gradients);
522 if (!persistent_) {
523 vspace.ReleaseBackwardFunction(trace.backward_function);
524 }
525 if (!s.ok()) {
526 cleanup();
527 return s;
528 }
529 } else {
530 in_gradients.resize(trace.input_tensor_id.size());
531 if (!persistent_) {
532 vspace.ReleaseBackwardFunction(trace.backward_function);
533 }
534 for (Gradient* grad : out_gradients) {
535 if (grad != nullptr) {
536 vspace.DeleteGradient(grad);
537 }
538 }
539 }
540 VLOG(1) << "Got " << in_gradients.size() << " in_gradients for "
541 << trace.input_tensor_id.size() << " sources";
542 for (int i = 0; i < in_gradients.size(); ++i) {
543 const int64 id = trace.input_tensor_id[i];
544 if (in_gradients[i] != nullptr) {
545 auto& unaggregated_grads = gradients[id];
546 unaggregated_grads.push_back(in_gradients[i]);
547 if (unaggregated_grads.size() > kMinAggregateCount) {
548 auto size_it = gradients_size.find(id);
549 int64 size;
550 if (size_it == gradients_size.end()) {
551 size = vspace.NumElements(unaggregated_grads[0]);
552 gradients_size.emplace(id, size);
553 } else {
554 size = size_it->second;
555 }
556 if (unaggregated_grads.size() * size * 4 > kMinAggregateBytes) {
557 Gradient* grad = vspace.AggregateGradients(unaggregated_grads);
558 unaggregated_grads.clear();
559 unaggregated_grads.push_back(grad);
560 }
561 }
562 }
563 auto usage_count_it = state.tensor_usage_counts.find(id);
564 if (usage_count_it == state.tensor_usage_counts.end()) {
565 VLOG(1) << "Tensor " << id << " not used";
566 continue;
567 }
568 usage_count_it->second--;
569 if (usage_count_it->second > 0) {
570 VLOG(1) << "Tensor " << id << " usage count " << usage_count_it->second;
571 continue;
572 }
573 auto tape_it = tensor_tape_.find(id);
574 if (tape_it == tensor_tape_.end()) {
575 VLOG(1) << "Tensor " << id
576 << " has no associated op. Deleting gradient";
577 auto grad_it = gradients.find(id);
578 if (grad_it != gradients.end()) {
579 for (auto g : grad_it->second) {
580 vspace.DeleteGradient(g);
581 }
582 gradients.erase(grad_it);
583 }
584 continue;
585 }
586 const int64 op_id = tape_it->second;
587 if (op_id == -1) {
588 VLOG(1) << "Tensor " << id << " is source";
589 continue;
590 }
591 auto missing_it = state.op_missing_tensor.find(op_id);
592 if (missing_it != state.op_missing_tensor.end()) {
593 missing_it->second--;
594 VLOG(1) << "Op " << op_id << " missing " << missing_it->second
595 << " output gradients";
596 if (missing_it->second == 0) {
597 op_stack.push_back(op_id);
598 }
599 }
600 }
601 }
602 CHECK(state.op_tape.empty());
603 result->reserve(source_tensor_ids.size());
604 gtl::FlatSet<int64> used_gradient_ids(source_tensor_ids.size());
605 for (auto is : source_tensor_ids) {
606 auto grad_it = gradients.find(is);
607 if (grad_it == gradients.end()) {
608 result->push_back(nullptr);
609 } else {
610 if (grad_it->second.size() > 1) {
611 Gradient* grad = vspace.AggregateGradients(grad_it->second);
612 grad_it->second.clear();
613 grad_it->second.push_back(grad);
614 }
615 result->push_back(grad_it->second[0]);
616 used_gradient_ids.insert(is);
617 }
618 }
619 VLOG(1) << "Final gradients size: "
620 << gradients.size() - used_gradient_ids.size();
621 for (auto grad_pair : gradients) {
622 if (used_gradient_ids.find(grad_pair.first) == used_gradient_ids.end()) {
623 for (const auto& g : grad_pair.second) {
624 vspace.DeleteGradient(g);
625 }
626 }
627 }
628 return Status::OK();
629}
