49class FileSystemRegistryImpl : public FileSystemRegistry {
50 public:
51 Status Register(const string& scheme, Factory factory) override;
52 FileSystem* Lookup(const string& scheme) override;
53 Status GetRegisteredFileSystemSchemes(std::vector<string>* schemes) override;
54
55 private:
56 mutable mutex mu_;
57 mutable std::unordered_map<string, std::unique_ptr<FileSystem>> registry_
58 GUARDED_BY(mu_);
59};
60
61Status FileSystemRegistryImpl::Register(const string& scheme,
62 FileSystemRegistry::Factory factory) {
63 mutex_lock lock(mu_);
64 if (!registry_.emplace(string(scheme), std::unique_ptr<FileSystem>(factory()))
65 .second) {
66 return errors::AlreadyExists("File factory for ", scheme,
67 " already registered");
68 }
69 return Status::OK();
70}
71
72FileSystem* FileSystemRegistryImpl::Lookup(const string& scheme) {
73 mutex_lock lock(mu_);
74 const auto found = registry_.find(scheme);
75 if (found == registry_.end()) {
76 return nullptr;
77 }
78 return found->second.get();
79}
80
81Status FileSystemRegistryImpl::GetRegisteredFileSystemSchemes(
82 std::vector<string>* schemes) {
83 mutex_lock lock(mu_);
84 for (const auto& e : registry_) {
85 schemes->push_back(e.first);
86 }
87 return Status::OK();
88}

92Status Env::GetFileSystemForFile(const string& fname, FileSystem** result) {
93 StringPiece scheme, host, path;
94 io::ParseURI(fname, &scheme, &host, &path);
95 FileSystem* file_system = file_system_registry_->Lookup(scheme.ToString());
96 if (!file_system) {
97 if (scheme.empty()) {
98 scheme = "[local]";
99 }
100
101 return errors::Unimplemented("File system scheme '", scheme,
102 "' not implemented (file: '", fname, "')");
103 }
104 *result = file_system;
105 return Status::OK();
106}

117Status Env::FlushFileSystemCaches() {
118 std::vector<string> schemes;
119 TF_RETURN_IF_ERROR(GetRegisteredFileSystemSchemes(&schemes));
120 for (const string& scheme : schemes) {
121 FileSystem* fs = nullptr;
122 TF_RETURN_IF_ERROR(
123 GetFileSystemForFile(io::CreateURI(scheme, "", ""), &fs));
124 fs->FlushCaches();
125 }
126 return Status::OK();
127}

163bool Env::FilesExist(const std::vector<string>& files,
164 std::vector<Status>* status) {
165 std::unordered_map<string, std::vector<string>> files_per_fs;
166 for (const auto& file : files) {
167 StringPiece scheme, host, path;
168 io::ParseURI(file, &scheme, &host, &path);
169 files_per_fs[scheme.ToString()].push_back(file);
170 }
171
172 std::unordered_map<string, Status> per_file_status;
173 bool result = true;
174 for (auto itr : files_per_fs) {
175 FileSystem* file_system = file_system_registry_->Lookup(itr.first);
176 bool fs_result;
177 std::vector<Status> local_status;
178 std::vector<Status>* fs_status = status ? &local_status : nullptr;
179 if (!file_system) {
180 fs_result = false;
181 if (fs_status) {
182 Status s = errors::Unimplemented("File system scheme '", itr.first,
183 "' not implemented");
184 local_status.resize(itr.second.size(), s);
185 }
186 } else {
187 fs_result = file_system->FilesExist(itr.second, fs_status);
188 }
189 if (fs_status) {
190 result &= fs_result;
191 for (int i = 0; i < itr.second.size(); ++i) {
192 per_file_status[itr.second[i]] = fs_status->at(i);
193 }
194 } else if (!fs_result) {
195 // Return early
196 return false;
197 }
198 }
199
200 if (status) {
201 for (const auto& file : files) {
202 status->push_back(per_file_status[file]);
203 }
204 }
205
206 return result;
207}

294string Env::GetExecutablePath() {
295 char exe_path[PATH_MAX] = {0};
296#ifdef __APPLE__
297 uint32_t buffer_size(0U);
298 _NSGetExecutablePath(nullptr, &buffer_size);
299 char unresolved_path[buffer_size];
300 _NSGetExecutablePath(unresolved_path, &buffer_size);
301 CHECK(realpath(unresolved_path, exe_path));
302#elif defined(__FreeBSD__)
303 int mib[4] = {CTL_KERN, KERN_PROC, KERN_PROC_PATHNAME, -1};
304 size_t exe_path_size = PATH_MAX;
305
306 if (sysctl(mib, 4, exe_path, &exe_path_size, NULL, 0) != 0) {
307 // Resolution of path failed
308 return "";
309 }
310#elif defined(PLATFORM_WINDOWS)
311 HMODULE hModule = GetModuleHandleW(NULL);
312 WCHAR wc_file_path[MAX_PATH] = {0};
313 GetModuleFileNameW(hModule, wc_file_path, MAX_PATH);
314 string file_path = WindowsFileSystem::WideCharToUtf8(wc_file_path);
315 std::copy(file_path.begin(), file_path.end(), exe_path);
316#else
317 CHECK_NE(-1, readlink("/proc/self/exe", exe_path, sizeof(exe_path) - 1));
318#endif
319 // Make sure it's null-terminated:
320 exe_path[sizeof(exe_path) - 1] = 0;
321
322 return exe_path;
323}

325bool Env::LocalTempFilename(string* filename) {
326 std::vector<string> dirs;
327 GetLocalTempDirectories(&dirs);
328
329 // Try each directory, as they might be full, have inappropriate
330 // permissions or have different problems at times.
331 for (const string& dir : dirs) {
332 *filename = io::JoinPath(dir, "tempfile-");
333 if (CreateUniqueFileName(filename, "")) {
334 return true;
335 }
336 }
337 return false;
338}

340bool Env::CreateUniqueFileName(string* prefix, const string& suffix) {
341#ifdef __APPLE__
342 uint64_t tid64;
343 pthread_threadid_np(nullptr, &tid64);
344 int32 tid = static_cast<int32>(tid64);
345 int32 pid = static_cast<int32>(getpid());
346#elif defined(__FreeBSD__)
347 // Has to be casted to long first, else this error appears:
348 // static_cast from 'pthread_t' (aka 'pthread *') to 'int32' (aka 'int')
349 // is not allowed
350 int32 tid = static_cast<int32>(static_cast<int64>(pthread_self()));
351 int32 pid = static_cast<int32>(getpid());
352#elif defined(PLATFORM_WINDOWS)
353 int32 tid = static_cast<int32>(GetCurrentThreadId());
354 int32 pid = static_cast<int32>(GetCurrentProcessId());
355#else
356 int32 tid = static_cast<int32>(pthread_self());
357 int32 pid = static_cast<int32>(getpid());
358#endif
359 uint64 now_microsec = NowMicros();
360
361 *prefix += strings::Printf("%s-%x-%d-%llx", port::Hostname().c_str(), tid,
362 pid, now_microsec);
363
364 if (!suffix.empty()) {
365 *prefix += suffix;
366 }
367 if (FileExists(*prefix).ok()) {
368 prefix->clear();
369 return false;
370 } else {
371 return true;
372 }
373}

379Status ReadFileToString(Env* env, const string& fname, string* data) {
380 uint64 file_size;
381 Status s = env->GetFileSize(fname, &file_size);
382 if (!s.ok()) {
383 return s;
384 }
385 std::unique_ptr<RandomAccessFile> file;
386 s = env->NewRandomAccessFile(fname, &file);
387 if (!s.ok()) {
388 return s;
389 }
390 gtl::STLStringResizeUninitialized(data, file_size);
391 char* p = gtl::string_as_array(data);
392 StringPiece result;
393 s = file->Read(0, file_size, &result, p);
394 if (!s.ok()) {
395 data->clear();
396 } else if (result.size() != file_size) {
397 s = errors::Aborted("File ", fname, " changed while reading: ", file_size,
398 " vs. ", result.size());
399 data->clear();
400 } else if (result.data() == p) {
401 // Data is already in the correct location
402 } else {
403 memmove(p, result.data(), result.size());
404 }
405 return s;
406}

408Status WriteStringToFile(Env* env, const string& fname,
409 const StringPiece& data) {
410 std::unique_ptr<WritableFile> file;
411 Status s = env->NewWritableFile(fname, &file);
412 if (!s.ok()) {
413 return s;
414 }
415 s = file->Append(data);
416 if (s.ok()) {
417 s = file->Close();
418 }
419 return s;
420}

422Status FileSystemCopyFile(FileSystem* src_fs, const string& src,
423 FileSystem* target_fs, const string& target) {
424 std::unique_ptr<RandomAccessFile> src_file;
425 TF_RETURN_IF_ERROR(src_fs->NewRandomAccessFile(src, &src_file));
426
427 std::unique_ptr<WritableFile> target_file;
428 TF_RETURN_IF_ERROR(target_fs->NewWritableFile(target, &target_file));
429
430 uint64 offset = 0;
431 std::unique_ptr<char[]> scratch(new char[kCopyFileBufferSize]);
432 Status s = Status::OK();
433 while (s.ok()) {
434 StringPiece result;
435 s = src_file->Read(offset, kCopyFileBufferSize, &result, scratch.get());
436 if (!(s.ok() || s.code() == error::OUT_OF_RANGE)) {
437 return s;
438 }
439 TF_RETURN_IF_ERROR(target_file->Append(result));
440 offset += result.size();
441 }
442 return target_file->Close();
443}

445// A ZeroCopyInputStream on a RandomAccessFile.
446namespace {
447class FileStream : public ::tensorflow::protobuf::io::ZeroCopyInputStream {
448 public:
449 explicit FileStream(RandomAccessFile* file) : file_(file), pos_(0) {}
450
451 void BackUp(int count) override { pos_ -= count; }
452 bool Skip(int count) override {
453 pos_ += count;
454 return true;
455 }
456 protobuf_int64 ByteCount() const override { return pos_; }
457 Status status() const { return status_; }
458
459 bool Next(const void** data, int* size) override {
460 StringPiece result;
461 Status s = file_->Read(pos_, kBufSize, &result, scratch_);
462 if (result.empty()) {
463 status_ = s;
464 return false;
465 }
466 pos_ += result.size();
467 *data = result.data();
468 *size = result.size();
469 return true;
470 }
471
472 private:
473 static const int kBufSize = 512 << 10;
474
475 RandomAccessFile* file_;
476 int64 pos_;
477 Status status_;
478 char scratch_[kBufSize];
479};

485 string serialized;
486 proto.AppendToString(&serialized);
487 return WriteStringToFile(env, fname, serialized);
488}

490Status ReadBinaryProto(Env* env, const string& fname,
491 ::tensorflow::protobuf::MessageLite* proto) {
492 std::unique_ptr<RandomAccessFile> file;
493 TF_RETURN_IF_ERROR(env->NewRandomAccessFile(fname, &file));
494 std::unique_ptr<FileStream> stream(new FileStream(file.get()));
495
496 // TODO(jiayq): the following coded stream is for debugging purposes to allow
497 // one to parse arbitrarily large messages for MessageLite. One most likely
498 // doesn't want to put protobufs larger than 64MB on Android, so we should
499 // eventually remove this and quit loud when a large protobuf is passed in.
500 ::tensorflow::protobuf::io::CodedInputStream coded_stream(stream.get());
501 // Total bytes hard limit / warning limit are set to 1GB and 512MB
502 // respectively.
503 coded_stream.SetTotalBytesLimit(1024LL << 20, 512LL << 20);
504
505 if (!proto->ParseFromCodedStream(&coded_stream)) {
506 TF_RETURN_IF_ERROR(stream->status());
507 return errors::DataLoss("Can't parse ", fname, " as binary proto");
508 }
509 return Status::OK();
510}

512Status WriteTextProto(Env* env, const string& fname,
513 const ::tensorflow::protobuf::Message& proto) {
514#if !defined(TENSORFLOW_LITE_PROTOS)
515 string serialized;
516 if (!::tensorflow::protobuf::TextFormat::PrintToString(proto, &serialized)) {
517 return errors::FailedPrecondition("Unable to convert proto to text.");
518 }
519 return WriteStringToFile(env, fname, serialized);
520#else
521 return errors::Unimplemented("Can't write text protos with protolite.");
522#endif
523}

525Status ReadTextProto(Env* env, const string& fname,
526 ::tensorflow::protobuf::Message* proto) {
527#if !defined(TENSORFLOW_LITE_PROTOS)
528 std::unique_ptr<RandomAccessFile> file;
529 TF_RETURN_IF_ERROR(env->NewRandomAccessFile(fname, &file));
530 std::unique_ptr<FileStream> stream(new FileStream(file.get()));
531
532 if (!::tensorflow::protobuf::TextFormat::Parse(stream.get(), proto)) {
533 TF_RETURN_IF_ERROR(stream->status());
534 return errors::DataLoss("Can't parse ", fname, " as text proto");
535 }
536 return Status::OK();
537#else
538 return errors::Unimplemented("Can't parse text protos with protolite.");
539#endif
540}
