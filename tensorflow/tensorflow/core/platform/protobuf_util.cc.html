16#include "tensorflow/core/platform/protobuf.h"

18namespace tensorflow {
19
20bool ParseProtoUnlimited(protobuf::MessageLite* proto,
21 const string& serialized) {
22 return ParseProtoUnlimited(proto, serialized.data(), serialized.size());
23}
24
25bool ParseProtoUnlimited(protobuf::MessageLite* proto, const void* serialized,
26 size_t size) {
27 protobuf::io::CodedInputStream coded_stream(
28 reinterpret_cast<const uint8*>(serialized), size);
29 coded_stream.SetTotalBytesLimit(INT_MAX, INT_MAX);
30 return proto->ParseFromCodedStream(&coded_stream);
31}
