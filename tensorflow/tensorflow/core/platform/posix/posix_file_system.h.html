24class PosixFileSystem : public FileSystem {
25 public:
26 PosixFileSystem() {}

28 ~PosixFileSystem() {}
29
30 Status NewRandomAccessFile(
31 const string& filename,
32 std::unique_ptr<RandomAccessFile>* result) override;
33
34 Status NewWritableFile(const string& fname,
35 std::unique_ptr<WritableFile>* result) override;
36
37 Status NewAppendableFile(const string& fname,
38 std::unique_ptr<WritableFile>* result) override;
39
40 Status NewReadOnlyMemoryRegionFromFile(
41 const string& filename,
42 std::unique_ptr<ReadOnlyMemoryRegion>* result) override;
43
44 Status FileExists(const string& fname) override;
45
46 Status GetChildren(const string& dir, std::vector<string>* result) override;
47
48 Status Stat(const string& fname, FileStatistics* stats) override;
49
50 Status GetMatchingPaths(const string& pattern,
51 std::vector<string>* results) override;
52
53 Status DeleteFile(const string& fname) override;
54
55 Status CreateDir(const string& name) override;
56
57 Status DeleteDir(const string& name) override;
58
59 Status GetFileSize(const string& fname, uint64* size) override;
60
61 Status RenameFile(const string& src, const string& target) override;
62
63 Status CopyFile(const string& src, const string& target) override;
64};
65
66Status IOError(const string& context, int err_number);
67
68class LocalPosixFileSystem : public PosixFileSystem {
69 public:
70 string TranslateName(const string& name) const override {
71 StringPiece scheme, host, path;
72 io::ParseURI(name, &scheme, &host, &path);
73 return path.ToString();
74 }
75};
