28// Comparator for two nodes. This is used in order to get a stable ording.
29using NodeComparator = std::function<bool(const Node*, const Node*)>;
30
31// Compares two node based on their ids.
32struct NodeComparatorID {
33 bool operator()(const Node* n1, const Node* n2) const {
34 return n1->id() < n2->id();
35 }
36};
37
38// Compare two nodes based on their names.
39struct NodeComparatorName {
40 bool operator()(const Node* n1, const Node* n2) const {
41 return n1->name() < n2->name();
42 }
43};
44
45// Perform a depth-first-search on g starting at the source node.
46// If enter is not empty, calls enter(n) before visiting any children of n.
47// If leave is not empty, calls leave(n) after visiting all children of n.
48// If stable_comparator is set, a stable ordering of visit is achieved by
49// sorting a node's neighbors first before visiting them.
50extern void DFS(const Graph& g, const std::function<void(Node*)>& enter,
51 const std::function<void(Node*)>& leave,
52 const NodeComparator& stable_comparator = {});
53
54// Perform a reverse depth-first-search on g starting at the sink node.
55// If enter is not empty, calls enter(n) before visiting any parents of n.
56// If leave is not empty, calls leave(n) after visiting all parents of n.
57// If stable_comparator is set, a stable ordering of visit is achieved by
58// sorting a node's neighbors first before visiting them.
59extern void ReverseDFS(const Graph& g, const std::function<void(Node*)>& enter,
60 const std::function<void(Node*)>& leave,
61 const NodeComparator& stable_comparator = {});
62
63// Perform a reverse depth-first-search on g starting at the 'start' nodes.
64// If enter is not empty, calls enter(n) before visiting any parents of n.
65// If leave is not empty, calls leave(n) after visiting all parents of n.
66// If stable_comparator is set, a stable ordering of visit is achieved by
67// sorting a node's neighbors first before visiting them.
68extern void ReverseDFSFrom(const Graph& g, gtl::ArraySlice<Node*> start,
69 const std::function<void(Node*)>& enter,
70 const std::function<void(Node*)>& leave,
71 const NodeComparator& stable_comparator = {});
72extern void ReverseDFSFrom(const Graph& g, gtl::ArraySlice<const Node*> start,
73 const std::function<void(const Node*)>& enter,
74 const std::function<void(const Node*)>& leave,
75 const NodeComparator& stable_comparator = {});
76
77// Stores in *order the post-order numbering of all nodes
78// in graph found via a depth first search starting at the source node.
79//
80// Note that this is equivalent to reverse topological sorting when the
81// graph does not have cycles.
82//
83// If stable_comparator is set, a stable ordering of visit is achieved by
84// sorting a node's neighbors first before visiting them.
85//
86// REQUIRES: order is not NULL.
87void GetPostOrder(const Graph& g, std::vector<Node*>* order,
88 const NodeComparator& stable_comparator = {});
89
90// Stores in *order the reverse post-order numbering of all nodes
91// If stable_comparator is set, a stable ordering of visit is achieved by
92// sorting a node's neighbors first before visiting them.
93void GetReversePostOrder(const Graph& g, std::vector<Node*>* order,
94 const NodeComparator& stable_comparator = {});
95
96// Prune nodes in "g" that are not in some path from the source node
97// to any node in 'nodes'. Returns true if changes were made to the graph.
98// Does not fix up source and sink edges.
99bool PruneForReverseReachability(Graph* g,
100 std::unordered_set<const Node*> nodes);
101
102// Connect all nodes with no incoming edges to source.
103// Connect all nodes with no outgoing edges to sink.
104//
105// Returns true if and only if 'g' is mutated.
106bool FixupSourceAndSinkEdges(Graph* g);
