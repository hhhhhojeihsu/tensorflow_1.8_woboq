34class OpLevelCostEstimator {
35 public:
36 OpLevelCostEstimator();
37 virtual ~OpLevelCostEstimator() {}
38
39 virtual Costs PredictCosts(const OpContext& op_context) const;
40
41 // Basic device performance info, sufficient for roofline estimate.
42 struct DeviceInfo {
43 double gigaops; // Billions of operations executed per second.
44 double gb_per_sec; // Bandwidth to main memory in GB per second.
45 };
46
47 // Returns basic device performance info.
48 virtual DeviceInfo GetDeviceInfo(const DeviceProperties& device) const;
49
50 protected:
51 // Predict cost of an op for which no accurate estimator is defined.
52 Costs PredictCostOfAnUnknownOp(const OpContext& op_context) const;
53
54 // Naive cost estimate based on the given operations count and total
55 // input/output tensor sizes of the given op_info combined.
56 Costs PredictOpCountBasedCost(double operations, const OpInfo& op_info) const;
57
58 // Naive cost estimate based on the given operations count and the given total
59 // io size in bytes. Sizes of op_info inputs and outputs are not taken into
60 // consideration.
61 Costs PredictOpCountBasedCost(double operations, double total_io_bytes,
62 const OpInfo& op_info) const;
63
64 // This family of routines counts the number of operations to perform the
65 // specified TensorFlow Op.
66 struct MatMulDimensions {
67 int m;
68 int n;
69 int k;
70 };
71 struct ConvolutionDimensions {
72 int64 batch; // Batch size.
73 int64 ix; // Input size x.
74 int64 iy; // Input size y.
75 int64 iz; // Input depth.
76 int64 kx; // Kernel x.
77 int64 ky; // Kernel y.
78 int64 oz; // Output depth.
79 int64 ox; // Output size x.
80 int64 oy; // Output size y.
81 int64 sx; // Stride x.
82 int64 sy; // Stride y.
83 Padding padding; // SAME or VALID.
84 };
85 enum ConvolutionFormat {
86 UNKNOWN_CONVOLUTION_FORMAT,
87 NHWC,
88 NCHW,
89 NCHW_VECT_C,
90 NCHW_VECT_W,
91 };
92 int64 CountConv2DOperations(const OpInfo& op_features,
93 bool* found_unknown_shapes) const;
94 int64 CountConv2DOperations(const OpInfo& op_features,
95 ConvolutionDimensions* conv_info,
96 bool* found_unknown_shapes) const;
97 int64 CountMatMulOperations(const OpInfo& op_features,
98 bool* found_unknown_shapes) const;
99 int64 CountMatMulOperations(const OpInfo& op_features,
100 MatMulDimensions* mat_mul,
101 bool* found_unknown_shapes) const;
102 int64 CountBatchMatMulOperations(const OpInfo& op_features,
103 bool* found_unknown_shapes) const;
104 int64 CountConv2DBackpropInputOperations(const OpInfo& op_features,
105 ConvolutionDimensions* conv_info,
106 bool* found_unknown_shapes) const;
107 int64 CountConv2DBackpropFilterOperations(const OpInfo& op_features,
108 ConvolutionDimensions* conv_info,
109 bool* found_unknown_shapes) const;
110
111 // Calculate the element count of an input/output tensor.
112 int64 CalculateTensorElementCount(const OpInfo::TensorProperties& tensor,
113 bool* found_unknown_shapes) const;
114
115 // Calculate the total size in bytes of an input/output tensor.
116 int64 CalculateTensorSize(const OpInfo::TensorProperties& tensor,
117 bool* found_unknown_shapes) const;
118
119 // Calculate the element count of the largest
120 // input of specified TensorFlow op.
121 int64 CalculateLargestInputCount(const OpInfo& op_features,
122 bool* found_unknown_shapes) const;
123
124 // Calculate the total size in bytes of the all
125 // the inputs of specified TensorFlow op.
126 int64 CalculateInputSize(const OpInfo& op_features,
127 bool* found_unknown_shapes) const;
128
129 // Calculate the total size in bytes of the all
130 // the outputs of specified TensorFlow op.
131 int64 CalculateOutputSize(const OpInfo& op_features,
132 bool* found_unknown_shapes) const;
133
134 // This family of routines predicts the costs to
135 // perform the specified TensorFlow Op on the
136 // device represented by a subclass. The default
137 // implementation just divides the operations to
138 // perform the op (from the "Count" routines,
139 // above) by the device peak operations per
140 // second.
141 // Implementation of costs other than
142 // execution_time is optional, depending on the
143 // device.
144 Costs PredictConv2D(const OpContext& op_context) const;
145 Costs PredictCwiseOp(const OpContext& op_context) const;
146 Costs PredictConv2DBackpropInput(const OpContext& op_context) const;
147 Costs PredictConv2DBackpropFilter(const OpContext& op_context) const;
148 Costs PredictFusedConv2DBiasActivation(const OpContext& op_context) const;
149 Costs PredictMatMul(const OpContext& op_context) const;
150 Costs PredictNoOp(const OpContext& op_context) const;
151 Costs PredictIdentity(const OpContext& op_context) const;
152 Costs PredictVariable(const OpContext& op_context) const;
153 Costs PredictBatchMatMul(const OpContext& op_context) const;
154 Costs PredictMetadata(const OpContext& op_context) const;
155 Costs PredictGatherOrSlice(const OpContext& op_context) const;
156 Costs PredictMaxPool(const OpContext& op_context) const;
157 Costs PredictMaxPoolGrad(const OpContext& op_context) const;
158 Costs PredictAvgPool(const OpContext& op_context) const;
159 Costs PredictAvgPoolGrad(const OpContext& op_context) const;
160 Costs PredictFusedBatchNorm(const OpContext& op_context) const;
161 Costs PredictFusedBatchNormGrad(const OpContext& op_context) const;
162
163 // Generic cost prediction method for fused operations.
164 Costs PredictFusedOp(const OpContext& op_context,
165 const std::vector<OpContext>& fused_op_contexts) const;
166
167 // Utility function for safe division. Returns 0
168 // if rhs is 0 or negative.
169 static double SafeDiv(const double lhs, const double rhs) {
170 if (rhs > 0) {
171 return lhs / rhs;
172 } else {
173 return 0.0;
174 }
175 }
176
177 // For convolution and its grad ops.
178 static ConvolutionDimensions ConvolutionDimensionsFromInputs(
179 const TensorShapeProto& original_image_shape,
180 const TensorShapeProto& original_filter_shape, const OpInfo& op_info,
181 bool* found_unknown_shapes);
182
183 // For Pooling, FusedBatchNorm, and their grad ops.
184 static ConvolutionDimensions OpDimensionsFromInputs(
185 const TensorShapeProto& original_image_shape, const OpInfo& op_info,
186 bool* found_unknown_shapes);
187
188 // Helper to construct child operation contexts for the component operations
189 // of fused ops.
190 static OpContext FusedChildContext(
191 const OpContext& parent, const string& op_name,
192 const OpInfo::TensorProperties& output,
193 const std::vector<OpInfo::TensorProperties>& inputs);
194
195 // Helper to construct tensor shapes.
196 static OpInfo::TensorProperties DescribeTensor(
197 DataType type, const std::vector<int64>& dims);
198
199 // Returns the Conv2D format for this operation.
200 static ConvolutionFormat GetConvolutionFormat(const OpContext& op_context);
201
202 // This method calculates the execution time depending on whether IO can
203 // overlap with computation. It assumes the memory and the compute times have
204 // already been calculated.
205 void CombineCostsAndUpdateExecutionTime(Costs* costs) const;
206
207 protected:
208 std::map<string, int> elementwise_ops_;
209 typedef std::function<Costs(const OpContext& op_context)> CostImpl;
210 std::map<string, CostImpl> device_cost_impl_;
211 // If true, assume compute and memory overlap; hence, the op cost is max of
212 // compute_time and memory_time, insteaf of sum of those two.
213 bool compute_memory_overlap_;
214
215 private:
216 friend class OpLevelCostEstimatorTest;
217};
