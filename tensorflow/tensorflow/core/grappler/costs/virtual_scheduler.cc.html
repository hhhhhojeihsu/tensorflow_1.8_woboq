40Costs CombineCosts(const Costs& left, const Costs& right) {
41 CHECK_NE(left.max_memory, kMemoryUnknown);
42 CHECK_NE(left.max_per_op_buffers, kMemoryUnknown);
43 CHECK_NE(left.max_per_op_streaming, kMemoryUnknown);
44
45 Costs result = left;
46 result.execution_time += right.execution_time;
47 result.compute_time += right.compute_time;
48 result.memory_time += right.memory_time;
49 if (right.inaccurate) {
50 result.inaccurate = true;
51 }
52 if (right.max_memory != kMemoryUnknown) {
53 result.max_memory += right.max_memory;
54 }
55 if (right.max_per_op_buffers != kMemoryUnknown) {
56 result.max_per_op_buffers =
57 std::max(left.max_per_op_buffers, right.max_per_op_buffers);
58 }
59 if (right.max_per_op_streaming != kMemoryUnknown) {
60 result.max_per_op_streaming =
61 std::max(left.max_per_op_streaming, right.max_per_op_streaming);
62 }
63 VLOG(4) << "costs execution_time=" << result.execution_time.count()
64 << " max_memory=" << result.max_memory
65 << " max_per_op_buffers=" << result.max_per_op_buffers
66 << " max_per_op_streaming=" << result.max_per_op_streaming;
67 return result;
68}

70// Key to the cached _Recv ops map, and its hash and predicate structures.
71struct RecvNodeDescriptor {
72 const NodeDef* node;
73 const int port_num;
74 const string device;
75
76 RecvNodeDescriptor(const NodeDef* node_, const int port_num_,
77 const string& device_)
78 : node(node_), port_num(port_num_), device(device_) {}
79};
80
81struct RecvNodeDescriptorHash {
82 std::size_t operator()(const RecvNodeDescriptor& recv_node) const {
83 return std::hash<const NodeDef*>()(recv_node.node) ^
84 std::hash<int>()(recv_node.port_num) ^
85 std::hash<string>()(recv_node.device);
86 }
87};
88
89struct RecvNodeDescriptorEqual {
90 bool operator()(const RecvNodeDescriptor& a,
91 const RecvNodeDescriptor& b) const {
92 return a.node == b.node && a.port_num == b.port_num && a.device == b.device;
93 }
94};

97// ReadyNodeManager
98const NodeDef* LIFOManager::GetCurrNode() {
99 CHECK(!nodes_.empty()) << "GetCurrNode(), but there's no ready node";
100 if (curr_pos_ == nodes_.end()) {
101 curr_pos_ = --(nodes_.rbegin().base()); // Last one in the list.
102 }
103 // Once curr_pos_ is set to a valid entry in the list, we keep using the
104 // cached curr_pos_ until RemoveCurrNode() is called. AddNode() will not
105 // change the GetCurrNode() return value.
106 return *curr_pos_;
107}
108
109void LIFOManager::RemoveCurrNode() {
110 // Make sure we have curr_pos_ ready to be removed.
111 GetCurrNode();
112 // Note curr_pos_ may not be pointing the last element if some nodes are
113 // added.
114 nodes_.erase(curr_pos_);
115
116 curr_pos_ = nodes_.end(); // Reset curr_pos_.
117}

119FirstReadyManager::FirstReadyManager() : ReadyNodeManager() {
120 std::make_heap(nodes_.begin(), nodes_.end());
121}
122
123void FirstReadyManager::Init(
124 const std::unordered_map<const NodeDef*, NodeState>* node_state) {
125 // Reset the node state since different instances of the scheduler can reuse
126 // the same node_manager.
127 node_state_ = node_state;
128 nodes_.clear();
129 waiting_queue_.clear();
130 greater_ = [this](const NodeDef* a, const NodeDef* b) -> bool {
131 if (node_state_->at(a).time_ready == node_state_->at(b).time_ready) {
132 // Use Node name as tie-breaker for deterministic node scheduling.
133 return a->name().compare(b->name()) > 0;
134 } else {
135 // Note: we need a node with minimum time_ready, not
136 // maximum; hence, using a > b for comparison function.
137 return node_state_->at(a).time_ready > node_state_->at(b).time_ready;
138 }
139 };
140}
141
142const NodeDef* FirstReadyManager::GetCurrNode() {
143 if (nodes_.empty()) {
144 // Nothing in the node_; probably, the very first call. Move
145 // waiting_queue_ to node_.
146 DrainWaitingQueue();
147 CHECK(!nodes_.empty()) << "GetCurrNode(), but there's no ready node";
148 }
149 return nodes_.front();
150}
151
152void FirstReadyManager::RemoveCurrNode() {
153 if (nodes_.empty()) {
154 // Make sure that there is a node to be removed at the front of nodes_.
155 GetCurrNode();
156 }
157 std::pop_heap(nodes_.begin(), nodes_.end(), greater_);
158 nodes_.pop_back();
159 DrainWaitingQueue();
160}
161
162bool FirstReadyManager::Empty() const {
163 return nodes_.empty() && waiting_queue_.empty();
164}
165
166void FirstReadyManager::DrainWaitingQueue() {
167 for (const auto* node : waiting_queue_) {
168 // push_heap in AddNode() and pop_heap in RemoveCurrNode() guarantees that
169 // the first element is the node with minimum time_ready.
170 nodes_.push_back(node);
171 std::push_heap(nodes_.begin(), nodes_.end(), greater_);
172 }
173 waiting_queue_.clear();
174}

176CompositeNodeManager::CompositeNodeManager()
177 : ReadyNodeManager(), send_manager_(), recv_manager_() {}

179void CompositeNodeManager::Init(
180 const std::unordered_map<const NodeDef*, NodeState>* node_state) {
181 node_state_ = node_state;
182 send_manager_.Init(node_state);
183 recv_manager_.Init(node_state);
184 curr_node_ = nullptr;
185}
186
187void CompositeNodeManager::AddNode(const NodeDef* node) {
188 if (IsSend(*node)) {
189 send_manager_.AddNode(node);
190 } else if (IsRecv(*node)) {
191 recv_manager_.AddNode(node);
192 } else {
193 const auto& device = node_state_->at(node).device_name;
194 ops_lifo_map_[device].AddNode(node);
195 }
196}
197
198const NodeDef* CompositeNodeManager::GetCurrNode() {
199 if (curr_node_) return curr_node_;
200
201 // Per-device LIFO for normal ops (not _Send / _Recv),
202 // FirstReady for _Send and _Recv (separately),
203 // Globally (among the LIFO-selected ops from each device and _Send and
204 // _Recv) FirstReady,
205 // Priorty order: _Send, _Recv, and then the rest, if time_ready is equal.
206 std::vector<std::pair<const NodeDef*, Costs::Duration>> candidates;
207 for (auto& ops_lifo : ops_lifo_map_) {
208 if (!ops_lifo.second.Empty()) {
209 const auto* op = ops_lifo.second.GetCurrNode();
210 candidates.emplace_back(op, node_state_->at(op).time_ready);
211 }
212 }
213 if (!send_manager_.Empty()) {
214 const auto* send = send_manager_.GetCurrNode();
215 candidates.emplace_back(send, node_state_->at(send).time_ready);
216 }
217 if (!recv_manager_.Empty()) {
218 const auto* recv = recv_manager_.GetCurrNode();
219 candidates.emplace_back(recv, node_state_->at(recv).time_ready);
220 }
221 CHECK(!candidates.empty());
222 auto first_ready = std::min_element(
223 candidates.begin(), candidates.end(),
224 [](const std::pair<const NodeDef*, Costs::Duration>& a,
225 const std::pair<const NodeDef*, Costs::Duration>& b) {
226 if (a.second == b.second) {
227 // Note that there can be only 1 Send and only 1 Recv in candidates,
228 // at most; hence, score is 2 for Send, 1 for Recv, and 0 for a
229 // normap op, and a_score and b_score are equal only if both are
230 // normal ops.
231 int a_score = 2 * IsSend(*a.first) + IsRecv(*a.first);
232 int b_score = 2 * IsSend(*b.first) + IsRecv(*b.first);
233 if (a_score == b_score) {
234 // Both are normal ops; use node name as tie breaker.
235 return a.first->name().compare(b.first->name()) < 0;
236 } else {
237 // Priortize by op type: _Send, _Recv, and normap ops.
238 return a_score > b_score;
239 }
240 } else {
241 return a.second < b.second;
242 }
243 });
244 // Next time we call GetCurrNode(), it just returns the cached one,
245 // curr_node_ until we call RemovCurrNode().
246 curr_node_ = first_ready->first;
247
248 return curr_node_;
249}
250
251void CompositeNodeManager::RemoveCurrNode() {
252 const auto* node = GetCurrNode();
253 if (IsSend(*node)) {
254 send_manager_.RemoveCurrNode();
255 } else if (IsRecv(*node)) {
256 recv_manager_.RemoveCurrNode();
257 } else {
258 const auto device = node_state_->at(node).device_name;
259 ops_lifo_map_[device].RemoveCurrNode();
260 }
261 // Reset curr_node_ so that GetCurrNode() finds another node.
262 curr_node_ = nullptr;
263}
264
265bool CompositeNodeManager::Empty() const {
266 // Empty if all the ready managers are empty.
267 bool empty = true;
268 for (const auto& ops_lifo : ops_lifo_map_) {
269 empty &= ops_lifo.second.Empty();
270 }
271 return empty && send_manager_.Empty() && recv_manager_.Empty();
272}

288ReadyNodeManager* VirtualScheduler::ReadyNodeManagerFactory(
289 const string& ready_node_manager) {
290 if (ready_node_manager == "FIFO") {
291 return new FIFOManager();
292 } else if (ready_node_manager == "LIFO") {
293 return new LIFOManager();
294 } else if (ready_node_manager == "FirstReady") {
295 return new FirstReadyManager();
296 } else if (ready_node_manager == "Composite") {
297 return new CompositeNodeManager();
298 }
299 LOG(FATAL) << "Not a valid ready node manager: " << ready_node_manager;
300}

302Status VirtualScheduler::Init() {
303 // Init() preprocesses the input grappler_item and graph_properties to extract
304 // necessary information for emulating tensorflow op scheduling and
305 // construct internal data structures (NodeState and DeviceState) for virtual
306 // scheduling.
307 ready_nodes_->Init(GetNodeStates());
308 // Construct graph properties.
309 Status status;
310 if (use_static_shapes_) {
311 status = graph_properties_.InferStatically(true);
312 } else {
313 status = graph_properties_.InferDynamically(cluster_);
314 }
315 if (!status.ok()) {
316 return status;
317 }
318
319 const auto& graph = grappler_item_->graph;
320 const auto& fetch_nodes = grappler_item_->fetch;
321 std::set<string> feed_nodes;
322 for (const auto& f : grappler_item_->feed) {
323 auto iter_and_inserted_flag = feed_nodes.insert(f.first);
324 QCHECK(iter_and_inserted_flag.second)
325 << "Duplicate feed node found: " << f.first;
326 }
327
328 // Get the nodes that would run to output fetch_nodes.
329 bool ill_formed = false;
330 const std::vector<const NodeDef*> fetch_fanin_nodes =
331 ComputeTransitiveFanin(graph, fetch_nodes, &ill_formed);
332 if (ill_formed) {
333 return errors::InvalidArgument(
334 "Ill formed graph or invalid set of fetch nodes specified");
335 }
336
337 // TODO(dyoon): this is a bit inefficient as name_to_node is already built in
338 // ComputeTransitiveFanin().
339 // Once ComputeTransitiveFanin is complete, only the nodes that can be reached
340 // from the fetch nodes are scheduled. So the scheduled nodes should be
341 // exactly the same as those executed for real. One possible discrepancy could
342 // be the control flow nodes, where tf only executes one path.
343 std::unordered_map<string, const NodeDef*> name_to_node;
344 for (const auto& node : fetch_fanin_nodes) {
345 name_to_node[node->name()] = node;
346 }
347
348 // TODO(dyoon): Instead of identifying _Send node here manually, add _Send
349 // to _Recv as control dependency when creating GrapplerItem.
350 std::unordered_map<string, const NodeDef*> name_to_send;
351 for (const auto& node : graph.node()) {
352 if (IsSend(node)) {
353 const auto& attr = node.attr();
354 name_to_send[attr.at("tensor_name").s()] = &node;
355 }
356 }
357
358 // To reuse _Recv ops.
359 std::unordered_map<RecvNodeDescriptor, const NodeDef*, RecvNodeDescriptorHash,
360 RecvNodeDescriptorEqual>
361 cached_recv_nodes;
362
363 // Build node_map; for each node, create its NodeState and connect its inputs
364 // and outputs.
365 for (const auto* curr_node : fetch_fanin_nodes) {
366 auto& curr_node_state = GetNodeStateOrCreateIt(curr_node);
367 const string curr_node_device = DeviceName(curr_node);
368 std::vector<string> inputs;
369 if (IsRecv(*curr_node)) {
370 const auto& attr = curr_node->attr();
371 if (attr.count("tensor_name")) {
372 const auto& send_node_name = attr.at("tensor_name").s();
373 auto it = name_to_send.find(send_node_name);
374 // If there is a _Send associated with the curr_node (_Recv), add it as
375 // input.
376 if (it != name_to_send.end()) {
377 const NodeDef* send = it->second;
378 inputs = {send->name()};
379 }
380 }
381 } else {
382 for (const string& input : curr_node->input()) {
383 inputs.push_back(input);
384 }
385 }
386 for (const string& input_node_name : inputs) {
387 // Note that input_node_name may be in <prefix><node_name>:<port_num>
388 // format, where <prefix> (e.g., "^" for control dependency) and
389 // ":<port_num>" may be omitted. NodeName() extracts only the node_name.
390 const NodeDef* input_node = name_to_node[NodeName(input_node_name)];
391
392 CHECK(input_node);
393 const string in_device = DeviceName(input_node);
394 const auto input_node_port_num = NodePosition(input_node_name);
395
396 if (curr_node_device == in_device) {
397 // Same device: connect input_node and curr_node directly.
398 curr_node_state.inputs.push_back(
399 std::make_pair(input_node, input_node_port_num));
400 auto& input_node_state = GetNodeStateOrCreateIt(input_node);
401 input_node_state.outputs[input_node_port_num].push_back(curr_node);
402 } else {
403 RecvNodeDescriptor recv_node(input_node, input_node_port_num,
404 curr_node_device);
405 auto it = cached_recv_nodes.find(recv_node);
406 if (it != cached_recv_nodes.end()) {
407 // Different device, but found an already-cached copy (a _Recv op);
408 // connect the _Recv to curr_node.
409 const NodeDef* recv_op = it->second;
410 // recv_op's output port is hard-coded to zero.
411 curr_node_state.inputs.push_back(std::make_pair(recv_op, 0));
412 auto& input_node_state = node_map_.at(recv_op);
413 input_node_state.outputs[0].push_back(curr_node);
414 } else {
415 // Different device, no cached copy; transfer input_node to the
416 // curr_node's device.
417 auto send_and_recv =
418 CreateSendRecv(input_node, curr_node, input_node_name);
419 // Note that CreateSendRecv() already connected input/output between
420 // _Send and _Recv ops.
421 const auto* send = send_and_recv.first;
422 const auto* recv = send_and_recv.second;
423 // recv_op's output port is hard-coded to zero.
424 curr_node_state.inputs.push_back(std::make_pair(recv, 0));
425 auto& input_node_state = GetNodeStateOrCreateIt(input_node);
426 input_node_state.outputs[input_node_port_num].push_back(send);
427
428 // Cache the _Recv op for future use.
429 cached_recv_nodes[recv_node] = recv;
430 }
431 }
432 }
433
434 // Special case: given feed nodes are ready at time 0.
435 const bool given_as_feed =
436 feed_nodes.find(curr_node->name()) != feed_nodes.end();
437
438 // Default case: node without inputs are ready at time 0.
439 // Note that we check inputs vector which may be different to
440 // curr_node->input(); e.g., we add Send as input to Recv.
441 const bool has_no_inputs = inputs.empty();
442
443 if (given_as_feed || has_no_inputs) {
444 curr_node_state.time_ready = Costs::Duration();
445 ready_nodes_->AddNode(curr_node);
446 VLOG(3) << "Added ready node: " << curr_node->name();
447 }
448
449 feed_nodes.erase(curr_node->name());
450
451 if (IsPersistentNode(curr_node)) {
452 auto& device_state = device_[curr_node_device];
453 for (int port_num = 0;
454 port_num < curr_node_state.output_properties.size(); ++port_num) {
455 device_state.persistent_nodes.insert(
456 std::make_pair(curr_node, port_num));
457 }
458 }
459 }
460
461 if (ready_nodes_->Empty()) {
462 return errors::InvalidArgument("No ready nodes in the graph.");
463 }
464
465 if (!feed_nodes.empty()) {
466 // This isn't always a bug: when the caller hasn't specified the exact list
467 // of feed and fetch nodes, by default we consider all placeholders as feed
468 // nodes, but some of them may not be needed for the default fetch node.
469 VLOG(1) << "Some feed nodes were not consumed by the fetch fanin: "
470 << str_util::Join(feed_nodes, ",");
471 }
472 initialized_ = true;
473 return Status::OK();
474}

476void VirtualScheduler::MaybeUpdateInputOutput(const NodeDef* node) {
477 CHECK(!initialized_) << "MaybeUpdateInputOutput is called after Init().";
478 // This method is called when NodeState is created and adds input and output
479 // properties for a few exceptional cases that GraphProperties cannot provide
480 // input/output properties.
481 if ((IsSend(*node) || IsRecv(*node)) && node->attr().count(kAttrInputSrc)) {
482 // _Send and _Recv ops created from VirtualScheduler have kAttrInputSrc
483 // attr; normal _Send and _Recv ops (from the input graph) do not have that
484 // attr.
485 auto& node_state = node_map_[node];
486 auto& inputs = node_state.input_properties;
487 auto& outputs = node_state.output_properties;
488
489 // _Send and _Recv ops are created from VirtualScheduler, so
490 // there should be no inputs TensorProperties.
491 CHECK(inputs.empty());
492 CHECK(outputs.empty());
493 const auto& attr = node->attr();
494 // This is the original input source to the _Send and _Recv, and this
495 // string includes "^" if it was control dependency, and output port
496 /// (e.g., ":2") if the input source had multiple outputs.
497 const auto& input_source_name = attr.at(kAttrInputSrc).s();
498 if (IsControlInput(input_source_name)) {
499 // Control dependency; regardless of the input source tensor size,
500 // send 4B.
501 OpInfo::TensorProperties control_message;
502 control_message.set_dtype(DT_FLOAT);
503 control_message.mutable_shape()->add_dim()->set_size(1);
504 auto* value = control_message.mutable_value();
505 value->add_float_val(1);
506 inputs.push_back(control_message);
507 outputs.push_back(control_message);
508 } else {
509 auto output_properties =
510 graph_properties_.GetOutputProperties(NodeName(input_source_name));
511 // Like with HasInputProperties, if a node does not have output
512 // properties, it's likely it was pruned during the shape inference run.
513 if (!output_properties.empty()) {
514 const auto input_node_port_num = NodePosition(input_source_name);
515 // Use the input source's output property as _Send and _Recv's input
516 // property.
517 CHECK_GT(output_properties.size(), input_node_port_num);
518 inputs.push_back(output_properties[input_node_port_num]);
519 outputs.push_back(output_properties[input_node_port_num]);
520 }
521 }
522 }

554std::pair<const NodeDef*, const NodeDef*> VirtualScheduler::CreateSendRecv(
555 const NodeDef* from, const NodeDef* to, const string& input_name) {
556 CHECK(!initialized_) << "CreateSendRecv is called after Init().";
557
558 // Connect "from" node to "to" node with _Send and _Recv such that
559 // from -> _Send -> _Recv -> to.
560 // _Send is placed on "Channel" device, and _Recv is on the same device
561 // as "to" node.
562 // input_node_name is the string from the "to" node to identify which output
563 // we get from the "from" node.
564
565 // Note that we use NodeState for scheduling, so _Send and _Recv
566 // NodeDefs created here need not be correct: in terms of name,
567 // input names, attrs, etc.
568
569 auto input_node_port_num = NodePosition(input_name);
570 string src_name;
571 if (input_node_port_num >= 0) {
572 src_name = strings::StrCat(from->name(), "_", input_node_port_num);
573 } else {
574 src_name = strings::StrCat(from->name(), "_minus1");
575 }
576
577 // _Send op.
578 auto* send = new NodeDef();
579 send->set_name("Send_" + src_name + "_from_" + SanitizedDeviceName(from) +
580 "_to_" + SanitizedDeviceName(to));
581 send->set_op("_Send");
582 send->add_input(from->name());
583 send->set_device(ChannelDeviceName(from, to));
584 auto& send_attr = *(send->mutable_attr());
585 send_attr[kAttrInputSrc].set_s(input_name);
586 send_attr[kAttrSrcDevice].set_s(DeviceName(from));
587 send_attr[kAttrDstDevice].set_s(DeviceName(to));
588
589 // _Recv op.
590 auto* recv = new NodeDef();
591 recv->set_name("Recv_" + src_name + "_on_" + SanitizedDeviceName(to));
592 recv->set_op("_Recv");
593 recv->add_input(send->name());
594 recv->set_device(DeviceName(to));
595 auto& recv_attr = *(recv->mutable_attr());
596 recv_attr[kAttrInputSrc].set_s(input_name);
597
598 // NodeState for _Send op.
599 auto& send_node_state = GetNodeStateOrCreateIt(send);
600 send_node_state.device_name = send->device(); // Set Channel device.
601 send_node_state.inputs.push_back(std::make_pair(from, input_node_port_num));
602 send_node_state.outputs[0].push_back(recv);
603
604 // NodeState for _Recv op.
605 auto& recv_node_state = GetNodeStateOrCreateIt(recv);
606 recv_node_state.inputs.push_back(std::make_pair(send, 0));
607 recv_node_state.outputs[0].push_back(to);
608
609 // Keep the created nodes.
610 additional_nodes_.emplace_back(std::unique_ptr<NodeDef>(send));
611 additional_nodes_.emplace_back(std::unique_ptr<NodeDef>(recv));
612
613 // Return _Send and _Recv.
614 return std::make_pair(send, recv);
615}

617OpContext VirtualScheduler::GetCurrNode() const {
618 const NodeDef* node = ready_nodes_->GetCurrNode();
619
620 // Get the device from the placer.
621 DeviceProperties device;
622 device = placer_.get_device(*node);
623
624 // Special case for _Send op.
625 if (IsSend(*node)) {
626 device.set_type(kChannelDevice);
627 }
628
629 // Construct OpContext.
630 OpContext op_context;
631 const auto& node_state = node_map_.at(node);
632 op_context.name = node->name();
633 op_context.device_name = node_state.device_name;
634 auto& op_info = op_context.op_info;
635 op_info.set_op(node->op());
636 *op_info.mutable_attr() = node->attr();
637 for (auto& input : node_state.input_properties) {
638 *op_info.add_inputs() = input;
639 }
640 for (auto& output : node_state.output_properties) {
641 *op_info.add_outputs() = output;
642 }
643 op_info.mutable_device()->Swap(&device);
644
645 if (grappler_item_->graph.has_library()) {
646 op_context.function_library = &grappler_item_->graph.library();
647 }
648 return op_context;
649}

651NodeState& VirtualScheduler::GetNodeStateOrCreateIt(const NodeDef* node) {
652 CHECK(!initialized_) << "GetNodeStateOrCreateIt is called after Init().";
653
654 auto it = node_map_.find(node);
655 if (it == node_map_.end()) {
656 // Not found; create a NodeState for this node.
657 it = node_map_.emplace(node, NodeState()).first;
658 auto& node_state = it->second;
659 node_state.input_properties =
660 graph_properties_.GetInputProperties(node->name());
661 node_state.output_properties =
662 graph_properties_.GetOutputProperties(node->name());
663
664 // Some ops may need further processing to the input / output properties:
665 // _Send and _Recv.
666 MaybeUpdateInputOutput(node);
667
668 if (!IsSend(*node)) {
669 node_state.device_name = DeviceName(node);
670 // For _Send op, device_name will be set to Channel in CreateSendRecv().
671 }
672
673 // Initialize output port related data:
674 // Assume the size of OutputProperties represents the number of output ports
675 // of this node.
676 for (size_t i = 0; i < node_state.output_properties.size(); ++i) {
677 node_state.time_no_references[i] = Costs::Duration::max();
678 node_state.num_outputs_executed[i] = 0;
679 // Populate an empty vector for each port. The caller will add nodes
680 // that use this port as input.
681 node_state.outputs[i] = {};
682 }
683 // Port_num -1 is for control dependency.
684 node_state.time_no_references[-1] = Costs::Duration::max();
685 node_state.num_outputs_executed[-1] = 0;
686 node_state.outputs[-1] = {};
687 }
688 return it->second;
689}

691int64 VirtualScheduler::CalculateOutputSize(
692 const std::vector<OpInfo::TensorProperties>& output_properties,
693 const int port_num) const {
694 if (port_num < 0) {
695 return 4; // 4B for control dependency.
696 }
697
698 if (port_num >= output_properties.size()) {
699 VLOG(3) << "VirtualScheduler::CalculateOutputSize() -- "
700 << "port_num: " << port_num
701 << " >= output_properties.size(): " << output_properties.size();
702 return 0;
703 }
704
705 const auto& output = output_properties[port_num];
706 int64 output_size = DataTypeSize(BaseType(output.dtype()));
707
708 for (const auto& dim : output.shape().dim()) {
709 auto dim_size = dim.size();
710 if (dim_size < 0) {
711 // Zero output size if there's any unknown dim.
712 output_size = 0;
713 VLOG(3) << "VirtualScheduler::CalculateOutputSize() -- "
714 << "unknown dim: " << output_size;
715 break;
716 }
717 output_size *= dim_size;
718 }
719
720 return output_size;
721}

723Costs& VirtualScheduler::FindOrCreateZero(const string& op_name,
724 std::map<string, Costs>* op_cost) {
725 auto it = op_cost->find(op_name);
726 if (it == op_cost->end()) {
727 // Note that default constructor of Costs sets some memory related fields
728 // to unknown values so we should explicitly initialize it with ZeroCosts.
729 it = op_cost->emplace(op_name, Costs::ZeroCosts()).first;
730 }
731 return it->second;
732}

734bool VirtualScheduler::MarkCurrNodeExecuted(const Costs& node_costs) {
735 // Update graph_costs_ and per-op costs.
736 graph_costs_ = CombineCosts(graph_costs_, node_costs);
737 const NodeDef* node = ready_nodes_->GetCurrNode();
738 const string& op_name = node->op();
739
740 // Also keep track of op counts and times per op (with their shapes).
741 OpContext op_context = GetCurrNode();
742 string node_description = GetOpDescription(op_context.op_info);
743 op_counts_[node_description] += 1;
744 op_costs_[node_description] =
745 std::make_pair(node_costs.execution_time.asMicroSeconds().count(),
746 !node_costs.inaccurate);
747
748 auto& op_cost = FindOrCreateZero(op_name, &op_to_cost_);
749 op_cost = CombineCosts(op_cost, node_costs);
750
751 // Update node and device states.
752 auto& node_state = node_map_[node];
753 auto& device = device_[node_state.device_name];
754 device.nodes_executed.push_back(node);
755 // Node is scheduled when the device is available AND all the inputs are
756 // ready; hence, time_scheduled is time_ready if time_ready > device curr
757 // time.
758 node_state.time_scheduled =
759 std::max(device.GetCurrTime(), node_state.time_ready);
760 // Override device curr time with the time_scheduled.
761 device.device_costs.execution_time = node_state.time_scheduled;
762 device.device_costs = CombineCosts(device.device_costs, node_costs);
763 auto curr_time = device.GetCurrTime();
764 node_state.time_finished = curr_time;
765
766 // Update device memory usage.
767 if (!IsPersistentNode(node)) {
768 for (const auto& port_num_output_pair : node_state.outputs) {
769 int port_num = port_num_output_pair.first;
770 // There's a chance that a specific output is not used at all.
771 if (node_state.outputs[port_num].empty()) {
772 node_state.time_no_references[port_num] = curr_time;
773 } else {
774 device.memory_usage +=
775 CalculateOutputSize(node_state.output_properties, port_num);
776 device.nodes_in_memory.insert(std::make_pair(node, port_num));
777 }
778 }
779 }
780
781 // Update device's per-op cost.
782 auto& device_op_cost = FindOrCreateZero(op_name, &device.op_to_cost);
783 device_op_cost = CombineCosts(device_op_cost, node_costs);
784
785 VLOG(3) << "Op scheduled -- name: " << node->name() << ", op: " << node->op()
786 << ", device: " << node->device()
787 << ", ready: " << node_state.time_ready.count()
788 << ", scheduled: " << node_state.time_scheduled.count()
789 << ", finished: " << node_state.time_finished.count();
790
791 // Increment num_inputs_ready of the output nodes
792 for (const auto& port_num_output_pair : node_state.outputs) {
793 for (auto* output_node : port_num_output_pair.second) {
794 auto& output_state = node_map_[output_node];
795 output_state.num_inputs_ready++;
796 // Execute a node as soon as all its inputs are ready. Merge nodes are
797 // special since they run as soon as one of their inputs becomes
798 // available.
799 if (output_state.num_inputs_ready == output_state.inputs.size() ||
800 IsMerge(*output_node)) {
801 // This output node is now ready.
802 output_state.time_ready = curr_time;
803 ready_nodes_->AddNode(output_node);
804 }
805 }
806 }
807
808 // Increment num_outputs_executed of the input nodes.
809 for (const auto& input_port : node_state.inputs) {
810 auto* input = input_port.first;
811 auto port = input_port.second;
812 auto& input_state = node_map_[input];
813 input_state.num_outputs_executed[port]++;
814 if (input_state.num_outputs_executed[port] ==
815 input_state.outputs[port].size() &&
816 !IsPersistentNode(input)) {
817 // All the outputs are executed; no reference to this output port of
818 // input node.
819 input_state.time_no_references[port] = curr_time;
820 auto& input_device = device_[input_state.device_name];
821 input_device.memory_usage -=
822 CalculateOutputSize(input_state.output_properties, port);
823
824 input_device.nodes_in_memory.erase(std::make_pair(input, port));
825 }
826 }
827
828 if (!IsPersistentNode(node)) {
829 // Now that output memory is added and used up nodes are deallocated,
830 // check max memory usage.
831 if (device.memory_usage > device.max_memory_usage) {
832 device.max_memory_usage = device.memory_usage;
833 device.mem_usage_snapshot_at_peak = device.nodes_in_memory;
834 }
835 }
836
837 // Remove the current node; assume FIFO.
838 ready_nodes_->RemoveCurrNode();
839
840 return !ready_nodes_->Empty();
841}

843Costs VirtualScheduler::Summary() const {
844 // Print out basic execution summary.
845 VLOG(1) << "Expected execution time: " << graph_costs_.execution_time.count();
846 VLOG(1) << "Expected compute time: " << graph_costs_.compute_time.count();
847 VLOG(1) << "Expected memory time: " << graph_costs_.memory_time.count();
848 VLOG(1) << "Expected max memory: " << graph_costs_.max_memory;
849 VLOG(1) << "Expected max per-op buffers: " << graph_costs_.max_per_op_buffers;
850 VLOG(1) << "Expected max per-op streaming buffers: "
851 << graph_costs_.max_per_op_streaming;
852
853 VLOG(1) << "Per-op execution time:";
854 for (const auto& op_cost_pair : op_to_cost_) {
855 const auto& op = op_cost_pair.first;
856 const auto& cost = op_cost_pair.second.execution_time.count();
857 const bool is_op_cost_accurate = !op_cost_pair.second.inaccurate;
858 if (cost) { // Skip printing out zero-cost ops.
859 VLOG(1) << " + " << op << " : " << (is_op_cost_accurate ? "" : "~")
860 << cost;
861 }
862 }
863
864 // Print per device summary
865 VLOG(1) << "Devices:";
866 Costs critical_path_costs = Costs::ZeroCosts();
867
868 for (const auto& device : device_) {
869 const auto& name = device.first;
870 const auto& state = device.second;
871
872 std::map<string, int64> op_to_memory;
873 // First profile only persistent memory usage.
874 int64 persistent_memory_usage = 0;
875 std::set<string> persisent_ops;
876 for (const auto& node_port : state.persistent_nodes) {
877 const auto* node = node_port.first;
878 const auto port = node_port.second;
879 const auto output_size =
880 CalculateOutputSize(node_map_.at(node).output_properties, port);
881 persistent_memory_usage += output_size;
882 op_to_memory[node->op()] += output_size;
883 persisent_ops.insert(node->op());
884 }
885 int64 max_memory_usage = persistent_memory_usage + state.max_memory_usage;
886 critical_path_costs.estimated_max_memory_per_device[name] =
887 max_memory_usage;
888
889 const Costs::NanoSeconds wall_time_ns = state.GetCurrTime();
890 VLOG(1) << "Device = " << name
891 << ", num_nodes = " << state.nodes_executed.size()
892 << ", wall_time_ns = " << wall_time_ns.count() << ", memory usage: "
893 << "persistent = "
894 << strings::HumanReadableNumBytes(persistent_memory_usage)
895 << ", peak = "
896 << strings::HumanReadableNumBytes(state.max_memory_usage)
897 << ", total = " << strings::HumanReadableNumBytes(max_memory_usage)
898 << ", at the end: "
899 << strings::HumanReadableNumBytes(state.memory_usage);
900
901 VLOG(1) << "Per-op execution time (and memory usage at peak memory usage):";
902
903 // Profile non-persistent op memory usage.
904 for (const auto& node_port : state.mem_usage_snapshot_at_peak) {
905 const auto* node = node_port.first;
906 const auto port = node_port.second;
907 op_to_memory[node->op()] +=
908 CalculateOutputSize(node_map_.at(node).output_properties, port);
909 }
910 Costs::NanoSeconds total_compute_time_ns;
911 bool is_total_cost_accurate = true;
912 for (const auto& op_cost_pair : state.op_to_cost) {
913 const auto& op = op_cost_pair.first;
914 const auto& cost = op_cost_pair.second.execution_time.count();
915 total_compute_time_ns += op_cost_pair.second.execution_time;
916 const bool is_op_cost_accurate = !op_cost_pair.second.inaccurate;
917 if (!is_op_cost_accurate) {
918 is_total_cost_accurate = false;
919 }
920
921 int64 op_mem_usage = 0;
922 auto it = op_to_memory.find(op);
923 if (it != op_to_memory.end()) {
924 op_mem_usage = it->second;
925 }
926
927 const float mem_usage_percent =
928 max_memory_usage > 0 ? Round2(100.0 * op_mem_usage / max_memory_usage)
929 : 0.0;
930 if (cost || mem_usage_percent > 1.0) {
931 // Print out only non-zero cost ops or ops with > 1% memory usage.
932 VLOG(1) << " + " << op << " : " << (is_op_cost_accurate ? "" : "~")
933 << cost << " (" << strings::HumanReadableNumBytes(op_mem_usage)
934 << " [" << mem_usage_percent << "%] "
935 << (persisent_ops.count(op) > 0 ? ": persistent op)" : ")");
936 }
937 }
938
939 int utilization = 0;
940 if (wall_time_ns.count() > 0) {
941 utilization = total_compute_time_ns.count() * 100 / wall_time_ns.count();
942 }
943 VLOG(1) << "Device = " << name << ", total_compute_time_ns = "
944 << (is_total_cost_accurate ? "" : "~")
945 << total_compute_time_ns.count()
946 << ", utilization = " << utilization << "%";
947
948 if (critical_path_costs.execution_time <= state.GetCurrTime()) {
949 critical_path_costs = state.device_costs;
950 }
951 }
952
953 if (VLOG_IS_ON(2)) {
954 // Also log the op description and their corresponding counts.
955 VLOG(2) << "Node description, counts, cost:";
956 for (const auto& item : op_counts_) {
957 int cost;
958 bool is_cost_accurate;
959 std::tie(cost, is_cost_accurate) = op_costs_.at(item.first);
960 VLOG(2) << "Node: " << item.first << ", Count: " << item.second
961 << ", Individual Cost: " << (is_cost_accurate ? "" : "~") << cost;
962 }
963 }
964
965 VLOG(1) << "Critical path execution time: "
966 << critical_path_costs.execution_time.count();
967 return critical_path_costs;
968}

970Costs VirtualScheduler::Summary(RunMetadata* metadata) {
971 if (metadata != nullptr) {
972 StepStats* stepstats = metadata->mutable_step_stats();
973 for (const auto& device : device_) {
974 GraphDef* device_partition_graph = metadata->add_partition_graphs();
975 DeviceStepStats* device_stepstats = stepstats->add_dev_stats();
976 device_stepstats->set_device(device.first);
977 for (const auto& node_def : device.second.nodes_executed) {
978 const NodeState& nodestate = node_map_.at(node_def);
979 NodeExecStats* node_stats = device_stepstats->add_node_stats();
980 uint64 total_output_size = 0;
981 for (int slot = 0; slot < nodestate.output_properties.size(); slot++) {
982 const auto& properties = nodestate.output_properties[slot];
983 NodeOutput* no = node_stats->add_output();
984 no->set_slot(slot);
985 TensorDescription* tensor_descr = no->mutable_tensor_description();
986 tensor_descr->set_dtype(properties.dtype());
987 *tensor_descr->mutable_shape() = properties.shape();
988 // Optional allocation description.
989 const auto tensor_size =
990 CalculateOutputSize(nodestate.output_properties, slot);
991 total_output_size += tensor_size;
992 tensor_descr->mutable_allocation_description()->set_requested_bytes(
993 tensor_size);
994 tensor_descr->mutable_allocation_description()->set_allocated_bytes(
995 tensor_size);
996 }
997 node_stats->set_timeline_label(node_def->op());
998 node_stats->set_node_name(node_def->name());
999 node_stats->set_op_start_rel_micros(0);
1000 node_stats->set_all_start_micros(
1001 nodestate.time_scheduled.asMicroSeconds().count());
1002 node_stats->set_op_end_rel_micros(
1003 nodestate.time_finished.asMicroSeconds().count() -
1004 nodestate.time_scheduled.asMicroSeconds().count());
1005 node_stats->set_all_end_rel_micros(
1006 nodestate.time_finished.asMicroSeconds().count() -
1007 nodestate.time_scheduled.asMicroSeconds().count());
1008 auto* mem_stats = node_stats->mutable_memory_stats();
1009 // VirtualScheduler does not specify scratch pad memory usage.
1010 mem_stats->set_temp_memory_size(0);
1011 int64 persistent_memory_size = 0;
1012 if (IsPersistentNode(node_def)) {
1013 persistent_memory_size = total_output_size;
1014 }
1015 mem_stats->set_persistent_memory_size(persistent_memory_size);
1016 *device_partition_graph->add_node() = *node_def;
1017 }
1018 }
1019 }
1020 return Summary();
1021}
