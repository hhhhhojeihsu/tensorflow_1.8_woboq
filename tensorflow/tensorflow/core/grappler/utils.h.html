35// A utility class to lookup a node and its outputs by node name.
36class NodeMap {

63// A vector with a set. The set stores the same elements as the vector, and
64// quickly answers whether a value is in the vector. Duplicated elements are not
65// allowed for now.
66template <class T>
67class SetVector {
68 public:
69 // Returns false if value already existed in the set, true otherwise.
70 bool PushBack(const T& value) {
71 if (!set_.insert(value).second) {
72 return false;
73 }
74 vector_.push_back(value);
75 return true;
76 }
77
78 T PopBack() {
79 T back = vector_.back();
80 set_.erase(back);
81 vector_.pop_back();
82 return back;
83 }
84
85 bool Exists(const T& value) const { return set_.find(value) != set_.end(); }
86
87 bool Empty() const { return vector_.empty(); }
88
89 void Reserve(int64 size) { vector_.reserve(size); }
90
91 private:
92 std::unordered_set<T> set_;
93 std::vector<T> vector_;
94};

180class SimpleGraphView {
181 public:
182 Status Initialize(const GraphDef& graph) {
183 return Initialize(graph, true, true);
184 }
185 Status Initialize(const GraphDef& graph, bool dedup_inputs,
186 bool dedup_outputs);
187
188 const GraphDef* graph() const { return graph_; }
189 inline int num_nodes() const { return index_to_name_.size(); }
190 inline const int index(const string& node_name) const {
191 const auto& it = name_to_index_.find(node_name);
192 DCHECK(it != name_to_index_.end());
193 return it == name_to_index_.end() ? -1 : it->second;
194 }
195 inline const string& node_name(int node_idx) const {
196 return index_to_name_[node_idx];
197 }
198 inline const gtl::InlinedVector<int, 4>& inputs(int node_idx) const {
199 return inputs_[node_idx];
200 }
201 inline const gtl::InlinedVector<int, 2>& outputs(int node_idx) const {
202 return outputs_[node_idx];
203 }
204
205 // Traverse the graph starting at `node_idx`, collecting indices of nodes
206 // visited in nodes_found. If a node has an op in `op_types_to_traverse`, the
207 // walk continues to its children. It is assumed that *graph_ was not modified
208 // after the call to Initialize().
209 void DepthFirstSearch(const std::unordered_set<string>& op_types_to_traverse,
210 int node_idx, std::set<int>* nodes_found) const;
211
212 string PrintToString() const;
213
214 private:
215 const GraphDef* graph_; // Not owned.
216 std::vector<string> index_to_name_;
217 std::unordered_map<string, int> name_to_index_;
218 std::vector<gtl::InlinedVector<int, 4>> inputs_;
219 std::vector<gtl::InlinedVector<int, 2>> outputs_;
220};
