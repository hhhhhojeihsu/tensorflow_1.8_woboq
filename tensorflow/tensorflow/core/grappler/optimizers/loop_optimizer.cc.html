48class LoopInvariantNodeMotionOptimizer {
49 public:
50 explicit LoopInvariantNodeMotionOptimizer(GraphDef* optimized_graph)
51 : optimized_graph_(optimized_graph) {}
52 virtual ~LoopInvariantNodeMotionOptimizer() = default;
53 Status Optimize();
54
55 private:
56 Status FindInvariantNodes(NodeDef* node);
57 Status RevertInvariantNodes();
58 Status MoveInvariantNodes(const int frame_id);
59 Status HandleInvariantNode(NodeDef* node, const int num_outputs,
60 const int frame_id);
61 Status HandleConst(NodeDef* node, const int num_outputs, const int frame_id);
62 Status HandleInvariantEnter(NodeDef* node, const int num_outputs);
63
64 GraphDef* optimized_graph_; // Not owned.
65 std::unique_ptr<NodeMap> node_map_;
66 std::map<NodeDef*, int> invariant_nodes_;
67 std::set<int> empty_set_;
68 // TODO(rmlarsen): Use vector instead of map, since frames ids are dense.
69 std::map<int, std::set<int>> frame_children_;
70 std::map<int, int> frame_parent_;
71 std::map<int, const NodeDef*> loop_cond_;
72 std::map<int, std::vector<NodeDef*>> invariant_enters_;
73 int new_enter_id_;
74};

76Status LoopInvariantNodeMotionOptimizer::HandleInvariantEnter(
77 NodeDef* node, const int num_outputs) {
78 auto consumers = node_map_->GetOutputs(node->name());
79 std::vector<string> enter_control_inputs;
80 string enter_input;
81 for (auto& input : node->input()) {
82 if (IsControlInput(input)) {
83 enter_control_inputs.push_back(input);
84 } else {
85 enter_input = input;
86 }
87 }
88 for (auto* consumer : consumers) {
89 if (invariant_nodes_.count(consumer)) {
90 for (int i = 0; i < consumer->input_size(); ++i) {
91 if (NodeName(consumer->input(i)) == node->name()) {
92 consumer->set_input(i, enter_input);
93 node_map_->AddOutput(NodeName(enter_input), consumer->name());
94 node_map_->RemoveOutput(node->name(), consumer->name());
95 }
96 }
97 for (auto& control_input : enter_control_inputs) {
98 consumer->add_input(control_input);
99 node_map_->AddOutput(NodeName(control_input), consumer->name());
100 }
101 }
102 }
103 return Status::OK();
104}

106Status LoopInvariantNodeMotionOptimizer::HandleConst(NodeDef* node,
107 const int num_outputs,
108 const int frame_id) {
109 NodeDef* const_node = nullptr;
110 if (num_outputs == 0) {
111 // all successor nodes are invariant
112 // Remove the control inputs from this frame to the const node,
113 // when moving it out of the frame (in parent frame)
114 const_node = node;
115 node_map_->RemoveInputs(node->name());
116 node->clear_input();
117 } else {
118 // some successor nodes are variant
119 // Have to keep the const node in the frame,
120 // so create a new one outside the frame (in parent frame)
121 const string const_node_name =
122 AddPrefixToNodeName(node->name(), kLoopOptimizer);
123 const_node = node_map_->GetNode(const_node_name);
124 if (const_node == nullptr) {
125 const_node = optimized_graph_->add_node();
126 const_node->set_name(const_node_name);
127 const_node->set_op("Const");
128 const_node->set_device(node->device());
129 *const_node->mutable_attr() = node->attr();
130 node_map_->AddNode(const_node->name(), const_node);
131 }
132 auto consumers = node_map_->GetOutputs(node->name());
133 for (auto* consumer : consumers) {
134 if (invariant_nodes_.count(consumer)) {
135 for (int i = 0; i < consumer->input_size(); ++i) {
136 if (NodeName(consumer->input(i)) == node->name()) {
137 if (IsControlInput(consumer->input(i))) {
138 *consumer->mutable_input(i) = AsControlDependency(*const_node);
139 } else {
140 *consumer->mutable_input(i) = const_node->name();
141 }
142 node_map_->AddOutput(const_node->name(), consumer->name());
143 node_map_->RemoveOutput(node->name(), consumer->name());
144 }
145 }
146 }
147 }
148 }
149 // add a control input from the parent frame
150 auto parent_it = frame_parent_.find(frame_id);
151 if (parent_it != frame_parent_.end()) {
152 int parent_id = parent_it->second;
153 auto loop_cond_it = loop_cond_.find(parent_id);
154 if (loop_cond_it == loop_cond_.end()) {
155 return errors::InvalidArgument("Frame ", frame_id,
156 " doesn't have a LoopCond node");
157 }
158 auto& loop_cond_name = loop_cond_it->second->name();
159 NodeDef* switch_node = nullptr;
160 for (auto* node : node_map_->GetOutputs(loop_cond_name)) {
161 if (node->op() == "Switch") {
162 switch_node = node;
163 break;
164 }
165 }
166 if (!switch_node) {
167 return errors::InvalidArgument("LoopCond node of Frame ", frame_id,
168 " doesn't connect to any Switch node");
169 }
170 string switch_output = StrCat(switch_node->name(), ":1");
171 const string ctrl_dep = ConstantFolding::AddControlDependency(
172 switch_output, optimized_graph_, node_map_.get());
173 const_node->add_input(ctrl_dep);
174 node_map_->AddOutput(NodeName(ctrl_dep), const_node->name());
175 }
176 return Status::OK();
177}

179Status LoopInvariantNodeMotionOptimizer::HandleInvariantNode(
180 NodeDef* node, const int num_outputs, const int frame_id) {
181 // have to remove control inputs to the invariant node from the same frame
182 // when moving this node out of this frame
183 for (int i = 0; i < node->input_size(); ++i) {
184 if (IsControlInput(node->input(i))) {
185 node->mutable_input()->SwapElements(i, node->input_size() - 1);
186 node->mutable_input()->RemoveLast();
187 }
188 }
189 if (num_outputs == 0) {
190 return Status::OK();
191 }
192
193 DataTypeVector input_types;
194 DataTypeVector output_types;
195 OpRegistryInterface* op_registry = OpRegistry::Global();
196 const OpRegistrationData* op_reg_data = nullptr;
197 TF_RETURN_IF_ERROR(op_registry->LookUp(node->op(), &op_reg_data));
198 TF_RETURN_IF_ERROR(InOutTypesForNode(*node, op_reg_data->op_def, &input_types,
199 &output_types));
200
201 auto consumers = node_map_->GetOutputs(node->name());
202 string fname = invariant_enters_[frame_id][0]->attr().at("frame_name").s();
203 int piterations =
204 invariant_enters_[frame_id][0]->attr().at("parallel_iterations").i();
205 for (auto* consumer : consumers) {
206 if (!invariant_nodes_.count(consumer)) {
207 for (int i = 0; i < consumer->input_size(); ++i) {
208 int port;
209 string node_name = ParseNodeName(consumer->input(i), &port);
210 if (node_name != node->name()) {
211 continue;
212 }
213 if (port < 0) {
214 return errors::InvalidArgument(
215 "Invariant node should not have control outputs "
216 "to variant node");
217 }
218 DataType output_type = output_types[port];
219 NodeDef* new_enter = optimized_graph_->add_node();
220 new_enter->set_op("Enter");
221 new_enter->set_device(node->device());
222 new_enter->set_name(AddPrefixToNodeName(
223 StrCat(fname, "_enter_", new_enter_id_++), kLoopOptimizer));
224 AttrValue data_type;
225 data_type.set_type(output_type);
226 new_enter->mutable_attr()->insert({"T", data_type});
227 AttrValue frame_name;
228 frame_name.set_s(fname);
229 new_enter->mutable_attr()->insert({"frame_name", frame_name});
230 AttrValue is_const;
231 is_const.set_b(true);
232 new_enter->mutable_attr()->insert({"is_constant", is_const});
233 AttrValue parallel_iterations;
234 parallel_iterations.set_i(piterations);
235 new_enter->mutable_attr()->insert(
236 {"parallel_iterations", parallel_iterations});
237 new_enter->add_input(consumer->input(i));
238 *consumer->mutable_input(i) = new_enter->name();
239 node_map_->AddNode(new_enter->name(), new_enter);
240 node_map_->AddOutput(node->name(), new_enter->name());
241 node_map_->AddOutput(new_enter->name(), consumer->name());
242 }
243 }
244 }
245 return Status::OK();
246}

248Status LoopInvariantNodeMotionOptimizer::MoveInvariantNodes(
249 const int frame_id) {
250 for (auto iter = invariant_nodes_.begin(); iter != invariant_nodes_.end();
251 ++iter) {
252 auto* invariant_node = iter->first;
253 const int num_outputs = iter->second;
254 if (IsEnter(*invariant_node)) {
255 TF_RETURN_IF_ERROR(HandleInvariantEnter(invariant_node, num_outputs));
256 } else if (IsConstant(*invariant_node)) {
257 TF_RETURN_IF_ERROR(HandleConst(invariant_node, num_outputs, frame_id));
258 } else {
259 TF_RETURN_IF_ERROR(
260 HandleInvariantNode(invariant_node, num_outputs, frame_id));
261 }
262 }
263 return Status::OK();
264}

266Status LoopInvariantNodeMotionOptimizer::RevertInvariantNodes() {
267 std::deque<const NodeDef*> reverted_nodes;
268 for (auto iter = invariant_nodes_.begin(); iter != invariant_nodes_.end();) {
269 bool erased = false;
270 const auto* node = iter->first;
271 if (!IsConstant(*node) && !IsEnter(*node) && iter->second > 0) {
272 auto& consumers = node_map_->GetOutputs(node->name());
273 for (auto* consumer : consumers) {
274 if (!invariant_nodes_.count(consumer)) {
275 for (const auto& input : consumer->input()) {
276 if (IsControlInput(input) && NodeName(input) == node->name()) {
277 reverted_nodes.push_back(node);
278 invariant_nodes_.erase(iter++);
279 erased = true;
280 break;
281 }
282 }
283 if (erased) break;
284 }
285 }
286 }
287 if (!erased) ++iter;
288 }
289 while (!reverted_nodes.empty()) {
290 const auto* node = reverted_nodes.front();
291 reverted_nodes.pop_front();
292 std::set<NodeDef*> producers;
293 for (const auto& input : node->input()) {
294 auto* producer = node_map_->GetNode(input);
295 auto iter = invariant_nodes_.find(producer);
296 if (iter != invariant_nodes_.end()) {
297 if (IsControlInput(input) && !IsConstant(*producer) &&
298 !IsEnter(*producer)) {
299 reverted_nodes.push_back(producer);
300 invariant_nodes_.erase(iter);
301 } else {
302 producers.insert(producer);
303 }
304 }
305 }
306 for (auto* producer : producers) {
307 auto iter = invariant_nodes_.find(producer);
308 if (iter != invariant_nodes_.end()) {
309 ++iter->second;
310 }
311 }
312 for (auto* consumer : node_map_->GetOutputs(node->name())) {
313 auto iter = invariant_nodes_.find(consumer);
314 if (iter != invariant_nodes_.end()) {
315 reverted_nodes.push_back(consumer);
316 invariant_nodes_.erase(iter);
317 }
318 }
319 }
320 return Status::OK();
321}

323Status LoopInvariantNodeMotionOptimizer::FindInvariantNodes(NodeDef* node) {
324 auto consumers = node_map_->GetOutputs(node->name());
325 invariant_nodes_.insert(std::make_pair(node, consumers.size()));
326 for (auto* consumer : consumers) {
327 if (invariant_nodes_.count(consumer) || ModifiesFrameInfo(*consumer)) {
328 continue;
329 }
330 bool is_invariant = true;
331 for (const auto& input : consumer->input()) {
332 if (!IsControlInput(input)) {
333 const string name = NodeName(input);
334 auto* producer = node_map_->GetNode(name);
335 if (!invariant_nodes_.count(producer)) {
336 if (IsConstant(*producer)) {
337 invariant_nodes_.insert(
338 std::make_pair(producer, node_map_->GetOutputs(name).size()));
339 } else {
340 is_invariant = false;
341 break;
342 }
343 }
344 }
345 }
346 if (is_invariant) {
347 std::set<NodeDef*> producers;
348 for (const auto& input : consumer->input()) {
349 auto* producer = node_map_->GetNode(input);
350 producers.insert(producer);
351 }
352 for (auto* producer : producers) {
353 auto iter = invariant_nodes_.find(producer);
354 if (iter != invariant_nodes_.end()) {
355 --iter->second;
356 }
357 }
358 TF_RETURN_IF_ERROR(FindInvariantNodes(consumer));
359 }
360 }
361 return Status::OK();
362}

364Status LoopInvariantNodeMotionOptimizer::Optimize() {
365 node_map_.reset(new NodeMap(optimized_graph_));
366 FrameMap frame_map;
367 int num_frames;
368 TF_RETURN_IF_ERROR(IdentifyFramesWithNodeMap(*optimized_graph_, *node_map_,
369 &frame_map, &num_frames));
370 std::deque<int> worklist;
371 for (auto iter = frame_map.begin(); iter != frame_map.end(); ++iter) {
372 auto* node = iter->first;
373 auto& frame_ids = iter->second;
374 if (frame_ids.size() >= 3) {
375 for (unsigned int i = 1; i < frame_ids.size() - 1; ++i) {
376 frame_parent_[frame_ids[i]] = frame_ids[i - 1];
377 frame_children_[frame_ids[i]].insert(frame_ids[i + 1]);
378 }
379 }
380 if (frame_ids.size() >= 2) {
381 frame_children_[frame_ids[0]].insert(frame_ids[1]);
382 frame_parent_[frame_ids.back()] = frame_ids[frame_ids.size() - 2];
383 }
384 if (frame_ids.size() >= 1) {
385 frame_children_.insert(std::make_pair(frame_ids.back(), empty_set_));
386 if (node->op() == "LoopCond") {
387 if (loop_cond_.count(frame_ids.back())) {
388 return errors::InvalidArgument(
389 "Loop ", frame_ids.back(),
390 " has more than one LoopCond node: ", node->name(), " and ",
391 loop_cond_[frame_ids.back()]->name());
392 }
393 loop_cond_[frame_ids.back()] = node;
394 }
395 if (IsEnter(*node) && node->attr().at("is_constant").b()) {
396 invariant_enters_[frame_ids.back()].push_back(
397 const_cast<NodeDef*>(node));
398 }
399 }
400 }
401
402 for (auto it = frame_children_.begin(); it != frame_children_.end(); ++it) {
403 if (it->second.size() == 0) {
404 worklist.push_back(it->first);
405 }
406 }
407
408 while (!worklist.empty()) {
409 int frame_id = worklist.front();
410 new_enter_id_ = 0;
411 worklist.pop_front();
412 auto parent_it = frame_parent_.find(frame_id);
413 if (parent_it != frame_parent_.end()) {
414 int parent_id = parent_it->second;
415 frame_children_[parent_id].erase(frame_id);
416 if (frame_children_[parent_id].size() == 0) {
417 worklist.push_back(parent_id);
418 }
419 }
420
421 if (invariant_enters_[frame_id].empty()) {
422 continue;
423 }
424 invariant_nodes_.clear();
425 for (auto* enter : invariant_enters_[frame_id]) {
426 TF_RETURN_IF_ERROR(FindInvariantNodes(enter));
427 }
428
429 // revert invariant nodes that have control outputs to variant nodes
430 TF_RETURN_IF_ERROR(RevertInvariantNodes());
431
432 TF_RETURN_IF_ERROR(MoveInvariantNodes(frame_id));
433 }
434 return Status::OK();
435}

437std::vector<int> GetStackPushNodesToConvert(
438 const SimpleGraphView& graph_view,
439 const std::unordered_set<string>& nodes_to_preserve, int stack_node_idx) {
440 VLOG(1) << "Stack node: " << graph_view.graph()->node(stack_node_idx).name();
441 const std::unordered_set<string> op_types_to_traverse(
442 {"Stack", "StackV2", "Enter", "RefEnter", "Switch", "RefSwitch",
443 "Identity", "RefIdentity"});
444 std::vector<int> nodes_to_convert;
445 std::set<int> fanout;
446 graph_view.DepthFirstSearch(op_types_to_traverse, stack_node_idx, &fanout);
447 for (int fanout_idx : fanout) {
448 const NodeDef& fanout_node = graph_view.graph()->node(fanout_idx);
449 VLOG(1) << "Fanout " << fanout_idx << " : " << fanout_node.name();
450 if (IsStackPushOp(fanout_node)) {
451 nodes_to_convert.push_back(fanout_idx);
452 } else if (IsStackOp(fanout_node) || IsStackCloseOp(fanout_node) ||
453 op_types_to_traverse.find(fanout_node.op()) !=
454 op_types_to_traverse.end()) {
455 continue;
456 } else if (!IsStackPopOp(fanout_node) ||
457 (!graph_view.outputs(fanout_idx).empty() ||
458 nodes_to_preserve.find(fanout_node.name()) !=
459 nodes_to_preserve.end())) {
460 // The node is either a stack pop with consumers or something unexpected
461 // so we leave the graph alone.
462 nodes_to_convert.clear();
463 break;
464 }
465 }
466 return nodes_to_convert;
467}

469Status RemoveStackOps(const GrapplerItem& item, GraphDef* optimized_graph) {
470 const std::unordered_set<string> nodes_to_preserve = item.NodesToPreserve();
471 const GraphDef& graph = item.graph;
472 *optimized_graph = graph;
473 NodeMap node_map(optimized_graph);
474 SimpleGraphView graph_view;
475 TF_RETURN_IF_ERROR(graph_view.Initialize(graph));
476 for (int node_idx = 0; node_idx < graph.node_size(); ++node_idx) {
477 if (IsStackOp(graph.node(node_idx))) {
478 for (int push_node_idx : GetStackPushNodesToConvert(
479 graph_view, nodes_to_preserve, node_idx)) {
480 // We found push nodes without corresponding pops. Convert them to
481 // Identity passing the data through and add a control dependency from
482 // the op supplying the stack handle.
483 NodeDef* push_node = optimized_graph->mutable_node(push_node_idx);
484 VLOG(1) << "Converting " << push_node_idx << " : "
485 << push_node->DebugString();
486 if (push_node->attr().count("swap_memory") != 0) {
487 push_node->mutable_attr()->erase("swap_memory");
488 }
489 push_node->set_op("Identity");
490 push_node->mutable_input()->SwapElements(0, 1);
491 const string ctrl_dep = ConstantFolding::AddControlDependency(
492 push_node->input(1), optimized_graph, &node_map);
493 push_node->set_input(1, ctrl_dep);
494 VLOG(1) << "After converting: " << push_node->DebugString();
495 }
496 }
497 }
498 return Status::OK();
499}

503Status LoopOptimizer::Optimize(Cluster* cluster, const GrapplerItem& item,
504 GraphDef* optimized_graph) {
505 *optimized_graph = item.graph;
506 // Set up helper data structures.
507 if (options_.enable_loop_invariant_node_motion) {
508 LoopInvariantNodeMotionOptimizer linm_optimizer(optimized_graph);
509 TF_RETURN_IF_ERROR(linm_optimizer.Optimize());
510 }
511 if (options_.enable_stack_push_removal) {
512 TF_RETURN_IF_ERROR(RemoveStackOps(item, optimized_graph));
513 }
514
515 return Status::OK();
516}
