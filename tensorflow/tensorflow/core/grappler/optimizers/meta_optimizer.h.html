19#include "tensorflow/core/framework/device_base.h"

25namespace tensorflow {
26namespace grappler {
27
28// Run the other grappler optimizers based on the specified rewriter config.
29class MetaOptimizer : public GraphOptimizer {
30 public:
31 MetaOptimizer(DeviceBase* cpu_device, const RewriterConfig& cfg)
32 : cpu_device_(cpu_device), cfg_(cfg) {}
33 ~MetaOptimizer() override {}
34
35 string name() const override { return "meta_optimizer"; };
36
37 Status Optimize(Cluster* cluster, const GrapplerItem& item,
38 GraphDef* optimized_graph) override;
39
40 void PrintResult();
41
42 void Feedback(Cluster* cluster, const GrapplerItem& item,
43 const GraphDef& optimized_graph, double result) override;
44
45 private:
46 std::unique_ptr<GraphOptimizer> NewOptimizer(const string& optimizer);
47 DeviceBase* const cpu_device_; // may be NULL
48 RewriterConfig cfg_;
49 std::vector<std::pair<string, string>> result_;
50};
51
52bool MetaOptimizerEnabled(const RewriterConfig& cfg);
53
54// Run the meta optimizer.
55//
56// If <cpu_device> is non-null, it is the device to be used for executing ops
57// during constant folding; if NULL, a new device is created for doing constant
58// folding. For performance, it is recommended to pass in an existing cpu_device
59// when possible.
60Status RunMetaOptimizer(const GrapplerItem& item, const RewriterConfig& cfg,
61 DeviceBase* cpu_device, Cluster* cluster,
62 GraphDef* optimized_graph);
63
64} // namespace grappler
65} // namespace tensorflow
