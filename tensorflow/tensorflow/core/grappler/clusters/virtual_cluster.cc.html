48Status VirtualCluster::Run(const GraphDef& graph,
49 const std::vector<std::pair<string, Tensor>>& feed,
50 const std::vector<string>& fetch,
51 RunMetadata* metadata) {
52 // Initialize a virtual scheduler to process the graph. Make sure to use
53 // static shape inference to prevent the schedulrer from calling the Run
54 // method on the cluster, and create an infinite loop.
55 GrapplerItem item;
56 item.graph = graph;
57 item.feed = feed;
58 item.fetch = fetch;
59 VirtualScheduler scheduler(&item, true, this, node_manager_.get());
60 TF_RETURN_IF_ERROR(scheduler.Init());
61
62 if (metadata) {
63 metadata->clear_step_stats();
64 metadata->clear_cost_graph();
65 metadata->clear_partition_graphs();
66 }
67
68 Costs node_costs;
69 int node_id = 0;
70 do {
71 OpContext op_context = scheduler.GetCurrNode();
72 node_costs = node_estimator_->PredictCosts(op_context);
73 if (metadata) {
74 CostGraphDef::Node* cost_node =
75 metadata->mutable_cost_graph()->add_node();
76 const string& op_name = op_context.name;
77 cost_node->set_id(node_id++);
78 cost_node->set_name(op_name);
79 cost_node->set_device(op_context.device_name);
80 cost_node->set_compute_cost(
81 node_costs.execution_time.asMicroSeconds().count());
82 cost_node->set_compute_time(
83 node_costs.compute_time.asMicroSeconds().count());
84 cost_node->set_memory_time(
85 node_costs.memory_time.asMicroSeconds().count());
86 for (const auto& output : op_context.op_info.outputs()) {
87 auto output_info = cost_node->add_output_info();
88 output_info->set_dtype(output.dtype());
89 *output_info->mutable_shape() = output.shape();
90
91 int64 size = DataTypeSize(output.dtype());
92 for (const auto& dim : output.shape().dim()) {
93 size *= std::max<int64>(1, dim.size());
94 }
95 output_info->set_size(size);
96 }
97 }
98 } while (scheduler.MarkCurrNodeExecuted(node_costs));
99
100 if (metadata) {
101 scheduler.Summary(metadata);
102 }
103
104 const std::unordered_map<string, DeviceProperties>& device = GetDevices();
105 std::unordered_map<string, int64> peak_mem_usage =
106 scheduler.GetPeakMemoryUsage();
107 for (const auto& mem_usage : peak_mem_usage) {
108 const string& device_name = mem_usage.first;
109 auto it = device.find(device_name);
110 if (it == device.end()) {
111 // It's probably the fake send/recv device. Eventually we'll need to
112 // remove this fake device to ensure proper memory accounting for
113 // multi-device settings.
114 continue;
115 }
116 const DeviceProperties& dev = it->second;
117 if (dev.memory_size() <= 0) {
118 // Available device memory unknown
119 continue;
120 }
121 int64 peak_mem = mem_usage.second;
122 if (peak_mem >= dev.memory_size()) {
123 return errors::ResourceExhausted(
124 "Graph requires ", peak_mem, " bytes of memory on device ",
125 device_name, " to run ", " but device only has ", dev.memory_size(),
126 " available.");
127 }
128 }
129
130 return Status::OK();
131}
