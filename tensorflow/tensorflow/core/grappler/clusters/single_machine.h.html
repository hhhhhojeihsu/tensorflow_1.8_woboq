26namespace tensorflow {
27namespace grappler {
28
29// Create a simple cluster that makes available to grappler a subset of the
30// nodes available on a single local computer.
31class SingleMachine : public Cluster {
32 public:
33 SingleMachine(int timeout_s, int num_cpu_cores, int num_gpus);
34 ~SingleMachine() override;
35
36 string type() const override { return "single_machine"; }
37
38 Status Provision() override;
39 Status Shutdown() override;
40
41 Status Initialize(const GrapplerItem& item) override;
42 Status Run(const GraphDef& item,
43 const std::vector<std::pair<string, Tensor>>& feed,
44 const std::vector<string>& fetch, RunMetadata* metadata) override;
45
46 Status EnablePeakMemoryStats(bool enable) override;
47
48 // It requires EnableAllocatorStats(true) be called before Provision().
49 Status GetPeakMemoryUsage(
50 std::unordered_map<string, uint64>* device_peak_memory) const override;
51
52 private:
53 Status RunWithTimeout(const std::vector<std::pair<string, Tensor>>& feed,
54 const std::vector<string>& fetch,
55 RunMetadata* run_metadata);
56 Status RunWithTimeout(const std::vector<std::pair<string, Tensor>>& feed,
57 const std::vector<string>& fetch,
58 RunMetadata* run_metadata, int64 timeout_s);
59 Status ResetSession();
60 Status CloseSession(bool use_timeout);
61 Status ShutdownSession();
62 void MergeCosts(CostGraphDef* graph_costs, const CostGraphDef& init_costs,
63 const CostGraphDef& queue_costs);
64
65 Status ClearAllocatorStats() const;
66
67 std::unique_ptr<Session> session_;
68 std::vector<QueueRunnerDef> queue_runner_defs_;
69 string last_graph_id_;
70 mutex last_graph_mu_;
71 const GraphDef* last_graph_ GUARDED_BY(last_graph_mu_) = nullptr;
72 std::vector<string> init_ops_;
73 int64 expected_init_time_s_;
74 std::unique_ptr<Coordinator> coordinator_;
75 std::unique_ptr<thread::ThreadPool> thread_pool_;
76
77 RunMetadata init_metadata_;
78
79 mutex close_mu_;
80 bool closing_ GUARDED_BY(close_mu_);
81
82 bool cpu_allocator_stats_enabled_ = false;
83};
84
85} // end namespace grappler
86} // end namespace tensorflow
