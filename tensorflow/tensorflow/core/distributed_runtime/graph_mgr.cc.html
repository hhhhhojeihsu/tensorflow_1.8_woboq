52GraphMgr::GraphMgr(const WorkerEnv* worker_env, DeviceMgr* device_mgr)
53 : worker_env_(worker_env), device_mgr_(device_mgr), table_(5) {
54 // The default value of sync_on_finish will be flipped soon and this
55 // environment variable will be removed as well.
56 Status status =
57 ReadBoolFromEnvVar("TF_SYNC_ON_FINISH", true, &sync_on_finish_);
58 if (!status.ok()) {
59 LOG(ERROR) << status.error_message();
60 }
61}

63GraphMgr::~GraphMgr() {
64 for (auto p : table_) p.second->Unref();

67GraphMgr::Item::~Item() {
68 for (const auto& unit : this->units) {
69 CHECK_NOTNULL(unit.device);
70 if (!graph_mgr->skip_cost_models_) {
71 graph_mgr->cost_model_manager_.RemoveCostModelForGraph(unit.graph);
72 }
73 delete unit.root;
74 unit.device->op_segment()->RemoveHold(this->session);
75 }
76}

78// NOTE: node->device_name() is not set by GraphConstructor. We
79// expects that NodeDef in GraphDef given to workers fully specifies
80// device names.
81static string SplitByDevice(const Node* node) {
82 return node->assigned_device_name();
83}
84
85// Validates "gdef" device specifications.
86static Status ValidateGraphDefForDevices(const GraphDef& gdef) {
87 DeviceNameUtils::ParsedName parsed;
88 for (const auto& ndef : gdef.node()) {
89 if (!DeviceNameUtils::ParseFullName(ndef.device(), &parsed)) {
90 return errors::InvalidArgument("Missing device name in: ",
91 SummarizeNodeDef(ndef));
92 }
93 }
94 return Status::OK();
95}

97Status GraphMgr::DecorateAndPublishGraphForDebug(
98 const DebugOptions& debug_options, Graph* graph, Device* device) {
99 std::unique_ptr<DebugGraphDecoratorInterface> decorator;
100 TF_RETURN_IF_ERROR(
101 DebugGraphDecoratorRegistry::CreateDecorator(debug_options, &decorator));
102 TF_RETURN_IF_ERROR(decorator->DecorateGraph(graph, device));
103 TF_RETURN_IF_ERROR(decorator->PublishGraph(*graph, device->name()));
104 return Status::OK();
105}

107// Creates executors given a graph definition "gdef" of a "session".
108// If a node in "gdef" is shared by other graphs in "session", the
109// same op kernel is reused. E.g., typically a params node is shared
110// by multiple graphs in a session.
111//
112// If "gdef" is assigned to multiple devices, extra nodes (e.g.,
113// send/recv nodes) maybe added. The extra nodes' name are generated
114// by calling "new_name(old_name)".
115//
116// "executors" are filled with one executor per device if success and
117// the caller takes the ownership of returned executors.
118Status GraphMgr::InitItem(const string& session, const GraphDef& gdef,
119 const GraphOptions& graph_options,
120 const DebugOptions& debug_options,
121 DistributedFunctionLibraryRuntime* cluster_flr,
122 Item* item) {
123 item->session = session;
124 item->lib_def.reset(
125 new FunctionLibraryDefinition(OpRegistry::Global(), gdef.library()));
126
127 TF_RETURN_IF_ERROR(ValidateGraphDefForDevices(gdef));
128
129 if (gdef.versions().producer() >= 5) {
130 // Validate the graph: we assume that merging two valid graphs
131 // should maintain graph validity.
132 TF_RETURN_IF_ERROR(graph::ValidateGraphDef(gdef, *item->lib_def));
133 }
134
135 item->proc_flr.reset(new ProcessFunctionLibraryRuntime(
136 device_mgr_, worker_env_->env, gdef.versions().producer(),
137 item->lib_def.get(), graph_options.optimizer_options(),
138 worker_env_->compute_pool, cluster_flr));
139
140 // Constructs the graph out of "gdef".
141 Graph graph(OpRegistry::Global());
142 GraphConstructorOptions opts;
143 opts.allow_internal_ops = true;
144 opts.expect_device_spec = true;
145 TF_RETURN_IF_ERROR(ConvertGraphDefToGraph(opts, gdef, &graph));
146
147 // Splits "graph" into multiple subgraphs by device names.
148 std::unordered_map<string, GraphDef> partitions;
149 PartitionOptions popts;
150 popts.node_to_loc = SplitByDevice;
151 popts.new_name = [this](const string& prefix) {
152 mutex_lock l(mu_);
153 return strings::StrCat(prefix, "_G", next_id_++);
154 };
155 popts.get_incarnation = [this](const string& name) -> int64 {
156 Device* device = nullptr;
157 Status s = device_mgr_->LookupDevice(name, &device);
158 if (s.ok()) {
159 return device->attributes().incarnation();
160 } else {
161 return PartitionOptions::kIllegalIncarnation;
162 }
163 };
164 popts.flib_def = &graph.flib_def();
165 popts.control_flow_added = true;
166 popts.scheduling_for_recvs = graph_options.enable_recv_scheduling();
167 TF_RETURN_IF_ERROR(Partition(popts, &graph, &partitions));
168 if (popts.scheduling_for_recvs) {
169 TF_RETURN_IF_ERROR(AddControlEdges(popts, &partitions));
170 }
171
172 std::unordered_map<string, std::unique_ptr<Graph>> partition_graphs;
173 for (const auto& partition : partitions) {
174 std::unique_ptr<Graph> device_graph(new Graph(OpRegistry::Global()));
175 GraphConstructorOptions device_opts;
176 // There are internal operations (e.g., send/recv) that we now allow.
177 device_opts.allow_internal_ops = true;
178 device_opts.expect_device_spec = true;
179 TF_RETURN_IF_ERROR(ConvertGraphDefToGraph(device_opts, partition.second,
180 device_graph.get()));
181 partition_graphs.emplace(partition.first, std::move(device_graph));
182 }
183
184 GraphOptimizationPassOptions optimization_options;
185 optimization_options.flib_def = item->lib_def.get();
186 optimization_options.partition_graphs = &partition_graphs;
187 TF_RETURN_IF_ERROR(OptimizationPassRegistry::Global()->RunGrouping(
188 OptimizationPassRegistry::POST_PARTITIONING, optimization_options));
189
190 LocalExecutorParams params;
191
192 item->units.reserve(partitions.size());
193 item->graph_mgr = this;
194 const auto& optimizer_opts = graph_options.optimizer_options();
195 GraphOptimizer optimizer(optimizer_opts);
196 for (auto& p : partition_graphs) {
197 const string& device_name = p.first;
198 std::unique_ptr<Graph>& subgraph = p.second;
199 item->units.resize(item->units.size() + 1);
200 ExecutionUnit* unit = &(item->units.back());
201
202 // Find the device.
203 Status s = device_mgr_->LookupDevice(device_name, &unit->device);
204 if (!s.ok()) {
205 // Remove the empty unit from the item as the item destructor wants all
206 // units to have valid devices.
207 item->units.pop_back();
208 return s;
209 }
210
211 // Give the device an opportunity to rewrite its subgraph.
212 TF_RETURN_IF_ERROR(unit->device->MaybeRewriteGraph(&subgraph));
213
214 // Top-level nodes in the graph uses the op segment to cache
215 // kernels. Therefore, as long as the executor is alive, we need
216 // to ensure the kernels cached for the session are alive.
217 auto opseg = unit->device->op_segment();
218 opseg->AddHold(session);
219
220 // Function library runtime.
221 FunctionLibraryRuntime* lib = item->proc_flr->GetFLR(unit->device->name());
222 if (lib == nullptr) {
223 return errors::InvalidArgument("Cannot find FLR for device: ",
224 unit->device->name());
225 }
226
227 // Construct the root executor for the subgraph.
228 params.device = unit->device;
229 params.function_library = lib;
230 params.create_kernel = [session, lib, opseg](const NodeDef& ndef,
231 OpKernel** kernel) {
232 // We do not share the kernel via the OpSegment if the node is
233 // stateless, or a function.
234 // NOTE(mrry): We must not share function kernels (implemented
235 // using `CallOp`) between subgraphs, because `CallOp::handle_`
236 // is tied to a particular subgraph. Even if the function itself
237 // is stateful, the `CallOp` that invokes it is not.
238 if (!lib->IsStateful(ndef.op()) ||
239 lib->GetFunctionLibraryDefinition()->Find(ndef.op()) != nullptr) {
240 return lib->CreateKernel(ndef, kernel);
241 }
242 auto create_fn = [lib, &ndef](OpKernel** kernel) {
243 return lib->CreateKernel(ndef, kernel);
244 };
245 // Kernels created for subgraph nodes need to be cached. On
246 // cache miss, create_fn() is invoked to create a kernel based
247 // on the function library here + global op registry.
248 return opseg->FindOrCreate(session, ndef.name(), kernel, create_fn);
249 };
250 params.delete_kernel = [lib](OpKernel* kernel) {
251 // If the node is stateful, opseg owns it. Otherwise, delete it.
252 if (kernel && !lib->IsStateful(kernel->type_string())) {
253 delete kernel;
254 }
255 };
256
257 optimizer.Optimize(lib, worker_env_->env, params.device, &subgraph,
258 /*shape_map=*/nullptr);
259
260 // EXPERIMENTAL: tfdbg inserts debug nodes (i.e., probes) to the graph.
261 if (!debug_options.debug_tensor_watch_opts().empty()) {
262 TF_RETURN_IF_ERROR(DecorateAndPublishGraphForDebug(
263 debug_options, subgraph.get(), params.device));
264 }
265
266 TF_RETURN_IF_ERROR(
267 EnsureMemoryTypes(DeviceType(unit->device->device_type()),
268 unit->device->name(), subgraph.get()));
269 unit->graph = subgraph.get();
270 unit->build_cost_model = graph_options.build_cost_model();
271 if (unit->build_cost_model > 0) {
272 skip_cost_models_ = false;
273 }
274 TF_RETURN_IF_ERROR(
275 NewLocalExecutor(params, std::move(subgraph), &unit->root));
276 }
277 return Status::OK();
278}

280Status GraphMgr::Register(const string& session, const GraphDef& gdef,
281 const GraphOptions& graph_options,
282 const DebugOptions& debug_options,
283 DistributedFunctionLibraryRuntime* cluster_flr,
284 string* handle) {
285 Item* item = new Item;
286 Status s =
287 InitItem(session, gdef, graph_options, debug_options, cluster_flr, item);
288 if (!s.ok()) {
289 item->Unref();
290 return s;
291 }
292
293 // Inserts one item into table_.
294 {
295 mutex_lock l(mu_);
296 *handle = strings::Printf("%016llx", ++next_id_);
297 item->handle = *handle;
298 CHECK(table_.insert({*handle, item}).second);
299 }
300 return Status::OK();
301}
302
303Status GraphMgr::Deregister(const string& handle) {
304 Item* item = nullptr;
305 // Removes one item from table_.
306 {
307 mutex_lock l(mu_);
308 auto iter = table_.find(handle);
309 if (iter == table_.end()) {
310 return errors::Aborted("Graph handle is not found: ", handle,
311 ". Possibly, this worker just restarted.");
312 }
313 item = iter->second;
314 table_.erase(iter);
315 }
316 item->Unref();
317 return Status::OK();
318}
319
320Status GraphMgr::DeregisterAll() {
321 std::vector<Item*> items;
322 // Removes all items from table_.
323 {
324 mutex_lock l(mu_);
325 for (const auto& entry : table_) {
326 items.push_back(entry.second);
327 }
328 table_.clear();
329 }
330 for (auto item : items) {
331 item->Unref();
332 }
333 return Status::OK();
334}

336Status GraphMgr::SendInputs(const int64 step_id, const NamedTensors& in) {
337 Rendezvous* rendezvous = worker_env_->rendezvous_mgr->Find(step_id);
338 std::vector<string> keys;
339 std::vector<Tensor> tensors_to_send;
340 keys.reserve(in.size());
341 tensors_to_send.reserve(in.size());
342 for (const auto& p : in) {
343 keys.push_back(p.first);
344 tensors_to_send.push_back(p.second);
345 }
346 Status s =
347 SendTensorsToRendezvous(rendezvous, nullptr, {}, keys, tensors_to_send);
348 rendezvous->Unref();
349 return s;
350}

352Status GraphMgr::RecvOutputs(const int64 step_id, NamedTensors* out) {
353 Rendezvous* rendezvous = worker_env_->rendezvous_mgr->Find(step_id);
354 Status s = RecvOutputsFromRendezvous(rendezvous, out, Rendezvous::Args());
355 rendezvous->Unref();
356 return s;

359void GraphMgr::RecvOutputsAsync(const int64 step_id, NamedTensors* out,
360 StatusCallback done) {
361 Rendezvous* rendezvous = worker_env_->rendezvous_mgr->Find(step_id);
362 std::vector<string> keys;
363 std::vector<Tensor>* received_keys = new std::vector<Tensor>;
364 keys.reserve(out->size());
365 received_keys->reserve(out->size());
366 for (const auto& p : *out) {
367 keys.push_back(p.first);
368 received_keys->push_back(p.second);
369 }
370 RecvOutputsFromRendezvousAsync(
371 rendezvous, nullptr, {}, keys, received_keys,
372 [done, rendezvous, received_keys, out, keys](const Status s) {
373 rendezvous->Unref();
374 for (int i = 0; i < keys.size(); ++i) {
375 (*out)[keys[i]] = (*received_keys)[i];
376 }
377 delete received_keys;
378 done(s);
379 });
380}

382void GraphMgr::ExecuteAsync(const string& handle, const int64 step_id,
383 WorkerSession* session, const ExecutorOpts& opts,
384 StepStatsCollector* collector,
385 MutableRunGraphResponseWrapper* response,
386 CancellationManager* cancellation_manager,
387 const NamedTensors& in, StatusCallback done) {
388 // Lookup an item. Holds one ref while executing.
389 Item* item = nullptr;
390 {
391 mutex_lock l(mu_);
392 auto iter = table_.find(handle);
393 if (iter != table_.end()) {
394 item = iter->second;
395 item->Ref();
396 }
397 }
398
399 if (item == nullptr) {
400 done(errors::Aborted("Graph handle is not found: ", handle));
401 return;
402 }
403
404 CostGraphDef* cost_graph = nullptr;
405 if (response != nullptr) {
406 cost_graph = response->mutable_cost_graph();
407 if (opts.record_partition_graphs()) {
408 for (const ExecutionUnit& unit : item->units) {
409 GraphDef graph_def;
410 unit.graph->ToGraphDef(&graph_def);
411 response->AddPartitionGraph(graph_def);
412 }
413 }
414 }
415
416 RemoteRendezvous* rendezvous = worker_env_->rendezvous_mgr->Find(step_id);
417 Status s = rendezvous->Initialize(session);
418
419 // Sends values specified by the caller.
420 if (s.ok()) {
421 std::vector<string> keys;
422 std::vector<Tensor> tensors_to_send;
423 keys.reserve(in.size());
424 tensors_to_send.reserve(in.size());
425 for (auto& p : in) {
426 keys.push_back(p.first);
427 tensors_to_send.push_back(p.second);
428 }
429 s = SendTensorsToRendezvous(rendezvous, nullptr, {}, keys, tensors_to_send);
430 }
431
432 if (!s.ok()) {
433 done(s);
434 item->Unref();
435 rendezvous->Unref();
436 return;
437 }
438
439 StartParallelExecutors(handle, step_id, item, rendezvous, collector,
440 cost_graph, cancellation_manager,
441 [item, rendezvous, done](const Status& s) {
442 done(s);
443 rendezvous->Unref();
444 item->Unref();
445 });
446}

448void GraphMgr::StartParallelExecutors(const string& handle, int64 step_id,
449 Item* item, Rendezvous* rendezvous,
450 StepStatsCollector* collector,
451 CostGraphDef* cost_graph,
452 CancellationManager* cancellation_manager,
453 StatusCallback done) {
454 const int num_units = item->units.size();
455 CHECK_GE(num_units, 1);
456 ScopedStepContainer* step_container = new ScopedStepContainer(
457 step_id,
458 [this](const string& name) { device_mgr_->ClearContainers({name}); });
459 // NOTE: Transfer one ref of rendezvous and item.
460 ExecutorBarrier* barrier =
461 new ExecutorBarrier(num_units, rendezvous,
462 [this, item, collector, cost_graph, step_container,
463 done](const Status& s) {
464 BuildCostModel(item, collector, cost_graph);
465 done(s);
466 delete step_container;
467 });
468 Executor::Args args;
469 {
470 mutex_lock l(mu_);
471 args.step_id = ++next_id_;
472 }
473 args.rendezvous = rendezvous;
474 args.cancellation_manager = cancellation_manager;
475 args.stats_collector = collector;
476 args.step_container = step_container;
477 args.sync_on_finish = sync_on_finish_;
478 if (LogMemory::IsEnabled()) {
479 LogMemory::RecordStep(args.step_id, handle);
480 }
481 thread::ThreadPool* pool = worker_env_->compute_pool;
482 using std::placeholders::_1;
483 // Line below is equivalent to this code, but does one less indirect call:
484 // args.runner = [pool](std::function<void()> fn) { pool->Schedule(fn); };
485 auto default_runner = std::bind(&thread::ThreadPool::Schedule, pool, _1);
486 for (const auto& unit : item->units) {
487 // TODO(zhengxq): if the device picks its own threadpool, we need to assign
488 // less threads to the main compute pool by default.
489 thread::ThreadPool* device_thread_pool =
490 unit.device->tensorflow_device_thread_pool();
491 if (!device_thread_pool) {
492 args.runner = default_runner;
493 } else {
494 args.runner =
495 std::bind(&thread::ThreadPool::Schedule, device_thread_pool, _1);
496 }
497 unit.root->RunAsync(args, barrier->Get());
498 }
499}

501void GraphMgr::BuildCostModel(Item* item, StepStatsCollector* collector,
502 CostGraphDef* cost_graph) {
503 if (collector && !skip_cost_models_) {
504 // Build the cost model
505 std::unordered_map<string, const Graph*> device_to_graph;
506 for (const auto& unit : item->units) {
507 if (unit.build_cost_model > 0) {
508 device_to_graph[unit.device->name()] = unit.graph;
509 }
510 }
511 collector->BuildCostModel(&cost_model_manager_, device_to_graph);
512
513 if (cost_graph != nullptr) {
514 for (const auto& unit : item->units) {
515 cost_model_manager_.AddToCostGraphDef(unit.graph, cost_graph)
516 .IgnoreError();
517 }
518 }
519 }
