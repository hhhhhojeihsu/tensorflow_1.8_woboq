27// Represents a pending asynchronous client call as a tag that can be
28// stored in a `grpc::CompletionQueue`.

30 public:
31 GrpcClientCQTag() {}

33

35 // Implementations of OnCompleted must delete *this.
36 virtual void OnCompleted(bool ok) = 0;
37
38 private:
39 TF_DISALLOW_COPY_AND_ASSIGN(GrpcClientCQTag);
40};
