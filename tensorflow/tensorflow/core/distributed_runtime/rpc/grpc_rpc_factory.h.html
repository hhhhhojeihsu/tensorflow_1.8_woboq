25namespace tensorflow {
26
27class GrpcRPCFactory : public RPCFactory {
28 public:
29 explicit GrpcRPCFactory(OpKernelConstruction* ctx, bool fail_fast,
30 int64 timeout_in_ms);
31
32 // Explicit destructor to control destruction order.
33 ~GrpcRPCFactory() override;
34
35 void Call(OpKernelContext* ctx, int64 num_elements, const Tensor& address_t,
36 const Tensor& method_t, const Tensor& request_t, const bool try_rpc,
37 Tensor* response_t, Tensor* status_code_t, Tensor* status_message_t,
38 AsyncOpKernel::DoneCallback done) override;
39
40 protected:
41 typedef std::shared_ptr<::grpc::Channel> ChannelPtr;
42 virtual ChannelPtr CreateChannelForAddress(const string& address);
43
44 private:
45 ::grpc::GenericStub* GetOrCreateStubForAddress(const string& address);
46
47 bool fail_fast_;
48 int64 timeout_in_ms_;
49 ::grpc::CompletionQueue completion_queue_;
50 Thread* polling_thread_; // Owned.
51
52 mutex mu_;
53 typedef std::unique_ptr<::grpc::GenericStub> StubPtr;
54 std::unordered_map<string, StubPtr> stubs_ GUARDED_BY(mu_);
55};
56
57} // namespace tensorflow
