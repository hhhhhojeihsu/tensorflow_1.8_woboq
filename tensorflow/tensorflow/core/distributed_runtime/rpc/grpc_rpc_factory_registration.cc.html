20namespace tensorflow {
21namespace {
22
23// Used for adding the grpc factory to the RPC factory registry.
24struct Value {
25 static RPCFactory* Function(OpKernelConstruction* ctx, bool fail_fast,
26 int64 timeout_in_ms) {
27 return new GrpcRPCFactory(ctx, fail_fast, timeout_in_ms);
28 }
29};
30
31REGISTER_RPC_FACTORY("grpc", Value::Function);
