22RandomAccessInputStream::RandomAccessInputStream(RandomAccessFile* file,
23 bool owns_file)
24 : file_(file), owns_file_(owns_file) {}

26RandomAccessInputStream::~RandomAccessInputStream() {
27 if (owns_file_) {
28 delete file_;
29 }
30}

32Status RandomAccessInputStream::ReadNBytes(int64 bytes_to_read,
33 string* result) {
34 if (bytes_to_read < 0) {
35 return errors::InvalidArgument("Cannot read negative number of bytes");
36 }
37 result->clear();
38 result->resize(bytes_to_read);
39 char* result_buffer = &(*result)[0];
40 StringPiece data;
41 Status s = file_->Read(pos_, bytes_to_read, &data, result_buffer);
42 if (data.data() != result_buffer) {
43 memmove(result_buffer, data.data(), data.size());
44 }
45 result->resize(data.size());
46 if (s.ok() || errors::IsOutOfRange(s)) {
47 pos_ += data.size();
48 } else {
49 return s;
50 }
51 // If the amount of data we read is less than what we wanted, we return an
52 // out of range error. We need to catch this explicitly since file_->Read()
53 // would not do so if at least 1 byte is read (b/30839063).
54 if (data.size() < bytes_to_read) {
55 return errors::OutOfRange("reached end of file");
56 }
57 return Status::OK();
58}

60// To limit memory usage, the default implementation of SkipNBytes() only reads
61// 8MB at a time.
62static constexpr int64 kMaxSkipSize = 8 * 1024 * 1024;
63
64Status RandomAccessInputStream::SkipNBytes(int64 bytes_to_skip) {
65 if (bytes_to_skip < 0) {
66 return errors::InvalidArgument("Can't skip a negative number of bytes");
67 }
68 std::unique_ptr<char[]> scratch(new char[kMaxSkipSize]);
69 // Try to read 1 bytes first, if we could complete the read then EOF is
70 // not reached yet and we could return.
71 if (bytes_to_skip > 0) {
72 StringPiece data;
73 Status s = file_->Read(pos_ + bytes_to_skip - 1, 1, &data, scratch.get());
74 if ((s.ok() || errors::IsOutOfRange(s)) && data.size() == 1) {
75 pos_ += bytes_to_skip;
76 return Status::OK();
77 }
78 }
79 // Read kDefaultSkipSize at a time till bytes_to_skip.
80 while (bytes_to_skip > 0) {
81 int64 bytes_to_read = std::min<int64>(kMaxSkipSize, bytes_to_skip);
82 StringPiece data;
83 Status s = file_->Read(pos_, bytes_to_read, &data, scratch.get());
84 if (s.ok() || errors::IsOutOfRange(s)) {
85 pos_ += data.size();
86 } else {
87 return s;
88 }
89 if (data.size() < bytes_to_read) {
90 return errors::OutOfRange("reached end of file");
91 }
92 bytes_to_skip -= bytes_to_read;
93 }
94 return Status::OK();
95}

97int64 RandomAccessInputStream::Tell() const { return pos_; }
