24// Return a new two level iterator. A two-level iterator contains an
25// index iterator whose values point to a sequence of blocks where
26// each block is itself a sequence of key,value pairs. The returned
27// two-level iterator yields the concatenation of all key/value pairs
28// in the sequence of blocks. Takes ownership of "index_iter" and
29// will delete it when no longer needed.
30//
31// Uses a supplied function to convert an index_iter value into
32// an iterator over the contents of the corresponding block.
33extern Iterator* NewTwoLevelIterator(
34 Iterator* index_iter,
35 Iterator* (*block_function)(void* arg, const StringPiece& index_value),
36 void* arg);
37
38} // namespace table
