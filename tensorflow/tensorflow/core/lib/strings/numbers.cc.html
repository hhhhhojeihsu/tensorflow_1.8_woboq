36template <typename T>
37T locale_independent_strtonum(const char* str, const char** endptr) {
38 static const std::unordered_map<string, T> special_nums = {
39 {"inf", std::numeric_limits<T>::infinity()},
40 {"+inf", std::numeric_limits<T>::infinity()},
41 {"-inf", -std::numeric_limits<T>::infinity()},
42 {"infinity", std::numeric_limits<T>::infinity()},
43 {"+infinity", std::numeric_limits<T>::infinity()},
44 {"-infinity", -std::numeric_limits<T>::infinity()},
45 {"nan", std::numeric_limits<T>::quiet_NaN()},
46 {"+nan", std::numeric_limits<T>::quiet_NaN()},
47 {"-nan", -std::numeric_limits<T>::quiet_NaN()},
48 };
49 std::stringstream s(str);
50
51 // Check if str is one of the special numbers.
52 string special_num_str;
53 s >> special_num_str;
54
55 for (int i = 0; i < special_num_str.length(); ++i) {
56 special_num_str[i] =
57 std::tolower(special_num_str[i], std::locale::classic());
58 }
59
60 auto entry = special_nums.find(special_num_str);
61 if (entry != special_nums.end()) {
62 *endptr = str + (s.eof() ? static_cast<std::iostream::pos_type>(strlen(str))
63 : s.tellg());
64 return entry->second;
65 } else {
66 // Perhaps it's a hex number
67 if (special_num_str.compare(0, 2, "0x") == 0 ||
68 special_num_str.compare(0, 3, "-0x") == 0) {
69 return strtol(str, const_cast<char**>(endptr), 16);
70 }
71 }
72 // Reset the stream
73 s.str(str);
74 s.clear();
75 // Use the "C" locale
76 s.imbue(std::locale::classic());
77
78 T result;
79 s >> result;
80
81 // Set to result to what strto{f,d} functions would have returned. If the
82 // number was outside the range, the stringstream sets the fail flag, but
83 // returns the +/-max() value, whereas strto{f,d} functions return +/-INF.
84 if (s.fail()) {
85 if (result == std::numeric_limits<T>::max() ||
86 result == std::numeric_limits<T>::infinity()) {
87 result = std::numeric_limits<T>::infinity();
88 s.clear(s.rdstate() & ~std::ios::failbit);
89 } else if (result == -std::numeric_limits<T>::max() ||
90 result == -std::numeric_limits<T>::infinity()) {
91 result = -std::numeric_limits<T>::infinity();
92 s.clear(s.rdstate() & ~std::ios::failbit);
93 }
94 }
95
96 if (endptr) {
97 *endptr =
98 str +
99 (s.fail() ? static_cast<std::iostream::pos_type>(0)
100 : (s.eof() ? static_cast<std::iostream::pos_type>(strlen(str))
101 : s.tellg()));
102 }
103 return result;
104}

110size_t FastInt32ToBufferLeft(int32 i, char* buffer) {
111 uint32 u = i;
112 size_t length = 0;
113 if (i < 0) {
114 *buffer++ = '-';
115 ++length;
116 // We need to do the negation in modular (i.e., "unsigned")
117 // arithmetic; MSVC++ apparently warns for plain "-u", so
118 // we write the equivalent expression "0 - u" instead.
119 u = 0 - u;
120 }
121 length += FastUInt32ToBufferLeft(u, buffer);
122 return length;
123}
124
125size_t FastUInt32ToBufferLeft(uint32 i, char* buffer) {
126 char* start = buffer;
127 do {
128 *buffer++ = ((i % 10) + '0');
129 i /= 10;
130 } while (i > 0);
131 *buffer = 0;
132 std::reverse(start, buffer);
133 return buffer - start;
134}

136size_t FastInt64ToBufferLeft(int64 i, char* buffer) {
137 uint64 u = i;
138 size_t length = 0;
139 if (i < 0) {
140 *buffer++ = '-';
141 ++length;
142 u = 0 - u;
143 }
144 length += FastUInt64ToBufferLeft(u, buffer);
145 return length;
146}

159static const double kDoublePrecisionCheckMax = DBL_MAX / 1.000000000000001;
160
161size_t DoubleToBuffer(double value, char* buffer) {
162 // DBL_DIG is 15 for IEEE-754 doubles, which are used on almost all
163 // platforms these days. Just in case some system exists where DBL_DIG
164 // is significantly larger -- and risks overflowing our buffer -- we have
165 // this assert.
166 static_assert(DBL_DIG < 20, "DBL_DIG is too big");
167
168 if (std::abs(value) <= kDoublePrecisionCheckMax) {
169 int snprintf_result =
170 snprintf(buffer, kFastToBufferSize, "%.*g", DBL_DIG, value);
171
172 // The snprintf should never overflow because the buffer is significantly
173 // larger than the precision we asked for.
174 DCHECK(snprintf_result > 0 && snprintf_result < kFastToBufferSize);
175
176 if (locale_independent_strtonum<double>(buffer, nullptr) == value) {
177 // Round-tripping the string to double works; we're done.
178 return snprintf_result;
179 }
180 // else: full precision formatting needed. Fall through.
181 }
182
183 int snprintf_result =
184 snprintf(buffer, kFastToBufferSize, "%.*g", DBL_DIG + 2, value);
185
186 // Should never overflow; see above.
187 DCHECK(snprintf_result > 0 && snprintf_result < kFastToBufferSize);
188
189 return snprintf_result;
190}

193char SafeFirstChar(StringPiece str) {
194 if (str.empty()) return '\0';
195 return str[0];
196}
197void SkipSpaces(StringPiece* str) {
198 while (isspace(SafeFirstChar(*str))) str->remove_prefix(1);
199}
200} // namespace

202bool safe_strto64(StringPiece str, int64* value) {
203 SkipSpaces(&str);
204
205 int64 vlimit = kint64max;
206 int sign = 1;
207 if (str_util::ConsumePrefix(&str, "-")) {
208 sign = -1;
209 // Different limit for positive and negative integers.
210 vlimit = kint64min;
211 }
212
213 if (!isdigit(SafeFirstChar(str))) return false;
214
215 int64 result = 0;
216 if (sign == 1) {
217 do {
218 int digit = SafeFirstChar(str) - '0';
219 if ((vlimit - digit) / 10 < result) {
220 return false;
221 }
222 result = result * 10 + digit;
223 str.remove_prefix(1);
224 } while (isdigit(SafeFirstChar(str)));
225 } else {
226 do {
227 int digit = SafeFirstChar(str) - '0';
228 if ((vlimit + digit) / 10 > result) {
229 return false;
230 }
231 result = result * 10 - digit;
232 str.remove_prefix(1);
233 } while (isdigit(SafeFirstChar(str)));
234 }
235
236 SkipSpaces(&str);
237 if (!str.empty()) return false;
238
239 *value = result;
240 return true;
241}

264bool safe_strto32(StringPiece str, int32* value) {
265 SkipSpaces(&str);
266
267 int64 vmax = kint32max;
268 int sign = 1;
269 if (str_util::ConsumePrefix(&str, "-")) {
270 sign = -1;
271 // Different max for positive and negative integers.
272 ++vmax;
273 }
274
275 if (!isdigit(SafeFirstChar(str))) return false;
276
277 int64 result = 0;
278 do {
279 result = result * 10 + SafeFirstChar(str) - '0';
280 if (result > vmax) {
281 return false;
282 }
283 str.remove_prefix(1);
284 } while (isdigit(SafeFirstChar(str)));
285
286 SkipSpaces(&str);
287
288 if (!str.empty()) return false;
289
290 *value = static_cast<int32>(result * sign);
291 return true;
292}

314bool safe_strtof(const char* str, float* value) {
315 const char* endptr;
316 *value = locale_independent_strtonum<float>(str, &endptr);
317 while (isspace(*endptr)) ++endptr;
318 // Ignore range errors from strtod/strtof.
319 // The values it returns on underflow and
320 // overflow are the right fallback in a
321 // robust setting.
322 return *str != '\0' && *endptr == '\0';
323}

336size_t FloatToBuffer(float value, char* buffer) {
337 // FLT_DIG is 6 for IEEE-754 floats, which are used on almost all
338 // platforms these days. Just in case some system exists where FLT_DIG
339 // is significantly larger -- and risks overflowing our buffer -- we have
340 // this assert.
341 static_assert(FLT_DIG < 10, "FLT_DIG is too big");
342
343 int snprintf_result =
344 snprintf(buffer, kFastToBufferSize, "%.*g", FLT_DIG, value);
345
346 // The snprintf should never overflow because the buffer is significantly
347 // larger than the precision we asked for.
348 DCHECK(snprintf_result > 0 && snprintf_result < kFastToBufferSize);
349
350 float parsed_value;
351 if (!safe_strtof(buffer, &parsed_value) || parsed_value != value) {
352 snprintf_result =
353 snprintf(buffer, kFastToBufferSize, "%.*g", FLT_DIG + 3, value);
354
355 // Should never overflow; see above.
356 DCHECK(snprintf_result > 0 && snprintf_result < kFastToBufferSize);
357 }
358 return snprintf_result;
359}

367bool StringToFp(const string& s, Fprint* fp) {
368 char junk;
369 uint64 result;
370 if (sscanf(s.c_str(), "%llx%c", &result, &junk) == 1) {
371 *fp = result;
372 return true;
373 } else {
374 return false;
375 }

378StringPiece Uint64ToHexString(uint64 v, char* buf) {
379 static const char* hexdigits = "0123456789abcdef";
380 const int num_byte = 16;
381 buf[num_byte] = '\0';
382 for (int i = num_byte - 1; i >= 0; i--) {
383 buf[i] = hexdigits[v & 0xf];
384 v >>= 4;
385 }
386 return StringPiece(buf, num_byte);
387}

389bool HexStringToUint64(const StringPiece& s, uint64* result) {
390 uint64 v = 0;
391 if (s.empty()) {
392 return false;
393 }
394 for (size_t i = 0; i < s.size(); i++) {
395 char c = s[i];
396 if (c >= '0' && c <= '9') {
397 v = (v << 4) + (c - '0');
398 } else if (c >= 'a' && c <= 'f') {
399 v = (v << 4) + 10 + (c - 'a');
400 } else if (c >= 'A' && c <= 'F') {
401 v = (v << 4) + 10 + (c - 'A');
402 } else {
403 return false;
404 }
405 }
406 *result = v;
407 return true;
408}

410string HumanReadableNum(int64 value) {
411 string s;
412 if (value < 0) {
413 s += "-";
414 value = -value;
415 }
416 if (value < 1000) {
417 Appendf(&s, "%lld", value);
418 } else if (value >= static_cast<int64>(1e15)) {
419 // Number bigger than 1E15; use that notation.
420 Appendf(&s, "%0.3G", static_cast<double>(value));
421 } else {
422 static const char units[] = "kMBT";
423 const char* unit = units;
424 while (value >= static_cast<int64>(1000000)) {
425 value /= static_cast<int64>(1000);
426 ++unit;
427 CHECK(unit < units + TF_ARRAYSIZE(units));
428 }
429 Appendf(&s, "%.2f%c", value / 1000.0, *unit);
430 }
431 return s;
432}

434string HumanReadableNumBytes(int64 num_bytes) {
435 if (num_bytes == kint64min) {
436 // Special case for number with not representable negation.
437 return "-8E";
438 }
439
440 const char* neg_str = (num_bytes < 0) ? "-" : "";
441 if (num_bytes < 0) {
442 num_bytes = -num_bytes;
443 }
444
445 // Special case for bytes.
446 if (num_bytes < 1024) {
447 // No fractions for bytes.
448 char buf[8]; // Longest possible string is '-XXXXB'
449 snprintf(buf, sizeof(buf), "%s%lldB", neg_str,
450 static_cast<int64>(num_bytes));
451 return string(buf);
452 }
453
454 static const char units[] = "KMGTPE"; // int64 only goes up to E.
455 const char* unit = units;
456 while (num_bytes >= static_cast<int64>(1024) * 1024) {
457 num_bytes /= 1024;
458 ++unit;
459 CHECK(unit < units + TF_ARRAYSIZE(units));
460 }
461
462 // We use SI prefixes.
463 char buf[16];
464 snprintf(buf, sizeof(buf), ((*unit == 'K') ? "%s%.1f%ciB" : "%s%.2f%ciB"),
465 neg_str, num_bytes / 1024.0, *unit);
466 return string(buf);
467}

469string HumanReadableElapsedTime(double seconds) {
470 string human_readable;
471
472 if (seconds < 0) {
473 human_readable = "-";
474 seconds = -seconds;
475 }
476
477 // Start with us and keep going up to years.
478 // The comparisons must account for rounding to prevent the format breaking
479 // the tested condition and returning, e.g., "1e+03 us" instead of "1 ms".
480 const double microseconds = seconds * 1.0e6;
481 if (microseconds < 999.5) {
482 strings::Appendf(&human_readable, "%0.3g us", microseconds);
483 return human_readable;
484 }
485 double milliseconds = seconds * 1e3;
486 if (milliseconds >= .995 && milliseconds < 1) {
487 // Round half to even in Appendf would convert this to 0.999 ms.
488 milliseconds = 1.0;
489 }
490 if (milliseconds < 999.5) {
491 strings::Appendf(&human_readable, "%0.3g ms", milliseconds);
492 return human_readable;
493 }
494 if (seconds < 60.0) {
495 strings::Appendf(&human_readable, "%0.3g s", seconds);
496 return human_readable;
497 }
498 seconds /= 60.0;
499 if (seconds < 60.0) {
500 strings::Appendf(&human_readable, "%0.3g min", seconds);
501 return human_readable;
502 }
503 seconds /= 60.0;
504 if (seconds < 24.0) {
505 strings::Appendf(&human_readable, "%0.3g h", seconds);
506 return human_readable;
507 }
508 seconds /= 24.0;
509 if (seconds < 30.0) {
510 strings::Appendf(&human_readable, "%0.3g days", seconds);
511 return human_readable;
512 }
513 if (seconds < 365.2425) {
514 strings::Appendf(&human_readable, "%0.3g months", seconds / 30.436875);
515 return human_readable;
516 }
517 seconds /= 365.2425;
518 strings::Appendf(&human_readable, "%0.3g years", seconds);
519 return human_readable;
520}
