40namespace tensorflow {

42class GPUOptions;
43
44// An object to keep track of pending Events in the StreamExecutor streams
45// and associated Tensors that cannot safely be deleted until the associated
46// Events are recorded.
47class EventMgr {
48 public:
49 EventMgr(perftools::gputools::StreamExecutor* se,
50 const GPUOptions& gpu_options);
51
52 ~EventMgr();
53
54 // Releases the references on the elements of "tensors" as soon as
55 // all events currently enqueued on "stream" have completed.
56 void ThenDeleteTensors(perftools::gputools::Stream* stream,
57 const TensorReferenceVector& tensors);
58
59 struct BufRec {
60 Allocator* alloc;
61 void* buf;
62 // operation and step_id are only populated when
63 // LogMemory::IsEnabled() is true.
64 string operation;
65 int64 step_id;
66 };
67
68 // Takes ownership of *bufrec.buf and calls bufrec.alloc->DeallocateRaw()
69 // on it as soon as all events currently enqueued on *stream have completed.
70 inline void ThenDeleteBuffer(perftools::gputools::Stream* stream,
71 BufRec bufrec) {
72 ToFreeVector to_free;
73 {
74 mutex_lock l(mu_);
75 QueueBuffer(stream, bufrec);
76 PollEvents(false, &to_free);
77 }
78 FreeMemory(to_free);
79 }
80
81 inline void ThenExecute(perftools::gputools::Stream* stream,
82 std::function<void()> func) {
83 ToFreeVector to_free;
84 {
85 mutex_lock l(mu_);
86 QueueFunc(stream, std::move(func));
87 PollEvents(false, &to_free);
88 }
89 FreeMemory(to_free);
90 }
91
92 private:
93 friend class TEST_EventMgrHelper;
94 perftools::gputools::StreamExecutor* const exec_;
95 const int64 deferred_bytes_threshold_;
96 const int32 polling_active_delay_usecs_;
97 mutex mu_;
98 condition_variable events_pending_ GUARDED_BY(mu_);
99
100 void FlushAccumulatedTensors() EXCLUSIVE_LOCKS_REQUIRED(mu_);
101
102 struct InUse {
103 perftools::gputools::Event* event;
104 TensorReferenceVector* mem;
105 BufRec bufrec;
106 std::function<void()> func;
107 };
108
109 typedef gtl::InlinedVector<InUse, 4> ToFreeVector;
110
111 void FreeMemory(const ToFreeVector& to_free) {
112 for (const auto& iu : to_free) {
113 if (iu.mem != nullptr) {
114 for (auto& t : *(iu.mem)) {
115 t.Unref();
116 }
117 delete iu.mem;
118 }
119 if (iu.bufrec.buf) {
120 if (LogMemory::IsEnabled()) {
121 LogMemory::RecordRawDeallocation(iu.bufrec.operation,
122 iu.bufrec.step_id, iu.bufrec.buf,
123 iu.bufrec.alloc, false);
124 }
125 iu.bufrec.alloc->DeallocateRaw(iu.bufrec.buf);
126 }
127 // The function must be called in another thread.
128 if (iu.func != nullptr) threadpool_.Schedule(iu.func);
129 }
130 }
131
132 // Stream-enqueue an unused Event and save with it a collection of
133 // Tensors and/or a BufRec to be deleted only after the Event
134 // records.
135 void QueueInUse(perftools::gputools::Stream* stream, InUse in_use)
136 EXCLUSIVE_LOCKS_REQUIRED(mu_);
137
138 void QueueTensors(perftools::gputools::Stream* stream,
139 TensorReferenceVector* tensors)
140 EXCLUSIVE_LOCKS_REQUIRED(mu_) {
141 QueueInUse(stream, {nullptr, tensors, BufRec(), nullptr});
142 }
143
144 void QueueBuffer(perftools::gputools::Stream* stream, BufRec bufrec)
145 EXCLUSIVE_LOCKS_REQUIRED(mu_) {
146 QueueInUse(stream, {nullptr, nullptr, bufrec, nullptr});
147 }
148
149 void QueueFunc(perftools::gputools::Stream* stream,
150 std::function<void()> func) EXCLUSIVE_LOCKS_REQUIRED(mu_) {
151 QueueInUse(stream, {nullptr, nullptr, BufRec(), std::move(func)});
152 }
153
154 // This function should be called at roughly the same tempo as
155 // QueueTensors() to check whether pending events have recorded,
156 // and then retire them. It appends InUse elements that need cleanup
157 // to "*to_free". The caller should call FreeMemory(to_free)
158 // when this returns.
159 void PollEvents(bool is_dedicated_poller, ToFreeVector* to_free)
160 EXCLUSIVE_LOCKS_REQUIRED(mu_);
161
162 // An internal polling loop that runs at a low frequency to clear
163 // straggler Events.
164 void PollLoop();
165
166 // Setup/Teardown functions for the polling loop.
167 void StartPollingLoop();
168 void StopPollingLoop();
169
170 // A stack of unused events
171 std::vector<perftools::gputools::Event*> free_events_ GUARDED_BY(mu_);
172
173 // Buffered list of tensors waiting to have an event queued for deletion
174 perftools::gputools::Stream* accumulated_stream_ GUARDED_BY(mu_);
175 TensorReferenceVector* accumulated_tensors_ GUARDED_BY(mu_);
176 // Sum of the TotalBytes() of the tensors in "accumulated_tensors_"
177 int64 accumulated_tensor_bytes_ GUARDED_BY(mu_);
178
179 // A FIFO queue of InUse events and associated tensors.
180 std::deque<InUse> used_events_ GUARDED_BY(mu_);
181
182 bool stop_polling_ GUARDED_BY(mu_);
183 std::unique_ptr<Notification> polling_stopped_;
184
185 // The main PollLoop for the event manager runs in this threadpool.
186 thread::ThreadPool threadpool_;
187};
