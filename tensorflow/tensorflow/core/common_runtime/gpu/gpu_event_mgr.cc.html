25EventMgr::EventMgr(gpu::StreamExecutor* se, const GPUOptions& gpu_options)
26 : exec_(se),
27 deferred_bytes_threshold_(gpu_options.deferred_deletion_bytes()
28 ? gpu_options.deferred_deletion_bytes()
29 : 8 * 1048576),
30 polling_active_delay_usecs_(gpu_options.polling_active_delay_usecs()
31 ? gpu_options.polling_active_delay_usecs()
32 : 10),
33 accumulated_stream_(nullptr),
34 accumulated_tensors_(new TensorReferenceVector),
35 accumulated_tensor_bytes_(0),
36 // threadpool_ has 1 thread for the polling loop, and one to execute
37 // event callback functions. Maybe we should have more?
38 threadpool_(Env::Default(), "GPU_Event_Manager", 2) {
39 StartPollingLoop();
40}
41
42EventMgr::~EventMgr() {
43 StopPollingLoop();
44
45 // Events are owned by this object.
46 for (auto& e : free_events_) {
47 delete e;
48 }
49 for (auto& t : *(accumulated_tensors_)) {
50 t.Unref();
51 }
52 delete accumulated_tensors_;
53 while (!used_events_.empty()) {
54 InUse* ue = &used_events_[0];
55 delete ue->event;
56 if (ue->mem != nullptr) {
57 for (auto& t : *(ue->mem)) {
58 t.Unref();
59 }
60 delete ue->mem;
61 }
62 if (ue->bufrec.buf) {
63 if (LogMemory::IsEnabled()) {
64 LogMemory::RecordRawDeallocation(ue->bufrec.operation,
65 ue->bufrec.step_id, ue->bufrec.buf,
66 ue->bufrec.alloc, false);
67 }
68 ue->bufrec.alloc->DeallocateRaw(ue->bufrec.buf);
69 }
70 if (ue->func != nullptr) threadpool_.Schedule(ue->func);
71 used_events_.pop_front();
72 }
73}

75void EventMgr::StartPollingLoop() {
76 CHECK(polling_stopped_ == nullptr);
77 {
78 mutex_lock l(mu_);
79 stop_polling_ = false;
80 }
81 polling_stopped_.reset(new Notification);
82 threadpool_.Schedule([this]() { PollLoop(); });
83}
84
85void EventMgr::StopPollingLoop() {
86 if (polling_stopped_) {
87 {
88 mutex_lock l(mu_);
89 stop_polling_ = true;
90 events_pending_.notify_all();
91 }
92 polling_stopped_->WaitForNotification();
93 polling_stopped_.reset(nullptr);
94 }
95}

97void EventMgr::ThenDeleteTensors(perftools::gputools::Stream* stream,
98 const TensorReferenceVector& tensors) {
99 mutex_lock l(mu_);
100 // TODO(jeff): We currently keep one accumulated_tensors_ object.
101 // If we start to use multiple streams heavily, we might want to keep
102 // separate vectors/byte counters per stream
103 if (!accumulated_tensors_->empty() && stream != accumulated_stream_) {
104 FlushAccumulatedTensors();
105 }
106 accumulated_stream_ = stream;
107 for (const auto& t : tensors) {
108 // accumulated_tensors_ takes over ownership of the reference to "t"
109 accumulated_tensors_->push_back(t);
110 accumulated_tensor_bytes_ += t.TotalBytes();
111 }
112 if (accumulated_tensor_bytes_ >= deferred_bytes_threshold_) {
113 FlushAccumulatedTensors();
114 }
115}
116
117void EventMgr::FlushAccumulatedTensors() {
118 DCHECK(!accumulated_tensors_->empty());
119 DCHECK(accumulated_stream_ != nullptr);
120 QueueTensors(accumulated_stream_, accumulated_tensors_);
121 accumulated_tensors_ = new TensorReferenceVector;
122 accumulated_tensor_bytes_ = 0;
123 accumulated_stream_ = nullptr;
124}

126// A polling loop to detect completion of GPU events.
127//
128// While one or more events is outstanding, poll for completed events. When no
129// events are outstanding, we sleep until one is enqueued.
130void EventMgr::PollLoop() {
131 ToFreeVector to_free;
132 while (true) {
133 bool events_still_pending;
134 {
135 mutex_lock l(mu_);
136 if (stop_polling_) {
137 break;
138 }
139 if (used_events_.empty()) {
140 events_pending_.wait(l);
141 }
142 PollEvents(true, &to_free);
143 events_still_pending = !used_events_.empty();
144 }
145 FreeMemory(to_free);
146 to_free.clear();
147
148 if (events_still_pending) {
149 Env::Default()->SleepForMicroseconds(polling_active_delay_usecs_);
150 }
151 }
152 polling_stopped_->Notify();
153}

155void EventMgr::QueueInUse(gpu::Stream* stream, InUse iu) {
156 VLOG(2) << "QueueInUse free_events_ " << free_events_.size()
157 << " used_events_ " << used_events_.size();
158 // Events are created on demand, and repeatedly reused. There is no
159 // limit placed here on the number of allocated Events.
160 if (free_events_.empty()) {
161 free_events_.push_back(new gpu::Event(exec_));
162 free_events_.back()->Init();
163 }
164 gpu::Event* e = free_events_.back();
165 free_events_.pop_back();
166 stream->ThenRecordEvent(e);
167 iu.event = e;
168 bool was_empty = used_events_.empty();
169 used_events_.push_back(iu);
170 // Maybe wake up the polling thread
171 if (was_empty) events_pending_.notify_all();
172}

174// This function must be called periodically to check whether pending
175// events have recorded, and then retire them. Initial observations
176// suggest that typical behavior in a TensorFlow program is to have
177// 0-3 events pending most of the time, but there are occasionally
178// spikes of up to several hundred outstanding.
179//
180// NOTE: If all events are on the same stream, no later event will
181// complete before an earlier event, except possibly if the earlier
182// event transitions to an error state, so there's no advantage in
183// looking past the first kPending event. However, if we're using
184// multiple streams there may be some gain in looking deeper.
185// As a compromise, PollEvent() calls that are triggered by the queueing
186// of a single event never look past the first kPending event. Calls
187// coming from the dedicated polling thread always sweep the full queue.
188//
189// Note that allowing the queue to grow very long could cause overall
190// GPU memory use to spike needlessly. An alternative strategy would
191// be to throttle new Op execution until the pending event queue
192// clears.
193void EventMgr::PollEvents(bool is_dedicated_poller,
194 gtl::InlinedVector<InUse, 4>* to_free) {
195 VLOG(2) << "PollEvents free_events_ " << free_events_.size()
196 << " used_events_ " << used_events_.size();
197 // Sweep the remaining events in order. If this is the dedicated
198 // polling thread, check the entire set. Otherwise, just sweep up to
199 // the first non-complete record that is still pending.
200 for (auto& iu : used_events_) {
201 if (iu.event == nullptr) continue;
202 gpu::Event::Status s = iu.event->PollForStatus();
203 switch (s) {
204 case gpu::Event::Status::kUnknown:
205 case gpu::Event::Status::kError:
206 // We don't expect to see these. Someday maybe propagate
207 // a Status error, but for now fail hard.
208 LOG(FATAL) << "Unexpected Event status: " << static_cast<int>(s);
209 break;
210 case gpu::Event::Status::kPending:
211 if (!is_dedicated_poller) return; // quit processing queue
212 break;
213 case gpu::Event::Status::kComplete:
214 // Make a copy of the InUse record so we can free it after releasing
215 // the lock
216 to_free->push_back(iu);
217 free_events_.push_back(iu.event);
218 // Mark this InUse record as completed.
219 iu.event = nullptr;
220 }
221 }
222 // Then clear any completed InUse records from the front of the queue.
223 while (!used_events_.empty()) {
224 InUse& iu = used_events_.front();
225 if (iu.event == nullptr) {
226 used_events_.pop_front();
227 } else {
228 break;
229 }
230 }
