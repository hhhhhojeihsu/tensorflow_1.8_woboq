15#ifndef TENSORFLOW_CORE_COMMON_RUNTIME_EAGER_COPY_TO_DEVICE_NODE_H_

18#include "tensorflow/core/common_runtime/device.h"
19#include "tensorflow/core/common_runtime/eager/eager_executor.h"
20#include "tensorflow/core/common_runtime/eager/tensor_handle.h"
21#include "tensorflow/core/framework/tensor.h"
22#include "tensorflow/core/lib/core/status.h"
23
24namespace tensorflow {
25
26class CopyToDeviceNode : public EagerNode {
27 public:
28 CopyToDeviceNode(TensorHandle* src, Device* dstd, EagerContext* ctx)
29 : EagerNode(ctx->NextId()),
30 src_(src),
31 dstd_(dstd),
32 ctx_(ctx),
33 dst_(new TensorHandle(id, src_->dtype, ctx)) {
34 src_->Ref();
35 dst_->Ref();
36 }
37
38 ~CopyToDeviceNode() override {
39 src_->Unref();
40 dst_->Unref();
41 }
42
43 Status Run() override {
44 TensorHandle* temp = nullptr;
45 TF_RETURN_IF_ERROR(src_->CopyToDevice(ctx_, dstd_, &temp));
46 const Tensor* tensor = nullptr;
47 Device* device = nullptr;
48 Device* op_device = nullptr;
49 Status status = temp->TensorAndDevice(&tensor, &device, &op_device);
50 // `temp` is a ready handle. So the following call should return OK.
51 TF_DCHECK_OK(status) << status.error_message();
52 DCHECK(tensor);
53 dst_->SetTensorAndDevice(*tensor, device, op_device);
54 temp->Unref();
55 return Status::OK();
56 }
57
58 TensorHandle* dst() { return dst_; }
59
60 private:
61 TensorHandle* src_;
62 Device* dstd_;
63 EagerContext* ctx_;
64 TensorHandle* dst_;
65};
66
67} // namespace tensorflow
