22EagerContext::EagerContext(const SessionOptions& opts,
23 ContextDevicePlacementPolicy default_policy,
24 bool async, std::unique_ptr<DeviceMgr> device_mgr,
25 Rendezvous* rendezvous)
26 : policy_(default_policy),
27 device_manager_(std::move(device_mgr)),
28 devices_(device_manager_->ListDevices()),
29 rendezvous_(rendezvous),
30 thread_pool_(NewThreadPoolFromSessionOptions(opts)),
31 pflr_(new ProcessFunctionLibraryRuntime(
32 device_manager_.get(), opts.env, TF_GRAPH_DEF_VERSION, &func_lib_def_,
33 {}, thread_pool_.get())),
34 log_device_placement_(opts.config.log_device_placement()),
35 async_default_(async) {
36 if (async_default_) {
37 executor_.EnableAsync();
38 }
39
40 for (auto* device : devices_) {
41 devices_map_[device->name()] = device;
42 }
43}
44
45bool EagerContext::Async() const {
46 mutex_lock l(async_map_mu_);
47 return gtl::FindWithDefault(thread_local_async_, std::this_thread::get_id(),
48 async_default_);
49}
50
51Status EagerContext::SetAsyncForThread(bool async) {
52 {
53 tensorflow::mutex_lock l(async_map_mu_);
54 thread_local_async_[std::this_thread::get_id()] = async;
55 }
56 if (async) {
57 executor_.EnableAsync();
58 } else {
59 // TODO(agarwal): Currently we add a wait here to handle cases where a
60 // sync op has a control dependency on an async op, and the latter has not
61 // executed yet. This wait can be removed by storing all the control
62 // inputs and waiting for them when executing ops.
63 return executor_.WaitForAllPendingNodes();
64 }
65 return Status::OK();
66}
67
68void EagerContext::ClearCaches() {
69 mutex_lock ml(cache_mu_);
70 gtl::STLDeleteValues(&kernel_cache_);
71}
72
73void EagerContext::SetThreadLocalDevicePlacementPolicy(
74 ContextDevicePlacementPolicy policy) {
75 mutex_lock ml(policy_map_mu_);
76 thread_local_policies_[std::this_thread::get_id()] = policy;
77}
78
79ContextDevicePlacementPolicy EagerContext::GetDevicePlacementPolicy() {
80 mutex_lock ml(policy_map_mu_);
81 auto policy_map_it = thread_local_policies_.find(std::this_thread::get_id());
82 if (policy_map_it != thread_local_policies_.end()) {
83 return policy_map_it->second;
84 }
85 return policy_;
86}
87
88EagerContext::~EagerContext() {
89 executor_.WaitForAllPendingNodes().IgnoreError();
90 ClearCaches();
91 rendezvous_->Unref();
92}

94bool EagerContext::FindFunctionByName(const string& name) {
95 mutex_lock l(functions_mu_);
96 return func_lib_def_.Find(name) != nullptr;
97}
98
99Status EagerContext::FindFunctionOpData(
100 const string& name, const tensorflow::OpRegistrationData** op_data) {
101 mutex_lock l(functions_mu_);
102 return func_lib_def_.LookUp(name, op_data);
103}
104
105const FunctionDef* EagerContext::FindFunctionDef(const string& name) {
106 mutex_lock l(functions_mu_);
107 return func_lib_def_.Find(name);
108}
109
110Status EagerContext::FindDeviceByName(const string& name, Device** result) {
111 auto it = devices_map_.find(name);
112 if (it == devices_map_.end()) {
113 return errors::InvalidArgument(name, " unknown device.");
114 }
115 *result = it->second;
116 return Status::OK();
117}

119Status EagerContext::AddFunctionDef(const FunctionDef& fdef) {
120 mutex_lock l(functions_mu_);
121 return func_lib_def_.AddFunctionDef(fdef);
122}

124KernelAndDevice* EagerContext::GetCachedKernel(Fprint128 cache_key) {
125 tf_shared_lock l(cache_mu_);
126 return gtl::FindPtrOrNull(kernel_cache_, cache_key);
127}
128
129void EagerContext::AddKernelToCache(Fprint128 cache_key,
130 KernelAndDevice* kernel) {
131 mutex_lock ml(cache_mu_);
132 gtl::InsertOrUpdate(&kernel_cache_, cache_key, kernel);
133}

135void EagerContext::SetShouldStoreMetadata(bool value) {
136 should_store_metadata_.store(value);

138 mutex_lock ml(metadata_mu_);
