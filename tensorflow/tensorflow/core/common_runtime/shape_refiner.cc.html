60constexpr char kArgOp[] = "_Arg";
61constexpr char kRetvalOp[] = "_Retval";

63// Runs shape inference for the given node using the given ShapeRefiner.
64// The node must be a sub-node of a function node and the outer_context is
65// the inference context of that function node in the outer graph.
66Status InferShapesForFunctionSubNode(const Node* node, ShapeRefiner* refiner,
67 InferenceContext* outer_context) {
68 TF_RETURN_IF_ERROR(refiner->AddNode(node));
69 InferenceContext* node_context = CHECK_NOTNULL(refiner->GetContext(node));
70
71 if (StringPiece(node->type_string()) == kArgOp) {
72 // Handle special node: function input.
73 // Shapes for these nodes are provided in the outer inference
74 // context.
75
76 int index;
77 TF_RETURN_IF_ERROR(GetNodeAttr(AttrSlice(node->def()), "index", &index));
78
79 if (index < 0 || outer_context->num_inputs() <= index) {
80 return errors::Internal(
81 "Function instantiation included invalid input index: ", index,
82 " not in [0, ", outer_context->num_inputs(), ").");
83 }
84
85 node_context->set_output(0, outer_context->input(index));
86
87 auto* resource = outer_context->input_handle_shapes_and_types(index);
88 if (resource) {
89 node_context->set_output_handle_shapes_and_types(0, *resource);
90 }
91 } else if (StringPiece(node->type_string()) == kRetvalOp) {
92 // Handle special node: function output.
93 // Shapes inferred for these nodes go into the outer inference
94 // context.
95
96 int index;
97 TF_RETURN_IF_ERROR(GetNodeAttr(AttrSlice(node->def()), "index", &index));
98
99 if (index < 0 || outer_context->num_outputs() <= index) {
100 return errors::Internal(
101 "Function instantiation included invalid output index: ", index,
102 " not in [0, ", outer_context->num_outputs(), ").");
103 }
104
105 // outer_context outlives node_context, therefore we need to create
106 // a new shape handle owned by outer_context instead.
107 ShapeHandle handle;
108 TensorShapeProto proto;
109 node_context->ShapeHandleToProto(node_context->input(0), &proto);
110 TF_RETURN_IF_ERROR(outer_context->MakeShapeFromShapeProto(proto, &handle));
111 outer_context->set_output(index, handle);
112
113 auto* resource = node_context->input_handle_shapes_and_types(0);
114 if (resource) {
115 outer_context->set_output_handle_shapes_and_types(index, *resource);
116 }
117 }
118
119 return Status::OK();
120}

124// TODO(cwhipkey): When an inference context inside function has
125// requested_input_tensor(i) or requested_input_tensor_as_partial_shape(i)
126// set when input(i) is an _Arg op, then this request should propagate to
127// context, and vice versa.
128//
129// NOTE: Recursive user-defined functions are not supported.
130// Maybe we won't support recursive functions at all in TF, because of
131// other maintainability issues.
132Status ShapeRefiner::InferShapesForFunction(
133 const tensorflow::FunctionDef* function_def, bool keep_nested_shapes,
134 ExtendedInferenceContext* outer_context) {
135 const Graph* graph;
136 auto it = functions_.find(function_def);
137 if (it != functions_.end()) {
138 graph = it->second.get();
139 } else {
140 InstantiationResult result;
141 TF_RETURN_IF_ERROR(InstantiateFunction(
142 *function_def, outer_context->get_context()->attrs(),
143 [this](const string& op, const OpDef** sig) {
144 return this->function_library_->LookUpOpDef(op, sig);
145 },
146 &result));
147
148 Graph* new_graph = new Graph(function_library_);
149 GraphConstructorOptions options;
150 options.allow_internal_ops = true;
151 TF_RETURN_IF_ERROR(
152 ConvertNodeDefsToGraph(options, result.nodes, new_graph));
153 functions_[function_def].reset(new_graph);
154 graph = new_graph;
155 }
156
157 std::unordered_set<const Node*> function_nodes;
158 Status inference_status = Status::OK();
159 {
160 auto node_shape_inference_lambda = [this, &outer_context, &function_nodes,
161 &inference_status](const Node* node) {
162 if (!inference_status.ok()) return;
163 inference_status = InferShapesForFunctionSubNode(
164 node, this, outer_context->get_context());
165 function_nodes.insert(node);
166 };
167
168 // Calls inference lambda for each node after visiting all predecessors.
169 // Ensures that we are adding nodes to ShapeRefiner in the topological
170 // order.
171 ReverseDFS(*graph, {}, node_shape_inference_lambda);
172 }
173
174 if (keep_nested_shapes && inference_status.ok()) {
175 // Fill the nested inferences map.
176 //
177 // The materialized function graph has extra nodes for arguments and
178 // return values, which are not explicitly listed in the FunctionDef,
179 // we filter out these special nodes here to not expose the implementation
180 // details and keep only inferences for the nodes listed in the FunctionDef.
181 std::unordered_map<string, const NodeDef*> user_defined_nodes;
182 for (const auto& node_def : function_def->node_def()) {
183 user_defined_nodes[node_def.name()] = &node_def;
184 }
185
186 std::unordered_map<string, std::unique_ptr<ExtendedInferenceContext>>
187 nested_inferences;
188 for (const Node* node : function_nodes) {
189 const string& node_name = node->name();
190 if (user_defined_nodes.find(node_name) != user_defined_nodes.end()) {
191 nested_inferences[node_name] = std::move(node_to_context_[node]);
192 node_to_context_.erase(node);
193 // By default InferenceContext refers to a NodeDef from Graph.
194 // Change it to the publicly accessible NodeDef of the function
195 // definition.
196 nested_inferences[node_name]->get_context()->node_def_ =
197 user_defined_nodes[node_name];
198 }
199 }
200 outer_context->set_nested_inferences(std::move(nested_inferences));
201 } else {
202 // Delete the contexts created for the functions nodes to save memory.
203 for (const Node* node : function_nodes) {
204 node_to_context_.erase(node);
205 }
206 }
207
208 return inference_status;
209}

211Status ShapeRefiner::AddNode(const Node* node) {
212 // For each 'input' of this node, fetch the corresponding shape
213 // from 'input's InferenceContext, and store into a vector
214 // indexed by 'node's input.
215 std::vector<const Node*> input_nodes(node->num_inputs());
216 std::vector<ShapeHandle> input_shapes(node->num_inputs());
217 std::vector<std::unique_ptr<std::vector<ShapeAndType>>>
218 input_handle_shapes_and_types(node->num_inputs());
219 for (const Edge* e : node->in_edges()) {
220 if (e->IsControlEdge()) continue;
221
222 const Node* input = e->src();
223 auto it = node_to_context_.find(input);
224 if (it == node_to_context_.end()) {
225 return errors::FailedPrecondition(
226 "Input ", e->dst_input(), " ('", input->name(), "') for '",
227 node->name(), "' was not previously added to ShapeRefiner.");
228 }
229
230 InferenceContext* c = it->second->get_context();
231 DCHECK_GE(e->dst_input(), 0);
232 input_nodes[e->dst_input()] = input;
233 input_shapes[e->dst_input()] = c->output(e->src_output());
234
235 // Only propagate handle data of edges which are carrying resource handles.
236 if (e->src()->output_type(e->src_output()) == DT_RESOURCE) {
237 const auto* in_v = c->output_handle_shapes_and_types(e->src_output());
238 if (in_v != nullptr) {
239 input_handle_shapes_and_types[e->dst_input()].reset(
240 new std::vector<ShapeAndType>(*in_v));
241 }
242 }
243 }
244
245 // Get the shape function for this node
246 const OpRegistrationData* op_reg_data;
247 TF_RETURN_IF_ERROR(ops_registry_->LookUp(node->type_string(), &op_reg_data));
248 if (op_reg_data->shape_inference_fn == nullptr &&
249 require_shape_inference_fns_) {
250 return errors::InvalidArgument(
251 "No shape inference function exists for op '", node->type_string(),
252 "', did you forget to define it?");
253 }
254
255 // This needs to be filled in with real data in a second pass.
256 std::vector<const Tensor*> input_tensors(node->num_inputs(), nullptr);
257 std::vector<ShapeHandle> input_tensors_as_shapes;
258
259 // Create the inference context for this node with the existing input shapes.
260 std::unique_ptr<InferenceContext> c(
261 new InferenceContext(graph_def_version_, &node->def(), node->op_def(),
262 input_shapes, input_tensors, input_tensors_as_shapes,
263 std::move(input_handle_shapes_and_types)));
264 if (!c->construction_status().ok()) {
265 return c->construction_status();
266 }
267
268 std::unique_ptr<ExtendedInferenceContext> ec(
269 new ExtendedInferenceContext(std::move(c), node));
270
271 // Run the shape inference function, and return if there was an error.
272 TF_RETURN_IF_ERROR(RunShapeFn(node, op_reg_data, ec.get()));
273
274 // Store the resulting context object in the map.
275 node_to_context_[node].swap(ec);
276
277 return Status::OK();
278}

280Status ShapeRefiner::SetShape(const Node* node, int output_port,
281 ShapeHandle shape) {
282 auto c = GetContext(node);
283 if (c == nullptr) {
284 return errors::Internal("Could not find context for ", node->name());
285 }
286
287 if (output_port < 0 || output_port >= node->num_outputs()) {
288 return errors::InvalidArgument(
289 "output_port '", output_port, "' is out of range, ", "node '",
290 node->name(), "' has ", node->num_outputs(), " outputs");
291 }
292
293 // Check compatibility, and merge the shapes.
294 ShapeHandle existing_shape = c->output(output_port);
295 TF_RETURN_IF_ERROR(c->Merge(existing_shape, shape, &shape));
296 c->set_output(output_port, shape);
297
298 // TODO(vrv): Do we need to propagate the new shape through all
299 // consumers that change their outputs? At the moment, python
300 // does not do this, but this seems like a nice feature.
301
302 // TODO(vrv): We might need to keep track of the fact that the
303 // existing shape is invalidated, in case we need to propagate
304 // this information to remote workers.
305 return Status::OK();
306}

308Status ShapeRefiner::UpdateNode(const Node* node, bool relax, bool* refined) {
309 auto it = node_to_context_.find(node);
310 if (it == node_to_context_.end()) {
311 *refined = true;
312 return AddNode(node);
313 }
314 ExtendedInferenceContext* node_ext_context = it->second.get();
315 InferenceContext* node_context = node_ext_context->get_context();
316
317 // Give up if the context wasn't successfully built by the AddNode() method.
318 TF_RETURN_IF_ERROR(node_context->construction_status());
319
320 // Check if the shapes of the nodes in the fan-in of this node have changed,
321 // and if they have update the node input shapes.
322 for (const Edge* e : node->in_edges()) {
323 if (e->IsControlEdge()) continue;
324
325 int dst_input = e->dst_input();
326 int src_output = e->src_output();
327
328 Node* input = e->src();
329 auto iter = node_to_context_.find(input);
330 if (iter == node_to_context_.end()) {
331 return errors::FailedPrecondition(
332 "Input ", dst_input, " ('", input->name(), "') for '", node->name(),
333 "' was not previously added to ShapeRefiner.");
334 }
335
336 InferenceContext* c = iter->second->get_context();
337 DCHECK_GE(dst_input, 0);
338 ShapeHandle existing_input = node_context->input(dst_input);
339 if (!relax) {
340 if (node_context->MergeInput(dst_input, c->output(src_output))) {
341 if (!SameDefinedShape(node_context, node_context->input(dst_input),
342 existing_input)) {
343 *refined = true;
344 }
345 }
346 } else {
347 if (node_context->RelaxInput(dst_input, c->output(src_output))) {
348 if (!SameDefinedShape(node_context, node_context->input(dst_input),
349 existing_input)) {
350 *refined = true;
351 }
352 }
353 }
354 if (node_context->requested_input_tensor_as_partial_shape(dst_input)) {
355 // The input value may have changed. Since we have no way to know if
356 // that's indeed the case, err on the safe side.
357 *refined = true;
358 }
359
360 // Also propagate handle shape and dtype of edges which are carrying
361 // resource handles.
362 if (e->src()->output_type(src_output) == DT_RESOURCE) {
363 auto* outputs = c->output_handle_shapes_and_types(src_output);
364 if (!outputs) continue;
365
366 if (!relax &&
367 node_context->MergeInputHandleShapesAndTypes(dst_input, *outputs)) {
368 *refined = true;
369 } else if (relax) {
370 std::vector<ShapeAndType> existing_inputs;
371 const std::vector<ShapeAndType>* inputs =
372 node_context->input_handle_shapes_and_types(dst_input);
373 if (inputs) {
374 existing_inputs = *inputs;
375 }
376 if (node_context->RelaxInputHandleShapesAndMergeTypes(dst_input,
377 *outputs)) {
378 if (IsUpdatedShapesOrTypes(
379 node_context, existing_inputs,
380 *node_context->input_handle_shapes_and_types(dst_input))) {
381 *refined = true;
382 }
383 }
384 }
385 }
386 }
387
388 if (!*refined) {
389 // No input shape has changed, we're done
390 return Status::OK();
391 }
392
393 // Get and run the shape function for this node to update the shapes of the
394 // outputs.
395 const OpRegistrationData* op_reg_data;
396 TF_RETURN_IF_ERROR(ops_registry_->LookUp(node->type_string(), &op_reg_data));
397 if (op_reg_data->shape_inference_fn == nullptr &&
398 require_shape_inference_fns_) {
399 return errors::InvalidArgument(
400 "No shape inference function exists for op '", node->type_string(),
401 "', did you forget to define it?");
402 }
403
404 if (!op_reg_data->shape_inference_fn) {
405 // There is nothing more we can infer
406 return Status::OK();
407 }
408
409 return RunShapeFn(node, op_reg_data, node_ext_context);
410}

412Status ShapeRefiner::EvaluateConstantTensorForEdge(const Node* node,
413 int dst_idx, bool* evaluated,
414 Tensor* result) {
415 *evaluated = false;
416 const Edge* input_edge;
417 TF_RETURN_IF_ERROR(node->input_edge(dst_idx, &input_edge));
418 OutputTensor tensor(input_edge->src(), input_edge->src_output());
419 return EvaluateConstantTensor(tensor, *this, *ops_registry_,
420 graph_def_version_, evaluated, result,
421 &graph_runner_, &const_tensor_map_,
422 kMaxTensorSize, disable_constant_propagation_);
423}

425Status ShapeRefiner::ConstantPartialShape(InferenceContext* target_context,
426 const Node* node, int dst_idx,
427 ShapeHandle* result) {
428 const Edge* input_edge;
429 TF_RETURN_IF_ERROR(node->input_edge(dst_idx, &input_edge));
430
431 InferenceContext* src_context = GetContext(input_edge->src());
432 if (src_context == nullptr) return errors::Internal("Missing src context");
433 ShapeHandle src_shape = src_context->output(input_edge->src_output());
434
435 if (src_context->Value(src_context->Rank(src_shape)) == 0) {
436 Tensor t;
437 bool evaluated = false;
438 TF_RETURN_IF_ERROR(
439 EvaluateConstantTensorForEdge(node, dst_idx, &evaluated, &t));
440 if (!evaluated) {
441 return errors::InvalidArgument(
442 "Received a shape scalar with unknown static value. A static value "
443 "of '-1' is required to represent an unknown shape.");
444 }
445 if (t.dims() == 0) {
446 if (t.dtype() == DT_INT32 && t.scalar<int32>()() == -1) {
447 *result = target_context->UnknownShape();
448 return Status::OK();
449 } else if (t.dtype() == DT_INT64 && t.scalar<int64>()() == -1) {
450 *result = target_context->UnknownShape();
451 return Status::OK();
452 }
453 }
454 return errors::InvalidArgument(
455 "Received an invalid shape scalar with a static value that is not "
456 "'-1': ",
457 t.DebugString());
458 }
459
460 TF_RETURN_IF_ERROR(src_context->WithRank(src_shape, 1, &src_shape));
461
462 const string& src_op = input_edge->src()->type_string();
463 if (src_context->Value(src_context->Dim(src_shape, 0)) == 0) {
464 // Source tensor is a vector of length 0, so the shape it
465 // represents is as scalar.
466 *result = target_context->Scalar();
467 } else if (src_op == "Shape") {
468 *result = src_context->input(0);
469 } else if (src_op == "ShapeN") {
470 *result = src_context->input(input_edge->src_output());
471 } else if (src_op == "Pack") {
472 std::vector<DimensionHandle> dims;
473 // Pack is concatenating its input scalars to form the shape tensor vector.
474 for (int i = 0; i < src_context->num_inputs(); ++i) {
475 Tensor scalar;
476 bool evaluated = false;
477 TF_RETURN_IF_ERROR(EvaluateConstantTensorForEdge(input_edge->src(), i,
478 &evaluated, &scalar));
479 if (evaluated) {
480 int64 size;
481 if (scalar.dtype() == DT_INT32) {
482 size = scalar.scalar<int32>()();
483 } else if (scalar.dtype() == DT_INT64) {
484 size = scalar.scalar<int64>()();
485 } else {
486 return errors::InvalidArgument("Pack input must be int32 or int64");
487 }
488 dims.push_back(size < 0 ? target_context->UnknownDim()
489 : target_context->MakeDim(size));
490 } else {
491 dims.push_back(target_context->UnknownDim());
492 }
493 }
494 *result = target_context->MakeShape(dims);
495 } else if (src_op == "Concat" || src_op == "ConcatV2") {
496 *result = target_context->Scalar();
497 // For Concat, input 0 is concat dim; for V2 it is the last input.
498 const int concat_dim =
499 src_op == "Concat" ? 0 : src_context->num_inputs() - 1;
500 // Concat is concatenating its input shape vectors.
501 for (int i = 0; i < src_context->num_inputs(); ++i) {
502 // Concat dim is ignored (and will always be a scalar).
503 if (i == concat_dim) continue;
504 ShapeHandle sub_result;
505 TF_RETURN_IF_ERROR(ConstantPartialShape(target_context, input_edge->src(),
506 i, &sub_result));
507 if (!target_context->RankKnown(sub_result)) {
508 // Failed to evaluate. Treat the output as completely unknown.
509 // TODO(cwhipkey): we could rely on all inputs being the same rank, so
510 // figure that rank out and append the right number of unknown dims.
511 *result = target_context->UnknownShape();
512 return Status::OK();
513 }
514 TF_RETURN_IF_ERROR(
515 target_context->Concatenate(*result, sub_result, result));
516 }
517 } else {
518 Tensor t;
519 bool evaluated = false;
520 TF_RETURN_IF_ERROR(
521 EvaluateConstantTensorForEdge(node, dst_idx, &evaluated, &t));
522 TF_RETURN_IF_ERROR(target_context->MakeShapeFromTensor(
523 evaluated ? &t : nullptr, src_shape, result));
524 }
525 return Status::OK();
526}

528Status ShapeRefiner::RunShapeFn(const Node* node,
529 const OpRegistrationData* op_reg_data,
530 ExtendedInferenceContext* ec) {
531 // This will be filled in with real data in a second pass.
532 std::vector<const Tensor*> input_tensors(node->num_inputs(), nullptr);
533 std::vector<Tensor> real_tensors(node->num_inputs());
534 std::vector<bool> attempted_materialization(node->num_inputs());
535 std::vector<bool> attempted_tensor_as_shape_conversion(node->num_inputs());
536 std::vector<ShapeHandle> input_tensors_as_shapes;
537
538 auto* c = ec->get_context();
539
540 c->set_input_tensors(input_tensors);
541 c->set_input_tensors_as_shapes(input_tensors_as_shapes);
542
543 // Run the shape inference function, and return if there was an error.
544 // Capture as lambda, because we might need to re-run inference later on.
545 auto run_inference_lambda = [&]() {
546 if (function_library_ && op_reg_data->is_function_op) {
547 // Special inference logic for user-defined functions.
548
549 auto* func_def = function_library_->Find(op_reg_data->op_def.name());
550 if (func_def) {
551 return InferShapesForFunction(func_def, keep_nested_shape_inferences_,
552 ec);
553 }
554 }
555
556 if (op_reg_data->shape_inference_fn) {
557 TF_RETURN_IF_ERROR(c->Run(op_reg_data->shape_inference_fn));
558 } else {
559 TF_RETURN_IF_ERROR(c->Run(shape_inference::UnknownShape));
560 }
561 return Status::OK();
562 };
563 TF_RETURN_IF_ERROR(run_inference_lambda());
564
565 // We must run the shape function repeatedly, in case users write
566 // shape functions where they only conditionally call input_tensor()
567 // based on the values of another input tensor.
568 bool rerun_shape_fn;
569 do {
570 // If the result of running shape inference would have benefitted
571 // from knowing the values of input tensors, try to materialize
572 // the results of those tensors, and then run the shape inference
573 // function again using those known tensors.
574 rerun_shape_fn = false;
575
576 // NOTE: It is possible to batch the extraction and
577 // materialization of inputs, instead of materializing one input
578 // at a time like we do below. If input-at-a-time computation
579 // becomes a bottleneck, we could separate ExtractConstantSubgraph
580 // into two functions: one that returns true if an input is
581 // derivable from constants, and another function that extracts
582 // the subgraph for multiple target nodes and executes the whole
583 // subgraph once.
584
585 for (int i = 0; i < c->num_inputs(); ++i) {
586 if (!c->requested_input_tensor(i)) {
587 continue;
588 }
589 // Check if we have not already filled in the requested input,
590 // and if not, try to materialize the tensors.
591 if (!attempted_materialization[i]) {
592 attempted_materialization[i] = true;
593
594 Tensor result;
595 bool evaluated = false;
596 TF_RETURN_IF_ERROR(
597 EvaluateConstantTensorForEdge(node, i, &evaluated, &result));
598 if (evaluated) {
599 real_tensors[i] = result;
600 input_tensors[i] = &real_tensors[i];
601 // We have more concrete information about a shape,
602 // so re-run shape inference.
603 rerun_shape_fn = true;
604 }
605 }
606 if (c->requested_input_tensor_as_partial_shape(i) &&
607 !attempted_tensor_as_shape_conversion[i]) {
608 attempted_tensor_as_shape_conversion[i] = true;
609 if (i >= input_tensors_as_shapes.size()) {
610 input_tensors_as_shapes.resize(i + 1);
611 }
612 ShapeHandle s;
613 TF_RETURN_IF_ERROR(ConstantPartialShape(c, node, i, &s));
614 input_tensors_as_shapes[i] = s;
615 rerun_shape_fn = true;
616 }
617 }
618
619 if (rerun_shape_fn) {
620 // We have more information about the shapes on this pass,
621 // so re-run shape inference.
622 c->set_input_tensors(input_tensors);
623 c->set_input_tensors_as_shapes(input_tensors_as_shapes);
624 TF_RETURN_IF_ERROR(run_inference_lambda());
625 }
626 } while (rerun_shape_fn);
627
628 return Status::OK();
629}

631bool ShapeRefiner::SameDefinedShape(InferenceContext* c, ShapeHandle s0,
632 ShapeHandle s1) {
633 if (s0.SameHandle(s1)) {
634 return true;
635 }
636 if (c->Rank(s0) != c->Rank(s1)) {
637 return false;
638 }
639 if (!c->RankKnown(s0) && !c->RankKnown(s1)) {
640 return false;
641 }
642 for (int i = 0; i < c->Rank(s0); ++i) {
643 if (!c->Dim(s0, i).SameHandle(c->Dim(s1, i))) {
644 int64 val0 = c->Value(c->Dim(s0, i));
645 int64 val1 = c->Value(c->Dim(s1, i));
646 if (val0 < 0 || val1 < 0 || val0 != val1) {
647 return false;
648 }
649 }
650 }
651
652 return true;
653}
654
655bool ShapeRefiner::IsUpdatedShapesOrTypes(
656 InferenceContext* c, const std::vector<ShapeAndType>& existing,
657 const std::vector<ShapeAndType>& updated) {
658 if (existing.size() != updated.size()) {
659 return true;
660 }
661 for (int i = 0; i < existing.size(); i++) {
662 if (!SameDefinedShape(c, existing[i].shape, updated[i].shape) ||
663 existing[i].dtype != updated[i].dtype) {
664 return true;
665 }
666 }
667 return false;
668}
