16#ifndef TENSORFLOW_CORE_COMMON_RUNTIME_RENAMED_DEVICE_H_
17#define TENSORFLOW_CORE_COMMON_RUNTIME_RENAMED_DEVICE_H_
18
19#include "tensorflow/core/common_runtime/device.h"
20#include "tensorflow/core/util/device_name_utils.h"
21
22namespace tensorflow {
23
24// Wraps a device with a new name, delegating work to the wrapped device.
25//
26// This class is used to wrap local devices when using clusterspec propagation
27// where the name of a particular device may change in the context of a given
28// session.
29class RenamedDevice : public Device {
30 public:
31 static Device* NewRenamedDevice(const string& new_base, Device* underlying,
32 bool owns_underlying,
33 bool isolate_session_state);
34
35 ~RenamedDevice() override;
36
37 // Below are virtual methods defined on DeviceBase
38 bool RequiresRecordingAccessedTensors() const override {
39 return underlying_->RequiresRecordingAccessedTensors();
40 }
41
42 const DeviceBase* UnderlyingDevice() const override {
43 return underlying_->UnderlyingDevice();
44 }
45 DeviceBase* UnderlyingDevice() override {
46 return underlying_->UnderlyingDevice();
47 }
48
49 const CpuWorkerThreads* tensorflow_cpu_worker_threads() const override {
50 return underlying_->tensorflow_cpu_worker_threads();
51 }
52
53 const GpuDeviceInfo* tensorflow_gpu_device_info() const override {
54 return underlying_->tensorflow_gpu_device_info();
55 }
56
57 Allocator* GetAllocator(AllocatorAttributes attr) override {
58 return underlying_->GetAllocator(attr);
59 }
60
61 Allocator* GetStepAllocator(AllocatorAttributes attr,
62 ResourceMgr* step_resource_manager) override {
63 return underlying_->GetStepAllocator(attr, step_resource_manager);
64 }
65
66 const Eigen::ThreadPoolDevice* eigen_cpu_device() override {
67 return underlying_->eigen_cpu_device();
68 }
69
70#ifdef TENSORFLOW_USE_SYCL
71 const Eigen::SyclDevice* eigen_sycl_device() const override {
72 return underlying_->eigen_sycl_device();
73 }
74#endif
75
76 PerOpGpuDevice* MakeGpuDevice() override {
77 return underlying_->MakeGpuDevice();
78 }
79
80 void ReinitializeGpuDevice(OpKernelContext* context, PerOpGpuDevice* device,
81 DeviceContext* dc, Allocator* allocator) override {
82 underlying_->ReinitializeGpuDevice(context, device, dc, allocator);
83 }
84
85 Status MakeTensorFromProto(const TensorProto& tensor_proto,
86 const AllocatorAttributes alloc_attrs,
87 Tensor* tensor) override {
88 return underlying_->MakeTensorFromProto(tensor_proto, alloc_attrs, tensor);
89 }
90
91 // Below are virtual methods defined on Device
92
93 void Compute(OpKernel* op_kernel, OpKernelContext* context) override {
94 underlying_->Compute(op_kernel, context);
95 }
96
97 void ComputeAsync(AsyncOpKernel* op_kernel, OpKernelContext* context,
98 AsyncOpKernel::DoneCallback done) override {
99 underlying_->ComputeAsync(op_kernel, context, std::move(done));
100 }
101
102 void ConsumeListOfAccessedTensors(
103 DeviceContext* context, const TensorReferenceVector& tensors) override {
104 underlying_->ConsumeListOfAccessedTensors(context, tensors);
105 }
106
107 Status Sync() override { return underlying_->Sync(); }
108
109 Status MaybeRewriteGraph(std::unique_ptr<Graph>* graph) override {
110 return underlying_->MaybeRewriteGraph(graph);
111 }
112
113 Status FillContextMap(const Graph* graph,
114 DeviceContextMap* device_context_map) override {
115 return underlying_->FillContextMap(graph, device_context_map);
116 }
117
118 // Returns the resource manager associated w/ this device.
119 ResourceMgr* resource_manager() override {
120 if (isolate_session_state_) {
121 return Device::resource_manager();
122 } else {
123 return underlying_->resource_manager();
124 }
125 }
126
127 private:
128 RenamedDevice(Device* underlying, const DeviceAttributes& attributes,
129 bool owns_underlying, bool isolate_session_state);
130 Device* const underlying_;
131 const bool owns_underlying_;
132 const bool isolate_session_state_;
133};
134
135} // namespace tensorflow
136
137#endif // TENSORFLOW_CORE_COMMON_RUNTIME_RENAMED_DEVICE_H_
