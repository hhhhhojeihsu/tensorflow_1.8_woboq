39ThreadPoolDevice::ThreadPoolDevice(const SessionOptions& options,
40 const string& name, Bytes memory_limit,
41 const DeviceLocality& locality,
42 Allocator* allocator)
43 : LocalDevice(options, Device::BuildDeviceAttributes(
44 name, DEVICE_CPU, memory_limit, locality)),
45 allocator_(allocator),
46 scoped_allocator_mgr_(new ScopedAllocatorMgr(name)) {}
47
48ThreadPoolDevice::~ThreadPoolDevice() {}
49
50void ThreadPoolDevice::Compute(OpKernel* op_kernel, OpKernelContext* context) {
51 // When TraceMe profiling is off (which is the default), the
52 // following TraceMe constructor is simply a conditional test of
53 // false value. Measurements show that its overhead is negligible.
54 port::Tracing::TraceMe trace_me(op_kernel->name(), op_kernel->type_string(),
55 op_kernel->IsExpensive());
56 if (port::Tracing::IsActive()) {
57 // TODO(pbar) We really need a useful identifier of the graph node.
58 const uint64 id = Hash64(op_kernel->name());
59 port::Tracing::ScopedActivity region(port::Tracing::EventCategory::kCompute,
60 id);
61 op_kernel->Compute(context);
62 } else {
63 op_kernel->Compute(context);
64 }
65}
66
67Allocator* ThreadPoolDevice::GetAllocator(AllocatorAttributes attr) {
68 return allocator_;
69}
70
71Allocator* ThreadPoolDevice::GetScopedAllocator(AllocatorAttributes attr,
72 int64 step_id) {
73 if (attr.scope_id > 0) {
74 return scoped_allocator_mgr_->GetContainer(step_id)->GetInstance(
75 attr.scope_id);
76 }
77 LOG(FATAL) << "Unexpected call to ThreadPoolDevice::GetScopedAllocator "
78 << "attr.scope_id = " << attr.scope_id;
79 return allocator_;
80}

82Status ThreadPoolDevice::MakeTensorFromProto(
83 const TensorProto& tensor_proto, const AllocatorAttributes alloc_attrs,
84 Tensor* tensor) {
85 if (tensor_proto.dtype() > 0 && tensor_proto.dtype() <= DataType_MAX) {
86 Tensor parsed(tensor_proto.dtype());
87 if (parsed.FromProto(cpu_allocator(), tensor_proto)) {
88 *tensor = std::move(parsed);
89 return Status::OK();
90 }
91 }
92 return errors::InvalidArgument("Cannot parse tensor from proto: ",
93 ProtoDebugString(tensor_proto));
94}
