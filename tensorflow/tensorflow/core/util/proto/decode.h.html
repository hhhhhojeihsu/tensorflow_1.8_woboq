16// Inline functions for parsing the protocol buffers wire format.
17//
18// These functions have been optimized at the expense of safety.
19// They are broken out into a separate file for readability but are
20// not intended for use by clients other than the decode_proto op.
21//
22// The calling code in the decode_proto op does some fairly
23// complicated things to ensure that this code is called
24// safely. Changes to this code should be thoroughly fuzz tested.

41// Converts an uint64 to an int64 without loss of information.
42// Unsigned values greater than INT64_MAX are represented as
43// negative numbers by wrapping (same as twos-complement bit equivalence).
44inline int64 WrapUnsignedAsSigned64(uint64 unsigned_value) {
45 // For a detailed explanation of why this works to wrap unsigned ints, see
46 // http://stackoverflow.com/questions/13150449/efficient-unsigned-to-signed-cast-avoiding-implementation-defined-behavior
47 // Both if tests should be optimized out.
48 if (unsigned_value <= INT64_MAX) {
49 return static_cast<int64>(unsigned_value);
50 }
51 // The C++ spec allows an architecture where this test is required.
52 if (unsigned_value >= INT64_MIN) {
53 return static_cast<int64>(unsigned_value - INT64_MIN) + INT64_MIN;
54 }
55 return 0; // This should never occur.
56}

75// Reads a single varint32 from a byte array.
76// It is the caller's responsibility to ensure that there is enough
77// space in the buffer.
78// The ok value will be set to false if the buffer does not contain
79// a valid varint.
80inline const uint8* ReadVarint64FromArray(const uint8* buffer, bool* ok,
81 uint64* value);
82
83// Reads a single varint32 from a byte array.
84// It is the caller's responsibility to ensure that there is enough
85// space in the buffer.
86// The ok value will be set to false if the buffer does not contain
87// a valid varint.
88// This is slightly less efficient than the private version in
89// coded_stream.cc but we duplicate less code by calling
90// the 64 bit version instead of copying the code.
91inline const uint8* ReadVarint32FromArray(const uint8* buffer, bool* ok,
92 uint32* value) {
93 uint64 tmp;
94 const uint8* buf = ReadVarint64FromArray(buffer, ok, &tmp);
95 *value = tmp & 0xffffffff;
96 return buf;
97}

99// Reads a single proto field value from a byte array into an array.
100// The array is part of a Tensor that was allocated by the caller
101// with type TensorType, while DeclaredType is the proto field type.
102template <class TensorType, enum WireFormatLite::FieldType DeclaredType>
103const uint8* ReadFromArray(const uint8* buf, TensorType* value);
104
105template <>
106inline const uint8* ReadFromArray<int32, WireFormatLite::TYPE_INT32>(
107 const uint8* buf, int32* value) {
108 uint32 temp;
109 bool unused_ok; // The Counting pass would have failed if this were corrupt.
110 buf = ReadVarint32FromArray(buf, &unused_ok, &temp);
111 *value = static_cast<int32>(temp);
112 return buf;
113}

263// Reads packed values from an array.
264// Stride is set to 1 for repeated fields, and 0 for non-repeated fields
265// (where any value overwrites previous values).
266template <class TensorType, enum WireFormatLite::FieldType DeclaredType>
267inline int ReadPackedPrimitives(const void* bufp, const size_t len,
268 const int index, const int stride,
269 void* datap) {
270 const uint8* buf = reinterpret_cast<const uint8*>(bufp);
271 const uint8* bound = buf + len;
272 TensorType* data = reinterpret_cast<TensorType*>(datap) + index;
273 int count;
274
275 // This could overrun the bound by stride-1. This is defended
276 // against in the caller, where it ensures that the input buffer
277 // contains complete values.
278 for (count = 0; buf < bound; count += stride) {
279 buf = ReadFromArray<TensorType, DeclaredType>(buf, data + count);
280 }
281 return count;
282}

284// Reads a primitive value field from a serialized proto.
285// The value is parsed from the serialized format, then static_cast
286// to the desired type for TensorFlow and stored.
287template <class ValueType, class TensorType,
288 enum WireFormatLite::FieldType DeclaredType>
289inline Status ReadPrimitive(CodedInputStream* input, int index, void* data) {
290 ValueType v;
291 if (!WireFormatLite::ReadPrimitive<ValueType, DeclaredType>(input, &v)) {
292 return errors::DataLoss("Failed reading primitive");
293 }
294
295 reinterpret_cast<TensorType*>(data)[index] = v;
296 return Status::OK();
297}

299// Reads a string, submessage, or other variable-length field from a
300// serialized proto.
301// May read all or part of a repeated field.
302inline Status ReadBytes(CodedInputStream* input, int index, void* datap) {
303 string* data = reinterpret_cast<string*>(datap) + index;
304 if (!WireFormatLite::ReadBytes(input, data)) {
305 return errors::DataLoss("Failed reading bytes");
306 }
307 return Status::OK();
308}

310// Reads a tag-delimited field (TYPE_GROUP) from a serialized proto,
311// as a bytestring.
312inline Status ReadGroupBytes(CodedInputStream* input, int field_number,
313 int index, void* datap) {
314 // WireFormatLite::SkipField has an option to emit the
315 // skipped bytes to an output stream. We could do better by implementing our
316 // own scanner but this is simpler for now.
317 // TODO(nix): there is a faster way to grab TYPE_GROUP bytes by relying
318 // on input->IsFlat() == true and using input->GetDirectBufferPointer()
319 // with input->CurrentPosition().
320 string* data = reinterpret_cast<string*>(datap) + index;
321 StringOutputStream string_stream(data);
322 CodedOutputStream out(&string_stream);
323 if (!WireFormatLite::SkipField(
324 input,
325 WireFormatLite::MakeTag(field_number,
326 WireFormatLite::WIRETYPE_START_GROUP),
327 &out)) {
328 return errors::DataLoss("Failed reading group");
329 }
330 return Status::OK();
331}

333// Reads a single field value from a CodedInputStream into a tensor.
334inline Status ReadValue(CodedInputStream* input,
335 WireFormatLite::FieldType field_type, int field_number,
336 DataType dtype, int index, void* datap) {
337 // Dispatch to the appropriately typed field reader based on the
338 // schema type.
339 switch (field_type) {
340 case WireFormatLite::TYPE_DOUBLE:
341 return ReadPrimitive<double, double, WireFormatLite::TYPE_DOUBLE>(
342 input, index, datap);
343 case WireFormatLite::TYPE_FLOAT:
344 if (dtype == DataType::DT_FLOAT) {
345 return ReadPrimitive<float, float, WireFormatLite::TYPE_FLOAT>(
346 input, index, datap);
347 }
348 if (dtype == DataType::DT_DOUBLE) {
349 return ReadPrimitive<float, double, WireFormatLite::TYPE_FLOAT>(
350 input, index, datap);
351 }
352 // Any case that reaches this point should have triggered an error
353 // already.
354 return errors::DataLoss("Failed reading TYPE_FLOAT");
355 case WireFormatLite::TYPE_INT64:
356 return ReadPrimitive<protobuf_int64, int64, WireFormatLite::TYPE_INT64>(
357 input, index, datap);
358 case WireFormatLite::TYPE_UINT64:
359 return ReadPrimitive<protobuf_uint64, int64, WireFormatLite::TYPE_UINT64>(
360 input, index, datap);
361 case WireFormatLite::TYPE_INT32:
362 return ReadPrimitive<int32, int32, WireFormatLite::TYPE_INT32>(
363 input, index, datap);
364 case WireFormatLite::TYPE_FIXED64:
365 return ReadPrimitive<protobuf_uint64, int64,
366 WireFormatLite::TYPE_FIXED64>(input, index, datap);
367 case WireFormatLite::TYPE_FIXED32:
368 if (dtype == DataType::DT_INT64) {
369 return ReadPrimitive<uint32, int64, WireFormatLite::TYPE_FIXED32>(
370 input, index, datap);
371 }
372 if (dtype == DataType::DT_INT32) {
373 return ReadPrimitive<uint32, int32, WireFormatLite::TYPE_FIXED32>(
374 input, index, datap);
375 }
376 // Any case that reaches this point should have triggered an error
377 // already.
378 return errors::DataLoss("Failed reading TYPE_FIXED32");
379 case WireFormatLite::TYPE_BOOL:
380 return ReadPrimitive<bool, bool, WireFormatLite::TYPE_BOOL>(input, index,
381 datap);
382 case WireFormatLite::TYPE_STRING:
383 return ReadBytes(input, index, datap);
384 case WireFormatLite::TYPE_GROUP:
385 return ReadGroupBytes(input, field_number, index, datap);
386 case WireFormatLite::TYPE_MESSAGE:
387 return ReadBytes(input, index, datap);
388 case WireFormatLite::TYPE_BYTES:
389 return ReadBytes(input, index, datap);
390 case WireFormatLite::TYPE_UINT32:
391 if (dtype == DataType::DT_INT64) {
392 return ReadPrimitive<uint32, int64, WireFormatLite::TYPE_UINT32>(
393 input, index, datap);
394 }
395 if (dtype == DataType::DT_INT32) {
396 return ReadPrimitive<uint32, int32, WireFormatLite::TYPE_UINT32>(
397 input, index, datap);
398 }
399 // Any case that reaches this point should have triggered an error
400 // already.
401 return errors::DataLoss("Failed reading TYPE_UINT32");
402 case WireFormatLite::TYPE_ENUM:
403 return ReadPrimitive<int32, int32, WireFormatLite::TYPE_ENUM>(
404 input, index, datap);
405 case WireFormatLite::TYPE_SFIXED32:
406 return ReadPrimitive<int32, int32, WireFormatLite::TYPE_SFIXED32>(
407 input, index, datap);
408 case WireFormatLite::TYPE_SFIXED64:
409 return ReadPrimitive<protobuf_int64, int64,
410 WireFormatLite::TYPE_SFIXED64>(input, index, datap);
411 case WireFormatLite::TYPE_SINT32:
412 return ReadPrimitive<int32, int32, WireFormatLite::TYPE_SINT32>(
413 input, index, datap);
414 case WireFormatLite::TYPE_SINT64:
415 return ReadPrimitive<protobuf_int64, int64, WireFormatLite::TYPE_SINT64>(
416 input, index, datap);
417 // default: intentionally omitted in order to enable static checking.
418 }
419 // Unreachable.
420 return errors::DataLoss("Failed reading unknown wire type");
421}

423// Reads and stores a length-delimited list of values.
424inline Status ReadPackedFromArray(const void* buf, size_t buf_size,
425 const WireFormatLite::FieldType field_type,
426 const int field_number, const DataType dtype,
427 const int stride, int* index, void* data) {
428 // Dispatch to the appropriately typed field reader based on the
429 // schema type.
430 switch (field_type) {
431 case WireFormatLite::TYPE_DOUBLE:
432 *index += ReadPackedPrimitives<double, WireFormatLite::TYPE_DOUBLE>(
433 buf, buf_size, *index, stride, data);
434 return Status::OK();
435 case WireFormatLite::TYPE_FLOAT:
436 *index += ReadPackedPrimitives<float, WireFormatLite::TYPE_FLOAT>(
437 buf, buf_size, *index, stride, data);
438 return Status::OK();
439 case WireFormatLite::TYPE_INT64:
440 *index += ReadPackedPrimitives<int64, WireFormatLite::TYPE_INT64>(
441 buf, buf_size, *index, stride, data);
442 return Status::OK();
443 case WireFormatLite::TYPE_UINT64:
444 *index += ReadPackedPrimitives<int64, WireFormatLite::TYPE_UINT64>(
445 buf, buf_size, *index, stride, data);
446 return Status::OK();
447 case WireFormatLite::TYPE_INT32:
448 *index += ReadPackedPrimitives<int32, WireFormatLite::TYPE_INT32>(
449 buf, buf_size, *index, stride, data);
450 return Status::OK();
451 case WireFormatLite::TYPE_FIXED64:
452 *index += ReadPackedPrimitives<int64, WireFormatLite::TYPE_FIXED64>(
453 buf, buf_size, *index, stride, data);
454 return Status::OK();
455 case WireFormatLite::TYPE_FIXED32:
456 if (dtype == DataType::DT_INT64) {
457 *index += ReadPackedPrimitives<int64, WireFormatLite::TYPE_FIXED32>(
458 buf, buf_size, *index, stride, data);
459 return Status::OK();
460 }
461 if (dtype == DataType::DT_INT32) {
462 *index += ReadPackedPrimitives<int32, WireFormatLite::TYPE_FIXED32>(
463 buf, buf_size, *index, stride, data);
464 return Status::OK();
465 }
466 // Any case that reaches this point should have triggered an error
467 // already.
468 return errors::DataLoss("Failed reading TYPE_FIXED32");
469 case WireFormatLite::TYPE_BOOL:
470 *index += ReadPackedPrimitives<bool, WireFormatLite::TYPE_BOOL>(
471 buf, buf_size, *index, stride, data);
472 return Status::OK();
473 case WireFormatLite::TYPE_STRING:
474 case WireFormatLite::TYPE_GROUP:
475 case WireFormatLite::TYPE_MESSAGE:
476 case WireFormatLite::TYPE_BYTES:
477 return errors::DataLoss("Non-primitive type encountered as packed");
478 case WireFormatLite::TYPE_UINT32:
479 if (dtype == DataType::DT_INT64) {
480 *index += ReadPackedPrimitives<int64, WireFormatLite::TYPE_UINT32>(
481 buf, buf_size, *index, stride, data);
482 return Status::OK();
483 }
484 if (dtype == DataType::DT_INT32) {
485 *index += ReadPackedPrimitives<int32, WireFormatLite::TYPE_UINT32>(
486 buf, buf_size, *index, stride, data);
487 return Status::OK();
488 }
489 // Any case that reaches this point should have triggered an error
490 // already.
491 return errors::DataLoss("Failed reading TYPE_UINT32");
492 case WireFormatLite::TYPE_ENUM:
493 *index += ReadPackedPrimitives<int32, WireFormatLite::TYPE_ENUM>(
494 buf, buf_size, *index, stride, data);
495 return Status::OK();
496 case WireFormatLite::TYPE_SFIXED32:
497 *index += ReadPackedPrimitives<int32, WireFormatLite::TYPE_SFIXED32>(
498 buf, buf_size, *index, stride, data);
499 return Status::OK();
500
501 case WireFormatLite::TYPE_SFIXED64:
502 *index += ReadPackedPrimitives<int64, WireFormatLite::TYPE_SFIXED64>(
503 buf, buf_size, *index, stride, data);
504 return Status::OK();
505
506 case WireFormatLite::TYPE_SINT32:
507 *index += ReadPackedPrimitives<int32, WireFormatLite::TYPE_SINT32>(
508 buf, buf_size, *index, stride, data);
509 return Status::OK();
510
511 case WireFormatLite::TYPE_SINT64:
512 *index += ReadPackedPrimitives<int64, WireFormatLite::TYPE_SINT64>(
513 buf, buf_size, *index, stride, data);
514 return Status::OK();
515 // default: intentionally omitted in order to enable static checking.
516 }
517 // Unreachable.
518 return errors::DataLoss("Failed reading unknown wire type");
519}

521// Reads a varint from the given buffer, write it to *value, and return the
522// new buffer pointer.
523// This was copied from coded_stream.cc where it is private.
524// Important: This routine may read as much as kMaxVarintBytes from
525// the buffer. It is the caller's responsibility to make sure that there is
526// enough space in the buffer.
527inline const uint8* ReadVarint64FromArray(const uint8* buffer, bool* ok,
528 uint64* value) {
529 const uint8* ptr = buffer;
530 uint32 b;
531
532 // Splitting into 32-bit pieces gives better performance on 32-bit
533 // processors.
534 uint32 part0 = 0, part1 = 0, part2 = 0;
535
536 b = *(ptr++);
537 part0 = b;
538 if (!(b & 0x80)) goto done;
539 part0 -= 0x80;
540 b = *(ptr++);
541 part0 += b << 7;
542 if (!(b & 0x80)) goto done;
543 part0 -= 0x80 << 7;
544 b = *(ptr++);
545 part0 += b << 14;
546 if (!(b & 0x80)) goto done;
547 part0 -= 0x80 << 14;
548 b = *(ptr++);
549 part0 += b << 21;
550 if (!(b & 0x80)) goto done;
551 part0 -= 0x80 << 21;
552 b = *(ptr++);
553 part1 = b;
554 if (!(b & 0x80)) goto done;
555 part1 -= 0x80;
556 b = *(ptr++);
557 part1 += b << 7;
558 if (!(b & 0x80)) goto done;
559 part1 -= 0x80 << 7;
560 b = *(ptr++);
561 part1 += b << 14;
562 if (!(b & 0x80)) goto done;
563 part1 -= 0x80 << 14;
564 b = *(ptr++);
565 part1 += b << 21;
566 if (!(b & 0x80)) goto done;
567 part1 -= 0x80 << 21;
568 b = *(ptr++);
569 part2 = b;
570 if (!(b & 0x80)) goto done;
571 part2 -= 0x80;
572 b = *(ptr++);
573 part2 += b << 7;
574 if (!(b & 0x80)) goto done;
575 // "part2 -= 0x80 << 7" is irrelevant because (0x80 << 7) << 56 is 0.
576
577 // We have overrun the maximum size of a varint (10 bytes). Assume
578 // the data is corrupt.
579 *ok = false;
580 return ptr;
581
582done:
583 *ok = true;
584 *value = (static_cast<uint64>(part0)) | (static_cast<uint64>(part1) << 28) |
585 (static_cast<uint64>(part2) << 56);
586 return ptr;
587}
