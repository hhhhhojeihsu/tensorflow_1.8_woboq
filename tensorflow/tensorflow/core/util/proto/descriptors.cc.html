25// Build a `DescriptorPool` from the named file or URI. The file or URI
26// must be available to the current TensorFlow environment.
27//
28// The file must contiain a serialized `FileDescriptorSet`. See
29// `GetDescriptorPool()` for more information.
30Status GetDescriptorPoolFromFile(
31 tensorflow::Env* env, const string& filename,
32 std::unique_ptr<tensorflow::protobuf::DescriptorPool>* owned_desc_pool) {
33 Status st = env->FileExists(filename);
34 if (!st.ok()) {
35 return st;
36 }
37
38 // Read and parse the FileDescriptorSet.
39 tensorflow::protobuf::FileDescriptorSet descs;
40 std::unique_ptr<tensorflow::ReadOnlyMemoryRegion> buf;
41 st = env->NewReadOnlyMemoryRegionFromFile(filename, &buf);
42 if (!st.ok()) {
43 return st;
44 }
45 if (!descs.ParseFromArray(buf->data(), buf->length())) {
46 return errors::InvalidArgument(
47 "descriptor_source contains invalid FileDescriptorSet: ", filename);
48 }
49
50 // Build a DescriptorPool from the FileDescriptorSet.
51 owned_desc_pool->reset(new tensorflow::protobuf::DescriptorPool());
52 for (const auto& filedesc : descs.file()) {
53 if ((*owned_desc_pool)->BuildFile(filedesc) == nullptr) {
54 return errors::InvalidArgument(
55 "Problem loading FileDescriptorProto (missing dependencies?): ",
56 filename);
57 }
58 }
59 return Status::OK();
60}

62} // namespace

64Status GetDescriptorPool(
65 tensorflow::Env* env, string const& descriptor_source,
66 tensorflow::protobuf::DescriptorPool const** desc_pool,
67 std::unique_ptr<tensorflow::protobuf::DescriptorPool>* owned_desc_pool) {
68 // Attempt to lookup the pool in the registry.
69 auto pool_fn = DescriptorPoolRegistry::Global()->Get(descriptor_source);
70 if (pool_fn != nullptr) {
71 return (*pool_fn)(desc_pool, owned_desc_pool);
72 }
73
74 // If there is no pool function registered for the given source, let the
75 // runtime find the file or URL.
76 Status status =
77 GetDescriptorPoolFromFile(env, descriptor_source, owned_desc_pool);
78 if (status.ok()) {
79 *desc_pool = owned_desc_pool->get();
80 }
81 *desc_pool = owned_desc_pool->get();
82 return status;
83}
