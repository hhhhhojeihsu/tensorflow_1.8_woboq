19#include <map>
20#include <string>
21
22#include "tensorflow/core/framework/op_kernel.h"
23#include "tensorflow/core/util/rpc/rpc_factory.h"

25namespace tensorflow {
26
27class RPCFactoryRegistry {
28 public:
29 typedef std::function<RPCFactory*(OpKernelConstruction* ctx, bool fail_fast,
30 int64 timeout_in_ms)>
31 RPCFactoryFn;
32
33 // Returns a pointer to a global RPCFactoryRegistry object.
34 static RPCFactoryRegistry* Global();
35
36 // Returns a pointer to an function that creates an RPC factory for the given
37 // protocol.
38 RPCFactoryFn* Get(const string& protocol);
39
40 // Registers a function that creates and RPC factory for the given protocol.
41 // The function should transfer the ownership of the factory to its caller.
42 void Register(const string& protocol, const RPCFactoryFn& factory_fn);
43
44 private:
45 std::map<string, RPCFactoryFn> fns_;
46};
47
48namespace rpc_factory_registration {
49
50class RPCFactoryRegistration {
51 public:
52 RPCFactoryRegistration(const string& protocol,
53 const RPCFactoryRegistry::RPCFactoryFn& factory_fn) {
54 RPCFactoryRegistry::Global()->Register(protocol, factory_fn);
55 }
56};
57
58} // namespace rpc_factory_registration
59
60#define REGISTER_RPC_FACTORY(protocol, factory_fn) \

63#define REGISTER_RPC_FACTORY_UNIQ_HELPER(ctr, protocol, factory_fn) \
64 REGISTER_RPC_FACTORY_UNIQ(ctr, protocol, factory_fn)
65
66#define REGISTER_RPC_FACTORY_UNIQ(ctr, protocol, factory_fn) \
67 static rpc_factory_registration::RPCFactoryRegistration \
68 rpc_factory_registration_fn_##ctr(protocol, factory_fn)
