16#ifndef TENSORFLOW_KERNELS_REVERSE_OP_H_
17#define TENSORFLOW_KERNELS_REVERSE_OP_H_
18
19#include "third_party/eigen3/unsupported/Eigen/CXX11/Tensor"
20#include "tensorflow/core/framework/tensor_types.h"
21
22namespace tensorflow {
23namespace functor {
24
25// Functor used by ReverseOp to do the computations.
26template <typename Device, typename T, int Dims>
27struct Reverse {
28 void operator()(const Device& d, typename TTypes<T, Dims>::ConstTensor input,
29 const Eigen::array<bool, Dims>& reverse_dims,
30 typename TTypes<T, Dims>::Tensor output) {
31 output.device(d) = input.reverse(reverse_dims);
32 }
33};
34
35template <typename Device, typename T>
36struct Reverse<Device, T, 0> {
37 void operator()(const Device& d, typename TTypes<T, 0>::ConstTensor input,
38 const Eigen::array<bool, 0>& reverse_dims,
39 typename TTypes<T, 0>::Tensor output) {
40 // Reversing a scalar is copying it.
41 output.device(d) = input;
42 }
43};
44
45} // namespace functor
46} // namespace tensorflow

48#endif // TENSORFLOW_KERNELS_MIRROR_OP_H_
