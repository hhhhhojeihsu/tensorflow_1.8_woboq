16#include "tensorflow/core/kernels/cast_op_impl.h"
17
18namespace tensorflow {
19
20typedef Eigen::ThreadPoolDevice CPUDevice;
21typedef Eigen::GpuDevice GPUDevice;
22
23std::function<void(OpKernelContext*, const Tensor&, Tensor*)>
24GetCpuCastFromUint16(DataType dst_dtype) {
25 CURRY_TYPES3(CAST_CASE, CPUDevice, uint16);
26 return nullptr;
27}
28
29#if GOOGLE_CUDA
30std::function<void(OpKernelContext*, const Tensor&, Tensor*)>
31GetGpuCastFromUint16(DataType dst_dtype) {
32 CURRY_TYPES3_NO_BF16(CAST_CASE, GPUDevice, uint16);
33 return nullptr;
34}
35#endif // GOOGLE_CUDA
36
37#ifdef TENSORFLOW_USE_SYCL

40GetSyclCastFromUint16(DataType dst_dtype) {

42 return nullptr;
43}
44#endif // TENSORFLOW_USE_SYCL
45
46} // namespace tensorflow
