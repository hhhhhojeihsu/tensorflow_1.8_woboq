34// Lookup table op that supports different table implementations specified by
35// the 'Container' template. Container must be derived from LookupInterface. The
36// key and value are of the templated type "key_dtype" and "value_dtype"
37// respectively.
38template <class Container, class key_dtype, class value_dtype>
39class LookupTableOp : public OpKernel {
40 public:
41 // ctx is not owned by this class.
42 explicit LookupTableOp(OpKernelConstruction* ctx)
43 : OpKernel(ctx), table_handle_set_(false) {
44 OP_REQUIRES_OK(ctx, ctx->allocate_persistent(tensorflow::DT_STRING,
45 tensorflow::TensorShape({2}),
46 &table_handle_, nullptr));
47 OP_REQUIRES_OK(
48 ctx, ctx->GetAttr("use_node_name_sharing", &use_node_name_sharing_));
49 }
50
51 // ctx is not owned by this function.
52 void Compute(OpKernelContext* ctx) override {
53 mutex_lock l(mu_);
54
55 if (!table_handle_set_) {
56 OP_REQUIRES_OK(ctx, cinfo_.Init(ctx->resource_manager(), def(),
57 use_node_name_sharing_));
58 }
59
60 auto creator = [ctx, this](lookup::LookupInterface** ret) {
61 lookup::LookupInterface* container = new Container(ctx, this);
62 if (!ctx->status().ok()) {
63 container->Unref();
64 return ctx->status();
65 }
66 if (ctx->track_allocations()) {
67 ctx->record_persistent_memory_allocation(
68 container->MemoryUsed() + table_handle_.AllocatedBytes());
69 }
70 *ret = container;
71 return Status::OK();
72 };
73
74 lookup::LookupInterface* table = nullptr;
75 OP_REQUIRES_OK(ctx,
76 cinfo_.resource_manager()
77 ->template LookupOrCreate<lookup::LookupInterface>(
78 cinfo_.container(), cinfo_.name(), &table, creator));
79 core::ScopedUnref unref_me(table);
80
81 OP_REQUIRES_OK(ctx, lookup::CheckTableDataTypes(
82 *table, DataTypeToEnum<key_dtype>::v(),
83 DataTypeToEnum<value_dtype>::v(), cinfo_.name()));
84
85 if (ctx->expected_output_dtype(0) == DT_RESOURCE) {
86 Tensor* handle;
87 OP_REQUIRES_OK(ctx, ctx->allocate_output(0, TensorShape({}), &handle));
88 handle->scalar<ResourceHandle>()() =
89 MakeResourceHandle<lookup::LookupInterface>(ctx, cinfo_.container(),
90 cinfo_.name());
91 } else {
92 if (!table_handle_set_) {
93 auto h = table_handle_.AccessTensor(ctx)->template flat<string>();
94 h(0) = cinfo_.container();
95 h(1) = cinfo_.name();
96 }
97 ctx->set_output_ref(0, &mu_, table_handle_.AccessTensor(ctx));
98 }
99 table_handle_set_ = true;
100 }
101
102 ~LookupTableOp() override {
103 // If the table object was not shared, delete it.
104 if (table_handle_set_ && cinfo_.resource_is_private_to_kernel()) {
105 TF_CHECK_OK(
106 cinfo_.resource_manager()->template Delete<lookup::LookupInterface>(
107 cinfo_.container(), cinfo_.name()));
108 }
109 }
110
111 private:
112 mutex mu_;
113 PersistentTensor table_handle_ GUARDED_BY(mu_);
114 bool table_handle_set_ GUARDED_BY(mu_);
115 ContainerInfo cinfo_;
116 bool use_node_name_sharing_;
117
118 TF_DISALLOW_COPY_AND_ASSIGN(LookupTableOp);
119};

123// Ensure that the compiler cannot elide a copy into a local, for
124// bounds checking on source tensors that might be updated asynchronously for
125// integral types. However non-integer variables are not allowed and therefore
126// the local copy is unnecessary.
127template <typename T>
128T SubtleMustCopyIfIntegral(const T& value) {
129 return internal::SubtleMustCopy(value);
130}
131
132inline const string& SubtleMustCopyIfIntegral(const string& value) {
133 return value;
134}
135
136inline const float SubtleMustCopyIfIntegral(const float value) { return value; }
137
138inline const double SubtleMustCopyIfIntegral(const double value) {
139 return value;
140}
141
142inline const Variant& SubtleMustCopyIfIntegral(const Variant& value) {
143 return value;
144}

146// Lookup table that wraps an unordered_map, where the key and value data type
147// is specified.
148//
149// This table is recommended for any variations to key values.
150//
151// For look up, the table is required to be initialized (allocated
152// and populated). Once the table is marked as initialized it becomes read-only.
153//
154// Sample use case:
155//
156// HashTable<int64, int64> table; // int64 -> int64.
157// table.Prepare(10); // Prepare the underlying data structure, the number of
158// // elements is required by interface, but not used.
159// // Populate the table, elements could be added in one or multiple calls.
160// table.Insert(key_tensor, value_tensor); // Populate the table.
161// ...
162// table.set_is_initialized();
163//
164// table.Find(in_t, &out_t, default_t)
165//
166template <class K, class V>
167class HashTable : public InitializableLookupTable {
168 public:
169 HashTable(OpKernelContext* ctx, OpKernel* kernel) {}
170
171 size_t size() const override {
172 // return the size of the table only if it's initialized, otherwise 0.
173 if (!is_initialized_) {
174 return 0;
175 }
176 std::atomic_thread_fence(std::memory_order_acquire);
177 return table_ ? table_->size() : 0;
178 }
179
180 DataType key_dtype() const override { return DataTypeToEnum<K>::v(); }
181
182 DataType value_dtype() const override { return DataTypeToEnum<V>::v(); }
183
184 protected:
185 Status DoPrepare(size_t unused) override {
186 if (is_initialized_) {
187 return errors::Aborted("HashTable already initialized.");
188 }
189 if (!table_) {
190 table_ = std::unique_ptr<std::unordered_map<K, V>>(
191 new std::unordered_map<K, V>());
192 }
193 return Status::OK();
194 };
195
196 Status DoLazyPrepare(std::function<int64(void)> unused) override {
197 constexpr size_t kUnusedSize = 0;
198 return DoPrepare(kUnusedSize);
199 }
200
201 Status DoInsert(const Tensor& keys, const Tensor& values) override {
202 if (!table_) {
203 return errors::FailedPrecondition("HashTable is not prepared.");
204 }
205
206 const auto key_values = keys.flat<K>();
207 const auto value_values = values.flat<V>();
208 for (int64 i = 0; i < key_values.size(); ++i) {
209 const K key = SubtleMustCopyIfIntegral(key_values(i));
210 const V value = SubtleMustCopyIfIntegral(value_values(i));
211 const V& previous_value = gtl::LookupOrInsert(table_.get(), key, value);
212 if (previous_value != value) {
213 return errors::FailedPrecondition(
214 "HashTable has different value for same key. Key ", key, " has ",
215 previous_value, " and trying to add value ", value);
216 }
217 }
218 return Status::OK();
219 }
220
221 Status DoFind(const Tensor& key, Tensor* value,
222 const Tensor& default_value) override {
223 const V default_val = default_value.flat<V>()(0);
224 const auto key_values = key.flat<K>();
225 auto value_values = value->flat<V>();
226
227 for (int64 i = 0; i < key_values.size(); ++i) {
228 value_values(i) = gtl::FindWithDefault(
229 *table_, SubtleMustCopyIfIntegral(key_values(i)), default_val);
230 }
231 return Status::OK();
232 }
233
234 int64 MemoryUsed() const override {
235 if (table_) {
236 const int64 num_elements = table_->size();
237 return num_elements * (sizeof(K) + sizeof(V));
238 } else {
239 return 0;
240 }
241 }
242
243 private:
244 std::unique_ptr<std::unordered_map<K, V>> table_;
245};
