59template <typename Device, typename T>
60struct LaunchGeneric {
61 void operator()(OpKernelContext* ctx, const Tensor& input,
62 const Tensor& filter, int row_stride, int col_stride,
63 int row_dilation, int col_dilation, const Padding& padding,
64 Tensor* output, TensorFormat data_format) {
65 CHECK(data_format == FORMAT_NHWC) << "Generic conv implementation only "
66 "supports NHWC tensor format for now.";
67 if (filter.dim_size(0) == 1 && filter.dim_size(1) == 1 && row_stride == 1 &&
68 col_stride == 1) {
69 // For 1x1 kernel, the 2D convolution is reduced to matrix
70 // multiplication.
71 //
72 // TODO(vrv): We should be able to call SpatialConvolution
73 // and it will produce the same result, but doing so
74 // led to NaNs during training. Using matmul instead for now.
75 int conv_width = 1; // Width for the convolution step.
76 for (int i = 0; i < 3; ++i) {
77 conv_width *= output->dim_size(i);
78 }
79
80 Eigen::array<Eigen::IndexPair<Eigen::DenseIndex>, 1> dim_pair;
81 dim_pair[0] = Eigen::IndexPair<Eigen::DenseIndex>(1, 0);
82 functor::MatMulConvFunctor<Device, T>()(
83 ctx->eigen_device<Device>(),
84 output->shaped<T, 2>({conv_width, filter.dim_size(3)}),
85 input.shaped<T, 2>({conv_width, filter.dim_size(2)}),
86 filter.shaped<T, 2>({filter.dim_size(2), filter.dim_size(3)}),
87 dim_pair);
88 } else if (filter.dim_size(0) == input.dim_size(1) &&
89 filter.dim_size(1) == input.dim_size(2) && row_dilation == 1 &&
90 col_dilation == 1 && padding == VALID) {
91 // If the input data and filter have the same height/width,
92 // the 2D convolution is reduced to matrix multiplication.
93 const int k = // Length of reduction dimension.
94 filter.dim_size(0) * filter.dim_size(1) * filter.dim_size(2);
95
96 Eigen::array<Eigen::IndexPair<Eigen::DenseIndex>, 1> dim_pair;
97 dim_pair[0] = Eigen::IndexPair<Eigen::DenseIndex>(1, 0);
98 functor::MatMulConvFunctor<Device, T>()(
99 ctx->eigen_device<Device>(),
100 output->shaped<T, 2>({input.dim_size(0), filter.dim_size(3)}),
101 input.shaped<T, 2>({input.dim_size(0), k}),
102 filter.shaped<T, 2>({k, filter.dim_size(3)}), dim_pair);
103 } else {
104 functor::SpatialConvolution<Device, T>()(
105 ctx->eigen_device<Device>(), output->tensor<T, 4>(),
106 input.tensor<T, 4>(), filter.tensor<T, 4>(), row_stride, col_stride,
107 row_dilation, col_dilation, BrainPadding2EigenPadding(padding));
108 }
109 }
110};
111} // namespace

113template <typename T>
114struct LaunchConv2DOp<CPUDevice, T> {
115 void operator()(OpKernelContext* ctx, bool use_cudnn, bool cudnn_use_autotune,
116 const Tensor& input, const Tensor& filter, int row_dilation,
117 int col_dilation, int row_stride, int col_stride,
118 const Padding& padding, Tensor* output,
119 TensorFormat data_format) {
120 if (data_format != FORMAT_NHWC) {
121 ctx->SetStatus(
122 errors::Unimplemented("Generic conv implementation only supports "
123 "NHWC tensor format for now."));
124 return;
125 }
126 LaunchGeneric<CPUDevice, T>()(ctx, input, filter, row_stride, col_stride,
127 row_dilation, col_dilation, padding, output,
128 data_format);
129 }

147// Conditionally launches DeepConv operation based on convolution parameters.
148template <>
149class LaunchDeepConvOp<CPUDevice, float> {
150 public:
151 static bool Run(OpKernelContext* ctx, const Tensor& input,
152 const Tensor& filter, int batch, int input_rows,
153 int input_cols, int in_depth, int filter_rows,
154 int filter_cols, int pad_rows, int pad_cols, int out_rows,
155 int out_cols, int out_depth, int dilation_rows,
156 int dilation_cols, int stride_rows, int stride_cols,
157 Tensor* output, TensorFormat data_format) {
158 if (data_format != FORMAT_NHWC || dilation_rows != 1 ||
159 dilation_cols != 1 ||
160 !CanUseDeepConv2D(stride_rows, stride_cols, filter_rows, filter_cols,
161 in_depth, out_depth, out_rows, out_cols)) {
162 return false;
163 }
164
165 Conv2DArgs args;
166 args.batch = batch;
167 args.in_rows = input_rows;
168 args.in_cols = input_cols;
169 args.in_depth = in_depth;
170 args.filter_rows = filter_rows;
171 args.filter_cols = filter_cols;
172 args.pad_rows = pad_rows;
173 args.pad_cols = pad_cols;
174 args.out_rows = out_rows;
175 args.out_cols = out_cols;
176 args.out_depth = out_depth;
177
178 auto input_ptr = input.template flat<float>().data();
179 auto filter_ptr = filter.template flat<float>().data();
180 auto output_ptr = output->template flat<float>().data();
181
182 functor::DeepConv2D<CPUDevice, float>()(ctx, args, input_ptr, filter_ptr,
183 output_ptr);
184 return true;
185 }

256template <typename Device, typename T>
257class Conv2DOp : public BinaryOp<T> {
258 public:
259 explicit Conv2DOp(OpKernelConstruction* context) : BinaryOp<T>(context) {
260 OP_REQUIRES_OK(context, context->GetAttr("dilations", &dilations_));
261 OP_REQUIRES_OK(context, context->GetAttr("strides", &strides_));
262 string data_format;
263 OP_REQUIRES_OK(context, context->GetAttr("data_format", &data_format));
264 OP_REQUIRES(context, FormatFromString(data_format, &data_format_),
265 errors::InvalidArgument("Invalid data format"));
266 OP_REQUIRES_OK(context, context->GetAttr("use_cudnn_on_gpu", &use_cudnn_));
267 use_cudnn_ &= CanUseCudnn();
268 cudnn_use_autotune_ = CudnnUseAutotune();
269 OP_REQUIRES(context, dilations_.size() == 4,
270 errors::InvalidArgument("Sliding window dilations field must "
271 "specify 4 dimensions"));
272 OP_REQUIRES(context, strides_.size() == 4,
273 errors::InvalidArgument("Sliding window strides field must "
274 "specify 4 dimensions"));
275 const int64 stride_n = GetTensorDim(strides_, data_format_, 'N');
276 const int64 stride_c = GetTensorDim(strides_, data_format_, 'C');
277 const int64 stride_h = GetTensorDim(strides_, data_format_, 'H');
278 const int64 stride_w = GetTensorDim(strides_, data_format_, 'W');
279 OP_REQUIRES(
280 context, stride_n == 1 && stride_c == 1,
281 errors::InvalidArgument("Current implementation does not yet support "
282 "strides in the batch and depth dimensions."));
283 OP_REQUIRES(context, stride_h > 0 && stride_w > 0,
284 errors::InvalidArgument(
285 "Row and column strides should be larger than 0."));
286
287 const int64 dilation_n = GetTensorDim(dilations_, data_format_, 'N');
288 const int64 dilation_c = GetTensorDim(dilations_, data_format_, 'C');
289 const int64 dilation_h = GetTensorDim(dilations_, data_format_, 'H');
290 const int64 dilation_w = GetTensorDim(dilations_, data_format_, 'W');
291 OP_REQUIRES(context, dilation_n == 1 && dilation_c == 1,
292 errors::InvalidArgument(
293 "Current implementation does not yet support "
294 "dilations in the batch and depth dimensions."));
295 OP_REQUIRES(
296 context, dilation_h > 0 && dilation_w > 0,
297 errors::InvalidArgument("Dilated rates should be larger than 0."));
298 OP_REQUIRES_OK(context, context->GetAttr("padding", &padding_));
299 }
300
301 void Compute(OpKernelContext* context) override {
302 // Input tensor is of the following dimensions:
303 // [ batch, in_rows, in_cols, in_depth ]
304
305 const Tensor& input = context->input(0);
306
307 // Input filter is of the following dimensions:
308 // [ filter_rows, filter_cols, in_depth, out_depth]
309 const Tensor& filter = context->input(1);
310
311 // For 2D convolution, there should be 4 dimensions.
312 OP_REQUIRES(context, input.dims() == 4,
313 errors::InvalidArgument("input must be 4-dimensional",
314 input.shape().DebugString()));
315 OP_REQUIRES(context, filter.dims() == 4,
316 errors::InvalidArgument("filter must be 4-dimensional: ",
317 filter.shape().DebugString()));
318
319 for (int i = 0; i < 3; i++) {
320 OP_REQUIRES(
321 context,
322 FastBoundsCheck(filter.dim_size(i), std::numeric_limits<int>::max()),
323 errors::InvalidArgument("filter too large"));
324 }
325
326 // The last dimension for input is in_depth. It must be the same as the
327 // filter's in_depth.
328 const int64 in_depth = GetTensorDim(input, data_format_, 'C');
329 OP_REQUIRES(context, in_depth == filter.dim_size(2),
330 errors::InvalidArgument(
331 "input and filter must have the same depth: ", in_depth,
332 " vs ", filter.dim_size(2)));
333
334 // The last dimension for filter is out_depth.
335 const int out_depth = static_cast<int>(filter.dim_size(3));
336
337 // The second dimension for input is rows/height.
338 // The first dimension for filter is rows/height.
339 const int64 input_rows_raw = GetTensorDim(input, data_format_, 'H');
340 OP_REQUIRES(
341 context,
342 FastBoundsCheck(input_rows_raw, std::numeric_limits<int>::max()),
343 errors::InvalidArgument("Input rows too large"));
344 const int input_rows = static_cast<int>(input_rows_raw);
345 const int filter_rows = static_cast<int>(filter.dim_size(0));
346
347 // The third dimension for input is columns/width.
348 // The second dimension for filter is columns/width.
349 const int64 input_cols_raw = GetTensorDim(input, data_format_, 'W');
350 OP_REQUIRES(
351 context,
352 FastBoundsCheck(input_cols_raw, std::numeric_limits<int>::max()),
353 errors::InvalidArgument("Input cols too large"));
354 const int input_cols = static_cast<int>(input_cols_raw);
355 const int filter_cols = static_cast<int>(filter.dim_size(1));
356
357 // The first dimension for input is batch.
358 const int64 batch_raw = GetTensorDim(input, data_format_, 'N');
359 OP_REQUIRES(context,
360 FastBoundsCheck(batch_raw, std::numeric_limits<int>::max()),
361 errors::InvalidArgument("batch is too large"));
362 const int batch = static_cast<int>(batch_raw);
363
364 // For now we take the stride and dilation from the second and third
365 // dimensions only (we do not support striding or dilation on the batch or
366 // depth dimension).
367 const int stride_rows = GetTensorDim(strides_, data_format_, 'H');
368 const int stride_cols = GetTensorDim(strides_, data_format_, 'W');
369
370 const int dilation_rows = GetTensorDim(dilations_, data_format_, 'H');
371 const int dilation_cols = GetTensorDim(dilations_, data_format_, 'W');
372
373 int64 out_rows = 0, out_cols = 0, pad_rows = 0, pad_cols = 0;
374 OP_REQUIRES_OK(context, GetWindowedOutputSizeV2(
375 input_rows, filter_rows, dilation_rows,
376 stride_rows, padding_, &out_rows, &pad_rows));
377 OP_REQUIRES_OK(context, GetWindowedOutputSizeV2(
378 input_cols, filter_cols, dilation_cols,
379 stride_cols, padding_, &out_cols, &pad_cols));
380 TensorShape out_shape =
381 ShapeFromFormat(data_format_, batch, out_rows, out_cols, out_depth);
382
383 // Output tensor is of the following dimensions:
384 // [ in_batch, out_rows, out_cols, out_depth ]
385 Tensor* output = nullptr;
386 OP_REQUIRES_OK(context, context->allocate_output(0, out_shape, &output));
387
388 VLOG(2) << "Conv2D: in_depth = " << in_depth
389 << ", input_cols = " << input_cols
390 << ", filter_cols = " << filter_cols
391 << ", input_rows = " << input_rows
392 << ", filter_rows = " << filter_rows
393 << ", stride_rows = " << stride_rows
394 << ", stride_cols = " << stride_cols
395 << ", dilation_rows = " << dilation_rows
396 << ", dilation_cols = " << dilation_cols
397 << ", out_depth = " << out_depth;
398
399 // If there is nothing to compute, return.
400 if (out_shape.num_elements() == 0) {
401 return;
402 }
403
404#ifdef TENSORFLOW_USE_LIBXSMM_CONVOLUTIONS
405 if (LaunchXsmmConvOp<Device, T>::Run(
406 context, input, filter, batch, input_rows, input_cols, in_depth,
407 filter_rows, filter_cols, pad_rows, pad_cols, out_rows, out_cols,
408 out_depth, dilation_rows, dilation_cols, stride_rows, stride_cols,
409 output, data_format_)) {
410 return;
411 }
412#endif
413
414 if (LaunchDeepConvOp<Device, T>::Run(
415 context, input, filter, batch, input_rows, input_cols, in_depth,
416 filter_rows, filter_cols, pad_rows, pad_cols, out_rows, out_cols,
417 out_depth, dilation_rows, dilation_cols, stride_rows, stride_cols,
418 output, data_format_)) {
419 return;
420 }
421
422 launcher_(context, use_cudnn_, cudnn_use_autotune_, input, filter,
423 dilation_rows, dilation_cols, stride_rows, stride_cols, padding_,
424 output, data_format_);
425 }
426
427 private:
428 std::vector<int32> dilations_;
429 std::vector<int32> strides_;
430 bool use_cudnn_;
431 Padding padding_;
432 TensorFormat data_format_;
433 LaunchConv2DOp<Device, T> launcher_;
434 bool cudnn_use_autotune_;
435
436 TF_DISALLOW_COPY_AND_ASSIGN(Conv2DOp);
437};

481template <typename T>
482void LaunchConv2DOp<GPUDevice, T>::operator()(
483 OpKernelContext* ctx, bool use_cudnn, bool cudnn_use_autotune,
484 const Tensor& input_param, const Tensor& filter, int row_dilation,
485 int col_dilation, int row_stride, int col_stride, const Padding& padding,
486 Tensor* output, TensorFormat data_format) {
487 using perftools::gputools::dnn::AlgorithmConfig;
488 using perftools::gputools::dnn::AlgorithmDesc;
489 using perftools::gputools::dnn::ProfileResult;
490 auto* stream = ctx->op_device_context()->stream();
491 OP_REQUIRES(ctx, stream, errors::Internal("No GPU stream available."));
492
493 if (!use_cudnn) {
494 ctx->SetStatus(
495 errors::Unimplemented("Conv2D for GPU is not currently supported "
496 "without cudnn"));
497 return;
498 }
499
500 Tensor input = input_param;
501
502 if (filter.dim_size(0) == 1 && filter.dim_size(1) == 1 && row_dilation == 1 &&
503 col_dilation == 1 && row_stride == 1 && col_stride == 1 &&
504 data_format == FORMAT_NHWC) {
505 // 1x1 filter, so call cublas directly.
506 const uint64 m = input.dim_size(0) * input.dim_size(1) * input.dim_size(2);
507 const uint64 k = filter.dim_size(2);
508 const uint64 n = filter.dim_size(3);
509
510 auto a_ptr = AsDeviceMemory(input.template flat<T>().data(),
511 input.template flat<T>().size());
512 auto b_ptr = AsDeviceMemory(filter.template flat<T>().data(),
513 filter.template flat<T>().size());
514 auto c_ptr = AsDeviceMemory(output->template flat<T>().data(),
515 output->template flat<T>().size());
516
517 auto no_transpose = perftools::gputools::blas::Transpose::kNoTranspose;
518 bool blas_launch_status =
519 stream
520 ->ThenBlasGemm(no_transpose, no_transpose, n, m, k, 1.0f, b_ptr, n,
521 a_ptr, k, 0.0f, &c_ptr, n)
522 .ok();
523 if (!blas_launch_status) {
524 ctx->SetStatus(errors::Internal("Blas SGEMM launch failed : m=", m,
525 ", n=", n, ", k=", k));
526 }
527 return;
528 } else if (filter.dim_size(0) == input.dim_size(1) &&
529 filter.dim_size(1) == input.dim_size(2) && row_dilation == 1 &&
530 col_dilation == 1 && padding == VALID &&
531 data_format == FORMAT_NHWC) {
532 // The input data and filter have the same height/width, so call cublas
533 // directly.
534 const uint64 m = input.dim_size(0);
535 const uint64 k =
536 filter.dim_size(0) * filter.dim_size(1) * filter.dim_size(2);
537 const uint64 n = filter.dim_size(3);
538
539 auto a_ptr = AsDeviceMemory(input.template flat<T>().data(),
540 input.template flat<T>().size());
541 auto b_ptr = AsDeviceMemory(filter.template flat<T>().data(),
542 filter.template flat<T>().size());
543 auto c_ptr = AsDeviceMemory(output->template flat<T>().data(),
544 output->template flat<T>().size());
545
546 auto no_transpose = perftools::gputools::blas::Transpose::kNoTranspose;
547 bool blas_launch_status =
548 stream
549 ->ThenBlasGemm(no_transpose, no_transpose, n, m, k, 1.0f, b_ptr, n,
550 a_ptr, k, 0.0f, &c_ptr, n)
551 .ok();
552 if (!blas_launch_status) {
553 ctx->SetStatus(errors::Internal("Blas SGEMM launch failed : m=", m,
554 ", n=", n, ", k=", k));
555 }
556 return;
557 }
558
559 int padding_rows = 0;
560 int padding_cols = 0;
561 const int64 in_batch = GetTensorDim(input, data_format, 'N');
562 int64 in_rows = GetTensorDim(input, data_format, 'H');
563 int64 in_cols = GetTensorDim(input, data_format, 'W');
564 const int64 in_depths = GetTensorDim(input, data_format, 'C');
565 const int64 out_batch = GetTensorDim(*output, data_format, 'N');
566 const int64 out_rows = GetTensorDim(*output, data_format, 'H');
567 const int64 out_cols = GetTensorDim(*output, data_format, 'W');
568 const int64 out_depths = GetTensorDim(*output, data_format, 'C');
569 const int64 patch_rows = filter.dim_size(0);
570 const int64 patch_cols = filter.dim_size(1);
571 if (padding == SAME) {
572 // Total padding on rows and cols is
573 // Pr = (R' - 1) * S + (Kr - 1) * Dr + 1 - R
574 // Pc = (C' - 1) * S + (Kc - 1) * Dc + 1 - C
575 // where (R', C') are output dimensions, (R, C) are input dimensions, S
576 // is stride, (Dr, Dc) are dilations, (Kr, Kc) are filter dimensions.
577 // We pad Pr/2 on the left and Pr - Pr/2 on the right, Pc/2 on the top
578 // and Pc - Pc/2 on the bottom. When Pr or Pc is odd, this means
579 // we pad more on the right and bottom than on the top and left.
580 padding_rows =
581 std::max<int>(0, (out_rows - 1) * row_stride +
582 (patch_rows - 1) * row_dilation + 1 - in_rows);
583 padding_cols =
584 std::max<int>(0, (out_cols - 1) * col_stride +
585 (patch_cols - 1) * col_dilation + 1 - in_cols);
586 const bool rows_odd = (padding_rows % 2 != 0);
587 const bool cols_odd = (padding_cols % 2 != 0);
588 if (rows_odd || cols_odd) {
589 Tensor transformed_input;
590 int64 new_in_rows = in_rows + rows_odd;
591 int64 new_in_cols = in_cols + cols_odd;
592 OP_REQUIRES_OK(
593 ctx,
594 ctx->allocate_temp(DataTypeToEnum<T>::value,
595 ShapeFromFormat(data_format, in_batch, new_in_rows,
596 new_in_cols, in_depths),
597 &transformed_input));
598
599 functor::PadInput<GPUDevice, T, int, 4>()(
600 ctx->eigen_device<GPUDevice>(), To32Bit(input_param.tensor<T, 4>()),
601 {{0, 0}}, {{rows_odd, cols_odd}},
602 To32Bit(transformed_input.tensor<T, 4>()), data_format);
603
604 input = transformed_input;
605 in_rows = new_in_rows;
606 in_cols = new_in_cols;
607 }
608 }
609
610 if (data_format == FORMAT_NHWC) {
611 // Convert the input tensor from NHWC to NCHW.
612 TensorShape nchw_shape =
613 ShapeFromFormat(FORMAT_NCHW, in_batch, in_rows, in_cols, in_depths);
614 if (in_depths > 1) {
615 Tensor transformed_input;
616 OP_REQUIRES_OK(ctx, ctx->allocate_temp(DataTypeToEnum<T>::value,
617 nchw_shape, &transformed_input));
618 functor::NHWCToNCHW<GPUDevice, T, 4>()(
619 ctx->eigen_device<GPUDevice>(),
620 const_cast<const Tensor&>(input).tensor<T, 4>(),
621 transformed_input.tensor<T, 4>());
622 input = transformed_input;
623 } else {
624 // If depth <= 1, then just reshape.
625 CHECK(input.CopyFrom(input, nchw_shape));
626 }
627 }
628
629 CHECK(padding_rows >= 0 && padding_cols >= 0)
630 << "Negative row or col paddings: (" << padding_rows << ", "
631 << padding_cols << ")";
632 perftools::gputools::dnn::BatchDescriptor input_desc;
633 input_desc.set_count(in_batch)
634 .set_feature_map_count(in_depths)
635 .set_height(in_rows)
636 .set_width(in_cols)
637 .set_layout(perftools::gputools::dnn::DataLayout::kBatchDepthYX);
638 perftools::gputools::dnn::BatchDescriptor output_desc;
639 output_desc.set_count(out_batch)
640 .set_height(out_rows)
641 .set_width(out_cols)
642 .set_feature_map_count(out_depths)
643 .set_layout(perftools::gputools::dnn::DataLayout::kBatchDepthYX);
644 perftools::gputools::dnn::FilterDescriptor filter_desc;
645 filter_desc.set_input_filter_height(filter.dim_size(0))
646 .set_input_filter_width(filter.dim_size(1))
647 .set_input_feature_map_count(filter.dim_size(2))
648 .set_output_feature_map_count(filter.dim_size(3));
649 perftools::gputools::dnn::ConvolutionDescriptor conv_desc;
650 conv_desc.set_vertical_dilation_rate(row_dilation)
651 .set_horizontal_dilation_rate(col_dilation)
652 .set_vertical_filter_stride(row_stride)
653 .set_horizontal_filter_stride(col_stride)
654 .set_zero_padding_height(padding_rows / 2)
655 .set_zero_padding_width(padding_cols / 2);
656
657 Tensor transformed_filter;
658 OP_REQUIRES_OK(ctx, ctx->allocate_temp(
659 DataTypeToEnum<T>::value,
660 TensorShape({filter.dim_size(3), filter.dim_size(2),
661 filter.dim_size(0), filter.dim_size(1)}),
662 &transformed_filter));
663
664 functor::TransformFilter<GPUDevice, T, int, 4>()(
665 ctx->eigen_device<GPUDevice>(), To32Bit(filter.tensor<T, 4>()),
666 To32Bit(transformed_filter.tensor<T, 4>()));
667
668 Tensor transformed_output;
669 OP_REQUIRES_OK(
670 ctx, ctx->allocate_temp(DataTypeToEnum<T>::value,
671 ShapeFromFormat(FORMAT_NCHW, out_batch, out_rows,
672 out_cols, out_depths),
673 &transformed_output));
674
675 auto input_ptr = AsDeviceMemory(input.template flat<T>().data(),
676 input.template flat<T>().size());
677 auto filter_ptr =
678 AsDeviceMemory(transformed_filter.template flat<T>().data(),
679 transformed_filter.template flat<T>().size());
680 auto output_ptr =
681 AsDeviceMemory(transformed_output.template flat<T>().data(),
682 transformed_output.template flat<T>().size());
683
684 static int64 ConvolveScratchSize = GetCudnnWorkspaceLimit(
685 // default value is in bytes despite the name of the environment variable
686 "TF_CUDNN_WORKSPACE_LIMIT_IN_MB", 1LL << 32 // 4GB
687 );
688
689 int device_id = stream->parent()->device_ordinal();
690 DataType dtype = input.dtype();
691 ConvParameters conv_parameters = {
692 in_batch, // batch
693 in_depths, // in_depths
694 {{in_rows, // in_rows
695 in_cols}}, // in_cols
696 out_depths, // out_depths
697 {{patch_rows, // filter_rows
698 patch_cols}}, // filter_cols
699 {{row_dilation, // dilation_rows
700 col_dilation}}, // dilation_cols
701 {{row_stride, // stride_rows
702 col_stride}}, // stride_cols
703 {{padding_rows, // padding_rows
704 padding_cols}}, // padding_cols
705 dtype, // tensor datatype
706 device_id, // device_id
707 };
708 AlgorithmConfig algorithm_config;
709 if (cudnn_use_autotune &&
710 !AutoTuneConv::GetInstance()->Find(conv_parameters, &algorithm_config)) {
711 std::vector<AlgorithmDesc> algorithms;
712 CHECK(stream->parent()->GetConvolveAlgorithms(
713 conv_parameters.ShouldIncludeWinogradNonfusedAlgo<T>(), &algorithms));
714 ProfileResult best_result;
715 ProfileResult best_result_no_scratch;
716 for (auto profile_algorithm : algorithms) {
717 // TODO(zhengxq): profile each algorithm multiple times to better
718 // accuracy.
719 CudnnScratchAllocator scratch_allocator(ConvolveScratchSize, ctx);
720 ProfileResult profile_result;
721 bool cudnn_launch_status =
722 stream
723 ->ThenConvolveWithAlgorithm(
724 input_desc, input_ptr, filter_desc, filter_ptr, conv_desc,
725 output_desc, &output_ptr, &scratch_allocator,
726 AlgorithmConfig(profile_algorithm), &profile_result)
727 .ok();
728 if (cudnn_launch_status) {
729 if (profile_result.is_valid()) {
730 if (profile_result.elapsed_time_in_ms() <
731 best_result.elapsed_time_in_ms()) {
732 best_result = profile_result;
733 }
734 if (scratch_allocator.TotalByteSize() == 0 &&
735 profile_result.elapsed_time_in_ms() <
736 best_result_no_scratch.elapsed_time_in_ms()) {
737 best_result_no_scratch = profile_result;
738 }
739 }
740 }
741 }
742 // TODO(yangzihao): refactor the profile result checking code into a common
743 // utility function.
744 OP_REQUIRES(ctx,
745 best_result.is_valid() || best_result_no_scratch.is_valid(),
746 errors::NotFound("No algorithm worked!"));
747 if (best_result.is_valid()) {
748 algorithm_config.set_algorithm(best_result.algorithm());
749 }
750 if (best_result_no_scratch.is_valid()) {
751 algorithm_config.set_algorithm_no_scratch(
752 best_result_no_scratch.algorithm());
753 }
754 AutoTuneConv::GetInstance()->Insert(conv_parameters, algorithm_config);
755 }
756
757 CudnnScratchAllocator scratch_allocator(ConvolveScratchSize, ctx);
758 bool cudnn_launch_status =
759 stream
760 ->ThenConvolveWithAlgorithm(input_desc, input_ptr, filter_desc,
761 filter_ptr, conv_desc, output_desc,
762 &output_ptr, &scratch_allocator,
763 algorithm_config, nullptr)
764 .ok();
765
766 if (!cudnn_launch_status) {
767 ctx->SetStatus(errors::Internal(
768 "cuDNN launch failure : input shape(", input.shape().DebugString(),
769 ") filter shape(", filter.shape().DebugString(), ")"));
770 }
771
772 // Convert the output tensor back from NHWC to NCHW.
773 if (data_format == FORMAT_NHWC) {
774 functor::NCHWToNHWC<GPUDevice, T, 4>()(
775 ctx->eigen_device<GPUDevice>(),
776 const_cast<const Tensor&>(transformed_output).tensor<T, 4>(),
777 output->tensor<T, 4>());
778 } else {
779 *output = transformed_output;
780 }
781}
