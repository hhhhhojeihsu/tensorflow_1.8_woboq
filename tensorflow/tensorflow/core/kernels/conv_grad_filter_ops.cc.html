54// Returns in 'col_data', image patches in storage order (height, width, depth)
55// extracted from image at 'input_data', which is required to be in storage
56// order (batch, height, width, depth).
57// Implementation written by Yangqing Jia (jiayq).
58template <typename T>
59void Im2col(const T* input_data, const int depth, const int height,
60 const int width, const int filter_h, const int filter_w,
61 const int pad_t, const int pad_l, const int pad_b, const int pad_r,
62 const int stride_h, const int stride_w, T* col_data) {
63 int height_col = (height + pad_t + pad_b - filter_h) / stride_h + 1;
64 int width_col = (width + pad_l + pad_r - filter_w) / stride_w + 1;
65
66 int h_pad = -pad_t;
67 for (int h = 0; h < height_col; ++h) {
68 int w_pad = -pad_l;
69 for (int w = 0; w < width_col; ++w) {
70 for (int ih = h_pad; ih < h_pad + filter_h; ++ih) {
71 for (int iw = w_pad; iw < w_pad + filter_w; ++iw) {
72 if (ih >= 0 && ih < height && iw >= 0 && iw < width) {
73 memcpy(col_data, input_data + (ih * width + iw) * depth,
74 sizeof(T) * depth);
75 } else {
76 // This should be simply padded with zero.
77 memset(col_data, 0, sizeof(T) * depth);
78 }
79 col_data += depth;
80 }
81 }
82 w_pad += stride_w;
83 }
84 h_pad += stride_h;
85 }
86}

95template <typename T>
96struct LaunchConv2DBackpropFilterOp<CPUDevice, T> {
97 void operator()(OpKernelContext* ctx, bool use_cudnn, bool cudnn_use_autotune,
98 const Tensor& out_backprop, const Tensor& input,
99 int row_stride, int col_stride, const Padding& padding,
100 Tensor* filter_backprop, TensorFormat data_format) {
101 const CPUDevice& d = ctx->eigen_device<CPUDevice>();
102 functor::SpatialConvolutionBackwardFilter<CPUDevice, T>()(
103 d, filter_backprop->tensor<T, 4>(), input.tensor<T, 4>(),
104 out_backprop.tensor<T, 4>(), row_stride, col_stride,
105 /*row_dilation=*/1, /*col_dilation=*/1);
106 }
107};

123template <>
124struct LaunchXsmmBackwardFilter<CPUDevice, float> {
125 bool operator()(OpKernelContext* context, const CPUDevice& d,
126 typename TTypes<float, 4>::ConstTensor input,
127 typename TTypes<float, 4>::Tensor filter,
128 typename TTypes<float, 4>::ConstTensor output, int input_rows,
129 int input_cols, int row_stride, int col_stride, int pad_h,
130 int pad_w, TensorFormat data_format) const {
131 auto batch = input.dimension(0);
132 auto in_depth = input.dimension(3);
133 auto out_depth = output.dimension(3);
134 auto filter_rows = filter.dimension(0);
135 auto filter_cols = filter.dimension(1);
136
137 auto num_threads =
138 context->device()->tensorflow_cpu_worker_threads()->num_threads;
139 // See libxsmm_dnn.h for this struct definition.
140 libxsmm_dnn_conv_desc desc;
141 desc.N = batch;
142 desc.C = in_depth;
143 desc.H = input_rows;
144 desc.W = input_cols;
145 desc.K = out_depth;
146 desc.R = filter_rows;
147 desc.S = filter_cols;
148 desc.u = row_stride;
149 desc.v = col_stride;
150 desc.pad_h = pad_h;
151 desc.pad_w = pad_w;
152 desc.pad_h_in = 0; // pad_rows; // ignored by libxsmm for now.
153 desc.pad_w_in = 0; // pad_cols; // ignored by libxsmm for now.
154 desc.pad_h_out = 0;
155 desc.pad_w_out = 0;
156 desc.threads = num_threads;
157 desc.algo = LIBXSMM_DNN_CONV_ALGO_DIRECT;
158 desc.buffer_format = LIBXSMM_DNN_TENSOR_FORMAT_NHWC;
159 desc.filter_format = LIBXSMM_DNN_TENSOR_FORMAT_RSCK;
160 desc.fuse_ops = LIBXSMM_DNN_CONV_FUSE_NONE;
161 desc.options = LIBXSMM_DNN_CONV_OPTION_NONE;
162 desc.datatype = LIBXSMM_DNN_DATATYPE_F32;
163
164 if (!CanUseXsmmConv2D(desc, data_format)) {
165 return false;
166 }
167
168 auto input_ptr = input.data();
169 auto filter_ptr = filter.data();
170 auto output_ptr = output.data();
171 bool success = functor::XsmmBkwFilterConv2D<CPUDevice, float>()(
172 context, desc, input_ptr, filter_ptr, output_ptr);
173 return success;
174 }
175};

178template <typename Device, class T>
179class Conv2DFastBackpropFilterOp : public OpKernel {
180 public:
181 explicit Conv2DFastBackpropFilterOp(OpKernelConstruction* context)
182 : OpKernel(context) {
183 string data_format;
184 OP_REQUIRES_OK(context, context->GetAttr("data_format", &data_format));
185 OP_REQUIRES(context, FormatFromString(data_format, &data_format_),
186 errors::InvalidArgument("Invalid data format"));
187 OP_REQUIRES(context, data_format_ == FORMAT_NHWC,
188 errors::InvalidArgument(
189 "Conv2DFastBackpropFilterOp only supports NHWC."));
190 OP_REQUIRES_OK(context, context->GetAttr("strides", &strides_));
191 OP_REQUIRES(context, strides_.size() == 4,
192 errors::InvalidArgument("Sliding window strides field must "
193 "specify 4 dimensions"));
194 OP_REQUIRES(
195 context, (strides_[0] == 1 && strides_[3] == 1),
196 errors::InvalidArgument("Current implementation does not yet support "
197 "strides in the batch and depth dimensions."));
198 OP_REQUIRES(context, strides_[1] > 0 && strides_[2] > 0,
199 errors::InvalidArgument(
200 "Row and column strides should be larger than 0."));
201 OP_REQUIRES_OK(context, context->GetAttr("padding", &padding_));
202 OP_REQUIRES_OK(context, context->GetAttr("dilations", &dilations_));
203 OP_REQUIRES(context, dilations_.size() == 4,
204 errors::InvalidArgument("Sliding window dilations field must "
205 "specify 4 dimensions"));
206 OP_REQUIRES(context, (dilations_[0] == 1 && dilations_[3] == 1),
207 errors::InvalidArgument(
208 "Current implementation does not yet support "
209 "dilations in the batch and depth dimensions."));
210 // TODO(yangzihao): Add a CPU implementation for dilated convolution.
211 OP_REQUIRES(context, (dilations_[1] == 1 && dilations_[2] == 1),
212 errors::InvalidArgument(
213 "Current Eigen and libxsmm implementations do not "
214 "yet support dilation rates larger than 1."));
215 }
216
217 void Compute(OpKernelContext* context) override {
218 const Tensor& input = context->input(0);
219 const Tensor& filter_sizes = context->input(1);
220 const Tensor& out_backprop = context->input(2);
221 OP_REQUIRES(
222 context, TensorShapeUtils::IsVector(filter_sizes.shape()),
223 errors::InvalidArgument(
224 "Conv2DBackpropFilter: filter_sizes input must be 1-dim, not ",
225 filter_sizes.dims()));
226 TensorShape filter_shape;
227 OP_REQUIRES_OK(context, TensorShapeUtils::MakeShape(
228 filter_sizes.vec<int32>(), &filter_shape));
229
230 ConvBackpropDimensions dims;
231 OP_REQUIRES_OK(
232 context,
233 ConvBackpropComputeDimensions(
234 type_string(), /*num_spatial_dims=*/2, input.shape(), filter_shape,
235 out_backprop.shape(), strides_, padding_, data_format_, &dims));
236
237 Tensor* filter_backprop = nullptr;
238 OP_REQUIRES_OK(context,
239 context->allocate_output(0, filter_shape, &filter_backprop));
240
241 // If there is nothing to compute, return.
242 if (filter_shape.num_elements() == 0) {
243 return;
244 }
245
246#if defined TENSORFLOW_USE_LIBXSMM_CONVOLUTIONS && \
247 defined TENSORFLOW_USE_LIBXSMM_BACKWARD_CONVOLUTIONS
248 int64 pad_top, pad_bottom;
249 int64 pad_left, pad_right;
250 OP_REQUIRES_OK(
251 context,
252 GetWindowedOutputSizeVerbose(
253 dims.spatial_dims[0].input_size, dims.spatial_dims[0].filter_size,
254 dims.spatial_dims[0].stride, padding_,
255 &dims.spatial_dims[0].output_size, &pad_top, &pad_bottom));
256 OP_REQUIRES_OK(
257 context,
258 GetWindowedOutputSizeVerbose(
259 dims.spatial_dims[1].input_size, dims.spatial_dims[1].filter_size,
260 dims.spatial_dims[1].stride, padding_,
261 &dims.spatial_dims[1].output_size, &pad_left, &pad_right));
262
263 if (pad_left == pad_right && pad_top == pad_bottom) {
264 if (LaunchXsmmBackwardFilter<Device, T>()(
265 context, context->eigen_device<Device>(), input.tensor<T, 4>(),
266 filter_backprop->tensor<T, 4>(), out_backprop.tensor<T, 4>(),
267 dims.spatial_dims[0].input_size, dims.spatial_dims[1].input_size,
268 static_cast<int>(dims.spatial_dims[0].stride),
269 static_cast<int>(dims.spatial_dims[1].stride),
270 static_cast<int>(pad_top), static_cast<int>(pad_left),
271 data_format_)) {
272 return;
273 }
274 }
275#endif
276
277 LaunchConv2DBackpropFilterOp<Device, T>()(
278 context, false, false, out_backprop, input, dims.spatial_dims[0].stride,
279 dims.spatial_dims[1].stride, padding_, filter_backprop, data_format_);
280 }
281
282 private:
283 std::vector<int32> dilations_;
284 std::vector<int32> strides_;
285 Padding padding_;
286 TensorFormat data_format_;
287
288 TF_DISALLOW_COPY_AND_ASSIGN(Conv2DFastBackpropFilterOp);
289};

292template <typename Device, class T>
293class Conv2DCustomBackpropFilterOp : public OpKernel {
294 public:
295 explicit Conv2DCustomBackpropFilterOp(OpKernelConstruction* context)
296 : OpKernel(context) {
297 string data_format;
298 OP_REQUIRES_OK(context, context->GetAttr("data_format", &data_format));
299 OP_REQUIRES(context, FormatFromString(data_format, &data_format_),
300 errors::InvalidArgument("Invalid data format"));
301 OP_REQUIRES(context, data_format_ == FORMAT_NHWC,
302 errors::InvalidArgument(
303 "Conv2DCustomBackpropFilterOp only supports NHWC."));
304 OP_REQUIRES_OK(context, context->GetAttr("strides", &strides_));
305 OP_REQUIRES(context, strides_.size() == 4,
306 errors::InvalidArgument("Sliding window strides field must "
307 "specify 4 dimensions"));
308 OP_REQUIRES(
309 context, (strides_[0] == 1 && strides_[3] == 1),
310 errors::InvalidArgument("Current implementation does not yet support "
311 "strides in the batch and depth dimensions."));
312 OP_REQUIRES(context, strides_[1] > 0 && strides_[2] > 0,
313 errors::InvalidArgument(
314 "Row and column strides should be larger than 0."));
315 OP_REQUIRES_OK(context, context->GetAttr("padding", &padding_));
316 OP_REQUIRES_OK(context, context->GetAttr("dilations", &dilations_));
317 OP_REQUIRES(context, dilations_.size() == 4,
318 errors::InvalidArgument("Sliding window dilations field must "
319 "specify 4 dimensions"));
320 OP_REQUIRES(context, (dilations_[0] == 1 && dilations_[3] == 1),
321 errors::InvalidArgument(
322 "Current implementation does not yet support "
323 "dilations in the batch and depth dimensions."));
324 // TODO(yangzihao): Add a CPU implementation for dilated convolution.
325 OP_REQUIRES(context, (dilations_[1] == 1 && dilations_[2] == 1),
326 errors::InvalidArgument(
327 "Current libxsmm and customized CPU implementations do "
328 "not yet support dilation rates larger than 1."));
329 }
330
331 void Compute(OpKernelContext* context) override {
332 const Tensor& input = context->input(0);
333 const Tensor& filter_sizes = context->input(1);
334 const Tensor& out_backprop = context->input(2);
335 OP_REQUIRES(
336 context, TensorShapeUtils::IsVector(filter_sizes.shape()),
337 errors::InvalidArgument(
338 "Conv2DCustomBackpropFilter: filter_sizes input must be 1-dim, "
339 "not ",
340 filter_sizes.dims()));
341 TensorShape filter_shape;
342 OP_REQUIRES_OK(context, TensorShapeUtils::MakeShape(
343 filter_sizes.vec<int32>(), &filter_shape));
344
345 ConvBackpropDimensions dims;
346 OP_REQUIRES_OK(context,
347 ConvBackpropComputeDimensions(
348 "Conv2DCustomBackpropFilter", /*num_spatial_dims=*/2,
349 input.shape(), filter_shape, out_backprop.shape(),
350 strides_, padding_, data_format_, &dims));
351
352 Tensor* filter_backprop;
353 OP_REQUIRES_OK(context,
354 context->allocate_output(0, filter_shape, &filter_backprop));
355
356 // If there is nothing to compute, return.
357 if (filter_shape.num_elements() == 0) {
358 return;
359 }
360
361 int64 pad_top, pad_bottom;
362 int64 pad_left, pad_right;
363 OP_REQUIRES_OK(
364 context,
365 GetWindowedOutputSizeVerbose(
366 dims.spatial_dims[0].input_size, dims.spatial_dims[0].filter_size,
367 dims.spatial_dims[0].stride, padding_,
368 &dims.spatial_dims[0].output_size, &pad_top, &pad_bottom));
369 OP_REQUIRES_OK(
370 context,
371 GetWindowedOutputSizeVerbose(
372 dims.spatial_dims[1].input_size, dims.spatial_dims[1].filter_size,
373 dims.spatial_dims[1].stride, padding_,
374 &dims.spatial_dims[1].output_size, &pad_left, &pad_right));
375#if defined TENSORFLOW_USE_LIBXSMM_CONVOLUTIONS && \
376 defined TENSORFLOW_USE_LIBXSMM_BACKWARD_CONVOLUTIONS
377 if (pad_left == pad_right && pad_top == pad_bottom) {
378 if (LaunchXsmmBackwardFilter<Device, T>()(
379 context, context->eigen_device<Device>(), input.tensor<T, 4>(),
380 filter_backprop->tensor<T, 4>(), out_backprop.tensor<T, 4>(),
381 dims.spatial_dims[0].input_size, dims.spatial_dims[1].input_size,
382 static_cast<int>(dims.spatial_dims[0].stride),
383 static_cast<int>(dims.spatial_dims[1].stride),
384 static_cast<int>(pad_top), static_cast<int>(pad_left),
385 data_format_)) {
386 return;
387 }
388 }
389#endif
390
391 // The total dimension size of each kernel.
392 const int filter_total_size = dims.spatial_dims[0].filter_size *
393 dims.spatial_dims[1].filter_size *
394 dims.in_depth;
395 // The output image size is the spatial size of the output.
396 const int output_image_size =
397 dims.spatial_dims[0].output_size * dims.spatial_dims[1].output_size;
398
399 // Shard 'batch' images into 'shard_size' groups of images to be fed
400 // into the parallel matmul. Calculate 'shard_size' by dividing the L3 cache
401 // size ('target_working_set_size') by the matmul size of an individual
402 // image ('work_unit_size').
403
404 // TODO(andydavis)
405 // *) Get L3 cache size from device at runtime (30MB is from ivybridge).
406 // *) Consider reducing 'target_working_set_size' if L3 is shared by
407 // other concurrently running tensorflow ops.
408 const size_t target_working_set_size = (30LL << 20) / sizeof(T);
409
410 const size_t size_A = output_image_size * filter_total_size;
411
412 const size_t size_B = output_image_size * dims.out_depth;
413
414 const size_t size_C = filter_total_size * dims.out_depth;
415
416 const size_t work_unit_size = size_A + size_B + size_C;
417
418 const size_t shard_size =
419 (target_working_set_size + work_unit_size - 1) / work_unit_size;
420
421 Tensor col_buffer;
422 OP_REQUIRES_OK(context,
423 context->allocate_temp(
424 DataTypeToEnum<T>::value,
425 TensorShape({static_cast<int64>(shard_size),
426 static_cast<int64>(output_image_size),
427 static_cast<int64>(filter_total_size)}),
428 &col_buffer));
429
430 // The input offset corresponding to a single input image.
431 const int input_offset = dims.spatial_dims[0].input_size *
432 dims.spatial_dims[1].input_size * dims.in_depth;
433 // The output offset corresponding to a single output image.
434 const int output_offset = dims.spatial_dims[0].output_size *
435 dims.spatial_dims[1].output_size * dims.out_depth;
436
437 const T* input_data = input.template flat<T>().data();
438 T* col_buffer_data = col_buffer.template flat<T>().data();
439 const T* out_backprop_data = out_backprop.template flat<T>().data();
440 T* filter_backprop_data = filter_backprop->template flat<T>().data();
441
442 typedef Eigen::TensorMap<Eigen::Tensor<T, 2, Eigen::RowMajor>,
443 Eigen::Unaligned>
444 TensorMap;
445 typedef Eigen::TensorMap<Eigen::Tensor<const T, 2, Eigen::RowMajor>,
446 Eigen::Unaligned>
447 ConstTensorMap;
448
449 TensorMap C(filter_backprop_data, filter_total_size, dims.out_depth);
450 C.setZero();
451
452 // Initialize contraction dims (we need to transpose 'A' below).
453 Eigen::array<Eigen::IndexPair<Eigen::DenseIndex>, 1> contract_dims;
454 contract_dims[0].first = 0;
455 contract_dims[0].second = 0;
456
457 auto worker_threads = *(context->device()->tensorflow_cpu_worker_threads());
458
459 for (int image_id = 0; image_id < dims.batch_size; image_id += shard_size) {
460 const int shard_limit =
461 std::min(static_cast<int>(shard_size),
462 static_cast<int>(dims.batch_size) - image_id);
463
464 auto shard = [&input_data, &col_buffer_data, &dims, &pad_top, &pad_left,
465 &pad_bottom, &pad_right, &input_offset,
466 &size_A](int64 start, int64 limit) {
467 for (int shard_id = start; shard_id < limit; ++shard_id) {
468 const T* input_data_shard = input_data + shard_id * input_offset;
469 T* col_data_shard = col_buffer_data + shard_id * size_A;
470
471 // When we compute the gradient with respect to the filters, we need
472 // to do im2col to allow gemm-type computation.
473 Im2col<T>(
474 input_data_shard, dims.in_depth, dims.spatial_dims[0].input_size,
475 dims.spatial_dims[1].input_size, dims.spatial_dims[0].filter_size,
476 dims.spatial_dims[1].filter_size, pad_top, pad_left, pad_bottom,
477 pad_right, dims.spatial_dims[0].stride,
478 dims.spatial_dims[1].stride, col_data_shard);
479 }
480 };
481 Shard(worker_threads.num_threads, worker_threads.workers, shard_limit,
482 size_A, shard);
483
484 ConstTensorMap A(col_buffer_data, output_image_size * shard_limit,
485 filter_total_size);
486 ConstTensorMap B(out_backprop_data, output_image_size * shard_limit,
487 dims.out_depth);
488
489 // Gradient with respect to filter.
490 C.device(context->eigen_cpu_device()) += A.contract(B, contract_dims);
491
492 input_data += input_offset * shard_limit;
493 out_backprop_data += output_offset * shard_limit;
494 }
495 }
496
497 private:
498 std::vector<int32> dilations_;
499 std::vector<int32> strides_;
500 Padding padding_;
501 TensorFormat data_format_;
502
503 TF_DISALLOW_COPY_AND_ASSIGN(Conv2DCustomBackpropFilterOp);
504};

539template <typename Device, class T>
540class Conv2DSlowBackpropFilterOp : public OpKernel {
541 public:
542 explicit Conv2DSlowBackpropFilterOp(OpKernelConstruction* context)
543 : OpKernel(context) {
544 string data_format;
545 OP_REQUIRES_OK(context, context->GetAttr("data_format", &data_format));
546 OP_REQUIRES(context, FormatFromString(data_format, &data_format_),
547 errors::InvalidArgument("Invalid data format"));
548 OP_REQUIRES_OK(context, context->GetAttr("strides", &strides_));
549 int stride_n = GetTensorDim(strides_, data_format_, 'N');
550 int stride_c = GetTensorDim(strides_, data_format_, 'C');
551 int stride_h = GetTensorDim(strides_, data_format_, 'H');
552 int stride_w = GetTensorDim(strides_, data_format_, 'W');
553 OP_REQUIRES(
554 context, (stride_n == 1 && stride_c == 1),
555 errors::InvalidArgument("Current implementation does not yet support "
556 "strides in the batch and depth dimensions."));
557 OP_REQUIRES(context, stride_h > 0 && stride_w > 0,
558 errors::InvalidArgument(
559 "Row and column strides should be larger than 0."));
560 OP_REQUIRES_OK(context, context->GetAttr("dilations", &dilations_));
561 OP_REQUIRES(context, dilations_.size() == 4,
562 errors::InvalidArgument("Sliding window dilations field must "
563 "specify 4 dimensions"));
564 int dilation_n = GetTensorDim(dilations_, data_format_, 'N');
565 int dilation_c = GetTensorDim(dilations_, data_format_, 'C');
566 int dilation_h = GetTensorDim(dilations_, data_format_, 'H');
567 int dilation_w = GetTensorDim(dilations_, data_format_, 'W');
568 OP_REQUIRES(context, dilation_n == 1 && dilation_c == 1,
569 errors::InvalidArgument(
570 "Current implementation does not yet support "
571 "dilations in the batch and depth dimensions."));
572 OP_REQUIRES(
573 context, dilation_h > 0 && dilation_w > 0,
574 errors::InvalidArgument("Dilated rates should be larger than 0."));
575 OP_REQUIRES_OK(context, context->GetAttr("use_cudnn_on_gpu", &use_cudnn_));
576 use_cudnn_ &= CanUseCudnn();
577 cudnn_use_autotune_ = CudnnUseAutotune();
578 OP_REQUIRES_OK(context, context->GetAttr("padding", &padding_));
579 }
580
581 void Compute(OpKernelContext* context) override {
582 const Tensor& input = context->input(0);
583 const Tensor& filter_sizes = context->input(1);
584 const Tensor& out_backprop = context->input(2);
585 OP_REQUIRES(
586 context, TensorShapeUtils::IsVector(filter_sizes.shape()),
587 errors::InvalidArgument(
588 "Conv2DBackpropFilter: filter_sizes input must be 1-dim, not ",
589 filter_sizes.dims()));
590 TensorShape filter_shape;
591 OP_REQUIRES_OK(context, TensorShapeUtils::MakeShape(
592 filter_sizes.vec<int32>(), &filter_shape));
593
594 Tensor* filter_backprop = nullptr;
595 OP_REQUIRES_OK(context,
596 context->allocate_output(0, filter_shape, &filter_backprop));
597
598 // If there is nothing to compute, return.
599 if (filter_shape.num_elements() == 0) {
600 return;
601 }
602 // If input is empty, set gradients to zero.
603 if (input.shape().num_elements() == 0) {
604 functor::SetZeroFunctor<Device, T> f;
605 f(context->eigen_device<Device>(), filter_backprop->flat<T>());
606 return;
607 }
608
609 // For now we take the stride from the second and third dimensions only (we
610 // do not support striding on the batch or depth dimension).
611 const int stride_rows = GetTensorDim(strides_, data_format_, 'H');
612 const int stride_cols = GetTensorDim(strides_, data_format_, 'W');
613 const int dilation_rows = GetTensorDim(dilations_, data_format_, 'H');
614 const int dilation_cols = GetTensorDim(dilations_, data_format_, 'W');
615
616 launcher_(context, use_cudnn_, cudnn_use_autotune_, out_backprop, input,
617 dilation_rows, dilation_cols, stride_rows, stride_cols, padding_,
618 filter_backprop, data_format_);
619 }
620
621 private:
622 std::vector<int32> dilations_;
623 std::vector<int32> strides_;
624 Padding padding_;
625 bool use_cudnn_;
626 TensorFormat data_format_;
627 LaunchConv2DBackpropFilterOp<Device, T> launcher_;
628 bool cudnn_use_autotune_;
629
630 TF_DISALLOW_COPY_AND_ASSIGN(Conv2DSlowBackpropFilterOp);
631};

633template <typename T>
634void LaunchConv2DBackpropFilterOp<Eigen::GpuDevice, T>::operator()(
635 OpKernelContext* ctx, bool use_cudnn, bool cudnn_use_autotune,
636 const Tensor& out_backprop, const Tensor& input, int row_dilation,
637 int col_dilation, int row_stride, int col_stride, const Padding& padding,
638 Tensor* filter_backprop, TensorFormat data_format) {
639 using perftools::gputools::dnn::AlgorithmConfig;
640 using perftools::gputools::dnn::AlgorithmDesc;
641 using perftools::gputools::dnn::ProfileResult;
642
643 std::vector<int32> dilations(4, 1);
644 dilations[GetTensorDimIndex(data_format, 'H')] = row_dilation;
645 dilations[GetTensorDimIndex(data_format, 'W')] = col_dilation;
646
647 std::vector<int32> strides(4, 1);
648 strides[GetTensorDimIndex(data_format, 'H')] = row_stride;
649 strides[GetTensorDimIndex(data_format, 'W')] = col_stride;
650 TensorShape filter_shape = filter_backprop->shape();
651
652 ConvBackpropDimensions dims;
653 OP_REQUIRES_OK(ctx, ConvBackpropComputeDimensionsV2(
654 "Conv2DSlowBackpropFilter", /*num_spatial_dims=*/2,
655 input.shape(), filter_shape, out_backprop.shape(),
656 dilations, strides, padding, data_format, &dims));
657
658 // TODO(yangzihao): The padding computations should be done in
659 // GetWindowedOutputSize() functions.
660 const int padding_rows =
661 (padding == VALID)
662 ? 0
663 : std::max<int>(0, (dims.spatial_dims[0].output_size - 1) *
664 dims.spatial_dims[0].stride +
665 (dims.spatial_dims[0].filter_size - 1) *
666 dims.spatial_dims[0].dilation +
667 1 - dims.spatial_dims[0].input_size);
668 const int padding_cols =
669 (padding == VALID)
670 ? 0
671 : std::max<int>(0, (dims.spatial_dims[1].output_size - 1) *
672 dims.spatial_dims[1].stride +
673 (dims.spatial_dims[1].filter_size - 1) *
674 dims.spatial_dims[1].dilation +
675 1 - dims.spatial_dims[1].input_size);
676
677 // TODO(zhengxq): cuDNN only supports equal padding on both sides, so only
678 // calling it when that is true. Remove this check when (if?) cuDNN starts
679 // supporting different padding.
680 bool rows_odd = (padding_rows % 2 != 0);
681 bool cols_odd = (padding_cols % 2 != 0);
682
683 auto* stream = ctx->op_device_context()->stream();
684 OP_REQUIRES(ctx, stream, errors::Internal("No GPU stream available."));
685
686 if (!use_cudnn) {
687 ctx->SetStatus(errors::Unimplemented(
688 "Conv2DBackprop for GPU is not currently supported "
689 "without cudnn"));
690 return;
691 }
692
693 bool cudnn_disable_conv_1x1_optimization_ = CudnnDisableConv1x1Optimization();
694 if (!cudnn_disable_conv_1x1_optimization_ &&
695 dims.spatial_dims[0].filter_size == 1 &&
696 dims.spatial_dims[1].filter_size == 1 &&
697 dims.spatial_dims[0].stride == 1 && dims.spatial_dims[1].stride == 1 &&
698 data_format == FORMAT_NHWC) {
699 const uint64 m = dims.in_depth;
700 const uint64 k = dims.batch_size * dims.spatial_dims[0].input_size *
701 dims.spatial_dims[1].input_size;
702 const uint64 n = dims.out_depth;
703
704 // The shape of output backprop is
705 // [batch, out_rows, out_cols, out_depth]
706 // From cublas's perspective, it is: n x k
707 auto a_ptr = AsDeviceMemory(out_backprop.template flat<T>().data(),
708 out_backprop.template flat<T>().size());
709
710 // The shape of input is
711 // [batch, in_rows, in_cols, in_depth],
712 // From cublas's perspective, it is: m x k
713 auto b_ptr = AsDeviceMemory(input.template flat<T>().data(),
714 input.template flat<T>().size());
715
716 // the shape of the filter backprop from the conv_2d should be
717 // [1, 1, in_depth, out_depth]
718 // From cublas's perspective, it is: n x m
719 auto c_ptr = AsDeviceMemory(filter_backprop->template flat<T>().data(),
720 filter_backprop->template flat<T>().size());
721
722 bool blas_launch_status =
723 stream
724 ->ThenBlasGemm(perftools::gputools::blas::Transpose::kNoTranspose,
725 perftools::gputools::blas::Transpose::kTranspose, n,
726 m, k, 1.0f, a_ptr, n, b_ptr, m, 0.0f, &c_ptr, n)
727 .ok();
728 if (!blas_launch_status) {
729 ctx->SetStatus(errors::Internal("Blas SGEMM launch failed : m=", m,
730 ", n=", n, ", k=", k));
731 }
732 return;
733 } else if (dims.spatial_dims[0].filter_size ==
734 dims.spatial_dims[0].input_size &&
735 dims.spatial_dims[1].filter_size ==
736 dims.spatial_dims[1].input_size &&
737 padding == VALID && data_format == FORMAT_NHWC) {
738 // The input data and filter have the same height/width, so call cublas
739 // directly.
740 const uint64 m = dims.spatial_dims[0].input_size *
741 dims.spatial_dims[1].input_size * dims.in_depth;
742 const uint64 k = dims.batch_size;
743 const uint64 n = dims.out_depth;
744
745 auto a_ptr = AsDeviceMemory(input.template flat<T>().data(),
746 input.template flat<T>().size());
747 auto b_ptr = AsDeviceMemory(out_backprop.template flat<T>().data(),
748 out_backprop.template flat<T>().size());
749 auto c_ptr = AsDeviceMemory(filter_backprop->template flat<T>().data(),
750 filter_backprop->template flat<T>().size());
751
752 bool blas_launch_status =
753 stream
754 ->ThenBlasGemm(perftools::gputools::blas::Transpose::kNoTranspose,
755 perftools::gputools::blas::Transpose::kTranspose, n,
756 m, k, 1.0f, b_ptr, n, a_ptr, m, 0.0f, &c_ptr, n)
757 .ok();
758 if (!blas_launch_status) {
759 ctx->SetStatus(errors::Internal("Blas SGEMM launch failed : m=", m,
760 ", n=", n, ", k=", k));
761 }
762 return;
763 }
764
765 Tensor compatible_input;
766 if (rows_odd || cols_odd) {
767 // If a padding dimension is odd, we have one more element on the right
768 // side or the bottom side. This is unsupported in cudnn. Therefore,
769 // we pad that extra element and make it compatible.
770 OP_REQUIRES_OK(
771 ctx, ctx->allocate_temp(
772 DataTypeToEnum<T>::value,
773 ShapeFromFormat(data_format, dims.batch_size,
774 dims.spatial_dims[0].input_size + rows_odd,
775 dims.spatial_dims[1].input_size + cols_odd,
776 dims.in_depth),
777 &compatible_input));
778
779 functor::PadInput<GPUDevice, T, int, 4>()(
780 ctx->template eigen_device<GPUDevice>(), To32Bit(input.tensor<T, 4>()),
781 {{0, 0}}, {{rows_odd, cols_odd}},
782 To32Bit(compatible_input.tensor<T, 4>()), data_format);
783 } else {
784 compatible_input = input;
785 }
786
787 CHECK(padding_rows >= 0 && padding_cols >= 0)
788 << "Negative row or col paddings: (" << padding_rows << ", "
789 << padding_cols << ")";
790 perftools::gputools::dnn::BatchDescriptor input_desc;
791 input_desc.set_count(dims.batch_size)
792 .set_height(GetTensorDim(compatible_input, data_format, 'H'))
793 .set_width(GetTensorDim(compatible_input, data_format, 'W'))
794 .set_feature_map_count(dims.in_depth)
795 .set_layout(perftools::gputools::dnn::DataLayout::kBatchDepthYX);
796 perftools::gputools::dnn::BatchDescriptor output_desc;
797 output_desc.set_count(dims.batch_size)
798 .set_height(dims.spatial_dims[0].output_size)
799 .set_width(dims.spatial_dims[1].output_size)
800 .set_feature_map_count(dims.out_depth)
801 .set_layout(perftools::gputools::dnn::DataLayout::kBatchDepthYX);
802 perftools::gputools::dnn::FilterDescriptor filter_desc;
803 filter_desc.set_input_filter_height(dims.spatial_dims[0].filter_size)
804 .set_input_filter_width(dims.spatial_dims[1].filter_size)
805 .set_input_feature_map_count(dims.in_depth)
806 .set_output_feature_map_count(dims.out_depth);
807 perftools::gputools::dnn::ConvolutionDescriptor conv_desc;
808 conv_desc.set_vertical_dilation_rate(dims.spatial_dims[0].dilation)
809 .set_horizontal_dilation_rate(dims.spatial_dims[1].dilation)
810 .set_vertical_filter_stride(dims.spatial_dims[0].stride)
811 .set_horizontal_filter_stride(dims.spatial_dims[1].stride)
812 .set_zero_padding_height(padding_rows / 2)
813 .set_zero_padding_width(padding_cols / 2);
814
815 // NOTE(zhengxq):
816 // cuDNN only supports the following layouts :
817 // Input : B x D x R x C
818 // Filter : OD x ID x R x C
819 // Whereas, we have
820 // Input : B x R x C x D
821 // Filter : R x C x ID x OD
822 // TransformFilter performs (R x C x ID x OD) => (OD x ID x R x C)
823 // The first TransformDepth performs
824 // (B x R x C x D) => (B x D x R x C).
825 // Since the tensor returned from cuDNN is B x D x R x C also,
826 // the second TransformDepth performs
827 // (B x D x R x C) => (B x R x C x D).
828
829 Tensor pre_transformed_filter_backprop;
830 OP_REQUIRES_OK(
831 ctx, ctx->allocate_temp(DataTypeToEnum<T>::value,
832 TensorShape({dims.out_depth, dims.in_depth,
833 dims.spatial_dims[0].filter_size,
834 dims.spatial_dims[1].filter_size}),
835 &pre_transformed_filter_backprop));
836
837 Tensor transformed_out_backprop;
838 if (data_format == FORMAT_NHWC) {
839 TensorShape nchw_shape = ShapeFromFormat(
840 FORMAT_NCHW, dims.batch_size, dims.spatial_dims[0].output_size,
841 dims.spatial_dims[1].output_size, dims.out_depth);
842 if (dims.out_depth > 1) {
843 OP_REQUIRES_OK(ctx,
844 ctx->allocate_temp(DataTypeToEnum<T>::value, nchw_shape,
845 &transformed_out_backprop));
846 functor::NHWCToNCHW<GPUDevice, T, 4>()(
847 ctx->eigen_device<GPUDevice>(), out_backprop.tensor<T, 4>(),
848 transformed_out_backprop.tensor<T, 4>());
849 } else {
850 // If depth <= 1, just reshape.
851 CHECK(transformed_out_backprop.CopyFrom(out_backprop, nchw_shape));
852 }
853 } else {
854 transformed_out_backprop = out_backprop;
855 }
856
857 Tensor transformed_input;
858 if (data_format == FORMAT_NHWC) {
859 TensorShape nchw_shape = ShapeFromFormat(
860 FORMAT_NCHW, GetTensorDim(compatible_input, data_format, 'N'),
861 GetTensorDim(compatible_input, data_format, 'H'),
862 GetTensorDim(compatible_input, data_format, 'W'),
863 GetTensorDim(compatible_input, data_format, 'C'));
864 if (nchw_shape.dim_size(1) > 1) {
865 OP_REQUIRES_OK(ctx, ctx->allocate_temp(DataTypeToEnum<T>::value,
866 nchw_shape, &transformed_input));
867 functor::NHWCToNCHW<GPUDevice, T, 4>()(
868 ctx->eigen_device<GPUDevice>(),
869 const_cast<const Tensor&>(compatible_input).tensor<T, 4>(),
870 transformed_input.tensor<T, 4>());
871 } else {
872 // If depth <= 1, just reshape.
873 CHECK(transformed_input.CopyFrom(compatible_input, nchw_shape));
874 }
875 } else {
876 transformed_input = compatible_input;
877 }
878
879 auto out_backprop_ptr =
880 AsDeviceMemory(transformed_out_backprop.template flat<T>().data(),
881 transformed_out_backprop.template flat<T>().size());
882 auto filter_backprop_ptr =
883 AsDeviceMemory(pre_transformed_filter_backprop.template flat<T>().data(),
884 pre_transformed_filter_backprop.template flat<T>().size());
885 auto input_ptr = AsDeviceMemory(transformed_input.template flat<T>().data(),
886 transformed_input.template flat<T>().size());
887
888 static int64 ConvolveBackwardFilterScratchSize = GetCudnnWorkspaceLimit(
889 "TF_CUDNN_WORKSPACE_LIMIT_IN_MB", 1LL << 32 // 4GB by default
890 );
891 int device_id = stream->parent()->device_ordinal();
892 DataType dtype = input.dtype();
893 ConvParameters conv_parameters = {
894 dims.batch_size, // batch
895 dims.in_depth, // in_depths
896 {{input_desc.height(), // in_rows
897 input_desc.width()}}, // in_cols
898 dims.out_depth, // out_depths
899 {{dims.spatial_dims[0].filter_size, // filter_rows
900 dims.spatial_dims[1].filter_size}}, // filter_cols
901 {{dims.spatial_dims[0].dilation, // dilation_rows
902 dims.spatial_dims[1].dilation}}, // dilation_cols
903 {{dims.spatial_dims[0].stride, // stride_rows
904 dims.spatial_dims[1].stride}}, // stride_cols
905 {{padding_rows, // padding_rows
906 padding_cols}}, // padding_cols
907 dtype, // tensor datatype
908 device_id, // device_id
909 };
910 AlgorithmConfig algorithm_config;
911 if (cudnn_use_autotune && !AutoTuneConvBwdFilter::GetInstance()->Find(
912 conv_parameters, &algorithm_config)) {
913 std::vector<AlgorithmDesc> algorithms;
914 CHECK(stream->parent()->GetConvolveBackwardFilterAlgorithms(
915 conv_parameters.ShouldIncludeWinogradNonfusedAlgo<T>(), &algorithms));
916 ProfileResult best_result;
917 ProfileResult best_result_no_scratch;
918 for (auto profile_algorithm : algorithms) {
919 // TODO(zhengxq): profile each algorithm multiple times to better
920 // accuracy.
921 CudnnScratchAllocator scratch_allocator(ConvolveBackwardFilterScratchSize,
922 ctx);
923 ProfileResult profile_result;
924 bool cudnn_launch_status =
925 stream
926 ->ThenConvolveBackwardFilterWithAlgorithm(
927 input_desc, input_ptr, output_desc, out_backprop_ptr,
928 conv_desc, filter_desc, &filter_backprop_ptr,
929 &scratch_allocator, AlgorithmConfig(profile_algorithm),
930 &profile_result)
931 .ok();
932 if (cudnn_launch_status) {
933 if (profile_result.is_valid()) {
934 if (profile_result.elapsed_time_in_ms() <
935 best_result.elapsed_time_in_ms()) {
936 best_result = profile_result;
937 }
938 if (scratch_allocator.TotalByteSize() == 0 &&
939 profile_result.elapsed_time_in_ms() <
940 best_result_no_scratch.elapsed_time_in_ms()) {
941 best_result_no_scratch = profile_result;
942 }
943 }
944 }
945 }
946 OP_REQUIRES(ctx,
947 best_result.is_valid() || best_result_no_scratch.is_valid(),
948 errors::NotFound("No algorithm worked!"));
949 if (best_result.is_valid()) {
950 algorithm_config.set_algorithm(best_result.algorithm());
951 }
952 if (best_result_no_scratch.is_valid()) {
953 algorithm_config.set_algorithm_no_scratch(
954 best_result_no_scratch.algorithm());
955 }
956 AutoTuneConvBwdFilter::GetInstance()->Insert(conv_parameters,
957 algorithm_config);
958 }
959 CudnnScratchAllocator scratch_allocator(ConvolveBackwardFilterScratchSize,
960 ctx);
961 bool cudnn_launch_status =
962 stream
963 ->ThenConvolveBackwardFilterWithAlgorithm(
964 input_desc, input_ptr, output_desc, out_backprop_ptr, conv_desc,
965 filter_desc, &filter_backprop_ptr, &scratch_allocator,
966 algorithm_config, nullptr)
967 .ok();
968
969 if (!cudnn_launch_status) {
970 ctx->SetStatus(errors::Internal(
971 "cuDNN Backward Filter function launch failure : input shape(",
972 input.shape().DebugString(), ") filter shape(",
973 filter_shape.DebugString(), ")"));
974 return;
975 }

977 auto toConstTensor = [](const Tensor& x) -> const Tensor { return x; };
978 functor::ReverseTransformFilter<GPUDevice, T, 4>()(
979 ctx->eigen_device<GPUDevice>(),
980 toConstTensor(pre_transformed_filter_backprop).template tensor<T, 4>(),
981 filter_backprop->tensor<T, 4>());
982}
