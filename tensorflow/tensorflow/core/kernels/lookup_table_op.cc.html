33// Lookup table that wraps an unordered_map, where the key and value data type
34// is specified. Each individual value must be a scalar. If vector values are
35// required, use MutableHashTableOfTensors.
36//
37// This table is mutable and thread safe - Insert can be called at any time.
38//
39// Sample use case:
40//
41// MutableHashTableOfScalars<int64, int64> table; // int64 -> int64.
42// // Populate the table, elements could be added in one or multiple calls.
43// table.Insert(key_tensor, value_tensor); // Populate the table.
44//
45// table.Find(in_t, &out_t, default_t)
46//
47template <class K, class V>
48class MutableHashTableOfScalars final : public LookupInterface {
49 public:
50 MutableHashTableOfScalars(OpKernelContext* ctx, OpKernel* kernel) {}
51
52 size_t size() const override {
53 mutex_lock l(mu_);
54 return table_.size();
55 }
56
57 Status Find(OpKernelContext* ctx, const Tensor& key, Tensor* value,
58 const Tensor& default_value) override {
59 const V default_val = default_value.flat<V>()(0);
60 const auto key_values = key.flat<K>();
61 auto value_values = value->flat<V>();
62
63 mutex_lock l(mu_);
64 for (int64 i = 0; i < key_values.size(); ++i) {
65 value_values(i) = gtl::FindWithDefault(
66 table_, SubtleMustCopyIfIntegral(key_values(i)), default_val);
67 }
68
69 return Status::OK();
70 }
71
72 Status DoInsert(bool clear, const Tensor& keys, const Tensor& values) {
73 const auto key_values = keys.flat<K>();
74 const auto value_values = values.flat<V>();
75
76 mutex_lock l(mu_);
77 if (clear) {
78 table_.clear();
79 }
80 for (int64 i = 0; i < key_values.size(); ++i) {
81 gtl::InsertOrUpdate(&table_, SubtleMustCopyIfIntegral(key_values(i)),
82 SubtleMustCopyIfIntegral(value_values(i)));
83 }
84 return Status::OK();
85 }
86
87 Status Insert(OpKernelContext* ctx, const Tensor& keys,
88 const Tensor& values) override {
89 return DoInsert(false, keys, values);
90 }
91
92 Status ImportValues(OpKernelContext* ctx, const Tensor& keys,
93 const Tensor& values) override {
94 return DoInsert(true, keys, values);
95 }
96
97 Status ExportValues(OpKernelContext* ctx) override {
98 mutex_lock l(mu_);
99 int64 size = table_.size();
100
101 Tensor* keys;
102 Tensor* values;
103 TF_RETURN_IF_ERROR(
104 ctx->allocate_output("keys", TensorShape({size}), &keys));
105 TF_RETURN_IF_ERROR(
106 ctx->allocate_output("values", TensorShape({size}), &values));
107
108 auto keys_data = keys->flat<K>();
109 auto values_data = values->flat<V>();
110 int64 i = 0;
111 for (auto it = table_.begin(); it != table_.end(); ++it, ++i) {
112 keys_data(i) = it->first;
113 values_data(i) = it->second;
114 }
115 return Status::OK();
116 }
117
118 DataType key_dtype() const override { return DataTypeToEnum<K>::v(); }
119
120 DataType value_dtype() const override { return DataTypeToEnum<V>::v(); }
121
122 TensorShape key_shape() const final { return TensorShape(); }
123
124 TensorShape value_shape() const override { return TensorShape(); }
125
126 int64 MemoryUsed() const override {
127 int64 ret = 0;
128 mutex_lock l(mu_);
129 for (unsigned i = 0; i < table_.bucket_count(); ++i) {
130 size_t bucket_size = table_.bucket_size(i);
131 if (bucket_size == 0) {
132 ret++;
133 } else {
134 ret += bucket_size;
135 }
136 }
137 return sizeof(MutableHashTableOfScalars) + ret;
138 }
139
140 private:
141 // TODO(andreasst): consider using a read/write lock or a concurrent map
142 mutable mutex mu_;
143 std::unordered_map<K, V> table_ GUARDED_BY(mu_);
144};

146// Lookup table that wraps an unordered_map. Behaves identical to
147// MutableHashTableOfScalars except that each value must be a vector.
148template <class K, class V>
149class MutableHashTableOfTensors final : public LookupInterface {
150 public:
151 MutableHashTableOfTensors(OpKernelContext* ctx, OpKernel* kernel) {
152 OP_REQUIRES_OK(ctx,
153 GetNodeAttr(kernel->def(), "value_shape", &value_shape_));
154 OP_REQUIRES(
155 ctx, TensorShapeUtils::IsVector(value_shape_),
156 errors::InvalidArgument("Default value must be a vector, got shape ",
157 value_shape_.DebugString()));
158 }
159
160 size_t size() const override {
161 mutex_lock l(mu_);
162 return table_.size();
163 }
164
165 Status Find(OpKernelContext* ctx, const Tensor& key, Tensor* value,
166 const Tensor& default_value) override {
167 const auto default_flat = default_value.flat<V>();
168 const auto key_values = key.flat<K>();
169 auto value_values = value->flat_inner_dims<V, 2>();
170 int64 value_dim = value_shape_.dim_size(0);
171
172 mutex_lock l(mu_);
173 for (int64 i = 0; i < key_values.size(); ++i) {
174 ValueArray* value_vec =
175 gtl::FindOrNull(table_, SubtleMustCopyIfIntegral(key_values(i)));
176 if (value_vec != nullptr) {
177 for (int64 j = 0; j < value_dim; j++) {
178 value_values(i, j) = value_vec->at(j);
179 }
180 } else {
181 for (int64 j = 0; j < value_dim; j++) {
182 value_values(i, j) = default_flat(j);
183 }
184 }
185 }
186
187 return Status::OK();
188 }
189
190 Status DoInsert(bool clear, const Tensor& keys, const Tensor& values) {
191 const auto key_values = keys.flat<K>();
192 const auto value_values = values.flat_inner_dims<V, 2>();
193 int64 value_dim = value_shape_.dim_size(0);
194
195 mutex_lock l(mu_);
196 if (clear) {
197 table_.clear();
198 }
199 for (int64 i = 0; i < key_values.size(); ++i) {
200 ValueArray value_vec;
201 for (int64 j = 0; j < value_dim; j++) {
202 V value = value_values(i, j);
203 value_vec.push_back(value);
204 }
205 gtl::InsertOrUpdate(&table_, SubtleMustCopyIfIntegral(key_values(i)),
206 value_vec);
207 }
208 return Status::OK();
209 }
210
211 Status Insert(OpKernelContext* ctx, const Tensor& keys,
212 const Tensor& values) override {
213 return DoInsert(false, keys, values);
214 }
215
216 Status ImportValues(OpKernelContext* ctx, const Tensor& keys,
217 const Tensor& values) override {
218 return DoInsert(true, keys, values);
219 }
220
221 Status ExportValues(OpKernelContext* ctx) override {
222 mutex_lock l(mu_);
223 int64 size = table_.size();
224 int64 value_dim = value_shape_.dim_size(0);
225
226 Tensor* keys;
227 Tensor* values;
228 TF_RETURN_IF_ERROR(
229 ctx->allocate_output("keys", TensorShape({size}), &keys));
230 TF_RETURN_IF_ERROR(ctx->allocate_output(
231 "values", TensorShape({size, value_dim}), &values));
232
233 auto keys_data = keys->flat<K>();
234 auto values_data = values->matrix<V>();
235 int64 i = 0;
236 for (auto it = table_.begin(); it != table_.end(); ++it, ++i) {
237 K key = it->first;
238 ValueArray value = it->second;
239 keys_data(i) = key;
240 for (int64 j = 0; j < value_dim; j++) {
241 values_data(i, j) = value[j];
242 }
243 }
244 return Status::OK();
245 }
246
247 DataType key_dtype() const override { return DataTypeToEnum<K>::v(); }
248
249 DataType value_dtype() const override { return DataTypeToEnum<V>::v(); }
250
251 TensorShape key_shape() const final { return TensorShape(); }
252
253 TensorShape value_shape() const override { return value_shape_; }
254
255 int64 MemoryUsed() const override {
256 int64 ret = 0;
257 mutex_lock l(mu_);
258 for (unsigned i = 0; i < table_.bucket_count(); ++i) {
259 size_t bucket_size = table_.bucket_size(i);
260 if (bucket_size == 0) {
261 ret++;
262 } else {
263 ret += bucket_size;
264 }
265 }
266 return sizeof(MutableHashTableOfTensors) + ret;
267 }
268
269 private:
270 TensorShape value_shape_;
271 // TODO(andreasst): consider using a read/write lock or a concurrent map
272 mutable mutex mu_;
273 typedef gtl::InlinedVector<V, 4> ValueArray;
274 std::unordered_map<K, ValueArray> table_ GUARDED_BY(mu_);
275};

296// Modeled after densehashtable in https://github.com/sparsehash/sparsehash
297template <class K, class V>
298class MutableDenseHashTable final : public LookupInterface {
299 public:
300 MutableDenseHashTable(OpKernelContext* ctx, OpKernel* kernel) {
301 OP_REQUIRES_OK(
302 ctx, GetNodeAttr(kernel->def(), "max_load_factor", &max_load_factor_));
303 OP_REQUIRES(ctx, max_load_factor_ > 0 && max_load_factor_ < 1,
304 errors::InvalidArgument(
305 "max_load_factor must be between 0 and 1, got: ",
306 max_load_factor_));
307
308 OP_REQUIRES_OK(ctx,
309 GetNodeAttr(kernel->def(), "value_shape", &value_shape_));
310 OP_REQUIRES(ctx,
311 TensorShapeUtils::IsScalar(value_shape_) ||
312 TensorShapeUtils::IsVector(value_shape_),
313 errors::InvalidArgument(
314 "Empty value must be a scalar or a vector, got shape ",
315 value_shape_.DebugString()));
316
317 const Tensor* empty_key_input;
318 OP_REQUIRES_OK(ctx, ctx->input("empty_key", &empty_key_input));
319 key_shape_ = empty_key_input->shape();
320 OP_REQUIRES(ctx,
321 TensorShapeUtils::IsScalar(key_shape_) ||
322 TensorShapeUtils::IsVector(key_shape_),
323 errors::InvalidArgument(
324 "Empty key must be a scalar or a vector, got shape ",
325 key_shape_.DebugString()));
326 empty_key_ = PersistentTensor(*empty_key_input);
327 empty_key_hash_ = HashKey(
328 empty_key_input->template shaped<K, 2>({1, key_shape_.num_elements()}),
329 0);
330
331 int64 initial_num_buckets;
332 OP_REQUIRES_OK(ctx, GetNodeAttr(kernel->def(), "initial_num_buckets",
333 &initial_num_buckets));
334 OP_REQUIRES_OK(ctx, AllocateBuckets(ctx, initial_num_buckets));
335 }
336
337 size_t size() const override LOCKS_EXCLUDED(mu_) {
338 mutex_lock l(mu_);
339 return num_entries_;
340 }
341
342 Status Find(OpKernelContext* ctx, const Tensor& key, Tensor* value,
343 const Tensor& default_value) override LOCKS_EXCLUDED(mu_) {
344 const int64 num_elements = key.dim_size(0);
345 const int64 key_size = key_shape_.num_elements();
346 const int64 value_size = value_shape_.num_elements();
347 if (key.NumElements() != num_elements * key_size) {
348 TensorShape expected_shape({num_elements});
349 expected_shape.AppendShape(key_shape_);
350 return errors::InvalidArgument("Expected key shape ",
351 expected_shape.DebugString(), " got ",
352 key.shape().DebugString());
353 }
354 const auto key_matrix = key.shaped<K, 2>({num_elements, key_size});
355 auto value_matrix = value->shaped<V, 2>({num_elements, value_size});
356 const auto default_flat = default_value.flat<V>();
357
358 mutex_lock l(mu_);
359 const auto key_buckets_matrix =
360 key_buckets_.AccessTensor(ctx)->template matrix<K>();
361 const auto value_buckets_matrix =
362 value_buckets_.AccessTensor(ctx)->template matrix<V>();
363 const auto empty_key_matrix =
364 empty_key_.AccessTensor(ctx)->template shaped<K, 2>({1, key_size});
365 const int64 bit_mask = num_buckets_ - 1;
366 // TODO(andreasst): parallelize using work_sharder
367 for (int64 i = 0; i < num_elements; ++i) {
368 const uint64 key_hash = HashKey(key_matrix, i);
369 if (empty_key_hash_ == key_hash &&
370 IsEqualKey(empty_key_matrix, 0, key_matrix, i)) {
371 return errors::InvalidArgument(
372 "Using the empty_key as a table key is not allowed");
373 }
374 int64 bucket_index = key_hash & bit_mask;
375 int64 num_probes = 0;
376 while (true) {
377 if (IsEqualKey(key_buckets_matrix, bucket_index, key_matrix, i)) {
378 for (int64 j = 0; j < value_size; ++j) {
379 // TODO(andreasst): check if we can get rid of SubtleMustCopy
380 // here and elsewhere in this file.
381 value_matrix(i, j) =
382 SubtleMustCopyIfIntegral(value_buckets_matrix(bucket_index, j));
383 }
384 break;
385 }
386 if (IsEqualKey(key_buckets_matrix, bucket_index, empty_key_matrix, 0)) {
387 for (int64 j = 0; j < value_size; ++j) {
388 value_matrix(i, j) = SubtleMustCopyIfIntegral(default_flat(j));
389 }
390 break;
391 }
392 ++num_probes;
393 bucket_index =
394 (bucket_index + num_probes) & bit_mask; // quadratic probing
395 if (num_probes >= num_buckets_) {
396 return errors::Internal(
397 "Internal error in MutableDenseHashTable lookup");
398 }
399 }
400 }
401 return Status::OK();
402 }
403
404 Status Insert(OpKernelContext* ctx, const Tensor& key,
405 const Tensor& value) override LOCKS_EXCLUDED(mu_) {
406 if (key.NumElements() != key.dim_size(0) * key_shape_.num_elements()) {
407 TensorShape expected_shape({key.dim_size(0)});
408 expected_shape.AppendShape(key_shape_);
409 return errors::InvalidArgument("Expected key shape ",
410 expected_shape.DebugString(), " got ",
411 key.shape().DebugString());
412 }
413 mutex_lock l(mu_);
414 // For simplicity we assume that all keys in the input result in inserts
415 // rather than updates. That means we may grow the table even though we
416 // don't need to. As long as the number of keys inserted in one call is
417 // small compared to the size of the map, the impact of this is minimal.
418 const int64 pending_num_entries = num_entries_ + key.dim_size(0);
419 if (pending_num_entries > num_buckets_ * max_load_factor_) {
420 int64 new_num_buckets = num_buckets_;
421 do {
422 new_num_buckets <<= 1;
423 } while (pending_num_entries > new_num_buckets * max_load_factor_);
424 TF_RETURN_IF_ERROR(Rebucket(ctx, new_num_buckets));
425 }
426 return DoInsert(ctx, key, value, false);
427 }
428
429 Status ImportValues(OpKernelContext* ctx, const Tensor& keys,
430 const Tensor& values) override LOCKS_EXCLUDED(mu_) {
431 mutex_lock l(mu_);
432 num_buckets_ = keys.dim_size(0);
433 key_buckets_ = PersistentTensor(keys);
434 value_buckets_ = PersistentTensor(values);
435 // Count the number of keys that are not the empty_key. This requires
436 // iterating through the whole table but that is OK as we only execute it
437 // during checkpoint restore.
438 num_entries_ = 0;
439 const auto empty_key_tensor =
440 empty_key_.AccessTensor(ctx)->template shaped<K, 2>(
441 {1, key_shape_.num_elements()});
442 const auto key_buckets_tensor =
443 key_buckets_.AccessTensor(ctx)->template matrix<K>();
444 for (int64 i = 0; i < num_buckets_; ++i) {
445 if (!IsEqualKey(key_buckets_tensor, i, empty_key_tensor, 0)) {
446 ++num_entries_;
447 }
448 }
449 return Status::OK();
450 }
451
452 Status ExportValues(OpKernelContext* ctx) override LOCKS_EXCLUDED(mu_) {
453 mutex_lock l(mu_);
454 Tensor key_buckets_tensor = *key_buckets_.AccessTensor(ctx);
455 Tensor value_buckets_tensor = *value_buckets_.AccessTensor(ctx);
456 TF_RETURN_IF_ERROR(ctx->set_output("keys", key_buckets_tensor));
457 TF_RETURN_IF_ERROR(ctx->set_output("values", value_buckets_tensor));
458 return Status::OK();
459 }
460
461 Status CheckKeyAndValueTensorsForImport(const Tensor& keys,
462 const Tensor& values) override {
463 TF_RETURN_IF_ERROR(CheckKeyAndValueTypes(keys, values));
464 TF_RETURN_IF_ERROR(CheckKeyShape(keys.shape()));
465
466 // The storage format in key_buckets_ and value_buckets_ is always vectors,
467 // even if the inputs are scalars. This is what eventually gets exported
468 // and is expected by the import method as well.
469 TensorShape key_shape = MaybeVectorizeShape(key_shape_);
470 TensorShape value_shape = MaybeVectorizeShape(value_shape_);
471
472 // Compute the final expected shape of the value by starting with the shape
473 // of all keys, removing the dimensions particular to each key and then
474 // appending the shape of a single value.
475 TensorShape expected_value_shape = keys.shape();
476 expected_value_shape.RemoveLastDims(key_shape.dims());
477 expected_value_shape.AppendShape(value_shape);
478 if (values.shape() != expected_value_shape) {
479 return errors::InvalidArgument(
480 "Expected shape ", expected_value_shape.DebugString(),
481 " for value, got ", values.shape().DebugString());
482 }
483 return Status::OK();
484 }
485
486 DataType key_dtype() const override { return DataTypeToEnum<K>::v(); }
487
488 DataType value_dtype() const override { return DataTypeToEnum<V>::v(); }
489
490 TensorShape key_shape() const override { return key_shape_; }
491
492 TensorShape value_shape() const override { return value_shape_; }
493
494 int64 MemoryUsed() const override {
495 mutex_lock l(mu_);
496 return sizeof(MutableDenseHashTable) + key_buckets_.AllocatedBytes() +
497 value_buckets_.AllocatedBytes() + empty_key_.AllocatedBytes();
498 }
499
500 private:
501 Status DoInsert(OpKernelContext* ctx, const Tensor& key, const Tensor& value,
502 bool ignore_empty_key) EXCLUSIVE_LOCKS_REQUIRED(mu_) {
503 const int64 num_elements = key.dim_size(0);
504 const int64 value_size = value_shape_.num_elements();
505 const int64 key_size = key_shape_.num_elements();
506 const auto key_matrix = key.shaped<K, 2>({num_elements, key_size});
507 auto value_matrix = value.shaped<V, 2>({num_elements, value_size});
508
509 auto key_buckets_matrix =
510 key_buckets_.AccessTensor(ctx)->template matrix<K>();
511 auto value_buckets_matrix =
512 value_buckets_.AccessTensor(ctx)->template matrix<V>();
513 const auto empty_key_tensor =
514 empty_key_.AccessTensor(ctx)->template shaped<K, 2>({1, key_size});
515 const int64 bit_mask = num_buckets_ - 1;
516 for (int64 i = 0; i < num_elements; ++i) {
517 const uint64 key_hash = HashKey(key_matrix, i);
518 if (empty_key_hash_ == key_hash &&
519 IsEqualKey(empty_key_tensor, 0, key_matrix, i)) {
520 if (ignore_empty_key) {
521 continue;
522 }
523 return errors::InvalidArgument(
524 "Using the empty_key as a table key is not allowed");
525 }
526 int64 bucket_index = key_hash & bit_mask;
527 int64 num_probes = 0;
528 while (true) {
529 if (IsEqualKey(key_buckets_matrix, bucket_index, key_matrix, i)) {
530 for (int64 j = 0; j < value_size; ++j) {
531 value_buckets_matrix(bucket_index, j) =
532 SubtleMustCopyIfIntegral(value_matrix(i, j));
533 }
534 break;
535 }
536 if (IsEqualKey(key_buckets_matrix, bucket_index, empty_key_tensor, 0)) {
537 ++num_entries_;
538 for (int64 j = 0; j < key_size; ++j) {
539 key_buckets_matrix(bucket_index, j) =
540 SubtleMustCopyIfIntegral(key_matrix(i, j));
541 }
542 for (int64 j = 0; j < value_size; ++j) {
543 value_buckets_matrix(bucket_index, j) =
544 SubtleMustCopyIfIntegral(value_matrix(i, j));
545 }
546 break;
547 }
548 ++num_probes;
549 bucket_index =
550 (bucket_index + num_probes) & bit_mask; // quadratic probing
551 if (num_probes >= num_buckets_) {
552 return errors::Internal(
553 "Internal error in MutableDenseHashTable insert");
554 }
555 }
556 }
557 return Status::OK();
558 }
559
560 Status AllocateBuckets(OpKernelContext* ctx, int64 new_num_buckets)
561 EXCLUSIVE_LOCKS_REQUIRED(mu_) {
562 if (new_num_buckets < 4 ||
563 ((new_num_buckets & (new_num_buckets - 1)) != 0)) {
564 return errors::InvalidArgument(
565 "Number of buckets must be at least 4 and a power of 2, got: ",
566 new_num_buckets);
567 }
568 num_buckets_ = new_num_buckets;
569 num_entries_ = 0;
570
571 const int64 key_size = key_shape_.num_elements();
572 Tensor* key_buckets_tensor;
573 TF_RETURN_IF_ERROR(ctx->allocate_persistent(
574 key_dtype(), TensorShape({num_buckets_, key_size}), &key_buckets_,
575 &key_buckets_tensor));
576 auto key_buckets_matrix = key_buckets_tensor->matrix<K>();
577 const auto empty_key_flat =
578 empty_key_.AccessTensor(ctx)->template flat<K>();
579 for (int64 i = 0; i < num_buckets_; ++i) {
580 for (int64 j = 0; j < key_size; ++j) {
581 key_buckets_matrix(i, j) = empty_key_flat(j);
582 }
583 }
584
585 const int64 value_size = value_shape_.num_elements();
586 Tensor* value_buckets_tensor;
587 TF_RETURN_IF_ERROR(ctx->allocate_persistent(
588 value_dtype(), TensorShape({num_buckets_, value_size}), &value_buckets_,
589 &value_buckets_tensor));
590 auto value_buckets_matrix = value_buckets_tensor->matrix<V>();
591 for (int64 i = 0; i < num_buckets_; ++i) {
592 for (int64 j = 0; j < value_size; ++j) {
593 // Initialize values to the default value for the type to avoid
594 // exposing uninitialized memory in ExportValues().
595 value_buckets_matrix(i, j) = V();
596 }
597 }
598 return Status::OK();
599 }
600
601 Status Rebucket(OpKernelContext* ctx, int64 num_new_buckets)
602 EXCLUSIVE_LOCKS_REQUIRED(mu_) {
603 Tensor old_key_buckets = *key_buckets_.AccessTensor(ctx);
604 Tensor old_value_buckets = *value_buckets_.AccessTensor(ctx);
605 TF_RETURN_IF_ERROR(AllocateBuckets(ctx, num_new_buckets));
606 return DoInsert(ctx, old_key_buckets, old_value_buckets, true);
607 }
608
609 uint64 HashKey(typename TTypes<K>::ConstMatrix key, int64 index) const {
610 if (key_shape_.num_elements() == 1) {
611 return HashScalar(key(index, 0));
612 }
613 uint64 result = 0;
614 for (int64 i = 0; i < key_shape_.num_elements(); ++i) {
615 result = Hash64Combine(result, HashScalar(key(index, i)));
616 }
617 return result;
618 }
619
620 // Use a template to allow this function to be used both with Matrix and
621 // ConstMatrix types.
622 template <typename MT2>
623 bool IsEqualKey(typename TTypes<K>::Matrix tensor1, int64 index1, MT2 tensor2,
624 int64 index2) const {
625 for (int64 i = 0; i < key_shape_.num_elements(); ++i) {
626 if (tensor1(index1, i) != tensor2(index2, i)) {
627 return false;
628 }
629 }
630 return true;
631 }
632
633 TensorShape key_shape_;
634 TensorShape value_shape_;
635 float max_load_factor_;
636 mutable mutex mu_;
637 int64 num_entries_ GUARDED_BY(mu_);
638 int64 num_buckets_ GUARDED_BY(mu_);
639 PersistentTensor key_buckets_ GUARDED_BY(mu_);
640 PersistentTensor value_buckets_ GUARDED_BY(mu_);
641 PersistentTensor empty_key_;
642 uint64 empty_key_hash_;
643};

647// Table lookup op. Perform the lookup operation on the given table.
648class LookupTableFindOp : public OpKernel {
649 public:
650 explicit LookupTableFindOp(OpKernelConstruction* ctx) : OpKernel(ctx) {}
651
652 void Compute(OpKernelContext* ctx) override {
653 lookup::LookupInterface* table;
654 OP_REQUIRES_OK(ctx, GetLookupTable("table_handle", ctx, &table));
655 core::ScopedUnref unref_me(table);
656
657 // Input 0 could be a STRING_REF or a RESOURCE
658 DataType expected_input_0 =
659 (ctx->input_dtype(0) == DT_RESOURCE) ? DT_RESOURCE : DT_STRING_REF;
660 DataTypeVector expected_inputs = {expected_input_0, table->key_dtype(),
661 table->value_dtype()};
662 DataTypeVector expected_outputs = {table->value_dtype()};
663 OP_REQUIRES_OK(ctx, ctx->MatchSignature(expected_inputs, expected_outputs));
664
665 const Tensor& key = ctx->input(1);
666 const Tensor& default_value = ctx->input(2);
667 OP_REQUIRES_OK(ctx, table->CheckFindArguments(key, default_value));
668
669 TensorShape output_shape = key.shape();
670 output_shape.RemoveLastDims(table->key_shape().dims());
671 output_shape.AppendShape(table->value_shape());
672 Tensor* out;
673 OP_REQUIRES_OK(ctx, ctx->allocate_output("values", output_shape, &out));
674
675 OP_REQUIRES_OK(ctx, table->Find(ctx, key, out, default_value));
676 }
677};
