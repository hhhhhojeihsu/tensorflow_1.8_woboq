29// See documentation in ../ops/dataset_ops.cc for a high-level
30// description of the following ops.
31
32class SqlDatasetOp : public DatasetOpKernel {
33 public:
34 explicit SqlDatasetOp(OpKernelConstruction* ctx) : DatasetOpKernel(ctx) {
35 OP_REQUIRES_OK(ctx, ctx->GetAttr("output_types", &output_types_));
36 OP_REQUIRES_OK(ctx, ctx->GetAttr("output_shapes", &output_shapes_));
37 for (const DataType& dt : output_types_) {
38 OP_REQUIRES(ctx,
39 dt == DT_STRING || dt == DT_INT8 || dt == DT_INT16 ||
40 dt == DT_INT32 || dt == DT_INT64 || dt == DT_UINT8 ||
41 dt == DT_UINT16 || dt == DT_BOOL || dt == DT_DOUBLE,
42 errors::InvalidArgument(
43 "Each element of `output_types_` must be one of: "
44 "DT_STRING, DT_INT8, DT_INT16, DT_INT32, DT_INT64, "
45 "DT_UINT8, DT_UINT16, DT_BOOL, DT_DOUBLE "));
46 }
47 for (const PartialTensorShape& pts : output_shapes_) {
48 OP_REQUIRES(ctx, pts.dims() == 0,
49 errors::InvalidArgument(
50 "Each element of `output_shapes_` must be a scalar."));
51 }
52 }
53 void MakeDataset(OpKernelContext* ctx, DatasetBase** output) override {
54 string driver_name;
55 OP_REQUIRES_OK(
56 ctx, ParseScalarArgument<string>(ctx, "driver_name", &driver_name));
57
58 string data_source_name;
59 OP_REQUIRES_OK(ctx, ParseScalarArgument<string>(ctx, "data_source_name",
60 &data_source_name));
61
62 string query;
63 OP_REQUIRES_OK(ctx, ParseScalarArgument<string>(ctx, "query", &query));
64
65 // TODO(b/64276826) Change this check when we add support for other
66 // databases.
67 OP_REQUIRES(ctx, driver_name == "sqlite",
68 errors::InvalidArgument(tensorflow::strings::Printf(
69 "The database type, %s, is not supported by SqlDataset. "
70 "The set of supported databases is: {'sqlite'}.",
71 driver_name.c_str())));
72
73 *output = new Dataset(driver_name, data_source_name, query, output_types_,
74 output_shapes_);
75 }
76
77 private:
78 class Dataset : public DatasetBase {
79 public:
80 Dataset(const string& driver_name, const string& data_source_name,
81 const string& query, const DataTypeVector& output_types,
82 const std::vector<PartialTensorShape>& output_shapes)
83 : driver_name_(driver_name),
84 data_source_name_(data_source_name),
85 query_(query),
86 output_types_(output_types),
87 output_shapes_(output_shapes) {}
88
89 std::unique_ptr<IteratorBase> MakeIterator(
90 const string& prefix) const override {
91 return std::unique_ptr<IteratorBase>(
92 new Iterator({this, strings::StrCat(prefix, "::Sql")}));
93 }
94
95 const DataTypeVector& output_dtypes() const override {
96 return output_types_;
97 }
98
99 const std::vector<PartialTensorShape>& output_shapes() const override {
100 return output_shapes_;
101 }
102
103 string DebugString() override { return "SqlDatasetOp::Dataset"; }
104
105 private:
106 class Iterator : public DatasetIterator<Dataset> {
107 public:
108 explicit Iterator(const Params& params)
109 : DatasetIterator<Dataset>(params) {}
110 ~Iterator() override {
111 if (query_connection_initialized_) {
112 Status s = query_connection_->Close();
113 if (!s.ok()) {
114 LOG(WARNING) << "Failed to close query connection: " << s;
115 }
116 }
117 }
118
119 Status GetNextInternal(IteratorContext* ctx,
120 std::vector<Tensor>* out_tensors,
121 bool* end_of_sequence) override {
122 mutex_lock l(mu_);
123 if (!query_connection_initialized_) {
124 query_connection_initialized_ = true;
125 query_connection_ = sql::DriverManager::CreateQueryConnection(
126 dataset()->driver_name_);
127 Status s = query_connection_->Open(dataset()->data_source_name_,
128 dataset()->query_,
129 dataset()->output_types_);
130 if (!s.ok()) {
131 LOG(WARNING) << "Failed to connect to database: " << s;
132 return s;
133 }
134 }
135 return query_connection_->GetNext(ctx, out_tensors, end_of_sequence);
136 }
137
138 private:
139 mutex mu_;
140 std::unique_ptr<sql::QueryConnection> query_connection_ GUARDED_BY(mu_);
141 bool query_connection_initialized_ GUARDED_BY(mu_) = false;
142 };
143 const string driver_name_;
144 const string data_source_name_;
145 const string query_;
146 const DataTypeVector output_types_;
147 const std::vector<PartialTensorShape> output_shapes_;
148 };
149 DataTypeVector output_types_;
150 std::vector<PartialTensorShape> output_shapes_;
151};
152
153REGISTER_KERNEL_BUILDER(Name("SqlDataset").Device(DEVICE_CPU), SqlDatasetOp);
