24// See documentation in ../ops/dataset_ops.cc for a high-level
25// description of the following op.
26
27class DenseToSparseBatchDatasetOp : public UnaryDatasetOpKernel {
28 public:
29 explicit DenseToSparseBatchDatasetOp(OpKernelConstruction* ctx)
30 : UnaryDatasetOpKernel(ctx) {}
31
32 void MakeDataset(OpKernelContext* ctx, DatasetBase* input,
33 DatasetBase** output) override {
34 // Create a new DenseToSparseBatchDatasetOp::Dataset, insert it in the
35 // step-local container, and return it as the output.
36 OP_REQUIRES(
37 ctx, input->output_dtypes().size() == 1,
38 errors::InvalidArgument("DenseToSparseBatchDataset only supports "
39 "inputs with a single component."));
40
41 int64 batch_size;
42 OP_REQUIRES_OK(ctx,
43 ParseScalarArgument<int64>(ctx, "batch_size", &batch_size));
44 OP_REQUIRES(
45 ctx, batch_size > 0,
46 errors::InvalidArgument("Batch size must be greater than zero."));
47
48 const Tensor* row_shape_t;
49 OP_REQUIRES_OK(ctx, ctx->input("row_shape", &row_shape_t));
50 OP_REQUIRES(ctx, TensorShapeUtils::IsVector(row_shape_t->shape()),
51 errors::InvalidArgument("row_shape must be a vector"));
52 PartialTensorShape row_shape;
53 OP_REQUIRES_OK(ctx, PartialTensorShape::MakePartialShape(
54 row_shape_t->vec<int64>().data(),
55 row_shape_t->NumElements(), &row_shape));
56
57 *output = nullptr;
58
59#define HANDLE_TYPE(T) \
60 case DataTypeToEnum<T>::value: { \
61 *output = new Dataset<T>(ctx, batch_size, row_shape, input); \
62 break; \
63 }
64
65 switch (input->output_dtypes()[0]) {
66 TF_CALL_DATASET_TYPES(HANDLE_TYPE);
67#undef HANDLE_TYPE
68 default:
69 OP_REQUIRES(ctx, false,
70 errors::Unimplemented(
71 "DenseToSparseBatchDataset unhandled data type: ",
72 input->output_dtypes()[0]));
73 }
74 }
75
76 private:
77 // TODO(mrry): Push the templated code down to the raw copying routine.
78 template <class T>
79 class Dataset : public GraphDatasetBase {
80 public:
81 Dataset(OpKernelContext* ctx, int64 batch_size,
82 const PartialTensorShape& row_shape, const DatasetBase* input)
83 : GraphDatasetBase(ctx),
84 batch_size_(batch_size),
85 row_shape_(row_shape),
86 input_(input) {
87 input_->Ref();
88
89 output_shapes_.reserve(1);
90 PartialTensorShape output_shape({-1});
91 output_shape.AppendShape(row_shape_);
92 output_shapes_.push_back(output_shape);
93 }
94
95 ~Dataset() override { input_->Unref(); }
96
97 std::unique_ptr<IteratorBase> MakeIterator(
98 const string& prefix) const override {
99 return std::unique_ptr<IteratorBase>(new Iterator(
100 {this, strings::StrCat(prefix, "::DenseToSparseBatch")}));
101 }
102
103 const DataTypeVector& output_dtypes() const override {
104 static DataTypeVector* output_dtypes_ = new DataTypeVector({DT_VARIANT});
105 return *output_dtypes_;
106 }
107
108 const std::vector<PartialTensorShape>& output_shapes() const override {
109 return output_shapes_;
110 }
111
112 string DebugString() override {
113 return strings::StrCat("DenseToSparseBatchDatasetOp(", batch_size_,
114 ")::Dataset");
115 }
116
117 protected:
118 Status AsGraphDefInternal(OpKernelContext* ctx, DatasetGraphDefBuilder* b,
119 Node** output) const override {
120 Node* input_node;
121 TF_RETURN_IF_ERROR(b->AddParentDataset(ctx, input_, &input_node));
122 Node* batch_size_node;
123 TF_RETURN_IF_ERROR(b->AddScalar(batch_size_, &batch_size_node));
124 Node* row_shape_node;
125 std::vector<int64> row_shape;
126 row_shape.reserve(
127 row_shape_.dims()); // not an unknown rank PartialTensorShape
128 for (int i = 0; i < row_shape_.dims(); i++)
129 row_shape.emplace_back(row_shape_.dim_size(i));
130 TF_RETURN_IF_ERROR(b->AddVector(row_shape, &row_shape_node));
131 TF_RETURN_IF_ERROR(b->AddDataset(
132 this, {input_node, batch_size_node, row_shape_node}, output));
133 return Status::OK();
134 }
135
136 private:
137 class Iterator : public DatasetIterator<Dataset<T>> {
138 public:
139 explicit Iterator(const typename Iterator::Params& params)
140 : DatasetIterator<Dataset<T>>(params),
141 input_impl_(params.dataset->input_->MakeIterator(params.prefix)) {}
142
143 Status GetNextInternal(IteratorContext* ctx,
144 std::vector<Tensor>* out_tensors,
145 bool* end_of_sequence) override {
146 // Each row of the output SparseTensor is an individual tensor
147 // from the input iterator.
148 std::vector<Tensor> batch_elements;
149 int64 total_elements = 0;
150 batch_elements.reserve(
151 DatasetIterator<Dataset<T>>::dataset()->batch_size_);
152 const PartialTensorShape& row_shape =
153 DatasetIterator<Dataset<T>>::dataset()->row_shape_;
154 const int row_ndims = row_shape.dims();
155
156 // Determine the size of the output tensors:
157 // * dense_shape will be [`row_shape + 1`].
158 Tensor dense_shape(ctx->allocator({}), DT_INT64, {row_ndims + 1});
159 auto dense_shape_vec = dense_shape.vec<int64>();
160 for (size_t i = 0; i < row_ndims; ++i) {
161 if (row_shape.dim_size(i) == -1) {
162 dense_shape_vec(i + 1) = 0;
163 } else {
164 dense_shape_vec(i + 1) = row_shape.dim_size(i);
165 }
166 }
167
168 {
169 mutex_lock l(mu_);
170 *end_of_sequence = false;
171 for (int i = 0;
172 i < DatasetIterator<Dataset<T>>::dataset()->batch_size_ &&
173 !*end_of_sequence;
174 ++i) {
175 std::vector<Tensor> batch_element_tuple;
176 TF_RETURN_IF_ERROR(input_impl_->GetNext(ctx, &batch_element_tuple,
177 end_of_sequence));
178 if (!*end_of_sequence) {
179 DCHECK_EQ(1, batch_element_tuple.size());
180 batch_elements.push_back(std::move(batch_element_tuple[0]));
181 total_elements += batch_element_tuple[0].NumElements();
182
183 // TODO(mrry): Investigate how to hoist this check when we
184 // have static information that renders it unnecessary.
185 if (batch_element_tuple[0].shape().dims() != row_ndims) {
186 return errors::InvalidArgument(
187 "Input element had shape (",
188 batch_element_tuple[0].shape().DebugString(),
189 ") that is incompatible with the row shape (",
190 row_shape.DebugString(), ").");
191 }
192 for (int j = 0; j < row_ndims; ++j) {
193 // Take the maximum in the dimension if -1 is given.
194 if (row_shape.dim_size(j) == -1) {
195 dense_shape_vec(j + 1) =
196 std::max(batch_element_tuple[0].dim_size(j),
197 dense_shape_vec(j + 1));
198 } else if (batch_element_tuple[0].dim_size(j) >
199 row_shape.dim_size(j)) {
200 return errors::DataLoss(
201 "Input element had shape (",
202 batch_element_tuple[0].shape().DebugString(),
203 ") that is larger than the row shape (",
204 row_shape.DebugString(), ").");
205 }
206 }
207 }
208 }
209 }
210
211 if (batch_elements.empty()) {
212 DCHECK(*end_of_sequence);
213 return Status::OK();
214 }
215
216 // * indices will be [`total_elements`, `row_shape + 1`].
217 // * values will be [`total_elements`].
218 Tensor indices(ctx->allocator({}), DT_INT64,
219 {total_elements, row_ndims + 1});
220 Tensor values(
221 ctx->allocator({}),
222 DatasetIterator<Dataset<T>>::dataset()->input_->output_dtypes()[0],
223 {total_elements});
224 auto indices_matrix = indices.matrix<int64>();
225 auto values_flat = values.flat<T>();
226
227 int64 current_position_in_values = 0;
228 for (int64 i = 0; i < batch_elements.size(); ++i) {
229 const Tensor& t = batch_elements[i];
230 const auto& t_flat = t.flat<T>();
231 // TODO(mrry): Replace with a memcpy or something more
232 // efficient. (Maybe an Eigen assign op?)
233 gtl::InlinedVector<int64, 4> strides(row_ndims);
234 if (!strides.empty()) {
235 strides[row_ndims - 1] = 1;
236 for (int64_t row_dim = strides.size() - 2; row_dim >= 0;
237 --row_dim) {
238 strides[row_dim] =
239 strides[row_dim + 1] * t.shape().dim_size(row_dim + 1);
240 }
241 }
242
243 for (int64 j = 0; j < t.NumElements(); ++j) {
244 values_flat(current_position_in_values) = t_flat(j);
245 indices_matrix(current_position_in_values, 0) = i;
246 int64 index = j;
247 for (size_t k = 0; k < strides.size(); ++k) {
248 indices_matrix(current_position_in_values, k + 1) =
249 index / strides[k];
250 index %= strides[k];
251 }
252 ++current_position_in_values;
253 }
254 }
255
256 dense_shape_vec(0) = batch_elements.size();
257
258 Tensor serialized_sparse(DT_VARIANT, TensorShape({3}));
259 auto serialized_sparse_t = serialized_sparse.vec<Variant>();
260 serialized_sparse_t(0) = std::move(indices);
261 serialized_sparse_t(1) = std::move(values);
262 serialized_sparse_t(2) = std::move(dense_shape);
263 out_tensors->push_back(std::move(serialized_sparse));
264
265 *end_of_sequence = false;
266 return Status::OK();
267 }
268
269 protected:
270 Status SaveInternal(IteratorStateWriter* writer) override {
271 mutex_lock l(mu_);
272 TF_RETURN_IF_ERROR(Iterator::SaveParent(writer, input_impl_));
273 return Status::OK();
274 }
275
276 Status RestoreInternal(IteratorContext* ctx,
277 IteratorStateReader* reader) override {
278 mutex_lock l(mu_);
279 TF_RETURN_IF_ERROR(Iterator::RestoreParent(ctx, reader, input_impl_));
280 return Status::OK();
281 }
282
283 private:
284 mutex mu_;
285 std::unique_ptr<IteratorBase> input_impl_ GUARDED_BY(mu_);
286 };
287
288 const int64 batch_size_;
289 const PartialTensorShape row_shape_;
290 const DatasetBase* const input_;
291 std::vector<PartialTensorShape> output_shapes_;
292 };
293};
