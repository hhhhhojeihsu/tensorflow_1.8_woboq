45Status VerifyTypesMatch(const DataTypeVector& expected,
46 const DataTypeVector& received) {
47 if (expected.size() != received.size()) {
48 return errors::InvalidArgument(
49 "Number of components does not match: expected ", expected.size(),
50 " types but got ", received.size(), ".");
51 }
52 for (size_t i = 0; i < expected.size(); ++i) {
53 if (expected[i] != received[i]) {
54 return errors::InvalidArgument("Data type mismatch at component ", i,
55 ": expected ", DataTypeString(expected[i]),
56 " but got ", DataTypeString(received[i]),
57 ".");
58 }
59 }
60 return Status::OK();
61}
62
63Status VerifyShapesCompatible(const std::vector<PartialTensorShape>& expected,
64 const std::vector<PartialTensorShape>& received) {
65 if (expected.size() != received.size()) {
66 return errors::InvalidArgument(
67 "Number of components does not match: expected ", expected.size(),
68 " shapes but got ", received.size(), ".");
69 }
70 for (size_t i = 0; i < expected.size(); ++i) {
71 if (!expected[i].IsCompatibleWith(received[i])) {
72 return errors::InvalidArgument("Incompatible shapes at component ", i,
73 ": expected ", expected[i].DebugString(),
74 " but got ", received[i].DebugString(),
75 ".");
76 }
77 }
78
79 return Status::OK();
80}

82class IteratorResource : public ResourceBase {
83 public:
84 IteratorResource(const DataTypeVector& output_dtypes,
85 const std::vector<PartialTensorShape>& output_shapes,
86 const int /*unused: graph_def_version*/,
87 std::unique_ptr<DeviceMgr> device_mgr,
88 std::unique_ptr<FunctionLibraryDefinition> flib_def,
89 std::unique_ptr<ProcessFunctionLibraryRuntime> pflr,
90 FunctionLibraryRuntime* lib)
91 : device_mgr_(std::move(device_mgr)),
92 flib_def_(std::move(flib_def)),
93 pflr_(std::move(pflr)),
94 lib_(lib),
95 iterator_(nullptr),
96 output_dtypes_(output_dtypes),
97 output_shapes_(output_shapes) {}
98
99 Status GetNext(IteratorContext* ctx, std::vector<Tensor>* out_tensors,
100 bool* end_of_sequence) {
101 std::shared_ptr<IteratorBase> captured_iterator(iterator_);
102 if (captured_iterator) {
103 if (lib_ != nullptr) {
104 ctx->set_lib(lib_);
105 }
106 return captured_iterator->GetNext(ctx, out_tensors, end_of_sequence);
107 } else {
108 return errors::FailedPrecondition(
109 "GetNext() failed because the iterator has not been initialized. "
110 "Ensure that you have run the initializer operation for this "
111 "iterator before getting the next element.");
112 }
113 }
114
115 Status Save(OpKernelContext* ctx, IteratorStateWriter* writer) {
116 std::shared_ptr<IteratorBase> captured_iterator(iterator_);
117 if (captured_iterator) {
118 return captured_iterator->Save(ctx, writer);
119 } else {
120 return errors::FailedPrecondition(
121 "Save() failed because the iterator has not been initialized. "
122 "Ensure that you have run the initializer operation for this "
123 "iterator before saving it.");
124 }
125 }

127 Status Restore(OpKernelContext* ctx, IteratorStateReader* reader) {
128 string serialized_graph_def;
129 TF_RETURN_IF_ERROR(reader->ReadScalar(GraphDatasetBase::kDatasetGraphKey,
130 &serialized_graph_def));
131 GraphDef graph_def;
132 if (!graph_def.ParseFromString(serialized_graph_def)) {
133 return errors::Internal("Error parsing dataset GraphDef.");
134 }
135 string output_node;
136 TF_RETURN_IF_ERROR(reader->ReadScalar(
137 GraphDatasetBase::kDatasetGraphOutputNodeKey, &output_node));
138 DatasetBase* dataset = nullptr;
139 Graph graph(OpRegistry::Global());
140 TF_RETURN_IF_ERROR(ImportGraphDef({}, graph_def, &graph, nullptr));
141 std::vector<Tensor> outputs;
142 GraphRunner graph_runner(ctx->env());
143
144 // Build a new FLR that knows about the functions in the graph, and use
145 // it for all operations on the restored iterator.
146 // NOTE(mrry): We clone the existing FLR and use it in the GraphRunner
147 // because some of the OpKernels in the graph might call functions that are
148 // only defined in the loaded GraphDef.
149 FunctionLibraryRuntime* lib;
150 std::unique_ptr<DeviceMgr> device_mgr(nullptr);
151 std::unique_ptr<FunctionLibraryDefinition> flib_def(nullptr);
152 std::unique_ptr<ProcessFunctionLibraryRuntime> pflr(nullptr);
153 TF_RETURN_IF_ERROR(ctx->function_library()->Clone(&flib_def, &pflr, &lib));
154 TF_RETURN_IF_ERROR(flib_def->AddLibrary(graph_def.library()));
155
156 TF_RETURN_IF_ERROR(
157 graph_runner.Run(&graph, lib, {}, {output_node}, &outputs));
158 TF_RETURN_IF_ERROR(GetDatasetFromVariantTensor(outputs[0], &dataset));
159
160 TF_RETURN_IF_ERROR(set_iterator(dataset->MakeIterator("Iterator")));
161 std::shared_ptr<IteratorBase> captured_iterator(iterator_);
162
163 if (captured_iterator) {
164 IteratorContext::Params params;
165 params.env = ctx->env();
166 params.runner = *(ctx->runner());
167 params.lib = lib;
168 DeviceBase* device = lib->device();
169 params.allocator_getter = [device](AllocatorAttributes attrs) {
170 return device->GetAllocator(attrs);
171 };
172 IteratorContext iter_ctx(std::move(params));
173
174 TF_RETURN_IF_ERROR(captured_iterator->Restore(&iter_ctx, reader));
175 mutex_lock l(mu_);
176 device_mgr_ = std::move(device_mgr);
177 lib_def_ = std::move(flib_def);
178 pflr_ = std::move(pflr);
179 lib_ = lib;
180 return Status::OK();
181 } else {
182 return errors::FailedPrecondition(
183 "Failed to restore iterator. Make sure the checkpoint ",
184 "is not corrupt. If the checkpoint does not contain the GraphDef, ",
185 "you will need to initialize your iterator before restoring.");
186 }
187 }
188
189 std::shared_ptr<const FunctionLibraryDefinition> function_library() {
190 tf_shared_lock l(mu_);
191 return lib_def_;
192 }
193
194 // Transfers ownership of iterator to this. This method is thread-safe.
195 Status set_iterator(std::unique_ptr<IteratorBase> iterator) {
196 if (iterator) {
197 TF_RETURN_IF_ERROR(
198 VerifyTypesMatch(output_dtypes_, iterator->output_dtypes()));
199 TF_RETURN_IF_ERROR(
200 VerifyShapesCompatible(output_shapes_, iterator->output_shapes()));
201 }
202 iterator_.reset(iterator.release());
203 return Status::OK();
204 }
205
206 void set_stats_aggregator(std::shared_ptr<StatsAggregator> stats_aggregator) {
207 mutex_lock l(mu_);
208 stats_aggregator_ = std::move(stats_aggregator);
209 }
210
211 std::shared_ptr<StatsAggregator> stats_aggregator() {
212 tf_shared_lock l(mu_);
213 return stats_aggregator_;
214 }
215
216 string DebugString() override { return "Iterator resource"; }
217
218 const DataTypeVector& output_dtypes() const { return output_dtypes_; }
219
220 const std::vector<PartialTensorShape>& output_shapes() const {
221 return output_shapes_;
222 }
223
224 private:
225 // The following (device_mgr_, flib_def_, pflr_) are only used when the
226 // IteratorResource is shared between sessions and in that case we create
227 // a new FLR. Otherwise these are set to null.
228 std::unique_ptr<DeviceMgr> device_mgr_;
229 std::unique_ptr<FunctionLibraryDefinition> flib_def_;
230 std::unique_ptr<ProcessFunctionLibraryRuntime> pflr_;
231 FunctionLibraryRuntime* lib_ = nullptr; // not owned.
232 std::shared_ptr<IteratorBase> iterator_;
233 mutex mu_;
234 std::shared_ptr<StatsAggregator> stats_aggregator_ GUARDED_BY(mu_);
235 std::shared_ptr<const FunctionLibraryDefinition> lib_def_ GUARDED_BY(mu_);
236 const DataTypeVector output_dtypes_;
237 const std::vector<PartialTensorShape> output_shapes_;
238};

240// Helper class for reading data from a VariantTensorData object.
241class VariantTensorDataReader : public IteratorStateReader {
242 public:
243 explicit VariantTensorDataReader(const VariantTensorData* data)
244 : data_(data) {
245 PreProcess();
246 }
247
248 // Returns OK iff the initialization was successful, i.e.,
249 // pre-processing did not have errors.
250 Status status() const { return status_; }
251
252 Status ReadScalar(StringPiece key, int64* val) override {
253 return ReadScalarInternal(key, val);
254 }
255
256 Status ReadScalar(StringPiece key, string* val) override {
257 return ReadScalarInternal(key, val);
258 }
259
260 Status ReadTensor(StringPiece key, Tensor* val) override {
261 return ReadTensorInternal(key, val);
262 }
263
264 bool Contains(StringPiece key) override {
265 return map_.find(key.ToString()) != map_.end();
266 }
267
268 private:
269 void PreProcess() {
270 string metadata;
271 data_->get_metadata(&metadata);
272 IteratorStateMetadata proto;
273 if (!proto.ParseFromString(metadata)) {
274 status_ = errors::Internal("Error parsing IteratorStateMetadata.");
275 return;
276 }
277 size_t num_entries = proto.keys_size();
278 CHECK_EQ(num_entries, data_->tensors_size());
279 for (size_t i = 0; i < num_entries; i++) {
280 map_[proto.keys(i)] = i;
281 }
282 }
283
284 template <typename T>
285 Status ReadScalarInternal(StringPiece key, T* val) {
286 if (map_.find(key.ToString()) == map_.end()) {
287 return errors::NotFound(key);
288 }
289 *val = data_->tensors(map_[key.ToString()]).scalar<T>()();
290 return Status::OK();
291 }
292
293 Status ReadTensorInternal(StringPiece key, Tensor* val) {
294 if (map_.find(key.ToString()) == map_.end()) {
295 return errors::NotFound(key);
296 }
297 *val = data_->tensors(map_[key.ToString()]);
298 return Status::OK();
299 }
300
301 std::map<string, size_t> map_;
302 const VariantTensorData* data_; // Not owned.
303 Status status_;
304};

306// Helper class for writing data to a VariantTensorData object.
307class VariantTensorDataWriter : public IteratorStateWriter {
308 public:
309 // Does not take ownership of data.
310 explicit VariantTensorDataWriter(VariantTensorData* data) : data_(data) {}
311
312 Status WriteScalar(StringPiece key, const int64 val) override {
313 return WriteScalarInternal(key, val);
314 }
315
316 Status WriteScalar(StringPiece key, const string& val) override {
317 return WriteScalarInternal(key, val);
318 }
319
320 Status WriteTensor(StringPiece key, const Tensor& val) override {
321 return WriteTensorInternal(key, val);
322 }
323
324 // Writes the metadata to `data_`.
325 Status Flush() {
326 string metadata;
327 if (!metadata_proto_.SerializeToString(&metadata)) {
328 return errors::Internal("Unable to serialize IteratorStateMetadata.");
329 }
330 data_->set_metadata(metadata);
331 return Status::OK();
332 }
333
334 private:
335 template <typename T>
336 Status WriteScalarInternal(StringPiece key, const T& val) {
337 Tensor val_t = Tensor(DataTypeToEnum<T>::v(), TensorShape({}));
338 val_t.scalar<T>()() = val;
339 return WriteTensorInternal(key, val_t);
340 }
341
342 Status WriteTensorInternal(StringPiece key, const Tensor& val) {
343 // Write key to the metadata proto. This gets written to `data_`
344 // when `Flush()` is called. We do this lazily to avoid multiple
345 // serialization calls.
346 metadata_proto_.add_keys(key.ToString());
347
348 // Update tensors.
349 *(data_->add_tensors()) = val;
350 return Status::OK();
351 }
352
353 VariantTensorData* data_;
354 // TODO(srbs): Set the version string.
355 IteratorStateMetadata metadata_proto_;
356};

358// Wrapper for encoding/decoding the iterator state stored in a Variant tensor.
359// The get() method returns an IteratorStateReader which can be used
360// to restore iterator state.
361//
362// Usage example:
363//
364// Encoding:
365//
366// Tensor t(DT_VARIANT, TensorShape({}));
367// t->scalar<Variant>()() = IteratorStateVariant(iterator_resource);
368//
369// Encode() sets the type_name of the VariantTensorData object to
370// IteratorStateVariant::TypeName().
371//
372// Decoding:
373//
374// Variant v = <VariantTensorDataProto object>;
375// DecodeUnaryVariant(&v);
376// IteratorStateVariant* wrapper = v.get<IteratorStateVariant>();
377// iterator_resource->Restore(ctx, wrapper->get())
378//
379// The type_name of the VariantTensorData object to be decoded must
380// match IteratorStateVariant::TypeName().
381class IteratorStateVariant {
382 public:
383 IteratorStateVariant() : data_(nullptr) {}
384 IteratorStateVariant(const IteratorStateVariant& other) : data_(nullptr) {
385 if (other.data_) {
386 Decode(*other.data_);
387 }
388 }
389 // Initializes this object with the current state of the iterator so
390 // that it can be written on the next call to Encode().
391 Status InitializeFromIterator(OpKernelContext* ctx,
392 IteratorResource* iterator_resource) {
393 data_.reset(new VariantTensorData());
394 data_->set_type_name(TypeName());
395 VariantTensorDataWriter writer(data_.get());
396 TF_RETURN_IF_ERROR(iterator_resource->Save(ctx, &writer));
397 TF_RETURN_IF_ERROR(writer.Flush());
398 return Status::OK();
399 }
400 string TypeName() const { return kIteratorVariantTypeName; }
401 void Encode(VariantTensorData* data) const { *data = *data_; }
402 bool Decode(const VariantTensorData& data) {
403 if (data.type_name() != TypeName()) {
404 return false;
405 }
406 std::unique_ptr<VariantTensorData> tensor_data(new VariantTensorData);
407 *tensor_data = data;
408 std::unique_ptr<VariantTensorDataReader> reader(
409 new VariantTensorDataReader(tensor_data.get()));
410 status_ = reader->status();
411 if (!status_.ok()) {
412 return false;
413 }
414 data_ = std::move(tensor_data);
415 reader_ = std::move(reader);
416 return true;
417 }
418 IteratorStateReader* get() { return reader_.get(); }
419 Status status() const { return status_; }
420 string DebugString() const {
421 if (data_) {
422 return strings::StrCat("IteratorStateVariant<",
423 "data: ", data_->DebugString(),
424 " status: ", status_.ToString(), ">");
425 } else {
426 return strings::StrCat("IteratorStateVariant<empty>");
427 }
428 }
429
430 private:
431 std::unique_ptr<IteratorStateReader> reader_;
432 Status status_;
433 std::unique_ptr<VariantTensorData> data_;
434};

439// to manually decode the returned Variant using MaybeDecodeAndCopy in
440// DeserializeIteratorOp which is not recommended.
441REGISTER_UNARY_VARIANT_DECODE_FUNCTION(IteratorStateVariant,

444class IteratorHandleOp : public OpKernel {
445 public:
446 explicit IteratorHandleOp(OpKernelConstruction* ctx)
447 : OpKernel(ctx), graph_def_version_(ctx->graph_def_version()) {
448 OP_REQUIRES_OK(ctx, ctx->GetAttr("output_types", &output_dtypes_));
449 OP_REQUIRES_OK(ctx, ctx->GetAttr("output_shapes", &output_shapes_));
450 OP_REQUIRES_OK(ctx, ctx->GetAttr("shared_name", &name_));
451 }
452
453 // The resource is deleted from the resource manager only when it is private
454 // to kernel. Ideally the resource should be deleted when it is no longer held
455 // by anyone, but it would break backward compatibility.
456 ~IteratorHandleOp() override {
457 if (resource_ != nullptr) {
458 resource_->Unref();
459 if (cinfo_.resource_is_private_to_kernel()) {
460 if (!cinfo_.resource_manager()
461 ->template Delete<IteratorResource>(cinfo_.container(),
462 cinfo_.name())
463 .ok()) {
464 // Do nothing; the resource can have been deleted by session resets.
465 }
466 }
467 }
468 }
469
470 void Compute(OpKernelContext* context) override LOCKS_EXCLUDED(mu_) {
471 {
472 mutex_lock l(mu_);
473 if (resource_ == nullptr) {
474 FunctionLibraryRuntime* lib;
475 std::unique_ptr<DeviceMgr> device_mgr(nullptr);
476 std::unique_ptr<FunctionLibraryDefinition> flib_def(nullptr);
477 std::unique_ptr<ProcessFunctionLibraryRuntime> pflr(nullptr);
478 // If the iterator is shared then we construct a new FLR, and pass that
479 // in. NOTE(mrry,rohanj): In this case it is not possible to call remote
480 // functions from the iterator. We may add this functionality if there
481 // is sufficient demand, but it will require a significant refactoring.
482 if (!name_.empty()) {
483 lib = CreatePrivateFLR(context, &device_mgr, &flib_def, &pflr);
484 } else {
485 OP_REQUIRES_OK(context, context->function_library()->Clone(
486 &flib_def, &pflr, &lib));
487 }
488
489 ResourceMgr* mgr = context->resource_manager();
490 OP_REQUIRES_OK(context, cinfo_.Init(mgr, def()));
491
492 IteratorResource* resource;
493 OP_REQUIRES_OK(
494 context,
495 mgr->LookupOrCreate<IteratorResource>(
496 cinfo_.container(), cinfo_.name(), &resource,
497 [lib, &device_mgr, &flib_def, &pflr,
498 this](IteratorResource** ret) EXCLUSIVE_LOCKS_REQUIRED(mu_) {
499 *ret = new IteratorResource(
500 output_dtypes_, output_shapes_, graph_def_version_,
501 std::move(device_mgr), std::move(flib_def),
502 std::move(pflr), lib);
503 return Status::OK();
504 }));
505
506 Status s = VerifyResource(resource);
507 if (TF_PREDICT_FALSE(!s.ok())) {
508 resource->Unref();
509 context->SetStatus(s);
510 return;
511 }
512
513 resource_ = resource;
514 }
515 }
516 OP_REQUIRES_OK(context, MakeResourceHandleToOutput(
517 context, 0, cinfo_.container(), cinfo_.name(),
518 MakeTypeIndex<IteratorResource>()));
519 }
520
521 private:
522 // During the first Compute(), resource is either created or looked up using
523 // shared_name. In the latter case, the resource found should be verified if
524 // it is compatible with this op's configuration. The verification may fail in
525 // cases such as two graphs asking queues of the same shared name to have
526 // inconsistent capacities.
527 Status VerifyResource(IteratorResource* resource) {
528 TF_RETURN_IF_ERROR(
529 VerifyTypesMatch(output_dtypes_, resource->output_dtypes()));
530 TF_RETURN_IF_ERROR(
531 VerifyShapesCompatible(output_shapes_, resource->output_shapes()));
532 return Status::OK();
533 }
534
535 template <typename To, typename From> // use like this: down_cast<T*>(foo);
536 static inline To down_cast(From* f) { // so we only accept pointers
537 static_assert(
538 (std::is_base_of<From, typename std::remove_pointer<To>::type>::value),
539 "target type not derived from source type");
540
541 // We skip the assert and hence the dynamic_cast if RTTI is disabled.
542#if !defined(__GNUC__) || defined(__GXX_RTTI)
543 // Uses RTTI in dbg and fastbuild. asserts are disabled in opt builds.
544 assert(f == nullptr || dynamic_cast<To>(f) != nullptr);
545#endif // !defined(__GNUC__) || defined(__GXX_RTTI)
546 return static_cast<To>(f);
547 }
548
549 FunctionLibraryRuntime* CreatePrivateFLR(
550 OpKernelContext* ctx, std::unique_ptr<DeviceMgr>* device_mgr,
551 std::unique_ptr<FunctionLibraryDefinition>* flib_def,
552 std::unique_ptr<ProcessFunctionLibraryRuntime>* pflr) {
553 // Wrap the existing device in order to see any captured resources
554 // in its resource manager. The existing device will outlive the
555 // IteratorResource, because we are storing the IteratorResource
556 // in that device's resource manager.
557 Device* wrapped_device = RenamedDevice::NewRenamedDevice(
558 ctx->device()->name(), down_cast<Device*>(ctx->device()),
559 false /* owns_underlying */, false /* isolate_session_state */);
560 device_mgr->reset(new DeviceMgr({wrapped_device}));
561 flib_def->reset(new FunctionLibraryDefinition(
562 *ctx->function_library()->GetFunctionLibraryDefinition()));
563 pflr->reset(new ProcessFunctionLibraryRuntime(
564 device_mgr->get(), ctx->env(), graph_def_version_, flib_def->get(),
565 {} /* TODO(mrry): OptimizerOptions? */,
566 nullptr /* TODO(mrry): ClusterFLR */));
567
568 return (*pflr)->GetFLR(ctx->device()->name());
569 }
570
571 mutex mu_;
572 ContainerInfo cinfo_; // Written once under mu_ then constant afterwards.
573 IteratorResource* resource_ GUARDED_BY(mu_) = nullptr;
574 DataTypeVector output_dtypes_;
575 std::vector<PartialTensorShape> output_shapes_;
576 const int graph_def_version_;
577 string name_;
578};

596class ToSingleElementOp : public AsyncOpKernel {
597 public:
598 explicit ToSingleElementOp(OpKernelConstruction* ctx)
599 : AsyncOpKernel(ctx),
600 thread_pool_(new thread::ThreadPool(
601 ctx->env(), ThreadOptions(),
602 strings::StrCat("to_single_element_op_thread_",
603 SanitizeThreadSuffix(name())),
604 1 /* num_threads */, false /* low_latency_hint */)) {}
605
606 void ComputeAsync(OpKernelContext* ctx, DoneCallback done) override {
607 // The call to `iterator->GetNext()` may block and depend on an
608 // inter-op thread pool thread, so we issue the call from the
609 // owned thread pool.
610 thread_pool_->Schedule([ctx, done]() {
611 DatasetBase* dataset;
612 OP_REQUIRES_OK_ASYNC(
613 ctx, GetDatasetFromVariantTensor(ctx->input(0), &dataset), done);
614 auto iterator = dataset->MakeIterator("SingleElementIterator");
615
616 IteratorContext::Params params;
617 params.env = ctx->env();
618 params.runner = *(ctx->runner());
619 params.lib = ctx->function_library();
620 DeviceBase* device = ctx->function_library()->device();
621 params.allocator_getter = [device](AllocatorAttributes attrs) {
622 return device->GetAllocator(attrs);
623 };
624
625 IteratorContext iter_ctx(std::move(params));
626
627 std::vector<Tensor> components;
628 components.reserve(dataset->output_dtypes().size());
629 bool end_of_sequence;
630
631 OP_REQUIRES_OK_ASYNC(
632 ctx, iterator->GetNext(&iter_ctx, &components, &end_of_sequence),
633 done);
634 OP_REQUIRES_ASYNC(ctx, !end_of_sequence,
635 errors::InvalidArgument("Dataset was empty."), done);
636
637 for (int i = 0; i < components.size(); ++i) {
638 // TODO(mrry): Check that the shapes match the shape attrs.
639 ctx->set_output(i, components[i]);
640 }
641
642 components.clear();
643 OP_REQUIRES_OK_ASYNC(
644 ctx, iterator->GetNext(&iter_ctx, &components, &end_of_sequence),
645 done);
646 OP_REQUIRES_ASYNC(
647 ctx, end_of_sequence,
648 errors::InvalidArgument("Dataset had more than one element."), done);
649
650 done();
651 });
652 }
653
654 private:
655 std::unique_ptr<thread::ThreadPool> thread_pool_;
656};

658class OneShotIteratorOp : public AsyncOpKernel {
659 public:
660 explicit OneShotIteratorOp(OpKernelConstruction* ctx)
661 : AsyncOpKernel(ctx),
662 thread_pool_(new thread::ThreadPool(
663 ctx->env(), ThreadOptions(),
664 strings::StrCat("one_shot_iterator_initialization_thread_",
665 SanitizeThreadSuffix(name())),
666 1 /* num_threads */, false /* low_latency_hint */)),
667 graph_def_version_(ctx->graph_def_version())
668
669 {
670 string shared_name;
671 OP_REQUIRES_OK(ctx, ctx->GetAttr("shared_name", &shared_name));
672 OP_REQUIRES(ctx, shared_name.empty(),
673 errors::InvalidArgument("OneShotIteratorOp does not currently "
674 "support the 'shared_name' attr."));
675 OP_REQUIRES_OK(ctx,
676 ctx->GetAttr("dataset_factory", &dataset_factory_func_));
677 OP_REQUIRES_OK(ctx, ctx->GetAttr("output_types", &output_dtypes_));
678 OP_REQUIRES_OK(ctx, ctx->GetAttr("output_shapes", &output_shapes_));
679 }
680
681 ~OneShotIteratorOp() override {
682 if (iterator_resource_ != nullptr) {
683 iterator_resource_->Unref();
684 if (!cinfo_.resource_manager()
685 ->Delete<IteratorResource>(cinfo_.container(), cinfo_.name())
686 .ok()) {
687 // Do nothing; the resource can have been deleted by session resets.
688 }
689 }
690 }
691
692 // NOTE(mrry): This is based on `ResourceOpKernel<T>::Compute()`,
693 // but due to the fact that `ResourceOpKernel<T>::CreateResource()`
694 // does not provide access to the `OpKernelContext*` and we need
695 // this to invoke the factory function, it's not possible to
696 // implement this kernel by implementing `CreateResource()`.
697 // Furthermore, due to the fact that this kernel might block when
698 // running the initialization function, we must implement this
699 // kernel as an async kernel.
700 void ComputeAsync(OpKernelContext* ctx, DoneCallback done) override {
701 {
702 mutex_lock l(mu_);
703 if (iterator_resource_ == nullptr && initialization_status_.ok()) {
704 // The initialization thread will call `done`.
705 if (!initialization_started_) {
706 // TODO(mrry): Convert the initialization code to use
707 // callbacks instead of wasting a thread.
708 thread_pool_->Schedule([this, ctx, done]() { Init(ctx, done); });
709 initialization_started_ = true;
710 } else {
711 done_callbacks_.emplace_back(ctx, std::move(done));
712 }
713 return;
714 }
715 }
716 ProduceOutput(ctx, std::move(done));
717 }
718
719 private:
720 void Init(OpKernelContext* ctx, DoneCallback done) {
721 IteratorResource* iterator = nullptr;
722 ContainerInfo cinfo;
723 Status s = TryInit(ctx, &iterator, &cinfo);
724
725 std::vector<std::pair<OpKernelContext*, DoneCallback>> callbacks_to_run;
726 {
727 mutex_lock l(mu_);
728 if (s.ok()) {
729 iterator_resource_ = iterator;
730 cinfo_ = cinfo;
731 }
732 initialization_status_ = s;
733 std::swap(done_callbacks_, callbacks_to_run);
734 }
735
736 for (auto&& ctx_done : callbacks_to_run) {
737 ProduceOutput(ctx_done.first, std::move(ctx_done.second));
738 }
739 ProduceOutput(ctx, std::move(done));
740 }
741
742 Status TryInit(OpKernelContext* ctx, IteratorResource** iterator,
743 ContainerInfo* cinfo) {
744 TF_RETURN_IF_ERROR(cinfo->Init(ctx->resource_manager(), def()));
745
746 FunctionLibraryRuntime* lib;
747 std::unique_ptr<FunctionLibraryDefinition> flib_def(nullptr);
748 std::unique_ptr<ProcessFunctionLibraryRuntime> pflr(nullptr);
749 TF_RETURN_IF_ERROR(ctx->function_library()->Clone(&flib_def, &pflr, &lib));
750
751 // Create an IteratorResource that will hold the iterator for this op.
752 TF_RETURN_IF_ERROR(
753 ctx->resource_manager()->LookupOrCreate<IteratorResource>(
754 cinfo->container(), cinfo->name(), iterator,
755 [lib, this, &flib_def, &pflr](IteratorResource** ret)
756 EXCLUSIVE_LOCKS_REQUIRED(mu_) {
757 *ret = new IteratorResource(
758 output_dtypes_, output_shapes_, graph_def_version_,
759 nullptr, std::move(flib_def), std::move(pflr), lib);
760 return Status::OK();
761 }));
762
763 core::ScopedUnref unref_iterator(*iterator);
764
765 TF_RETURN_IF_ERROR(
766 VerifyTypesMatch(output_dtypes_, (*iterator)->output_dtypes()));
767 TF_RETURN_IF_ERROR(
768 VerifyShapesCompatible(output_shapes_, (*iterator)->output_shapes()));
769
770 // Call the dataset_factory_func_ to create a new dataset,
771 // over which this op will iterate.
772 FunctionLibraryRuntime::Handle f_handle;
773 TF_RETURN_IF_ERROR(ctx->function_library()->Instantiate(
774 dataset_factory_func_.name(), AttrSlice(&dataset_factory_func_.attr()),
775 &f_handle));
776 FunctionLibraryRuntime::Options opts;
777 opts.cancellation_manager = ctx->cancellation_manager();
778 // Choose a step ID that is guaranteed not to clash with any
779 // Session-generated step ID. DirectSession only generates
780 // non-negative step IDs (contiguous, starting from 0), and
781 // MasterSession generates 56-bit random step IDs whose MSB is
782 // always 0, so a negative random step ID should suffice.
783 opts.step_id = -std::abs(static_cast<int64>(random::New64()));
784 ScopedStepContainer step_container(opts.step_id, [ctx](const string& name) {
785 ctx->resource_manager()->Cleanup(name).IgnoreError();
786 });
787 opts.step_container = &step_container;
788 opts.runner = ctx->runner();
789 Notification n;
790 Status factory_status;
791 std::vector<Tensor> return_values;
792 ctx->function_library()->Run(opts, f_handle, {}, &return_values,
793 [&n, &factory_status](Status s) {
794 factory_status.Update(s);
795 n.Notify();
796 });
797 n.WaitForNotification();
798 TF_RETURN_IF_ERROR(factory_status);
799 if (return_values.size() != 1 || return_values[0].dtype() != DT_VARIANT ||
800 !TensorShapeUtils::IsScalar(return_values[0].shape())) {
801 return errors::InvalidArgument(
802 "The `dataset_factory` function must return "
803 "a single scalar of dtype DT_VARIANT.");
804 }
805
806 // Create an iterator for the dataset that was created in the
807 // factory function.
808 DatasetBase* dataset;
809 TF_RETURN_IF_ERROR(GetDatasetFromVariantTensor(return_values[0], &dataset));
810 TF_RETURN_IF_ERROR(
811 (*iterator)->set_iterator(dataset->MakeIterator("Iterator")));
812
813 (*iterator)->Ref();
814 return Status::OK();
815 }
816
817 void ProduceOutput(OpKernelContext* ctx, const DoneCallback& done) {
818 Tensor* handle;
819 OP_REQUIRES_OK_ASYNC(ctx, ctx->allocate_output(0, TensorShape({}), &handle),
820 done);
821 Status s;
822 {
823 mutex_lock l(mu_);
824 s = initialization_status_;
825 if (s.ok()) {
826 handle->scalar<ResourceHandle>()() =
827 MakeResourceHandle<IteratorResource>(ctx, cinfo_.container(),
828 cinfo_.name());
829 }
830 }
831 OP_REQUIRES_OK_ASYNC(ctx, s, done);
832 done();
833 }
834
835 NameAttrList dataset_factory_func_;
836 DataTypeVector output_dtypes_;
837 std::vector<PartialTensorShape> output_shapes_;
838
839 std::unique_ptr<thread::ThreadPool> thread_pool_;
840
841 mutex mu_;
842 ContainerInfo cinfo_ GUARDED_BY(mu_);
843 IteratorResource* iterator_resource_ GUARDED_BY(mu_) = nullptr;
844
845 bool initialization_started_ GUARDED_BY(mu_) = false;
846 Status initialization_status_ GUARDED_BY(mu_);
847 std::vector<std::pair<OpKernelContext*, DoneCallback>> done_callbacks_
848 GUARDED_BY(mu_);
849 const int graph_def_version_;
850};

852class IteratorGetNextOp : public AsyncOpKernel {
853 public:
854 explicit IteratorGetNextOp(OpKernelConstruction* ctx)
855 : AsyncOpKernel(ctx),
856 thread_pool_(new thread::ThreadPool(
857 ctx->env(), ThreadOptions(),
858 strings::StrCat("iterator_get_next_thread_",
859 SanitizeThreadSuffix(name())),
860 1 /* num_threads */, false /* low_latency_hint */)) {}
861
862 void ComputeAsync(OpKernelContext* ctx, DoneCallback done) override {
863 IteratorResource* iterator;
864 OP_REQUIRES_OK_ASYNC(
865 ctx, LookupResource(ctx, HandleFromInput(ctx, 0), &iterator), done);
866 // The call to `iterator->GetNext()` may block and depend on an
867 // inter-op thread pool thread, so we issue the call from the
868 // owned thread pool.
869 thread_pool_->Schedule(std::bind(
870 [ctx, iterator](DoneCallback done) {
871 std::vector<Tensor> components;
872 bool end_of_sequence = false;
873
874 IteratorContext::Params params;
875 params.env = ctx->env();
876 params.stats_aggregator_getter = [iterator]() {
877 return iterator->stats_aggregator();
878 };
879 params.runner = *(ctx->runner());
880 params.function_library = iterator->function_library();
881 DeviceBase* device = ctx->function_library()->device();
882 params.allocator_getter = [device](AllocatorAttributes attrs) {
883 return device->GetAllocator(attrs);
884 };
885 IteratorContext iter_ctx(std::move(params));
886
887 Status s =
888 iterator->GetNext(&iter_ctx, &components, &end_of_sequence);
889 // NOTE(mrry): We must unref the iterator before calling `done()`, to
890 // avoid destruction races.
891 iterator->Unref();
892
893 if (!s.ok()) {
894 ctx->SetStatus(s);
895 } else if (end_of_sequence) {
896 ctx->SetStatus(errors::OutOfRange("End of sequence"));
897 } else {
898 for (int i = 0; i < components.size(); ++i) {
899 // TODO(mrry): Check that the shapes match the shape attrs.
900 ctx->set_output(i, components[i]);
901 }
902 }
903 done();
904 },
905 std::move(done)));
906 }
907
908 private:
909 std::unique_ptr<thread::ThreadPool> thread_pool_;
910};

1025 ctx, ctx->allocate_output(0, TensorShape({}), &resource_handle_t));
