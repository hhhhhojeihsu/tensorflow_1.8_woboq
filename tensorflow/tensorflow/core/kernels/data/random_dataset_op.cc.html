27// See documentation in ../ops/dataset_ops.cc for a high-level
28// description of the following op.
29
30class RandomDatasetOp : public DatasetOpKernel {
31 public:
32 explicit RandomDatasetOp(OpKernelConstruction* ctx) : DatasetOpKernel(ctx) {}
33
34 void MakeDataset(OpKernelContext* ctx, DatasetBase** output) override {
35 int64 seed;
36 OP_REQUIRES_OK(ctx, ParseScalarArgument<int64>(ctx, "seed", &seed));
37
38 int64 seed2;
39 OP_REQUIRES_OK(ctx, ParseScalarArgument<int64>(ctx, "seed2", &seed2));
40
41 // By TensorFlow convention, passing 0 for both seeds indicates
42 // that the shuffling should be seeded non-deterministically.
43 if (seed == 0 && seed2 == 0) {
44 seed = random::New64();
45 seed2 = random::New64();
46 }
47
48 *output = new Dataset(ctx, seed, seed2);
49 }
50
51 private:
52 class Dataset : public GraphDatasetBase {
53 public:
54 Dataset(OpKernelContext* ctx, int64 seed, int64 seed2)
55 : GraphDatasetBase(ctx), seed_(seed), seed2_(seed2) {}
56
57 std::unique_ptr<IteratorBase> MakeIterator(
58 const string& prefix) const override {
59 return std::unique_ptr<IteratorBase>(
60 new Iterator({this, strings::StrCat(prefix, "::Random")}));
61 }
62
63 const DataTypeVector& output_dtypes() const override {
64 static DataTypeVector* dtypes = new DataTypeVector({DT_INT64});
65 return *dtypes;
66 }
67
68 const std::vector<PartialTensorShape>& output_shapes() const override {
69 static std::vector<PartialTensorShape>* shapes =
70 new std::vector<PartialTensorShape>({{}});
71 return *shapes;
72 }
73
74 string DebugString() override {
75 return strings::StrCat("RandomDatasetOp(", seed_, ", ", seed2_,
76 ")::Dataset");
77 }
78
79 protected:
80 Status AsGraphDefInternal(DatasetGraphDefBuilder* b,
81 Node** output) const override {
82 Node* seed = nullptr;
83 Node* seed2 = nullptr;
84 TF_RETURN_IF_ERROR(b->AddScalar(seed_, &seed));
85 TF_RETURN_IF_ERROR(b->AddScalar(seed2_, &seed2));
86 TF_RETURN_IF_ERROR(b->AddDataset(this, {seed, seed2}, output));
87 return Status::OK();
88 }
89
90 private:
91 class Iterator : public DatasetIterator<Dataset> {
92 public:
93 explicit Iterator(const Params& params)
94 : DatasetIterator<Dataset>(params),
95 parent_generator_(dataset()->seed_, dataset()->seed2_),
96 generator_(&parent_generator_) {}
97
98 Status GetNextInternal(IteratorContext* ctx,
99 std::vector<Tensor>* out_tensors,
100 bool* end_of_sequence) override {
101 mutex_lock l(mu_);
102 Tensor value_tensor(ctx->allocator({}), DT_INT64, {});
103 value_tensor.scalar<int64>()() = Random();
104 out_tensors->emplace_back(std::move(value_tensor));
105 *end_of_sequence = false;
106 return Status::OK();
107 }
108
109 protected:
110 Status SaveInternal(IteratorStateWriter* writer) override {
111 mutex_lock l(mu_);
112 TF_RETURN_IF_ERROR(writer->WriteScalar(full_name("num_random_samples"),
113 num_random_samples_));
114 return Status::OK();
115 }
116
117 Status RestoreInternal(IteratorContext* ctx,
118 IteratorStateReader* reader) override {
119 mutex_lock l(mu_);
120 TF_RETURN_IF_ERROR(reader->ReadScalar(full_name("num_random_samples"),
121 &num_random_samples_));
122 parent_generator_ =
123 random::PhiloxRandom(dataset()->seed_, dataset()->seed2_);
124 generator_ = random::SingleSampleAdapter<random::PhiloxRandom>(
125 &parent_generator_);
126 generator_.Skip(num_random_samples_);
127 return Status::OK();
128 }
129
130 private:
131 random::SingleSampleAdapter<random::PhiloxRandom>::ResultType Random()
132 EXCLUSIVE_LOCKS_REQUIRED(mu_) {
133 num_random_samples_++;
134 auto out = generator_();
135 return out;
136 }
137 mutex mu_;
138 random::PhiloxRandom parent_generator_ GUARDED_BY(mu_);
139 random::SingleSampleAdapter<random::PhiloxRandom> generator_
140 GUARDED_BY(mu_);
141 int64 num_random_samples_ GUARDED_BY(mu_) = 0;
142 };
143
144 const int64 seed_;
145 const int64 seed2_;
146 };
147};
148
149REGISTER_KERNEL_BUILDER(Name("RandomDataset").Device(DEVICE_CPU),
150 RandomDatasetOp);
