42class CallFrameBase : public CallFrameInterface {
43 public:
44 explicit CallFrameBase(DataTypeSlice ret_types)
45 : ret_types_(ret_types), retvals_(ret_types.size()) {}
46
47 // Caller methods.
48 Status ConsumeRetvals(std::vector<Tensor>* retvals) {
49 retvals->reserve(retvals_.size());
50 int i = 0;
51 for (auto&& val : retvals_) {
52 if (!val) {
53 return errors::Internal("No return value for index ", i, ".");
54 }
55 retvals->emplace_back(std::move(val.value()));
56 ++i;
57 }
58 return Status::OK();
59 }
60
61 size_t num_retvals() const override { return retvals_.size(); }
62
63 // Callee methods.
64 Status SetRetval(int index, const Tensor& val) override {
65 if (index < retvals_.size() && val.dtype() == ret_types_[index] &&
66 !retvals_[index]) {
67 retvals_[index] = val;
68 return Status::OK();
69 } else if (index >= retvals_.size()) {
70 return errors::InvalidArgument("Return value ", index,
71 " is out of range.");
72 } else if (val.dtype() != ret_types_[index]) {
73 return errors::InvalidArgument("Expected type ",
74 DataTypeString(ret_types_[index]),
75 " for return value ", index, " but got ",
76 DataTypeString(val.dtype()), ".");
77 } else {
78 return errors::Internal("Attempted to set return value ", index,
79 " more than once.");
80 }
81 }
82
83 private:
84 DataTypeSlice ret_types_;
85 std::vector<gtl::optional<Tensor>> retvals_;
86 TF_DISALLOW_COPY_AND_ASSIGN(CallFrameBase);
87};

89class OwnedArgsCallFrame : public CallFrameBase {
90 public:
91 OwnedArgsCallFrame(std::vector<Tensor>&& args,
92 const std::vector<Tensor>* captured_inputs,
93 DataTypeSlice ret_types)
94 : CallFrameBase(ret_types),
95 args_(std::move(args)),
96 captured_inputs_(captured_inputs) {}
97
98 size_t num_args() const override {
99 return args_.size() + captured_inputs_->size();
100 }
101
102 // Callee methods.
103 Status GetArg(int index, Tensor* val) const override {
104 if (index < args_.size() && args_[index].IsInitialized()) {
105 // TODO(mrry): Consider making `CallFrameInterface::GetArg` non-const in
106 // order to be able to `std::move(args_[index])` into `*val`.
107 *val = args_[index];
108 return Status::OK();
109 } else if (index < args_.size() + captured_inputs_->size()) {
110 *val = (*captured_inputs_)[index - args_.size()];
111 return Status::OK();
112 } else if (index >= args_.size() + captured_inputs_->size()) {
113 return errors::InvalidArgument("Argument ", index, " is out of range.");
114 } else {
115 return errors::Internal("Attempted to get argument ", index,
116 " more than once.");
117 }
118 }
119
120 private:
121 std::vector<Tensor> args_;
122 const std::vector<Tensor>* const captured_inputs_; // Not owned.
123};

161Status CapturedFunction::MaybeInstantiate(
162 IteratorContext* ctx, FunctionLibraryRuntime::Handle* out_handle) {
163 mutex_lock l(mu_);
164 if (lib_ == nullptr) {
165 // The context's runtime will be used for all subsequent calls.
166 lib_ = ctx->lib();
167 DCHECK(f_handle_ == kInvalidHandle);
168 FunctionLibraryRuntime::InstantiateOptions inst_opts;
169 inst_opts.overlay_lib = ctx->function_library().get();
170 inst_opts.state_handle = std::to_string(random::New64());
171 TF_RETURN_IF_ERROR(lib_->Instantiate(func_.name(), AttrSlice(&func_.attr()),
172 inst_opts, &f_handle_));
173 const FunctionBody* fbody = lib_->GetFunctionBody(f_handle_);
174 if (fbody == nullptr) {
175 return errors::Internal("Failed to instantiate function body.");
176 }
177 ret_types_ = fbody->ret_types;
178 } else {
179 // TODO(mrry): Consider moving this under a shared lock, as it is
180 // the common case.
181 if (ctx->lib() != lib_) {
182 return errors::Internal(
183 "Captured function was called with a different "
184 "FunctionLibraryRuntime*, which is not permitted.");
185 }
186 }
187 *out_handle = f_handle_;
188 return Status::OK();
189}

191Status CapturedFunction::Run(IteratorContext* ctx, std::vector<Tensor>&& args,
192 std::vector<Tensor>* rets) {
193 FunctionLibraryRuntime::Handle handle;
194 TF_RETURN_IF_ERROR(MaybeInstantiate(ctx, &handle));
195
196 FunctionLibraryRuntime::Options f_opts;
197 f_opts.step_id = CapturedFunction::generate_step_id();
198 ScopedStepContainer step_container(f_opts.step_id, [ctx](const string& name) {
199 ctx->lib()->device()->resource_manager()->Cleanup(name).IgnoreError();
200 });
201 f_opts.step_container = &step_container;
202 f_opts.runner = ctx->runner();
203 // TODO(mrry): Add cancellation manager support to IteratorContext
204 // so that we can cancel running map functions. The local
205 // cancellation manager here is created so that we can run kernels
206 // (such as queue kernels) that depend on the non-nullness of
207 // `OpKernelContext::cancellation_manager()`, but additional effort
208 // will be required to plumb it through the `IteratorContext`.
209 CancellationManager c_mgr;
210 f_opts.cancellation_manager = &c_mgr;
211
212 OwnedArgsCallFrame frame(std::move(args), &captured_inputs_, ret_types_);
213 Notification n;
214 Status s;
215 ctx->lib()->Run(f_opts, handle, &frame, [&n, &s](Status func_status) {
216 s.Update(func_status);
217 n.Notify();
218 });
219 n.WaitForNotification();
220 TF_RETURN_IF_ERROR(s);
221 return frame.ConsumeRetvals(rets);
222}

259Status CapturedFunction::Instantiate(IteratorContext* ctx) {
260 FunctionLibraryRuntime::Handle unused_handle;
261 TF_RETURN_IF_ERROR(MaybeInstantiate(ctx, &unused_handle));
262 mutex_lock l(mu_);
263 if (captured_runner_ == nullptr) {
264 captured_runner_ = *ctx->runner();
265 }
266 return Status::OK();
267}

269Status CapturedFunction::RunInstantiated(const std::vector<Tensor>& args,
270 std::vector<Tensor>* rets) {
271 FunctionLibraryRuntime* lib;
272 FunctionLibraryRuntime::Handle handle;
273 std::function<void(std::function<void()>)>* runner;
274 {
275 tf_shared_lock l(mu_);
276 if (lib_ == nullptr) {
277 return errors::FailedPrecondition(
278 "`CapturedFunction::Instantiate()` must be called before a call to "
279 "`CapturedFunction::RunInstantiated()`.");
280 }
281 lib = lib_;
282 handle = f_handle_;
283 runner = &captured_runner_;
284 }
285
286 FunctionLibraryRuntime::Options f_opts;
287 f_opts.step_id = CapturedFunction::generate_step_id();
288 ScopedStepContainer step_container(f_opts.step_id, [lib](const string& name) {
289 lib->device()->resource_manager()->Cleanup(name).IgnoreError();
290 });
291 f_opts.step_container = &step_container;
292 f_opts.runner = runner;
293 // TODO(mrry): Add cancellation manager support to IteratorContext
294 // so that we can cancel running map functions. The local
295 // cancellation manager here is created so that we can run kernels
296 // (such as queue kernels) that depend on the non-nullness of
297 // `OpKernelContext::cancellation_manager()`, but additional effort
298 // will be required to plumb it through the `IteratorContext`.
299 CancellationManager c_mgr;
300 f_opts.cancellation_manager = &c_mgr;
301
302 BorrowedArgsCallFrame frame(args, &captured_inputs_, ret_types_);
303 Notification n;
304 Status s;
305
306 lib->Run(f_opts, handle, &frame, [&n, &s](Status func_status) {
307 s.Update(func_status);
308 n.Notify();
309 });
310 n.WaitForNotification();
311 TF_RETURN_IF_ERROR(s);
312 return frame.ConsumeRetvals(rets);
313}

315void CapturedFunction::RunAsync(IteratorContext* ctx,
316 std::vector<Tensor>&& args,
317 std::vector<Tensor>* rets,
318 FunctionLibraryRuntime::DoneCallback done) {
319 // NOTE(mrry): This method does not transfer ownership of `ctx`, and it may
320 // be deleted before `done` is called. Take care not to capture `ctx` in any
321 // code that may execute asynchronously in this function.
322 FunctionLibraryRuntime::Handle handle;
323 Status s = MaybeInstantiate(ctx, &handle);
324 if (!s.ok()) {
325 done(s);
326 return;
327 }
328 auto frame =
329 new OwnedArgsCallFrame(std::move(args), &captured_inputs_, ret_types_);
330
331 FunctionLibraryRuntime::Options f_opts;
332 f_opts.step_id = CapturedFunction::generate_step_id();
333 ResourceMgr* resource_mgr = ctx->lib()->device()->resource_manager();
334 auto step_container = new ScopedStepContainer(
335 f_opts.step_id, [resource_mgr](const string& name) {
336 resource_mgr->Cleanup(name).IgnoreError();
337 });
338 f_opts.step_container = step_container;
339 f_opts.runner = ctx->runner();
340 // TODO(mrry): Add cancellation manager support to IteratorContext
341 // so that we can cancel running map functions. The local
342 // cancellation manager here is created so that we can run kernels
343 // (such as queue kernels) that depend on the non-nullness of
344 // `OpKernelContext::cancellation_manager()`, but additional effort
345 // will be required to plumb it through the `IteratorContext`.
346 auto c_mgr = new CancellationManager;
347 f_opts.cancellation_manager = c_mgr;
348
349 tf_shared_lock l(mu_);
350 ctx->lib()->Run(f_opts, handle, frame,
351 std::bind(
352 [rets, step_container, c_mgr, frame](
353 FunctionLibraryRuntime::DoneCallback done,
354 // Begin unbound arguments.
355 Status s) {
356 delete step_container;
357 delete c_mgr;
358 if (s.ok()) {
359 s = frame->ConsumeRetvals(rets);
360 }
361 delete frame;
362 done(s);
363 },
364 std::move(done), std::placeholders::_1));
365}
