28class StatsAggregatorImpl : public StatsAggregator {
29 public:
30 StatsAggregatorImpl() {}
31
32 void AddToHistogram(const string& name,
33 gtl::ArraySlice<double> values) override {
34 mutex_lock l(mu_);
35 histogram::Histogram& histogram = histograms_[name];
36 for (double value : values) {
37 histogram.Add(value);
38 }
39 }
40
41 void EncodeToProto(Summary* out_summary) override {
42 mutex_lock l(mu_);
43 for (const auto& pair : histograms_) {
44 const string& name = pair.first;
45 const histogram::Histogram& histogram = pair.second;
46
47 Summary::Value* value = out_summary->add_value();
48 value->set_tag(name);
49 histogram.EncodeToProto(value->mutable_histo(),
50 false /* doesn't preserve zero buckets */);
51 }
52 }
53
54 private:
55 mutex mu_;
56 std::unordered_map<string, histogram::Histogram> histograms_ GUARDED_BY(mu_);
57 TF_DISALLOW_COPY_AND_ASSIGN(StatsAggregatorImpl);
58};
59
60class StatsAggregatorHandleOp
61 : public ResourceOpKernel<StatsAggregatorResource> {
62 public:
63 explicit StatsAggregatorHandleOp(OpKernelConstruction* ctx)
64 : ResourceOpKernel<StatsAggregatorResource>(ctx) {}
65
66 private:
67 Status CreateResource(StatsAggregatorResource** ret) override
68 EXCLUSIVE_LOCKS_REQUIRED(mu_) {
69 *ret = new StatsAggregatorResource(
70 std::unique_ptr<StatsAggregator>(new StatsAggregatorImpl));
71 return Status::OK();
72 }
73
74 Status VerifyResource(StatsAggregatorResource* resource) override {
75 return Status::OK();
76 }
77};
78
79class StatsAggregatorSummaryOp : public OpKernel {
80 public:
81 explicit StatsAggregatorSummaryOp(OpKernelConstruction* ctx)
82 : OpKernel(ctx) {}
83
84 void Compute(OpKernelContext* ctx) override {
85 const Tensor& resource_handle_t = ctx->input(0);
86 OP_REQUIRES(ctx, TensorShapeUtils::IsScalar(resource_handle_t.shape()),
87 errors::InvalidArgument("resource_handle must be a scalar"));
88
89 StatsAggregatorResource* resource;
90 OP_REQUIRES_OK(ctx,
91 LookupResource(ctx, HandleFromInput(ctx, 0), &resource));
92 core::ScopedUnref unref_iterator(resource);
93
94 Tensor* summary_t;
95 OP_REQUIRES_OK(ctx, ctx->allocate_output(0, TensorShape({}), &summary_t));
96 Summary summary;
97 resource->stats_aggregator()->EncodeToProto(&summary);
98 summary_t->scalar<string>()() = summary.SerializeAsString();
99 }
100};

102REGISTER_KERNEL_BUILDER(Name("StatsAggregatorHandle").Device(DEVICE_CPU),
103 StatsAggregatorHandleOp);
104REGISTER_KERNEL_BUILDER(Name("StatsAggregatorSummary").Device(DEVICE_CPU),
105 StatsAggregatorSummaryOp);
