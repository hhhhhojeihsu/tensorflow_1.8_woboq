88 template <> \
89 void Relu6<GPUDevice, T>::operator()( \

91 typename TTypes<T>::Tensor activations); \

104 typename TTypes<T>::Tensor activations); \

107 template <> \
108 void EluGrad<GPUDevice, T>::operator()( \

121 void SeluGrad<GPUDevice, T>::operator()( \

125 extern template struct SeluGrad<GPUDevice, T>;
