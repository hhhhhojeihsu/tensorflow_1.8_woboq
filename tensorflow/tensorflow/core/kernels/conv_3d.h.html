16// Functors for 3d convolution.
17
18#ifndef TENSORFLOW_KERNELS_CONV_3D_H_
19#define TENSORFLOW_KERNELS_CONV_3D_H_
20
21#include "tensorflow/core/framework/tensor_types.h"
22#include "tensorflow/core/kernels/eigen_cuboid_convolution.h"
23
24namespace tensorflow {
25namespace functor {
26
27// Applies a 3D convolution to a batch of multi-channel volumes.
28template <typename Device, typename T>
29struct CuboidConvolution;
30
31typedef Eigen::ThreadPoolDevice CPUDevice;
32
33template <typename T>
34struct CuboidConvolution<CPUDevice, T> {
35 void operator()(const CPUDevice& d, typename TTypes<T, 5>::Tensor output,
36 typename TTypes<T, 5>::ConstTensor input,
37 typename TTypes<T, 5>::ConstTensor filter, int stride_planes,
38 int stride_rows, int stride_cols,
39 const Eigen::PaddingType& padding) {
40 output.device(d) = Eigen::CuboidConvolution(
41 input, filter, stride_planes, stride_rows, stride_cols, padding);
42 }
43};
44
45} // namespace functor
46} // namespace tensorflow
47
48#endif // TENSORFLOW_KERNELS_CONV_3D_H_
