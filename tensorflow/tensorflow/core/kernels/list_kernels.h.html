37// Variant compatible type for a list of tensors. This is mutable but instances
38// should never be mutated after stored in a variant tensor.
39struct TensorList {
40 public:
41 TensorList() {}
42 TensorList(const TensorList& other);
43
44 static const char kTypeName[];
45 string TypeName() const { return kTypeName; }
46
47 void Encode(VariantTensorData* data) const;
48
49 bool Decode(const VariantTensorData& data);
50
51 // TODO(apassos) fill this out
52 string DebugString() const { return "TensorList"; }
53
54 std::vector<Tensor> tensors;
55 PartialTensorShape element_shape;
56 DataType element_dtype;
57};

61template <typename Device, typename T>
62class TensorListStack : public OpKernel {
63 public:
64 typedef std::vector<std::unique_ptr<typename TTypes<T, 2>::ConstMatrix>>
65 ConstMatrixVector;
66 explicit TensorListStack(OpKernelConstruction* c) : OpKernel(c) {
67 OP_REQUIRES_OK(c, c->GetAttr("element_dtype", &element_dtype_));
68 OP_REQUIRES_OK(c, c->GetAttr("num_elements", &num_elements_));
69 }
70
71 ~TensorListStack() {}
72
73 void Compute(OpKernelContext* c) override {
74 const TensorList* l = c->input(0).scalar<Variant>()().get<TensorList>();
75 OP_REQUIRES(c, l != nullptr,
76 errors::InvalidArgument(
77 "Input handle is not a list. Saw: '",
78 c->input(0).scalar<Variant>()().DebugString(), "'"));
79 OP_REQUIRES(c, element_dtype_ == l->element_dtype,
80 errors::InvalidArgument("Invalid data types; op elements ",
81 DataTypeString(element_dtype_),
82 " but list elements ",
83 DataTypeString(l->element_dtype)));
84 OP_REQUIRES(c, l->element_shape.IsFullyDefined(),
85 errors::InvalidArgument("Tried to stack elements from a list "
86 "with non-fully-defined shape: ",
87 l->element_shape.DebugString()));
88 if (num_elements_ != -1) {
89 OP_REQUIRES(c, l->tensors.size() == num_elements_,
90 errors::InvalidArgument("Operation expected a list with ",
91 num_elements_,
92 " elements but got a list with ",
93 l->tensors.size(), " elements."));
94 }
95 TensorShape resulting_shape;
96 resulting_shape.AddDim(l->tensors.size());
97 for (TensorShapeDim s : l->element_shape) {
98 resulting_shape.AddDim(s.size);
99 }
100 Tensor* output;
101 OP_REQUIRES_OK(c, c->allocate_output(0, resulting_shape, &output));
102 if (output->NumElements() == 0) {
103 return;
104 }
105
106 ConstMatrixVector inputs_flat;
107 inputs_flat.reserve(l->tensors.size());
108 for (const auto& t : l->tensors) {
109 OP_REQUIRES(
110 c, l->element_shape.IsCompatibleWith(t.shape()),
111 errors::InvalidArgument(
112 "Tensor with invalid shape in list. List element shape shape: ",
113 l->element_shape.DebugString(),
114 " and tensor shape: ", t.shape().DebugString()));
115 inputs_flat.emplace_back(new typename TTypes<T, 2>::ConstMatrix(
116 t.shaped<T, 2>({1, t.NumElements()})));
117 }
118 auto output_flat = output->shaped<T, 2>({1, output->NumElements()});
119
120#if GOOGLE_CUDA
121 if (std::is_same<Device, Eigen::GpuDevice>::value) {
122 ConcatGPU<T>(c, inputs_flat, output, &output_flat);
123 return;
124 }
125#endif // GOOGLE_CUDA
126 ConcatCPU<T>(c->device(), inputs_flat, &output_flat);
127 }
128
129 private:
130 int num_elements_;
131 DataType element_dtype_;
132};

134template <typename Device, typename T>
135class TensorListFromTensor : public OpKernel {
136 public:
137 TensorListFromTensor(OpKernelConstruction* c) : OpKernel(c) {}
138
139 void Compute(OpKernelContext* c) override {
140 Tensor* output_tensor;
141 AllocatorAttributes attr;
142 attr.set_on_host(true);
143 OP_REQUIRES_OK(c, c->allocate_output(0, {}, &output_tensor, attr));
144 PartialTensorShape element_shape;
145 OP_REQUIRES_OK(c, TensorShapeFromTensor(c->input(1), &element_shape));
146 TensorList output_list;
147 const Tensor& t = c->input(0);
148 output_list.element_dtype = t.dtype();
149 TensorShape output_shape(t.shape());
150 output_shape.RemoveDim(0);
151 OP_REQUIRES(c, element_shape.IsCompatibleWith(output_shape),
152 errors::InvalidArgument(
153 "Specified a list with shape ", element_shape.DebugString(),
154 " from a tensor with shape ", output_shape.DebugString()));
155 output_list.element_shape = element_shape;
156 output_list.tensors.reserve(t.shape().dim_size(0));
157 for (int i = 0; i < t.shape().dim_size(0); ++i) {
158 Tensor tmp = t.Slice(i, i + 1);
159 TensorShape tmp_shape = tmp.shape();
160 tmp_shape.RemoveDim(0);
161 OP_REQUIRES(c, tmp.CopyFrom(tmp, tmp_shape),
162 errors::Unknown("Unexpected shape error."));
163 // TODO(apassos) maybe not always align; but weird compiler bugs seem to
164 // prevent this.
165 Tensor aligned;
166 OP_REQUIRES_OK(c, c->allocate_temp(tmp.dtype(), tmp.shape(), &aligned));
167 aligned.flat<T>().device(c->eigen_device<Device>()) =
168 tmp.unaligned_flat<T>();
169 output_list.tensors.push_back(aligned);
170 }
171 output_tensor->scalar<Variant>()() = std::move(output_list);
172 }
173};

175template <typename Device>
176Status TensorListBinaryAdd(OpKernelContext* c, const TensorList& a,
177 const TensorList& b, TensorList* out) {
178 if (a.element_dtype != b.element_dtype) {
179 return errors::InvalidArgument(
180 "Trying to add two lists of tensors of different dtypes. One is ",
181 DataTypeString(a.element_dtype), " and the other is ",
182 DataTypeString(b.element_dtype));
183 }
184 out->element_dtype = a.element_dtype;
185 if (!a.element_shape.IsCompatibleWith(b.element_shape)) {
186 return errors::InvalidArgument(
187 "Trying to add two lists of tensors with incompatible element shapes. "
188 "One is ",
189 a.element_shape.DebugString(), " and the other is ",
190 b.element_shape.DebugString());
191 }
192
193 TF_RETURN_IF_ERROR(
194 a.element_shape.MergeWith(b.element_shape, &out->element_shape));
195 if (a.tensors.size() != b.tensors.size()) {
196 return errors::InvalidArgument(
197 "Trying to add two lists of tensors with different lengths. One is ",
198 a.tensors.size(), " and the other is ", b.tensors.size());
199 }
200 out->tensors.reserve(a.tensors.size());
201 for (int i = 0; i < a.tensors.size(); ++i) {
202 const Tensor& a_tensor = a.tensors[i];
203 const Tensor& b_tensor = b.tensors[i];
204 if (a_tensor.dtype() == DT_INVALID) {
205 out->tensors.push_back(b_tensor);
206 continue;
207 }
208 if (b_tensor.dtype() == DT_INVALID) {
209 out->tensors.push_back(a_tensor);
210 continue;
211 }
212 if (a_tensor.shape() != b_tensor.shape()) {
213 // TODO(apassos) support broadcasting additions here?
214 return errors::InvalidArgument(
215 "Trying to add two tensors with incompatible element shapes. "
216 "One is ",
217 a_tensor.shape().DebugString(), " and the other is ",
218 b_tensor.shape().DebugString(), " in position ", i);
219 }
220 Tensor out_tensor;
221 TF_RETURN_IF_ERROR(
222 c->allocate_temp(a_tensor.dtype(), a_tensor.shape(), &out_tensor));
223 out->tensors.push_back(out_tensor);
224 switch (out_tensor.dtype()) {
225#define DTYPE_CASE(dtype) \
226 case DataTypeToEnum<dtype>::value: \
227 out_tensor.flat<dtype>().device(c->eigen_device<Device>()) = \
228 a_tensor.flat<dtype>() + b_tensor.flat<dtype>(); \
229 break;
230
231 TF_CALL_NUMBER_TYPES(DTYPE_CASE)
232
233#undef DTYPE_CASE
234 default:
235 return errors::InvalidArgument("Trying to add unsupported dtype ",
236 out_tensor.dtype());
237 }
238 }
239 return Status::OK();
240}

242template <typename Device>
243Status TensorListZerosLike(OpKernelContext* c, const TensorList& x,
244 TensorList* y) {
245 y->element_dtype = x.element_dtype;
246 y->element_shape = x.element_shape;
247 y->tensors.reserve(x.tensors.size());
248 for (const Tensor& t : x.tensors) {
249 Tensor out_tensor;
250 TF_RETURN_IF_ERROR(c->allocate_temp(t.dtype(), t.shape(), &out_tensor));
251 switch (out_tensor.dtype()) {
252#define DTYPE_CASE(dtype) \
253 case DataTypeToEnum<dtype>::value: \
254 out_tensor.flat<dtype>().device(c->eigen_device<Device>()) = \
255 out_tensor.flat<dtype>().constant(dtype(0)); \
256 break;
257
258 TF_CALL_NUMBER_TYPES(DTYPE_CASE)
259
260#undef DTYPE_CASE
261 default:
262 return errors::InvalidArgument(
263 "Trying to compute zeros_like for unsupported dtype",
264 out_tensor.dtype());
265 }
266 }
267 return Status::OK();
268}
