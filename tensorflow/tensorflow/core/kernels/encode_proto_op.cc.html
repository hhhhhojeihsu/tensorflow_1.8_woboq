45// Computes the total serialized size for a packed repeated field.
46// For fixed-size types this can just multiply, but for variable-sized
47// types it has to iterate through the values in the tensor.
48template <WireFormatLite::FieldType FieldType, typename TensorT>
49size_t TotalPackedSize(const Tensor& input, int message_index, int size);
50
51template <>
52size_t TotalPackedSize<WireFormatLite::TYPE_DOUBLE, double>(const Tensor& input,
53 int message_index,
54 int size) {
55 return size * WireFormatLite::kDoubleSize;
56}

60 int message_index,
61 int size) {

67 int message_index,

72template <>
73size_t TotalPackedSize<WireFormatLite::TYPE_INT64, int64>(const Tensor& input,
74 int message_index,
75 int size) {
76 size_t data_size = 0;
77 auto input_t = input.flat_inner_dims<int64>();
78 for (int64 i = 0; i < size; i++) {
79 data_size += WireFormatLite::Int64Size(
80 input_t(static_cast<int64>(message_index), i));
81 }
82 return data_size;

216// Writes a possibly repeated primitive field.
217// TensorFlow does not have unsigned types, so we decode them to signed and
218// encode them back to unsigned.
219template <typename TensorT, typename ProtoT,
220 WireFormatLite::FieldType FieldType,
221 void Writer(ProtoT, CodedOutputStream*)>
222void WriteField(const FieldDescriptor& field_desc, const Tensor& input,
223 int message_index, int size, CodedOutputStream* output) {
224 auto wire_type = WireFormatLite::WireTypeForFieldType(
225 WireFormatLite::FieldType(field_desc.type()));
226
227 auto input_t = input.flat_inner_dims<TensorT>();
228 if (field_desc.options().packed()) {
229 // Write the tag for the packed field.
230 WireFormatLite::WriteTag(field_desc.number(),
231 WireFormatLite::WIRETYPE_LENGTH_DELIMITED, output);
232
233 // Write the total packed length.
234 size_t data_size =
235 TotalPackedSize<FieldType, TensorT>(input, message_index, size);
236 output->WriteVarint32(data_size);
237
238 // Write individual values.
239 for (int64 i = 0; i < size; i++) {
240 // Note implicit cast from signed to unsigned.
241 const ProtoT& value = input_t(static_cast<int64>(message_index), i);
242 Writer(value, output);
243 }
244 } else {
245 for (int64 i = 0; i < size; i++) {
246 WireFormatLite::WriteTag(field_desc.number(), wire_type, output);
247
248 // Note implicit cast from signed to unsigned.
249 const ProtoT& value = input_t(static_cast<int64>(message_index), i);
250 Writer(value, output);
251 }
252 }
253}

255// Writes a possibly repeated string, bytes, or message field.
256template <typename T, void Writer(int, const T&, CodedOutputStream*)>
257void WriteVarLenField(const FieldDescriptor& field_desc, const Tensor& input,
258 int message_index, int size, CodedOutputStream* output) {
259 auto input_t = input.flat_inner_dims<T>();
260 for (int64 i = 0; i < size; i++) {
261 const T& value = input_t(static_cast<int64>(message_index), i);
262 // TODO(nix): there doesn't seem to be an inlined version of
263 // WireFormatLite::WriteString or its relatives, which might allow a
264 // small speedup.
265 Writer(field_desc.number(), value, output);
266 }
267}

269// Writes a group field.
270// Groups are treated like submessages, but tag-delimited
271// instead of length-delimited. WireFormatLite handles this
272// differently so we code it ourselves.
273void WriteGroup(const FieldDescriptor& field_desc, const Tensor& input,
274 int message_index, int size, CodedOutputStream* output) {
275 auto input_t = input.flat_inner_dims<string>();
276 for (int64 i = 0; i < size; i++) {
277 const string& value = input_t(static_cast<int64>(message_index), i);
278 WireFormatLite::WriteTag(field_desc.number(),
279 WireFormatLite::WIRETYPE_START_GROUP, output);
280 // Note the use of WriteRaw instead of WriteString to skip the length.
281 output->WriteRaw(value.data(), value.size());
282 WireFormatLite::WriteTag(field_desc.number(),
283 WireFormatLite::WIRETYPE_END_GROUP, output);
284 }
285}

287// Writes a (possibly repeated) field into an output stream.
288// It is the caller's responsibility to ensure that the type of
289// the input tensor is compatible with the type of the proto
290// field descriptor, and that (message_index, size-1) is within
291// bounds.
292void WriteField(const FieldDescriptor& field_desc, const Tensor& input,
293 int message_index, int size, CodedOutputStream* output) {
294 DataType tf_type = input.dtype();
295
296 switch (field_desc.type()) {
297 case WireFormatLite::TYPE_DOUBLE:
298 return WriteField<double, double, WireFormatLite::TYPE_DOUBLE,
299 WireFormatLite::WriteDoubleNoTag>(
300 field_desc, input, message_index, size, output);
301 case WireFormatLite::TYPE_FLOAT:
302 switch (tf_type) {
303 case DataType::DT_FLOAT:
304 return WriteField<float, float, WireFormatLite::TYPE_FLOAT,
305 WireFormatLite::WriteFloatNoTag>(
306 field_desc, input, message_index, size, output);
307 case DataType::DT_DOUBLE:
308 return WriteField<double, float, WireFormatLite::TYPE_FLOAT,
309 WireFormatLite::WriteFloatNoTag>(
310 field_desc, input, message_index, size, output);
311 default:
312 return;
313 }
314 case WireFormatLite::TYPE_INT64:
315 return WriteField<int64, protobuf_int64, WireFormatLite::TYPE_INT64,
316 WireFormatLite::WriteInt64NoTag>(
317 field_desc, input, message_index, size, output);
318 case WireFormatLite::TYPE_UINT64:
319 return WriteField<int64, protobuf_uint64, WireFormatLite::TYPE_UINT64,
320 WireFormatLite::WriteUInt64NoTag>(
321 field_desc, input, message_index, size, output);
322 case WireFormatLite::TYPE_INT32:
323 return WriteField<int32, int32, WireFormatLite::TYPE_INT32,
324 WireFormatLite::WriteInt32NoTag>(
325 field_desc, input, message_index, size, output);
326 case WireFormatLite::TYPE_FIXED64:
327 return WriteField<int64, protobuf_uint64, WireFormatLite::TYPE_FIXED64,
328 WireFormatLite::WriteFixed64NoTag>(
329 field_desc, input, message_index, size, output);
330 case WireFormatLite::TYPE_FIXED32:
331 switch (tf_type) {
332 case DataType::DT_INT64:
333 return WriteField<int64, uint32, WireFormatLite::TYPE_FIXED32,
334 WireFormatLite::WriteFixed32NoTag>(
335 field_desc, input, message_index, size, output);
336 case DataType::DT_INT32:
337 return WriteField<int32, uint32, WireFormatLite::TYPE_FIXED32,
338 WireFormatLite::WriteFixed32NoTag>(
339 field_desc, input, message_index, size, output);
340 default:
341 return;
342 }
343 case WireFormatLite::TYPE_BOOL:
344 return WriteField<bool, bool, WireFormatLite::TYPE_BOOL,
345 WireFormatLite::WriteBoolNoTag>(
346 field_desc, input, message_index, size, output);
347 case WireFormatLite::TYPE_STRING:
348 return WriteVarLenField<string, WireFormatLite::WriteString>(
349 field_desc, input, message_index, size, output);
350 case WireFormatLite::TYPE_GROUP:
351 return WriteGroup(field_desc, input, message_index, size, output);
352 case WireFormatLite::TYPE_MESSAGE:
353 return WriteVarLenField<string, WireFormatLite::WriteBytes>(
354 field_desc, input, message_index, size, output);
355 case WireFormatLite::TYPE_BYTES:
356 return WriteVarLenField<string, WireFormatLite::WriteBytes>(
357 field_desc, input, message_index, size, output);
358 case WireFormatLite::TYPE_UINT32:
359 switch (tf_type) {
360 case DataType::DT_INT64:
361 return WriteField<int64, uint32, WireFormatLite::TYPE_UINT32,
362 WireFormatLite::WriteUInt32NoTag>(
363 field_desc, input, message_index, size, output);
364 case DataType::DT_INT32:
365 return WriteField<int32, uint32, WireFormatLite::TYPE_UINT32,
366 WireFormatLite::WriteUInt32NoTag>(
367 field_desc, input, message_index, size, output);
368 default:
369 return;
370 }
371 case WireFormatLite::TYPE_ENUM:
372 return WriteField<int32, int32, WireFormatLite::TYPE_ENUM,
373 WireFormatLite::WriteEnumNoTag>(
374 field_desc, input, message_index, size, output);
375 case WireFormatLite::TYPE_SFIXED32:
376 return WriteField<int32, int32, WireFormatLite::TYPE_SFIXED32,
377 WireFormatLite::WriteSFixed32NoTag>(
378 field_desc, input, message_index, size, output);
379 case WireFormatLite::TYPE_SFIXED64:
380 return WriteField<int64, protobuf_int64, WireFormatLite::TYPE_SFIXED64,
381 WireFormatLite::WriteSFixed64NoTag>(
382 field_desc, input, message_index, size, output);
383 case WireFormatLite::TYPE_SINT32:
384 return WriteField<int32, int32, WireFormatLite::TYPE_SINT32,
385 WireFormatLite::WriteSInt32NoTag>(
386 field_desc, input, message_index, size, output);
387 case WireFormatLite::TYPE_SINT64:
388 return WriteField<int64, protobuf_int64, WireFormatLite::TYPE_SINT64,
389 WireFormatLite::WriteSInt64NoTag>(
390 field_desc, input, message_index, size, output);
391 // default: intentionally omitted in order to enable static checking.
392 }
393}

395// Checks that a Protobuf field is compatible with a TensorFlow datatype.
396// This is separated from WriteField to lift it out of the inner loop.
397bool IsCompatibleType(const FieldDescriptor& field_desc, DataType tf_type) {
398 switch (field_desc.type()) {
399 case WireFormatLite::TYPE_DOUBLE:
400 return tf_type == DataType::DT_DOUBLE;
401 case WireFormatLite::TYPE_FLOAT:
402 return tf_type == DataType::DT_FLOAT || tf_type == DataType::DT_DOUBLE;
403 case WireFormatLite::TYPE_INT64:
404 case WireFormatLite::TYPE_SFIXED64:
405 case WireFormatLite::TYPE_SINT64:
406 return tf_type == DataType::DT_INT64;
407 case WireFormatLite::TYPE_UINT64:
408 return tf_type == DataType::DT_INT64;
409 case WireFormatLite::TYPE_INT32:
410 case WireFormatLite::TYPE_ENUM:
411 case WireFormatLite::TYPE_SFIXED32:
412 case WireFormatLite::TYPE_SINT32:
413 return tf_type == DataType::DT_INT32;
414 case WireFormatLite::TYPE_FIXED64:
415 return tf_type == DataType::DT_INT64;
416 case WireFormatLite::TYPE_FIXED32:
417 case WireFormatLite::TYPE_UINT32:
418 return tf_type == DataType::DT_INT64 || tf_type == DataType::DT_INT32;
419 case WireFormatLite::TYPE_BOOL:
420 return tf_type == DataType::DT_BOOL;
421 case WireFormatLite::TYPE_STRING:
422 case WireFormatLite::TYPE_GROUP:
423 case WireFormatLite::TYPE_MESSAGE:
424 case WireFormatLite::TYPE_BYTES:
425 return tf_type == DataType::DT_STRING;
426 // default: intentionally omitted in order to enable static checking.
427 }
428 return false;
429}

431class EncodeProtoOp : public OpKernel {
432 public:
433 explicit EncodeProtoOp(OpKernelConstruction* context) : OpKernel(context) {
434 string descriptor_source;
435 OP_REQUIRES_OK(context,
436 context->GetAttr("descriptor_source", &descriptor_source));
437 // We always get back a desc_pool, but we may not own it. If we own it,
438 // owned_desc_pool_ will be filled in.
439 DescriptorPool const* desc_pool;
440 OP_REQUIRES_OK(context, GetDescriptorPool(context->env(), descriptor_source,
441 &desc_pool, &owned_desc_pool_));
442
443 string message_type;
444 OP_REQUIRES_OK(context, context->GetAttr("message_type", &message_type));
445 const Descriptor* message_desc =
446 desc_pool->FindMessageTypeByName(message_type);
447 OP_REQUIRES(context, message_desc != nullptr,
448 errors::InvalidArgument("No descriptor found for message type ",
449 message_type));
450
451 OP_REQUIRES_OK(context, context->GetAttr("field_names", &field_names_));
452
453 // Gather the field descriptors for the given field_names.
454 field_descs_.resize(field_names_.size());
455 for (int i = 0; i < field_names_.size(); i++) {
456 const string& name = field_names_[i];
457 auto field_desc = message_desc->FindFieldByName(name);
458 OP_REQUIRES(context, field_desc != nullptr,
459 errors::InvalidArgument("Unknown field: ", name,
460 " in message type ", message_type));
461
462 field_descs_[i] = field_desc;
463 }
464
465 // Build a list of indices into field_descs sorted by increasing
466 // field_number. This will be used to output fields in sorted order,
467 // which is strongly encouraged when serializing protobufs.
468 sorted_field_index_.resize(field_names_.size());
469 // Start with the fields sorted by current index.
470 for (int i = 0; i < field_names_.size(); i++) sorted_field_index_[i] = i;
471 // Then sort the field indices by their proto field number.
472 std::sort(sorted_field_index_.begin(), sorted_field_index_.end(),
473 [this](int a, int b) -> bool {
474 return field_descs_[a]->number() < field_descs_[b]->number();
475 });
476 }
477
478 void Compute(OpKernelContext* cx) override {
479 const Tensor* sizes_tensor;
480 OP_REQUIRES_OK(cx, cx->input("sizes", &sizes_tensor));
481
482 OpInputList values;
483 OP_REQUIRES_OK(cx, cx->input_list("values", &values));
484
485 OP_REQUIRES(cx, field_descs_.size() == values.size(),
486 errors::InvalidArgument(
487 "Length of inputs list must match field_names"));
488
489 // Check the arguments for consistency.
490 TensorShape common_prefix;
491 int message_count;
492 for (int i = 0; i < field_descs_.size(); i++) {
493 const Tensor& v = values[i];
494
495 // The type of each value tensor must match the corresponding field.
496 OP_REQUIRES(cx, IsCompatibleType(*field_descs_[i], v.dtype()),
497 errors::InvalidArgument(
498 "Incompatible type for field " + field_names_[i] +
499 ". Saw dtype: ",
500 DataTypeString(v.dtype()),
501 " but field type is: ", field_descs_[i]->type_name()));
502
503 // All value tensors must have the same shape prefix (i.e. batch size).
504 TensorShape shape_prefix = v.shape();
505 shape_prefix.RemoveDim(shape_prefix.dims() - 1);
506
507 // Do some initialization on the first input value. The rest will
508 // have to match this one.
509 if (i == 0) {
510 OP_REQUIRES(cx, v.dims() >= 1,
511 errors::InvalidArgument(
512 "Expected value to be at least a vector, saw shape: ",
513 v.shape().DebugString()));
514 common_prefix = shape_prefix;
515 message_count = common_prefix.num_elements();
516 } else {
517 OP_REQUIRES(cx, shape_prefix == common_prefix,
518 errors::InvalidArgument(
519 "Values must match up to the last dimension"));
520 }
521 }
522
523 TensorShape expected_sizes_shape = common_prefix;
524 expected_sizes_shape.AddDim(field_descs_.size());
525
526 OP_REQUIRES(cx, sizes_tensor->shape() == expected_sizes_shape,
527 errors::InvalidArgument(
528 "sizes should be batch_size + [len(field_names)]. Saw: ",
529 sizes_tensor->shape().DebugString(),
530 " but expected: ", expected_sizes_shape.DebugString()));
531
532 auto sizes = sizes_tensor->flat_inner_dims<int32>();
533
534 for (int i = 0; i < field_descs_.size(); ++i) {
535 const Tensor& v = values[i];
536 int max_size = v.dim_size(v.dims() - 1);
537
538 // The last dimension of a value tensor must be greater than the
539 // corresponding
540 // size in the sizes tensor.
541 for (int message_index = 0; message_index < message_count;
542 message_index++) {
543 OP_REQUIRES(
544 cx, sizes(message_index, i) <= max_size,
545 errors::InvalidArgument(
546 "Size to write must not be larger than value tensor; but saw: ",
547 sizes(message_index, i), " > ", max_size, " at message ",
548 message_index, " field ", i));
549 }
550 }
551
552 // This pointer is owned by the context.
553 Tensor* output_tensor;
554 OP_REQUIRES_OK(cx, cx->allocate_output(0, common_prefix, &output_tensor));
555
556 auto bufs = output_tensor->flat<string>();
557 for (int message_index = 0; message_index < message_count;
558 message_index++) {
559 // TODO(nix): possibly optimize allocation here by calling
560 // bufs(message_index).reserve(DEFAULT_BUF_SIZE);
561 StringOutputStream output_string(&bufs(message_index));
562 CodedOutputStream out(&output_string);
563 // Write fields in ascending field_number order.
564 for (int i : sorted_field_index_) {
565 auto& field_desc = *field_descs_[i];
566 const Tensor& v = values[i];
567 int size = sizes(message_index, i);
568 if (!size) continue;
569 WriteField(field_desc, v, message_index, size, &out);
570 }
571 }
572 }
573
574 private:
575 std::vector<string> field_names_;
576 std::vector<const FieldDescriptor*> field_descs_;
577
578 // Owned_desc_pool_ is null when using descriptor_source=local.
579 std::unique_ptr<DescriptorPool> owned_desc_pool_;
580
581 // Contains indices into field_names_, sorted by field number since
582 // that's the order of writing.
583 std::vector<int> sorted_field_index_;
584
585 TF_DISALLOW_COPY_AND_ASSIGN(EncodeProtoOp);
586};
