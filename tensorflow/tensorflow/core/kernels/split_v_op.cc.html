47template <typename Device, typename T, typename Tlen>
48class SplitVOpBase : public OpKernel {
49 public:
50 explicit SplitVOpBase(OpKernelConstruction* c) : OpKernel(c) {}
51
52 void ComputeEasyCases(OpKernelContext* context, bool* done,
53 std::vector<Tlen>* split_sizes_vec) {
54 const int32 num_split = context->num_outputs();
55 const Tensor& input = context->input(0);
56 const TensorShape& input_shape = input.shape();
57 const Tensor& split_tensor = context->input(1);
58 const Tensor& split_dim_tensor = context->input(2);
59
60 OP_REQUIRES(context, split_dim_tensor.NumElements() == 1,
61 errors::InvalidArgument("split_dim_tensor must have "
62 "exactly one element."));
63
64 const int32 split_dim_orig = split_dim_tensor.flat<int32>()(0);
65 const int32 split_dim =
66 split_dim_orig < 0 ? split_dim_orig + input.dims() : split_dim_orig;
67
68 OP_REQUIRES(
69 context,
70 split_tensor.dims() == 1 && split_tensor.NumElements() == num_split,
71 errors::InvalidArgument("size of the split_tensor must be 1-D and have "
72 "the same elements as outputs got ",
73 split_tensor.dims(), " -D and ",
74 split_tensor.NumElements(), " elements"));
75
76 auto split_sizes_d = split_tensor.vec<Tlen>();
77
78 split_sizes_vec->resize(split_sizes_d.size());
79
80 std::copy(split_sizes_d.data(), split_sizes_d.data() + split_sizes_d.size(),
81 split_sizes_vec->begin());
82
83 OP_REQUIRES(
84 context, num_split > 0,
85 errors::InvalidArgument(
86 "Number of ways to split should be > 0, but got ", num_split));
87
88 OP_REQUIRES(
89 context, 0 <= split_dim && split_dim < input.dims(),
90 errors::InvalidArgument("-input rank(-", input.dims(),
91 ") <= split_dim < input rank (", input.dims(),
92 "), but got ", split_dim_orig));
93
94 Tlen input_size_split_dim = input_shape.dim_size(split_dim);
95
96 // Special case 1: num_split == 1. Nothing to do.
97 if (num_split == 1) {
98 context->set_output(0, context->input(0));
99 OP_REQUIRES(
100 context, (*split_sizes_vec)[0] == input_size_split_dim,
101 errors::InvalidArgument("If there is only one output, it must have "
102 "the same size as the input. Input size: ",
103 input_size_split_dim,
104 " output size: ", (*split_sizes_vec)[0]));
105 *done = true;
106 return;
107 }
108
109 // Determine sizes of output, in case of a -1 input value
110 int neg_one_dim = -1;
111 Tlen determined_size = 0;
112 for (int d = 0; d < split_sizes_vec->size(); ++d) {
113 Tlen size = (*split_sizes_vec)[d];
114
115 if (size == -1) {
116 OP_REQUIRES(context, neg_one_dim == -1,
117 errors::InvalidArgument("There can only be one -1 in the "
118 "input."));
119 neg_one_dim = d;
120 } else {
121 determined_size += size;
122 }
123 }
124
125 OP_REQUIRES(
126 context,
127 (neg_one_dim == -1 && determined_size == input_size_split_dim) ||
128 (neg_one_dim >= 0 && determined_size <= input_size_split_dim),
129 errors::InvalidArgument("Determined shape must either match "
130 "input shape along split_dim exactly if "
131 "fully specified, or be less than the size of "
132 "the input along split_dim if not fully "
133 "specified. Got: ",
134 determined_size));
135
136 if (neg_one_dim >= 0) {
137 (*split_sizes_vec)[neg_one_dim] = input_size_split_dim - determined_size;
138 }
139
140 // Special case 2: split along the 1st dimension. We can share the
141 // underlying buffer.
142 //
143 // Apply this optimization conservatively: if input is aligned,
144 // the resulting tensors must be aligned. It's conservative
145 // because if the immediate consumer of the resulting tensors are
146 // not using eigen for computation, its perfectly fine to avoid
147 // the copying.
148 if ((split_dim == 0) && IsInnerDimsSizeAligned<T>(input_shape)) {
149 Tlen start = 0;
150 for (int i = 0; i < num_split; ++i) {
151 context->set_output(i,
152 input.Slice(start, start + (*split_sizes_vec)[i]));
153 start += (*split_sizes_vec)[i];
154 }
155 *done = true;
156 return;
157 }
158 }
159
160 template <typename IndexType>
161 std::tuple<IndexType, IndexType, IndexType> SetDims(
162 const TensorShape& input_shape, const int32 split_dim) const {
163 static_assert(std::is_integral<IndexType>::value,
164 "IndexType must be an integer type");
165 int32 prefix_dim_size = 1;
166 for (int i = 0; i < split_dim; ++i) {
167 prefix_dim_size *= input_shape.dim_size(i);
168 }
169
170 // Caller must ensure that dim_size and suffix_dim_size are <
171 // std::numeric_limits<IndexType>::max()
172 IndexType split_dim_size =
173 static_cast<IndexType>(input_shape.dim_size(split_dim));
174
175 IndexType suffix_dim_size = 1;
176 for (int i = split_dim + 1; i < input_shape.dims(); ++i) {
177 suffix_dim_size *= static_cast<IndexType>(input_shape.dim_size(i));
178 }
179 return std::make_tuple(prefix_dim_size, split_dim_size, suffix_dim_size);
180 }
181};

183template <typename T, typename Tlen, typename InputReshapedType, int NDims>
184class SplitVOpCPUImpl {
185 public:
186 template <typename MakeSizesType, typename ReshapeResultType>
187 void operator()(OpKernelContext* context,
188 const InputReshapedType& input_reshaped,
189 const std::vector<int64>& split_start_points,
190 const TensorShape& input_shape, int32 split_dim,
191 Eigen::DenseIndex prefix_dim_size,
192 Eigen::DenseIndex split_dim_size,
193 Eigen::DenseIndex suffix_dim_size,
194 std::vector<Tlen>& split_sizes_vec,
195 const MakeSizesType& make_sizes,
196 const ReshapeResultType& reshape_result) const {
197 Eigen::DSizes<Eigen::DenseIndex, NDims> indices;
198 for (int i = 0; i < NDims; ++i) {
199 indices[i] = 0;
200 }
201 const auto num_threads =
202 context->device()->tensorflow_cpu_worker_threads()->num_threads;
203 // TODO(jewillco): Tune heuristic further.
204 const auto input_element_count = input_shape.num_elements();
205 const int num_split = split_start_points.size();
206 const bool use_parallelism_between_outputs =
207 (num_split >= 4 &&
208 input_element_count >= std::max(num_threads, num_split) * 4096 &&
209 input_element_count < num_split * 180 * 1024);
210
211 auto range_output_func = [&indices, context, &input_shape, split_dim,
212 &split_sizes_vec, &split_start_points,
213 use_parallelism_between_outputs, &input_reshaped,
214 &make_sizes,
215 &reshape_result](int64 start, int64 limit) {
216 for (int64 i = start; i < limit; ++i) {
217 TensorShape output_shape(input_shape);
218 output_shape.set_dim(split_dim, split_sizes_vec[i]);
219 Tensor* result = nullptr;
220 OP_REQUIRES_OK(context,
221 context->allocate_output(i, output_shape, &result));
222
223 const auto sizes = make_sizes(split_sizes_vec[i]);
224
225 if (sizes.TotalSize() > 0) {
226 auto result_shaped = reshape_result(result, split_sizes_vec[i]);
227
228 auto current_indices = indices;
229 current_indices[NDims - 2] = split_start_points[i];
230 if (use_parallelism_between_outputs) {
231 // Use sequential implementation for single output.
232 result_shaped = input_reshaped.slice(current_indices, sizes);
233 } else {
234 // This implementation may be parallel internally.
235 functor::Split<CPUDevice, T, NDims>()(
236 context->eigen_device<CPUDevice>(), result_shaped,
237 input_reshaped, current_indices, sizes);
238 }
239 }
240 }
241 };
242 if (use_parallelism_between_outputs) {
243 // Run in parallel, disabling parallelism in functor.
244 Shard(num_split,
245 context->device()->tensorflow_cpu_worker_threads()->workers,
246 num_split, input_element_count / num_split, range_output_func);
247 } else {
248 // Run sequentially, but allow internal parallelism in functor.
249 range_output_func(0, num_split);
250 }
251 }
252};

254template <typename T, typename Tlen>
255class SplitVOpCPU : public SplitVOpBase<CPUDevice, T, Tlen> {
256 public:
257 typedef SplitVOpBase<CPUDevice, T, Tlen> Base;
258 explicit SplitVOpCPU(OpKernelConstruction* c) : Base(c) {}
259
260 void Compute(OpKernelContext* context) override {
261 bool done = false;
262 std::vector<Tlen> split_sizes_vec;
263 Base::ComputeEasyCases(context, &done, &split_sizes_vec);
264 if (!context->status().ok() || done) {
265 return;
266 }
267 const int32 num_split = Base::num_outputs();
268 const Tensor& input = context->input(0);
269 const TensorShape& input_shape = input.shape();
270 const int32 split_dim_orig = context->input(2).flat<int32>()(0);
271 const int32 split_dim =
272 split_dim_orig < 0 ? split_dim_orig + input.dims() : split_dim_orig;
273
274 // Android also uses int32 indexing, so check here also.
275 OP_REQUIRES(
276 context,
277 FastBoundsCheck(input.NumElements(),
278 std::numeric_limits<Eigen::DenseIndex>::max()),
279 errors::InvalidArgument("Split requires input size < ",
280 std::numeric_limits<Eigen::DenseIndex>::max()));
281
282 Eigen::DenseIndex prefix_dim_size;
283 Eigen::DenseIndex split_dim_size;
284 Eigen::DenseIndex suffix_dim_size;
285
286 std::tie(prefix_dim_size, split_dim_size, suffix_dim_size) =
287 Base::template SetDims<Eigen::DenseIndex>(input_shape, split_dim);
288 std::vector<int64> split_start_points(num_split);
289 for (int i = 0; i < num_split; ++i) {
290 if (i == 0) {
291 split_start_points[i] = 0;
292 } else {
293 split_start_points[i] =
294 split_start_points[i - 1] + split_sizes_vec[i - 1];
295 }
296 }
297
298 if (prefix_dim_size == 1) {
299 auto input_reshaped =
300 input.shaped<T, 2>({split_dim_size, suffix_dim_size});
301 auto make_sizes = [&](Eigen::DenseIndex split_size) {
302 return Eigen::DSizes<Eigen::DenseIndex, 2>{split_size, suffix_dim_size};
303 };
304 auto reshape_result = [&](Tensor* result, Tlen split_size) {
305 return result->shaped<T, 2>({split_size, suffix_dim_size});
306 };
307 SplitVOpCPUImpl<T, Tlen, decltype(input_reshaped), 2>{}(
308 context, input_reshaped, split_start_points, input_shape, split_dim,
309 prefix_dim_size, split_dim_size, suffix_dim_size, split_sizes_vec,
310 make_sizes, reshape_result);
311 } else {
312 auto input_reshaped = input.shaped<T, 3>(
313 {prefix_dim_size, split_dim_size, suffix_dim_size});
314 auto make_sizes = [&](Eigen::DenseIndex split_size) {
315 return Eigen::DSizes<Eigen::DenseIndex, 3>{prefix_dim_size, split_size,
316 suffix_dim_size};
317 };
318 auto reshape_result = [&](Tensor* result, Tlen split_size) {
319 return result->shaped<T, 3>(
320 {prefix_dim_size, split_size, suffix_dim_size});
321 };
322 SplitVOpCPUImpl<T, Tlen, decltype(input_reshaped), 3>{}(
323 context, input_reshaped, split_start_points, input_shape, split_dim,
324 prefix_dim_size, split_dim_size, suffix_dim_size, split_sizes_vec,
325 make_sizes, reshape_result);
326 }
327 }
328};

340// Partial specialization for GPU
341template <typename T, typename Tlen>
342class SplitVOpGPU : public SplitVOpBase<GPUDevice, T, Tlen> {
343 public:
344 typedef SplitVOpBase<GPUDevice, T, Tlen> Base;
345 explicit SplitVOpGPU(OpKernelConstruction* c) : Base(c) {}
346
347 void Compute(OpKernelContext* context) override {
348 bool done = false;
349 std::vector<Tlen> split_sizes_vec;
350 Base::ComputeEasyCases(context, &done, &split_sizes_vec);
351 if (!context->status().ok() || done) {
352 return;
353 }
354 const int32 num_split = Base::num_outputs();
355 const Tensor& input = context->input(0);
356 const TensorShape& input_shape = input.shape();
357 const int32 split_dim_orig = context->input(2).flat<int32>()(0);
358 const int32 split_dim =
359 split_dim_orig < 0 ? split_dim_orig + input.dims() : split_dim_orig;
360 OP_REQUIRES(
361 context,
362 FastBoundsCheck(input.NumElements(), std::numeric_limits<int32>::max()),
363 errors::InvalidArgument("Split on GPU requires input size "
364 "< max int32"));
365
366 int32 prefix_dim_size;
367 int32 split_dim_size;
368 int32 suffix_dim_size;
369 std::tie(prefix_dim_size, split_dim_size, suffix_dim_size) =
370 Base::template SetDims<int32>(input_shape, split_dim);
371
372 // use the same approach as concat (see documentation there)
373 // reshape to 2D
374
375 if (num_split > 16) {
376 CudaDeviceArrayOnHost<T*> ptrs(context, num_split);
377 OP_REQUIRES_OK(context, ptrs.Init());
378
379 CudaDeviceArrayOnHost<Tlen> offsets(context, num_split + 1);
380 OP_REQUIRES_OK(context, offsets.Init());
381
382 Tlen offset = 0;
383 int entry = split_sizes_vec[0];
384 bool fixed_size =
385 std::all_of(split_sizes_vec.begin(), split_sizes_vec.end(),
386 [&entry](int n) { return n == entry; });
387
388 for (int i = 0; i < num_split; ++i) {
389 TensorShape output_shape(input_shape);
390 output_shape.set_dim(split_dim, split_sizes_vec[i]);
391 Tensor* result = nullptr;
392 OP_REQUIRES_OK(context,
393 context->allocate_output(i, output_shape, &result));
394 ptrs.Set(i, result->flat<T>().data());
395 offsets.Set(i, offset);
396 offset += split_sizes_vec[i] * suffix_dim_size;
397 }
398 offsets.Set(num_split, offset);
399 OP_REQUIRES_OK(context, ptrs.Finalize());
400 OP_REQUIRES_OK(context, offsets.Finalize());
401
402 if (input.NumElements() > 0) {
403 SplitVOpGPULaunch<T, Tlen>().Run(
404 context->eigen_device<GPUDevice>(), fixed_size,
405 input.flat<T>().data(), prefix_dim_size,
406 input.NumElements() / prefix_dim_size, offsets.data(), ptrs.data());
407 OP_REQUIRES(
408 context, context->op_device_context()->stream()->ok(),
409 errors::Internal("Launch of gpu kernel for SplitVOp failed"));
410 }
411 } else {
412 Eigen::DenseIndex prefix_dim_size;
413 Eigen::DenseIndex split_dim_size;
414 Eigen::DenseIndex suffix_dim_size;
415
416 std::tie(prefix_dim_size, split_dim_size, suffix_dim_size) =
417 Base::template SetDims<Eigen::DenseIndex>(input_shape, split_dim);
418 auto input_reshaped = input.shaped<T, 2>(
419 {prefix_dim_size, split_dim_size * suffix_dim_size});
420
421 Eigen::DSizes<Eigen::DenseIndex, 2> indices{0, 0};
422
423 for (int i = 0; i < num_split; ++i) {
424 TensorShape output_shape(input_shape);
425 output_shape.set_dim(split_dim, split_sizes_vec[i]);
426 Tensor* result = nullptr;
427 OP_REQUIRES_OK(context,
428 context->allocate_output(i, output_shape, &result));
429
430 Eigen::DSizes<Eigen::DenseIndex, 2> sizes{
431 prefix_dim_size, split_sizes_vec[i] * suffix_dim_size};
432
433 if (sizes.TotalSize() > 0) {
434 auto result_shaped = result->shaped<T, 2>(
435 {prefix_dim_size, split_sizes_vec[i] * suffix_dim_size});
436
437 functor::SplitCustom<GPUDevice, T>()(
438 context->eigen_device<GPUDevice>(), result_shaped, input_reshaped,
439 indices, sizes);
440 }
441 indices[1] += split_sizes_vec[i] * suffix_dim_size;
442 }
443 }
444 }
