44template <typename T>
45struct MemCpyFunctor {
46 // Returns true if the copy was made with memcpy, false otherwise.
47 bool Copy(const Tensor& input, const gtl::InlinedVector<int64, 4>& begin,
48 const gtl::InlinedVector<int64, 4>& end, Tensor* result) {
49 if (DataTypeCanUseMemcpy(DataTypeToEnum<T>::v())) {
50 auto in = input.tensor<T, 2>();
51 auto output = result->tensor<T, 2>();
52 // TODO(agarwal): Consider multi-threading if size[0] is large
53 for (int row_in = begin[0], row_out = 0; row_in < end[0];
54 ++row_in, ++row_out) {
55 if (row_in + 1 < end[0]) {
56 port::prefetch<port::PREFETCH_HINT_T0>(&output(row_in + 1, 0));
57 port::prefetch<port::PREFETCH_HINT_T0>(&in(row_in + 1, begin[1]));
58 }
59 memcpy(&output(row_out, 0), &in(row_in, begin[1]),
60 (end[1] - begin[1]) * sizeof(T));
61 }
62 return true;
63 }
64 return false;
65 }
66};
67
68template <>
69struct MemCpyFunctor<ResourceHandle> {
70 bool Copy(const Tensor& input, const gtl::InlinedVector<int64, 4>& begin,
71 const gtl::InlinedVector<int64, 4>& end, Tensor* result) {
72 return false;
73 }
74};

78template <typename Device, typename T>
79class StridedSliceOp : public OpKernel {
80 public:
81 explicit StridedSliceOp(OpKernelConstruction* context) : OpKernel(context) {
82 OP_REQUIRES_OK(context, context->GetAttr("begin_mask", &begin_mask));
83 OP_REQUIRES_OK(context, context->GetAttr("end_mask", &end_mask));
84 OP_REQUIRES_OK(context, context->GetAttr("ellipsis_mask", &ellipsis_mask));
85 OP_REQUIRES_OK(context, context->GetAttr("new_axis_mask", &new_axis_mask));
86 OP_REQUIRES_OK(context,
87 context->GetAttr("shrink_axis_mask", &shrink_axis_mask));
88 }
89
90 void Compute(OpKernelContext* context) override {
91 TensorShape processing_shape, final_shape;
92 bool is_identity = true;
93 bool slice_dim0 = true;
94 bool is_simple_slice = true;
95 gtl::InlinedVector<int64, 4> begin;
96 gtl::InlinedVector<int64, 4> end;
97 gtl::InlinedVector<int64, 4> strides;
98
99 OP_REQUIRES_OK(
100 context, ValidateStridedSliceOp(
101 &context->input(1), &context->input(2), context->input(3),
102 context->input(0).shape(), begin_mask, end_mask,
103 ellipsis_mask, new_axis_mask, shrink_axis_mask,
104 &processing_shape, &final_shape, &is_identity,
105 &is_simple_slice, &slice_dim0, &begin, &end, &strides));
106 const Tensor& input = context->input(0);
107
108 // Optimization #1, slice is a no-op plus reshape
109 if (is_identity) {
110 VLOG(1) << "Strided slice identity ";
111 Tensor tmp;
112 OP_REQUIRES(context, tmp.CopyFrom(input, final_shape),
113 errors::Internal("Copy failed"));
114 context->set_output(0, tmp);
115 return;
116 }
117
118 // Optimization #2, slice is memory contiguous (only occurs in dim 0)
119 if (slice_dim0 && IsDim0SliceAligned<T>(input.shape(), begin[0], end[0])) {
120 OP_REQUIRES(context, input.dims() >= 1,
121 errors::InvalidArgument(
122 "Input must have rank at least 1, got: ", input.dims()));
123 // Otherwise, is_identity should be true.
124 VLOG(1) << "Strided slice dim 0: " << input.shape().DebugString();
125 OP_REQUIRES(
126 context, begin[0] <= end[0],
127 errors::InvalidArgument("begin[0] (", begin[0],
128 ") must less or equal to end[0] (", end[0]));
129 Tensor slice = input.Slice(begin[0], end[0]);
130 Tensor tmp;
131 OP_REQUIRES(context, tmp.CopyFrom(slice, final_shape),
132 errors::Internal("Copy failed"));
133 context->set_output(0, tmp);
134 return;
135 }
136
137 Tensor* result = nullptr;
138 OP_REQUIRES_OK(context, context->allocate_output(0, final_shape, &result));
139 const int input_dims = input.dims();
140 const int processing_dims = processing_shape.dims();
141
142 if (processing_shape.num_elements() > 0) {
143 // Optimization #3, slice has stride 1 in all dimensions
144 // Optimization #3A, slice has only two dimensions
145 // TODO(aselle): Here we are restricting to processing_shape and
146 // final_shape being 2D. This isn't strictly necessary, but I don't
147 // want to blow up code gen size, because to shape<> you need static
148 // NDIM and T
149 if (is_simple_slice && std::is_same<Device, CPUDevice>::value &&
150 input_dims == 2 && processing_shape.dims() == 2 &&
151 final_shape.dims() == 2) {
152 MemCpyFunctor<T> functor;
153 if (functor.Copy(input, begin, end, result)) {
154 return;
155 }
156 }
157
158#define HANDLE_DIM(NDIM) \
159 if (processing_dims == NDIM) { \
160 HandleStridedSliceCase<Device, T, NDIM>(context, begin, end, strides, \
161 processing_shape, is_simple_slice, \
162 result); \
163 return; \
164 }
165
166 HANDLE_DIM(1);
167 HANDLE_DIM(2);
168 HANDLE_DIM(3);
169 HANDLE_DIM(4);
170 HANDLE_DIM(5);
171 HANDLE_DIM(6);
172 HANDLE_DIM(7);
173
174#undef HANDLE_DIM
175
176 OP_REQUIRES(
177 context, false,
178 errors::Unimplemented("Unhandled input dimensions ", input_dims));
179 }
180 }
181
182 private:
183 int32 begin_mask, end_mask;
184 int32 ellipsis_mask, new_axis_mask, shrink_axis_mask;
185};

187template <typename Device, typename T>
188class StridedSliceGradOp : public OpKernel {
189 public:
190 explicit StridedSliceGradOp(OpKernelConstruction* context)
191 : OpKernel(context) {
192 OP_REQUIRES_OK(context, context->GetAttr("begin_mask", &begin_mask));
193 OP_REQUIRES_OK(context, context->GetAttr("end_mask", &end_mask));
194 OP_REQUIRES_OK(context, context->GetAttr("ellipsis_mask", &ellipsis_mask));
195 OP_REQUIRES_OK(context, context->GetAttr("new_axis_mask", &new_axis_mask));
196 OP_REQUIRES_OK(context,
197 context->GetAttr("shrink_axis_mask", &shrink_axis_mask));
198 }
199
200 void Compute(OpKernelContext* context) override {
201 TensorShape processing_shape, final_shape;
202 bool is_identity = true;
203 bool slice_dim0 = true;
204 bool is_simple_slice = true;
205 gtl::InlinedVector<int64, 4> begin;
206 gtl::InlinedVector<int64, 4> end;
207 gtl::InlinedVector<int64, 4> strides;
208
209 TensorShape input_shape;
210 const Tensor& input_shape_tensor = context->input(0);
211 OP_REQUIRES(
212 context, input_shape_tensor.dims() == 1,
213 errors::InvalidArgument("shape must be 1-D, got shape.shape = ",
214 input_shape_tensor.shape().DebugString()));
215 if (input_shape_tensor.dtype() == DT_INT32) {
216 OP_REQUIRES_OK(
217 context, TensorShapeUtils::MakeShape(input_shape_tensor.vec<int32>(),
218 &input_shape));
219 } else if (input_shape_tensor.dtype() == DT_INT64) {
220 OP_REQUIRES_OK(
221 context, TensorShapeUtils::MakeShape(input_shape_tensor.vec<int64>(),
222 &input_shape));
223 } else {
224 LOG(FATAL) << "shape must have type int32 or int64.";
225 }
226
227 OP_REQUIRES_OK(
228 context,
229 ValidateStridedSliceOp(
230 &context->input(1), &context->input(2), context->input(3),
231 input_shape, begin_mask, end_mask, ellipsis_mask, new_axis_mask,
232 shrink_axis_mask, &processing_shape, &final_shape, &is_identity,
233 &is_simple_slice, &slice_dim0, &begin, &end, &strides));
234
235 // Check to make sure dy is consistent with the original slice
236 TensorShape dy_shape = context->input(4).shape();
237 OP_REQUIRES(
238 context, final_shape == dy_shape,
239 errors::InvalidArgument("shape of dy was ", dy_shape.DebugString(),
240 " instead of ", final_shape.DebugString()));
241
242 if (!context->status().ok()) return;
243
244 // const int input_dims = input.dims();
245 const int processing_dims = processing_shape.dims();
246 Tensor* result = nullptr;
247 OP_REQUIRES_OK(context, context->allocate_output(0, input_shape, &result));
248
249 if (processing_shape.dims() == 0) {
250 auto in = context->input(4);
251 OP_REQUIRES(context, result->CopyFrom(in, processing_shape),
252 errors::Internal("Copy failed"));
253 return;
254 }
255
256#define HANDLE_DIM(NDIM) \
257 if (processing_dims == NDIM) { \
258 HandleStridedSliceGradCase<Device, T, NDIM>(context, begin, end, strides, \
259 processing_shape, \
260 is_simple_slice, result); \
261 return; \
262 }
263
264 HANDLE_DIM(1);
265 HANDLE_DIM(2);
266 HANDLE_DIM(3);
267 HANDLE_DIM(4);
268 HANDLE_DIM(5);
269 HANDLE_DIM(6);
270 HANDLE_DIM(7);
271
272#undef HANDLE_DIM
273 }
274
275 private:
276 int32 begin_mask, end_mask;
277 int32 ellipsis_mask, new_axis_mask, shrink_axis_mask;
278};

280template <typename Device, typename T>
281class StridedSliceAssignOp : public OpKernel {
282 public:
283 explicit StridedSliceAssignOp(OpKernelConstruction* context)
284 : OpKernel(context) {
285 OP_REQUIRES_OK(context, context->GetAttr("begin_mask", &begin_mask));
286 OP_REQUIRES_OK(context, context->GetAttr("end_mask", &end_mask));
287 OP_REQUIRES_OK(context, context->GetAttr("ellipsis_mask", &ellipsis_mask));
288 OP_REQUIRES_OK(context, context->GetAttr("new_axis_mask", &new_axis_mask));
289 OP_REQUIRES_OK(context,
290 context->GetAttr("shrink_axis_mask", &shrink_axis_mask));
291 }
292
293 void Compute(OpKernelContext* context) override {
294 TensorShape processing_shape, final_shape;
295 bool is_identity = true;
296 bool slice_dim0 = true;
297 bool is_simple_slice = true;
298 gtl::InlinedVector<int64, 4> begin;
299 gtl::InlinedVector<int64, 4> end;
300 gtl::InlinedVector<int64, 4> strides;
301
302 Tensor old_lhs;
303 if (context->input_dtype(0) == DT_RESOURCE) {
304 Var* v;
305 OP_REQUIRES_OK(context,
306 LookupResource(context, HandleFromInput(context, 0), &v));
307 old_lhs = *v->tensor();
308 OP_REQUIRES(context, old_lhs.dtype() == DataTypeToEnum<T>::value,
309 errors::InvalidArgument(
310 "l-value dtype ", DataTypeString(old_lhs.dtype()),
311 " does not match r-value dtype ",
312 DataTypeString(DataTypeToEnum<T>::value)));
313 } else {
314 context->forward_ref_input_to_ref_output(0, 0);
315 old_lhs = context->mutable_input(0, true);
316 }
317
318 OP_REQUIRES_OK(
319 context,
320 ValidateStridedSliceOp(
321 &context->input(1), &context->input(2), context->input(3),
322 old_lhs.shape(), begin_mask, end_mask, ellipsis_mask, new_axis_mask,
323 shrink_axis_mask, &processing_shape, &final_shape, &is_identity,
324 &is_simple_slice, &slice_dim0, &begin, &end, &strides));
325
326 if (processing_shape.num_elements()) {
327 const Tensor& input = context->input(4);
328 TensorShape input_shape = input.shape();
329 TensorShape original_shape = old_lhs.shape();
330 // TODO(aselle): This check is too strong, we only should need
331 // input_shape to be broadcastable to final_shape
332 OP_REQUIRES(
333 context, final_shape == input_shape,
334 errors::Unimplemented(
335 "sliced l-value shape ", final_shape.DebugString(),
336 " does not match r-value shape ", input_shape.DebugString(),
337 ". Automatic broadcasting not ", "yet implemented."));
338 const int processing_dims = processing_shape.dims();
339
340 // 0-dimensional case implies the left and right are exactly the same
341 // scalar shape
342
343// Handle general dimensions
344#define HANDLE_DIM(NDIM) \
345 if (processing_dims == NDIM) { \
346 HandleStridedSliceAssignCase<Device, T, NDIM>()( \
347 context, begin, end, strides, processing_shape, is_simple_slice, \
348 &old_lhs); \
349 return; \
350 }
351 HANDLE_DIM(0);
352 HANDLE_DIM(1);
353 HANDLE_DIM(2);
354 HANDLE_DIM(3);
355 HANDLE_DIM(4);
356 HANDLE_DIM(5);
357 HANDLE_DIM(6);
358 HANDLE_DIM(7);
359#undef HANDLE_DIM
360
361 OP_REQUIRES(context, false,
362 errors::Unimplemented("Unhandled input dimensions ",
363 processing_dims));
364 }
365 }
366
367 private:
368 int32 begin_mask, end_mask;
369 int32 ellipsis_mask, new_axis_mask, shrink_axis_mask;
370};
