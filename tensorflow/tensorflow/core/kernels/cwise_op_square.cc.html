16#include "tensorflow/core/kernels/cwise_ops_common.h"

18namespace tensorflow {
19REGISTER8(UnaryOp, CPU, "Square", functor::square, float, Eigen::half, double,
20 int32, int64, complex64, complex128, bfloat16);
21
22#if GOOGLE_CUDA
23REGISTER4(UnaryOp, GPU, "Square", functor::square, float, Eigen::half, double,
24 int64);
25
26// A special GPU kernel for int32.
27// TODO(b/25387198): Also enable int32 in device memory. This kernel
28// registration requires all int32 inputs and outputs to be in host memory.
29REGISTER_KERNEL_BUILDER(Name("Square")
30 .Device(DEVICE_GPU)
31 .HostMemory("x")
32 .HostMemory("y")
33 .TypeConstraint<int32>("T"),
34 UnaryOp<CPUDevice, functor::square<int32>>);
35#endif
36
37#ifdef TENSORFLOW_USE_SYCL
38REGISTER3(UnaryOp, SYCL, "Square", functor::square, float, double, int64);
39REGISTER_KERNEL_BUILDER(Name("Square")
40 .Device(DEVICE_SYCL)
41 .HostMemory("x")
42 .HostMemory("y")
43 .TypeConstraint<int32>("T"),
44 UnaryOp<CPUDevice, functor::square<int32>>);
45#endif // TENSORFLOW_USE_SYC
46} // namespace tensorflow
