27class ScopedAllocatorOp : public OpKernel {
28 public:
29 explicit ScopedAllocatorOp(OpKernelConstruction* context)
30 : OpKernel(context) {
31 OP_REQUIRES_OK(context, context->GetAttr("T", &dtype_));
32 OP_REQUIRES_OK(context, context->GetAttr("shapes", &shapes_));
33 OP_REQUIRES_OK(context, context->GetAttr("sa_name", &name_));
34 OP_REQUIRES_OK(context, context->GetAttr("id", &id_));
35 OP_REQUIRES_OK(context, context->GetAttr("expected_call_count",
36 &expected_call_count_));
37 device_ = context->device();
38 // Precalculate the size of the backing tensor and the offsets of
39 // the subtensors to be allocated from it, taking into account
40 // alignment considerations.
41 ScopedAllocatorMgr::PopulateFields(id_, shapes_, dtype_, &fields_);
42 size_t num_bytes = fields_.back().offset + fields_.back().bytes;
43 num_elements_ = num_bytes / DataTypeSize(dtype_);
44 OP_REQUIRES(context, num_bytes % DataTypeSize(dtype_) == 0,
45 errors::InvalidArgument(
46 "Number of bytes ", num_bytes,
47 " must be divisible by size of datatype ", dtype_));
48 }
49
50 void Compute(OpKernelContext* context) override {
51 ScopedAllocatorMgr* sam = device_->GetScopedAllocatorMgr();
52 if (!sam) {
53 context->SetStatus(errors::Internal(
54 "ScopedAllocatorMgr not supported on device ", device_->name()));
55 return;
56 }
57 Tensor* backing_tensor = nullptr;
58 AllocatorAttributes attr = context->output_alloc_attr(0);
59 Status s =
60 context->allocate_output(0, {num_elements_}, &backing_tensor, attr);
61 VLOG(1) << "_ScopedAllocatorOp new backing tensor size "
62 << backing_tensor->TotalBytes() << " num_elements_ "
63 << num_elements_ << " buffer " << DMAHelper::buffer(backing_tensor)
64 << " base addr " << DMAHelper::base(backing_tensor);
65 if (s.ok()) {
66 s = sam->AddScopedAllocator(*backing_tensor, context->step_id(), id_,
67 name_, fields_, expected_call_count_);
68 }
69 if (!s.ok()) {
70 context->SetStatus(s);
71 }
72 }
73
74 private:
75 std::vector<TensorShape> shapes_;
76 DataType dtype_;
77 int64 num_elements_;
78 std::vector<ScopedAllocator::Field> fields_;
79 string name_;
80 int32 id_;
81 int32 expected_call_count_;
82 DeviceBase* device_;
83};
84
85REGISTER_KERNEL_BUILDER(Name("_ScopedAllocator").Device(DEVICE_CPU),
86 ScopedAllocatorOp);
87
88REGISTER_KERNEL_BUILDER(Name("_ScopedAllocator").Device(DEVICE_GPU),
89 ScopedAllocatorOp);

91class ScopedAllocatorConcatOp : public OpKernel {
92 public:
93 explicit ScopedAllocatorConcatOp(OpKernelConstruction* context)
94 : OpKernel(context) {
95 OP_REQUIRES_OK(context, context->GetAttr("shape", &shape_));
96 OP_REQUIRES_OK(context, context->GetAttr("T", &dtype_));
97 // This stuff is just for debugging
98 OP_REQUIRES_OK(context, context->GetAttr("sa_name", &name_));
99 OP_REQUIRES_OK(context, context->GetAttr("id", &id_));
100 device_ = context->device();
101 }
102
103 void Compute(OpKernelContext* context) override {
104 const Tensor& backing_tensor = context->input(0);
105 // Check that type matches.
106 OP_REQUIRES(
107 context, backing_tensor.dtype() == dtype_,
108 errors::InvalidArgument("Backing tensor type ", backing_tensor.dtype(),
109 " does not match expected type ", dtype_));
110 // Check that backing tensor is at least as large as the shape of the
111 // output.
112 OP_REQUIRES(context, backing_tensor.NumElements() >= shape_.num_elements(),
113 errors::InvalidArgument("Backing tensor num elements ",
114 backing_tensor.NumElements(),
115 " is not equal to expected ",
116 shape_.num_elements()));
117 VLOG(1) << "_ScopedAllocatorConcatOp outputting backing tensor at "
118 << DMAHelper::base(&backing_tensor);
119 Tensor backing_copy(backing_tensor);
120 context->set_output(0, backing_copy);
121 const TensorBuffer* backing_buf = DMAHelper::buffer(&backing_copy);
122 const void* backing_tensor_lb = backing_buf->data();
123 const void* backing_tensor_ub = static_cast<const void*>(
124 static_cast<const char*>(backing_tensor_lb) + backing_buf->size());
125 // Check that all inputs lie entirely within the backing tensor.
126 for (int i = 1; i < context->num_inputs(); ++i) {
127 const TensorBuffer* input_buf = DMAHelper::buffer(&context->input(i));
128 const void* input_lb = input_buf->data();
129 OP_REQUIRES(
130 context, input_lb >= backing_tensor_lb,
131 errors::InvalidArgument("Lower bound check fail for input ", i,
132 " to node ", context->op_kernel().name()));
133 const void* input_ub = static_cast<const void*>(
134 static_cast<const char*>(input_lb) + input_buf->size());
135 OP_REQUIRES(
136 context, input_ub <= backing_tensor_ub,
137 errors::InvalidArgument("Upper bound check fail for input ", i,
138 " to node ", context->op_kernel().name()));
139 }
140 }
141
142 private:
143 TensorShape shape_;
144 DataType dtype_;
145 string name_;
146 int32 id_;
147 DeviceBase* device_;
148};

150REGISTER_KERNEL_BUILDER(Name("_ScopedAllocatorConcat").Device(DEVICE_CPU),
151 ScopedAllocatorConcatOp);

153REGISTER_KERNEL_BUILDER(Name("_ScopedAllocatorConcat").Device(DEVICE_GPU),
154 ScopedAllocatorConcatOp);

156class ScopedAllocatorSplitOp : public OpKernel {
157 public:
158 explicit ScopedAllocatorSplitOp(OpKernelConstruction* context)
159 : OpKernel(context) {
160 OP_REQUIRES_OK(context, context->GetAttr("T", &dtype_));
161 // This stuff is just for debugging
162 OP_REQUIRES_OK(context, context->GetAttr("sa_name", &name_));
163 OP_REQUIRES_OK(context, context->GetAttr("id", &id_));
164 device_ = context->device();
165 }
166
167 void Compute(OpKernelContext* context) override {
168 Tensor backing_copy(context->input(0));
169 // Check that type matches.
170 OP_REQUIRES(
171 context, backing_copy.dtype() == dtype_,
172 errors::InvalidArgument("Backing tensor type ", backing_copy.dtype(),
173 " does not match expected type ", dtype_));
174 const TensorBuffer* backing_buf = DMAHelper::buffer(&backing_copy);
175 const void* backing_tensor_lb = backing_buf->data();
176 const void* backing_tensor_ub = static_cast<const void*>(
177 static_cast<const char*>(backing_tensor_lb) + backing_buf->size());
178 for (int i = 1; i < context->num_inputs(); ++i) {
179 VLOG(1) << "_ScopedAllocatorSplitOp assigning input " << i
180 << " to output " << i - 1 << " buf addr "
181 << DMAHelper::base(&context->input(i));
182 Tensor copy(context->input(i));
183 OP_REQUIRES(
184 context, copy.dtype() == dtype_,
185 errors::InvalidArgument("Input ", i, " tensor type ", copy.dtype(),
186 " does not match expected type ", dtype_));
187 context->set_output(i - 1, copy);
188 const TensorBuffer* input_buf = DMAHelper::buffer(&copy);
189 const void* input_lb = input_buf->data();
190 OP_REQUIRES(
191 context, input_lb >= backing_tensor_lb,
192 errors::InvalidArgument("Lower bound check fail for input ", i,
193 " to node ", context->op_kernel().name()));
194 const void* input_ub = static_cast<const void*>(
195 static_cast<const char*>(input_lb) + input_buf->size());
196 OP_REQUIRES(
197 context, input_ub <= backing_tensor_ub,
198 errors::InvalidArgument("Upper bound check fail for input ", i,
199 " to node ", context->op_kernel().name()));
200 }
201 }
202
203 private:
204 DataType dtype_;
205 string name_;
206 int32 id_;
207 DeviceBase* device_;
208};
