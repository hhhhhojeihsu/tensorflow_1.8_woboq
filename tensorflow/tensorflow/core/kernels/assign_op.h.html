25namespace tensorflow {

27// TODO(jeff): Get rid of use_exclusive_lock_ option
28
29// Computes *input[0] = input[1]
30class AssignOp : public OpKernel {
31 public:
32 explicit AssignOp(OpKernelConstruction* context) : OpKernel(context) {
33 OP_REQUIRES_OK(context,
34 context->GetAttr("use_locking", &use_exclusive_lock_));
35 OP_REQUIRES_OK(context,
36 context->GetAttr("validate_shape", &validate_shape_));
37 OP_REQUIRES(context, IsRefType(context->input_type(0)),
38 errors::InvalidArgument("lhs input needs to be a ref type"));
39 }
40
41 void Compute(OpKernelContext* context) override {
42 const Tensor& rhs = context->input(1);
43
44 // We always return the input ref.
45 context->forward_ref_input_to_ref_output(0, 0);
46
47 // We can't always know how this value will be used downstream,
48 // so make conservative assumptions in specifying constraints on
49 // the memory allocation attributes.
50 // TODO(rmlarsen): These conservative constraints make buffer
51 // forwarding unlikely to happen very often. Try to use graph analysis
52 // (possibly the InferAllocAttr pass in the executer) to improve the
53 // situation.
54 AllocatorAttributes attr;
55 attr.set_gpu_compatible(true);
56 attr.set_nic_compatible(true);
57
58 {
59 mutex_lock l(*context->input_ref_mutex(0));
60 const Tensor& old_lhs = context->mutable_input(0, /* lock_held */ true);
61 const bool same_shape = old_lhs.shape().IsSameSize(rhs.shape());
62 if (validate_shape_) {
63 OP_REQUIRES(
64 context, same_shape,
65 errors::InvalidArgument(
66 "Assign requires shapes of both tensors to match. lhs shape= ",
67 old_lhs.shape().DebugString(),
68 " rhs shape= ", rhs.shape().DebugString()));
69 }
70
71 // In the code below we try to minimize the amount of memory allocation
72 // and copying by trying the following two shortcuts:
73 // 1. If we can reuse the rhs buffer we avoid both a memory allocation
74 // and copying.
75 // 2. If the lhs is initialized and has the same number of elements as the
76 // rhs we can avoid a memory allocation.
77
78 // 1. Try to reuse the rhs.
79 std::unique_ptr<Tensor> input_alias = context->forward_input(
80 1, OpKernelContext::Params::kNoReservation /*output_index*/,
81 old_lhs.dtype(), old_lhs.shape(), DEVICE_MEMORY, attr);
82 if (input_alias != nullptr) {
83 // Transfer ownership to the ref.
84 context->replace_ref_input(0, *input_alias.release(),
85 /* lock_held */ true);
86 return;
87 }
88
89 // 2. Try to copy into an existing buffer.
90 if (old_lhs.IsInitialized() &&
91 old_lhs.shape().num_elements() == rhs.shape().num_elements()) {
92 // The existing lhs tensor has already been initialized and the right
93 // hand side can fit in the underlying buffer.
94 Tensor reshaped_old_lhs;
95 if (same_shape) {
96 reshaped_old_lhs = old_lhs;
97 } else {
98 CHECK(reshaped_old_lhs.CopyFrom(old_lhs, rhs.shape()));
99 context->replace_ref_input(0, reshaped_old_lhs, /* lock_held */ true);
100 }
101 if (use_exclusive_lock_) {
102 Copy(context, &reshaped_old_lhs, rhs);
103 return;
104 }
105 } else {
106 // Create a new persistent tensor whose shape matches the right hand
107 // side, hand off to lhs and copy the rhs into it.
108 PersistentTensor copy;
109 Tensor* copyTensor = nullptr;
110 OP_REQUIRES_OK(
111 context, context->allocate_persistent(old_lhs.dtype(), rhs.shape(),
112 &copy, &copyTensor, attr));
113 // We track memory of variables in variable ops instead of in this
114 // assign op.
115 context->clear_recorded_memory();
116 context->replace_ref_input(0, *copyTensor, /* lock_held */ true);
117 if (use_exclusive_lock_) {
118 Copy(context, copyTensor, rhs);
119 return;
120 }
121 }
122 }
123
124 // The tensor has already been initialized and the right hand side
125 // matches the left hand side's shape. We have been told to do the
126 // copy outside the lock.
127 Tensor old_unlocked_lhs = context->mutable_input(0, /* lock_held */ false);
128 Copy(context, &old_unlocked_lhs, rhs);
129 }
130
131 virtual void Copy(OpKernelContext* context, Tensor* lhs,
132 const Tensor& rhs) = 0;
133
134 bool use_exclusive_lock_;
135 bool validate_shape_;
136};
