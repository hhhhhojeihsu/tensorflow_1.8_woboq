34// This is for use with ResourceVariables to ensure *tensor has a
35// reference count of 1 before you update it.
36// REQUIRES: If you pass in variable->tensor(), *variable->mu() must be held.
37template <typename Device, typename T>
38Status PrepareToUpdateVariable(OpKernelContext* ctx, Tensor* tensor) {
39 if (!tensor->RefCountIsOne()) {
40 // Tensor's buffer is in use by some read, so we need to copy before
41 // updating.
42 PersistentTensor unused;
43 Tensor* tmp;
44 if (std::is_same<T, Variant>::value) {
45 AllocatorAttributes attr;
46 attr.set_on_host(true);
47 TF_RETURN_IF_ERROR(ctx->allocate_persistent(
48 tensor->dtype(), tensor->shape(), &unused, &tmp, attr));
49
50 const auto elements_in = tensor->flat<Variant>();
51 auto elements_out = tmp->flat<Variant>();
52 for (int64 i = 0; i < elements_in.size(); ++i) {
53 elements_out(i) = elements_in(i);
54 }
55 } else {
56 AllocatorAttributes attr;
57 attr.set_gpu_compatible(true);
58 attr.set_nic_compatible(true);
59 TF_RETURN_IF_ERROR(ctx->allocate_persistent(
60 tensor->dtype(), tensor->shape(), &unused, &tmp, attr));
61 functor::DenseUpdate<Device, T, ASSIGN> copy_functor;
62 copy_functor(ctx->eigen_device<Device>(), tmp->flat<T>(),
63 const_cast<const Tensor*>(tensor)->flat<T>());
64 }
65 *tensor = *tmp;
66 }
67 return Status::OK();
68}

70// This gives you `*out`, a tensor you can update, corresponding to a
71// variable passed as input index `input`. This handles the
72// differences between reference and resource variables. For resource
73// variables, we ensure `*out` has a reference count of 1 (using
74// PrepareToUpdateVariable() to copy if necessary) unless
75// sparse && !lock_held, in which case it never copies.
76template <typename Device, typename T>
77Status GetInputTensorFromVariable(OpKernelContext* ctx, int input,
78 bool lock_held, bool sparse, Tensor* out) {
79 if (ctx->input_dtype(input) == DT_RESOURCE) {
80 Var* var;
81 if (LookupResource(ctx, HandleFromInput(ctx, input), &var).ok()) {
82 core::ScopedUnref unref_var(var);
83 if (lock_held) {
84 TF_RETURN_IF_ERROR(
85 PrepareToUpdateVariable<Device, T>(ctx, var->tensor()));
86 *out = *var->tensor();
87 } else {
88 mutex_lock ml(*var->mu());
89 if (!sparse) {
90 TF_RETURN_IF_ERROR(
91 PrepareToUpdateVariable<Device, T>(ctx, var->tensor()));
92 }
93 *out = *var->tensor();
94 }
95 return Status::OK();
96 } else {
97 return errors::Internal("Invalid variable reference.");
98 }
99 }
100 *out = ctx->mutable_input(input, lock_held);
101 return Status::OK();
102}
