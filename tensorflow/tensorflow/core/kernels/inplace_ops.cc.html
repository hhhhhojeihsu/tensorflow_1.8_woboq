34template <typename Device, typename T>
35Status DoParallelConcatUpdate(const Device& d, const Tensor& value, int32 loc,
36 Tensor* output) {
37 auto Tvalue = value.shaped<T, 2>({1, value.NumElements()});
38 auto Toutput = output->flat_outer_dims<T>();
39 auto nrows = Toutput.dimension(0);
40 auto r = (loc % nrows + nrows) % nrows; // Guard index range.
41 Toutput.template chip<0>(r).device(d) = Tvalue.template chip<0>(0);
42 return Status::OK();
43}
44
45template <>
46Status DoParallelConcat(const CPUDevice& d, const Tensor& value, int32 loc,
47 Tensor* output) {
48 CHECK_EQ(value.dtype(), output->dtype());
49 switch (value.dtype()) {
50#define CASE(type) \
51 case DataTypeToEnum<type>::value: \
52 return DoParallelConcatUpdate<CPUDevice, type>(d, value, loc, output);
53 TF_CALL_NUMBER_TYPES(CASE);
54 TF_CALL_string(CASE);
55 TF_CALL_variant(CASE);
56#undef CASE
57 default:
58 return errors::InvalidArgument("Unsupported data type: ", value.dtype());
59 }
60}
61
62#ifdef TENSORFLOW_USE_SYCL
63template <>
64Status DoParallelConcat(const SyclDevice& d, const Tensor& value, int32 loc,
65 Tensor* output) {
66 CHECK_EQ(value.dtype(), output->dtype());
67 switch (value.dtype()) {
68#define CASE(type) \
69 case DataTypeToEnum<type>::value: \
70 return DoParallelConcatUpdate<SyclDevice, type>(d, value, loc, output);
71 TF_CALL_GPU_NUMBER_TYPES_NO_HALF(CASE);

83template <typename Device>
84class ParallelConcatUpdate : public OpKernel {
85 public:
86 explicit ParallelConcatUpdate(OpKernelConstruction* ctx) : OpKernel(ctx) {
87 OP_REQUIRES_OK(ctx, ctx->GetAttr("loc", &loc_));
88 }
89
90 void Compute(OpKernelContext* ctx) override {
91 auto value = ctx->input(0);
92 auto update = ctx->input(1);
93
94 OP_REQUIRES(
95 ctx, value.dims() == update.dims(),
96 errors::InvalidArgument("value and update shape doesn't match: ",
97 value.shape().DebugString(), " vs. ",
98 update.shape().DebugString()));
99 for (int i = 1; i < value.dims(); ++i) {
100 OP_REQUIRES(
101 ctx, value.dim_size(i) == update.dim_size(i),
102 errors::InvalidArgument("value and update shape doesn't match ",
103 value.shape().DebugString(), " vs. ",
104 update.shape().DebugString()));
105 }
106 OP_REQUIRES(ctx, 1 == update.dim_size(0),
107 errors::InvalidArgument("update shape doesn't match: ",
108 update.shape().DebugString()));
109
110 Tensor output = value; // This creates an alias intentionally.
111 const auto& d = ctx->eigen_device<Device>();
112 OP_REQUIRES_OK(
113 ctx, ::tensorflow::functor::DoParallelConcat(d, update, loc_, &output));
114 ctx->set_output(0, output);
115 }
116
117 private:
118 int32 loc_;
119};

251class InplaceOpBase : public OpKernel {
252 public:
253 explicit InplaceOpBase(OpKernelConstruction* ctx) : OpKernel(ctx) {}
254
255 void Compute(OpKernelContext* ctx) override {
256 auto x = ctx->input(0);
257 auto i = ctx->input(1);
258 auto v = ctx->input(2);
259
260 OP_REQUIRES(ctx, TensorShapeUtils::IsVector(i.shape()),
261 errors::InvalidArgument("i must be a vector. ",
262 i.shape().DebugString()));
263 OP_REQUIRES(ctx, x.dims() == v.dims(),
264 errors::InvalidArgument(
265 "x and v shape doesn't match (ranks differ): ",
266 x.shape().DebugString(), " vs. ", v.shape().DebugString()));
267 for (int i = 1; i < x.dims(); ++i) {
268 OP_REQUIRES(
269 ctx, x.dim_size(i) == v.dim_size(i),
270 errors::InvalidArgument("x and v shape doesn't match at index ", i,
271 " : ", x.shape().DebugString(), " vs. ",
272 v.shape().DebugString()));
273 }
274 OP_REQUIRES(ctx, i.dim_size(0) == v.dim_size(0),
275 errors::InvalidArgument(
276 "i and x shape doesn't match at index 0: ",
277 i.shape().DebugString(), " vs. ", v.shape().DebugString()));
278
279 Tensor y = x; // This creates an alias intentionally.
280 OP_REQUIRES_OK(ctx, DoCompute(ctx, i, v, &y));
281 ctx->set_output(0, y);
282 }
283
284 protected:
285 virtual Status DoCompute(OpKernelContext* ctx, const Tensor& i,
286 const Tensor& v, Tensor* y) = 0;
287};

293template <typename T>
294void DoInplaceOp(const CPUDevice& d, InplaceOpType op, const Tensor& i,
295 const Tensor& v, Tensor* y) {
296 auto Ti = i.flat<int32>();
297 auto Tv = v.flat_outer_dims<T>();
298 auto Ty = y->flat_outer_dims<T>();
299 auto nrows = Ty.dimension(0);
300 for (int64 j = 0; j < Ti.size(); ++j) {
301 auto r = (Ti(j) % nrows + nrows) % nrows; // Guard index range.
302 switch (op) {
303 case I_UPDATE:
304 Ty.template chip<0>(r).device(d) = Tv.template chip<0>(j);
305 break;
306 case I_ADD:
307 Ty.template chip<0>(r).device(d) += Tv.template chip<0>(j);
308 break;
309 case I_SUB:
310 Ty.template chip<0>(r).device(d) -= Tv.template chip<0>(j);
311 break;
312 }
313 }
314}
315
316// String type only supports inplace update.
317void DoInplaceStringUpdateOp(const CPUDevice& d, const Tensor& i,
318 const Tensor& v, Tensor* y) {
319 auto Ti = i.flat<int32>();
320 auto Tv = v.flat_outer_dims<string>();
321 auto Ty = y->flat_outer_dims<string>();
322 auto nrows = Ty.dimension(0);
323 for (int64 j = 0; j < Ti.size(); ++j) {
324 auto r = (Ti(j) % nrows + nrows) % nrows; // Guard index range.
325 Ty.template chip<0>(r).device(d) = Tv.template chip<0>(j);
326 }
327}
328
329template <>
330Status DoInplace(const CPUDevice& device, InplaceOpType op, const Tensor& i,
331 const Tensor& v, Tensor* y) {
332 CHECK_EQ(v.dtype(), y->dtype());
333 if (op == I_UPDATE) {
334 if (v.dtype() == DT_STRING) {
335 DoInplaceStringUpdateOp(device, i, v, y);
336 return Status::OK();
337 } else if (v.dtype() == DT_BOOL) {
338 DoInplaceOp<bool>(device, op, i, v, y);
339 return Status::OK();
340 }
341 }
342 switch (v.dtype()) {
343#define CASE(type) \
344 case DataTypeToEnum<type>::value: \
345 DoInplaceOp<type>(device, op, i, v, y); \
346 break;
347 TF_CALL_NUMBER_TYPES(CASE);
348#undef CASE
349 default:
350 return errors::InvalidArgument("Unsupported data type: ", v.dtype());
351 }
352 return Status::OK();
353}

371class CopyOpBase : public OpKernel {
372 public:
373 explicit CopyOpBase(OpKernelConstruction* ctx) : OpKernel(ctx) {}
374
375 void Compute(OpKernelContext* ctx) override {
376 auto x = ctx->input(0);
377 Tensor* y;
378 OP_REQUIRES_OK(ctx, ctx->allocate_output(0, x.shape(), &y));
379 OP_REQUIRES_OK(ctx, DoCompute(ctx, x, y));
380 }
381
382 protected:
383 virtual Status DoCompute(OpKernelContext* ctx, const Tensor& x,
384 Tensor* y) = 0;
385};
386
387template <typename Device>
388class CopyOp : public CopyOpBase {
389 public:
390 explicit CopyOp(OpKernelConstruction* ctx) : CopyOpBase(ctx) {}
391
392 protected:
393 Status DoCompute(OpKernelContext* ctx, const Tensor& x, Tensor* y) override {
394 const auto& d = ctx->eigen_device<Device>();
395 return ::tensorflow::functor::DoCopy(d, x, y);
396 }

405template <>
406Status DoCopy(const CPUDevice& device, const Tensor& x, Tensor* y) {
407 CHECK_EQ(x.dtype(), y->dtype());
408 switch (x.dtype()) {
409#define CASE(type) \
410 case DataTypeToEnum<type>::value: \
411 y->flat<type>().device(device) = x.flat<type>(); \
412 break;
413
414 TF_CALL_NUMBER_TYPES(CASE);
415 TF_CALL_bool(CASE);
416#undef CASE
417 default:
418 return errors::InvalidArgument("Unsupported data type: ", x.dtype());
419 }
420 return Status::OK();
421}

426template <typename Device, typename T>
427class EmptyOp : public OpKernel {
428 public:
429 explicit EmptyOp(OpKernelConstruction* ctx) : OpKernel(ctx) {
430 OP_REQUIRES_OK(ctx, ctx->GetAttr("init", &init_));
431 }
432
433 void Compute(OpKernelContext* ctx) override {
434 const Tensor& shape = ctx->input(0);
435 OP_REQUIRES(
436 ctx, TensorShapeUtils::IsVector(shape.shape()),
437 errors::InvalidArgument("shape must be a vector of int32, got shape ",
438 shape.shape().DebugString()));
439 auto dims = shape.flat<int32>();
440 TensorShape out_shape;
441 OP_REQUIRES_OK(ctx, TensorShapeUtils::MakeShape(
442 reinterpret_cast<const int32*>(dims.data()),
443 dims.size(), &out_shape));
444 Tensor* out = nullptr;
445 OP_REQUIRES_OK(ctx, ctx->allocate_output(0, out_shape, &out));
446
447 if (init_) {
448 functor::SetZeroFunctor<Device, T>()(ctx->eigen_device<Device>(),
449 out->flat<T>());
450 }
451 }
452
453 private:
454 bool init_;
455};
