32InferenceContext::InferenceContext(
33 int graph_def_version, const NodeDef* node_def, const OpDef& op_def,
34 const std::vector<TensorShapeProto>& input_shapes,
35 const std::vector<const Tensor*>& input_tensors,
36 const std::vector<TensorShapeProto>& input_tensors_as_shapes,
37 const std::vector<
38 std::unique_ptr<std::vector<std::pair<TensorShapeProto, DataType>>>>&
39 input_handle_shapes_and_types)
40 : graph_def_version_(graph_def_version),
41 node_def_(CHECK_NOTNULL(node_def)) {
42 std::vector<ShapeHandle> input_tensors_as_shape_handles;
43 for (const TensorShapeProto& p : input_tensors_as_shapes) {
44 ShapeHandle shape;
45 construction_status_.Update(MakeShapeFromShapeProto(p, &shape));
46 if (!construction_status_.ok()) {
47 return;
48 }
49 input_tensors_as_shape_handles.push_back(shape);
50 }
51 PreInputInit(op_def, input_tensors, input_tensors_as_shape_handles);
52 if (!construction_status_.ok()) return;
53 for (const TensorShapeProto& p : input_shapes) {
54 ShapeHandle shape;
55 construction_status_.Update(MakeShapeFromShapeProto(p, &shape));
56 if (!construction_status_.ok()) {
57 return;
58 }
59 inputs_.push_back(shape);
60 }
61
62 std::vector<std::unique_ptr<std::vector<ShapeAndType>>> handle_data(
63 input_shapes.size());
64 for (int i = 0; i < input_handle_shapes_and_types.size(); ++i) {
65 const auto& v = input_handle_shapes_and_types[i];
66 if (v == nullptr) {
67 continue;
68 }
69 handle_data[i].reset(new std::vector<ShapeAndType>(v->size()));
70 auto& new_v = *handle_data[i];
71 for (int j = 0; j < v->size(); ++j) {
72 const auto& p = (*v)[j];
73 construction_status_.Update(
74 MakeShapeFromShapeProto(p.first, &new_v[j].shape));
75 if (!construction_status_.ok()) {
76 return;
77 }
78 new_v[j].dtype = p.second;
79 }
80 }
81 PostInputInit(std::move(handle_data));
82}

154Status InferenceContext::Run(
155 const std::function<Status(shape_inference::InferenceContext* c)>& fn) {
156 Status s = fn(this);
157 if (!s.ok()) {
158 return AttachContext(s);
159 }
160#ifndef NDEBUG
161 for (int i = 0; i < num_outputs(); ++i) {
162 DCHECK(output(i).IsSet())
163 << i << " for " << node_def_->name() << " of type " << node_def_->op();
164 }
165#endif // NDEBUG
166 return s;
167}

169Status InferenceContext::set_output(StringPiece output_name,
170 const std::vector<ShapeHandle>& shapes) {
171 auto result = output_name_map_.find(output_name);
172 if (result == output_name_map_.end()) {
173 return errors::InvalidArgument("Unknown output name: ", output_name);
174 } else {
175 const int start = result->second.first;
176 const int size = result->second.second - start;
177 if (size != shapes.size()) {
178 return errors::InvalidArgument("Must have exactly ", shapes.size(),
179 " shapes.");
180 }
181 for (int i = 0; i < size; ++i) {
182 outputs_[i + start] = shapes[i];
183 }
184 }
185 return Status::OK();
186}
187
188Status InferenceContext::input(StringPiece input_name,
189 std::vector<ShapeHandle>* output) const {
190 const auto result = input_name_map_.find(input_name);
191 if (result == input_name_map_.end()) {
192 return errors::InvalidArgument("Unknown input name: ", input_name);
193 } else {
194 output->clear();
195 for (int i = result->second.first; i < result->second.second; ++i) {
196 output->push_back(inputs_[i]);
197 }
198 }
199 return Status::OK();
200}
201
202Status InferenceContext::output(StringPiece output_name,
203 std::vector<ShapeHandle>* output) const {
204 const auto result = output_name_map_.find(output_name);
205 if (result == output_name_map_.end()) {
206 return errors::InvalidArgument("Unknown output name: ", output_name);
207 } else {
208 output->clear();
209 for (int i = result->second.first; i < result->second.second; ++i) {
210 output->push_back(outputs_[i]);
211 }
212 }
213 return Status::OK();
214}

218void InferenceContext::PreInputInit(
219 const OpDef& op_def, const std::vector<const Tensor*>& input_tensors,
220 const std::vector<ShapeHandle>& input_tensors_as_shapes) {
221 input_tensors_ = input_tensors;
222 input_tensors_as_shapes_ = input_tensors_as_shapes;
223
224 construction_status_ = NameRangesForNode(*node_def_, op_def, &input_name_map_,
225 &output_name_map_);
226 if (!construction_status_.ok()) return;
227
228 int num_outputs = 0;
229 for (const auto& e : output_name_map_) {
230 num_outputs = std::max(num_outputs, e.second.second);
231 }
232 for (int i = 0; i < num_outputs; ++i) {
233 outputs_.push_back(nullptr);
234 }
235 output_handle_shapes_and_types_.resize(num_outputs);
236}
237
238void InferenceContext::PostInputInit(
239 std::vector<std::unique_ptr<std::vector<ShapeAndType>>> input_handle_data) {
240 int num_inputs_from_node_def = 0;
241 for (const auto& e : input_name_map_) {
242 num_inputs_from_node_def =
243 std::max(num_inputs_from_node_def, e.second.second);
244 }
245
246 // Allow passing empty shapes/dtypes to avoid changing every single test.
247 if (input_handle_data.empty()) {
248 input_handle_shapes_and_types_.resize(inputs_.size());
249 } else {
250 if (input_handle_data.size() != inputs_.size()) {
251 construction_status_ = errors::InvalidArgument(
252 "Wrong number of handle shapes passed; expected ", inputs_.size(),
253 " got ", input_handle_data.size());
254 return;
255 }
256 input_handle_shapes_and_types_ = std::move(input_handle_data);
257 }
258
259 if (inputs_.size() != num_inputs_from_node_def) {
260 construction_status_ = errors::InvalidArgument(
261 "Wrong number of inputs passed: ", inputs_.size(), " while ",
262 num_inputs_from_node_def, " expected based on NodeDef");
263 return;
264 }
265
266 CHECK_LE(input_tensors_.size(), inputs_.size());
267 input_tensors_.resize(inputs_.size());
268 requested_input_tensor_.resize(inputs_.size());
269 requested_input_tensor_as_partial_shape_.resize(inputs_.size());
270}

272void InferenceContext::ShapeHandleToProto(ShapeHandle handle,
273 TensorShapeProto* proto) {
274 if (!RankKnown(handle)) {
275 proto->set_unknown_rank(true);
276 return;
277 }
278
279 for (int32 i = 0; i < Rank(handle); ++i) {
280 DimensionHandle dim = Dim(handle, i);
281 auto* dim_shape = proto->add_dim();
282 if (ValueKnown(dim)) {
283 dim_shape->set_size(Value(dim));
284 } else {
285 dim_shape->set_size(-1);
286 }
287 }
288}

290bool InferenceContext::FullyDefined(ShapeHandle s) {
291 if (!RankKnown(s)) return false;
292 for (int i = 0; i < Rank(s); ++i) {
293 if (!ValueKnown(Dim(s, i))) return false;
294 }
295 return true;
296}

298DimensionHandle InferenceContext::NumElements(ShapeHandle s) {
299 const auto rank = Rank(s);
300 if (rank == kUnknownRank) return UnknownDim();
301 bool found_unknown = false;
302 int64 size = 1;
303 for (int i = 0; i < rank; ++i) {
304 int64 dim_val = Value(Dim(s, i));
305 if (dim_val == kUnknownDim) {
306 found_unknown = true;
307 } else if (dim_val == 0) {
308 return MakeDim(0);
309 } else {
310 size *= dim_val;
311 }
312 }
313 if (found_unknown) {
314 return UnknownDim();
315 } else {
316 return MakeDim(size);
317 }
318}

339Status InferenceContext::WithRank(ShapeHandle shape, int64 rank,
340 ShapeHandle* out) {
341 if (rank > kint32max) {
342 return errors::InvalidArgument("Rank cannot exceed kint32max");
343 }
344 const int32 existing = Rank(shape);
345 if (existing == rank) {
346 *out = shape;
347 return Status::OK();
348 }
349 if (existing == kUnknownRank) {
350 std::vector<DimensionHandle> dims;
351 dims.reserve(rank);
352 for (int i = 0; i < rank; ++i) {
353 dims.push_back(UnknownDim());
354 }
355 ShapeHandle shp = shape_manager_.MakeShape(dims);
356 return Merge(shape, shp, out);
357 }
358 *out = nullptr;
359
360 return errors::InvalidArgument("Shape must be rank ", rank, " but is rank ",
361 existing);
362}
363
364Status InferenceContext::WithRankAtLeast(ShapeHandle shape, int64 rank,
365 ShapeHandle* out) {
366 if (rank > kint32max) {
367 return errors::InvalidArgument("Rank cannot exceed kint32max");
368 }
369 const int32 existing = Rank(shape);
370 if (existing >= rank || existing == kUnknownRank) {
371 *out = shape;
372 return Status::OK();
373 }
374 *out = nullptr;
375 return errors::InvalidArgument("Shape must be at least rank ", rank,
376 " but is rank ", existing);
377}
378
379Status InferenceContext::WithRankAtMost(ShapeHandle shape, int64 rank,
380 ShapeHandle* out) {
381 if (rank > kint32max) {
382 return errors::InvalidArgument("Rank cannot exceed kint32max");
383 }
384 const int32 existing = Rank(shape);
385 if (existing <= rank || existing == kUnknownRank) {
386 *out = shape;
387 return Status::OK();
388 }
389 *out = nullptr;
390 return errors::InvalidArgument("Shape must be at most rank ", rank,
391 " but is rank ", existing);
392}

394Status InferenceContext::WithValue(DimensionHandle dim, int64 value,
395 DimensionHandle* out) {
396 const int64 existing = Value(dim);
397 if (existing == value) {
398 *out = dim;
399 return Status::OK();
400 }
401 if (existing == kUnknownDim) {
402 DimensionHandle d = MakeDim(value);
403 return Merge(dim, d, out);
404 }
405 *out = nullptr;
406 return errors::InvalidArgument("Dimension must be ", value, " but is ",
407 existing);
408}

410void InferenceContext::Relax(DimensionHandle d_old, DimensionHandle d_new,
411 DimensionHandle* out) {
412 if (d_old.SameHandle(d_new)) {
413 *out = d_old;
414 } else if (!ValueKnown(d_old) && !ValueKnown(d_new)) {
415 // The node will be fed by the dimension d_new instead of d_old: any
416 // equality assertion between d_old and other input dimension on this node
417 // may not be true anymore, so forget them all.
418 ForgetMerges();
419 // Return the new shape handle to force the relaxation to propagate to the
420 // fanout of the context.
421 *out = d_new;
422 } else if (!ValueKnown(d_new)) {
423 ForgetMerges();
424 *out = d_new;
425 } else if (Value(d_old) == Value(d_new)) {
426 // Return the old shape handle. This will stop the relaxation in the fanout
427 // of the context.
428 *out = d_old;
429 } else {
430 // Return a new handle that encodes a different unknown dim.
431 ForgetMerges();
432 *out = UnknownDim();
433 }
434}

436Status InferenceContext::Merge(DimensionHandle d0, DimensionHandle d1,
437 DimensionHandle* out) {
438 if (d0.SameHandle(d1)) {
439 *out = d0;
440 return Status::OK();
441 } else if (!ValueKnown(d1)) {
442 *out = d0;
443 merged_dims_.emplace_back(d0, d1);
444 return Status::OK();
445 } else if (!ValueKnown(d0)) {
446 *out = d1;
447 merged_dims_.emplace_back(d0, d1);
448 return Status::OK();
449 } else if (Value(d0) == Value(d1)) {
450 *out = d0;
451 return Status::OK();
452 } else {
453 *out = nullptr;
454 return errors::InvalidArgument("Dimensions must be equal, but are ",
455 Value(d0), " and ", Value(d1));
456 }
457}

459Status InferenceContext::MergePrefix(ShapeHandle s, ShapeHandle prefix,
460 ShapeHandle* s_out,
461 ShapeHandle* prefix_out) {
462 *s_out = *prefix_out = nullptr;
463 if (!RankKnown(prefix) || !RankKnown(s)) {
464 *s_out = s;
465 *prefix_out = prefix;
466 return Status::OK();
467 }
468 const int32 rank = Rank(prefix);
469 TF_RETURN_IF_ERROR(WithRankAtLeast(s, rank, &s));
470
471 // Merge the prefix dims and create the new output shapes.
472 std::vector<DimensionHandle> dims;
473 dims.resize(rank);
474 for (int i = 0; i < rank; ++i) {
475 TF_RETURN_IF_ERROR(Merge(Dim(s, i), Dim(prefix, i), &dims[i]));
476 }
477 *prefix_out = MakeShape(dims);
478 for (int i = rank; i < Rank(s); ++i) dims.push_back(Dim(s, i));
479 *s_out = MakeShape(dims);
480 return Status::OK();
481}

483void InferenceContext::Relax(ShapeHandle s_old, ShapeHandle s_new,
484 ShapeHandle* out) {
485 if (s_old.SameHandle(s_new)) {
486 *out = s_old;
487 return;
488 } else if (!RankKnown(s_new) || !s_old.IsSet()) {
489 ForgetMerges();
490 *out = s_new;
491 return;
492 }
493
494 const int32 rank = Rank(s_old);
495 if (rank != Rank(s_new)) {
496 ForgetMerges();
497 *out = UnknownShape();
498 return;
499 }
500
501 bool return_s_old = true;
502 for (int i = 0; i < rank; ++i) {
503 auto d0 = Dim(s_old, i);
504 auto d1 = Dim(s_new, i);
505 if (d0.SameHandle(d1)) continue;
506
507 auto v0 = Value(d0);
508 auto v1 = Value(d1);
509 if (v0 == kUnknownDim || v1 == kUnknownDim || v0 != v1) {
510 return_s_old = false;
511 break;
512 }
513 }
514 if (return_s_old) {
515 *out = s_old;
516 return;
517 }
518
519 // Relax dims.
520 std::vector<DimensionHandle> dims(rank);
521 for (int i = 0; i < rank; ++i) {
522 Relax(Dim(s_old, i), Dim(s_new, i), &dims[i]);
523 }
524 ForgetMerges();
525 *out = MakeShape(dims);
526}

528Status InferenceContext::Merge(ShapeHandle s0, ShapeHandle s1,
529 ShapeHandle* out) {
530 if (s0.SameHandle(s1)) {
531 *out = s0;
532 return Status::OK();
533 } else if (!RankKnown(s1)) {
534 *out = s0;
535 merged_shapes_.emplace_back(s0, s1);
536 return Status::OK();
537 } else if (!RankKnown(s0)) {
538 *out = s1;
539 merged_shapes_.emplace_back(s0, s1);
540 return Status::OK();
541 }
542
543 const int32 rank = Rank(s0);
544 if (rank != Rank(s1)) {
545 *out = nullptr;
546 return errors::InvalidArgument("Shapes must be equal rank, but are ", rank,
547 " and ", Rank(s1));
548 }
549
550 bool return_s0 = true;
551 bool return_s1 = true;
552 for (int i = 0; i < rank; ++i) {
553 auto d0 = Dim(s0, i);
554 auto d1 = Dim(s1, i);
555 if (d0.SameHandle(d1)) continue;
556
557 auto v0 = Value(d0);
558 auto v1 = Value(d1);
559 if (v0 == kUnknownDim) {
560 if (v1 != kUnknownDim) {
561 return_s0 = false;
562 }
563 } else if (v1 == kUnknownDim) {
564 return_s1 = false;
565 } else if (v0 != v1) {
566 *out = nullptr;
567 return errors::InvalidArgument(
568 "Dimension ", i, " in both shapes must be equal, but are ", Value(d0),
569 " and ", Value(d1), ". Shapes are ", DebugString(s0), " and ",
570 DebugString(s1), ".");
571 }
572 }
573
574 merged_shapes_.emplace_back(s0, s1);
575
576 if (return_s0 || return_s1) {
577 *out = return_s0 ? s0 : s1;
578 return Status::OK();
579 }
580
581 // Merge dims.
582 std::vector<DimensionHandle> dims(rank, nullptr);
583 for (int i = 0; i < rank; ++i) {
584 // Invariant for merge was checked earlier, so CHECK is ok.
585 TF_CHECK_OK(Merge(Dim(s0, i), Dim(s1, i), &dims[i]));
586 }
587
588 Status s = ReturnCreatedShape(dims, out);
589 if (s.ok()) {
590 // Merge the new shape with s0. Since s0 and s1 are merged, this implies
591 // that s1 and out are also merged.
592 merged_shapes_.emplace_back(s0, *out);
593 }
594 return s;
595}

598 ShapeHandle* out) {
599 return Subshape(s, start, std::numeric_limits<int64>::max() /* end */, out);

601
602Status InferenceContext::Subshape(ShapeHandle s, int64 start_in, int64 end_in,
603 ShapeHandle* out) {
604 int64 start = start_in;
605 int64 end = end_in;
606 const int32 rank = Rank(s);
607 if (start == 0 && ((RankKnown(s) && end >= rank) ||
608 end == std::numeric_limits<int64>::max())) {
609 *out = s;
610 return Status::OK();
611 }
612 if (!RankKnown(s)) {
613 return ReturnUnknownShape(out);
614 }
615
616 if (start > rank) start = rank;
617 if (end > rank) end = rank;
618 if (start < 0) {
619 start = rank + start;
620 if (start < 0) {
621 *out = nullptr;
622 return errors::InvalidArgument("Subshape start out of bounds: ", start_in,
623 ", for shape with rank ", rank);
624 }
625 }
626
627 if (end < 0) {
628 end = rank + end;
629 if (end < 0) {
630 *out = nullptr;
631 return errors::InvalidArgument("Subshape end out of bounds: ", end_in,
632 ", for shape with rank ", rank);
633 }
634 }
635 if (start > end) {
636 *out = nullptr;
637 return errors::InvalidArgument(
638 "Subshape must have computed start <= end, but is ", start, " and ",
639 end, " (computed from start ", start_in, " and end ", end_in,
640 " over shape with rank ", rank, ")");
641 }
642 std::vector<DimensionHandle> dims;
643 dims.reserve(end - start);
644 for (int i = start; i < end; ++i) {
645 dims.push_back(Dim(s, i));
646 }
647 return ReturnCreatedShape(dims, out);
648}

650Status InferenceContext::Concatenate(ShapeHandle s1, ShapeHandle s2,
651 ShapeHandle* out) {
652 if (!RankKnown(s1) || !RankKnown(s2)) {
653 return ReturnUnknownShape(out);
654 }
655 const int32 s1_rank = Rank(s1);
656 const int32 s2_rank = Rank(s2);
657 const int32 rank = s1_rank + s2_rank;
658 std::vector<DimensionHandle> dims;
659 dims.reserve(rank);
660 for (int i = 0; i < s1_rank; ++i) dims.push_back(Dim(s1, i));
661 for (int i = 0; i < s2_rank; ++i) dims.push_back(Dim(s2, i));
662 return ReturnCreatedShape(dims, out);
663}

665Status InferenceContext::ReplaceDim(ShapeHandle s, int64 dim_index_in,
666 DimensionHandle new_dim, ShapeHandle* out) {
667 if (!RankKnown(s)) {
668 return ReturnUnknownShape(out);
669 }
670 int64 dim_index = dim_index_in;
671 if (dim_index < 0) {
672 dim_index = s->dims_.size() + dim_index;
673 }
674 if (!FastBoundsCheck(dim_index, s->dims_.size())) {
675 *out = nullptr;
676 return errors::InvalidArgument("Out of range dim_index ", dim_index_in,
677 " for shape with ", s->dims_.size(),
678 " dimensions");
679 }
680 std::vector<DimensionHandle> dims(s->dims_);
681 dims[dim_index] = new_dim;
682 return ReturnCreatedShape(dims, out);
683}

773Status InferenceContext::InternalMakeShapeFromTensor(
774 bool treat_unknown_scalar_tensor_as_unknown_shape, const Tensor* t,
775 ShapeHandle tensor_shape, ShapeHandle* out) {
776 // Only callers who have set
777 if (!treat_unknown_scalar_tensor_as_unknown_shape) {
778 TF_RETURN_IF_ERROR(WithRank(tensor_shape, 1, &tensor_shape));
779 }
780 if (t == nullptr) {
781 // This is guarded by the check above.
782 if (Rank(tensor_shape) == 0) {
783 return ReturnUnknownShape(out);
784 }
785 // Shape tensor is not known, but if the shape of the shape tensor is then
786 // the right number of unknown dims can be created.
787 DimensionHandle shape_dim = Dim(tensor_shape, 0);
788 if (!ValueKnown(shape_dim)) {
789 return ReturnUnknownShape(out);
790 }
791 const auto num_dims = Value(shape_dim);
792 std::vector<DimensionHandle> dims;
793 dims.reserve(num_dims);
794 for (int i = 0; i < num_dims; i++) dims.push_back(UnknownDim());
795 return ReturnCreatedShape(dims, out);
796 }
797
798 if (t->shape().dims() == 0) {
799 if (t->dtype() == DataType::DT_INT32) {
800 auto flat_t = t->scalar<int32>();
801 if (flat_t() != -1) {
802 *out = nullptr;
803 return errors::InvalidArgument(
804 "Input tensor must be rank 1, or if its rank 0 it must have value "
805 "-1 "
806 "(representing an unknown shape). Saw value: ",
807 flat_t());
808 }
809 return ReturnUnknownShape(out);
810 } else if (t->dtype() == DataType::DT_INT64) {
811 auto flat_t = t->scalar<int64>();
812 if (flat_t() != -1) {
813 *out = nullptr;
814 return errors::InvalidArgument(
815 "Input tensor must be rank 1, or if its rank 0 it must have value "
816 "-1 "
817 "(representing an unknown shape). Saw value: ",
818 flat_t());
819 }
820 return ReturnUnknownShape(out);
821 } else {
822 *out = nullptr;
823 return errors::InvalidArgument(
824 "Input tensor must be int32 or int64, but was ",
825 DataTypeString(t->dtype()));
826 }
827 }
828
829 if (t->shape().dims() != 1) {
830 *out = nullptr;
831 return errors::InvalidArgument(
832 "Input tensor must be rank 1, but was rank ", t->shape().dims(), ".",
833 ((t->shape().dims() == 0)
834 ? "If it is rank 0 rank 0 it must have statically known value -1 "
835 "(representing an unknown shape). "
836 : " "),
837 "Saw tensor shape ", t->shape().DebugString());
838 }
839 std::vector<DimensionHandle> dims;
840 if (t->dtype() == DataType::DT_INT32) {
841 auto flat_t = t->flat<int32>();
842 for (int i = 0; i < flat_t.size(); ++i) {
843 const int32 val = flat_t(i);
844 if (val < -1) {
845 return errors::InvalidArgument(
846 "Invalid value in tensor used for shape: ", val);
847 }
848 // -1 will become an unknown dim.
849 dims.push_back(MakeDim(val));
850 }
851 } else if (t->dtype() == DataType::DT_INT64) {
852 auto flat_t = t->flat<int64>();
853 for (int i = 0; i < flat_t.size(); ++i) {
854 const int64 val = flat_t(i);
855 if (val < -1) {
856 return errors::InvalidArgument(
857 "Invalid value in tensor used for shape: ", val);
858 }
859 // -1 will become an unknown dim.
860 dims.push_back(MakeDim(val));
861 }
862 } else {
863 *out = nullptr;
864 return errors::InvalidArgument(
865 "Input tensor must be int32 or int64, but was ",
866 DataTypeString(t->dtype()));
867 }
868
869 return ReturnCreatedShape(dims, out);
870}

872Status InferenceContext::MakeShapeFromPartialTensorShape(
873 const PartialTensorShape& partial_shape, ShapeHandle* out) {
874 *out = nullptr;
875 if (partial_shape.dims() == -1) {
876 return ReturnUnknownShape(out);
877 }
878 const int num_dims = partial_shape.dims();
879 std::vector<DimensionHandle> dims(num_dims);

881 // -1 is unknown in PartialTensorShape and in InferenceContext, so this size
882 // can be passed directly to MakeDim.
883 dims[i] = MakeDim(partial_shape.dim_size(i));
884 }
885 return ReturnCreatedShape(dims, out);

902Status InferenceContext::GetScalarFromTensor(const Tensor* t, int64* val) {
903 // Caller must ensure that <t> is not NULL.
904 const int rank = t->dims();

938Status InferenceContext::MakeDimForScalarInputWithNegativeIndexing(
939 int idx, int input_rank, DimensionHandle* out) {
940 int64 val;
941 const Tensor* t = input_tensor(idx);
942 if (t == nullptr) {
943 *out = UnknownDim();
944 return Status::OK();
945 }
946 TF_RETURN_IF_ERROR(GetScalarFromTensor(t, &val));
947 if (val < 0) {

949 *out = UnknownDim();
950 return Status::OK();

952 return errors::InvalidArgument("Dimension size, given by scalar input ",
953 val, " must be in range [-", input_rank,
954 ", ", input_rank, ")");
955 } else {
956 val += input_rank;
957 }
958 } else if (input_rank >= 0 && val >= input_rank) {
959 return errors::InvalidArgument("Dimension size, given by scalar input ",
960 val, " must be in range [-", input_rank,
961 ", ", input_rank, ")");
962 }
963 *out = MakeDim(val);
964 return Status::OK();

967Status InferenceContext::Divide(DimensionHandle dividend,
968 DimensionOrConstant divisor,
969 bool evenly_divisible, DimensionHandle* out) {
970 const int64 divisor_value = Value(divisor);
971 if (divisor_value == 1) {
972 *out = dividend;
973 } else if (!ValueKnown(dividend) ||
974 (divisor.dim.IsSet() && !ValueKnown(divisor.dim))) {
975 *out = UnknownDim();
976 } else {
977 const int64 v = Value(dividend);
978 if (divisor_value <= 0) {
979 return errors::InvalidArgument("Divisor must be positive but is ",
980 divisor_value);
981 }
982 if (evenly_divisible && (v % divisor_value) != 0) {
983 return errors::InvalidArgument(
984 "Dimension size must be evenly divisible by ", divisor_value,
985 " but is ", v);
986 }
987 *out = MakeDim(v / divisor_value);
988 }
989 return Status::OK();
990}

992Status InferenceContext::Add(DimensionHandle first, DimensionOrConstant second,
993 DimensionHandle* out) {
994 const int64 first_value = Value(first);
995 const int64 second_value = Value(second);
996 // Special cases.
997 if (first_value == 0) {
998 *out = MakeDim(second);
999 } else if (second_value == 0) {
1000 *out = first;
1001 } else if (first_value == kUnknownDim || second_value == kUnknownDim) {
1002 *out = UnknownDim();
1003 } else {
1004 // Invariant: Both values are known and positive. Still in run-time we can
1005 // get pair of values which cannot be store in output. Check below will
1006 // report error. We still need to avoid undefined behavior of signed
1007 // overflow and use unsigned addition.
1008 const int64 sum = static_cast<uint64>(first_value) + second_value;
1009 if (sum < 0) {
1010 return errors::InvalidArgument("Dimension size overflow from adding ",
1011 first_value, " and ", second_value);
1012 }
1013 *out = MakeDim(sum);
1014 }
1015 return Status::OK();
1016}

1018Status InferenceContext::Subtract(DimensionHandle first,
1019 DimensionOrConstant second,
1020 DimensionHandle* out) {
1021 const int64 first_value = Value(first);
1022 const int64 second_value = Value(second);
1023 // Special cases.
1024 if (second_value == 0) {
1025 *out = first;
1026 } else if (first_value == kUnknownDim || second_value == kUnknownDim) {
1027 *out = UnknownDim();
1028 } else {
1029 // Invariant: Both values are known, first_value is non-negative, and
1030 // second_value is positive.
1031 if (first_value < second_value) {
1032 return errors::InvalidArgument(
1033 "Negative dimension size caused by subtracting ", second_value,
1034 " from ", first_value);
1035 }
1036 *out = MakeDim(first_value - second_value);
1037 }
1038 return Status::OK();
1039}

1041Status InferenceContext::Multiply(DimensionHandle first,
1042 DimensionOrConstant second,
1043 DimensionHandle* out) {
1044 const int64 first_value = Value(first);
1045 const int64 second_value = Value(second);
1046 // Special cases.
1047 if (first_value == 0) {
1048 *out = first;
1049 } else if (second_value == 0) {
1050 *out = MakeDim(second);
1051 } else if (first_value == 1) {
1052 *out = MakeDim(second);
1053 } else if (second_value == 1) {
1054 *out = first;
1055 } else if (first_value == kUnknownDim || second_value == kUnknownDim) {
1056 *out = UnknownDim();
1057 } else {
1058 // Invariant: Both values are known and greater than 1.
1059 const int64 product = first_value * second_value;
1060 if (product < 0) {
1061 return errors::InvalidArgument(
1062 "Negative dimension size caused by overflow when multiplying ",
1063 first_value, " and ", second_value);
1064 }
1065 *out = MakeDim(product);
1066 }
1067 return Status::OK();

1106Status InferenceContext::AttachContext(const Status& status) {
1107 std::vector<string> input_shapes;
1108 for (const ShapeHandle& input_shape : inputs_) {
1109 input_shapes.emplace_back(DebugString(input_shape));
1110 }
1111
1112 // Add information about the input tensors and partial tensor shapes used.
1113 std::vector<string> input_from_tensors_str;
1114 std::vector<string> input_from_tensors_as_shape_str;
1115 for (int i = 0; i < inputs_.size(); ++i) {
1116 if (requested_input_tensor_as_partial_shape_[i] &&
1117 i < input_tensors_as_shapes_.size() &&
1118 input_tensors_as_shapes_[i].IsSet() &&
1119 RankKnown(input_tensors_as_shapes_[i])) {
1120 input_from_tensors_as_shape_str.push_back(strings::StrCat(
1121 "input[", i, "] = ", DebugString(input_tensors_as_shapes_[i])));
1122 } else if (requested_input_tensor_[i] && i < input_tensors_.size() &&
1123 input_tensors_[i] != nullptr) {
1124 input_from_tensors_str.push_back(strings::StrCat(
1125 "input[", i, "] = <",
1126 input_tensors_[i]->SummarizeValue(256 /* max_values */), ">"));
1127 }
1128 }
1129
1130 string error_context = strings::StrCat(
1131 " for '", node_def_->name(), "' (op: '", node_def_->op(),
1132 "') with input shapes: ", str_util::Join(input_shapes, ", "));
1133 if (!input_from_tensors_str.empty()) {
1134 strings::StrAppend(&error_context, " and with computed input tensors: ",
1135 str_util::Join(input_from_tensors_str, ", "));
1136 }
1137 if (!input_from_tensors_as_shape_str.empty()) {
1138 strings::StrAppend(&error_context,
1139 " and with input tensors computed as partial shapes: ",
1140 str_util::Join(input_from_tensors_as_shape_str, ","));
1141 }
1142
1143 strings::StrAppend(&error_context, ".");
1144 return Status(status.code(),
1145 strings::StrCat(status.error_message(), error_context));
1146}

1148bool InferenceContext::MergeHandleShapesAndTypes(
1149 const std::vector<ShapeAndType>& shapes_and_types,
1150 std::vector<ShapeAndType>* to_update) {
1151 if (shapes_and_types.size() != to_update->size()) {
1152 return false;
1153 }
1154 std::vector<ShapeAndType> new_values(shapes_and_types.size());
1155 bool refined = false;
1156 for (int i = 0; i < shapes_and_types.size(); ++i) {
1157 const ShapeAndType& existing = (*to_update)[i];
1158 if (shapes_and_types[i].dtype == existing.dtype) {
1159 new_values[i].dtype = existing.dtype;
1160 } else {
1161 if (existing.dtype != DT_INVALID) {
1162 return false;
1163 } else {
1164 new_values[i].dtype = shapes_and_types[i].dtype;
1165 refined = true;
1166 }
1167 }
1168 if (!Merge(existing.shape, shapes_and_types[i].shape, &new_values[i].shape)
1169 .ok()) {
1170 // merge failed, ignore the new value.
1171 new_values[i].shape = existing.shape;
1172 }
1173 if (!existing.shape.SameHandle(new_values[i].shape)) {
1174 refined = true;
1175 }
1176 }
1177 if (!refined) {
1178 return false;
1179 }
1180 for (int i = 0; i < new_values.size(); ++i) {
1181 (*to_update)[i] = new_values[i];
1182 }
1183 return true;
1184}
