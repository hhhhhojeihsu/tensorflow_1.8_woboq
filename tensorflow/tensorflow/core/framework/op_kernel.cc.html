48Status MatchSignatureHelper(const DataTypeSlice expected_inputs,
49 const DataTypeSlice expected_outputs,
50 const DataTypeSlice inputs,
51 const DataTypeSlice outputs) {
52 bool signature_mismatch = false;
53
54 if (inputs.size() != expected_inputs.size()) signature_mismatch = true;
55 for (size_t i = 0; !signature_mismatch && i < inputs.size(); ++i) {
56 if (!TypesCompatible(expected_inputs[i], inputs[i])) {
57 signature_mismatch = true;
58 }
59 }
60
61 if (outputs.size() != expected_outputs.size()) signature_mismatch = true;
62 for (size_t i = 0; !signature_mismatch && i < outputs.size(); ++i) {
63 if (!TypesCompatible(expected_outputs[i], outputs[i])) {
64 signature_mismatch = true;
65 }
66 }
67
68 if (signature_mismatch) {
69 return errors::InvalidArgument(
70 "Signature mismatch, have: ", DataTypeSliceString(inputs), "->",
71 DataTypeSliceString(outputs),
72 " expected: ", DataTypeSliceString(expected_inputs), "->",
73 DataTypeSliceString(expected_outputs));
74 }
75 return Status::OK();
76}

121Status OpKernel::InputRange(StringPiece input_name, int* start,
122 int* stop) const {
123 const auto result = input_name_map_.find(input_name);
124 if (result == input_name_map_.end()) {
125 return errors::InvalidArgument("Unknown input name: ", input_name);
126 } else {
127 *start = result->second.first;
128 *stop = result->second.second;
129 return Status::OK();
130 }
131}

137 return errors::InvalidArgument("Unknown output name: ", output_name);
138 } else {
139 *start = result->second.first;
140 *stop = result->second.second;
141 return Status::OK();
142 }

145Status OpKernel::MakeShape(const Tensor& shape, TensorShape* out) const {
146 if (!IsLegacyVector(shape.shape())) {
147 return errors::InvalidArgument(
148 "shape must be a vector of {int32,int64}, got shape ",
149 shape.shape().DebugString());
150 }
151 if (shape.dtype() == DataType::DT_INT32) {
152 auto vec = shape.flat<int32>();
153 return TensorShapeUtils::MakeShape(vec.data(), vec.size(), out);
154 } else if (shape.dtype() == DataType::DT_INT64) {
155 auto vec = shape.flat<int64>();
156 return TensorShapeUtils::MakeShape(vec.data(), vec.size(), out);
157 } else {
158 return errors::InvalidArgument("shape must be a vector of {int32,int64}.");
159 }
160}

217Status OpKernelConstruction::allocate_temp(DataType type,
218 const TensorShape& shape,
219 Tensor* out_temp) {
220 AllocationAttributes attr;
221 attr.allocation_will_be_logged = true;
222 Tensor new_temp(allocator_, type, shape, attr);
223
224 if (!new_temp.IsInitialized()) {
225 return errors::ResourceExhausted(
226 "OOM when allocating temporary tensor with shape", shape.DebugString());
227 }
228 if (LogMemory::IsEnabled()) {
229 LogMemory::RecordTensorAllocation(
230 def_->name(), LogMemory::OP_KERNEL_CONSTRUCTION_STEP_ID, new_temp);
231 }
232 *out_temp = new_temp;
233 return Status::OK();
234}
235
236Status OpKernelConstruction::allocate_persistent(
237 DataType type, const TensorShape& shape, PersistentTensor* out_persistent,
238 Tensor** out_tensor) {
239 // for now just do the same thing as allocate_temp
240 // TODO(misard) add specific memory tracking for persistent tensors
241 Tensor persistent;
242 Status s = allocate_temp(type, shape, &persistent);
243 if (!s.ok()) {
244 return s;
245 }
246 *out_persistent = PersistentTensor(persistent);
247 Tensor* allocated = out_persistent->AccessTensor(this);
248 if (out_tensor) {
249 *out_tensor = allocated;
250 }
251 return s;
252}

275OpKernelContext::~OpKernelContext() {
276 for (TensorValue& value : outputs_) {
277 if (!value.is_ref()) {
278 delete value.tensor;
279 }
280 }
281 if (params_->record_tensor_accesses) referenced_tensors_.Destroy();
282}

284Allocator* OpKernelContext::get_allocator(AllocatorAttributes attr) {
285 Allocator* allocator = nullptr;
286 if (attr.scope_id > 0) {
287 allocator = params_->device->GetScopedAllocator(attr, step_id());
288 CHECK(allocator);
289 } else {
290 allocator = params_->device->GetStepAllocator(attr, resource_manager());
291 }
292 if (track_allocations()) {
293 mutex_lock lock(mu_);
294 for (const auto& wrapped : wrapped_allocators_) {
295 if (wrapped.first == allocator) {
296 return wrapped.second;
297 }
298 }
299 TrackingAllocator* wrapped_allocator =
300 new TrackingAllocator(allocator, params_->track_allocations);
301 wrapped_allocators_.push_back(std::make_pair(allocator, wrapped_allocator));
302 return wrapped_allocator;
303 } else {
304 return allocator;
305 }
306}

318Status OpKernelContext::input(StringPiece name, const Tensor** tensor) {
319 int start, stop;
320 TF_RETURN_IF_ERROR(params_->op_kernel->InputRange(name, &start, &stop));
321 if (stop != start + 1) {
322 return errors::InvalidArgument("OpKernel used list-valued input name '",
323 name,
324 "' when single-valued input was "
325 "expected");
326 }
327 if (input_is_ref(start)) {
328 return errors::InvalidArgument("OpKernel used ref input name '", name,
329 "' when non-ref input was expected");
330 }
331 *tensor = (*params_->inputs)[start].tensor;
332 record_tensor_reference(**tensor);
333 return Status::OK();
334}

416bool OpKernelContext::forward_input_to_output_with_shape(
417 int input_index, int output_index, const TensorShape& output_shape,
418 Tensor** output) {
419 const auto output_attr = params_->output_attr_array == nullptr
420 ? AllocatorAttributes()
421 : output_alloc_attr(output_index);
422 std::unique_ptr<Tensor> new_tensor = forward_input(
423 input_index, output_index, expected_output_dtype(output_index),
424 output_shape, output_memory_type(output_index), output_attr);
425 if (new_tensor != nullptr) {
426 // Transfer ownership to the output slot in OpKernelContext.
427 outputs_[output_index] = TensorValue(new_tensor.release());
428 *output = outputs_[output_index].tensor;
429 return true;
430 } else {
431 return false;
432 }
433}

435Status OpKernelContext::forward_input_to_output_with_shape(
436 StringPiece input_name, StringPiece output_name,
437 const TensorShape& output_shape, Tensor** output) {
438 int input_index, output_index, stop;
439 TF_RETURN_IF_ERROR(
440 params_->op_kernel->InputRange(input_name, &input_index, &stop));
441 if (stop != input_index + 1) {
442 return errors::InvalidArgument("OpKernel used list-valued input name '",
443 input_name,
444 "' when single-valued input was "
445 "expected");
446 }
447 TF_RETURN_IF_ERROR(
448 params_->op_kernel->OutputRange(output_name, &output_index, &stop));
449 if (stop != output_index + 1) {
450 return errors::InvalidArgument("OpKernel used list-valued output name '",
451 output_name,
452 "' when single-valued output was "
453 "expected");
454 }
455 if (!forward_input_to_output_with_shape(input_index, output_index,
456 output_shape, output)) {
457 return errors::FailedPrecondition("OpKernel could not forward input '",
458 input_name, "' to output '", output_name);
459 }
460 return Status::OK();
461}

463std::unique_ptr<Tensor> OpKernelContext::forward_input(
464 int input_index, int output_index, DataType output_dtype,
465 const TensorShape& output_shape, MemoryType output_memory_type,
466 const AllocatorAttributes& output_attr) {
467 DCHECK_GE(input_index, 0);
468 DCHECK_LT(input_index, num_inputs());
469 const TensorValue& input = (*params_->inputs)[input_index];
470 // Check whether at graph construction time this output was marked
471 // either for no forwarding or with a reservation for this input.
472 // If it's reserved for this input we'll skip the refcount and
473 // AllocatorAttribute checks.
474 // TODO(tucker): Maybe we should skip all of the checks?
475 bool never_forward =
476 (params_->forward_from_array != nullptr && output_index >= 0 &&
477 params_->forward_from_array[output_index] == Params::kNeverForward);
478 if (never_forward) return nullptr;
479 bool forward_expected =
480 (params_->forward_from_array != nullptr && output_index >= 0 &&
481 params_->forward_from_array[output_index] == input_index);
482 if (!forward_expected && params_->forward_from_array != nullptr) {
483 // Check for possibly conflicting forward.
484 for (int i = 0; i < num_outputs(); ++i) {
485 if (params_->forward_from_array[i] == input_index) {
486 // This input is reserved for output i.
487 return nullptr;
488 }
489 }
490 }
491 // Check that input tensor exists and is not a ref.
492 if (input.tensor == nullptr || input.is_ref()) {
493 CHECK(!forward_expected);
494 return nullptr;
495 }
496 // Check that input type matches.
497 if (input_dtype(input_index) != output_dtype) {
498 CHECK(!forward_expected);
499 return nullptr;
500 }
501 // Check that the input and output sizes are compatible.
502 if (input.tensor->shape().num_elements() != output_shape.num_elements()) {
503 CHECK(!forward_expected);
504 return nullptr;
505 }
506 // Check that input and output memory types match, i.e.
507 // that they either both live in host or both live in device memory.
508 if (input_memory_type(input_index) != output_memory_type) {
509 CHECK(!forward_expected);
510 return nullptr;
511 }
512 if (!forward_expected) {
513 if (!input->RefCountIsOne()) {
514 return nullptr;
515 }
516 // Check that output allocator attributes are not more restrictive than
517 // input allocator attributes.
518 const auto input_attr = params_->input_alloc_attrs == nullptr
519 ? AllocatorAttributes()
520 : input_alloc_attr(input_index);
521 if (!output_attr.IsEqualOrLessRestrictiveThan(input_attr)) {
522 return nullptr;
523 }
524 }
525 // TODO(rmlarsen): Use MakeUnique here. There is already a copy in
526 // tensorflow/compiler/xla/ptr_util.h. Perhaps this should be part of
527 // general cleanup of ownership in this code.
528 std::unique_ptr<Tensor> output_tensor(new Tensor());
529 CHECK(output_tensor->CopyFrom(*input.tensor, output_shape));
530 return output_tensor;
531}

533Status OpKernelContext::forward_input_or_allocate_temp(
534 gtl::ArraySlice<int> candidate_input_indices, DataType type,
535 const TensorShape& shape, const AllocatorAttributes& allocator_attr,
536 Tensor* out_temp) {
537 for (int input_index : candidate_input_indices) {
538 std::unique_ptr<Tensor> new_tensor =
539 forward_input(input_index, Params::kNoReservation /*output_index*/,
540 type, shape, DEVICE_MEMORY, allocator_attr);
541 if (new_tensor != nullptr) {
542 *out_temp = std::move(*new_tensor);
543 return Status::OK();
544 }
545 }
546 return allocate_temp(type, shape, out_temp, allocator_attr);
547}

671Status OpKernelContext::allocate_tensor(
672 DataType type, const TensorShape& shape, Tensor* out_tensor,
673 AllocatorAttributes attr, const AllocationAttributes& allocation_attr) {
674 Allocator* a = get_allocator(attr);
675 AllocationAttributes logged_attr(allocation_attr);
676 logged_attr.allocation_will_be_logged = true;
677 Tensor new_tensor(a, type, shape, logged_attr);
678
679 if (!new_tensor.IsInitialized()) {
680 return errors::ResourceExhausted(
681 "OOM when allocating tensor with shape", shape.DebugString(),
682 " and type ", DataTypeString(type), " on ", params_->device->name(),
683 " by allocator ", a->Name());
684 }
685 if (params_->log_memory) {
686 LogMemory::RecordTensorAllocation(params_->op_kernel->name(),
687 params_->step_id, new_tensor);
688 }
689 record_tensor_reference(new_tensor);
690 *out_tensor = std::move(new_tensor);
691 return Status::OK();
692}

694Status OpKernelContext::allocate_output(int index, const TensorShape& shape,
695 Tensor** output,
696 AllocatorAttributes attr) {
697 DCHECK_GE(index, 0);
698 DCHECK_LT(index, outputs_.size());
699 const DataType type = params_->op_kernel->output_type(index);
700 DCHECK(!IsRefType(type));
701 DCHECK(mutable_output(index) == nullptr);
702 Tensor* output_tensor = new Tensor();
703 Status s = allocate_tensor(type, shape, output_tensor, attr);
704 if (s.ok()) {
705 outputs_[index] = TensorValue(output_tensor);
706 *output = outputs_[index].tensor;
707 }
708 return s;
709}

711Status OpKernelContext::allocate_temp(
712 DataType type, const TensorShape& shape, Tensor* out_temp,
713 AllocatorAttributes allocator_attr,
714 const AllocationAttributes& allocation_attr) {
715 Status s =
716 allocate_tensor(type, shape, out_temp, allocator_attr, allocation_attr);
717 if (track_allocations() && s.ok() && out_temp->TotalBytes() > 0) {
718 Allocator* a = get_allocator(allocator_attr);
719 if (a->TracksAllocationSizes()) {
720 int64 alloc_size = a->AllocatedSize(out_temp->tensor_data().data());
721 record_temp_memory_allocation(alloc_size, *out_temp);

728 const TensorShape& shape,
729 PersistentTensor* out_persistent,
730 Tensor** out_tensor,
731 AllocatorAttributes attr) {
732 Tensor persistent;

734 if (s.ok()) {
735 *out_persistent = PersistentTensor(persistent);
736 if (out_tensor) {

746 }
747 }
748 }
749 return s;

765void OpKernelContext::set_output(int index, const Tensor& tensor) {
766 DCHECK_GE(index, 0);
767 DCHECK_LT(index, outputs_.size());
768 DCHECK(!IsRefType(params_->op_kernel->output_type(index)));
769 DCHECK_EQ(mutable_output(index), nullptr);
770 record_tensor_reference(tensor);
771 outputs_[index] = TensorValue(new Tensor(tensor));
772 if (track_allocations() && tensor.TotalBytes() > 0) {
773 mutex_lock l(stats_mu_);
774 if (!temp_tensor_buffer_and_size_) {
775 return;
776 }
777 auto it = std::find_if(temp_tensor_buffer_and_size_->begin(),
778 temp_tensor_buffer_and_size_->end(),
779 [&tensor](const std::pair<const void*, int64>& e) {
780 return e.first == static_cast<const void*>(
781 tensor.tensor_data().data());
782 });
783 if (it != temp_tensor_buffer_and_size_->end()) {
784 temp_memory_allocated_ -= it->second;
785 temp_tensor_buffer_and_size_->erase(it);
786 }
787 }

972// Helper for AttrsMatch().
973bool InTypeList(DataType dt, const AttrValue& type_list) {
974 for (int in_list : type_list.list().type()) {
975 if (dt == in_list) return true;
976 }
977 return false;

980// Returns whether the attrs satisfy the constraints in the kernel_def. Returns
981// an error if attrs in kernel_def are not found, or have a mismatching type.
982Status AttrsMatch(AttrSlice attrs, const KernelDef& kernel_def, bool* match) {
983 *match = false;
984 for (const auto& constraint : kernel_def.constraint()) {
985 if (constraint.allowed_values().list().type_size() == 0) {
986 return errors::Unimplemented(
987 "KernelDef '", ProtoShortDebugString(kernel_def),
988 " has constraint on attr '", constraint.name(),
989 "' with unsupported type: ",
990 SummarizeAttrValue(constraint.allowed_values()));
991 }
992
993 const AttrValue* found = attrs.Find(constraint.name());
994 if (found) {
995 if (found->type() != DT_INVALID) {
996 if (!InTypeList(found->type(), constraint.allowed_values())) {
997 return Status::OK();
998 }
999 } else {
1000 if (!AttrValueHasType(*found, "list(type)").ok()) {
1001 return errors::InvalidArgument(
1002 "KernelDef '", ProtoShortDebugString(kernel_def),
1003 "' has constraint on attr '", constraint.name(),
1004 "' that has value '", SummarizeAttrValue(*found),
1005 "' that does not have type 'type' or 'list(type)' in NodeDef "
1006 "'",
1007 attrs.SummarizeNode(), "'");
1008 }
1009
1010 for (int t : found->list().type()) {
1011 if (!InTypeList(static_cast<DataType>(t),
1012 constraint.allowed_values())) {
1013 return Status::OK();
1014 }
1015 }
1016 }
1017 } else {
1018 return errors::InvalidArgument(
1019 "OpKernel '", kernel_def.op(), "' has constraint on attr '",
1020 constraint.name(), "' not in NodeDef '", attrs.SummarizeNode(),
1021 "', KernelDef: '", ProtoShortDebugString(kernel_def), "'");
1022 }
1023 }
1024 *match = true;
1025 return Status::OK();
1026}

1030// TODO(irving): Replace with const Node& version below.
1031Status FindKernelRegistration(const DeviceType& device_type,
1032 const NodeDef& node_def,
1033 const KernelRegistration** reg,
1034 bool* was_attr_mismatch) {
1035 *reg = nullptr;
1036 *was_attr_mismatch = false;
1037 // Label defaults to empty if not found in NodeDef.
1038 const string& label = GetNodeAttrString(node_def, kKernelAttr);
1039
1040 const string key = Key(node_def.op(), device_type, label);
1041 auto regs = GlobalKernelRegistryTyped()->equal_range(key);
1042 for (auto iter = regs.first; iter != regs.second; ++iter) {
1043 // If there is a kernel registered for the op and device_type,
1044 // check that the attrs match.
1045 bool match;
1046 TF_RETURN_IF_ERROR(AttrsMatch(node_def, iter->second.def, &match));
1047 if (match) {
1048 if (*reg != nullptr) {
1049 return errors::InvalidArgument(
1050 "Multiple OpKernel registrations match NodeDef '",
1051 SummarizeNodeDef(node_def), "': '",
1052 ProtoShortDebugString((*reg)->def), "' and '",
1053 ProtoShortDebugString(iter->second.def), "'");
1054 }
1055 *reg = &iter->second;
1056 } else {
1057 *was_attr_mismatch = true;
1058 }
1059 }
1060 return Status::OK();
1061}

1065// TODO(irving): Change const NodeDef& to const Node&
1066Status FindKernelDef(const DeviceType& device_type, const NodeDef& node_def,
1067 const KernelDef** def, string* kernel_class_name) {
1068 const KernelRegistration* reg = nullptr;
1069 bool was_attr_mismatch;
1070 TF_RETURN_IF_ERROR(
1071 FindKernelRegistration(device_type, node_def, &reg, &was_attr_mismatch));
1072 if (reg == nullptr) {
1073 Status s = errors::NotFound(
1074 "No registered '", node_def.op(), "' OpKernel for ",
1075 DeviceTypeString(device_type), " devices compatible with node ",
1076 SummarizeNodeDef(node_def));
1077 if (was_attr_mismatch) {
1078 errors::AppendToMessage(
1079 &s, " (OpKernel was found, but attributes didn't match)");
1080 }
1081 errors::AppendToMessage(
1082 &s, ". Registered:", KernelsRegisteredForOp(node_def.op()));
1083 return s;
1084 }
1085 if (def != nullptr) *def = &reg->def;
1086 if (kernel_class_name != nullptr) *kernel_class_name = reg->kernel_class_name;
1087 return Status::OK();

1090Status SupportedDeviceTypesForNode(
1091 const std::vector<DeviceType>& prioritized_types, const NodeDef& def,
1092 DeviceTypeVector* device_types) {
1093 // TODO(zhifengc): Changes the callers (SimplePlacer and
1094 // DynamicPlacer) to consider the possibility that 'def' is call to
1095 // a user-defined function and only calls this
1096 // SupportedDeviceTypesForNode for primitive ops.
1097 const OpRegistrationData* op_reg_data;
1098 const Status s = OpRegistry::Global()->LookUp(def.op(), &op_reg_data);
1099 if (s.ok()) {
1100 for (const DeviceType& device_type : prioritized_types) {
1101 const KernelRegistration* reg = nullptr;
1102 bool was_attr_mismatch;
1103 TF_RETURN_IF_ERROR(
1104 FindKernelRegistration(device_type, def, &reg, &was_attr_mismatch));
1105 if (reg != nullptr) device_types->push_back(device_type);
1106 }
1107 } else {
1108 // Assumes that all device types support this node.
1109 for (const DeviceType& device_type : prioritized_types) {
1110 device_types->push_back(device_type);
1111 }
1112 }
1113 return Status::OK();
1114}

1152
1153Status CreateOpKernel(DeviceType device_type, DeviceBase* device,
1154 Allocator* allocator, FunctionLibraryRuntime* flib,
1155 const NodeDef& node_def, int graph_def_version,
1156 OpKernel** kernel) {
1157 VLOG(1) << "Instantiating kernel for node: " << SummarizeNodeDef(node_def);
1158
1159 // Look up the Op registered for this op name.
1160 const OpDef* op_def = nullptr;
1161 Status s = OpRegistry::Global()->LookUpOpDef(node_def.op(), &op_def);
1162 if (!s.ok()) return s;
1163
1164 // Validate node_def against OpDef.
1165 s = ValidateNodeDef(node_def, *op_def);
1166 if (!s.ok()) return s;
1167
1168 // Look up kernel registration.
1169 const KernelRegistration* registration;
1170 bool was_attr_mismatch;
1171 s = FindKernelRegistration(device_type, node_def, &registration,
1172 &was_attr_mismatch);
1173 if (!s.ok()) {
1174 errors::AppendToMessage(&s, " when instantiating ", node_def.op());
1175 return s;
1176 }
1177 if (registration == nullptr) {
1178 s.Update(errors::NotFound("No registered '", node_def.op(),
1179 "' OpKernel for ", DeviceTypeString(device_type),
1180 " devices compatible with node ",
1181 SummarizeNodeDef(node_def)));
1182 if (was_attr_mismatch) {
1183 errors::AppendToMessage(
1184 &s, " (OpKernel was found, but attributes didn't match)");
1185 }
1186 errors::AppendToMessage(
1187 &s, ". Registered:", KernelsRegisteredForOp(node_def.op()));
1188 return s;
1189 }
1190
1191 // Get signature from the OpDef & NodeDef
1192 DataTypeVector inputs;
1193 DataTypeVector outputs;
1194 s.Update(InOutTypesForNode(node_def, *op_def, &inputs, &outputs));
1195 if (!s.ok()) {
1196 errors::AppendToMessage(&s, " for node: ", SummarizeNodeDef(node_def));
1197 return s;
1198 }
1199
1200 // We are creating a kernel for an op registered in
1201 // OpRegistry::Global(), we consult the kernel registry to decide
1202 // the kernel's input and output memory types.
1203 MemoryTypeVector input_memory_types;
1204 MemoryTypeVector output_memory_types;
1205 TF_RETURN_IF_ERROR(MemoryTypesForNode(OpRegistry::Global(), device_type,
1206 node_def, &input_memory_types,
1207 &output_memory_types));
1208
1209 // Everything needed for OpKernel construction.
1210 OpKernelConstruction context(
1211 device_type, device, allocator, &node_def, op_def, flib, inputs,
1212 input_memory_types, outputs, output_memory_types, graph_def_version, &s);
1213 *kernel = (*registration->factory)(&context);
1214 if (!s.ok()) {
1215 delete *kernel;
1216 *kernel = nullptr;
1217 }
1218 return s;
1219}

1235Status ValidateKernelRegistrations(const OpRegistryInterface& op_registry) {
1236 for (const auto& key_registration : *GlobalKernelRegistryTyped()) {
1237 const KernelDef& kernel_def(key_registration.second.def);
1238 const OpRegistrationData* op_reg_data;
1239 const Status status = op_registry.LookUp(kernel_def.op(), &op_reg_data);
1240 if (!status.ok()) {
1241 // TODO(josh11b): Make this a hard error.
1242 LOG(ERROR) << "OpKernel ('" << ProtoShortDebugString(kernel_def)
1243 << "') for unknown op: " << kernel_def.op();
1244 continue;
1245 }
1246 const OpDef& op_def = op_reg_data->op_def;
1247 for (const auto& host_memory_arg : kernel_def.host_memory_arg()) {
1248 if (!FindArgInOp(host_memory_arg, op_def.input_arg()) &&
1249 !FindArgInOp(host_memory_arg, op_def.output_arg())) {
1250 return errors::InvalidArgument(
1251 "HostMemory arg '", host_memory_arg,
1252 "' not found in OpDef: ", SummarizeOpDef(op_def));
1253 }
1254 }
1255 }
1256 return Status::OK();
