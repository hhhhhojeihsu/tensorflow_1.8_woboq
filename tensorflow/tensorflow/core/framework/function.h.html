45// FunctionDefHelper::Create is a convenient helper to construct a
46// FunctionDef proto.
47// E.g.,
48// FunctionDef my_func = FunctionDefHelper::Create(
49// "my_func_name",
50// {"x:T", "y:T" /* one string per argument */},
51// {"z:T" /* one string per return value */},
52// {"T: {float, double}" /* one string per attribute */},
53// {
54// {{"o"}, "Mul", {"x", "y"}, {{"T", "$T"}}}
55// /* one entry per function node */
56// },
57// /* Mapping between function returns and function node outputs. */
58// {{"z", "o:z"}});
59//
60// For the old Function::Node approach, use FunctionDefHelper::Define()
61// E.g.,
62// FunctionDef my_func = FunctionDefHelper::Define(
63// "my_func_name",
64// {"x:T", "y:T" /* one string per argument */},
65// {"z:T" /* one string per return value */},
66// {"T: {float, double}" /* one string per attribute */},
67// {
68// {{"z"}, "Mul", {"x", "y"}, {{"T", "$T"}}}
69// /* one entry per function node */
70// });
71class FunctionDefHelper {
72 public:
73 // AttrValueWrapper has copy constructors for the type T so that
74 // it's easy to construct a simple AttrValue proto.
75 //
76 // If T is a string type (const char*, string, or StringPiece), and
77 // it starts with "$", we construct a AttrValue of "placeholder".
78 //
79 // E.g.,
80 // std::<string, AttrValueWrapper> x = {"T", "$T"}
81 // is a named attr value placeholder.
82 struct AttrValueWrapper {
83 AttrValue proto;
84
85 AttrValueWrapper() {}
86
87 template <typename T>
88 AttrValueWrapper(T val) { // NOLINT(runtime/explicit)
89 SetAttrValue(val, &proto);
90 }
91
92 private:
93 void InitFromString(StringPiece val);
94 };

97 static AttrValueWrapper FunctionRef(
98 const string& name,
99 gtl::ArraySlice<std::pair<string, AttrValueWrapper>> attrs);
100 static AttrValueWrapper FunctionRef(const string& name) {
101 return FunctionRef(name, {});
102 }

143 // Helpers to construct a constant scalar.
144 template <typename T>
145 static Node Const(const string& name, const T& val) {
146 Node n = {{name}, "Const"};
147 const DataType dtype = DataTypeToEnum<T>::value;
148 n.attr.push_back({"dtype", dtype});
149 Tensor t(dtype, TensorShape({}));
150 t.scalar<T>()() = val;
151 n.attr.push_back({"value", t});
152 return n;
153 }
154
155 template <typename T>
156 static Node Const(const string& name, gtl::ArraySlice<T> vals) {
157 Node n = {{name}, "Const"};
158 const DataType dtype = DataTypeToEnum<T>::value;
159 n.attr.push_back({"dtype", dtype});
160 int64 num = vals.size();
161 Tensor t(dtype, TensorShape({num}));
162 for (size_t i = 0; i < vals.size(); ++i) {
163 t.flat<T>()(i) = vals[i];
164 }
165 n.attr.push_back({"value", t});
166 return n;
167 }

199// and opdef is filled with a pointer to the corresponding signature

202 GetFunctionSignature;

249// Represents a function call frame. I.e., the data structure used to
250// pass arguments to a function and retrieve its results.
251//
252// Runtime must arrange accesses to one FunctionCallFrame s.t.
253// 1. SetArgs() happens before any GetArg();
254// 2. GetRetvals happens after all SetRetval();
255class FunctionCallFrame : public CallFrameInterface {
256 public:
257 FunctionCallFrame(DataTypeSlice arg_types, DataTypeSlice ret_types);
258 ~FunctionCallFrame();
259
260 // Caller methods.
261 Status SetArgs(gtl::ArraySlice<Tensor> args);
262 Status GetRetvals(std::vector<Tensor>* rets) const;
263 Status ConsumeRetvals(std::vector<Tensor>* rets);
264
265 size_t num_args() const override { return arg_types_.size(); }
266 size_t num_retvals() const override { return ret_types_.size(); }
267
268 // Callee methods.
269 Status GetArg(int index, Tensor* val) const override;
270 Status SetRetval(int index, const Tensor& val) override;
271
272 private:
273 DataTypeVector arg_types_;
274 DataTypeVector ret_types_;
275 gtl::InlinedVector<Tensor, 4> args_;
276 struct Retval {
277 bool has_val = false;
278 Tensor val;
279 };
280 gtl::InlinedVector<Retval, 4> rets_;
281
282 TF_DISALLOW_COPY_AND_ASSIGN(FunctionCallFrame);
283};

419 // Instantiate a function with the given "attrs".
420 //
421 // Returns OK and fills in "handle" if the instantiation succeeds.
422 // Otherwise returns an error and "handle" is undefined.
423 struct InstantiateOptions {
424 // The canonical device name of the device on which the function
425 // should be instantiated. If empty, the function will be
426 // instantiated on the local device.
427 string target;
428
429 // This interface is EXPERIMENTAL and subject to change.
430 //
431 // If non-null, the runtime will use `overlay_lib` to resolve
432 // function(s) named in `function_name` and `attrs`. Otherwise,
433 // the runtime will use its internal library.
434 // NOTE(mrry): If provided, all functions defined in `overlay_lib`
435 // must be self-contained, and cannot refer to functions defined
436 // in other libraries.
437 // TODO(mrry): Provide a mechanism for sharing core functions
438 // between a set of libraries (e.g. by allowing a
439 // `FunctionLibraryDefinition` to store an `outer_scope` pointer
440 // and implementing name resolution across libraries).
441 const FunctionLibraryDefinition* overlay_lib = nullptr;
442
443 // This interface is EXPERIMENTAL and subject to change.
444 //
445 // If non-empty, the runtime will use `state_handle` to identify
446 // cached state related the instantiated function. Two functions
447 // of the same name and attrs, instantiated with the same
448 // `state_handle` will have the same handle and share the same
449 // state (in stateful kernels); and two functions with different
450 // values for `state_handle` will have independent state.
451 string state_handle;
452 };
453 typedef uint64 Handle;
454 virtual Status Instantiate(const string& function_name, AttrSlice attrs,
455 const InstantiateOptions& options,
456 Handle* handle) = 0;

458 Handle* handle) {
459 return Instantiate(function_name, attrs, {}, handle);
460 }

465 // Returns the function body for the instantiated function given its

472 // Asynchronously invokes the instantiated function identified by
473 // "handle".
474 //
475 // If function execution succeeds, "done" is called with OK and
476 // "*rets" is filled with the function's return values. Otheriwse,
477 // "done" is called with an error status.
478 //
479 // Does not take ownership of "rets".
480 // In the cross-process scenario, runner isn't used for making the Async
481 // RPC calls.
482 struct Options {
483 // The id of the step that is calling this function.
484 int64 step_id = 0;
485 Rendezvous* rendezvous = nullptr;
486 CancellationManager* cancellation_manager = nullptr;
487 ScopedStepContainer* step_container = nullptr;
488 StepStatsCollector* stats_collector = nullptr;
489
490 std::function<void(std::function<void()>)>* runner = nullptr;
491
492 // Parameters for remote function execution.
493 bool remote_execution = false;
494 string source_device = ""; // Fully specified device name.
495
496 // Allocator attributes specifying where the args are / rets should be put.
497 // These should either be {} or match the length of args / retvals. If {},

499 std::vector<AllocatorAttributes> args_alloc_attrs;
500 std::vector<AllocatorAttributes> rets_alloc_attrs;
501
502 // If true, we create a new IntraProcessRendezvous, else use the existing
503 // one.
504 bool create_rendezvous = false;
505 };

609// To register a gradient function for a builtin op, one should use
610// REGISTER_OP_GRADIENT(<op_name>, <c++ grad factory>);
611//
612// Typically, the c++ grad factory is a plan function that can be
613// converted into ::tensorflow::gradient::Creator, which is
614// std::function<Status(const AttrSlice&, FunctionDef*)>.
615//
616// A ::tensorflow::gradient::Creator should populate in FunctionDef* with a
617// definition of a brain function which compute the gradient for the
618// <op_name> when the <op_name> is instantiated with the given attrs.
619//
620// E.g.,
621//
622// Status MatMulGrad(const AttrSlice& attrs, FunctionDef* g) {
623// bool transpose_a;
624// TF_RETURN_IF_ERROR(attrs.Get("transpose_a", &transpose_a));
625// bool transpose_b;
626// TF_RETURN_IF_ERROR(attrs.Get("transpose_b", &transpose_b));
627// DataType dtype;
628// TF_RETURN_IF_ERROR(attrs.Get("dtype", &dtype));
629// if (!transpose_a && !transpose_b) {
630// *g = FunctionDefHelper::Define(
631// "MatMulGrad",
632// {"x:T ", "y:T", "dz:T"}, // Inputs to this function
633// {"dx:T", "dy:T"}, // Outputs from this function
634// {"T: {float, double}"}, // Attributes needed by this function
635// {
636// {{"x_t"}, "Transpose", {"x"}, {{"T", "$T"}}},
637// {{"y_t"}, "Transpose", {"y"}, {{"T", "$T"}}},
638// {{"dx"}, "MatMul", {"dz", "y_t"}, {{"T", "$T"}}},
639// {{"dy"}, "MatMul", {"x_", "dz"}, {{"T", "$T"}}},
640// });
641// } else {
642// ... ...
643// }
644// return Status::OK();
645// }
646//
647// NOTE: $T is substituted with the type variable "T" when the
648// gradient function MatMul is instantiated.
649//
650// TODO(zhifengc): Better documentation somewhere.
651
652// Macros to define a gradient function factory for a primitive
653// operation.
654#define REGISTER_OP_GRADIENT(name, fn) \
655 REGISTER_OP_GRADIENT_UNIQ_HELPER(__COUNTER__, name, fn)
656
657#define REGISTER_OP_NO_GRADIENT(name) \
