26Tensor DeepCopy(const Tensor& other) {
27 Tensor tmp = Tensor(other.dtype(), other.shape());
28 if (DataTypeCanUseMemcpy(other.dtype())) {
29 if (other.NumElements() > 0) {
30 StringPiece other_data = other.tensor_data();
31
32 // We use StringPiece as a convenient map over the tensor buffer,
33 // but we cast the type to get to the underlying buffer to do the
34 // copy.
35 StringPiece tmp_data = tmp.tensor_data();
36 memcpy(const_cast<char*>(tmp_data.data()), other_data.data(),
37 other_data.size());
38 }
39 } else if (other.dtype() == DT_STRING) {
40 tmp.flat<string>() = other.flat<string>();
41 } else {
42 CHECK_EQ(DT_VARIANT, other.dtype());
43 tmp.flat<Variant>() = other.flat<Variant>();
44 }
45 return tmp;
46}

48Status Concat(const gtl::ArraySlice<Tensor>& tensors, Tensor* result) {
49 if (tensors.empty()) {
50 return errors::InvalidArgument("Cannot concatenate zero tensors");
51 }
52 int64 total_dim0_size = 0;
53 for (const Tensor& tensor : tensors) {
54 if (tensor.dims() == 0) {
55 return errors::InvalidArgument(
56 "Cannot concatenate a zero-dimensional tensor");
57 }
58 total_dim0_size += tensor.dim_size(0);
59 }
60 TensorShape shape = tensors[0].shape();
61 shape.set_dim(0, total_dim0_size);
62
63 const DataType dtype = tensors[0].dtype();
64 for (int i = 1; i < tensors.size(); ++i) {
65 if (tensors[i].dtype() != dtype) {
66 return errors::InvalidArgument(
67 "Cannot concatenate tensors that have different data types");
68 }
69 }
70 *result = Tensor(dtype, shape);
71
72 // We use StringPiece as a convenient map over the tensor buffer,
73 // but we cast the type to get to the underlying buffer to do the
74 // copy.
75 StringPiece to_data = result->tensor_data();
76
77 if (DataTypeCanUseMemcpy(dtype)) {
78 int64 offset = 0;
79 for (const Tensor& tensor : tensors) {
80 StringPiece from_data = tensor.tensor_data();
81 CHECK_LE(offset + from_data.size(), to_data.size());
82 memcpy(const_cast<char*>(to_data.data()) + offset, from_data.data(),
83 from_data.size());
84
85 offset += from_data.size();
86 }
87 } else {
88 if (dtype != DT_STRING) {
89 return errors::Internal("Unexpected data type");
90 }
91 string* to_strings =
92 reinterpret_cast<string*>(const_cast<char*>(to_data.data()));
93
94 int64 offset = 0;
95 for (const Tensor& tensor : tensors) {
96 auto from_strings = tensor.flat<string>();
97 CHECK_LE(offset + tensor.NumElements(), result->NumElements());
98 for (int i = 0; i < tensor.NumElements(); ++i) {
99 to_strings[offset + i] = from_strings(i);
100 }
101
102 offset += tensor.NumElements();
103 }
104 }
105
106 return Status::OK();
107}

109Status Split(const Tensor& tensor, const gtl::ArraySlice<int64>& sizes,
110 std::vector<Tensor>* result) {
111 if (tensor.dims() == 0) {
112 return errors::InvalidArgument("Cannot split a zero-dimensional tensor");
113 }
114 int64 total_size = 0;
115 for (int64 size : sizes) {
116 total_size += size;
117 }
118 if (total_size != tensor.dim_size(0)) {
119 return errors::InvalidArgument(
120 "The values in 'sizes' do not sum to the zeroth-dimension size of "
121 "'tensor'");
122 }
123
124 StringPiece from_data = tensor.tensor_data();
125
126 if (DataTypeCanUseMemcpy(tensor.dtype())) {
127 int64 offset = 0;
128 for (int64 size : sizes) {
129 TensorShape shape = tensor.shape();
130 shape.set_dim(0, size);
131 result->emplace_back(tensor.dtype(), shape);
132 Tensor* split = &(*result)[result->size() - 1];
133
134 // We use StringPiece as a convenient map over the tensor buffer,
135 // but we cast the type to get to the underlying buffer to do the
136 // copy.
137 StringPiece to_data = split->tensor_data();
138 CHECK_LE(offset + to_data.size(), from_data.size());
139 memcpy(const_cast<char*>(to_data.data()), from_data.data() + offset,
140 to_data.size());
141
142 offset += to_data.size();
143 }
144 } else {
145 if (tensor.dtype() != DT_STRING) {
146 return errors::Internal("Unexpected data type");
147 }
148 auto from_strings = tensor.flat<string>();
149
150 int64 offset = 0;
151 for (int64 size : sizes) {
152 TensorShape shape = tensor.shape();
153 shape.set_dim(0, size);
154 result->emplace_back(tensor.dtype(), shape);
155 Tensor& split = (*result)[result->size() - 1];
156 string* to_strings = reinterpret_cast<string*>(
157 const_cast<char*>(split.tensor_data().data()));
158
159 CHECK_LE(offset + split.NumElements(), tensor.NumElements());
160 for (int i = 0; i < split.NumElements(); ++i) {
161 to_strings[i] = from_strings(offset + i);
162 }
163
164 offset += split.NumElements();
165 }
166 }
167
168 return Status::OK();
169}
