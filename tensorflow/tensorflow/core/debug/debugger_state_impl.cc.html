23DebuggerState::DebuggerState(const DebugOptions& debug_options) {
24 for (const DebugTensorWatch& watch :
25 debug_options.debug_tensor_watch_opts()) {
26 for (const string& url : watch.debug_urls()) {
27 debug_urls_.insert(url);
28 }
29 }
30}
31
32DebuggerState::~DebuggerState() {
33 for (const string& debug_url : debug_urls_) {
34 DebugIO::CloseDebugURL(debug_url).IgnoreError();
35 }
36}
37
38Status DebuggerState::PublishDebugMetadata(
39 const int64 global_step, const int64 session_run_index,
40 const int64 executor_step_index, const std::vector<string>& input_names,
41 const std::vector<string>& output_names,
42 const std::vector<string>& target_names) {
43 return DebugIO::PublishDebugMetadata(global_step, session_run_index,
44 executor_step_index, input_names,
45 output_names, target_names, debug_urls_);
46}
47
48Status DebugGraphDecorator::DecorateGraph(Graph* graph, Device* device) {
49 DebugNodeInserter::DeparallelizeWhileLoops(graph, device);
50 return DebugNodeInserter::InsertNodes(
51 debug_options_.debug_tensor_watch_opts(), graph, device);
52}
53
54Status DebugGraphDecorator::PublishGraph(const Graph& graph,
55 const string& device_name) {
56 std::unordered_set<string> debug_urls;
57 for (const DebugTensorWatch& watch :
58 debug_options_.debug_tensor_watch_opts()) {
59 for (const string& url : watch.debug_urls()) {
60 debug_urls.insert(url);
61 }
62 }
63
64 return DebugIO::PublishGraph(graph, device_name, debug_urls);
65}
