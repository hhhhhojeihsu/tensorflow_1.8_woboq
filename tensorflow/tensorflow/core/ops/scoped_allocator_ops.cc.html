21REGISTER_OP("_ScopedAllocator")
22 .Output("output: T")
23 .Attr("shapes: list(shape)")
24 .Attr("T: type")
25 .Attr("sa_name: string")
26 .Attr("id: int")
27 .Attr("expected_call_count: int")
28 .SetIsStateful()
29 .SetShapeFn(shape_inference::ExplicitShape)
30 .Doc(R"doc(
31Allocates a mutable tensor that becomes available to appropriately annotated
32downstream Ops as backing store for their output tensor allocations via the
33ScopedAllocatorMgr.
34Returns a reference to this value.
35
36This is an experimental op for internal use only. It is possible to use this
37op in unsafe ways.
38)doc");

40REGISTER_OP("_ScopedAllocatorConcat")
41 .Output("output: T")
42 .Input("backing: T")
43 .Input("inputs: N * T")
44 .Attr("shape: shape")
45 .Attr("T: type")
46 .Attr("sa_name: string")
47 .Attr("id: int")
48 .Attr("N: int >= 2")
49 .SetIsStateful()
50 .SetShapeFn(shape_inference::ExplicitShape)
51 .Doc(R"doc(
52Acts like a Concat Op that merges multple tensors into one, however it must
53only be used in conjunction with a ScopedAllocator which is backing the memory
54of all of its input tensors so that actually it just outputs a read-only
55reference to that ScopedAllocator's backing tensor.
56
57This is an experimental op for internal use only. It is possible to use this
58op in unsafe ways.
59)doc");
