88Status CrossEntropyGrad(const AttrSlice& attrs, FunctionDef* g) {
89 // clang-format off
90 *g = FDH::Define(
91 // Arg defs
92 {"features: T", "labels: T", "dcost_dloss: T", "donotcare: T"},
93 // Ret val defs
94 {"dcost_dfeatures: T", "dcost_dlabels: T"},
95 // Attr defs
96 {{"T: {float, double}"}},
97 // Nodes
98 {
99 // _, dloss_dfeatures = CrossEntropy(features, labels)
100 {{"donotcare_loss", "dloss_dfeatures"}, "CrossEntropy",
101 {"features", "labels"}, {{"T", "$T"}}},
102 // dcost_dloss is of shape [batch_size].
103 // dcost_dloss_mat is of shape [batch_size, 1].
104 FDH::Const("neg1", -1),
105 {{"dcost_dloss_mat"}, "ExpandDims", {"dcost_dloss", "neg1"},
106 {{"T", "$T"}}},
107 // chain rule: dcost/dfeatures = dcost/dloss * dloss/dfeatures
108 {{"dcost_dfeatures"}, "Mul", {"dcost_dloss_mat", "dloss_dfeatures"},
109 {{"T", "$T"}}},
110 {{"dcost_dlabels"}, "ZerosLike", {"labels"}, {{"T", "$T"}}},
111 });
112 // clang-format on
113 return Status::OK();
114}
115REGISTER_OP_GRADIENT("CrossEntropy", CrossEntropyGrad);
