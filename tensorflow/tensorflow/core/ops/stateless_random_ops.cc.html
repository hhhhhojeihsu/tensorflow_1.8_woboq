19namespace tensorflow {

21using shape_inference::DimensionHandle;
22using shape_inference::ShapeHandle;
23
24static Status StatelessShape(shape_inference::InferenceContext* context) {
25 // Check seed shape
26 ShapeHandle seed;
27 TF_RETURN_IF_ERROR(context->WithRank(context->input(1), 1, &seed));
28 DimensionHandle unused;
29 TF_RETURN_IF_ERROR(context->WithValue(context->Dim(seed, 0), 2, &unused));
30
31 // Set output shape
32 shape_inference::ShapeHandle out;
33 TF_RETURN_IF_ERROR(context->MakeShapeFromShapeTensor(0, &out));
34 context->set_output(0, out);
35 return Status::OK();
36}
37
38#define REGISTER_STATELESS_OP(name) \
39 REGISTER_OP(name) \
40 .Input("shape: T") \
41 .Input("seed: Tseed") \
42 .Output("output: dtype") \
43 .Attr("dtype: {half,float,double} = DT_FLOAT") \
44 .Attr("T: {int32, int64} = DT_INT32") \
45 .Attr("Tseed: {int32, int64} = DT_INT64") \
46 .SetShapeFn(StatelessShape)
47
