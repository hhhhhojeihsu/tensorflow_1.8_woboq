78REGISTER_OP("StringJoin")
79 .Input("inputs: N * string")
80 .Attr("N: int")
81 .Attr("separator: string = ''")
82 .Output("output: string")
83 .SetShapeFn([](InferenceContext* c) {
84 // If all inputs are scalars, then return a scalar.
85 bool all_scalar = true;
86 for (int i = 0; i < c->num_inputs(); ++i) {
87 if (c->Rank(c->input(i)) != 0) all_scalar = false;
88 }
89 if (all_scalar) {
90 c->set_output(0, c->Scalar());
91 return Status::OK();
92 }
93
94 // At least one input is unknown or a scalar.
95 // Merge the non-scalars to find the output shape.
96 // Don't merge inputs with unknown rank, as they can actually be scalars
97 // or the output shape.
98 ShapeHandle out = c->UnknownShape();
99 for (int i = 0; i < c->num_inputs(); ++i) {
100 if (c->RankKnown(c->input(i)) && c->Rank(c->input(i)) != 0) {
101 TF_RETURN_IF_ERROR(c->Merge(out, c->input(i), &out));
102 }
103 }
104 c->set_output(0, out);
105 return Status::OK();
106 });

108REGISTER_OP("StringSplit")
109 .Input("input: string")
110 .Input("delimiter: string")
111 .Output("indices: int64")
112 .Output("values: string")
113 .Output("shape: int64")
114 .Attr("skip_empty: bool = true")
115 .SetShapeFn([](InferenceContext* c) {
116 ShapeHandle unused;
117 TF_RETURN_IF_ERROR(c->WithRank(c->input(0), 1, &unused));
118 TF_RETURN_IF_ERROR(c->WithRank(c->input(1), 0, &unused));
119
120 c->set_output(0, c->Matrix(InferenceContext::kUnknownDim, 2));
121 c->set_output(1, c->Vector(InferenceContext::kUnknownDim));
122 c->set_output(2, c->Vector(2));
123 return Status::OK();
124 });

137REGISTER_OP("Substr")
138 .Input("input: string")
139 .Input("pos: T")
140 .Input("len: T")
141 .Output("output: string")
142 .Attr("T: {int32, int64}")
143 .SetShapeFn([](InferenceContext* c) {
144 ShapeHandle pos_shape = c->input(1);
145 ShapeHandle len_shape = c->input(2);
146 ShapeHandle unused;
147 // Check that pos/len have same rank
148 TF_RETURN_IF_ERROR(c->WithRank(pos_shape, c->Rank(len_shape), &unused));
149 // Check that dimensions are equal
150 for (int32 i = 0; i < c->Rank(pos_shape); ++i) {
151 DimensionHandle pos_dim = c->Dim(pos_shape, i);
152 DimensionHandle len_dim = c->Dim(len_shape, i);
153 if (c->Value(pos_dim) != c->Value(len_dim)) {
154 return errors::InvalidArgument(
155 "pos and len shapes must match: ", c->DebugString(pos_shape),
156 " vs. ", c->DebugString(len_shape));
157 }
158 }
159 // c->input(0) is the ShapeHandle to input strings
160 // BroadcastBinaryOpShapeFn infers shape from c->input(0) and c->input(1).
161 return shape_inference::BroadcastBinaryOpShapeFn(c);
162 });
