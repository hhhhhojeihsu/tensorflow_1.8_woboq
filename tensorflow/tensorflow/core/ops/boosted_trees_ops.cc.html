37REGISTER_OP("BoostedTreesCalculateBestGainsPerFeature")
38 .Input("node_id_range: int32")
39 .Input("stats_summary_list: num_features * float32")
40 .Input("l1: float")
41 .Input("l2: float")
42 .Input("tree_complexity: float")
43 .Input("min_node_weight: float")
44 .Attr("max_splits: int >= 1")
45 .Attr("num_features: int >= 1") // not passed but populated automatically.
46 .Output("node_ids_list: num_features * int32")
47 .Output("gains_list: num_features * float32")
48 .Output("thresholds_list: num_features * int32")
49 .Output("left_node_contribs_list: num_features * float32")
50 .Output("right_node_contribs_list: num_features * float32")
51 .SetShapeFn([](shape_inference::InferenceContext* c) {
52 // Confirms the rank of the inputs and sets the shape of the outputs.
53 int max_splits;
54 int num_features;
55 TF_RETURN_IF_ERROR(c->GetAttr("max_splits", &max_splits));
56 TF_RETURN_IF_ERROR(c->GetAttr("num_features", &num_features));
57 shape_inference::ShapeHandle node_id_range_shape;
58 shape_inference::ShapeHandle unused_shape;
59 TF_RETURN_IF_ERROR(c->WithRank(c->input(0), 1, &node_id_range_shape));
60 TF_RETURN_IF_ERROR(
61 c->Merge(node_id_range_shape, c->MakeShape({2}), &unused_shape));
62 // Checks that all stats summary entries are of the same shape.
63 shape_inference::ShapeHandle summary_shape_base;
64 TF_RETURN_IF_ERROR(c->WithRank(c->input(1), 3, &summary_shape_base));
65 TF_RETURN_IF_ERROR(c->Merge(summary_shape_base,
66 c->MakeShape({max_splits, -1, 2}),
67 &unused_shape));
68 for (int i = 1; i < num_features; ++i) {
69 shape_inference::ShapeHandle summary_shape;
70 TF_RETURN_IF_ERROR(c->WithRank(c->input(1 + i), 3, &summary_shape));
71 TF_RETURN_IF_ERROR(
72 c->Merge(summary_shape_base, summary_shape, &unused_shape));
73 }
74 TF_RETURN_IF_ERROR(
75 c->WithRank(c->input(num_features + 1), 0, &unused_shape));
76 TF_RETURN_IF_ERROR(
77 c->WithRank(c->input(num_features + 2), 0, &unused_shape));
78 TF_RETURN_IF_ERROR(
79 c->WithRank(c->input(num_features + 3), 0, &unused_shape));
80 // Sets the output lists.
81 std::vector<shape_inference::ShapeHandle> output_shapes_vec(
82 num_features, c->MakeShape({-1}));
83 TF_RETURN_IF_ERROR(c->set_output("node_ids_list", output_shapes_vec));
84 TF_RETURN_IF_ERROR(c->set_output("gains_list", output_shapes_vec));
85 TF_RETURN_IF_ERROR(c->set_output("thresholds_list", output_shapes_vec));
86 std::vector<shape_inference::ShapeHandle> output_shapes_contribs(
87 num_features, c->MakeShape({-1, 1}));
88 TF_RETURN_IF_ERROR(
89 c->set_output("left_node_contribs_list", output_shapes_contribs));
90 TF_RETURN_IF_ERROR(
91 c->set_output("right_node_contribs_list", output_shapes_contribs));
92 return Status::OK();
93 });

137REGISTER_OP("BoostedTreesMakeStatsSummary")
138 .Input("node_ids: int32")
139 .Input("gradients: float")
140 .Input("hessians: float")
141 .Input("bucketized_features_list: num_features * int32")
142 .Attr("max_splits: int >= 1")
143 .Attr("num_buckets: int >= 1")
144 .Attr("num_features: int >= 1")
145 .Output("stats_summary: float")
146 .SetShapeFn([](shape_inference::InferenceContext* c) {
147 // Sets the shape of the output as a Rank 4 Tensor.
148 int max_splits;
149 int num_buckets;
150 int num_features;
151 TF_RETURN_IF_ERROR(c->GetAttr("max_splits", &max_splits));
152 TF_RETURN_IF_ERROR(c->GetAttr("num_buckets", &num_buckets));
153 TF_RETURN_IF_ERROR(c->GetAttr("num_features", &num_features));
154 shape_inference::ShapeHandle node_ids_shape;
155 shape_inference::ShapeHandle gradients_shape;
156 shape_inference::ShapeHandle hessians_shape;
157 shape_inference::ShapeHandle bucketized_feature_shape;
158 shape_inference::ShapeHandle unused_shape;
159 shape_inference::DimensionHandle unused_dim;
160 TF_RETURN_IF_ERROR(c->WithRank(c->input(0), 1, &node_ids_shape));
161 TF_RETURN_IF_ERROR(c->WithRank(c->input(1), 2, &gradients_shape));
162 TF_RETURN_IF_ERROR(c->WithRank(c->input(2), 2, &hessians_shape));
163 TF_RETURN_IF_ERROR(c->Merge(c->Dim(node_ids_shape, 0),
164 c->Dim(gradients_shape, 0), &unused_dim));
165 TF_RETURN_IF_ERROR(
166 c->Merge(gradients_shape, hessians_shape, &unused_shape));
167 for (int f = 0; f < num_features; ++f) {
168 TF_RETURN_IF_ERROR(
169 c->WithRank(c->input(3 + f), 1, &bucketized_feature_shape));
170 TF_RETURN_IF_ERROR(c->Merge(c->Dim(node_ids_shape, 0),
171 c->Dim(bucketized_feature_shape, 0),
172 &unused_dim));
173 }
174 c->set_output(0,
175 c->MakeShape({num_features, max_splits, num_buckets, 2}));
176 return Status::OK();
177 });

179REGISTER_OP("BoostedTreesPredict")
180 .Input("tree_ensemble_handle: resource")
181 .Input("bucketized_features: num_bucketized_features * int32")
182 .Attr("num_bucketized_features: int >= 1") // Inferred.
183 .Attr("logits_dimension: int")
184 .Output("logits: float")
185 .SetShapeFn([](shape_inference::InferenceContext* c) {
186 shape_inference::ShapeHandle feature_shape;
187 int num_bucketized_features;
188 TF_RETURN_IF_ERROR(
189 c->GetAttr("num_bucketized_features", &num_bucketized_features));
190 shape_inference::ShapeHandle unused_input;
191 for (int i = 0; i < num_bucketized_features; ++i) {
192 TF_RETURN_IF_ERROR(c->WithRank(c->input(i + 1), 1, &feature_shape));
193 // Check that the shapes of all bucketized features are the same.
194 TF_RETURN_IF_ERROR(c->Merge(c->input(1), feature_shape, &unused_input));
195 }
196
197 int logits_dimension;
198 TF_RETURN_IF_ERROR(c->GetAttr("logits_dimension", &logits_dimension));
199 auto logits_shape =
200 c->MakeShape({c->Dim(feature_shape, 0), logits_dimension});
201 // Logits.
202 c->set_output(0, logits_shape);
203 return Status::OK();
204 });

218REGISTER_OP("BoostedTreesTrainingPredict")
219 .Input("tree_ensemble_handle: resource")
220 .Input("cached_tree_ids: int32")
221 .Input("cached_node_ids: int32")
222 .Input("bucketized_features: num_bucketized_features * int32")
223 .Attr("num_bucketized_features: int >= 1")
224 .Attr("logits_dimension: int")
225 .Output("partial_logits: float")
226 .Output("tree_ids: int32")
227 .Output("node_ids: int32")
228 .SetShapeFn([](shape_inference::InferenceContext* c) {
229 shape_inference::ShapeHandle feature_shape;
230 int num_bucketized_features;
231 TF_RETURN_IF_ERROR(
232 c->GetAttr("num_bucketized_features", &num_bucketized_features));
233
234 shape_inference::ShapeHandle unused_input;
235 for (int i = 0; i < num_bucketized_features; ++i) {
236 TF_RETURN_IF_ERROR(c->WithRank(c->input(i + 3), 1, &feature_shape));
237 TF_RETURN_IF_ERROR(
238 c->Merge(c->input(i + 3), feature_shape, &unused_input));
239 }
240 // all inputs/outputs except logits should have same shape.
241 TF_RETURN_IF_ERROR(c->Merge(c->input(1), feature_shape, &unused_input));
242 TF_RETURN_IF_ERROR(c->Merge(c->input(2), feature_shape, &unused_input));
243
244 int logits_dimension;
245 TF_RETURN_IF_ERROR(c->GetAttr("logits_dimension", &logits_dimension));
246 auto logits_shape =
247 c->MakeShape({c->Dim(feature_shape, 0), logits_dimension});
248 // Partial logits.
249 c->set_output(0, logits_shape);
250 // Tree ids.
251 c->set_output(1, c->MakeShape({c->Dim(feature_shape, 0)}));
252 // Node ids.
253 c->set_output(2, c->MakeShape({c->Dim(feature_shape, 0)}));
254 return Status::OK();
255 });

257REGISTER_OP("BoostedTreesUpdateEnsemble")
258 .Input("tree_ensemble_handle: resource")
259 .Input("feature_ids: int32")
260 .Input("node_ids: num_features * int32")
261 .Input("gains: num_features * float")
262 .Input("thresholds: num_features * int32")
263 .Input("left_node_contribs: num_features * float")
264 .Input("right_node_contribs: num_features * float")
265 .Input("max_depth: int32")
266 .Input("learning_rate: float")
267 .Attr("pruning_mode: int >=0")
268 .Attr("num_features: int >= 0") // Inferred.
269 .SetShapeFn([](shape_inference::InferenceContext* c) {
270 shape_inference::ShapeHandle shape_handle;
271 int num_features;
272 TF_RETURN_IF_ERROR(c->GetAttr("num_features", &num_features));
273
274 // Feature_ids, should be one for each feature.
275 shape_inference::ShapeHandle feature_ids_shape;
276 TF_RETURN_IF_ERROR(c->WithRank(c->input(1), 1, &feature_ids_shape));
277 TF_RETURN_IF_ERROR(
278 c->Merge(c->input(1), c->Vector(num_features), &shape_handle));
279
280 for (int i = 0; i < num_features; ++i) {
281 // Node ids.
282 TF_RETURN_IF_ERROR(c->WithRank(c->input(i + 2), 1, &shape_handle));
283 auto shape_rank_1 = c->MakeShape({c->Dim(shape_handle, 0)});
284 auto shape_rank_2 = c->MakeShape({c->Dim(shape_handle, 0), 1});
285
286 // Gains.
287 TF_RETURN_IF_ERROR(
288 c->WithRank(c->input(i + num_features + 2), 1, &shape_handle));
289 // TODO(nponomareva): replace this with input("name",vector of shapes).
290 TF_RETURN_IF_ERROR(c->Merge(c->input(i + num_features + 2),
291 shape_rank_1, &shape_handle));
292 // Thresholds.
293 TF_RETURN_IF_ERROR(
294 c->WithRank(c->input(i + num_features * 2 + 2), 1, &shape_handle));
295 TF_RETURN_IF_ERROR(c->Merge(c->input(i + num_features * 2 + 2),
296 shape_rank_1, &shape_handle));
297 // Left and right node contribs.
298 TF_RETURN_IF_ERROR(
299 c->WithRank(c->input(i + num_features * 3 + 2), 2, &shape_handle));
300 TF_RETURN_IF_ERROR(c->Merge(c->input(i + num_features * 3 + 2),
301 shape_rank_2, &shape_handle));
302 TF_RETURN_IF_ERROR(
303 c->WithRank(c->input(i + num_features * 4 + 2), 2, &shape_handle));
304 TF_RETURN_IF_ERROR(c->Merge(c->input(i + num_features * 4 + 2),
305 shape_rank_2, &shape_handle));
306 }
307 return Status::OK();
308 });
