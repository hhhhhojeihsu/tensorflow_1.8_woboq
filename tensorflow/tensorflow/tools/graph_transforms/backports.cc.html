30// Switch any ConcatV2 nodes to the v1 version, swapping the input order.
31Status BackportConcatV2Transform(const GraphDef& input_graph_def,
32 const TransformFuncContext& context,
33 GraphDef* output_graph_def) {
34 TF_RETURN_IF_ERROR(ReplaceMatchingOpTypes(
35 input_graph_def, {"ConcatV2"},
36 [](const NodeMatch& match, const std::set<string>& input_nodes,
37 const std::set<string>& output_nodes,
38 std::vector<NodeDef>* new_nodes) {
39 const NodeDef& concat_v2_node = match.node;
40 NodeDef concat_node = concat_v2_node;
41 concat_node.set_op("Concat");
42 // The last input is inserted at the head of the inputs, because Concat
43 // expects the dimension as the first input (not the last as in
44 // ConcatV2).
45 concat_node.mutable_input()->Clear();
46 const string& dim_input =
47 concat_v2_node.input(concat_v2_node.input_size() - 1);
48 concat_node.add_input(dim_input);
49 for (int i = 0; i < (concat_v2_node.input_size() - 1); ++i) {
50 concat_node.add_input(concat_v2_node.input(i));
51 }
52 // Tidx attribute must be deleted because it's not used in Concat.
53 concat_node.mutable_attr()->erase("Tidx");
54 new_nodes->push_back(concat_node);
55 return Status::OK();
56 },
57 {true}, output_graph_def));
58
59 return Status::OK();
60}
61
62REGISTER_GRAPH_TRANSFORM("backport_concatv2", BackportConcatV2Transform);

64// Switch any TensorArrayV3 nodes to the v2 version, removing the second output.
65Status BackportTensorArrayV3Transform(const GraphDef& input_graph_def,
66 const TransformFuncContext& context,
67 GraphDef* output_graph_def) {
68 std::map<string, string> inputs_to_rename;
69 GraphDef replaced_graph_def;
70 TF_RETURN_IF_ERROR(ReplaceMatchingOpTypes(
71 input_graph_def, {"TensorArrayV3|TensorArrayGradV3"},
72 [&inputs_to_rename](const NodeMatch& match,
73 const std::set<string>& input_nodes,
74 const std::set<string>& output_nodes,
75 std::vector<NodeDef>* new_nodes) {
76 const NodeDef& tensor_array_v3_node = match.node;
77
78 // All we need to do here is rename the op type, since the attributes
79 // remain the same.
80 NodeDef tensor_array_v2_node = tensor_array_v3_node;
81 if (tensor_array_v3_node.op() == "TensorArrayV3") {
82 tensor_array_v2_node.set_op("TensorArrayV2");
83 } else {
84 tensor_array_v2_node.set_op("TensorArrayGradV2");
85 }
86
87 // The v3 version has a second 'flow' output that's not present in v2,
88 // so substitute a dummy constant instead in any places that use it.
89 NodeDef replacement_flow_node;
90 replacement_flow_node.set_op("Const");
91 SetNodeAttr("dtype", DT_FLOAT, &replacement_flow_node);
92 replacement_flow_node.set_name(tensor_array_v3_node.name() +
93 "/replacement_flow_node");
94 Tensor replacement_flow_tensor(DT_FLOAT, {});
95 // I'm picking an arbitrary value for the gradient flow here, for lack
96 // of a better alternative.
97 replacement_flow_tensor.flat<float>()(0) = 1.0f;
98 SetNodeTensorAttr<float>("value", replacement_flow_tensor,
99 &replacement_flow_node);
100 inputs_to_rename[tensor_array_v3_node.name() + ":1"] =
101 replacement_flow_node.name();
102
103 new_nodes->push_back(tensor_array_v2_node);
104 new_nodes->push_back(replacement_flow_node);
105 return Status::OK();
106 },
107 {true}, &replaced_graph_def));
108 // Update the graph so that any nodes that referred to removed inputs now
109 // pull from the substitute constants we've added.
110 GraphDef renamed_graph_def;
111 TF_RETURN_IF_ERROR(RenameNodeInputs(replaced_graph_def, inputs_to_rename,
112 std::unordered_set<string>(),
113 &renamed_graph_def));
114 TF_RETURN_IF_ERROR(ReplaceMatchingOpTypes(
115 renamed_graph_def,
116 {"TensorArrayWriteV3|TensorArrayReadV3|TensorArrayGatherV3|"
117 "TensorArrayScatterV3|TensorArrayConcatV3|TensorArraySplitV3|"
118 "TensorArraySizeV3|TensorArrayCloseV3"},
119 [](const NodeMatch& match, const std::set<string>& input_nodes,
120 const std::set<string>& output_nodes,
121 std::vector<NodeDef>* new_nodes) {
122 const NodeDef& v3_node = match.node;
123 NodeDef v2_node = v3_node;
124 v2_node.set_op(v3_node.op().substr(0, v3_node.op().size() - 1) + "2");
125 new_nodes->push_back(v2_node);
126 return Status::OK();
127 },
128 {true}, output_graph_def));
129 return Status::OK();
130}

132REGISTER_GRAPH_TRANSFORM("backport_tensor_array_v3",
133 BackportTensorArrayV3Transform);
