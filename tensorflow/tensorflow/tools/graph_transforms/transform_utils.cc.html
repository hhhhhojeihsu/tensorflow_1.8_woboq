31void RecordMatchedNodes(const NodeMatch& match,
32 std::set<string>* matched_nodes) {
33 matched_nodes->insert(match.node.name());
34 for (const NodeMatch& input_match : match.inputs) {
35 RecordMatchedNodes(input_match, matched_nodes);
36 }

44void MatchedNodesAsArray(const NodeMatch& match, std::vector<NodeDef>* result) {
45 std::set<string> found_nodes;
46 std::vector<NodeMatch> current_matches = {match};
47 while (!current_matches.empty()) {
48 std::vector<NodeMatch> next_matches;
49 for (const NodeMatch& current_match : current_matches) {
50 if (found_nodes.count(current_match.node.name())) {
51 continue;
52 }
53 found_nodes.insert(current_match.node.name());
54 result->push_back(current_match.node);
55 for (const NodeMatch& input_match : current_match.inputs) {
56 next_matches.push_back(input_match);
57 }
58 }
59 current_matches = next_matches;
60 }
61}

82void NodeNamePartsFromInput(const string& input_name, string* prefix,
83 string* node_name, string* suffix) {
84 std::vector<string> input_parts = str_util::Split(input_name, ':');
85 if (input_parts.size() < 2) {
86 *suffix = "";
87 } else {
88 *suffix = ":" + input_parts[1];
89 }
90 StringPiece node_name_piece(input_parts[0]);
91 if (str_util::ConsumePrefix(&node_name_piece, "^")) {
92 *prefix = "^";
93 } else {
94 *prefix = "";
95 }
96 *node_name = node_name_piece.ToString();
97}

99string NodeNameFromInput(const string& input_name) {
100 string prefix;
101 string node_name;
102 string suffix;
103 NodeNamePartsFromInput(input_name, &prefix, &node_name, &suffix);
104 return node_name;
105}

107string CanonicalInputName(const string& input_name) {
108 string prefix;
109 string node_name;
110 string suffix;
111 NodeNamePartsFromInput(input_name, &prefix, &node_name, &suffix);
112 if (suffix.empty()) {
113 suffix = ":0";
114 }
115 return prefix + node_name + suffix;
116}

118uint64 HashNodeDef(const NodeDef& node) {
119 uint64 hash = Hash64String(node.op());
120 hash = Hash64Combine(hash, Hash64String(node.name()));
121 for (const string& input : node.input()) {
122 hash = Hash64Combine(hash, Hash64String(CanonicalInputName(input)));
123 }
124 hash = Hash64Combine(hash, Hash64String(node.device()));
125 std::vector<string> attr_names;
126 attr_names.reserve(node.attr().size());
127 for (const auto& attr : node.attr()) {
128 attr_names.push_back(attr.first);
129 }
130 std::sort(attr_names.begin(), attr_names.end());
131 string attr_serialized;
132 for (const string& attr_name : attr_names) {
133 auto attr = node.attr().at(attr_name);
134 attr.SerializeToString(&attr_serialized);
135 hash = Hash64Combine(hash, Hash64String(attr_serialized));
136 }
137 return hash;
138}

182Status SortByExecutionOrder(const GraphDef& input_graph_def,
183 GraphDef* output_graph_def) {
184 const int num_nodes = input_graph_def.node_size();
185 std::vector<int> ready;
186 std::vector<int> pending_count;
187 pending_count.reserve(num_nodes);
188 std::vector<gtl::InlinedVector<int, 4>> outputs(num_nodes);
189
190 std::map<string, int> name_index;
191 for (int i = 0; i < input_graph_def.node_size(); ++i) {
192 const NodeDef& node(input_graph_def.node(i));
193 name_index[node.name()] = i;
194 }
195
196 // Parse the inputs for each node.
197 for (int n = 0; n < num_nodes; ++n) {
198 const NodeDef& node_def(input_graph_def.node(n));
199 if (IsMerge(node_def)) {
200 // for merge only wait for one non-control input.
201 int32 num_control_edges = 0;
202 for (int i = 0; i < node_def.input_size(); ++i) {
203 if (str_util::StartsWith(node_def.input(i), "^")) {
204 num_control_edges++;
205 }
206 }
207 pending_count.push_back(num_control_edges + 1);
208 } else {
209 pending_count.push_back(node_def.input_size());
210 }
211 if (node_def.input_size() == 0) {
212 ready.push_back(n);
213 continue;
214 }
215 for (int i = 0; i < node_def.input_size(); ++i) {
216 const string& input_name = node_def.input(i);
217 const string& input_node_name = NodeNameFromInput(input_name);
218 if (!name_index.count(input_node_name)) {
219 return errors::InvalidArgument("Node '", node_def.name(),
220 "': Unknown input node '",
221 node_def.input(i), "'");
222 }
223 outputs[name_index[input_node_name]].push_back(n);
224 }
225 }
226
227 int processed = 0;
228 output_graph_def->Clear();
229 // Process the NodeDefs in topological order.
230 // Code above sets this up by filling in ready_ with nodes that have no
231 // inputs, pending_counts_ with the number of inputs for each node and
232 // outputs_ with the outputs of each node.
233 while (!ready.empty()) {
234 int o = ready.back();
235 ready.pop_back();
236 ++processed;
237 const NodeDef& node_def(input_graph_def.node(o));
238 *output_graph_def->mutable_node()->Add() = node_def;
239
240 // Update pending_count for outputs.
241 for (size_t i = 0; i < outputs[o].size(); ++i) {
242 const int output = outputs[o][i];
243 pending_count[output]--;
244 if (pending_count[output] == 0) {
245 ready.push_back(output);
246 }
247 }
248 }
249
250 if (processed < input_graph_def.node_size()) {
251 return errors::InvalidArgument(input_graph_def.node_size() - processed,
252 " nodes in a cycle");
253 }
254 return Status::OK();
255}

282Status GraphMatcher::GetOpTypeMatches(const OpTypePattern& pattern,
283 std::vector<NodeMatch>* matches) {
284 std::set<string> matched_nodes;
285 for (const NodeDef& node : graph_def_.node()) {
286 // Skip any nodes that are already part of a match.
287 if (matched_nodes.count(node.name())) {
288 continue;
289 }
290 NodeMatch match;
291 if (DoesOpTypeMatch(node, pattern, matched_nodes, &match)) {
292 RecordMatchedNodes(match, &matched_nodes);
293 matches->push_back(match);
294 }
295 }
296 return Status::OK();
297}

299bool GraphMatcher::DoesOpTypeMatch(
300 const NodeDef& node, const OpTypePattern& pattern,
301 const std::set<string>& previously_matched_nodes, NodeMatch* match) {
302 VLOG(1) << "Looking at node " << node.DebugString();
303 VLOG(1) << "pattern=" << pattern.DebugString();
304 VLOG(1) << "match=" << match->DebugString();
305 if (previously_matched_nodes.count(node.name())) {
306 VLOG(1) << "node " << node.name() << " has been previously matched";
307 return false;
308 }
309 bool pattern_matched = false;
310 if (pattern.op == "*") {
311 pattern_matched = true;
312 } else {
313 std::vector<string> pattern_ops = str_util::Split(pattern.op, '|');
314 for (const string& pattern_op : pattern_ops) {
315 if (node.op() == pattern_op) {
316 pattern_matched = true;
317 }
318 }
319 }
320 if (!pattern_matched) {
321 VLOG(1) << "node.op() != pattern.op()";
322 return false;
323 }
324 match->node = node;
325 // Ignore any control inputs for pattern-matching purposes
326 std::vector<string> non_control_inputs;
327 for (const string& input : node.input()) {
328 if (!input.empty() && (input[0] != '^')) {
329 non_control_inputs.push_back(input);
330 }
331 }
332 if (pattern.inputs.empty()) {
333 // If there are no inputs, assume that's the end of the pattern.
334 return true;
335 }
336 if (non_control_inputs.size() != pattern.inputs.size()) {
337 VLOG(1) << "non_control_inputs.size() != pattern.inputs.size()";
338 return false;
339 }
340 for (int i = 0; i < pattern.inputs.size(); ++i) {
341 const string& input_node_name = NodeNameFromInput(non_control_inputs[i]);
342 const NodeDef& input_node = *(node_map_[input_node_name]);
343 const OpTypePattern& input_pattern = pattern.inputs[i];
344 match->inputs.push_back(NodeMatch());
345 NodeMatch* input_match = &(match->inputs.back());
346 if (!DoesOpTypeMatch(input_node, input_pattern, previously_matched_nodes,
347 input_match)) {
348 return false;
349 }
350 }
351 return true;
352}

354Status ReplaceMatchingOpTypes(
355 const GraphDef& input_graph_def, const OpTypePattern& pattern,
356 const std::function<Status(const NodeMatch&, const std::set<string>&,
357 const std::set<string>&, std::vector<NodeDef>*)>&
358 node_generator,
359 const ReplaceMatchingOpTypesOptions& options, GraphDef* output_graph_def) {
360 // Start off by retrieving all the matching subgraphs.
361 GraphMatcher matcher(input_graph_def);
362 std::vector<NodeMatch> matches;
363 TF_RETURN_IF_ERROR(matcher.GetOpTypeMatches(pattern, &matches));
364
365 // Do some housekeeping so we can easily look up the resulting matches given
366 // a node name.
367 std::set<string> matched_nodes;
368 std::map<string, const NodeMatch*> matches_by_head_name;
369 for (const NodeMatch& match : matches) {
370 matches_by_head_name[match.node.name()] = &match;
371 RecordMatchedNodes(match, &matched_nodes);
372 }
373 std::map<string, std::vector<const NodeDef*>> outputs_map;
374 MapNodesToOutputs(input_graph_def, &outputs_map);
375
376 // Go through all the nodes in the input graph, see if they are part of a
377 // match or if they can be left untouched.
378 output_graph_def->Clear();
379 for (const NodeDef& input_node : input_graph_def.node()) {
380 if (matches_by_head_name.count(input_node.name())) {
381 // This node is the beginning of a match, so call the replacement function
382 // after setting up some information it will need.
383 const NodeMatch* match = matches_by_head_name[input_node.name()];
384 std::vector<NodeDef> matched_nodes_array;
385 MatchedNodesAsArray(*match, &matched_nodes_array);
386 // This tells us whether a node is part of the current match.
387 std::set<string> matched_nodes_lookup;
388 for (const NodeDef& matched_node : matched_nodes_array) {
389 matched_nodes_lookup.insert(matched_node.name());
390 }
391 // These are helper arrays that the replacement function can use to tell
392 // whether it can safely remove an internal node (because nothing outside
393 // of the match uses it) or whether external nodes depend on it.
394 std::set<string> input_nodes;
395 std::set<string> output_nodes;
396 for (const NodeDef& matched_node : matched_nodes_array) {
397 // Look through all of this node's inputs, and if any of them come from
398 // outside the match, then this should be noted as one of the external
399 // inputs of the subgraph.
400 for (const string& input_name : matched_node.input()) {
401 string input_node_name = NodeNameFromInput(input_name);
402 if (!matched_nodes_lookup.count(input_node_name)) {
403 input_nodes.insert(matched_node.name());
404 }
405 }
406 // Do a reverse input lookup, to see which other nodes use the current
407 // one as an input. If any of those nodes are outside the match
408 // subgraph, then the current node is marked as an output node that
409 // shouldn't be removed.
410 if (outputs_map.count(matched_node.name())) {
411 for (const NodeDef* dependent_node :
412 outputs_map[matched_node.name()]) {
413 if (!matched_nodes_lookup.count(dependent_node->name())) {
414 output_nodes.insert(matched_node.name());
415 }
416 }
417 }
418 }
419 // Call the generator function and add all the returned nodes to the
420 // graph.
421 std::vector<NodeDef> new_nodes;
422 TF_RETURN_IF_ERROR(
423 node_generator(*match, input_nodes, output_nodes, &new_nodes));
424 std::set<string> new_node_names;
425 for (const NodeDef& new_node : new_nodes) {
426 new_node_names.insert(new_node.name());
427 }
428 // Check to make sure the generator function preserved all of the nodes
429 // that are used elsewhere in the graph, and add them back in if not.
430 bool abort_replacement = false;
431 if (!options.allow_inconsistencies) {
432 for (const string& expected_output : output_nodes) {
433 if (!new_node_names.count(expected_output)) {
434 LOG(WARNING) << "Expected " << expected_output
435 << " to be preserved.";
436 abort_replacement = true;
437 }
438 }
439 }
440 if (abort_replacement) {
441 LOG(WARNING) << "Generator function didn't preserve needed nodes, "
442 << "copying old replacements back in instead.";
443 std::vector<NodeDef> old_nodes;
444 MatchedNodesAsArray(*match, &old_nodes);
445 for (const NodeDef& old_node : old_nodes) {
446 NodeDef* added_node = output_graph_def->mutable_node()->Add();
447 *added_node = old_node;
448 }
449 } else {
450 for (const NodeDef& new_node : new_nodes) {
451 NodeDef* added_node = output_graph_def->mutable_node()->Add();
452 *added_node = new_node;
453 }
454 }
455 } else if (!matched_nodes.count(input_node.name())) {
456 // This node isn't part of any match, so just copy it over.
457 NodeDef* added_node = output_graph_def->mutable_node()->Add();
458 *added_node = input_node;
459 } else {
460 // Do nothing, because this is an internal part of a matching subgraph,
461 // and so will have been replaced by a new replacement subgraph.
462 }
463 }
464
465 return Status::OK();
466}

468Status RenameNodeInputs(const GraphDef& input_graph_def,
469 const std::map<string, string>& inputs_to_rename,
470 const std::unordered_set<string>& nodes_to_ignore,
471 GraphDef* output_graph_def) {
472 std::map<string, std::vector<std::pair<string, string>>>
473 canonical_inputs_to_rename;
474 for (const auto& input_to_rename : inputs_to_rename) {
475 canonical_inputs_to_rename[NodeNameFromInput(input_to_rename.first)]
476 .push_back({input_to_rename.first, input_to_rename.second});
477 }
478
479 output_graph_def->Clear();
480 for (const NodeDef& node : input_graph_def.node()) {
481 NodeDef* new_node = output_graph_def->mutable_node()->Add();
482 *new_node = node;
483 new_node->mutable_input()->Clear();
484 for (const string& input_name : node.input()) {
485 std::set<string> already_visited;
486 string new_input_name = input_name;
487 while (
488 canonical_inputs_to_rename.count(NodeNameFromInput(new_input_name))) {
489 string input_node_name = NodeNameFromInput(new_input_name);
490 if (already_visited.count(input_node_name)) {
491 return errors::InvalidArgument(
492 "RenameNodeInputs argument contains a cycle for ",
493 input_node_name);
494 }
495 already_visited.insert(input_node_name);
496 if (nodes_to_ignore.count(node.name())) {
497 break;
498 }
499 bool any_match_found = false;
500 for (const std::pair<string, string>& input_to_rename :
501 canonical_inputs_to_rename.at(input_node_name)) {
502 const string& source_name = input_to_rename.first;
503 const string& dest_name = input_to_rename.second;
504 bool is_match;
505 string match_name;
506 if (str_util::EndsWith(source_name, ":*")) {
507 is_match = true;
508 string prefix;
509 string unused_node_name;
510 string suffix;
511 NodeNamePartsFromInput(new_input_name, &prefix, &unused_node_name,
512 &suffix);
513 match_name = prefix + dest_name + suffix;
514 } else {
515 is_match = (CanonicalInputName(source_name) ==
516 CanonicalInputName(new_input_name));
517 match_name = dest_name;
518 }
519 if (is_match) {
520 new_input_name = match_name;
521 any_match_found = true;
522 }
523 }
524 if (!any_match_found) {
525 break;
526 }
527 }
528 *(new_node->mutable_input()->Add()) = new_input_name;
529 }
530 }
531 return Status::OK();
532}

548void FindInvalidInputs(const GraphDef& graph_def,
549 std::vector<std::pair<string, string>>* invalid_inputs) {
550 std::map<string, const NodeDef*> node_map;
551 MapNamesToNodes(graph_def, &node_map);
552
553 for (const NodeDef& node : graph_def.node()) {
554 for (const string& input : node.input()) {
555 string input_node = NodeNameFromInput(input);
556 if (!node_map.count(input_node)) {
557 invalid_inputs->push_back({node.name(), input_node});
558 }
559 }
560 }
561}

563Status IsGraphValid(const GraphDef& graph_def) {
564 std::vector<std::pair<string, string>> invalid_inputs;
565 FindInvalidInputs(graph_def, &invalid_inputs);
566 if (!invalid_inputs.empty()) {
567 std::map<string, const NodeDef*> node_map;
568 MapNamesToNodes(graph_def, &node_map);
569 for (const std::pair<string, string>& invalid_input : invalid_inputs) {
570 LOG(ERROR) << "Invalid input " << invalid_input.second << " for node "
571 << invalid_input.first << " - "
572 << node_map[invalid_input.first]->DebugString();
573 }
574 return errors::Internal(
575 "Invalid graph with inputs referring to nonexistent nodes");
576 }
577 return Status::OK();
578}

609Status TransformFuncContext::GetOneStringParameter(const string& name,
610 const string& default_value,
611 string* result) const {
612 const int params_count = CountParameters(name);
613 if (params_count == 0) {
614 *result = default_value;
615 return Status::OK();
616 } else if (params_count == 1) {
617 *result = params.at(name).at(0);
618 return Status::OK();
619 } else {
620 return errors::InvalidArgument("Expected a single '", name,
621 "' parameter, but found ", params_count,
622 " occurrences");
623 }

626Status TransformFuncContext::GetOneInt32Parameter(const string& name,
627 int32 default_value,
628 int32* result) const {

638 " argument as a number:", string_value);

660Status TransformFuncContext::GetOneFloatParameter(const string& name,
661 float default_value,
662 float* result) const {
663 const int params_count = CountParameters(name);
664 if (params_count == 0) {
665 *result = default_value;
666 return Status::OK();
667 }
668 string string_value;
669 TF_RETURN_IF_ERROR(GetOneStringParameter(name, "", &string_value));

671 return errors::InvalidArgument(
672 "Couldn't interpret the ", name,
673 " argument as a float number:", string_value);
674 }
675 return Status::OK();
676}
677
678Status TransformFuncContext::GetOneBoolParameter(const string& name,
679 bool default_value,
680 bool* result) const {
681 const int params_count = CountParameters(name);
682 if (params_count == 0) {
683 *result = default_value;
684 return Status::OK();
685 }
686 string string_value;
687 TF_RETURN_IF_ERROR(GetOneStringParameter(name, "", &string_value));
688 if (string_value == "true" || string_value == "1") {
689 *result = true;
690 } else if (string_value == "false" || string_value == "0") {
691 *result = false;
692 } else {
693 return errors::InvalidArgument("Couldn't interpret the ", name,
694 " argument as a boolean:", string_value,
695 " (expected true, false, 0 or 1)");
696 }
697 return Status::OK();
698}
