30// Renames all nodes not uses as graph inputs or outputs to short numerical
31// forms.
32Status ObfuscateNames(const GraphDef& input_graph_def,
33 const TransformFuncContext& context,
34 GraphDef* output_graph_def) {
35 std::unordered_set<string> required_nodes;
36 for (const string& input : context.input_names) {
37 required_nodes.insert(input);
38 }
39 for (const string& output : context.output_names) {
40 required_nodes.insert(output);
41 }
42
43 const string valid_chars =
44 "0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ";
45 const int64 chars_size = valid_chars.size();
46
47 std::map<string, string> new_names;
48 int64 name_index = 0;
49 for (const NodeDef& input_node : input_graph_def.node()) {
50 const string& old_name = input_node.name();
51 string new_name;
52 if (required_nodes.count(old_name)) {
53 new_name = old_name;
54 } else {
55 do {
56 int64 remaining = name_index;
57 new_name = "";
58 while (true) {
59 const int64 remainder = (remaining % chars_size);
60 const char current_char = valid_chars[remainder];
61 new_name = current_char + new_name;
62 remaining /= chars_size;
63 if (remaining <= 0) {
64 break;
65 }
66 }
67 ++name_index;
68 } while (required_nodes.count(new_name));
69 }
70 new_names[old_name] = new_name;
71 }
72
73 output_graph_def->Clear();
74 for (const NodeDef& input_node : input_graph_def.node()) {
75 NodeDef* node = output_graph_def->mutable_node()->Add();
76 *node = input_node;
77 const string& old_name = input_node.name();
78 node->set_name(new_names[old_name]);
79 node->mutable_input()->Clear();
80 for (const string& input_name : input_node.input()) {
81 string prefix;
82 string input_node_name;
83 string suffix;
84 NodeNamePartsFromInput(input_name, &prefix, &input_node_name, &suffix);
85 if (new_names.count(input_node_name) == 0) {
86 return errors::InvalidArgument("No node named ", input_node_name,
87 " for input to ", old_name);
88 }
89 string new_input_name = prefix + new_names[input_node_name] + suffix;
90 *(node->mutable_input()->Add()) = new_input_name;
91 }
92 }
93
94 return Status::OK();
95}
96
97REGISTER_GRAPH_TRANSFORM("obfuscate_names", ObfuscateNames);
