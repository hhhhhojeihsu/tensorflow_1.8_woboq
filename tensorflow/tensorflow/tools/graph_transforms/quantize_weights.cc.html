32// Converts any large float constants into eight-bit equivalents, with a
33// Dequantize op so that subsequent nodes can still access the results in a
34// float form.
35Status QuantizeWeights(const GraphDef& input_graph_def,
36 const TransformFuncContext& context,
37 GraphDef* output_graph_def) {
38 int32 minimum_size;
39 TF_RETURN_IF_ERROR(
40 context.GetOneInt32Parameter("minimum_size", 1024, &minimum_size));
41 TF_RETURN_IF_ERROR(ReplaceMatchingOpTypes(
42 input_graph_def, {"Const"},
43 [minimum_size](const NodeMatch& match,
44 const std::set<string>& input_nodes,
45 const std::set<string>& output_nodes,
46 std::vector<NodeDef>* new_nodes) {
47 const NodeDef& old_const_node = match.node;
48 if (!old_const_node.attr().count("dtype")) {
49 return errors::InvalidArgument("No 'dtype' attribute for Const node ",
50 old_const_node.name());
51 }
52 if (!old_const_node.attr().count("value")) {
53 return errors::InvalidArgument("No 'value' attribute for Const node ",
54 old_const_node.name());
55 }
56 const DataType old_dtype = old_const_node.attr().at("dtype").type();
57 Tensor old_tensor;
58 if (!old_tensor.FromProto(old_const_node.attr().at("value").tensor())) {
59 return errors::InvalidArgument("Decoding Tensor failed for node",
60 old_const_node.name());
61 }
62 const size_t num_elements = old_tensor.NumElements();
63 // If this isn't a float constant, or it's too small, then reuse the
64 // same node with no changes.
65 if ((old_dtype != DT_FLOAT) || (num_elements < minimum_size)) {
66 new_nodes->push_back(old_const_node);
67 return Status::OK();
68 }
69 const float* old_values = old_tensor.flat<float>().data();
70 float min = std::numeric_limits<float>::max();
71 float max = std::numeric_limits<float>::min();
72 for (int i = 0; i < num_elements; ++i) {
73 const float value = old_values[i];
74 min = std::min(min, value);
75 max = std::max(max, value);
76 }
77 // Make sure the quantization range includes 0.0f. Not all quantized
78 // Ops behave properly if 0.0f is not in the range.
79 min = std::min(min, 0.0f);
80 max = std::max(0.0f, max);
81 // min_value == max_value is a tricky case. It can occur for general
82 // tensors, and of course for scalars. The quantized ops cannot deal
83 // with this case, so we set max_value to something else.
84 // It's a tricky question what is the numerically best solution to
85 // deal with this degeneracy.
86 // TODO(petewarden): Better use a tolerance than a hard comparison?
87 if (min == max) {
88 if (std::abs(min) < 0.000001f) {
89 max = min + 1.0f;
90 } else if (min > 0) {
91 max = 2.0f * min;
92 } else {
93 max = min / 2.0f;
94 }
95 }
96 Tensor quantized_tensor(DT_QUINT8, old_tensor.shape());
97 FloatTensorToQuantizedInPlace<quint8>(old_tensor, min, max,
98 &quantized_tensor);
99
100 NodeDef quantized_const_node;
101 quantized_const_node.set_op("Const");
102 quantized_const_node.set_name(old_const_node.name() +
103 "_quantized_const");
104 SetNodeAttr("dtype", DT_QUINT8, &quantized_const_node);
105 SetNodeTensorAttr<float>("value", quantized_tensor,
106 &quantized_const_node);
107 new_nodes->push_back(quantized_const_node);
108
109 NodeDef min_node;
110 min_node.set_op("Const");
111 min_node.set_name(old_const_node.name() + "_quantized_min");
112 SetNodeAttr("dtype", DT_FLOAT, &min_node);
113 Tensor min_tensor(DT_FLOAT, {});
114 min_tensor.scalar<float>()() = min;
115 SetNodeTensorAttr<float>("value", min_tensor, &min_node);
116 new_nodes->push_back(min_node);
117
118 NodeDef max_node;
119 max_node.set_op("Const");
120 max_node.set_name(old_const_node.name() + "_quantized_max");
121 SetNodeAttr("dtype", DT_FLOAT, &max_node);
122 Tensor max_tensor(DT_FLOAT, {});
123 max_tensor.scalar<float>()() = max;
124 SetNodeTensorAttr<float>("value", max_tensor, &max_node);
125 new_nodes->push_back(max_node);
126
127 NodeDef dequantize_node;
128 dequantize_node.set_op("Dequantize");
129 dequantize_node.set_name(old_const_node.name());
130 SetNodeAttr("T", DT_QUINT8, &dequantize_node);
131 SetNodeAttr("mode", "MIN_FIRST", &dequantize_node);
132 AddNodeInput(quantized_const_node.name(), &dequantize_node);
133 AddNodeInput(min_node.name(), &dequantize_node);
134 AddNodeInput(max_node.name(), &dequantize_node);
135 new_nodes->push_back(dequantize_node);
136
137 return Status::OK();
138 },
139 {}, output_graph_def));
140
141 return Status::OK();
142}
143
144REGISTER_GRAPH_TRANSFORM("quantize_weights", QuantizeWeights);
