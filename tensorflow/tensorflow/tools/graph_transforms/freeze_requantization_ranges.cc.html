24struct MinMaxRecord {
25 string name;
26 float min;
27 float max;
28};
29
30// Try to parse a log file containing loosely-structured lines, some of which
31// are the min/max logs we want.
32Status ExtractMinMaxRecords(const string& log_file_name,
33 std::vector<MinMaxRecord>* records) {
34 string file_data;
35 TF_RETURN_IF_ERROR(
36 ReadFileToString(Env::Default(), log_file_name, &file_data));
37 const string print_suffix("__print__");
38 const string requant_prefix("__requant_min_max:");
39 std::vector<string> file_lines = str_util::Split(file_data, '\n');
40 for (const string& file_line : file_lines) {
41 // We expect to find a line with components separated by semicolons, so to
42 // start make sure that the basic structure is in place/
43 if (!str_util::StrContains(file_line,
44 print_suffix + ";" + requant_prefix)) {
45 continue;
46 }
47 std::vector<string> line_parts = str_util::Split(file_line, ';');
48 if (line_parts.size() < 2) {
49 continue;
50 }
51 // Now we want to figure out which components have the name and min max
52 // values by scanning for the prefix we expect.
53 bool min_max_found = false;
54 int min_max_index;
55 for (int i = 1; i < line_parts.size(); ++i) {
56 if (str_util::StartsWith(line_parts[i], requant_prefix)) {
57 min_max_found = true;
58 min_max_index = i;
59 }
60 }
61 if (!min_max_found) {
62 continue;
63 }
64 // Finally we need to break out the values from the strings, and parse them
65 // into a form we can use.
66 string min_max_string = line_parts[min_max_index];
67 std::vector<string> min_max_parts = str_util::Split(min_max_string, '[');
68 if ((min_max_parts.size() != 3) || (min_max_parts[0] != requant_prefix)) {
69 continue;
70 }
71 string min_string = min_max_parts[1];
72 std::vector<string> min_string_parts = str_util::Split(min_string, ']');
73 if (min_string_parts.size() != 2) {
74 continue;
75 }
76 string min_number_string = min_string_parts[0];
77 float min;
78 if (!strings::safe_strtof(min_number_string.c_str(), &min)) {
79 continue;
80 }
81 string max_string = min_max_parts[2];
82 std::vector<string> max_string_parts = str_util::Split(max_string, ']');
83 if (max_string_parts.size() != 2) {
84 continue;
85 }
86 string max_number_string = max_string_parts[0];
87 float max;
88 if (!strings::safe_strtof(max_number_string.c_str(), &max)) {
89 continue;
90 }
91 StringPiece name_string = line_parts[min_max_index - 1];
92 if (!str_util::EndsWith(name_string, print_suffix)) {
93 continue;
94 }
95 string name =
96 name_string.substr(0, name_string.size() - print_suffix.size())
97 .ToString();
98 records->push_back({name, min, max});
99 }
100 return Status::OK();
101}

103// Uses the observed min/max values for requantization captured in a log file to
104// replace costly RequantizationRange ops with simple Consts.
105Status FreezeRequantizationRanges(const GraphDef& input_graph_def,
106 const TransformFuncContext& context,
107 GraphDef* output_graph_def) {
108 string min_max_log_file;
109 TF_RETURN_IF_ERROR(
110 context.GetOneStringParameter("min_max_log_file", "", &min_max_log_file));
111 if (min_max_log_file.empty()) {
112 return errors::InvalidArgument(
113 "You must pass a file name to min_max_log_file");
114 }
115 float min_percentile;
116 TF_RETURN_IF_ERROR(
117 context.GetOneFloatParameter("min_percentile", 5.0f, &min_percentile));
118 float max_percentile;
119 TF_RETURN_IF_ERROR(
120 context.GetOneFloatParameter("max_percentile", 5.0f, &max_percentile));
121
122 std::vector<MinMaxRecord> records;
123 TF_RETURN_IF_ERROR(ExtractMinMaxRecords(min_max_log_file, &records));
124 if (records.empty()) {
125 return errors::InvalidArgument(
126 "No min/max range logs were found in the log file");
127 }
128
129 std::map<string, const NodeDef*> node_map;
130 MapNamesToNodes(input_graph_def, &node_map);
131 bool any_missing_nodes = false;
132 std::map<string, std::vector<MinMaxRecord>> records_by_node;
133 for (const MinMaxRecord& record : records) {
134 records_by_node[record.name].push_back(record);
135 if (!node_map.count(record.name)) {
136 any_missing_nodes = true;
137 LOG(WARNING) << "Node from log not found in graph: " << record.name;
138 }
139 }
140 if (any_missing_nodes) {
141 return errors::InvalidArgument(
142 "Nodes were found in the log file that aren't present in the graph");
143 }
144
145 // Now find out the largest and smallest min/max values for the node.
146 std::map<string, std::pair<float, float>> range_for_nodes;
147 for (const auto& record_info : records_by_node) {
148 const string& name = record_info.first;
149 const std::vector<MinMaxRecord> records = record_info.second;
150 std::vector<float> mins;
151 std::vector<float> maxs;
152 for (const MinMaxRecord& record : records) {
153 mins.push_back(record.min);
154 maxs.push_back(record.max);
155 }
156 std::sort(mins.begin(), mins.end());
157 std::sort(maxs.begin(), maxs.end());
158 int min_index = std::round(mins.size() * (min_percentile / 100.0f));
159 if (min_index < 0) {
160 min_index = 0;
161 }
162 int max_index =
163 std::round(maxs.size() * (1.0f - (max_percentile / 100.0f)));
164 if (max_index > (maxs.size() - 1)) {
165 max_index = maxs.size() - 1;
166 }
167 const float min = mins[min_index];
168 const float max = maxs[max_index];
169 range_for_nodes[name] = {min, max};
170 }
171 std::map<string, string> inputs_to_rename;
172 GraphDef frozen_graph_def;
173 for (const NodeDef& node : input_graph_def.node()) {
174 if (range_for_nodes.count(node.name())) {
175 if (node.op() != "RequantizationRange") {
176 return errors::InvalidArgument(
177 "Node is expected to be a RequantizationRange op: ", node.name(),
178 ", but is: ", node.op());
179 }
180 const float min_value = range_for_nodes.at(node.name()).first;
181 NodeDef* min_node = frozen_graph_def.mutable_node()->Add();
182 min_node->set_op("Const");
183 min_node->set_name(node.name() + "/frozen_min");
184 SetNodeAttr("dtype", DT_FLOAT, min_node);
185 Tensor min_tensor(DT_FLOAT, {});
186 min_tensor.flat<float>()(0) = min_value;
187 SetNodeTensorAttr<float>("value", min_tensor, min_node);
188 inputs_to_rename[node.name() + ":0"] = min_node->name() + ":0";
189
190 const float max_value = range_for_nodes.at(node.name()).second;
191 NodeDef* max_node = frozen_graph_def.mutable_node()->Add();
192 max_node->set_op("Const");
193 max_node->set_name(node.name() + "/frozen_max");
194 SetNodeAttr("dtype", DT_FLOAT, max_node);
195 Tensor max_tensor(DT_FLOAT, {});
196 max_tensor.flat<float>()(0) = max_value;
197 SetNodeTensorAttr<float>("value", max_tensor, max_node);
198 inputs_to_rename[node.name() + ":1"] = max_node->name() + ":0";
199 } else {
200 NodeDef* new_node = frozen_graph_def.mutable_node()->Add();
201 *new_node = node;
202 }
203 }
204 return RenameNodeInputs(frozen_graph_def, inputs_to_rename,
205 std::unordered_set<string>(), output_graph_def);
206}

209 FreezeRequantizationRanges);
