22namespace internal_statusor {

24void Helper::HandleInvalidStatusCtorArg(Status* status) {
25 const char* kMessage =
26 "An OK status is not a valid constructor argument to StatusOr<T>";
27 LOG(ERROR) << kMessage;
28 // Fall back to tensorflow::error::INTERNAL.
29 *status = ::tensorflow::errors::Internal(kMessage);
30}
31
32void Helper::Crash(const Status& status) {
33 LOG(FATAL) << "Attempting to fetch value instead of handling error "
34 << status;
35}
