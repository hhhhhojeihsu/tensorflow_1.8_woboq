37namespace xla {

39template <typename T>
40class Array2D : public Array<T> {
41 public:
42 Array2D() : Array<T>(std::vector<int64>{0, 0}) {}
43
44 Array2D(const int64 n1, const int64 n2)
45 : Array<T>(std::vector<int64>{n1, n2}) {}
46
47 Array2D(const int64 n1, const int64 n2, const T value)
48 : Array<T>({n1, n2}, value) {}
49
50 // Creates an array from the given nested initializer list. The outer
51 // initializer list is the first dimension; the inner is the second dimension.
52 // For example, {{1, 2, 3}, {4, 5, 6}} results in an array with n1=2 and n2=3.
53 Array2D(std::initializer_list<std::initializer_list<T>> values)
54 : Array<T>(values) {}
55
56 // Creates an array of a floating-point type (half, bfloat16, float,
57 // or double) from the given nested initializer list of float values.
58 template <typename T2, typename = typename std::enable_if<
59 (std::is_same<T, Eigen::half>::value ||
60 std::is_same<T, bfloat16>::value ||
61 std::is_same<T, float>::value ||
62 std::is_same<T, double>::value) &&
63 std::is_same<T2, float>::value>::type>
64 Array2D(std::initializer_list<std::initializer_list<T2>> values)
65 : Array<T>(values) {}
66
67 Array2D(const Array2D<T>& other) : Array<T>(other) {}
68
69 int64 n1() const { return this->dim(0); }
70 int64 n2() const { return this->dim(1); }
71
72 int64 height() const { return this->dim(0); }
73 int64 width() const { return this->dim(1); }
74
75 // Fills the array with a pattern of values of the form:
76 //
77 // (rowno << log2ceil(width) | colno) + start_value
78 //
79 // This makes it easy to see distinct row/column values in the array.
80 void FillUnique(T start_value = 0) {
81 for (int64 i0 = 0; i0 < n1(); ++i0) {
82 for (int64 i1 = 0; i1 < n2(); ++i1) {
83 (*this)(i0, i1) =
84 ((i0 << tensorflow::Log2Ceiling64(n2())) | i1) + start_value;
85 }
86 }
87 }
88
89 // Applies f to all cells in this array, in row-major order.
90 void Each(std::function<void(int64, int64, T*)> f) {
91 for (int64 i0 = 0; i0 < n1(); ++i0) {
92 for (int64 i1 = 0; i1 < n2(); ++i1) {
93 f(i0, i1, &(*this)(i0, i1));
94 }
95 }
96 }
97};
98
99// Returns a linspace-populated Array2D in the range [from, to] (inclusive)
100// with dimensions n1 x n2.
101template <typename NativeT = float>
102std::unique_ptr<Array2D<NativeT>> MakeLinspaceArray2D(double from, double to,
103 int64 n1, int64 n2) {
104 auto array = MakeUnique<Array2D<NativeT>>(n1, n2);
105 int64 count = n1 * n2;
106 NativeT step =
107 static_cast<NativeT>((count > 1) ? (to - from) / (count - 1) : 0);
108 auto set = [&array, n1, n2](int64 index, NativeT value) {
109 (*array)(index / n2, index % n2) = value;
110 };
111 for (int64 i = 0; i < count - 1; ++i) {
112 set(i, (static_cast<NativeT>(from) +
113 static_cast<NativeT>(i) * static_cast<NativeT>(step)));
114 }
115 set(count - 1, static_cast<NativeT>(to));
116 return array;
117}
