25class Helper {
26 public:
27 // Move type-agnostic error handling to the .cc.
28 static void HandleInvalidStatusCtorArg(Status*);
29 TF_ATTRIBUTE_NORETURN static void Crash(const Status& status);

32// Construct an instance of T in `p` through placement new, passing Args... to
33// the constructor.
34// This abstraction is here mostly for the gcc performance fix.
35template <typename T, typename... Args>
36void PlacementNew(void* p, Args&&... args) {
37#if defined(__GNUC__) && !defined(__clang__)
38 // Teach gcc that 'p' cannot be null, fixing code size issues.
39 if (p == nullptr) __builtin_unreachable();
40#endif
41 new (p) T(std::forward<Args>(args)...);
42}

44// Helper base class to hold the data and all operations.
45// We move all this to a base class to allow mixing with the appropriate
46// TraitsBase specialization.
47template <typename T>
48class StatusOrData {
49 template <typename U>
50 friend class StatusOrData;
51
52 public:
53 StatusOrData() = delete;
54
55 StatusOrData(const StatusOrData& other) {
56 if (other.ok()) {
57 MakeValue(other.data_);
58 MakeStatus();
59 } else {
60 MakeStatus(other.status_);
61 }
62 }
63
64 StatusOrData(StatusOrData&& other) noexcept {
65 if (other.ok()) {
66 MakeValue(std::move(other.data_));
67 MakeStatus();
68 } else {
69 MakeStatus(std::move(other.status_));
70 }
71 }
72
73 template <typename U>
74 StatusOrData(const StatusOrData<U>& other) {
75 if (other.ok()) {
76 MakeValue(other.data_);
77 MakeStatus();
78 } else {
79 MakeStatus(other.status_);
80 }
81 }
82
83 template <typename U>
84 StatusOrData(StatusOrData<U>&& other) {
85 if (other.ok()) {
86 MakeValue(std::move(other.data_));
87 MakeStatus();
88 } else {
89 MakeStatus(std::move(other.status_));
90 }
91 }
92
93 explicit StatusOrData(const T& value) : data_(value) { MakeStatus(); }
94 explicit StatusOrData(T&& value) : data_(std::move(value)) { MakeStatus(); }
95
96 explicit StatusOrData(const Status& status) : status_(status) {
97 EnsureNotOk();
98 }
99 explicit StatusOrData(Status&& status) : status_(std::move(status)) {
100 EnsureNotOk();
101 }
102
103 StatusOrData& operator=(const StatusOrData& other) {
104 if (this == &other) return *this;
105 if (other.ok())
106 Assign(other.data_);
107 else
108 Assign(other.status_);
109 return *this;
110 }
111
112 StatusOrData& operator=(StatusOrData&& other) {
113 if (this == &other) return *this;
114 if (other.ok())
115 Assign(std::move(other.data_));
116 else
117 Assign(std::move(other.status_));
118 return *this;
119 }
120
121 ~StatusOrData() {
122 if (ok()) {
123 status_.~Status();
124 data_.~T();
125 } else {
126 status_.~Status();
127 }
128 }
129
130 void Assign(const T& value) {
131 if (ok()) {
132 data_.~T();
133 MakeValue(value);
134 } else {
135 MakeValue(value);
136 status_ = Status::OK();
137 }
138 }
139
140 void Assign(T&& value) {
141 if (ok()) {
142 data_.~T();
143 MakeValue(std::move(value));
144 } else {
145 MakeValue(std::move(value));
146 status_ = Status::OK();
147 }
148 }
149
150 void Assign(const Status& status) {
151 Clear();
152 status_ = status;
153 EnsureNotOk();
154 }
155
156 void Assign(Status&& status) {
157 Clear();
158 status_ = std::move(status);
159 EnsureNotOk();
160 }
161
162 bool ok() const { return status_.ok(); }
163
164 protected:
165 // status_ will always be active after the constructor.
166 // We make it a union to be able to initialize exactly how we need without
167 // waste.
168 // Eg. in the copy constructor we use the default constructor of Status in
169 // the ok() path to avoid an extra Ref call.
170 union {
171 Status status_;
172 };
173
174 // data_ is active iff status_.ok()==true
175 struct Dummy {};
176 union {
177 // When T is const, we need some non-const object we can cast to void* for
178 // the placement new. dummy_ is that object.
179 Dummy dummy_;
180 T data_;
181 };
182
183 void Clear() {
184 if (ok()) data_.~T();
185 }
186
187 void EnsureOk() const {
188 if (!ok()) Helper::Crash(status_);
189 }
190
191 void EnsureNotOk() {
192 if (ok()) Helper::HandleInvalidStatusCtorArg(&status_);
193 }
194
195 // Construct the value (ie. data_) through placement new with the passed
196 // argument.
197 template <typename Arg>
198 void MakeValue(Arg&& arg) {
199 internal_statusor::PlacementNew<T>(&dummy_, std::forward<Arg>(arg));
200 }
201
202 // Construct the status (ie. status_) through placement new with the passed
203 // argument.
204 template <typename... Args>
205 void MakeStatus(Args&&... args) {
206 internal_statusor::PlacementNew<Status>(&status_,
207 std::forward<Args>(args)...);
208 }
209};

211// Helper base class to allow implicitly deleted constructors and assignment
212// operations in StatusOr.
213// TraitsBase will explicitly delete what it can't support and StatusOr will
214// inherit that behavior implicitly.
215template <bool Copy, bool Move>
216struct TraitsBase {
217 TraitsBase() = default;
218 TraitsBase(const TraitsBase&) = default;
219 TraitsBase(TraitsBase&&) = default;
220 TraitsBase& operator=(const TraitsBase&) = default;
221 TraitsBase& operator=(TraitsBase&&) = default;
222};
223
224template <>
225struct TraitsBase<false, true> {
226 TraitsBase() = default;
227 TraitsBase(const TraitsBase&) = delete;
228 TraitsBase(TraitsBase&&) = default;
229 TraitsBase& operator=(const TraitsBase&) = delete;
230 TraitsBase& operator=(TraitsBase&&) = default;
231};
232
233template <>
234struct TraitsBase<false, false> {
235 TraitsBase() = default;
236 TraitsBase(const TraitsBase&) = delete;
237 TraitsBase(TraitsBase&&) = delete;
238 TraitsBase& operator=(const TraitsBase&) = delete;
239 TraitsBase& operator=(TraitsBase&&) = delete;
240};
