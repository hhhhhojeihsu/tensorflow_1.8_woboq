33void SetDebugOptionsDefaults(DebugOptions* flags) {
34 flags->set_xla_enable_fast_math(true);
35 flags->set_xla_llvm_enable_alias_scope_metadata(true);
36 flags->set_xla_llvm_enable_noalias_metadata(true);
37 flags->set_xla_llvm_enable_invariant_load_metadata(true);
38 flags->set_xla_llvm_disable_expensive_passes(false);
39 flags->set_xla_backend_optimization_level(3);
40 flags->set_xla_cpu_multi_thread_eigen(true);
41 flags->set_xla_gpu_cuda_data_dir("./cuda_sdk_lib");
42 flags->set_xla_eliminate_hlo_implicit_broadcast(true);
43#ifdef INTEL_MKL
44 flags->set_xla_cpu_use_mkl_dnn(true);
45#endif // INTEL_MKL
46
47 // Set cudnn batchnorm off by default; it does not provide a performance win
48 // on average.
49 flags->set_xla_gpu_use_cudnn_batchnorm(false);
50}

52// Allocates flag_values and flag_objects; this function must not be called more
53// than once - its call done via call_once.
54void AllocateFlags() {
55 flag_values = new DebugOptions;
56
57 SetDebugOptionsDefaults(flag_values);
58
59 // Returns a lambda that calls "member_setter" on "flag_values" with the
60 // argument passed in to the lambda.
61 auto bool_setter_for = [](void (DebugOptions::*member_setter)(bool)) {
62 return [member_setter](bool value) {
63 (flag_values->*member_setter)(value);
64 return true;
65 };
66 };
67
68 // Returns a lambda that calls "member_setter" on "flag_values" with the
69 // argument passed in to the lambda.
70 auto int32_setter_for = [](void (DebugOptions::*member_setter)(int32)) {
71 return [member_setter](int32 value) {
72 (flag_values->*member_setter)(value);
73 return true;
74 };
75 };
76
77 // Custom "sub-parser" lambda for xla_disable_hlo_passes.
78 auto setter_for_xla_disable_hlo_passes = [](string comma_separated_values) {
79 std::vector<string> disabled_passes =
80 tensorflow::str_util::Split(comma_separated_values, ',');
81 for (const auto& passname : disabled_passes) {
82 flag_values->add_xla_disable_hlo_passes(passname);
83 }
84 return true;
85 };
86
87 // Custom "sub-parser" lambda for xla_backend_extra_options.
88 auto setter_for_xla_backend_extra_options =
89 [](string comma_separated_values) {
90 auto* extra_options_map =
91 flag_values->mutable_xla_backend_extra_options();
92 impl::parse_xla_backend_extra_options(extra_options_map,
93 comma_separated_values);
94 return true;
95 };
96
97 // Custom "sub-parser" lambda for xla_reduce_precision.
98 auto setter_for_xla_reduce_precision =
99 [](string reduce_precision_option_value) {
100 HloReducePrecisionOptions* option_proto =
101 flag_values->add_hlo_reduce_precision_options();
102 return impl::parse_xla_reduce_precision_option(
103 option_proto, reduce_precision_option_value);
104 };

304void AppendDebugOptionsFlags(std::vector<tensorflow::Flag>* flag_list) {
305 std::call_once(flags_init, &AllocateFlags);
306 flag_list->insert(flag_list->end(), flag_objects->begin(),
307 flag_objects->end());
308}
309
310xla::DebugOptions GetDebugOptionsFromFlags() {
311 std::call_once(flags_init, &AllocateFlags);
312 return *flag_values;
313}
