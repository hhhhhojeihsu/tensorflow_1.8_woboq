58// RAII timer that logs with a given label the wall clock time duration in human
59// readable form. This differs from base's ElapsedTimer primarily in that it
60// spits out the human-readable duration form.
61//
62// By default, the timing traces are only printed at VLOG(1) and above:
63//
64// XLA_SCOPED_LOGGING_TIMER("fooing bar"); // nop if !VLOG_IS_ON(1).
65//
66// but you can control this via:
67//
68// XLA_SCOPED_LOGGING_TIMER_LEVEL("fooing bar", 2); // nop if !VLOG_IS_ON(2)
69//
70#define XLA_SCOPED_LOGGING_TIMER(label) \
71 XLA_SCOPED_LOGGING_TIMER_HELPER(label, 1, __COUNTER__)
72#define XLA_SCOPED_LOGGING_TIMER_LEVEL(label, level) \
73 XLA_SCOPED_LOGGING_TIMER_HELPER(label, level, __COUNTER__)
74
75// Helper for implementing macros above. Do not use directly.
76//
77// Forces the evaluation of "counter", which we expect is equal to __COUNTER__.
78#define XLA_SCOPED_LOGGING_TIMER_HELPER(label, level, counter) \
79 XLA_SCOPED_LOGGING_TIMER_HELPER2(label, level, counter)
80
81// Helper for macros above. Don't use directly.
82#define XLA_SCOPED_LOGGING_TIMER_HELPER2(label, level, counter) \
83 ::xla::ScopedLoggingTimer XLA_ScopedLoggingTimerInstance##counter( \
84 label, VLOG_IS_ON(level))
85
86// RAII timer for XLA_SCOPED_LOGGING_TIMER and XLA_SCOPED_LOGGING_TIMER_LEVEL
87// macros above. Recommended usage is via the macros so you don't have to give
88// the timer a name or worry about calling VLOG_IS_ON yourself.
89struct ScopedLoggingTimer {
90 // The timer does nothing if enabled is false. This lets you pass in your
91 // file's VLOG_IS_ON value.
92 ScopedLoggingTimer(const string& label, bool enabled);
93 ~ScopedLoggingTimer();
94
95 bool enabled;
96 string label;
97 uint64 start_micros;
98};

161// Compares two containers for equality. Returns true iff the two containers
162// have the same size and all their elements compare equal using their
163// operator==. Like std::equal, but forces size equality.
164template <typename Container1T, typename Container2T>
165bool ContainersEqual(const Container1T& c1, const Container2T& c2) {
166 return ((c1.size() == c2.size()) &&
167 std::equal(std::begin(c1), std::end(c1), std::begin(c2)));
168}
169
170template <typename Container1T,
171 typename ElementType = typename Container1T::value_type>
172bool ContainersEqual(const Container1T& c1,
173 std::initializer_list<ElementType> il) {
174 tensorflow::gtl::ArraySlice<ElementType> c2{il};
175 return ContainersEqual(c1, c2);
176}

178// Compares two containers for equality. Returns true iff the two containers
179// have the same size and all their elements compare equal using the predicate
180// p. Like std::equal, but forces size equality.
181template <typename Container1T, typename Container2T, class PredicateT>
182bool ContainersEqual(const Container1T& c1, const Container2T& c2,
183 PredicateT p) {
184 return ((c1.size() == c2.size()) &&
185 std::equal(std::begin(c1), std::end(c1), std::begin(c2), p));
186}

188// Performs a copy of count values from src to dest, using different strides for
189// source and destination. The source starting index is src_base, while the
190// destination one is dest_base.
191template <typename D, typename S>
192void StridedCopy(tensorflow::gtl::MutableArraySlice<D> dest, int64 dest_base,
193 int64 dest_stride, tensorflow::gtl::ArraySlice<S> src,
194 int64 src_base, int64 src_stride, int64 count) {
195 for (; count > 0; --count, dest_base += dest_stride, src_base += src_stride) {
196 dest[dest_base] = static_cast<D>(src[src_base]);
197 }
198}

248// Checks whether permutation is a permutation of the [0, rank) integer range.

251// Applies `permutation` on `input` and returns the permuted array.
252// For each i, output[permutation[i]] = input[i].
253//
254// Precondition:
255// 1. `permutation` is a permutation of 0..permutation.size()-1.
256// 2. permutation.size() == input.size().
257template <template <typename...> class C, typename T>
258std::vector<T> Permute(tensorflow::gtl::ArraySlice<int64> permutation,
259 C<T> input) {
260 tensorflow::gtl::ArraySlice<T> data(input);
261 CHECK(IsPermutation(permutation, data.size()));
262 std::vector<T> output(data.size());
263 for (size_t i = 0; i < permutation.size(); ++i) {
264 output[permutation[i]] = data[i];
265 }
266 return output;
267}

297// Formats the container as a comma-separated string. StrAppend must support
298// appending the elements of the container. Prefix is prepended and suffix is
299// appended to the returned string.
300template <typename Container>
301string CommaSeparatedString(const Container& c, const char* prefix = "",
302 const char* suffix = "") {
303 // Not using Join() since the implementation here is simple anyway and this
304 // avoids copying the string to append prefix.
305 string comma_separated = prefix;
306 const char* separator = "";
307 for (const auto& entry : c) {
308 tensorflow::strings::StrAppend(&comma_separated, separator, entry);
309 separator = ", ";
310 }
311 comma_separated += suffix;
312 return comma_separated;
313}

363// Rounds the value up to a multiple of the divisor by first calling CeilOfRatio
364// then multiplying by the divisor. For example: RoundUpToNearest(13, 8) => 16
365template <typename T>
366T RoundUpToNearest(T value, T divisor) {
367 return CeilOfRatio(value, divisor) * divisor;
368}

370// Rounds the value down to a multiple of the divisor by first calling
371// FloorOfRatio then multiplying by the divisor. For example:
372// RoundDownToNearest(13, 8) => 8
373template <typename T>
374T RoundDownToNearest(T value, T divisor) {
375 return FloorOfRatio(value, divisor) * divisor;

413// Returns the start indices of consecutive non-overlapping subsequences of `a`
414// and `b` with the same product, i.e. `(i, j)` so
415// • a = {a[0 = i_0], ..., a[i_1 - 1], a[i_1], ... , a[i_2 - 1], ...}
416// • b = {b[0 = j_0], ..., b[j_1 - 1], b[j_1], ... , b[j_2 - 1], ...}

419// b[j_k] × b[j_k + 1] × ... × b[j_(k+1) - 1]
420// where `CommonFactors(a, b)[CommonFactors(a, b).size - 1] = (a.size, b.size)`
