16#include "tensorflow/compiler/xla/service/hlo_subcomputation_unification.h"

18#include <unordered_map>
19
20namespace xla {
21
22StatusOr<bool> HloSubcomputationUnification::Run(HloModule* module) {
23 // For each computation C in the module, find the first computation C0 in the
24 // computations_ list that is identical to C, and adds canon[C] = C0.
25 std::unordered_map<HloComputation*, HloComputation*> canon;
26 const auto& computations = module->computations();
27 for (auto i = computations.begin(); i != computations.end(); ++i) {
28 for (auto j = computations.begin(); j != i; ++j) {
29 // Do not waste time comparing `*i` with `*j` if `*j` is not canonical.
30 if (canon.find(*j) == canon.end() && **i == **j) {
31 canon[*i] = *j;
32 break;
33 }
34 }
35 }
36
37 if (canon.empty()) {
38 return false;
39 }
40
41 module->ReplaceComputations(canon);
42 return true;
43}
