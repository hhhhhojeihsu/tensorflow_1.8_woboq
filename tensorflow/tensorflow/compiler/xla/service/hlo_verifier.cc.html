112Status ShapeVerifier::HandleOutfeed(HloInstruction* outfeed) {
113 // Outfeed has a separate shape field for the value which is outfed to the
114 // host. The shape of the instruction itself is always nil because the outfeed
115 // produces no HLO value in the graph.
116 if (!ShapeUtil::Compatible(outfeed->outfeed_shape(),
117 outfeed->operand(0)->shape())) {
118 return InvalidArgument(
119 "Expected outfeed to have shape compatible with operand's shape %s, "
120 "actual shape is %s:\n%s",
121 ShapeUtil::HumanString(outfeed->operand(0)->shape()).c_str(),
122 ShapeUtil::HumanString(outfeed->outfeed_shape()).c_str(),
123 outfeed->ToString().c_str());
124 }
125 return CheckShape(outfeed, ShapeUtil::MakeNil());
126}

169Status ShapeVerifier::HandleBroadcast(HloInstruction* broadcast) {
170 // HLO broadcast has no exact analog at the proto level so there is no
171 // ShapeInference method. Check the output shape explicitly.
172 const Shape& operand_shape = broadcast->operand(0)->shape();
173 // Check for mixed precision.
174 TF_RETURN_IF_ERROR(CheckShape(broadcast, broadcast->shape()));
175 TF_RET_CHECK(ShapeUtil::Rank(operand_shape) ==
176 broadcast->dimensions().size());
177 for (int64 i = 0; i < ShapeUtil::Rank(operand_shape); ++i) {
178 int64 output_dimension = broadcast->dimensions()[i];
179 TF_RET_CHECK(broadcast->shape().dimensions(output_dimension) ==
180 operand_shape.dimensions(i))
181 << broadcast->ToString() << " operand shape " << operand_shape;
182 }
183 return tensorflow::Status::OK();
184}
185
186Status ShapeVerifier::HandleBroadcastDimOne(HloInstruction* broadcastDimOne) {
187 const Shape& operand_shape = broadcastDimOne->operand(0)->shape();
188 int64 operand_rank = ShapeUtil::Rank(operand_shape);
189 const Shape& output_shape = broadcastDimOne->shape();
190 // Check for mixed precision.
191 TF_RETURN_IF_ERROR(CheckShape(broadcastDimOne, output_shape));
192 TF_RET_CHECK(operand_rank == ShapeUtil::Rank(output_shape));
193 for (int64 i = 0; i < operand_rank; ++i) {
194 int64 operand_dimension = operand_shape.dimensions(i);
195 int64 output_dimension = output_shape.dimensions(i);
196 TF_RET_CHECK(operand_dimension == 1 ||
197 operand_dimension == output_dimension)
198 << "Dimension " << i << " of broadcastDimOne "
199 << broadcastDimOne->ToString() << " is " << operand_dimension
200 << ", expected 1 or " << output_dimension;
201 }
202 return tensorflow::Status::OK();
203}

263Status ShapeVerifier::HandleMap(HloInstruction* map) {
264 std::vector<const Shape*> operand_shapes;
265 int64 max_operand_rank = 0;
266 for (const HloInstruction* operand : map->operands()) {
267 operand_shapes.push_back(&operand->shape());
268 max_operand_rank =
269 std::max(max_operand_rank, ShapeUtil::Rank(operand->shape()));
270 }
271 // TODO(b/65689298) Remove code below once Map is generalized to accept
272 // arbitrary map dimensions.

274 std::iota(map_dims.begin(), map_dims.end(), 0);
275 return CheckShape(map, ShapeInference::InferMapShape(
276 operand_shapes,
277 map->to_apply()->ComputeProgramShape(), map_dims));
278}

391// Checks that the instruction does not have mixed precision floating point
392// inputs.
393Status CheckMixedPrecisionOperands(const HloInstruction* instruction) {
394 switch (instruction->opcode()) {
395 // White list the following opcodes for mixed-precision check, because they
396 // involve data pass through or grouping via tuples, where the precisions
397 // of buffers can be different.
398 case HloOpcode::kCall:
399 case HloOpcode::kConditional:
400 case HloOpcode::kConstant:
401 case HloOpcode::kCrossReplicaSum:
402 case HloOpcode::kCustomCall:
403 case HloOpcode::kFusion:
404 case HloOpcode::kGetTupleElement:
405 case HloOpcode::kInfeed:
406 case HloOpcode::kOutfeed:
407 case HloOpcode::kParameter:
408 case HloOpcode::kRecv:
409 case HloOpcode::kRecvDone:
410 case HloOpcode::kReducePrecision:
411 case HloOpcode::kSelect:
412 case HloOpcode::kSend:
413 case HloOpcode::kSendDone:
414 case HloOpcode::kTuple:
415 case HloOpcode::kWhile:
416 break;
417 default: {
418 PrimitiveType fp_type = PRIMITIVE_TYPE_INVALID;
419 for (auto operand : instruction->operands()) {
420 TF_RETURN_IF_ERROR(ShapeUtil::ForEachSubshapeWithStatus(
421 operand->shape(),
422 [&](const Shape& subshape, const ShapeIndex& index) {
423 if (!ShapeUtil::ElementIsFloating(subshape)) {
424 return Status::OK();
425 }
426 if (fp_type == PRIMITIVE_TYPE_INVALID) {
427 fp_type = subshape.element_type();
428 } else if (fp_type != subshape.element_type()) {
429 return FailedPrecondition(
430 "Seen floating point types of different precisions in "
431 "%s, but mixed precision is disallowed.",
432 instruction->ToString().c_str());
433 }
434 return Status::OK();
435 }));
436 }
437 }
438 }
439 return Status::OK();
440}

452Status ShapeVerifier::CheckShape(const HloInstruction* instruction,
453 const Shape& inferred_shape) {
454 // If allow_mixed_precision_ is false, check if there are operands with
455 // different precisions. We need this check because ShapeInference allows
456 // mixed precision inputs.
457 if (!allow_mixed_precision_) {
458 TF_RETURN_IF_ERROR(CheckMixedPrecisionOperands(instruction));
459 }
460
461 // Check if the output shape matches the expected shape.
462 bool compatible;
463 // We treat BF16 and F32 as compatible types if mixed precision is allowed,
464 // but only when the instruction defines the BF16/F32 buffer.
465 switch (instruction->opcode()) {
466 case HloOpcode::kSelect:
467 if (ShapeUtil::IsTuple(inferred_shape) || !allow_mixed_precision_) {
468 // Select only defines the top-level buffer, which in this case is the
469 // tuple, so we cannot allow mixed precision.
470 compatible =
471 ShapeUtil::Compatible(instruction->shape(), inferred_shape);
472 } else {
473 compatible = ShapeUtil::CompatibleIgnoringFpPrecision(
474 instruction->shape(), inferred_shape);
475 }
476 break;
477 case HloOpcode::kGetTupleElement:
478 case HloOpcode::kTuple:
479 // Tuple and GetTupleElement do not define BF16/F32 buffers, so mixed
480 // precision is disallowed.
481 case HloOpcode::kConstant:
482 case HloOpcode::kBitcast:
483 case HloOpcode::kBitcastConvert:
484 case HloOpcode::kCall:
485 case HloOpcode::kConditional:
486 case HloOpcode::kConvert:
487 case HloOpcode::kCustomCall:
488 case HloOpcode::kInfeed:
489 case HloOpcode::kOutfeed:
490 case HloOpcode::kParameter:
491 case HloOpcode::kRecv:
492 case HloOpcode::kRecvDone:
493 case HloOpcode::kSend:
494 case HloOpcode::kSendDone:
495 case HloOpcode::kWhile:
496 // The above opcodes should match the expected shapes exactly.
497 compatible = ShapeUtil::Compatible(instruction->shape(), inferred_shape);
498 break;
499 default:
500 if (allow_mixed_precision_) {
501 compatible = ShapeUtil::CompatibleIgnoringFpPrecision(
502 instruction->shape(), inferred_shape);
503 } else {
504 compatible =
505 ShapeUtil::Compatible(instruction->shape(), inferred_shape);
506 }
507 }
508 if (!compatible) {
509 return InvalidArgument(
510 "Expected instruction to have shape compatible with %s, actual "
511 "shape is %s:\n%s",
512 ShapeUtil::HumanString(inferred_shape).c_str(),
513 ShapeUtil::HumanString(instruction->shape()).c_str(),
514 instruction->ToString().c_str());
515 }
516 return tensorflow::Status::OK();
517}

570 tensorflow::gtl::ArraySlice<HloComputation*> computations) {

577// Verifies various invariants about the structure of the HLO:
578//
579// (1) each instruction has a non-null parent() set to the HloComputation which
580// contains it.
581//
582// (2) each computation has a non-null parent() set to the HloModule which
583// contains it.
584//
585// (3) the operands of each instruction are in the same computation as the
586// instruction.
587Status VerifyHloStructure(HloModule* module) {
588 for (const HloComputation* computation : module->computations()) {
589 if (computation->parent() == nullptr) {
590 return FailedPrecondition("Computation %s has a null parent pointer",
591 computation->name().c_str());
592 }
593 if (computation->parent() != module) {
594 return FailedPrecondition(
595 "Computation %s parent() does not point to parent module",
596 computation->name().c_str());
597 }
598
599 for (const HloInstruction* instruction : computation->instructions()) {
600 if (instruction->parent() == nullptr) {
601 return FailedPrecondition("Instruction %s has a null parent pointer",
602 instruction->name().c_str());
603 }
604 if (instruction->parent() != computation) {
605 return FailedPrecondition(
606 "Instruction %s parent() does not point to parent computation",
607 instruction->name().c_str());
608 }
609 }
610 }
611
612 // Check that operands are in the same computation separately from verifying
613 // parent() correctness so conditions like a null HloInstruction::parent() are
614 // identified and reported explicitly above rather than reporting a mismatched
615 // operand.
616 for (const HloComputation* computation : module->computations()) {
617 for (const HloInstruction* instruction : computation->instructions()) {
618 for (int i = 0; i < instruction->operand_count(); ++i) {
619 const HloInstruction* operand = instruction->operand(i);
620 if (operand->parent() != instruction->parent()) {
621 return FailedPrecondition(
622 "Operand %d (%s) of instruction %s is in a different "
623 "computation: %s vs %s",
624 i, operand->name().c_str(), instruction->name().c_str(),
625 operand->parent()->name().c_str(),
626 instruction->parent()->name().c_str());
627 }
628 }
629 }
630 }
631 return tensorflow::Status::OK();
632}

634Status HloVerifier::CheckFusionInstruction(HloInstruction* fusion) const {
635 // The parent fusion instruction of the fusion computation must be 'fusion'.
636 HloComputation* fused_computation = fusion->fused_instructions_computation();
637 if (fusion != fused_computation->FusionInstruction()) {
638 return FailedPrecondition(
639 "Instruction of fused computation does not match expected instruction "
640 "%s.",
641 fusion->ToString().c_str());
642 }
643
644 // Fused root instruction and fused parameters must all be owned by the fusion
645 // computation.
646 bool root_owned = false;
647 const std::vector<HloInstruction*>& fused_parameters =
648 fusion->fused_parameters();
649 const HloInstruction* fused_root = fusion->fused_expression_root();
650 std::vector<bool> parameter_owned(fused_parameters.size(), false);
651 for (auto* instruction : fused_computation->instructions()) {
652 if (fused_root == instruction) {
653 if (root_owned) {
654 return FailedPrecondition("Root appears more than once in %s.",
655 fusion->ToString().c_str());
656 }
657 root_owned = true;
658 }
659 for (int i = 0; i < fused_parameters.size(); ++i) {
660 if (fused_parameters[i] == instruction) {
661 if (parameter_owned[i]) {
662 return FailedPrecondition("Parameter appears more than once in %s.",
663 fusion->ToString().c_str());
664 }
665 parameter_owned[i] = true;
666 }
667 }
668 }
669 if (!root_owned) {
670 return FailedPrecondition("Root not found in computation of %s.",
671 fusion->ToString().c_str());
672 }
673 // Make sure all the parameter_owned entries are set
674 for (int i = 0; i < parameter_owned.size(); i++) {
675 if (!parameter_owned[i]) {
676 return FailedPrecondition("Parameter %d not found in computation of %s.",
677 i, fusion->ToString().c_str());
678 }
679 }
680
681 // Fused root must have no users.
682 if (fused_root->user_count() != 0) {
683 return FailedPrecondition("Root of %s may not have users.",
684 fusion->ToString().c_str());
685 }
686
687 // All uses of fused instructions must be in the fusion computation, and every
688 // non-root instruction must have at least one use.
689 for (auto* instruction :
690 fusion->fused_instructions_computation()->instructions()) {
691 if (instruction != fused_root) {
692 if (instruction->user_count() == 0) {
693 return FailedPrecondition(
694 "Non-root instruction %s in %s must have users.",
695 instruction->ToString().c_str(), fusion->ToString().c_str());
696 }
697 for (auto& user : instruction->users()) {
698 if (fused_computation != user->parent()) {
699 return FailedPrecondition(
700 "Non-root instruction %s in %s may not have external users.",
701 instruction->ToString().c_str(), fusion->ToString().c_str());
702 }
703 }
704 }
705 }
706
707 // Fused parameter instructions must be numbered contiguously and match up
708 // (shapes compatible) with their respective operand.
709 CHECK_EQ(fusion->operands().size(), fused_parameters.size());
710 std::vector<bool> parameter_numbers(fused_parameters.size(), false);
711 for (auto fused_param : fused_parameters) {
712 int64 param_no = fused_param->parameter_number();
713 if (param_no < 0) {
714 return FailedPrecondition(
715 "Unexpected negative parameter number %lld in %s.", param_no,
716 fusion->ToString().c_str());
717 }
718 if (param_no >= fused_parameters.size()) {
719 return FailedPrecondition(
720 "Unexpected parameter number %lld in %s: higher then number of "
721 "parameters %lu.",
722 param_no, fusion->ToString().c_str(), fused_parameters.size());
723 }
724 if (parameter_numbers[param_no]) {
725 return FailedPrecondition(
726 "Did not expect parameter number %lld more than once in %s.",
727 param_no, fusion->ToString().c_str());
728 }
729 parameter_numbers[param_no] = true;
730 if (!ShapeUtil::Compatible(fused_param->shape(),
731 fusion->operand(param_no)->shape())) {
732 return FailedPrecondition(
733 "Shape mismatch between parameter number %lld and its operand in %s.",
734 param_no, fusion->ToString().c_str());
735 }
736 }
737 // Make sure all the parameter_numbers entries were seen
738 for (int i = 0; i < parameter_numbers.size(); i++) {
739 if (!parameter_numbers[i]) {
740 return FailedPrecondition("Did not see parameter number %d in %s.", i,
741 fusion->ToString().c_str());
742 }
743 }
744
745 // TODO(b/65423525): We'd like to check that all operands are distinct.
746 // This is currently disabled due to the invariant being violated by
747 // multi-output fusion.
748 return tensorflow::Status::OK();
749}
750
751StatusOr<bool> HloVerifier::Run(HloModule* module) {
752 TF_RETURN_IF_ERROR(VerifyHloStructure(module));
753
754 tensorflow::gtl::FlatMap<string, const HloInstruction*> instructions;
755
756 for (auto* computation : module->computations()) {
757 for (const auto& instruction : computation->instructions()) {
758 TF_RET_CHECK(instruction->parent() == computation);
759 if (instruction->opcode() == HloOpcode::kFusion) {
760 TF_RETURN_IF_ERROR(CheckFusionInstruction(instruction));
761 TF_RET_CHECK(
762 ContainersEqual(instruction->called_computations(),
763 {instruction->fused_instructions_computation()}))
764 << "Fusion HLO calls computations other than the "
765 "fused_instructions_computation: "
766 << instruction->ToString()
767 << " instruction->fused_instructions_computation(): "
768 << instruction->fused_instructions_computation()->ToString()
769 << " instruction->called_computations(): "
770 << ComputationsToString(instruction->called_computations());
771
772 for (const auto& fused : instruction->fused_instructions()) {
773 TF_RET_CHECK(fused->parent() ==
774 instruction->fused_instructions_computation())
775 << "Fused HLO was missing a parent: " << fused->ToString()
776 << " parent: " << fused->parent()
777 << " computation: " << computation;
778 }
779 } else if (instruction->opcode() == HloOpcode::kBroadcast) {
780 // If you see this failure then someone has confused the difference
781 // between the HLO broadcast op, and the UserComputation broadcast
782 // op. See https://groups.google.com/forum/#!topic/xla-dev/9LqijHmTt_I
783 // or ComputationLowerer::Visit()
784 TF_RET_CHECK(instruction->dimensions().size() ==
785 ShapeUtil::Rank(instruction->operand(0)->shape()))
786 << "Broadcast HLO (" << instruction->ToShortString()
787 << ") has invalid number of dimensions: "
788 << instruction->dimensions().size()
789 << " != " << ShapeUtil::Rank(instruction->operand(0)->shape());
790 } else if (instruction->opcode() == HloOpcode::kWhile) {
791 auto* while_cond = instruction->while_condition();
792 auto* while_body = instruction->while_body();
793 TF_RET_CHECK(while_cond->num_parameters() == 1)
794 << "While condition must have exactly 1 parameter; had "
795 << while_cond->num_parameters() << ": " << while_cond->ToString();
796 TF_RET_CHECK(while_body->num_parameters() == 1)
797 << "While body must have exactly 1 parameter; had "
798 << while_body->num_parameters() << ": " << while_body->ToString();
799 TF_RET_CHECK(instruction->operand_count() == 1)
800 << "While loop must have exactly one operand; had "
801 << instruction->operand_count() << ": " << instruction->ToString();
802
803 auto* init = instruction->operand(0);
804 auto* cond_param = while_cond->parameter_instruction(0);
805 TF_RET_CHECK(ShapeUtil::Compatible(init->shape(), cond_param->shape()))
806 << "While condition's parameter must have the same shape as the "
807 "loop's 'init'. init: "
808 << init->ToString() << ", param: " << cond_param->ToString();
809 auto* cond_root = while_cond->root_instruction();
810 TF_RET_CHECK(ShapeUtil::Compatible(cond_root->shape(),
811 ShapeUtil::MakeShape(PRED, {})))
812 << "While condition should have shape PRED: "
813 << cond_root->ToString();
814
815 auto* body_param = while_body->parameter_instruction(0);
816 TF_RET_CHECK(ShapeUtil::Compatible(init->shape(), body_param->shape()))
817 << "While body's parameter must have the same shape as the loop's "
818 "'init'. init: "
819 << init->ToString() << ", param: " << body_param->ToString();
820 auto* body_root = while_body->root_instruction();
821 TF_RET_CHECK(ShapeUtil::Compatible(init->shape(), body_root->shape()))
822 << "While body should have same shape as the loop's 'init'. init: "
823 << init->ToString() << ", body: " << body_root->ToString();
824 }
825
826 auto previous = instructions.find(instruction->name());
827 TF_RET_CHECK(previous == instructions.end())
828 << "HLO has name that is not unique within module:\n"
829 << instruction->ToString()
830 << " in computation: " << computation->name()
831 << "\nPrevious HLO with same name:\n"
832 << previous->second->ToString()
833 << " in computation: " << previous->second->parent()->name();
834 instructions[instruction->name()] = instruction;
835 }
836
837 std::unique_ptr<ShapeVerifier> shape_verifier = shape_verifier_factory_();
838 TF_RETURN_IF_ERROR(computation->Accept(shape_verifier.get()));
839 }
840
841 return false;
842}
