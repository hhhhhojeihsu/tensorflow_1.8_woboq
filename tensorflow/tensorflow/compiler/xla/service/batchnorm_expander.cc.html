47// BatchNormExpanderVisitor traverses the HLO computation and rewrites BatchNorm
48// operations into smaller operations.
49class BatchNormExpanderVisitor : public DfsHloVisitorWithDefault {
50 public:
51 // Default visitor action is to do nothing and return OK.
52 Status DefaultAction(HloInstruction* /*hlo_instruction*/) override {
53 return Status::OK();
54 }
55
56 Status HandleBatchNormTraining(HloInstruction* batch_norm) override;
57
58 Status HandleBatchNormInference(HloInstruction* batch_norm) override;
59
60 Status HandleBatchNormGrad(HloInstruction* batch_norm) override;
61
62 // Runs the visitor on a computation.
63 static bool Run(HloComputation* computation, bool rewrite_training_op,
64 bool rewrite_inference_op, bool rewrite_grad_op,
65 bool use_fusion);
66
67 // Returns whether any batch norm ops were rewritten.
68 const bool changed() const { return changed_; }
69
70 ~BatchNormExpanderVisitor() override = default;
71
72 private:
73 explicit BatchNormExpanderVisitor(HloComputation* computation,
74 bool rewrite_training_op,
75 bool rewrite_inference_op,
76 bool rewrite_grad_op, bool use_fusion)
77 : computation_(computation),
78 rewrite_training_op_(rewrite_training_op),
79 rewrite_inference_op_(rewrite_inference_op),
80 rewrite_grad_op_(rewrite_grad_op),
81 use_fusion_(use_fusion) {}
82
83 HloComputation* GetScalarBinaryComputation(PrimitiveType primitive_type,
84 HloOpcode opcode) {
85 HloComputation::Builder b("scalar_computation");
86 auto scalar_lhs = b.AddInstruction(HloInstruction::CreateParameter(
87 0, ShapeUtil::MakeShape(primitive_type, {}), "scalar_lhs"));
88 auto scalar_rhs = b.AddInstruction(HloInstruction::CreateParameter(
89 1, ShapeUtil::MakeShape(primitive_type, {}), "scalar_rhs"));
90 auto scalar_op = b.AddInstruction(
91 HloInstruction::CreateBinary(ShapeUtil::MakeShape(primitive_type, {}),
92 opcode, scalar_lhs, scalar_rhs));
93 return computation_->parent()->AddEmbeddedComputation(b.Build(scalar_op));
94 }
95
96 // Current HloComputation instance the BatchNormExpander is
97 // traversing.
98 HloComputation* computation_;
99
100 bool rewrite_training_op_;
101 bool rewrite_inference_op_;
102 bool rewrite_grad_op_;
103 bool use_fusion_;
104
105 // Whether rewrite has occurred.
106 bool changed_ = false;
107
108 // Replaces the existing HLO instruction old_instruction, with
109 // new_instruction, and marks the optimizer status as changed.
110 // Returns the Status representing the result of the replace operation.
111 Status ReplaceWithNewInstruction(
112 HloInstruction* old_instruction,
113 std::unique_ptr<HloInstruction> new_instruction) {
114 TF_RETURN_IF_ERROR(computation_->ReplaceWithNewInstruction(
115 old_instruction, std::move(new_instruction)));
116 changed_ = true;
117 return Status::OK();
118 }
119
120 // Replaces the existing HLO instruction old_instruction, with
121 // new_instruction, and marks the optimizer status as changed.
122 // Returns the Status representing the result of the replace operation.
123 Status ReplaceInstruction(HloInstruction* old_instruction,
124 HloInstruction* new_instruction) {
125 TF_RETURN_IF_ERROR(
126 computation_->ReplaceInstruction(old_instruction, new_instruction));
127 changed_ = true;
128 return Status::OK();
129 }
130};
131
132bool BatchNormExpanderVisitor::Run(HloComputation* computation,
133 bool rewrite_training_op,
134 bool rewrite_inference_op,
135 bool rewrite_grad_op, bool use_fusion) {
136 BatchNormExpanderVisitor visitor(
137 computation,
138 /*rewrite_training_op=*/rewrite_training_op,
139 /*rewrite_inference_op=*/rewrite_inference_op,
140 /*rewrite_grad_op=*/rewrite_grad_op,
141 /*use_fusion=*/use_fusion);
142 TF_CHECK_OK(computation->Accept(&visitor));
143 return visitor.changed_;
144}

146Status BatchNormExpanderVisitor::HandleBatchNormTraining(
147 HloInstruction* batch_norm) {
148 if (!rewrite_training_op_) {
149 return Status::OK();
150 }
151
152 std::vector<HloInstruction*> added_instructions;
153 auto add = [&](std::unique_ptr<HloInstruction> inst) {
154 HloInstruction* added_inst = computation_->AddInstruction(std::move(inst));
155 added_inst->set_metadata(batch_norm->metadata());
156 added_instructions.push_back(added_inst);
157 return added_inst;
158 };
159 int64 instruction_count_before = computation_->instruction_count();
160
161 // Expand batch norm training into smaller HLO ops.
162 HloInstruction* operand = batch_norm->mutable_operand(0);
163 const Shape operand_shape = operand->shape();
164 PrimitiveType ptype = operand_shape.element_type();
165 int64 feature_index = batch_norm->feature_index();
166 const int64 feature_count = operand_shape.dimensions(feature_index);
167 const int64 size_in_elements = ShapeUtil::ElementsIn(operand_shape);
168 auto elements_per_feature_literal =
169 Literal::CreateR0<float>(size_in_elements / feature_count);
170 TF_ASSIGN_OR_RETURN(elements_per_feature_literal,
171 elements_per_feature_literal->Convert(ptype));
172 auto elements_per_feature = add(
173 HloInstruction::CreateConstant(std::move(elements_per_feature_literal)));
174
175 HloInstruction* scale = batch_norm->mutable_operand(1);
176 HloInstruction* offset = batch_norm->mutable_operand(2);
177 const Shape feature_shape = scale->shape();
178
179 auto zero_literal = Literal::CreateR0(0.0f);
180 TF_ASSIGN_OR_RETURN(zero_literal, zero_literal->Convert(ptype));
181 auto zero = add(HloInstruction::CreateConstant(std::move(zero_literal)));
182
183 auto epsilon_literal = Literal::CreateR0(batch_norm->epsilon());
184 TF_ASSIGN_OR_RETURN(epsilon_literal, epsilon_literal->Convert(ptype));
185 auto epsilon =
186 add(HloInstruction::CreateConstant(std::move(epsilon_literal)));
187 std::vector<int64> dimensions_without_feature;
188
189 for (int64 i = 0; i < ShapeUtil::Rank(operand_shape); ++i) {
190 if (i != feature_index) {
191 dimensions_without_feature.push_back(i);
192 }
193 }
194
195 auto scale_broadcasted = add(
196 HloInstruction::CreateBroadcast(operand_shape, scale, {feature_index}));
197
198 auto offset_broadcasted = add(
199 HloInstruction::CreateBroadcast(operand_shape, offset, {feature_index}));
200
201 HloComputation* add_reduce_computation =
202 GetScalarBinaryComputation(ptype, HloOpcode::kAdd);
203
204 // X^2.
205 auto operand_squared = add(HloInstruction::CreateBinary(
206 operand_shape, HloOpcode::kMultiply, operand, operand));
207 // Sum[X].
208 auto sum = add(HloInstruction::CreateReduce(feature_shape, operand, zero,
209 dimensions_without_feature,
210 add_reduce_computation));
211
212 // Sum[X^2].
213 auto squared_sum = add(HloInstruction::CreateReduce(
214 feature_shape, operand_squared, zero, dimensions_without_feature,
215 add_reduce_computation));
216
217 // Fuse two parallel reduces together to improve performance.
218 if (use_fusion_ && !batch_norm->has_sharding()) {
219 auto tuple = add(HloInstruction::CreateTuple({sum, squared_sum}));
220
221 auto fused = computation_->CreateFusionInstruction(
222 {tuple, sum, squared_sum, operand_squared},
223 HloInstruction::FusionKind::kInput);
224
225 sum = add(HloInstruction::CreateGetTupleElement(feature_shape, fused, 0));
226
227 squared_sum =
228 add(HloInstruction::CreateGetTupleElement(feature_shape, fused, 1));
229 }
230
231 // E[X].
232 auto mean = add(HloInstruction::CreateBinary(
233 feature_shape, HloOpcode::kDivide, sum, elements_per_feature));
234
235 auto mean_broadcasted = add(
236 HloInstruction::CreateBroadcast(operand_shape, mean, {feature_index}));
237
238 // E[X^2].
239 auto square_mean = add(HloInstruction::CreateBinary(
240 feature_shape, HloOpcode::kDivide, squared_sum, elements_per_feature));
241
242 // E^2[X].
243 auto mean_square = add(HloInstruction::CreateBinary(
244 feature_shape, HloOpcode::kMultiply, mean, mean));
245
246 // Var[X].
247 auto var = add(HloInstruction::CreateBinary(
248 feature_shape, HloOpcode::kSubtract, square_mean, mean_square));
249
250 auto var_broadcasted =
251 add(HloInstruction::CreateBroadcast(operand_shape, var, {feature_index}));
252
253 // Var[X] + epsilon.
254 auto var_add_epsilon = add(HloInstruction::CreateBinary(
255 operand_shape, HloOpcode::kAdd, var_broadcasted, epsilon));
256
257 auto neg_half_literal = Literal::CreateR0(-0.5f);
258 TF_ASSIGN_OR_RETURN(neg_half_literal, neg_half_literal->Convert(ptype));
259 auto neg_half =
260 add(HloInstruction::CreateConstant(std::move(neg_half_literal)));
261
262 // 1 / Sqrt[Var[X] + epsilon].
263 auto rsqrt_var_add_epsilon = add(HloInstruction::CreateBinary(
264 operand_shape, HloOpcode::kPower, var_add_epsilon, neg_half));
265
266 // X - E[X].
267 auto operand_minus_mean = add(HloInstruction::CreateBinary(
268 operand_shape, HloOpcode::kSubtract, operand, mean_broadcasted));
269
270 // (X - E[X]) / Sqrt[Var[X] + epsilon].
271 auto normalized = add(
272 HloInstruction::CreateBinary(operand_shape, HloOpcode::kMultiply,
273 operand_minus_mean, rsqrt_var_add_epsilon));
274
275 // (X - E[X]) / Sqrt[Var[X] + epsilon] * scale.
276 auto scaled_normalized = add(HloInstruction::CreateBinary(
277 operand_shape, HloOpcode::kMultiply, normalized, scale_broadcasted));
278
279 // (X - E[X]) / Sqrt[Var[X] + epsilon] * scale + offset.
280 auto shifted_normalized = add(HloInstruction::CreateBinary(
281 operand_shape, HloOpcode::kAdd, scaled_normalized, offset_broadcasted));
282
283 auto tuple = HloInstruction::CreateTuple({shifted_normalized, mean, var});
284
285 if (batch_norm->has_sharding()) {
286 int64 instruction_count_after = computation_->instruction_count();
287 CHECK_EQ(instruction_count_after,
288 instruction_count_before + added_instructions.size());
289 HloSharding operand_sharding =
290 batch_norm->sharding().GetAsShapeTree(batch_norm->shape()).element({0});
291 for (HloInstruction* inst : added_instructions) {
292 if (ShapeUtil::Equal(inst->shape(), operand_shape)) {
293 inst->set_sharding(operand_sharding);
294 } else {
295 inst->set_sharding(HloSharding::Replicate());
296 }
297 }
298 tuple->set_sharding(batch_norm->sharding());
299 }
300 TF_CHECK_OK(ReplaceWithNewInstruction(batch_norm, std::move(tuple)));
301 return Status::OK();
302}

304Status BatchNormExpanderVisitor::HandleBatchNormInference(
305 HloInstruction* batch_norm) {
306 if (!rewrite_inference_op_) {
307 return Status::OK();
308 }
309 // Expand batch norm inference into smaller HLO ops.
310 HloInstruction* operand = batch_norm->mutable_operand(0);
311 const Shape operand_shape = operand->shape();
312 int64 feature_index = batch_norm->feature_index();
313 PrimitiveType ptype = operand_shape.element_type();
314
315 HloInstruction* scale = batch_norm->mutable_operand(1);
316 HloInstruction* offset = batch_norm->mutable_operand(2);
317 HloInstruction* mean = batch_norm->mutable_operand(3);
318 HloInstruction* var = batch_norm->mutable_operand(4);
319 const Shape feature_shape = scale->shape();
320
321 auto epsilon_literal = Literal::CreateR0(batch_norm->epsilon());
322 TF_ASSIGN_OR_RETURN(epsilon_literal, epsilon_literal->Convert(ptype));
323 auto epsilon = computation_->AddInstruction(
324 HloInstruction::CreateConstant(std::move(epsilon_literal)));
325
326 std::vector<int64> dimensions_without_feature;
327
328 for (int64 i = 0; i < ShapeUtil::Rank(operand_shape); ++i) {
329 if (i != feature_index) {
330 dimensions_without_feature.push_back(i);
331 }
332 }
333
334 std::vector<HloInstruction*> added_instructions;
335 auto add = [&](std::unique_ptr<HloInstruction> inst) {
336 HloInstruction* added_inst = computation_->AddInstruction(std::move(inst));
337 added_inst->set_metadata(batch_norm->metadata());
338 added_instructions.push_back(added_inst);
339 return added_inst;
340 };
341 int64 instruction_count_before = computation_->instruction_count();
342
343 auto scale_broadcasted = add(
344 HloInstruction::CreateBroadcast(operand_shape, scale, {feature_index}));
345
346 auto offset_broadcasted = add(
347 HloInstruction::CreateBroadcast(operand_shape, offset, {feature_index}));
348
349 auto mean_broadcasted = add(
350 HloInstruction::CreateBroadcast(operand_shape, mean, {feature_index}));
351
352 auto var_broadcasted =
353 add(HloInstruction::CreateBroadcast(operand_shape, var, {feature_index}));
354
355 // Var[X] + epsilon.
356 auto var_add_epsilon = add(HloInstruction::CreateBinary(
357 operand_shape, HloOpcode::kAdd, var_broadcasted, epsilon));
358
359 auto neg_half_literal = Literal::CreateR0(-0.5f);
360 TF_ASSIGN_OR_RETURN(neg_half_literal, neg_half_literal->Convert(ptype));
361 auto neg_half =
362 add(HloInstruction::CreateConstant(std::move(neg_half_literal)));
363
364 // 1 / Sqrt[Var[X] + epsilon].
365 auto rsqrt_var_add_epsilon = add(HloInstruction::CreateBinary(
366 operand_shape, HloOpcode::kPower, var_add_epsilon, neg_half));
367
368 // X - E[X].
369 auto operand_minus_mean = add(HloInstruction::CreateBinary(
370 operand_shape, HloOpcode::kSubtract, operand, mean_broadcasted));
371
372 // (X - E[X]) / Sqrt[Var[X] + epsilon].
373 auto normalized = add(
374 HloInstruction::CreateBinary(operand_shape, HloOpcode::kMultiply,
375 operand_minus_mean, rsqrt_var_add_epsilon));
376
377 // (X - E[X]) / Sqrt[Var[X] + epsilon] * scale.
378 auto scaled_normalized = add(HloInstruction::CreateBinary(
379 operand_shape, HloOpcode::kMultiply, normalized, scale_broadcasted));
380
381 // (X - E[X]) / Sqrt[Var[X] + epsilon] * scale + offset.
382 auto shifted_normalized = HloInstruction::CreateBinary(
383 operand_shape, HloOpcode::kAdd, scaled_normalized, offset_broadcasted);
384
385 int64 instruction_count_after = computation_->instruction_count();
386 CHECK_EQ(instruction_count_after,
387 instruction_count_before + added_instructions.size());
388 if (batch_norm->has_sharding()) {
389 for (HloInstruction* inst : added_instructions) {
390 if (ShapeUtil::Equal(inst->shape(), operand_shape)) {
391 inst->set_sharding(batch_norm->sharding());
392 } else {
393 inst->set_sharding(HloSharding::Replicate());
394 }
395 }
396 shifted_normalized->set_sharding(batch_norm->sharding());
397 }
398 TF_CHECK_OK(
399 ReplaceWithNewInstruction(batch_norm, std::move(shifted_normalized)));
400 return Status::OK();
401}

403Status BatchNormExpanderVisitor::HandleBatchNormGrad(
404 HloInstruction* batch_norm) {
405 // Use the following formulas to calculate gradients:
406 // scale_grad =
407 // sum(output_grad * (activation - mean(activation))) * rsqrt(var + epsilon)
408 //
409 // offset_grad =
410 // sum(output_grad)
411 //
412 // activation_grad =
413 // 1/N * scale * rsqrt(var + epsilon) *
414 // (N * output_grad - sum(output_grad) - (activation - mean(activation)) *
415 // sum(output_grad * (activation - mean(activation))) / (variance +
416 // epsilon))
417 if (!rewrite_grad_op_) {
418 return Status::OK();
419 }
420 std::vector<HloInstruction*> added_instructions;
421 auto add = [&](std::unique_ptr<HloInstruction> inst) {
422 HloInstruction* added_inst = computation_->AddInstruction(std::move(inst));
423 added_inst->set_metadata(batch_norm->metadata());
424 added_instructions.push_back(added_inst);
425 return added_inst;
426 };
427 int64 instruction_count_before = computation_->instruction_count();
428
429 HloInstruction* activation = batch_norm->mutable_operand(0);
430 const Shape activation_shape = activation->shape();
431 PrimitiveType ptype = activation_shape.element_type();
432 HloInstruction* scale = batch_norm->mutable_operand(1);
433 const Shape feature_shape = scale->shape();
434 HloInstruction* mean = batch_norm->mutable_operand(2);
435 HloInstruction* variance = batch_norm->mutable_operand(3);
436 HloInstruction* grad_output = batch_norm->mutable_operand(4);
437
438 int64 feature_index = batch_norm->feature_index();
439
440 const int64 size_in_elements = ShapeUtil::ElementsIn(activation_shape);
441 const int64 feature_count = activation_shape.dimensions(feature_index);
442 auto elements_per_feature_literal =
443 Literal::CreateR0<float>(size_in_elements / feature_count);
444 TF_ASSIGN_OR_RETURN(elements_per_feature_literal,
445 elements_per_feature_literal->Convert(ptype));
446 auto elements_per_feature = add(
447 HloInstruction::CreateConstant(std::move(elements_per_feature_literal)));
448
449 auto zero_literal = Literal::CreateR0(0.0f);
450 TF_ASSIGN_OR_RETURN(zero_literal, zero_literal->Convert(ptype));
451 auto zero = add(HloInstruction::CreateConstant(std::move(zero_literal)));
452
453 auto neg_half_literal = Literal::CreateR0(-0.5f);
454 TF_ASSIGN_OR_RETURN(neg_half_literal, neg_half_literal->Convert(ptype));
455 auto neg_half =
456 add(HloInstruction::CreateConstant(std::move(neg_half_literal)));
457
458 auto epsilon_literal = Literal::CreateR0(batch_norm->epsilon());
459 TF_ASSIGN_OR_RETURN(epsilon_literal, epsilon_literal->Convert(ptype));
460 auto epsilon =
461 add(HloInstruction::CreateConstant(std::move(epsilon_literal)));
462
463 std::vector<int64> dimensions_without_feature;
464
465 for (int64 i = 0; i < ShapeUtil::Rank(activation_shape); ++i) {
466 if (i != feature_index) {
467 dimensions_without_feature.push_back(i);
468 }
469 }
470
471 auto scale_broadcasted = add(HloInstruction::CreateBroadcast(
472 activation_shape, scale, {feature_index}));
473 auto variance_broadcasted = add(HloInstruction::CreateBroadcast(
474 activation_shape, variance, {feature_index}));
475
476 // E[X].
477 auto mean_broadcasted = add(
478 HloInstruction::CreateBroadcast(activation_shape, mean, {feature_index}));
479
480 // rsqrt[Var[X] + epsilon].
481 auto rsqrt_var_add_epsilon_broadcasted = add(HloInstruction::CreateBinary(
482 activation_shape, HloOpcode::kPower,
483 add(HloInstruction::CreateBinary(activation_shape, HloOpcode::kAdd,
484 variance_broadcasted, epsilon)),
485 neg_half));
486
487 auto rsqrt_var_add_epsilon = add(HloInstruction::CreateBinary(
488 feature_shape, HloOpcode::kPower,
489 add(HloInstruction::CreateBinary(feature_shape, HloOpcode::kAdd, variance,
490 epsilon)),
491 neg_half));
492
493 // X - E[X].
494 auto activation_minus_mean = add(HloInstruction::CreateBinary(
495 activation_shape, HloOpcode::kSubtract, activation, mean_broadcasted));
496
497 // Grad[Y] * (X - E[X]).
498 auto grad_output_times_activiation_minus_mean =
499 add(HloInstruction::CreateBinary(activation_shape, HloOpcode::kMultiply,
500 grad_output, activation_minus_mean));
501
502 HloComputation* add_reduce_computation =
503 GetScalarBinaryComputation(ptype, HloOpcode::kAdd);
504
505 // sum(Grad[Y] * (X - E[X])).
506 auto sum_grad_output_times_activiation_minus_mean =
507 add(HloInstruction::CreateReduce(
508 feature_shape, grad_output_times_activiation_minus_mean, zero,
509 dimensions_without_feature, add_reduce_computation));
510
511 // Grad[beta] = Sum(Grad[Y]).
512 auto grad_beta = add(HloInstruction::CreateReduce(
513 feature_shape, grad_output, zero, dimensions_without_feature,
514 add_reduce_computation));
515
516 if (use_fusion_ && !batch_norm->has_sharding()) {
517 auto tuple = add(HloInstruction::CreateTuple(
518 {sum_grad_output_times_activiation_minus_mean, grad_beta}));
519
520 auto fused = computation_->CreateFusionInstruction(
521 {tuple, sum_grad_output_times_activiation_minus_mean, grad_beta},
522 HloInstruction::FusionKind::kInput);
523
524 sum_grad_output_times_activiation_minus_mean =
525 add(HloInstruction::CreateGetTupleElement(feature_shape, fused, 0));
526
527 grad_beta =
528 add(HloInstruction::CreateGetTupleElement(feature_shape, fused, 1));
529 }
530
531 // Grad[scale] = Sum(Grad[Y] * (X - E[X]) * rsqrt[Var[X] + epsilon]).
532 auto grad_scale = add(HloInstruction::CreateBinary(
533 feature_shape, HloOpcode::kMultiply,
534 sum_grad_output_times_activiation_minus_mean, rsqrt_var_add_epsilon));
535
536 // I2 = Sum(Grad[Y])
537 auto i2 = add(HloInstruction::CreateBroadcast(activation_shape, grad_beta,
538 {feature_index}));
539
540 // I3 = Sum(Grad[Y] * (X - E[X]))
541 auto i3 = add(HloInstruction::CreateBroadcast(
542 activation_shape, sum_grad_output_times_activiation_minus_mean,
543 {feature_index}));
544
545 // I4 = (X - E[X]) * I3
546 auto i4 = add(HloInstruction::CreateBinary(
547 activation_shape, HloOpcode::kMultiply, i3, activation_minus_mean));
548
549 // I5 = I4 / (Var[X] + epsilon)
550 auto i5 = add(HloInstruction::CreateBinary(
551 activation_shape, HloOpcode::kDivide, i4,
552 add(HloInstruction::CreateBinary(activation_shape, HloOpcode::kAdd,
553 variance_broadcasted, epsilon))));
554
555 // scale * rsqrt[Var[X] + epsilon] * 1/N
556 auto scale_times_rsqrt_var_add_epsilon = add(HloInstruction::CreateBinary(
557 activation_shape, HloOpcode::kMultiply, scale_broadcasted,
558 rsqrt_var_add_epsilon_broadcasted));
559
560 scale_times_rsqrt_var_add_epsilon = add(HloInstruction::CreateBinary(
561 activation_shape, HloOpcode::kDivide, scale_times_rsqrt_var_add_epsilon,
562 elements_per_feature));
563
564 auto i1 =
565 add(HloInstruction::CreateBinary(activation_shape, HloOpcode::kMultiply,
566 grad_output, elements_per_feature));
567
568 // I6 = I1 - I2 - I5
569 auto i6 = add(HloInstruction::CreateBinary(
570 activation_shape, HloOpcode::kSubtract,
571 add(HloInstruction::CreateBinary(activation_shape, HloOpcode::kSubtract,
572 i1, i2)),
573 i5));
574
575 // Grad[X] = scale * rsqrt[Var[X] + epsilon] * 1/N * I6.
576 auto grad_activation =
577 add(HloInstruction::CreateBinary(activation_shape, HloOpcode::kMultiply,
578 scale_times_rsqrt_var_add_epsilon, i6));
579 auto tuple =
580 HloInstruction::CreateTuple({grad_activation, grad_scale, grad_beta});
581 if (batch_norm->has_sharding()) {
582 int64 instruction_count_after = computation_->instruction_count();
583 CHECK_EQ(instruction_count_after,
584 instruction_count_before + added_instructions.size());
585 HloSharding activation_sharding =
586 batch_norm->sharding().GetAsShapeTree(batch_norm->shape()).element({0});
587 for (HloInstruction* inst : added_instructions) {
588 if (ShapeUtil::Equal(inst->shape(), activation_shape)) {
589 inst->set_sharding(activation_sharding);
590 } else {
591 inst->set_sharding(HloSharding::Replicate());
592 }
593 }
594 tuple->set_sharding(batch_norm->sharding());
595 }
596
597 TF_CHECK_OK(ReplaceWithNewInstruction(batch_norm, std::move(tuple)));
598
599 return Status::OK();
600}

602StatusOr<bool> BatchNormExpander::Run(HloModule* module) {
603 XLA_VLOG_LINES(2, "BatchNormExpander::Run(), before:\n" + module->ToString());
604 bool changed = false;
605 for (auto* comp : module->MakeNonfusionComputations()) {
606 if (BatchNormExpanderVisitor::Run(comp, rewrite_training_op_,
607 rewrite_inference_op_, rewrite_grad_op_,
608 use_fusion_)) {
609 changed = true;
610 }
611 }
612 XLA_VLOG_LINES(2, "BatchNormExpander::Run(), after:\n" + module->ToString());
613 return changed;
614}
