35bool HloOrdering::ExecutesBefore(const HloInstruction* a,
36 const HloInstruction* b) const {
37 // 'a' and 'b' may be in different computations. In this case, find the
38 // callgraph ancestor instructions which call (potentially transitively) the
39 // computations containing 'a' and 'b' and use these ancestor instructions to
40 // compare order.
41 const HloInstruction* a_ancestor;
42 const HloInstruction* b_ancestor;
43 std::tie(a_ancestor, b_ancestor) =
44 call_graph_->NearestAncestorsInSameComputation(
45 const_cast<HloInstruction*>(a), const_cast<HloInstruction*>(b));
46
47 if (a_ancestor == nullptr) {
48 // Ancestors in a common computation could not be found so consider the
49 // instructions 'a' and 'b' to be unordered.
50 return false;
51 }
52 // a_ancestor and b_ancestor must be either both null or both non-null.
53 CHECK_NE(b_ancestor, nullptr);
54 CHECK_EQ(a_ancestor->parent(), b_ancestor->parent());
55
56 // If the common ancestor is a while instruction there is an additional
57 // ordering criteria which may apply. The condition computation is considered
58 // to execute before the body computation so if 'a' is in the condition and
59 // 'b' is in the body, then 'a' executes before 'b'.
60 if (a_ancestor == b_ancestor && a_ancestor->opcode() == HloOpcode::kWhile) {
61 const HloComputation* body = a_ancestor->while_body();
62 const HloComputation* condition = a_ancestor->while_condition();
63 if (call_graph_->InstructionIsNestedIn(a, condition) &&
64 call_graph_->InstructionIsNestedIn(b, body)) {
65 return true;
66 }
67 }
68
69 // If the common ancestor is a conditional instruction, even though the true
70 // and false computations are not really ordered per-se, we define the true
71 // computation to be ordered before the false one.
72 // This ensures that buffers can still be shared among the two computations
73 // as they will forcibly have disjoint liveness.
74 if (a_ancestor == b_ancestor &&
75 a_ancestor->opcode() == HloOpcode::kConditional) {
76 const HloComputation* true_computation = a_ancestor->true_computation();
77 const HloComputation* false_computation = a_ancestor->false_computation();
78 if (call_graph_->InstructionIsNestedIn(a, true_computation) &&
79 call_graph_->InstructionIsNestedIn(b, false_computation)) {
80 return true;
81 }
82 // If 'b' is the conditional ancestor, and 'a' is within the true or false
83 // computations, 'a' executes before 'b'.
84 if (b == a_ancestor &&
85 (call_graph_->InstructionIsNestedIn(a, true_computation) ||
86 call_graph_->InstructionIsNestedIn(a, false_computation))) {
87 return true;
88 }
89 }
90
91 return ExecutesBeforeInSameComputation(a_ancestor, b_ancestor);
92}

94bool HloOrdering::IsDefinedBefore(const HloValue& a, const HloValue& b) const {
95 // If 'b' is an entry param then 'a' cannot be defined before 'b' because 'b'
96 // is live into the module.
97 const HloModule* module = b.defining_instruction()->parent()->parent();
98 if (b.defining_instruction()->parent() == module->entry_computation() &&
99 b.defining_instruction()->opcode() == HloOpcode::kParameter) {
100 return false;
101 }
102
103 // Phi values require special handling. Because XLA does not have a phi
104 // instruction, the definition instruction of the phis values are
105 // placeholders: either the subcomputation parameter (body or condition) or
106 // the while instruction. However, the program point where these values are
107 // logically defined does not necessarily coincide exactly with program point
108 // of these place-holder instructions. So we explicitly define the following
109 // order for phi values:
110 //
111 // body/condition parameter phi:
112 // Defined before all values defined in its computation excepting other
113 // phis.
114 //
115 // while phi:
116 // defined after all values defined in the condition or body.
117 //
118 auto is_body_or_condition_phi = [](const HloValue& v) {
119 return v.is_phi() &&
120 v.defining_instruction()->opcode() == HloOpcode::kParameter;
121 };
122 if (is_body_or_condition_phi(a) && !is_body_or_condition_phi(b) &&
123 call_graph_->InstructionIsNestedIn(b.defining_instruction(),
124 a.defining_instruction()->parent())) {
125 return true;
126 }
127 if (is_body_or_condition_phi(b) &&
128 call_graph_->InstructionIsNestedIn(a.defining_instruction(),
129 b.defining_instruction()->parent())) {
130 return false;
131 }
132
133 // If 'b' is a while phi and 'a' is in the body or condition, then 'a'
134 // executes before 'b'.
135 if (b.is_phi() && b.defining_instruction()->opcode() == HloOpcode::kWhile &&
136 (call_graph_->InstructionIsNestedIn(
137 a.defining_instruction(), b.defining_instruction()->while_body()) ||
138 call_graph_->InstructionIsNestedIn(
139 a.defining_instruction(),
140 b.defining_instruction()->while_condition()))) {
141 return true;
142 }
143 // If 'b' is a conditional phi and 'a' is in the true or false computation,
144 // then 'a' executes before 'b'.
145 if (b.is_phi() &&
146 b.defining_instruction()->opcode() == HloOpcode::kConditional &&
147 (call_graph_->InstructionIsNestedIn(
148 a.defining_instruction(),
149 b.defining_instruction()->true_computation()) ||
150 call_graph_->InstructionIsNestedIn(
151 a.defining_instruction(),
152 b.defining_instruction()->false_computation()))) {
153 return true;
154 }
155 return ExecutesBefore(a.defining_instruction(), b.defining_instruction());
156}

158/* static */
159bool HloOrdering::UseIsBeforeValueDefinition(
160 const HloUse& use, const HloValue& value,
161 const HloDataflowAnalysis& dataflow) const {
162 VLOG(4) << "UseIsBeforeValueDefinition(use=" << use
163 << ", value=" << value.ToShortString() << ")";
164 if (ExecutesBefore(use.instruction, value.defining_instruction())) {
165 VLOG(4) << " use instruction executes before value-defining instruction";
166 return true;
167 }
168
169 // If the use is at the instruction where the value is defined, then the use
170 // is before the def if the instruction allows buffer sharing (in place
171 // computation).
172 if (use.instruction == value.defining_instruction() &&
173 CanShareOperandBufferWithUser(
174 use.instruction->mutable_operand(use.operand_number),
175 use.operand_index, value.defining_instruction(),
176 value.defining_index(), dataflow)) {
177 VLOG(4) << " use is value def, and instruction can share use buffer";
178 return true;
179 }
180
181 // The use at a while is an input to a phi, and logically occurs before values
182 // are defined in the body or condition computations.
183 if (use.instruction->opcode() == HloOpcode::kWhile) {
184 const HloInstruction* xla_while = use.instruction;
185 if (call_graph_->InstructionIsNestedIn(value.defining_instruction(),
186 xla_while->while_body()) ||
187 call_graph_->InstructionIsNestedIn(value.defining_instruction(),
188 xla_while->while_condition())) {
189 VLOG(4) << " use is while " << use.instruction->name()
190 << " and def is in condition or body";
191 return true;
192 }
193 }
194
195 // Similarly if the value is defined at a while, it logically occurs after any
196 // uses in the body or condition computations.
197 if (value.defining_instruction()->opcode() == HloOpcode::kWhile) {
198 CHECK(value.is_phi());
199 const HloInstruction* xla_while = value.defining_instruction();
200 if (call_graph_->InstructionIsNestedIn(use.instruction,
201 xla_while->while_body()) ||
202 call_graph_->InstructionIsNestedIn(use.instruction,
203 xla_while->while_condition())) {
204 VLOG(4) << " value is while " << value.defining_instruction()->name()
205 << " and use is in condition or body";
206 return true;
207 }
208 }
209
210 // The use at a call occurs before values that are defined in the called
211 // computation.
212 if (use.instruction->opcode() == HloOpcode::kCall) {
213 const HloInstruction* call = use.instruction;
214 if (call_graph_->InstructionIsNestedIn(value.defining_instruction(),
215 call->to_apply())) {
216 VLOG(4) << " use is call " << use.instruction->name()
217 << " and def is in called computation";
218 return true;
219 }
220 }
221
222 if (use.instruction->opcode() == HloOpcode::kConditional) {
223 const HloInstruction* conditional = use.instruction;
224 if (call_graph_->InstructionIsNestedIn(value.defining_instruction(),
225 conditional->true_computation())) {
226 VLOG(4) << " use is conditional " << use.instruction->name()
227 << " and def is in TRUE computation";
228 return true;
229 }
230 if (call_graph_->InstructionIsNestedIn(value.defining_instruction(),
231 conditional->false_computation())) {
232 VLOG(4) << " use is conditional " << use.instruction->name()
233 << " and def is in FALSE computation";
234 return true;
235 }
236 }
237
238 VLOG(4) << " use is not before value";
239 return false;
240}

242bool HloOrdering::LiveRangeStrictlyBefore(
243 const HloValue& a, const HloValue& b,
244 const HloDataflowAnalysis& dataflow) const {
245 VLOG(4) << "LiveRangeStrictlyBefore(a = " << a.ToShortString()
246 << ", b = " << b.ToShortString() << ")";
247 if (!IsDefinedBefore(a, b)) {
248 VLOG(4) << a << " not defined before " << b;
249 return false;
250 }
251 // All uses of 'a' must be before 'b' is defined.
252 for (const HloUse& use : a.uses()) {
253 if (!UseIsBeforeValueDefinition(use, b, dataflow)) {
254 VLOG(4) << "use of " << a << " (" << use << ") not before " << b
255 << " is defined";
256 return false;
257 }
258 }
259 return true;
260}
261
262bool HloOrdering::MayInterfere(const HloValue& a, const HloValue& b,
263 const HloDataflowAnalysis& dataflow) const {
264 // Buffers without disjoint liveness may interfere.
265 return !LiveRangeStrictlyBefore(a, b, dataflow) &&
266 !LiveRangeStrictlyBefore(b, a, dataflow);
267}

269HloOrderingProto HloOrdering::ToProto() const {
270 HloOrderingProto proto;
271 for (const auto& computation : module_->computations()) {
272 const std::vector<const HloInstruction*>* sequence =
273 SequentialOrder(*computation);
274 if (sequence != nullptr) {
275 HloOrderingProto::SequentialComputation* proto_computation =
276 proto.add_sequential_computations();
277 proto_computation->set_computation_name(computation->name());
278 for (const HloInstruction* instruction : *sequence) {
279 *proto_computation->add_instruction_names() = instruction->name();
280 }
281 }
282 }
283 return proto;
284}

289bool PredecessorHloOrdering::ExecutesBeforeInSameComputation(
290 const HloInstruction* a, const HloInstruction* b) const {
291 CHECK_EQ(a->parent(), b->parent());
292
293 // 'a' executes before 'b' if 'a' is in the strict predecessor set of 'b'.
294 return a != b && predecessors_.at(a->parent())->IsReachable(a, b);
295}

297string PredecessorHloOrdering::ToStringHelper(const string& name) const {
298 std::vector<string> pieces;
299 pieces.push_back(name);
300 for (auto* computation : module_->MakeNonfusionComputations()) {
301 pieces.push_back(tensorflow::strings::Printf("computation %s:",
302 computation->name().c_str()));
303 const auto all = computation->MakeInstructionPostOrder();
304 for (auto instruction : all) {
305 pieces.push_back(tensorflow::strings::Printf(
306 " %s predecessors:", instruction->name().c_str()));
307 for (auto predecessor : all) {
308 if (predecessors_.at(computation)
309 ->IsReachable(predecessor, instruction)) {
310 pieces.push_back(
311 tensorflow::strings::Printf(" %s", predecessor->name().c_str()));
312 }
313 }
314 }
315 }
316 return tensorflow::str_util::Join(pieces, "\n");
317}

319DependencyHloOrdering::DependencyHloOrdering(const HloModule* module)
320 : PredecessorHloOrdering(module) {
321 // Compute predecessor relationships between all instructions to determine
322 // ordering based on dependencies. ExecutesBefore will return true iff there
323 // exists a path in the HLO computation graph from 'a' to 'b'.
324 for (auto* computation : module->MakeNonfusionComputations()) {
325 predecessors_.emplace(computation, computation->ComputeReachability());
326 }
327}
328
329string DependencyHloOrdering::ToString() const {
330 return ToStringHelper("DependencyHloOrdering");
331}

333SequentialHloOrdering::SequentialHloOrdering(
334 const HloModule* module, const HloModuleSequence& module_sequence)
335 : HloOrdering(module), module_sequence_(module_sequence) {
336 // Create a map from instruction to its order position.
337 for (auto computation_order : module_sequence_) {
338 const std::vector<const HloInstruction*>& order = computation_order.second;
339 for (int i = 0; i < order.size(); ++i) {
340 DCHECK_EQ(0, order_position_.count(order[i]));
341 order_position_.emplace(order[i], i);
342 }
343 }
344}
345
346bool SequentialHloOrdering::ExecutesBeforeInSameComputation(
347 const HloInstruction* a, const HloInstruction* b) const {
348 CHECK_EQ(a->parent(), b->parent());
349 // If either instruction is not in the order, then 'a' and 'b' are unordered.
350 if (order_position_.count(a) == 0 || order_position_.count(b) == 0) {
351 return false;
352 }
353 return order_position_.at(a) < order_position_.at(b);
354}
355
356const std::vector<const HloInstruction*>*
357SequentialHloOrdering::SequentialOrder(
358 const HloComputation& computation) const {
359 auto find_it = module_sequence_.find(&computation);
360 return find_it == module_sequence_.end() ? nullptr : &find_it->second;
361}
362
363string SequentialHloOrdering::ToString() const {
364 std::vector<string> pieces;
365 pieces.push_back("SequentialHloOrdering");
366 for (auto* computation : module_->computations()) {
367 pieces.push_back(tensorflow::strings::Printf("computation %s order:",
368 computation->name().c_str()));
369 // Gather all instructions in the module sequence for this computation and
370 // sort them by their position.
371 std::vector<const HloInstruction*> instructions;
372 for (auto& instruction_position : order_position_) {
373 const HloInstruction* instruction = instruction_position.first;
374 if (instruction->parent() == computation) {
375 instructions.push_back(instruction);
376 }
377 }
378 std::sort(instructions.begin(), instructions.end(),
379 [this](const HloInstruction* a, const HloInstruction* b) {
380 return order_position_.at(a) < order_position_.at(b);
381 });
382 for (auto instruction : instructions) {
383 pieces.push_back(
384 tensorflow::strings::Printf(" %s", instruction->name().c_str()));
385 }
386 }
387 return tensorflow::str_util::Join(pieces, "\n");
388}

390std::ostream& operator<<(
391 std::ostream& out,
392 const SequentialHloOrdering::HloModuleSequence& module_sequence) {
393 for (auto computation_pair : module_sequence) {
394 const HloComputation* computation = computation_pair.first;
395 const std::vector<const HloInstruction*>& computation_sequence =
396 computation_pair.second;
397 out << "Computation " << computation->name() << ":\n";
398 for (auto* instruction : computation_sequence) {
399 out << " " << instruction->name() << "\n";
400 }
401 }
402 return out;
403}
