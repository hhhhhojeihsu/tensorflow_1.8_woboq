42namespace cpu {

44// CPU-targeting implementation of the XLA Executable interface.
45//
46// Wraps a JIT-ed object that can be executed "on device". We JIT for the host
47// architecture, so JIT-ed code and host code share the same ABI.
48class CpuExecutable : public Executable {
49 public:
50 CpuExecutable(std::unique_ptr<SimpleOrcJIT> jit,
51 std::unique_ptr<const BufferAssignment> assignment,
52 std::unique_ptr<const HloModule> hlo_module,
53 const string& entry_function_name,
54 std::unique_ptr<HloProfilePrinterData> hlo_profile_printer_data,
55 std::unique_ptr<HloProfileIndexMap> hlo_profile_index_map);
56 ~CpuExecutable() override {}
57
58 StatusOr<std::unique_ptr<ShapedBuffer>> ExecuteOnStream(
59 const ServiceExecutableRunOptions* run_options,
60 tensorflow::gtl::ArraySlice<const ShapedBuffer*> arguments,
61 HloExecutionProfile* hlo_execution_profile) override;
62
63 StatusOr<std::unique_ptr<ShapedBuffer>> ExecuteAsyncOnStream(
64 const ServiceExecutableRunOptions* run_options,
65 tensorflow::gtl::ArraySlice<const ShapedBuffer*> arguments) override;
66
67 // This should be called after set_ir_module_string.
68 const string& ir_module_string() const { return ir_module_string_; }
69
70 void set_ir_module_string(const string& ir_module_string) {
71 ir_module_string_ = ir_module_string;
72 }
73
74 static int64 ShapeSizeBytes(const Shape& shape);
75
76 // Type of the computation function we expect in the JIT.
77 using ComputeFunctionType = void (*)(
78 void* /*result*/, const ExecutableRunOptions* /*run_options*/,
79 const void** /*args*/, void** /*temps*/, int64* /*profile_counters*/);
80
81 const ComputeFunctionType& compute_function() const {
82 return compute_function_;
83 }
84
85 const BufferAssignment& buffer_assignment() const { return *assignment_; }
86
87 private:
88 // Allocate buffers required for execution and assign them to the elements of
89 // "buffers". "buffers" should be sized to the number of buffers in buffer
90 // assignment. Each vector element corresponds to a particular Index. If
91 // a vector element already contains a non-null DeviceMemoryBase, then no
92 // buffer is assigned for this element.
93 Status AllocateBuffers(
94 DeviceMemoryAllocator* memory_allocator, int device_ordinal,
95 std::vector<perftools::gputools::DeviceMemoryBase>* buffers);
96
97 // Calls the generated function performing the computation with the given
98 // arguments using the supplied buffers.
99 Status ExecuteComputeFunction(
100 const ExecutableRunOptions* run_options,
101 tensorflow::gtl::ArraySlice<const ShapedBuffer*> arguments,
102 tensorflow::gtl::ArraySlice<perftools::gputools::DeviceMemoryBase>
103 buffers,
104 HloExecutionProfile* hlo_execution_profile);
105
106 // Create a ShapedBuffer for holding the result of the computation. The
107 // addresses (DeviceMemoryBases) are set according to buffer assignment.
108 // 'buffers_in_result' should point to a vector of the same size as
109 // 'allocated_buffers'. An element in buffers_in_result is set to true if the
110 // corresponding buffer is live out of the computation (and thus contained in
111 // the returned ShapedBuffer).
112 StatusOr<std::unique_ptr<ShapedBuffer>> CreateResultShapedBuffer(
113 const ServiceExecutableRunOptions* run_options,
114 tensorflow::gtl::ArraySlice<perftools::gputools::DeviceMemoryBase>
115 allocated_buffers,
116 std::vector<bool>* buffers_in_result);
117
118 // Returns the points-to set of the root instruction of the entry
119 // computation. Uses points-to analysis from buffer assignment.
120 const PointsToSet& GetRootPointsToSet() const;
121
122 // The JIT containing compiled modules.
123 const std::unique_ptr<SimpleOrcJIT> jit_;
124
125 // Buffer assignment for the buffers we need to allocate.
126 const std::unique_ptr<const BufferAssignment> assignment_;
127
128 // The LLVM IR, in string format, of the unoptimized module generated for this
129 // CpuExecutable. We save a string instead of an llvm::Module* because leaving
130 // llvm::Module* in a singleton can cause the heap checker to emit false
131 // positives.
132 string ir_module_string_;
133
134 ComputeFunctionType compute_function_;
135
136 // Entry function name for the computation.
137 const string entry_function_name_;
138
139 TF_DISALLOW_COPY_AND_ASSIGN(CpuExecutable);
140};
