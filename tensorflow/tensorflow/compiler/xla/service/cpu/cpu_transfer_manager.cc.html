43class CpuInfeedBuffer : public cpu::runtime::XfeedBuffer {
44 public:
45 explicit CpuInfeedBuffer(int32 length)
46 : length_(length),
47 buffer_(new char[length]),
48 device_memory_(buffer_, length_) {}
49 ~CpuInfeedBuffer() override { delete[] buffer_; }
50
51 int32 length() override { return length_; }
52 void* data() override { return buffer_; }
53 void Done(StatusOr<Shape> /*shape*/) override { delete this; }
54
55 se::DeviceMemoryBase* device_memory() { return &device_memory_; }
56
57 private:
58 int32 length_;
59 char* buffer_;
60 se::DeviceMemoryBase device_memory_;
61};
62
63class CpuOutfeedBuffer : public cpu::runtime::XfeedBuffer {
64 public:
65 CpuOutfeedBuffer(void* destination, int32 length)
66 : destination_(destination), length_(length) {}
67
68 StatusOr<Shape> WaitForNotification() {
69 done_.WaitForNotification();
70 return status_;
71 }
72
73 int32 length() override { return length_; }
74 void* data() override { return destination_; }
75 void Done(StatusOr<Shape> shape) override {
76 status_ = std::move(shape);
77 done_.Notify();
78 }
79
80 private:
81 void* destination_;
82 int32 length_;
83 StatusOr<Shape> status_;
84 tensorflow::Notification done_;
85};

93Status CpuTransferManager::TransferLiteralToInfeed(se::StreamExecutor* executor,
94 const Literal& literal) {
95 const Shape& shape = literal.shape();
96 VLOG(2) << "Transferring literal to infeed with shape: "
97 << ShapeUtil::HumanString(shape);
98
99 if (!ShapeUtil::IsTuple(shape)) {
100 int64 size = GetByteSizeRequirement(shape);
101 return TransferBufferToInfeed(executor, size, literal.untyped_data());
102 }
103
104 if (ShapeUtil::IsNestedTuple(shape)) {
105 return Unimplemented(
106 "Infeed with a nested tuple shape is not supported: %s",
107 ShapeUtil::HumanString(literal.shape()).c_str());
108 }
109
110 // For a tuple, we transfer each of its elements to the device and
111 // enqueue the resulting destination device addresses with the
112 // infeed manager.
113 std::vector<cpu::runtime::XfeedBuffer*> buffers;
114 buffers.reserve(ShapeUtil::TupleElementCount(shape));
115 auto cleanup = tensorflow::gtl::MakeCleanup([&buffers]() {
116 for (cpu::runtime::XfeedBuffer* b : buffers) {
117 b->Done(Cancelled("Failed to infeed buffer to device."));
118 }
119 });
120
121 for (int64 i = 0; i < ShapeUtil::TupleElementCount(shape); ++i) {
122 const Shape& tuple_element_shape = ShapeUtil::GetSubshape(shape, {i});
123 int64 tuple_element_size = GetByteSizeRequirement(tuple_element_shape);
124 TF_ASSIGN_OR_RETURN(
125 cpu::runtime::XfeedBuffer * buffer,
126 TransferBufferToInfeedInternal(executor, tuple_element_size,
127 literal.untyped_data({i})));
128 buffers.push_back(buffer);
129 }
130
131 cpu::runtime::XfeedManager* xfeed_manager = cpu::runtime::GetXfeedManager();
132 xfeed_manager->infeed()->EnqueueBuffersAtomically(buffers);
133
134 cleanup.release();
135 return Status::OK();
136}

138Status CpuTransferManager::TransferBufferToInfeed(se::StreamExecutor* executor,
139 int64 size,
140 const void* source) {
141 TF_ASSIGN_OR_RETURN(cpu::runtime::XfeedBuffer * buffer,
142 TransferBufferToInfeedInternal(executor, size, source));
143
144 cpu::runtime::XfeedManager* xfeed_manager = cpu::runtime::GetXfeedManager();
145 xfeed_manager->infeed()->EnqueueBuffersAtomically({buffer});
146
147 return Status::OK();
148}
149
150StatusOr<cpu::runtime::XfeedBuffer*>
151CpuTransferManager::TransferBufferToInfeedInternal(se::StreamExecutor* executor,
152 int64 size,
153 const void* source) {
154 if (size > std::numeric_limits<int32>::max()) {
155 return InvalidArgument("Infeed shape is too large: needs %lld bytes", size);
156 }
157
158 if (size <= 0) {
159 return InvalidArgument("Infeed shape must have positive size; got %lld",
160 size);
161 }
162
163 int32 size_32 = static_cast<int32>(size);
164 CpuInfeedBuffer* queued_buffer = new CpuInfeedBuffer(size_32);
165 Status s =
166 TransferBufferToDevice(executor, /*size=*/size,
167 /*source=*/source, queued_buffer->device_memory());
168
169 if (!s.ok()) {
170 queued_buffer->Done(s);
171 return s;
172 }
173 return queued_buffer;
174}

176Status CpuTransferManager::TransferLiteralFromOutfeed(
177 se::StreamExecutor* executor, const Shape& literal_shape,
178 Literal* literal) {
179 if (!ShapeUtil::IsTuple(literal_shape)) {
180 int64 size = GetByteSizeRequirement(literal_shape);
181 // Note: OSS build didn't like implicit conversion from
182 // literal_shape.dimensions() to the array slice on 2017-07-10.
183 tensorflow::gtl::ArraySlice<int64> dimensions(
184 tensorflow::bit_cast<const int64*>(literal_shape.dimensions().data()),
185 literal_shape.dimensions().size());
186 *literal = std::move(*Literal::CreateFromDimensions(
187 literal_shape.element_type(), dimensions));
188 TF_ASSIGN_OR_RETURN(Shape received_shape,
189 TransferArrayBufferFromOutfeed(
190 executor, literal->untyped_data(), size));
191 TF_RET_CHECK(ShapeUtil::Compatible(received_shape, literal->shape()))
192 << "Shape received from outfeed "
193 << ShapeUtil::HumanString(received_shape)
194 << " did not match the shape that was requested for outfeed: "
195 << ShapeUtil::HumanString(literal_shape);
196 TF_RET_CHECK(size == GetByteSizeRequirement(received_shape));
197 *literal->mutable_shape_do_not_use() = received_shape;
198 return Status::OK();
199 }
200
201 if (ShapeUtil::IsNestedTuple(literal_shape)) {
202 return Unimplemented(
203 "Nested tuple outfeeds are not yet implemented on CPU.");
204 }
205
206 std::vector<std::unique_ptr<Literal>> elements;
207 std::vector<std::pair<void*, int64>> buffer_data;
208 for (int64 i = 0; i < literal_shape.tuple_shapes_size(); ++i) {
209 const Shape& tuple_element_shape =
210 ShapeUtil::GetTupleElementShape(literal_shape, i);
211 // Note: OSS build didn't like implicit conversion from
212 // literal_shape.dimensions() to the array slice on 2017-07-10.
213 tensorflow::gtl::ArraySlice<int64> dimensions(
214 tensorflow::bit_cast<const int64*>(
215 tuple_element_shape.dimensions().data()),
216 tuple_element_shape.dimensions().size());
217 auto empty = Literal::CreateFromDimensions(
218 tuple_element_shape.element_type(), dimensions);
219 int64 size = GetByteSizeRequirement(tuple_element_shape);
220 buffer_data.push_back({empty->untyped_data(), size});
221 elements.push_back(std::move(empty));
222 }
223
224 TF_ASSIGN_OR_RETURN(Shape received_shape,
225 TransferTupleBuffersFromOutfeed(executor, buffer_data));
226
227 TF_RET_CHECK(ShapeUtil::Compatible(received_shape, literal_shape))
228 << "Shape received from outfeed "
229 << ShapeUtil::HumanString(received_shape)
230 << " did not match the shape that was requested for outfeed: "
231 << ShapeUtil::HumanString(literal_shape);
232 TF_RET_CHECK(GetByteSizeRequirement(literal_shape) ==
233 GetByteSizeRequirement(received_shape));
234
235 for (int64 i = 0; i < literal_shape.tuple_shapes_size(); ++i) {
236 *elements[i]->mutable_shape_do_not_use() = received_shape.tuple_shapes(i);
237 }
238 *literal = std::move(*Literal::MakeTupleOwned(std::move(elements)));
239 TF_RET_CHECK(ShapeUtil::Equal(literal->shape(), literal_shape));
240 return Status::OK();
241}

253 return TransferBuffersFromOutfeedInternal(
254 executor, {{destination, size_bytes}}, /*is_tuple=*/false);

257StatusOr<Shape> CpuTransferManager::TransferBuffersFromOutfeedInternal(
258 perftools::gputools::StreamExecutor* executor,
259 tensorflow::gtl::ArraySlice<std::pair<void*, int64>> buffer_data,
260 bool is_tuple) {
261 std::vector<std::unique_ptr<CpuOutfeedBuffer>> buffers;
262 for (auto b : buffer_data) {
263 int64 size = b.second;
264 if (size > std::numeric_limits<int32>::max()) {
265 return InvalidArgument("Outfeed shape is too large: needs %lld bytes",
266 size);
267 }
268
269 if (size <= 0) {
270 return InvalidArgument("Outfeed shape must have positive size; got %lld",
271 size);
272 }
273
274 int32 size_32 = static_cast<int32>(size);
275 VLOG(2)
276 << "Enqueueing outfeed buffer (for the device to populate) of length "
277 << size_32 << "B";
278 buffers.emplace_back(MakeUnique<CpuOutfeedBuffer>(b.first, size_32));
279 }
280
281 std::vector<cpu::runtime::XfeedBuffer*> buffer_pointers;
282 buffer_pointers.reserve(buffers.size());
283 for (auto& b : buffers) {
284 buffer_pointers.push_back(b.get());
285 }
286
287 cpu::runtime::XfeedManager* xfeed_manager = cpu::runtime::GetXfeedManager();
288 xfeed_manager->outfeed()->EnqueueBuffersAtomically(buffer_pointers);
289 VLOG(2) << "Waiting for buffer to be notified as populated.";
290 std::vector<Shape> outfed_shapes;
291 for (auto& buffer : buffers) {
292 TF_ASSIGN_OR_RETURN(Shape outfed_shape, buffer->WaitForNotification());
293 outfed_shapes.push_back(std::move(outfed_shape));
294 }
295 if (is_tuple) {
296 return ShapeUtil::MakeTupleShape(outfed_shapes);
297 }
298 TF_RET_CHECK(outfed_shapes.size() == 1);
299 return std::move(outfed_shapes[0]);
300}
