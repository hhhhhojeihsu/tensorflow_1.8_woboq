28// We want to change the layout of constant arrays to be column major when all
29// of their users are dot operations that can be made faster with the flipped
30// layout. To avoid going quadriatic over the # of instructions, we cache this
31// property in should_make_rhs_col_major -- it maps a constant to true if all of
32// the users of said constant are dot operations that can be sped up. This
33// cache is populated lazily as we encounter dot operations traversing the
34// instruction stream.
35
36namespace {
37using ::tensorflow::gtl::nullopt;
38using ::tensorflow::gtl::optional;
39
40using ShouldMakeOperandColMajorCache =
41 tensorflow::gtl::FlatMap<const HloInstruction*, bool>;
42} // namespace
43
44static bool ShouldMakeAllUsersColMajor(const HloInstruction* instruction) {
45 for (auto* user : instruction->users()) {
46 optional<int64> operand_idx = ProfitableToMakeDotOperandColumnMajor(*user);
47 if (!operand_idx || user->operand(*operand_idx) != instruction ||
48 std::count(user->operands().begin(), user->operands().end(),
49 instruction) != 1) {
50 return false;
51 }
52 }
53 return true;
54}
55
56static optional<int64> ShouldMakeOperandColumnMajor(
57 ShouldMakeOperandColMajorCache* cache, const HloInstruction& instruction) {
58 optional<int64> operand_idx =
59 ProfitableToMakeDotOperandColumnMajor(instruction);
60 if (!operand_idx) {
61 return nullopt;
62 }
63
64 const HloInstruction* operand = instruction.operand(*operand_idx);
65 if (operand->opcode() != HloOpcode::kConstant) {
66 return nullopt;
67 }
68
69 auto it = cache->find(operand);
70 if (it == cache->end()) {
71 auto insert_result =
72 cache->insert({operand, ShouldMakeAllUsersColMajor(operand)});
73 CHECK(insert_result.second);
74 it = insert_result.first;
75 }
76
77 return it->second ? operand_idx : nullopt;
78}

80static Shape RowMajorShape(const Shape& old_shape) {
81 Shape new_shape(old_shape);
82 std::vector<int64> dimension_order(new_shape.dimensions_size());
83 std::iota(dimension_order.rbegin(), dimension_order.rend(), 0);
84 *new_shape.mutable_layout() = LayoutUtil::MakeLayout(dimension_order);
85 return new_shape;
86}
87
88static Shape ColMajorShape(const Shape& old_shape) {
89 Shape new_shape(old_shape);
90 std::vector<int64> dimension_order(new_shape.dimensions_size());
91 std::iota(dimension_order.begin(), dimension_order.end(), 0);
92 *new_shape.mutable_layout() = LayoutUtil::MakeLayout(dimension_order);
93 return new_shape;
94}

96Status CpuLayoutAssignment::AddBackendConstraints(
97 LayoutConstraints* constraints) {
98 ShouldMakeOperandColMajorCache cache;
99
100 const HloComputation* computation = constraints->computation();
101 for (auto* instruction : computation->instructions()) {
102 if (instruction->opcode() == HloOpcode::kConvolution &&
103 PotentiallyImplementedAsEigenConvolution(*instruction)) {
104 const HloInstruction* convolution = instruction;
105 const HloInstruction* lhs_instruction = convolution->operand(0);
106 const HloInstruction* rhs_instruction = convolution->operand(1);
107
108 // In order to implement `convolution` with Eigen convolution, the layouts
109 // of the input, filter, and output need to be row-major.
110 //
111 // These constraints are not hard constraints. Ideally, we should decide
112 // which layouts to choose according to some cost model.
113 Shape output_shape(RowMajorShape(convolution->shape()));
114 Shape input_shape(RowMajorShape(lhs_instruction->shape()));
115 Shape filter_shape(RowMajorShape(rhs_instruction->shape()));
116
117 // Set layouts of the instructions' shapes.
118 TF_RETURN_IF_ERROR(
119 constraints->SetOperandLayout(input_shape, convolution, 0));
120 TF_RETURN_IF_ERROR(
121 constraints->SetOperandLayout(filter_shape, convolution, 1));
122 TF_RETURN_IF_ERROR(
123 constraints->SetInstructionLayout(output_shape, convolution));
124 } else if (optional<int64> op_idx =
125 ShouldMakeOperandColumnMajor(&cache, *instruction)) {
126 const HloInstruction* op = instruction->operand(*op_idx);
127 TF_RETURN_IF_ERROR(constraints->SetOperandLayout(
128 ColMajorShape(op->shape()), instruction, *op_idx));
129 } else if (PotentiallyImplementedAsEigenDot(*instruction)) {
130 const HloInstruction* dot = instruction;
131 // In order to implement `dot` with Eigen dot, the layouts of the lhs,
132 // rhs, and output need to be row-major.
133 //
134 // These constraints are not hard constraints. Ideally, we should decide
135 // which layouts to choose according to some cost model.
136 Shape output_shape(RowMajorShape(dot->shape()));
137
138 const HloInstruction* lhs_instruction = dot->operand(0);
139 Shape lhs_shape(RowMajorShape(lhs_instruction->shape()));
140 TF_RETURN_IF_ERROR(constraints->SetOperandLayout(lhs_shape, dot, 0));
141
142 // dot is a kDot or a kTransposeDot fusion node. In the latter case, if
143 // it represents X @ X, it may have just one operand.
144 if (dot->operand_count() > 1) {
145 const HloInstruction* rhs_instruction = dot->operand(1);
146 Shape rhs_shape(RowMajorShape(rhs_instruction->shape()));
147 TF_RETURN_IF_ERROR(constraints->SetOperandLayout(rhs_shape, dot, 1));
148 }
149
150 // Set layouts of the instructions' shapes.
151 TF_RETURN_IF_ERROR(constraints->SetInstructionLayout(output_shape, dot));
152 } else {
153 for (int64 operand_no = 0; operand_no < instruction->operand_count();
154 ++operand_no) {
155 // Skip operands which already have a constraint.
156 if (constraints->OperandLayout(instruction, operand_no) != nullptr) {
157 continue;
158 }
159 // Skip over forwarded operands.
160 if (constraints->OperandBufferForwarded(instruction, operand_no)) {
161 continue;
162 }
163 // Skip operands with non-array shapes.
164 if (!ShapeUtil::IsArray(instruction->operand(operand_no)->shape())) {
165 continue;
166 }
167 Shape operand_shape(
168 RowMajorShape(instruction->operand(operand_no)->shape()));
169 TF_RETURN_IF_ERROR(constraints->SetOperandLayout(
170 operand_shape, instruction, operand_no));
171 }
172 // Skip over the root instruction for the top-level computation.
173 if (computation->parent()->entry_computation() == computation &&
174 computation->root_instruction() == instruction) {
175 continue;
176 }
177 // Skip instructions which don't produce array shapes (tuples, opaque,
178 // etc.).
179 if (!ShapeUtil::IsArray(instruction->shape())) {
180 continue;
181 }
182 }
183 }
184 return tensorflow::Status::OK();
185}
