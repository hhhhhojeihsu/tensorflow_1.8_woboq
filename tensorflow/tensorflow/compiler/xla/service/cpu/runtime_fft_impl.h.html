35// Computes either a forward or reverse complex-to-complex FFT.
36template <bool Forward, int FFTRank, typename EigenDevice>
37void EigenFftC2C(const EigenDevice& device, complex64* out, complex64* operand,
38 int64 input_batch, int64 fft_length0, int64 fft_length1,
39 int64 fft_length2) {
40 // Create the axes (which are always trailing).
41 const auto axes = Eigen::ArrayXi::LinSpaced(FFTRank, 1, FFTRank);
42 constexpr auto direction = Forward ? Eigen::FFT_FORWARD : Eigen::FFT_REVERSE;
43
44 const std::array<int64, 3> fft_shape = {
45 {fft_length0, fft_length1, fft_length2}};
46
47 Eigen::DSizes<Eigen::DenseIndex, FFTRank + 1> dims;
48 dims[0] = input_batch;
49 for (int i = 0; i < FFTRank; i++) {
50 dims[i + 1] = fft_shape[i];
51 }
52 const Eigen::TensorMap<Eigen::Tensor<complex64, FFTRank + 1, Eigen::RowMajor>,
53 Eigen::Aligned>
54 input(operand, dims);
55 Eigen::TensorMap<Eigen::Tensor<complex64, FFTRank + 1, Eigen::RowMajor>,
56 Eigen::Aligned>
57 output(out, dims);
58 output.device(device) = input.template fft<Eigen::BothParts, direction>(axes);
59}

61// Computes a forward real->complex FFT, slicing out redundant negative
62// frequencies from the innermost dimension.
63template <int FFTRank, typename EigenDevice>
64void EigenFftR2C(const EigenDevice& device, complex64* out, float* operand,
65 int64 input_batch, int64 fft_length0, int64 fft_length1,
66 int64 fft_length2) {
67 const std::array<int64, 3> fft_shape = {
68 {fft_length0, fft_length1, fft_length2}};
69
70 Eigen::DSizes<Eigen::DenseIndex, FFTRank + 1> in_dims;
71 in_dims[0] = input_batch;
72 Eigen::DSizes<Eigen::DenseIndex, FFTRank + 1> out_dims;
73 out_dims[0] = input_batch;
74 TensorShape temp_shape{input_batch};
75 for (int i = 0; i < FFTRank; i++) {
76 in_dims[i + 1] = fft_shape[i];
77 out_dims[i + 1] = i == FFTRank - 1 ? fft_shape[i] / 2 + 1 : fft_shape[i];
78 temp_shape.AddDim(fft_shape[i]);
79 }
80 const Eigen::TensorMap<Eigen::Tensor<float, FFTRank + 1, Eigen::RowMajor>,
81 Eigen::Aligned>
82 input(operand, in_dims);
83 Eigen::TensorMap<Eigen::Tensor<complex64, FFTRank + 1, Eigen::RowMajor>,
84 Eigen::Aligned>
85 output(out, out_dims);
86
87 // Create the axes (which are always trailing).
88 const auto axes = Eigen::ArrayXi::LinSpaced(FFTRank, 1, FFTRank);
89
90 // Compute the full FFT using a temporary tensor.
91 Tensor temp(DataTypeToEnum<complex64>::v(), temp_shape);
92 auto full_fft = temp.flat_inner_dims<complex64, FFTRank + 1>();
93 const Eigen::DSizes<Eigen::DenseIndex, FFTRank + 1> zero_start_indices;
94 full_fft.device(device) =
95 input.template fft<Eigen::BothParts, Eigen::FFT_FORWARD>(axes);
96
97 // Slice away the negative frequency components.
98 output.device(device) = full_fft.slice(zero_start_indices, out_dims);
99}

101// Computes a reverse complex->real FFT, reconstructing redundant negative
102// frequencies using reverse conjugate on innermost dimension after doing IFFT
103// on outer dimensions.
104template <int FFTRank, typename EigenDevice>
105void EigenFftC2R(const EigenDevice& device, float* out, complex64* operand,
106 int64 input_batch, int64 fft_length0, int64 fft_length1,
107 int64 fft_length2) {
108 const std::array<int64, 3> fft_shape = {
109 {fft_length0, fft_length1, fft_length2}};
110
111 Eigen::DSizes<Eigen::DenseIndex, FFTRank + 1> in_dims;
112 in_dims[0] = input_batch;
113 Eigen::DSizes<Eigen::DenseIndex, FFTRank + 1> out_dims;
114 out_dims[0] = input_batch;
115 TensorShape temp_shape{input_batch};
116 for (int i = 0; i < FFTRank; i++) {
117 in_dims[i + 1] = i == FFTRank - 1 ? fft_shape[i] / 2 + 1 : fft_shape[i];
118 out_dims[i + 1] = fft_shape[i];
119 temp_shape.AddDim(fft_shape[i]);
120 }
121 const Eigen::TensorMap<Eigen::Tensor<complex64, FFTRank + 1, Eigen::RowMajor>,
122 Eigen::Aligned>
123 input(operand, in_dims);
124 Eigen::TensorMap<Eigen::Tensor<float, FFTRank + 1, Eigen::RowMajor>,
125 Eigen::Aligned>
126 output(out, out_dims);
127
128 // Calculate the shape of the temporary tensor for the full FFT and the
129 // region we will slice from input given fft_shape. We slice input to
130 // fft_shape on its inner-most dimensions, except the last (which we
131 // slice to fft_shape[-1] / 2 + 1).
132 Tensor temp(DataTypeToEnum<complex64>::v(), temp_shape);
133 auto full_fft = temp.flat_inner_dims<complex64, FFTRank + 1>();
134
135 // Calculate the starting point and range of the source of
136 // negative frequency part.
137 auto neg_sizes = in_dims;
138 neg_sizes[FFTRank] = fft_shape[FFTRank - 1] - in_dims[FFTRank];
139 Eigen::DSizes<Eigen::DenseIndex, FFTRank + 1> neg_target_indices;
140 neg_target_indices[FFTRank] = in_dims[FFTRank];
141
142 const Eigen::DSizes<Eigen::DenseIndex, FFTRank + 1> zero_start_indices;
143 Eigen::DSizes<Eigen::DenseIndex, FFTRank + 1> neg_start_indices;
144 neg_start_indices[FFTRank] = 1;
145
146 full_fft.slice(zero_start_indices, in_dims).device(device) = input;
147
148 // First, conduct IFFTs on outer dimensions. We save computation (and
149 // avoid touching uninitialized memory) by slicing full_fft to the
150 // subregion we wrote input to.
151 if (FFTRank > 1) {
152 const auto outer_axes =
153 Eigen::ArrayXi::LinSpaced(FFTRank - 1, 1, FFTRank - 1);
154 full_fft.slice(zero_start_indices, in_dims).device(device) =
155 full_fft.slice(zero_start_indices, in_dims)
156 .template fft<Eigen::BothParts, Eigen::FFT_REVERSE>(outer_axes);
157 }
158
159 // Reconstruct the full FFT by appending reversed and conjugated
160 // spectrum as the negative frequency part.
161 Eigen::array<bool, FFTRank + 1> reverse_last_axis;
162 for (auto i = 0; i <= FFTRank; i++) {
163 reverse_last_axis[i] = i == FFTRank;
164 }
165
166 if (neg_sizes[FFTRank] != 0) {
167 full_fft.slice(neg_target_indices, neg_sizes).device(device) =
168 full_fft.slice(neg_start_indices, neg_sizes)
169 .reverse(reverse_last_axis)
170 .conjugate();
171 }
172
173 auto inner_axis = Eigen::array<int, 1>{FFTRank};
174 output.device(device) =
175 full_fft.template fft<Eigen::RealPart, Eigen::FFT_REVERSE>(inner_axis);
176}

178template <int FFTRank, typename EigenDevice>
179void EigenFftWithRank(const EigenDevice& device, void* out, void* operand,
180 int32 fft_type, int64 input_batch, int64 fft_length0,
181 int64 fft_length1, int64 fft_length2) {
182 CHECK(::xla::FftType_IsValid(fft_type)) << fft_type;
183 switch (fft_type) {
184 case ::xla::FftType::FFT:
185 EigenFftC2C<true, FFTRank, EigenDevice>(
186 device, static_cast<complex64*>(out),
187 static_cast<complex64*>(operand), input_batch, fft_length0,
188 fft_length1, fft_length2);
189 break;
190 case ::xla::FftType::IFFT:
191 EigenFftC2C<false, FFTRank, EigenDevice>(
192 device, static_cast<complex64*>(out),
193 static_cast<complex64*>(operand), input_batch, fft_length0,
194 fft_length1, fft_length2);
195 break;
196 case ::xla::FftType::RFFT:
197 EigenFftR2C<FFTRank, EigenDevice>(
198 device, static_cast<complex64*>(out), static_cast<float*>(operand),
199 input_batch, fft_length0, fft_length1, fft_length2);
200 break;
201 case ::xla::FftType::IRFFT:
202 EigenFftC2R<FFTRank, EigenDevice>(
203 device, static_cast<float*>(out), static_cast<complex64*>(operand),
204 input_batch, fft_length0, fft_length1, fft_length2);
205 break;
206 default:
207 LOG(FATAL) << "Unsupported FFT type: " << fft_type;
208 }
209}

213template <typename EigenDevice>
214void EigenFftImpl(const EigenDevice& device, void* out, void* operand,
215 int32 fft_type, int32 fft_rank, int64 input_batch,
216 int64 fft_length0, int64 fft_length1, int64 fft_length2) {
217 switch (fft_rank) {
218 case 1:
219 internal::EigenFftWithRank<1, EigenDevice>(
220 device, out, operand, fft_type, input_batch, fft_length0, 0, 0);
221 break;
222 case 2:
223 internal::EigenFftWithRank<2, EigenDevice>(device, out, operand, fft_type,
224 input_batch, fft_length0,
225 fft_length1, 0);
226 break;
227 case 3:
228 internal::EigenFftWithRank<3, EigenDevice>(device, out, operand, fft_type,
229 input_batch, fft_length0,
230 fft_length1, fft_length2);
231 break;
232 default:
233 LOG(FATAL) << "Unsupported FFT rank " << fft_rank;
234 }
235}
