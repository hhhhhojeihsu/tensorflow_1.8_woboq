27TF_ATTRIBUTE_NO_SANITIZE_MEMORY void __xla_cpu_runtime_EigenConvF32(
28 const void* run_options_ptr, float* out, float* lhs, float* rhs,
29 int64 input_batch, int64 input_rows, int64 input_cols, int64 input_channels,
30 int64 kernel_rows, int64 kernel_cols, int64 kernel_channels,
31 int64 kernel_filters, int64 output_rows, int64 output_cols,
32 int64 row_stride, int64 col_stride, int64 padding_top, int64 padding_bottom,
33 int64 padding_left, int64 padding_right, int64 lhs_row_dilation,
34 int64 lhs_col_dilation, int64 rhs_row_dilation, int64 rhs_col_dilation) {
35 const xla::ExecutableRunOptions* run_options =
36 static_cast<const xla::ExecutableRunOptions*>(run_options_ptr);
37 tensorflow::xla::EigenConvImpl(
38 *run_options->intra_op_thread_pool(), out, lhs, rhs, input_batch,
39 input_rows, input_cols, input_channels, kernel_rows, kernel_cols,
40 kernel_channels, kernel_filters, output_rows, output_cols, row_stride,
41 col_stride, padding_top, padding_bottom, padding_left, padding_right,
42 lhs_row_dilation, lhs_col_dilation, rhs_row_dilation, rhs_col_dilation);
43}
