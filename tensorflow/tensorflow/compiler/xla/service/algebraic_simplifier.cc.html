49// Returns whether operand is a literal with the given value.
50bool IsLiteralWithValue(const HloInstruction* operand, int8 value) {
51 return operand->opcode() == HloOpcode::kConstant &&
52 operand->literal().IsAll(value);
53}
54
55bool IsAll(const HloInstruction* op, int8 value) {
56 if (IsLiteralWithValue(op, value)) {
57 return true;
58 }
59 if (op->opcode() == HloOpcode::kBroadcast && IsAll(op->operand(0), value)) {
60 return true;
61 }
62 return false;
63}

65// Returns whether the given transpose produces a result which is bit-wise
66// identical to its operand and thus may be replaced with a bitcast.
67bool TransposeIsBitcast(const HloInstruction* transpose) {
68 CHECK_EQ(HloOpcode::kTranspose, transpose->opcode());
69 const HloInstruction* operand = transpose->operand(0);
70 return ShapeUtil::TransposeIsBitcast(operand->shape(), transpose->shape(),
71 transpose->dimensions());
72}

74// Returns true if the given reshape produces a result which is bit-wise
75// identical to its operand and thus may be replaced with a bitcast.
76//
77// This function is conservative -- even if this function returns false, the
78// reshape may still be a bitcast. For example, a reshape from [28x28] to [784].
79bool ReshapeIsBitcast(
80 const HloInstruction* reshape,
81 const AlgebraicSimplifier::ValidBitcastCallback& valid_bitcast_callback) {
82 CHECK_EQ(HloOpcode::kReshape, reshape->opcode());
83
84 const HloInstruction* operand = reshape->operand(0);
85 // Can't insert bitcasts if the compiler used a memory layout which isn't
86 // compatible.
87 return ShapeUtil::ReshapeIsBitcast(operand->shape(), reshape->shape()) &&
88 valid_bitcast_callback(operand->shape(), reshape->shape());
89}

91// Adds a scalar computation to the module to enable optimizations with dot
92// converting into reduction.
93HloComputation* CreateScalarBinaryComputation(HloModule* module,
94 PrimitiveType primitive_type,
95 HloOpcode opcode) {
96 HloComputation::Builder b("scalar_computation");
97 auto scalar_lhs = b.AddInstruction(HloInstruction::CreateParameter(
98 0, ShapeUtil::MakeShape(F32, {}), "scalar_lhs"));
99 auto scalar_rhs = b.AddInstruction(HloInstruction::CreateParameter(
100 1, ShapeUtil::MakeShape(F32, {}), "scalar_rhs"));
101 auto scalar_op = b.AddInstruction(
102 HloInstruction::CreateBinary(ShapeUtil::MakeShape(primitive_type, {}),
103 opcode, scalar_lhs, scalar_rhs));
104 HloComputation* scalar_computation =
105 module->AddEmbeddedComputation(b.Build(scalar_op));
106 return scalar_computation;
107}

320bool AlgebraicSimplifierVisitor::SameShape(const HloInstruction* lhs,
321 const HloInstruction* rhs) const {
322 if (is_layout_sensitive_) {
323 return ShapeUtil::Equal(lhs->shape(), rhs->shape());
324 } else {
325 return ShapeUtil::Compatible(lhs->shape(), rhs->shape());
326 }

329void AlgebraicSimplifierVisitor::ReplaceWithBitcast(
330 HloInstruction* instruction) {
331 CHECK_EQ(1, instruction->operand_count());
332 CHECK_EQ(ShapeUtil::ElementsIn(instruction->shape()),
333 ShapeUtil::ElementsIn(instruction->operand(0)->shape()));
334 CHECK_EQ(ShapeUtil::ByteSizeOf(instruction->shape()),
335 ShapeUtil::ByteSizeOf(instruction->operand(0)->shape()));
336
337 auto bitcast = computation_->AddInstruction(
338 HloInstruction::CreateUnary(instruction->shape(), HloOpcode::kBitcast,
339 instruction->mutable_operand(0)));
340 TF_CHECK_OK(ReplaceInstruction(instruction, bitcast));
341}

343bool AlgebraicSimplifierVisitor::ReplaceInstructionIfSameShape(
344 HloInstruction* old_instruction, HloInstruction* new_instruction) {
345 if (!SameShape(old_instruction, new_instruction)) {
346 return false;
347 }
348 TF_CHECK_OK(ReplaceInstruction(old_instruction, new_instruction));
349 return true;

352Status AlgebraicSimplifierVisitor::HandleAdd(HloInstruction* add) {
353 auto lhs = add->mutable_operand(0);
354 auto rhs = add->mutable_operand(1);
355 // A + 0 => A
356 VLOG(10) << "trying transform [A + 0 => A]: " << add->ToString();
357 if (IsAll(rhs, 0) && ReplaceInstructionIfSameShape(add, lhs)) {
358 return Status::OK();
359 }
360 // 0 + A => A
361 VLOG(10) << "trying transform [0 + A => A]: " << add->ToString();
362 if (IsAll(lhs, 0) && ReplaceInstructionIfSameShape(add, rhs)) {
363 return Status::OK();
364 }
365
366 // Canonicalization: Put constants on the right. This makes the reassociation
367 // rules below simpler.
368 VLOG(10) << "trying transform [Const + A => A + Const]";
369 if (lhs->IsConstant() && !rhs->IsConstant()) {
370 return ReplaceWithNewInstruction(
371 add,
372 HloInstruction::CreateBinary(add->shape(), HloOpcode::kAdd, rhs, lhs));
373 }
374
375 // Reassociate to allow constant folding.
376 //
377 // Note: This is not general. For example, we won't reassociate
378 //
379 // (A + C1) + (B + C2) => A + B + (C1 + C2).
380 //
381 VLOG(10) << "trying transform [(A + C1) + C2 => A + (C1 + C2)]";
382 if (rhs->IsConstant() && lhs->opcode() == HloOpcode::kAdd &&
383 !lhs->operand(0)->IsConstant() && lhs->operand(1)->IsConstant()) {
384 auto* c1 = lhs->mutable_operand(1);
385 auto* c2 = rhs;
386
387 TF_ASSIGN_OR_RETURN(auto* sum_of_constants,
388 MakeBinaryHlo(HloOpcode::kAdd, c1, c2));
389 return ReplaceWithNewInstruction(
390 add, HloInstruction::CreateBinary(add->shape(), HloOpcode::kAdd,
391 lhs->mutable_operand(0),
392 sum_of_constants));
393 }
394
395 return Status::OK();
396}

398Status AlgebraicSimplifierVisitor::HandleBitcast(HloInstruction* bitcast) {
399 // If a bitcast feeds a bitcast, make it a single bitcast.
400 if (bitcast->operand(0)->opcode() == HloOpcode::kBitcast) {
401 return ReplaceWithNewInstruction(
402 bitcast, HloInstruction::CreateUnary(
403 bitcast->shape(), HloOpcode::kBitcast,
404 bitcast->mutable_operand(0)->mutable_operand(0)));
405 }
406 // All bitcasts can be eliminated (assuming layout constraints are
407 // satisified).
408 ReplaceInstructionIfSameShape(bitcast, bitcast->mutable_operand(0));
409 return Status::OK();
410}

412Status AlgebraicSimplifierVisitor::HandleBitcastConvert(
413 HloInstruction* bitcast) {
414 // Eliminate bitcast converts between same shape.
415 ReplaceInstructionIfSameShape(bitcast, bitcast->mutable_operand(0));
416 return Status::OK();
417}

432Status AlgebraicSimplifierVisitor::HandleConcatenate(
433 HloInstruction* concatenate) {
434 tensorflow::gtl::ArraySlice<HloInstruction*> operands(
435 concatenate->operands());
436 if (operands.size() == 1) {
437 // Unary concatenates are useless.
438 ReplaceInstructionIfSameShape(concatenate, operands[0]);
439 return Status::OK();
440 }
441 // Filter out and remove empty operands.
442 std::vector<HloInstruction*> nonempty_operands;
443 for (HloInstruction* operand : operands) {
444 if (!ShapeUtil::HasZeroElements(operand->shape())) {
445 nonempty_operands.push_back(operand);
446 }
447 }
448 if (nonempty_operands.size() < operands.size()) {
449 HloInstruction* replacement;
450 if (nonempty_operands.empty()) {
451 replacement = operands[0];
452 } else if (nonempty_operands.size() == 1) {
453 replacement = nonempty_operands[0];
454 } else {
455 replacement =
456 computation_->AddInstruction(concatenate->CloneWithNewOperands(
457 concatenate->shape(), nonempty_operands));
458 }
459 VLOG(10) << "trying to replace " << concatenate->ToString() << " with "
460 << replacement->ToString();
461 ReplaceInstructionIfSameShape(concatenate, replacement);
462 } else if (operands.size() == 2) {
463 // A binary concat with a broadcasted scalar as an operand can be converted
464 // into a pad which is simpler to fold into other operations.
465 bool is_effective_low_pad =
466 operands[0]->opcode() == HloOpcode::kBroadcast &&
467 ShapeUtil::IsScalar(operands[0]->operand(0)->shape());
468 bool is_effective_high_pad =
469 operands[1]->opcode() == HloOpcode::kBroadcast &&
470 ShapeUtil::IsScalar(operands[1]->operand(0)->shape());
471 if (!is_effective_low_pad && !is_effective_high_pad) {
472 return Status::OK();
473 }
474 PaddingConfig padding_config;
475 for (int64 dim = 0; dim < ShapeUtil::Rank(operands[0]->shape()); ++dim) {
476 auto padding_config_dim = padding_config.add_dimensions();
477 padding_config_dim->set_edge_padding_high(0);
478 padding_config_dim->set_edge_padding_low(0);
479 padding_config_dim->set_interior_padding(0);
480 if (dim == concatenate->concatenate_dimension()) {
481 if (is_effective_low_pad) {
482 padding_config_dim->set_edge_padding_low(
483 operands[0]->shape().dimensions(dim));
484 } else {
485 padding_config_dim->set_edge_padding_high(
486 operands[1]->shape().dimensions(dim));
487 }
488 }
489 }
490 int64 operand_to_pad = is_effective_low_pad ? 1 : 0;
491 int64 pad_value_operand = is_effective_low_pad ? 0 : 1;
492 HloInstruction* pad =
493 computation_->AddInstruction(HloInstruction::CreatePad(
494 concatenate->shape(), operands[operand_to_pad],
495 operands[pad_value_operand]->mutable_operand(0), padding_config));
496 return ReplaceInstruction(concatenate, pad);
497 }
498 return Status::OK();
499}

501static HloInstruction* BuildTupleConstant(HloComputation* computation,
502 const Literal& literal) {
503 if (ShapeUtil::IsTuple(literal.shape())) {
504 std::vector<HloInstruction*> elems;
505 elems.reserve(ShapeUtil::TupleElementCount(literal.shape()));
506 for (int i = 0; i < ShapeUtil::TupleElementCount(literal.shape()); ++i) {
507 elems.push_back(
508 BuildTupleConstant(computation, LiteralView::Create(literal, {i})));
509 }
510 return computation->AddInstruction(HloInstruction::CreateTuple(elems));
511 } else {
512 return computation->AddInstruction(
513 HloInstruction::CreateConstant(literal.CloneToUnique()));
514 }
515}

517Status AlgebraicSimplifierVisitor::HandleConstant(HloInstruction* constant) {
518 // Tuple constants aren't directly supported by any backend. Expand them into
519 // explicit Tuple instructions.
520 if (ShapeUtil::IsTuple(constant->shape())) {
521 return ReplaceInstruction(
522 constant, BuildTupleConstant(computation_, constant->literal()));
523 }
524
525 // If a literal is all the same element replace it with a scalar broadcast.
526 if (ShapeUtil::ElementsIn(constant->shape()) > 1 &&
527 constant->literal().IsAllFirst()) {
528 std::unique_ptr<Literal> unique_scalar =
529 MakeUnique<Literal>(constant->literal().GetFirstScalarLiteral());
530 HloInstruction* scalar = computation_->AddInstruction(
531 HloInstruction::CreateConstant(std::move(unique_scalar)));
532 return ReplaceWithNewInstruction(
533 constant,
534 HloInstruction::CreateBroadcast(constant->shape(), scalar, {}));
535 }
536 return Status::OK();
537}

539Status AlgebraicSimplifierVisitor::HandleSubtract(HloInstruction* sub) {
540 auto lhs = sub->mutable_operand(0);
541 auto rhs = sub->mutable_operand(1);
542 // A - 0 => A
543 VLOG(10) << "trying transform [A - 0 => A]: " << sub->ToString();
544 if (IsAll(rhs, 0) && ReplaceInstructionIfSameShape(sub, lhs)) {
545 return Status::OK();
546 }
547
548 // Canonicalize subtraction of a constant to addition.
549 VLOG(10) << "trying transform [A - Const => A + (-Const)]";
550 if (rhs->IsConstant() && !lhs->IsConstant()) {
551 HloInstruction* negative_const = computation_->AddInstruction(
552 HloInstruction::CreateUnary(rhs->shape(), HloOpcode::kNegate, rhs));
553 return ReplaceWithNewInstruction(
554 sub, HloInstruction::CreateBinary(sub->shape(), HloOpcode::kAdd, lhs,
555 negative_const));
556 }
557
558 return Status::OK();
559}

561Status AlgebraicSimplifierVisitor::HandleDivide(HloInstruction* divide) {
562 auto lhs = divide->mutable_operand(0);
563 auto rhs = divide->mutable_operand(1);
564 // A/1 => A
565 VLOG(10) << "trying transform [A/1 => A]: " << divide->ToString();
566 if (IsAll(rhs, 1) && ReplaceInstructionIfSameShape(divide, lhs)) {
567 return Status::OK();
568 }
569
570 // exp(A)/exp(B) => exp(A-B)
571 if (lhs->opcode() == HloOpcode::kExp && rhs->opcode() == HloOpcode::kExp) {
572 VLOG(10) << "transform [exp(A)/exp(B) => exp(A-B)]: " << divide->ToString();
573 HloInstruction* subtract =
574 computation_->AddInstruction(HloInstruction::CreateBinary(
575 divide->shape(), HloOpcode::kSubtract, lhs->mutable_operand(0),
576 rhs->mutable_operand(0)));
577 return ReplaceWithNewInstruction(
578 divide, HloInstruction::CreateUnary(divide->shape(), HloOpcode::kExp,
579 subtract));
580 }
581
582 // A/exp(B) => A*exp(-B)
583 if (rhs->opcode() == HloOpcode::kExp) {
584 VLOG(10) << "transform [A/exp(B) => A*exp(-B)]: " << divide->ToString();
585 HloInstruction* negate =
586 computation_->AddInstruction(HloInstruction::CreateUnary(
587 divide->shape(), HloOpcode::kNegate, rhs->mutable_operand(0)));
588 HloInstruction* new_exp = computation_->AddInstruction(
589 HloInstruction::CreateUnary(divide->shape(), HloOpcode::kExp, negate));
590 return ReplaceWithNewInstruction(
591 divide, HloInstruction::CreateBinary(
592 divide->shape(), HloOpcode::kMultiply, lhs, new_exp));
593 }
594
595 // A/pow(B,C) => A*pow(B,-C)
596 if (rhs->opcode() == HloOpcode::kPower) {
597 VLOG(10) << "transform [A/pow(B,C) => A*pow(B,-C)]: " << divide->ToString();
598 // The output shape of the created negate operator should be the same as the
599 // input.
600 const Shape& negate_shape = rhs->operand(1)->shape();
601 HloInstruction* negate =
602 computation_->AddInstruction(HloInstruction::CreateUnary(
603 negate_shape, HloOpcode::kNegate, rhs->mutable_operand(1)));
604 // And the power operator should retain the output shape of the old one.
605 const Shape& new_power_shape = rhs->shape();
606 HloInstruction* new_power = computation_->AddInstruction(
607 HloInstruction::CreateBinary(new_power_shape, HloOpcode::kPower,
608 rhs->mutable_operand(0), negate));
609 return ReplaceWithNewInstruction(
610 divide, HloInstruction::CreateBinary(
611 divide->shape(), HloOpcode::kMultiply, lhs, new_power));
612 }
613
614 // Simplifying integral division would produce unexpected results.
615 if (ShapeUtil::ElementIsIntegral(divide->shape())) {
616 return Status::OK();
617 }
618
619 // A / Const => A * (1 / Const)
620 //
621 // (Backends can do this transformation, but generally only if the constant is
622 // a scalar.)
623 if (lhs->opcode() != HloOpcode::kConstant &&
624 rhs->opcode() == HloOpcode::kConstant) {
625 HloInstruction* one =
626 computation_->AddInstruction(HloInstruction::CreateConstant(
627 Literal::One(lhs->shape().element_type()).CloneToUnique()));
628 HloInstruction* inverse =
629 computation_->AddInstruction(HloInstruction::CreateBinary(
630 rhs->shape(), HloOpcode::kDivide, one, rhs));
631 return ReplaceWithNewInstruction(
632 divide, HloInstruction::CreateBinary(
633 divide->shape(), HloOpcode::kMultiply, lhs, inverse));
634 }
635
636 // (A / B) / (C / D) => (A / B)*(D / C) => (A * D) / (B * C)
637 if (lhs->opcode() == HloOpcode::kDivide &&
638 rhs->opcode() == HloOpcode::kDivide) {
639 TF_ASSIGN_OR_RETURN(auto a_times_d, MakeBinaryHlo(HloOpcode::kMultiply,
640 lhs->mutable_operand(0),
641 rhs->mutable_operand(1)));
642 TF_ASSIGN_OR_RETURN(auto b_times_c, MakeBinaryHlo(HloOpcode::kMultiply,
643 lhs->mutable_operand(1),
644 rhs->mutable_operand(0)));
645 TF_ASSIGN_OR_RETURN(auto new_divide, MakeBinaryHlo(HloOpcode::kDivide,
646 a_times_d, b_times_c));
647
648 return ReplaceInstruction(divide, new_divide);
649 }
650
651 // (A / B) / C => A / (B * C)
652 if (lhs->opcode() == HloOpcode::kDivide) {
653 TF_ASSIGN_OR_RETURN(
654 auto b_times_c,
655 MakeBinaryHlo(HloOpcode::kMultiply, lhs->mutable_operand(1), rhs));
656 return ReplaceWithNewInstruction(
657 divide,
658 HloInstruction::CreateBinary(divide->shape(), HloOpcode::kDivide,
659 lhs->mutable_operand(0), b_times_c));
660 }
661
662 // A / (B / C) => (A*C) / B
663 if (rhs->opcode() == HloOpcode::kDivide) {
664 TF_ASSIGN_OR_RETURN(auto a_times_c, MakeBinaryHlo(HloOpcode::kMultiply, lhs,
665 rhs->mutable_operand(1)));
666 return ReplaceWithNewInstruction(
667 divide,
668 HloInstruction::CreateBinary(divide->shape(), HloOpcode::kDivide,
669 a_times_c, rhs->mutable_operand(0)));
670 }
671
672 return Status::OK();
673}

675StatusOr<bool> AlgebraicSimplifierVisitor::HandleDotStrengthReduction(
676 HloInstruction* dot) {
677 HloInstruction* lhs = dot->mutable_operand(0);
678 HloInstruction* rhs = dot->mutable_operand(1);
679 int64 lhs_collapsing_dim =
680 dot->dot_dimension_numbers().lhs_contracting_dimensions(0);
681 if (lhs->IsRank2Transpose()) {
682 lhs = lhs->mutable_operand(0);
683 lhs_collapsing_dim = 1 - lhs_collapsing_dim;
684 }
685 const int64 lhs_kept_dim = 1 - lhs_collapsing_dim;
686
687 int64 rhs_collapsing_dim =
688 dot->dot_dimension_numbers().rhs_contracting_dimensions(0);
689 if (rhs->IsRank2Transpose()) {
690 rhs = rhs->mutable_operand(0);
691 rhs_collapsing_dim = 1 - rhs_collapsing_dim;
692 }
693 const int64 rhs_kept_dim = 1 - rhs_collapsing_dim;
694
695 auto reshape_if_necessary = [&](HloInstruction* hlo) {
696 if (ShapeUtil::SameDimensions(hlo->shape(), dot->shape())) {
697 return hlo;
698 }
699 return computation_->AddInstruction(
700 HloInstruction::CreateReshape(dot->shape(), hlo));
701 };
702
703 auto broadcast_to_dim = [&](HloInstruction* hlo, const Shape& shape,
704 int64 dim) {
705 return computation_->AddInstruction(
706 HloInstruction::CreateBroadcast(shape, hlo, {dim}));
707 };
708
709 auto multiply = [&](HloInstruction* local_lhs, HloInstruction* local_rhs) {
710 return computation_->AddInstruction(HloInstruction::CreateBinary(
711 local_lhs->shape(), HloOpcode::kMultiply, local_lhs, local_rhs));
712 };
713
714 // Strength reduce dot(a[K] , b[K]) =
715 // reshape(result.shape,
716 // reduce_sum(multiply(a, b), {0}))
717 if (ShapeUtil::Rank(rhs->shape()) == 1 &&
718 ShapeUtil::Rank(lhs->shape()) == 1) {
719 TF_RETURN_IF_ERROR(
720 ReplaceInstruction(dot, reshape_if_necessary(AddReduce(
721 multiply(Flatten(lhs), Flatten(rhs)), 0))));
722 return true;
723 }
724
725 if (ShapeUtil::IsEffectiveScalar(rhs->shape()) &&
726 ShapeUtil::IsEffectiveScalar(lhs->shape())) {
727 TF_RETURN_IF_ERROR(ReplaceInstruction(
728 dot, reshape_if_necessary(multiply(Flatten(lhs), Flatten(rhs)))));
729 return true;
730 }
731
732 // Simplify outer product into multiply with implicit broadcasting.
733 //
734 // A dot(a[M, 1], b[1, N]) = multiply(a [M,1], b [1, N])
735 if (ShapeUtil::Rank(rhs->shape()) == 2 &&
736 rhs->shape().dimensions(rhs_collapsing_dim) == 1) {
737 TF_RETURN_IF_ERROR(ReplaceInstruction(
738 dot, multiply(broadcast_to_dim(Flatten(lhs), dot->shape(), 0),
739 broadcast_to_dim(Flatten(rhs), dot->shape(), 1))));
740 return true;
741 }
742
743 // Strength reduce dot(a[1, K], b) =
744 // reshape(result.shape,
745 // reduce_sum(
746 // multiply(broadcast(reshape(a, [K]), {0}), b),
747 // {0})
748 // )
749 // )
750 if (ShapeUtil::Rank(lhs->shape()) == 1 ||
751 (ShapeUtil::Rank(lhs->shape()) == 2 &&
752 lhs->shape().dimensions(lhs_kept_dim) == 1)) {
753 if (ShapeUtil::Rank(rhs->shape()) == 1) {
754 TF_RETURN_IF_ERROR(ReplaceInstruction(
755 dot,
756 reshape_if_necessary(AddReduce(multiply(Flatten(lhs), rhs), 0))));
757 return true;
758 }
759 TF_RETURN_IF_ERROR(ReplaceInstruction(
760 dot, reshape_if_necessary(
761 AddReduce(multiply(broadcast_to_dim(Flatten(lhs), rhs->shape(),
762 rhs_collapsing_dim),
763 rhs),
764 rhs_collapsing_dim))));
765 return true;
766 }
767
768 // Strength reduce dot(a, b[K, 1]) =
769 // reshape(result.shape,
770 // reduce_sum(multiply(a, broadcast(reshape([K],b), {1})), {0})
771 // )
772 if (ShapeUtil::Rank(rhs->shape()) == 1 ||
773 (ShapeUtil::Rank(rhs->shape()) == 2 &&
774 rhs->shape().dimensions(rhs_kept_dim) == 1)) {
775 TF_RETURN_IF_ERROR(ReplaceInstruction(
776 dot, reshape_if_necessary(AddReduce(
777 multiply(lhs, broadcast_to_dim(Flatten(rhs), lhs->shape(),
778 lhs_collapsing_dim)),
779 lhs_collapsing_dim))));
780 return true;
781 }
782 return false;
783}

785StatusOr<HloInstruction*> AlgebraicSimplifierVisitor::OptimizeDotOfConcat(
786 HloInstruction* dot) {
787 const DotDimensionNumbers& dnums = dot->dot_dimension_numbers();
788 if (dnums.lhs_contracting_dimensions_size() != 1 ||
789 dnums.lhs_batch_dimensions_size() != 0) {
790 return nullptr;
791 }
792
793 const int64 lhs_contracting_dim = dnums.lhs_contracting_dimensions(0);
794 const int64 rhs_contracting_dim = dnums.rhs_contracting_dimensions(0);
795 HloInstruction* lhs = dot->mutable_operand(0);
796 HloInstruction* rhs = dot->mutable_operand(1);
797
798 TF_ASSIGN_OR_RETURN(
799 HloInstruction * optimized_lhs_concat,
800 OptimizeDotOfConcatHelper(dot->shape(), lhs, lhs_contracting_dim, rhs,
801 rhs_contracting_dim, /*swapped=*/false));
802 if (optimized_lhs_concat) {
803 return optimized_lhs_concat;
804 }
805
806 return OptimizeDotOfConcatHelper(dot->shape(), rhs, rhs_contracting_dim, lhs,
807 lhs_contracting_dim, /*swapped=*/true);
808}

810StatusOr<HloInstruction*> AlgebraicSimplifierVisitor::OptimizeDotOfConcatHelper(
811 const Shape& dot_shape, HloInstruction* lhs, int64 lhs_contracting_dim,
812 HloInstruction* rhs, int64 rhs_contracting_dim, bool swapped) {
813 bool can_optimize = lhs->opcode() == HloOpcode::kConcatenate &&
814 lhs->concatenate_dimension() == lhs_contracting_dim &&
815 rhs->opcode() == HloOpcode::kConstant;
816 if (!can_optimize) {
817 return nullptr;
818 }
819
820 // We're replacing this:
821 //
822 // +-----+-----+-----+ +-------------------+
823 // | | | | | |
824 // | | | | | R_0 |
825 // | | | | | |
826 // | | | | +-------------------+
827 // | | | | | |
828 // | L_0 | L_1 | L_2 | * | R_1 |
829 // | | | | | |
830 // | | | | +-------------------+
831 // | | | | | |
832 // | | | | | R_2 |
833 // | | | | | |
834 // +-----+-----+-----+ +-------------------+
835 //
836 // with this:
837 //
838 // [Sum over i]
839 //
840 // +-----+ +-------------------+
841 // | | | |
842 // | | * | R_i |
843 // | | | |
844 // | | +-------------------+
845 // | |
846 // | L_i |
847 // | |
848 // | |
849 // | |
850 // | |
851 // | |
852 // +-----+
853 //
854 // where the LHS is a concatenate operation (so we can "split" the LHS tensor
855 // for free) and the RHS is a constant tensor (and thus can be split at
856 // compile time). In the future, we may also want to do this when both the
857 // LHS and the RHS are concatenate operations that line up along the dimension
858 // being contracted over.
859 //
860 // We should be able to generalize this transform to work on a non-constant
861 // RHS when/if we have in-place slices or support input-fusing slices into
862 // Dots.
863
864 // Dimension numbers for the new dot instructions we'll create (L_i * R_i in
865 // the diagram above).
866 DotDimensionNumbers new_dot_dnums;
867 new_dot_dnums.add_lhs_contracting_dimensions(swapped ? rhs_contracting_dim
868 : lhs_contracting_dim);
869 new_dot_dnums.add_rhs_contracting_dimensions(swapped ? lhs_contracting_dim
870 : rhs_contracting_dim);
871
872 // Here we use the MKN notation, where the contracted dimension has K
873 // elements and the two non-contracted dimensions have M and N elements.
874 HloInstruction* add_result = nullptr;
875 int64 rhs_contracting_dim_offset = 0;
876 int64 n = rhs->shape().dimensions(1 - rhs_contracting_dim);
877 for (HloInstruction* concat_op : lhs->operands()) {
878 int64 sub_k = concat_op->shape().dimensions(lhs_contracting_dim);
879 Shape rhs_slice_shape(rhs->shape());
880 rhs_slice_shape.set_dimensions(rhs_contracting_dim, sub_k);
881
882 std::array<int64, 2> start_indices;
883 start_indices[rhs_contracting_dim] = rhs_contracting_dim_offset;
884 start_indices[1 - rhs_contracting_dim] = 0;
885
886 std::array<int64, 2> limit_indices;
887 limit_indices[rhs_contracting_dim] = rhs_contracting_dim_offset + sub_k;
888 limit_indices[1 - rhs_contracting_dim] = n;
889
890 HloInstruction* rhs_slice =
891 computation_->AddInstruction(HloInstruction::CreateSlice(
892 rhs_slice_shape, rhs, /*start_indices=*/start_indices,
893 /*limit_indices=*/limit_indices, /*strides=*/{1, 1}));
894
895 // TODO(b/69062148): We can get rid of `swapped` once all backends support
896 // "non-canonical" contraction dimensions (that contracts dimension 1 of the
897 // LHS with dimension 0 of the RHS). But for now we keep the same
898 // contraction dimensions as the incoming dot operation to ensure the new
899 // dot operations can be lowered.
900 HloInstruction *new_dot_lhs, *new_dot_rhs;
901 if (swapped) {
902 new_dot_lhs = rhs_slice;
903 new_dot_rhs = concat_op;
904 } else {
905 new_dot_lhs = concat_op;
906 new_dot_rhs = rhs_slice;
907 }
908
909 auto* new_dot = computation_->AddInstruction(HloInstruction::CreateDot(
910 dot_shape, new_dot_lhs, new_dot_rhs, new_dot_dnums));
911
912 if (add_result) {
913 add_result = computation_->AddInstruction(HloInstruction::CreateBinary(
914 dot_shape, HloOpcode::kAdd, add_result, new_dot));
915 } else {
916 add_result = new_dot;
917 }
918
919 rhs_contracting_dim_offset += sub_k;
920 }
921
922 return add_result;
923}

925Status AlgebraicSimplifierVisitor::HandleDot(HloInstruction* dot) {
926 auto lhs = dot->mutable_operand(0);
927 auto rhs = dot->mutable_operand(1);
928
929 // Only optimize F32 dot operations where the dot, rhs and lhs are rank 2 or
930 // below.
931 if (dot->shape().element_type() != F32 || ShapeUtil::Rank(lhs->shape()) > 2 ||
932 ShapeUtil::Rank(rhs->shape()) > 2 || ShapeUtil::Rank(dot->shape()) > 2) {
933 return Status::OK();
934 }
935
936 // Replace a zero element dot with a broadcast of the constant 0.
937 if (ShapeUtil::HasZeroElements(dot->shape()) ||
938 ShapeUtil::HasZeroElements(lhs->shape()) ||
939 ShapeUtil::HasZeroElements(rhs->shape())) {
940 auto zero = computation_->AddInstruction(
941 HloInstruction::CreateConstant(Literal::CreateR0(0.0f)));
942 return ReplaceWithNewInstruction(
943 dot, HloInstruction::CreateBroadcast(dot->shape(), zero, {}));
944 }
945
946 TF_ASSIGN_OR_RETURN(HloInstruction * dot_of_concat_optimized,
947 OptimizeDotOfConcat(dot));
948 if (dot_of_concat_optimized) {
949 VLOG(10) << "Replaced dot(concat(...), constant) with add(dot(..., "
950 "constant)...)";
951 return ReplaceInstruction(dot, dot_of_concat_optimized);
952 }
953
954 if (enable_dot_strength_reduction_ && !is_layout_sensitive_) {
955 TF_ASSIGN_OR_RETURN(bool did_strength_reduction,
956 HandleDotStrengthReduction(dot));
957 if (did_strength_reduction) {
958 return Status::OK();
959 }
960 }
961
962 // Simplify dot(transpose(a), transpose(b)) to transpose(dot(b,a)).
963 if (lhs->IsRank2Transpose() && rhs->IsRank2Transpose()) {
964 DotDimensionNumbers dot_dimension_numbers;
965 dot_dimension_numbers.add_lhs_contracting_dimensions(1);
966 dot_dimension_numbers.add_rhs_contracting_dimensions(0);
967 auto new_dot = computation_->AddInstruction(HloInstruction::CreateDot(
968 ShapeUtil::PermuteDimensions({1, 0}, dot->shape()),
969 rhs->mutable_operand(0), lhs->mutable_operand(0),
970 dot_dimension_numbers));
971 return ReplaceWithNewInstruction(
972 dot, HloInstruction::CreateTranspose(dot->shape(), new_dot, {1, 0}));
973 }
974
975 return Status::OK();
976}

978Status AlgebraicSimplifierVisitor::HandleMultiply(HloInstruction* multiply) {
979 auto lhs = multiply->mutable_operand(0);
980 auto rhs = multiply->mutable_operand(1);
981 // A*1 => A
982 VLOG(10) << "trying transform [A*1 => A]: " << multiply->ToString();
983 if (IsAll(rhs, 1) && ReplaceInstructionIfSameShape(multiply, lhs)) {
984 return Status::OK();
985 }
986 // 1*A => A
987 VLOG(10) << "trying transform [1*A => A]: " << multiply->ToString();
988 if (IsAll(lhs, 1) && ReplaceInstructionIfSameShape(multiply, rhs)) {
989 return Status::OK();
990 }
991
992 // exp(A) * exp(B) => exp(A+B)
993 if (lhs->opcode() == HloOpcode::kExp && rhs->opcode() == HloOpcode::kExp) {
994 auto add = computation_->AddInstruction(HloInstruction::CreateBinary(
995 multiply->shape(), HloOpcode::kAdd, lhs->mutable_operand(0),
996 rhs->mutable_operand(0)));
997 return ReplaceWithNewInstruction(
998 multiply,
999 HloInstruction::CreateUnary(multiply->shape(), HloOpcode::kExp, add));
1000 }
1001 return Status::OK();
1002}

1045// Return whether the given reshape instruction leaves the dimensions at the
1046// given input indices unmodified, and returns their output indices.
1047//
1048// Example:
1049// input_dim_indices = {2, 3}
1050// input shape = T[a, b, x, y, cd]
1051// output shape = T[ab, x, 1, y, c, d]
1052// return value = {1, 3}
1053//
1054// Precondition: input_dim_indices is sorted.
1055std::pair<bool, std::vector<int64>> ReshapeLeavesDimensionsUnmodified(
1056 const HloInstruction* hlo,
1057 tensorflow::gtl::ArraySlice<int64> input_dim_indices) {
1058 CHECK_EQ(HloOpcode::kReshape, hlo->opcode());
1059 CHECK(std::is_sorted(input_dim_indices.begin(), input_dim_indices.end()));
1060
1061 std::vector<int64> output_dim_indices;
1062 std::vector<std::pair<int64, int64>> unmodified_dims =
1063 ShapeUtil::DimensionsUnmodifiedByReshape(hlo->operand(0)->shape(),
1064 hlo->shape());
1065 size_t i = 0; // index to unmodified_dims
1066 for (int64 input_dim_index : input_dim_indices) {
1067 // Search unmodified_dims for input_dim_index. We can search from the last
1068 // matching position because input_dim_indices is guaranteed to be sorted.
1069 while (i < unmodified_dims.size() &&
1070 unmodified_dims[i].first < input_dim_index) {
1071 ++i;
1072 }
1073 if (i >= unmodified_dims.size() ||
1074 unmodified_dims[i].first != input_dim_index) {
1075 return std::make_pair(false, std::vector<int64>());
1076 }
1077 output_dim_indices.push_back(unmodified_dims[i].second);
1078 }
1079 return std::make_pair(true, output_dim_indices);
1080}

1122Status AlgebraicSimplifierVisitor::HandleBroadcast(HloInstruction* broadcast) {
1123 auto operand = broadcast->mutable_operand(0);
1124 auto dims = broadcast->dimensions();
1125 // A degenerate broadcast of a reshape that does not change the number of
1126 // elements can be replaced by a reshape.
1127 if (std::is_sorted(dims.begin(), dims.end()) &&
1128 ShapeUtil::ElementsIn(broadcast->shape()) ==
1129 ShapeUtil::ElementsIn(operand->shape())) {
1130 VLOG(10) << "transform broadcast(X) -> reshape(X) where "
1131 "n(broadcast(X)) == n(X)";
1132 return ReplaceWithNewInstruction(
1133 broadcast, HloInstruction::CreateReshape(broadcast->shape(), operand));
1134 }
1135
1136 // A degenerate broadcast that has the same input and output rank can be
1137 // converted into a transpose.
1138 if (ShapeUtil::Rank(broadcast->shape()) ==
1139 ShapeUtil::Rank(operand->shape()) &&
1140 ShapeUtil::ElementsIn(broadcast->shape()) ==
1141 ShapeUtil::ElementsIn(operand->shape())) {
1142 VLOG(10) << "transform broadcast(X) -> transpose(X) where "
1143 "n(broadcast(X)) == n(X)";
1144 return ReplaceWithNewInstruction(
1145 broadcast,
1146 HloInstruction::CreateTranspose(broadcast->shape(), operand, dims));
1147 }
1148
1149 // A broadcast of a reshape which merely inserts 1-sized dimensions can
1150 // elide its operand.
1151 {
1152 bool merely_inserts_or_deletes_1_sized_dimensions;
1153 std::vector<int64> inserted_indices, deleted_indices;
1154 std::tie(merely_inserts_or_deletes_1_sized_dimensions, deleted_indices,
1155 inserted_indices) =
1156 operand->ReshapeMerelyInsertsOrDeletes1SizedDimensions();
1157 if (merely_inserts_or_deletes_1_sized_dimensions &&
1158 deleted_indices.empty()) {
1159 std::reverse(inserted_indices.begin(), inserted_indices.end());
1160 for (auto inserted_index : inserted_indices) {
1161 dims.erase(dims.begin() + inserted_index);
1162 }
1163 return ReplaceWithNewInstruction(
1164 broadcast,
1165 HloInstruction::CreateBroadcast(broadcast->shape(),
1166 operand->mutable_operand(0), dims));
1167 }
1168 }

1170 // A Broadcast that feeds a unary element-wise operation can sink the
1171 // broadcast after the unary element-wise operation.
1172 TF_ASSIGN_OR_RETURN(
1173 bool sink_succeeded,
1174 TryToSinkReshapeOrBroadcastAfterOpWithUniqueNonScalarOperand(broadcast));
1175 changed_ |= sink_succeeded;
1176 if (sink_succeeded) {
1177 return Status::OK();
1178 }
1179
1180 // A scalar broadcast feeding an instruction which only permutes (reshape,
1181 // transpose, sort, reverse) or selects a subset of operand elements (slice,
1182 // dynamic slice) can be replaced with a broadcast directly to the output
1183 // shape of the instruction.
1184 if (ShapeUtil::IsScalar(operand->shape())) {
1185 for (HloInstruction* user : broadcast->users()) {
1186 // Skip if the broadcast user has no uses itself.
1187 if (user->user_count() == 0 && user != computation_->root_instruction()) {
1188 continue;
1189 }
1190 if (OutputIsPermutationOfOperandElements(user, broadcast) ||
1191 OutputIsSubsetOfOperandElements(user, broadcast)) {
1192 VLOG(10) << "transform permuting/subset of a scalar broadcast into "
1193 << "a single broadcast";
1194 HloInstruction* new_broadcast = computation_->AddInstruction(
1195 HloInstruction::CreateBroadcast(user->shape(), operand, {}));
1196 // Use HloInstruction::ReplaceAllUsesWith instead of
1197 // HloComputation::ReplaceWithNewInstruction because we are replacing an
1198 // instruction other than the visited instruction.
1199 changed_ = true;
1200 return user->ReplaceAllUsesWith(new_broadcast);
1201 }
1202 }
1203 return Status::OK();
1204 }
1205
1206 // Merge two consecutive broadcasts into a single one.
1207 if (operand->opcode() == HloOpcode::kBroadcast) {
1208 std::vector<int64> new_dimensions;
1209 for (auto dim : operand->dimensions()) {
1210 new_dimensions.push_back(dims[dim]);
1211 }
1212 return ReplaceWithNewInstruction(
1213 broadcast,
1214 HloInstruction::CreateBroadcast(
1215 broadcast->shape(), operand->mutable_operand(0), new_dimensions));
1216 }
1217 return Status::OK();
1218}

1262Status AlgebraicSimplifierVisitor::HandlePad(HloInstruction* pad) {
1263 if (ShapeUtil::HasZeroElements(pad->operand(0)->shape())) {
1264 return ReplaceWithNewInstruction(
1265 pad, HloInstruction::CreateBroadcast(pad->shape(),
1266 pad->mutable_operand(1), {}));
1267 }
1268 // Eliminate nop pads (padding all zero), and replace a pad with negative
1269 // padding with a pad with non-negative padding followed by a slice.
1270 bool all_zero = true;
1271 bool has_negative = false;
1272 for (auto& padding_dimension : pad->padding_config().dimensions()) {
1273 if (padding_dimension.edge_padding_low() < 0 ||
1274 padding_dimension.edge_padding_high() < 0) {
1275 has_negative = true;
1276 }
1277 if (padding_dimension.edge_padding_low() != 0 ||
1278 padding_dimension.edge_padding_high() != 0) {
1279 all_zero = false;
1280 }
1281 }
1282
1283 if (all_zero) {
1284 ReplaceInstructionIfSameShape(pad, pad->mutable_operand(0));
1285 return Status::OK();
1286 }
1287
1288 if (has_negative) {
1289 // Pad has negative padding. Replace with a pad with the non-negative
1290 // padding followed by a slice which effectively performs the negative
1291 // padding.
1292 // TODO(b/34628603): Add support for negative padding in the backends, or
1293 // change kPad semantics to disallow negative padding and use slice
1294 // instead.
1295
1296 // First construct the padding config with non-negative entries and the
1297 // compute the shape of this new pad instruction.
1298 PaddingConfig nonzero_padding = pad->padding_config();
1299 for (int i = 0; i < pad->padding_config().dimensions_size(); ++i) {
1300 PaddingConfig::PaddingConfigDimension* padding_dimension =
1301 nonzero_padding.mutable_dimensions(i);
1302 // Set negative padding to zero.
1303 if (padding_dimension->edge_padding_low() < 0) {
1304 padding_dimension->set_edge_padding_low(0);
1305 }
1306 if (padding_dimension->edge_padding_high() < 0) {
1307 padding_dimension->set_edge_padding_high(0);
1308 }
1309 }
1310
1311 TF_ASSIGN_OR_RETURN(HloInstruction * nonzero_pad,
1312 MakePadHlo(pad->mutable_operand(0),
1313 pad->mutable_operand(1), nonzero_padding));
1314 // Copy the layout from the original pad instructions. The new pad and the
1315 // slice instruction should all have the same layout.
1316 TF_RETURN_IF_ERROR(LayoutUtil::CopyLayoutBetweenShapes(
1317 pad->shape(), nonzero_pad->mutable_shape()));
1318
1319 // Second, construct the slice instruction to perform the negative padding.
1320 std::vector<int64> start_indices;
1321 std::vector<int64> end_indices;
1322 std::vector<int64> strides;
1323 for (int64 i = 0; i < pad->padding_config().dimensions_size(); ++i) {
1324 const PaddingConfig::PaddingConfigDimension& padding_dimension =
1325 pad->padding_config().dimensions(i);
1326 int64 start = 0;
1327 if (padding_dimension.edge_padding_low() < 0) {
1328 start = -1 * padding_dimension.edge_padding_low();
1329 }
1330 int64 end = nonzero_pad->shape().dimensions(i);
1331 if (padding_dimension.edge_padding_high() < 0) {
1332 end += padding_dimension.edge_padding_high();
1333 }
1334 start_indices.push_back(start);
1335 end_indices.push_back(end);
1336 strides.push_back(1);
1337 }
1338
1339 TF_ASSIGN_OR_RETURN(
1340 HloInstruction * slice,
1341 MakeSliceHlo(nonzero_pad, start_indices, end_indices, strides));
1342
1343 // Verify that the slice shape matches the pad shape.
1344 TF_RET_CHECK(ShapeUtil::Compatible(slice->shape(), pad->shape()));
1345
1346 return ReplaceInstruction(pad, slice);
1347 }
1348
1349 return Status::OK();
1350}

1352Status AlgebraicSimplifierVisitor::HandlePower(HloInstruction* power) {
1353 VLOG(10) << "trying transform [pow(A, 0) => 1]: " << power->ToString();
1354 auto lhs = power->mutable_operand(0);
1355 auto rhs = power->mutable_operand(1);
1356 if (IsAll(rhs, 0)) {
1357 auto one = HloInstruction::CreateConstant(
1358 Literal::One(power->shape().element_type()).CloneToUnique());
1359 std::unique_ptr<HloInstruction> ones;
1360 if (ShapeUtil::IsScalar(power->shape())) {
1361 ones = std::move(one);
1362 } else {
1363 ones = HloInstruction::CreateBroadcast(
1364 power->shape(), computation_->AddInstruction(std::move(one)), {});
1365 }
1366 return ReplaceWithNewInstruction(power, std::move(ones));
1367 }
1368
1369 VLOG(10) << "trying transform [pow(A, 1) => A]: " << power->ToString();
1370 if (IsAll(rhs, 1) && ReplaceInstructionIfSameShape(power, lhs)) {
1371 return Status::OK();
1372 }
1373
1374 // pow(exp(A),B) => exp(A*B)
1375 if (lhs->opcode() == HloOpcode::kExp) {
1376 auto a_times_b = computation_->AddInstruction(HloInstruction::CreateBinary(
1377 power->shape(), HloOpcode::kMultiply, lhs->operands()[0], rhs));
1378 return ReplaceWithNewInstruction(
1379 power, HloInstruction::CreateUnary(power->shape(), HloOpcode::kExp,
1380 a_times_b));
1381 }
1382 VLOG(10) << "trying transform [pow(A, 2) => A*A]: " << power->ToString();
1383 if (IsAll(rhs, 2)) {
1384 return ReplaceWithNewInstruction(
1385 power, HloInstruction::CreateBinary(power->shape(),
1386 HloOpcode::kMultiply, lhs, lhs));
1387 }
1388
1389 VLOG(10) << "trying transform [pow(A, -1) => 1/A]: " << power->ToString();
1390 if (IsAll(rhs, -1)) {
1391 auto* one = computation_->AddInstruction(HloInstruction::CreateConstant(
1392 Literal::One(rhs->shape().element_type()).CloneToUnique()));
1393
1394 // Explicitly broadcast scalar 1 to the output shape, to avoid implicit
1395 // broadcast in divide HLO as we are trying to eliminate implicit
1396 // broadcasting at HLO level.
1397 auto* broadcast_one = computation_->AddInstruction(
1398 HloInstruction::CreateBroadcast(power->shape(), one, {}));
1399 return ReplaceWithNewInstruction(
1400 power, HloInstruction::CreateBinary(power->shape(), HloOpcode::kDivide,
1401 broadcast_one, lhs));
1402 }
1403
1404 VLOG(10) << "trying transform [pow(pow(A, X), Y) => pow(A, X*Y)]: "
1405 << power->ToString();
1406
1407 // Don't perform this optimization if either of the exponents is complex; this
1408 // identity is true only for real-valued exponents. In addition, we cowardly
1409 // refuse to do this transformation if the two expontents have different
1410 // element types.
1411 if (lhs->opcode() == HloOpcode::kPower &&
1412 !ShapeUtil::ElementIsComplex(lhs->operand(1)->shape()) &&
1413 !ShapeUtil::ElementIsComplex(rhs->shape()) &&
1414 ShapeUtil::SameElementType(lhs->operand(1)->shape(), rhs->shape())) {
1415 auto exponent_product =
1416 computation_->AddInstruction(HloInstruction::CreateBinary(
1417 rhs->shape(), HloOpcode::kMultiply, lhs->mutable_operand(1), rhs));
1418 return ReplaceWithNewInstruction(
1419 power, HloInstruction::CreateBinary(power->shape(), HloOpcode::kPower,
1420 lhs->mutable_operand(0),
1421 exponent_product));
1422 }
1423
1424 return Status::OK();
1425}

1427// TODO(b/74536353): do this simplification for BroadcastDimOne as well.
1428StatusOr<bool> AlgebraicSimplifierVisitor::
1429 TryToSinkReshapeOrBroadcastAfterOpWithUniqueNonScalarOperand(
1430 HloInstruction* reshape_or_broadcast) {
1431 bool changed = false;
1432 if (ShapeUtil::IsScalar(reshape_or_broadcast->shape())) {
1433 return false;
1434 }
1435 HloInstruction* operand = reshape_or_broadcast->mutable_operand(0);
1436 for (HloInstruction* user : reshape_or_broadcast->users()) {
1437 if (user->user_count() == 0 && user != computation_->root_instruction()) {
1438 continue;
1439 }
1440 // Do not move reshapes or broadcasts past copies since the shape the copy
1441 // will operate on will change.
1442 if (user->opcode() == HloOpcode::kCopy) {
1443 continue;
1444 }
1445 // Do not change the shape of fusion nodes in case there a multiple shapes
1446 // inside the fusion node already.
1447 if (user->opcode() == HloOpcode::kFusion) {
1448 continue;
1449 }
1450 if (!user->IsElementwise()) {
1451 continue;
1452 }
1453
1454 int64 reshape_or_broadcast_operand_index = -1;
1455 // Find the unique non-scalar operand or continue if there isn't one.
1456 int64 scalar_count = 0;
1457 for (int64 i = 0; i < user->operand_count(); ++i) {
1458 if (ShapeUtil::IsScalar(user->operand(i)->shape())) {
1459 ++scalar_count;
1460 } else {
1461 reshape_or_broadcast_operand_index = i;
1462 }
1463 }
1464 if (scalar_count != user->operand_count() - 1) {
1465 continue;
1466 }
1467 VLOG(4) << "Sinking reshape or broadcast after user:";
1468 VLOG(4) << " old reshape/broadcast: " << reshape_or_broadcast->ToString();
1469 VLOG(4) << " old user: " << user->ToString();
1470 CHECK_EQ(user->operand(reshape_or_broadcast_operand_index),
1471 reshape_or_broadcast);
1472 auto new_user_operands = user->operands();
1473 new_user_operands[reshape_or_broadcast_operand_index] = operand;
1474 auto new_user = computation_->AddInstruction(user->CloneWithNewOperands(
1475 ShapeUtil::MakeShapeWithLayout(
1476 user->shape().element_type(),
1477 AsInt64Slice(operand->shape().dimensions()),
1478 LayoutUtil::MinorToMajor(operand->shape())),
1479 new_user_operands));
1480 VLOG(4) << " new user: " << new_user->ToString();
1481 HloInstruction* new_reshape_or_broadcast = nullptr;
1482 if (reshape_or_broadcast->opcode() == HloOpcode::kReshape) {
1483 new_reshape_or_broadcast =
1484 computation_->AddInstruction(HloInstruction::CreateReshape(
1485 ShapeUtil::MakeShapeWithLayout(
1486 user->shape().element_type(),
1487 AsInt64Slice(reshape_or_broadcast->shape().dimensions()),
1488 LayoutUtil::MinorToMajor(reshape_or_broadcast->shape())),
1489 new_user));
1490 } else {
1491 TF_RET_CHECK(reshape_or_broadcast->opcode() == HloOpcode::kBroadcast);
1492 new_reshape_or_broadcast =
1493 computation_->AddInstruction(HloInstruction::CreateBroadcast(
1494 ShapeUtil::MakeShapeWithLayout(
1495 user->shape().element_type(),
1496 AsInt64Slice(reshape_or_broadcast->shape().dimensions()),
1497 LayoutUtil::MinorToMajor(reshape_or_broadcast->shape())),
1498 new_user, reshape_or_broadcast->dimensions()));
1499 }
1500 VLOG(4) << " new reshape/broadcast: "
1501 << new_reshape_or_broadcast->ToString();
1502 TF_RETURN_IF_ERROR(user->ReplaceAllUsesWith(new_reshape_or_broadcast));
1503 changed = true;
1504 }
1505 return changed;
1506}

1508Status AlgebraicSimplifierVisitor::HandleReshape(HloInstruction* reshape) {
1509 auto operand = reshape->mutable_operand(0);
1510
1511 // Reshape directly to empty constant if the shape contains zero-element
1512 // dimension.
1513 if (ShapeUtil::HasZeroElements(reshape->shape())) {
1514 auto empty_constant = HloInstruction::CreateConstant(
1515 Literal::CreateFromShape(reshape->shape()));
1516
1517 return ReplaceWithNewInstruction(reshape, std::move(empty_constant));
1518 }
1519
1520 // Delete no-op reshapes, i.e. where shape = operand shape.
1521 if (SameShape(reshape, operand)) {
1522 VLOG(10) << "deleting no-op reshape";
1523 return ReplaceInstruction(reshape, operand);
1524 }
1525
1526 // Merge reshapes.
1527 if (HloOpcode::kReshape == operand->opcode()) {
1528 return ReplaceWithNewInstruction(
1529 reshape, HloInstruction::CreateReshape(reshape->shape(),
1530 operand->mutable_operand(0)));
1531 }
1532
1533 if (HloOpcode::kBroadcast == reshape->operand(0)->opcode()) {
1534 auto opt_dims = ReshapeLeavesDimensionsUnmodified(
1535 reshape, reshape->operand(0)->dimensions());
1536 if (opt_dims.first) {
1537 return ReplaceWithNewInstruction(
1538 reshape,
1539 HloInstruction::CreateBroadcast(
1540 reshape->shape(), reshape->mutable_operand(0)->mutable_operand(0),
1541 opt_dims.second));
1542 }
1543 }
1544
1545 // A Reshape that feeds a unary element-wise operation can sink the
1546 // reshape after the unary element-wise operation.
1547 TF_ASSIGN_OR_RETURN(
1548 bool sink_succeeded,
1549 TryToSinkReshapeOrBroadcastAfterOpWithUniqueNonScalarOperand(reshape));
1550 changed_ |= sink_succeeded;
1551 if (sink_succeeded) {
1552 return Status::OK();
1553 }
1554
1555 // Make this a bitcast if possible.
1556 if (is_layout_sensitive_ &&
1557 ReshapeIsBitcast(reshape, valid_bitcast_callback_)) {
1558 ReplaceWithBitcast(reshape);
1559 return Status::OK();
1560 }
1561
1562 return Status::OK();
1563}

1601Status AlgebraicSimplifierVisitor::HandleDynamicUpdateSlice(
1602 HloInstruction* dynamic_update_slice) {
1603 auto update = dynamic_update_slice->mutable_operand(1);
1604 auto start_indices = dynamic_update_slice->operand(2);
1605 // DynamicUpdateSlice on a scalar just passes through the update argument.
1606 if (ShapeUtil::IsScalar(dynamic_update_slice->shape())) {
1607 return ReplaceInstruction(dynamic_update_slice, update);
1608 }
1609
1610 // DynamicUpdateSlice where operand and update have the same size and
1611 // start_indices are all zero is simply equal to update.
1612 //
1613 // (We require start_indices to be all zero because we want this optimization
1614 // not to affect the visible behavior of this op even when the indices are out
1615 // of range. Currently dynamic-update-slice wraps out-of-range indices, so
1616 // we can only remove the op if its indices never wrap.)
1617 if (IsAll(start_indices, 0) && SameShape(dynamic_update_slice, update)) {
1618 return ReplaceInstruction(dynamic_update_slice, update);
1619 }
1620
1621 // If any dimension of update is 0, elide the DynamicUpdateSlice. This
1622 // optimization becomes invalid should we later prefer to warn about out of
1623 // bound indices.
1624 if (ShapeUtil::HasZeroElements(update->shape())) {
1625 return ReplaceInstruction(dynamic_update_slice,
1626 dynamic_update_slice->mutable_operand(0));
1627 }
1628 return Status::OK();

1631Status AlgebraicSimplifierVisitor::HandleReduce(HloInstruction* reduce) {
1632 auto arg = reduce->mutable_operand(0);
1633 auto init_value = reduce->mutable_operand(1);
1634 tensorflow::gtl::ArraySlice<int64> dimensions(reduce->dimensions());
1635 HloComputation* function = reduce->to_apply();
1636 if (ShapeUtil::HasZeroElements(arg->shape()) ||
1637 ShapeUtil::HasZeroElements(reduce->shape())) {
1638 return ReplaceWithNewInstruction(
1639 reduce,
1640 HloInstruction::CreateBroadcast(reduce->shape(), init_value, {}));
1641 }
1642
1643 // A Transpose feeding a reduce can simply permute the reduction dimensions
1644 // field if the output of the reduce is a vector or scalar. Higher ranked
1645 // result may require a transpose of the output.
1646 if (ShapeUtil::Rank(reduce->shape()) <= 1 &&
1647 arg->opcode() == HloOpcode::kTranspose) {
1648 auto transpose_dimensions = arg->dimensions();
1649 std::vector<int64> new_reduce_dimensions;
1650 for (auto dim : dimensions) {
1651 new_reduce_dimensions.push_back(transpose_dimensions[dim]);
1652 }
1653 return ReplaceWithNewInstruction(
1654 reduce, HloInstruction::CreateReduce(
1655 reduce->shape(), arg->mutable_operand(0), init_value,
1656 new_reduce_dimensions, function));
1657 }
1658
1659 // A reshape that collapses multiple dimensions into a dimension being
1660 // reduced can just reduce all of those dimensions instead of doing a
1661 // collapsing reshape before a reduction.
1662 if (arg->opcode() == HloOpcode::kReshape) {
1663 std::vector<std::pair<int64, int64>> unmodified_dims =
1664 ShapeUtil::DimensionsUnmodifiedByReshape(arg->operand(0)->shape(),
1665 arg->shape());
1666 std::vector<bool> arg_dim_in_output(ShapeUtil::Rank(arg->shape()), true);
1667 std::vector<bool> arg_dim_unmodified(ShapeUtil::Rank(arg->shape()), false);
1668 for (auto dim : dimensions) {
1669 arg_dim_in_output[dim] = false;
1670 }
1671 for (auto dim_pair : unmodified_dims) {
1672 arg_dim_unmodified[dim_pair.second] = true;
1673 }
1674 // The goal is to verify that all dimensions that are not removed in the
1675 // reduce are unmodified by the reshape. For example:
1676 // reduce(reshape([A,B*C], a[A,B,C]),[1]) = reduce(a[A, B, C], [1, 2])
1677 bool can_move_reshape_into_reduce = true;
1678 for (int64 i = 0; i < arg_dim_in_output.size(); ++i) {
1679 if (arg_dim_in_output[i] && !arg_dim_unmodified[i]) {
1680 can_move_reshape_into_reduce = false;
1681 }
1682 }
1683 if (can_move_reshape_into_reduce) {
1684 changed_ = true;
1685 std::unordered_set<int64> dimensions_not_to_reduce;
1686 for (auto dim_pair : unmodified_dims) {
1687 if (arg_dim_in_output[dim_pair.second]) {
1688 dimensions_not_to_reduce.insert(dim_pair.first);
1689 }
1690 }
1691 std::vector<int64> new_reduce_dimensions;
1692 for (int64 i = 0; i < ShapeUtil::Rank(arg->operand(0)->shape()); ++i) {
1693 if (dimensions_not_to_reduce.count(i) == 0) {
1694 new_reduce_dimensions.push_back(i);
1695 }
1696 }
1697 return ReplaceWithNewInstruction(
1698 reduce, HloInstruction::CreateReduce(
1699 reduce->shape(), arg->mutable_operand(0), init_value,
1700 new_reduce_dimensions, function));
1701 }
1702 }
1703 if (ShapeUtil::ElementsIn(reduce->shape()) ==
1704 ShapeUtil::ElementsIn(arg->shape()) ||
1705 ShapeUtil::HasZeroElements(arg->shape())) {
1706 auto reshape = computation_->AddInstruction(
1707 HloInstruction::CreateReshape(reduce->shape(), arg));
1708 return ReplaceWithNewInstruction(
1709 reduce, HloInstruction::CreateMap(reduce->shape(),
1710 {reshape, init_value}, function));
1711 }
1712 return Status::OK();
1713}

1715Status AlgebraicSimplifierVisitor::HandleReduceWindow(
1716 HloInstruction* reduce_window) {
1717 if (ShapeUtil::HasZeroElements(reduce_window->operand(0)->shape())) {
1718 return ReplaceWithNewInstruction(
1719 reduce_window,
1720 HloInstruction::CreateBroadcast(reduce_window->shape(),
1721 reduce_window->mutable_operand(1), {}));
1722 }
1723 auto operand = reduce_window->mutable_operand(0);
1724 const Window& window = reduce_window->window();
1725 auto function = reduce_window->to_apply();
1726 if (ShapeUtil::IsScalar(operand->shape())) {
1727 TF_RET_CHECK(ShapeUtil::IsScalar(reduce_window->shape()));
1728 return ReplaceWithNewInstruction(
1729 reduce_window,
1730 HloInstruction::CreateMap(reduce_window->shape(),
1731 {operand, reduce_window->mutable_operand(1)},
1732 function));
1733 }
1734
1735 // This optimization folds a pad op into reduce_window.
1736 HloInstruction* pad;
1737 const HloInstruction* convert = nullptr;
1738 if (operand->opcode() == HloOpcode::kPad) {
1739 pad = operand;
1740 } else if (operand->opcode() == HloOpcode::kConvert &&
1741 operand->operand(0)->opcode() == HloOpcode::kPad) {
1742 convert = operand;
1743 pad = operand->mutable_operand(0);
1744 } else {
1745 VLOG(10) << "Not folding pad into reduce-window as there is no pad.";
1746 return Status::OK();
1747 }
1748
1749 VLOG(10) << "Considering folding Pad: " << pad->ToString()
1750 << "\ninto reduce-window: " << reduce_window->ToString()
1751 << (convert != nullptr ? tensorflow::strings::StrCat(
1752 "\nvia convert: ", convert->ToString())
1753 : "");
1754
1755 // Do not fold interior padding into ReduceWindow since the backends do not
1756 // support it.
1757 const PaddingConfig& pad_config = pad->padding_config();
1758 if (HasInteriorPadding(pad_config)) {
1759 VLOG(10) << "Not folding pad into reduce-window due to interior padding.";
1760 return Status::OK();
1761 }
1762
1763 // If reduce_window already has padding, the pad value of the pad op and the
1764 // init value of reduce_window must match to allow folding the pad.
1765 const HloInstruction* pad_value = pad->operand(1);
1766 const HloInstruction* reduce_init_value = reduce_window->operand(1);
1767 if (pad_value != reduce_init_value) {
1768 auto literals_are_equivalent = [&] {
1769 auto& pad_literal = pad_value->literal();
1770 auto& reduce_init_literal = reduce_init_value->literal();
1771 if (pad_literal == reduce_init_literal) {
1772 return true;
1773 }
1774 auto converted_pad_literal = pad_literal.ConvertToShape(
1775 reduce_init_value->shape(), /*round_f32_to_bf16=*/true);
1776 if (!converted_pad_literal.ok()) {
1777 return false;
1778 }
1779 return *converted_pad_literal.ValueOrDie() == reduce_init_literal;
1780 };
1781 // The pad value is usually a constant, so we handle that case and do not
1782 // try to get more fancy about proving equivalence in cases beyond that.
1783 if (pad_value->opcode() != HloOpcode::kConstant ||
1784 reduce_init_value->opcode() != HloOpcode::kConstant ||
1785 !literals_are_equivalent()) {
1786 VLOG(10) << "Not folding pad into reduce-window due to different pad "
1787 "values.";
1788 return Status::OK();
1789 }
1790 }
1791
1792 // If the pad puts a single non-identity value in each window that we're
1793 // reducing, then this is a broadcast.
1794 HloInstruction* pad_operand = pad->mutable_operand(0);
1795 auto is_effective_broadcast = [&] {
1796 if (window_util::HasStride(window)) {
1797 VLOG(10) << "Window has stride.";
1798 return false;
1799 }
1800 if (!window_util::HasSymmetricPadding(pad_config)) {
1801 VLOG(10) << "Window has uneven padding.";
1802 return false;
1803 }
1804 for (int64 i = 0; i < pad_config.dimensions_size(); ++i) {
1805 const auto& pad_dimension = pad_config.dimensions(i);
1806 if ((pad_dimension.edge_padding_low() != 0 ||
1807 pad_dimension.edge_padding_high() != 0) &&
1808 pad_operand->shape().dimensions(i) != 1) {
1809 VLOG(10) << "Found non-trivial dimension being padded: " << i;
1810 return false;
1811 }
1812 }
1813 VLOG(10) << "Found to be padding trivial dimensions only.";
1814
1815 for (int64 i = 0; i < window.dimensions_size(); ++i) {
1816 const auto& pad_dimension = pad_config.dimensions(i);
1817 const WindowDimension& window_dimension = window.dimensions(i);
1818 bool dimension_has_padding = (pad_dimension.edge_padding_low() != 0 ||
1819 pad_dimension.edge_padding_high() != 0);
1820 if (dimension_has_padding &&
1821 window_dimension.size() < pad_dimension.edge_padding_low() + 1) {
1822 VLOG(10) << "Found window did not cover single unpadded element in "
1823 "dimension: "
1824 << i;
1825 return false;
1826 }
1827 if (pad_operand->shape().dimensions(i) != 1 &&
1828 window_dimension.size() != 1) {
1829 VLOG(10) << "Found window covers more than one element in non-trivial "
1830 "dimension: "
1831 << i;
1832 return false;
1833 }
1834 }
1835 VLOG(10) << "Found window covers a single unpadded element.";
1836 return true;
1837 };
1838
1839 HloInstruction* new_reduce_window_operand;
1840 if (convert != nullptr) {
1841 new_reduce_window_operand =
1842 computation_->AddInstruction(HloInstruction::CreateConvert(
1843 ShapeUtil::ChangeElementType(pad_operand->shape(),
1844 convert->shape().element_type()),
1845 pad_operand));
1846 } else {
1847 new_reduce_window_operand = pad_operand;
1848 }
1849
1850 if (is_effective_broadcast()) {
1851 VLOG(10) << "Replacing pad/reduce-window with (implicit) broadcast.";
1852 auto fadd = [this](std::unique_ptr<HloInstruction> x) {
1853 return computation_->AddInstruction(std::move(x));
1854 };
1855 return ReplaceWithNewInstruction(
1856 reduce_window, HloInstruction::CreateBroadcastSequence(
1857 /*output_shape=*/reduce_window->shape(),
1858 /*operand=*/new_reduce_window_operand, fadd));
1859 }
1860
1861 // Carry out the folding of the pad into reduce_window.
1862 VLOG(10) << "Folding pad into reduce-window.";
1863 Window new_window = window;
1864 const int64 rank = ShapeUtil::Rank(reduce_window->shape());
1865 TF_RET_CHECK(pad_config.dimensions_size() == rank);
1866 TF_RET_CHECK(window.dimensions_size() == rank);
1867 for (int64 i = 0; i < rank; ++i) {
1868 const auto& pad_dim = pad_config.dimensions(i);
1869 auto& window_dim = *new_window.mutable_dimensions(i);
1870 window_dim.set_padding_low(window_dim.padding_low() +
1871 pad_dim.edge_padding_low());
1872 window_dim.set_padding_high(window_dim.padding_high() +
1873 pad_dim.edge_padding_high());
1874 }
1875
1876 return ReplaceWithNewInstruction(
1877 reduce_window, HloInstruction::CreateReduceWindow(
1878 /*shape=*/reduce_window->shape(),
1879 /*operand=*/new_reduce_window_operand,
1880 /*init_value=*/reduce_window->mutable_operand(1),
1881 /*window=*/new_window,
1882 /*reduce_computation=*/function));
1883}

1885Status AlgebraicSimplifierVisitor::HandleTranspose(HloInstruction* transpose) {
1886 auto operand = transpose->mutable_operand(0);
1887 if (std::is_sorted(transpose->dimensions().begin(),
1888 transpose->dimensions().end())) {
1889 VLOG(10) << "deleting no-op transpose";
1890 return ReplaceInstruction(transpose, operand);
1891 }
1892
1893 if (HloOpcode::kTranspose == operand->opcode()) {
1894 return ReplaceWithNewInstruction(
1895 transpose, HloInstruction::CreateTranspose(
1896 transpose->shape(), operand->mutable_operand(0),
1897 ComposePermutations(operand->dimensions(),
1898 transpose->dimensions())));
1899 }
1900
1901 if (is_layout_sensitive_ && TransposeIsBitcast(transpose)) {
1902 ReplaceWithBitcast(transpose);
1903 return Status::OK();
1904 }
1905
1906 return Status::OK();
1907}

1909Status AlgebraicSimplifierVisitor::HandleConvolution(
1910 HloInstruction* convolution) {
1911 auto lhs = convolution->mutable_operand(0);
1912 auto rhs = convolution->mutable_operand(1);
1913 if (ShapeUtil::HasZeroElements(lhs->shape()) ||
1914 ShapeUtil::HasZeroElements(rhs->shape())) {
1915 return ReplaceWithNewInstruction(
1916 convolution,
1917 HloInstruction::CreateBroadcast(
1918 convolution->shape(),
1919 computation_->AddInstruction(HloInstruction::CreateConvert(
1920 ShapeUtil::MakeShape(convolution->shape().element_type(), {}),
1921 computation_->AddInstruction(
1922 HloInstruction::CreateConstant(Literal::CreateR0(0.0f))))),
1923 {}));
1924 }
1925 const auto& window = convolution->window();
1926 if (!enable_conv_simplification_) {
1927 return Status::OK();
1928 }
1929 // HandleConvolution tries to replace a convolution with a DOT instruction.
1930 //
1931 // Only add when bitcasts can be used:
1932 // - if bitcasts are not supported, then reshapes could be used but will
1933 // end up with another copy.
1934 // - if bitcasts are supported, the simplifier will be called again with
1935 // bitcasts_ == true.
1936
1937 // TODO(cwhipkey): b/31337498, make this layout insensitive.
1938 if (!is_layout_sensitive_) {
1939 return Status::OK();
1940 }
1941
1942 const ConvolutionDimensionNumbers& dnums =
1943 convolution->convolution_dimension_numbers();
1944 const Shape& input_shape = lhs->shape();
1945 const Shape& filter_shape = rhs->shape();
1946 const Shape& convolution_shape = convolution->shape();
1947 TF_RET_CHECK(LayoutUtil::HasLayout(input_shape));
1948 TF_RET_CHECK(LayoutUtil::HasLayout(filter_shape));
1949 TF_RET_CHECK(LayoutUtil::HasLayout(convolution_shape));
1950
1951 // Require the spatial dimensions in the kernel to have a bound of one.
1952 for (int64 i = 0; i < dnums.kernel_spatial_dimensions_size(); ++i) {
1953 if (filter_shape.dimensions(dnums.kernel_spatial_dimensions(i)) != 1) {
1954 return Status::OK();
1955 }
1956 }
1957
1958 // Stride ignores part of the output, which matrix multiplication does not do,
1959 // so require no stride. Padding and base (lhs) dilation both implicitly
1960 // extend the data, which matrix multiplication also does not do, so require
1961 // no padding and no base (lhs) dilation. Window (rhs) dilation has no effect
1962 // for a 1x1 window, so window dilation is no problem.
1963 if (window_util::HasStride(window) || window_util::HasPadding(window) ||
1964 window_util::HasBaseDilation(window)) {
1965 return Status::OK();
1966 }
1967
1968 // Also, the shapes must align for a rowmajor matmul:
1969 // - the input and output have the same layout.
1970 // - for input/output, the channel dimension must be the most minor. Other
1971 // spatial dims can be in any order.
1972 // - for filters, the input channel dimension must be more major than the
1973 // output channel dimension. The width+height don't matter because
1974 // they are 1.
1975 //
1976 // These constraints are harsh. If the channel dimension is the most major
1977 // and/or the layout of input/output feature dimensions are reversed, we can
1978 // still convert Conv into more efficient Matmul with operand transposition
1979 // (such as the transposition flags in cuBLAS SGEMM).
1980 if (!LayoutUtil::Equal(input_shape.layout(), convolution_shape.layout()) ||
1981 LayoutUtil::Minor(input_shape.layout(), 0) !=
1982 dnums.input_feature_dimension() ||
1983 LayoutUtil::Minor(convolution_shape.layout(), 0) !=
1984 dnums.output_feature_dimension() ||
1985 // The input feature dimension should come later in the minor-to-major
1986 // order.
1987 (PositionInContainer(LayoutUtil::MinorToMajor(filter_shape),
1988 dnums.kernel_input_feature_dimension()) <
1989 PositionInContainer(LayoutUtil::MinorToMajor(filter_shape),
1990 dnums.kernel_output_feature_dimension()))) {
1991 return Status::OK();
1992 }
1993
1994 auto add_bitcast = [&](Shape shape, HloInstruction* operand) {
1995 std::vector<int64> dims(operand->shape().dimensions_size());
1996 std::iota(dims.begin(), dims.end(), 0);
1997 return computation_->AddInstruction(
1998 HloInstruction::CreateUnary(shape, HloOpcode::kBitcast, operand));
1999 };
2000
2001 // Replace it with a dot, with bitcasts around it to get the right shape.
2002 const int64 input_channels =
2003 input_shape.dimensions(dnums.input_feature_dimension());
2004 const int64 output_channels =
2005 filter_shape.dimensions(dnums.kernel_output_feature_dimension());
2006
2007 // Computes the product of the non-feature dimensions.
2008 int64 conv_width = 1;
2009 for (int i = 0; i < input_shape.dimensions_size(); ++i) {
2010 if (i != dnums.input_feature_dimension()) {
2011 conv_width *= input_shape.dimensions(i);
2012 }
2013 }
2014
2015 // We already checked feature_dimension is most minor, so data in input_shape
2016 // and row-major {conv_width,input_channels} are bitwise identical.
2017 const Shape new_input_shape = ShapeUtil::MakeShapeWithDescendingLayout(
2018 input_shape.element_type(), {conv_width, input_channels});
2019 // We already checked input_feature_dimension is more major than
2020 // output_feature_dimension, so data in filter_shape and row-major
2021 // {input_channels,output_channels} are bitwise identical.
2022 const Shape new_filter_shape = ShapeUtil::MakeShapeWithDescendingLayout(
2023 filter_shape.element_type(), {input_channels, output_channels});
2024 const Shape dot_output_shape = ShapeUtil::MakeShapeWithDescendingLayout(
2025 convolution_shape.element_type(), {conv_width, output_channels});
2026
2027 // We cannot insert bitcasts if the layouts will not be compatible.
2028 // TODO(b/33178038): Consider inserting a transpose if a bitcast would be
2029 // invalid.
2030 if (!valid_bitcast_callback_(input_shape, new_input_shape) ||
2031 !valid_bitcast_callback_(filter_shape, new_filter_shape) ||
2032 !valid_bitcast_callback_(dot_output_shape, convolution_shape)) {
2033 return Status::OK();
2034 }
2035
2036 auto new_lhs = add_bitcast(new_input_shape, lhs);
2037 auto new_rhs = add_bitcast(new_filter_shape, rhs);
2038 DotDimensionNumbers dot_dimension_numbers;
2039 dot_dimension_numbers.add_lhs_contracting_dimensions(1);
2040 dot_dimension_numbers.add_rhs_contracting_dimensions(0);
2041 auto dot = computation_->AddInstruction(HloInstruction::CreateDot(
2042 dot_output_shape, new_lhs, new_rhs, dot_dimension_numbers));
2043 return ReplaceInstruction(convolution, add_bitcast(convolution_shape, dot));
2044}

2052 return false;

2058 return true;

2063 // min_operand operand
2064 // \ /
