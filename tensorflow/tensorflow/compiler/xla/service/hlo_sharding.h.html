35namespace xla {

37// HLO shardings describe how an HLO instruction is split across multiple
38// computations.
39class HloSharding {
40 public:
41 // Creates a trivial sharding that replicates a maximal tile across all
42 // devices.
43 static HloSharding Replicate() { return HloSharding(); }
44
45 // Creates a sharding that emulates device placement; a tile shape equal to
46 // the input shape (one tile) assigned to a single device.
47 static HloSharding AssignDevice(int64 device_id);
48
49 // Creates a new sharding which splits a shape into tiles each with shape
50 // `tile_shape`. Each tile is assigned to one device, which is specified by
51 // `tile_assignment`. Any tensor not a multiple of the tile size in any
52 // dimension is implicitly padded to the tile size.
53 //
54 // e.g. Tile({2, 2}, {0, 1}) on a tensor of shape {3, 2} would look like:
55 // 2 1 padding
56 // <------><->
57 // +----+----+
58 // | 0 | 1 |
59 // +----+----+
60 //
61 // Split into two tiles, one of which is implicitly padded by one.
62 static HloSharding Tile(const Shape& tile_shape,
63 const Array<int64>& tile_assignment) {
64 return HloSharding(tile_shape, tile_assignment);
65 }
66
67 // Creates a new sharding which splits a one-dimensional input shape into
68 // `num_tiles` tiles.
69 static HloSharding Tile1D(const Shape& input_shape, int64 num_tiles);
70
71 // Creates a new sharding for a tuple type. The given ShapeTree must have
72 // elements for every leaf shape contained in the tuple.
73 static HloSharding Tuple(const ShapeTree<HloSharding>& sub_shardings) {
74 std::vector<HloSharding> flattened_list;
75 flattened_list.reserve(
76 std::distance(sub_shardings.leaf_begin(), sub_shardings.leaf_end()));
77 for (const auto& index_to_sharding : sub_shardings.leaves()) {
78 flattened_list.push_back(index_to_sharding.second);
79 }
80 return HloSharding(flattened_list);
81 }
82
83 // Creates a new sharding for a tuple type. The requested tuple shape must not
84 // be nested. For nested tuples, use the ShapeTree overload.
85 static HloSharding Tuple(const Shape& tuple_shape,
86 tensorflow::gtl::ArraySlice<HloSharding> shardings) {
87 CHECK(ShapeUtil::IsTuple(tuple_shape));
88 CHECK(!ShapeUtil::IsNestedTuple(tuple_shape));
89 std::vector<HloSharding> flattened_list(shardings.begin(), shardings.end());
90 CHECK_EQ(flattened_list.size(), ShapeUtil::TupleElementCount(tuple_shape));
91 return HloSharding(flattened_list);
92 }
93
94 // Create a new sharding from a protobuf OpSharding.
95 static StatusOr<HloSharding> FromProto(const OpSharding& proto);
96
97 // Checks whether device is a reserved device number. A reserved device number
98 // has usually a special meaning, with dedicated handling logic.
99 static bool IsReservedDevice(int64 device) { return device < 0; }
100
101 OpSharding ToProto() const;
102 string ToString() const;
103
104 // Validate that this sharding can be applied to a tensor with shape `shape`.
105 Status Validate(const Shape& shape, int64 num_devices) const;
106
107 // Returns true if the sharding has tuple type.
108 bool IsTuple() const { return tuple_; }
109
110 // Returns true if the sharding is trivial: replicate on all devices.
111 bool IsReplicated() const {
112 if (!IsTuple()) {
113 return replicated_;
114 }
115 return std::all_of(tuple_elements_.begin(), tuple_elements_.end(),
116 [](const HloSharding& s) { return s.IsReplicated(); });
117 }
118
119 // Returns true if the tile size is the same as the input size.
120 bool IsTileMaximal() const {
121 if (!IsTuple()) {
122 return maximal_;
123 }
124 return std::all_of(tuple_elements_.begin(), tuple_elements_.end(),
125 [](const HloSharding& s) { return s.IsTileMaximal(); });
126 }
127
128 // Returns true if the sharding defines an operation on the given device.
129 bool UsesDevice(int64 device) const;
130
131 // Returns the tile that should be executed on the given device.
132 // REQUIRES: !IsTuple()
133 std::vector<int64> TileIndexForDevice(int64 device) const;
134
135 // Returns the device that should execute the given tile.
136 // It is an error to call this if is_replicated() is true.
137 // REQUIRES: !IsTuple()
138 int64 DeviceForTileIndex(tensorflow::gtl::ArraySlice<int64> index) const;
139
140 // Given a device ID, returns the offset within the input space of the
141 // tile that should be executed on the given core. This returns the lower
142 // extent of the tile in the input space.
143 // REQUIRES: !IsTuple()
144 std::vector<int64> TileOffsetForDevice(int64 device) const;
145
146 // Given a device ID, returns the limit within the input space of the
147 // tile that should be executed on the given core. This returns the upper
148 // extent of the tile in the input space.
149 // REQUIRES: !IsTuple()
150 std::vector<int64> TileLimitForDevice(int64 device) const;
151
152 // Returns the single device this op operates on.
153 // REQUIRES: !IsTuple&& !Replicated() && IsTileMaximal()
154 StatusOr<int64> UniqueDevice() const;
155
156 // Returns true if this op only uses a single device.
157 bool HasUniqueDevice() const;
158
159 // Returns the ShapeTree containing the shardings for each element of this
160 // tuple, if IsTuple, or a ShapeTree with a single element containing this
161 // sharding. Only the leaf elements are populated. This creates a new
162 // ShapeTree object so is not cheap.
163 ShapeTree<HloSharding> GetAsShapeTree(const Shape& shape) const {
164 if (IsTuple()) {
165 ShapeTree<HloSharding> result(shape, HloSharding::Replicate());
166 CHECK_EQ(std::distance(result.leaf_begin(), result.leaf_end()),
167 tuple_elements_.size());
168 auto it = tuple_elements_.begin();
169 for (auto& index_to_sharding : result.leaves()) {
170 index_to_sharding.second = *it++;
171 }
172 return result;
173 } else {
174 return ShapeTree<HloSharding>(shape, *this);
175 }
176 }
177
178 bool operator==(const HloSharding& other) const {
179 return replicated_ == other.replicated_ && maximal_ == other.maximal_ &&
180 ShapeUtil::Compatible(tile_shape_, other.tile_shape_) &&
181 tile_assignment_ == other.tile_assignment_ &&
182 tuple_elements_ == other.tuple_elements_;
183 }
184 bool operator!=(const HloSharding& other) const { return !(*this == other); }
185
186 size_t Hash() const {
187 if (!tuple_) {
188 size_t h = 0;
189 for (const auto& element : tuple_elements_) {
190 h = tensorflow::Hash64Combine(h, element.Hash());
191 }
192 return h;
193 }
194 if (replicated_) {
195 return 0;
196 }
197 size_t h = 0;
198 for (uint32 v : tile_assignment_) {
199 h = tensorflow::Hash64Combine(h, std::hash<uint32>{}(v));
200 }
201 for (uint32 v : tile_shape_.dimensions()) {
202 h = tensorflow::Hash64Combine(h, std::hash<uint32>{}(v));
203 }
204 return h;
205 }
206
207 // Gets the tile shape.
208 // REQUIRES: !IsTileMaximal() && !IsTuple()
209 const Shape& tile_shape() const { return tile_shape_; }
210 // Gets the tile assignment tensor.
211 // REQUIRES: !IsReplicated() && !IsTuple()
212 const Array<int64>& tile_assignment() const { return tile_assignment_; }
213
214 // Returns the flattened list of all the leaf shardings in a tuple shape, by
215 // pre-order walk (ShapeTree iterator order).
216 // REQUIRES: IsTuple().
217 const std::vector<HloSharding>& tuple_elements() const {
218 return tuple_elements_;
219 }
220
221 // Return a new sharding that can apply to the given new shape.
222 // If this sharding is tile-maximal, the returned sharding will be the same as
223 // this sharding. If this sharding is not tile-maximal, the returned
224 // sharding's tile size will differ:
225 // - Non-sharded dimensions will be adapted to be the same as `new_shape`;
226 // tile_dimension(i) = new_shape.dimensions(i);
227 // - Sharded dimensions will be kept the same unless `transform` is supplied
228 // in which case tile_dimension(i) = transform(i, tile_dimension(i));
229 // REQUIRES: !IsTuple().
230 HloSharding TransformShardedTileShape(
231 const Shape& new_shape,
232 const std::function<int64(int64, int64)>& transform = nullptr) const;
233
234 private:
235 HloSharding()
236 : replicated_(true),
237 maximal_(true),
238 tuple_(false),
239 tile_shape_(),
240 tile_assignment_({0}) {}
241 explicit HloSharding(int64 device_id)
242 : replicated_(false),
243 maximal_(true),
244 tuple_(false),
245 tile_shape_(),
246 tile_assignment_({1}, device_id) {}
247 HloSharding(const Shape& tile_shape, const Array<int64>& tile_assignment)
248 : replicated_(false),
249 maximal_(false),
250 tuple_(false),
251 tile_shape_(tile_shape),
252 tile_assignment_(tile_assignment) {}
253 HloSharding(const std::vector<HloSharding>& tuple_shardings)
254 : replicated_(false),
255 maximal_(false),
256 tuple_(true),
257 tile_assignment_({0}),
258 tuple_elements_(tuple_shardings) {}
259
260 // Internal helper to validate a tuple sharding.
261 Status ValidateTuple(const Shape& shape, int64 num_devices) const;
262 // Internal helper to validate a non-tuple (leaf) sharding.
263 Status ValidateNonTuple(const Shape& shape, int64 num_devices) const;
264
265 bool replicated_;
266 bool maximal_;
267 bool tuple_;
268 Shape tile_shape_;
269 Array<int64> tile_assignment_;
270 // Only non-empty when tuple_ is true, but because empty tuples are allowed
271 // may also be empty even then. This is a flattened list of all the leaf
272 // shardings in a tuple shape, by pre-order walk (ShapeTree iterator order).
273 std::vector<HloSharding> tuple_elements_;
274};
