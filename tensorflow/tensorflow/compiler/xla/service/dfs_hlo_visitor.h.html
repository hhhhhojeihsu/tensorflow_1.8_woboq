39// A postorder depth-first HloInstruction visitor. When Handle* is called on an
40// instruction, all its operands were already visited. User code can subclass
41// this to iterate over an HloInstruction DAG. The Handle* routines have
42// operands / data unpacked for ease of use in the visitor subclass.
43//
44// No instruction will ever be visited twice; however, the root instruction will
45// be reported again when the traversal is done via a call to FinishVisit.
46//
47// A subclass must override at least
48// (either HandleElementwiseUnary or all the Handle methods for unary ops) and
49// (either HandleElementwiseBinary or all the Handle methods for binary ops)).
50// The default Handle methods for (unary, binary) ops call
51// (HandleElementwiseUnary, HandleElementwiseBinary).
52// The default (HandleElementwiseUnary, HandleElementwiseBinary) return an
53// "unimplemented" error status.
54//
55// Note: this may change to an iterator in the future for flexibility purposes.
56//
57// Users should not use this class directly, but use the type-aliases
58// DfsHloVisitor/ConstDfsHloVisitor instead.
59template <typename HloInstructionPtr>
60class DfsHloVisitorBase {
61 static_assert(
62 std::is_same<HloInstruction*, HloInstructionPtr>::value ||
63 std::is_same<const HloInstruction*, HloInstructionPtr>::value,
64 "Template argument expected to be HloInstruction* or const "
65 "HloInstruction*");
66
67 public:
68 DfsHloVisitorBase() {}

238 // 3 possible visitation states of HLO instructions. Each instruction's
239 // state only flows one way: kNotVisited -> kVisiting -> kVisited.
240 enum VisitState {
241 kNotVisited = 0,
242 kVisiting = 1,
243 kVisited = 2,
244 };
245
246 VisitState GetVisitState(int id) { return visit_state_.GetState(id); }
247 VisitState GetVisitState(const HloInstruction& instruction);
248
249 // Resize internal state if necessary to hold state for ids <= num.
250 // This call is purely a performance hint and can be omitted without
251 // affecting correctness.
252 void ReserveVisitStates(int num) { visit_state_.Reserve(num); }
253
254 // Useful when we want to visit the same computation more than once with the
255 // same visitor.
256 void ResetVisitStates() { visit_state_.Reset(); }
257
258 void SetVisitState(int id, VisitState state) {
259 visit_state_.SetState(id, state);
260 }
261
262 // Sets the visitation state of the given instruction as kVisiting.
263 //
264 // Precondition: current state must be kNotVisited.
265 void SetVisiting(const HloInstruction& instruction);
266
267 // Sets the visitation state of the given instruction as kVisited.
268 //
269 // Precondition: current state must be either kNotVisited or kVisiting.
270 void SetVisited(const HloInstruction& instruction);

272 // Returns whether the state of the given instruction is kVisiting.
273 bool IsVisiting(const HloInstruction& instruction) {
274 return GetVisitState(instruction) == kVisiting;
275 }
276
277 // Returns whether the state of the given instruction is kVisited.
278 bool DidVisit(const HloInstruction& instruction) {
279 return GetVisitState(instruction) == kVisited;
280 }
281
282 // Returns whether the state of the given instruction is kNotVisited.
283 bool NotVisited(const HloInstruction& instruction) {
284 return GetVisitState(instruction) == kNotVisited;
285 }

287 // This method should be overridden by subclasses that wish to run some
288 // operation on an op before its Handle* visitor method is called.

290 // For any HLO op, the order of calls is:
291 //
292 // Preprocess(op);
293 // Handle/OpType/(op);
294 // Postprocess(op);

297 // own preprocessing.

308 private:
309 class DFSVisitStates {
310 public:
311 DFSVisitStates() {}
312 void Reserve(uint64 num) {
313 states_.reserve((num + kStatesPerWord - 1) / kStatesPerWord);
314 }
315 VisitState GetState(uint64 id) {
316 uint64 word_index = id / kStatesPerWord;
317 if (word_index >= states_.size()) {
318 return VisitState::kNotVisited;
319 }
320 static_assert(static_cast<int>(VisitState::kVisited) < 3,
321 "VisitState must fit in two bits");
322 uint64 w = states_[word_index];
323 uint32 shift = 2 * (id % kStatesPerWord); // 2 bits per state
324 return static_cast<VisitState>((w >> shift) & 0x3);
325 }
326 void SetState(uint64 id, VisitState state) {
327 uint64 word_index = id / kStatesPerWord;
328 if (word_index >= states_.size()) {
329 states_.resize(word_index + 1, 0);
330 }
331 uint64* w = &states_[word_index];
332 uint32 shift = 2 * (id % kStatesPerWord); // 2 bits per state
333 uint64 mask = 0x3ull << shift;
334 *w = (*w & ~mask) | (static_cast<uint64>(state) << shift);
335 DCHECK_EQ(GetState(id), state);
336 }
337 void Reset() { states_.clear(); }
338
339 private:
340 static const uint32 kStatesPerWord = sizeof(uint64) / 2 /*bits per entry*/;
341 // Map from id to two-bit states. We store 32 such states per 64-bit
342 // value
343 std::vector<uint64> states_;
344 };
345
346 DFSVisitStates visit_state_;
347
