184tensorflow::Status VerifyReducerShape(const ProgramShape& reducer_shape,
185 const Shape& init_value_shape,
186 const PrimitiveType& input_element_type) {
187 if (reducer_shape.parameters_size() != 2) {
188 return InvalidArgument(
189 "Reduction function must take 2 parameters, but "
190 "takes %d parameter(s).",
191 reducer_shape.parameters_size());
192 }
193
194 const Shape& accumulator_shape = reducer_shape.result();
195 if (ShapeUtil::Rank(accumulator_shape) != 0) {
196 return InvalidArgument(
197 "Reduction function must have rank 0 (rank %lld reduction function "
198 "given).",
199 ShapeUtil::Rank(accumulator_shape));
200 }
201
202 // Check that the accumulator can be passed in as the first argument.
203 // Note: comparing here and below with Compatible since we don't care about
204 // layout in scalars - see b/26668201 for a longer-term vision.
205 if (!ShapeUtil::Compatible(accumulator_shape, reducer_shape.parameters(0))) {
206 return InvalidArgument(
207 "Reduction function's first parameter shape differs from the "
208 "result shape: %s vs %s",
209 ShapeUtil::HumanString(reducer_shape.parameters(0)).c_str(),
210 ShapeUtil::HumanString(accumulator_shape).c_str());
211 }
212
213 // Check that init_value's shape is suitable for reducer_shape.
214 if (!ShapeUtil::CompatibleIgnoringFpPrecision(accumulator_shape,
215 init_value_shape)) {
216 return InvalidArgument(
217 "Reduction function's accumulator shape differs from the "
218 "init_value shape: %s vs %s",
219 ShapeUtil::HumanString(accumulator_shape).c_str(),
220 ShapeUtil::HumanString(init_value_shape).c_str());
221 }
222
223 // Check that the inputs can be passed in as the second argument.
224 const Shape& input_element_shape =
225 ShapeUtil::MakeShape(input_element_type, {});
226 if (!ShapeUtil::CompatibleIgnoringFpPrecision(input_element_shape,
227 reducer_shape.parameters(1))) {
228 return InvalidArgument(
229 "Reduction function's second parameter shape differs from the "
230 "input type element type: %s vs %s",
231 ShapeUtil::HumanString(reducer_shape.parameters(1)).c_str(),
232 ShapeUtil::HumanString(input_element_shape).c_str());
233 }
234
235 // Currently the accumulator and inputs must be the same type,
236 // though that restriction could be relaxed.
237 if (!ShapeUtil::CompatibleIgnoringFpPrecision(accumulator_shape,
238 reducer_shape.parameters(1))) {
239 return InvalidArgument(
240 "Reduction function's second parameter shape must "
241 "match the result shape, but got %s vs %s.",
242 ShapeUtil::HumanString(reducer_shape.parameters(1)).c_str(),
243 ShapeUtil::HumanString(accumulator_shape).c_str());
244 }
245
246 return tensorflow::Status::OK();
247}

249StatusOr<Shape> InferWindowOutputShape(const Shape& base_shape,
250 const Window& window,
251 PrimitiveType element_type,
252 bool allow_negative_padding) {
253 if (window.dimensions_size() != ShapeUtil::Rank(base_shape)) {
254 return InvalidArgument(
255 "Window has dimension %d but base shape has dimension %lld.",
256 window.dimensions_size(), ShapeUtil::Rank(base_shape));
257 }
258
259 std::vector<int64> output_dimensions(window.dimensions_size());
260 for (int64 i = 0; i < window.dimensions_size(); ++i) {
261 const auto& dim = window.dimensions(i);
262 if (dim.size() <= 0) {
263 return InvalidArgument("Window %s has a non-positive dimension.",
264 window.DebugString().c_str());
265 }
266 if (dim.stride() <= 0) {
267 return InvalidArgument("Window %s has a non-positive stride.",
268 window.DebugString().c_str());
269 }
270 if (!allow_negative_padding && dim.padding_low() < 0) {
271 return InvalidArgument("Window %s has a negative low padding.",
272 window.DebugString().c_str());
273 }
274 if (!allow_negative_padding && dim.padding_high() < 0) {
275 return InvalidArgument("Window %s has a negative high padding.",
276 window.DebugString().c_str());
277 }
278 if (dim.base_dilation() < 1) {
279 return InvalidArgument(
280 "Window %s has a non-positive base area dilation factor.",
281 window.DebugString().c_str());
282 }
283 if (dim.window_dilation() < 1) {
284 return InvalidArgument(
285 "Window %s has a non-positive window dilation factor.",
286 window.DebugString().c_str());
287 }
288
289 const int64 dilated_base = window_util::DilatedBound(
290 ShapeUtil::GetDimension(base_shape, i), dim.base_dilation());
291 const int64 padded_dilated_base =
292 dim.padding_low() + dilated_base + dim.padding_high();
293 const int64 dilated_window =
294 window_util::DilatedBound(dim.size(), dim.window_dilation());
295
296 output_dimensions[i] = window_util::StridedBound(
297 padded_dilated_base, dilated_window, dim.stride());
298 }
299
300 return ShapeUtil::MakeShape(element_type, output_dimensions);
301}

320/* static */ StatusOr<Shape> ShapeInference::InferUnaryOpShape(
321 UnaryOperation operation, const Shape& arg) {
322 TF_RETURN_IF_ERROR(ExpectNotTupleOrOpaque(arg, "operand of unary operation"));
323
324 TF_DCHECK_OK(ShapeUtil::ValidateShapeWithOptionalLayout(arg));
325 switch (operation) {
326 case UNOP_FLOOR:
327 case UNOP_CEIL:
328 if (!ShapeUtil::ElementIsFloating(arg)) {
329 return InvalidArgument(
330 "Expected element type in shape to be floating for floor/ceil "
331 "operation; got %s.",
332 PrimitiveType_Name(arg.element_type()).c_str());
333 }
334 return arg;
335 case UNOP_COS:
336 case UNOP_SIN:
337 case UNOP_EXP:
338 case UNOP_LOG:
339 case UNOP_TANH:
340 if (!ShapeUtil::ElementIsFloating(arg) &&
341 !ShapeUtil::ElementIsComplex(arg)) {
342 return InvalidArgument(
343 "Expected element type in shape to be floating or complex for "
344 "sin/cos/exp/log/tanh operation; got %s.",
345 PrimitiveType_Name(arg.element_type()).c_str());
346 }
347 return arg;
348 case UNOP_REAL:
349 case UNOP_IMAG:
350 if (!ShapeUtil::ElementIsComplex(arg)) {
351 return InvalidArgument(
352 "Expected element type in shape to be complex for real/imag "
353 "operation; got %s.",
354 PrimitiveType_Name(arg.element_type()).c_str());
355 }
356 return ShapeUtil::ChangeElementType(arg, F32);
357 case UNOP_ABS:
358 if (ShapeUtil::ElementIsComplex(arg)) {
359 return ShapeUtil::ChangeElementType(
360 arg, primitive_util::ComplexComponentType(arg.element_type()));
361 }
362 return arg;
363 case UNOP_NEGATE:
364 case UNOP_ROUND_NEAREST_AFZ:
365 case UNOP_SIGN:
366 case UNOP_SORT:
367 return arg;
368
369 case UNOP_NOT:
370 if (arg.element_type() != PRED &&
371 !primitive_util::IsIntegralType(arg.element_type())) {
372 return InvalidArgument(
373 "Expected pred or an integral element type in argument to Not "
374 "operation; got %s.",
375 PrimitiveType_Name(arg.element_type()).c_str());
376 }
377 return arg;
378
379 case UNOP_IS_FINITE:
380 if (!ShapeUtil::ElementIsFloating(arg)) {
381 return InvalidArgument(
382 "Expected element type in shape to be floating point for IsFinite "
383 "operation; got %s.",
384 PrimitiveType_Name(arg.element_type()).c_str());
385 }
386 return ShapeUtil::ChangeElementType(arg, PRED);
387
388 default:
389 return InvalidArgument(
390 "Unknown operation for unary shape inference: \"%s\".",
391 UnaryOperation_Name(operation).c_str());
392 }
393}

395/* static */ StatusOr<Shape> ShapeInference::InferConcatOpShape(
396 tensorflow::gtl::ArraySlice<const Shape*> arg_shapes,
397 const int64 dimension) {
398 if (arg_shapes.empty()) {
399 return InvalidArgument("Concatenate expects at least one argument.");
400 }
401 if (dimension < 0 || dimension >= ShapeUtil::Rank(*arg_shapes[0])) {
402 return InvalidArgument("Concatenate dimension out of bounds: %lld.",
403 dimension);
404 }
405 const Shape* arg_shape = nullptr;
406 PrimitiveType element_type = PRIMITIVE_TYPE_INVALID;
407 for (const Shape* shape : arg_shapes) {
408 TF_RETURN_IF_ERROR(
409 ExpectNotTupleOrOpaque(*shape, "operand of concatenation"));
410 if (!arg_shape) {
411 arg_shape = shape;
412 element_type = arg_shape->element_type();
413 continue;
414 }
415 if (ShapeUtil::Rank(*arg_shape) != ShapeUtil::Rank(*shape)) {
416 return InvalidArgument(
417 "Cannot concatenate arrays with different ranks: %lld (%s) vs %lld "
418 "(%s).",
419 ShapeUtil::Rank(*arg_shape),
420 ShapeUtil::HumanString(*arg_shape).c_str(), ShapeUtil::Rank(*shape),
421 ShapeUtil::HumanString(*shape).c_str());
422 }
423 if (!ShapeUtil::SameElementTypeIgnoringFpPrecision(*arg_shape, *shape)) {
424 return InvalidArgument(
425 "Cannot concatenate arrays with different element types: %s vs %s.",
426 PrimitiveType_Name(arg_shape->element_type()).c_str(),
427 PrimitiveType_Name(shape->element_type()).c_str());
428 }
429 for (int64 dimension_number = 0;
430 dimension_number < ShapeUtil::Rank(*arg_shape); ++dimension_number) {
431 if (arg_shape->dimensions(dimension_number) !=
432 shape->dimensions(dimension_number)) {
433 if (dimension_number == dimension) {
434 continue; // It's okay to differ in the dimension we're
435 // concatenating.
436 }
437 return InvalidArgument(
438 "Cannot concatenate arrays that differ in dimensions other than "
439 "the one being concatenated (the other array dimensions must be "
440 "the same): %s vs %s in dimension %lld.",
441 ShapeUtil::HumanString(*arg_shape).c_str(),
442 ShapeUtil::HumanString(*shape).c_str(), dimension);
443 }
444 }
445 element_type = ShapeUtil::HigherPrecisionElementType(*shape, *arg_shape);
446 }
447
448 std::vector<int64> new_dimensions(arg_shape->dimensions().begin(),
449 arg_shape->dimensions().end());
450 for (size_t i = 1; i < arg_shapes.size(); ++i) {
451 new_dimensions[dimension] += arg_shapes[i]->dimensions(dimension);
452 }
453 return ShapeUtil::MakeShape(element_type, new_dimensions);
454}

456/* static */ StatusOr<Shape> ShapeInference::InferConvertShape(
457 const Shape& operand_shape, PrimitiveType new_element_type) {
458 auto old_element_type = operand_shape.element_type();
459 if (primitive_util::IsComplexType(old_element_type) &&
460 !primitive_util::IsComplexType(new_element_type)) {
461 return Unimplemented(
462 "Conversion from complex to real type %s => %s is not implemented.",
463 ShapeUtil::HumanString(operand_shape).c_str(),
464 PrimitiveType_Name(new_element_type).c_str());
465 }
466 if (ShapeUtil::IsTuple(operand_shape) || new_element_type == TUPLE) {
467 // Note: we may want to support tuple conversions via this operation in the
468 // future, by recursing into the tuple elements to check all sub-conversions
469 // are valid. For now we just reject them, though.
470 return InvalidArgument(
471 "Convert does not allow tuples, so cannot convert from %s to %s.",
472 ShapeUtil::HumanString(operand_shape).c_str(),
473 PrimitiveType_Name(new_element_type).c_str());
474 }
475
476 return ShapeUtil::ChangeElementType(operand_shape, new_element_type);
477}

508/* static */ StatusOr<Shape> ShapeInference::InferReducePrecisionShape(
509 const Shape& operand_shape, const int exponent_bits,
510 const int mantissa_bits) {
511 if (!ShapeUtil::ElementIsFloating(operand_shape)) {
512 return InvalidArgument(
513 "Expected element type in shape to be floating point for "
514 "ReducePrecision operation; got %s.",
515 PrimitiveType_Name(operand_shape.element_type()).c_str());
516 }
517 if (exponent_bits < 1) {
518 // One exponent bit is necessary to distinguish 0 from infinity. Having
519 // no exponent bits doesn't produce a sensible number, so we require at
520 // least one.
521 return InvalidArgument("Expected exponent_bits >= 1; got %d.",
522 exponent_bits);
523 }
524 if (mantissa_bits < 0) {
525 // A number with no mantissa bits is still meaningful, however.
526 return InvalidArgument("Expected non-negative mantissa_bits; got %d.",
527 mantissa_bits);
528 }
529 return operand_shape;
530}

532/* static */ StatusOr<Shape> ShapeInference::InferPadShape(
533 const Shape& operand_shape, const Shape& padding_value_shape,
534 const PaddingConfig& padding_config) {
535 if (ShapeUtil::IsTuple(operand_shape)) {
536 return InvalidArgument(
537 "Pad operation does not support tuple-shape operands.");
538 }
539 if (!ShapeUtil::IsScalar(padding_value_shape)) {
540 return InvalidArgument(
541 "Pad operation does not support non-scalar padding values.");
542 }
543 if (ShapeUtil::Rank(operand_shape) != padding_config.dimensions_size()) {
544 return InvalidArgument(
545 "The rank of the operand and the padding configuration do not match: "
546 "%s vs %s.",
547 ShapeUtil::HumanString(operand_shape).c_str(),
548 padding_config.ShortDebugString().c_str());
549 }
550 if (!ShapeUtil::SameElementTypeIgnoringFpPrecision(operand_shape,
551 padding_value_shape)) {
552 return InvalidArgument(
553 "The element types of the operands to Pad do not match.");
554 }
555 std::vector<int64> dimensions(ShapeUtil::Rank(operand_shape));
556 for (int64 i = 0; i < operand_shape.dimensions_size(); ++i) {
557 dimensions[i] = operand_shape.dimensions(i) +
558 padding_config.dimensions(i).edge_padding_low() +
559 padding_config.dimensions(i).edge_padding_high() +
560 std::max<int64>(operand_shape.dimensions(i) - 1, 0LL) *
561 padding_config.dimensions(i).interior_padding();
562 }
563 return ShapeUtil::MakeShape(
564 ShapeUtil::HigherPrecisionElementType(operand_shape, padding_value_shape),
565 dimensions);
566}

568// Current DotDimensionNumbers Requirements:
569//
570// Contracting Dimensions:
571// *) Exactly one contracting dimension on both lhs and rhs.
572// *) Contracting dimension size must be the same on both lhs and rhs.
573// *) Contracting dimension numbers do not need to be the same (i.e. transposes
574// are passed on to emitter implementations).
575//
576// Batch Dimensions:
577// *) Same number of batch dimensions on both lhs and rhs.
578// *) Same batch dimension numbers (and sizes) on both lhs and rhs.
579// *) Batch dimension numbers must be ordered before contracting and
580// non-contracting/non-batch dimension numbers.
581//
582// Non-Contracting-Non-Batch Dimensions:
583// *) Can be 0 (matrix-vector) or 1 (matrix-matrix).
584//
585
586namespace {
587
588Status ValidateDotDimensionNumbers(
589 const Shape& lhs, const Shape& rhs,
590 const DotDimensionNumbers& dimension_numbers) {
591 // Check that dimension numbers are in range.
592 auto dims_in_range =
593 [](const int64 rank, tensorflow::gtl::ArraySlice<int64> contracting_dims,
594 tensorflow::gtl::ArraySlice<int64> batch_dims) -> bool {
595 auto in_range = [&rank](int64 i) -> bool { return 0 <= i && i < rank; };
596 return std::all_of(contracting_dims.begin(), contracting_dims.end(),
597 in_range) &&
598 std::all_of(batch_dims.begin(), batch_dims.end(), in_range);
599 };
600
601 tensorflow::gtl::ArraySlice<int64> lhs_contracting_dimensions =
602 AsInt64Slice(dimension_numbers.lhs_contracting_dimensions());
603 tensorflow::gtl::ArraySlice<int64> rhs_contracting_dimensions =
604 AsInt64Slice(dimension_numbers.rhs_contracting_dimensions());
605 tensorflow::gtl::ArraySlice<int64> lhs_batch_dimensions =
606 AsInt64Slice(dimension_numbers.lhs_batch_dimensions());
607 tensorflow::gtl::ArraySlice<int64> rhs_batch_dimensions =
608 AsInt64Slice(dimension_numbers.rhs_batch_dimensions());
609
610 if (!dims_in_range(ShapeUtil::Rank(lhs), lhs_contracting_dimensions,
611 lhs_batch_dimensions) ||
612 !dims_in_range(ShapeUtil::Rank(rhs), rhs_contracting_dimensions,
613 rhs_batch_dimensions)) {
614 return InvalidArgument("A dimension number is out of range in Dot: %s.",
615 dimension_numbers.DebugString().c_str());
616 }
617
618 // Check that dimension numbers are unique.
619 auto dims_unique = [](tensorflow::gtl::ArraySlice<int64> contracting_dims,
620 tensorflow::gtl::ArraySlice<int64> batch_dims) -> bool {
621 tensorflow::gtl::FlatSet<int64> dim_set;
622 auto is_unique = [&dim_set](int64 i) -> bool {
623 return dim_set.insert(i).second;
624 };
625 return std::all_of(contracting_dims.begin(), contracting_dims.end(),
626 is_unique) &&
627 std::all_of(batch_dims.begin(), batch_dims.end(), is_unique);
628 };
629
630 if (!dims_unique(lhs_contracting_dimensions, lhs_batch_dimensions) ||
631 !dims_unique(rhs_contracting_dimensions, rhs_batch_dimensions)) {
632 return InvalidArgument("A dimension number is not unique in Dot: %s.",
633 dimension_numbers.DebugString().c_str());
634 }
635
636 // Check that the count of non-contracting-non-batch dimensions is in {0, 1}.
637 const int64 lhs_non_contracting_non_batch_dims =
638 ShapeUtil::Rank(lhs) -
639 dimension_numbers.lhs_contracting_dimensions_size() -
640 dimension_numbers.lhs_batch_dimensions_size();
641 const int64 rhs_non_contracting_non_batch_dims =
642 ShapeUtil::Rank(rhs) -
643 dimension_numbers.rhs_contracting_dimensions_size() -
644 dimension_numbers.rhs_batch_dimensions_size();
645 if (lhs_non_contracting_non_batch_dims < 0 ||
646 lhs_non_contracting_non_batch_dims > 1 ||
647 rhs_non_contracting_non_batch_dims < 0 ||
648 rhs_non_contracting_non_batch_dims > 1) {
649 return InvalidArgument(
650 "Batch and contracting dimension number mismatch with rank.");
651 }
652
653 // Check that batch dimension numbers are ordered before all others, and
654 // that they are monotonically increasing.
655 std::vector<int64> batch_dim_numbers(lhs_batch_dimensions.size());
656 std::iota(batch_dim_numbers.begin(), batch_dim_numbers.end(), 0);
657 if (!std::equal(batch_dim_numbers.begin(), batch_dim_numbers.end(),
658 lhs_batch_dimensions.begin()) ||
659 !std::equal(batch_dim_numbers.begin(), batch_dim_numbers.end(),
660 rhs_batch_dimensions.begin())) {
661 return InvalidArgument(
662 "Batch dimension numbers must precede non-batch dimensions and be"
663 "monotonically increasing.");
664 }
665
666 return Status::OK();
667}

671/* static */ StatusOr<Shape> ShapeInference::InferDotOpShape(
672 const Shape& lhs, const Shape& rhs,
673 const DotDimensionNumbers& dimension_numbers) {
674 TF_RETURN_IF_ERROR(ExpectNotTupleOrOpaque(lhs, "lhs of dot"));
675 TF_RETURN_IF_ERROR(ExpectNotTupleOrOpaque(rhs, "rhs of dot"));
676
677 auto fail = [lhs, rhs](const string& addendum) -> Status {
678 string message = tensorflow::strings::Printf(
679 "Cannot infer shape for dot operation: %s <dot> %s.",
680 ShapeUtil::HumanString(lhs).c_str(),
681 ShapeUtil::HumanString(rhs).c_str());
682 if (!addendum.empty()) {
683 message += " " + addendum;
684 }
685 return InvalidArgument("%s", message.c_str());
686 };
687
688 // Check if both element types are the same.
689 if (!ShapeUtil::SameElementTypeIgnoringFpPrecision(lhs, rhs)) {
690 return fail("Element types do not match.");
691 }
692
693 if ((ShapeUtil::Rank(lhs) < 1) || (ShapeUtil::Rank(rhs) < 1)) {
694 return fail("Dot only supports rank 1 or above.");
695 }
696
697 // Validate basic properties of dot dimension numbers.
698 TF_RETURN_IF_ERROR(ValidateDotDimensionNumbers(lhs, rhs, dimension_numbers));
699
700 // Check that there is only one contracting dimension for both lhs and rhs.
701 if (dimension_numbers.lhs_contracting_dimensions_size() !=
702 dimension_numbers.rhs_contracting_dimensions_size() ||
703 dimension_numbers.lhs_contracting_dimensions_size() != 1) {
704 return fail("Must specify one contracting dimension for both lhs and rhs.");
705 }
706
707 // Check that contracting dimension sizes match.
708 const int64 lhs_contracting_dimension =
709 dimension_numbers.lhs_contracting_dimensions(0);
710 const int64 rhs_contracting_dimension =
711 dimension_numbers.rhs_contracting_dimensions(0);
712 if (lhs.dimensions(lhs_contracting_dimension) !=
713 rhs.dimensions(rhs_contracting_dimension)) {
714 return fail("Contracting dimension sizes do not match.");
715 }
716
717 // Check that number of batch dimensions match.
718 if (dimension_numbers.lhs_batch_dimensions_size() !=
719 dimension_numbers.rhs_batch_dimensions_size()) {
720 return fail("Must the same number of batch dimensions for lhs and rhs.");
721 }
722
723 // Check that batch dimension numbers and sizes match.
724 for (int64 i = 0; i < dimension_numbers.lhs_batch_dimensions_size(); ++i) {
725 if (dimension_numbers.lhs_batch_dimensions(i) !=
726 dimension_numbers.rhs_batch_dimensions(i) ||
727 lhs.dimensions(dimension_numbers.lhs_batch_dimensions(i)) !=
728 rhs.dimensions(dimension_numbers.rhs_batch_dimensions(i))) {
729 return fail("Batch dimension numbers and sizes must match for lhs/rhs.");
730 }
731 }
732
733 // The ranks of lhs and rhs are decremented by 1 respectively due to the
734 // contraction, and added for the rank of the result. When an input tensor is
735 // a scalar, its contribution to the rank of the result is 0.
736 // Generate the result dimensions in order, rhs dimensions followed by lhs
737 // dimensions except the contracted and batch dimensions.
738 std::vector<int64> dimensions;
739 std::unordered_set<int64> rhs_batch_dims(
740 dimension_numbers.rhs_batch_dimensions().begin(),
741 dimension_numbers.rhs_batch_dimensions().end());
742 for (int64 i = 0; i < ShapeUtil::Rank(lhs); i++) {
743 if (i != lhs_contracting_dimension) {
744 dimensions.push_back(lhs.dimensions(i));
745 }
746 }
747 for (int64 i = 0; i < ShapeUtil::Rank(rhs); i++) {
748 if (i != rhs_contracting_dimension && rhs_batch_dims.count(i) == 0) {
749 dimensions.push_back(rhs.dimensions(i));
750 }
751 }
752 Shape result = ShapeUtil::MakeShape(
753 ShapeUtil::HigherPrecisionElementType(lhs, rhs), dimensions);
754
755 TF_DCHECK_OK(ShapeUtil::ValidateShapeWithOptionalLayout(result));
756 VLOG(2) << "inferred dot shape: " << ShapeUtil::HumanString(result);
757 return result;
758}

760/* static */ StatusOr<Shape>
761ShapeInference::InferDegenerateDimensionBroadcastShape(
762 BinaryOperation operation, const Shape& lhs, const Shape& rhs) {
763 TF_RET_CHECK(ShapeUtil::Rank(lhs) == ShapeUtil::Rank(rhs));
764
765 // The shapes have to be compatible. That is, if some dimension d has a
766 // different size in the two shapes, one of them has to be 1 (a "degenerate"
767 // dimension). In that case, the output shape has the non-1 dimension size
768 // from the lhs/rhs pair in every index.
769 std::vector<int64> output_dimensions(ShapeUtil::Rank(lhs));
770 for (int64 i = 0; i < ShapeUtil::Rank(lhs); ++i) {
771 if (lhs.dimensions(i) == rhs.dimensions(i)) {
772 output_dimensions[i] = lhs.dimensions(i);
773 } else if (lhs.dimensions(i) == 1) {
774 output_dimensions[i] = rhs.dimensions(i);
775 } else if (rhs.dimensions(i) == 1) {
776 output_dimensions[i] = lhs.dimensions(i);
777 } else {
778 return InvalidArgument(
779 "Binary op %s with incompatible shapes: %s and %s.",
780 BinaryOperation_Name(operation).c_str(),
781 ShapeUtil::HumanString(lhs).c_str(),
782 ShapeUtil::HumanString(rhs).c_str());
783 }
784 }
785 return ShapeUtil::MakeShape(ShapeUtil::HigherPrecisionElementType(lhs, rhs),
786 output_dimensions);
787}

789/* static */ StatusOr<Shape> ShapeInference::InferInDimBroadcastShape(
790 BinaryOperation operation, const Shape& smaller_shape,
791 const Shape& larger_shape,
792 tensorflow::gtl::ArraySlice<int64> broadcast_dimensions) {
793 if (broadcast_dimensions.empty() && !ShapeUtil::IsScalar(smaller_shape)) {
794 // Reject "magic" inference for binops on different shapes, requiring
795 // the user to provide an explicit broadcast dimension in this case.
796 // See b/25177275 for more details.
797 return InvalidArgument("Automatic shape inference not supported: %s and %s",
798 ShapeUtil::HumanString(smaller_shape).c_str(),
799 ShapeUtil::HumanString(larger_shape).c_str());
800 } else if (broadcast_dimensions.size() != ShapeUtil::Rank(smaller_shape)) {
801 return InvalidArgument(
802 "Size of broadcast_dimensions has to match lower-rank operand's "
803 "rank; "
804 " lower-rank operand's rank is %lld, size of broadcast_dimensions is "
805 "%zu.",
806 ShapeUtil::Rank(smaller_shape), broadcast_dimensions.size());
807 }
808
809 // broadcast_dimensions is a sequence of dimensions; its length is equal to
810 // the rank of the lower-rank operand. The lower-rank operand's dimensions
811 // have to be compatible with the higher-rank operand's dimensions at indices
812 // specified by broadcast_dimensions. Here compatible means the dimension
813 // sizes are equal or in one of the shapes the dimension size is
814 // one. Examples:
815 //
816 // smaller_shape larger_shape broadcast_dimensions output_shape
817 // [] [2, 3] {} [2, 3]
818 // [3] [4, 3] {1} [4, 3]
819 // [2, 3] [2, 3, 4] {0, 1} [2, 3, 4]
820 // [2, 1] [2, 3, 4] {0, 2} [2, 3, 1]
821 // [2, 3] [2, 1, 4] {0, 1} [2, 3, 4]
822 //
823 // The column output_shape may not be the final shape of the XLA
824 // operation. After the "InDim" broadcasting implemented in this function
825 // expands the rank, degenerate-dimension broadcasting (implemented in
826 // InferDegenerateDimensionBroadcastShape) broadcasts dimensions of size one
827 // up to match the dimension size of the other operand. For example, consider
828 // the row in the table above with a smaller_shape of [2, 1]. The shape
829 // returned by this function is [2, 3, 1] (output_shape) however, the result
830 // shape of the XLA operation is [2, 3, 4] after degenerate-dimension
831 // broadcasting.
832 //
833 // Invalid broadcasts:
834 //
835 // smaller_shape=[3], larger_shape=[4, 3], broadcast_dimensions={0}
836 // Reason: Dimension zero** of larger_shape (size 4) is not compatible with
837 // dimension zero of smaller_shape(size 3). **Zero here comes from the value
838 // in broadcast_dimensions.
839 //
840 // smaller_shape=[2, 1], larger_shape=[2, 3, 4], broadcast_dimensions={1, 2}
841 // Reason: Dimension one of larger_shape (size 3) is not compatible with
842 // dimension zero of smaller_shape(size 2)
843
844 // The output shape is initially the larger_shape. Sizes of dimensions
845 // specified in broadcast_dimensions are then changed to match the
846 // corresponding dimension size in smaller_shape.
847 Shape output_shape(larger_shape);
848 output_shape.set_element_type(
849 ShapeUtil::HigherPrecisionElementType(larger_shape, smaller_shape));
850
851 for (int i = 0; i < smaller_shape.dimensions_size(); ++i) {
852 int64 dimension_to_match = broadcast_dimensions.at(i);
853 if (dimension_to_match < 0) {
854 return InvalidArgument(
855 "Broadcast dimension number (%lld) cannot be negative.",
856 dimension_to_match);
857 }
858 if (dimension_to_match >= larger_shape.dimensions_size()) {
859 return InvalidArgument(
860 "Broadcast dimension number (%lld) too large; higher-rank "
861 "operand has rank %d.",
862 dimension_to_match, larger_shape.dimensions_size());
863 }
864 int64 small_dimension_size = smaller_shape.dimensions(i);
865 int64 large_dimension_size = larger_shape.dimensions(dimension_to_match);
866 // Dimension sizes must be compatible: match or be degenerate (degenerate
867 // case is handled by degenerate dimension broadcasting which occurs after
868 // InDim broadcasting).
869 if (small_dimension_size != large_dimension_size &&
870 small_dimension_size != 1 && large_dimension_size != 1) {
871 return InvalidArgument(
872 "Broadcast dimension %d mismatch: %lld != %lld; %s and %s.", i,
873 small_dimension_size, large_dimension_size,
874 ShapeUtil::HumanString(smaller_shape).c_str(),
875 ShapeUtil::HumanString(larger_shape).c_str());
876 }
877 // Make sure the broadcast dimensions are listed in a strictly increasing
878 // order.
879 if (i > 0 && broadcast_dimensions.at(i - 1) >= dimension_to_match) {
880 return InvalidArgument(
881 "Broadcast dimensions order is wrong: %lld comes after %lld.",
882 dimension_to_match, broadcast_dimensions.at(i - 1));
883 }
884
885 output_shape.set_dimensions(dimension_to_match, small_dimension_size);
886 }
887
888 return output_shape;
889}

891/* static */ StatusOr<Shape> ShapeInference::InferElementwiseBinaryOpShape(
892 BinaryOperation operation, const Shape& lhs, const Shape& rhs,
893 tensorflow::gtl::ArraySlice<int64> broadcast_dimensions) {
894 TF_RETURN_IF_ERROR(
895 ExpectNotTupleOrOpaque(lhs, "lhs of elementwise binary operation"));
896 TF_RETURN_IF_ERROR(
897 ExpectNotTupleOrOpaque(rhs, "rhs of elementwise binary operation"));
898
899 if (!ShapeUtil::SameElementTypeIgnoringFpPrecision(lhs, rhs)) {
900 return InvalidArgument(
901 "Binary op %s with different element types: %s and %s.",
902 BinaryOperation_Name(operation).c_str(),
903 ShapeUtil::HumanString(lhs).c_str(),
904 ShapeUtil::HumanString(rhs).c_str());
905 }
906
907 if (ShapeUtil::Rank(lhs) == ShapeUtil::Rank(rhs)) {
908 std::vector<int64> identity_dims(ShapeUtil::Rank(lhs));
909 std::iota(identity_dims.begin(), identity_dims.end(), 0);
910 if (!broadcast_dimensions.empty() &&
911 broadcast_dimensions != identity_dims) {
912 return InvalidArgument(
913 "Broadcast dimensions field must either be not set or be the "
914 "identity on binary operations with operands of the same rank.");
915 }
916 }
917
918 if (ShapeUtil::CompatibleIgnoringFpPrecision(lhs, rhs)) {
919 // If the shapes are the same other than layout, the output shape is the
920 // same (elementwise op).
921 return ShapeUtil::ChangeElementType(
922 lhs, ShapeUtil::HigherPrecisionElementType(lhs, rhs));
923 }
924
925 if (ShapeUtil::Rank(lhs) == ShapeUtil::Rank(rhs)) {
926 return InferDegenerateDimensionBroadcastShape(operation, lhs, rhs);
927 } else {
928 // Ranks do not match, so perform InDim broadcasting using
929 // broadcast_dimensions. Scalar broadcasting is a special case of this.
930 const Shape& larger_shape =
931 ShapeUtil::Rank(lhs) > ShapeUtil::Rank(rhs) ? lhs : rhs;
932 const Shape& smaller_shape =
933 ShapeUtil::Rank(lhs) > ShapeUtil::Rank(rhs) ? rhs : lhs;
934
935 // After InDim broadcasting, perform degenerate dimensions broadcasting.
936 TF_ASSIGN_OR_RETURN(
937 Shape indim_broadcast_shape,
938 InferInDimBroadcastShape(operation, smaller_shape, larger_shape,
939 broadcast_dimensions));
940
941 return InferDegenerateDimensionBroadcastShape(
942 operation, indim_broadcast_shape, larger_shape);
943 }
944}

1103/* static */ StatusOr<Shape> ShapeInference::InferMapShape(
1104 tensorflow::gtl::ArraySlice<const Shape*> arg_shapes,
1105 const ProgramShape& to_apply,
1106 tensorflow::gtl::ArraySlice<int64> dimensions) {
1107 if (arg_shapes.empty()) {
1108 return InvalidArgument("Map expects at least one argument.");
1109 }
1110
1111 // All arguments must have the same shape.
1112 const Shape* arg_shape = arg_shapes[0];
1113 for (size_t i = 1; i < arg_shapes.size(); ++i) {
1114 TF_RETURN_IF_ERROR(
1115 ExpectNotTupleOrOpaque(*arg_shapes[i], "operand of map"));
1116
1117 if (ShapeUtil::CompatibleIgnoringFpPrecision(*arg_shapes[i], *arg_shape)) {
1118 continue;
1119 }
1120 if (!ShapeUtil::IsTuple(*arg_shapes[i]) &&
1121 !ShapeUtil::IsTuple(*arg_shape) &&
1122 ShapeUtil::SameElementTypeIgnoringFpPrecision(*arg_shapes[i],
1123 *arg_shape)) {
1124 if (ShapeUtil::IsScalar(*arg_shapes[i])) {
1125 continue;
1126 }
1127 if (ShapeUtil::IsScalar(*arg_shape)) {
1128 arg_shape = arg_shapes[i];
1129 continue;
1130 }
1131 }
1132
1133 std::vector<string> pieces;
1134 for (const Shape* shape : arg_shapes) {
1135 pieces.push_back(ShapeUtil::HumanString(*shape));
1136 }
1137 return InvalidArgument(
1138 "Map operation requires all operands to have the same shape; got: "
1139 "%s.",
1140 Join(pieces, ", ").c_str());
1141 }
1142
1143 // Check that dimensions.size == arg_shape.dimensions_size() (we currently
1144 // only support mapping across all dimensions: i.e. scalar map functions).
1145 if (dimensions.size() != arg_shape->dimensions_size()) {
1146 return InvalidArgument(
1147 "Map applied to a subset of dimensions currently not supported: "
1148 "arg_dimension_size: %d, requested_map_dimensions_size: %zu.",
1149 arg_shape->dimensions_size(), dimensions.size());
1150 }
1151
1152 // Check that requested map dimensions numbers are monotonically increasing.
1153 for (int i = 0; i < dimensions.size(); ++i) {
1154 if (dimensions[i] != i) {
1155 return InvalidArgument(
1156 "Map requires monotonically increasing dimension numbers; got: %s.",
1157 Join(dimensions, ", ").c_str());
1158 }
1159 }
1160
1161 // The applied function's arity equals the number of arguments.
1162 if (arg_shapes.size() != to_apply.parameters_size()) {
1163 return InvalidArgument(
1164 "Map applied function arity must match number of arguments; got: "
1165 "arity: %d, arguments: %zu.",
1166 to_apply.parameters_size(), arg_shapes.size());
1167 }
1168
1169 // The parameters should all be scalars, and the output too.
1170 const Shape& output_shape = to_apply.result();
1171 if (!ShapeUtil::IsScalar(output_shape)) {
1172 return InvalidArgument(
1173 "Mapped computation's result has to be a scalar; got: %s.",
1174 ShapeUtil::HumanString(output_shape).c_str());
1175 }
1176
1177 for (int i = 0; i < to_apply.parameters_size(); ++i) {
1178 const Shape& parameter_shape = to_apply.parameters(i);
1179
1180 if (!ShapeUtil::IsScalar(parameter_shape)) {
1181 return InvalidArgument(
1182 "Mapped computation's parameter has to be a scalar; "
1183 "got parameter %d shape: %s.",
1184 i, ShapeUtil::HumanString(parameter_shape).c_str());
1185 }
1186
1187 if (!ShapeUtil::SameElementTypeIgnoringFpPrecision(parameter_shape,
1188 *arg_shape)) {
1189 return InvalidArgument(
1190 "Mapped computation's parameter type has to match argument element "
1191 "type; got parameter %d shape: %s, argument shape: %s.",
1192 i, ShapeUtil::HumanString(parameter_shape).c_str(),
1193 ShapeUtil::HumanString(*arg_shape).c_str());
1194 }
1195 }
1196
1197 return ShapeUtil::MakeShape(output_shape.element_type(),
1198 AsInt64Slice(arg_shape->dimensions()));
1199}

1613/* static */ StatusOr<Shape> ShapeInference::InferConvolveShape(
1614 const Shape& lhs, const Shape& rhs, const Window& window,
1615 const ConvolutionDimensionNumbers& dnums) {
1616 TF_RETURN_IF_ERROR(ExpectNotTupleOrOpaque(lhs, "lhs of convolution"));
1617 TF_RETURN_IF_ERROR(ExpectNotTupleOrOpaque(rhs, "rhs of convolution"));
1618
1619 if (!ShapeUtil::SameElementTypeIgnoringFpPrecision(lhs, rhs)) {
1620 return InvalidArgument(
1621 "Convolution with different element types: %s and %s.",
1622 ShapeUtil::HumanString(lhs).c_str(),
1623 ShapeUtil::HumanString(rhs).c_str());
1624 }
1625 if (dnums.input_spatial_dimensions_size() !=
1626 dnums.kernel_spatial_dimensions_size()) {
1627 return InvalidArgument(
1628 "Both arguments to convolution must have same number of dimensions.\n"
1629 "Window: %s",
1630 window.DebugString().c_str());
1631 }
1632
1633 const int num_spatial_dims = dnums.input_spatial_dimensions_size();
1634 if (window.dimensions_size() != num_spatial_dims) {
1635 return InvalidArgument(
1636 "Window must have same number of dimensions as dimension numbers.\n"
1637 "Window: %s\nDimension numbers: %s.",
1638 window.DebugString().c_str(), dnums.DebugString().c_str());
1639 }
1640
1641 const int num_dims = num_spatial_dims + 2;
1642 if (ShapeUtil::Rank(lhs) != num_dims) {
1643 return InvalidArgument(
1644 "The LHS argument to a convolution should have rank %d; lhs: %s.",
1645 num_dims, ShapeUtil::HumanString(lhs).c_str());
1646 }
1647 if (ShapeUtil::Rank(rhs) != num_dims) {
1648 return InvalidArgument(
1649 "The RHS argument to a convolution should have rank %d; lhs: %s.",
1650 num_dims, ShapeUtil::HumanString(lhs).c_str());
1651 }
1652 TF_DCHECK_OK(ShapeUtil::ValidateShapeWithOptionalLayout(lhs));
1653 TF_DCHECK_OK(ShapeUtil::ValidateShapeWithOptionalLayout(rhs));
1654
1655 // Verifies that the input and window dimensions are a permutation of
1656 // the dimension numbers.
1657 std::vector<int64> input_dnums(num_dims);
1658 input_dnums[0] = dnums.input_batch_dimension();
1659 input_dnums[1] = dnums.input_feature_dimension();
1660 std::copy(dnums.input_spatial_dimensions().begin(),
1661 dnums.input_spatial_dimensions().end(), input_dnums.begin() + 2);
1662 std::sort(input_dnums.begin(), input_dnums.end());
1663
1664 std::vector<int64> window_dnums(num_dims);
1665 window_dnums[0] = dnums.kernel_input_feature_dimension();
1666 window_dnums[1] = dnums.kernel_output_feature_dimension();
1667 std::copy(dnums.kernel_spatial_dimensions().begin(),
1668 dnums.kernel_spatial_dimensions().end(), window_dnums.begin() + 2);
1669 std::sort(window_dnums.begin(), window_dnums.end());
1670
1671 std::vector<int64> output_dnums(num_dims);
1672 output_dnums[0] = dnums.output_batch_dimension();
1673 output_dnums[1] = dnums.output_feature_dimension();
1674 std::copy(dnums.output_spatial_dimensions().begin(),
1675 dnums.output_spatial_dimensions().end(), output_dnums.begin() + 2);
1676 std::sort(output_dnums.begin(), output_dnums.end());
1677
1678 std::vector<int64> expected_dnums(num_dims);
1679 std::iota(expected_dnums.begin(), expected_dnums.end(), 0);
1680
1681 const auto in_range = [num_dims](int64 i) { return 0 <= i && i < num_dims; };
1682 if (!std::all_of(input_dnums.begin(), input_dnums.end(), in_range) ||
1683 !std::all_of(window_dnums.begin(), window_dnums.end(), in_range) ||
1684 !std::all_of(output_dnums.begin(), output_dnums.end(), in_range)) {
1685 return InvalidArgument(
1686 "A dimension number is out of range in convolution: %s.",
1687 dnums.DebugString().c_str());
1688 }
1689
1690 if (input_dnums != expected_dnums) {
1691 return InvalidArgument(
1692 "Input dimensions of convolution must contain each dimension exactly "
1693 "once: %s.",
1694 dnums.DebugString().c_str());
1695 }
1696 if (window_dnums != expected_dnums) {
1697 return InvalidArgument(
1698 "Window dimensions of convolution must contain each dimension exactly "
1699 "once: %s.",
1700 dnums.DebugString().c_str());
1701 }
1702 if (output_dnums != expected_dnums) {
1703 return InvalidArgument(
1704 "Output dimensions of convolution must contain each dimension exactly "
1705 "once: %s.",
1706 dnums.DebugString().c_str());
1707 }
1708
1709 std::vector<int64> input_spatial_dims(num_spatial_dims);
1710 for (int i = 0; i < num_spatial_dims; ++i) {
1711 input_spatial_dims[i] = lhs.dimensions(dnums.input_spatial_dimensions(i));
1712 }
1713 const int64 input_features = lhs.dimensions(dnums.input_feature_dimension());
1714 const int64 input_batch = lhs.dimensions(dnums.input_batch_dimension());
1715
1716 std::vector<int64> kernel_spatial_dims(num_spatial_dims);
1717 for (int i = 0; i < num_spatial_dims; ++i) {
1718 kernel_spatial_dims[i] = rhs.dimensions(dnums.kernel_spatial_dimensions(i));
1719 }
1720 const int64 kernel_input_features =
1721 rhs.dimensions(dnums.kernel_input_feature_dimension());
1722 const int64 kernel_output_features =
1723 rhs.dimensions(dnums.kernel_output_feature_dimension());
1724
1725 if (input_features != kernel_input_features) {
1726 return InvalidArgument(
1727 "Expected LHS feature dimension (value %lld) to match RHS "
1728 "input feature dimension (value %lld); got <conv>(%s, %s)\n"
1729 "Dimension numbers: {%s}.",
1730 input_features, kernel_input_features,
1731 ShapeUtil::HumanString(lhs).c_str(),
1732 ShapeUtil::HumanString(rhs).c_str(), dnums.DebugString().c_str());
1733 }
1734 std::vector<int64> window_dims(num_spatial_dims);
1735 for (int i = 0; i < num_spatial_dims; ++i) {
1736 window_dims[i] = window.dimensions(i).size();
1737 }
1738 if (kernel_spatial_dims != window_dims) {
1739 return InvalidArgument(
1740 "Window dimensions do not match RHS shape:\n\t"
1741 "RHS shape: %s\n\t"
1742 "Window: {%s}\n\t"
1743 "Dimension numbers: {%s}.",
1744 ShapeUtil::HumanString(rhs).c_str(), window.ShortDebugString().c_str(),
1745 dnums.ShortDebugString().c_str());
1746 }
1747
1748 Shape base_shape =
1749 ShapeUtil::MakeShape(lhs.element_type(), input_spatial_dims);
1750 TF_ASSIGN_OR_RETURN(
1751 Shape window_output_shape,
1752 InferWindowOutputShape(base_shape, window, lhs.element_type(),
1753 /*allow_negative_padding=*/true));

1766/* static */ StatusOr<Shape> ShapeInference::InferFftShape(
1767 const Shape& in, const FftType fft_type,
1768 const tensorflow::gtl::ArraySlice<int64> fft_length) {
1769 const int64 fft_rank = fft_length.size();
1770 if (fft_rank < 1 || fft_rank > 3) {
1771 return InvalidArgument("FFT only supports ranks 1-3; got %lld.", fft_rank);
1772 }
1773#define RET_CHECK_RANK(x) \
1774 if (x.dimensions_size() < fft_rank) { \
1775 return InvalidArgument( \
1776 "FFT of rank %lld requires input of at least " \
1777 "same rank; got input of rank %d", \
1778 fft_rank, x.dimensions_size()); \
1779 }
1780 switch (fft_type) {
1781 case FFT:
1782 case IFFT:
1783 if (in.element_type() != C64) {
1784 return InvalidArgument("%s requires C64 input type, found %s.",
1785 FftType_Name(fft_type).c_str(),
1786 PrimitiveType_Name(in.element_type()).c_str());
1787 }
1788 RET_CHECK_RANK(in);
1789 return in;
1790 case RFFT: {
1791 if (in.element_type() != F32) {
1792 return InvalidArgument("RFFT requires F32 input type, found %s.",
1793 PrimitiveType_Name(in.element_type()).c_str());
1794 }
1795 RET_CHECK_RANK(in);
1796 for (int i = 0; i < fft_rank; i++) {
1797 if (in.dimensions(in.dimensions_size() - fft_rank + i) !=
1798 fft_length[i]) {
1799 return InvalidArgument(
1800 "RFFT requires innermost dimensions match fft_length but "
1801 "dimension %lld is %lld and should be %lld.",
1802 in.dimensions_size() - fft_rank + i,
1803 in.dimensions(in.dimensions_size() - fft_rank + i),
1804 fft_length[i]);
1805 }
1806 }
1807 Shape result = ShapeUtil::ChangeElementType(in, C64);
1808 result.set_dimensions(result.dimensions_size() - 1,
1809 fft_length[fft_rank - 1] / 2 + 1);
1810 return result;
1811 }
1812 case IRFFT: {
1813 if (in.element_type() != C64) {
1814 return InvalidArgument("IRFFT requires C64 input type, found %s.",
1815 PrimitiveType_Name(in.element_type()).c_str());
1816 }
1817 RET_CHECK_RANK(in);
1818 Shape result = ShapeUtil::ComplexComponentShape(in);
1819 for (int i = 0; i < fft_rank - 1; i++) {
1820 if (in.dimensions(in.dimensions_size() - fft_rank + i) !=
1821 fft_length[i]) {
1822 return InvalidArgument(
1823 "IRFFT requires all but one innermost dimensions match "
1824 "fft_length, but dimension %lld is %lld and should be %lld.",
1825 in.dimensions_size() - fft_rank + i,
1826 in.dimensions(in.dimensions_size() - fft_rank + i),
1827 fft_length[i]);
1828 }
1829 }
1830 if (in.dimensions(in.dimensions_size() - 1) !=
1831 fft_length[fft_rank - 1] / 2 + 1) {
1832 return InvalidArgument(
1833 "IRFFT requires innermost dimension matches fft_length/2+1, but "
1834 "dimension %d is %lld and should be %lld.",
1835 in.dimensions_size() - 1, in.dimensions(in.dimensions_size() - 1),
1836 fft_length[fft_rank - 1] / 2 + 1);
1837 }
1838 result.set_dimensions(result.dimensions_size() - 1,
1839 fft_length[fft_rank - 1]);
1840 return result;
1841 }
1842 default:
1843 LOG(FATAL) << "Unexpected fft_type: " << fft_type;
1844 }
1845#undef RET_CHECK_RANK
1846}

1961/* static */ StatusOr<Shape> ShapeInference::InferSliceShape(
1962 const Shape& arg, tensorflow::gtl::ArraySlice<int64> starts,
1963 tensorflow::gtl::ArraySlice<int64> limits,
1964 tensorflow::gtl::ArraySlice<int64> strides) {
1965 auto error = [&](const string& message) {
1966 return InvalidArgument(
1967 "%s in slice operation; argument shape: %s; starts: {%s}; limits: "
1968 "{%s}; strides: {%s}.",
1969 message.c_str(), ShapeUtil::HumanString(arg).c_str(),
1970 Join(starts, ",").c_str(), Join(limits, ",").c_str(),
1971 Join(strides, ",").c_str());
1972 };
1973 TF_RETURN_IF_ERROR(ExpectNotTupleOrOpaque(arg, "operand of slice"));
1974 VLOG(2) << tensorflow::strings::Printf(
1975 "slicing shape %s starts={%s} limits={%s}",
1976 ShapeUtil::HumanString(arg).c_str(), Join(starts, ", ").c_str(),
1977 Join(limits, ", ").c_str());
1978
1979 if (starts.size() != limits.size()) {
1980 return error(Printf("slice start and limit sizes differ: %zu vs %zu",
1981 starts.size(), limits.size()));
1982 }
1983
1984 if (starts.size() != strides.size()) {
1985 return error(Printf("slice start and strides sizes differ: %zu vs %zu",
1986 starts.size(), strides.size()));
1987 }
1988

1990 return InvalidArgument(
1991 "Slice index count does not match argument rank: %zu vs %lld.",

1996 for (int64 dimension = 0; dimension < starts.size(); ++dimension) {
1997 int64 start_index = starts[dimension];

1999 int64 stride = strides[dimension];
2000 if (start_index < 0) {
2001 return InvalidArgument("Negative start index to slice: %lld.",
2002 start_index);
2003 }

2007 "size (%lld)",
2008 limit_index, arg.dimensions(dimension)));
2009 }
2010 VLOG(2) << tensorflow::strings::Printf("starts[%lld] = %lld", dimension,
2011 start_index);
2012 VLOG(2) << tensorflow::strings::Printf("limits[%lld] = %lld", dimension,
2013 limit_index);

2015 return error(
2016 Printf("limit index (%lld) must be greater or equal to "
2017 "start index (%lld) in slice with positive stride",
2018 limit_index, start_index));
2019 }
2020 if (stride <= 0) {
2021 return InvalidArgument("Stride (%lld) must be positive.", stride);
2022 }
2023 sizes.push_back((limit_index - start_index + stride - 1) / stride);
2024 }
2025
2026 return ShapeUtil::MakeShape(arg.element_type(), sizes);

2322 if (ShapeUtil::ElementsIn(operand) != ShapeUtil::ElementsIn(inferred_shape)) {
2323 return InvalidArgument(
2324 "Reshape operation has mismatched element counts: from=%lld (%s) "
2325 "to=%lld (%s).",
2326 ShapeUtil::ElementsIn(operand), ShapeUtil::HumanString(operand).c_str(),
2327 ShapeUtil::ElementsIn(inferred_shape),
2328 ShapeUtil::HumanString(inferred_shape).c_str());
2329 }
2330
2331 std::vector<int64> indices(ShapeUtil::Rank(operand));
2332 std::iota(indices.begin(), indices.end(), 0);
2333 if (dimensions.size() != ShapeUtil::Rank(operand) ||
2334 !std::is_permutation(dimensions.begin(), dimensions.end(),
2335 indices.begin())) {
2336 return InvalidArgument(
2337 "Reshape dimensions [%s] are not a permutation of the operand "
2338 "dimensions (operand shape is %s).",
2339 Join(dimensions, ",").c_str(), ShapeUtil::HumanString(operand).c_str());
2340 }
2341
2342 return inferred_shape;
2343}

2346 const Shape& operand, tensorflow::gtl::ArraySlice<int64> dimensions) {

2348
2349 std::vector<int64> indices(ShapeUtil::Rank(operand));
2350 std::iota(indices.begin(), indices.end(), 0);
2351 if (dimensions.size() != ShapeUtil::Rank(operand) ||
2352 !std::is_permutation(dimensions.begin(), dimensions.end(),
2353 indices.begin())) {
2354 return InvalidArgument(
2355 "Transpose dimensions not a permutation of the operand dimensions.");
2356 }

2359 // we need output[i]=input[dimensions[i]] which is
2360 // Permute(Inverse(dimensions),input).

2443 string argument_shapes =

2453 }

2460 return InvalidArgument(
2461 "Call parameter must match argument; got parameter %d shape: %s, "

2464 ShapeUtil::HumanString(arg_shape).c_str());
2465 }
2466 }

2471static Status ValidateGatherDimensionNumbers(
2472 const Shape& input_shape,
2473 tensorflow::gtl::ArraySlice<int64> gather_indices_shape,
2474 const GatherDimensionNumbers& dim_numbers) {
2475 if (!c_is_sorted(dim_numbers.output_window_dims())) {
2476 return InvalidArgument(
2477 "Output window dimensions in gather op must be ascending; got: %s.",
2478 Join(dim_numbers.output_window_dims(), ", ").c_str());
2479 }
2480
2481 if (c_adjacent_find(dim_numbers.output_window_dims()) !=
2482 dim_numbers.output_window_dims().end()) {
2483 return InvalidArgument(
2484 "Output window dimensions in gather op must not repeat; got: %s.",
2485 Join(dim_numbers.output_window_dims(), ", ").c_str());
2486 }
2487
2488 const int64 output_window_dim_count = dim_numbers.output_window_dims_size();
2489 const int64 output_shape_rank =
2490 output_window_dim_count + gather_indices_shape.size() - 1;
2491
2492 for (int i = 0; i < dim_numbers.output_window_dims_size(); ++i) {
2493 int64 window_index = dim_numbers.output_window_dims(i);
2494 if (window_index < 0 || window_index >= output_shape_rank) {
2495 return InvalidArgument(
2496 "Window index %d in gather op is out of bounds; got %lld, but should "
2497 "have been in [0,%lld).",
2498 i, window_index, output_shape_rank);
2499 }
2500 }
2501
2502 if (dim_numbers.gather_dims_to_operand_dims_size() !=
2503 gather_indices_shape[dim_numbers.index_vector_dim()]) {
2504 return InvalidArgument(
2505 "Gather op has %d elements in gather_dims_to_operand_dims and the "
2506 "bound of dimension index_vector_dim=%lld of gather_indices is "
2507 "%lld. These two numbers must be equal.",
2508 dim_numbers.gather_dims_to_operand_dims_size(),
2509 dim_numbers.index_vector_dim(),
2510 gather_indices_shape[dim_numbers.index_vector_dim()]);
2511 }
2512
2513 for (int i = 0; i < dim_numbers.gather_dims_to_operand_dims_size(); i++) {
2514 int64 gather_dim_to_input_dim = dim_numbers.gather_dims_to_operand_dims(i);
2515 if (gather_dim_to_input_dim < 0 ||
2516 gather_dim_to_input_dim >= input_shape.dimensions_size()) {
2517 return InvalidArgument(
2518 "Invalid gather_dims_to_operand_dims mapping; domain is [0, %d), "
2519 "got: %d->%lld.",
2520 input_shape.dimensions_size(), i, gather_dim_to_input_dim);
2521 }
2522 }
2523
2524 std::vector<int64> sorted_gather_dims_to_operand_dims(
2525 dim_numbers.gather_dims_to_operand_dims().begin(),
2526 dim_numbers.gather_dims_to_operand_dims().end());
2527
2528 c_sort(sorted_gather_dims_to_operand_dims);
2529
2530 if (c_adjacent_find(sorted_gather_dims_to_operand_dims) !=
2531 sorted_gather_dims_to_operand_dims.end()) {
2532 return InvalidArgument(
2533 "Repeated dimensions are not allowed in gather_dims_to_operand_dims; "
2534 "got: %s.",
2535 Join(dim_numbers.gather_dims_to_operand_dims(), ", ").c_str());
2536 }
2537
2538 for (int64 elided_dim : dim_numbers.elided_window_dims()) {
2539 if (elided_dim < 0 || elided_dim >= input_shape.dimensions_size()) {
2540 return InvalidArgument(
2541 "Invalid elided_window_dims set in gather op; valid range is [0, "
2542 "%d), got: %lld.",
2543 input_shape.dimensions_size(), elided_dim);
2544 }
2545 }
2546
2547 if (!c_is_sorted(dim_numbers.elided_window_dims())) {
2548 return InvalidArgument(
2549 "elided_window_dims in gather op must be sorted; got: %s",
2550 Join(dim_numbers.elided_window_dims(), ", ").c_str());
2551 }
2552
2553 if (c_adjacent_find(dim_numbers.elided_window_dims()) !=
2554 dim_numbers.elided_window_dims().end()) {
2555 return InvalidArgument(
2556 "Repeated dimensions not allowed in elided_window_dims in gather op; "
2557 "got: %s.",
2558 Join(dim_numbers.elided_window_dims(), ", ").c_str());
2559 }
2560
2561 return Status::OK();
2562}

2564/*static*/ StatusOr<Shape> ShapeInference::InferGatherShape(
2565 const Shape& input_shape, const Shape& gather_indices_shape,
2566 const GatherDimensionNumbers& gather_dim_numbers,
2567 tensorflow::gtl::ArraySlice<int64> window_bounds) {
2568 TF_RETURN_IF_ERROR(
2569 ExpectNotTupleOrOpaque(input_shape, "input tensor operand gather op"));
2570 TF_RETURN_IF_ERROR(ExpectNotTupleOrOpaque(
2571 gather_indices_shape, "gather indices operand of gather op"));
2572
2573 if (!ShapeUtil::ElementIsIntegral(gather_indices_shape)) {
2574 return InvalidArgument(
2575 "Gather indices parameter must be an integral tensor; got %s.",
2576 ShapeUtil::HumanString(gather_indices_shape).c_str());
2577 }
2578
2579 // We implicitly reshape gather indices of shape P[A,B,C] to P[A,B,C,1] if
2580 // index_vector_dim is rank(P). The bounds of this expanded shape is
2581 // stored in expanded_gather_indices_shape.
2582
2583 if (gather_indices_shape.dimensions_size() <
2584 gather_dim_numbers.index_vector_dim() ||
2585 gather_dim_numbers.index_vector_dim() < 0) {
2586 return InvalidArgument(
2587 "Gather index leaf dimension must be within [0, rank(gather_indices) + "
2588 "1). rank(gather_indices) is %d and gather index leaf dimension is "
2589 "%lld.",
2590 gather_indices_shape.dimensions_size(),
2591 gather_dim_numbers.index_vector_dim());
2592 }
2593
2594 std::vector<int64> expanded_gather_indices_shape;
2595 expanded_gather_indices_shape.reserve(gather_indices_shape.dimensions_size());
2596 c_copy(gather_indices_shape.dimensions(),
2597 std::back_inserter(expanded_gather_indices_shape));
2598 if (expanded_gather_indices_shape.size() ==
2599 gather_dim_numbers.index_vector_dim()) {
2600 expanded_gather_indices_shape.push_back(1);
2601 }
2602
2603 TF_RETURN_IF_ERROR(ValidateGatherDimensionNumbers(
2604 input_shape, expanded_gather_indices_shape, gather_dim_numbers));
2605
2606 if (window_bounds.size() != input_shape.dimensions_size()) {
2607 return InvalidArgument(
2608 "Gather op must have one window bound for every input dimension; got: "
2609 "len(window_bounds)=%lu, input_shape.rank=%d.",
2610 window_bounds.size(), input_shape.dimensions_size());
2611 }
2612
2613 if (window_bounds.size() !=
2614 gather_dim_numbers.output_window_dims_size() +
2615 gather_dim_numbers.elided_window_dims_size()) {
2616 return InvalidArgument(
2617 "All components of the window index in a gather op must either be a "
2618 "output window index or explicitly elided; got len(window_bounds)=%lu, "
2619 "output_window_bounds=%s, elided_window_bounds=%s.",
2620 window_bounds.size(),
2621 Join(gather_dim_numbers.output_window_dims(), ",").c_str(),
2622 Join(gather_dim_numbers.elided_window_dims(), ",").c_str());
2623 }
2624
2625 for (int i = 0; i < window_bounds.size(); i++) {
2626 int64 window_bound = window_bounds[i];
2627 int64 corresponding_input_bound = input_shape.dimensions(i);
2628 if (window_bound < 0 || window_bound > corresponding_input_bound) {
2629 return InvalidArgument(
2630 "Window bound at index %d in gather op is out of range, must be "
2631 "within "
2632 "[0, %lld), got %lld.",
2633 i, corresponding_input_bound + 1, window_bound);
2634 }
2635 }
2636
2637 for (int i = 0; i < gather_dim_numbers.elided_window_dims_size(); i++) {
2638 if (window_bounds[gather_dim_numbers.elided_window_dims(i)] != 1) {
2639 return InvalidArgument(
2640 "Gather op can only elide window indices with bound 1, but bound is "
2641 "%lld for index %lld at position %d.",
2642 window_bounds[gather_dim_numbers.elided_window_dims(i)],
2643 gather_dim_numbers.elided_window_dims(i), i);
2644 }
2645 }
2646
2647 int64 result_rank = gather_dim_numbers.output_window_dims_size() +
2648 (expanded_gather_indices_shape.size() - 1);
2649 int64 window_dims_seen = 0;
2650 int64 gather_dims_seen = 0;
2651 std::vector<int64> output_dim_bounds;
2652 output_dim_bounds.reserve(result_rank);
2653 for (int64 i = 0; i < result_rank; i++) {
2654 int64 current_bound;
2655 bool is_window_index =
2656 c_binary_search(gather_dim_numbers.output_window_dims(), i);
2657 if (is_window_index) {
2658 while (c_binary_search(gather_dim_numbers.elided_window_dims(),
2659 window_dims_seen)) {
2660 window_dims_seen++;
2661 }
2662 current_bound = window_bounds[window_dims_seen++];
2663 } else {
2664 if (gather_dims_seen == gather_dim_numbers.index_vector_dim()) {
2665 gather_dims_seen++;
2666 }
2667 current_bound = expanded_gather_indices_shape[gather_dims_seen++];
2668 }
2669
2670 output_dim_bounds.push_back(current_bound);
2671 }
2672
2673 return ShapeUtil::MakeShape(input_shape.element_type(), output_dim_bounds);
2674}
