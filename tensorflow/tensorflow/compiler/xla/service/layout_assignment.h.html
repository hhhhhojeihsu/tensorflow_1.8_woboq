46// Abstract base class for layout constraints. These constraint objects are
47// gathered together in LayoutConstraints object.
48class LayoutConstraint {
49 public:
50 LayoutConstraint(bool mandatory, bool dfs)
51 : mandatory_(mandatory), dfs_(dfs) {}
52 virtual ~LayoutConstraint() = default;
53
54 virtual string ToString() const = 0;
55
56 // True if this constraint cannot be overwritten by a different constraint.
57 bool mandatory() const { return mandatory_; }
58
59 // When true, propagate in DFS. When false, constraint will propagate in BFS.
60 bool dfs() const { return dfs_; }
61
62 private:
63 bool mandatory_;
64 bool dfs_;
65};
66
67std::ostream& operator<<(std::ostream& out, const LayoutConstraint& constraint);

69// Layout constraint on a single LogicalBuffer. This constrains the layout of an
70// array produced by a particular instruction.
71class BufferLayoutConstraint : public LayoutConstraint {
72 public:
73 BufferLayoutConstraint(const Layout& layout, const LogicalBuffer& buffer,
74 bool mandatory, bool dfs);
75
76 const LogicalBuffer& buffer() const { return *buffer_; }
77 const Layout& layout() const { return layout_; }
78
79 string ToString() const override;
80
81 private:
82 Layout layout_;
83 const LogicalBuffer* buffer_;
84};
85
86// Constraint on the layout of the operand of an instruction. The constrained
87// shape can be arbitrarily shaped (array or tuple). This is a constraint on the
88// use of a shaped value and is not a hard constraint on the instruction(s)
89// which define the value as copies may be inserted between the definition and
90// use.
91class OperandLayoutConstraint : public LayoutConstraint {
92 public:
93 OperandLayoutConstraint(const ShapeLayout& shape_layout,
94 const HloInstruction* instruction, int64 operand_no,
95 bool mandatory, bool dfs);
96
97 const ShapeLayout& shape_layout() const { return shape_layout_; }
98 const HloInstruction* instruction() const { return instruction_; }
99 const int64 operand_no() const { return operand_no_; }
100 const HloInstruction* operand() const {
101 return instruction_->operand(operand_no_);
102 }
103
104 string ToString() const override;
105
106 private:
107 ShapeLayout shape_layout_;
108 const HloInstruction* instruction_;
109 int64 operand_no_;
110};
111
112// Constraint on the layout of the result of the entry computation.
113class ResultLayoutConstraint : public LayoutConstraint {
114 public:
115 explicit ResultLayoutConstraint(const ShapeLayout& shape_layout,
116 bool dfs = false)
117 : LayoutConstraint(/*mandatory=*/true, dfs),
118 shape_layout_(shape_layout) {}
119
120 const ShapeLayout& shape_layout() const { return shape_layout_; }
121 string ToString() const override;
122
123 private:
124 const ShapeLayout shape_layout_;
125};

127// Class encapsulating the layout constraints of the values in a HLO
128// computation.
129class LayoutConstraints {
130 public:
131 LayoutConstraints(const TuplePointsToAnalysis& points_to_analysis,
132 HloComputation* computation);
133 ~LayoutConstraints() = default;
134
135 const HloComputation* computation() const { return computation_; }
136 HloComputation* computation() { return computation_; }
137 const TuplePointsToAnalysis& points_to_analysis() const {
138 return points_to_analysis_;
139 }
140
141 // Return a vector containing the constraints which have been added to the
142 // LayoutConstraints object since the construction of the object or since the
143 // last time ConsumeAddedConstraints() has been called. This is used to
144 // identify newly added constraints when propagating layouts.
145 std::vector<const LayoutConstraint*> ConsumeAddedConstraints() {
146 std::vector<const LayoutConstraint*> ret_vec(std::move(added_constraints_));
147 added_constraints_.clear();
148 return ret_vec;
149 }
150 void ClearAddedConstraints() { added_constraints_.clear(); }
151
152 // Returns the layout of a LogicalBuffer, the layout of the operand of the
153 // instruction, or the layout of the result of the computation, respectively,
154 // if it has been constrained. Otherwise return nullptr.
155 const Layout* BufferLayout(const LogicalBuffer& buffer) const;
156 const BufferLayoutConstraint* GetBufferLayoutConstraint(
157 const LogicalBuffer& buffer) const;
158 const ShapeLayout* OperandLayout(const HloInstruction* instruction,
159 int64 operand_no) const;
160 const OperandLayoutConstraint* GetOperandLayoutConstraint(
161 const HloInstruction* instruction, int64 operand_no) const;
162 const ShapeLayout* ResultLayout() const;
163
164 // Add a constraint on the layout of a LogicalBuffer, the layout of the
165 // operand of the instruction, or the layout of the result of the computation,
166 // respectively.
167 Status SetBufferLayout(const Layout& layout, const LogicalBuffer& buffer,
168 bool mandatory = true, bool dfs = true);
169 Status SetOperandLayout(const Shape& shape_with_layout,
170 const HloInstruction* instruction, int64 operand_no,
171 bool mandatory = true, bool dfs = true);
172 Status SetResultLayout(const Shape& shape_with_layout, bool dfs = true);
173
174 // Convenience wrapper around SetOperandLayout for setting the layout of a
175 // operand using a Layout object. The operand must be array-shaped.
176 Status SetArrayOperandLayout(const Layout& layout,
177 const HloInstruction* instruction,
178 int64 operand_no, bool mandatory = true,
179 bool dfs = true);
180
181 // Convenience wrapper around SetBufferLayout. Sets the layouts of all buffers
182 // created by the instruction to the layouts in the given shape. The
183 // instruction must define every logical buffer in its output.
184 Status SetInstructionLayout(const Shape& shape_with_layout,
185 const HloInstruction* instruction,
186 bool mandatory = true, bool dfs = true);
187
188 // Returns true if any buffer in the given operand is forwarded to the output
189 // of the given instruction. For example, the Tuple instruction forwards the
190 // buffers of its operands and would return true for each of its operands.
191 bool OperandBufferForwarded(const HloInstruction* instruction,
192 int64 operand_no) const;
193
194 // Returns the set of logical buffers (by LogicalBuffer:Id) which do not
195 // yet have a layout constraint
196 const std::set<LogicalBuffer::Id>& unconstrained_buffer_ids() const {
197 return unconstrained_buffer_ids_;
198 }
199
200 string ToString() const;
201
202 private:
203 // Find a bufferset in the bufferset cache. This is useful since we can
204 // currently create the flattened buffer set for the same instruction many
205 // times, which is often slow.
206 PointsToSet::BufferSet* GetBufferSet(const HloInstruction* instruction) const;
207
208 // The set of BufferLayoutConstraints applied to the computation.
209 std::unordered_map<const LogicalBuffer*, BufferLayoutConstraint>
210 buffer_constraints_;
211
212 // The set of OperandLayoutConstraints applied to the computation.
213 using OperandConstraintKey = std::pair<const HloInstruction*, int64>;
214 std::map<OperandConstraintKey, OperandLayoutConstraint> operand_constraints_;
215
216 // The result constraint for the computation (can be null).
217 std::unique_ptr<ResultLayoutConstraint> result_constraint_;
218
219 // A vector which holds constraints as they are added. Can be cleared with
220 // ClearAddedConstraints.
221 std::vector<const LayoutConstraint*> added_constraints_;
222
223 // Points-to analysis for the module. Used to propagate constraints through
224 // the HLO graph.
225 const TuplePointsToAnalysis& points_to_analysis_;
226
227 // Array-shaped buffers which have not yet been constrained.
228 std::set<LogicalBuffer::Id> unconstrained_buffer_ids_;
229
230 mutable tensorflow::gtl::FlatMap<const HloInstruction*,
231 std::unique_ptr<PointsToSet::BufferSet>>
232 buffer_sets_cache_;
233
234 HloComputation* computation_;
235};

237// Contains constraints on the layout of channels; sends and recvs.
238class ChannelLayoutConstraints {
239 public:
240 // Construct an empty constraint set.
241 ChannelLayoutConstraints() {}
242
243 // Returns true if channel_id has a layout constraint.
244 bool IsChannelConstrained(int64 channel_id) const {
245 return constraints_.count(channel_id) > 0;
246 }
247
248 // Given `shape`, apply the layout for `channel_id`. `channel_id` must already
249 // be constrained.
250 Shape LayoutShapeForChannel(Shape shape, int64 channel_id) const {
251 CHECK(IsChannelConstrained(channel_id));
252 *shape.mutable_layout() = constraints_.at(channel_id);
253 return shape;
254 }
255
256 // Returns the layout constraint for `channel_id`, which must already be
257 // constrained.
258 Layout LayoutForChannel(int64 channel_id) const {
259 CHECK(IsChannelConstrained(channel_id));
260 return constraints_.at(channel_id);
261 }
262
263 // Adds a new layout constraint for `channel_id`. If a constraint for
264 // `channel_id` already exists, this operation requires that the new layout is
265 // the same as the previously constrained layout.
266 void ConstrainChannel(int64 channel_id, const Layout& layout) {
267 CHECK(!IsChannelConstrained(channel_id) ||
268 LayoutUtil::Equal(layout, constraints_[channel_id]));
269 constraints_[channel_id] = layout;
270 }
271
272 private:
273 std::unordered_map<int64, Layout> constraints_;
274};

276// HLO pass which assigns layouts to all instructions in the HLO module while
277// satisfying all necessary invariants and minimizing cost.
278class LayoutAssignment : public HloPassInterface {
279 public:
280 // entry_computation_layout is modified to populate a layout for the result in
281 // the case that no particular layout is requested.
282 //
283 // channel_constraints is both an input and output. Any sends or recvs that
284 // are present in channel_constraints will be layed out as constrained. Any
285 // unconstrained sends or recvs will be layed out as locally optimal and their
286 // layout will be added as a constraint to channel_constraints.
287 //
288 // If channel_constraints is nullptr, no kSend or kRecvs must be contained
289 // within any module passed to `Run`.
290 explicit LayoutAssignment(
291 ComputationLayout* entry_computation_layout,
292 ChannelLayoutConstraints* channel_constraints = nullptr);
293 ~LayoutAssignment() override {}
294 tensorflow::StringPiece name() const override { return "layout-assignment"; }
295
296 // Assign layouts to the given module. Returns whether the module was changed
297 // (any layouts were changed).
298 StatusOr<bool> Run(HloModule* module) override;
299
300 protected:
301 // These methods, invoked by PropagateConstraints, propagate a layout
302 // constraint to its neighbors (i.e. operands and users) in order to minimize
303 // the cost of the instructions being constrainted on. New constraints are
304 // added to the given constraint set.
305 //
306 // Backends can override these methods with backend-specific propagation
307 // rules.
308 virtual Status PropagateBufferConstraint(
309 const BufferLayoutConstraint& layout_constraint,
310 LayoutConstraints* constraints);
311 virtual Status PropagateOperandConstraint(
312 const OperandLayoutConstraint& layout_constraint,
313 LayoutConstraints* constraints);
314 virtual Status PropagateResultConstraint(
315 const ResultLayoutConstraint& layout_constraint,
316 LayoutConstraints* constraints);
317
318 // By default LayoutAssignment ensures that inputs and outputs of CustomCalls
319 // have the "major-first" layout (i.e. {n, n-1, ..., 0}).
320 //
321 // If this function returns true, LayoutAssignment does not set a layout for
322 // the given CustomCall. It's up to the backend to set one in
323 // AddBackendConstraints, if necessary.
324 //
325 // Precondition: instruction->opcode() == HloOpcode::kCustomCall.
326 virtual bool CustomCallRequiresMajorFirstLayout(
327 const HloInstruction* /*instruction*/) {
328 return true;
329 }
330
331 // Called after layouts of an instruction have been finalized to allow
332 // subclasses to check for platform specific assumptions.
333 virtual Status Verify(const HloInstruction* instruction) {
334 return Status::OK();
335 }
336
337 // Propagates a buffer layout constraint into the operands that use it.
338 Status PropagateBufferConstraintToUses(
339 const BufferLayoutConstraint& layout_constraint,
340 LayoutConstraints* constraints);
341
342 // Propagates a layout constraint on the use of the result of the given
343 // instruction to the definitions of the LogicalBuffers which make up the
344 // result.
345 Status PropagateUseConstraintToDefs(const ShapeLayout& shape_layout,
346 const HloInstruction* instruction,
347 LayoutConstraints* constraints);
348
349 // Chooses a layout of operand `operand_no` of `instruction` that minimizes
350 // the cost of `instruction`. `output_layout` is the layout of `instruction`.
351 // Returns null if it can't decide the best layout.
352 // Precondition: `instruction` and the operand are array-shaped.
353 std::unique_ptr<Layout> ChooseOperandLayoutFromOutputLayout(
354 const Layout& output_layout, const HloInstruction* instruction,
355 int64 operand_no);
356 // Given the layout of `user`'s `operand_no`-th operand, chooses a layout of
357 // `user` that minimizes its cost on that operand. Returns null if it can't
358 // decide the best layout.
359 // Precondition: `user` and the operand are array-shaped.
360 std::unique_ptr<Layout> ChooseOutputLayoutFromOperandLayout(
361 const Layout& operand_layout, const HloInstruction* user,
362 int64 operand_no);
363
364 private:
365 // Adds constraints which must be satisfied for correctness on all
366 // backends. Called once prior to propagating constraints.
367 Status AddMandatoryConstraints(
368 const ComputationLayout& computation_layout,
369 const ChannelLayoutConstraints* channel_constraints,
370 HloComputation* computation, LayoutConstraints* constraints);
371
372 // This method can be overridden to add backend-specific constraints to the
373 // layout of the instructions of a computation. This method is called after
374 // all mandatory constraints have been added via AddMandatoryConstraints
375 // and before propagating constraints.
376 virtual Status AddBackendConstraints(LayoutConstraints* constraints) {
377 return Status::OK();
378 }
379
380 // Construct contraints and assign layouts to all instructions in the
381 // computation satisfying the given ComputationLayout. Layouts constraints are
382 // added, then propagated until all LogicalBuffers in the computation are
383 // constrained.
384 Status RunOnComputation(const ComputationLayout& computation_layout,
385 const TuplePointsToAnalysis& points_to_analysis,
386 HloComputation* computation,
387 ChannelLayoutConstraints* channel_constraints);
388
389 // Assign layouts to the instructions of a computation which satisfy the given
390 // layout constraints. Copies may be added to satisfy the constraints. The
391 // given LayoutConstraints must have layout constraints every logical buffer
392 // in the computation.
393 Status AssignLayouts(const LayoutConstraints& constraints,
394 HloComputation* computation);
395
396 // Propagates layout constraints from a set of initial constraints in order to
397 // minimize the local cost of the computation. This propagation is *not*
398 // required for correctness.
399 Status PropagateConstraints(LayoutConstraints* constraints);
400
401 // Check that all layouts in the module have been set and satisfy all
402 // necessary conditions.
403 Status CheckLayouts(HloModule* module);
404
405 ComputationLayout* entry_computation_layout_;
406
407 protected:
408 // Map containing the layouts of all computations assigned so
409 // far. Computations are handled in a topological sort where computations are
410 // handled before their caller instructions so the layouts of caller
411 // instructions can be set to match the computation.
412 std::map<HloComputation*, ComputationLayout> computation_layouts_;
413 ChannelLayoutConstraints* channel_layout_constraints_;
414};
