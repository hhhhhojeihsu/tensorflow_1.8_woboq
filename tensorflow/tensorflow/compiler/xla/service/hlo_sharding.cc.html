30HloSharding HloSharding::Tile1D(const Shape& input_shape, int64 num_tiles) {
31 CHECK_EQ(1, ShapeUtil::Rank(input_shape));
32 CHECK_GT(num_tiles, 1);
33 std::vector<int64> dimensions(1, num_tiles);
34 Shape tile_shape = input_shape;
35 auto& tile_dimension = (*tile_shape.mutable_dimensions())[0];
36 tile_dimension = CeilOfRatio(static_cast<int64>(tile_dimension), num_tiles);
37 Array<int64> assignment(dimensions);
38 std::iota(assignment.begin(), assignment.end(), 0);
39 return HloSharding(tile_shape, assignment);
40}

42string HloSharding::ToString() const {
43 if (IsTuple()) {
44 std::vector<string> parts;
45 parts.reserve(tuple_elements_.size());
46 for (const HloSharding& element : tuple_elements_) {
47 parts.push_back(element.ToString());
48 }
49 return StrCat("{", tensorflow::str_util::Join(parts, ", "), "}");
50 }
51
52 string result = StrCat("{", (replicated_ ? " replicated" : ""),
53 (maximal_ ? " maximal" : ""));
54
55 if (replicated_) {
56 return "{replicated}";
57 } else if (maximal_) {
58 return StrCat(
59 "{maximal device=", static_cast<int64>(*tile_assignment_.begin()), "}");
60 } else {
61 return StrCat("{", ShapeUtil::HumanString(tile_shape_), " ", "devices=[",
62 Join(tile_assignment_.dimensions(), ","), "]",
63 Join(tile_assignment_, ","), "}");
64 }
65}

67bool HloSharding::UsesDevice(int64 device) const {
68 if (IsTuple()) {
69 return std::any_of(
70 tuple_elements_.begin(), tuple_elements_.end(),
71 [&](const HloSharding& s) { return s.UsesDevice(device); });
72 }
73 const auto& devices = tile_assignment_;
74 return replicated_ ||
75 std::find(devices.begin(), devices.end(), device) != devices.end();
76}

78std::vector<int64> HloSharding::TileIndexForDevice(int64 device) const {
79 CHECK(!ShapeUtil::IsTuple(tile_shape_));
80 CHECK(!maximal_);
81 CHECK(!IsTuple());
82 std::vector<int64> ret_index;
83 tile_assignment_.Each([&](tensorflow::gtl::ArraySlice<int64> index, int64 d) {
84 if (d == device) {
85 ret_index = {index.begin(), index.end()};
86 }
87 });
88 CHECK(!ret_index.empty());
89 return ret_index;
90}

92int64 HloSharding::DeviceForTileIndex(
93 tensorflow::gtl::ArraySlice<int64> index) const {
94 CHECK(!replicated_);
95 CHECK(!IsTuple());
96 if (maximal_) {
97 return *tile_assignment_.begin();
98 }
99 CHECK_EQ(ShapeUtil::Rank(tile_shape_), tile_assignment_.dimensions().size());
100 return tile_assignment_(index);
101}

103std::vector<int64> HloSharding::TileOffsetForDevice(int64 device) const {
104 CHECK(!IsTuple());
105
106 std::vector<int64> index = TileIndexForDevice(device);
107 if (maximal_) {
108 // Index will always be all zeroes if we're maximal, and tile_shape_ is not
109 // valid.
110 return index;
111 }
112 for (int64 i = 0; i < index.size(); ++i) {
113 index[i] *= tile_shape_.dimensions(i);
114 }
115 return index;
116}
117
118std::vector<int64> HloSharding::TileLimitForDevice(int64 device) const {
119 CHECK(!IsTuple());
120 CHECK(!maximal_); // Maximal shardings do not have a valid tile shape.
121
122 std::vector<int64> index = TileIndexForDevice(device);
123 for (int64 i = 0; i < index.size(); ++i) {
124 index[i] = (index[i] + 1) * tile_shape_.dimensions(i);
125 }
126 return index;
127}

129StatusOr<int64> HloSharding::UniqueDevice() const {
130 if (IsTuple()) {
131 if (tuple_elements_.empty()) {
132 return tensorflow::errors::InvalidArgument(
133 "UniqueDevice() called on empty tuple");
134 }
135 std::vector<StatusOr<int64>> results;
136 std::transform(tuple_elements_.begin(), tuple_elements_.end(),
137 std::back_inserter(results),
138 [](const HloSharding& s) { return s.UniqueDevice(); });
139 if (std::all_of(results.begin(), results.end(),
140 [&](const StatusOr<int64>& s) {
141 return s.ok() && results[0].ok() &&
142 s.ValueOrDie() == results[0].ValueOrDie();
143 })) {
144 return results[0];
145 } else {
146 return tensorflow::errors::InvalidArgument(
147 "Tuple did not contain a unique device");
148 }
149 }
150 if (!replicated_ && maximal_ && !IsTuple()) {
151 return static_cast<int64>(*tile_assignment_.begin());
152 }
153 return tensorflow::errors::InvalidArgument(
154 "UniqueDevice() called on sharding that executes on multiple devices");
155}

157bool HloSharding::HasUniqueDevice() const {
158 if (IsTuple()) {
159 return UniqueDevice().status().ok();
160 } else {
161 return !IsReplicated() && IsTileMaximal();
162 }

165Status HloSharding::ValidateTuple(const Shape& shape, int64 num_devices) const {
166 if (!ShapeUtil::IsTuple(shape)) {
167 return tensorflow::errors::InvalidArgument(
168 StrCat("Sharding is tuple-shaped but validation shape is not."));
169 }
170 // The easiest way to get the number of elements in a nested tuple is just to
171 // create a shape tree. We could call GetAsShapeTree, but that will try and
172 // apply our tuple_shardings_ to the shape tree, and that might cause a crash
173 // at this point as we haven't validated them.
174 ShapeTree<bool> bool_shape_tree(shape, false);
175 int64 num_leaves =
176 std::distance(bool_shape_tree.leaf_begin(), bool_shape_tree.leaf_end());
177 if (num_leaves != tuple_elements_.size()) {
178 return tensorflow::errors::InvalidArgument(
179 StrCat("Validation tuple shape has ", num_leaves,
180 " leaf elements, but this sharding contains ",
181 tuple_elements_.size(), " elements."));
182 }
183
184 // Now we've validated the number of tuple elements, it's safe to request a
185 // shape tree.
186 ShapeTree<HloSharding> shape_tree = GetAsShapeTree(shape);
187 for (const auto& index_to_sharding : shape_tree.leaves()) {
188 if (index_to_sharding.first.empty()) {
189 // An empty tuple has a ShapeTree with a single leaf at the empty index.
190 continue;
191 }
192 Status status = index_to_sharding.second.ValidateNonTuple(
193 ShapeUtil::GetSubshape(shape, index_to_sharding.first), num_devices);
194 if (!status.ok()) {
195 tensorflow::errors::AppendToMessage(
196 &status, StrCat("Note: While validating sharding tuple element ",
197 index_to_sharding.first.ToString(), " which is ",
198 index_to_sharding.second.ToString()));
199 return status;
200 }
201 }
202 return Status::OK();
203}

205Status HloSharding::Validate(const Shape& shape, int64 num_devices) const {
206 Status status = IsTuple() ? ValidateTuple(shape, num_devices)
207 : ValidateNonTuple(shape, num_devices);
208 if (!status.ok()) {
209 tensorflow::errors::AppendToMessage(
210 &status, StrCat("Note: While validating sharding ", ToString(),
211 " against shape ", ShapeUtil::HumanString(shape)));
212 }
213 return status;
214}

216Status HloSharding::ValidateNonTuple(const Shape& shape,
217 int64 num_devices) const {
218 if (ShapeUtil::IsTuple(shape)) {
219 return tensorflow::errors::InvalidArgument(
220 StrCat("Validation shape is a tuple but sharding is not."));
221 }
222 if (replicated_) {
223 return Status::OK();
224 }
225
226 // All tile assignments must be less than the number of available cores and
227 // unique.
228 Status status = Status::OK();
229 std::set<int64> seen_cores;
230 tile_assignment_.Each(
231 [&](tensorflow::gtl::ArraySlice<int64> indices, int32 core) {
232 // Don't overwrite a bad status, so we report the first error.
233 if (status.ok()) {
234 if (core >= num_devices) {
235 status = tensorflow::errors::InvalidArgument(StrCat(
236 "core ", core, " > ", num_devices, " in tile assignment"));
237 } else if (seen_cores.count(core) != 0) {
238 status = tensorflow::errors::InvalidArgument(
239 StrCat("core ", core, " is not unique in tile assignment"));
240 }
241 }
242 seen_cores.insert(core);
243 });
244 if (!status.ok()) {
245 return status;
246 }
247
248 if (IsTileMaximal()) {
249 return Status::OK();
250 }
251
252 // The tile rank must be the same as the input rank.
253 if (ShapeUtil::Rank(shape) != ShapeUtil::Rank(tile_shape_)) {
254 return tensorflow::errors::InvalidArgument(
255 "Tile rank is different to the input rank. sharding=", ToString(),
256 ", input_shape=", ShapeUtil::HumanString(shape));
257 }
258
259 // The tile shape must not be the same as the input shape without maximal_
260 // also set. If this is the case, we're not actually sharded and the correct
261 // constructor should have been used.
262 if (ShapeUtil::Equal(shape, tile_shape_)) {
263 return tensorflow::errors::InvalidArgument(
264 "Tile shape is the same as the input shape. If a replicated sharding "
265 "was intended, use HloSharding::Replicated(). If a device placement "
266 "was intended, use HloSharding::AssignDevice()");
267 }
268
269 // The tile shape must not be greater than the input shape in any dimension.
270 for (int64 i = 0, e = ShapeUtil::Rank(shape); i != e; ++i) {
271 auto tile_dim = tile_shape_.dimensions(i);
272 auto shape_dim = shape.dimensions(i);
273 if (tile_dim > shape_dim) {
274 return tensorflow::errors::InvalidArgument(
275 StrCat("Tile is larger than input shape (dimension ", i, ", ",
276 tile_dim, " > ", shape_dim));
277 }
278 }
279
280 // The tile assignment tensor must be exactly dimensioned to ceil(shape[dim]
281 // tile[dim]) for every dimension contained within tile.
282 for (int64 i = 0, e = tile_assignment_.dimensions().size(); i != e; ++i) {
283 int64 expected_dim =
284 CeilOfRatio(shape.dimensions(i), tile_shape_.dimensions(i));
285 if (tile_assignment_.dimensions()[i] != expected_dim) {
286 return tensorflow::errors::InvalidArgument(
287 StrCat("Tile assignment tensor has incorrect shape. Dimension ", i,
288 " expected ", expected_dim, " but got ",
289 tile_assignment_.dimensions()[i]));
290 }
291 }
292
293 return Status::OK();
294}

296/*static*/ StatusOr<HloSharding> HloSharding::FromProto(
297 const OpSharding& proto) {
298 if (proto.type() == OpSharding::Type::OpSharding_Type_TUPLE) {
299 std::vector<HloSharding> tuple_shardings;
300 tuple_shardings.reserve(proto.tuple_shardings().size());
301 for (const OpSharding& tuple_sharding_proto : proto.tuple_shardings()) {
302 TF_ASSIGN_OR_RETURN(HloSharding sharding,
303 HloSharding::FromProto(tuple_sharding_proto));
304 tuple_shardings.push_back(sharding);
305 }
306 return HloSharding(tuple_shardings);
307 } else if (proto.type() == OpSharding::Type::OpSharding_Type_REPLICATED) {
308 return Replicate();
309 } else if (proto.type() == OpSharding::Type::OpSharding_Type_MAXIMAL ||
310 proto.tile_assignment_devices().size() == 1) {
311 return HloSharding(proto.tile_assignment_devices(0));
312 }
313 // Some versions of gcc cannot infer the TileAssignment constructor from a
314 // braced initializer-list, so create one manually.
315 std::vector<int64> devices(proto.tile_assignment_devices().begin(),
316 proto.tile_assignment_devices().end());
317 Array<int64> tile_assignment(
318 std::vector<int64>(proto.tile_assignment_dimensions().begin(),
319 proto.tile_assignment_dimensions().end()));
320 std::copy(proto.tile_assignment_devices().begin(),
321 proto.tile_assignment_devices().end(), tile_assignment.begin());
322 return HloSharding(proto.tile_shape(), tile_assignment);
323}

325OpSharding HloSharding::ToProto() const {
326 OpSharding result;
327
328 if (IsTuple()) {
329 for (const HloSharding& element : tuple_elements_) {
330 *result.add_tuple_shardings() = element.ToProto();
331 }
332 result.set_type(OpSharding::Type::OpSharding_Type_TUPLE);
333 return result;
334 }
335
336 *result.mutable_tile_shape() = tile_shape_;
337 for (int64 dim : tile_assignment_.dimensions()) {
338 result.add_tile_assignment_dimensions(dim);
339 }
340 for (auto device : tile_assignment_) {
341 result.add_tile_assignment_devices(device);
342 }
343 if (IsReplicated()) {
344 result.set_type(OpSharding::Type::OpSharding_Type_REPLICATED);
345 } else if (IsTileMaximal()) {
346 result.set_type(OpSharding::Type::OpSharding_Type_MAXIMAL);
347 } else {
348 result.set_type(OpSharding::Type::OpSharding_Type_OTHER);
349 }
350 return result;
351}

353HloSharding HloSharding::TransformShardedTileShape(
354 const Shape& new_shape,
355 const std::function<int64(int64, int64)>& transform) const {
356 CHECK(!IsTuple());
357 if (IsTileMaximal()) {
358 return *this;
359 }
360 CHECK_EQ(ShapeUtil::Rank(new_shape), ShapeUtil::Rank(tile_shape()));
361 Shape new_tile_shape;
362 new_tile_shape.set_element_type(tile_shape().element_type());
363 for (int64 i = 0; i < ShapeUtil::Rank(new_shape); ++i) {
364 int64 dim;
365 if (tile_assignment().dim(i) == 1) {
366 dim = new_shape.dimensions(i);
367 } else if (transform) {
368 dim = transform(i, tile_shape().dimensions(i));
369 } else {
370 dim = tile_shape().dimensions(i);
371 }
372 new_tile_shape.add_dimensions(dim);
373 }
374 TF_CHECK_OK(
375 LayoutUtil::CopyLayoutBetweenShapes(tile_shape_, &new_tile_shape));
376 return HloSharding::Tile(new_tile_shape, tile_assignment());
377}
