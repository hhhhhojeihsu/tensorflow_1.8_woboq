52// Options to configure the service when it is created.
53class ServiceOptions {
54 public:
55 // Set the platform backing the service, or nullptr for the default platform.
56 ServiceOptions& set_platform(perftools::gputools::Platform* platform);
57 perftools::gputools::Platform* platform() const;
58
59 // Set the number of replicas to use when compiling replicated
60 // programs.
61 ServiceOptions& set_number_of_replicas(int number_of_replicas);
62 int number_of_replicas() const;
63
64 // Sets the thread pool size for parallel execution of an individual operator.
65 ServiceOptions& set_intra_op_parallelism_threads(int num_threads);
66 int intra_op_parallelism_threads() const;
67
68 private:
69 perftools::gputools::Platform* platform_ = nullptr;
70 int number_of_replicas_ = 1;
71 int intra_op_parallelism_threads_ = -1;
72};

74// The XLA service object, which is the same across all platforms. It maintains
75// the service state of computations and allocations, and delegates
76// target-specific requests to the target-specific infrastructure
77// (target-specific compiler, StreamExecutor).
78class Service : public ServiceInterface {
79 public:
80 // Factory method for creating a new Service.
81 static StatusOr<std::unique_ptr<Service>> NewService(
82 perftools::gputools::Platform* platform = nullptr);
83 static StatusOr<std::unique_ptr<Service>> NewService(
84 const ServiceOptions& options);
85
86 // Creates a new computation with the given name.
87 // A unique ComputationHandle is returned.
88 tensorflow::Status Computation(const ComputationRequest* arg,
89 ComputationResponse* result) override;
90
91 // Unregisters a previously-allocated global handle.
92 //
93 // If the handle given is not currently allocated, a NOT_FOUND status is
94 // returned.
95 tensorflow::Status Unregister(const UnregisterRequest* arg,
96 UnregisterResponse* result) override;
97
98 // Deconstructs a tuple. Returns a newly created GlobalDataHandle for each
99 // element in the tuple.
100 tensorflow::Status DeconstructTuple(
101 const DeconstructTupleRequest* arg,
102 DeconstructTupleResponse* result) override;
103
104 // Modifies the provided computation so that subsequent executions
105 // will compute the provided ComputationDataHandle, rather than the
106 // last expression enqueued on that Computation.
107 tensorflow::Status SetReturnValue(const SetReturnValueRequest* arg,
108 SetReturnValueResponse* results) override;
109
110 // Executes a computation with the provided global data passed as
111 // immutable arguments. Returns global data output and execution timing.
112 tensorflow::Status Execute(const ExecuteRequest* arg,
113 ExecuteResponse* result) override;
114
115 // Executes a computation with the provided global data passed as
116 // immutable arguments. The request contains the whole computation graph.
117 // Returns global data output and execution timing.
118 //
119 // TODO(b/74197823): This is a part of a NOT YET ready refactor.
120 tensorflow::Status ExecuteGraph(const ExecuteGraphRequest* arg,
121 ExecuteResponse* result) override;
122
123 // Executes one or more computations in parallel with the provided global data
124 // passed as immutable arguments. Returns global data output for each
125 // computation.
126 tensorflow::Status ExecuteParallel(const ExecuteParallelRequest* arg,
127 ExecuteParallelResponse* result) override;
128
129 // Executes one or more computations in parallel with the provided global data
130 // passed as immutable arguments. Returns global data output for each
131 // computation.
132 //
133 // TODO(b/74197823): This is a part of a NOT YET ready refactor.
134 tensorflow::Status ExecuteGraphParallel(
135 const ExecuteGraphParallelRequest* arg,
136 ExecuteParallelResponse* result) override;
137
138 // Requests one or more device handles from the target.
139 //
140 // When N device handles are requested and the number of replicas is R, at
141 // least N * R devices must be available. The devices are assigned based on
142 // the device ordinals such that the first R available devices are assigned to
143 // the first set of replicas, and the next R devices to the second set of
144 // replicas, etc. Each returned device handle represents the device with the
145 // replica id 0.
146 tensorflow::Status GetDeviceHandles(
147 const GetDeviceHandlesRequest* arg,
148 GetDeviceHandlesResponse* result) override;
149
150 // Asynchronously executes a computation with provided arguments. Invokes
151 // the provided computation with the provided global data passed as
152 // immutable arguments. Returns a handle to the execution.
153 //
154 // (Note: The corresponding function in xla::Client was removed as part of
155 // b/64116060, in an attempt to simplify our API. We're keeping this around
156 // for now in case we want to expose this to clients in a different way.)
157 tensorflow::Status ExecuteAsync(const ExecuteAsyncRequest* arg,
158 ExecuteAsyncResponse* result) override;
159
160 // Waits until the specified execution is complete and returns the result.
161 // Calling this API multiple times with the same execution handle returns the
162 // method with an error since the execution handle is destroyed after the
163 // first call.
164 tensorflow::Status WaitForExecution(
165 const WaitForExecutionRequest* arg,
166 WaitForExecutionResponse* result) override;
167
168 // Requests that global data be transferred to the client in literal form.
169 tensorflow::Status TransferToClient(
170 const TransferToClientRequest* arg,
171 TransferToClientResponse* result) override;
172
173 // Transfers data from a literal provided by the client, into device memory.
174 tensorflow::Status TransferToServer(
175 const TransferToServerRequest* arg,
176 TransferToServerResponse* result) override;
177
178 // Transfers data from a literal provided by the client, into the Infeed
179 // buffer of the device.
180 tensorflow::Status TransferToInfeed(
181 const TransferToInfeedRequest* arg,
182 TransferToInfeedResponse* result) override;
183
184 // Transfers data from the Outfeed othe device to the literal provided by the
185 // client.
186 tensorflow::Status TransferFromOutfeed(
187 const TransferFromOutfeedRequest* arg,
188 TransferFromOutfeedResponse* result) override;
189
190 // Resets devices, clearing all existing state on all the devices associated
191 // with this service (including memory allocated on the devices).
192 //
193 // ResetDevice may only be called where no previous Execution state on the
194 // device is used by the next Execution.
195 //
196 // ResetDevice should be called before an Execution that expect the device to
197 // be in the reset state. For example, if the prior Execution modifies device
198 // state (e.g., architectural state) that the next Execution depends on.
199 tensorflow::Status ResetDevice(const ResetDeviceRequest* arg,
200 ResetDeviceResponse* result) override;
201
202 // Tests if an expression is a compile-time constant.
203 tensorflow::Status IsConstant(const IsConstantRequest* arg,
204 IsConstantResponse* result) override;
205
206 // Computes the value of a constant expression.
207 tensorflow::Status ComputeConstant(const ComputeConstantRequest* arg,
208 ComputeConstantResponse* result) override;
209
210 // Returns the shape (with layout) of an array associated with a given data
211 // handle.
212 tensorflow::Status GetShape(const GetShapeRequest* arg,
213 GetShapeResponse* result) override;
214
215 // Returns the program shape of the computation associated with the given
216 // handle.
217 tensorflow::Status GetComputationShape(
218 const GetComputationShapeRequest* arg,
219 GetComputationShapeResponse* result) override;
220
221 /////
222 // Computation-oriented methods.
223
224 // Enqueues an Op on the computation.
225 tensorflow::Status Op(const OpRequest* arg, OpResponse* result) override;
226
227 // Retrieves the inferred shape for a value within a computation.
228 tensorflow::Status GetLocalShape(const GetLocalShapeRequest* arg,
229 GetLocalShapeResponse* result) override;
230
231 // Retrieves the statistics of a computation.
232 tensorflow::Status GetComputationStats(
233 const ComputationStatsRequest* arg,
234 ComputationStatsResponse* result) override;
235
236 // Retrieves the statistics of a computation.
237 //
238 // TODO(b/74197823): This is a part of a NOT YET ready refactor.
239 tensorflow::Status GetComputationGraphStats(
240 const ComputationGraphStatsRequest* arg,
241 ComputationStatsResponse* result) override;
242
243 // Snapshots the current state of a computation handle into a serializable
244 // protocol buffer form, so it can be loaded via
245 // LoadComputationSnapshot.
246 tensorflow::Status SnapshotComputation(
247 const SnapshotComputationRequest* arg,
248 SnapshotComputationResponse* result) override;
249
250 // Loads a computation from a serialized protocol buffer created via
251 // SnapshotComputation.
252 tensorflow::Status LoadComputationSnapshot(
253 const LoadComputationSnapshotRequest* arg,
254 LoadComputationSnapshotResponse* result) override;
255
256 // Creates a unique channel handle that can be used for Send/Recv
257 // instructions.
258 tensorflow::Status CreateChannelHandle(
259 const CreateChannelHandleRequest* arg,
260 CreateChannelHandleResponse* result) override;
261
262 // Returns the ComputationTracker of the current service instance.
263 // Only used in unit tests to access user computations from client.
264 const ComputationTracker& computation_tracker() {
265 return computation_tracker_;
266 }
267
268 // Returns the backend used to execute computations.
269 const Backend& backend() const { return *execute_backend_; }
270 Backend* mutable_backend() { return execute_backend_.get(); }

272 private:
273 // A private overload for Service itself, used by other methods within this
274 // class.
275 StatusOr<std::unique_ptr<HloModuleConfig>> CreateModuleConfig(
276 const ProgramShape& program_shape,
277 tensorflow::gtl::ArraySlice<const ShapedBuffer*> arguments,
278 const ExecutionOptions& execution_options,
279 const UserComputation* user_computation = nullptr);
280
281 // Picks a parallel response and fills the result.
282 Status PickParallelResponse(const ExecuteParallelResponse& parallel_result,
283 ExecuteResponse* result);
284
285 // Prepare the executors for executing parallel.
286 StatusOr<std::vector<perftools::gputools::StreamExecutor*>> GetExecutors(
287 const ExecutionOptions& execution_options, int64 requests_size,
288 int64 request_index) const;
289
290 // Prepare the arguments for executing parallel.
291 StatusOr<std::vector<std::vector<const ShapedBuffer*>>> GetArguments(
292 const ExecutionOptions& execution_options,
293 tensorflow::gtl::ArraySlice<const GlobalDataHandle*> arguments);
294
295 protected:
296 friend class LocalExecutable;
297
298 // The constructor is private. Use the NewService factory to create new
299 // service objects.
300 Service(const ServiceOptions& options,
301 std::unique_ptr<Backend> execute_backend);
302
303 // Resolves the given argument handles in the allocation tracker and returns
304 // the corresponding allocations for every replica. The function also verifies
305 // that each allocation matches the execution platform and device ordinal of
306 // the corresponding replica.
307 StatusOr<std::vector<std::vector<const ShapedBuffer*>>>
308 ResolveAndValidateArguments(
309 tensorflow::gtl::ArraySlice<const GlobalDataHandle*> arguments,
310 tensorflow::gtl::ArraySlice<perftools::gputools::StreamExecutor*>
311 stream_executors);
312
313 // Create a Hlo module config for the given program shape and arguments.
314 // execution_options is optional; if not given a default is used.
315 StatusOr<std::unique_ptr<HloModuleConfig>> CreateModuleConfig(
316 const ProgramShape& program_shape,
317 tensorflow::gtl::ArraySlice<const Shape*> argument_shapes,
318 const ExecutionOptions* execution_options,
319 const UserComputation* user_computation = nullptr);
320
321 // Builds an Executable for the given parameters.
322 //
323 // If device_allocator is not null, the compiler may use it to allocate temp
324 // buffers, which the compiler is responsible for freeing. The allocator
325 // given here need not match the allocator used when running the executable.
326 StatusOr<std::unique_ptr<Executable>> BuildExecutable(
327 const VersionedComputationHandle& versioned_handle,
328 std::unique_ptr<HloModuleConfig> module_config, Backend* backend,
329 perftools::gputools::StreamExecutor* executor,
330 DeviceMemoryAllocator* device_allocator = nullptr);
331
332 // Builds an Executable for the given HLO module proto.
333 //
334 // TODO(b/74197823): This is a part of a NOT YET ready refactor.
335 StatusOr<std::unique_ptr<Executable>> BuildExecutable(
336 const HloModuleProto& module_proto,
337 std::unique_ptr<HloModuleConfig> module_config, Backend* backend,
338 perftools::gputools::StreamExecutor* executor,
339 DeviceMemoryAllocator* device_allocator = nullptr);
340
341 // Same as BuildExecutable() above, but builds a list of Executables for the
342 // given computations that may interact with each other.
343 StatusOr<std::vector<std::unique_ptr<Executable>>> BuildExecutables(
344 std::vector<VersionedComputationHandle> versioned_handles,
345 std::vector<std::unique_ptr<HloModuleConfig>> module_configs,
346 Backend* backend,
347 std::vector<std::vector<perftools::gputools::StreamExecutor*>> executors,
348 DeviceMemoryAllocator* device_allocator);
349 StatusOr<std::vector<std::unique_ptr<Executable>>> BuildExecutables(
350 const std::vector<const HloModuleProto*>& module_protos,
351 std::vector<std::unique_ptr<HloModuleConfig>> module_configs,
352 Backend* backend,
353 std::vector<std::vector<perftools::gputools::StreamExecutor*>> executors,
354 DeviceMemoryAllocator* device_allocator);
355
356 // Similar to BuildExecutable, but look in the compilation cache for the
357 // executable first. If the executable is not in the cache, it is built and
358 // inserted into the cache.
359 StatusOr<std::shared_ptr<Executable>> BuildAndCacheExecutable(
360 const VersionedComputationHandle& versioned_handle,
361 std::unique_ptr<HloModuleConfig> module_config, Backend* backend,
362 perftools::gputools::StreamExecutor* executor, ExecutionProfile* profile,
363 DeviceMemoryAllocator* device_allocator = nullptr);
364
365 // Runs the given executable with the given arguments and register the result
366 // in the allocation tracker. The handle of the result from the tracker is
367 // returned. If the parameter "profile" is not null, it points to an
368 // ExecutionProfile object which will be filled in with profile data.
369 StatusOr<GlobalDataHandle> ExecuteAndRegisterResult(
370 Executable* executable,
371 const tensorflow::gtl::ArraySlice<std::vector<const ShapedBuffer*>>
372 arguments,
373 Backend* backend, const string& result_tag, ExecutionProfile* profile);
374
375 // Runs the given executables with the given arguments and register the result
376 // from each executable in the allocation tracker. The handles of the result
377 // from the tracker are returned.
378 StatusOr<std::vector<GlobalDataHandle>> ExecuteParallelAndRegisterResult(
379 tensorflow::gtl::ArraySlice<Executable*> executables,
380 tensorflow::gtl::ArraySlice<std::vector<std::vector<const ShapedBuffer*>>>
381 arguments,
382 Backend* backend,
383 tensorflow::gtl::ArraySlice<DeviceHandle> device_handles,
384 tensorflow::gtl::ArraySlice<string> result_tags,
385 ExecutionProfile* profile);
386
387 // Convenience function for adding a function to a user computation.
388 template <typename RequestT, typename ResponseT>
389 tensorflow::Status AddInstruction(
390 const RequestT* arg, ResponseT* result,
391 const std::function<StatusOr<ComputationDataHandle>(UserComputation*)>&
392 adder);
393
394 // Executes a single computation which has more than one target device.
395 // The N devices are expected to all return an empty tuple, but one, which
396 // will be the result of this computation.
397 tensorflow::Status ExecuteOneToN(const ExecuteRequest* arg,
398 ExecuteResponse* result);
399 tensorflow::Status ExecuteOneToN(const ExecuteGraphRequest* arg,
400 ExecuteResponse* result);
401
402 // Convenience function which checks whether the given shape_with_layout
403 // (presumably passed by the client to set the result layout) is valid for the
404 // given computation result shape.
405 tensorflow::Status ValidateResultShapeWithLayout(
406 const Shape& shape_with_layout, const Shape& result_shape) const;
407
408 // Returns the stream executors assigned to the replicas represented by the
409 // given device handle. Each device_handle is a virtual replicated device that
410 // represents a set of physical devices for the replicas.
411 StatusOr<std::vector<perftools::gputools::StreamExecutor*>> Replicas(
412 const Backend& backend, const DeviceHandle& device_handle) const;
413
414 Status MaybeDumpHloModule(const HloModule& module) const;
415
416 // Returns the device handle that represents the replicated device for a
417 // single computation that is not model-parallelized.
418 DeviceHandle SingleComputationDeviceHandle() const;
419
420 ServiceOptions options_;
421
422 // Tracks computations built via the API.
423 ComputationTracker computation_tracker_;
424
425 // Tracks channels created via the API.
426 ChannelTracker channel_tracker_;
427
428 // Tracks allocations made via the API and computation execution.
429 AllocationTracker allocation_tracker_;
430
431 // Tracks asynchronously launched executions via the API.
432 ExecutionTracker execution_tracker_;
433
434 // Cache containing previously built Executables.
435 CompilationCache compilation_cache_;
436
437 // Backend to compile and execute computations on.
438 std::unique_ptr<Backend> execute_backend_;
