60// Creates and returns a copy of the given instruction with a different
61// layout. Tuple-shaped instructions will be deep-copied, and the last Tuple
62// instruction producing the copy is returned.
63StatusOr<HloInstruction*> CreateCopyWithNewLayout(
64 const Shape& shape_with_layout, HloInstruction* instruction) {
65 TF_RET_CHECK(LayoutUtil::HasLayout(shape_with_layout));
66 DCHECK(ShapeUtil::Compatible(shape_with_layout, instruction->shape()))
67 << ShapeUtil::HumanString(shape_with_layout) << " "
68 << ShapeUtil::HumanString(instruction->shape())
69 << " instruction: " << instruction->ToString();
70
71 if (ShapeUtil::IsTuple(instruction->shape())) {
72 // Deep-copy tuples.
73 std::vector<HloInstruction*> element_copies;
74 for (int64 i = 0; i < ShapeUtil::TupleElementCount(instruction->shape());
75 ++i) {
76 HloInstruction* gte = instruction->parent()->AddInstruction(
77 HloInstruction::CreateGetTupleElement(
78 ShapeUtil::GetSubshape(instruction->shape(), {i}), instruction,
79 i));
80
81 // Recurse to copy each elements.
82 TF_ASSIGN_OR_RETURN(
83 HloInstruction * element_copy,
84 CreateCopyWithNewLayout(
85 ShapeUtil::GetSubshape(shape_with_layout, {i}), gte));
86 element_copies.push_back(element_copy);
87 }
88 // Gather element copies into a tuple with a new Tuple instruction.
89 HloInstruction* tuple_copy = instruction->parent()->AddInstruction(
90 HloInstruction::CreateTuple(element_copies));
91 LayoutUtil::ClearLayout(tuple_copy->mutable_shape());
92 TF_RETURN_IF_ERROR(LayoutUtil::CopyLayoutBetweenShapes(
93 shape_with_layout, tuple_copy->mutable_shape()));
94 return tuple_copy;
95 } else if (ShapeUtil::IsArray(instruction->shape())) {
96 HloInstruction* copy =
97 instruction->parent()->AddInstruction(HloInstruction::CreateUnary(
98 instruction->shape(), HloOpcode::kCopy, instruction));
99 LayoutUtil::ClearLayout(copy->mutable_shape());
100 TF_RETURN_IF_ERROR(LayoutUtil::CopyLayoutBetweenShapes(
101 shape_with_layout, copy->mutable_shape()));
102
103 return copy;
104 } else {
105 return FailedPrecondition(
106 "Can only copy array and tuple shaped instructions");
107 }
108}

110// Creates a copy of the given operand if the operand's layout does not match
111// the given layout. This copy replaces the use in the given instruction. Tuple
112// operands will be deep-copied.
113Status CopyOperandIfLayoutsDiffer(const ShapeLayout& operand_layout,
114 HloInstruction* instruction,
115 int64 operand_no) {
116 HloInstruction* operand = instruction->mutable_operand(operand_no);
117 TF_RET_CHECK(operand_layout.LayoutIsSet());
118 TF_RET_CHECK(LayoutUtil::HasLayout(operand->shape()));
119
120 if (ShapeUtil::Equal(operand_layout.shape(), operand->shape())) {
121 // Operand layout already matches our constraint. Nothing to do.
122 return Status::OK();
123 }
124
125 TF_ASSIGN_OR_RETURN(HloInstruction * operand_copy,
126 CreateCopyWithNewLayout(operand_layout.shape(), operand));
127
128 return instruction->ReplaceOperandWith(operand_no, operand_copy);
129}

179LayoutConstraints::LayoutConstraints(
180 const TuplePointsToAnalysis& points_to_analysis,
181 HloComputation* computation)
182 : points_to_analysis_(points_to_analysis), computation_(computation) {
183 // Gather all array-shaped logical buffers into unconstrained_buffer_ids.
184 for (LogicalBuffer::Id id = 0; id < points_to_analysis_.num_logical_buffers();
185 id++) {
186 auto& buffer = points_to_analysis_.logical_buffer(id);
187 // The points to analysis is computed per module, restrict constraints to
188 // array buffers in this computation.
189 if (buffer.IsArray() && buffer.instruction()->parent() == computation) {
190 unconstrained_buffer_ids_.insert(buffer.id());
191 }
192 }
193}

195PointsToSet::BufferSet* LayoutConstraints::GetBufferSet(
196 const HloInstruction* instruction) const {
197 auto it = buffer_sets_cache_.find(instruction);
198 if (it != buffer_sets_cache_.end()) {
199 return it->second.get();
200 }
201 auto& buffer_set =
202 buffer_sets_cache_
203 .emplace(instruction, MakeUnique<PointsToSet::BufferSet>())
204 .first->second;
205 const auto& points_to_set = points_to_analysis_.GetPointsToSet(instruction);
206 points_to_set.ForEachElement(
207 [&buffer_set](const ShapeIndex& /*index*/,
208 const PointsToSet::BufferList& buffers) {
209 buffer_set->insert(buffers.begin(), buffers.end());
210 });
211 return buffer_set.get();
212}

214bool LayoutConstraints::OperandBufferForwarded(
215 const HloInstruction* instruction, int64 operand_no) const {
216 // The operand is potentially forwarded if the intersection of points-to sets
217 // of the operand and the instruction is non-empty.
218 PointsToSet::BufferSet* output_buffers = GetBufferSet(instruction);
219 PointsToSet::BufferSet* operand_buffers =
220 GetBufferSet(instruction->operand(operand_no));
221 for (const LogicalBuffer* output_buffer : *output_buffers) {
222 if (operand_buffers->count(output_buffer) > 0) {
223 return true;
224 }
225 }
226 return false;
227}

229Status LayoutConstraints::SetBufferLayout(const Layout& layout,
230 const LogicalBuffer& buffer,
231 bool mandatory, bool dfs) {
232 VLOG(3) << "SetBufferLayout : " << buffer << " : "
233 << LayoutUtil::HumanString(layout);
234
235 TF_RETURN_IF_ERROR(points_to_analysis_.VerifyBuffer(buffer));
236 if (!buffer.IsArray()) {
237 return FailedPrecondition(
238 "Layout of buffer %s cannot be constrained because buffer is not "
239 "array-shaped, has shape: %s",
240 buffer.ToString().c_str(),
241 ShapeUtil::HumanString(buffer.shape()).c_str());
242 }
243 TF_RETURN_IF_ERROR(
244 LayoutUtil::ValidateLayoutForShape(layout, buffer.shape()));
245
246 const BufferLayoutConstraint* curr_constraint =
247 GetBufferLayoutConstraint(buffer);
248 if (curr_constraint != nullptr) {
249 if (LayoutUtil::Equal(curr_constraint->layout(), layout)) {
250 // New constraint matches existing constraint. Nothing to do.
251 return Status::OK();
252 }
253 if (curr_constraint->mandatory()) {
254 return FailedPrecondition(
255 "Buffer %s already has the layout constraint %s, cannot add "
256 "incompatible constraint %s",
257 buffer.ToString().c_str(),
258 LayoutUtil::HumanString(curr_constraint->layout()).c_str(),
259 LayoutUtil::HumanString(layout).c_str());
260 }
261 }
262
263 auto iter = buffer_constraints_.find(&buffer);
264 bool overwrite = iter != buffer_constraints_.end();
265 if (!overwrite) {
266 iter = buffer_constraints_
267 .insert(std::make_pair(
268 &buffer,
269 BufferLayoutConstraint(layout, buffer, mandatory, dfs)))
270 .first;
271 } else {
272 iter->second = BufferLayoutConstraint(layout, buffer, mandatory, dfs);
273 }
274 added_constraints_.push_back(&iter->second);
275
276 // Remove buffer from the set of unconstrained buffers.
277 TF_RET_CHECK(unconstrained_buffer_ids_.count(buffer.id()) ==
278 static_cast<int>(!overwrite));
279 unconstrained_buffer_ids_.erase(buffer.id());
280
281 return Status::OK();
282}

284Status LayoutConstraints::SetOperandLayout(const Shape& shape_with_layout,
285 const HloInstruction* instruction,
286 int64 operand_no, bool mandatory,
287 bool dfs) {
288 VLOG(3) << "SetOperandLayout : " << instruction->name() << ", operand "
289 << operand_no << " : "
290 << ShapeUtil::HumanStringWithLayout(shape_with_layout);
291
292 const OperandLayoutConstraint* curr_shape_layout =
293 GetOperandLayoutConstraint(instruction, operand_no);
294 if (curr_shape_layout != nullptr) {
295 if (curr_shape_layout->shape_layout().MatchesLayoutInShape(
296 shape_with_layout)) {
297 // New constraint matches existing constraint. Nothing to do.
298 return Status::OK();
299 }
300 if (curr_shape_layout->mandatory()) {
301 return FailedPrecondition(
302 "Operand %lld of instruction %s already has a layout constraint "
303 "%s, cannot add incompatible constraint %s",
304 operand_no, instruction->name().c_str(),
305 curr_shape_layout->shape_layout().ToString().c_str(),
306 ShapeUtil::HumanStringWithLayout(shape_with_layout).c_str());
307 }
308 }
309
310 // If any buffers in the operand occur in the output of the instruction, then
311 // return an error. This case is not handled because such a constraint changes
312 // layouts beyond this immediate use and is complicated to handle.
313 if (OperandBufferForwarded(instruction, operand_no)) {
314 return FailedPrecondition(
315 "Cannot constraint layout of operand %lld of instruction %s "
316 "because instruction forwards operand's LogicalBuffer(s)",
317 operand_no, instruction->name().c_str());
318 }
319
320 auto key = std::make_pair(instruction, operand_no);
321 auto iter = operand_constraints_.find(key);
322 if (iter == operand_constraints_.end()) {
323 auto pair = std::make_pair(
324 key, OperandLayoutConstraint(ShapeLayout(shape_with_layout),
325 instruction, operand_no, mandatory, dfs));
326 iter = operand_constraints_.insert(pair).first;
327 } else {
328 iter->second =
329 OperandLayoutConstraint(ShapeLayout(shape_with_layout), instruction,
330 operand_no, mandatory, dfs);
331 }
332 added_constraints_.push_back(&iter->second);
333
334 return Status::OK();
335}

348Status LayoutConstraints::SetResultLayout(const Shape& shape_with_layout,
349 bool dfs) {
350 VLOG(3) << "SetResultLayout : "
351 << ShapeUtil::HumanStringWithLayout(shape_with_layout);
352
353 const ShapeLayout* curr_shape_layout = ResultLayout();
354 if (curr_shape_layout != nullptr) {
355 if (!curr_shape_layout->MatchesLayoutInShape(shape_with_layout)) {
356 return FailedPrecondition(
357 "Result of computation %s already has the layout constraint %s, "
358 "cannot add incompatible constraint %s",
359 computation_->name().c_str(), curr_shape_layout->ToString().c_str(),
360 ShapeUtil::HumanStringWithLayout(shape_with_layout).c_str());
361 }
362 // New constraint matches existing constraint. Nothing to do.
363 return Status::OK();
364 }
365
366 result_constraint_.reset(
367 new ResultLayoutConstraint(ShapeLayout(shape_with_layout), dfs));
368 added_constraints_.push_back(result_constraint_.get());
369
370 return Status::OK();
371}

373Status LayoutConstraints::SetInstructionLayout(
374 const Shape& shape_with_layout, const HloInstruction* instruction,
375 bool mandatory, bool dfs) {
376 VLOG(3) << "SetInstructionLayout : " << instruction->name() << ", "
377 << ShapeUtil::HumanStringWithLayout(shape_with_layout);
378
379 if (!ShapeUtil::Compatible(shape_with_layout, instruction->shape())) {
380 return FailedPrecondition(
381 "Instruction %s of shape %s cannot be assigned incompatible layout %s",
382 instruction->name().c_str(),
383 ShapeUtil::HumanString(instruction->shape()).c_str(),
384 ShapeUtil::HumanStringWithLayout(shape_with_layout).c_str());
385 }
386
387 // Create a BufferLayoutConstraint for each array shape in the output of the
388 // instruction.
389 return ShapeUtil::ForEachSubshapeWithStatus(
390 shape_with_layout,
391 [this, instruction, mandatory](const Shape& subshape,
392 const ShapeIndex& index) -> Status {
393 // The precondition for this method is that the instruction defines all
394 // buffers in its output.
395 auto buffers =
396 points_to_analysis_.GetPointsToSet(instruction).element(index);
397 CHECK_EQ(1, buffers.size());
398 CHECK_EQ(buffers[0]->instruction(), instruction);
399
400 if (ShapeUtil::IsArray(subshape)) {
401 return SetBufferLayout(subshape.layout(), *buffers[0], mandatory);
402 } else {
403 return Status::OK();
404 }
405 });
406}

465 if (ResultLayout() != nullptr) {
466 tensorflow::strings::StrAppend(&output, " => ", ResultLayout()->ToString(),

472Status LayoutAssignment::AddMandatoryConstraints(
473 const ComputationLayout& computation_layout,
474 const ChannelLayoutConstraints* channel_constraints,
475 HloComputation* computation, LayoutConstraints* constraints) {
476 VLOG(3) << "Adding mandatory layout constraints to computation "
477 << computation->name();
478
479 // Constrain layouts of instructions which define values with pre-existing
480 // layouts.
481 for (auto* instruction : computation->instructions()) {
482 Shape const* shape_with_layout = nullptr;
483 if (instruction->opcode() == HloOpcode::kInfeed) {
484 // Infeed layouts must match the layout of the original inserted
485 // instruction.
486 // TODO(b/31425034): Change infeeds to be more like parameters, with
487 // shapes in the ComputationLayout.
488 DCHECK(!LayoutUtil::IsPadded(instruction->shape()));
489 TF_RETURN_IF_ERROR(
490 constraints->SetInstructionLayout(instruction->shape(), instruction));
491 } else if (instruction->opcode() == HloOpcode::kOutfeed) {
492 // Constrain the input to the Outfeed instruction to be the expected
493 // layout of the Outfeed.
494 TF_RETURN_IF_ERROR(constraints->SetOperandLayout(
495 instruction->outfeed_shape(), instruction, 0));
496 } else if (instruction->opcode() == HloOpcode::kParameter) {
497 // Parameter layouts must match the respective layout in
498 // ComputationLayout.
499 shape_with_layout =
500 &computation_layout.parameter_layout(instruction->parameter_number())
501 .shape();
502 }
503 if (shape_with_layout != nullptr) {
504 TF_RETURN_IF_ERROR(
505 constraints->SetInstructionLayout(*shape_with_layout, instruction));
506 }
507
508 if (instruction->opcode() == HloOpcode::kSend ||
509 instruction->opcode() == HloOpcode::kRecv) {
510 CHECK(channel_constraints)
511 << "Multi-module layout assignment requires ChannelLayoutConstraints";
512 int64 channel_id = instruction->channel_id();
513 if (!channel_constraints->IsChannelConstrained(channel_id)) {
514 continue;
515 }
516 if (instruction->opcode() == HloOpcode::kSend) {
517 // TODO(b/68493863): Change to use SetOperandLayout().
518 const Shape send_buffer_shape = instruction->operand(0)->shape();
519 TF_RET_CHECK(ShapeUtil::IsArray(send_buffer_shape));
520 Shape new_buffer_shape = channel_constraints->LayoutShapeForChannel(
521 send_buffer_shape, instruction->channel_id());
522 TF_RETURN_IF_ERROR(constraints->SetInstructionLayout(
523 new_buffer_shape, instruction->operand(0)));
524 } else {
525 const Shape recv_buffer_shape =
526 ShapeUtil::GetTupleElementShape(instruction->shape(), 0);
527 TF_RET_CHECK(ShapeUtil::IsArray(recv_buffer_shape));
528 TF_ASSIGN_OR_RETURN(
529 const LogicalBuffer* buffer,
530 constraints->points_to_analysis().GetBufferDefinedAt(instruction,
531 {0}));
532 Shape new_shape = channel_constraints->LayoutShapeForChannel(
533 recv_buffer_shape, instruction->channel_id());
534 TF_RETURN_IF_ERROR(
535 constraints->SetBufferLayout(new_shape.layout(), *buffer));
536 }
537 }
538 }
539
540 // Constrain layouts of instructions which call computations which have
541 // already been assigned layouts. Instructions which call computations in a
542 // parallel element-wise context (eg, map or reduce) do not need layout
543 // constraints because they operate on scalars.
544 for (auto* instruction : computation->instructions()) {
545 if (instruction->opcode() == HloOpcode::kCall) {
546 // kCall instruction operands and output must match the ComputationLayout
547 // of the called computation.
548 const ComputationLayout& called_computation_layout =
549 FindOrDie(computation_layouts_, instruction->to_apply());
550 TF_RETURN_IF_ERROR(constraints->SetInstructionLayout(
551 called_computation_layout.result_layout().shape(), instruction));
552 TF_RET_CHECK(instruction->operand_count() ==
553 called_computation_layout.parameter_count());
554 for (int64 i = 0; i < instruction->operand_count(); ++i) {
555 TF_RETURN_IF_ERROR(constraints->SetOperandLayout(
556 called_computation_layout.parameter_layout(i).shape(), instruction,
557 i));
558 }
559 } else if (instruction->opcode() == HloOpcode::kWhile) {
560 // Layout of input and output of kWhile instruction must be equal and must
561 // match both input and output of body computation. Also, the input of
562 // condition computation must match kWhile layout.
563 HloComputation* body = instruction->while_body();
564 HloComputation* condition = instruction->while_condition();
565 const HloInstruction* init = instruction->operand(0);
566 const ComputationLayout& body_layout =
567 FindOrDie(computation_layouts_, body);
568 const ComputationLayout& condition_layout =
569 FindOrDie(computation_layouts_, condition);
570
571 // Check a few invariants irrespective of layout.
572 CHECK_EQ(1, instruction->operand_count());
573 CHECK_EQ(1, body->num_parameters());
574 CHECK_EQ(1, condition->num_parameters());
575 DCHECK(ShapeUtil::Compatible(body_layout.result_shape(),
576 body_layout.parameter_shape(0)));
577 DCHECK(ShapeUtil::Compatible(body_layout.result_shape(),
578 condition_layout.parameter_shape(0)));
579 DCHECK(ShapeUtil::Compatible(body_layout.result_shape(), init->shape()));
580
581 // Return error if earlier layout assignment of the embedded computations
582 // has produced conflicting layouts.
583 if (!ShapeUtil::Equal(body_layout.result_shape(),
584 body_layout.parameter_shape(0))) {
585 return InternalError(
586 "Parameter and result of body computation %s of while instruction "
587 "%s have different layouts: %s vs %s",
588 body->name().c_str(), instruction->name().c_str(),
589 ShapeUtil::HumanString(body_layout.result_shape()).c_str(),
590 ShapeUtil::HumanString(body_layout.parameter_shape(0)).c_str());
591 }
592 if (!ShapeUtil::Equal(body->root_instruction()->shape(),
593 condition->parameter_instruction(0)->shape())) {
594 return InternalError(
595 "Parameter of condition computation %s of while instruction "
596 "%s does not match body computation %s result: %s vs %s",
597 condition->name().c_str(), instruction->name().c_str(),
598 body->name().c_str(),
599 ShapeUtil::HumanString(condition_layout.parameter_shape(0)).c_str(),
600 ShapeUtil::HumanString(body_layout.result_shape()).c_str());
601 }
602
603 // Constrain the output and the operand of the while instruction to match
604 // the computations.
605 TF_RETURN_IF_ERROR(constraints->SetInstructionLayout(
606 body_layout.result_shape(), instruction));
607 TF_RETURN_IF_ERROR(constraints->SetOperandLayout(
608 body_layout.result_shape(), instruction, 0));
609 } else if (instruction->opcode() == HloOpcode::kConditional) {
610 // The layout of the true and false computations must match, and must
611 // be the layout of the kConditional instruction.
612 TF_RET_CHECK(instruction->operand_count() == 3);
613
614 HloComputation* true_computation = instruction->true_computation();
615 HloComputation* false_computation = instruction->false_computation();
616 const HloInstruction* true_operand = instruction->operand(1);
617 const HloInstruction* false_operand = instruction->operand(2);
618
619 TF_RET_CHECK(true_computation->num_parameters() == 1);
620 TF_RET_CHECK(false_computation->num_parameters() == 1);
621 ComputationLayout& true_computation_layout =
622 FindOrDie(computation_layouts_, true_computation);
623 ComputationLayout& false_computation_layout =
624 FindOrDie(computation_layouts_, false_computation);
625
626 DCHECK(ShapeUtil::Compatible(true_operand->shape(),
627 true_computation_layout.parameter_shape(0)));
628 DCHECK(ShapeUtil::Compatible(
629 false_operand->shape(), false_computation_layout.parameter_shape(0)));
630
631 TF_RETURN_IF_ERROR(constraints->SetInstructionLayout(
632 true_computation_layout.result_shape(), instruction));
633 TF_RETURN_IF_ERROR(constraints->SetOperandLayout(
634 true_computation_layout.parameter_shape(0), instruction, 1,
635 /*mandatory=*/true));
636 TF_RETURN_IF_ERROR(constraints->SetOperandLayout(
637 false_computation_layout.parameter_shape(0), instruction, 2,
638 /*mandatory=*/true));
639 } else if (instruction->opcode() == HloOpcode::kCustomCall) {
640 if (!CustomCallRequiresMajorFirstLayout(instruction)) {
641 continue;
642 }
643 // Add constraints for kCustomCall instruction operands and instructions.
644 // For now we only support major-first layouts for all inputs and outputs.
645 Shape result_shape = ShapeUtil::MakeShapeWithDescendingLayout(
646 instruction->shape().element_type(),
647 AsInt64Slice(instruction->shape().dimensions()));
648 TF_RETURN_IF_ERROR(
649 constraints->SetInstructionLayout(result_shape, instruction));
650 for (int64 i = 0; i < instruction->operand_count(); ++i) {
651 const Shape& operand_shape = instruction->operand(i)->shape();
652 // Opaque operands don't get a layout constraint.
653 if (ShapeUtil::IsOpaque(operand_shape)) {
654 continue;
655 }
656
657 Shape row_major_operand_shape =
658 ShapeUtil::MakeShapeWithDescendingLayout(
659 operand_shape.element_type(),
660 AsInt64Slice(operand_shape.dimensions()));
661 TF_RETURN_IF_ERROR(constraints->SetOperandLayout(
662 row_major_operand_shape, instruction, i));
663 }
664 }
665 }
666
667 // Finally set the result layout to match ComputationLayout.
668 return constraints->SetResultLayout(
669 computation_layout.result_layout().shape());
670}

796Status LayoutAssignment::CheckLayouts(HloModule* module) {
797 TF_ASSIGN_OR_RETURN(auto points_to_analysis,
798 TuplePointsToAnalysis::Run(module));
799 for (auto* computation : module->MakeNonfusionComputations()) {
800 for (auto* instruction : computation->instructions()) {
801 // Verify every instruction has a layout and the layout is valid for the
802 // shape.
803 TF_RET_CHECK(LayoutUtil::HasLayout(instruction->shape()));
804 TF_RETURN_IF_ERROR(ShapeUtil::ValidateShape(instruction->shape()));
805
806 // Use points-to analysis to verify that every subshape element in the
807 // output of the instruction matches the layout of the logical buffer
808 // which could be the source of the subshape value.
809 const PointsToSet& points_to_set =
810 points_to_analysis->GetPointsToSet(instruction);
811 TF_RETURN_IF_ERROR(points_to_set.ForEachElementWithStatus(
812 [&instruction](ShapeIndex index,
813 const PointsToSet::BufferList& buffers) -> Status {
814 if (ShapeUtil::IsLeafIndex(instruction->shape(), index)) {
815 const Shape& instruction_subshape =
816 ShapeUtil::GetSubshape(instruction->shape(), index);
817 for (const LogicalBuffer* buffer : buffers) {
818 if (!ShapeUtil::Equal(instruction_subshape, buffer->shape())) {
819 return InternalError(
820 "Layout of instruction %s at index {%s} does not match "
821 "source LogicalBuffer %s: %s vs %s",
822 instruction->name().c_str(),
823 tensorflow::str_util::Join(index, ",").c_str(),
824 buffer->ToString().c_str(),
825 ShapeUtil::HumanStringWithLayout(instruction_subshape)
826 .c_str(),
827 ShapeUtil::HumanStringWithLayout(buffer->shape())
828 .c_str());
829 }
830 }
831 }
832 return Status::OK();
833 }));
834
835 // Verify instructions that have special layout constraints.
836 switch (instruction->opcode()) {
837 case HloOpcode::kCall:
838 TF_RETURN_IF_ERROR(CheckCallLayout(
839 instruction,
840 FindOrDie(computation_layouts_, instruction->to_apply())));
841 break;
842 case HloOpcode::kCustomCall:
843 if (CustomCallRequiresMajorFirstLayout(instruction)) {
844 TF_RETURN_IF_ERROR(CheckCustomCallLayout(instruction));
845 }
846 break;
847 case HloOpcode::kFusion:
848 TF_RETURN_IF_ERROR(CheckFusionLayout(instruction));
849 break;
850 case HloOpcode::kParameter:
851 TF_RETURN_IF_ERROR(CheckParameterLayout(
852 instruction,
853 FindOrDie(computation_layouts_, instruction->parent())));
854 break;
855 case HloOpcode::kConstant:
856 TF_RETURN_IF_ERROR(CheckConstantLayout(instruction));
857 break;
858 case HloOpcode::kWhile:
859 TF_RETURN_IF_ERROR(CheckWhileLayout(
860 instruction,
861 FindOrDie(computation_layouts_, instruction->while_condition()),
862 FindOrDie(computation_layouts_, instruction->while_body())));
863 break;
864 case HloOpcode::kConditional:
865 TF_RETURN_IF_ERROR(CheckConditionalLayout(
866 instruction,
867 FindOrDie(computation_layouts_, instruction->true_computation()),
868 FindOrDie(computation_layouts_,
869 instruction->false_computation())));
870 break;
871 default:
872 break;
873 }
874 }
875 }
876
877 // Finally verify the result layout matches the layout of the entry
878 // computation root.
879 TF_RET_CHECK(ShapeUtil::Equal(
880 module->entry_computation()->root_instruction()->shape(),
881 FindOrDie(computation_layouts_, module->entry_computation())
882 .result_layout()
883 .shape()));
884
885 return Status::OK();
886}

907std::unique_ptr<Layout> LayoutAssignment::ChooseOperandLayoutFromOutputLayout(
908 const Layout& output_layout, const HloInstruction* instruction,
909 int64 operand_no) {
910 const HloInstruction* operand = instruction->operand(operand_no);
911
912 CHECK(ShapeUtil::IsArray(instruction->shape()));
913 CHECK(ShapeUtil::IsArray(operand->shape()));
914
915 if (instruction->IsElementwiseOnOperand(operand_no) &&
916 !ShapeUtil::IsScalar(operand->shape()) &&
917 ShapeUtil::Rank(operand->shape()) ==
918 ShapeUtil::Rank(instruction->shape())) {
919 // Assign operands the same layout as the instruction, so that
920 // 1) the elementwise operation can reuse its operand's buffer, and
921 // 2) the input and output elements can reuse the same linear index.
922 //
923 // TODO(jingyue): Other operations, such as kSlice and kConcat, can benefit
924 // from assigning the same layout to input and output.
925 return MakeUnique<Layout>(output_layout);
926 }
927
928 if (instruction->opcode() == HloOpcode::kReshape) {
929 // Prefer the operand layout that makes the reshape an bitcast. If any
930 // dimension bound is 1 in the operand shape, there may be several such
931 // layouts. So if 'output_layout' is the default layout, try if the
932 // reshape is a bitcast when using the same layout. This may avoid copy
933 // operations. For similar reasons, if the operand and output have the same
934 // rank, try to match the operand's layout to the output.
935 if (ShapeUtil::TrueRank(operand->shape()) == 1 &&
936 ShapeUtil::Rank(instruction->shape()) == 1) {
937 // Don't assign a layout in case of R1 -> effective R1 reshape.
938 return nullptr;
939 }
940 const Shape& output_shape = instruction->shape();
941 Shape output_shape_with_layout = ShapeUtil::MakeShapeWithLayout(
942 output_shape.element_type(), AsInt64Slice(output_shape.dimensions()),
943 LayoutUtil::MinorToMajor(output_layout));
944 Shape operand_shape = operand->shape();
945 *operand_shape.mutable_layout() =
946 LayoutUtil::GetDefaultLayoutForShape(operand_shape);
947 if (ShapeUtil::ReshapeIsBitcast(operand_shape, output_shape_with_layout)) {
948 return MakeUnique<Layout>(operand_shape.layout());
949 }
950 if (ShapeUtil::Rank(operand_shape) == ShapeUtil::Rank(output_shape)) {
951 *operand_shape.mutable_layout() = output_layout;
952 if (ShapeUtil::ReshapeIsBitcast(operand_shape,
953 output_shape_with_layout)) {
954 return MakeUnique<Layout>(output_layout);
955 }
956 }
957 auto aligned_operand_shape =
958 ShapeUtil::AlignLayouts(output_shape_with_layout, operand_shape);
959 if (aligned_operand_shape) {
960 auto operand_layout = aligned_operand_shape.value().layout();
961 TF_CHECK_OK(
962 LayoutUtil::ValidateLayoutForShape(operand_layout, operand_shape));
963 return MakeUnique<Layout>(operand_layout);
964 }
965 }
966
967 if (instruction->opcode() == HloOpcode::kTranspose) {
968 // Pick the operand layout that makes the transpose a bitcast.
969 int64 rank = ShapeUtil::Rank(instruction->shape());
970 std::vector<int64> new_minor_to_major(rank);
971 for (int64 i = 0; i < rank; ++i) {
972 int64 output_dim = LayoutUtil::Minor(output_layout, i);
973 int64 operand_dim = instruction->dimensions(output_dim);
974 new_minor_to_major[i] = operand_dim;
975 }
976 Layout operand_layout = LayoutUtil::MakeLayout(new_minor_to_major);
977 TF_CHECK_OK(
978 LayoutUtil::ValidateLayoutForShape(operand_layout, operand->shape()));
979 return MakeUnique<Layout>(operand_layout);
980 }
981
982 return nullptr;
983}

985std::unique_ptr<Layout> LayoutAssignment::ChooseOutputLayoutFromOperandLayout(
986 const Layout& operand_layout, const HloInstruction* user,
987 int64 operand_no) {
988 const HloInstruction* operand = user->operand(operand_no);
989
990 CHECK(ShapeUtil::IsArray(user->shape()) &&
991 ShapeUtil::IsArray(operand->shape()));
992
993 if (user->IsElementwiseOnOperand(operand_no) &&
994 !ShapeUtil::IsScalar(operand->shape()) &&
995 ShapeUtil::Rank(operand->shape()) == ShapeUtil::Rank(user->shape())) {
996 // Assign users the same layout as the operand.
997 return MakeUnique<Layout>(operand_layout);
998 }
999
1000 if (user->opcode() == HloOpcode::kReshape) {
1001 // Prefer the user layout that makes the reshape an bitcast. If any
1002 // dimension bound is 1 in the user shape, there may be several such
1003 // layouts. So if 'operand_layout' is the default layout, try if the
1004 // reshape is a bitcast when using the same layout. This may avoid copy
1005 // operations. For similar reasons, if the operand and output have the same
1006 // rank, try to match the outputs's layout to the operand.
1007 if (ShapeUtil::Rank(operand->shape()) == 1 &&
1008 ShapeUtil::TrueRank(user->shape()) == 1) {
1009 // Don't assign a layout in case of R1 -> effective R1 reshape.
1010 return nullptr;
1011 }
1012 Shape operand_shape_with_layout = ShapeUtil::MakeShapeWithLayout(
1013 operand->shape().element_type(),
1014 AsInt64Slice(operand->shape().dimensions()),
1015 LayoutUtil::MinorToMajor(operand_layout));
1016 Shape output_shape = user->shape();
1017 *output_shape.mutable_layout() =
1018 LayoutUtil::GetDefaultLayoutForShape(output_shape);
1019 if (ShapeUtil::ReshapeIsBitcast(output_shape, operand_shape_with_layout)) {
1020 return MakeUnique<Layout>(output_shape.layout());
1021 }
1022 if (ShapeUtil::Rank(operand->shape()) == ShapeUtil::Rank(output_shape)) {
1023 *output_shape.mutable_layout() = operand_layout;
1024 if (ShapeUtil::ReshapeIsBitcast(output_shape,
1025 operand_shape_with_layout)) {
1026 return MakeUnique<Layout>(operand_layout);
1027 }
1028 }
1029 auto aligned_user_shape =
1030 ShapeUtil::AlignLayouts(operand_shape_with_layout, output_shape);
1031 if (aligned_user_shape) {
1032 auto user_layout = aligned_user_shape.value().layout();
1033 TF_CHECK_OK(
1034 LayoutUtil::ValidateLayoutForShape(user_layout, output_shape));
1035 return MakeUnique<Layout>(user_layout);
1036 }
1037 }
1038
1039 if (user->opcode() == HloOpcode::kTranspose) {
1040 // Pick the user layout that makes the transpose a bitcast.
1041 int64 rank = ShapeUtil::Rank(user->shape());
1042 std::vector<int64> new_minor_to_major(rank);
1043 auto inverse_dimensions = InversePermutation(user->dimensions());
1044 for (int64 i = 0; i < rank; ++i) {
1045 int64 operand_dim = LayoutUtil::Minor(operand_layout, i);
1046 int64 user_dim = inverse_dimensions[operand_dim];
1047 new_minor_to_major[i] = user_dim;
1048 }
1049 Layout user_layout = LayoutUtil::MakeLayout(new_minor_to_major);
1050 TF_CHECK_OK(LayoutUtil::ValidateLayoutForShape(user_layout, user->shape()));
1051 return MakeUnique<Layout>(user_layout);
1052 }
1053
1054 return nullptr;
1055}

1057Status LayoutAssignment::PropagateConstraints(LayoutConstraints* constraints) {
1058 // Gathers all initial constraints in a worklist and propagates them in
1059 // depth-first order. DFS order seems to be better than BFS because a
1060 // constraint is propagated as far as possible before propagating unrelated
1061 // constraints which makes it less likely that conflicting constraints will be
1062 // propagated to instructions. However, we should experiment with other orders
1063 // too.
1064 std::deque<const LayoutConstraint*> worklist;
1065
1066 // Lambda for moving newly added constraints to the worklist.
1067 auto add_new_constraints_to_worklist = [constraints, &worklist]() {
1068 // Add constraints to the front of the deque for DFS ordering.
1069 for (auto* constraint : constraints->ConsumeAddedConstraints()) {
1070 if (constraint->dfs()) {
1071 worklist.push_front(constraint);
1072 } else {
1073 worklist.push_back(constraint);
1074 }
1075 }
1076 };
1077 add_new_constraints_to_worklist();
1078
1079 while (!worklist.empty()) {
1080 const LayoutConstraint* layout_constraint = worklist.front();
1081 worklist.pop_front();
1082 VLOG(2) << "Propagating " << layout_constraint->ToString()
1083 << " to its neighbors.";
1084 if (auto* buffer_constraint =
1085 dynamic_cast<const BufferLayoutConstraint*>(layout_constraint)) {
1086 TF_RETURN_IF_ERROR(
1087 PropagateBufferConstraint(*buffer_constraint, constraints));
1088 } else if (auto* operand_constraint =
1089 dynamic_cast<const OperandLayoutConstraint*>(
1090 layout_constraint)) {
1091 TF_RETURN_IF_ERROR(
1092 PropagateOperandConstraint(*operand_constraint, constraints));
1093 } else if (auto* result_constraint =
1094 dynamic_cast<const ResultLayoutConstraint*>(
1095 layout_constraint)) {
1096 TF_RETURN_IF_ERROR(
1097 PropagateResultConstraint(*result_constraint, constraints));
1098 } else {
1099 LOG(FATAL) << "Invalid constraint type: " << *layout_constraint;
1100 }
1101
1102 add_new_constraints_to_worklist();
1103 }
1104 return Status::OK();
1105}

1137Status LayoutAssignment::PropagateUseConstraintToDefs(
1138 const ShapeLayout& shape_layout, const HloInstruction* instruction,
1139 LayoutConstraints* constraints) {
1140 // Try to set all logical buffers which may be sources of the given operand to
1141 // match the given layout.
1142 const PointsToSet& points_to_set =
1143 constraints->points_to_analysis().GetPointsToSet(instruction);
1144 return points_to_set.ForEachElementWithStatus(
1145 [this, &shape_layout, constraints](
1146 const ShapeIndex& index,
1147 const PointsToSet::BufferList& buffers) -> Status {
1148 if (ShapeUtil::IsLeafIndex(shape_layout.shape(), index)) {
1149 for (const LogicalBuffer* buffer : buffers) {
1150 if (constraints->BufferLayout(*buffer) == nullptr &&
1151 ShapeUtil::IsArray(buffer->shape())) {
1152 TF_RETURN_IF_ERROR(constraints->SetBufferLayout(
1153 ShapeUtil::GetSubshape(shape_layout.shape(), index).layout(),
1154 *buffer, /*mandatory=*/true));
1155 }
1156 }
1157 }
1158 return Status::OK();
1159 });
1160}

1162Status LayoutAssignment::PropagateOperandConstraint(
1163 const OperandLayoutConstraint& operand_constraint,
1164 LayoutConstraints* constraints) {
1165 // Try to set the layout of the logical buffers in the given operand to match
1166 // the constrained layout. This avoids copies.
1167 TF_RETURN_IF_ERROR(
1168 PropagateUseConstraintToDefs(operand_constraint.shape_layout(),
1169 operand_constraint.operand(), constraints));
1170
1171 // For array-shaped operands and user instructions try to pick a minimum cost
1172 // layout. For example, if the operand of a elementwise instruction is
1173 // constained to a certain layout we want the output of the instruction to
1174 // have the same layout.
1175 const HloInstruction* operand = operand_constraint.operand();
1176 const HloInstruction* user = operand_constraint.instruction();
1177 if (!ShapeUtil::IsArray(operand->shape()) ||
1178 !ShapeUtil::IsArray(user->shape())) {
1179 return Status::OK();
1180 }
1181
1182 // Only try to choose a low cost layout if the instruction 'user' defines its
1183 // output (ie, doesn't forward a buffer from elsewhere).
1184 if (constraints->OperandBufferForwarded(user,
1185 operand_constraint.operand_no())) {
1186 return Status::OK();
1187 }
1188 TF_ASSIGN_OR_RETURN(
1189 const LogicalBuffer* buffer,
1190 constraints->points_to_analysis().GetBufferDefinedAt(user, /*index=*/{}));
1191
1192 if (constraints->BufferLayout(*buffer) == nullptr) {
1193 std::unique_ptr<Layout> layout = ChooseOutputLayoutFromOperandLayout(
1194 operand_constraint.shape_layout().layout(), user,
1195 operand_constraint.operand_no());
1196 if (layout != nullptr) {
1197 TF_RETURN_IF_ERROR(
1198 constraints->SetBufferLayout(*layout, *buffer, /*mandatory=*/false));
1199 }
1200 }
1201 return Status::OK();
1202}

1204Status LayoutAssignment::PropagateBufferConstraint(
1205 const BufferLayoutConstraint& buffer_constraint,
1206 LayoutConstraints* constraints) {
1207 // Only propagate array layouts.
1208 const LogicalBuffer& buffer = buffer_constraint.buffer();
1209 if (!buffer.IsArray()) {
1210 return Status::OK();
1211 }
1212
1213 // If this buffer is the result of an array-shaped op (as opposed to an array
1214 // element in a tuple) try to propagate the layout to its operands.
1215 if (buffer.IsTopLevel()) {
1216 const HloInstruction* instruction = buffer.instruction();
1217 // Propagate the def-constraint on an instruction to the use-constraints on
1218 // its operands (use-def propagation).
1219 for (int64 operand_no = 0; operand_no < instruction->operand_count();
1220 ++operand_no) {
1221 if (constraints->OperandLayout(instruction, operand_no) == nullptr &&
1222 ShapeUtil::IsArray(instruction->operand(operand_no)->shape())) {
1223 std::unique_ptr<Layout> operand_layout =
1224 ChooseOperandLayoutFromOutputLayout(buffer_constraint.layout(),
1225 instruction, operand_no);
1226 if (operand_layout != nullptr) {
1227 TF_RETURN_IF_ERROR(constraints->SetArrayOperandLayout(
1228 *operand_layout, instruction, operand_no, /*mandatory=*/true));
1229 }
1230 }
1231 }
1232 }
1233 return PropagateBufferConstraintToUses(buffer_constraint, constraints);

1236Status LayoutAssignment::PropagateBufferConstraintToUses(
1237 const BufferLayoutConstraint& buffer_constraint,
1238 LayoutConstraints* constraints) {
1239 const LogicalBuffer& buffer = buffer_constraint.buffer();
1240 TF_RET_CHECK(buffer.IsArray());
1241
1242 // Propagate the layout to all array uses of the logical buffer. This skips
1243 // uses of the buffer where the buffer is the element of a tuple.
1244 for (const auto& user_operand_no :
1245 GetArrayUsesOfBuffer(buffer, constraints->points_to_analysis())) {
1246 const HloInstruction* user = user_operand_no.first;
1247 int64 operand_no = user_operand_no.second;
1248 // Only add an operand constraint if the user does not forward the buffer
1249 // because this case is not handled is SetOperandLayout.
1250 if (constraints->OperandLayout(user, operand_no) == nullptr &&
1251 !constraints->OperandBufferForwarded(user, operand_no)) {
1252 TF_RETURN_IF_ERROR(constraints->SetArrayOperandLayout(
1253 buffer_constraint.layout(), user, operand_no, /*mandatory=*/false));
1254 }
1255 }
1256
1257 return Status::OK();
1258}

1272// Infers the layout of the array at the given index in the given instruction's
1273// output using points-to analysis. Precondition: The given instruction must
1274// not produce this array value (that is, the array is forwarded from the
1275// instruction's operands).
1276StatusOr<Layout> InferArrayLayout(
1277 const TuplePointsToAnalysis& points_to_analysis,
1278 HloInstruction* instruction, const ShapeIndex& index) {
1279 // This function should only be called for array shapes which don't yet have
1280 // layouts.
1281 const Shape& subshape = ShapeUtil::GetSubshape(instruction->shape(), index);
1282 TF_RET_CHECK(ShapeUtil::IsArray(subshape));
1283 TF_RET_CHECK(!subshape.has_layout());
1284
1285 // The instruction should not define the buffer at this index.
1286 TF_RET_CHECK(
1287 !points_to_analysis.InstructionDefinesBufferAtIndex(instruction, index))
1288 << instruction->ToString();
1289
1290 const auto& source_buffers =
1291 points_to_analysis.GetPointsToSet(instruction).element(index);
1292 TF_RET_CHECK(!source_buffers.empty());
1293
1294 // Verify the layout is the same for every LogicalBuffer which this location
1295 // ('instruction' and 'index') points to.
1296 const Layout* first_buffer_layout = nullptr;
1297 for (const LogicalBuffer* source_buffer : source_buffers) {
1298 if (!source_buffer->shape().has_layout()) {
1299 // This should not happen because we've assigned layouts to all
1300 // instructions preceding this one.
1301 return InternalError("LogicalBuffer %s does not have a layout",
1302 source_buffer->ToString().c_str());
1303 }
1304
1305 if (first_buffer_layout == nullptr) {
1306 first_buffer_layout = &source_buffer->shape().layout();
1307 } else if (!LayoutUtil::Equal(source_buffer->shape().layout(),
1308 *first_buffer_layout)) {
1309 // The points-to set is ambiguous for this index and the different source
1310 // buffers have different layouts. This case is possible in valid XLA
1311 // computations because we do not propagate BufferLayoutConstraints to all
1312 // LogicalBuffers which may alias the constrained LogicalBuffer at some
1313 // point in the computation.
1314 return FailedPrecondition(
1315 "Array at index {%s} in instruction %s aliases buffers %s "
1316 "and %s which have different layouts",
1317 tensorflow::str_util::Join(index, ",").c_str(),
1318 instruction->name().c_str(), source_buffers[0]->ToString().c_str(),
1319 source_buffer->ToString().c_str());
1320 }
1321 }
1322
1323 return *first_buffer_layout;
1324}
1325
1326// For fusion instructions, set the layout of each fused parameter instruction
1327// to match the layout of its corresponding fusion instruction operand. Also,
1328// set the layout of the fused root to match the layout of the fusion
1329// instruction itself.
1330Status SetFusionLayouts(HloInstruction* fusion) {
1331 TF_RET_CHECK(fusion->opcode() == HloOpcode::kFusion);
1332 for (auto* fused_instruction :
1333 fusion->fused_instructions_computation()->MakeInstructionPostOrder()) {
1334 if (fused_instruction->opcode() == HloOpcode::kParameter) {
1335 const HloInstruction* fusion_operand =
1336 fusion->operand(fused_instruction->parameter_number());
1337 DCHECK(ShapeUtil::Compatible(fusion_operand->shape(),
1338 fused_instruction->shape()));
1339 TF_RETURN_IF_ERROR(LayoutUtil::CopyLayoutBetweenShapes(
1340 fusion_operand->shape(), fused_instruction->mutable_shape()));
1341 } else if (fused_instruction == fusion->fused_expression_root()) {
1342 // The layout of the root of the fused expression must match the fusion
1343 // instruction layout.
1344 DCHECK(
1345 ShapeUtil::Compatible(fusion->shape(), fused_instruction->shape()));
1346 TF_RETURN_IF_ERROR(LayoutUtil::CopyLayoutBetweenShapes(
1347 fusion->shape(), fused_instruction->mutable_shape()));
1348 } else if (fused_instruction->opcode() == HloOpcode::kGetTupleElement) {
1349 // A GTE inherits its layout from its operand (which should ultimately be
1350 // a parameter).
1351 TF_RETURN_IF_ERROR(LayoutUtil::CopyLayoutBetweenShapes(
1352 fused_instruction->operand(0)->shape().tuple_shapes(
1353 fused_instruction->tuple_index()),
1354 fused_instruction->mutable_shape()));
1355 } else if (fused_instruction->opcode() == HloOpcode::kConstant) {
1356 // Give constants the layout of their literal.
1357 TF_RETURN_IF_ERROR(LayoutUtil::CopyLayoutBetweenShapes(
1358 fused_instruction->literal().shape(),
1359 fused_instruction->mutable_shape()));
1360 } else if (fused_instruction->opcode() == HloOpcode::kInfeed) {
1361 // Nop; leave the infeed layout alone.
1362 } else {
1363 // Other instructions don't have layouts inside of fusion nodes.
1364 LayoutUtil::ClearLayout(fused_instruction->mutable_shape());
1365 }
1366 }
1367
1368 return Status::OK();
1369}

1373Status LayoutAssignment::AssignLayouts(const LayoutConstraints& constraints,
1374 HloComputation* computation) {
1375 VLOG(2) << "Assigning layouts to computation: " << computation->name();
1376 XLA_VLOG_LINES(2, computation->ToString());
1377 XLA_VLOG_LINES(2, constraints.ToString());
1378
1379 for (HloInstruction* instruction : computation->MakeInstructionPostOrder()) {
1380 LayoutUtil::ClearLayout(instruction->mutable_shape());
1381
1382 // Create a copy of an operand if the operand instruction's layout does not
1383 // match the use constraint (OperandLayoutConstraint).
1384 for (int64 operand_no = 0; operand_no < instruction->operand_count();
1385 ++operand_no) {
1386 const ShapeLayout* operand_layout =
1387 constraints.OperandLayout(instruction, operand_no);
1388 if (operand_layout != nullptr) {
1389 TF_RETURN_IF_ERROR(CopyOperandIfLayoutsDiffer(*operand_layout,
1390 instruction, operand_no));
1391 }
1392 }
1393
1394 // Set the layouts of the array shapes this instruction defines as indicated
1395 // by the respective BufferLayoutConstraints. Any array shapes in the output
1396 // of the instruction which are not defined by the instruction (eg, array
1397 // elements in a Tuple instruction) will be assigned below via inference.
1398 for (const LogicalBuffer* buffer :
1399 constraints.points_to_analysis().GetBuffersDefinedByInstruction(
1400 instruction)) {
1401 if (!ShapeUtil::IsArray(buffer->shape())) {
1402 continue;
1403 }
1404
1405 TF_RET_CHECK(buffer->instruction() == instruction);
1406 const Layout* buffer_layout = constraints.BufferLayout(*buffer);
1407 TF_RET_CHECK(buffer_layout != nullptr);
1408
1409 if (instruction->opcode() == HloOpcode::kConstant) {
1410 // For constants, we also need to change the layout of the internal
1411 // literal.
1412 instruction->RelayoutConstant(*buffer_layout, buffer->index());
1413 } else {
1414 Shape* buffer_subshape = ShapeUtil::GetMutableSubshape(
1415 instruction->mutable_shape(), buffer->index());
1416 *buffer_subshape->mutable_layout() = *buffer_layout;
1417 }
1418 }
1419
1420 // Any remaining layouts in the output of the instruction must be
1421 // inferrable using points-to analysis.
1422 TF_RETURN_IF_ERROR(ShapeUtil::ForEachMutableSubshapeWithStatus(
1423 instruction->mutable_shape(),
1424 [instruction, &constraints](Shape* subshape, const ShapeIndex& index) {
1425 if (subshape->has_layout() || !ShapeUtil::IsArray(*subshape)) {
1426 return Status::OK();
1427 }
1428 // Set Layout of subshape to match layout of LogicalBuffer which
1429 // produces it.
1430 TF_ASSIGN_OR_RETURN(*subshape->mutable_layout(),
1431 InferArrayLayout(constraints.points_to_analysis(),
1432 instruction, index));
1433 return Status::OK();
1434 }));
1435
1436 // Fusion instructions require some layouts to be set on fused instructions
1437 // inside the fusion instruction.
1438 if (instruction->opcode() == HloOpcode::kFusion) {
1439 TF_RETURN_IF_ERROR(SetFusionLayouts(instruction));
1440 }
1441
1442 // Execute extra verification step once the layout has been finalized.
1443 TF_RETURN_IF_ERROR(Verify(instruction));
1444
1445 // Verify all layouts in the shape have been set.
1446 TF_RET_CHECK(LayoutUtil::HasLayout(instruction->shape()));
1447 }
1448
1449 // Copy the root instruction's result if its layout does not match the result
1450 // layout constraint.
1451 if (constraints.ResultLayout() != nullptr &&
1452 !constraints.ResultLayout()->MatchesLayoutInShape(
1453 computation->root_instruction()->shape())) {
1454 TF_ASSIGN_OR_RETURN(
1455 HloInstruction * new_root,
1456 CreateCopyWithNewLayout(constraints.ResultLayout()->shape(),
1457 computation->root_instruction()));
1458 computation->set_root_instruction(new_root);
1459 }
1460
1461 return Status::OK();
1462}

1464Status LayoutAssignment::RunOnComputation(
1465 const ComputationLayout& computation_layout,
1466 const TuplePointsToAnalysis& points_to_analysis,
1467 HloComputation* computation,
1468 ChannelLayoutConstraints* channel_constraints) {
1469 DCHECK(computation_layout.LayoutIsSet());
1470 InsertOrDie(&computation_layouts_, computation, computation_layout);
1471 VLOG(2) << "LayoutAssignment::RunOnComputation(" << computation->name()
1472 << ")";
1473 VLOG(2) << " ComputationLayout = " << computation_layout.ToString();
1474
1475 // Construct LayoutConstraints with all layout constraints of the computation.
1476 LayoutConstraints constraints(points_to_analysis, computation);
1477
1478 // Add constraints required for correctness on all backends (eg, entry
1479 // parameter layout constraints).
1480 TF_RETURN_IF_ERROR(AddMandatoryConstraints(
1481 computation_layout, channel_constraints, computation, &constraints));
1482
1483 // Add any backend-specific constraints.
1484 TF_RETURN_IF_ERROR(AddBackendConstraints(&constraints));
1485
1486 // Propagates layouts from mandatory and backend constraints.
1487 TF_RETURN_IF_ERROR(PropagateConstraints(&constraints));
1488
1489 // While any unconstrained buffers remain, pick an arbitrary buffer, give it a
1490 // layout and propagate the change.
1491 while (!constraints.unconstrained_buffer_ids().empty()) {
1492 int unconstrained_count = constraints.unconstrained_buffer_ids().size();
1493
1494 // Arbitrarily pick the first unconstrained buffer and give it the default
1495 // layout (or the literal layout, in case of constants). By construction
1496 // unconstrained_buffers() has a stable sort based on LogicalBuffer::Id.
1497 const LogicalBuffer& buffer = points_to_analysis.GetBuffer(
1498 *constraints.unconstrained_buffer_ids().begin());
1499 const HloInstruction* instruction = buffer.instruction();
1500 Layout new_layout =
1501 instruction->opcode() == HloOpcode::kConstant
1502 ? ShapeUtil::GetSubshape(instruction->literal().shape(),
1503 buffer.index())
1504 .layout()
1505 : LayoutUtil::GetDefaultLayoutForShape(buffer.shape());
1506 TF_RETURN_IF_ERROR(constraints.SetBufferLayout(new_layout, buffer,
1507 /*mandatory=*/false));
1508
1509 TF_RETURN_IF_ERROR(PropagateConstraints(&constraints));
1510
1511 // To verify progress has been made, check that the number of unconstrained
1512 // buffers has been reduced.
1513 CHECK_LT(constraints.unconstrained_buffer_ids().size(),
1514 unconstrained_count);
1515 }
1516
1517 // All logical buffers should have constraints at this point. All that
1518 // remains is assign the constraints to the buffers and infer layouts for
1519 // aliased buffers.
1520 TF_RETURN_IF_ERROR(AssignLayouts(constraints, computation));
1521
1522 // Record the layouts assigned for any communication ops in
1523 // channel_constraints so that they are constrained for future modules.
1524 for (HloInstruction* instruction : computation->instructions()) {
1525 if (instruction->opcode() == HloOpcode::kSend) {
1526 channel_constraints->ConstrainChannel(
1527 instruction->channel_id(), instruction->operand(0)->shape().layout());
1528 } else if (instruction->opcode() == HloOpcode::kRecvDone) {
1529 channel_constraints->ConstrainChannel(instruction->channel_id(),
1530 instruction->shape().layout());
1531 }
1532 }
1533 return Status::OK();
1534}

1536StatusOr<bool> LayoutAssignment::Run(HloModule* module) {
1537 VLOG(2) << "Running layout assignment on module " << module->name();
1538 XLA_VLOG_LINES(3, module->ToString());
1539 if (VLOG_IS_ON(10)) {
1540 hlo_graph_dumper::DumpGraph(*module->entry_computation(),
1541 "before layout assignment",
1542 module->config().debug_options());
1543 }
1544
1545 TF_ASSIGN_OR_RETURN(auto points_to_analysis,
1546 TuplePointsToAnalysis::Run(module));
1547
1548 // Assign layouts to computations in an order such that a callee computation
1549 // is handled before its caller computation. This ensures that the layout of
1550 // all callers of a computation will agree.
1551 std::list<HloComputation*> computation_post_order =
1552 module->MakeComputationPostOrder();
1553 for (auto* computation : module->MakeComputationPostOrder()) {
1554 if (computation->IsFusionComputation()) {
1555 continue;
1556 }
1557 // Clear existing layouts of the instructions. All layouts must be assigned
1558 // by the LayoutAssignment pass, except for those on infeeds, parameters,
1559 // and the computation result. The latter two are specified in
1560 // computation_layout, so we only need to keep the existing layouts for
1561 // infeeds. Clearing the layouts here avoids hiding potential bugs in the
1562 // layout assignment pass that may accidently use the existing layout.
1563 for (HloInstruction* instruction : computation->instructions()) {
1564 if (instruction->opcode() == HloOpcode::kBitcast) {
1565 // bitcasts are inherently layout sensitive and so a bitcast instruction
1566 // present in the IR before layout assignment is a bug.
1567 return InternalError(
1568 "Unexpected bitcast operation seen during layout assignment: %s.",
1569 instruction->ToString().c_str());
1570 }
1571 if (instruction->opcode() != HloOpcode::kInfeed) {
1572 LayoutUtil::ClearLayout(instruction->mutable_shape());
1573 }
1574 }
1575 if (computation == module->entry_computation()) {
1576 TF_RETURN_IF_ERROR(RunOnComputation(
1577 *entry_computation_layout_, *points_to_analysis,
1578 module->entry_computation(), channel_layout_constraints_));
1579 } else {
1580 ComputationLayout computation_layout(computation->ComputeProgramShape());
1581 // Setting all embedded computations to the default layout is potentially
1582 // suboptimal.
1583 computation_layout.SetToDefaultLayout();
1584 TF_RETURN_IF_ERROR(RunOnComputation(computation_layout,
1585 *points_to_analysis, computation,
1586 channel_layout_constraints_));
1587 }
1588 }
1589
1590 TF_RETURN_IF_ERROR(CheckLayouts(module));
1591
1592 VLOG(3) << "After layout assignment:";
1593 XLA_VLOG_LINES(3, module->ToString());
1594 if (VLOG_IS_ON(10)) {
1595 hlo_graph_dumper::DumpGraph(*module->entry_computation(),
1596 "after layout assignment",
1597 module->config().debug_options());
1598 }
1599
1600 // All layouts are reset then reassigned by this pass.
1601 return true;
1602}
