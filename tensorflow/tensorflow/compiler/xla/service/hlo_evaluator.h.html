32namespace xla {
33
34// Responsible for evaluating HLO and obtain literal as the evaluation results.
35//
36// This class is not thread-safe.
37class HloEvaluator : public DfsHloVisitorWithDefault {
38 public:
39 // Only evaluate up to max_loop_iterations per while-loop execution if
40 // specified.
41 explicit HloEvaluator(int64 max_loop_iterations = -1);
42
43 // Evaluates an HLO module and an array of pointers to literals.
44 // Returns the evaluated result as a literal if successful.
45 // Precondition: The indices of arg_literals correspond to the parameter
46 // numbers of the HLO parameters in the computation. See comment below for an
47 // example.
48 // `LiteralPtr` accepts either std::unique_ptr<Literal> or const Literal*
49 // type.
50 template <typename LiteralPtr>
51 StatusOr<std::unique_ptr<Literal>> Evaluate(
52 const HloModule& module,
53 tensorflow::gtl::ArraySlice<LiteralPtr> arg_literals);
54
55 // Evaluates an HLO computation and an array of pointers to literals.
56 // Returns the evaluated result as a literal if successful.
57 // Precondition: The indices of arg_literals correspond to the parameter
58 // numbers of the HLO parameters in the computation. For e.g., consider the
59 // following graph:
60 //
61 // *
62 // / \
63 // + Parameter1
64 // / \
65 // / \
66 // Parameter0 Constant
67 //
68 // where Parameter0 has parameter_number 0 and Parameter1 has parameter_number
69 // 1 in this computation. The input literals array will then have its first
70 // literal map to Parameter0 and the second map to Parameter1.
71 // `LiteralPtr` accepts either std::unique_ptr<Literal> or const Literal*
72 // type.
73 template <typename LiteralPtr>
74 StatusOr<std::unique_ptr<Literal>> Evaluate(
75 const HloComputation& computation,
76 tensorflow::gtl::ArraySlice<LiteralPtr> arg_literals);
77
78 // Evaluates a single HLO instruction and an array of pointers to literals.
79 // Return the evaluated result as literal if successful.
80 // Precondition:
81 // 1. argument literals correspond to the input instruction's parameters in
82 // their post-ordering.
83 // 2. the instruction's operands must be of either Parameter or Constant type.
84 // `LiteralPtr` accepts either std::unique_ptr<Literal> or const Literal*
85 // type.
86 template <typename LiteralPtr>
87 StatusOr<std::unique_ptr<Literal>> Evaluate(
88 HloInstruction* instruction,
89 tensorflow::gtl::ArraySlice<LiteralPtr> arg_literals);
90
91 // Evaluates a single HLO instruction with constant operands.
92 // Returns the evaluated result as literal if successful.
93 // Precondition:
94 // 1. all operands of the input instruction are constants.
95 // 2. the instruction is not a Parameter operation.
96 StatusOr<std::unique_ptr<Literal>> Evaluate(HloInstruction* instruction);
97
98 // Same as Evaluate, except returning nullptr on error.
99 std::unique_ptr<Literal> TryEvaluate(HloInstruction* instruction);
100
101 // Evaluates a single HLO instruction, substituting the given literals for
102 // some of the instruction's operands.
103 //
104 // For example, given instruction = op(A, B, C) and the map
105 // {A = x, C = y}, this evaluates op(x, B, y).
106 StatusOr<std::unique_ptr<Literal>> EvaluateWithSubstitutions(
107 const HloInstruction* instruction,
108 const std::unordered_map<const HloInstruction*, const Literal*>&
109 substitutions);
110
111 protected:
112 // Templated DfsHloVisitor. Typically ReturnT here indicates the resulting
113 // literal type of each evaluated Handle* method of a TypedVisitor.
114 // There are however a few notable exceptions to this rule, notably:
115 // - HandleCompare and HandleIsFinite: where the resulting literal type is
116 // always boolean.
117 // These operations are handled outside of the parent HloEvaluator handlers
118 // instead of from within TypedVisitor.
119 //
120 // Type params:
121 // - ReturnT: The type of input and output of each operation.
122 // - ElementwiseT: The type in which internal computation are done.
123 template <typename ReturnT, typename ElementwiseT = ReturnT>
124 class TypedVisitor;
125
126 // Wraps around instruction handling to infer types before dispatching to
127 // the corresponding typed Visitor.
128 Status DefaultAction(HloInstruction* hlo) override {
129 return hlo->Visit(typed_visitors_.at(hlo->shape().element_type()).get());
130 }
131
132 Status Preprocess(HloInstruction* hlo) override;
133
134 Status Postprocess(HloInstruction* hlo) override;
135
136 // Operations that are type-agnostic or always return a specific type, such as
137 // HandleIsFinite where boolean is always returned.
138 //
139 Status HandleParameter(HloInstruction* parameter) override;
140
141 Status HandleConstant(HloInstruction* constant) override;
142
143 Status HandleConcatenate(HloInstruction* concatenate) override;
144
145 Status HandleReshape(HloInstruction* reshape) override;
146
147 Status HandleTranspose(HloInstruction* transpose) override;
148
149 Status HandleIsFinite(HloInstruction* is_finite) override;
150
151 Status HandleCompare(HloInstruction* compare) override;
152
153 Status HandleTuple(HloInstruction* tuple) override;
154
155 Status HandleGather(HloInstruction* gather) override;
156
157 Status HandleGetTupleElement(HloInstruction* get_tuple_element) override;
158
159 Status HandleCopy(HloInstruction* copy) override;
160
161 Status HandleConditional(HloInstruction* conditional) override;
162
163 Status HandleCall(HloInstruction* call) override;
164
165 Status HandleFusion(HloInstruction* fusion) override;
166
167 Status HandleWhile(HloInstruction* while_hlo) override;
168
169 Status HandleSelect(HloInstruction* select) override;
170
171 private:
172 // Returns the already-evaluated literal result for the instruction.
173 // A Constant instruction is considered evaluated and its literal will be
174 // returned directly without looking up the cache.
175 // Crash with log if the given instruction has not been evaluated previously.
176 const Literal& GetEvaluatedLiteralFor(const HloInstruction* hlo) {
177 if (hlo->IsConstant()) {
178 return hlo->literal();
179 }
180 auto it = evaluated_.find(hlo);
181 CHECK(it != evaluated_.end())
182 << "could not find evaluated value for: " << hlo->ToString();
183 return *(it->second);
184 }
185
186 // Map from a primitive type to its associated (templated) DfsHloVisitor.
187 // Note: the hash function here is only needed because current gcc std::hash
188 // does not specialize for enum types. This should however be fixed in the
189 // future: https://gcc.gnu.org/bugzilla/show_bug.cgi?id=60970#c5
190 tensorflow::gtl::FlatMap<PrimitiveType, std::unique_ptr<DfsHloVisitor>,
191 std::hash<int>>
192 typed_visitors_;
193
194 // Tracks the HLO instruction and its evaluated literal result.
195 // TODO(b/35950897): have better memory management here to free instructions
196 // that are no longer a parent for any other subsequent instruction in
197 // post-orderring.
198 // Must be cleared for each evaluation.
199 tensorflow::gtl::FlatMap<const HloInstruction*, std::unique_ptr<Literal>>
200 evaluated_;
201
202 // Caches pointers to input literals, assuming they are in post-order.
203 // Literals are not owned by this class, and they must outlive the lifetime of
204 // each invocation to the Evaluate* method.
205 // Must be cleared for each evaluation.
206 std::vector<const Literal*> arg_literals_;
207
208 // Max loop iterations to execute with no maximum if negative.
209 int64 max_loop_iterations_;
210
211 TF_DISALLOW_COPY_AND_ASSIGN(HloEvaluator);
212};
213
214} // namespace xla
