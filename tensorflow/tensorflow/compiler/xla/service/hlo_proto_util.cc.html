24HloProto MakeHloProto(const HloModule& module,
25 const BufferAssignment& assignment) {
26 HloOrderingProto proto_ordering =
27 assignment.liveness().hlo_ordering().ToProto();
28 BufferAssignmentProto proto_assignment = assignment.ToProto();
29 HloProto proto = MakeHloProto(module);
30 proto.mutable_hlo_ordering()->Swap(&proto_ordering);
31 proto.mutable_buffer_assignment()->Swap(&proto_assignment);
32 return proto;
33}
34
35HloProto MakeHloProto(const HloModule& module) {
36 HloModuleProto proto_module = module.ToProto();
37 HloProto proto;
38 proto.mutable_hlo_module()->Swap(&proto_module);
39 return proto;
40}

42StatusOr<std::vector<const Shape*>> EntryComputationParameterShapes(
43 const HloProto& hlo_proto) {
44 if (!hlo_proto.has_hlo_module()) {
45 return NotFound("HloProto missing HloModuleProto.");
46 }
47 if (!hlo_proto.hlo_module().has_program_shape()) {
48 return NotFound("HloProto missing program shape.");
49 }
50
51 std::vector<const Shape*> parameter_shapes;
52 const auto& program_shape = hlo_proto.hlo_module().program_shape();
53 for (const Shape& shape : program_shape.parameters()) {
54 parameter_shapes.push_back(&shape);
55 }
56 return parameter_shapes;
57}
58
59StatusOr<const Shape*> EntryComputationOutputShape(const HloProto& hlo_proto) {
60 if (!hlo_proto.has_hlo_module()) {
61 return NotFound("HloProto missing HloModuleProto.");
62 }
63 if (!hlo_proto.hlo_module().has_program_shape()) {
64 return NotFound("HloProto missing program shape.");
65 }
66 if (!hlo_proto.hlo_module().program_shape().has_result()) {
67 return NotFound("HloProto missing result in its program shape");
68 }
69
70 return &hlo_proto.hlo_module().program_shape().result();
71}
