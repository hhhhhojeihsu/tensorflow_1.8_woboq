36// Tries to replace a conditional with a call operation of the corresponding
37// computation. If the given conditional has a constant predicate, tries to
38// replace it with a call to its true/false computation as appropirate and then
39// inline that computation.
40//
41// Returns true if it made a change to the graph.
42static StatusOr<bool> TryRemoveConditional(HloInstruction* conditional) {
43 CHECK_EQ(conditional->opcode(), HloOpcode::kConditional);
44 // Do not remove conditionals that contain side-effecting instructions or
45 // have control predecessors/successors in either true/false computation.
46 if (!conditional->parent()->IsRemovable(conditional) ||
47 conditional->HasSideEffect()) {
48 VLOG(2) << "Not attempting to remove conditional as it is not removable or "
49 "has side effect: "
50 << conditional->ToShortString();
51 return false;
52 }
53
54 if (conditional->operand(0)->opcode() != HloOpcode::kConstant) {
55 VLOG(2) << "Not attempting to remove conditional as its predicate is not a "
56 "compile-time constant: "
57 << conditional->ToShortString();
58 return false;
59 }
60
61 auto computation = conditional->parent();
62 HloInstruction* call_op;
63 if (conditional->operand(0)->literal().Get<bool>({})) {
64 call_op = computation->AddInstruction(HloInstruction::CreateCall(
65 conditional->shape(), {conditional->mutable_operand(1)},
66 conditional->true_computation()));
67 } else {
68 call_op = computation->AddInstruction(HloInstruction::CreateCall(
69 conditional->shape(), {conditional->mutable_operand(2)},
70 conditional->false_computation()));
71 }
72
73 TF_RETURN_IF_ERROR(computation->ReplaceInstruction(conditional, call_op));
74 TF_RETURN_IF_ERROR(CallInliner::Inline(call_op).status());
75
76 return true;
77}

79StatusOr<bool> ConditionalSimplifier::Run(HloModule* module) {
80 XLA_VLOG_LINES(
81 3, "ConditionalSimplifier::Run(), before:\n" + module->ToString());
82 bool changed = false;
83
84 // Gather all the conditional ops in our module. We do this ahead of time so
85 // we don't have to worry about mutating the lists of computations or
86 // instructions as we iterate.
87 std::vector<HloInstruction*> conditional_ops;
88 for (auto* comp : module->computations()) {
89 for (auto* instr : comp->instructions()) {
90 if (instr->opcode() == HloOpcode::kConditional) {
91 conditional_ops.push_back(instr);
92 }
93 }
94 }
95
96 for (HloInstruction* conditional_op : conditional_ops) {
97 TF_ASSIGN_OR_RETURN(bool result, TryRemoveConditional(conditional_op));
98 changed |= result;
99 }
100
101 XLA_VLOG_LINES(3,
102 "ConditionalSimplifier::Run(), after:\n" + module->ToString());
103 return changed;
104}
