42Status HloCostAnalysis::Preprocess(const HloInstruction* hlo) {
43 // Set current instruction cost values to reasonable default values. Each
44 // handler can overwrite these values. In Postprocess, these values are
45 // accumulated and written to the per-instruction maps.
46 current_properties_.clear();
47 current_should_compute_bottleneck_time_ = true;
48
49 // The default number of bytes accessed for an instruction is the sum of the
50 // sizes of the inputs and outputs. The default ShapeUtil::ByteSizeOf does not
51 // handle opaque types.
52 float bytes_accessed = shape_size_(hlo->shape());
53 for (const HloInstruction* operand : hlo->operands()) {
54 bytes_accessed += shape_size_(operand->shape());
55 }
56 current_properties_[kBytesAccessedKey] = bytes_accessed;
57
58 return Status::OK();
59}
60
61Status HloCostAnalysis::Postprocess(const HloInstruction* hlo) {
62 if (current_should_compute_bottleneck_time_) {
63 // Compute the time as the time of the bottleneck, i.e. the slowest property
64 // given the per-second rate of each property.
65 float optimal_seconds = 0.0f;
66 for (const auto& property : current_properties_) {
67 if (property.first != kOptimalSecondsKey) {
68 optimal_seconds = std::max(
69 optimal_seconds,
70 property.second /
71 GetProperty(property.first, per_second_rates_, INFINITY));
72 }
73 }
74 current_properties_[kOptimalSecondsKey] = optimal_seconds;
75 }
76
77 TF_RET_CHECK(hlo_properties_.emplace(hlo, current_properties_).second);
78 for (const auto& property : current_properties_) {
79 properties_sum_[property.first] += property.second;
80 }
81
82 return Status::OK();
83}

85Status HloCostAnalysis::HandleElementwiseOp(
86 const HloInstruction* hlo_instruction) {
87 const auto& shape = hlo_instruction->shape();
88 // For element-wise operations, the number of computations is the same as the
89 // number of elements in the output shape.
90 auto computation_count = ShapeUtil::ElementsIn(shape);
91 auto opcode = hlo_instruction->opcode();
92 // We treat transcendental operations separately since one transcendental
93 // operation can correspond to several floating point ops.
94 if (opcode == HloOpcode::kExp || opcode == HloOpcode::kPower ||
95 opcode == HloOpcode::kTanh || opcode == HloOpcode::kSin ||
96 opcode == HloOpcode::kCos) {
97 current_properties_[kTranscendentalsKey] = computation_count;
98 } else {
99 // Note: transcendental operations are considered a separate category from
100 // FLOPs.
101 current_properties_[kFlopsKey] = computation_count;
102 }
103 return Status::OK();
104}

107 const Properties& properties,
108 const float default_value) {
109 auto key_value = properties.find(key);
110 return key_value == properties.end() ? default_value : key_value->second;

202Status HloCostAnalysis::HandleDot(const HloInstruction* dot) {
203 const Shape& lhs_shape = dot->operand(0)->shape();
204 const Shape& rhs_shape = dot->operand(1)->shape();
205 const DotDimensionNumbers& dnums = dot->dot_dimension_numbers();
206 // Count of elements along the reduction dimension (last dimension for the
207 // rhs).
208 int64 reduction_width =
209 lhs_shape.dimensions(dnums.lhs_contracting_dimensions(0));
210 // First divide by reduction width before multiplying by rhs elements to avoid
211 // overflow.
212 int64 fma_count;
213 if (reduction_width == 0) {
214 fma_count = 0;
215 } else {
216 fma_count = (ShapeUtil::ElementsIn(lhs_shape) / reduction_width) *
217 ShapeUtil::ElementsIn(rhs_shape);
218 }
219
220 // We count an FMA operation as 2 floating point operations.
221 current_properties_[kFlopsKey] = kFmaFlops * fma_count;
222 return Status::OK();
223}

237Status HloCostAnalysis::HandleMap(const HloInstruction* map) {
238 // Compute properties of the mapped function.
239 TF_ASSIGN_OR_RETURN(const Properties sub_properties,
240 ProcessSubcomputation(map->to_apply()));
241
242 // Compute the cost of all elements for this Map operation.
243 const int64 element_count = ShapeUtil::ElementsIn(map->shape());
244 for (const auto& property : sub_properties) {
245 if (property.first != kBytesAccessedKey) {
246 current_properties_[property.first] = property.second * element_count;
247 }
248 }
249 return Status::OK();
250}

252Status HloCostAnalysis::HandleReduce(const HloInstruction* reduce) {
253 auto arg = reduce->operand(0);
254 HloComputation* function = reduce->to_apply();
255 // Compute the cost of the user function.
256 TF_ASSIGN_OR_RETURN(const Properties sub_properties,
257 ProcessSubcomputation(function));
258
259 // Compute the cost of all elements for this Reduce operation.
260 int64 reduction_count = ShapeUtil::ElementsIn(arg->shape()) -
261 ShapeUtil::ElementsIn(reduce->shape());
262 for (const auto& property : sub_properties) {
263 if (property.first != kBytesAccessedKey) {
264 current_properties_[property.first] = property.second * reduction_count;
265 }
266 }
267 return Status::OK();
268}

270Status HloCostAnalysis::HandleReduceWindow(
271 const HloInstruction* reduce_window) {
272 const Window& window = reduce_window->window();
273 auto function = reduce_window->to_apply();
274 // Compute the properties of the reduction function.
275 TF_ASSIGN_OR_RETURN(const Properties sub_properties,
276 ProcessSubcomputation(function));
277
278 // Compute the cost of all elements for this ReduceWindow operation. For each
279 // output element there are window_size - 1 reductions to perform.
280 int64 window_element_count = 1;
281 for (const auto& dimension : window.dimensions()) {
282 window_element_count *= dimension.size();
283 }
284 const int64 output_element_count =
285 ShapeUtil::ElementsIn(reduce_window->shape());
286 const int64 reduction_count =
287 (window_element_count - 1) * output_element_count;
288 for (const auto& property : sub_properties) {
289 if (property.first != kBytesAccessedKey) {
290 current_properties_[property.first] = property.second * reduction_count;
291 }
292 }
293 return Status::OK();
294}

296Status HloCostAnalysis::HandleSelectAndScatter(
297 const HloInstruction* instruction) {
298 // Compute the properties of the select and scatter function.
299 // Compute the properties of the reduction function.
300 TF_ASSIGN_OR_RETURN(const Properties select_properties,
301 ProcessSubcomputation(instruction->select()));
302 TF_ASSIGN_OR_RETURN(const Properties scatter_properties,
303 ProcessSubcomputation(instruction->scatter()));
304
305 // Compute the cost of all elements for this operation. For each scatter
306 // source element there are window_size - 1 select computations to perform and
307 // 1 scatter computation to perform.
308 const auto source = instruction->operand(1);
309 const auto source_element_count = ShapeUtil::ElementsIn(source->shape());
310 int64 window_element_count = 1;
311 for (const auto& dimension : instruction->window().dimensions()) {
312 window_element_count *= dimension.size();
313 }
314 const int64 select_count = source_element_count * (window_element_count - 1);
315 for (const auto& property : select_properties) {
316 if (property.first != kBytesAccessedKey) {
317 current_properties_[property.first] += property.second * select_count;
318 }
319 }
320 for (const auto& property : scatter_properties) {
321 if (property.first != kBytesAccessedKey) {
322 current_properties_[property.first] +=
323 property.second * source_element_count;
324 }
325 }
326 return Status::OK();
327}

387Status HloCostAnalysis::HandleConvolution(const HloInstruction* convolution) {
388 auto lhs = convolution->operand(0);
389 auto rhs = convolution->operand(1);
390 Window window = convolution->window();
391 const auto& result_shape = convolution->shape();
392 const Shape& lhs_shape = lhs->shape();
393 const Shape& rhs_shape = rhs->shape();
394
395 const auto& dnums = convolution->convolution_dimension_numbers();
396
397 const int64 input_batch_dim = dnums.input_batch_dimension();
398 const int64 input_feature_dim = dnums.input_feature_dimension();
399 const int64 output_feature_dim = dnums.output_feature_dimension();
400 const int64 input_feature =
401 ShapeUtil::GetDimension(lhs_shape, input_feature_dim);
402 const int64 output_feature =
403 ShapeUtil::GetDimension(result_shape, output_feature_dim);
404 const int64 batch = ShapeUtil::GetDimension(lhs_shape, input_batch_dim);
405
406 DimensionVector kernel_limits;
407 DimensionVector output_limits;
408 DimensionVector input_limits;
409 if (window.dimensions().empty()) {
410 window = window_util::MakeWindow({1});
411 kernel_limits.push_back(1);
412 output_limits.push_back(1);
413 input_limits.push_back(1);
414 } else {
415 for (int64 spatial_dimension = 0;
416 spatial_dimension < window.dimensions_size(); ++spatial_dimension) {
417 // Spatial dimension number for kernel (rhs).
418 const int64 kernel_spatial_dim =
419 dnums.kernel_spatial_dimensions(spatial_dimension);
420 const int64 kernel_limit = rhs_shape.dimensions(kernel_spatial_dim);
421 kernel_limits.push_back(kernel_limit);
422
423 // Spatial dimension number for output.
424 const int64 output_spatial_dim =
425 dnums.output_spatial_dimensions(spatial_dimension);
426 const int64 output_limit = result_shape.dimensions(output_spatial_dim);
427 output_limits.push_back(output_limit);
428
429 // Spatial dimension number for input (lhs).
430 const int64 input_spatial_dim =
431 dnums.input_spatial_dimensions(spatial_dimension);
432 const int64 input_limit = lhs_shape.dimensions(input_spatial_dim);
433 input_limits.push_back(input_limit);
434 }
435 }
436
437 DimensionVector valid_position_counts;
438
439 // Loop over each spatial dimension.
440 for (int64 spatial_dimension = 0;
441 spatial_dimension < window.dimensions_size(); ++spatial_dimension) {
442 int64 valid_position_count = 0;
443 // Loop over each point in the kernel.
444 for (int64 kernel_idx = 0; kernel_idx < kernel_limits[spatial_dimension];
445 ++kernel_idx) {
446 // Loop over each point in the output.
447 for (int64 output_idx = 0; output_idx < output_limits[spatial_dimension];
448 ++output_idx) {
449 // Calculate lhs (input) index without taking base dilation into
450 // account.
451 const auto& window_dim = window.dimensions(spatial_dimension);
452 const int64 undilated_index = output_idx * window_dim.stride() -
453 window_dim.padding_low() +
454 kernel_idx * window_dim.window_dilation();
455
456 // Calculate the actual lhs (input) index after dilation. Avoid the
457 // division as an optimization.
458 const int64 lhs_spatial_index =
459 window_dim.base_dilation() > 1
460 ? undilated_index / window_dim.base_dilation()
461 : undilated_index;
462
463 // Skip if the lhs (input) index is to be dilated.
464 if (undilated_index != lhs_spatial_index * window_dim.base_dilation()) {
465 continue;
466 }
467
468 // Skip if input index is not in bound.
469 if (lhs_spatial_index < 0 ||
470 lhs_spatial_index >= input_limits[spatial_dimension]) {
471 continue;
472 }
473
474 valid_position_count += 1;
475 }
476 }
477 valid_position_counts.push_back(valid_position_count);
478 }
479
480 const int64 fma_count =
481 input_feature * output_feature * batch * Product(valid_position_counts);
482 current_properties_[kFlopsKey] = fma_count * kFmaFlops;
483 return Status::OK();
484}

486Status HloCostAnalysis::HandleFft(const HloInstruction* fft) {
487 auto real_shape =
488 ShapeUtil::IsTuple(fft->operand(0)->shape())
489 ? ShapeUtil::GetTupleElementShape(fft->operand(0)->shape(), 0)
490 : fft->operand(0)->shape();
491 constexpr int kFmaPerComplexMul = 4;
492 int64 log_factors = 1;
493 for (int64 dim : fft->fft_length()) {
494 log_factors *= tensorflow::Log2Floor(dim);
495 }
496 current_properties_[kFlopsKey] = kFmaFlops * kFmaPerComplexMul * log_factors *
497 ShapeUtil::ElementsIn(real_shape);
498 return Status::OK();
499}

501Status HloCostAnalysis::HandleCrossReplicaSum(const HloInstruction* crs) {
502 // We assume 2 replicas, so that each output element is the sum of two input
503 // elements.
504 //
505 // TODO(b/33004697): Compute correct cost here, taking the actual number of
506 // replicas into account.
507 double flops = 0.0;
508 ShapeUtil::ForEachSubshape(
509 crs->shape(), [&, this](const Shape& subshape, const ShapeIndex&) {
510 if (ShapeUtil::IsArray(subshape)) {
511 flops += ShapeUtil::ElementsIn(subshape);
512 }
513 });
514 current_properties_[kFlopsKey] = flops;
515 return Status::OK();

527Status HloCostAnalysis::HandleFusion(const HloInstruction* fusion) {
528 // Compute the properties of the fused expression and attribute them to the
529 // fusion node. Use a dummy shape_size to avoid any errors from trying to
530 // calculate the size of a shape that does not have a layout, since nodes
531 // inside fusion nodes do not necessarily have a layout assigned.
532 ShapeSizeFunction shape_size = [](const Shape& shape) { return 0; };
533 TF_ASSIGN_OR_RETURN(
534 current_properties_,
535 ProcessSubcomputation(fusion->fused_instructions_computation(),
536 &shape_size));
537
538 // Fusion nodes that produce a tuple also produce the entries in the tuple.
539 // Ignore the memory accessed inside fused ops, since fusion is supposed to
540 // prevent intermediate data from touching slow memory.
541 current_properties_[kBytesAccessedKey] = 0;
542 ShapeUtil::ForEachSubshape(
543 fusion->shape(),
544 [this](const Shape& subshape, const ShapeIndex& /*shape_index*/) {
545 current_properties_[kBytesAccessedKey] += shape_size_(subshape);
546 });
547
548 for (const HloInstruction* operand : fusion->operands()) {
549 current_properties_[kBytesAccessedKey] += shape_size_(operand->shape());
550 }
551
552 return Status::OK();
553}

572Status HloCostAnalysis::HandleSort(const HloInstruction* sort) {
573 // This assumes a comparison based N*log(N) algorithm. As for all ops, the
574 // actual properties of the op depend on the backend implementation.

576 current_properties_[kFlopsKey] = elements * tensorflow::Log2Ceiling(elements);
577 return Status::OK();

580Status HloCostAnalysis::HandleWhile(const HloInstruction* xla_while) {
581 // Since the number of iterations of the while node will not always be
582 // something that we can statically analyze, we cannot precisely compute the
583 // cost of a while node. For now compute the cost of a single iteration.
584 //
585 // TODO(b/26346211): Improve the cost analysis for while nodes.
586 TF_ASSIGN_OR_RETURN(const Properties body_properties,
587 ProcessSubcomputation(xla_while->while_body()));
588
589 TF_ASSIGN_OR_RETURN(const Properties condition_properties,
590 ProcessSubcomputation(xla_while->while_condition()));
591
592 current_properties_.clear();
593 for (const auto& property : body_properties) {
594 current_properties_[property.first] += property.second;
595 }
596 for (const auto& property : condition_properties) {
597 current_properties_[property.first] += property.second;
598 }
599 current_should_compute_bottleneck_time_ = false;
600
601 return Status::OK();
602}

604Status HloCostAnalysis::HandleConditional(const HloInstruction* conditional) {
605 // Compute the cost of the true and false computations and take the maximum
606 // from those for each property.
607 TF_ASSIGN_OR_RETURN(const Properties true_computation_properties,
608 ProcessSubcomputation(conditional->true_computation()));
609 TF_ASSIGN_OR_RETURN(const Properties false_computation_properties,
610 ProcessSubcomputation(conditional->false_computation()));
611 current_properties_ = true_computation_properties;
612 for (const auto& property : false_computation_properties) {
613 if (!tensorflow::gtl::InsertIfNotPresent(&current_properties_, property)) {
614 current_properties_[property.first] =
615 std::max(current_properties_[property.first], property.second);
616 }
617 }
618 current_should_compute_bottleneck_time_ = false;
619
620 return Status::OK();
621}

664StatusOr<HloCostAnalysis::Properties> HloCostAnalysis::ProcessSubcomputation(
665 HloComputation* computation, const ShapeSizeFunction* shape_size) {
666 if (shape_size == nullptr) {
667 shape_size = &shape_size_;
668 }
669 HloCostAnalysis visitor(*shape_size, per_second_rates_);
670 TF_RETURN_IF_ERROR(computation->Accept(&visitor));
671 return visitor.properties();
672}
