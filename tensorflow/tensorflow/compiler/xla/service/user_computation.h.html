39// A UserComputation is the built-up computation that users create via the
40// XLA Service interface.
41//
42// The XLA service adds instructions to a user computation via this
43// interface. The state of the computation is stored as a SessionComputation
44// proto which holds a record of all operation-building requests received by the
45// XLA service.
46//
47// UserComputations are lowered to HloComputations which are passed to the high
48// level compiler interface.
49class UserComputation {
50 public:
51 // Factory used when restoring a computation from serialized session
52 // computation (computation snapshot) data. Remaps any references to
53 // computation handle via the old_to_new mapping.
54 //
55 // An error will occur if the old_to_new mapping cannot resolve a reference to
56 // a computation that is present in session_computation.
57 static StatusOr<std::unique_ptr<UserComputation>> MakeWithRemapping(
58 const SessionComputation& session_computation,
59 const ComputationHandle& handle,
60 const std::map<int64, ComputationHandle>& old_to_new);
61

63 explicit UserComputation(const string& name, const ComputationHandle& handle);

271 // Computes and returns the program shape for the user computation -- gathers
272 // parameters and result type into a single proto. A shared_ptr is used
273 // because the returned pointer refers to an internally cached value which may
274 // be discarded by the UserComputation object. This avoid unnecessary copies.
275 //
276 // If the parameter space is not dense (i.e. there are holes in the parameter
277 // numbers provided) then an error status is returned.
278 StatusOr<std::shared_ptr<const ProgramShape>> ComputeProgramShape(
279 VersionedComputationHandle::Version version) const;

281 // Returns true if the given data handle does not depend on any parameter with
282 // index higher then num_parameters. That is, the value can be computed at
283 // compile time if we know the first num_parameters arguments.
284 StatusOr<bool> IsConstant(const ComputationDataHandle& handle,
285 int64 num_parameters);

287 // Returns the output shape of the operation indicated by the given handle.

290 // Sets metadata on the Hlo instruction referenced by the given handle.

293
294 // Sets the device assignment on the Hlo instruction referenced by 'handle'.
295 Status SetOpSharding(const ComputationDataHandle& handle,
296 const OpSharding& sharding);

298 // Builds a HLO computation from the UserComputation. The parameter "resolver"
299 // is a function which returns a pointer to the HloComputation corresponding
300 // to the given ComputationHandle at the given version. The resolver is used
301 // for operations, such as map, which call other computations and need a
302 // pointer to the called HloComputation to construct the respective HLO
303 // instructions. If include_unreachable_instructions is true, then
304 // instructions which are not reachable from the root are lowered into
305 // HloInstructions.
306 using HloComputationResolver =
307 std::function<HloComputation*(const VersionedComputationHandle& handle)>;
308 StatusOr<std::unique_ptr<HloComputation>> BuildHloComputation(
309 VersionedComputationHandle::Version version,
310 HloComputationResolver hlo_resolver, const DebugOptions& debug_options,
311 bool include_unreachable_instructions = true) const;
312
313 // Return a vector containing the embedded computations used by this
314 // UserComputation. Only embedded computations which are called directly by
315 // this UserComputation are included. That is, the transitive closure of
316 // embedded computations is not included.
317 std::vector<VersionedComputationHandle> GetEmbeddedComputations(
318 VersionedComputationHandle::Version version) const;
319
320 // Returns the number of OperationRequest objects in this UserComputation.
321 // The 'version' of a computation is identical to the number of
322 // OperationRequests in the UserComputation.
323 int64 request_count(VersionedComputationHandle::Version version) const {
324 return version;
325 }
326
327 // Returns a copy of the internal session state for this computation -- this
328 // is useful for serializing the guts of a user computation, though references
329 // to other handles (e.g. referred-to computations) must be handled with care
330 // in the serialization / de-serialization process.
331 SessionComputation CloneSessionComputation(
332 VersionedComputationHandle::Version version) const;
333
334 // Warning: typically we don't want to look up computation data handles until
335 // the computation is finished being built, for consistency purposes. We
336 // expose this routine for error reporting purposes so that we can provide
337 // more meaningful error messages from the XLA service layer.
338 //
339 // Returns the operation request that the handle comes from.
340 StatusOr<const OperationRequest*> LookUpRequestForErrorReporting(
341 const ComputationDataHandle& handle) const;
342
343 // Retrieves the parameter metadata for the given parameter number.
344 //
345 // If the parameter number is invalid for this computation, nullopt is
346 // returned. When the return value has_value(), nullptr will never be
347 // the held value.
348 tensorflow::gtl::optional<const OpMetadata*> ParameterMetadata(
349 int parameter_number) const;
350
351 private:
352 // Warning: dangerous mutating operation that doesn't respect versioning.
353 // This is only used at initialization time when constructing from a
354 // SessionComputation a la MakeWithRemapping.
355 //
356 // Remaps references to old computations (with handle values in the keys of
357 // old_to_new) to the computation handle given in the values. This is useful
358 // when loading computations from snapshots, to finish initialization, before
359 // the user computation is released into the wild.
360 Status RemapEmbeddedComputations(
361 const std::map<int64, ComputationHandle>& old_to_new)
362 EXCLUSIVE_LOCKS_REQUIRED(mutex_);
363
364 // Returns the OperationRequest corresponding to the given handle.
365 StatusOr<const OperationRequest*> LookUpRequest(
366 const ComputationDataHandle& handle) const
367 EXCLUSIVE_LOCKS_REQUIRED(mutex_);
368
369 // Creates a new ComputationDataHandle with the next available handle value.
370 ComputationDataHandle CreateComputationDataHandle()
371 EXCLUSIVE_LOCKS_REQUIRED(mutex_);
372
373 // Checks whether the parameter numbers of the parameter operations are
374 // contiguous starting from zero. Returns appropriate error status if not.
375 Status CheckParametersAreContiguous(
376 VersionedComputationHandle::Version version) const
377 EXCLUSIVE_LOCKS_REQUIRED(mutex_);
378
379 VersionedComputationHandle GetVersionedHandleInternal() const
380 EXCLUSIVE_LOCKS_REQUIRED(mutex_);
381
382 // Name of the computation.
383 string name_;
384
385 mutable tensorflow::mutex mutex_;
386
387 // State of the computation as a record of all operation-building requests.
388 SessionComputation session_computation_ GUARDED_BY(mutex_);
389
390 // Mapping from parameter number to operation request containing the
391 // respective ParameterRequest.
392 std::map<int64, OperationRequest*> parameters_ GUARDED_BY(mutex_);
393
394 // The next ComputationDataHandle value to assign. Handle values are assigned
395 // sequentially.
396 int64 next_handle_value_ GUARDED_BY(mutex_);
397
398 // If handle_to_return_.has_handle() then an Execution of this Computation
399 // will compute the value represented by handle_to_return_, otherwise it will
400 // compute the value of (next_handle_value_ - 1).
401 ComputationDataHandle handle_to_return_ GUARDED_BY(mutex_);
402
403 // Memoized ProgramShape and its version. A shared_ptr is used because
404 // references to this object are returned by ComputeProgramShape.
405 mutable int64 program_shape_version_ GUARDED_BY(mutex_) = 0;
406 mutable std::shared_ptr<const ProgramShape> program_shape_ GUARDED_BY(mutex_);
407
408 TF_DISALLOW_COPY_AND_ASSIGN(UserComputation);
409};
