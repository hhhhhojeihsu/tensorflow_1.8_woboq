47Status GenericTransferManager::WriteSingleTupleIndexTable(
48 perftools::gputools::StreamExecutor* executor,
49 tensorflow::gtl::ArraySlice<se::DeviceMemoryBase> elements,
50 const Shape& shape, perftools::gputools::DeviceMemoryBase* region) {
51 TF_RET_CHECK(elements.size() == ShapeUtil::TupleElementCount(shape));
52
53 std::vector<const void*> element_pointers;
54 for (const se::DeviceMemoryBase& element : elements) {
55 element_pointers.push_back(element.opaque());
56 }
57 return TransferBufferToDevice(executor, GetByteSizeRequirement(shape),
58 element_pointers.data(), region);
59}

61StatusOr<std::unique_ptr<Literal>>
62GenericTransferManager::TransferLiteralFromDevice(
63 se::StreamExecutor* executor, const ShapedBuffer& device_buffer) {
64 VLOG(2) << "transferring literal from device ordinal "
65 << executor->device_ordinal() << "; device buffer: " << device_buffer;
66 TF_RET_CHECK(executor->device_ordinal() == device_buffer.device_ordinal());
67
68 // The on-host and on-device shape should always be the same for the generic
69 // transfer manager.
70 TF_RET_CHECK(ShapeUtil::Equal(device_buffer.on_device_shape(),
71 device_buffer.on_host_shape()));
72
73 std::unique_ptr<Literal> literal =
74 Literal::CreateFromShape(device_buffer.on_host_shape());
75
76 TF_RETURN_IF_ERROR(ShapeUtil::ForEachSubshapeWithStatus(
77 device_buffer.on_host_shape(),
78 [&](const Shape& subshape, const ShapeIndex& index) -> Status {
79 if (!ShapeUtil::IsTuple(subshape)) {
80 TF_RETURN_IF_ERROR(TransferBufferFromDevice(
81 executor,
82 /*source=*/device_buffer.buffer(index),
83 /*size=*/GetByteSizeRequirement(subshape),
84 /*destination=*/
85 literal->untyped_data(index)));
86 }
87
88 return Status::OK();
89 }));
90 return std::move(literal);
91}

93Status GenericTransferManager::TransferLiteralToDevice(
94 se::StreamExecutor* executor, const Literal& literal,
95 const ShapedBuffer& device_buffer) {
96 const Shape& shape = literal.shape();
97 VLOG(2) << "transferring literal shape to device: "
98 << ShapeUtil::HumanString(shape)
99 << "; device buffer: " << device_buffer;
100
101 // The on-host and on-device shape should always be the same for the generic
102 // transfer manager.
103 TF_RET_CHECK(ShapeUtil::Equal(device_buffer.on_device_shape(),
104 device_buffer.on_host_shape()));
105
106 TF_RET_CHECK(
107 ShapeUtil::Compatible(literal.shape(), device_buffer.on_host_shape()));
108 TF_RET_CHECK(executor->device_ordinal() == device_buffer.device_ordinal());
109
110 TF_RETURN_IF_ERROR(WriteTupleIndexTables(executor, device_buffer));
111
112 return ShapeUtil::ForEachSubshapeWithStatus(
113 device_buffer.on_host_shape(),
114 [&](const Shape& device_subshape, const ShapeIndex& index) -> Status {
115 se::DeviceMemoryBase device_memory = device_buffer.buffer(index);
116 if (ShapeUtil::IsArray(device_subshape)) {
117 TF_RET_CHECK(GetByteSizeRequirement(device_subshape) ==
118 device_memory.size());
119 // Element is array-shaped: transfer array data to device buffer.
120 const auto subliteral = LiteralView::Create(literal, index);
121 std::unique_ptr<Literal> relayed_out_literal;
122 const void* source;
123 if (LayoutUtil::Equal(device_subshape.layout(),
124 subliteral.shape().layout())) {
125 source = subliteral.untyped_data();
126 } else {
127 // Relayout data before transferring.
128 relayed_out_literal = subliteral.Relayout(device_subshape.layout(),
129 /*shape_index=*/{});
130 source = relayed_out_literal->untyped_data();
131 }
132 return TransferBufferToDevice(
133 executor,
134 /*size=*/GetByteSizeRequirement(device_subshape), source,
135 &device_memory);
136 }
137 return Status::OK();
138 });
139}
