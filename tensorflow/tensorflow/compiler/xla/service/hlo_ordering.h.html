34// Base class for describing a partial ordering of HLO instructions. Used to
35// determine live range overlap of HLO instruction output buffers.
36class HloOrdering {
37 public:
38 HloOrdering(const HloModule* module)
39 : module_(module), call_graph_(CallGraph::Build(module)) {}
40 virtual ~HloOrdering() = default;
41
42 // Returns true if instruction 'a' executes before instruction 'b'. This is
43 // not reflexive, that is, an instruction does not execute before itself.
44 bool ExecutesBefore(const HloInstruction* a, const HloInstruction* b) const;
45
46 // Returns whether the value 'a' is defined before the value 'b' under the
47 // given ordering.
48 bool IsDefinedBefore(const HloValue& a, const HloValue& b) const;
49
50 // Returns whether the given use is before the given value definition under
51 // the given ordering.
52 bool UseIsBeforeValueDefinition(const HloUse& use, const HloValue& value,
53 const HloDataflowAnalysis& dataflow) const;
54 // Returns whether the given values interfere. Two values interfere if they
55 // may both be simultaneously live.
56 bool MayInterfere(const HloValue& a, const HloValue& b,
57 const HloDataflowAnalysis& dataflow) const;
58
59 // Returns true if the live range of the given value 'a' is strictly before
60 // the live range of value 'b' using the given HLO ordering.
61 bool LiveRangeStrictlyBefore(const HloValue& a, const HloValue& b,
62 const HloDataflowAnalysis& dataflow) const;
63
64 // Returns the sequential instruction order for the given computation, or
65 // nullptr if the computation does not have a sequential ordering.
66 virtual const std::vector<const HloInstruction*>* SequentialOrder(
67 const HloComputation& computation) const = 0;
68
69 // Return the call graph of the module used to compute ordering.
70 const CallGraph& call_graph() const { return *call_graph_; }
71
72 virtual string ToString() const = 0;
73
74 // Returns the serialized representation of this ordering.
75 // Only sequential computation orders are represented.
76 HloOrderingProto ToProto() const;
77
78 protected:
79 // Returns true if instruction 'a' executes before instruction 'b'.
80 // Precondition: 'a' and 'b' are in the same computation.
81 //
82 // Derived classes should implement this method for determining order of
83 // instructions in the same computation. ExecutesBefore() analyzes the
84 // callgraph and uses this method to determine ordering of instructions in
85 // different computations.
86 virtual bool ExecutesBeforeInSameComputation(
87 const HloInstruction* a, const HloInstruction* b) const = 0;
88
89 const HloModule* module_;
90
91 std::unique_ptr<CallGraph> call_graph_;
92};
93
94// Base class for partial orderings implemented by a map of predecessors for
95// each instruction. Subclasses should fill in predecessors_.
96class PredecessorHloOrdering : public HloOrdering {
97 public:
98 ~PredecessorHloOrdering() override = default;
99
100 // Returns nullptr indicating the computation does not have a sequential
101 // ordering.
102 const std::vector<const HloInstruction*>* SequentialOrder(
103 const HloComputation& computation) const override {
104 return nullptr;
105 }
106
107 HloReachabilityMap& reachability_map(const HloComputation* computation) {
108 return *predecessors_.at(computation);
109 }
110 const HloReachabilityMap& reachability_map(
111 const HloComputation* computation) const {
112 return *predecessors_.at(computation);
113 }
114
115 protected:
116 explicit PredecessorHloOrdering(const HloModule* module);
117 string ToStringHelper(const string& name) const;
118
119 bool ExecutesBeforeInSameComputation(const HloInstruction* a,
120 const HloInstruction* b) const override;
121
122 // For each computation in the module, this is the set of the instruction's
123 // predecessors. An instruction is an element of its own predecessor set.
124 //
125 // Subclasses should fill this in to define the desired ordering.
126 tensorflow::gtl::FlatMap<const HloComputation*,
127 std::unique_ptr<HloReachabilityMap>>
128 predecessors_;
129};
130
131// An HLO ordering based on data dependencies in the HLO graph. In this partial
132// order, instruction A executes before instruction B only if there is a path
133// from A to B in the HLO graph. For example, given the following graph:
134/*
135 param
136 / \
137 negate exp
138 \ /
139 add
140*/
141// DependencyHloOrdering gives the following executes-before relations:
142// param executes before negate, exp, and add
143// negate executes before add
144// exp executes before add
145// add executes before nothing
146// negate and exp are not ordered because the dependencies allow either to
147// execute before the other (or in parallel). DependencyHloOrdering ordering
148// allows maximum parallelism and enables any execution order which satisfies
149// data dependencies. This requires pessimistic assumptions about buffer live
150// ranges and can result in more memory used than more constrained orderings.
151class DependencyHloOrdering : public PredecessorHloOrdering {
152 public:
153 explicit DependencyHloOrdering(const HloModule* module);
154 ~DependencyHloOrdering() override = default;
155
156 string ToString() const override;
157};
158
159// An HLO ordering based on a total order of instructions in each computation.
160// The computation total order is a sequencing of all of its instructions in
161// the computation (eg, {inst0, inst1, inst2,...}) as in single-threaded
162// execution. For example, given the following HLO graph:
163/*
164 param
165 / \
166 negate exp
167 \ /
168 add
169*/
170// and the following sequence:
171//
172// {param, negate, exp, add}
173//
174// SequentialHloOrdering gives the following executes-before relations:
175// param executes before negate, exp, and add
176// negate executes before exp and add
177// exp executes before add
178// add executes before nothing
179// This is more constrained than DependencyHloOrdering in this example because
180// negate and exp are ordered (negate before exp). This enables param to share
181// the same buffer as exp (param buffer is dead after exp). Generally, this
182// ordering enables more buffer sharing (reduced memory usage) because buffer
183// interference is reduced relative to DependencyHloOrdering.
184class SequentialHloOrdering : public HloOrdering {
185 public:
186 // A sequence of instructions for each computation in the module.
187 using HloModuleSequence =
188 tensorflow::gtl::FlatMap<const HloComputation*,
189 std::vector<const HloInstruction*>>;
190
191 SequentialHloOrdering(const HloModule* module,
192 const HloModuleSequence& module_sequence);
193 ~SequentialHloOrdering() override = default;
194
195 // Returns the sequential instruction order for the given computation.
196 const std::vector<const HloInstruction*>* SequentialOrder(
197 const HloComputation& computation) const override;
198
199 string ToString() const override;
200
201 protected:
202 bool ExecutesBeforeInSameComputation(const HloInstruction* a,
203 const HloInstruction* b) const override;
204
205 const HloModuleSequence module_sequence_;
206
207 // The position of every instruction in the HLO module in its respective
208 // computation sequence (a value of zero indicates the instruction is first in
209 // the sequence, etc). Instructions from all computations are contained in
210 // this map so more than one instruction may have the same position
211 // value. This is not a problem because ExecutesBefore also verifies
212 // instructions are in the same computation.
213 tensorflow::gtl::FlatMap<const HloInstruction*, int> order_position_;
214};
215
216std::ostream& operator<<(
217 std::ostream& out,
218 const SequentialHloOrdering::HloModuleSequence& module_sequence);
