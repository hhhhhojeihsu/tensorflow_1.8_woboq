47std::unique_ptr<HloComputation> HloComputation::Builder::Build(
48 HloInstruction* root_instruction) {
49 int parameter_count = 0;
50 for (auto& instruction : instructions_) {
51 if (instruction->opcode() == HloOpcode::kParameter) {
52 parameter_count++;
53 }
54 }
55 // If root_instruction is not specified use the last added instruction.
56 HloInstruction* root =
57 root_instruction ? root_instruction : last_added_instruction_;
58 CHECK_NE(nullptr, root);
59 return WrapUnique(new HloComputation(name_, parameter_count, &instructions_,
60 root, fusion_instruction_));
61}

63HloComputation::HloComputation(
64 const string& name, int parameter_count,
65 std::vector<std::unique_ptr<HloInstruction>>* instructions,
66 HloInstruction* root_instruction, HloInstruction* fusion_instruction)
67 : name_(name),
68 unique_id_(-1),
69 root_instruction_(root_instruction),
70 fusion_instruction_(fusion_instruction) {
71 param_instructions_.resize(parameter_count, nullptr);
72 bool root_found = false;
73 for (auto& instruction : *instructions) {
74 if (instruction->opcode() == HloOpcode::kParameter) {
75 int64 param_no = instruction->parameter_number();
76 CHECK(param_no >= 0 && param_no < parameter_count)
77 << "\nERROR: invalid parameter number. Expected [0, "
78 << parameter_count << "), got " << param_no;
79 CHECK(param_instructions_[param_no] == nullptr)
80 << "\nERROR: parameter number " << param_no
81 << " already allocated in this computation";
82 param_instructions_[param_no] = instruction.get();
83 }
84 root_found |= instruction.get() == root_instruction_;
85 AddInstructionInternal(std::move(instruction));
86 }
87 CHECK(root_found)
88 << "\nERROR: root instruction is not present in computation.";
89}

91HloInstruction* HloComputation::AddInstruction(
92 std::unique_ptr<HloInstruction> instruction) {
93 CHECK(instruction->opcode() != HloOpcode::kParameter)
94 << "Parameter instructions cannot be added to a computation after "
95 << "it has been built";
96 return AddInstructionInternal(std::move(instruction));
97}

99HloInstruction* HloComputation::AddInstructionInternal(
100 std::unique_ptr<HloInstruction> instruction) {
101 if (parent() != nullptr) {
102 instruction->UniquifyName(&parent()->instruction_name_uniquer());
103 instruction->SetUniqueId(parent()->NewUniqueInstructionId());
104 }
105 instruction->set_parent(this);
106 HloInstruction* pinst = instruction.get();
107 instruction_iterators_[pinst] =
108 instructions_.insert(instructions_.end(), std::move(instruction));
109 return pinst;
110}

112HloInstruction* HloComputation::AddParameter(
113 std::unique_ptr<HloInstruction> instruction) {
114 CHECK(instruction->opcode() == HloOpcode::kParameter);
115 CHECK(IsFusionComputation());
116 CHECK(fusion_instruction_->operand_count() == param_instructions_.size());
117 instruction->set_parent(this);
118 param_instructions_.push_back(instruction.get());
119 AddInstructionInternal(std::move(instruction));
120 return instructions_.back().get();
121}

123Status HloComputation::RemoveParameter(int64 param_no) {
124 CHECK_GE(param_no, 0);
125 CHECK_LT(param_no, param_instructions_.size());
126 CHECK(IsFusionComputation());
127 HloInstruction* param_instruction = param_instructions_[param_no];
128 auto param_instruction_iterator = param_instructions_.begin() + param_no;
129 param_instructions_.erase(param_instruction_iterator);
130 // Throw removed fused parameter instruction away.
131 TF_RETURN_IF_ERROR(RemoveInstruction(param_instruction));
132
133 while (param_no < param_instructions_.size()) {
134 param_instruction = param_instructions_[param_no];
135 string param_name = param_instruction->name();
136 // Fusion parameters are named foo.param_1, bar.param_2, etc. We are
137 // renumbering the parameters, so replace the final number in the name with
138 // the updated value.
139 const string param_underscore = ".param_";
140 size_t index = param_name.rfind(param_underscore);
141 if (index == string::npos) {
142 string after_param = name().substr(index + param_underscore.size());
143 int64 numeric_suffix;
144 if (tensorflow::strings::safe_strto64(after_param, &numeric_suffix)) {
145 param_name =
146 StrCat(param_name.substr(0, index), param_underscore, param_no);
147 }
148 }
149
150 HloInstruction* new_instr =
151 AddInstructionInternal(HloInstruction::CreateParameter(
152 param_no, param_instruction->shape(), param_name));
153 TF_RETURN_IF_ERROR(param_instruction->ReplaceAllUsesWith(new_instr));
154 param_instructions_[param_no] = new_instr;
155 TF_RETURN_IF_ERROR(RemoveInstruction(param_instruction));
156 param_no++;
157 }
158
159 return Status::OK();
160}

162bool HloComputation::IsRemovable(const HloInstruction* instruction) {
163 // If the instruction has control predecessors or successors then we cannot
164 // remove the instruction without violating ordering constraints (added, for
165 // example, to avert interference due to buffer aliasing).
166 if (!instruction->control_predecessors().empty() ||
167 !instruction->control_successors().empty()) {
168 return false;
169 }
170
171 if (instruction->opcode() == HloOpcode::kParameter &&
172 !IsFusionComputation()) {
173 return false;
174 }
175
176 return true;
177}

179bool HloComputation::HasSideEffect() const {
180 for (auto* instruction : instructions()) {
181 if (instruction->HasSideEffect()) {
182 return true;
183 }
184 }
185 return false;
186}

188Status HloComputation::RemoveInstructionAndUnusedOperands(
189 HloInstruction* instruction) {
190 TF_RET_CHECK(root_instruction() != instruction);
191
192 TF_RET_CHECK(instruction->user_count() == 0);
193 TF_RET_CHECK(IsRemovable(instruction))
194 << "Cannot remove instruction: " << instruction->ToString();
195 std::unordered_set<HloInstruction*> removed;
196 std::queue<HloInstruction*> worklist;
197 worklist.push(instruction);
198 while (!worklist.empty()) {
199 HloInstruction* item = worklist.front();
200 worklist.pop();
201
202 if (removed.count(item) != 0 || item->user_count() != 0 ||
203 item == root_instruction() || !IsRemovable(item) ||
204 item->HasSideEffect()) {
205 continue;
206 }
207 for (int i = 0; i < item->operand_count(); ++i) {
208 worklist.push(item->mutable_operand(i));
209 }
210
211 TF_RETURN_IF_ERROR(RemoveInstruction(item));
212 removed.insert(item);
213 }
214 return Status::OK();
215}

217Status HloComputation::RemoveInstruction(HloInstruction* instruction) {
218 VLOG(2) << "Removing instruction " << instruction->name()
219 << " from computation " << name();
220 TF_RET_CHECK(IsRemovable(instruction))
221 << "cannot remove instruction: " << instruction->ToString();
222 TF_RET_CHECK(root_instruction() != instruction)
223 << "cannot remove root instruction " << instruction->name();
224 TF_RET_CHECK(instruction->user_count() == 0)
225 << "instruction " << instruction->name()
226 << " has users and cannot be removed";
227 TF_RET_CHECK(instruction->control_predecessors().empty())
228 << "instruction " << instruction->name()
229 << " has control predecessors and cannot be removed";
230 TF_RET_CHECK(instruction->control_successors().empty())
231 << "instruction " << instruction->name()
232 << " has control successors and cannot be removed";
233
234 TF_RET_CHECK(instruction_iterators_.count(instruction) != 0);
235 auto inst_it = instruction_iterators_.at(instruction);
236 (*inst_it)->set_parent(nullptr);
237 instruction->DetachFromOperands();
238 instructions_.erase(inst_it);
239 return Status::OK();
240}

242void HloComputation::set_root_instruction(
243 HloInstruction* new_root_instruction) {
244 // The shape of the root (ignoring layout) is an invariant of the computation
245 // for non-fusion cases.
246 if (!IsFusionComputation()) {
247 CHECK(ShapeUtil::Compatible(new_root_instruction->shape(),
248 root_instruction_->shape()))
249 << new_root_instruction->shape().ShortDebugString()
250 << " is incompatible with "
251 << root_instruction_->shape().ShortDebugString();
252 }
253 bool root_found = false;
254 for (auto& instruction : instructions_) {
255 if (new_root_instruction == instruction.get()) {
256 root_found = true;
257 break;
258 }
259 }
260 DCHECK(root_found);
261
262 root_instruction_ = new_root_instruction;
263}

267// Helper class which computes the post order of an expression rooted at a
268// particular instruction.
269class InstructionPostOrderer : public DfsHloVisitorWithDefault {
270 public:
271 // added_instructions is the set of instructions which have already been
272 // accounted for in the post order in previous invocations of
273 // GetOrder. Without this mechanism, instructions which are predecessors of
274 // multiple root instructions of the computation can be added to the post
275 // order more than once.
276 static std::list<HloInstruction*> GetOrder(
277 HloInstruction* root,
278 tensorflow::gtl::FlatSet<HloInstruction*>* added_instructions) {
279 InstructionPostOrderer orderer(added_instructions);
280 TF_CHECK_OK(root->Accept(&orderer));
281 return std::move(orderer.post_order_);
282 }
283
284 private:
285 explicit InstructionPostOrderer(
286 tensorflow::gtl::FlatSet<HloInstruction*>* added_instructions)
287 : added_instructions_(added_instructions) {}
288 ~InstructionPostOrderer() override {}
289
290 Status DefaultAction(HloInstruction* hlo_instruction) override {
291 if (added_instructions_->count(hlo_instruction) == 0) {
292 post_order_.push_back(hlo_instruction);
293 added_instructions_->insert(hlo_instruction);
294 }
295 return Status::OK();
296 }
297
298 std::list<HloInstruction*> post_order_;
299 tensorflow::gtl::FlatSet<HloInstruction*>* added_instructions_;
300};
301
302// Helper which builds a post order of the HLO call graph.
303void ComputeComputationPostOrder(
304 HloComputation* computation,
305 tensorflow::gtl::FlatSet<HloComputation*>* visited,
306 std::list<HloComputation*>* post_order) {
307 if (visited->insert(computation).second) {
308 for (auto* instruction : computation->instructions()) {
309 for (HloComputation* called_computation :
310 instruction->called_computations()) {
311 ComputeComputationPostOrder(called_computation, visited, post_order);
312 }
313 }
314 post_order->push_back(computation);
315 }
316}

318} // namespace

320std::list<HloInstruction*> HloComputation::MakeInstructionPostOrder() const {
321 std::list<HloInstruction*> post_order;
322 std::list<HloInstruction*> trace_instructions;
323 tensorflow::gtl::FlatSet<HloInstruction*> added_instructions;
324 for (auto& instruction : instructions_) {
325 if (instruction->opcode() == HloOpcode::kTrace) {
326 // Trace instructions aren't handled by the DFS visitor. Add trace
327 // instructions to the post order at the end (necessarily they have no
328 // users).
329 trace_instructions.push_back(instruction.get());
330 } else if (instruction->users().empty()) {
331 post_order.splice(post_order.end(),
332 InstructionPostOrderer::GetOrder(instruction.get(),
333 &added_instructions));
334 }
335 }
336 post_order.splice(post_order.end(), trace_instructions);
337 CHECK_EQ(instructions_.size(), post_order.size())
338 << "number of instructions does not match post order size";
339 return post_order;
340}
341
342std::list<HloComputation*> HloComputation::MakeEmbeddedComputationsList()
343 const {
344 tensorflow::gtl::FlatSet<HloComputation*> visited;
345 std::list<HloComputation*> post_order;
346
347 // To avoid special handling of this computation, cast away const of
348 // 'this'. 'this' is immediately removed from the post order after
349 // construction.
350 ComputeComputationPostOrder(const_cast<HloComputation*>(this), &visited,
351 &post_order);
352
353 // We don't want to include this computation in the post order.
354 CHECK_EQ(this, post_order.back());
355 post_order.pop_back();
356
357 return post_order;
358}

360string HloComputation::ToString(const HloPrintOptions& options) const {
361 std::ostringstream s;
362 for (int i = 0; i < options.indent_amount(); i++) {
363 s << " ";
364 }
365 if (options.print_percent()) {
366 s << "%";
367 }
368 s << name();
369 if (options.print_program_shape()) {
370 s << " " << ShapeUtil::HumanString(ComputeProgramShape());
371 }
372 s << " {\n";
373 for (const HloInstruction* instruction : MakeInstructionPostOrder()) {
374 for (int i = 0; i < options.indent_amount(); i++) {
375 s << " ";
376 }
377 s << " " << (instruction == root_instruction_ ? "ROOT " : "")
378 << instruction->ToString(options) << "\n";
379 }
380 for (int i = 0; i < options.indent_amount(); i++) {
381 s << " ";
382 }
383 s << "}";
384 return s.str();
385}

387HloComputationProto HloComputation::ToProto() const {
388 HloComputationProto proto;
389 CHECK(unique_id_ != -1)
390 << "This computation does not have a valid id. Please make sure the "
391 "computation is inside a module before dumping it.";
392 proto.set_id(unique_id_);
393 proto.set_name(name_);
394 for (const HloInstruction* instruction : MakeInstructionPostOrder()) {
395 HloInstructionProto instruction_proto = instruction->ToProto();
396 proto.add_instructions()->Swap(&instruction_proto);
397 }
398 proto.set_root_id(root_instruction()->unique_id());
399 *proto.mutable_program_shape() = ComputeProgramShape();
400 return proto;
401}

403/* static */ StatusOr<std::unique_ptr<HloComputation>>
404HloComputation::CreateFromProto(
405 HloModule* module, const HloComputationProto& proto,
406 const tensorflow::gtl::FlatMap<int64, HloComputation*>& computation_map) {
407 std::vector<std::unique_ptr<HloInstruction>> instructions;
408 tensorflow::gtl::FlatMap<int64, HloInstruction*> instruction_map;
409 int64 parameter_count = 0;
410 for (const HloInstructionProto& instruction_proto : proto.instructions()) {
411 TF_ASSIGN_OR_RETURN(
412 std::unique_ptr<HloInstruction> instruction,
413 HloInstruction::CreateFromProto(module, instruction_proto,
414 instruction_map, computation_map));
415 if (instruction->opcode() == HloOpcode::kParameter) {
416 parameter_count++;
417 }
418 TF_RET_CHECK(!ContainsKey(instruction_map, instruction_proto.id()));
419 instruction_map[instruction_proto.id()] = instruction.get();
420 instructions.push_back(std::move(instruction));
421 }
422
423 TF_RET_CHECK(proto.root_id() != -1);
424 TF_RET_CHECK(ContainsKey(instruction_map, proto.root_id()));
425 HloInstruction* root = instruction_map.at(proto.root_id());
426 return WrapUnique(new HloComputation(proto.name(), parameter_count,
427 &instructions, root,
428 /*fusion_instruction=*/nullptr));
429}

431void HloComputation::FuseInstructionsInto(
432 tensorflow::gtl::ArraySlice<HloInstruction*> instructions_to_fuse,
433 HloInstruction* fusion_instruction) {
434 CHECK_EQ(HloOpcode::kFusion, fusion_instruction->opcode());
435 HloInstruction* root = instructions_to_fuse.front();
436 TF_CHECK_OK(root->ReplaceAllUsesWith(fusion_instruction));
437 if (root == root_instruction()) {
438 set_root_instruction(fusion_instruction);
439 }
440 TF_CHECK_OK(RemoveInstruction(root));
441 for (size_t i = 1; i < instructions_to_fuse.size(); ++i) {
442 HloInstruction* instruction = instructions_to_fuse[i];
443 fusion_instruction->FuseInstruction(instruction);
444 if (instruction->user_count() == 0) {
445 TF_CHECK_OK(RemoveInstruction(instruction));
446 }
447 }
448}
449
450HloInstruction* HloComputation::CreateFusionInstruction(
451 tensorflow::gtl::ArraySlice<HloInstruction*> instructions_to_fuse,
452 HloInstruction::FusionKind fusion_kind) {
453 HloInstruction* root = instructions_to_fuse.front();
454 HloInstruction* fusion_instruction = AddInstruction(
455 HloInstruction::CreateFusion(root->shape(), fusion_kind, root));
456 FuseInstructionsInto(instructions_to_fuse, fusion_instruction);
457 return fusion_instruction;
458}

460StatusOr<HloInstruction*> HloComputation::DeepCopyHelper(
461 HloInstruction* instruction, const ShapeTree<bool>* indices_to_copy,
462 ShapeTree<HloInstruction*>* copies_added, ShapeIndex* index) {
463 if (ShapeUtil::IsArray(instruction->shape())) {
464 if (indices_to_copy == nullptr || indices_to_copy->element(*index)) {
465 // Use kCopy to copy array elements
466 HloInstruction* copy = AddInstruction(HloInstruction::CreateUnary(
467 instruction->shape(), HloOpcode::kCopy, instruction));
468 if (copies_added != nullptr) {
469 *copies_added->mutable_element(*index) = copy;
470 }
471 return copy;
472 } else {
473 // Array elements which are not to be copied are passed through
474 // transparently.
475 return instruction;
476 }
477 } else if (ShapeUtil::IsTuple(instruction->shape())) {
478 std::vector<HloInstruction*> elements;
479 for (int64 i = 0; i < ShapeUtil::TupleElementCount(instruction->shape());
480 i++) {
481 HloInstruction* gte =
482 AddInstruction(HloInstruction::CreateGetTupleElement(
483 ShapeUtil::GetTupleElementShape(instruction->shape(), i),
484 instruction, i));
485
486 index->push_back(i);
487 TF_ASSIGN_OR_RETURN(
488 HloInstruction * element,
489 DeepCopyHelper(gte, indices_to_copy, copies_added, index));
490 elements.push_back(element);
491 index->pop_back();
492 }
493 return AddInstruction(HloInstruction::CreateTuple(elements));
494 } else {
495 return FailedPrecondition(
496 "Can only copy array and tuple shaped instructions");
497 }
498}
499
500StatusOr<HloInstruction*> HloComputation::DeepCopyInstruction(
501 HloInstruction* instruction, const ShapeTree<bool>* indices_to_copy,
502 ShapeTree<HloInstruction*>* copies_added) {
503 if (instruction->parent() != this) {
504 return FailedPrecondition(
505 "Can't deep copy instruction %s: instruction is not in computation %s",
506 instruction->name().c_str(), name().c_str());
507 }
508 if (indices_to_copy != nullptr &&
509 !ShapeUtil::Compatible(instruction->shape(), indices_to_copy->shape())) {
510 return FailedPrecondition(
511 "Can't deep copy instruction %s: given shape tree of indices to copy "
512 "has incompatible shapes: %s vs. %s",
513 instruction->name().c_str(),
514 ShapeUtil::HumanString(instruction->shape()).c_str(),
515 ShapeUtil::HumanString(indices_to_copy->shape()).c_str());
516 }
517
518 ShapeIndex index;
519 return DeepCopyHelper(instruction, indices_to_copy, copies_added, &index);
520}

522ProgramShape HloComputation::ComputeProgramShape() const {
523 ProgramShape program_shape;
524
525 for (auto* param_instruction : param_instructions_) {
526 *program_shape.add_parameters() = param_instruction->shape();
527 *program_shape.add_parameter_names() = param_instruction->name();
528 }
529 *program_shape.mutable_result() = root_instruction_->shape();
530
531 return program_shape;
532}

534bool HloComputation::operator==(const HloComputation& other) const {
535 std::set<std::pair<const HloInstruction*, const HloInstruction*>> visited;
536 std::function<bool(const HloInstruction*, const HloInstruction*)> eq =
537 [&visited, &eq](const HloInstruction* a, const HloInstruction* b) {
538 // If <a,b> are visited but not identical, the recursion should have
539 // been aborted. So, if <a,b> are visited at this point, they must be
540 // identical.
541 if (visited.count(std::make_pair(a, b)) > 0) {
542 return true;
543 }
544 visited.emplace(a, b);
545 return a->Identical(
546 *b, eq, [](const HloComputation* a, const HloComputation* b) {
547 return *a == *b;
548 });
549 };
550 return eq(root_instruction(), other.root_instruction());
551}

553Status HloComputation::ReplaceWithNewInstruction(
554 HloInstruction* old_instruction,
555 std::unique_ptr<HloInstruction> new_instruction) {
556 return ReplaceInstruction(old_instruction,
557 AddInstruction(std::move(new_instruction)));
558}
559
560Status HloComputation::ReplaceInstruction(HloInstruction* old_instruction,
561 HloInstruction* new_instruction) {
562 TF_RET_CHECK(
563 ShapeUtil::Compatible(old_instruction->shape(), new_instruction->shape()))
564 << ShapeUtil::HumanString(old_instruction->shape()) << " vs "
565 << ShapeUtil::HumanString(new_instruction->shape());
566
567 VLOG(10) << "transformed " << old_instruction->ToString() << " to "
568 << new_instruction->ToString();
569 // Try to add metadata for HLO instructions that are created to replace
570 // existing HLO instructions (e.g. during optimizations). The assumption is
571 // that the old instruction and the new instruction would perform the same
572 // function, and that they would be correlated to the same TF op. This might
573 // not always be correct since HLO optimizations can cross TF op boundaries.
574 // But still this seems to be better than nothing.
575 if (new_instruction->metadata().op_name().empty()) {
576 new_instruction->set_metadata(old_instruction->metadata());
577 }
578 TF_RETURN_IF_ERROR(old_instruction->ReplaceAllUsesWith(new_instruction));
579 return RemoveInstructionAndUnusedOperands(old_instruction);
580}

582std::unique_ptr<HloReachabilityMap> HloComputation::ComputeReachability()
583 const {
584 const std::list<HloInstruction*> all = MakeInstructionPostOrder();
585 auto result = MakeUnique<HloReachabilityMap>(all);
586
587 std::vector<HloInstruction*> inputs;
588 for (const HloInstruction* hlo : all) {
589 inputs.assign(hlo->operands().begin(), hlo->operands().end());
590 inputs.insert(inputs.end(), hlo->control_predecessors().begin(),
591 hlo->control_predecessors().end());
592 result->SetReachabilityToUnion(inputs, hlo);
593 }
594 return result;
595}
596
597void HloComputation::UpdateReachabilityThroughInstruction(
598 const HloInstruction* instruction, HloReachabilityMap* reachability_map) {
599 std::queue<const HloInstruction*> worklist;
600 worklist.push(instruction);
601
602 std::vector<HloInstruction*> inputs;
603
604 while (!worklist.empty()) {
605 const HloInstruction* item = worklist.front();
606 worklist.pop();
607
608 inputs.assign(item->operands().begin(), item->operands().end());
609 inputs.insert(inputs.end(), item->control_predecessors().begin(),
610 item->control_predecessors().end());
611
612 if (reachability_map->SetReachabilityToUnion(inputs, item)) {
613 // Add immediate successors to worklist.
614 for (const HloInstruction* user : item->users()) {
615 worklist.push(user);
616 }
617 for (const HloInstruction* succ : item->control_successors()) {
618 worklist.push(succ);
619 }
620 }
621 }
622}

624std::vector<HloInstruction*> HloComputation::CollectUnreachableRoots() const {
625 std::vector<HloInstruction*> unreachable_roots;
626 for (auto* instruction : instructions()) {
627 if (instruction->user_count() == 0 &&
628 instruction->control_successors().empty() &&
629 instruction != root_instruction()) {
630 unreachable_roots.push_back(instruction);
631 }
632 }
633 VLOG(3) << "Unreachable roots:"
634 << tensorflow::str_util::Join(
635 unreachable_roots, "\n\t",
636 [](string* out, const HloInstruction* hlo) {
637 tensorflow::strings::StrAppend(out, hlo->ToString());
638 });
639 return unreachable_roots;
640}

642template <typename HloInstructionPtr>
643Status HloComputation::Accept(
644 DfsHloVisitorBase<HloInstructionPtr>* visitor) const {
645 // Visit unreachable roots. Beware that the visitor might delete the currently
646 // visited root, which would invalidate iterators if the unreachable roots
647 // weren't computed ahead of time.
648 for (HloInstruction* root : CollectUnreachableRoots()) {
649 VLOG(3) << "Traversing unreachable root: " << root->ToString();
650 // Call FinishVisit only at the end.
651 TF_RETURN_IF_ERROR(root->Accept(visitor, /*call_finish_visit=*/false));
652 }
653 // Visit the computation root instruction last.
654 return root_instruction()->Accept(visitor, /*call_finish_visit=*/true);
655}

657// Explicit instantiations.
658template Status HloComputation::Accept(DfsHloVisitor* visitor) const;
659template Status HloComputation::Accept(ConstDfsHloVisitor* visitor) const;

661Status HloComputation::AcceptWithOperandOrder(
662 DfsHloVisitor* visitor,
663 const HloInstruction::CompareFunction& operand_order) const {
664 // Visit unreachable roots. Beware that the visitor might delete the currently
665 // visited root, which would invalidate iterators if the unreachable roots
666 // weren't computed ahead of time.
667 for (HloInstruction* root : CollectUnreachableRoots()) {
668 TF_RETURN_IF_ERROR(
669 root->AcceptWithOperandOrder(visitor, operand_order,
670 /*call_finish_visit=*/false));
671 }
672 // Visit the computation root instruction last.
673 return root_instruction()->AcceptWithOperandOrder(visitor, operand_order,
674 /*call_finish_visit=*/true);
675}

677template <typename HloInstructionPtr>
678Status HloComputation::AcceptOrdered(
679 DfsHloVisitorBase<HloInstructionPtr>* visitor,
680 const std::vector<const HloInstruction*>& order) const {
681 VLOG(3) << "Accepting visitor with order.";
682 for (HloInstruction* root : CollectUnreachableRoots()) {
683 TF_RET_CHECK(std::find(order.begin(), order.end(), root) != order.end())
684 << root->ToString();
685 }
686 TF_RET_CHECK(order.size() == instruction_count());
687 std::unordered_set<const HloInstruction*> visited;
688 for (const HloInstruction* instruction : order) {
689 VLOG(3) << "Visiting ordered: " << instruction->ToString();
690 TF_RET_CHECK(instruction_iterators_.count(instruction) == 1)
691 << "Instruction " << instruction->name() << " is not in computation "
692 << name();
693 TF_RET_CHECK(visited.count(instruction) == 0)
694 << "Instruction " << instruction->name()
695 << " appears more than once in order";
696 HloInstruction* mutable_instruction =
697 const_cast<HloInstruction*>(instruction);
698 TF_RETURN_IF_ERROR(visitor->Preprocess(mutable_instruction));
699 TF_RETURN_IF_ERROR(mutable_instruction->Visit(visitor));
700 visitor->SetVisited(*mutable_instruction);
701 TF_RETURN_IF_ERROR(visitor->Postprocess(mutable_instruction));
702 visited.insert(instruction);
703 }
704 TF_RETURN_IF_ERROR(visitor->FinishVisit(root_instruction()));
705 return Status::OK();
706}

733
734std::unique_ptr<HloComputation> HloComputation::CloneWithReplacements(
735 std::unordered_map<const HloInstruction*, std::unique_ptr<HloInstruction>>
736 replacements,
737 HloModule* module, const string& suffix) {
738 // Look up instr in the replacements map, and return either the replacement,
739 // or instr, if the replacement isn't present.
740 //
741 // Note: This can return null, indicating that instr should not be present in
742 // the new computation.
743 auto replace = [&](HloInstruction* instr) {
744 auto it = replacements.find(instr);
745 if (it == replacements.end()) {
746 return instr;
747 }
748 return it->second.get();
749 };
750
751 VLOG(1) << "Cloning " << name() << " --> " << suffix << "\n";
752 std::vector<HloInstruction*> postorder;
753 for (HloInstruction* instr : MakeInstructionPostOrder()) {
754 if (HloInstruction* replacement = replace(instr)) {
755 postorder.push_back(replacement);
756 }
757 }
758
759 std::unordered_map<HloInstruction*, HloInstruction*> clone_map;
760 std::vector<std::unique_ptr<HloInstruction>> instructions;
761 std::unique_ptr<HloInstruction> new_instr = nullptr;
762 for (auto instr : postorder) {
763 std::vector<HloInstruction*> new_operands;
764 for (auto operand : instr->operands()) {
765 auto replaced_operand = replace(operand);
766 // If replaced_operand is null, that means 'replacements' asked us not to
767 // include operand in the new computation. But we can't do that, because
768 // operand is used by instr.
769 CHECK_NE(replaced_operand, nullptr)
770 << "replacements map tried to eliminate a used instruction "
771 << operand->ToString() << ", used by " << instr->ToString();
772 new_operands.push_back(FindOrDie(clone_map, replaced_operand));
773 }
774 new_instr =
775 instr->CloneWithNewOperands(instr->shape(), new_operands, module);
776 InsertOrDie(&clone_map, instr, new_instr.get());
777 instructions.push_back(std::move(new_instr));
778 }
779 Builder builder(name() + "." + suffix);
780 for (auto& instr : instructions) {
781 builder.AddInstruction(std::move(instr));
782 }
783 auto result = builder.Build(
784 /*root_instruction=*/FindOrDie(clone_map, replace(root_instruction())));
785
786 // Clone control dependencies.
787 for (auto instr : postorder) {
788 HloInstruction* new_instr = FindOrDie(clone_map, instr);
789 for (auto successor : instr->control_successors()) {
790 auto replaced_successor = replace(successor);
791
792 // successor may not be in clone_map, because it might have been
793 // removed by the replacements map.
794 if (replaced_successor == nullptr) {
795 continue;
796 }
797
798 TF_CHECK_OK(new_instr->AddControlDependencyTo(
799 FindOrDie(clone_map, replaced_successor)));
800 }
801 }
802
803 // We cloned the elements of 'replacements', so they're all going to be
804 // destroyed. HloInstructions need to be detached from their operands before
805 // they're destroyed, otherwise they stick around in the operands' users lists
806 // and cause use-after-frees.
807 for (auto& kv : replacements) {
808 if (std::unique_ptr<HloInstruction>& new_instr = kv.second) {
809 new_instr->DetachFromOperands();
810 }
811 }
812
813 return result;
814}
