35// IrArray represents an XLA array at the LLVM IR level. This class
36// encapsulates a base pointer to the buffer holding the array (as an LLVM
37// Value) and the shape of the array. The class includes methods for emitting
38// LLVM IR sequences which access elements of the array at a multidimensional
39// index (eg, [x, y, z] in a 3-dimensional array). Arbitrary shape and layouts
40// are supported.
41class IrArray {
42 public:
43 // A multidimensional index into an IrArray. The index for dimension zero is
44 // first in the vector. This is the reverse order of the notation used for
45 // describing the dimensions of an array. That is, for a [4 x 3 x 2] array
46 // dimension zero has size 2, dimension one has size 3, and dimension two has
47 // size 4. Thus the index {1, 2, 3} indexes the last element of this [4 x 3 x
48 // 2] array.
49 //
50 // This may also keep a linear index and the layout and dimensions it was
51 // emitted for; if the shape where this `Index` is used matches, the linear
52 // index may be used, potentially sparing the cost of computing the
53 // multidimensional index, which LLVM DCE can delete.
54 class Index {
55 public:
56 // Constructs an empty zero-dimensional index.
57 Index() {}
58
59 // Constructs an index of rank "size". Each dimension of the index is
60 // initialized to "value".
61 explicit Index(size_t size, llvm::Value* value = nullptr)
62 : multidim_(size, value) {}
63
64 // Constructs an index from multi-dimensional index "multidim". The linear
65 // index is set to nullptr.
66 explicit Index(tensorflow::gtl::ArraySlice<llvm::Value*> multidim)
67 : multidim_(multidim.begin(), multidim.end()) {}
68
69 // Constructs an index from linear index "linear" and computes the
70 // multi-dimensional index from "linear" and "shape". "ir_builder" is the IR
71 // builder to emit the index of each dimension in the multi-dimensional
72 // index.
73 //
74 // Precondition: "shape" has a layout.
75 Index(llvm::Value* linear, const Shape& shape,
76 llvm::IRBuilder<>* ir_builder);
77
78 // Constructs an index from the given multi-dimensional index and the shape
79 // that it indexes into.
80 //
81 // Precondition: "shape" has a layout.
82 Index(tensorflow::gtl::ArraySlice<llvm::Value*> multidim,
83 const Shape& shape, llvm::IRBuilder<>* ir_builder);
84
85 // Constructs an index from both a multi-dimensional index and a linear
86 // index. "shape" has the same meaning as that in the constructor that takes
87 // only a linear index.
88 Index(tensorflow::gtl::ArraySlice<llvm::Value*> multidim,
89 llvm::Value* linear, const Shape& shape);
90
91 const std::vector<llvm::Value*>& multidim() const { return multidim_; }
92 llvm::Value* linear() const { return linear_; }
93
94 size_t size() const { return multidim().size(); }
95
96 llvm::Value* operator[](size_t i) const { return multidim()[i]; }
97 llvm::Value*& operator[](size_t i) { return multidim()[i]; }
98
99 void push_back(llvm::Value* value) { multidim().push_back(value); }
100
101 using iterator = std::vector<llvm::Value*>::iterator;
102 using const_iterator = std::vector<llvm::Value*>::const_iterator;
103
104 iterator begin() { return multidim().begin(); }
105 iterator end() { return multidim().end(); }
106
107 const_iterator begin() const { return multidim().begin(); }
108 const_iterator end() const { return multidim().end(); }
109
110 llvm::Value* back() const { return multidim().back(); }
111
112 bool LinearValidOnShape(const Shape& a) const;
113
114 // Given that "this" is the target index of a reshape from `operand_shape`
115 // to `shape`, returns the source index.
116 Index SourceIndexOfReshape(const Shape& shape, const Shape& operand_shape,
117 llvm::IRBuilder<>* builder) const;
118
119 // Returns the index into the source operand from which a slice operation
120 // selects a value to be placed into index "this". The slice is described
121 // by starting indices `starts` and stride values `strides`.
122 //
123 // Precondition: "this" is an index into a slice whose shape is `shape`.
124 Index SourceIndexOfSlice(const Shape& shape,
125 tensorflow::gtl::ArraySlice<int64> starts,
126 tensorflow::gtl::ArraySlice<int64> strides,
127 llvm::IRBuilder<>* builder) const;
128
129 // Given that "this" is the target index of a transpose from `operand_shape`
130 // to `shape` with the given dimension mapping, returns the source index.
131 Index SourceIndexOfTranspose(
132 const Shape& shape, const Shape& operand_shape,
133 tensorflow::gtl::ArraySlice<int64> dimension_mapping,
134 llvm::IRBuilder<>* builder) const;
135
136 // Given that "this" is the target index of a bitcast from `operand_shape`
137 // to `shape`, returns the source index.
138 Index SourceIndexOfBitcast(const Shape& shape, const Shape& operand_shape,
139 llvm::IRBuilder<>* builder) const;
140
141 // Given that "this" is the target index of a broadcast from `operand_shape`
142 // to `shape` with the given dimension mapping, returns the source index.
143 Index SourceIndexOfBroadcast(
144 const Shape& shape, const Shape& operand_shape,
145 tensorflow::gtl::ArraySlice<int64> dimension_mapping,
146 llvm::IRBuilder<>* builder) const;
147
148 // Linearizes the index into the given shape, i.e. reshapes it to rank-1 and
149 // returns the index into the sole dimension 0 of the new shape.
150 llvm::Value* Linearize(tensorflow::gtl::ArraySlice<int64> dimensions,
151 llvm::IRBuilder<>* builder) const;
152
153 private:
154 // Changing the multi-dimensional index invalidates the linear index.
155 std::vector<llvm::Value*>& multidim() {
156 linear_ = nullptr;
157 return multidim_;
158 }
159
160 std::vector<llvm::Value*> multidim_;
161
162 // These values are purely for efficiency; `multidim_` is enough to find the
163 // element at a given `Index`, but if a loop is emitted with a linear index
164 // space, that linear index can be saved in `linear_`, and the layout and
165 // dimensions of the shape the loop was emitted for in `layout_` and
166 // `dims_`, and if the `Index` is used in another array, and its layout and
167 // dimensions match, the linear index can be used, sparing the cost of
168 // computing `multidim_`, which LLVM DCE could potentially so delete.
169 // Modifying `multidim_` after construction nullifies `linear_`, lest it
170 // be used wrongly, as it would be valid no more.
171 // If a loop is emitted with a multidimensional index space, `linear_` would
172 // be null and `layout_` and `dims_` would be ignored.
173 llvm::Value* linear_ = nullptr;
174 Layout layout_;
175 std::vector<int64> dims_;
176 };
177
178 // Default constructor. Constructs an IrArray in a null status.
179 IrArray() : base_ptr_(nullptr), shape_(nullptr) {}
180
181 // Construct an IrArray with the given base pointer and shape. base_ptr is a
182 // pointer type pointing to the first element(lowest address) of the array.
183 IrArray(llvm::Value* base_ptr, const Shape& shape);
184
185 // Default implementations of copying and moving.
186 IrArray(IrArray&& other) = default;
187 IrArray(const IrArray& other) = default;
188 IrArray& operator=(IrArray&& other) = default;
189 IrArray& operator=(const IrArray& other) = default;
190
191 llvm::Value* GetBasePointer() const { return base_ptr_; }
192 llvm::Type* GetElementLlvmType() const { return element_type_; }
193
194 const Shape& GetShape() const {
195 CHECK(shape_ != nullptr);
196 return *shape_;
197 }

243 // Promises LLVM that the data pointed to by this IrArray never changes after
244 // it's first loaded.
245 //
246 // The temporal scope of this promise is the "whole program" from LLVM's point
247 // of view, but how this translates to HLOs differs between backends.
248 //
249 // In the single-threaded CPU backend, we emit one function that
250 // runs all the HLOs in sequence, so the whole program is the whole HLO
251 // module.
252 //
253 // In the GPU backend, we emit one GPU kernel per top-level HLO (i.e. per HLO
254 // in the entry computation). From LLVM's perspective, launching a new kernel
255 // is like launching a new program, and so the whole program is one top-level
256 // HLO. Since the scope of the promise is smaller than in the CPU backend, we
257 // can mark more things as invariant in the GPU backend.
258 //
259 // Marking loads as invariant is particularly helpful on GPUs because
260 // invariant loads can be lowered to PTX ld.global.nc (equivalent to CUDA's
261 // __ldg intrinsic). These loads use a special cache, and can be
262 // significantly faster than regular loads.
263 void MarkInvariantOverWholeProgram(llvm::LLVMContext* context) {
264 if (is_invariant_) {
265 return;
266 }
267 is_invariant_ = true;
268 AddMetadata(llvm::LLVMContext::MD_invariant_load,
269 llvm::MDNode::get(*context, {}));
270 }
271
272 const std::map<int, llvm::MDNode*>& metadata() const { return metadata_; }
273
274 // Bumps the "which_dimension" value within the provided index by the provided
275 // addend.
276 static Index BumpIndex(const Index& index, int64 which_dimension,
277 int64 addend, llvm::IRBuilder<>* ir_builder);
278
279 private:
280 // Add the specified LLVM IR metadata to loads/stores associated with this
281 // IrArray.
282 void AddMetadata(int kind, llvm::MDNode* md) {
283 InsertOrDie(&metadata_, kind, md);
284 }
285
286 // Address of the base of the array as an LLVM Value.
287 llvm::Value* base_ptr_;
288
289 // The LLVM type of the elements in the array.
290 llvm::Type* element_type_;
291
292 // Shape of the XLA array.
293 const Shape* shape_;
294
295 // The list of key/value pairs used when attaching metadata to emitted
296 // loads/stores for this array. They keys are the metadata kinds and the
297 // values are the metadata nodes.
298 std::map<int, llvm::MDNode*> metadata_;
299
300 bool is_invariant_ = false;
301};
