16#ifndef TENSORFLOW_COMPILER_XLA_SERVICE_INTERFACE_H_
17#define TENSORFLOW_COMPILER_XLA_SERVICE_INTERFACE_H_
18
19#include "tensorflow/compiler/xla/xla.pb.h"
20#include "tensorflow/core/lib/core/status.h"
21
22namespace xla {
23
24// Defines the interface for an XLA service on the client side. This service
25// helps abstract around the actual implementation of a service - the service
26// can be local (running in the same process), or remote - in which case an RPC
27// stub is used as the implementation.
28class ServiceInterface {
29 public:
30 ServiceInterface() {}
31 virtual ~ServiceInterface() = default;
32
33 // TODO(b/31824348): Convert to use StatusOr.
34 virtual tensorflow::Status TransferToClient(
35 const TransferToClientRequest* arg, TransferToClientResponse* result) = 0;
36
37 virtual tensorflow::Status TransferToServer(
38 const TransferToServerRequest* arg, TransferToServerResponse* result) = 0;
39
40 virtual tensorflow::Status TransferToInfeed(
41 const TransferToInfeedRequest* arg, TransferToInfeedResponse* result) = 0;
42
43 virtual tensorflow::Status TransferFromOutfeed(
44 const TransferFromOutfeedRequest* arg,
45 TransferFromOutfeedResponse* result) = 0;
46
47 virtual tensorflow::Status ResetDevice(const ResetDeviceRequest* arg,
48 ResetDeviceResponse* result) = 0;
49
50 virtual tensorflow::Status LoadComputationSnapshot(
51 const LoadComputationSnapshotRequest* request,
52 LoadComputationSnapshotResponse* result) = 0;
53
54 virtual tensorflow::Status Execute(const ExecuteRequest* arg,
55 ExecuteResponse* result) = 0;
56
57 virtual tensorflow::Status ExecuteGraph(const ExecuteGraphRequest* arg,
58 ExecuteResponse* result) = 0;
59
60 virtual tensorflow::Status ExecuteParallel(
61 const ExecuteParallelRequest* arg, ExecuteParallelResponse* result) = 0;
62
63 virtual tensorflow::Status ExecuteGraphParallel(
64 const ExecuteGraphParallelRequest* arg,
65 ExecuteParallelResponse* result) = 0;
66
67 virtual tensorflow::Status ExecuteAsync(const ExecuteAsyncRequest* arg,
68 ExecuteAsyncResponse* result) = 0;
69
70 virtual tensorflow::Status WaitForExecution(
71 const WaitForExecutionRequest* arg, WaitForExecutionResponse* result) = 0;
72
73 virtual tensorflow::Status DeconstructTuple(
74 const DeconstructTupleRequest* arg, DeconstructTupleResponse* result) = 0;
75
76 virtual tensorflow::Status GetComputationStats(
77 const ComputationStatsRequest* arg, ComputationStatsResponse* result) = 0;
78
79 virtual tensorflow::Status GetComputationGraphStats(
80 const ComputationGraphStatsRequest* arg,
81 ComputationStatsResponse* result) = 0;
82
83 virtual tensorflow::Status GetComputationShape(
84 const GetComputationShapeRequest* arg,
85 GetComputationShapeResponse* result) = 0;
86
87 virtual tensorflow::Status GetShape(const GetShapeRequest* arg,
88 GetShapeResponse* result) = 0;
89
90 virtual tensorflow::Status CreateChannelHandle(
91 const CreateChannelHandleRequest* arg,
92 CreateChannelHandleResponse* result) = 0;
93
94 virtual tensorflow::Status GetDeviceHandles(
95 const GetDeviceHandlesRequest* arg, GetDeviceHandlesResponse* result) = 0;
96
97 // Methods used by ComputationBuilder.
98 virtual tensorflow::Status Computation(const ComputationRequest* arg,
99 ComputationResponse* result) = 0;
100
101 virtual tensorflow::Status Op(const OpRequest* arg, OpResponse* result) = 0;
102
103 virtual tensorflow::Status GetLocalShape(const GetLocalShapeRequest* arg,
104 GetLocalShapeResponse* result) = 0;
105
106 virtual tensorflow::Status SetReturnValue(
107 const SetReturnValueRequest* arg, SetReturnValueResponse* results) = 0;
108
109 virtual tensorflow::Status IsConstant(const IsConstantRequest* arg,
110 IsConstantResponse* result) = 0;
111
112 virtual tensorflow::Status ComputeConstant(
113 const ComputeConstantRequest* arg, ComputeConstantResponse* result) = 0;
114
115 // Methods used by Computation.
116 virtual tensorflow::Status SnapshotComputation(
117 const SnapshotComputationRequest* ag,
118 SnapshotComputationResponse* result) = 0;
119
120 // Methods used by GlobalData.
121 virtual tensorflow::Status Unregister(const UnregisterRequest* arg,
122 UnregisterResponse* result) = 0;
123};
124
125} // namespace xla
126
127#endif // TENSORFLOW_COMPILER_XLA_SERVICE_INTERFACE_H_
