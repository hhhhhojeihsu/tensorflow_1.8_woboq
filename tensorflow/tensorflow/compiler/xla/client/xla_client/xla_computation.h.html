27// The computation graph that the user builds up with the XlaBuilder.
28//
29// TODO(b/74197823): Replace xla::Computation with this one.
30class XlaComputation {
31 public:
32 XlaComputation() : unique_id_(-1) {}
33
34 XlaComputation(const XlaComputation&) = delete;
35 XlaComputation& operator=(const XlaComputation&) = delete;
36
37 XlaComputation(XlaComputation&& from) = default;
38
39 XlaComputation& operator=(XlaComputation&& from) = default;
40
41 // Returns the "program shape" (parameter and return shapes) for this
42 // computation.
43 StatusOr<ProgramShape> GetProgramShape() const;
44
45 const HloModuleProto& proto() const { return proto_; }
46
47 private:
48 XlaComputation(const int64 unique_id) : unique_id_(unique_id) {}
49 HloModuleProto* mutable_proto() { return &proto_; }
50 friend class XlaBuilder;
51
52 int64 unique_id_;
53 HloModuleProto proto_;
54};
