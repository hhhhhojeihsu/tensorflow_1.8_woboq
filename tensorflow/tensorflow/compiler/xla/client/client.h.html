35namespace xla {

37// XLA service's client object -- wraps the service with convenience and
38// lifetime-oriented methods.
39class Client {
40 public:
41 explicit Client(ServiceInterface* stub);
42 virtual ~Client();
43
44 // Executes the computation with the given arguments and returns the global
45 // data that was produced from the execution.
46 // * If execution_options is not nullptr, these options are passed to the
47 // service to affect how it compiles our computation. (The pointer does not
48 // need to live beyond this call.)
49 // * If execution_options.device_handles is not empty, the computation is
50 // executed on the devices associated with the handles by partitioning the
51 // computation based on the attached sharding attributes. Otherwise, a
52 // device is chosen by the service.
53 // * If execution_profile is not nullptr then the pointed-to ExecutionProfile
54 // will be filled with profile data from the execution.
55 StatusOr<std::unique_ptr<GlobalData>> Execute(
56 const Computation& computation,
57 tensorflow::gtl::ArraySlice<GlobalData*> arguments,
58 const ExecutionOptions* execution_options = nullptr,
59 ExecutionProfile* execution_profile = nullptr);
60
61 // Executes the computation with the given arguments and returns the global
62 // data that was produced from the execution.
63 // * If execution_options is not nullptr, these options are passed to the
64 // service to affect how it compiles our computation. (The pointer does not
65 // need to live beyond this call.)
66 // * If execution_profile is not nullptr then the pointed-to ExecutionProfile
67 // will be filled with profile data from the execution.
68 //
69 // TODO(b/74197823): This is a part of a NOT YET ready refactor.
70 StatusOr<std::unique_ptr<GlobalData>> Execute(
71 const XlaComputation& computation,
72 tensorflow::gtl::ArraySlice<GlobalData*> arguments,
73 const ExecutionOptions* execution_options = nullptr,
74 ExecutionProfile* execution_profile = nullptr);
75
76 // A struct to represent a computation instance to be executed.
77 // * If execution_options.device_handles is not empty, the computation is
78 // executed on the devices associated with the handles by partitioning the
79 // computation based on the attached sharding attributes. Otherwise, a
80 // device is chosen by the service.
81 struct ComputationInstance {
82 const Computation& computation;
83 std::vector<GlobalData*> arguments;
84 ExecutionOptions execution_options;
85 ExecutionProfile* execution_profile;
86
87 ComputationInstance(const Computation& computation,
88 std::vector<GlobalData*> arguments,
89 ExecutionOptions execution_options,
90 ExecutionProfile* execution_profile)
91 : computation(computation),
92 arguments(std::move(arguments)),
93 execution_options(execution_options),
94 execution_profile(execution_profile) {}
95 };
96
97 // Executes a list ComputationInstances and returns global data produced from
98 // each computation.
99 StatusOr<std::vector<std::unique_ptr<GlobalData>>> ExecuteParallel(
100 tensorflow::gtl::ArraySlice<ComputationInstance> computations);
101
102 // A struct to represent a computation instance to be executed.
103 // * If execution_options.device_handles is not empty, the computation is
104 // executed on the devices associated with the handles by partitioning the
105 // computation based on the attached sharding attributes. Otherwise, a
106 // device is chosen by the service.
107 //
108 // TODO(b/74197823): This is a part of a NOT YET ready refactor.
109 struct XlaComputationInstance {
110 const XlaComputation& computation;
111 std::vector<GlobalData*> arguments;
112 ExecutionOptions execution_options;
113 ExecutionProfile* execution_profile;
114
115 XlaComputationInstance(const XlaComputation& computation,
116 std::vector<GlobalData*> arguments,
117 ExecutionOptions execution_options,
118 ExecutionProfile* execution_profile)
119 : computation(computation),
120 arguments(std::move(arguments)),
121 execution_options(execution_options),
122 execution_profile(execution_profile) {}
123 };
124
125 // Executes a list XlaComputationInstances and returns global data produced
126 // from each computation.
127 //
128 // TODO(b/74197823): This is a part of a NOT YET ready refactor.
129 StatusOr<std::vector<std::unique_ptr<GlobalData>>> ExecuteParallel(
130 tensorflow::gtl::ArraySlice<XlaComputationInstance> computations);
131
132 // Requests device_count device handles available on the target. The returned
133 // device handles are used to specify the devices to execute the computations
134 // (see ExecuteParallel) or to transfer data (see TransferToServer or
135 // TransferToInfeed).
136 StatusOr<std::vector<DeviceHandle>> GetDeviceHandles(int64 device_count);
137
138 // Transfer the global data provided to this client process, which is
139 // returned in the provided literal. Use sparingly to avoid transfer
140 // overheads.
141 //
142 // If shape_with_layout is not nullptr, it points to a shape whose layout will
143 // be the layout of the returned literal.
144 StatusOr<std::unique_ptr<Literal>> Transfer(
145 const GlobalData& data, const Shape* shape_with_layout = nullptr);
146
147 // Transfer the given literal to the server. This allocates memory on the
148 // device and copies the literal's contents over. Returns a global data handle
149 // that can be used to refer to this value from the client.
150 //
151 // If device_handle is not nullptr, data is transferred to the associated
152 // device (and its replicas if replication is enabled). Otherwise, data is
153 // transferred to the default device (and its replicas).
154 StatusOr<std::unique_ptr<GlobalData>> TransferToServer(
155 const Literal& literal, const DeviceHandle* device_handle = nullptr);
156
157 // Transfer the given literal to the Infeed interface of the device.
158 //
159 // device_handle and replica_id together specify a particular device; a device
160 // assigned for the given replica_id among the replicas that the given device
161 // handle belongs to.
162 Status TransferToInfeed(const Literal& literal, int64 replica_id = 0,
163 const DeviceHandle* device_handle = nullptr);
164
165 // Transfers from the Outfeed of the device.
166 //
167 // device_handle and replica_id together specify a particular device; a device
168 // assigned for the given replica_id among the replicas that the given device
169 // handle belongs to.
170 StatusOr<std::unique_ptr<Literal>> TransferFromOutfeed(
171 const Shape* shape_with_layout, int64 replica_id = 0,
172 const DeviceHandle* device_handle = nullptr);
173
174 // Resets the device, clearing all existing state on the device.
175 Status ResetDevice();
176
177 // Executes the computation with the given arguments and transfers the result
178 // to the client as a literal. Parameters are defined the same as for
179 // Execute() and Transfer().
180 StatusOr<std::unique_ptr<Literal>> ExecuteAndTransfer(
181 const Computation& computation,
182 tensorflow::gtl::ArraySlice<GlobalData*> arguments,
183 const ExecutionOptions* execution_options = nullptr,
184 ExecutionProfile* execution_profile = nullptr);
185
186 // Executes the computation with the given arguments and transfers the result
187 // to the client as a literal. Parameters are defined the same as for
188 // Execute() and Transfer().
189 //
190 // TODO(b/74197823): This is a part of a NOT YET ready refactor.
191 StatusOr<std::unique_ptr<Literal>> ExecuteAndTransfer(
192 const XlaComputation& computation,
193 tensorflow::gtl::ArraySlice<GlobalData*> arguments,
194 const ExecutionOptions* execution_options = nullptr,
195 ExecutionProfile* execution_profile = nullptr);
196
197 // Unregister the memory for the given GlobalData on the device.
198 Status Unregister(const GlobalData& data);
199
200 // Returns a vector of global data handles that point to the tuple elements.
201 StatusOr<std::vector<std::unique_ptr<GlobalData>>> DeconstructTuple(
202 const GlobalData& data);
203
204 // Retrieves the statistics of the given computation.
205 StatusOr<ComputationStats> GetComputationStats(
206 const Computation& computation, const DebugOptions& debug_options) const;
207
208 // Retrieves the statistics of the given computation.
209 //
210 // TODO(b/74197823): This is a part of a NOT YET ready refactor.
211 StatusOr<ComputationStats> GetComputationStats(
212 const XlaComputation& computation,
213 const DebugOptions& debug_options) const;
214
215 // Returns the Shape of the given array specified by 'data'. The shape
216 // includes the Layout of the array as it is stored on the service.
217 StatusOr<Shape> GetShape(const GlobalData& data);
218
219 // As above, but returns the shape of the provided computation (parameter
220 // types/names and return type).
221 StatusOr<std::unique_ptr<ProgramShape>> GetComputationShape(
222 const Computation& computation);
223
224 // As above, but returns the shape of the provided computation (parameter
225 // types/names and return type).
226 //
227 // TODO(b/74197823): This is a part of a NOT YET ready refactor.
228 StatusOr<std::unique_ptr<ProgramShape>> GetComputationShape(
229 const XlaComputation& computation);
230
231 // Creates a channel handle that can be used to transfer data between
232 // two computations via a pair of Send and Recv instructions.
233 StatusOr<ChannelHandle> CreateChannelHandle();
234
235 StatusOr<Computation> LoadSnapshot(const SessionModule& module);
236
237 ServiceInterface* stub() { return stub_; }
238
239 private:
240 // Returns the execution statistics (e.g., gflop/s) as a string from the
241 // ExecutionProfile returned from an execution of the computation.
242 StatusOr<string> ExecutionStatsAsString(const Computation& computation,
243 const ExecutionProfile& profile);
244 StatusOr<string> ExecutionStatsAsString(const XlaComputation& computation,
245 const ExecutionProfile& profile);
246
247 ServiceInterface* stub_; // Stub that this client is connected on.
248
249 TF_DISALLOW_COPY_AND_ASSIGN(Client);
250};

252} // namespace xla
