24namespace xla {
25
26// Class containing options for building an LocalExecutable with
27// LocalClient::Compile.
28class ExecutableBuildOptions {
29 public:
30 // If set, this is the device to build the computation for. Valid
31 // device_ordinal values are: 0 to # of devices - 1. These values are
32 // identical to the device ordinal values used by StreamExecutor. The built
33 // executable will be executable on any device equivalent to the specified
34 // device as determined by Backend::devices_equivalent(). A value of -1
35 // indicates this option has not been set.
36 ExecutableBuildOptions& set_device_ordinal(int device_ordinal);
37 int device_ordinal() const;
38
39 // If set, this specifies the layout of the result of the computation. If not
40 // set, the service will chose the layout of the result. A Shape is used to
41 // store the layout to accommodate tuple result shapes. A value of nullptr
42 // indicates the option has not been set.
43 ExecutableBuildOptions& set_result_layout(const Shape& shape_with_layout);
44 const Shape* result_layout() const;
45
46 // If set, this specifies an allocator that can be used to allocate temporary
47 // space on the device during compilation. For example, the compiler might
48 // want to run various algorithms on the device and pick the fastest one -- it
49 // might allocate buffers for use by these algorithms using this allocator.
50 //
51 // This does not need to be the same as the DeviceMemoryAllocator passed when
52 // running the executable.
53 ExecutableBuildOptions& set_device_allocator(
54 DeviceMemoryAllocator* allocator);
55 DeviceMemoryAllocator* device_allocator() const;
56
57 // If set, specifies a regexp of HLO graphs to dump (as in DebugOptions).
58 ExecutableBuildOptions& set_generate_hlo_graph(string regex);
59 const tensorflow::gtl::optional<string>& generate_hlo_graph() const;
60
61 // If set, specifies a dirpath to dump the end-of-optimization-pipeline HLO
62 // protobuf to (as in DebugOptions).
63 ExecutableBuildOptions& set_dump_optimized_hlo_proto_to(
64 tensorflow::StringPiece dirpath);
65 const tensorflow::gtl::optional<string>& dump_optimized_hlo_proto_to() const;
66
67 // If set, specifies a dirpath to dump the per-pass-in-pipeline HLO protobufs
68 // to (as in DebugOptions).
69 ExecutableBuildOptions& set_dump_per_pass_hlo_proto_to(
70 tensorflow::StringPiece dirpath);
71 const tensorflow::gtl::optional<string>& dump_per_pass_hlo_proto_to() const;
72
73 // If true, specifies that we should record an HLO profile during execution
74 // and log it after execution (as in DebugOptions). If nullopt the default is
75 // used.
76 ExecutableBuildOptions& set_hlo_profile(bool enabled);
77 tensorflow::gtl::optional<bool> hlo_profile() const;
78
79 // Returns a string representation of the build options, suitable for
80 // debugging.
81 string ToString() const;
82
83 private:
84 tensorflow::gtl::optional<bool> hlo_profile_;
85 int device_ordinal_ = -1;
86 Shape result_layout_;
87 bool result_layout_set_ = false;
88 tensorflow::gtl::optional<string> generate_hlo_graph_;
89 tensorflow::gtl::optional<string> dump_optimized_hlo_proto_to_;
90 tensorflow::gtl::optional<string> dump_per_pass_hlo_proto_to_;
91 DeviceMemoryAllocator* device_allocator_ = nullptr;
92};
93
94} // namespace xla
