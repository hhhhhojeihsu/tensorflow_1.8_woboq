30using InstructionGenerator =
31 ComputationDataHandle (*)(ComputationBuilder*, const ComputationDataHandle&,
32 const ComputationDataHandle&);
33
34Computation CreateScalarComputation(const string& name, PrimitiveType type,
35 ComputationBuilder* builder,
36 InstructionGenerator generator) {
37 std::unique_ptr<ComputationBuilder> b;
38 if (type == PRED) {
39 b = builder->CreateSubBuilder(name);
40 } else {
41 b = builder->CreateSubBuilder(
42 tensorflow::strings::StrCat(name, "_", PrimitiveType_Name(type)));
43 }
44
45 const Shape scalar = ShapeUtil::MakeShape(type, {});
46 auto lhs = b->Parameter(0, scalar, "lhs");
47 auto rhs = b->Parameter(1, scalar, "rhs");
48 generator(b.get(), lhs, rhs);
49 return b->BuildAndNoteError();
50}

128StatusOr<ComputationDataHandle> Any(const ComputationDataHandle& predicates,
129 ComputationBuilder* builder) {
130 auto f = builder->ConstantR0<bool>(false);
131 Computation logical_or = CreateScalarOrComputation(builder);
132 TF_ASSIGN_OR_RETURN(std::unique_ptr<Shape> predicates_shape,
133 builder->GetShape(predicates));
134 std::vector<int64> all_dimensions(ShapeUtil::Rank(*predicates_shape));
135 std::iota(all_dimensions.begin(), all_dimensions.end(), 0);
136 return builder->Reduce(predicates, f, logical_or, all_dimensions);
137}

200StatusOr<XlaOp> Any(const XlaOp& predicates, XlaBuilder* builder) {
201 auto f = builder->ConstantR0<bool>(false);
202 XlaComputation logical_or = CreateScalarOrComputation(builder);
203 TF_ASSIGN_OR_RETURN(const Shape& predicates_shape,
204 builder->GetShape(predicates));
205 std::vector<int64> all_dimensions(ShapeUtil::Rank(predicates_shape));
206 std::iota(all_dimensions.begin(), all_dimensions.end(), 0);
207 return builder->Reduce(predicates, f, logical_or, all_dimensions);
208}
