24Computation::Computation() : parent_(nullptr) {}
25
26Computation::Computation(ServiceInterface* parent,
27 const ComputationHandle& handle)
28 : handle_(handle), parent_(parent) {}
29
30Computation::Computation(Computation&& computation)
31 : handle_(std::move(computation.handle_)), parent_(computation.parent_) {
32 computation.ResetWithoutFreeing();
33}
34
35void Computation::Reset() {
36 // TODO(b/34469253) deallocate any owned computation.
37 ResetWithoutFreeing();
38}
39
40StatusOr<std::unique_ptr<SessionModule>> Computation::Snapshot() const {
41 SnapshotComputationRequest request;
42 *request.mutable_computation() = handle_;
43 SnapshotComputationResponse response;
44
45 TF_RETURN_IF_ERROR(parent_->SnapshotComputation(&request, &response));
46
47 return WrapUnique(response.release_module());
48}
49
50Computation::~Computation() { Reset(); }
51
52Computation& Computation::operator=(Computation&& computation) {
53 if (&computation != this) {
54 Reset();
55 handle_ = computation.handle_;
56 parent_ = computation.parent_;
57 computation.ResetWithoutFreeing();
58 }
59 return *this;
60}
61
62void Computation::ResetWithoutFreeing() {
63 handle_.Clear();
64 parent_ = nullptr;
65}
66
67StatusOr<ProgramShape> Computation::GetProgramShape() const {
68 GetComputationShapeRequest request;
69 *request.mutable_computation() = handle_;
70 GetComputationShapeResponse response;
71
72 TF_RETURN_IF_ERROR(parent_->GetComputationShape(&request, &response));
73
74 return std::move(*response.mutable_program_shape());
75}
