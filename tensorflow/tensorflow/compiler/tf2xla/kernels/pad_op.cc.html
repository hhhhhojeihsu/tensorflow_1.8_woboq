23namespace tensorflow {
24namespace {
25
26class PadOp : public XlaOpKernel {
27 public:
28 explicit PadOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {}
29
30 void Compile(XlaOpKernelContext* ctx) override {
31 const TensorShape input_shape = ctx->InputShape(0);
32 const TensorShape pad_shape = ctx->InputShape(1);
33 const int dims = input_shape.dims();
34 OP_REQUIRES(
35 ctx,
36 TensorShapeUtils::IsMatrix(pad_shape) && pad_shape.dim_size(1) == 2,
37 errors::InvalidArgument("paddings must be a matrix with 2 columns: ",
38 pad_shape.DebugString()));
39 const int fixed_dims =
40 (allow_legacy_scalars() && dims == 0 && pad_shape.dim_size(0) == 1)
41 ? 1
42 : dims;
43 OP_REQUIRES(
44 ctx, fixed_dims == pad_shape.dim_size(0),
45 errors::InvalidArgument(
46 "The first dimension of paddings must be the rank of inputs",
47 pad_shape.DebugString(), " ", input_shape.DebugString()));
48
49 if (fixed_dims == 0) {
50 // Tensor is rank 0. Return it unchanged.
51 ctx->SetOutput(0, ctx->Input(0));
52 return;
53 }
54
55 // Evaluate the 'padding' constant input, reshaping to a matrix.
56 xla::Literal pad_literal;
57 OP_REQUIRES_OK(
58 ctx, ctx->ConstantInputReshaped(1, {fixed_dims, 2}, &pad_literal));
59
60 xla::PaddingConfig config;
61 for (int i = 0; i < fixed_dims; ++i) {
62 auto* dim = config.add_dimensions();
63 int before = pad_literal.Get<int32>({i, 0});
64 int after = pad_literal.Get<int32>({i, 1});
65 OP_REQUIRES(ctx, before >= 0 && after >= 0,
66 errors::InvalidArgument("Paddings must be non-negative: ",
67 before, " ", after));
68 dim->set_edge_padding_low(before);
69 dim->set_edge_padding_high(after);
70 }
71
72 // PadV2 added a "constant_values" input that indicates the pad value.
73 xla::ComputationDataHandle constant_values;
74 if (ctx->num_inputs() == 3) {
75 OP_REQUIRES(ctx, TensorShapeUtils::IsScalar(ctx->InputShape(2)),
76 errors::InvalidArgument("constant_values must be a scalar."));
77 ctx->SetOutput(0,
78 ctx->builder()->Pad(ctx->Input(0), ctx->Input(2), config));
79 } else {
80 auto zero = XlaHelpers::Zero(ctx->builder(), input_type(0));
81 ctx->SetOutput(0, ctx->builder()->Pad(ctx->Input(0), zero, config));
82 }
83 }
84};
85
86REGISTER_XLA_OP(Name("Pad").CompileTimeConstInput("paddings"), PadOp);
87REGISTER_XLA_OP(Name("PadV2").CompileTimeConstInput("paddings"), PadOp);
