24namespace tensorflow {
25namespace {
26
27class L2LossOp : public XlaOpKernel {
28 public:
29 explicit L2LossOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {}
30
31 void Compile(XlaOpKernelContext* ctx) override {
32 std::vector<int64> dims(ctx->InputShape(0).dims());
33 std::iota(dims.begin(), dims.end(), 0);
34
35 DataType dtype = ctx->input_type(0);
36 xla::ComputationBuilder* const b = ctx->builder();
37
38 // output = sum(t ** 2) / 2
39 const DataType accumulation_type = XlaHelpers::SumAccumulationType(dtype);
40 auto t =
41 XlaHelpers::ConvertElementType(b, ctx->Input(0), accumulation_type);
42 auto square = b->Mul(t, t);
43 auto reduce = b->Reduce(square, XlaHelpers::Zero(b, accumulation_type),
44 *ctx->GetOrCreateAdd(accumulation_type), dims);
45 auto deconverted = XlaHelpers::ConvertElementType(b, reduce, dtype);
46 auto two = XlaHelpers::IntegerLiteral(b, dtype, 2);
47 ctx->SetOutput(0, b->Div(deconverted, two));
48 }
49};
50
51REGISTER_XLA_OP(Name("L2Loss"), L2LossOp);
52
