41Status GetStackShape(xla::ComputationBuilder* builder, XlaResource* resource,
42 TensorShape* stack_shape) {
43 auto shape_or_status = builder->GetShape(resource->value());
44 if (!shape_or_status.ok()) {
45 return shape_or_status.status();
46 }
47 xla::Shape shape = *shape_or_status.ValueOrDie();
48 TF_RET_CHECK(xla::ShapeUtil::IsTuple(shape));
49 return XLAShapeToTensorShape(xla::ShapeUtil::GetTupleElementShape(shape, 0),
50 stack_shape);
51}
52
53// Since the element shape is not provided to the Stack operator,
54// we lazily initialize the Stack at the time of the first write.
55//
56// If a Stack `resource` has not been initialized, constructs storage for the
57// Stack with elements of `elem_shape`. For both initialized and
58// uninitialized Stacks, checks that the tensor has a type compatible with
59// 'dtype' and shape compatible with 'elem_shape'.
60//
61// TODO(phawkins): consider changing the API of the stack operators to
62// allow an optional element shape at stack construction time.
63Status MaybeInitializeStack(xla::ComputationBuilder* builder,
64 XlaResource* resource, DataType dtype,
65 const TensorShape& elem_shape) {
66 if (resource->type() != dtype) {
67 return errors::InvalidArgument(
68 "Stack dtype is ", DataTypeString(resource->type()),
69 " but op has dtype ", DataTypeString(dtype), ".");
70 }
71
72 TensorShape stack_shape;
73 stack_shape.AddDim(resource->tensor_array_size());
74 stack_shape.AppendShape(elem_shape);
75
76 if (!resource->initialized()) {
77 // Stack has not been initialized.
78 xla::ComputationDataHandle zero =
79 XlaHelpers::Zero(builder, resource->type());
80 TF_RETURN_IF_ERROR(resource->SetTypeAndShape(dtype, elem_shape));
81 TF_RETURN_IF_ERROR(resource->SetZeroValue(builder));
82 } else {
83 // Checks the expected shape matches the actual shape.
84 TensorShape actual_shape;
85 TF_RETURN_IF_ERROR(GetStackShape(builder, resource, &actual_shape));
86 if (stack_shape != actual_shape) {
87 return errors::InvalidArgument(
88 "Mismatched Stack shapes: ", stack_shape.DebugString(), " vs ",
89 actual_shape.DebugString());
90 }
91 }
92 return Status::OK();
93}

95class StackOp : public XlaOpKernel {
96 public:
97 explicit StackOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {
98 OP_REQUIRES_OK(ctx, ctx->GetAttr("elem_type", &dtype_));
99 OP_REQUIRES_OK(ctx, ctx->GetAttr("stack_name", &stack_name_));
100 }
101
102 void Compile(XlaOpKernelContext* ctx) override {
103 int64 size;
104 OP_REQUIRES_OK(ctx, ctx->ConstantInputAsIntScalar(0, &size));
105 OP_REQUIRES(
106 ctx, size >= 0,
107 errors::InvalidArgument(
108 "XLA compilation requires a fixed stack size upper bound. If "
109 "you are using tf.while_loop, set the maximum_iterations parameter "
110 "to fix this issue."));
111
112 // We defer initializing the Stack resource until we see the first push.
113 // Otherwise we do not know the shape of the stack elements.
114 xla::ComputationDataHandle value;
115 XlaContext& xc = XlaContext::Get(ctx);
116 XlaResource* resource;
117 string name = strings::StrCat("Stack: ", stack_name_);
118 OP_REQUIRES_OK(
119 ctx, xc.CreateResource(XlaResource::kStack, -1, std::move(name), dtype_,
120 TensorShape(), value, /*tensor_array_size=*/size,
121 /*tensor_array_gradients=*/{}, &resource));
122 ctx->SetResourceOutput(0, resource);
123 }
124
125 private:
126 DataType dtype_;
127 string stack_name_;
128
129 TF_DISALLOW_COPY_AND_ASSIGN(StackOp);
130};
131
132REGISTER_XLA_OP(Name("StackV2").CompileTimeConstInput("max_size"), StackOp);

134class StackPushOp : public XlaOpKernel {
135 public:
136 explicit StackPushOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {
137 OP_REQUIRES_OK(ctx, ctx->GetAttr("T", &dtype_));
138 }
139
140 void Compile(XlaOpKernelContext* ctx) override {
141 xla::ComputationBuilder* b = ctx->builder();
142 TensorShape elem_shape = ctx->InputShape(1);
143
144 XlaResource* resource;
145 OP_REQUIRES_OK(ctx, ctx->GetResourceInput(0, &resource));
146
147 // Initializes the Stack, if the element shape was not already known.
148 OP_REQUIRES_OK(ctx, MaybeInitializeStack(b, resource, dtype_, elem_shape));
149
150 xla::ComputationDataHandle ta = b->GetTupleElement(resource->value(), 0);
151 xla::ComputationDataHandle index = b->GetTupleElement(resource->value(), 1);
152 xla::ComputationDataHandle value = ctx->Input(1);
153
154 // start_indices of the DynamicUpdateSlice are [index, 0, 0, ..., 0].
155 auto start_indices =
156 b->Pad(b->Reshape(index, {1}), b->ConstantR0<int32>(0),
157 xla::MakeEdgePaddingConfig({{0, elem_shape.dims()}}));
158
159 TensorShape slice_shape = elem_shape;
160 slice_shape.InsertDim(0, 1LL);
161 auto update = b->Reshape(value, slice_shape.dim_sizes());
162
163 // TODO(phawkins): We don't check the index is in bounds --- there is no
164 // error mechanism in XLA.
165 OP_REQUIRES_OK(ctx, resource->SetValue(b->Tuple(
166 {b->DynamicUpdateSlice(ta, update, start_indices),
167 b->Add(index, b->ConstantR0<int32>(1))})));
168
169 ctx->SetOutput(0, value);
170 }
171
172 private:
173 DataType dtype_;
174
175 TF_DISALLOW_COPY_AND_ASSIGN(StackPushOp);
176};
177
178REGISTER_XLA_OP(Name("StackPushV2"), StackPushOp);

180class StackPopOp : public XlaOpKernel {
181 public:
182 explicit StackPopOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {
183 OP_REQUIRES_OK(ctx, ctx->GetAttr("elem_type", &dtype_));
184 }
185
186 void Compile(XlaOpKernelContext* ctx) override {
187 xla::ComputationBuilder* b = ctx->builder();
188
189 XlaResource* resource;
190 OP_REQUIRES_OK(ctx, ctx->GetResourceInput(0, &resource));
191
192 // There is a somewhat subtle issue here: here "uninitialized" means we have
193 // not yet seen a pop in the order that we compile operators, not the order
194 // that we run them. However, in practice the two orders should be the same
195 // for the sole user of the stack operators (loop gradients).
196 OP_REQUIRES(ctx, resource->initialized(),
197 errors::InvalidArgument("Stack pop on uninitialized stack"));
198
199 TensorShape stack_shape;
200 OP_REQUIRES_OK(ctx, GetStackShape(b, resource, &stack_shape));
201
202 xla::ComputationDataHandle state = resource->value();
203 xla::ComputationDataHandle ta = b->GetTupleElement(state, 0);
204 xla::ComputationDataHandle index = b->GetTupleElement(state, 1);
205
206 index = b->Sub(index, b->ConstantR0<int32>(1));
207 OP_REQUIRES_OK(ctx, resource->SetValue(b->Tuple({ta, index})));
208
209 // start_indices of the DynamicSlice are [index, 0, 0, ..., 0].
210 auto start_indices =
211 b->Pad(b->Reshape(index, {1}), b->ConstantR0<int32>(0),
212 xla::MakeEdgePaddingConfig({{0, stack_shape.dims() - 1}}));
213
214 auto slice_shape = stack_shape.dim_sizes();
215 slice_shape[0] = 1LL;
216
217 // TODO(phawkins): We don't check the index is in bounds --- there is no
218 // error mechanism in XLA.
219 xla::ComputationDataHandle read =
220 b->DynamicSlice(ta, start_indices, slice_shape);
221
222 // Remove the leading '1' dimension.
223 std::vector<int64> value_shape(slice_shape.begin() + 1, slice_shape.end());
224 ctx->SetOutput(0, b->Reshape(read, value_shape));
225 }
226
227 private:
228 DataType dtype_;
229
230 TF_DISALLOW_COPY_AND_ASSIGN(StackPopOp);
231};
232
233REGISTER_XLA_OP(Name("StackPopV2"), StackPopOp);
