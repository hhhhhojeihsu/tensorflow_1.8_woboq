29class SoftmaxOp : public XlaOpKernel {
30 public:
31 explicit SoftmaxOp(OpKernelConstruction* ctx) : XlaOpKernel(ctx) {
32 log_ = str_util::StartsWith(type_string(), "Log");
33 }
34
35 void Compile(XlaOpKernelContext* ctx) override {
36 const TensorShape logits_shape = ctx->InputShape(0);
37 OP_REQUIRES(ctx, TensorShapeUtils::IsMatrix(logits_shape),
38 errors::InvalidArgument("logits must be 2-dimensional"));
39
40 const int kBatchDim = 0;
41 const int kClassDim = 1;
42
43 const DataType type = input_type(0);
44 auto logits = ctx->Input(0);
45
46 xla::ComputationBuilder* const b = ctx->builder();
47 const xla::Computation& max_func = *ctx->GetOrCreateMax(type);
48
49 // Find the max in each batch, resulting in a tensor of shape [batch]
50 auto logits_max =
51 b->Reduce(logits, XlaHelpers::MinValue(b, type), max_func, {kClassDim});
52 // Subtract the max in batch b from every element in batch b. Broadcasts
53 // along the batch dimension.
54 auto shifted_logits = b->Sub(logits, logits_max, {kBatchDim});
55 auto exp_shifted = b->Exp(shifted_logits);
56 const DataType accumulation_type = XlaHelpers::SumAccumulationType(type);
57 auto converted =
58 XlaHelpers::ConvertElementType(b, exp_shifted, accumulation_type);
59 auto reduce =
60 b->Reduce(converted, XlaHelpers::Zero(b, accumulation_type),
61 *ctx->GetOrCreateAdd(accumulation_type), {kClassDim});
62 auto sum = XlaHelpers::ConvertElementType(b, reduce, type);
63 auto softmax =
64 log_
65 // softmax = shifted_logits - log(sum(exp(shifted_logits)))
66 ? b->Sub(shifted_logits, b->Log(sum), {kBatchDim})
67 // softmax = exp(shifted_logits) / sum(exp(shifted_logits))
68 : b->Div(exp_shifted, sum, {kBatchDim});
69 ctx->SetOutput(0, softmax);
70 }
71
72 private:
73 bool log_;
74};
75
76REGISTER_XLA_OP(Name("Softmax"), SoftmaxOp);
77REGISTER_XLA_OP(Name("LogSoftmax"), SoftmaxOp);

79std::pair<xla::ComputationDataHandle, xla::ComputationDataHandle>
80CrossEntropyWithLogits(XlaOpKernelContext* ctx, DataType type,
81 const xla::ComputationDataHandle& logits,
82 const xla::ComputationDataHandle& labels) {
83 const xla::Computation& max_func = *ctx->GetOrCreateMax(type);
84
85 const int kBatchDim = 0;
86 const int kClassDim = 1;
87
88 xla::ComputationBuilder* b = ctx->builder();
89 // Find the max in each batch, resulting in a tensor of shape [batch]
90 auto logits_max =
91 b->Reduce(logits, XlaHelpers::MinValue(b, type), max_func, {kClassDim});
92
93 // Subtract the max in batch b from every element in batch b.
94 // Broadcasts along the batch dimension.
95 auto shifted_logits = b->Sub(logits, logits_max, {kBatchDim});
96
97 // exp(logits - max_logits)
98 auto exp_shifted_logits = b->Exp(shifted_logits);
99
100 // sum_{class} (exp(logits - max_logits))
101 const DataType accumulation_type = XlaHelpers::SumAccumulationType(type);
102 auto converted =
103 XlaHelpers::ConvertElementType(b, exp_shifted_logits, accumulation_type);
104 auto reduce = b->Reduce(converted, XlaHelpers::Zero(b, accumulation_type),
105 *ctx->GetOrCreateAdd(accumulation_type), {kClassDim});
106 auto sum_exp = XlaHelpers::ConvertElementType(b, reduce, type);
107
108 // log(sum(exp(logits - max_logits)))
109 auto log_sum_exp = b->Log(sum_exp);
110
111 // sum(-labels *
112 // ((logits - max_logits) - log(sum(exp(logits - max_logits)))))
113 // along classes
114 // (The subtraction broadcasts along the batch dimension.)
115 auto sub = b->Sub(shifted_logits, log_sum_exp, {kBatchDim});
116 auto mul = b->Mul(b->Neg(labels), sub);
117 auto sum =
118 b->Reduce(XlaHelpers::ConvertElementType(b, mul, accumulation_type),
119 XlaHelpers::Zero(b, accumulation_type),
120 *ctx->GetOrCreateAdd(accumulation_type), {kClassDim});
121 auto loss = XlaHelpers::ConvertElementType(b, sum, type);
122
123 // backprop: prob - labels, where
124 // prob = exp(logits - max_logits) / sum(exp(logits - max_logits))
125 // (where the division broadcasts along the batch dimension)
126 xla::ComputationDataHandle backprop =
127 b->Sub(b->Div(exp_shifted_logits, sum_exp, {kBatchDim}), labels);
128 return {loss, backprop};
129}

131class SoftmaxXentWithLogitsOp : public XlaOpKernel {
132 public:
133 explicit SoftmaxXentWithLogitsOp(OpKernelConstruction* ctx)
134 : XlaOpKernel(ctx) {}
135
136 void Compile(XlaOpKernelContext* ctx) override {
137 const TensorShape logits_shape = ctx->InputShape(0);
138 const TensorShape labels_shape = ctx->InputShape(1);
139 OP_REQUIRES(ctx, logits_shape.IsSameSize(labels_shape),
140 errors::InvalidArgument(
141 "logits and labels must be same size: logits_size=",
142 logits_shape.DebugString(),
143 " labels_size=", labels_shape.DebugString()));
144 OP_REQUIRES(ctx, TensorShapeUtils::IsMatrix(logits_shape),
145 errors::InvalidArgument("logits must be 2-dimensional"));
146 // As we already tested that both inputs have the same shape no need to
147 // check that "labels" is a matrix too.
148
149 const DataType type = input_type(0);
150 auto logits = ctx->Input(0);
151 auto labels = ctx->Input(1);
152
153 xla::ComputationDataHandle loss, backprop;
154 std::tie(loss, backprop) =
155 CrossEntropyWithLogits(ctx, type, logits, labels);
156 ctx->SetOutput(0, loss);
157 ctx->SetOutput(1, backprop);
158 }
159};

161REGISTER_XLA_OP(Name("SoftmaxCrossEntropyWithLogits"), SoftmaxXentWithLogitsOp);

163class SparseSoftmaxXentWithLogitsOp : public XlaOpKernel {
164 public:
165 explicit SparseSoftmaxXentWithLogitsOp(OpKernelConstruction* ctx)
166 : XlaOpKernel(ctx) {}
167
168 void Compile(XlaOpKernelContext* ctx) override {
169 const TensorShape logits_shape = ctx->InputShape(0);
170 const TensorShape labels_shape = ctx->InputShape(1);
171 OP_REQUIRES(ctx, TensorShapeUtils::IsMatrix(logits_shape),
172 errors::InvalidArgument("logits must be 2-D, but got shape ",
173 logits_shape.DebugString()));
174 OP_REQUIRES(ctx, TensorShapeUtils::IsVector(labels_shape),
175 errors::InvalidArgument("labels must be 1-D, but got shape ",
176 labels_shape.DebugString()));
177 OP_REQUIRES(ctx, logits_shape.dim_size(0) == labels_shape.dim_size(0),
178 errors::InvalidArgument(
179 "logits and labels must have the same first dimension, "
180 "got logits shape ",
181 logits_shape.DebugString(), " and labels shape ",
182 labels_shape.DebugString()));
183 OP_REQUIRES(ctx, logits_shape.dim_size(1) > 0,
184 errors::InvalidArgument(
185 "Must have at least one class, but got logits shape ",
186 logits_shape.DebugString()));
187
188 int64 batch_size = logits_shape.dim_size(0);
189 int64 depth = logits_shape.dim_size(1);
190
191 DataType logits_type = input_type(0);
192 DataType indices_type = input_type(1);
193
194 xla::ComputationDataHandle indices = ctx->Input(1);
195
196 xla::ComputationBuilder* builder = ctx->builder();
197 xla::ComputationDataHandle labels;
198 OP_REQUIRES_OK(ctx,
199 XlaHelpers::OneHot(
200 builder, depth, /*axis=*/1, input_type(1), labels_shape,
201 indices, XlaHelpers::One(builder, logits_type),
202 XlaHelpers::Zero(builder, logits_type), &labels));
203
204 // If any of the indices are out of range, we must populate the labels with
205 // NaNs to obey the interface contract of
206 // tf.nn.sparse_softmax_cross_entropy_with_logits.
207 // Builds a vector of {batch_size} that is 0 if the index is in range, or
208 // NaN otherwise; then add that vector to the labels to force out-of-range
209 // values to NaNs.
210 xla::ComputationDataHandle nan_or_zero = builder->Select(
211 builder->And(
212 builder->Le(XlaHelpers::Zero(builder, indices_type), indices),
213 builder->Lt(indices, XlaHelpers::IntegerLiteral(
214 builder, indices_type, depth))),
215 builder->Broadcast(XlaHelpers::Zero(builder, logits_type),
216 {batch_size}),
217 builder->Broadcast(XlaHelpers::FloatLiteral(builder, logits_type, NAN),
218 {batch_size}));
219 labels = builder->Add(labels, nan_or_zero, {0});
220
221 xla::ComputationDataHandle loss, backprop;
222 std::tie(loss, backprop) =
223 CrossEntropyWithLogits(ctx, logits_type, ctx->Input(0), labels);
224 ctx->SetOutput(0, loss);
225 ctx->SetOutput(1, backprop);
226 }
227};
228
229REGISTER_XLA_OP(Name("SparseSoftmaxCrossEntropyWithLogits"),
230 SparseSoftmaxXentWithLogitsOp);
