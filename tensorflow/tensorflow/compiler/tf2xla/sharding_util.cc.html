24namespace {
25const char kDeviceSuffixReplicatedCore[] = "REPLICATED_CORE";
26const char kShardingAttribute[] = "_XlaSharding";
27} // namespace
28
29namespace {
30xla::StatusOr<tensorflow::gtl::optional<xla::OpSharding>>
31GetShardingFromNodeDef(const NodeDef& node_def) {
32 if (!HasNodeAttr(node_def, kShardingAttribute)) {
33 return tensorflow::gtl::optional<xla::OpSharding>();
34 }
35 string value;
36 xla::OpSharding sharding;
37 TF_RETURN_IF_ERROR(GetNodeAttr(node_def, kShardingAttribute, &value));
38 if (!sharding.ParseFromString(value)) {
39 return xla::InvalidArgument(
40 "Experimental _XlaSharding attribute was not a valid encoded "
41 "xla::OpSharding proto.");
42 }
43 return tensorflow::gtl::optional<xla::OpSharding>(sharding);
44}
45
46Status CoreOutOfRangeError(int core, int num_cores_per_replica) {
47 return errors::InvalidArgument(
48 "Invalid replicated core id: ", core,
49 "; num_cores_per_replica=", num_cores_per_replica);
50}
51} // namespace

53xla::StatusOr<tensorflow::gtl::optional<xla::OpSharding>>
54ParseShardingFromDevice(
55 const string& device_name, int num_cores_per_replica,
56 tensorflow::gtl::optional<xla::OpSharding> explicit_sharding) {
57 if (device_name.empty()) {
58 return tensorflow::gtl::optional<xla::OpSharding>();
59 }
60 DeviceNameUtils::ParsedName parsed_device;
61 if (!DeviceNameUtils::ParseFullName(device_name, &parsed_device)) {
62 return errors::InvalidArgument("Malformed assigned device '", device_name,
63 "'");
64 }
65
66 if (explicit_sharding.has_value()) {
67 return explicit_sharding;
68 } else if (!parsed_device.has_type || !parsed_device.has_id ||
69 !str_util::StrContains(parsed_device.type,
70 kDeviceSuffixReplicatedCore)) {
71 return tensorflow::gtl::optional<xla::OpSharding>();
72 } else {
73 const int core = parsed_device.id;
74 if (core < 0 || core >= num_cores_per_replica) {
75 return CoreOutOfRangeError(core, num_cores_per_replica);
76 }
77 return tensorflow::gtl::optional<xla::OpSharding>(
78 xla::sharding_builder::AssignDevice(core));
79 }
80}
81
82xla::StatusOr<tensorflow::gtl::optional<xla::OpSharding>>
83ParseShardingFromDevice(const NodeDef& node_def, int num_cores_per_replica) {
84 const string& device_name = node_def.device();
85 TF_ASSIGN_OR_RETURN(tensorflow::gtl::optional<xla::OpSharding> sharding,
86 GetShardingFromNodeDef(node_def));
87 return ParseShardingFromDevice(device_name, num_cores_per_replica, sharding);
88}
89
90xla::StatusOr<tensorflow::gtl::optional<xla::OpSharding>>
91ParseShardingFromDevice(const Node& node, int num_cores_per_replica) {
92 string device_name = node.assigned_device_name();
93 if (device_name.empty()) {
94 device_name = node.requested_device();
95 }
96 TF_ASSIGN_OR_RETURN(tensorflow::gtl::optional<xla::OpSharding> sharding,
97 GetShardingFromNodeDef(node.def()));
98 return ParseShardingFromDevice(device_name, num_cores_per_replica, sharding);
99}
100
101void SetShardingDeviceAssignmentFromNode(const Node& src, Node* dst) {
102 string device_name = src.assigned_device_name();
103 if (device_name.empty()) {
104 device_name = src.requested_device();
105 }
106 dst->set_assigned_device_name(device_name);
107 if (const AttrValue* attr = src.attrs().Find(kShardingAttribute)) {
108 dst->AddAttr(kShardingAttribute, *attr);
109 }
110}
