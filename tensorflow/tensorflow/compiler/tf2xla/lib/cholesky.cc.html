34// def cholesky_unblocked(a):
35// assert len(a.shape) == 2 and a.shape[-2] == a.shape[-1]
36// n = a.shape[-2]
37// l = np.zeros_like(a)
38// for j in xrange(n):
39// r = l[..., j, :j]
40// l[..., j, j] = np.sqrt(a[..., j, j] - np.dot(r, r))
41// l[..., j+1:, j] = (a[..., j+1:, j] - np.dot(l[..., j+1:, :j],
42// np.transpose(r))) / l[..., j, j]
43// return l
44xla::StatusOr<xla::ComputationDataHandle> CholeskyUnblocked(
45 xla::ComputationBuilder* builder, const xla::ComputationDataHandle& a) {
46 TF_ASSIGN_OR_RETURN(std::unique_ptr<xla::Shape> shape, builder->GetShape(a));
47 xla::ComputationDataHandle l = Zeros(builder, *shape);
48 const int64 n = xla::ShapeUtil::GetDimension(*shape, -2);
49 for (int j = 0; j < n; ++j) {
50 // Picture of block structure:
51 // ... \
52 // \
53 // -- r -- d
54 // |\
55 // B c \
56 // | \
57 // | ...
58 //
59 // ^
60 // column j
61 TF_ASSIGN_OR_RETURN(auto d,
62 SliceInMinorDims(builder, a, {j, j}, {j + 1, j + 1}));
63 TF_ASSIGN_OR_RETURN(auto c,
64 SliceInMinorDims(builder, a, {j + 1, j}, {n, j + 1}));
65 xla::ComputationDataHandle new_d_squared = d;
66 xla::ComputationDataHandle br;
67 if (j > 0) {
68 TF_ASSIGN_OR_RETURN(auto r,
69 SliceInMinorDims(builder, l, {j, 0}, {j + 1, j}));
70 TF_ASSIGN_OR_RETURN(auto b,
71 SliceInMinorDims(builder, l, {j + 1, 0}, {n, j}));
72 TF_ASSIGN_OR_RETURN(auto r_squared,
73 BatchDot(builder, r, r, /*transpose_x=*/false,
74 /*transpose_y=*/true, /*conjugate_x=*/false,
75 /*conjugate_y=*/false));
76 new_d_squared = builder->Sub(new_d_squared, r_squared);
77
78 TF_ASSIGN_OR_RETURN(br, BatchDot(builder, b, r, /*transpose_x=*/false,
79 /*transpose_y=*/true,
80 /*conjugate_x=*/false,
81 /*conjugate_y=*/false));
82 }
83 auto new_d_inv = builder->Pow(
84 new_d_squared, FloatLiteral(builder, shape->element_type(), -0.5));
85 auto new_d = builder->Mul(new_d_inv, new_d_squared);
86 TF_ASSIGN_OR_RETURN(l, UpdateSliceInMinorDims(builder, l, new_d, {j, j}));
87
88 if (j > 0) {
89 c = builder->Sub(c, br);
90 }
91 auto new_c = builder->Mul(c, new_d_inv);
92 TF_ASSIGN_OR_RETURN(l,
93 UpdateSliceInMinorDims(builder, l, new_c, {j + 1, j}));
94 }
95 return l;
96}

100xla::StatusOr<xla::ComputationDataHandle> Cholesky(
101 xla::ComputationBuilder* builder, xla::ComputationDataHandle a,
102 int64 block_size) {
103 TF_ASSIGN_OR_RETURN(std::unique_ptr<xla::Shape> a_shape,
104 builder->GetShape(a));
105 const int ndims = xla::ShapeUtil::Rank(*a_shape);
106 if (ndims < 2) {
107 return errors::InvalidArgument(
108 "Arguments to Cholesky must have rank >= 2: ", ndims);
109 }
110
111 const int64 n = xla::ShapeUtil::GetDimension(*a_shape, -1);
112 if (n != xla::ShapeUtil::GetDimension(*a_shape, -2)) {
113 return errors::InvalidArgument(
114 "Arguments to Cholesky must be square matrices: ",
115 xla::ShapeUtil::HumanString(*a_shape));
116 }
117
118 if (block_size < 1) {
119 return errors::InvalidArgument(
120 "block_size argument to Cholesky must be >= 1; got ", block_size);
121 }
122
123 // Blocked left-looking Cholesky factorization.
124 // Algorithm 1 from
125 // Haidar, Azzam, et al. "High-performance Cholesky factorization for GPU-only
126 // execution." Proceedings of General Purpose GPUs. ACM, 2017.
127 xla::ComputationDataHandle l = Zeros(builder, *a_shape);
128 for (int64 i = 0; i < n; i += block_size) {
129 int64 k = std::min(block_size, n - i);
130 if (i > 0) {
131 // TODO(phawkins): consider implementing SYRK for the diagonal part of
132 // the panel.
133 // a[i:, i:i+k] -= np.dot(l[i:, :i], np.transpose(l[i:i+k, :i]))
134 TF_ASSIGN_OR_RETURN(auto lhs,
135 SliceInMinorDims(builder, l, {i, 0}, {n, i}));
136 TF_ASSIGN_OR_RETURN(auto rhs,
137 SliceInMinorDims(builder, l, {i, 0}, {i + k, i}));
138 TF_ASSIGN_OR_RETURN(auto delta,
139 BatchDot(builder, lhs, rhs, /*transpose_x=*/false,
140 /*transpose_y=*/true, /*conjugate_x=*/false,
141 /*conjugate_y=*/false));
142 TF_ASSIGN_OR_RETURN(auto before,
143 SliceInMinorDims(builder, a, {i, i}, {n, i + k}));
144 TF_ASSIGN_OR_RETURN(
145 a, UpdateSliceInMinorDims(builder, a, builder->Sub(before, delta),
146 {i, i}));
147 }
148
149 // l[i:i+k, i:i+k] = cholesky_unblocked(a[i:i+k, i:i+k])
150 TF_ASSIGN_OR_RETURN(auto x,
151 SliceInMinorDims(builder, a, {i, i}, {i + k, i + k}));
152 TF_ASSIGN_OR_RETURN(auto factorized, CholeskyUnblocked(builder, x));
153 TF_ASSIGN_OR_RETURN(l,
154 UpdateSliceInMinorDims(builder, l, factorized, {i, i}));
155
156 if (i + k < n) {
157 // l[i+k:, i:i+k] = trsm_right_transpose(l[i:i+k, i:i+k], a[i+k:, i:i+k])
158 TF_ASSIGN_OR_RETURN(auto panel,
159 SliceInMinorDims(builder, a, {i + k, i}, {n, i + k}));
160 TF_ASSIGN_OR_RETURN(auto update,
161 TriangularSolve(builder, factorized, panel,
162 /*left_side=*/false,
163 /*lower=*/true,
164 /*transpose_a=*/true,
165 /*conjugate_a=*/false,
166 /*block_size=*/8));
167 TF_ASSIGN_OR_RETURN(
168 l, UpdateSliceInMinorDims(builder, l, update, {i + k, i}));
169 }
170 }
171 return l;
172}
