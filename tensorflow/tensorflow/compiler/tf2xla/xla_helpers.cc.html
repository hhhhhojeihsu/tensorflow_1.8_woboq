35Status ArgMinMax(xla::ComputationBuilder* builder, XlaOpKernelContext* ctx,
36 const xla::ComputationDataHandle& input,
37 const TensorShape& input_shape, DataType input_type,
38 DataType output_type, int axis, bool is_min,
39 xla::ComputationDataHandle* argminmax) {
40 xla::ComputationDataHandle init_value;
41 const xla::Computation* reducer;
42 if (is_min) {
43 init_value = XlaHelpers::MaxValue(builder, input_type);
44 reducer = ctx->GetOrCreateMin(input_type);
45 } else {
46 init_value = XlaHelpers::MinValue(builder, input_type);
47 reducer = ctx->GetOrCreateMax(input_type);
48 }
49
50 xla::PrimitiveType xla_output_type;
51 TF_RETURN_IF_ERROR(DataTypeToPrimitiveType(output_type, &xla_output_type));
52
53 xla::ComputationDataHandle input_max = builder->Reduce(
54 input, init_value, *reducer, /*dimensions_to_reduce=*/{axis});
55 std::vector<int64> broadcast_dims(input_shape.dims() - 1);
56 std::iota(broadcast_dims.begin(), broadcast_dims.begin() + axis, 0);
57 std::iota(broadcast_dims.begin() + axis, broadcast_dims.end(), axis + 1);
58 // Compute a mask that has 1s for elements equal to the maximum.
59 xla::ComputationDataHandle partial_mask = builder->ConvertElementType(
60 builder->Eq(input, input_max, broadcast_dims), xla_output_type);
61
62 // In order to make identity elements for a bitwise And, we:
63 // Left shift the 1 to the leftmost bit, yielding 0x10...0
64 // Arithmetic right shift the 1 back to the rightmost bit, yielding
65 // 0xFF...F
66 int32 bits_in_type =
67 xla::ShapeUtil::ByteSizeOfPrimitiveType(xla_output_type) * 8 - 1;
68 xla::ComputationDataHandle shift_amount =
69 XlaHelpers::IntegerLiteral(builder, output_type, bits_in_type);
70 xla::ComputationDataHandle full_mask = builder->ShiftRightArithmetic(
71 builder->ShiftLeft(partial_mask, shift_amount), shift_amount);
72
73 // And with the vector [0, 1, 2, ...] to convert each 0xFF...F into its
74 // index.
75 xla::ComputationDataHandle iota;
76
77 const int64 axis_size = input_shape.dim_size(axis);
78 TF_RETURN_IF_ERROR(XlaHelpers::Iota(builder, output_type, axis_size, &iota));
79 xla::ComputationDataHandle product =
80 builder->And(full_mask, iota, /*broadcast_dimensions=*/{axis});
81
82 // If there are multiple maximum elements, choose the one with the highest
83 // index.
84 xla::ComputationDataHandle output =
85 builder->Reduce(product, XlaHelpers::MinValue(builder, output_type),
86 *ctx->GetOrCreateMax(output_type),
87 /*dimensions_to_reduce=*/{axis});
88 *argminmax = output;
89 return Status::OK();
90}

122xla::ComputationDataHandle XlaHelpers::Epsilon(xla::ComputationBuilder* b,
123 DataType data_type) {
124 switch (data_type) {
125 case DT_HALF:
126 return b->ConstantR0<Eigen::half>(
127 static_cast<Eigen::half>(Eigen::NumTraits<Eigen::half>::epsilon()));
128 case DT_BFLOAT16:
129 return b->ConstantR0<bfloat16>(bfloat16::epsilon());
130 case DT_FLOAT:
131 return b->ConstantR0<float>(std::numeric_limits<float>::epsilon());
132 case DT_DOUBLE:
133 return b->ConstantR0<double>(std::numeric_limits<double>::epsilon());
134 default:
135 LOG(FATAL) << "Unsupported type in XlaHelpers::Epsilon: "
136 << DataTypeString(data_type);
137 }
138}

155/* static */ Status XlaHelpers::ReshapeLiteral(
156 const xla::Literal& input, gtl::ArraySlice<int64> dimensions,
157 xla::Literal* output) {
158 if (xla::ShapeUtil::IsTuple(input.shape())) {
159 return errors::InvalidArgument("ReshapeLiteral does not support tuples.");
160 }
161 xla::Shape shape =
162 xla::ShapeUtil::MakeShape(input.shape().element_type(), dimensions);
163 int64 elements_before = xla::ShapeUtil::ElementsIn(input.shape());
164 int64 elements_after = xla::ShapeUtil::ElementsIn(shape);
165 if (elements_before != elements_after) {
166 return errors::InvalidArgument(
167 "Shapes before and after ReshapeLiteral have different numbers of "
168 "elements.");
169 }
170
171 *output = input.Clone();
172 output->mutable_shape_do_not_use()->Swap(&shape);
173 return Status::OK();
174}

176template <typename T>
177static Tensor MakeLinspaceTensor(const TensorShape& shape, int64 depth) {
178 Tensor linspace(DataTypeToEnum<T>::v(), shape);
179 auto linspace_flat = linspace.flat<T>();
180 for (int64 i = 0; i < depth; ++i) {
181 linspace_flat(i) = i;
182 }
183 return linspace;
184}

206Status XlaHelpers::Iota(xla::ComputationBuilder* builder, DataType dtype,
207 int64 size, xla::ComputationDataHandle* iota) {
208 TensorShape linspace_shape({size});
209 Tensor linspace;
210 switch (dtype) {
211 case DT_UINT8:
212 linspace = MakeLinspaceTensor<uint8>(linspace_shape, size);
213 break;
214 case DT_INT32:
215 linspace = MakeLinspaceTensor<int32>(linspace_shape, size);
216 break;
217 case DT_INT64:
218 linspace = MakeLinspaceTensor<int64>(linspace_shape, size);
219 break;
220 default:
221 return errors::InvalidArgument("Invalid argument type ",
222 DataTypeString(dtype));
223 }
224 xla::Literal linspace_literal;
225 TF_RETURN_IF_ERROR(HostTensorToLiteral(linspace, &linspace_literal));
226 *iota = builder->ConstantLiteral(linspace_literal);
227 return Status::OK();
228}

230Status XlaHelpers::OneHot(xla::ComputationBuilder* builder, int64 depth,
231 int axis, DataType index_type,
232 const TensorShape& indices_shape,
233 const xla::ComputationDataHandle& indices,
234 const xla::ComputationDataHandle& on_value,
235 const xla::ComputationDataHandle& off_value,
236 xla::ComputationDataHandle* one_hot) {
237 const int indices_dims = indices_shape.dims();
238 const int output_dims = indices_dims + 1;
239
240 TensorShape output_shape = indices_shape;
241 output_shape.InsertDim(axis, depth);
242
243 // Build a Tensor populated with values 0, 1, 2, ... depth.
244 std::vector<int64> linspace_dims(output_dims, 1);
245 linspace_dims[axis] = depth;
246 TensorShape linspace_shape(linspace_dims);
247 Tensor linspace;
248 switch (index_type) {
249 case DT_UINT8:
250 linspace = MakeLinspaceTensor<uint8>(linspace_shape, depth);
251 break;
252 case DT_INT32:
253 linspace = MakeLinspaceTensor<int32>(linspace_shape, depth);
254 break;
255 case DT_INT64:
256 linspace = MakeLinspaceTensor<int64>(linspace_shape, depth);
257 break;
258 default:
259 return errors::InvalidArgument("Invalid argument type ",
260 DataTypeString(index_type));
261 }
262 xla::Literal linspace_literal;
263 TF_RETURN_IF_ERROR(HostTensorToLiteral(linspace, &linspace_literal));
264
265 // Broadcast the linspace constant across the indices along the new axis,
266 // and test equality at each position.
267 std::vector<int64> broadcast_dims(indices_shape.dims());
268 std::iota(broadcast_dims.begin(), broadcast_dims.begin() + axis, 0);
269 std::iota(broadcast_dims.begin() + axis, broadcast_dims.end(), axis + 1);
270 xla::ComputationDataHandle one_hot_bool = builder->Eq(
271 indices, builder->ConstantLiteral(linspace_literal), broadcast_dims);
272
273 // Selects the user-provided off_value and on_value values.
274 *one_hot = builder->Select(
275 one_hot_bool, builder->Broadcast(on_value, output_shape.dim_sizes()),
276 builder->Broadcast(off_value, output_shape.dim_sizes()));
277 return Status::OK();
278}
