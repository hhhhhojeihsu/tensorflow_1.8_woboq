25static void ReplaceAll(string* text, StringPiece from, StringPiece to) {
26 size_t pos = 0;
27 while ((pos = text->find(from.data(), pos, from.size())) != string::npos) {
28 text->replace(pos, from.size(), to.data(), to.size());
29 pos += to.size();
30 if (from.empty()) {
31 pos++; // Match at the beginning of the text and after every byte
32 }
33 }
34}
35
36void ReplaceAllPairs(string* text,
37 const std::vector<std::pair<string, string>>& replace) {
38 for (const std::pair<string, string>& from_to : replace) {
39 ReplaceAll(text, from_to.first, from_to.second);
40 }
41}
