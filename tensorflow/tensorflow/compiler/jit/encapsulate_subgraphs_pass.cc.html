61bool AreAllParentsConst(const Node& n,
62 const gtl::FlatSet<const Node*>& runtime_const_nodes) {
63 if (n.type_string() == "GuaranteeConst" || n.type_string() == "Const") {
64 // If the current node is itself a cast-to-const, no need
65 // to look at the incoming edges.
66 return true;
67 }
68
69 bool all_parents_const = true;
70 bool atleast_one_non_control_edge = false;
71 for (const Edge* in : n.in_edges()) {
72 atleast_one_non_control_edge =
73 atleast_one_non_control_edge || !in->IsControlEdge();
74 if (!in->IsControlEdge() && runtime_const_nodes.count(in->src()) == 0) {
75 all_parents_const = false;
76 break;
77 }
78 }
79 return all_parents_const && atleast_one_non_control_edge;
80}

82void MarkGuaranteedConstants(
83 const Graph& graph,
84 const std::vector<std::pair<const Node*, Node*>>& src_arg_pairs) {
85 gtl::FlatSet<const Node*> guaranteed_const_nodes;
86 std::vector<const Node*> srcs;
87 srcs.reserve(src_arg_pairs.size());
88 for (const auto& src_arg : src_arg_pairs) {
89 srcs.push_back(src_arg.first);
90 }
91 ReverseDFSFrom(graph, srcs, /*enter=*/nullptr,
92 /*leave=*/[&guaranteed_const_nodes](const Node* n) {
93 // TODO(vinuraja): Doesn't work in the presence of loops.
94 if (AreAllParentsConst(*n, guaranteed_const_nodes)) {
95 guaranteed_const_nodes.insert(n);
96 }
97 });
98
99 for (auto& src_arg : src_arg_pairs) {
100 if (guaranteed_const_nodes.count(src_arg.first) != 0) {
101 VLOG(1) << "Guaranteed const found: " << src_arg.first->DebugString();
102 src_arg.second->AddAttr("_is_guaranteed_constant", true);
103 }
104 }
105}

107// A node/slot pair.
108// TODO(phawkins): is there a common definition of this?
109struct NodeSlot {
110 NodeSlot() : node(nullptr), slot(-1), dtype(DT_INVALID) {}
111 NodeSlot(const Node* node, int slot)
112 : node(node), slot(slot), dtype(DT_INVALID) {}
113 NodeSlot(const Node* node, int slot, DataType dtype)
114 : node(node), slot(slot), dtype(dtype) {}
115
116 const Node* node;
117 int slot;
118
119 // Optional: used to record the destination type of a source NodeSlot in case
120 // the source output is a Ref type that is cast to a Tensor at the
121 // destination.
122 DataType dtype;
123
124 bool operator==(const NodeSlot& other) const {
125 return node == other.node && slot == other.slot && dtype == other.dtype;
126 }
127
128 // Leave dtype out of the hash since there are never two NodeSlots with the
129 // same node and slot and different dtypes.
130 struct Hasher {
131 uint64 operator()(NodeSlot const& s) const {
132 return Hash64Combine(std::hash<const Node*>()(s.node),
133 std::hash<int>()(s.slot));
134 }
135 };
136
137 struct PairHasher {
138 uint64 operator()(std::pair<NodeSlot, NodeSlot> const& s) const {
139 return Hash64Combine(Hasher()(s.first), Hasher()(s.second));
140 }
141 };
142};

181 // A subgraph of the input, all marked with a common 'group_attribute'
182 // value. A subgraph may contain multiple `outside_compilation' clusters.
183 //
184 // In the following simple example, A, B, ..., E are nodes in the original
185 // graph. The group attributes and outside_compilation attributes g and oc are
186 // each shown as either 0 or empty.
187 //
188 // A --> B --> C --> D --> E
189 // g: g:0 g:0 g:0 g:
190 // oc: oc: oc:0 oc: oc:
191 //
192 // The example is rewritten to two graphs; one on the host and one to be
193 // compiled. The host graph is as follows. RAH is a RecvAtHost node receiving
194 // input from the compiled cluster, and SFH is a SendFromHost node sending
195 // input back to the compiled cluster. Dotted edges are control edges. A
196 // 'sequencing' node S is inserted, and both RAH and SFH are connected via S
197 // to E (and in general all nodes that depend on nodes in the compiled
198 // cluster) to ensure that they are not pruned.
199 //
200 // A --> Call --> E
201 // ^
202 // .
203 // ........> S
204 // .... ^
205 // .. .
206 // RAH --> C --> SFH
207 //
208 // The compiled cluster is as follows. HC is a HostCompute node which is the
209 // source of a channel to the RAH node above and the destination of a channel
210 // from the SFH node above.
211 //
212 // Arg --> B --> HC --> D --> Retval
213 //
214 // The channels HC/RAH and SFH/HC each transmit multiple tensors, so there is
215 // at most one RAH and SFH in each outside_compilation cluster. This design is
216 // preferred over adding separate Arg/Retval nodes for each transmitted value
217 // because it allows optimizations to the host code that would like to limit
218 // communication between host and device and, e.g., raise only one interrupt
219 // per channel rather than one per transmitted value.
220 //
221 // The shapes of the outputs from the HC node in general cannot be determined
222 // until the shapes of its inputs are known at compile time, since e.g.,
223 // above, the shape of C's outputs aren't known until the shape of its inputs
224 // are known. If the shapes of the HC's outputs can be determined during the
225 // rewrite, they are stored in the node's 'shapes' attr. Otherwise a minimal
226 // graph is stored in the shape_inference_graph attr. This graph can be used
227 // when compiling the HC Op to determined the shape of the SFH inputs given
228 // the shapes of any ancestor RAH outputs. If it can be determined that the
229 // shape of the SFH inputs will not be inferrable even once the shapes of the
230 // RAH outputs are known, an error is returned by the rewriter.

681Status Encapsulator::Subgraph::RecordArg(
682 const Edge* edge, const std::unordered_map<const Node*, Node*>& node_images,
683 std::vector<std::pair<const Node*, Node*>>* src_arg_pairs) {
684 Node* src_node = edge->src();
685 int src_slot = edge->src_output();
686 std::unordered_map<NodeSlot, int, NodeSlot::Hasher>::iterator iter;
687 bool inserted;
688 std::tie(iter, inserted) =
689 args_by_src_.emplace(NodeSlot(src_node, src_slot), args_by_src_.size());
690 int arg_index = iter->second;
691 if (inserted) {
692 NodeDef arg_def;
693 NodeDefBuilder builder(
694 strings::StrCat(src_node->name(), "_", src_slot, "_arg"), kArgOp);
695 DataType dtype = edge->dst()->input_type(edge->dst_input());
696 builder.Attr("T", dtype);
697 builder.Attr("index", arg_index);
698 Status s = builder.Finalize(&arg_def);
699 if (!s.ok()) return s;
700
701 Node* arg = graph_->AddNode(arg_def, &s);
702 if (!s.ok()) return s;
703
704 src_arg_pairs->push_back({src_node, arg});
705 args_.push_back(arg);
706 }
707 Node* dst_node = edge->dst();
708 Node* dst_image = node_images.at(dst_node);
709 int dst_slot = edge->dst_input();
710 args_by_dst_[NodeSlot(dst_node, dst_slot)] = arg_index;
711 graph_->AddEdge(args_[arg_index], 0, dst_image, dst_slot);
712 return Status::OK();
713}

786Status Encapsulator::Subgraph::AddHostComputes(
787 const string& subgraph_name,
788 const std::unordered_map<const Node*, Node*>& node_images) {
789 for (auto& oc_subgraph_iter : outside_compilation_subgraphs_) {
790 const string& oc_subgraph_name = oc_subgraph_iter.first;
791 OutsideCompilationSubgraph& oc_subgraph = oc_subgraph_iter.second;
792 if (!oc_subgraph.inputs.empty() || !oc_subgraph.control_inputs.empty() ||
793 !oc_subgraph.outputs_by_src.empty() ||
794 !oc_subgraph.control_outputs.empty()) {
795 // Build a _HostCompute node.
796 std::vector<NodeDefBuilder::NodeOut> inputs(oc_subgraph.inputs.size());
797 std::vector<DataType> input_dtypes(oc_subgraph.inputs.size(), DT_INVALID);
798 std::vector<DataType> output_dtypes(oc_subgraph.outputs_by_src.size(),
799 DT_INVALID);
800
801 for (const auto& input_src : oc_subgraph.inputs) {
802 const Node* src_node = input_src.first.node;
803 Node* src_image = node_images.at(src_node);
804 int src_slot = input_src.first.slot;
805 int input_index = input_src.second;
806
807 DataType dtype = src_node->output_type(src_slot);
808 inputs[input_index].Reset(src_image->name(), src_slot, dtype);
809 input_dtypes[input_index] = dtype;
810 }
811
812 for (const auto& output : oc_subgraph.outputs_by_src) {
813 DataType dtype = output.first.dtype;
814 int output_index = output.second;
815 output_dtypes[output_index] = dtype;
816 }
817
818 NodeDef host_compute_def;
819 NodeDefBuilder builder(strings::StrCat("outside_compilation_",
820 oc_subgraph_name, "_host_compute"),
821 kHostComputeOp);
822 builder.Input(inputs);
823 builder.Attr("Tinputs", input_dtypes);
824 builder.Attr("Toutputs", output_dtypes);
825 builder.Attr("key",
826 strings::StrCat("host_compute_channel_", subgraph_name, "_",
827 oc_subgraph_name));
828 builder.Attr("_outside_compilation_subgraph", oc_subgraph_name);
829 Status s = builder.Finalize(&host_compute_def);
830 if (!s.ok()) return s;
831
832 Node* host_compute = graph_->AddNode(host_compute_def, &s);
833 if (!s.ok()) return s;
834 oc_subgraph.host_compute_name = host_compute->name();
835
836 // Connect the _HostCompute node to its producers in the subgraph.
837 for (auto& input_src : oc_subgraph.inputs) {
838 const Node* src_node = input_src.first.node;
839 Node* src_image = node_images.at(src_node);
840 int src_slot = input_src.first.slot;
841 int input_index = input_src.second;
842 graph_->AddEdge(src_image, src_slot, host_compute, input_index);
843 }
844
845 // Connect the _HostCompute node to its control edge producers in the
846 // subgraph.
847 for (const auto& src_node : oc_subgraph.control_inputs) {
848 Node* src_image = node_images.at(src_node);
849 graph_->AddControlEdge(src_image, host_compute);
850 }
851
852 // Connect the consumers in the subgraph to the _HostCompute node.
853 for (const auto& output : oc_subgraph.outputs_by_dst) {
854 const Node* dst_node = output.first.node;
855 Node* dst_image = node_images.at(dst_node);
856 int dst_slot = output.first.slot;
857 int output_index = output.second;
858
859 graph_->AddEdge(host_compute, output_index, dst_image, dst_slot);
860 }
861
862 // Connect the control edge consumers in the subgraph to the _HostCompute
863 // node.
864 for (const auto& dst_node : oc_subgraph.control_outputs) {
865 Node* dst_image = node_images.at(dst_node);
866 graph_->AddControlEdge(host_compute, dst_image);
867 }
868 }
869 }
870
871 return Status::OK();
872}

898Status Encapsulator::Subgraph::BuildFunctionDef(
899 const string& name_in, const RewriteSubgraphFn& rewrite_subgraph_fn,
900 bool reuse_existing_functions, FunctionLibraryDefinition* library) {
901 // name_in is copied here because name may be modified below if
902 // rewrite_subgraph_fn is true.
903 string name = name_in;
904 call_node_def_.set_op(name);
905 call_node_def_.set_name(name);
906 call_node_def_.set_device(device_);
907
908 if (rewrite_subgraph_fn) {
909 // Initialize the input and output permutations to the identity.
910 std::vector<int> input_permutation(args_by_src_.size());
911 std::iota(input_permutation.begin(), input_permutation.end(), 0);
912 std::vector<int> output_permutation(results_.size());
913 std::iota(output_permutation.begin(), output_permutation.end(), 0);
914
915 TF_RETURN_IF_ERROR(rewrite_subgraph_fn(
916 &graph_, &input_permutation, &output_permutation, &call_node_def_));
917
918 // Apply the input/output permutations to the 'args_by_...' and 'results_'
919 // mappings, so when we build edges in BuildOutputGraph() we
920 // connect them to the right input/output positions.
921 if (input_permutation.size() != args_by_src_.size()) {
922 return errors::InvalidArgument("Input permutation has incorrect size.");
923 }
924 if (output_permutation.size() != results_.size()) {
925 return errors::InvalidArgument("Output permutation has incorrect size.");
926 }
927 for (auto& arg : args_by_src_) {
928 arg.second = input_permutation[arg.second];
929 }
930 for (auto& arg : args_by_dst_) {
931 arg.second = input_permutation[arg.second];
932 }
933 for (auto& result : results_) {
934 result.second = output_permutation[result.second];
935 }
936
937 name = call_node_def_.op();
938 }
939
940 function_def_name_ = name;
941
942 FunctionDef fdef;
943 TF_RETURN_IF_ERROR(GraphToFunctionDef(*graph_, name, &fdef));
944
945 if (VLOG_IS_ON(1)) {
946 VLOG(2) << "Build function def " << name;
947 dump_graph::DumpGraphToFile(
948 strings::StrCat("encapsulate_fdef_graph_", name), *graph_, library);
949 dump_graph::DumpFunctionDefToFile(
950 strings::StrCat("encapsulate_fdef_", name), fdef);
951 }
952
953 if (!reuse_existing_functions || library->Find(name) == nullptr) {
954 TF_RETURN_IF_ERROR(library->AddFunctionDef(fdef));
955 }
956 return Status::OK();
957}

1019Status Encapsulator::Subgraph::BuildParallelCheckOp(
1020 const std::unordered_map<const Node*, Node*>& node_images,
1021 Graph* graph_out) {
1022 // Build an index mapping output positions to node/slot pairs in the
1023 // original graph.
1024 std::vector<NodeSlot> results_by_num(results_.size());
1025 for (const auto& entry : results_) {
1026 results_by_num[entry.second] = entry.first;
1027 }
1028
1029 // Build a parallel check NodeDef.
1030 int num_results = results_by_num.size();
1031 std::vector<DataType> result_dtypes(num_results);
1032 std::vector<NodeDefBuilder::NodeOut> expected_outputs(num_results);
1033 std::vector<NodeDefBuilder::NodeOut> actual_outputs(num_results);
1034 for (int i = 0; i < num_results; ++i) {
1035 const NodeSlot& node_slot = results_by_num[i];
1036 result_dtypes[i] = node_slot.node->output_type(node_slot.slot);
1037 expected_outputs[i] =
1038 NodeDefBuilder::NodeOut(node_images.at(node_slot.node)->name(),
1039 node_slot.slot, result_dtypes[i]);
1040 actual_outputs[i] =
1041 NodeDefBuilder::NodeOut(call_node_def_.name(), i, result_dtypes[i]);
1042 }
1043 // Assign the parallel check op to a CPU on the same task as the cluster it is
1044 // checking.
1045 string device, dummy;
1046 if (!DeviceNameUtils::SplitDeviceName(
1047 call_node_inputs_->assigned_device_name(), &device, &dummy)) {
1048 return errors::InvalidArgument("Could not parse device name");
1049 }
1050 strings::StrAppend(&device, "/cpu:0");
1051
1052 NodeDef check_def;
1053 TF_RETURN_IF_ERROR(
1054 NodeDefBuilder(graph_out->NewName(strings::StrCat(call_node_def_.name(),
1055 "_parallel_check")),
1056 "ParallelCheck")
1057 .Device(device)
1058 .Attr("T", result_dtypes)
1059 .Input(expected_outputs)
1060 .Input(actual_outputs)
1061 .Finalize(&check_def));
1062
1063 Status s;
1064 Node* check_op = graph_out->AddNode(check_def, &s);
1065 if (!s.ok()) return s;
1066 check_op->set_assigned_device_name(device);
1067
1068 // TODO(phawkins): it seems redundant to call AddEdge as well as
1069 // pass Inputs to the NodeDefBuilder, but I have been unable to find a
1070 // way to avoid it.
1071 for (int i = 0; i < num_results; ++i) {
1072 const NodeSlot& node_slot = results_by_num[i];
1073 graph_out->AddEdge(node_images.at(node_slot.node), node_slot.slot, check_op,
1074 i);
1075 graph_out->AddEdge(call_node_inputs_, i, check_op, num_results + i);
1076 }
1077
1078 call_node_outputs_ = check_op;
1079 return Status::OK();
1080}

1122Status Encapsulator::Subgraph::AddRecvAtHostNode(
1123 const string& group_attribute, const string& subgraph_name,
1124 const string& outside_compilation_attribute, const string& oc_subgraph_name,
1125 OutsideCompilationSubgraph* oc_subgraph, Graph* graph_out) {
1126 if (host_compute_key_placeholder_ == nullptr) {
1127 TF_RETURN_IF_ERROR(AddHostComputeKeyPlaceholder(oc_subgraph, graph_out));
1128 }
1129
1130 std::vector<DataType> dtypes(oc_subgraph->inputs.size(), DT_INVALID);
1131
1132 for (const auto& input : oc_subgraph->inputs) {
1133 const Node* src_node = input.first.node;
1134 int src_slot = input.first.slot;
1135 int input_index = input.second;
1136
1137 DataType dtype = src_node->output_type(src_slot);
1138 dtypes[input_index] = dtype;
1139 }
1140
1141 NodeDef recv_def;
1142 NodeDefBuilder builder(strings::StrCat("outside_compilation_", subgraph_name,
1143 "_", oc_subgraph_name, "_recv"),
1144 kRecvAtHostOp);
1145 builder.Device(device_);
1146 builder.Attr("Toutputs", dtypes);
1147 // The correct device_ordinal will be inserted during replication in a
1148 // subsequent rewrite.
1149 builder.Attr("device_ordinal", 0);
1150 builder.Attr("key", strings::StrCat("host_compute_channel_", subgraph_name,
1151 "_", oc_subgraph_name));
1152 builder.Attr(group_attribute, subgraph_name);
1153 builder.Attr(outside_compilation_attribute, oc_subgraph_name);
1154 builder.Input(host_compute_key_placeholder_->name(), 0, DT_STRING);
1155 Status s = builder.Finalize(&recv_def);
1156 if (!s.ok()) return s;
1157
1158 oc_subgraph->recv_at_host = graph_out->AddNode(recv_def, &s);
1159 if (!s.ok()) return s;
1160 graph_out->AddEdge(host_compute_key_placeholder_, 0,
1161 oc_subgraph->recv_at_host, 0);
1162
1163 // Add a control dependency forcing the RecvAtHost to run before the subgraph
1164 // completes. This has no effect on execution order but prevents the
1165 // RecvAtHost being pruned.
1166 TF_RETURN_IF_ERROR(MakeSequencingNode(subgraph_name, graph_out));
1167 graph_out->AddControlEdge(oc_subgraph->recv_at_host, sequencer_);
1168
1169 return Status::OK();

1172Status Encapsulator::Subgraph::AddSendFromHostNode(
1173 const std::unordered_map<const Node*, Node*>& node_images,
1174 const string& group_attribute, const string& subgraph_name,
1175 const string& outside_compilation_attribute, const string& oc_subgraph_name,
1176 OutsideCompilationSubgraph* oc_subgraph, Graph* graph_out) {
1177 if (host_compute_key_placeholder_ == nullptr) {
1178 TF_RETURN_IF_ERROR(AddHostComputeKeyPlaceholder(oc_subgraph, graph_out));
1179 }
1180
1181 std::vector<DataType> dtypes(oc_subgraph->outputs_by_src.size(), DT_INVALID);
1182 std::vector<NodeDefBuilder::NodeOut> inputs(
1183 oc_subgraph->outputs_by_src.size());
1184
1185 for (const auto& output : oc_subgraph->outputs_by_src) {
1186 const Node* src_node = output.first.node;
1187 Node* src_image = node_images.at(src_node);
1188 int src_slot = output.first.slot;
1189 int output_index = output.second;
1190
1191 DataType dtype = src_node->output_type(src_slot);
1192 dtypes[output_index] = dtype;
1193 inputs[output_index].Reset(src_image->name(), src_slot, dtype);
1194 }
1195
1196 NodeDef send_def;
1197 NodeDefBuilder builder(strings::StrCat("outside_compilation_", subgraph_name,
1198 "_", oc_subgraph_name, "_send"),
1199 kSendFromHostOp);
1200 builder.Device(device_);
1201 builder.Attr("Tinputs", dtypes);
1202 builder.Attr("key", strings::StrCat("host_compute_channel_", subgraph_name,
1203 "_", oc_subgraph_name));
1204 // The correct device_ordinal will be inserted during replication in a
1205 // subsequent rewrite.
1206 builder.Attr("device_ordinal", 0);
1207 builder.Attr(group_attribute, subgraph_name);
1208 builder.Attr(outside_compilation_attribute, oc_subgraph_name);
1209 builder.Input(inputs);
1210 builder.Input(host_compute_key_placeholder_->name(), 0, DT_STRING);
1211 Status s = builder.Finalize(&send_def);
1212 if (!s.ok()) return s;
1213
1214 oc_subgraph->send_from_host = graph_out->AddNode(send_def, &s);
1215 if (!s.ok()) return s;
1216 graph_out->AddEdge(host_compute_key_placeholder_, 0,
1217 oc_subgraph->send_from_host, inputs.size());
1218
1219 // Add a control dependency forcing the SendFromHost to run before the
1220 // subgraph completes. This has no effect on execution order but prevents the
1221 // RecvAtHost being pruned.
1222 TF_RETURN_IF_ERROR(MakeSequencingNode(subgraph_name, graph_out));
1223 graph_out->AddControlEdge(oc_subgraph->send_from_host, sequencer_);
1224
1225 return Status::OK();
1226}

1309Status Encapsulator::CopySubgraphEdges(
1310 const std::unordered_map<const Node*, Node*>& node_images,
1311 std::vector<std::pair<const Node*, Node*>>* src_arg_pairs) {
1312 for (const Edge* edge : graph_in_->edges()) {
1313 string src_func_id;
1314 string src_outside_compilation_id;
1315 TF_RETURN_IF_ERROR(GetFunctionNameAttr(edge->src(), &src_func_id,
1316 &src_outside_compilation_id));
1317 string dst_func_id;
1318 string dst_outside_compilation_id;
1319 TF_RETURN_IF_ERROR(GetFunctionNameAttr(edge->dst(), &dst_func_id,
1320 &dst_outside_compilation_id));
1321 Node* src_image = gtl::FindWithDefault(node_images, edge->src(), nullptr);
1322 Node* dst_image = gtl::FindWithDefault(node_images, edge->dst(), nullptr);
1323
1324 // Copy edges that are local to a subgraph.
1325 if (IsInSubgraph(src_func_id, src_outside_compilation_id) &&
1326 IsInSubgraph(dst_func_id, dst_outside_compilation_id) &&
1327 src_func_id == dst_func_id) {
1328 Graph* g = subgraphs_[src_func_id].GetGraph();
1329 if (edge->IsControlEdge()) {
1330 g->AddControlEdge(src_image, dst_image);
1331 } else {
1332 g->AddEdge(src_image, edge->src_output(), dst_image, edge->dst_input());
1333 }
1334 continue;
1335 }
1336
1337 // Record 'src' as an output of its subgraph, if applicable.
1338 if (IsInSubgraph(src_func_id, src_outside_compilation_id)) {
1339 if (!edge->IsControlEdge()) {
1340 DataType dtype = edge->src()->output_type(edge->src_output());
1341 if (IsRefType(dtype)) {
1342 return errors::InvalidArgument(
1343 "Ref Tensors (e.g., Variables) are not supported as results: "
1344 "tensor ",
1345 edge->src()->name(), ":", edge->src_output());
1346 }
1347 }
1348
1349 Subgraph& src_subgraph = subgraphs_[src_func_id];
1350 if (src_func_id == dst_func_id) {
1351 // src is in the subgraph and dst is outside_compilation in the same
1352 // subgraph.
1353 src_subgraph.RecordOutsideCompilationInputOrControl(
1354 dst_outside_compilation_id, edge);
1355 } else {
1356 // Ignore control edges leaving the subgraph. We will lift them onto the
1357 // enclosing call operators in BuildOutputGraph().
1358 if (!edge->IsControlEdge()) {
1359 TF_RETURN_IF_ERROR(src_subgraph.RecordResult(edge, node_images));
1360 }
1361 }
1362 }
1363
1364 // Record 'dst' as an input of its subgraph, if applicable.
1365 if (IsInSubgraph(dst_func_id, dst_outside_compilation_id)) {
1366 // Look at the type of the destination not the source, since Ref output
1367 // Tensors can be automatically cast to non-Ref Tensors at the
1368 // destination.
1369 if (!edge->IsControlEdge()) {
1370 DataType dtype = edge->dst()->input_type(edge->dst_input());
1371 if (IsRefType(dtype)) {
1372 return errors::InvalidArgument(
1373 "Ref Tensors (e.g., Variables) are not supported as args: "
1374 "tensor ",
1375 edge->src()->name(), ":", edge->src_output());
1376 }
1377 }
1378
1379 Subgraph& dst_subgraph = subgraphs_[dst_func_id];
1380 if (src_func_id == dst_func_id) {
1381 // dst is in the subgraph and src is outside_compilation in the same
1382 // subgraph.
1383 dst_subgraph.RecordOutsideCompilationOutputOrControl(
1384 src_outside_compilation_id, edge);
1385 } else {
1386 // Ignore control edges entering the subgraph. We will lift them onto
1387 // the enclosing call operators in BuildOutputGraph().
1388 if (!edge->IsControlEdge()) {
1389 TF_RETURN_IF_ERROR(
1390 dst_subgraph.RecordArg(edge, node_images, src_arg_pairs));
1391 }
1392 }
1393 }
1394 }
1395 return Status::OK();
1396}

1398Status Encapsulator::SplitIntoSubgraphs() {
1399 Status s;
1400
1401 // Map from input graph nodes to subgraph nodes.
1402 std::unordered_map<const Node*, Node*> node_images;
1403
1404 // Each entry of src_arg_pairs is a pair whose first element is a node in the
1405 // original graph that has an output edge in the subgraph, and whose second
1406 // element is the arg node in the subgraph that it sends to. The vector will
1407 // be filled in below in AddArgs.
1408 std::vector<std::pair<const Node*, Node*>> src_arg_pairs;
1409
1410 TF_RETURN_IF_ERROR(CopySubgraphNodes(&node_images));
1411 TF_RETURN_IF_ERROR(CopySubgraphEdges(node_images, &src_arg_pairs));
1412
1413 // For each subgraph, add the nodes that deal with inputs and outputs its
1414 // nested outside_compilation subgraphs. These could not be added earlier
1415 // during CopySubgraphEdges since we need to discover all the types of the
1416 // inputs and outputs for an outside_compilation subgraph before creating a
1417 // single input and output node for it.
1418 for (auto& entry : subgraphs_) {
1419 Subgraph& subgraph = entry.second;
1420 TF_RETURN_IF_ERROR(subgraph.AddHostComputes(entry.first, node_images));
1421 }
1422
1423 MarkGuaranteedConstants(*graph_in_, src_arg_pairs);
1424
1425 for (auto& entry : subgraphs_) {
1426 Subgraph& subgraph = entry.second;
1427 FixupSourceAndSinkEdges(subgraph.GetGraph());
1428 }
1429
1430 return s;
1431}

1593Status Encapsulator::CopyEdgeToOutputGraph(
1594 const Edge* edge, const string& src_func_id,
1595 const string& src_outside_compilation_id, const string& dst_func_id,
1596 const string& dst_outside_compilation_id,
1597 const std::unordered_map<const Node*, Node*>& node_images,
1598 bool parallel_checking, Graph* graph_out,
1599 std::unordered_set<std::pair<NodeSlot, NodeSlot>, NodeSlot::PairHasher>*
1600 edges_added) {
1601 Node* src_image;
1602 TF_RETURN_IF_ERROR(FindOutputImageOfEdgeSrc(
1603 src_func_id, src_outside_compilation_id, dst_func_id,
1604 dst_outside_compilation_id, node_images, edge->src(), &src_image));
1605 Node* dst_image;
1606 TF_RETURN_IF_ERROR(FindOutputImageOfEdgeDst(
1607 src_func_id, src_outside_compilation_id, dst_func_id,
1608 dst_outside_compilation_id, node_images, edge->dst(), &dst_image));
1609
1610 // If this is a control edge then copy it and return. Lift control edges onto
1611 // the enclosing call operator.
1612 if (edge->IsControlEdge()) {
1613 // Add the control edge, if we have not already added it, using the images
1614 // determined above (potentially call operators or RecvAtHost/SendFromHost).
1615 if (edges_added->emplace(NodeSlot(src_image, -1), NodeSlot(dst_image, -1))
1616 .second) {
1617 graph_out->AddControlEdge(src_image, dst_image);
1618 }
1619
1620 // If parallel checking is enabled, also add a control edge to the
1621 // corresponding parallel check op.
1622 if (parallel_checking) {
1623 graph_out->AddControlEdge(src_image, node_images.at(edge->dst()));
1624 }
1625 return Status::OK();
1626 }
1627
1628 int src_output =
1629 FindOutputSlotOfEdgeSrc(src_func_id, src_outside_compilation_id,
1630 dst_func_id, dst_outside_compilation_id, edge);
1631
1632 int dst_input =
1633 FindOutputSlotOfEdgeDst(src_func_id, src_outside_compilation_id,
1634 dst_func_id, dst_outside_compilation_id, edge);
1635
1636 if (IsInSubgraph(dst_func_id, dst_outside_compilation_id) &&
1637 parallel_checking) {
1638 // If we are parallel checking, also feed the tensor as an input to the
1639 // corresponding parallel check subgraph.
1640 graph_out->AddEdge(src_image, src_output, node_images.at(edge->dst()),
1641 edge->dst_input());
1642 }
1643
1644 // Add the edge, if we have not already added it.
1645 if (edges_added
1646 ->emplace(NodeSlot(src_image, src_output),
1647 NodeSlot(dst_image, dst_input))
1648 .second) {
1649 graph_out->AddEdge(src_image, src_output, dst_image, dst_input);
1650 }
1651 return Status::OK();
1652}

1654Status Encapsulator::AddEdgesToOutputGraph(
1655 const std::unordered_map<const Node*, Node*>& node_images,
1656 bool parallel_checking, Graph* graph_out) {
1657 // Set of edges already added to the output graph, represented as (src, dst)
1658 // pairs. We use the set to deduplicate edges; multiple edges in the input
1659 // graph may map to one edge in the output graph.
1660 std::unordered_set<std::pair<NodeSlot, NodeSlot>, NodeSlot::PairHasher>
1661 edges_added;
1662
1663 for (const Edge* edge : graph_in_->edges()) {
1664 string src_func_id;
1665 string src_outside_compilation_id;
1666 TF_RETURN_IF_ERROR(GetFunctionNameAttr(edge->src(), &src_func_id,
1667 &src_outside_compilation_id));
1668 string dst_func_id;
1669 string dst_outside_compilation_id;
1670 TF_RETURN_IF_ERROR(GetFunctionNameAttr(edge->dst(), &dst_func_id,
1671 &dst_outside_compilation_id));
1672
1673 // Ignore edges that are strictly contained within one subgraph, unless
1674 // we are constructing parallel check graphs.
1675 if (IsInSubgraph(src_func_id, src_outside_compilation_id) &&
1676 IsInSubgraph(dst_func_id, dst_outside_compilation_id) &&
1677 src_func_id == dst_func_id) {
1678 if (parallel_checking) {
1679 Node* src_image = node_images.at(edge->src());
1680 Node* dst_image = node_images.at(edge->dst());
1681 if (edge->IsControlEdge()) {
1682 graph_out->AddControlEdge(src_image, dst_image);
1683 } else {
1684 graph_out->AddEdge(src_image, edge->src_output(), dst_image,
1685 edge->dst_input());
1686 }
1687 }
1688 continue;
1689 }
1690
1691 // We have an edge that crosses a cluster boundary or is entirely within the
1692 // unclustered graph.
1693 TF_RETURN_IF_ERROR(CopyEdgeToOutputGraph(
1694 edge, src_func_id, src_outside_compilation_id, dst_func_id,
1695 dst_outside_compilation_id, node_images, parallel_checking, graph_out,
1696 &edges_added));
1697 }
1698
1699 for (auto& subgraph_entry : subgraphs_) {
1700 Subgraph& subgraph = subgraph_entry.second;
1701 subgraph.ConnectSequencerToCallNode(graph_out);
1702 }
1703
1704 return Status::OK();
1705}

1731// Adds a copy of node_in to graph_out and adds the mapping to
1732// copied_node_images.
1733Status CopyShapeInferenceNodeToGraph(
1734 Node* node_in, const Node* send_node,
1735 const std::unordered_map<Node*, Node*>& dummy_node_images,
1736 FunctionLibraryDefinition* library,
1737 std::unordered_map<Node*, Node*>* copied_node_images, Graph* graph_out) {
1738 // Once all the ancestor nodes have been added to graph_out, add this node
1739 // and connect it to its ancestors.
1740 Node* node_out = graph_out->CopyNode(node_in);
1741 (*copied_node_images)[node_in] = node_out;
1742 // Don't bother to build the shape inference graph if there's a node with no
1743 // shape inference function, since it would just result in an error later at
1744 // compile time.
1745 const OpRegistrationData* op_reg_data;
1746 TF_RETURN_IF_ERROR(library->LookUp(node_in->type_string(), &op_reg_data));
1747 if (op_reg_data->shape_inference_fn == nullptr) {
1748 return errors::InvalidArgument(
1749 "Shape inference is not possible for outside_compilation "
1750 "SendFromHost node ",
1751 send_node->name(), " because it depends on node ", node_in->name(),
1752 " which does not have a shape inference function registered.");
1753 }
1754 // Add all the edges to the newly copied node.
1755 for (const Edge* in_edge : node_in->in_edges()) {
1756 if (!in_edge->IsControlEdge()) {
1757 Node* src = in_edge->src();
1758 const auto iter = dummy_node_images.find(src);
1759 if (iter == dummy_node_images.end()) {
1760 // The src is a copied node so use the original output port.
1761 graph_out->AddEdge((*copied_node_images)[in_edge->src()],
1762 in_edge->src_output(), node_out,
1763 in_edge->dst_input());
1764 } else {
1765 // The src is a dummy node so use output port 0.
1766 graph_out->AddEdge(iter->second, 0, node_out, in_edge->dst_input());
1767 }
1768 }
1769 }
1770 return Status::OK();
1771}

1775Status Encapsulator::DoStaticShapeInferenceForOutsideCompilationSend(
1776 const Graph& graph_in, const ShapeRefiner& shape_refiner,
1777 const std::unordered_set<string>& recv_at_host_nodes, Node* send_node,
1778 FunctionLibraryDefinition* library,
1779 std::vector<TensorShapeProto>* static_shape_out,
1780 std::unique_ptr<Graph>* graph_out) {
1781 // Maps from nodes in graph_in to nodes in graph_out.
1782 //
1783 // When an edge has fully defined shape the source node in graph_in is
1784 // replaced in graph_out by a dummy constant node. The mapping from nodes
1785 // in graph_in to dummy nodes is stored in dummy_node_images.
1786 //
1787 // When a node in graph_in has at least one ancestor that doesn't have fully
1788 // defined shape, it is copied into graph_out. The mapping from nodes in
1789 // graph_in to copied nodes is stored in copied_node_images.
1790 //
1791 // The two types of node are treated differently because, when adding edges to
1792 // graph_out, an output from a dummy node always uses port 0, whereas an
1793 // output from a copied node uses the same port that was used in graph_in.
1794 std::unordered_map<Node*, Node*> dummy_node_images;
1795 std::unordered_map<Node*, Node*> copied_node_images;
1796
1797 graph_out->reset(new Graph(graph_in.op_registry()));
1798 (*graph_out)->set_versions(graph_in.versions());
1799 // The final input to the send node is the dynamic key, which we don't include
1800 // in the static shapes.
1801 static_shape_out->resize(send_node->num_inputs() - 1);
1802
1803 // We don't use the standard ReverseDFS because we want to cut off traversal
1804 // whenever we find an output with fully defined shape.
1805 // TODO(misard) make this work properly in the presence of control flow.
1806 struct Work {
1807 Node* node;
1808 bool leave; // Are we entering or leaving node?
1809 };
1810 std::vector<Work> stack({{send_node, false}});
1811 std::vector<bool> visited(graph_in.num_node_ids(), false);
1812 while (!stack.empty()) {
1813 Work w = stack.back();
1814 stack.pop_back();
1815 Node* n = w.node;
1816
1817 if (w.leave) {
1818 TF_RETURN_IF_ERROR(CopyShapeInferenceNodeToGraph(
1819 n, send_node, dummy_node_images, library, &copied_node_images,
1820 graph_out->get()));
1821 } else {
1822 if (visited[n->id()]) continue;
1823 visited[n->id()] = true;
1824
1825 // Arrange to revisit when all done with all inputs.
1826 stack.push_back(Work{n, true});
1827
1828 bool has_parent_with_unknown_shape = false;
1829 for (const Edge* in_edge : n->in_edges()) {
1830 if (!in_edge->IsControlEdge()) {
1831 Node* src_node = in_edge->src();
1832 int src_port = in_edge->src_output();
1833 shape_inference::InferenceContext* context =
1834 shape_refiner.GetContext(src_node);
1835 shape_inference::ShapeHandle shape = context->output(src_port);
1836 if (context->FullyDefined(shape)) {
1837 // This ancestor has known shape, so instead of adding it to the
1838 // stack, add a dummy node with that shape to graph_out and
1839 // continue.
1840 TensorShapeProto proto;
1841 context->ShapeHandleToProto(shape, &proto);
1842 if (dummy_node_images.find(src_node) == dummy_node_images.end()) {
1843 dummy_node_images[src_node] = AddDummyShapedNode(
1844 src_node->output_type(src_port), proto, graph_out->get());
1845 }
1846 // The final input to the send node is the dynamic key, which we
1847 // don't include in the static shapes.
1848 if (n == send_node &&
1849 in_edge->dst_input() < static_shape_out->size()) {
1850 (*static_shape_out)[in_edge->dst_input()] = proto;
1851 }
1852 } else {
1853 has_parent_with_unknown_shape = true;
1854 if (!visited[src_node->id()]) {
1855 if (VLOG_IS_ON(2)) {
1856 TensorShapeProto proto;
1857 context->ShapeHandleToProto(shape, &proto);
1858 }
1859 stack.push_back({src_node, false});
1860 }
1861 }
1862 }
1863 }
1864 if (!has_parent_with_unknown_shape) {
1865 if (n == send_node) {
1866 // The shapes of all the inputs to send_node are statically known. We
1867 // won't have to do any inference at compile time so return now: the
1868 // shapes were stored in static_shape_out above.
1869 graph_out->reset();
1870 return Status::OK();
1871 } else {
1872 // Any shape that is being processed is either the original send node
1873 // or has at least one output with statically-unknown shape. If the
1874 // latter and it doesn't have any inputs with statically-unknown
1875 // shape, then check that it is of the recv nodes that we can fill in
1876 // the shape of at run-time later. If it isn't one of those, then we
1877 // won't have any additional knowledge at compile time, so we already
1878 // know we won't be able to do shape inference and we can return an
1879 // error now.
1880 if (recv_at_host_nodes.find(n->name()) == recv_at_host_nodes.end()) {
1881 return errors::InvalidArgument(
1882 "Shape inference is not possible for outside_compilation "
1883 "SendFromHost node ",
1884 send_node->name(), " because shape of node ", n->name(),
1885 " will not be known at compilation time.");
1886 }
1887 }
1888 }
1889 }
1890 }
1891
1892 return Status::OK();
1893}

1895Status Encapsulator::MakePrunedGraphCopyAndInline(
1896 const Graph& graph, const std::vector<Node*>& sink_nodes,
1897 std::unique_ptr<Graph>* pruned_graph,
1898 std::unordered_map<const Node*, Node*>* node_images,
1899 FunctionLibraryDefinition* library) {
1900 // First copy all ancestor nodes of sink_nodes into a new graph.
1901 pruned_graph->reset(new Graph(library));
1902 (*pruned_graph)->set_versions(graph.versions());
1903 ReverseDFSFrom(graph, sink_nodes,
1904 /*enter=*/nullptr,
1905 /*leave=*/[&](Node* n) {
1906 if (!n->IsSource()) {
1907 Node* copied = (*pruned_graph)->CopyNode(n);
1908 node_images->emplace(n, copied);
1909 }
1910 });
1911
1912 // Add all the edges between copied nodes.
1913 for (auto entry : *node_images) {
1914 const Node* orig = entry.first;
1915 Node* image = entry.second;
1916 for (const Edge* out_edge : orig->out_edges()) {
1917 auto iter = node_images->find(out_edge->dst());
1918 if (iter != node_images->end()) {
1919 // The source and destination are both in the copied graph.
1920 (*pruned_graph)
1921 ->AddEdge(image, out_edge->src_output(), iter->second,
1922 out_edge->dst_input());
1923 }
1924 }
1925 }
1926
1927 // Find all the function call nodes, and inline them.
1928 std::vector<Node*> function_nodes;
1929 for (auto node : (*pruned_graph)->nodes()) {
1930 const OpRegistrationData* op_reg_data;
1931 TF_RETURN_IF_ERROR(library->LookUp(node->type_string(), &op_reg_data));
1932 if (op_reg_data->is_function_op) {
1933 function_nodes.push_back(node);
1934 }
1935 }
1936 for (auto node : function_nodes) {
1937 VLOG(2) << "Inlining function " << node->name();
1938 const FunctionDef* fdef = library->Find(node->type_string());
1939 if (fdef == nullptr) {
1940 return errors::Internal("Failed to find function ", node->type_string(),
1941 " in function library.");
1942 }
1943 FunctionBody* fbody = nullptr;
1944 TF_RETURN_IF_ERROR(
1945 FunctionDefToBodyHelper(*fdef, node->attrs(), library,
1946 [library](const string& op, const OpDef** sig) {
1947 return library->LookUpOpDef(op, sig);
1948 },
1949 &fbody));
1950 InlineFunctionBody(*library, pruned_graph->get(), node, fbody);
1951 delete fbody;
1952 }
1953
1954 return Status::OK();
1955}

1957Status Encapsulator::MakeGraphForOutsideCompilationSends(
1958 const Graph& graph, std::unique_ptr<Graph>* pruned_graph,
1959 ShapeRefiner* shape_refiner,
1960 std::unordered_map<const Node*, Node*>* node_images,
1961 FunctionLibraryDefinition* library) {
1962 // Find all the send_from_host nodes in all subgraphs, to use as roots for the
1963 // pruning.
1964 std::vector<Node*> send_from_host_nodes;
1965 for (auto& subgraph_entry : subgraphs_) {
1966 Subgraph& subgraph = subgraph_entry.second;
1967 std::vector<string> outside_compilation_names;
1968 subgraph.GetOutsideCompilationSubgraphNames(&outside_compilation_names);
1969 for (const auto& name : outside_compilation_names) {
1970 Node* send_node = subgraph.GetSendFromHostNode(name);
1971 if (send_node != nullptr) {
1972 send_from_host_nodes.push_back(send_node);
1973 }
1974 }
1975 }
1976
1977 // Make a copy of all the graph nodes needed to evaluate the send_from_host
1978 // nodes, inlining any functions as needed.
1979 TF_RETURN_IF_ERROR(MakePrunedGraphCopyAndInline(
1980 graph, send_from_host_nodes, pruned_graph, node_images, library));
1981
1982 // Perform shape inference on the pruned graph.
1983 shape_refiner->set_require_shape_inference_fns(false);
1984 FixupSourceAndSinkEdges(pruned_graph->get());
1985 std::vector<Node*> post_order;
1986 GetReversePostOrder(*(*pruned_graph), &post_order);
1987 for (auto node : post_order) {
1988 // Ignore the status returned by the shape_refiner. At this point we want
1989 // the best effort shapes, even if no shape function is registered for a
1990 // node.
1991 Status status = shape_refiner->AddNode(node);
1992 if (!status.ok()) {
1993 VLOG(1) << "Shape inference failed for node: " << status;
1994 }
1995 }
1996
1997 return Status::OK();
1998}

2000Status Encapsulator::GetShapeInfoForOutsideCompilationSends(
2001 Graph* graph_out, FunctionLibraryDefinition* library) {
2002 std::unique_ptr<Graph> pruned_graph;
2003 ShapeRefiner shape_refiner(graph_out->versions(), graph_out->op_registry());
2004 std::unordered_map<const Node*, Node*> node_images;
2005 TF_RETURN_IF_ERROR(MakeGraphForOutsideCompilationSends(
2006 *graph_out, &pruned_graph, &shape_refiner, &node_images, library));
2007
2008 if (VLOG_IS_ON(1)) {
2009 dump_graph::DumpGraphToFile("pruned_graph_for_shape_inference",
2010 *pruned_graph, library);
2011 }
2012
2013 for (auto& subgraph_entry : subgraphs_) {
2014 const string& subgraph_name = subgraph_entry.first;
2015 Subgraph& subgraph = subgraph_entry.second;
2016 // Find all the recv_at_host nodes in this subgraph.
2017 std::vector<string> outside_compilation_names;
2018 subgraph.GetOutsideCompilationSubgraphNames(&outside_compilation_names);
2019 std::unordered_set<string> recv_at_host_names;
2020 for (const auto& oc_name : outside_compilation_names) {
2021 Node* recv_node = subgraph.GetRecvAtHostNode(oc_name);
2022 if (recv_node != nullptr) {
2023 recv_at_host_names.insert(recv_node->name());
2024 }
2025 }
2026 // For each send_from_host node, do as much shape inference as possible
2027 // without knowing the shape of the recv_at_host nodes, and store the
2028 // result, along with enough information to complete the job at compile time
2029 // once the recv_at_host shapes are known.
2030 for (const auto& oc_name : outside_compilation_names) {
2031 Node* send_node = subgraph.GetSendFromHostNode(oc_name);
2032 std::vector<TensorShapeProto> static_shape;
2033 std::unique_ptr<Graph> graph;
2034 if (send_node != nullptr) {
2035 TF_RETURN_IF_ERROR(DoStaticShapeInferenceForOutsideCompilationSend(
2036 *pruned_graph, shape_refiner, recv_at_host_names,
2037 node_images[send_node], library, &static_shape, &graph));
2038 if (graph == nullptr) {
2039 VLOG(2) << "Send node " << send_node->name() << " shapes";
2040 for (int i = 0; i < static_shape.size(); ++i) {
2041 VLOG(2) << static_shape[i].DebugString();
2042 }
2043 } else {
2044 if (VLOG_IS_ON(2)) {
2045 GraphDef graphdef;
2046 graph->ToGraphDef(&graphdef);
2047 VLOG(2) << "Send node " << send_node->name() << " graph\n"
2048 << graphdef.DebugString();
2049 }
2050 }
2051 }
2052 TF_RETURN_IF_ERROR(subgraph.AddShapeInferenceInfo(
2053 subgraph_name, oc_name, static_shape, graph.get(), library));
2054 }
2055 if (!outside_compilation_names.empty()) {
2056 TF_RETURN_IF_ERROR(subgraph.ReplaceFunctionDef(library));
2057 }
2058 }
2059
2060 return Status::OK();

2140Status EncapsulateSubgraphsPass::Run(
2141 const GraphOptimizationPassOptions& options) {
2142 VLOG(1) << "EncapsulateSubgraphsPass::Run";
2143 legacy_flags::EncapsulateSubgraphsPassFlags* flags =
2144 legacy_flags::GetEncapsulateSubgraphsPassFlags();
2145 if (VLOG_IS_ON(1)) {
2146 dump_graph::DumpGraphToFile("before_encapsulate_subgraphs", **options.graph,
2147 options.flib_def);
2148 }
2149
2150 std::unique_ptr<Graph> graph_out;
2151 FunctionLibraryDefinition* const library = options.flib_def;
2152
2153 OptimizerOptions opts;
2154 std::unique_ptr<ProcessFunctionLibraryRuntime> pflr(
2155 new ProcessFunctionLibraryRuntime(nullptr, options.session_options->env,
2156 TF_GRAPH_DEF_VERSION, library, opts));
2157 FunctionLibraryRuntime* flr =
2158 pflr->GetFLR(ProcessFunctionLibraryRuntime::kDefaultFLRDevice);
2159
2160 auto rewrite_subgraph = [flr](std::unique_ptr<Graph>* subgraph,
2161 std::vector<int>* input_permutation,
2162 std::vector<int>* output_permutation,
2163 NodeDef* node) {
2164 // Optimize the subgraph.
2165 OptimizeGraph(flr, subgraph);
2166
2167 const int num_args = input_permutation->size();
2168 std::vector<bool> const_args(num_args);
2169 TF_RETURN_IF_ERROR(BackwardsConstAnalysis(**subgraph, &const_args));
2170
2171 DataTypeVector arg_types(num_args);
2172 TF_RETURN_IF_ERROR(GetArgTypes(**subgraph, &arg_types));
2173
2174 // Compute a permutation of the arguments such that the constant arguments
2175 // are first.
2176 const int num_consts =
2177 std::count(const_args.begin(), const_args.end(), true);
2178
2179 const int num_resources =
2180 std::count(arg_types.begin(), arg_types.end(), DT_RESOURCE);
2181 const int num_nonconsts = num_args - num_resources - num_consts;
2182 if (num_nonconsts < 0) {
2183 return errors::Internal("num_nonconsts should be >= 0, was ",
2184 num_nonconsts);
2185 }
2186
2187 int const_pos = 0;
2188 int arg_pos = num_consts;
2189 int resource_pos = num_consts + num_nonconsts;
2190 for (int i = 0; i < num_args; ++i) {
2191 if (const_args[i]) {
2192 if (arg_types[i] == DT_RESOURCE) {
2193 return errors::Internal(
2194 "Resource arguments cannot be constant (argument ", i, ")");
2195 }
2196 (*input_permutation)[i] = const_pos;
2197 ++const_pos;
2198 } else if (arg_types[i] == DT_RESOURCE) {
2199 (*input_permutation)[i] = resource_pos;
2200 ++resource_pos;
2201 } else {
2202 (*input_permutation)[i] = arg_pos;
2203 ++arg_pos;
2204 }
2205 }
2206
2207 // Renumber argument nodes in the graph.
2208 TF_RETURN_IF_ERROR(RenumberArguments(subgraph->get(), *input_permutation));
2209
2210 // TODO(phawkins): add a forward is-constant analysis, similarly split
2211 // outputs into host-memory constants and device-memory non-constants.
2212
2213 AddNodeAttr(kXlaCompiledKernelAttr, true, node);
2214 AddNodeAttr(kXlaNumConstantArgsAttr, num_consts, node);
2215 AddNodeAttr(kXlaNumResourceArgsAttr, num_resources, node);
2216 return Status::OK();
2217 };

2219 TF_RETURN_IF_ERROR(EncapsulateSubgraphsInFunctions(
2220 kXlaClusterAttr, kXlaOutsideCompilationAttr, **options.graph,
2221 rewrite_subgraph, flags->tf_xla_parallel_checking,
2222 /*reuse_existing_functions=*/false, &graph_out, library));

2226 options.flib_def);

2229 *options.graph = std::move(graph_out);
