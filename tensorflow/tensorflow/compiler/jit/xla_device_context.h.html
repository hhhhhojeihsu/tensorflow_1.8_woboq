30// The allocator used for Tensors assigned to the XLA device. The allocator
31// ignores the alignment and size of the request and always returns a new,
32// empty, XlaTensor.
33class XlaDeviceAllocator : public Allocator {
34 public:
35 XlaDeviceAllocator();
36 ~XlaDeviceAllocator() override;
37
38 string Name() override;
39
40 void* AllocateRaw(size_t alignment, size_t num_bytes) override;
41 void DeallocateRaw(void* ptr) override;
42 void GetStats(AllocatorStats* stats) override;
43};
44
45// Helper class for managing data transfers between host and XLA devices.
46class XlaTransferManager {
47 public:
48 explicit XlaTransferManager(perftools::gputools::Stream* stream,
49 xla::LocalClient* client,
50 bool transfer_as_literal);
51
52 void CopyCPUTensorToDevice(const Tensor* cpu_tensor, Device* device,
53 Tensor* device_tensor, StatusCallback done) const;
54 void CopyDeviceTensorToCPU(const Tensor* device_tensor,
55 StringPiece tensor_name, Device* device,
56 Tensor* cpu_tensor, StatusCallback done);
57 perftools::gputools::Stream* stream() const { return stream_; }
58
59 private:
60 Status TransferLiteralToDevice(const Tensor& host_tensor,
61 Tensor* device_tensor) const;
62 Status TransferLiteralFromDevice(Tensor* host_tensor,
63 const Tensor& device_tensor) const;
64
65 // Stream obtained from a Device, used to transfer tensors between
66 // CPU and device.
67 perftools::gputools::Stream* stream_;
68 // For the underlying memory allocator and XLA's TransferManager.
69 xla::LocalClient* client_;
70 // Transfer manager, for marshalling data to and from the device.
71 xla::TransferManager* transfer_manager_;
72 // True if we must use XLA's TransferManager for correct device transfers.
73 bool transfer_as_literal_;
74};
75
76// DeviceContext for operators assigned to XlaDevice devices. The
77// implementation must inherit from DeviceContext but otherwise just
78// wraps the methods in XlaTransferManager.
79class XlaDeviceContext : public DeviceContext {
80 public:
81 explicit XlaDeviceContext(perftools::gputools::Stream* stream,
82 xla::LocalClient* client, bool transfer_as_literal);
83
84 void CopyCPUTensorToDevice(const Tensor* cpu_tensor, Device* device,
85 Tensor* device_tensor,
86 StatusCallback done) const override;
87 void CopyDeviceTensorToCPU(const Tensor* device_tensor,
88 StringPiece tensor_name, Device* device,
89 Tensor* cpu_tensor, StatusCallback done) override;
90 perftools::gputools::Stream* stream() const override {
91 return manager_.stream();
92 }
93
94 private:
95 XlaTransferManager manager_;
96};
